89 boost::shared_ptr<FloatingRateCouponPricer> LinearTSRCmsCouponPricerBuilder::engineImpl(const Currency& ccy) {
90 
91  const string& ccyCode = ccy.code();
92  Real rev = parseCcyReversion(engineParameters_, ccyCode);
93  string policy = engineParameter("Policy");
94 
95  Handle<Quote> revQuote(boost::shared_ptr<Quote>(new SimpleQuote(rev)));
96  Handle<SwaptionVolatilityStructure> vol = market_->swaptionVol(ccyCode, configuration(MarketContext::pricing));
97  Handle<YieldTermStructure> yts = market_->discountCurve(ccyCode, configuration(MarketContext::pricing));
98 
99  string lowerBoundStr =
100  (vol->volatilityType() == ShiftedLognormal) ? "LowerRateBoundLogNormal" : "LowerRateBoundNormal";
101  string upperBoundStr =
102  (vol->volatilityType() == ShiftedLognormal) ? "UpperRateBoundLogNormal" : "UpperRateBoundNormal";
103 
104  LinearTsrPricer::Settings settings;
105  if (policy == "RateBound") {
106  Real lower = parseReal(engineParameter(lowerBoundStr));
107  Real upper = parseReal(engineParameter(upperBoundStr));
108  settings.withRateBound(lower, upper);
109  } else if (policy == "VegaRatio") {
110  Real lower = parseReal(engineParameter(lowerBoundStr));
111  Real upper = parseReal(engineParameter(upperBoundStr));
112  Real vega = parseReal(engineParameter("VegaRatio"));
113  settings.withVegaRatio(vega, lower, upper);
114  } else if (policy == "PriceThreshold") {
115  Real lower = parseReal(engineParameter(lowerBoundStr));
116  Real upper = parseReal(engineParameter(upperBoundStr));
117  Real threshold = parseReal(engineParameter("PriceThreshold"));
118  settings.withPriceThreshold(threshold, lower, upper);
119  } else if (policy == "BsStdDev") {
120  Real lower = parseReal(engineParameter(lowerBoundStr));
121  Real upper = parseReal(engineParameter(upperBoundStr));
122  Real stddevs = parseReal(engineParameter("BSStdDevs"));
123  settings.withPriceThreshold(stddevs, lower, upper);
124  } else
125  QL_FAIL("unknown string for policy parameter");
126 
127  boost::shared_ptr<FloatingRateCouponPricer> pricer =
128  boost::make_shared<LinearTsrPricer>(vol, revQuote, yts, settings);
129 
130  // Return the cached pricer
131  return pricer;
132 }
