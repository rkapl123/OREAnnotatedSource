121 // calls parseDateOrPeriod and returns a Date (either the supplied date or asof+period)
122 Date getDateFromDateOrPeriod(const string& token, Date asof, QuantLib::Calendar cal, QuantLib::BusinessDayConvention bdc) {
123  Period term; // gets populated by parseDateOrPeriod
124  Date expiryDate; // gets populated by parseDateOrPeriod
125  bool tmpIsDate; // gets populated by parseDateOrPeriod
126  parseDateOrPeriod(token, expiryDate, term, tmpIsDate); // checks if the market string contains a date or a period
127  if (!tmpIsDate)
128  expiryDate = cal.adjust(asof + term, bdc);
129  return expiryDate;
130 }

133 boost::shared_ptr<MarketDatum> parseMarketDatum(const Date& asof, const string& datumName, const Real& value) {
134 
135  vector<string> tokens;
136  boost::split(tokens, datumName, boost::is_any_of("/"));
137  QL_REQUIRE(tokens.size() > 2, "more than 2 tokens expected in " << datumName);
138 
139  MarketDatum::InstrumentType instrumentType = parseInstrumentType(tokens[0]);
140  MarketDatum::QuoteType quoteType = parseQuoteType(tokens[1]);
141 
142  switch (instrumentType) {
143 
144  case MarketDatum::InstrumentType::ZERO: {
145  // ZERO/RATE/EUR/EUR1D/A365/1Y
146  QL_REQUIRE(quoteType == MarketDatum::QuoteType::RATE || quoteType == MarketDatum::QuoteType::YIELD_SPREAD,
147  "Invalid quote type for " << datumName);
148  QL_REQUIRE(tokens.size() == 6, "6 tokens expected in " << datumName);
149  const string& ccy = tokens[2];
150  DayCounter dc = parseDayCounter(tokens[4]);
151  // token 5 can be a date, or tenor
152  Date date = Date();
153  Period tenor = Period();
154  bool isDate;
155  parseDateOrPeriod(tokens[5], date, tenor, isDate);
156  return boost::make_shared<ZeroQuote>(value, asof, datumName, quoteType, ccy, date, dc, tenor);
157  }
158 
159  case MarketDatum::InstrumentType::DISCOUNT: {
160  // DISCOUNT/RATE/EUR/EUR1D/1Y
161  // DISCOUNT/RATE/EUR/EUR1D/2016-12-15
162  QL_REQUIRE(tokens.size() == 5, "5 tokens expected in " << datumName);
163  const string& ccy = tokens[2];
164  // token 4 can be a date, or tenor
165  Date date = Date();
166  Period tenor = Period();
167  bool isDate;
168  parseDateOrPeriod(tokens[4], date, tenor, isDate);
169  if (!isDate) {
170  // we can't assume any calendar here, so we do the minimal adjustment with a weekend only calendar
171  QL_REQUIRE(tenor != Period(), "neither date nor tenor recognised");
172  date = WeekendsOnly().adjust(asof + tenor);
173  }
174  return boost::make_shared<DiscountQuote>(value, asof, datumName, quoteType, ccy, date);
175  }
176 
177  case MarketDatum::InstrumentType::MM: {
178  QL_REQUIRE(tokens.size() == 5 || tokens.size() == 6, "5 or 6 tokens expected in " << datumName);
179  const string& ccy = tokens[2];
180  Size offset = 0;
181  string indexName;
182  if (tokens.size() == 6) {
183  indexName = tokens[3];
184  offset = 1;
185  }
186  Period fwdStart = parsePeriod(tokens[3 + offset]);
187  Period term = parsePeriod(tokens[4 + offset]);
188  return boost::make_shared<MoneyMarketQuote>(value, asof, datumName, quoteType, ccy, fwdStart, term, indexName);
189  }
190 
191  case MarketDatum::InstrumentType::MM_FUTURE: {
192  QL_REQUIRE(tokens.size() == 6, "6 tokens expected in " << datumName);
193  const string& ccy = tokens[2];
194  const string& expiry = tokens[3];
195  const string& contract = tokens[4];
196  Period term = parsePeriod(tokens[5]);
197  return boost::make_shared<MMFutureQuote>(value, asof, datumName, quoteType, ccy, expiry, contract, term);
198  }
199 
200  case MarketDatum::InstrumentType::OI_FUTURE: {
201  QL_REQUIRE(tokens.size() == 6, "6 tokens expected in " << datumName);
202  const string& ccy = tokens[2];
203  const string& expiry = tokens[3];
204  const string& contract = tokens[4];
205  Period term = parsePeriod(tokens[5]);
206  return boost::make_shared<OIFutureQuote>(value, asof, datumName, quoteType, ccy, expiry, contract, term);
207  }
208 
209  case MarketDatum::InstrumentType::FRA: {
210  QL_REQUIRE(tokens.size() == 5, "5 tokens expected in " << datumName);
211  const string& ccy = tokens[2];
212  Period fwdStart = parsePeriod(tokens[3]);
213  Period term = parsePeriod(tokens[4]);
214  return boost::make_shared<FRAQuote>(value, asof, datumName, quoteType, ccy, fwdStart, term);
215  }
216 
217  case MarketDatum::InstrumentType::IMM_FRA: {
218  QL_REQUIRE(tokens.size() == 5, "5 tokens expected in " << datumName);
219  const string& ccy = tokens[2];
220  string imm1 = tokens[3];
221  string imm2 = tokens[4];
222  unsigned int m1 = parseInteger(imm1);
223  unsigned int m2 = parseInteger(imm2);
224  QL_REQUIRE(m2 > m1, "Second IMM date must be after the first in " << datumName);
225  return boost::make_shared<ImmFraQuote>(value, asof, datumName, quoteType, ccy, m1, m2);
226  }
227 
228  case MarketDatum::InstrumentType::IR_SWAP: {
229  QL_REQUIRE(tokens.size() == 6 || tokens.size() == 7, "6 or 7 tokens expected in " << datumName);
230  const string& ccy = tokens[2];
231  Size offset = 0;
232  string indexName;
233  if (tokens.size() == 7) {
234  indexName = tokens[3];
235  offset = 1;
236  }
237  Period fwdStart = parsePeriod(tokens[3 + offset]);
238  Period tenor = parsePeriod(tokens[4 + offset]);
239  Period term = parsePeriod(tokens[5 + offset]);
240  return boost::make_shared<SwapQuote>(value, asof, datumName, quoteType, ccy, fwdStart, term, tenor, indexName);
241  }
242 
243  case MarketDatum::InstrumentType::BASIS_SWAP: {
244  // An optional identifier as a penultimate token supports the following two versions:
245  // BASIS_SWAP/BASIS_SPREAD/3M/1D/USD/5Y
246  // BASIS_SWAP/BASIS_SPREAD/3M/1D/USD/foobar/5Y
247  QL_REQUIRE(tokens.size() == 6 || tokens.size() == 7, "Either 6 or 7 tokens expected in " << datumName);
248  Period flatTerm = parsePeriod(tokens[2]);
249  Period term = parsePeriod(tokens[3]);
250  const string& ccy = tokens[4];
251  Period maturity;
252  if (tokens.size() == 7) {
253  maturity = parsePeriod(tokens[6]);
254  } else {
255  maturity = parsePeriod(tokens[5]);
256  }
257  return boost::make_shared<BasisSwapQuote>(value, asof, datumName, quoteType, flatTerm, term, ccy, maturity);
258  }
259 
260  case MarketDatum::InstrumentType::BMA_SWAP: {
261  QL_REQUIRE(tokens.size() == 5, "5 tokens expected in " << datumName);
262  const string& ccy = tokens[2];
263  Period term = parsePeriod(tokens[3]);
264  Period maturity = parsePeriod(tokens[4]);
265  return boost::make_shared<BMASwapQuote>(value, asof, datumName, quoteType, term, ccy, maturity);
266  }
267 
268  case MarketDatum::InstrumentType::CC_BASIS_SWAP: {
269  QL_REQUIRE(tokens.size() == 7, "7 tokens expected in " << datumName);
270  const string& flatCcy = tokens[2];
271  Period flatTerm = parsePeriod(tokens[3]);
272  const string& ccy = tokens[4];
273  Period term = parsePeriod(tokens[5]);
274  Period maturity = parsePeriod(tokens[6]);
275  return boost::make_shared<CrossCcyBasisSwapQuote>(value, asof, datumName, quoteType, flatCcy, flatTerm, ccy,
276  term, maturity);
277  }
278 
279  case MarketDatum::InstrumentType::CC_FIX_FLOAT_SWAP: {
280  // CC_FIX_FLOAT_SWAP/RATE/USD/3M/TRY/1Y/5Y
281  QL_REQUIRE(tokens.size() == 7, "7 tokens expected in " << datumName);
282  Period floatTenor = parsePeriod(tokens[3]);
283  Period fixedTenor = parsePeriod(tokens[5]);
284  Period maturity = parsePeriod(tokens[6]);
285  return boost::make_shared<CrossCcyFixFloatSwapQuote>(value, asof, datumName, quoteType, tokens[2], floatTenor,
286  tokens[4], fixedTenor, maturity);
287  }
288 
289  case MarketDatum::InstrumentType::CDS: {
290  // CDS/[CONV_]CREDIT_SPREAD/Name/Seniority/ccy/term
291  // CDS/[CONV_]CREDIT_SPREAD/Name/Seniority/ccy/term/runningSpread
292  // CDS/[CONV_]CREDIT_SPREAD/Name/Seniority/ccy/doc/term
293  // CDS/[CONV_]CREDIT_SPREAD/Name/Seniority/ccy/doc/term/runningSpread
294  // CDS/PRICE/Name/Seniority/ccy/term
295  // CDS/PRICE/Name/Seniority/ccy/term/runningSpread
296  // CDS/PRICE/Name/Seniority/ccy/doc/term
297  // CDS/PRICE/Name/Seniority/ccy/doc/term/runningSpread
298  QL_REQUIRE(tokens.size() == 6 || tokens.size() == 7 || tokens.size() == 8,
299  "6, 7 or 8 tokens expected in " << datumName);
300  const string& underlyingName = tokens[2];
301  const string& seniority = tokens[3];
302  const string& ccy = tokens[4];
303 
304  string docClause;
305  Period term;
306  Real runningSpread = Null<Real>();
307  if (tokens.size() == 6) {
308  term = parsePeriod(tokens[5]);
309  } else if (tokens.size() == 8) {
310  docClause = tokens[5];
311  term = parsePeriod(tokens[6]);
312  runningSpread = parseReal(tokens[7]) / 10000;
313  } else {
314  // 7 tokens => [5]/[6] = doc/term or term/runningSpread
315  CdsDocClause cdsDocClause;
316  if (tryParse<CdsDocClause>(tokens[5], cdsDocClause, &parseCdsDocClause)) {
317  docClause = tokens[5];
318  term = parsePeriod(tokens[6]);
319  } else {
320  term = parsePeriod(tokens[5]);
321  runningSpread = parseReal(tokens[6]) / 10000;
322  }
323  }
324 
325  return boost::make_shared<CdsQuote>(value, asof, datumName, quoteType, underlyingName,
326  seniority, ccy, term, docClause, runningSpread);
327  }
328 
329  case MarketDatum::InstrumentType::HAZARD_RATE: {
330  QL_REQUIRE(tokens.size() == 6 || tokens.size() == 7, "6 or 7 tokens expected in " << datumName);
331  const string& underlyingName = tokens[2];
332  const string& seniority = tokens[3];
333  const string& ccy = tokens[4];
334  string docClause = tokens.size() == 7 ? tokens[5] : "";
335  Period term = parsePeriod(tokens.back());
336  return boost::make_shared<HazardRateQuote>(value, asof, datumName, underlyingName, seniority, ccy, term,
337  docClause);
338  }
339 
340  case MarketDatum::InstrumentType::RECOVERY_RATE: {
341  QL_REQUIRE(tokens.size() == 3 || tokens.size() == 5 || tokens.size() == 6,
342  "3, 5 or 6 tokens expected in " << datumName);
343  const string& underlyingName = tokens[2]; // issuer name for CDS, security ID for bond specific RRs
344  string seniority = "";
345  string ccy = "";
346  string docClause = "";
347  if (tokens.size() >= 5) {
348  // CDS
349  seniority = tokens[3];
350  ccy = tokens[4];
351  if (tokens.size() == 6)
352  docClause = tokens[5];
353  }
354  return boost::make_shared<RecoveryRateQuote>(value, asof, datumName, underlyingName, seniority, ccy, docClause);
355  }
356 
357  case MarketDatum::InstrumentType::CAPFLOOR: {
358  QL_REQUIRE(tokens.size() == 8 || tokens.size() == 4, "Either 4 or 8 tokens expected in " << datumName);
359  const string& ccy = tokens[2];
360  if (tokens.size() == 8) {
361  Period term = parsePeriod(tokens[3]);
362  Period tenor = parsePeriod(tokens[4]);
363  bool atm = parseBool(tokens[5].c_str());
364  bool relative = parseBool(tokens[6].c_str());
365  Real strike = parseReal(tokens[7]);
366  return boost::make_shared<CapFloorQuote>(value, asof, datumName, quoteType, ccy, term, tenor, atm, relative,
367  strike);
368  } else {
369  Period indexTenor = parsePeriod(tokens[3]);
370  return boost::make_shared<CapFloorShiftQuote>(value, asof, datumName, quoteType, ccy, indexTenor);
371  }
372  }
373 
374  case MarketDatum::InstrumentType::SWAPTION: {
375  QL_REQUIRE(tokens.size() == 4 || tokens.size() == 6 || tokens.size() == 7,
376  "4, 6 or 7 tokens expected in " << datumName);
377  const string& ccy = tokens[2];
378  Period expiry = tokens.size() >= 6 ? parsePeriod(tokens[3]) : Period(0 * QuantLib::Days);
379  Period term = tokens.size() >= 6 ? parsePeriod(tokens[4]) : parsePeriod(tokens[3]);
380  if (tokens.size() >= 6) { // volatility
381  const string& dimension = tokens[5];
382  Real strike = 0.0;
383  if (dimension == "ATM")
384  QL_REQUIRE(tokens.size() == 6, "6 tokens expected in ATM quote " << datumName);
385  else if (dimension == "Smile") {
386  QL_REQUIRE(tokens.size() == 7, "7 tokens expected in Smile quote " << datumName);
387  strike = parseReal(tokens[6]);
388  } else
389  QL_FAIL("Swaption vol quote dimension " << dimension << " not recognised");
390  return boost::make_shared<SwaptionQuote>(value, asof, datumName, quoteType, ccy, expiry, term, dimension,
391  strike);
392  } else { // SLN volatility shift
393  return boost::make_shared<SwaptionShiftQuote>(value, asof, datumName, quoteType, ccy, term);
394  }
395  }
396 
397  case MarketDatum::InstrumentType::BOND_OPTION: {
398  QL_REQUIRE(tokens.size() == 4 || tokens.size() == 6, "4 or 6 tokens expected in " << datumName);
399  const string& qualifier = tokens[2];
400  Period expiry = tokens.size() == 6 ? parsePeriod(tokens[3]) : Period(0 * QuantLib::Days);
401  Period term = tokens.size() == 6 ? parsePeriod(tokens[4]) : parsePeriod(tokens[3]);
402  if (tokens.size() == 6) { // volatility
403  QL_REQUIRE(tokens[5] == "ATM", "only ATM allowed for bond option quotes");
404  return boost::make_shared<BondOptionQuote>(value, asof, datumName, quoteType, qualifier, expiry, term);
405  } else { // SLN volatility shift
406  return boost::make_shared<BondOptionShiftQuote>(value, asof, datumName, quoteType, qualifier, term);
407  }
408  }
409 
410  case MarketDatum::InstrumentType::FX_SPOT: {
411  QL_REQUIRE(tokens.size() == 4, "4 tokens expected in " << datumName);
412  const string& unitCcy = tokens[2];
413  const string& ccy = tokens[3];
414  return boost::make_shared<FXSpotQuote>(value, asof, datumName, quoteType, unitCcy, ccy);
415  }
416 
417  case MarketDatum::InstrumentType::FX_FWD: {
418  QL_REQUIRE(tokens.size() == 5, "5 tokens expected in " << datumName);
419  const string& unitCcy = tokens[2];
420  const string& ccy = tokens[3];
421  Period term = parsePeriod(tokens[4]);
422  return boost::make_shared<FXForwardQuote>(value, asof, datumName, quoteType, unitCcy, ccy, term);
423  }
424 
425  case MarketDatum::InstrumentType::FX_OPTION: {
426  QL_REQUIRE(tokens.size() == 6, "6 tokens expected in " << datumName);
427  const string& unitCcy = tokens[2];
428  const string& ccy = tokens[3];
429  Period expiry = parsePeriod(tokens[4]);
430  const string& strike = tokens[5];
431  return boost::make_shared<FXOptionQuote>(value, asof, datumName, quoteType, unitCcy, ccy, expiry, strike);
432  }
433 
434  case MarketDatum::InstrumentType::ZC_INFLATIONSWAP: {
435  QL_REQUIRE(tokens.size() == 4, "4 tokens expected in " << datumName);
436  const string& index = tokens[2];
437  Period term = parsePeriod(tokens[3]);
438  return boost::make_shared<ZcInflationSwapQuote>(value, asof, datumName, index, term);
439  }
440 
441  case MarketDatum::InstrumentType::YY_INFLATIONSWAP: {
442  QL_REQUIRE(tokens.size() == 4, "4 tokens expected in " << datumName);
443  const string& index = tokens[2];
444  Period term = parsePeriod(tokens[3]);
445  return boost::make_shared<YoYInflationSwapQuote>(value, asof, datumName, index, term);
446  }
447 
448  case MarketDatum::InstrumentType::ZC_INFLATIONCAPFLOOR: {
449  QL_REQUIRE(tokens.size() == 6, "6 tokens expected in " << datumName);
450  const string& index = tokens[2];
451  Period term = parsePeriod(tokens[3]);
452  QL_REQUIRE(tokens[4] == "C" || tokens[4] == "F",
453  "excepted C or F for Cap or Floor at position 5 in " << datumName);
454  bool isCap = tokens[4] == "C";
455  string strike = tokens[5];
456  return boost::make_shared<ZcInflationCapFloorQuote>(value, asof, datumName, quoteType, index, term, isCap,
457  strike);
458  }
459 
460  case MarketDatum::InstrumentType::YY_INFLATIONCAPFLOOR: {
461  QL_REQUIRE(tokens.size() == 6, "6 tokens expected in " << datumName);
462  const string& index = tokens[2];
463  Period term = parsePeriod(tokens[3]);
464  QL_REQUIRE(tokens[4] == "C" || tokens[4] == "F",
465  "excepted C or F for Cap or Floor at position 5 in " << datumName);
466  bool isCap = tokens[4] == "C";
467  string strike = tokens[5];
468  return boost::make_shared<YyInflationCapFloorQuote>(value, asof, datumName, quoteType, index, term, isCap,
469  strike);
470  }
471 
472  case MarketDatum::InstrumentType::SEASONALITY: {
473  QL_REQUIRE(tokens.size() == 5, "5 tokens expected in " << datumName);
474  const string& index = tokens[3];
475  const string& type = tokens[2];
476  const string& month = tokens[4];
477  return boost::make_shared<SeasonalityQuote>(value, asof, datumName, index, type, month);
478  }
479  case MarketDatum::InstrumentType::EQUITY_SPOT: {
480  QL_REQUIRE(tokens.size() == 4, "4 tokens expected in " << datumName);
481  QL_REQUIRE(quoteType == MarketDatum::QuoteType::PRICE, "Invalid quote type for " << datumName);
482  const string& equityName = tokens[2];
483  const string& ccy = tokens[3];
484  return boost::make_shared<EquitySpotQuote>(value, asof, datumName, quoteType, equityName, ccy);
485  }
486 
487  case MarketDatum::InstrumentType::EQUITY_FWD: {
488  QL_REQUIRE(tokens.size() == 5, "5 tokens expected in " << datumName);
489  QL_REQUIRE(quoteType == MarketDatum::QuoteType::PRICE, "Invalid quote type for " << datumName);
490  const string& equityName = tokens[2];
491  const string& ccy = tokens[3];
492  Date expiryDate = getDateFromDateOrPeriod(tokens[4], asof);
493  return boost::make_shared<EquityForwardQuote>(value, asof, datumName, quoteType, equityName, ccy, expiryDate);
494  }
495 
496  case MarketDatum::InstrumentType::EQUITY_DIVIDEND: {
497  QL_REQUIRE(tokens.size() == 5, "5 tokens expected in " << datumName);
498  QL_REQUIRE(quoteType == MarketDatum::QuoteType::RATE, "Invalid quote type for " << datumName);
499  const string& equityName = tokens[2];
500  const string& ccy = tokens[3];
501  Date tenorDate = getDateFromDateOrPeriod(tokens[4], asof);
502  return boost::make_shared<EquityDividendYieldQuote>(value, asof, datumName, quoteType, equityName, ccy,
503  tenorDate);
504  }
505 
506  case MarketDatum::InstrumentType::EQUITY_OPTION: {
507  QL_REQUIRE(tokens.size() >= 6 || tokens.size() <= 9, "6 - 9 tokens expected in " << datumName);
508  QL_REQUIRE(quoteType == MarketDatum::QuoteType::RATE_LNVOL || quoteType == MarketDatum::QuoteType::PRICE,
509  "Invalid quote type for " << datumName);
510  const string& equityName = tokens[2];
511  const string& ccy = tokens[3];
512  string expiryString = tokens[4];
513  // do we have a call or put flag as the last token?
514  bool hasCallPutToken = tokens.back() == "C" || tokens.back() == "P";
515  // the following tokens represent the strike, except the last one, if we have a call/put token
516  string strikeStr;
517  for(Size i=5; i < tokens.size() - (hasCallPutToken ? 1 : 0); ++i) {
518  strikeStr += (i > 5 ? "/" : "") + tokens[i];
519  }
520  boost::shared_ptr<BaseStrike> strike;
521  // we support ATM, ATMF as aliases for ATM/AtmSpot, ATM/AtmFwd, plus absolute strikes and MNY/[Spot/Fwd]/1.2
522  if(strikeStr == "ATM")
523  strike = boost::make_shared<AtmStrike>(QuantLib::DeltaVolQuote::AtmType::AtmSpot);
524  else if(strikeStr == "ATMF")
525  strike = boost::make_shared<AtmStrike>(QuantLib::DeltaVolQuote::AtmType::AtmFwd);
526  else
527  strike = parseBaseStrike(strikeStr);
528  bool isCall = true;
529  if (hasCallPutToken) {
530  QL_REQUIRE(tokens.back() == "C" || tokens.back() == "P",
531  "excepted C or P for Call or Put at position " << tokens.size() << " in " << datumName);
532  isCall = tokens.back() == "C";
533  }
534 
535  // note how we only store the expiry string - to ensure we can support both Periods and Dates being specified in
536  // the vol curve-config.
537  return boost::make_shared<EquityOptionQuote>(value, asof, datumName, quoteType, equityName, ccy, expiryString,
538  strike, isCall);
539  }
540 
541  case MarketDatum::InstrumentType::BOND: {
542  QL_REQUIRE(tokens.size() == 3, "3 tokens expected in " << datumName);
543  const string& securityID = tokens[2];
544  if (quoteType == MarketDatum::QuoteType::YIELD_SPREAD)
545  return boost::make_shared<SecuritySpreadQuote>(value, asof, datumName, securityID);
546  else if (quoteType == MarketDatum::QuoteType::PRICE)
547  return boost::make_shared<BondPriceQuote>(value, asof, datumName, securityID);
548  }
549 
550  case MarketDatum::InstrumentType::CDS_INDEX: {
551  QL_REQUIRE(tokens.size() == 5, "5 tokens expected in " << datumName);
552  QL_REQUIRE(quoteType == MarketDatum::QuoteType::BASE_CORRELATION, "Invalid quote type for " << datumName);
553  const string& cdsIndexName = tokens[2];
554  Period term = parsePeriod(tokens[3]);
555  Real detachmentPoint = parseReal(tokens[4]);
556  return boost::make_shared<BaseCorrelationQuote>(value, asof, datumName, quoteType, cdsIndexName, term,
557  detachmentPoint);
558  }
559 
560  case MarketDatum::InstrumentType::INDEX_CDS_OPTION: {
561 
562  // Expects the following form. The strike is optional. The index term is optional for backwards compatibility.
563  // INDEX_CDS_OPTION/RATE_LNVOL/<INDEX_NAME>[/<INDEX_TERM>]/<EXPIRY>[/<STRIKE>]
564  QL_REQUIRE(tokens.size() >= 4 || tokens.size() <= 6, "4, 5 or 6 tokens expected in " << datumName);
565  QL_REQUIRE(quoteType == MarketDatum::QuoteType::RATE_LNVOL, "Invalid quote type for " << datumName);
566 
567  boost::shared_ptr<Expiry> expiry;
568  boost::shared_ptr<BaseStrike> strike;
569  string indexTerm;
570  if (tokens.size() == 6) {
571  // We have been given an index term, an expiry and a strike.
572  indexTerm = tokens[3];
573  expiry = parseExpiry(tokens[4]);
574  strike = parseBaseStrike(tokens[5]);
575  } else if (tokens.size() == 5) {
576  // We have been given either 1) an index term and an expiry or 2) an expiry and a strike.
577  // If the last token is a number, we have 2) an expiry and a strike.
578  Real tmp;
579  if (tryParseReal(tokens[4], tmp)) {
580  expiry = parseExpiry(tokens[3]);
581  strike = parseBaseStrike(tokens[4]);
582  } else {
583  indexTerm = tokens[3];
584  expiry = parseExpiry(tokens[4]);
585  }
586  } else {
587  // We have just been given the expiry.
588  expiry = parseExpiry(tokens[3]);
589  }
590 
591  return boost::make_shared<IndexCDSOptionQuote>(value, asof, datumName, tokens[2], expiry, indexTerm, strike);
592  }
593 
594  case MarketDatum::InstrumentType::COMMODITY_SPOT: {
595  QL_REQUIRE(tokens.size() == 4, "4 tokens expected in " << datumName);
596  QL_REQUIRE(quoteType == MarketDatum::QuoteType::PRICE, "Invalid quote type for " << datumName);
597 
598  return boost::make_shared<CommoditySpotQuote>(value, asof, datumName, quoteType, tokens[2], tokens[3]);
599  }
600 
601  case MarketDatum::InstrumentType::COMMODITY_FWD: {
602  // Expects the following form:
603  // COMMODITY_FWD/PRICE/<COMDTY_NAME>/<CCY>/<DATE/TENOR>
604  QL_REQUIRE(tokens.size() == 5, "5 tokens expected in " << datumName);
605  QL_REQUIRE(quoteType == MarketDatum::QuoteType::PRICE, "Invalid quote type for " << datumName);
606 
607  // The last token can be a string defining a special tenor i.e. ON, TN or SN
608  if (tokens[4] == "ON") {
609  return boost::make_shared<CommodityForwardQuote>(value, asof, datumName, quoteType, tokens[2], tokens[3],
610  1 * Days, 0 * Days);
611  } else if (tokens[4] == "TN") {
612  return boost::make_shared<CommodityForwardQuote>(value, asof, datumName, quoteType, tokens[2], tokens[3],
613  1 * Days, 1 * Days);
614  } else if (tokens[4] == "SN") {
615  return boost::make_shared<CommodityForwardQuote>(value, asof, datumName, quoteType, tokens[2], tokens[3],
616  1 * Days);
617  }
618 
619  // The last token can be a date or a standard tenor
620  Date date;
621  Period tenor;
622  bool isDate;
623  parseDateOrPeriod(tokens[4], date, tenor, isDate);
624 
625  if (isDate) {
626  return boost::make_shared<CommodityForwardQuote>(value, asof, datumName, quoteType, tokens[2], tokens[3],
627  date);
628  } else {
629  return boost::make_shared<CommodityForwardQuote>(value, asof, datumName, quoteType, tokens[2], tokens[3],
630  tenor);
631  }
632  }
633 
634  case MarketDatum::InstrumentType::COMMODITY_OPTION: {
635  // Expects one of the following forms:
636  // COMMODITY_OPTION/<QT>/<COMDTY_NAME>/<CCY>/<EXPIRY>/<STRIKE>
637  // COMMODITY_OPTION/<QT>/<COMDTY_NAME>/<CCY>/<EXPIRY>/<STRIKE>/<OT>
638  // where QT = RATE_LNVOL or PRICE and OT = C (for Call) or P (for Put)
639  using QT = MarketDatum::QuoteType;
640  QL_REQUIRE(tokens.size() >= 6, "At least 6 tokens expected in " << datumName);
641  QL_REQUIRE(quoteType == QT::RATE_LNVOL || quoteType == QT::PRICE,
642  "Quote type for " << datumName << " should be 'RATE_LNVOL' or 'PRICE'");
643 
644  boost::shared_ptr<Expiry> expiry = parseExpiry(tokens[4]);
645 
646  // If the last token is C or P, process it and update bounds of strike portion.
647  auto itStkBeg = tokens.begin() + 5;
648  auto itStkEnd = tokens.end();
649  Option::Type optionType = Option::Call;
650  if (tokens.back() == "C" || tokens.back() == "P") {
651  if (tokens.back() == "P")
652  optionType = Option::Put;
653  itStkEnd = prev(itStkEnd);
654  }
655 
656  // Parse the strike
657  QL_REQUIRE(itStkBeg != itStkEnd, "");
658  string strStrike = boost::algorithm::join(boost::make_iterator_range(itStkBeg, itStkEnd), "/");
659  boost::shared_ptr<BaseStrike> strike = parseBaseStrike(strStrike);
660 
661  return boost::make_shared<CommodityOptionQuote>(value, asof, datumName, quoteType, tokens[2],
662  tokens[3], expiry, strike, optionType);
663  }
664 
665  case MarketDatum::InstrumentType::CORRELATION: {
666  // Expects the following form:
667  // CORRELATION/RATE/<INDEX1>/<INDEX2>/<TENOR>/<STRIKE>
668  QL_REQUIRE(tokens.size() == 6, "6 tokens expected in " << datumName);
669  QL_REQUIRE(quoteType == MarketDatum::QuoteType::RATE || quoteType == MarketDatum::QuoteType::PRICE,
670  "Quote type for " << datumName << " should be 'CORRELATION' or 'PRICE'");
671 
672  return boost::make_shared<CorrelationQuote>(value, asof, datumName, quoteType, tokens[2], tokens[3], tokens[4],
673  tokens[5]);
674  }
675 
676  case MarketDatum::InstrumentType::CPR: {
677  QL_REQUIRE(tokens.size() == 3, "3 tokens expected in " << datumName);
678  const string& securityID = tokens[2];
679  QL_REQUIRE(quoteType == MarketDatum::QuoteType::RATE, "Invalid quote type for " << datumName);
680  return boost::make_shared<CPRQuote>(value, asof, datumName, securityID);
681  }
682 
683  default:
684  QL_FAIL("Cannot convert \"" << datumName << "\" to MarketDatum");
685  } // switch instrument type
