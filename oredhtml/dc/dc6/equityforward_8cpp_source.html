40 void EquityForward::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
41  Currency ccy = parseCurrencyWithMinors(currency_);
42 
43  // get the equity currency from the market
44  Currency equityCcy = engineFactory->market()->equityCurve(eqName())->currency();
45 
46  // ensure forward currency matches the equity currency
47  QL_REQUIRE(!equityCcy.empty(), "No equity currency in equityCurve for equity " << eqName());
48  QL_REQUIRE(ccy == equityCcy, "EquityForward currency " << ccy << " does not match equity currency " << equityCcy << " for trade " << id());
49 
50  // convert strike to the major currency if needed
51  Real strike;
52  if (!strikeCurrency_.empty()) {
53  Currency strikeCcy = parseCurrencyWithMinors(strikeCurrency_);
54  strike = convertMinorToMajorCurrency(strikeCurrency_, strike_);
55  // ensure strike currency matches equity currency
56  QL_REQUIRE(strikeCcy == equityCcy, "Stike currency " << ccy << " does not match equity currency " << equityCcy << " for trade " << id());
57  } else {
58  WLOG("No Strike Currency provide for trade " << id() << ", assuming trade currency " << ccy);
59  strike = convertMinorToMajorCurrency(currency_, strike_);
60  }
61 
62  QuantLib::Position::Type longShort = parsePositionType(longShort_);
63  Date maturity = parseDate(maturityDate_);
64 
65  string name = eqName();
66 
67  boost::shared_ptr<Instrument> inst =
68  boost::make_shared<QuantExt::EquityForward>(name, ccy, longShort, quantity_, maturity, strike);
69 
70  // Pricing engine
71  boost::shared_ptr<EngineBuilder> builder = engineFactory->builder(tradeType_);
72  QL_REQUIRE(builder, "No builder found for " << tradeType_);
73  boost::shared_ptr<EquityForwardEngineBuilder> eqFwdBuilder =
74  boost::dynamic_pointer_cast<EquityForwardEngineBuilder>(builder);
75  inst->setPricingEngine(eqFwdBuilder->engine(name, ccy));
76 
77  // set up other Trade details
78  instrument_ = boost::shared_ptr<InstrumentWrapper>(new VanillaInstrument(inst));
79  npvCurrency_ = ccy.code();
80  maturity_ = maturity;
81  // Notional - we really need todays spot to get the correct notional.
82  // But rather than having it move around we use strike * quantity
83  notional_ = strike * quantity_;
84  notionalCurrency_ = ccy.code();
85 
86  additionalData_["underlyingSecurityId"] = name;
87  additionalData_["strike"] = strike;
88  additionalData_["strikeCurrency"] = strikeCurrency_;
89  additionalData_["quantity"] = quantity_;
90 }

92 void EquityForward::fromXML(XMLNode* node) {
93  Trade::fromXML(node);
94  XMLNode* eNode = XMLUtils::getChildNode(node, "EquityForwardData");
95 
96  longShort_ = XMLUtils::getChildValue(eNode, "LongShort", true);
97  maturityDate_ = XMLUtils::getChildValue(eNode, "Maturity", true);
98  XMLNode* tmp = XMLUtils::getChildNode(eNode, "Underlying");
99  if (!tmp)
100  tmp = XMLUtils::getChildNode(eNode, "Name");
101  equityUnderlying_.fromXML(tmp);
102  currency_ = XMLUtils::getChildValue(eNode, "Currency", true);
103  strike_ = XMLUtils::getChildValueAsDouble(eNode, "Strike", true);
