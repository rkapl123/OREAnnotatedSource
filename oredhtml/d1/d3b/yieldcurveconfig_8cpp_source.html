37 YieldCurveSegment::Type parseYieldCurveSegment(const string& s) {
38  if (iequals(s, "Zero"))
39  return YieldCurveSegment::Type::Zero;
40  else if (iequals(s, "Zero Spread"))
41  return YieldCurveSegment::Type::ZeroSpread;
42  else if (iequals(s, "Discount"))
43  return YieldCurveSegment::Type::Discount;
44  else if (iequals(s, "Deposit"))
45  return YieldCurveSegment::Type::Deposit;
46  else if (iequals(s, "FRA"))
47  return YieldCurveSegment::Type::FRA;
48  else if (iequals(s, "Future"))
49  return YieldCurveSegment::Type::Future;
50  else if (iequals(s, "OIS"))
51  return YieldCurveSegment::Type::OIS;
52  else if (iequals(s, "Swap"))
53  return YieldCurveSegment::Type::Swap;
54  else if (iequals(s, "Average OIS"))
55  return YieldCurveSegment::Type::AverageOIS;
56  else if (iequals(s, "Tenor Basis Swap"))
57  return YieldCurveSegment::Type::TenorBasis;
58  else if (iequals(s, "Tenor Basis Two Swaps"))
59  return YieldCurveSegment::Type::TenorBasisTwo;
60  else if (iequals(s, "BMA Basis Swap"))
61  return YieldCurveSegment::Type::BMABasis;
62  else if (iequals(s, "FX Forward"))
63  return YieldCurveSegment::Type::FXForward;
64  else if (iequals(s, "Cross Currency Basis Swap"))
65  return YieldCurveSegment::Type::CrossCcyBasis;
66  else if (iequals(s, "Cross Currency Fix Float Swap"))
67  return YieldCurveSegment::Type::CrossCcyFixFloat;
68  else if (iequals(s, "Discount Ratio"))
69  return YieldCurveSegment::Type::DiscountRatio;
70  else if (iequals(s, "FittedBond"))
71  return YieldCurveSegment::Type::FittedBond;
72  else if (iequals(s, "Yield Plus Default"))
73  return YieldCurveSegment::Type::YieldPlusDefault;
74  else if (iequals(s, "Weighted Average"))
75  return YieldCurveSegment::Type::WeightedAverage;
76  else if (iequals(s, "Ibor Fallback"))
77  return YieldCurveSegment::Type::IborFallback;
78  QL_FAIL("Yield curve segment type " << s << " not recognized");
79 }

233 void YieldCurveConfig::fromXML(XMLNode* node) {
234 
235  XMLUtils::checkNode(node, "YieldCurve");
236 
237  // Read in the mandatory nodes.
238  curveID_ = XMLUtils::getChildValue(node, "CurveId", true);
239  curveDescription_ = XMLUtils::getChildValue(node, "CurveDescription", true);
240  currency_ = XMLUtils::getChildValue(node, "Currency", true);
241  discountCurveID_ = XMLUtils::getChildValue(node, "DiscountCurve", true);
242 
243  // Read in the segments.
244  XMLNode* segmentsNode = XMLUtils::getChildNode(node, "Segments");
245  if (segmentsNode) {
246  for (XMLNode* child = XMLUtils::getChildNode(segmentsNode); child; child = XMLUtils::getNextSibling(child)) {
247 
248  boost::shared_ptr<YieldCurveSegment> segment;
249  string childName = XMLUtils::getNodeName(child);
250 
251  if (childName == "Direct") {
252  segment.reset(new DirectYieldCurveSegment());
253  } else if (childName == "Simple") {
254  segment.reset(new SimpleYieldCurveSegment());
255  } else if (childName == "AverageOIS") {
256  segment.reset(new AverageOISYieldCurveSegment());
257  } else if (childName == "TenorBasis") {
258  segment.reset(new TenorBasisYieldCurveSegment());
259  } else if (childName == "CrossCurrency") {
260  segment.reset(new CrossCcyYieldCurveSegment());
261  } else if (childName == "ZeroSpread") {
262  segment.reset(new ZeroSpreadedYieldCurveSegment());
263  } else if (childName == "DiscountRatio") {
264  segment.reset(new DiscountRatioYieldCurveSegment());
265  } else if (childName == "FittedBond") {
266  segment.reset(new FittedBondYieldCurveSegment());
267  } else if (childName == "WeightedAverage") {
268  segment.reset(new WeightedAverageYieldCurveSegment());
269  } else if (childName == "YieldPlusDefault") {
270  segment.reset(new YieldPlusDefaultYieldCurveSegment());
271  } else if(childName == "IborFallback"){
272  segment.reset(new IborFallbackCurveSegment());
273  } else {
274  QL_FAIL("Yield curve segment node name '" << childName << "' not recognized.");
275  }
276 
277  if (segment) {
278  try {
279  segment->fromXML(child);
280  } catch (std::exception& ex) {
281  ALOG("Exception parsing yield curve segment XML Node, name = " << childName << " and curveID = "
282  << curveID_ << " : " << ex.what());
283  }
284  } else {
285  LOG("Unable to build yield curve segment for name = " << childName << " and curveID = " << curveID_);
286  }
287  curveSegments_.push_back(segment);
288  }
289  } else {
290  QL_FAIL("No Segments node in XML doc for yield curve ID = " << curveID_);
291  }
292 
293  // Read in the optional nodes.
294 
295  // Empty strings if not there (or if there and empty).
296  interpolationVariable_ = XMLUtils::getChildValue(node, "InterpolationVariable", false);
297  interpolationMethod_ = XMLUtils::getChildValue(node, "InterpolationMethod", false);
298  zeroDayCounter_ = XMLUtils::getChildValue(node, "YieldCurveDayCounter", false);
299 
300  // Add hardcoded defaults for now.
301  if (interpolationVariable_.empty()) {
302  interpolationVariable_ = "Discount";
303  }
304  if (interpolationMethod_.empty()) {
305  interpolationMethod_ = interpolationVariable_ == "Zero" ? "Linear" : "LogLinear";
306  }
307  if (zeroDayCounter_.empty()) {
308  zeroDayCounter_ = "A365";
309  }
310  XMLNode* nodeToTest = XMLUtils::getChildNode(node, "Extrapolation");
311  if (nodeToTest) {
312  extrapolation_ = XMLUtils::getChildValueAsBool(node, "Extrapolation", false);
313  } else {
314  extrapolation_ = true;
315  }
316 
317  // Optional bootstrap configuration
318  if (XMLNode* n = XMLUtils::getChildNode(node, "BootstrapConfig")) {
319  bootstrapConfig_.fromXML(n);
320  }
321 
322  // Tolerance is deprecated in favour of Accuracy in BootstrapConfig. However, if it is
323  // still provided, use it as the accuracy and global accuracy in the bootstrap.
324  if (XMLUtils::getChildNode(node, "Tolerance")) {
325  Real accuracy = XMLUtils::getChildValueAsDouble(node, "Tolerance", false);
326  bootstrapConfig_ =
327  BootstrapConfig(accuracy, accuracy, bootstrapConfig_.dontThrow(), bootstrapConfig_.maxAttempts(),
328  bootstrapConfig_.maxFactor(), bootstrapConfig_.minFactor());
329  }
330 
331  populateRequiredCurveIds();
332 }

383 void YieldCurveSegment::fromXML(XMLNode* node) {
384  typeID_ = XMLUtils::getChildValue(node, "Type", true);
385  string name = XMLUtils::getNodeName(node);
386 
387  // Check if curve type is valid for the given segment name
388  std::map<std::string, std::list<std::string>> validSegmentTypes = {
389  {"Direct", {"Zero", "Discount"}},
390  {"Simple", {"Deposit", "FRA", "Future", "OIS", "Swap", "BMA Basis Swap"}},
391  {"AverageOIS", {"Average OIS"}},
392  {"TenorBasis", {"Tenor Basis Swap", "Tenor Basis Two Swaps"}},
393  {"CrossCurrency", {"FX Forward", "Cross Currency Basis Swap", "Cross Currency Fix Float Swap"}},
394  {"ZeroSpread", {"Zero Spread"}},
395  {"FittedBond", {"FittedBond"}},
396  {"YieldPlusDefault", {"Yield Plus Default"}},
397  {"WeightedAverage", {"Weighted Average"}},
398  {"DiscountRatio", {"Discount Ratio"}},
399  {"IborFallback", {"Ibor Fallback"}}};
400 
401  std::list<std::string> validTypes = validSegmentTypes.at(name);
402  QL_REQUIRE(std::find(validTypes.begin(), validTypes.end(), typeID_) != validTypes.end(),
403  "The curve type " << typeID_ << " is not a valid " << name << " curve segment type");
404 
405  if (name == "DiscountRatio") {
406  } else if (name == "AverageOIS") {
407  XMLNode* quotesNode = XMLUtils::getChildNode(node, "Quotes");
408  if (quotesNode) {
409  for (XMLNode* child = XMLUtils::getChildNode(quotesNode, "CompositeQuote"); child;
410  child = XMLUtils::getNextSibling(child)) {
411  quotes_.push_back(quote(XMLUtils::getChildValue(child, "RateQuote", true)));
412  quotes_.push_back(quote(XMLUtils::getChildValue(child, "SpreadQuote", true)));
413  }
414  } else {
415  QL_FAIL("No Quotes in segment. Remove segment or add quotes.");
416  }
417  } else {
418  quotes_.clear();
419  XMLNode* quotesNode = XMLUtils::getChildNode(node, "Quotes");
420  if (quotesNode) {
421  for (auto n : XMLUtils::getChildrenNodes(quotesNode, "Quote")) {
422  string attr = XMLUtils::getAttribute(n, "optional"); // return "" if not present
423  bool opt = (!attr.empty() && parseBool(attr));
424  quotes_.emplace_back(quote(XMLUtils::getNodeValue(n), opt));
425  }
426  }
427  }
428  type_ = parseYieldCurveSegment(typeID_);
429  conventionsID_ = XMLUtils::getChildValue(node, "Conventions", false);
430 }

432 XMLNode* YieldCurveSegment::toXML(XMLDocument& doc) {
433  XMLNode* node = doc.allocNode("Segment");
434  XMLUtils::addChild(doc, node, "Type", typeID_);
435  if (!quotes_.empty()) {
436  XMLNode* quotesNode = doc.allocNode("Quotes");
437  // Special case handling for AverageOIS where the quotes are stored as pairs
438  // Spread and Rate.
439  if (type_ == YieldCurveSegment::Type::AverageOIS) {
440  QL_REQUIRE(quotes_.size() % 2 == 0, "Invalid quotes vector should be even")
441  for (Size i = 0; i < quotes_.size(); i = i + 2) {
442  string rateQuote = quotes_[i].first;
443  string spreadQuote = quotes_[i + 1].first;
444 
445  XMLNode* compositeQuoteNode = doc.allocNode("CompositeQuote");
446  XMLUtils::addChild(doc, compositeQuoteNode, "SpreadQuote", spreadQuote);
447  XMLUtils::addChild(doc, compositeQuoteNode, "RateQuote", rateQuote);
448  XMLUtils::appendNode(quotesNode, compositeQuoteNode);
449  }
450  } else {
451  for (auto q : quotes_) {
452  XMLNode* qNode = doc.allocNode("Quote", q.first);
453  if (q.second)
454  XMLUtils::addAttribute(doc, qNode, "optional", "true");
455  XMLUtils::appendNode(quotesNode, qNode);
456  }
457  }
458  XMLUtils::appendNode(node, quotesNode);
459  }
460 
461  XMLUtils::addChild(doc, node, "Conventions", conventionsID_);
462  return node;
463 }
