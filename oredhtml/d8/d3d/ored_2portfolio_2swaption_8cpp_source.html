50 void Swaption::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
51 
52  QL_REQUIRE(swap_.size() == 2, "underlying swap must have 2 legs");
53 
54  // we can assume 2 legs now
55  bool isCrossCcy = swap_[0].currency() != swap_[1].currency();
56 
57  bool isFixedFloating = (swap_[0].legType() == "Fixed" && swap_[1].legType() == "Floating") ||
58  (swap_[1].legType() == "Fixed" && swap_[0].legType() == "Floating");
59 
60  // First elimate these
61  QL_REQUIRE(!isCrossCcy, "Cross Currency Swaptions not supported");
62  QL_REQUIRE(isFixedFloating, "Basis Swaptions not supported");
63 
64  // check at least one exercise date is given
65  QL_REQUIRE(!option_.exerciseDates().empty(), "No exercise dates given");
66 
67  // check if all exercise dates (adjusted by the notice period) are in the past and if so, build an expired
68  // instrument
69  Period noticePeriod = option_.noticePeriod().empty() ? 0 * Days : parsePeriod(option_.noticePeriod());
70  Calendar noticeCal = option_.noticeCalendar().empty() ? NullCalendar() : parseCalendar(option_.noticeCalendar());
71  BusinessDayConvention noticeBdc =
72  option_.noticeConvention().empty() ? Unadjusted : parseBusinessDayConvention(option_.noticeConvention());
73  std::vector<Date> exerciseDates(option_.exerciseDates().size());
74  for (Size i = 0; i < option_.exerciseDates().size(); ++i) {
75  exerciseDates.push_back(noticeCal.advance(parseDate(option_.exerciseDates()[i]), -noticePeriod, noticeBdc));
76  }
77 
78  Date latestExerciseDate = *std::max_element(exerciseDates.begin(), exerciseDates.end());
79  if (QuantLib::detail::simple_event(latestExerciseDate).hasOccurred()) {
80  exercise_ = boost::make_shared<BermudanExercise>(exerciseDates);
81  legs_ = {{boost::make_shared<QuantLib::SimpleCashFlow>(0.0, latestExerciseDate)}};
82  instrument_ = boost::shared_ptr<InstrumentWrapper>(
83  new VanillaInstrument(boost::make_shared<QuantLib::Swap>(legs_, std::vector<bool>(1, false)), 1.0, {}, {}));
84  legCurrencies_ = {swap_[0].currency()};
85  legPayers_ = {false};
86  npvCurrency_ = swap_[0].currency();
87  notional_ = Null<Real>();
88  notionalCurrency_ = npvCurrency_;
89  maturity_ = latestExerciseDate;
90  return;
91  }
92 
93  // build swaption with at least one alive exercise date
94  bool fixedFirst = swap_[0].legType() == "Fixed";
95  boost::shared_ptr<FixedLegData> fixedLegData =
96  boost::dynamic_pointer_cast<FixedLegData>(swap_[fixedFirst ? 0 : 1].concreteLegData());
97  boost::shared_ptr<FloatingLegData> floatingLegData =
98  boost::dynamic_pointer_cast<FloatingLegData>(swap_[fixedFirst ? 1 : 0].concreteLegData());
99  underlyingIndex_ = floatingLegData->index();
100 
101  bool isNonStandard =
102  (swap_[0].notionals().size() > 1 || swap_[1].notionals().size() > 1 || floatingLegData->spreads().size() > 1 ||
103  floatingLegData->gearings().size() > 1 || fixedLegData->rates().size() > 1);
104 
105  Exercise::Type exerciseType = parseExerciseType(option_.style());
106 
107  // treat non standard europeans as bermudans, also if exercise fees are present
108  QL_REQUIRE(exerciseType == Exercise::Bermudan || exerciseType == Exercise::European,
109  "Exercise type " << option_.style() << " not implemented for Swaptions");
110  if (exerciseType == Exercise::Bermudan || (isNonStandard && exerciseType == Exercise::European) ||
111  !option_.exerciseFees().empty())
112  buildBermudan(engineFactory);
113  else
114  buildEuropean(engineFactory);
115 
116  // add required fixings, we add the required fixing for the underlying swap, which might be more
117  // than actually required, i.e. we are conservative here
118  std::map<string,string> indexNameMapper = {{underlyingIndexQlName_, underlyingIndex_}};
119  if (engineFactory->iborFallbackConfig().isIndexReplaced(underlyingIndex_)) {
120  string rfrIndexName = engineFactory->iborFallbackConfig().fallbackData(underlyingIndex_).rfrIndex;
121  // we don't support convention based rfr fallback indices, with ore ticket 1758 this might change
122  indexNameMapper[parseIborIndex(rfrIndexName)->name()] = rfrIndexName;
123  }
124  addToRequiredFixings(underlyingLeg_, boost::make_shared<FixingDateGetter>(requiredFixings_, indexNameMapper));
125 }

136 void Swaption::buildEuropean(const boost::shared_ptr<EngineFactory>& engineFactory) {
137 
138  LOG("Building European Swaption " << id());
139 
140  // Swaption details
141  Settlement::Type settleType = parseSettlementType(option_.settlement());
142  Settlement::Method settleMethod = option_.settlementMethod() == ""
143  ? defaultMethod(settleType)
144  : parseSettlementMethod(option_.settlementMethod());
145  QL_REQUIRE(option_.exerciseDates().size() == 1, "Only one exercise date expected for European Option");
146 
147  Period noticePeriod = option_.noticePeriod().empty() ? 0 * Days : parsePeriod(option_.noticePeriod());
148  Calendar noticeCal = option_.noticeCalendar().empty() ? NullCalendar() : parseCalendar(option_.noticeCalendar());
149  BusinessDayConvention noticeBdc =
150  option_.noticeConvention().empty() ? Unadjusted : parseBusinessDayConvention(option_.noticeConvention());
151  Date exDate = noticeCal.advance(parseDate(option_.exerciseDates().front()), -noticePeriod, noticeBdc);
152  QL_REQUIRE(exDate >= Settings::instance().evaluationDate(), "Exercise date expected in the future.");
153 
154  boost::shared_ptr<VanillaSwap> swap = buildVanillaSwap(engineFactory, exDate);
155  underlyingLeg_ = swap->floatingLeg();
156  underlyingFixedLeg_ = swap->fixedLeg();
157 
158  Position::Type positionType = parsePositionType(option_.longShort());
159  Real multiplier = (positionType == QuantLib::Position::Long ? 1.0 : -1.0);
160 
161  string ccy = swap_[0].currency();
162  Currency currency = parseCurrency(ccy);
163 
164  // Check for existence of swaption engine builder.
165  string tt("EuropeanSwaption");
166  boost::shared_ptr<EngineBuilder> builder = engineFactory->builder(tt);
167  QL_REQUIRE(builder, "No builder found for " << tt);
168  boost::shared_ptr<EuropeanSwaptionEngineBuilder> swaptionBuilder =
169  boost::dynamic_pointer_cast<EuropeanSwaptionEngineBuilder>(builder);
170 
171  exercise_ = boost::make_shared<EuropeanExercise>(exDate);
172 
173  // Build Swaption
174  boost::shared_ptr<QuantLib::Swaption> swaption(new QuantLib::Swaption(swap, exercise_, settleType, settleMethod));
175 
176  std::vector<boost::shared_ptr<Instrument>> additionalInstruments;
177  std::vector<Real> additionalMultipliers;
178  addPremiums(additionalInstruments, additionalMultipliers, 1.0, option_.premiumData(), -multiplier, currency,
179  engineFactory, swaptionBuilder->configuration(MarketContext::pricing));
180 
181  // Set the engine.
182  swaption->setPricingEngine(swaptionBuilder->engine(currency));
183 
184  // Now set the instrument wrapper, depending on delivery
185  if (settleType == Settlement::Physical) {
186  // tracks state for any option flavour, including physical delivery
187  instrument_ = boost::shared_ptr<InstrumentWrapper>(
188  new EuropeanOptionWrapper(swaption, positionType == Position::Long ? true : false, exDate,
189  settleType == Settlement::Physical ? true : false, swap, 1.0, 1.0,
190  additionalInstruments, additionalMultipliers));
191  // maturity of underlying
192  maturity_ = std::max(swap->fixedSchedule().dates().back(), swap->floatingSchedule().dates().back());
193  } else {
194  instrument_ = boost::shared_ptr<InstrumentWrapper>(
195  new VanillaInstrument(swaption, multiplier, additionalInstruments, additionalMultipliers));
196  // Align with ISDA AANA/GRID guidance as of November 2020
197  // maturity_ = exDate;
198  maturity_ = std::max(swap->fixedSchedule().dates().back(), swap->floatingSchedule().dates().back());
199  }
200 
201  DLOG("Building European Swaption done");
202 }

204 void Swaption::buildBermudan(const boost::shared_ptr<EngineFactory>& engineFactory) {
205  DLOG("Building Bermudan Swaption " << id());
206 
207  QL_REQUIRE(swap_.size() >= 1, "underlying swap must have at least 1 leg");
208 
209  string ccy_str = swap_[0].currency();
210  Currency currency = parseCurrency(ccy_str);
211 
212  bool fixedFirst = swap_[0].legType() == "Fixed";
213  boost::shared_ptr<FixedLegData> fixedLegData =
214  boost::dynamic_pointer_cast<FixedLegData>(swap_[fixedFirst ? 0 : 1].concreteLegData());
215  boost::shared_ptr<FloatingLegData> floatingLegData =
216  boost::dynamic_pointer_cast<FloatingLegData>(swap_[fixedFirst ? 1 : 0].concreteLegData());
217  underlyingIndex_ = floatingLegData->index();
218 
219  bool isNonStandard =
220  (swap_[0].notionals().size() > 1 || swap_[1].notionals().size() > 1 || floatingLegData->spreads().size() > 1 ||
221  floatingLegData->gearings().size() > 1 || fixedLegData->rates().size() > 1);
222 
223  boost::shared_ptr<VanillaSwap> vanillaSwap;
224  boost::shared_ptr<NonstandardSwap> nonstandardSwap;
225  boost::shared_ptr<Swap> swap;
226  if (!isNonStandard) {
227  vanillaSwap = buildVanillaSwap(engineFactory);
228  underlyingLeg_ = vanillaSwap->floatingLeg();
229  underlyingFixedLeg_ = vanillaSwap->fixedLeg();
230  swap = vanillaSwap;
231  } else {
232  nonstandardSwap = buildNonStandardSwap(engineFactory);
233  underlyingLeg_ = nonstandardSwap->floatingLeg();
234  underlyingFixedLeg_ = nonstandardSwap->fixedLeg();
235  swap = nonstandardSwap;
236  }
237 
238  DLOG("Swaption::Build(): Underlying Start = " << QuantLib::io::iso_date(swap->startDate()));
239 
240  ExerciseBuilder eb(option_, {swap->leg(1)});
241  QL_REQUIRE(!eb.noticeDates().empty(), "BermudanSwaption: Swaption does not have any alive exercise dates");
242  exercise_ = eb.exercise();
243  maturity_ = eb.noticeDates().back();
244 
245  Settlement::Type delivery = parseSettlementType(option_.settlement());
246  Settlement::Method deliveryMethod =
247  option_.settlementMethod() == "" ? defaultMethod(delivery) : parseSettlementMethod(option_.settlementMethod());
248 
249  if (delivery == Settlement::Cash && deliveryMethod == Settlement::ParYieldCurve)
250  WLOG("Cash-settled Bermudan Swaption (id = "
251  << id() << ") with ParYieldCurve settlement method not supported by Lgm engine. "
252  << "Approximate pricing using CollateralizedCashPrice pricing methodology");
253 
254  // Build swaption
255  DLOG("Build Swaption instrument");
256  boost::shared_ptr<QuantLib::Instrument> swaption;
257  if (isNonStandard) {
258  swaption = boost::shared_ptr<QuantLib::Instrument>(
259  new QuantLib::NonstandardSwaption(nonstandardSwap, exercise_, delivery, deliveryMethod));
260  // workaround for missing registration in QL versions < 1.13
261  swaption->registerWithObservables(nonstandardSwap);
262  } else
263  swaption = boost::shared_ptr<QuantLib::Instrument>(
264  new QuantLib::Swaption(vanillaSwap, exercise_, delivery, deliveryMethod));
265 
266  QuantLib::Position::Type positionType = parsePositionType(option_.longShort());
267  if (delivery == Settlement::Physical)
268  maturity_ = swap->maturityDate();
269 
270  DLOG("Get/Build Bermudan Swaption engine");
271  cpu_timer timer;
272 
273  string tt("BermudanSwaption");
274  boost::shared_ptr<EngineBuilder> builder = engineFactory->builder(tt);
275  QL_REQUIRE(builder, "No builder found for " << tt);
276  boost::shared_ptr<BermudanSwaptionEngineBuilder> swaptionBuilder =
277  boost::dynamic_pointer_cast<BermudanSwaptionEngineBuilder>(builder);
278 
279  // determine strikes for calibration basket (simple approach, a la summit)
280  std::vector<Real> strikes(eb.noticeDates().size(), Null<Real>());
281  if (!isNonStandard) {
282  Real tmp = vanillaSwap->fixedRate() - vanillaSwap->spread();
283  std::fill(strikes.begin(), strikes.end(), tmp);
284  DLOG("calibration strike is " << tmp << "(fixed rate " << vanillaSwap->fixedRate() << ", spread "
285  << vanillaSwap->spread() << ")");
286  } else {
287  for (Size i = 0; i < eb.noticeDates().size(); ++i) {
288  const Schedule& fix = nonstandardSwap->fixedSchedule();
289  const Schedule& flt = nonstandardSwap->floatingSchedule();
290  Size fixIdx =
291  std::lower_bound(fix.dates().begin(), fix.dates().end(), eb.noticeDates()[i]) - fix.dates().begin();
292  Size fltIdx =
293  std::lower_bound(flt.dates().begin(), flt.dates().end(), eb.noticeDates()[i]) - flt.dates().begin();
294  // play safe
295  fixIdx = std::min(fixIdx, nonstandardSwap->fixedRate().size() - 1);
296  fltIdx = std::min(fltIdx, nonstandardSwap->spreads().size() - 1);
297  strikes[i] = nonstandardSwap->fixedRate()[fixIdx] - nonstandardSwap->spreads()[fltIdx];
298  DLOG("calibration strike for ex date "
299  << QuantLib::io::iso_date(eb.noticeDates()[i]) << " is " << strikes[i] << " (fixed rate "
300  << nonstandardSwap->fixedRate()[fixIdx] << ", spread " << nonstandardSwap->spreads()[fltIdx] << ")");
301  }
302  }
303 
304  boost::shared_ptr<PricingEngine> engine =
305  swaptionBuilder->engine(id(), isNonStandard, ccy_str, eb.noticeDates(), swap->maturityDate(), strikes);
306 
307  timer.stop();
308  DLOG("Swaption model calibration time: " << timer.format(default_places, "%w") << " s");
309 
310  swaption->setPricingEngine(engine);
311 
312  // timer.restart();
313  // Real npv = swaption->NPV();
314  // Real pt = timer.elapsed();
315  // DLOG("Swaption pricing time: " << pt*1000 << " ms");
316 
317  boost::shared_ptr<EngineBuilder> tmp = engineFactory->builder("Swap");
318  boost::shared_ptr<SwapEngineBuilderBase> swapBuilder = boost::dynamic_pointer_cast<SwapEngineBuilderBase>(tmp);
319  QL_REQUIRE(swapBuilder, "No Swap Builder found for Swaption " << id());
320  boost::shared_ptr<PricingEngine> swapEngine = swapBuilder->engine(currency);
321 
322  std::vector<boost::shared_ptr<Instrument>> underlyingSwaps =
323  buildUnderlyingSwaps(swapEngine, swap, eb.noticeDates());
324 
325  // If premium data is provided
326  // 1) build the fee trade and pass it to the instrument wrapper for pricing
327  // 2) add fee payment as additional trade leg for cash flow reporting
328  std::vector<boost::shared_ptr<Instrument>> additionalInstruments;
329  std::vector<Real> additionalMultipliers;
330  Real multiplier = positionType == Position::Long ? 1.0 : -1.0;
331  addPremiums(additionalInstruments, additionalMultipliers, 1.0, option_.premiumData(), -multiplier, currency,
332  engineFactory, swaptionBuilder->configuration(MarketContext::pricing));
333 
334  // instrument_ = boost::shared_ptr<InstrumentWrapper> (new VanillaInstrument (swaption, multiplier));
335  instrument_ = boost::shared_ptr<InstrumentWrapper>(
336  new BermudanOptionWrapper(swaption, positionType == Position::Long ? true : false, eb.noticeDates(),
337  delivery == Settlement::Physical ? true : false, underlyingSwaps, 1.0, 1.0,
338  additionalInstruments, additionalMultipliers));
339 
340  DLOG("Building Bermudan Swaption done");
341 }

343 boost::shared_ptr<VanillaSwap> Swaption::buildVanillaSwap(const boost::shared_ptr<EngineFactory>& engineFactory,
344  const Date& firstExerciseDate) {
345  // Limitation to vanilla for now
346  QL_REQUIRE(swap_.size() == 2 && swap_[0].currency() == swap_[1].currency() &&
347  swap_[0].isPayer() != swap_[1].isPayer() && swap_[0].notionals() == swap_[1].notionals() &&
348  swap_[0].notionals().size() == 1,
349  "Not a Vanilla Swap");
350 
351  Size fixedLegIndex, floatingLegIndex;
352  if (swap_[0].legType() == "Floating" && swap_[1].legType() == "Fixed") {
353  floatingLegIndex = 0;
354  fixedLegIndex = 1;
355  } else if (swap_[1].legType() == "Floating" && swap_[0].legType() == "Fixed") {
356  floatingLegIndex = 1;
357  fixedLegIndex = 0;
358  } else {
359  QL_FAIL("Invalid leg types " << swap_[0].legType() << " + " << swap_[1].legType());
360  }
361  boost::shared_ptr<FixedLegData> fixedLegData =
362  boost::dynamic_pointer_cast<FixedLegData>(swap_[fixedLegIndex].concreteLegData());
363  boost::shared_ptr<FloatingLegData> floatingLegData =
364  boost::dynamic_pointer_cast<FloatingLegData>(swap_[floatingLegIndex].concreteLegData());
365 
366  QL_REQUIRE(fixedLegData->rates().size() == 1, "Vanilla Swaption: constant rate required");
367  QL_REQUIRE(floatingLegData->spreads().size() <= 1, "Vanilla Swaption: constant spread required");
368 
369  boost::shared_ptr<EngineBuilder> tmp = engineFactory->builder("Swap");
370  boost::shared_ptr<SwapEngineBuilderBase> swapBuilder = boost::dynamic_pointer_cast<SwapEngineBuilderBase>(tmp);
371  QL_REQUIRE(swapBuilder, "No Swap Builder found for Swaption " << id());
372 
373  // Get Trade details
374  string ccy = swap_[0].currency();
375  Currency currency = parseCurrency(ccy);
376  Real nominal = swap_[0].notionals().back();
377 
378  Real rate = fixedLegData->rates().front();
379  Real spread = floatingLegData->spreads().empty() ? 0.0 : floatingLegData->spreads().front();
380  string indexName = floatingLegData->index();
381 
382  Schedule fixedSchedule = makeSchedule(swap_[fixedLegIndex].schedule());
383  DayCounter fixedDayCounter = parseDayCounter(swap_[fixedLegIndex].dayCounter());
384 
385  Schedule floatingSchedule = makeSchedule(swap_[floatingLegIndex].schedule());
386  Handle<IborIndex> index =
387  engineFactory->market()->iborIndex(indexName, swapBuilder->configuration(MarketContext::pricing));
388  underlyingIndexQlName_ = index->name();
389  DayCounter floatingDayCounter = parseDayCounter(swap_[floatingLegIndex].dayCounter());
390 
391  BusinessDayConvention paymentConvention = parseBusinessDayConvention(swap_[floatingLegIndex].paymentConvention());
392 
393  VanillaSwap::Type type = swap_[fixedLegIndex].isPayer() ? VanillaSwap::Payer : VanillaSwap::Receiver;
394 
395  // We treat overnight and bma indices approximately as ibor indices and warn about this in the log
396  if (boost::dynamic_pointer_cast<OvernightIndex>(*index) ||
397  boost ::dynamic_pointer_cast<QuantExt::BMAIndexWrapper>(*index))
398  ALOG("Swaption trade " << id() << " on ON or BMA index '" << underlyingIndex_
399  << "' built, will treat the index approximately as an ibor index");
400 
401  // only take into account accrual periods with start date on or after first exercise date (if given)
402  if (firstExerciseDate != Null<Date>()) {
403  std::vector<Date> fixDates = fixedSchedule.dates();
404  auto it1 = std::lower_bound(fixDates.begin(), fixDates.end(), firstExerciseDate);
405  fixDates.erase(fixDates.begin(), it1);
406  // check we have at least 1 to stop set fault on vector(fixDates.size() - 1, true) but maybe check should be 2
407  QL_REQUIRE(fixDates.size() >= 2,
408  "Not enough schedule dates are left in Swaption fixed leg (check exercise dates)");
409  fixedSchedule = Schedule(fixDates, fixedSchedule.calendar(), Unadjusted, boost::none, boost::none, boost::none,
410  boost::none, std::vector<bool>(fixDates.size() - 1, true));
411  std::vector<Date> floatingDates = floatingSchedule.dates();
412  auto it2 = std::lower_bound(floatingDates.begin(), floatingDates.end(), firstExerciseDate);
413  floatingDates.erase(floatingDates.begin(), it2);
414  QL_REQUIRE(floatingDates.size() >= 2,
415  "Not enough schedule dates are left in Swaption floating leg (check exercise dates)");
416  floatingSchedule = Schedule(floatingDates, floatingSchedule.calendar(), Unadjusted, boost::none, boost::none,
417  boost::none, boost::none, std::vector<bool>(floatingDates.size() - 1, true));
418  }
419 
420  // Build a vanilla (bullet) swap underlying
421  boost::shared_ptr<VanillaSwap> swap(new VanillaSwap(type, nominal, fixedSchedule, rate, fixedDayCounter,
422  floatingSchedule, *index, spread, floatingDayCounter,
423  paymentConvention));
424 
425  swap->setPricingEngine(swapBuilder->engine(currency));
426 
427  // Set other ore::data::Trade details
428  npvCurrency_ = ccy;
429  notional_ = Null<Real>();
430  notionalCurrency_ = ccy;
431  legCurrencies_ = vector<string>(2, ccy);
432  legs_.push_back(swap->fixedLeg());
433  legs_.push_back(swap->floatingLeg());
434  legPayers_.push_back(swap_[fixedLegIndex].isPayer());
435  legPayers_.push_back(swap_[floatingLegIndex].isPayer());
436 
437  return swap;
438 }

440 boost::shared_ptr<NonstandardSwap>
441 Swaption::buildNonStandardSwap(const boost::shared_ptr<EngineFactory>& engineFactory) {
442  QL_REQUIRE(swap_.size() == 2, "Two swap legs expected");
443  QL_REQUIRE(swap_[0].currency() == swap_[1].currency(), "single currency swap expected");
444  QL_REQUIRE(swap_[0].isPayer() != swap_[1].isPayer(), "pay and receive leg expected");
445 
446  Size fixedLegIndex, floatingLegIndex;
447  if (swap_[0].legType() == "Floating" && swap_[1].legType() == "Fixed") {
448  floatingLegIndex = 0;
449  fixedLegIndex = 1;
450  } else if (swap_[1].legType() == "Floating" && swap_[0].legType() == "Fixed") {
451  floatingLegIndex = 1;
452  fixedLegIndex = 0;
453  } else {
454  QL_FAIL("Invalid leg types " << swap_[0].legType() << " + " << swap_[1].legType());
455  }
456  boost::shared_ptr<FixedLegData> fixedLegData =
457  boost::dynamic_pointer_cast<FixedLegData>(swap_[fixedLegIndex].concreteLegData());
458  boost::shared_ptr<FloatingLegData> floatingLegData =
459  boost::dynamic_pointer_cast<FloatingLegData>(swap_[floatingLegIndex].concreteLegData());
460 
461  boost::shared_ptr<EngineBuilder> tmp = engineFactory->builder("Swap");
462  boost::shared_ptr<SwapEngineBuilderBase> swapBuilder = boost::dynamic_pointer_cast<SwapEngineBuilderBase>(tmp);
463  QL_REQUIRE(swapBuilder, "No Swap Builder found for Swaption " << id());
464 
465  // Get Trade details
466  string ccy = swap_[0].currency();
467  Currency currency = parseCurrency(ccy);
468  Schedule fixedSchedule = makeSchedule(swap_[fixedLegIndex].schedule());
469  Schedule floatingSchedule = makeSchedule(swap_[floatingLegIndex].schedule());
470  vector<Real> fixedNominal = buildScheduledVectorNormalised(
471  swap_[fixedLegIndex].notionals(), swap_[fixedLegIndex].notionalDates(), fixedSchedule, (Real)Null<Real>());
472  vector<Real> floatNominal =
473  buildScheduledVectorNormalised(swap_[floatingLegIndex].notionals(), swap_[floatingLegIndex].notionalDates(),
474  floatingSchedule, (Real)Null<Real>());
475  vector<Real> fixedRate = buildScheduledVectorNormalised(fixedLegData->rates(), fixedLegData->rateDates(),
476  fixedSchedule, (Real)Null<Real>());
477  vector<Real> spreads = buildScheduledVectorNormalised(floatingLegData->spreads(), floatingLegData->spreadDates(),
478  floatingSchedule, (Real)Null<Real>());
479  // gearings are optional, i.e. may be empty
480  vector<Real> gearings = buildScheduledVectorNormalised(floatingLegData->gearings(), floatingLegData->gearingDates(),
481  floatingSchedule, 1.0);
482  string indexName = floatingLegData->index();
483  DayCounter fixedDayCounter = parseDayCounter(swap_[fixedLegIndex].dayCounter());
484  Handle<IborIndex> index =
485  engineFactory->market()->iborIndex(indexName, swapBuilder->configuration(MarketContext::pricing));
486  underlyingIndexQlName_ = index->name();
487  DayCounter floatingDayCounter = parseDayCounter(swap_[floatingLegIndex].dayCounter());
488  BusinessDayConvention paymentConvention = parseBusinessDayConvention(swap_[floatingLegIndex].paymentConvention());
489 
490  // We treat overnight and bma indices approximately as ibor indices and warn about this in the log
491  if (boost::dynamic_pointer_cast<OvernightIndex>(*index) ||
492  boost ::dynamic_pointer_cast<QuantExt::BMAIndexWrapper>(*index))
493  ALOG("Swaption trade " << id() << " on ON or BMA index '" << underlyingIndex_
494  << "' built, will treat the index approximately as an ibor index");
495 
496  VanillaSwap::Type type = swap_[fixedLegIndex].isPayer() ? VanillaSwap::Payer : VanillaSwap::Receiver;
497 
498  // Build a vanilla (bullet) swap underlying
499  boost::shared_ptr<NonstandardSwap> swap(new NonstandardSwap(type, fixedNominal, floatNominal, fixedSchedule,
500  fixedRate, fixedDayCounter, floatingSchedule, *index,
501  gearings, spreads, floatingDayCounter,
502  false, // no intermediate notional exchanges
503  false, // no final notional exchanges
504  paymentConvention));
505 
506  swap->setPricingEngine(swapBuilder->engine(currency));
507 
508  // Set other ore::data::Trade details
509  npvCurrency_ = ccy;
510  notional_ = Null<Real>();
511  notionalCurrency_ = ccy;
512  legCurrencies_ = vector<string>(2, ccy);
513  legs_.push_back(swap->fixedLeg());
514  legs_.push_back(swap->floatingLeg());
515  legPayers_.push_back(swap_[fixedLegIndex].isPayer());
516  legPayers_.push_back(swap_[floatingLegIndex].isPayer());
517 
518  return swap;

521 std::vector<boost::shared_ptr<Instrument>>
522 Swaption::buildUnderlyingSwaps(const boost::shared_ptr<PricingEngine>& swapEngine,
523  const boost::shared_ptr<Swap>& underlyingSwap, const std::vector<Date>& exerciseDates) {
524  std::vector<boost::shared_ptr<Instrument>> swaps;
525  for (Size i = 0; i < exerciseDates.size(); ++i) {
526  std::vector<Leg> legs(2);
527  std::vector<bool> payer(2);
528  for (Size j = 0; j < legs.size(); ++j) {
529  legs[j] = underlyingSwap->leg(j);
530  // if in the swap data the fixed leg is the first component, the legs in the underlying
531  // swap match the order of the legs in the swap data, otherwise they are swapped
532  payer[j] = swap_[0].legType() == "Fixed" ? swap_[j].isPayer() : swap_[1 - j].isPayer();
533  boost::shared_ptr<Coupon> coupon = boost::dynamic_pointer_cast<Coupon>(legs[j].front());
534  while (legs[j].size() > 0 && coupon->accrualStartDate() < exerciseDates[i]) {
535  legs[j].erase(legs[j].begin());
536  coupon = boost::dynamic_pointer_cast<Coupon>(legs[j].front());
537  }
538  }
539  boost::shared_ptr<Swap> newSwap(new Swap(legs, payer));
540  newSwap->setPricingEngine(swapEngine);
541  swaps.push_back(newSwap);
542  if (legs[0].size() > 0 && legs[1].size() > 0) {
543  boost::shared_ptr<Coupon> coupon1 = boost::dynamic_pointer_cast<Coupon>(legs[0].front());
544  boost::shared_ptr<Coupon> coupon2 = boost::dynamic_pointer_cast<Coupon>(legs[1].front());
545  DLOG("Added underlying Swap start " << QuantLib::io::iso_date(coupon1->accrualStartDate()) << " "
546  << QuantLib::io::iso_date(coupon2->accrualStartDate()) << " "
547  << "exercise " << QuantLib::io::iso_date(exerciseDates[i]));
548  } else {
549  WLOG("Added underlying Swap with at least one empty leg for exercise "
550  << QuantLib::io::iso_date(exerciseDates[i]) << "!");
551  }
552  }
553  return swaps;
554 }

567 const std::map<std::string,boost::any>& Swaption::additionalData() const {
568  Size numLegs = swap_.size();
569  // use the build time as of date to determine current notionals
570  Date asof = Settings::instance().evaluationDate();
571  for (Size i = 0; i < numLegs; ++i) {
572  string legID = to_string(i+1);
573  additionalData_["legType[" + legID + "]"] = swap_[i].legType();
574  additionalData_["isPayer[" + legID + "]"] = swap_[i].isPayer();
575  additionalData_["notionalCurrency[" + legID + "]"] = swap_[i].currency();
576  for (Size j = 0; j < legs_[i].size(); ++j) {
577  boost::shared_ptr<CashFlow> flow = legs_[i][j];
578  // pick flow with earliest future payment date on this leg
579  if (flow->date() > asof) {
580  additionalData_["amount[" + legID + "]"] = flow->amount();
581  additionalData_["paymentDate[" + legID + "]"] = to_string(flow->date());
582  boost::shared_ptr<Coupon> coupon = boost::dynamic_pointer_cast<Coupon>(flow);
583  if (coupon) {
584  additionalData_["currentNotional[" + legID + "]"] = coupon->nominal();
585  additionalData_["rate[" + legID + "]"] = coupon->rate();
586  boost::shared_ptr<FloatingRateCoupon> frc = boost::dynamic_pointer_cast<FloatingRateCoupon>(flow);
587  if (frc) {
588  additionalData_["index[" + legID + "]"] = frc->index()->name();
589  additionalData_["spread[" + legID + "]"] = frc->spread();
590  }
591  }
592  break;
593  }
594  }
595  if (legs_[i].size() > 0) {
596  boost::shared_ptr<Coupon> coupon = boost::dynamic_pointer_cast<Coupon>(legs_[i][0]);
597  if (coupon)
598  additionalData_["originalNotional[" + legID + "]"] = coupon->nominal();
599  }
600  }
601  return additionalData_;
602 }
