32 void OptionData::fromXML(XMLNode* node) {
33  XMLUtils::checkNode(node, "OptionData");
34  longShort_ = XMLUtils::getChildValue(node, "LongShort", true);
35  callPut_ = XMLUtils::getChildValue(node, "OptionType", false);
36  payoffType_ = XMLUtils::getChildValue(node, "PayoffType", false);
37  style_ = XMLUtils::getChildValue(node, "Style", false);
38  noticePeriod_ = XMLUtils::getChildValue(node, "NoticePeriod", false);
39  noticeCalendar_ = XMLUtils::getChildValue(node, "NoticeCalendar", false);
40  noticeConvention_ = XMLUtils::getChildValue(node, "NoticeConvention", false);
41  settlement_ = XMLUtils::getChildValue(node, "Settlement", false);
42  settlementMethod_ = XMLUtils::getChildValue(node, "SettlementMethod", false);
43  payoffAtExpiry_ = XMLUtils::getChildValueAsBool(node, "PayOffAtExpiry", false);
44  premiumData_.fromXML(node);
45  exerciseFeeTypes_.clear();
46  exerciseFeeDates_.clear();
47  vector<std::reference_wrapper<vector<string>>> attrs;
48  attrs.push_back(exerciseFeeTypes_);
49  attrs.push_back(exerciseFeeDates_);
50  exerciseFees_ = XMLUtils::getChildrenValuesWithAttributes<Real>(node, "ExerciseFees", "ExerciseFee",
51  {"type", "startDate"}, attrs, &parseReal);
52  exerciseFeeSettlementPeriod_ = XMLUtils::getChildValue(node, "ExerciseFeeSettlementPeriod", false);
53  exerciseFeeSettlementCalendar_ = XMLUtils::getChildValue(node, "ExerciseFeeSettlementCalendar", false);
54  exerciseFeeSettlementConvention_ = XMLUtils::getChildValue(node, "ExerciseFeeSettlementConvention", false);
55  exercisePrices_ = XMLUtils::getChildrenValuesAsDoubles(node, "ExercisePrices", "ExercisePrice", false);
56  exerciseDates_ = XMLUtils::getChildrenValues(node, "ExerciseDates", "ExerciseDate", false);
57 
58  automaticExercise_ = boost::none;
59  if (XMLNode* n = XMLUtils::getChildNode(node, "AutomaticExercise"))
60  automaticExercise_ = parseBool(XMLUtils::getNodeValue(n));
61 
62  exerciseData_ = boost::none;
63  if (XMLNode* n = XMLUtils::getChildNode(node, "ExerciseData")) {
64  exerciseData_ = OptionExerciseData();
65  exerciseData_->fromXML(n);
66  }
67 
68  paymentData_ = boost::none;
69  if (XMLNode* n = XMLUtils::getChildNode(node, "PaymentData")) {
70  paymentData_ = OptionPaymentData();
71  paymentData_->fromXML(n);
72  }
73 }

75 XMLNode* OptionData::toXML(XMLDocument& doc) {

85  if (noticeCalendar_ != "")

87  if (noticeConvention_ != "")
88  XMLUtils::addChild(doc, node, "NoticeConvention", noticeConvention_);
89  if (settlement_ != "")
90  XMLUtils::addChild(doc, node, "Settlement", settlement_);
91  if (settlementMethod_ != "")
92  XMLUtils::addChild(doc, node, "SettlementMethod", settlementMethod_);
93  XMLUtils::addChild(doc, node, "PayOffAtExpiry", payoffAtExpiry_);
94  XMLUtils::appendNode(node, premiumData_.toXML(doc));
95  XMLUtils::addChildrenWithOptionalAttributes(doc, node, "ExerciseFees", "ExerciseFee", exerciseFees_,
96  {"type", "startDate"}, {exerciseFeeTypes_, exerciseFeeDates_});
97  if (exerciseFeeSettlementPeriod_ != "")
98  XMLUtils::addChild(doc, node, "ExerciseFeeSettlementPeriod", exerciseFeeSettlementPeriod_);
99  if (exerciseFeeSettlementCalendar_ != "")
100  XMLUtils::addChild(doc, node, "ExerciseFeeSettlementCalendar", exerciseFeeSettlementCalendar_);
101  if (exerciseFeeSettlementConvention_ != "")
102  XMLUtils::addChild(doc, node, "ExerciseFeeSettlementConvention", exerciseFeeSettlementConvention_);
103  XMLUtils::addChildren(doc, node, "ExercisePrices", "ExercisePrice", exercisePrices_);
104  XMLUtils::addChildren(doc, node, "ExerciseDates", "ExerciseDate", exerciseDates_);
105 
106  if (automaticExercise_)
107  XMLUtils::addChild(doc, node, "AutomaticExercise", *automaticExercise_);
108 
109  if (exerciseData_) {
110  XMLUtils::appendNode(node, exerciseData_->toXML(doc));
111  }
112 
113  if (paymentData_) {
114  XMLUtils::appendNode(node, paymentData_->toXML(doc));
115  }
116 
117  return node;
118 }

120 ExerciseBuilder::ExerciseBuilder(const OptionData& optionData, const std::vector<Leg> legs) {
121 
122  // only keep a) future exercise dates and b) exercise dates that exercise into a whole
123  // accrual period of the underlying; TODO handle exercises into broken periods?
124 
125  // determine last accrual start date present in the underlying legs
126 
127  Date lastAccrualStartDate = Date::minDate();
128  for (auto const& l : legs) {
129  for (auto const& c : l) {
130  if (auto cpn = boost::dynamic_pointer_cast<Coupon>(c))
131  lastAccrualStartDate = std::max(lastAccrualStartDate, cpn->accrualStartDate());
132  }
133  }
134 
135  // get notiice period, calendar, bdc
136 
137  Period noticePeriod = optionData.noticePeriod().empty() ? 0 * Days : parsePeriod(optionData.noticePeriod());
138  Calendar noticeCal =
139  optionData.noticeCalendar().empty() ? NullCalendar() : parseCalendar(optionData.noticeCalendar());
140  BusinessDayConvention noticeBdc =
141  optionData.noticeConvention().empty() ? Unadjusted : parseBusinessDayConvention(optionData.noticeConvention());
142 
143  // build vector of sorted exercise dates
144 
145  std::vector<QuantLib::Date> sortedExerciseDates;
146  for (auto const& d : optionData.exerciseDates())
147  sortedExerciseDates.push_back(parseDate(d));
148  std::sort(sortedExerciseDates.begin(), sortedExerciseDates.end());
149 
150  // build vector of alive exercise dates and corresponding notive dates
151 
152  std::vector<bool> isExerciseDateAlive(sortedExerciseDates.size(), false);
153 
154  for (Size i = 0; i < sortedExerciseDates.size(); i++) {
155  Date noticeDate = noticeCal.advance(sortedExerciseDates[i], -noticePeriod, noticeBdc);
156  if (noticeDate > Settings::instance().evaluationDate() && noticeDate <= lastAccrualStartDate) {
157  isExerciseDateAlive[i] = true;
158  noticeDates_.push_back(noticeDate);
159  exerciseDates_.push_back(sortedExerciseDates[i]);
160  DLOG("Got notice date " << QuantLib::io::iso_date(noticeDate) << " using notice period " << noticePeriod
161  << ", convention " << noticeBdc << ", calendar " << noticeCal.name()
162  << " from exercise date " << exerciseDates_.back());
163  }
164  if (noticeDate > lastAccrualStartDate)
165  WLOG("Remove notice date " << ore::data::to_string(noticeDate) << " (exercise date "
166  << sortedExerciseDates[i] << ") after last accrual start date "
167  << ore ::data::to_string(lastAccrualStartDate));
168  }
169 
170  // if we do not have any notice dates left, we are done (a nullptr will be returned by exercise())
171 
172  if (noticeDates_.empty())
173  return;
174 
175  // build exercise instance
176 
177  exercise_ = boost::make_shared<BermudanExercise>(noticeDates_);
178 
179  // build feee and rebated exercise instance, if any fees are present
180 
181  if (!optionData.exerciseFees().empty()) {
182 
183  // build an exercise date "schedule" by adding the maximum possible date at the end
184 
185  std::vector<Date> exDatesPlusInf(sortedExerciseDates);
186  exDatesPlusInf.push_back(Date::maxDate());
187  vector<double> allRebates = buildScheduledVectorNormalised(optionData.exerciseFees(),
188  optionData.exerciseFeeDates(), exDatesPlusInf, 0.0);
189 
190  // filter on alive rebates, so that we can a vector of rebates corresponding to the exerciseDates vector
191 
192  vector<double> rebates;
193  for (Size i = 0; i < sortedExerciseDates.size(); ++i) {
194  if (isExerciseDateAlive[i])
195  rebates.push_back(allRebates[i]);
196  }
197 
198  // flip the sign of the fee to get a rebate
199 
200  for (auto& r : rebates)
201  r = -r;
202  vector<string> feeType = buildScheduledVectorNormalised<string>(
203  optionData.exerciseFeeTypes(), optionData.exerciseFeeDates(), exDatesPlusInf, "");
204 
205  // convert relative to absolute fees if required
206 
207  for (Size i = 0; i < rebates.size(); ++i) {
208 
209  // default to Absolute
210 
211  if (feeType[i].empty())
212  feeType[i] = "Absolute";
213 
214  if (feeType[i] == "Percentage") {
215 
216  // get next coupon after exercise to determine relevant notional
217 
218  std::set<std::pair<Date, Real>> notionals;
219  for (auto const& l : legs) {
220  for (auto const& c : l) {
221  if (auto cpn = boost::dynamic_pointer_cast<Coupon>(c)) {
222  if (cpn->accrualStartDate() >= exerciseDates_[i])
223  notionals.insert(std::make_pair(cpn->accrualStartDate(), cpn->nominal()));
224  }
225  }
226  }
227 
228  if (notionals.empty())
229  rebates[i] = 0.0; // notional is zero
230  else {
231  Real feeNotional = notionals.begin()->second;
232  DLOG("Convert percentage rebate "
233  << rebates[i] << " to absolute reabte " << rebates[i] * feeNotional << " using nominal "
234  << feeNotional << " for exercise date " << QuantLib::io::iso_date(exerciseDates_[i]));
235  rebates[i] *= feeNotional; // multiply percentage fee by relevant notional
236  }
237 
238  } else {
239  QL_REQUIRE(feeType[i] == "Absolute", "fee type must be Absolute or Relative");
240  }
241  }
242 
243  Period feeSettlPeriod = optionData.exerciseFeeSettlementPeriod().empty()
244  ? 0 * Days
245  : parsePeriod(optionData.exerciseFeeSettlementPeriod());
246 
247  Calendar feeSettlCal = optionData.exerciseFeeSettlementCalendar().empty()
248  ? NullCalendar()
249  : parseCalendar(optionData.exerciseFeeSettlementCalendar());
250 
251  BusinessDayConvention feeSettlBdc =
252  optionData.exerciseFeeSettlementConvention().empty()
253  ? Unadjusted
254  : parseBusinessDayConvention(optionData.exerciseFeeSettlementConvention());
255 
256  exercise_ = boost::make_shared<QuantExt::RebatedExercise>(*exercise_, exerciseDates_, rebates, feeSettlPeriod,
257  feeSettlCal, feeSettlBdc);
258 
259  // log rebates
260 
261  auto dbgEx = boost::static_pointer_cast<QuantExt::RebatedExercise>(exercise_);
262 
263  for (Size i = 0; i < exerciseDates_.size(); ++i) {
264  DLOG("Got rebate " << dbgEx->rebate(i) << " with payment date "
265  << QuantLib::io::iso_date(dbgEx->rebatePaymentDate(i)) << " (exercise date="
266  << QuantLib::io::iso_date(exerciseDates_[i]) << ") using rebate settl period "
267  << feeSettlPeriod << ", calendar " << feeSettlCal << ", convention " << feeSettlBdc);
268  }
269  }
270 }
