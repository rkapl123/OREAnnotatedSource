86 void TodaysMarket::initialise(const Date& asof) {
87 
88  asof_ = asof;
89 
90  calibrationInfo_ = boost::make_shared<TodaysMarketCalibrationInfo>();
91  calibrationInfo_->asof = asof;
92 
93  // Fixings
94 
95  if (loadFixings_) {
96  // Apply them now in case a curve builder needs them
97  LOG("Todays Market Loading Fixings");
98  applyFixings(loader_->loadFixings(), conventions_);
99  LOG("Todays Market Loading Fixing done.");
100  }
101 
102  // Dividends - apply them now in case a curve builder needs them
103 
104  LOG("Todays Market Loading Dividends");
105  applyDividends(loader_->loadDividends());
106  LOG("Todays Market Loading Dividends done.");
107 
108  // Add all FX quotes from the loader to Triangulation
109 
110  for (auto& md : loader_->loadQuotes(asof_)) {
111  if (md->asofDate() == asof_ && md->instrumentType() == MarketDatum::InstrumentType::FX_SPOT) {
112  boost::shared_ptr<FXSpotQuote> q = boost::dynamic_pointer_cast<FXSpotQuote>(md);
113  QL_REQUIRE(q, "Failed to cast " << md->name() << " to FXSpotQuote");
114  fxT_.addQuote(q->unitCcy() + q->ccy(), q->quote());
115  }
116  }
117 
118  // build the dependency graph for all configurations and build all FX Spots
119  DependencyGraph dg(asof_, params_, curveConfigs_, conventions_, iborFallbackConfig_);
120  map<string, string> buildErrors;
121 
122  for (const auto& configuration : params_->configurations()) {
123  // Build the graph of objects to build for the current configuration
124  dg.buildDependencyGraph(configuration.first, buildErrors);
125  }
126  dependencies_ = dg.dependencies();
127 
128  // build the fx spots in all configurations upfront (managing dependencies would be messy due to triangulation)
129 
130  for (const auto& configuration : params_->configurations()) {
131  Graph& g = dependencies_[configuration.first];
132  VertexIterator v, vend;
133  for (std::tie(v, vend) = boost::vertices(g); v != vend; ++v) {
134  if (g[*v].obj == MarketObject::FXSpot) {
135  try {
136  buildNode(configuration.first, g[*v]);
137  } catch (const std::exception& e) {
138  buildErrors[g[*v].curveSpec->name()] = e.what();
139  ALOG("error while building node " << g[*v] << " in configuration " << configuration.first << ": "
140  << e.what());
141  }
142  }
143  }
144  }
145 
146  // if market is not build lazily, sort the dependency graph and build the objects
147 
148  if (!lazyBuild_) {
149 
150  for (const auto& configuration : params_->configurations()) {
151 
152  LOG("Build objects in TodaysMarket configuration " << configuration.first);
153 
154  // Sort the graph topologically
155 
156  Graph& g = dependencies_[configuration.first];
157  IndexMap index = boost::get(boost::vertex_index, g);
158  std::vector<Vertex> order;
159  try {
160  boost::topological_sort(g, std::back_inserter(order));
161  } catch (const std::exception& e) {
162  // topological_sort() might have produced partial results, that we have to discard
163  order.clear();
164  // set error (most likely a circle), and output cycles if any
165  buildErrors["CurveDependencyGraph"] = "Topological sort of dependency graph failed for configuration " +
166  configuration.first + " (" + ore::data::to_string(e.what()) +
167  "). Got cylcle(s): " + getCycles(g);
168  }
169 
170  TLOG("Can build objects in the following order:");
171  for (auto const& m : order) {
172  TLOG("vertex #" << index[m] << ": " << g[m]);
173  }
174 
175  // Build the objects in the graph in topological order
176 
177  Size countSuccess = 0, countError = 0;
178  for (auto const& m : order) {
179  try {
180  buildNode(configuration.first, g[m]);
181  ++countSuccess;
182  DLOG("built node " << g[m] << " in configuration " << configuration.first);
183  } catch (const std::exception& e) {
184  if (g[m].curveSpec)
185  buildErrors[g[m].curveSpec->name()] = e.what();
186  else
187  buildErrors[g[m].name] = e.what();
188  ++countError;
189  ALOG("error while building node " << g[m] << " in configuration " << configuration.first << ": "
190  << e.what());
191  }
192  }
193 
194  LOG("Loaded CurvesSpecs: success: " << countSuccess << ", error: " << countError);
195  }
196 
197  } else {
198  LOG("Build objects in TodaysMarket lazily, i.e. when requested.");
199  }
200 
201  // output errors from initialisation phase
202 
203  if (!buildErrors.empty()) {
204  for (auto const& error : buildErrors) {
205  ALOG(StructuredCurveErrorMessage(error.first, "Failed to Build Curve", error.second));
206  }
207  if (!continueOnError_) {
208  string errStr;
209  for (auto const& error : buildErrors) {
210  errStr += "(" + error.first + ": " + error.second + "); ";
211  }
212  QL_FAIL("Cannot build all required curves! Building failed for: " << errStr);
213  }
214  }

218 void TodaysMarket::buildNode(const std::string& configuration, Node& node) const {
219 
220  // if the node is already built, there is nothing to do
221 
222  if (node.built)
223  return;
224 
225  if (node.curveSpec == nullptr) {
226 
227  // not spec-based node, this can only be a SwapIndexCurve
228 
229  QL_REQUIRE(node.obj == MarketObject::SwapIndexCurve,
230  "market object '" << node.obj << "' (" << node.name << ") without curve spec, this is unexpected.");
231  const string& swapIndexName = node.name;
232  const string& discountIndex = node.mapping;
233  addSwapIndex(swapIndexName, discountIndex, configuration);
234  DLOG("Added SwapIndex " << swapIndexName << " with DiscountingIndex " << discountIndex);
235  requiredSwapIndices_[configuration][swapIndexName] =
236  swapIndices_.at(std::make_pair(configuration, swapIndexName)).currentLink();
237 
238  } else {
239 
240  // spec-based node
241 
242  auto spec = node.curveSpec;
243  switch (spec->baseType()) {
244 
245  // Yield
246  case CurveSpec::CurveType::Yield: {
247  boost::shared_ptr<YieldCurveSpec> ycspec = boost::dynamic_pointer_cast<YieldCurveSpec>(spec);
248  QL_REQUIRE(ycspec, "Failed to convert spec " << *spec << " to yield curve spec");
249 
250  auto itr = requiredYieldCurves_.find(ycspec->name());
251  if (itr == requiredYieldCurves_.end()) {
252  DLOG("Building YieldCurve for asof " << asof_);
253  boost::shared_ptr<YieldCurve> yieldCurve =
254  boost::make_shared<YieldCurve>(asof_, *ycspec, *curveConfigs_, *loader_, *conventions_,
255  requiredYieldCurves_, requiredDefaultCurves_, fxT_, referenceData_,
256  iborFallbackConfig_, preserveQuoteLinkage_, requiredDiscountCurves_);
257  calibrationInfo_->yieldCurveCalibrationInfo[ycspec->name()] = yieldCurve->calibrationInfo();
258  itr = requiredYieldCurves_.insert(make_pair(ycspec->name(), yieldCurve)).first;
259  DLOG("Added YieldCurve \"" << ycspec->name() << "\" to requiredYieldCurves map");
260  if (itr->second->currency().code() != ycspec->ccy()) {
261  WLOG("Warning: YieldCurve has ccy " << itr->second->currency() << " but spec has ccy "
262  << ycspec->ccy());
263  }
264  }
265 
266  if (node.obj == MarketObject::DiscountCurve) {
267  DLOG("Adding DiscountCurve(" << node.name << ") with spec " << *ycspec << " to configuration "
268  << configuration);
269  yieldCurves_[make_tuple(configuration, YieldCurveType::Discount, node.name)] = itr->second->handle();
270  // Also add to requiredDiscountCurves
271  requiredDiscountCurves_.insert(make_pair(node.name, itr->second));
272 
273  } else if (node.obj == MarketObject::YieldCurve) {
274  DLOG("Adding YieldCurve(" << node.name << ") with spec " << *ycspec << " to configuration "
275  << configuration);
276  yieldCurves_[make_tuple(configuration, YieldCurveType::Yield, node.name)] = itr->second->handle();
277 
278  } else if (node.obj == MarketObject::IndexCurve) {
279  DLOG("Adding Index(" << node.name << ") with spec " << *ycspec << " to configuration "
280  << configuration);
281  // ibor fallback handling
282  auto tmpIndex = parseIborIndex(node.name, itr->second->handle(),
283  conventions_->has(node.name, Convention::Type::IborIndex) ||
284  conventions_->has(node.name, Convention::Type::OvernightIndex)
285  ? conventions_->get(node.name)
286  : nullptr);
287  if (iborFallbackConfig_.isIndexReplaced(node.name, asof_)) {
288  auto fallbackData = iborFallbackConfig_.fallbackData(node.name);
289  boost::shared_ptr<IborIndex> rfrIndex;
290  auto f = iborIndices_.find(make_pair(configuration, fallbackData.rfrIndex));
291  if (f == iborIndices_.end()) {
292  f = iborIndices_.find(make_pair(Market::defaultConfiguration, fallbackData.rfrIndex));
293  QL_REQUIRE(f != iborIndices_.end(),
294  "Failed to build ibor fallback index '"
295  << node.name << "', did not find rfr index '" << fallbackData.rfrIndex
296  << "' in configuration '" << configuration
297  << "' or default - is the rfr index configuration in todays market parameters?");
298  }
299  auto oi = boost::dynamic_pointer_cast<OvernightIndex>(*f->second);
300  QL_REQUIRE(oi,
301  "Found rfr index '"
302  << fallbackData.rfrIndex << "' as falback for ibor index '" << node.name
303  << "', but this is not an overnight index. Are the fallback rules correct here?");
304  tmpIndex = boost::make_shared<QuantExt::FallbackIborIndex>(
305  tmpIndex, oi, fallbackData.spread, fallbackData.switchDate,
306  iborFallbackConfig_.useRfrCurveInTodaysMarket());
307  TLOG("built ibor fall back index for '" << node.name << "' in configuration " << configuration
308  << " using rfr index '" << fallbackData.rfrIndex
309  << "', spread " << fallbackData.spread
310  << ", will use rfr curve in t0 market: " << std::boolalpha
311  << iborFallbackConfig_.useRfrCurveInTodaysMarket());
312  }
313  iborIndices_[make_pair(configuration, node.name)] = Handle<IborIndex>(tmpIndex);
314  } else {
315  QL_FAIL("unexpected market object type '"
316  << node.obj << "' for yield curve, should be DiscountCurve, YieldCurve, IndexCurve");
317  }
318  break;
319  }
320 
321  // FX Spot
322  case CurveSpec::CurveType::FX: {
323  boost::shared_ptr<FXSpotSpec> fxspec = boost::dynamic_pointer_cast<FXSpotSpec>(spec);
324  QL_REQUIRE(fxspec, "Failed to convert spec " << *spec << " to fx spot spec");
325  auto itr = requiredFxSpots_.find(fxspec->name());
326  if (itr == requiredFxSpots_.end()) {
327  DLOG("Building FXSpot for asof " << asof_);
328  boost::shared_ptr<FXSpot> fxSpot = boost::make_shared<FXSpot>(asof_, *fxspec, fxT_);
329  itr = requiredFxSpots_.insert(make_pair(fxspec->name(), fxSpot)).first;
330  fxT_.addQuote(fxspec->subName().substr(0, 3) + fxspec->subName().substr(4, 3), itr->second->handle());
331  }
332  LOG("Adding FXSpot (" << node.name << ") with spec " << *fxspec << " to configuration " << configuration);
333  fxSpots_[configuration].addQuote(node.name, itr->second->handle());
334  break;
335  }
336 
337  // FX Vol
338  case CurveSpec::CurveType::FXVolatility: {
339  boost::shared_ptr<FXVolatilityCurveSpec> fxvolspec =
340  boost::dynamic_pointer_cast<FXVolatilityCurveSpec>(spec);
341  QL_REQUIRE(fxvolspec, "Failed to convert spec " << *spec);
342 
343  // have we built the curve already ?
344  auto itr = requiredFxVolCurves_.find(fxvolspec->name());
345  if (itr == requiredFxVolCurves_.end()) {
346  DLOG("Building FXVolatility for asof " << asof_);
347  boost::shared_ptr<FXVolCurve> fxVolCurve = boost::make_shared<FXVolCurve>(
348  asof_, *fxvolspec, *loader_, *curveConfigs_, fxT_, requiredYieldCurves_, requiredFxVolCurves_,
349  requiredCorrelationCurves_, *conventions_);
350  calibrationInfo_->fxVolCalibrationInfo[fxvolspec->name()] = fxVolCurve->calibrationInfo();
351  itr = requiredFxVolCurves_.insert(make_pair(fxvolspec->name(), fxVolCurve)).first;
352  }
353 
354  DLOG("Adding FXVol (" << node.name << ") with spec " << *fxvolspec << " to configuration "

356  fxVols_[make_pair(configuration, node.name)] =
357  Handle<BlackVolTermStructure>(itr->second->volTermStructure());

366

368  if (itr == requiredSwaptionVolCurves_.end()) {
369  DLOG("Building Swaption Volatility for asof " << asof_);
370  boost::shared_ptr<SwaptionVolCurve> swaptionVolCurve = boost::make_shared<SwaptionVolCurve>(

749 void TodaysMarket::require(const MarketObject o, const string& name, const string& configuration) const {
750 
751  // if the market is not lazily built, do nothing
752 
753  if (!lazyBuild_)
754  return;
755 
756  // search the node (o, name) in the dependency graph
757 
758  DLOG("market object " << o << "(" << name << ") required for configuration '" << configuration << "'");
759 
760  auto tmp = dependencies_.find(configuration);
761  if (tmp == dependencies_.end()) {
762  if (configuration != Market::defaultConfiguration) {
763  ALOG(StructuredCurveErrorMessage(ore::data::to_string(o) + "(" + name + ")", "Failed to Build Curve",
764  "Configuration '" + configuration +
765  "' not known, retry with default configuration."));
766  require(o, name, Market::defaultConfiguration);
767  return;
768  } else {
769  ALOG(StructuredCurveErrorMessage(ore::data::to_string(o) + "(" + name + ")", "Failed to Build Curve",
770  "Configuration 'default' not known, this is unexpected. Do nothing."));
771  return;
772  }
773  }
774 
775  Vertex node = nullptr;
776 
777  Graph& g = tmp->second;
778  IndexMap index = boost::get(boost::vertex_index, g);
779 
780  VertexIterator v, vend;
781  bool found = false;
782  for (std::tie(v, vend) = boost::vertices(g); v != vend; ++v) {
783  if (g[*v].obj == o) {
784  if (o == MarketObject::Correlation) {
785  // split the required name and the node name and compare the tokens
786  found = getCorrelationTokens(name) == getCorrelationTokens(g[*v].name);
787  } else {
788  found = (g[*v].name == name);
789  }
790  if (found) {
791  node = *v;
792  break;
793  }
794  }
795  }
796 
797  // if we did not find a node, we retry with the default configuration, as required by the interface
798 
799  if (!found && configuration != Market::defaultConfiguration) {
800  DLOG("not found, retry with default configuration");
801  require(o, name, Market::defaultConfiguration);
802  return;
803  }
804 
805  // if we still have no node, we do nothing, the error handling is done in MarketImpl
806 
807  if (!found) {
808  DLOG("not found, do nothing");
809  return;
810  }
811 
812  // if the node is already built, we are done
813 
814  if (g[node].built) {
815  DLOG("node already built, do nothing.");
816  return;
817  }
818 
819  // run a DFS from the found node to identify the required nodes to be built and get a possible order to do this
820 
821  map<string, string> buildErrors;
822  std::vector<Vertex> order;
823  bool foundCycle = false;
824 
825  DfsVisitor<Vertex> dfs(order, foundCycle);
826  auto colorMap = boost::make_vector_property_map<boost::default_color_type>(index);
827  boost::depth_first_visit(g, node, dfs, colorMap);
828 
829  if (foundCycle) {
830  order.clear();
831  buildErrors[g[node].curveSpec ? g[node].curveSpec->name() : g[node].name] = "found cycle";
832  }
833 
834  // build the nodes
835 
836  TLOG("Can build objects in the following order:");
837  for (auto const& m : order) {
838  TLOG("vertex #" << index[m] << ": " << g[m] << (g[m].built ? " (already built)" : " (not yet built)"));
839  }
840 
841  Size countSuccess = 0, countError = 0;
842  for (auto const& m : order) {
843  if (g[m].built)
844  continue;
845  try {
846  buildNode(configuration, g[m]);
847  ++countSuccess;
848  DLOG("built node " << g[m] << " in configuration " << configuration);
849  } catch (const std::exception& e) {
850  if (g[m].curveSpec)
851  buildErrors[g[m].curveSpec->name()] = e.what();
852  else
853  buildErrors[g[m].name] = e.what();
854  ++countError;
855  ALOG("error while building node " << g[m] << " in configuration " << configuration << ": " << e.what());
856  }
857  }
858 
859  if (countSuccess + countError > 0) {
860  DLOG("Loaded CurvesSpecs: success: " << countSuccess << ", error: " << countError);
861  }
862 
863  // output errors
864 
865  if (!buildErrors.empty()) {
866  for (auto const& error : buildErrors) {
867  ALOG(StructuredCurveErrorMessage(error.first, "Failed to Build Curve", error.second));
868  }
869  if (!continueOnError_) {
870  string errStr;
871  for (auto const& error : buildErrors) {
872  errStr += "(" + error.first + ": " + error.second + "); ";
873  }
874  QL_FAIL("Cannot build all required curves! Building failed for: " << errStr);
875  }
876  }
877 } // TodaysMarket::require()
