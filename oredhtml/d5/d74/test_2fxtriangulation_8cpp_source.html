90 BOOST_AUTO_TEST_CASE(testValues) {
91 
92  // Tolerance for comparisons
93  Real tol = 1e-12;
94 
95  // Check inverse
96  BOOST_CHECK_CLOSE(fx.getQuote("USDEUR")->value(), 1.0 / 1.0861, tol);

98 
99  // Check Triangulation
100  BOOST_CHECK_CLOSE(fx.getQuote("USDJPY")->value(), 128.51 / 1.0861, tol);
101  BOOST_CHECK_CLOSE(fx.getQuote("JPYUSD")->value(), 1.0861 / 128.51, tol);
102  BOOST_CHECK_CLOSE(fx.getQuote("USDGBP")->value(), 0.74519 / 1.0861, tol);
103  BOOST_CHECK_CLOSE(fx.getQuote("GBPUSD")->value(), 1.0861 / 0.74519, tol);
104  BOOST_CHECK_CLOSE(fx.getQuote("NOKSEK")->value(), 9.2640 / 9.6810, tol);
105 
106  // Check Triangulation where the EUR quote is reversed
107  BOOST_CHECK_CLOSE(fx.getQuote("ZZZUSD")->value(), 3.141 * 1.0861, tol);
108  BOOST_CHECK_CLOSE(fx.getQuote("USDZZZ")->value(), 1 / (3.141 * 1.0861), tol);

111 BOOST_AUTO_TEST_CASE(testMoreThanOneStep) {
112 
113  // Larger tolerance for multiple steps
114  Real tol = 1e-8;
115 
116  // Check that we don't handle more than one step
117  // EURUSD + EURAUD + AUDNZD => USDNZD
118  BOOST_CHECK_THROW(fx.getQuote("USDNZD"), QuantLib::Error);
119  // but if we cache EURNZD first....
120  BOOST_CHECK_CLOSE(fx.getQuote("EURNZD")->value(), 1.6450, tol);
121  // then we should be able to get it with one step
122  BOOST_CHECK_CLOSE(fx.getQuote("USDNZD")->value(), 1.6450 / 1.0861, tol);
