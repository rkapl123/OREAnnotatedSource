130 // MXN TIIE
131 // If tenor equates to 28 Days, i.e. tenor is 4W or 28D, ensure that the index is created
132 // with a tenor of 4W under the hood. Things work better this way especially cap floor stripping.
133 // We do the same with 91D -> 3M (a la KRW CD below)
134 template <> class IborIndexParserWithPeriod<MXNTiie> : public IborIndexParser {
135 public:
136  boost::shared_ptr<IborIndex> build(Period p, const Handle<YieldTermStructure>& h) const override {
137  if (p.units() == Days && p.length() == 28) {
138  return boost::make_shared<MXNTiie>(4 * Weeks, h);
139  } else if (p.units() == Days && p.length() == 91) {
140  return boost::make_shared<MXNTiie>(3 * Months, h);
141  } else {
142  return boost::make_shared<MXNTiie>(p, h);
143  }
144  }
145 
146  string family() const override { return MXNTiie(4 * Weeks).familyName(); }
147 };

164 // CNY REPOFIX
165 // If tenor equates to 7 Days, i.e. tenor is 1W or 7D, ensure that the index is created

174  } else {
175  return boost::make_shared<CNYRepoFix>(p, h);

177  }

182 // Helper function to check that index name to index object is a one-to-one mapping
183 void checkOneToOne(const map<string, boost::shared_ptr<OvernightIndex>>& onIndices,
184  const map<string, boost::shared_ptr<IborIndexParser>>& iborIndices) {
185 
186  // Should not attempt to add the same family name to the set if the provided mappings are one to one
187  set<string> familyNames;
188 
189  for (const auto& kv : onIndices) {
190  auto p = familyNames.insert(kv.second->familyName());
191  QL_REQUIRE(p.second, "Duplicate mapping for overnight index family " << *p.first << " not allowed");
192  }
193 
194  for (const auto& kv : iborIndices) {
195  auto p = familyNames.insert(kv.second->family());
196  QL_REQUIRE(p.second, "Duplicate mapping for ibor index family " << *p.first << " not allowed");
197  }
198 }

239 boost::shared_ptr<IborIndex> parseIborIndex(const string& s, string& tenor, const Handle<YieldTermStructure>& h,
240  const boost::shared_ptr<Convention>& c) {
241 
242  // Check the index string is of the required form before doing anything
243  vector<string> tokens;
244  split(tokens, s, boost::is_any_of("-"));
245  QL_REQUIRE(tokens.size() == 2 || tokens.size() == 3,
246  "Two or three tokens required in " << s << ": CCY-INDEX or CCY-INDEX-TERM");
247 
248  // Variables used below
249  string indexStem = tokens[0] + "-" + tokens[1];
250  if (tokens.size() == 3) {
251  tenor = tokens[2];
252  } else {
253  tenor = "";
254  }
255 
256  // if we have a convention given, set up the index using this convention, this overrides the parsing from
257  // hardcoded strings below if there is an overlap
258  if (c) {
259  QL_REQUIRE(c->id() == s, "ibor index convention id ('"
260  << c->id() << "') not matching ibor index string to parse ('" << s << "'");
261  Currency ccy = parseCurrency(tokens[0]);
262  if (auto conv = boost::dynamic_pointer_cast<OvernightIndexConvention>(c)) {
263  QL_REQUIRE(tenor.empty(), "no tenor allowed for convention based overnight index ('" << s << "')");
264  return boost::make_shared<OvernightIndex>(tokens[0] + "-" + tokens[1], conv->settlementDays(), ccy,
265  parseCalendar(conv->fixingCalendar()),
266  parseDayCounter(conv->dayCounter()), h);
267 
268  } else if (auto conv = boost::dynamic_pointer_cast<IborIndexConvention>(c)) {
269  QL_REQUIRE(!tenor.empty(), "no tenor given for convention based Ibor index ('" << s << "'");
270  return boost::make_shared<IborIndex>(tokens[0] + "-" + tokens[1], parsePeriod(tenor),
271  conv->settlementDays(), ccy, parseCalendar(conv->fixingCalendar()),
272  parseBusinessDayConvention(conv->businessDayConvention()),
273  conv->endOfMonth(), parseDayCounter(conv->dayCounter()), h);
274  } else {
275  QL_FAIL("invalid convention passed to parseIborIndex(): expected OvernightIndexConvention or "
276  "IborIndexConvention");
277  }
278  }
279

555 boost::shared_ptr<BondIndex> parseBondIndex(const string& name) {
556 
557  // Make sure the prefix is correct
558  string prefix = name.substr(0, 5);
559  QL_REQUIRE(prefix == "BOND-", "A bond index string must start with 'BOND-' but got " << prefix);
560 
561  // Now take the remainder of the string
562  // for spot indices, this should just be the bond name
563  // for future indices, this is of the form NAME-YYYY-MM or NAME-YYYY-MM-DD where NAME is the commodity name
564  // (possibly containing hyphens) and YYYY-MM(-DD) is the expiry date of the futures contract
565  Date expiry;
566  string nameWoPrefix = name.substr(5);
567  string bondName = nameWoPrefix;
568 
569  // Check for form NAME-YYYY-MM-DD
570  if (nameWoPrefix.size() > 10) {
571  string test = nameWoPrefix.substr(nameWoPrefix.size() - 10);
572  if (boost::regex_match(test, boost::regex("\\d{4}-\\d{2}-\\d{2}"))) {
573  expiry = parseDate(test);
574  bondName = nameWoPrefix.substr(0, nameWoPrefix.size() - test.size() - 1);
575  }
576  }
577 
578  // Check for form NAME-YYYY-MM if NAME-YYYY-MM-DD failed
579  if (expiry == Date() && nameWoPrefix.size() > 7) {
580  string test = nameWoPrefix.substr(nameWoPrefix.size() - 7);
581  if (boost::regex_match(test, boost::regex("\\d{4}-\\d{2}"))) {
582  expiry = parseDate(test + "-01");
583  bondName = nameWoPrefix.substr(0, nameWoPrefix.size() - test.size() - 1);
584  }
585  }
586 
587  // Create and return the required future index
588  if (expiry != Date()) {
589  return boost::make_shared<BondFuturesIndex>(expiry, bondName);
590  } else {
591  return boost::make_shared<BondIndex>(bondName);
592  }

596 boost::shared_ptr<QuantExt::CommodityIndex> parseCommodityIndex(const string& name, bool hasPrefix,
597  const Calendar& cal,
598  const Handle<PriceTermStructure>& ts,
599  const Conventions& conventions) {
600 
601  // Whether we check for "COMM-" prefix depends on hasPrefix.
602  string commName = name;
603  if (hasPrefix) {
604  // Make sure the prefix is correct
605  string prefix = name.substr(0, 5);
606  QL_REQUIRE(prefix == "COMM-", "A commodity index string must start with 'COMM-' but got " << prefix);
607  commName = name.substr(5);
608  }
609 
610  // Now take the remainder of the string
611  // for spot indices, this should just be the commodity name (possibly containing hyphens)
612  // for future indices, this is of the form NAME-YYYY-MM or NAME-YYYY-MM-DD where NAME is the commodity name
613  // (possibly containing hyphens) and YYYY-MM(-DD) is the expiry date of the futures contract
614  Date expiry;
615 
616  // Check for form NAME-YYYY-MM-DD
617  if (commName.size() > 10) {
618  string test = commName.substr(commName.size() - 10);
619  if (boost::regex_match(test, boost::regex("\\d{4}-\\d{2}-\\d{2}"))) {
620  expiry = parseDate(test);
621  commName = commName.substr(0, commName.size() - test.size() - 1);
622  }
623  }
624 
625  // Check for form NAME-YYYY-MM if NAME-YYYY-MM-DD failed
626  if (expiry == Date() && commName.size() > 7) {
627  string test = commName.substr(commName.size() - 7);
628  if (boost::regex_match(test, boost::regex("\\d{4}-\\d{2}"))) {
629  expiry = parseDate(test + "-01");
630  commName = commName.substr(0, commName.size() - test.size() - 1);
631  }
632  }
633 
634  // Name to use when creating the index. This may be updated if we have a commodity future convention and IndexName
635  // is provided by the convention.
636  string indexName = commName;
637 
638  // Do we have a commodity future convention for the commodity.
639  pair<bool, boost::shared_ptr<Convention>> p = conventions.get(commName, Convention::Type::CommodityFuture);
640  boost::shared_ptr<CommodityFutureConvention> convention;
641  if (p.first) {
642 
643  convention = boost::dynamic_pointer_cast<CommodityFutureConvention>(p.second);
644 
645  if (!convention->indexName().empty())
646  indexName = convention->indexName();
647 
648  // If we have provided OffPeakPowerIndexData, we use that to construct the off peak power commodity index.
649  if (convention->offPeakPowerIndexData()) {
650 
651  const auto& oppIdxData = *convention->offPeakPowerIndexData();
652 
653  if (expiry == Date()) {
654  // If expiry is still not set use any date (off peak index is calendar daily)
655  expiry = Settings::instance().evaluationDate();
656  }
657  string suffix = "-" + to_string(expiry);
658 
659  auto offPeakIndex = boost::dynamic_pointer_cast<CommodityFuturesIndex>(parseCommodityIndex(
660  oppIdxData.offPeakIndex() + suffix, conventions, false));
661  auto peakIndex = boost::dynamic_pointer_cast<CommodityFuturesIndex>(parseCommodityIndex(
662  oppIdxData.peakIndex() + suffix, conventions, false));
663 
664  return boost::make_shared<OffPeakPowerIndex>(indexName, expiry, offPeakIndex, peakIndex,
665  oppIdxData.offPeakHours(), oppIdxData.peakCalendar(), ts);
666  }
667  }
668 
669  // Create and return the required future index
670  if (expiry != Date() || convention) {
671 
672  // If expiry is empty, just use any valid expiry.
673  if (expiry == Date()) {
674  ConventionsBasedFutureExpiry feCalc(*convention);
675  expiry = feCalc.nextExpiry();
676  }
677 
678  bool keepDays = convention && convention->contractFrequency() == Daily;
679 
680  Calendar cdr = cal;
681  if (convention && cdr == NullCalendar()) {
682  cdr = convention->calendar();
683  }
684 
685  return boost::make_shared<CommodityFuturesIndex>(indexName, expiry, cdr, keepDays, ts);
686 
687  } else {
688  return boost::make_shared<CommoditySpotIndex>(commName, cal, ts);
689  }
690 }

697 boost::shared_ptr<Index> parseIndex(const string& s, const boost::shared_ptr<Conventions>& conventions) {
698  boost::shared_ptr<QuantLib::Index> ret_idx;
699  // if we have an ibor index convention we parse s using this (and throw if we don't succeed)
700  if (conventions && (conventions->has(s, Convention::Type::IborIndex) ||
701  conventions->has(s, Convention::Type::OvernightIndex))) {
702  ret_idx = parseIborIndex(s, Handle<YieldTermStructure>(), conventions->get(s));
703  } else {
704  // otherwise we try to parse the ibor index without conventions
705  try {
706  ret_idx = parseIborIndex(s, Handle<YieldTermStructure>(), nullptr);
707  } catch (...) {
708  }
709  }
710  if (!ret_idx) {
711  // if we have a swap index convention we parse s using this (and throw if we don't succeed)
712  if (conventions && conventions->has(s, Convention::Type::SwapIndex)) {
713  auto c = boost::dynamic_pointer_cast<SwapIndexConvention>(conventions->get(s));
714  QL_REQUIRE(conventions->has(c->conventions(), Convention::Type::Swap),
715  "do not have swap conventions for '" << c->conventions()
716  << "', required from swap index convention '" << s << "'");
717  ret_idx =
718  parseSwapIndex(s, Handle<YieldTermStructure>(), Handle<YieldTermStructure>(),
719  boost::dynamic_pointer_cast<IRSwapConvention>(conventions->get(c->conventions())), c);
720  } else {
721  // otherwise try to parse a swap index without conventions
722  try {
723  ret_idx = parseSwapIndex(s, Handle<YieldTermStructure>(), Handle<YieldTermStructure>(), nullptr);
724  } catch (...) {
725  }
726  }
727  }
728  if (!ret_idx) {
729  try {
730  ret_idx = parseZeroInflationIndex(s, false, Handle<ZeroInflationTermStructure>(), conventions);
731  } catch (...) {
732  }
733  }
734  if (!ret_idx) {
735  try {
736  ret_idx = parseFxIndex(s);
737  } catch (...) {
738  }
739  }
740  if (!ret_idx) {
741  try {
742  ret_idx = parseEquityIndex(s);
743  } catch (...) {
744  }
745  }
746  if (!ret_idx) {
747  try {
748  ret_idx = parseBondIndex(s);
749  } catch (...) {
750  }
751  }
752  if (!ret_idx) {
753  try {
754  ret_idx = conventions ? parseCommodityIndex(s, *conventions) : parseCommodityIndex(s);
755  } catch (...) {
756  }
757  }
758  if (!ret_idx) {
759  try {
760  ret_idx = parseGenericIndex(s);
761  } catch (...) {
762  }
763  }
764  QL_REQUIRE(ret_idx, "parseIndex \"" << s << "\" not recognized");
765  return ret_idx;
766 }

791 string internalIndexName(const string& indexName) {
792 
793  // Check that the indexName string is of the required form
794  vector<string> tokens;
795  split(tokens, indexName, boost::is_any_of("-"));
796  QL_REQUIRE(tokens.size() == 2 || tokens.size() == 3,
797  "Two or three tokens required in " << indexName << ": CCY-INDEX or CCY-INDEX-TERM");
798 
799  // Static map of allowable alternative external names to our unique internal name
800  static map<string, string> m = {{"DKK-TNR", "DKK-DKKOIS"}, {"EUR-EURIB", "EUR-EURIBOR"}, {"CAD-BA", "CAD-CDOR"},
801  {"EUR-ESTR", "EUR-ESTER"}, {"EUR-STR", "EUR-ESTER"}};
802 
803  // Is start of indexName covered by the map? If so, update it.
804  string tmpName = tokens[0] + "-" + tokens[1];
805  if (m.count(tmpName) == 1) {
806  tmpName = m.at(tmpName);
807  }
808 
809  // If there were only two tokens, return the possibly updated two tokens.
810  if (tokens.size() == 2) {
811  return tmpName;
812  }
813 
814  // Check if we have an overnight index
815  // This covers cases like USD-FedFunds-1D and returns USD-FedFunds
816  // (no need to check convention based overnight indices, they are always of the form CCY-INDEX)
817  if (isOvernightIndex(tmpName)) {
818  Period p = parsePeriod(tokens[2]);
819  QL_REQUIRE(p == 1 * Days,
820  "The period " << tokens[2] << " is not compatible with the overnight index " << tmpName);
821  return tmpName;
822  }
823 
824  // Allow USD-SIFMA-1W or USD-SIFMA-7D externally. USD-SIFMA is used internally.
825  if (tmpName == "USD-SIFMA" && (tokens[2] == "1W" || tokens[2] == "7D")) {
826  return tmpName;
827  }
828 
829  return tmpName + "-" + tokens[2];
830 }
