26 void Envelope::fromXML(XMLNode* node) {
27  XMLUtils::checkNode(node, "Envelope");
28  counterparty_ = XMLUtils::getChildValue(node, "CounterParty", true);
29  nettingSetId_ = XMLUtils::getChildValue(node, "NettingSetId", false);
30 
31  portfolioIds_.clear();
32  XMLNode* portfolioNode = XMLUtils::getChildNode(node, "PortfolioIds");
33  if (portfolioNode) {
34  for (auto const& c : XMLUtils::getChildrenNodes(portfolioNode, "PortfolioId"))
35  portfolioIds_.insert(XMLUtils::getNodeValue(c));
36  }
37 
38  additionalFields_.clear();
39  XMLNode* additionalNode = XMLUtils::getChildNode(node, "AdditionalFields");
40  if (additionalNode) {
41  for (XMLNode* child = XMLUtils::getChildNode(additionalNode); child; child = XMLUtils::getNextSibling(child)) {
42  additionalFields_[XMLUtils::getNodeName(child)] = XMLUtils::getNodeValue(child);
43  }
44  }
45 }

47 XMLNode* Envelope::toXML(XMLDocument& doc) {
48  XMLNode* node = doc.allocNode("Envelope");
49  XMLUtils::addChild(doc, node, "CounterParty", counterparty_);
50  XMLUtils::addChild(doc, node, "NettingSetId", nettingSetId_);
51  XMLNode* portfolioNode = doc.allocNode("PortfolioIds");
52  XMLUtils::appendNode(node, portfolioNode);
53  for (const auto& p : portfolioIds_)
54  XMLUtils::addChild(doc, portfolioNode, "PortfolioId", p);
55  XMLNode* additionalNode = doc.allocNode("AdditionalFields");
56  XMLUtils::appendNode(node, additionalNode);
57  for (const auto& it : additionalFields_)
58  XMLUtils::addChild(doc, additionalNode, it.first, it.second);
59  return node;
60 }
