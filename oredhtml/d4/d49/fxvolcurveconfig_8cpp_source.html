53 const vector<string>& FXVolatilityCurveConfig::quotes() {
54  if (quotes_.size() == 0) {
55  vector<string> tokens;
56  boost::split(tokens, fxSpotID(), boost::is_any_of("/"));
57  QL_REQUIRE(tokens.size() == 3, "Expected 3 tokens FX/CCY1/CCY2 in fxSpotID (" << fxSpotID() << ")");
58  quotes_.push_back("FX/RATE/" + tokens[1] + "/" + tokens[2]);
59  string base = "FX_OPTION/RATE_LNVOL/" + tokens[1] + "/" + tokens[2] + "/";
60  for (auto e : expiries_) {
61  quotes_.push_back(base + e + "/ATM");
62  if (dimension_ == Dimension::SmileVannaVolga || dimension_ == Dimension::SmileBFRR) {
63  for (auto const& d : smileDelta_) {
64  quotes_.push_back(base + e + "/" + to_string(d) + "RR");
65  quotes_.push_back(base + e + "/" + to_string(d) + "BF");
66  }
67  } else if (dimension_ == Dimension::SmileDelta) {
68  for (auto d : deltas_) {
69  quotes_.push_back(base + e + "/" + d);
70  }
71  }
72  }
73  }
74  return quotes_;
75 }

77 void FXVolatilityCurveConfig::fromXML(XMLNode* node) {
78  XMLUtils::checkNode(node, "FXVolatility");
79 
80  curveID_ = XMLUtils::getChildValue(node, "CurveId", true);
81  curveDescription_ = XMLUtils::getChildValue(node, "CurveDescription", true);
82  string dim = XMLUtils::getChildValue(node, "Dimension", true);
83  string cal = XMLUtils::getChildValue(node, "Calendar");
84  string smileInterp = XMLUtils::getChildValue(node, "SmileInterpolation");
85 
86  fxSpotID_ = XMLUtils::getChildValue(node, "FXSpotID", true);
87 
88  vector<string> tokens;
89  boost::split(tokens, fxSpotID_, boost::is_any_of("/"));
90  QL_REQUIRE(tokens.size() == 3, "Expected 3 tokens FX/CCY1/CCY2 in fxSpotID (" << fxSpotID_ << ")");
91  if (cal == "") {
92  cal = tokens[1] + "," + tokens[2];
93  }
94  calendar_ = parseCalendar(cal);
95 
96  string dc = XMLUtils::getChildValue(node, "DayCounter");
97  if (dc == "")
98  dc = "A365";
99  dayCounter_ = parseDayCounter(dc);
100 
101  if (dim == "ATMTriangulated") {
102  dimension_ = Dimension::ATMTriangulated;
103  baseVolatility1_ = XMLUtils::getChildValue(node, "BaseVolatility1", true);
104  baseVolatility2_ = XMLUtils::getChildValue(node, "BaseVolatility2", true);
105 
106  string fxIndexTag = XMLUtils::getChildValue(node, "FXIndexTag");
107  if (fxIndexTag == "")
108  fxIndexTag = "GENERIC";
109  fxIndexTag_ = fxIndexTag;
110  } else {
111  if (dim == "ATM") {
112  dimension_ = Dimension::ATM;
113  } else if (dim == "Smile") {
114 
115  conventionsID_ = XMLUtils::getChildValue(node, "Conventions", false);
116  string smileType = XMLUtils::getChildValue(node, "SmileType");
117  if (smileType == "" || smileType == "VannaVolga") {
118  dimension_ = Dimension::SmileVannaVolga;
119 
120  // only read smile interpolation method if dimension is smile.
121  if (smileInterp == "") {
122  smileInterpolation_ = SmileInterpolation::VannaVolga2; // default to VannaVolga 2nd approximation
123  } else if (smileInterp == "VannaVolga1") {
124  smileInterpolation_ = SmileInterpolation::VannaVolga1;
125  } else if (smileInterp == "VannaVolga2") {
126  smileInterpolation_ = SmileInterpolation::VannaVolga2;
127  } else {
128  QL_FAIL("SmileInterpolation " << smileInterp << " not supported");
129  }
130 
131  string sDelta = XMLUtils::getChildValue(node, "SmileDelta");
132  if (sDelta == "")
133  smileDelta_ = {25};
134  else
135  smileDelta_ = parseListOfValues<Size>(sDelta, &parseInteger);
136  } else if (smileType == "Delta") {
137  dimension_ = Dimension::SmileDelta;
138  // only read smile interpolation method if dimension is smile.
139  if (smileInterp == "" || smileInterp == "Linear") {
140  smileInterpolation_ = SmileInterpolation::Linear;
141  } else if (smileInterp == "Cubic") {
142  smileInterpolation_ = SmileInterpolation::Cubic;
143  } else {
144  QL_FAIL("SmileInterpolation " << smileInterp << " not supported");
145  }
146  deltas_ = XMLUtils::getChildrenValuesAsStrings(node, "Deltas", true);
147 
148  // check that these are valid deltas
149  for (auto d : deltas_) {
150  QL_REQUIRE(d == "ATM" || d.back() == 'P' || d.back() == 'C',
151  "this is not a valid value for delta, " << d);
152  if (d != "ATM") {
153  parseReal(d.substr(0, d.size() - 1));
154  }
155  }
156  } else if (smileType == "BFRR") {
157  dimension_ = Dimension::SmileBFRR;
158  if (smileInterp == "" || smileInterp == "Cubic") {
159  smileInterpolation_ = SmileInterpolation::Cubic;
160  } else if (smileInterp == "Linear") {
161  smileInterpolation_ = SmileInterpolation::Linear;
162  } else {
163  QL_FAIL("SmileInterpolation " << smileInterp << " not supported");
164  }
165  string sDelta = XMLUtils::getChildValue(node, "SmileDelta");
166  if (sDelta == "")
167  smileDelta_ = {10, 25};
168  else
169  smileDelta_ = parseListOfValues<Size>(sDelta, &parseInteger);
170  } else {
171  QL_FAIL("SmileType '" << smileType << "' not supported, expected VannaVolga, Delta, BFRR");
172  }
173  } else {
174  QL_FAIL("Dimension " << dim << " not supported yet");
175  }
176  expiries_ = XMLUtils::getChildrenValuesAsStrings(node, "Expiries", true);
177 
178  bool curvesRequired = dimension_ == Dimension::SmileVannaVolga || dimension_ == Dimension::SmileDelta ||
179  dimension_ == Dimension::SmileBFRR;
180  fxForeignYieldCurveID_ = XMLUtils::getChildValue(node, "FXForeignCurveID", curvesRequired);
181  fxDomesticYieldCurveID_ = XMLUtils::getChildValue(node, "FXDomesticCurveID", curvesRequired);
182  }
183 
184  if (auto tmp = XMLUtils::getChildNode(node, "Report")) {
185  reportConfig_.fromXML(tmp);
186  }
187 
188  populateRequiredCurveIds();
189 }

191 XMLNode* FXVolatilityCurveConfig::toXML(XMLDocument& doc) {
192  XMLNode* node = doc.allocNode("FXVolatility");
193 
194  XMLUtils::addChild(doc, node, "CurveId", curveID_);
195  XMLUtils::addChild(doc, node, "CurveDescription", curveDescription_);
196  if (dimension_ == Dimension::ATM) {
197  XMLUtils::addChild(doc, node, "Dimension", "ATM");
198  } else if (dimension_ == Dimension::ATMTriangulated) {
199  XMLUtils::addChild(doc, node, "Dimension", "ATMTriangulated");
200  XMLUtils::addChild(doc, node, "FXSpotID", fxSpotID_);
201  XMLUtils::addChild(doc, node, "FXIndexTag", fxIndexTag_);
202  XMLUtils::addChild(doc, node, "BaseVolatility1", baseVolatility1_);
203  XMLUtils::addChild(doc, node, "BaseVolatility2", baseVolatility2_);
204 
205  return node;
206  } else if (dimension_ == Dimension::SmileVannaVolga) {
207  XMLUtils::addChild(doc, node, "Dimension", "Smile");
208  XMLUtils::addChild(doc, node, "SmileType", "VannaVolga");
209  // only write smile interpolation if dimension is smile
210  if (smileInterpolation_ == SmileInterpolation::VannaVolga1) {
211  XMLUtils::addChild(doc, node, "SmileInterpolation", "VannaVolga1");
212  } else if (smileInterpolation_ == SmileInterpolation::VannaVolga2) {
213  XMLUtils::addChild(doc, node, "SmileInterpolation", "VannaVolga2");
214  } else {
215  QL_FAIL("Unknown SmileInterpolation in FXVolatilityCurveConfig::toXML()");
216  }
217  XMLUtils::addGenericChildAsList(doc, node, "SmileDelta", deltas_);
218  XMLUtils::addChild(doc, node, "Conventions", to_string(conventionsID_));
219  } else if (dimension_ == Dimension::SmileDelta) {
220  XMLUtils::addChild(doc, node, "Dimension", "Smile");
221  XMLUtils::addChild(doc, node, "SmileType", "Delta");
222  // only write smile interpolation if dimension is smile
223  if (smileInterpolation_ == SmileInterpolation::Linear) {
224  XMLUtils::addChild(doc, node, "SmileInterpolation", "Linear");
225  } else if (smileInterpolation_ == SmileInterpolation::Cubic) {
226  XMLUtils::addChild(doc, node, "SmileInterpolation", "Cubic");
227  } else {
228  QL_FAIL("Unknown SmileInterpolation in FXVolatilityCurveConfig::toXML()");
229  }
230  XMLUtils::addChild(doc, node, "Conventions", to_string(conventionsID_));
231  XMLUtils::addGenericChildAsList(doc, node, "Deltas", deltas_);
232  } else if (dimension_ == Dimension::SmileBFRR) {
233  XMLUtils::addChild(doc, node, "Dimension", "Smile");
234  XMLUtils::addChild(doc, node, "SmileType", "BFRR");
235  if (smileInterpolation_ == SmileInterpolation::Linear) {
236  XMLUtils::addChild(doc, node, "SmileInterpolation", "Linear");
237  } else if (smileInterpolation_ == SmileInterpolation::Cubic) {
238  XMLUtils::addChild(doc, node, "SmileInterpolation", "Cubic");
239  } else {
240  QL_FAIL("Unknown SmileInterpolation in FXVolatilityCurveConfig::toXML()");
241  }
242  XMLUtils::addGenericChildAsList(doc, node, "SmileDelta", smileDelta_);
243  XMLUtils::addChild(doc, node, "Conventions", to_string(conventionsID_));
244  } else {
245  QL_FAIL("Unkown Dimension in FXVolatilityCurveConfig::toXML()");
246  }
247  XMLUtils::addGenericChildAsList(doc, node, "Expiries", expiries_);
248  XMLUtils::addChild(doc, node, "FXSpotID", fxSpotID_);
249  if (!fxForeignYieldCurveID_.empty())
250  XMLUtils::addChild(doc, node, "FXForeignCurveID", fxForeignYieldCurveID_);
251  if (!fxDomesticYieldCurveID_.empty())
252  XMLUtils::addChild(doc, node, "FXDomesticCurveID", fxDomesticYieldCurveID_);
253  XMLUtils::addChild(doc, node, "Calendar", to_string(calendar_));
254  XMLUtils::addChild(doc, node, "DayCounter", to_string(dayCounter_));
255  XMLUtils::appendNode(node, reportConfig_.toXML(doc));
256 
257  return node;
258 }

260 void FXVolatilityCurveConfig::populateRequiredCurveIds() {
261  if (!fxDomesticYieldCurveID_.empty() && !fxForeignYieldCurveID_.empty()) {
262  std::vector<string> domTokens, forTokens;
263  split(domTokens, fxDomesticYieldCurveID_, boost::is_any_of("/"));
264  split(forTokens, fxForeignYieldCurveID_, boost::is_any_of("/"));
265 
266  if (domTokens.size() == 3 && domTokens[0] == "Yield") {
267  requiredCurveIds_[CurveSpec::CurveType::Yield].insert(domTokens[2]);
268  } else if (domTokens.size() == 1) {
269  requiredCurveIds_[CurveSpec::CurveType::Yield].insert(fxDomesticYieldCurveID_);
270  } else {
271  QL_FAIL("Cannot determine the required domestic yield curve for fx vol curve " << curveID_);
272  }
273 
274  if (forTokens.size() == 3 && forTokens[0] == "Yield") {
275  requiredCurveIds_[CurveSpec::CurveType::Yield].insert(forTokens[2]);
276  } else if (forTokens.size() == 1) {
277  requiredCurveIds_[CurveSpec::CurveType::Yield].insert(fxForeignYieldCurveID_);
278  } else {
279  QL_FAIL("Cannot determine the required foreign yield curve for fx vol curve " << curveID_);
280  }
281  }
282 
283  if (dimension_ == Dimension::ATMTriangulated) {
284  requiredCurveIds_[CurveSpec::CurveType::FXVolatility].insert(baseVolatility1_);
285  requiredCurveIds_[CurveSpec::CurveType::FXVolatility].insert(baseVolatility2_);
286 
287  vector<string> tokens;
288  boost::split(tokens, fxSpotID_, boost::is_any_of("/"));
289  QL_REQUIRE(tokens.size() == 3, "unexpected fxSpot format: " << fxSpotID_);
290  auto forTarget = tokens[1];
291  auto domTarget = tokens[2];
292 
293  // we need to include inverse ccy pairs as well
294  QL_REQUIRE(baseVolatility1_.size() == 6, "invalid ccy pair length");
295  QL_REQUIRE(baseVolatility2_.size() == 6, "invalid ccy pair length");
296  auto forBase1 = baseVolatility1_.substr(0, 3);
297  auto domBase1 = baseVolatility1_.substr(3);
298  auto forBase2 = baseVolatility2_.substr(0, 3);
299  auto domBase2 = baseVolatility2_.substr(3);
300 
301  requiredCurveIds_[CurveSpec::CurveType::FXVolatility].insert(domBase1 + forBase1);
302  requiredCurveIds_[CurveSpec::CurveType::FXVolatility].insert(domBase2 + forBase2);
303 
304  // we need to establish the common currency between the two pairs to include the correlations
305  std::string baseCcy = "";
306  if (forBase1 == forBase2 || forBase1 == domBase2) {
307  baseCcy = forBase1;
308  } else {
309  QL_REQUIRE(domBase1 == forBase2 || domBase1 == domBase2, "no common currency found for baseVolatilities");
310  baseCcy = domBase1;
311  }
312 
313  // straight pair
314  std::string forIndex = "FX-" + fxIndexTag_ + "-" + forTarget + "-" + baseCcy;
315  std::string domIndex = "FX-" + fxIndexTag_ + "-" + domTarget + "-" + baseCcy;
316  requiredCurveIds_[CurveSpec::CurveType::Correlation].insert(forIndex + "&" + domIndex);
317  // inverse pair
318  requiredCurveIds_[CurveSpec::CurveType::Correlation].insert(domIndex + "&" + forIndex);
319  // inverse fx index1
320  std::string forIndexInverse = "FX-" + fxIndexTag_ + "-" + baseCcy + "-" + forTarget;
321  requiredCurveIds_[CurveSpec::CurveType::Correlation].insert(forIndexInverse + "&" + domIndex);
322  requiredCurveIds_[CurveSpec::CurveType::Correlation].insert(domIndex + "&" + forIndexInverse);
323  // inverse fx index2
324  std::string domIndexInverse = "FX-" + fxIndexTag_ + "-" + baseCcy + "-" + domTarget;
325  requiredCurveIds_[CurveSpec::CurveType::Correlation].insert(forIndex + "&" + domIndexInverse);
326  requiredCurveIds_[CurveSpec::CurveType::Correlation].insert(domIndexInverse + "&" + forIndex);
327  // both fx indices inverted
328  requiredCurveIds_[CurveSpec::CurveType::Correlation].insert(forIndexInverse + "&" + domIndexInverse);
329  requiredCurveIds_[CurveSpec::CurveType::Correlation].insert(domIndexInverse + "&" + forIndexInverse);
330  }
