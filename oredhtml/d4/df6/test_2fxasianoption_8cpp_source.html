42  class TestMarket : public MarketImpl {
43  public:
44  TestMarket(const Real spot, const Date& expiry, const Rate domRate, const Rate forRate,
45  const Volatility flatVolatility) {
46  // Reference date and common day counter
47  asof_ = Date(01, Feb, 2021);
48  DayCounter dayCounter = Actual360();
49 
50  // Add USD/JPY discount curves
51  Handle<YieldTermStructure> domestic(boost::make_shared<FlatForward>(asof_, domRate, dayCounter));
52  Handle<YieldTermStructure> foreign(boost::make_shared<FlatForward>(asof_, forRate, dayCounter));
53  yieldCurves_[make_tuple(Market::defaultConfiguration, YieldCurveType::Discount, "USD")] = domestic;
54  yieldCurves_[make_tuple(Market::defaultConfiguration, YieldCurveType::Discount, "JPY")] = foreign;
55 
56  // Add fx spot
57  fxSpots_[Market::defaultConfiguration].addQuote("JPYUSD", Handle<Quote>(boost::make_shared<SimpleQuote>(spot)));
58 
59  // Add USDJPY volatilities
60  Handle<BlackVolTermStructure> volatility(
61  boost::make_shared<BlackConstantVol>(asof_, TARGET(), flatVolatility, dayCounter));
62  fxVols_[make_pair(Market::defaultConfiguration, "JPYUSD")] = volatility;
63  }
64  };

120  for (const auto& a : asians) {
121  Time deltaT = a.length / (a.fixings - 1);
122  Date expiry;
123  vector<Date> fixingDates(a.fixings);
124  vector<std::string> strFixingDates(a.fixings);
125  for (Size i = 0; i < a.fixings; ++i) {

141  {"RequiredSamples", "2047"}, {"Seed", "0"}};
142  engineFactory = boost::make_shared<EngineFactory>(engineData, market);
143

147
