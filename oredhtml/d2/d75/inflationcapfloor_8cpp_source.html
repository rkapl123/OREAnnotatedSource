107  // build market

109  Settings::instance().evaluationDate() = today;

111  boost::shared_ptr<TestMarket> market = boost::make_shared<TestMarket>(today);
112  // Test if EUR yoy inflation curve is empty
113  Handle<YoYInflationIndex> infidx = market->yoyInflationIndex("EUHICPXT");
114  QL_REQUIRE(!infidx.empty(), "EUHICPXT inflation index not found");
115 
116  // envelope
117  Envelope env("CP");
118 
119  // Start/End date
120  Date startDate = today;
121  Date endDate = today + 5 * Years;
122 
123  // date 2 string
124  std::ostringstream oss;
125  oss << io::iso_date(startDate);
126  string start(oss.str());
127  oss.str("");
128  oss.clear();
129  oss << io::iso_date(endDate);
130  string end(oss.str());
131 
132  // Schedules
133  string conv = "F";
134  string rule = "Forward";
135  ScheduleData scheduleYY(ScheduleRules(start, end, "1Y", "TARGET", conv, conv, rule));
136 
137  // Leg variables
138  string dc = "ACT/ACT";
139  vector<Real> notional(1, 10000000);
140  string paymentConvention = "F";
141 
142  // EUR YoY Leg
143  bool isPayerYY = false;
144  string indexYY = "EUHICPXT";
145  string lag = "3M";
146  LegData legYY(boost::make_shared<YoYLegData>(indexYY, lag, 0), isPayerYY, "EUR", scheduleYY, dc, notional,
147  vector<string>(), paymentConvention, false, true);
148 
149  std::vector<double> caps(1, 0.009);
150  // Build capfloor trades
151  boost::shared_ptr<Trade> yyCap(new ore::data::CapFloor(env, "Long", legYY, caps, std::vector<double>()));
152 
153  // engine data and factory
154  boost::shared_ptr<EngineData> engineData = boost::make_shared<EngineData>();
155  engineData->model("YYCapFloor") = "YYCapModel";
156  engineData->engine("YYCapFloor") = "YYCapEngine";
157  boost::shared_ptr<EngineFactory> engineFactory = boost::make_shared<EngineFactory>(engineData, market);
158  engineFactory->registerBuilder(boost::make_shared<YoYCapFloorEngineBuilder>());
159 
160  // build capfloor and portfolio
161  boost::shared_ptr<Portfolio> portfolio(new Portfolio());
162  yyCap->id() = "YoY_Cap";
163 
164  portfolio->add(yyCap);
165  portfolio->build(engineFactory);
166 
167  // check YoY cap NPV against pure QL pricing
168  Schedule schedule(startDate, endDate, 1 * Years, TARGET(), Following, Following, DateGeneration::Forward, false);
169  Handle<YieldTermStructure> nominalTs = market->discountCurve("EUR");
170  Leg yyLeg =
171  yoyInflationLeg(schedule, TARGET(), market->yoyInflationIndex("EUHICPXT").currentLink(), Period(3, Months))
172  .withNotionals(10000000)
173  .withPaymentDayCounter(ActualActual())
174  .withPaymentAdjustment(Following)
175  .withRateCurve(nominalTs);
176 
177  boost::shared_ptr<YoYInflationCapFloor> qlCap(new YoYInflationCap(yyLeg, caps));
178 
179  Handle<QuantLib::YoYOptionletVolatilitySurface> hovs = market->yoyCapFloorVol("EUHICPXT");
180  auto dscEngine = boost::make_shared<YoYInflationBachelierCapFloorEngine>(
181  market->yoyInflationIndex("EUHICPXT").currentLink(), hovs, nominalTs);
182  qlCap->setPricingEngine(dscEngine);
183  BOOST_CHECK_CLOSE(yyCap->instrument()->NPV(), qlCap->NPV(), 1E-8); // this is 1E-10 rel diff
184 }
