32 void FxForward::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
33  // If you Buy EURUSD forward, then you buy EUR and sell USD.
34  // EUR = foreign, USD = Domestic.
35  // You pay in USD, so the Domestic / Sold ccy is the "payer" currency
36  Currency boughtCcy = parseCurrency(boughtCurrency_);
37  Currency soldCcy = parseCurrency(soldCurrency_);
38  Date maturityDate = parseDate(maturityDate_);
39 
40  // Derive settlement date from payment data parameters
41  Date payDate;
42  if (payDate_.empty()) {
43  //LOG("Attempting paydate advance");
44  Period payLag = payLag_.empty() ? 0 * Days : parsePeriod(payLag_);
45  Calendar payCalendar = payCalendar_.empty() ? NullCalendar() : parseCalendar(payCalendar_);
46  BusinessDayConvention payConvention =
47  payConvention_.empty() ? Unadjusted : parseBusinessDayConvention(payConvention_);
48  payDate = payCalendar.advance(maturityDate, payLag, payConvention);
49  } else {
50  payDate = parseDate(payDate_);
51  QL_REQUIRE(payDate >= maturityDate, "FX Forward settlement date should equal or exceed the maturity date.");
52  }
53 
54  // Add required FX fixing for Cash settlement. If fixingDate == payDate, no fixing is actually required.
55  // Currently, fxIndexCalendar_ is "" (NullCalendar) and fxIndexDays_ is 0, so fixingDate = maturityDate always.
56  Date fixingDate;
57  Calendar fxIndexCalendar = fxIndexCalendar_.empty() ? NullCalendar() : parseCalendar(fxIndexCalendar_);
58  fixingDate = fxIndexCalendar.advance(maturityDate, -static_cast<Integer>(fxIndexDays_), Days);
59 
60  boost::shared_ptr<QuantExt::FxIndex> fxIndex;
61  Currency payCcy;
62 
63  if (payCurrency_.empty()) {
64  // If settlement currency is not set, set it to the domestic currency.
65  //LOG("Settlement currency was not specified, defaulting to " << soldCcy.code());
66  payCcy = soldCcy;
67  } else {
68  payCcy = parseCurrency(payCurrency_);
69  QL_REQUIRE(payCcy == boughtCcy || payCcy == soldCcy,
70  "Settlement currency must be either " << boughtCcy.code() << " or " << soldCcy.code() << ".");
71  }
72 
73  bool fixingRequired = (settlement_ == "Cash") && (payDate > fixingDate);
74  if (fixingRequired) {
75  QL_REQUIRE(!fxIndex_.empty(), "FX settlement index must be specified for non-deliverable forwards");
76  Currency nonPayCcy = payCcy == boughtCcy ? soldCcy : boughtCcy;
77  fxIndex = buildFxIndex(fxIndex_, payCcy.code(), nonPayCcy.code(), engineFactory->market(),
78  engineFactory->configuration(MarketContext::pricing), fxIndexCalendar_, fxIndexDays_);
79  requiredFixings_.addFixingDate(fixingDate, fxIndex_, payDate);
80  }
81 
82  QL_REQUIRE(tradeActions().empty(), "TradeActions not supported for FxForward");
83 
84  DLOG("Build FxForward with maturity date " << QuantLib::io::iso_date(maturityDate) << " and pay date "
85  << QuantLib::io::iso_date(payDate));
86 
87  boost::shared_ptr<QuantLib::Instrument> instrument =
88  boost::make_shared<QuantExt::FxForward>(boughtAmount_, boughtCcy, soldAmount_, soldCcy, maturityDate, false,
89  settlement_ == "Physical", payDate, payCcy, fixingDate, fxIndex);
90  instrument_.reset(new VanillaInstrument(instrument));
91 
92  npvCurrency_ = soldCurrency_;
93  notional_ = Null<Real>(); // soldAmount_;
94  notionalCurrency_ = ""; // soldCurrency_;
95  maturity_ = std::max(payDate, maturityDate);
96 
97  // Set up Legs
98  legs_ = {{boost::make_shared<SimpleCashFlow>(boughtAmount_, payDate)},
99  {boost::make_shared<SimpleCashFlow>(soldAmount_, payDate)}};
100  legCurrencies_ = {boughtCurrency_, soldCurrency_};
101  legPayers_ = {false, true};
102 
103  // set Pricing engine
104  boost::shared_ptr<EngineBuilder> builder = engineFactory->builder(tradeType_);
105  QL_REQUIRE(builder, "No builder found for " << tradeType_);
106  boost::shared_ptr<FxForwardEngineBuilderBase> fxBuilder =
107  boost::dynamic_pointer_cast<FxForwardEngineBuilderBase>(builder);
108 
109  instrument_->qlInstrument()->setPricingEngine(fxBuilder->engine(boughtCcy, soldCcy));
110 
111  DLOG("FxForward leg 0: " << legs_[0][0]->date() << " " << legs_[0][0]->amount());
112  DLOG("FxForward leg 1: " << legs_[1][0]->date() << " " << legs_[1][0]->amount());
113 
114  additionalData_["soldCurrency"] = soldCurrency_;
115  additionalData_["boughtCurrency"] = boughtCurrency_;
116  additionalData_["soldAmount"] = soldAmount_;
117  additionalData_["boughtAmount"] = boughtAmount_;
118 }

120 QuantLib::Real FxForward::notional() const {
121  // try to get the notional from the additional results of the instrument
122  try {
123  return instrument_->qlInstrument()->result<Real>("currentNotional");
124  } catch (const std::exception& e) {
125  if (strcmp(e.what(), "currentNotional not provided"))
126  ALOG("error when retrieving notional: " << e.what());
127  }
128  // if not provided, return null
129  return Null<Real>();
130 }

144 void FxForward::fromXML(XMLNode* node) {
145  Trade::fromXML(node);
146  XMLNode* fxNode = XMLUtils::getChildNode(node, "FxForwardData");
147  QL_REQUIRE(fxNode, "No FxForwardData Node");
148  maturityDate_ = XMLUtils::getChildValue(fxNode, "ValueDate", true);
149  boughtCurrency_ = XMLUtils::getChildValue(fxNode, "BoughtCurrency", true);
150  soldCurrency_ = XMLUtils::getChildValue(fxNode, "SoldCurrency", true);
151  boughtAmount_ = XMLUtils::getChildValueAsDouble(fxNode, "BoughtAmount", true);
152  soldAmount_ = XMLUtils::getChildValueAsDouble(fxNode, "SoldAmount", true);
153  settlement_ = XMLUtils::getChildValue(fxNode, "Settlement", false);
154  if (settlement_ == "")
155  settlement_ = "Physical";
156 
157  if (XMLNode* settlementDataNode = XMLUtils::getChildNode(fxNode, "SettlementData")) {
158  payCurrency_ = XMLUtils::getChildValue(settlementDataNode, "Currency", false);
159  fxIndex_ = XMLUtils::getChildValue(settlementDataNode, "FXIndex", false);
160  payDate_ = XMLUtils::getChildValue(settlementDataNode, "Date", false);
161 
162  if (payDate_.empty()) {
163  if (XMLNode* rulesNode = XMLUtils::getChildNode(settlementDataNode, "Rules")) {
164  payLag_ = XMLUtils::getChildValue(rulesNode, "PaymentLag", false);
165  payCalendar_ = XMLUtils::getChildValue(rulesNode, "PaymentCalendar", false);
166  payConvention_ = XMLUtils::getChildValue(rulesNode, "PaymentConvention", false);
167  }
168  }
169  }
170 }

172 XMLNode* FxForward::toXML(XMLDocument& doc) {
173  XMLNode* node = Trade::toXML(doc);
174  XMLNode* fxNode = doc.allocNode("FxForwardData");
175  XMLUtils::appendNode(node, fxNode);
176  XMLUtils::addChild(doc, fxNode, "ValueDate", maturityDate_);
177  XMLUtils::addChild(doc, fxNode, "BoughtCurrency", boughtCurrency_);
178  XMLUtils::addChild(doc, fxNode, "BoughtAmount", boughtAmount_);
179  XMLUtils::addChild(doc, fxNode, "SoldCurrency", soldCurrency_);
180  XMLUtils::addChild(doc, fxNode, "SoldAmount", soldAmount_);
181  XMLUtils::addChild(doc, fxNode, "Settlement", settlement_);
182 
183  XMLNode* settlementDataNode = doc.allocNode("SettlementData");
184  XMLUtils::appendNode(fxNode, settlementDataNode);
185 
186  if (!payCurrency_.empty())
187  XMLUtils::addChild(doc, settlementDataNode, "Currency", payCurrency_);
188  if (!fxIndex_.empty())
189  XMLUtils::addChild(doc, settlementDataNode, "FXIndex", fxIndex_);
190  if (!payDate_.empty()) {
191  XMLUtils::addChild(doc, settlementDataNode, "Date", payDate_);
192  } else {
193  XMLNode* rulesNode = doc.allocNode("Rules");
194  XMLUtils::appendNode(settlementDataNode, rulesNode);
195  if (!payLag_.empty())
196  XMLUtils::addChild(doc, rulesNode, "PaymentLag", payLag_);
197  if (!payCalendar_.empty())
198  XMLUtils::addChild(doc, rulesNode, "PaymentCalendar", payCalendar_);
199  if (!payConvention_.empty())
200  XMLUtils::addChild(doc, rulesNode, "PaymentConvention", payConvention_);
201  }
202 
203  return node;
204 }
