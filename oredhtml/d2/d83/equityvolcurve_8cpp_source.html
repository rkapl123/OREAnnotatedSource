54 EquityVolCurve::EquityVolCurve(Date asof, EquityVolatilityCurveSpec spec, const Loader& loader,
55  const CurveConfigurations& curveConfigs, const Handle<EquityIndex>& eqIndex,
56  const map<string, boost::shared_ptr<EquityCurve>>& requiredEquityCurves,
57  const map<string, boost::shared_ptr<EquityVolCurve>>& requiredEquityVolCurves) {
58 
59  try {
60  LOG("EquityVolCurve: start building equity volatility structure with ID " << spec.curveConfigID());
61 
62  auto config = *curveConfigs.equityVolCurveConfig(spec.curveConfigID());
63 
64  calendar_ = parseCalendar(config.calendar());
65  // if calendar is null use currency
66  if (calendar_ == NullCalendar())
67  calendar_ = parseCalendar(config.ccy());
68  dayCounter_ = parseDayCounter(config.dayCounter());
69 
70  if (config.isProxySurface()) {
71  buildVolatility(asof, spec, curveConfigs, requiredEquityCurves, requiredEquityVolCurves);
72  } else {
73  QL_REQUIRE(config.quoteType() == MarketDatum::QuoteType::PRICE ||
74  config.quoteType() == MarketDatum::QuoteType::RATE_LNVOL,
75  "EquityVolCurve: Only lognormal volatilities and option premiums supported for equity "
76  "volatility surfaces.");
77 
78  // Do different things depending on the type of volatility configured
79  boost::shared_ptr<VolatilityConfig> vc = config.volatilityConfig();
80  if (auto cvc = boost::dynamic_pointer_cast<ConstantVolatilityConfig>(vc)) {
81  buildVolatility(asof, config, *cvc, loader);
82  } else if (auto vcc = boost::dynamic_pointer_cast<VolatilityCurveConfig>(vc)) {
83  buildVolatility(asof, config, *vcc, loader);
84  } else if (auto vssc = boost::dynamic_pointer_cast<VolatilityStrikeSurfaceConfig>(vc)) {
85  buildVolatility(asof, config, *vssc, loader, eqIndex);
86  } else if (auto vmsc = boost::dynamic_pointer_cast<VolatilityMoneynessSurfaceConfig>(vc)) {
87  buildVolatility(asof, config, *vmsc, loader, eqIndex);
88  } else if (auto vdsc = boost::dynamic_pointer_cast<VolatilityDeltaSurfaceConfig>(vc)) {
89  buildVolatility(asof, config, *vdsc, loader, eqIndex);
90  } else {
91  QL_FAIL("Unexpected VolatilityConfig in EquityVolatilityConfig");
92  }
93  }
94  DLOG("EquityVolCurve: finished building equity volatility structure with ID " << spec.curveConfigID());
95 
96  buildCalibrationInfo(asof, curveConfigs, config, eqIndex);
97 
98  } catch (exception& e) {
99  QL_FAIL("Equity volatility curve building failed : " << e.what());
100  } catch (...) {
101  QL_FAIL("Equity volatility curve building failed: unknown error");
102  }

105 void EquityVolCurve::buildVolatility(const Date& asof, const EquityVolatilityCurveConfig& vc,
106  const ConstantVolatilityConfig& cvc, const Loader& loader) {
107  LOG("EquityVolCurve: start building constant volatility structure");
108 
109  QL_REQUIRE(cvc.quoteType() == MarketDatum::QuoteType::RATE_LNVOL ||
110  cvc.quoteType() == MarketDatum::QuoteType::RATE_SLNVOL ||
111  cvc.quoteType() == MarketDatum::QuoteType::RATE_NVOL,
112  "Quote for Equity Constant Volatility Config must be a Volatility");
113 
114  // Loop over all market datums and find the single quote
115  // Return error if there are duplicates (this is why we do not use loader.get() method)
116  Real quoteValue = Null<Real>();
117  for (const boost::shared_ptr<MarketDatum>& md : loader.loadQuotes(asof)) {
118  if (md->asofDate() == asof && md->instrumentType() == MarketDatum::InstrumentType::EQUITY_OPTION) {
119 
120  boost::shared_ptr<EquityOptionQuote> q = boost::dynamic_pointer_cast<EquityOptionQuote>(md);
121 
122  if (q->name() == cvc.quote()) {
123  TLOG("Found the constant volatility quote " << q->name());
124  QL_REQUIRE(quoteValue == Null<Real>(), "Duplicate quote found for quote with id " << cvc.quote());
125  // convert quote from minor to major currency if needed
126  quoteValue = convertMinorToMajorCurrency(q->ccy(), q->quote()->value());
127  }
128  }
129  }
130  QL_REQUIRE(quoteValue != Null<Real>(), "Quote not found for id " << cvc.quote());
131 
132  DLOG("Creating BlackConstantVol structure");
133  vol_ = boost::make_shared<BlackConstantVol>(asof, calendar_, quoteValue, dayCounter_);
134 
135  LOG("EquityVolCurve: finished building constant volatility structure");
136 }

138 void EquityVolCurve::buildVolatility(const Date& asof, const EquityVolatilityCurveConfig& vc,
139  const VolatilityCurveConfig& vcc, const Loader& loader) {
140 
141  LOG("EquityVolCurve: start building 1-D volatility curve");
142 
143  QL_REQUIRE(vcc.quoteType() == MarketDatum::QuoteType::RATE_LNVOL ||
144  vcc.quoteType() == MarketDatum::QuoteType::RATE_SLNVOL ||
145  vcc.quoteType() == MarketDatum::QuoteType::RATE_NVOL,
146  "Quote for Equity Constant Volatility Config must be a Volatility");
147 
148  // Must have at least one quote
149  QL_REQUIRE(vcc.quotes().size() > 0, "No quotes specified in config " << vc.curveID());
150 
151  // Check if we are using a regular expression to select the quotes for the curve. If we are, the quotes should
152  // contain exactly one element.
153  auto wildcard = getUniqueWildcard(vcc.quotes());
154 
155  // curveData will be populated with the expiry dates and volatility values.
156  map<Date, Real> curveData;
157 
158  // Different approaches depending on whether we are using a regex or searching for a list of explicit quotes.
159  if (wildcard) {
160  DLOG("Have single quote with pattern " << (*wildcard).regex());
161 
162  // Loop over quotes and process commodity option quotes matching pattern on asof
163  for (const boost::shared_ptr<MarketDatum>& md : loader.loadQuotes(asof)) {
164 
165  // Go to next quote if the market data point's date does not equal our asof
166  if (md->asofDate() != asof)
167  continue;
168 
169  auto q = boost::dynamic_pointer_cast<EquityOptionQuote>(md);
170  if (q && (*wildcard).matches(q->name()) && q->quoteType() == vc.quoteType()) {
171 
172  TLOG("The quote " << q->name() << " matched the pattern");
173 
174  Date expiryDate = getDateFromDateOrPeriod(q->expiry(), asof, calendar_);
175  if (expiryDate > asof) {
176  // Add the quote to the curve data
177  QL_REQUIRE(curveData.count(expiryDate) == 0, "Duplicate quote for the expiry date "
178  << io::iso_date(expiryDate)
179  << " provided by equity volatility config "
180  << vc.curveID());
181  // convert quote from minor to major currency if needed
182  curveData[expiryDate] = convertMinorToMajorCurrency(q->ccy(), q->quote()->value());
183 
184  TLOG("Added quote " << q->name() << ": (" << io::iso_date(expiryDate) << "," << fixed
185  << setprecision(9) << q->quote()->value() << ")");
186  }
187  }
188  }
189  // Check that we have quotes in the end
190  QL_REQUIRE(curveData.size() > 0, "No quotes found matching regular expression " << vcc.quotes()[0]);
191 
192  } else {
193 
194  DLOG("Have " << vcc.quotes().size() << " explicit quotes");
195 
196  // Loop over quotes and process commodity option quotes that are explicitly specified in the config
197  for (const boost::shared_ptr<MarketDatum>& md : loader.loadQuotes(asof)) {
198  // Go to next quote if the market data point's date does not equal our asof
199  if (md->asofDate() != asof)
200  continue;
201 
202  if (auto q = boost::dynamic_pointer_cast<EquityOptionQuote>(md)) {
203 
204  // Find quote name in configured quotes.
205  auto it = find(vcc.quotes().begin(), vcc.quotes().end(), q->name());
206 
207  if (it != vcc.quotes().end()) {
208  TLOG("Found the configured quote " << q->name());
209 
210  Date expiryDate = getDateFromDateOrPeriod(q->expiry(), asof, calendar_);
211  QL_REQUIRE(expiryDate > asof, "Equity volatility quote '" << q->name()
212  << "' has expiry in the past ("
213  << io::iso_date(expiryDate) << ")");
214  QL_REQUIRE(curveData.count(expiryDate) == 0, "Duplicate quote for the date "
215  << io::iso_date(expiryDate)
216  << " provided by equity volatility config "
217  << vc.curveID());
218 
219  // convert quote from minor to major currency if needed
220  curveData[expiryDate] = convertMinorToMajorCurrency(q->ccy(), q->quote()->value());
221 
222  TLOG("Added quote " << q->name() << ": (" << io::iso_date(expiryDate) << "," << fixed
223  << setprecision(9) << q->quote()->value() << ")");
224  }
225  }
226  }
227 
228  // Check that we have found all of the explicitly configured quotes
229  QL_REQUIRE(curveData.size() == vcc.quotes().size(), "Found " << curveData.size() << " quotes, but "
230  << vcc.quotes().size()
231  << " quotes were given in config.");
232  }
233 
234  // Create the dates and volatility vector
235  vector<Date> dates;
236  vector<Volatility> volatilities;
237  for (const auto& datum : curveData) {
238  dates.push_back(datum.first);
239  volatilities.push_back(datum.second);
240  TLOG("Added data point (" << io::iso_date(dates.back()) << "," << fixed << setprecision(9)
241  << volatilities.back() << ")");
242  }
243 
244  DLOG("Creating BlackVarianceCurve object.");
245  auto tmp = boost::make_shared<BlackVarianceCurve>(asof, dates, volatilities, dayCounter_);
246 
247  // set max expiry date (used in buildCalibrationInfo())
248  if (!dates.empty())
249  maxExpiry_ = dates.back();
250 
251  // Set the interpolation.
252  if (vcc.interpolation() == "Linear") {
253  DLOG("Interpolation set to Linear.");
254  } else if (vcc.interpolation() == "Cubic") {
255  DLOG("Setting interpolation to Cubic.");
256  tmp->setInterpolation<Cubic>();
257  } else if (vcc.interpolation() == "LogLinear") {
258  DLOG("Setting interpolation to LogLinear.");
259  tmp->setInterpolation<LogLinear>();
260  } else {
261  DLOG("Interpolation " << vcc.interpolation() << " not recognised so leaving it Linear.");
262  }
263 
264  // Set the volatility_ member after we have possibly updated the interpolation.
265  vol_ = tmp;
266 
267  // Set the extrapolation
268  if (parseExtrapolation(vcc.extrapolation()) == Extrapolation::Flat) {
269  DLOG("Enabling BlackVarianceCurve flat volatility extrapolation.");
270  vol_->enableExtrapolation();
271  } else if (parseExtrapolation(vcc.extrapolation()) == Extrapolation::None) {
272  DLOG("Disabling BlackVarianceCurve extrapolation.");
273  vol_->disableExtrapolation();
274  } else if (parseExtrapolation(vcc.extrapolation()) == Extrapolation::UseInterpolator) {
275  DLOG("BlackVarianceCurve does not support using interpolator for extrapolation "
276  << "so default to flat volatility extrapolation.");
277  vol_->enableExtrapolation();
278  } else {
279  DLOG("Unexpected extrapolation so default to flat volatility extrapolation.");
280  vol_->enableExtrapolation();
281  }
282 
283  LOG("EquityVolCurve: finished building 1-D volatility curve");
284 }

286 void EquityVolCurve::buildVolatility(const Date& asof, EquityVolatilityCurveConfig& vc,
287  const VolatilityStrikeSurfaceConfig& vssc, const Loader& loader,
288  const QuantLib::Handle<EquityIndex>& eqIndex) {
289  try {
290 
291  QL_REQUIRE(vssc.expiries().size() > 0, "No expiries defined");
292  QL_REQUIRE(vssc.strikes().size() > 0, "No strikes defined");
293 
294  // check for wild cards
295  bool expiriesWc = find(vssc.expiries().begin(), vssc.expiries().end(), "*") != vssc.expiries().end();
296  bool strikesWc = find(vssc.strikes().begin(), vssc.strikes().end(), "*") != vssc.strikes().end();
297  if (expiriesWc) {
298  QL_REQUIRE(vssc.expiries().size() == 1, "Wild card expiriy specified but more expiries also specified.");
299  }
300  if (strikesWc) {
301  QL_REQUIRE(vssc.strikes().size() == 1, "Wild card strike specified but more strikes also specified.");
302  }
303  bool wildcard = strikesWc || expiriesWc;
304 
305  vector<Real> callStrikes, putStrikes;
306  vector<Real> callData, putData;
307  vector<Date> callExpiries, putExpiries;
308 
309  // In case of wild card we need the following granularity within the mkt data loop
310  bool strikeRelevant = strikesWc;
311  bool expiryRelevant = expiriesWc;
312  bool quoteRelevant = true;
313 
314  // We loop over all market data, looking for quotes that match the configuration
315  Size callQuotesAdded = 0;
316  Size putQuotesAdded = 0;
317  for (auto& md : loader.loadQuotes(asof)) {
318  // skip irrelevant data
319  if (md->asofDate() == asof && md->instrumentType() == MarketDatum::InstrumentType::EQUITY_OPTION &&
320  md->quoteType() == vc.quoteType()) {
321  boost::shared_ptr<EquityOptionQuote> q = boost::dynamic_pointer_cast<EquityOptionQuote>(md);
322  // todo - for now we will ignore ATM, ATMF quotes both for explicit strikes and in case of strike wild
323  // card. ----
324  auto absoluteStrike = boost::dynamic_pointer_cast<AbsoluteStrike>(q->strike());
325  if (absoluteStrike && (q->eqName() == vc.curveID() && q->ccy() == vc.ccy())) {
326  if (!expiriesWc) {
327  auto j = std::find(vssc.expiries().begin(), vssc.expiries().end(), q->expiry());
328  expiryRelevant = j != vssc.expiries().end();
329  }
330  if (!strikesWc) {
331  auto i = std::find_if(vssc.strikes().begin(), vssc.strikes().end(),
332  [&absoluteStrike](const std::string& x) {
333  return close_enough(parseReal(x), absoluteStrike->strike());
334  });
335  strikeRelevant = i != vssc.strikes().end();
336  }
337  quoteRelevant = strikeRelevant && expiryRelevant;
338 
339  // add quote to vectors, if relevant
340  // If a quote doesn't include a call/put flag (an Implied Vol for example), it
341  // defaults to a call. For an explicit surface we expect either a call and put
342  // for every point, or just a vol at every point
343  if (quoteRelevant) {
344  Date tmpDate = getDateFromDateOrPeriod(q->expiry(), asof, calendar_);
345  QL_REQUIRE(tmpDate >= asof,
346  "Option quote for a past date (" << ore::data::to_string(tmpDate) << ")");
347  if (tmpDate == asof) {
348  DLOG("Option quote for as of date (" << ore::data::to_string(tmpDate) << ") ignored.");
349  continue;
350  }
351  // get values and strikes, convert from minor to major currency if needed
352  Real quoteValue = q->quote()->value();
353  if (vc.quoteType() == MarketDatum::QuoteType::PRICE)
354  quoteValue = convertMinorToMajorCurrency(q->ccy(), quoteValue);
355  Real strikeValue = convertMinorToMajorCurrency(q->ccy(), absoluteStrike->strike());
356 
357  if (q->isCall()) {
358  callStrikes.push_back(strikeValue);
359  callData.push_back(quoteValue);
360  callExpiries.push_back(tmpDate);
361  callQuotesAdded++;
362  } else {
363  putStrikes.push_back(strikeValue);
364  putData.push_back(quoteValue);
365  putExpiries.push_back(tmpDate);
366  putQuotesAdded++;
367  }
368  }
369  }
370  }
371  }
372 
373  QL_REQUIRE(callQuotesAdded > 0, "No valid equity volatility quotes provided");
374  bool callSurfaceOnly = false;
375  if (callQuotesAdded > 0 && putQuotesAdded == 0) {
376  QL_REQUIRE(vc.quoteType() != MarketDatum::QuoteType::PRICE,
377  "For Premium quotes, call and put quotes must be supplied.");
378  DLOG("EquityVolatilityCurve " << vc.curveID() << ": Only one set of quotes, can build surface directly");
379  callSurfaceOnly = true;
380  }
381  // Check loaded quotes
382  if (!wildcard) {
383  Size explicitGridSize = vssc.expiries().size() * vssc.strikes().size();
384  QL_REQUIRE(callQuotesAdded == explicitGridSize,
385  "EquityVolatilityCurve " << vc.curveID() << ": " << callQuotesAdded << " quotes provided but "
386  << explicitGridSize << " expected.");
387  if (!callSurfaceOnly) {
388  QL_REQUIRE(callQuotesAdded == putQuotesAdded,
389  "Call and Put quotes must match for explicitly defined surface, "
390  << callQuotesAdded << " call quotes, and " << putQuotesAdded << " put quotes");
391  DLOG("EquityVolatilityCurve " << vc.curveID() << ": Complete set of " << callQuotesAdded
392  << ", call and put quotes found.");
393  }
394  }
395 
396  QL_REQUIRE(callStrikes.size() == callData.size() && callData.size() == callExpiries.size(),
397  "Quotes loaded don't produce strike,vol,expiry vectors of equal length.");
398  QL_REQUIRE(putStrikes.size() == putData.size() && putData.size() == putExpiries.size(),
399  "Quotes loaded don't produce strike,vol,expiry vectors of equal length.");
400  DLOG("EquityVolatilityCurve " << vc.curveID() << ": Found " << callQuotesAdded << ", call quotes and "
401  << putQuotesAdded << " put quotes using wildcard.");
402 
403  // Set the strike extrapolation which only matters if extrapolation is turned on for the whole surface.
404  bool flatStrikeExtrap = true;
405  bool flatTimeExtrap = true;
406  if (vssc.extrapolation()) {
407 
408  auto strikeExtrapType = parseExtrapolation(vssc.strikeExtrapolation());
409  if (strikeExtrapType == Extrapolation::UseInterpolator) {
410  DLOG("Strike extrapolation switched to using interpolator.");
411  flatStrikeExtrap = false;
412  } else if (strikeExtrapType == Extrapolation::None) {
413  DLOG("Strike extrapolation cannot be turned off on its own so defaulting to flat.");
414  } else if (strikeExtrapType == Extrapolation::Flat) {
415  DLOG("Strike extrapolation has been set to flat.");
416  } else {
417  DLOG("Strike extrapolation " << strikeExtrapType << " not expected so default to flat.");
418  }
419 
420  auto timeExtrapType = parseExtrapolation(vssc.timeExtrapolation());
421  if (timeExtrapType == Extrapolation::UseInterpolator) {
422  DLOG("Time extrapolation switched to using interpolator.");
423  flatTimeExtrap = false;
424  } else if (timeExtrapType == Extrapolation::None) {
425  DLOG("Time extrapolation cannot be turned off on its own so defaulting to flat.");
426  } else if (timeExtrapType == Extrapolation::Flat) {
427  DLOG("Time extrapolation has been set to flat.");
428  } else {
429  DLOG("Time extrapolation " << timeExtrapType << " not expected so default to flat.");
430  }
431 
432  } else {
433  DLOG("Extrapolation is turned off for the whole surface so the time and"
434  << " strike extrapolation settings are ignored");
435  }
436 
437  // set max expiry date (used in buildCalibrationInfo())
438  maxExpiry_ = Date::minDate();
439  for (auto const& d : callExpiries)
440  maxExpiry_ = std::max(maxExpiry_, d);
441  for (auto const& d : putExpiries)
442  maxExpiry_ = std::max(maxExpiry_, d);
443  if (maxExpiry_ == Date::minDate())
444  maxExpiry_ = Date();
445 
446  bool preferOutOfTheMoney = vc.preferOutOfTheMoney() && *vc.preferOutOfTheMoney();
447 
448  if (vc.quoteType() == MarketDatum::QuoteType::PRICE) {
449 
450  // Create the 1D solver options used in the price stripping.
451  Solver1DOptions solverOptions = vc.solverConfig();
452 
453  DLOG("Building a option price surface for calls and puts");
454  boost::shared_ptr<OptionPriceSurface> callSurface =
455  boost::make_shared<OptionPriceSurface>(asof, callExpiries, callStrikes, callData, dayCounter_);
456  boost::shared_ptr<OptionPriceSurface> putSurface =
457  boost::make_shared<OptionPriceSurface>(asof, putExpiries, putStrikes, putData, dayCounter_);
458 
459  DLOG("CallSurface contains " << callSurface->expiries().size() << " expiries.");
460 
461  DLOG("Stripping equity volatility surface from the option premium surfaces");
462  boost::shared_ptr<EquityOptionSurfaceStripper> eoss = boost::make_shared<EquityOptionSurfaceStripper>(
463  eqIndex, callSurface, putSurface, calendar_, dayCounter_, vc.exerciseType(), flatStrikeExtrap,
464  flatStrikeExtrap, flatTimeExtrap, preferOutOfTheMoney, solverOptions);
465  vol_ = eoss->volSurface();
466 
467  } else if (vc.quoteType() == MarketDatum::QuoteType::RATE_LNVOL) {
468 
469  if (callExpiries.size() == 1 && callStrikes.size() == 1) {
470  LOG("EquityVolCurve: Building BlackConstantVol");
471  vol_ = boost::shared_ptr<BlackVolTermStructure>(
472  new BlackConstantVol(asof, Calendar(), callData[0], dayCounter_));
473  } else {
474  // create a vol surface from the calls
475  boost::shared_ptr<BlackVarianceSurfaceSparse> callSurface =
476  boost::make_shared<BlackVarianceSurfaceSparse>(asof, calendar_, callExpiries, callStrikes, callData,
477  dayCounter_, flatStrikeExtrap, flatStrikeExtrap,
478  flatTimeExtrap);
479 
480  if (callSurfaceOnly) {
481  // if only a call surface provided use that
482  vol_ = callSurface;
483  } else {
484  // otherwise create a vol surface from puts and strip for a final surface
485  boost::shared_ptr<BlackVarianceSurfaceSparse> putSurface =
486  boost::make_shared<BlackVarianceSurfaceSparse>(asof, calendar_, putExpiries, putStrikes,
487  putData, dayCounter_, flatStrikeExtrap,
488  flatStrikeExtrap, flatTimeExtrap);
489 
490  boost::shared_ptr<EquityOptionSurfaceStripper> eoss =
491  boost::make_shared<EquityOptionSurfaceStripper>(
492  eqIndex, callSurface, putSurface, calendar_, dayCounter_, Exercise::European,
493  flatStrikeExtrap, flatStrikeExtrap, flatTimeExtrap, preferOutOfTheMoney);
494  vol_ = eoss->volSurface();
495  }
496  }
497  } else {
498  QL_FAIL("EquityVolatility: Invalid quote type provided.");
499  }
500  DLOG("Setting BlackVarianceSurfaceSparse extrapolation to " << to_string(vssc.extrapolation()));
501  vol_->enableExtrapolation(vssc.extrapolation());
502 
503  } catch (std::exception& e) {
504  QL_FAIL("equity vol curve building failed :" << e.what());
505  } catch (...) {
506  QL_FAIL("equity vol curve building failed: unknown error");
507  }

511 vector<Real> checkMoneyness(const vector<string>& strMoneynessLevels) {
512 
513  using boost::adaptors::transformed;
514  using boost::algorithm::join;
515 
516  vector<Real> moneynessLevels = parseVectorOfValues<Real>(strMoneynessLevels, &parseReal);
