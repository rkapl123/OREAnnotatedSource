56 class TestMarket : public MarketImpl {
57 public:
58  TestMarket(Real hazardRate, Real recoveryRate, Real liborRate) {
59  asof_ = Date(3, Feb, 2016);
60  // build discount
61  yieldCurves_[make_tuple(Market::defaultConfiguration, YieldCurveType::Discount, "EUR")] =
62  flatRateYts(liborRate);
63  defaultCurves_[make_pair(Market::defaultConfiguration, "CreditCurve_A")] = flatRateDcs(hazardRate);
64  recoveryRates_[make_pair(Market::defaultConfiguration, "CreditCurve_A")] =
65  Handle<Quote>(boost::make_shared<SimpleQuote>(recoveryRate));
66  // build ibor index
67  Handle<IborIndex> hEUR(ore::data::parseIborIndex(
68  "EUR-EURIBOR-6M", yieldCurves_[make_tuple(Market::defaultConfiguration, YieldCurveType::Discount, "EUR")]));
69  iborIndices_[make_pair(Market::defaultConfiguration, "EUR-EURIBOR-6M")] = hEUR;
70  }

72 private:
73  Handle<YieldTermStructure> flatRateYts(Real forward) {
74  boost::shared_ptr<YieldTermStructure> yts(new FlatForward(0, NullCalendar(), forward, ActualActual()));
75  yts->enableExtrapolation();
76  return Handle<YieldTermStructure>(yts);
77  }

79  boost::shared_ptr<DefaultProbabilityTermStructure> dcs(new FlatHazardRate(asof_, forward, SimpleDayCounter()));
80  return Handle<DefaultProbabilityTermStructure>(dcs);
81  }

107  ScheduleData fixedSchedule(ScheduleRules(start, end, fixtenor, calStr, conv, conv, rule));

110  boost::shared_ptr<FixedLegData> fixedLegRateData = boost::make_shared<FixedLegData>(vector<double>(1, rate));

118  return cds;

156 Real cdsNpv(const MarketInputs& m, const TradeInputs& t) {
157 
158  // build market
159  boost::shared_ptr<Market> market = boost::make_shared<TestMarket>(m.hazardRate, m.recoveryRate, m.liborRate);
160  Settings::instance().evaluationDate() = market->asofDate();
161 
162  CommonVars vars;
163  boost::shared_ptr<ore::data::CreditDefaultSwap> cds = vars.makeCDS(t.endDate, t.fixedRate);
164 
165  // Build and price
166  boost::shared_ptr<EngineData> engineData = boost::make_shared<EngineData>();
167  engineData->model("CreditDefaultSwap") = "DiscountedCashflows";
168  engineData->engine("CreditDefaultSwap") = "MidPointCdsEngine";
169 
170  boost::shared_ptr<EngineFactory> engineFactory = boost::make_shared<EngineFactory>(engineData, market);
171 
172  cds->build(engineFactory);
173 
174  return cds->instrument()->NPV();
175 }
