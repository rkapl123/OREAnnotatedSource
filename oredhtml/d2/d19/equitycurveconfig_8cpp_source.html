49 void EquityCurveConfig::fromXML(XMLNode* node) {
50  XMLUtils::checkNode(node, "EquityCurve");
51 
52  curveID_ = XMLUtils::getChildValue(node, "CurveId", true);
53  curveDescription_ = XMLUtils::getChildValue(node, "CurveDescription", true);
54  forecastingCurve_ = XMLUtils::getChildValue(node, "ForecastingCurve", true);
55  currency_ = XMLUtils::getChildValue(node, "Currency", true);
56  calendar_ = XMLUtils::getChildValue(node, "Calendar", false);
57  type_ = parseEquityCurveConfigType(XMLUtils::getChildValue(node, "Type", true));
58  if (type_ == EquityCurveConfig::Type::OptionPremium)
59  exerciseStyle_ = parseExerciseType(XMLUtils::getChildValue(node, "ExerciseStyle", true));
60  equitySpotQuoteID_ = XMLUtils::getChildValue(node, "SpotQuote", true);
61  dayCountID_ = XMLUtils::getChildValue(node, "DayCounter", false);
62  fwdQuotes_ = XMLUtils::getChildrenValues(node, "Quotes", "Quote");
63  quotes_ = fwdQuotes_;
64  if (equitySpotQuoteID_ != "")
65  quotes_.insert(quotes_.begin(), equitySpotQuoteID_);
66 
67  XMLNode* divInterpNode = XMLUtils::getChildNode(node, "DividendInterpolation");
68  if (divInterpNode) {
69  divInterpVariable_ = XMLUtils::getChildValue(divInterpNode, "InterpolationVariable", true);
70  divInterpMethod_ = XMLUtils::getChildValue(divInterpNode, "InterpolationMethod", true);
71  } else {
72  divInterpVariable_ = "Zero";
73  divInterpMethod_ = divInterpVariable_ == "Zero" ? "Linear" : "LogLinear";
74  }
75 
76  extrapolation_ = XMLUtils::getChildValueAsBool(node, "Extrapolation"); // defaults to true
77 
78  if (type_ == Type::NoDividends) {
79  QL_REQUIRE(fwdQuotes_.size() == 0,
80  "Invalid EquityCurveConfig, no Quotes should be present when type=NoDividends");
81  QL_REQUIRE(divInterpNode == nullptr,
82  "Invalid EquityCurveConfig, no DividendInterpolation should be present when type=NoDividends");
83  } else {
84  QL_REQUIRE(fwdQuotes_.size() > 0, "Invalid EquityCurveConfig, Quotes should be present when type!=NoDividends");
85  }
86  populateRequiredCurveIds();
87 }

89 XMLNode* EquityCurveConfig::toXML(XMLDocument& doc) {
90  XMLNode* node = doc.allocNode("EquityCurve");
91  XMLUtils::addChild(doc, node, "CurveId", curveID_);
92  XMLUtils::addChild(doc, node, "CurveDescription", curveDescription_);
93  XMLUtils::addChild(doc, node, "Currency", currency_);
94  XMLUtils::addChild(doc, node, "Calendar", calendar_);
95  XMLUtils::addChild(doc, node, "ForecastingCurve", forecastingCurve_);
96  XMLUtils::addChild(doc, node, "Type", to_string(type_));
97  if (type_ == EquityCurveConfig::Type::OptionPremium)
98  XMLUtils::addChild(doc, node, "ExerciseStyle", to_string(exerciseStyle_));
99  XMLUtils::addChild(doc, node, "SpotQuote", equitySpotQuoteID_);
100  XMLUtils::addChildren(doc, node, "Quotes", "Quote", fwdQuotes_);
101  XMLUtils::addChild(doc, node, "DayCounter", dayCountID_);
102 
103  if (type_ != Type::NoDividends) {
104  XMLNode* divInterpNode = XMLUtils::addChild(doc, node, "DividendInterpolation");
105  XMLUtils::addChild(doc, divInterpNode, "InterpolationVariable", divInterpVariable_);
106  XMLUtils::addChild(doc, divInterpNode, "InterpolationMethod", divInterpMethod_);
107  }
108  XMLUtils::addChild(doc, node, "Extrapolation", extrapolation_);
109 
110  return node;
111 }

130 EquityCurveConfig::Type parseEquityCurveConfigType(const std::string& str) {
131  if (str == "DividendYield")
132  return EquityCurveConfig::Type::DividendYield;
133  else if (str == "ForwardPrice")
134  return EquityCurveConfig::Type::ForwardPrice;
135  else if (str == "ForwardDividendPrice")
136  return EquityCurveConfig::Type::ForwardDividendPrice;
137  else if (str == "OptionPremium")
138  return EquityCurveConfig::Type::OptionPremium;
139  else if (str == "NoDividends")
140  return EquityCurveConfig::Type::NoDividends;
141  QL_FAIL("Invalid EquityCurveConfig::Type " << str);
142 }
