85string SimmBucketMapperBase::bucket(const RiskType& riskType, const string& qualifier) const {
86
87 QL_REQUIRE(hasBuckets(riskType), "The risk type " << riskType << " does not have buckets");
88
89 // Vol risk type bucket mappings are stored in their non-vol counterparts
90 auto lookupRiskType = riskType;
91 auto nv = nonVolRiskTypeMap.find(riskType);
92 if (nv != nonVolRiskTypeMap.end()) {
93 lookupRiskType = nv->second;
94 }
95
96 // Deal with RiskType::IRCurve
97 if (lookupRiskType == RiskType::IRCurve) {
98 return irBucket(qualifier);
99 }
100
101 string bucket;
102 string lookupName = qualifier;
103
104 bool haveMapping = has(lookupRiskType, lookupName, false);
105 bool haveFallback = has(lookupRiskType, lookupName, true);
106 bool noBucket = !haveMapping && !haveFallback;
107
108 if (noBucket && nameMapper_ != nullptr && lookupRiskType == RiskType::Equity) {
109 // if we have a simm name mapping we do a reverse lookup on the name as the CRIF qualifier isn't in reference data
110 lookupName = nameMapper_->externalName(qualifier);
111 haveMapping = has(lookupRiskType, lookupName, false);
112 haveFallback = has(lookupRiskType, lookupName, true);
113 noBucket = !haveMapping && ! haveFallback;
114 }
115
116 // Do some checks and return the lookup
117 if (noBucket) {
118 if (lookupRiskType == RiskType::Commodity) {
119 bool commIndex = false;
120 // first check is there is an entry under equity reference, this may tell us if it is an index
121 if (refDataManager_ != nullptr && refDataManager_->hasData("Equity", lookupName)) {
122 auto refData = boost::dynamic_pointer_cast<ore::data::EquityReferenceDatum>(
123 refDataManager_->getData("Equity", lookupName));
124 if (refData->equityData().isIndex)
125 commIndex = true;
126 }
127
128
129 // Commodity does not allow "Residual", but bucket 16 is "Other", bucket 17 for Indices
130 // FAQ Methodology and implementation 20180124 (H3) says to use "Other"
131 bucket = commIndex ? "17" : "16";
132 TLOG("Don't have any bucket mappings for the "
133 << "combination of risk type " << riskType << " and qualifier " << lookupName
134 << " - assigning to bucket " << bucket);
135
136 } else if (lookupRiskType == RiskType::Equity) {
137 // check if it is an index
138 if (refDataManager_ != nullptr && refDataManager_->hasData("Equity", lookupName)) {
139 auto refData = boost::dynamic_pointer_cast<ore::data::EquityReferenceDatum>(refDataManager_->getData("Equity", lookupName));
140 // if the equity is an index we assign to bucket 11
141 if (refData->equityData().isIndex) {
142 TLOG("Don't have any bucket mappings for the "
143 << "combination of risk type " << riskType << " and qualifier " << lookupName
144 << " - assigning to bucket 11");
145 bucket = "11";
146 }
147 }
148 }
149
150 if (bucket.empty()) {
151 TLOG("Don't have any bucket mappings for the "
152 << "combination of risk type " << riskType << " and qualifier " << lookupName
153 << " - assigning to Residual/Other bucket");
154 bucket = "Residual";
155 }
156
157 FailedMapping fm;
158 fm.name = qualifier;
159 fm.lookupName = lookupName;
160 fm.riskType = riskType;
161 fm.lookupRiskType = lookupRiskType;
162 failedMappings_.insert(fm);
163
164 } else {
165 // We may have several mappings per qualifier, pick the first valid one that matches the fallback flag
166 Date today = Settings::instance().evaluationDate();
167 for (auto m : bucketMapping_.at(lookupRiskType).at(lookupName)) {
168 if (m.validToDate() >= today && m.validFromDate() <= today && m.fallback() == !haveMapping) {
169 bucket = m.bucket();
170 return bucket;
171 }
172 }
173 TLOG("bucket mapping for risk type " << riskType << " and qualifier " << qualifier << " inactive, return Residual");
174 bucket = "Residual";
175 }
176
177 return bucket;
178}

182bool SimmBucketMapperBase::has(const RiskType& riskType, const string& qualifier,
183 boost::optional<bool> fallback) const {
184
185 // Vol risk type bucket mappings are stored in their non-vol counterparts
186 auto lookupRiskType = riskType;
187 auto nv = nonVolRiskTypeMap.find(riskType);
188 if (nv != nonVolRiskTypeMap.end()) {
189 lookupRiskType = nv->second;
190 }
191
192 if (lookupRiskType == RiskType::IRCurve)
193 return true;
194
195 auto bm = bucketMapping_.find(lookupRiskType);
196 if (bm == bucketMapping_.end())
197 return false;
198
199 auto q = bm->second.find(qualifier);
200 if (q == bm->second.end())
201 return false;
202
203 Date today = Settings::instance().evaluationDate();
204
205 // We may have several mappings (several periods, override and fallback mappings)
206 // So pick the first valid one
207
208 for (const auto& m : q->second) {
209 Date validTo = m.validToDate();
210 Date validFrom = m.validFromDate();
211 if (validTo >= today && validFrom <= today && (fallback ? *fallback == m.fallback() : true))
212 return true;
213 }
214 return false;
215}

217void SimmBucketMapperBase::fromXML(XMLNode* node) {
218 XMLUtils::checkNode(node, "SIMMBucketMappings");
219
220 // Every time a call to fromXML is made, reset the bucket mapper to its initial state
221 reset();
222
223 LOG("Start parsing SIMMBucketMappings");
224
225 for (XMLNode* rtNode = XMLUtils::getChildNode(node); rtNode; rtNode = XMLUtils::getNextSibling(rtNode)) {
226 // Get the risk type that we are dealing with
227 string strRiskType = XMLUtils::getNodeName(rtNode);
228 RiskType riskType = parseSimmRiskType(strRiskType);
229
230 checkRiskType(riskType);
231
232 QL_REQUIRE(bucketMapping_.count(riskType) == 0,
233 "Can only have one node for each risk type. " << strRiskType << " appears more than once.");
234 bucketMapping_[riskType] = {};
235
236 // Loop over and add the bucket mappings for this risk type
237 for (XMLNode* mappingNode = XMLUtils::getChildNode(rtNode, "Mapping"); mappingNode;
238 mappingNode = XMLUtils::getNextSibling(mappingNode, "Mapping")) {
239
240 string validTo = XMLUtils::getChildValue(mappingNode, "ValidTo", false);
241 string validFrom = XMLUtils::getChildValue(mappingNode, "ValidFrom", false);
242 string qualifier = XMLUtils::getChildValue(mappingNode, "Qualifier", false);
243 string bucket = XMLUtils::getChildValue(mappingNode, "Bucket", false);
244 if (bucket == "" || qualifier == "") {
245 ALOG("skip bucket mapping for quaifier '" << qualifier << "' and bucket '" << bucket << "'");
246 continue;
247 }
248 string fallbackString = XMLUtils::getChildValue(mappingNode, "Fallback", false);
249 bool fallback = false;
250 if (fallbackString != "")
251 fallback = ore::data::parseBool(fallbackString);
252
253 if (validTo != "") {
254 // Check whether the provided validTo element is a valid date, ALERT and ignore if it is not
255 try {
256 ore::data::parseDate(validTo);
257 } catch(std::exception&) {
258 ALOG("Cannot parse bucket mapping validTo " << validTo << " for qualifier " << qualifier << ", ignore");
259 validTo = "";
260 }
261 }
262 if (validFrom != "") {
263 // Check whether the provided validFrom element is a valid date, ALERT and ignore if it is not
264 try {
265 ore::data::parseDate(validFrom);
266 } catch(std::exception&) {
267 ALOG("Cannot parse bucket mapping validFrom " << validFrom << " for qualifier " << qualifier << ", ignore");
268 validFrom = "";
269 }
270 }
271
272 // Add mapping
273 BucketMapping mapping(bucket, validFrom, validTo, fallback);
274 if (bucketMapping_[riskType].find(qualifier) == bucketMapping_[riskType].end())
275 bucketMapping_[riskType][qualifier] = std::set<BucketMapping>();
276 bucketMapping_[riskType][qualifier].insert(mapping);
277 TLOG("Added SIMM bucket mapping: {" << riskType << ": {" << qualifier << ", " << bucket << ", " << validFrom << ", " << validTo << ", " << fallback << "}}");
278 if (bucket != "Residual") {
279 int bucketInt = ore::data::parseInteger(bucket);
280 QL_REQUIRE(bucketInt >= 1, "found bucket " << bucket << ", expected >= 1");
281 }
282 }
283 }
284
285 LOG("Finished parsing SIMMBucketMappings");
286}

288XMLNode* SimmBucketMapperBase::toXML(ore::data::XMLDocument& doc) {
289 XMLNode* node = doc.allocNode("SIMMBucketMappings");
290 for (auto const& b : bucketMapping_) {
291 XMLNode* n = doc.allocNode(ore::data::to_string(b.first));
292 XMLUtils::appendNode(node, n);
293 for (auto const& bms : b.second) {
294 for (auto const& bm : bms.second) {
295 XMLNode* m = doc.allocNode("Mapping");
296 XMLUtils::appendNode(n, m);
297 if (!bms.first.empty())
298 XMLUtils::addChild(doc, m, "Qualifier", bms.first);
299 if (!bm.validTo().empty())
300 XMLUtils::addChild(doc, m, "ValidTo", bm.validTo());
301 if (!bm.validFrom().empty())
302 XMLUtils::addChild(doc, m, "ValidFrom", bm.validFrom());
303 if (!bm.bucket().empty())
304 XMLUtils::addChild(doc, m, "Bucket", bm.bucket());
305 if (bm.fallback())
306 XMLUtils::addChild(doc, m, "Fallback", bm.fallback());
307 }
308 }
309 }
310 return node;
311}

313void SimmBucketMapperBase::addMapping(const RiskType& riskType, const string& qualifier, const string& bucket,
314 const string& validFrom, const string& validTo, bool fallback) {
315
316 // Possibly map to non-vol counterpart for lookup
317 RiskType rt = riskType;
318 if (nonVolRiskTypeMap.count(riskType) > 0) {
319 rt = nonVolRiskTypeMap.at(riskType);
320 }
321
322 if (rt == RiskType::IRCurve) {
323 // IR has internal mapping so return early - no need for warning
324 // Could catch commodity here but allow extra mappings there
325 return;
326 }
327
328 QL_REQUIRE(hasBuckets(riskType),
329 "Tried to add a bucket mapping for risk type " << riskType << " but it does not have buckets.");
330
331 if (bucketMapping_[rt].find(qualifier) == bucketMapping_[rt].end())
332 bucketMapping_[rt][qualifier] = std::set<BucketMapping>();
333
334 std::string vf = validFrom, vt = validTo;
335 if (vf != "") {
336 try {
337 ore::data::parseDate(vf);
338 } catch(std::exception& e) {
339 ALOG("Error parsing validFrom date, ignore: " << e.what());
340 vf = "";
341 }
342 }
343 if (vt != "") {
344 try {
345 ore::data::parseDate(vt);
346 } catch(std::exception& e) {
347 ALOG("Error parsing validTo date, ignore: " << e.what());
348 vt = "";
349 }
350 }
351
352 bucketMapping_[rt][qualifier].insert(BucketMapping(bucket, vt, vf, fallback));
353}

355string SimmBucketMapperBase::irBucket(const string& qualifier) const {
356 if (qualifier == "USD" || qualifier == "EUR" || qualifier == "GBP" || qualifier == "AUD" || qualifier == "CAD" ||
357 qualifier == "CHF" || qualifier == "DKK" || qualifier == "HKD" || qualifier == "KRW" || qualifier == "NOK" ||
358 qualifier == "NZD" || qualifier == "SEK" || qualifier == "SGD" || qualifier == "TWD")
359 return "1";
360 if (qualifier == "JPY")
361 return "2";
362 return "3";
363}

365void SimmBucketMapperBase::checkRiskType(const RiskType& riskType) const {
366 QL_REQUIRE(riskType != RiskType::IRCurve, "Risk type " << RiskType::IRCurve << " is mapped to buckets internally.");
367
368 QL_REQUIRE(hasBuckets(riskType), "The risk type " << riskType << " does not have buckets.");

371 << "mappings are stored in their non-vol counterparts. Use "

373 << riskType << ".");
374}
