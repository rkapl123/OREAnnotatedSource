26ClonedScenarioGenerator::ClonedScenarioGenerator(const boost::shared_ptr<ScenarioGenerator>& scenarioGenerator,
27 const std::vector<Date>& dates, const Size nSamples) {
28 DLOG("Build cloned scenario generator for " << dates.size() << " dates and " << nSamples << " samples.");
29 scenarioGenerator->reset();
30 scenarios_.resize(nSamples * dates.size());
31 for (Size i = 0; i < nSamples; ++i) {
32 for (Size j = 0; j < dates.size(); ++j) {
33 scenarios_[i * dates.size() + j] = scenarioGenerator->next(dates[j])->clone();
34 }
35 }
36}

38boost::shared_ptr<Scenario> ClonedScenarioGenerator::next(const Date& d) {
39 QL_REQUIRE(i_ < scenarios_.size(), "ClonedScenarioGenerator::next(" << d << "): no more scenarios stored.");
40 return scenarios_[i_++];
41}
