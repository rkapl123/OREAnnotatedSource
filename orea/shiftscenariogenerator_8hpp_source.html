119 //! Apply 1d triangular shift to 1d data such as yield curves, public to allow test suite access
120 /*!
121 Apply triangular shaped shifts to the underlying curve where the triangle
122 reaches from the previous to the next shift tenor point with peak at the
123 current shift tenor point.
124 At the initial and final shift tenor the shape is replaced such that the
125 full shift is applied to all curve grid points to the left of the first
126 shift point and to the right of the last shift point, respectively.
127 The procedure guarantees that no sensitivity to original curve points is
128 "missed" when the shift curve is less granular, e.g.
129 original curve |...|...|...|...|...|...|...|...|...|
130 shift curve ......|...........|...........|......
131 */
132 void applyShift(
133 //! Number of the shift curve tenor point to be shifted here
134 Size j,
135 //! Shift size interpreted as either absolute or relative shift
136 Real shiftSize,
137 //! Upwards shift if true, otherwise downwards
138 bool up,
139 //! Absolute: newValue = oldValue + shiftSize. Relative: newValue = oldValue * (1 + shiftSize)
140 ShiftType type,
141 //! Shift tenors expressed as times
142 const vector<Time>& shiftTimes,
143 //! Input curve values such as zero rates
144 const vector<Real>& values,
145 //! Tenor points of the input curve, expressed as times
146 const vector<Time>& times,
147 //! Resulting shifted curve with same tenor structure as the input curve
148 vector<Real>& shiftedValues,
149 //! Initialise shiftedValues vector before applying this shift j (yes for sensitivity, no for stress)
150 bool initialise);
151
152 //! Apply 2d shift to 2d matrix such as swaption volatilities, public to allow test suite access
153 /*! This is the 2d generalisation of the 1d version of applyShift()
154 */
155 void applyShift(
156 //! Index of the shift tenor in "expiry" direction
157 Size j,
158 //! Index of the shift tenor in "term" (Swaptions) or "strike" (Caps) direction
159 Size k,
160 //! Shift size interpreted as either absolute or relative shift
161 Real shiftSize,
162 //! Upwards shift if true, otherwise downwards
163 bool up,
164 //! Absolute: newValue = oldValue + shiftSize. Relative: newValue = oldValue * (1 + shiftSize)
165 ShiftType type,
166 //! Coordinate time in "expiry" direction of the shift curve
167 const vector<Time>& shiftX,
168 //! Coordinate time in "term" or "strike" direction of the shift curve
169 const vector<Time>& shiftY,
170 //! Coordinate time in "expiry" direction of the underlying data
171 const vector<Time>& dataX,
172 //! Coordinate time in "term" or "strike" direction of the underlying data
173 const vector<Time>& dataY,
174 //! Matrix of input data
175 const vector<vector<Real>>& data,
176 //! Matrix of shifted result data
177 vector<vector<Real>>& shiftedData,
178 //! Initialise shiftedData vector before applying this shift j/k (yes for sensitivity, no for stress)
179 bool initialise);
180
181 //! return the base scenario
182 boost::shared_ptr<Scenario> baseScenario() const { return scenarios_.front(); }
