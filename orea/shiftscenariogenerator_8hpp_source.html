118  //! Apply 1d triangular shift to 1d data such as yield curves, public to allow test suite access
119  /*!
120  Apply triangular shaped shifts to the underlying curve where the triangle
121  reaches from the previous to the next shift tenor point with peak at the
122  current shift tenor point.
123  At the initial and final shift tenor the shape is replaced such that the
124  full shift is applied to all curve grid points to the left of the first
125  shift point and to the right of the last shift point, respectively.
126  The procedure guarantees that no sensitivity to original curve points is
127  "missed" when the shift curve is less granular, e.g.
128  original curve |...|...|...|...|...|...|...|...|...|
129  shift curve ......|...........|...........|......
130  */
131  void applyShift(
132  //! Number of the shift curve tenor point to be shifted here
133  Size j,
134  //! Shift size interpreted as either absolute or relative shift
135  Real shiftSize,
136  //! Upwards shift if true, otherwise downwards
137  bool up,
138  //! Absolute: newValue = oldValue + shiftSize. Relative: newValue = oldValue * (1 + shiftSize)
139  ShiftType type,
140  //! Shift tenors expressed as times
141  const vector<Time>& shiftTimes,
142  //! Input curve values such as zero rates
143  const vector<Real>& values,
144  //! Tenor points of the input curve, expressed as times
145  const vector<Time>& times,
146  //! Resulting shifted curve with same tenor structure as the input curve
147  vector<Real>& shiftedValues,
148  //! Initialise shiftedValues vector before applying this shift j (yes for sensitivity, no for stress)
149  bool initialise);
150 
151  //! Apply 2d shift to 2d matrix such as swaption volatilities, public to allow test suite access
152  /*! This is the 2d generalisation of the 1d version of applyShift()
153  */
154  void applyShift(
155  //! Index of the shift tenor in "expiry" direction
156  Size j,
157  //! Index of the shift tenor in "term" (Swaptions) or "strike" (Caps) direction
158  Size k,
159  //! Shift size interpreted as either absolute or relative shift
160  Real shiftSize,
161  //! Upwards shift if true, otherwise downwards
162  bool up,
163  //! Absolute: newValue = oldValue + shiftSize. Relative: newValue = oldValue * (1 + shiftSize)
164  ShiftType type,
165  //! Coordinate time in "expiry" direction of the shift curve
166  const vector<Time>& shiftX,
167  //! Coordinate time in "term" or "strike" direction of the shift curve
168  const vector<Time>& shiftY,
169  //! Coordinate time in "expiry" direction of the underlying data
