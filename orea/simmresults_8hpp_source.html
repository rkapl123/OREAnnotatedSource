35/*! A container for SIMM results broken down by product class, risk class
36 and margin type.
37*/
38class SimmResults {
39public:
40 typedef SimmConfiguration::ProductClass ProductClass;
41 typedef SimmConfiguration::RiskClass RiskClass;
42 typedef SimmConfiguration::MarginType MarginType;
43 typedef std::tuple<ProductClass, RiskClass, MarginType, std::string> Key;
44
45 SimmResults(const std::string& ccy = "") : ccy_(ccy){};
46
47 /*! Add initial margin value \p im to the results container for the given combination of
48 SIMM <em>product class</em>, <em>risk class</em> and <em>margin type</em>
49
50 \remark If there is already a result in the container for that combination, it is
51 overwritten if overwrite=true. Otherwise, the amounts are added together.
52 Can check this using the <code>has</code> method before adding.
53 */
54 void add(const SimmConfiguration::ProductClass& pc, const SimmConfiguration::RiskClass& rc,
55 const SimmConfiguration::MarginType& mt, const std::string& b, QuantLib::Real im,
56 const std::string& calculationCurrency, const bool overwrite);
57
58 //! Convert SIMM amounts to a different currency
59 void convert(const boost::shared_ptr<ore::data::Market>& market, const std::string& currency);
60 void convert(QuantLib::Real fxSpot, const std::string& currency);
61
62 /*! Get the initial margin value from the results container for the given combination of
63 SIMM <em>product class</em>, <em>risk class</em> and <em>margin type</em>
64
65 \warning returns <code>QuantLib::Null<QuantLib::Real></code> if there is no initial margin
66 value in the results for the given combination. Can avoid this by first checking
67 the results using the <code>has</code> method.
68 */
69 QuantLib::Real get(const SimmConfiguration::ProductClass& pc, const SimmConfiguration::RiskClass& rc,
70 const SimmConfiguration::MarginType& mt, const std::string b) const;
71
72 /*! Check if there is an initial margin value in the results container for the given combination of
73 SIMM <em>product class</em>, <em>risk class</em> and <em>margin type</em>
74 */
75 bool has(const SimmConfiguration::ProductClass& pc, const SimmConfiguration::RiskClass& rc,
76 const SimmConfiguration::MarginType& mt, const std::string b) const;
77
78 //! Return true if the container is empty, otherwise false
79 bool empty() const;
80
81 //! Clear the results from the container
82 void clear();
83
84 //! Return the map containing the results
85 // Not nice to do this i.e. exposing inner implementation but it
86 // allows to iterate over results cleanly. Would be better to subclass
87 // std::iterator in this class.
88 const std::map<Key, QuantLib::Real>& data() const { return data_; }
89
90 const std::string& currency() const { return ccy_; }
91
92 std::string& currency() { return ccy_; }
93
94private:
95 std::map<Key, QuantLib::Real> data_;
96 std::string ccy_;
97};
