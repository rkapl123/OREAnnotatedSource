31ExposureCalculator::ExposureCalculator(
32 const boost::shared_ptr<Portfolio>& portfolio, const boost::shared_ptr<NPVCube>& cube,
33 const boost::shared_ptr<CubeInterpretation> cubeInterpretation,
34 const boost::shared_ptr<Market>& market,
35 bool exerciseNextBreak, const string& baseCurrency, const string& configuration,
36 const Real quantile, const CollateralExposureHelper::CalculationType calcType, const bool multiPath,
37 const bool flipViewXVA)
38 : portfolio_(portfolio), cube_(cube), cubeInterpretation_(cubeInterpretation),
39 market_(market), exerciseNextBreak_(exerciseNextBreak),
40 baseCurrency_(baseCurrency), configuration_(configuration),
41 quantile_(quantile), calcType_(calcType),
42 multiPath_(multiPath), dates_(cube->dates()),
43 today_(market_->asofDate()), dc_(ActualActual(ActualActual::ISDA)), flipViewXVA_(flipViewXVA) {
44
45 QL_REQUIRE(portfolio_, "portfolio is null");
46
47 if (multiPath) {
48 exposureCube_ = boost::make_shared<SinglePrecisionInMemoryCubeN>(
49 market->asofDate(), portfolio_->ids(), dates_,
50 cube_->samples(), EXPOSURE_CUBE_DEPTH);// EPE, ENE, allocatedEPE, allocatedENE
51 } else {
52 exposureCube_ = boost::make_shared<DoublePrecisionInMemoryCubeN>(
53 market->asofDate(), portfolio_->ids(), dates_,
54 1, EXPOSURE_CUBE_DEPTH);// EPE, ENE, allocatedEPE, allocatedENE
55 }
56
57 set<string> nettingSetIdsSet;
58 for (const auto& trade : portfolio->trades())
59 nettingSetIdsSet.insert(trade->envelope().nettingSetId());
60 nettingSetIds_= vector<string>(nettingSetIdsSet.begin(), nettingSetIdsSet.end());
61
62 times_ = vector<Real>(dates_.size(), 0.0);
63 for (Size i = 0; i < dates_.size(); i++)
64 times_[i] = dc_.yearFraction(today_, cube_->dates()[i]);
65
66 boost::shared_ptr<RegularCubeInterpretation> regularCubeInterpretation =
67 boost::dynamic_pointer_cast<RegularCubeInterpretation>(cubeInterpretation_);
68 isRegularCubeStorage_ = (regularCubeInterpretation != NULL);
69}

71void ExposureCalculator::build() {
72 LOG("Compute trade exposure profiles, " << (flipViewXVA_ ? "inverted (flipViewXVA = Y)" : "regular (flipViewXVA = N)"));
73
74 for (Size i = 0; i < portfolio_->size(); ++i) {
75 string tradeId = portfolio_->trades()[i]->id();
76 string nettingSetId = portfolio_->trades()[i]->envelope().nettingSetId();
77 LOG("Aggregate exposure for trade " << tradeId);
78 if (nettingSetDefaultValue_.find(nettingSetId) == nettingSetDefaultValue_.end()) {
79 nettingSetDefaultValue_[nettingSetId] = vector<vector<Real>>(dates_.size(), vector<Real>(cube_->samples(), 0.0));
80 nettingSetCloseOutValue_[nettingSetId] = vector<vector<Real>>(dates_.size(), vector<Real>(cube_->samples(), 0.0));
81 }
82
83 // Identify the next break date if provided, default is trade maturity.
84 Date nextBreakDate = portfolio_->trades()[i]->maturity();
85 TradeActions ta = portfolio_->trades()[i]->tradeActions();
86 if (exerciseNextBreak_ && !ta.empty()) {
87 // loop over actions and pick next mutual break, if available
88 vector<TradeAction> actions = ta.actions();
89 for (Size j = 0; j < actions.size(); ++j) {
90 DLOG("TradeAction for " << tradeId << ", actionType " << actions[j].type() << ", actionOwner "
91 << actions[j].owner());
92 // FIXME: Introduce enumeration and parse text when building trade
93 if (actions[j].type() == "Break" && actions[j].owner() == "Mutual") {
94 QuantLib::Schedule schedule = ore::data::makeSchedule(actions[j].schedule());
95 vector<Date> dates = schedule.dates();
96 std::sort(dates.begin(), dates.end());
97 Date today = Settings::instance().evaluationDate();
98 for (Size k = 0; k < dates.size(); ++k) {
99 if (dates[k] > today && dates[k] < nextBreakDate) {
100 nextBreakDate = dates[k];
101 DLOG("Next break date for trade " << tradeId << ": "
102 << QuantLib::io::iso_date(nextBreakDate));
103 break;
104 }
105 }
106 }
107 }
108 }
109
110 Handle<YieldTermStructure> curve = market_->discountCurve(baseCurrency_, configuration_);
111 Real npv0;
112 if (flipViewXVA_) {
113 npv0 = -cube_->getT0(i);
114 } else {
115 npv0 = cube_->getT0(i);
116 }
117 vector<Real> epe(dates_.size() + 1, 0.0);
118 vector<Real> ene(dates_.size() + 1, 0.0);
119 vector<Real> ee_b(dates_.size() + 1, 0.0);
120 vector<Real> eee_b(dates_.size() + 1, 0.0);
121 vector<Real> pfe(dates_.size() + 1, 0.0);
122 epe[0] = std::max(npv0, 0.0);
123 ene[0] = std::max(-npv0, 0.0);
124 ee_b[0] = epe[0];
125 eee_b[0] = ee_b[0];
126 pfe[0] = std::max(npv0, 0.0);
127 exposureCube_->setT0(epe[0], tradeId, ExposureIndex::EPE);
128 exposureCube_->setT0(ene[0], tradeId, ExposureIndex::ENE);
129 for (Size j = 0; j < dates_.size(); ++j) {
130 Date d = cube_->dates()[j];
131 vector<Real> distribution(cube_->samples(), 0.0);
132 for (Size k = 0; k < cube_->samples(); ++k) {
133 // RL 2020-07-17
134 // 1) If the calculation type is set to NoLag:
135 // Collateral balances are NOT delayed by the MPoR, but we use the close-out NPV.
136 // 2) Otherwise:
137 // Collateral balances are delayed by the MPoR (if possible, i.e. the valuation
138 // grid has MPoR spacing), and we use the default date NPV.
139 // This is the treatment in the ORE releases up to June 2020).
140 Real defaultValue =
141 d > nextBreakDate && exerciseNextBreak_ ? 0.0 : cubeInterpretation_->getDefaultNpv(cube_, i, j, k);
142 Real closeOutValue;
143 if (isRegularCubeStorage_ && j == dates_.size() - 1)
144 closeOutValue = defaultValue;
145 else
146 closeOutValue = d > nextBreakDate && exerciseNextBreak_
147 ? 0.0
148 : cubeInterpretation_->getCloseOutNpv(cube_, i, j, k);
149 Real npv = calcType_ == CollateralExposureHelper::CalculationType::NoLag ? closeOutValue : defaultValue;
150 epe[j + 1] += max(npv, 0.0) / cube_->samples();
151 ene[j + 1] += max(-npv, 0.0) / cube_->samples();
152 nettingSetDefaultValue_[nettingSetId][j][k] += defaultValue;
153 nettingSetCloseOutValue_[nettingSetId][j][k] += closeOutValue;
154 distribution[k] = npv;
155 if (multiPath_) {
156 exposureCube_->set(max(npv, 0.0), tradeId, d, k, ExposureIndex::EPE);
157 exposureCube_->set(max(-npv, 0.0), tradeId, d, k, ExposureIndex::ENE);
158 }
159 }
160 if (!multiPath_) {
161 exposureCube_->set(epe[j + 1], tradeId, d, 0, ExposureIndex::EPE);
162 exposureCube_->set(ene[j + 1], tradeId, d, 0, ExposureIndex::ENE);
163 }
164 ee_b[j + 1] = epe[j + 1] / curve->discount(cube_->dates()[j]);
165 eee_b[j + 1] = std::max(eee_b[j], ee_b[j + 1]);
166 std::sort(distribution.begin(), distribution.end());
167 Size index = Size(floor(quantile_ * (cube_->samples() - 1) + 0.5));
168 pfe[j + 1] = std::max(distribution[index], 0.0);
169 }
170 ee_b_[tradeId] = ee_b;
171 eee_b_[tradeId] = eee_b;
172 pfe_[tradeId] = pfe;
173
174 Real epe_b = 0.0;
175 Real eepe_b = 0.0;
176
177 Size t = 0;
178 Calendar cal = WeekendsOnly();
179 /*The time average in the EEPE calculation is taken over the first year of the exposure evolution
180 (or until maturity if all positions of the netting set mature before one year).
181 This one year point is actually taken to be today+1Y+4D, so that the 1Y point on the dateGrid is always
182 included.
183 This may effect DateGrids with daily data points*/
184 Date maturity = std::min(cal.adjust(today_ + 1 * Years + 4 * Days), portfolio_->trades()[i]->maturity());
185 QuantLib::Real maturityTime = dc_.yearFraction(today_, maturity);
186
187 while (t < dates_.size() && times_[t] <= maturityTime)
188 ++t;
189
190 if (t > 0) {
191 vector<double> weights(t);
192 weights[0] = times_[0];
193 for (Size k = 1; k < t; k++)
194 weights[k] = times_[k] - times_[k - 1];
195 double totalWeights = std::accumulate(weights.begin(), weights.end(), 0.0);
196 for (Size k = 0; k < t; k++)
197 weights[k] /= totalWeights;
198
199 for (Size k = 0; k < t; k++) {
200 epe_b += ee_b[k] * weights[k];
201 eepe_b += eee_b[k] * weights[k];
202 }
203 }
204 epe_b_[tradeId] = epe_b;
205 eepe_b_[tradeId] = eepe_b;
206 }
207}

209vector<Real> ExposureCalculator::getMeanExposure(const string& tid, ExposureIndex index) {
210 vector<Real> exp(dates_.size() + 1, 0.0);
211 exp[0] = exposureCube_->getT0(tid, index);
212 for (Size i = 0; i < dates_.size(); i++) {
213 for (Size k = 0; k < exposureCube_->samples(); k++) {
214 exp[i + 1] += exposureCube_->get(tid, dates_[i], k, index);
215 }
216 exp[i + 1] /= exposureCube_->samples();
217 }
218 return exp;
219}
