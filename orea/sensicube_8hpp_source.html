40 //! Naieve concrete implementation of NPVSensiCube
41 template <typename T> class SensiCube : public ore::analytics::NPVSensiCube {
42 public:
43  SensiCube(const std::vector<std::string>& ids, const QuantLib::Date& asof, QuantLib::Size samples, const T& t = T())
44  : ids_(ids), asof_(asof), dates_(1, asof_), samples_(samples), t0Data_(ids.size(), t),
45  tradeNPVs_(ids.size(), std::map<Size, T>()) {}

47  //! load cube from an archive
48  void load(const std::string& fileName) override {
49  std::ifstream ifs(fileName.c_str(), std::fstream::binary);
50  QL_REQUIRE(ifs.is_open(), "error opening file " << fileName);
51  boost::archive::binary_iarchive ia(ifs);
52  ia >> *this;
53  }
54 
55  //! write cube to an archive
56  void save(const std::string& fileName) const override {
57  std::ofstream ofs(fileName.c_str(), std::fstream::binary);
58  QL_REQUIRE(ofs.is_open(), "error opening file " << fileName);
59  boost::archive::binary_oarchive oa(ofs);
60  oa << *this;
61  }

77  Real getT0(QuantLib::Size i, QuantLib::Size) const override {
78  this->check(i, 0, 0);

82  //! Set a value in the cube
83  void setT0(QuantLib::Real value, QuantLib::Size i, QuantLib::Size) override {
84  this->check(i, 0, 0);
85  this->t0Data_[i] = static_cast<T>(value);
86  }
87 
88  //! Get a value from the cube
89  Real get(QuantLib::Size i, QuantLib::Size j, QuantLib::Size k, QuantLib::Size) const override {
90  this->check(i, j, k);
91 
92  auto itr = this->tradeNPVs_[i].find(k);
93  if (itr != tradeNPVs_[i].end()) {
94  return itr->second;
95  } else {
96  return this->t0Data_[i];
97  }
98  }
99 
100  //! Set a value in the cube
101  void set(QuantLib::Real value, QuantLib::Size i, QuantLib::Size j, QuantLib::Size k, QuantLib::Size) override {
102  this->check(i, j, k);
103  this->tradeNPVs_[i][k] = static_cast<T>(value);
104  }
105 
106  const std::map<QuantLib::Size, QuantLib::Real>& getTradeNPVs(QuantLib::Size i) const override {
107  return tradeNPVs_[i];
108  }
109 
110 private:

112  template <class Archive> void serialize(Archive& ar, const unsigned int) {
113  ar& ids_;
114  ar& asof_;

116  ar& t0Data_;

118  }

125 protected:

129  void check(QuantLib::Size i, QuantLib::Size j, QuantLib::Size k) const {
130  QL_REQUIRE(i < numIds(), "Out of bounds on ids (i=" << i << ")");
131  QL_REQUIRE(j < depth(), "Out of bounds on depth (j=" << j << ")");
132  QL_REQUIRE(k < samples(), "Out of bounds on samples (k=" << k << ")");
133  }
134 };
