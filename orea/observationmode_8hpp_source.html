39private:

42public:
43 //! Allowable mode mode

45
46 Mode mode() { return mode_; }
47
48 void setMode(Mode s) { mode_ = s; }
49
50 void setMode(const std::string& s) {
51 if (s == "None")
52 mode_ = Mode::None;
53 else if (s == "Disable")
54 mode_ = Mode::Disable;
55 else if (s == "Defer")
56 mode_ = Mode::Defer;
57 else if (s == "Unregister")
58 mode_ = Mode::Unregister;
59 else {
60 QL_FAIL("Invalid ObserverMode string " << s);
61 }
62 }

64private:
