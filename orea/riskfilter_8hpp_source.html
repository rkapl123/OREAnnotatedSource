37//! Risk Filter
38/*! The risk filter class groups risk factor keys w.r.t. a risk class (IR, FX, EQ...) and a risk type (delta-gamma,
39 * vega, ...). It can e.g. be used to break down a var report.
40 */
41class RiskFilter {
42public:
43 //! for both risk class and risk type, index 0 stands for "all"
44 RiskFilter(const Size riskClassIndex, const Size riskTypeIndex);
45
46 bool allowed(const RiskFactorKey::KeyType& t) const;
47 const std::string& riskClassLabel() const { return riskClassLabel_[riskClassIndex_]; }
48 const std::string& riskTypeLabel() const { return riskTypeLabel_[riskTypeIndex_]; }
49
50 static Size numberOfRiskClasses() { return riskClassLabel_.size(); }
51 static Size numberOfRiskTypes() { return riskTypeLabel_.size(); }
52
53private:
54 static const std::vector<std::string> riskClassLabel_;
55 static const std::vector<std::string> riskTypeLabel_;
56 const Size riskClassIndex_, riskTypeIndex_;
57 std::set<RiskFactorKey::KeyType> allowed_;
58 bool neg_;
59};
