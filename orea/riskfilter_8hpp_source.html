38//! Risk Filter
39/*! The risk filter class groups risk factor keys w.r.t. a risk class (IR, FX, EQ...) and a risk type (delta-gamma,
40 * vega, ...). It can e.g. be used to break down a var report.
41 */
42class RiskFilter : public ScenarioFilter {
43public:
44 //! for both risk class and risk type, index 0 stands for "all"
45 RiskFilter(const Size riskClassIndex, const Size riskTypeIndex);
46
47 bool allow(const RiskFactorKey& t) const override;
48 const std::string& riskClassLabel() const { return riskClassLabel_[riskClassIndex_]; }
49 const std::string& riskTypeLabel() const { return riskTypeLabel_[riskTypeIndex_]; }
50
51 static Size numberOfRiskClasses() { return riskClassLabel_.size(); }
52 static Size numberOfRiskTypes() { return riskTypeLabel_.size(); }
53
54private:
55 static const std::vector<std::string> riskClassLabel_;
56 static const std::vector<std::string> riskTypeLabel_;
57 const Size riskClassIndex_, riskTypeIndex_;
58 std::set<RiskFactorKey::KeyType> allowed_;
59 bool neg_;
60};
