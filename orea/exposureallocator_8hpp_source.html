36//! Exposure allocator base class
37/*!
38 Derived classes implement a constructor with the relevant additional input data
39 and a build function that performs the XVA calculations for all netting sets and
40 along all paths.
41*/
42class ExposureAllocator {
43public:
44
45 enum class AllocationMethod {
46 None,
47 Marginal, // Pykhtin & Rosen, 2010
48 RelativeFairValueGross,
49 RelativeFairValueNet,
50 RelativeXVA
51 };
52
53 ExposureAllocator(
54 const boost::shared_ptr<Portfolio>& portfolio,
55 const boost::shared_ptr<NPVCube>& tradeExposureCube,
56 const boost::shared_ptr<NPVCube>& nettedExposureCube,
57 const Size allocatedTradeEpeIndex = 2, const Size allocatedTradeEneIndex = 3,
58 const Size tradeEpeIndex = 0, const Size tradeEneIndex = 1,
59 const Size nettingSetEpeIndex = 1, const Size nettingSetEneIndex = 2);
60
61 virtual ~ExposureAllocator() {}
62 const boost::shared_ptr<NPVCube>& exposureCube() { return tradeExposureCube_; }
63 //! Compute exposures along all paths and fill result structures
64 virtual void build();
65
66
67protected:
68 virtual Real calculateAllocatedEpe(const string& tid, const string& nid, const Date& date, const Size sample) = 0;
69 virtual Real calculateAllocatedEne(const string& tid, const string& nid, const Date& date, const Size sample) = 0;
70 boost::shared_ptr<Portfolio> portfolio_;
71 boost::shared_ptr<NPVCube> tradeExposureCube_;
72 boost::shared_ptr<NPVCube> nettedExposureCube_;
73 Size tradeEpeIndex_;
74 Size tradeEneIndex_;
75 Size allocatedTradeEpeIndex_;
76 Size allocatedTradeEneIndex_;
77 Size nettingSetEpeIndex_;
78 Size nettingSetEneIndex_;
79 map<string, Real> nettingSetValueToday_, nettingSetPositiveValueToday_, nettingSetNegativeValueToday_;
80};
