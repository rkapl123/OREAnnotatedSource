Detailed Description

Scenario Generator using cross asset model paths.

The generator expects

  • a calibrated model,
  • an associated multi path generator (i.e. providing paths for all factors of the model ordered as described in the model),
  • a scenario factory that provides building scenario class instances,
  • the configuration of market curves to be simulated
  • a simulation date grid that starts in the future, i.e. does not include today's date
  • the associated time grid including t=0

398  } else {
399  QL_FAIL("CrossAssetModelScenarioGenerator: expected inflation model to be JY or DK.");
400  }
401 
402  scenarios[i]->add(cpiKeys_[j], cpi);
403  }
404 
405  // Zero inflation curves
406  for (Size j = 0; j < zeroInfCurves_.size(); ++j) {
407 
408  auto tup = zeroInfCurves_[j];
409 
410  // State variables needed depends on model, 3 for JY and 2 for DK.
411  auto idx = std::get<0>(tup);
412  Array state(3);
413  state[0] = sample.value[model_->pIdx(INF, idx, 0)][i + 1];
414  state[1] = sample.value[model_->pIdx(INF, idx, 1)][i + 1];
415  if (std::get<2>(tup) == DK) {
416  state.resize(2);
417  } else {
418  state[2] = sample.value[model_->pIdx(IR, std::get<1>(tup))][i + 1];
419  }
420 
421  // Update the term structure's date and state.
422  auto ts = std::get<3>(tup);
423  ts->move(dates_[i], state);
424 
425  // Populate the zero inflation scenario values based on the current date and state.
426  for (Size k = 0; k < ten_zinf_[j].size(); k++) {
427  Time T = dc.yearFraction(dates_[i], dates_[i] + ten_zinf_[j][k]);
428  scenarios[i]->add(zeroInflationKeys_[j * ten_zinf_[j].size() + k], ts->zeroRate(T));
429  }
430  }
431 
432  // YoY inflation curves
433  for (Size j = 0; j < yoyInfCurves_.size(); ++j) {
434 
435  auto tup = yoyInfCurves_[j];
436 
437  // For YoY model implied term structure, JY and DK both need 3 state variables.
438  auto idx = std::get<0>(tup);
439  Array state(3);
440  state[0] = sample.value[model_->pIdx(INF, idx, 0)][i + 1];
441  state[1] = sample.value[model_->pIdx(INF, idx, 1)][i + 1];
442  state[2] = sample.value[model_->pIdx(IR, std::get<1>(tup))][i + 1];
443 
444  // Update the term structure's date and state.
445  auto ts = std::get<3>(tup);
446  ts->move(dates_[i], state);
447 
448  // Create the YoY pillar dates from the tenors.
449  vector<Date> pillarDates(ten_yinf_[j].size());
450  for (Size k = 0; k < pillarDates.size(); ++k)
451  pillarDates[k] = dates_[i] + ten_yinf_[j][k];
452 
453  // Use the YoY term structure's YoY rates to populate the scenarios.
454  auto yoyRates = ts->yoyRates(pillarDates);
455  for (Size k = 0; k < pillarDates.size(); ++k) {
456  scenarios[i]->add(yoyInflationKeys_[j * ten_yinf_[j].size() + k], yoyRates.at(pillarDates[k]));
457  }
458  }
459 
460  // Credit curves
461  for (Size j = 0; j < n_cr_; ++j) {
462  if (model_->modelType(CR, j) == ModelType::LGM1F) {
463  Real z = sample.value[model_->pIdx(CR, j, 0)][i + 1];
464  Real y = sample.value[model_->pIdx(CR, j, 1)][i + 1];
465  lgmDefaultCurves_[j]->move(dates_[i], z, y);
466  for (Size k = 0; k < ten_dfc_[j].size(); k++) {
467  Date d = dates_[i] + ten_dfc_[j][k];
468  Time T = dc.yearFraction(dates_[i], d);
469  Real survProb = std::max(lgmDefaultCurves_[j]->survivalProbability(T), 0.00001);
470  scenarios[i]->add(defaultCurveKeys_[j * ten_dfc_[j].size() + k], survProb);
471  }
472  } else if (model_->modelType(CR, j) == ModelType::CIRPP) {
473  Real y = sample.value[model_->pIdx(CR, j, 0)][i + 1];
474  cirppDefaultCurves_[j]->move(dates_[i], y);
475  for (Size k = 0; k < ten_dfc_[j].size(); k++) {
476  Date d = dates_[i] + ten_dfc_[j][k];
477  Time T = dc.yearFraction(dates_[i], d);
478  Real survProb = std::max(cirppDefaultCurves_[j]->survivalProbability(T), 0.00001);
479  scenarios[i]->add(defaultCurveKeys_[j * ten_dfc_[j].size() + k], survProb);
480  }
481  }
482  }
483 
484  // TODO: Further risk factor classes are added here
485  }
486  return scenarios;
487 }
