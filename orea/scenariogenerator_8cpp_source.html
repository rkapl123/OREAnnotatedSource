200 void test_lgm(bool sobol, bool antithetic, bool brownianBridge) {
201 
202  TestData d;
203 
204  // Simulation date grid
205  Date today = d.referenceDate;
206  std::vector<Period> tenorGrid = {1 * Years, 2 * Years, 3 * Years, 5 * Years, 7 * Years, 10 * Years};
207  ore::analytics::DateGrid grid(tenorGrid);
208 
209  // Model
210  boost::shared_ptr<QuantExt::LGM> model = d.lgm;
211 
212  // State process
213  boost::shared_ptr<StochasticProcess1D> stateProcess = model->stateProcess();
214 
215  // Simulation market parameters, we just need the yield curve structure here
216  BOOST_TEST_MESSAGE("set up sim market parameters");
217  boost::shared_ptr<ScenarioSimMarketParameters> simMarketConfig(new ScenarioSimMarketParameters);
218  simMarketConfig->setYieldCurveTenors("", {3 * Months, 6 * Months, 1 * Years, 2 * Years, 3 * Years, 4 * Years,
219  5 * Years, 7 * Years, 10 * Years, 12 * Years, 15 * Years, 20 * Years,
220  30 * Years, 40 * Years, 50 * Years});
221  simMarketConfig->setSimulateFXVols(false);
222  simMarketConfig->setSimulateEquityVols(false);
223  simMarketConfig->setYieldCurveDayCounters("", "ACT/ACT");
224  // Multi path generator: Pseudo Random
225  BigNatural seed = 42;
226  // bool antithetic = true;
227  boost::shared_ptr<QuantExt::MultiPathGeneratorBase> pathGen;
228  if (sobol) {
229  if (brownianBridge)
230  pathGen = boost::make_shared<MultiPathGeneratorSobolBrownianBridge>(stateProcess, grid.timeGrid(),
231  SobolBrownianGenerator::Diagonal, seed);
232  else
233  pathGen = boost::make_shared<MultiPathGeneratorSobol>(stateProcess, grid.timeGrid(), seed);
234  } else
235  pathGen =
236  boost::make_shared<MultiPathGeneratorMersenneTwister>(stateProcess, grid.timeGrid(), seed, antithetic);
237 
238  // Scenario factory
239  // We assume different implementations of the scenario objects which are more or less
240  // optimized w.r.t. memory usage. Hence we use the scenario factory here to avoid
241  // switching in the scenario generator class below.
242  boost::shared_ptr<ScenarioFactory> scenarioFactory(new SimpleScenarioFactory);
243 
244  // Scenario Generator
245  boost::shared_ptr<LgmScenarioGenerator> scenGen =
246  boost::make_shared<LgmScenarioGenerator>(model, pathGen, scenarioFactory, simMarketConfig, today, grid);
247 
248  // Basic martingale tests
249  Size samples = 10000;
250  Real eur = 0.0, eur2 = 0.0;
251  for (Size i = 0; i < samples; i++) {
252  for (Date d : grid.dates()) {
253  boost::shared_ptr<Scenario> scenario = scenGen->next(d);
254  if (d == grid.dates().back()) { // in 10 years from today
255  RiskFactorKey key(RiskFactorKey::KeyType::DiscountCurve, "EUR", 8);
256  Real eur10yDiscount = scenario->get(key);
257  Real numeraire = scenario->getNumeraire();
258  eur += eur10yDiscount / numeraire;
259  eur2 += 1.0 / numeraire;
260  }
261  }
262  }
263 
264  eur /= samples;
265  eur2 /= samples;
266 
267  Real relTolerance = 0.01;
268  Real eurExpected = d.market->discountCurve("EUR")->discount(20.);
269  BOOST_CHECK_MESSAGE(fabs(eur - eurExpected) / eurExpected < relTolerance,
270  "EUR 20Y Discount mismatch: " << eur << " vs " << eurExpected);
271  Real eurExpected2 = d.market->discountCurve("EUR")->discount(10.);
272  BOOST_CHECK_MESSAGE(fabs(eur2 - eurExpected2) / eurExpected2 < relTolerance,
273  "EUR 10Y Discount mismatch: " << eur2 << " vs " << eurExpected2);
274 
275  BOOST_TEST_MESSAGE("LGM " << (sobol ? "Sobol " : "MersenneTwister ") << (antithetic ? "Antithetic" : "")
276  << (brownianBridge ? "BrownianBridge" : ""));
277  BOOST_TEST_MESSAGE("EUR 20Y Discount: " << eur << " vs " << eurExpected);
278  BOOST_TEST_MESSAGE("EUR 10Y Discount: " << eur2 << " vs " << eurExpected2);
279 }
