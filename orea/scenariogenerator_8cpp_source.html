214void test_lgm(bool sobol, bool antithetic, bool brownianBridge) {
215
216 BOOST_TEST_MESSAGE("call test_lgm with sobol=" << sobol << " antithetic=" << antithetic << " brownianBridge=" << brownianBridge);
217 TestData d;
218
219 // Simulation date grid
220 Date today = d.referenceDate;
221 std::vector<Period> tenorGrid = {1 * Years, 2 * Years, 3 * Years, 5 * Years, 7 * Years, 10 * Years};
222 DateGrid grid(tenorGrid);
223
224 // Model
225 boost::shared_ptr<QuantExt::LGM> model = d.lgm;
226
227 // State process
228 boost::shared_ptr<StochasticProcess1D> stateProcess =
229 boost::dynamic_pointer_cast<StochasticProcess1D>(model->stateProcess());
230
231 // Simulation market parameters, we just need the yield curve structure here
232 BOOST_TEST_MESSAGE("set up sim market parameters");
233 boost::shared_ptr<ScenarioSimMarketParameters> simMarketConfig(new ScenarioSimMarketParameters);
234 simMarketConfig->setYieldCurveTenors("", {3 * Months, 6 * Months, 1 * Years, 2 * Years, 3 * Years, 4 * Years,
235 5 * Years, 7 * Years, 10 * Years, 12 * Years, 15 * Years, 20 * Years,
236 30 * Years, 40 * Years, 50 * Years});
237 simMarketConfig->setSimulateFXVols(false);
238 simMarketConfig->setSimulateEquityVols(false);
239 // Multi path generator: Pseudo Random
240 BigNatural seed = 42;
241 // bool antithetic = true;
242 boost::shared_ptr<QuantExt::MultiPathGeneratorBase> pathGen;
243 if (sobol) {
244 if (brownianBridge)
245 pathGen = boost::make_shared<MultiPathGeneratorSobolBrownianBridge>(stateProcess, grid.timeGrid(),
246 SobolBrownianGenerator::Diagonal, seed);
247 else
248 pathGen = boost::make_shared<MultiPathGeneratorSobol>(stateProcess, grid.timeGrid(), seed);
249 } else
250 pathGen =
251 boost::make_shared<MultiPathGeneratorMersenneTwister>(stateProcess, grid.timeGrid(), seed, antithetic);
252
253 // Scenario factory
254 // We assume different implementations of the scenario objects which are more or less
255 // optimized w.r.t. memory usage. Hence we use the scenario factory here to avoid
256 // switching in the scenario generator class below.
257 boost::shared_ptr<ScenarioFactory> scenarioFactory(new SimpleScenarioFactory);
258
259 // Scenario Generator
260 boost::shared_ptr<LgmScenarioGenerator> scenGen =
261 boost::make_shared<LgmScenarioGenerator>(model, pathGen, scenarioFactory, simMarketConfig, today, grid);
262
263 // Basic martingale tests
264 Size samples = 10000;
265 Real eur = 0.0, eur2 = 0.0;
266 for (Size i = 0; i < samples; i++) {
267 for (Date d : grid.dates()) {
268 boost::shared_ptr<Scenario> scenario = scenGen->next(d);
269 if (d == grid.dates().back()) { // in 10 years from today
270 RiskFactorKey key(RiskFactorKey::KeyType::DiscountCurve, "EUR", 8);
271 Real eur10yDiscount = scenario->get(key);
272 Real numeraire = scenario->getNumeraire();
273 eur += eur10yDiscount / numeraire;
274 eur2 += 1.0 / numeraire;
275 }
276 }
277 }
278
279 eur /= samples;
280 eur2 /= samples;
281
282 Real relTolerance = 0.01;
283 Real eurExpected = d.market->discountCurve("EUR")->discount(20.);
284 BOOST_CHECK_MESSAGE(fabs(eur - eurExpected) / eurExpected < relTolerance,
285 "EUR 20Y Discount mismatch: " << eur << " vs " << eurExpected);
286 Real eurExpected2 = d.market->discountCurve("EUR")->discount(10.);
287 BOOST_CHECK_MESSAGE(fabs(eur2 - eurExpected2) / eurExpected2 < relTolerance,
288 "EUR 10Y Discount mismatch: " << eur2 << " vs " << eurExpected2);
289
290 BOOST_TEST_MESSAGE("LGM " << (sobol ? "Sobol " : "MersenneTwister ") << (antithetic ? "Antithetic" : "")
291 << (brownianBridge ? "BrownianBridge" : ""));
292 BOOST_TEST_MESSAGE("EUR 20Y Discount: " << eur << " vs " << eurExpected);
293 BOOST_TEST_MESSAGE("EUR 10Y Discount: " << eur2 << " vs " << eurExpected2);
294}

322void test_crossasset(bool sobol, bool antithetic, bool brownianBridge) {
323 TestData d;
324
325 // Simulation date grid
326 Date today = d.referenceDate;
327 std::vector<Period> tenorGrid = {1 * Years, 2 * Years, 3 * Years, 5 * Years, 7 * Years, 10 * Years};
328 boost::shared_ptr<DateGrid> grid = boost::make_shared<DateGrid>(tenorGrid);
329
330 // Model
331 boost::shared_ptr<QuantExt::CrossAssetModel> model = d.ccLgm;
332
333 // State process
334 boost::shared_ptr<StochasticProcess> stateProcess = model->stateProcess();
335
336 // Simulation market parameters, we just need the yield curve structure here
337 BOOST_TEST_MESSAGE("set up sim market parameters");
338 boost::shared_ptr<ScenarioSimMarketParameters> simMarketConfig(new ScenarioSimMarketParameters);
339 simMarketConfig->setYieldCurveTenors("", {3 * Months, 6 * Months, 1 * Years, 2 * Years, 3 * Years, 4 * Years,
340 5 * Years, 7 * Years, 10 * Years, 12 * Years, 15 * Years, 20 * Years,
341 30 * Years, 40 * Years, 50 * Years});
342 simMarketConfig->setSimulateFXVols(false);
343 simMarketConfig->setSimulateEquityVols(false);
344 simMarketConfig->setZeroInflationTenors("", {3 * Months, 6 * Months, 1 * Years, 2 * Years, 3 * Years, 4 * Years,
345 5 * Years, 7 * Years, 10 * Years, 12 * Years, 15 * Years, 20 * Years,
346 30 * Years, 40 * Years, 50 * Years});
347
348 // Multi path generator
349 BigNatural seed = 42;
350 if (auto tmp = boost::dynamic_pointer_cast<CrossAssetStateProcess>(stateProcess)) {
351 tmp->resetCache(grid->timeGrid().size() - 1);
352 }
353 boost::shared_ptr<QuantExt::MultiPathGeneratorBase> pathGen;
354 if (sobol) {
355 if (brownianBridge)
356 pathGen = boost::make_shared<MultiPathGeneratorSobolBrownianBridge>(stateProcess, grid->timeGrid(),
357 SobolBrownianGenerator::Diagonal, seed);
358 else
359 pathGen = boost::make_shared<MultiPathGeneratorSobol>(stateProcess, grid->timeGrid(), seed);
360 } else
361 pathGen =
362 boost::make_shared<MultiPathGeneratorMersenneTwister>(stateProcess, grid->timeGrid(), seed, antithetic);
363
364 // Scenario factory
365 // We assume different implementations of the scenario objects which are more or less
366 // optimized w.r.t. memory usage. Hence we use the scenario factory here to avoid
367 // switching in the scenario generator class below.
368 boost::shared_ptr<ScenarioFactory> scenarioFactory(new SimpleScenarioFactory);
369
370 // Scenario Generator
371 boost::shared_ptr<CrossAssetModelScenarioGenerator> scenGen = boost::make_shared<CrossAssetModelScenarioGenerator>(
372 model, pathGen, scenarioFactory, simMarketConfig, today, grid, d.market);
373
374 // Basic martingale tests
375 Size samples = 10000;
376 Real eur = 0.0, usd = 0.0, gbp = 0.0, eur2 = 0.0, usd2 = 0.0, gbp2 = 0.0, eur3 = 0.0;
377
378 cpu_timer timer;
379 for (Size i = 0; i < samples; i++) {
380 for (Date d : grid->dates()) {
381 boost::shared_ptr<Scenario> scenario = scenGen->next(d);
382
383 if (d == grid->dates().back()) { // in 10 years from today
384 RiskFactorKey eurKey(RiskFactorKey::KeyType::DiscountCurve, "EUR", 8);
385 RiskFactorKey usdKey(RiskFactorKey::KeyType::DiscountCurve, "USD", 8);
386 RiskFactorKey gbpKey(RiskFactorKey::KeyType::DiscountCurve, "GBP", 8);
387 RiskFactorKey usdeurKey(RiskFactorKey::KeyType::FXSpot, "USDEUR");
388 RiskFactorKey gbpeurKey(RiskFactorKey::KeyType::FXSpot, "GBPEUR");
389 RiskFactorKey euhicpKey(RiskFactorKey::KeyType::CPIIndex, "EUHICPXT");
390
391 Real usdeurFX = scenario->get(usdeurKey);
392 Real gbpeurFX = scenario->get(gbpeurKey);
393 Real numeraire = scenario->getNumeraire();
394 Real eur10yDiscount = scenario->get(eurKey);
395 Real gbp10yDiscount = scenario->get(gbpKey);
396 Real usd10yDiscount = scenario->get(usdKey);
397 Real euhicp = scenario->get(euhicpKey);
