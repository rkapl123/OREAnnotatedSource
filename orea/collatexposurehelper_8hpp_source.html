36//! Collateral Exposure Helper
37/*!
38 This class contains helper functions to aid in the calculation
39 of collateralised exposures.
40
41 It can be used to calculate margin requirements in the presence of
42 e.g. thresholds and minimum transfer amounts, update collateral
43 account details with e.g. new margin call info, and return
44 collateralised exposures to the user/invoker.
45
46 For further information refer to the detailed ORE documentation.
47
48 \ingroup analytics
49*/
50class CollateralExposureHelper {
51public:
52 /*!
53 Enumeration 'CalculationType' specifies how the collateralised
54 exposures should be calculated (please refer to Sungard white-paper
55 titled "Closing In On the CloseOut"):
56 - 'Symmetric' => margin calls only settled after margin period of risk
57 - 'AsymmetricCVA' => margin requested from ctp only settles after margin period of risk
58 -- (our margin postings settle instantaneously)
59 - 'AsymmetricDVA' => margin postings to ctp only settle after margin period of risk
60 -- (margin calls to receive collateral from counterparty settle instantaneously)
61 - 'NoLag' => margin calls/postings settled without margin period of risk delay
62 */
63 enum CalculationType { Symmetric, AsymmetricCVA, AsymmetricDVA, NoLag };
64
65 /*!
66 Calculates CSA margin requirement, taking the following into account
67 - uncollateralised value
68 - collateral value
69 - threshold
70 - minimum transfer amount
71 - independent amount
72 */
73 static Real marginRequirementCalc(const boost::shared_ptr<CollateralAccount>& collat, const Real& uncollatValue,
74 const Date& simulationDate);
75
76 /*!
77 Performs linear interpolation between dates to
78 estimate the value as of simulationDate.
79 Flat extrapolation at far end.
80 */
81 template <class T>
82 static Real estimateUncollatValue(const Date& simulationDate, const Real& npv_t0, const Date& date_t0,
83 const vector<vector<T>>& scenPvProfiles, const unsigned& scenIndex,
84 const vector<Date>& dateGrid);
85
86 /*!
87 Checks if margin call is in need of update, and updates if necessary
88 */
89 static void updateMarginCall(const boost::shared_ptr<CollateralAccount>& collat, const Real& uncollatValue,
90 const Date& simulationDate, const Real& accrualFactor,
91 const CalculationType& calcType = Symmetric, const bool& eligMarginReqDateUs = true,
92 const bool& eligMarginReqDateCtp = true);
93
94 /*!
95 Computes the Credit Support Amount for the portfolio, given an unsecured exposure as input
96 All calculations done in CSA currency
97 */
98 static Real creditSupportAmount(
99 const boost::shared_ptr<ore::data::NettingSetDefinition>& nettingSet,
100 const Real& uncollatValueCsaCur);
101
102 /*!
103 Takes a netting set (and scenario exposures) as input
104 and returns collateral balance paths per scenario
105 */
106 static boost::shared_ptr<vector<boost::shared_ptr<CollateralAccount>>> collateralBalancePaths(
107 const boost::shared_ptr<NettingSetDefinition>& csaDef, const Real& nettingSetPv, const Date& date_t0,
108 const vector<vector<Real>>& nettingSetValues, const Date& nettingSet_maturity, const vector<Date>& dateGrid,
109 const Real& csaFxTodayRate, const vector<vector<Real>>& csaFxScenarioRates, const Real& csaTodayCollatCurve,
110 const vector<vector<Real>>& csaScenCollatCurves, const CalculationType& calcType = Symmetric);
111};
