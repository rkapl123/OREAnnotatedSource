50class TestMarket : public ore::data::MarketImpl {
51public:
52 TestMarket(Date asof);
53
54private:
55 Handle<YieldTermStructure> flatRateYts(Real forward) {
56 boost::shared_ptr<YieldTermStructure> yts(
57 new FlatForward(Settings::instance().evaluationDate(), forward, ActualActual(ActualActual::ISDA)));
58 return Handle<YieldTermStructure>(yts);
59 }
60 Handle<BlackVolTermStructure> flatRateFxv(Volatility forward) {
61 boost::shared_ptr<BlackVolTermStructure> fxv(
62 new BlackConstantVol(Settings::instance().evaluationDate(), NullCalendar(), forward, ActualActual(ActualActual::ISDA)));
63 return Handle<BlackVolTermStructure>(fxv);
64 }
65 Handle<YieldTermStructure> flatRateDiv(Real dividend) {
66 boost::shared_ptr<YieldTermStructure> yts(
67 new FlatForward(Settings::instance().evaluationDate(), dividend, ActualActual(ActualActual::ISDA)));
68 return Handle<YieldTermStructure>(yts);
69 }
70 Handle<QuantLib::SwaptionVolatilityStructure>
71 flatRateSvs(Volatility forward, VolatilityType type = ShiftedLognormal, Real shift = 0.0) {
72 boost::shared_ptr<QuantLib::SwaptionVolatilityStructure> svs(
73 new QuantLib::ConstantSwaptionVolatility(Settings::instance().evaluationDate(), NullCalendar(),
74 ModifiedFollowing, forward, ActualActual(ActualActual::ISDA), type, shift));
75 return Handle<QuantLib::SwaptionVolatilityStructure>(svs);
76 }
77 Handle<QuantExt::CreditCurve> flatRateDcs(Volatility forward) {
78 boost::shared_ptr<DefaultProbabilityTermStructure> dcs(
79 new FlatHazardRate(asof_, forward, ActualActual(ActualActual::ISDA)));
80 return Handle<QuantExt::CreditCurve>(
81 boost::make_shared<QuantExt::CreditCurve>(Handle<DefaultProbabilityTermStructure>(dcs)));
82 }
83 Handle<OptionletVolatilityStructure> flatRateCvs(Volatility vol, VolatilityType type = Normal, Real shift = 0.0) {
84 boost::shared_ptr<OptionletVolatilityStructure> ts(
85 new QuantLib::ConstantOptionletVolatility(Settings::instance().evaluationDate(), NullCalendar(),
86 ModifiedFollowing, vol, ActualActual(ActualActual::ISDA), type, shift));
87 return Handle<OptionletVolatilityStructure>(ts);
88 }
89 Handle<QuantExt::CorrelationTermStructure> flatCorrelation(Real correlation = 0.0) {
90 boost::shared_ptr<QuantExt::CorrelationTermStructure> ts(
91 new QuantExt::FlatCorrelation(Settings::instance().evaluationDate(), correlation, ActualActual(ActualActual::ISDA)));
92 return Handle<QuantExt::CorrelationTermStructure>(ts);
93 }
94 Handle<CPICapFloorTermPriceSurface> flatRateCps(Handle<ZeroInflationIndex> infIndex,
95 const std::vector<Rate> cStrikes, std::vector<Rate> fStrikes,
96 std::vector<Period> cfMaturities, Matrix cPrice, Matrix fPrice) {
97 boost::shared_ptr<CPICapFloorTermPriceSurface> ts(
98 new InterpolatedCPICapFloorTermPriceSurface<QuantLib::Bilinear>(
99 1.0, 0.0, infIndex->availabilityLag(), infIndex->zeroInflationTermStructure()->calendar(), Following,
100 ActualActual(ActualActual::ISDA), infIndex.currentLink(), CPI::AsIndex, discountCurve(infIndex->currency().code()), cStrikes, fStrikes, cfMaturities,
101 cPrice, fPrice));
102 return Handle<CPICapFloorTermPriceSurface>(ts);
103 }
104 Handle<QuantLib::CPIVolatilitySurface> flatCpiVolSurface(Volatility v) {
105 Natural settleDays = 0;
106 Calendar cal = TARGET();
107 BusinessDayConvention bdc = Following;
108 DayCounter dc = Actual365Fixed();
109 Period lag = 2 * Months;
110 Frequency freq = Annual;
111 bool interp = false;
112 boost::shared_ptr<ConstantCPIVolatility> cpiCapFloorVolSurface =
113 boost::make_shared<ConstantCPIVolatility>(v, settleDays, cal, bdc, dc, lag, freq, interp);
114
115 return Handle<QuantLib::CPIVolatilitySurface>(cpiCapFloorVolSurface);
116 }
117 Handle<ZeroInflationIndex> makeZeroInflationIndex(string index, vector<Date> dates, vector<Rate> rates,
118 boost::shared_ptr<ZeroInflationIndex> ii,
119 Handle<YieldTermStructure> yts);
120 Handle<YoYInflationIndex> makeYoYInflationIndex(string index, vector<Date> dates, vector<Rate> rates,
121 boost::shared_ptr<YoYInflationIndex> ii,
122 Handle<YieldTermStructure> yts);
123};
