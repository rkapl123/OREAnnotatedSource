35void SimmResults::add(const SimmConfiguration::ProductClass& pc, const SimmConfiguration::RiskClass& rc,
36 const SimmConfiguration::MarginType& mt, const string& b, Real im,
37 const string& calculationCurrency, const bool overwrite) {
38
39 // Add the value as long as the currencies are matching. If the SimmResults container does not yet have
40 // a currency, we set it to be that of the incoming value
41 if (ccy_.empty())
42 ccy_ = calculationCurrency;
43 else
44 QL_REQUIRE(calculationCurrency == ccy_, "Cannot add value to SimmResults in a different currency ("
45 << calculationCurrency << "). Expected " << ccy_ << ".");
46
47 QL_REQUIRE(im >= 0.0, "Cannot add negative IM " << im << " result to SimmResults for RiskClass=" << rc
48 << ", MarginType=" << mt << ", and Bucket=" << b);
49
50 const auto key = make_tuple(pc, rc, mt, b);
51 const bool hasResults = data_.find(key) != data_.end();
52 if (hasResults && !overwrite)
53 data_[key] += im;
54 else
55 data_[key] = im;
56}

58void SimmResults::convert(const boost::shared_ptr<ore::data::Market>& market, const string& currency) {
59 // Get corresponding FX spot rate
60 Real fxSpot = market->fxRate(ccy_ + currency)->value();
61
62 convert(fxSpot, currency);
63}

65void SimmResults::convert(Real fxSpot, const string& currency) {
66 // Check that target currency is valid
67 QL_REQUIRE(ore::data::checkCurrency(currency), "Cannot convert SIMM results. The target currency ("
68 << currency << ") must be a valid ISO currency code");
69
70 // Skip if already in target currency
71 if (ccy_ == currency)
72 return;
73
74 // Convert SIMM results to target currency
75 for (auto& sr : data_)
76 sr.second *= fxSpot;
77
78 // Update currency
79 ccy_ = currency;
80}

82Real SimmResults::get(const SimmConfiguration::ProductClass& pc, const SimmConfiguration::RiskClass& rc,
83 const SimmConfiguration::MarginType& mt, const string b) const {
84 if (has(pc, rc, mt, b)) {
85 return data_.at(make_tuple(pc, rc, mt, b));

87 return Null<Real>();
