238//! ParSensitivityConverter class
239/*!
240 1) Build Jacobi matrix containing sensitivities of par rates (first index) w.r.t. zero shifts (second index)
241 2) Convert zero rate and optionlet vol sensitivities into par rate/vol sensitivities
242
243 Let: p_ij denote curve i's par instrument j (curve may be a discount or an index curve)
244 z_kl denote curve k's zero shift l (curve may be a discount or an index curve)
245 i,k run from 0 to I
246 j,l run from 0 to J
247
248 Organisation of the matrix:
249 z_00 z_01 z_02 ... z_0J ... z_10 z_11 ... z_1J ... z_I0 z_I1 ... z_IJ
250 p_00
251 p_01
252 p_02
253 ...
254 p_0J
255 p_10
256 p_11
257 ...
258 p_1J
259 ...
260 p_I0
261 p_I1
262 ...
263 p_IJ
264
265 The curve numbering starts with discount curves (by ccy), followed by index curves (by index name).
266 The number J of par instruments respectively zero shifts can differ between discount and index curves.
267 The number of zero shifts matches the number of par instruments.
268 The Jacobi matrix is therefore quadratic by construction.
269 */
270class ParSensitivityConverter {
271public:
272 /*! Constructor where \p parSensitivities is the par rate sensitivities w.r.t. zero shifts \p shiftSizes gives
273 the absolute zero and par shift sizes for each risk factor key.
274 */
275 ParSensitivityConverter(const ParSensitivityAnalysis::ParContainer& parSensitivities,
276 const std::map<ore::analytics::RiskFactorKey, std::pair<QuantLib::Real, QuantLib::Real>>& shiftSizes);
277
278 //! Inspectors
279 //@{
280 //! Return the set of raw, i.e. zero, risk factor keys
281 //! The ordering in this set defines the order of the columns in the Jacobi matrix
282 const std::set<ore::analytics::RiskFactorKey>& rawKeys() { return rawKeys_; }
283 //! Return the set of par risk factor keys
284 //! The ordering in this set defines the order of the rows in the Jacobi matrix
285 const std::set<ore::analytics::RiskFactorKey>& parKeys() { return parKeys_; }
286 //@}
287
288 //! Takes an array of zero sensitivities and returns an array of par sensitivities
289 /*! \param zeroSensitivities array of zero sensitivities ordered according to rawKeys()
290
291 \return array of par sensitivities ordered according to parKeys()
292
293 \todo should we use a compressed_vector for the input / output parameter or both ?
294 */
295 boost::numeric::ublas::vector<Real>
296 convertSensitivity(const boost::numeric::ublas::vector<Real>& zeroSensitivities);
297
298 //! Write the inverse of the transposed Jacobian to the \p reportOut
299 void writeConversionMatrix(ore::data::Report& reportOut) const;
300
301private:
302 std::set<ore::analytics::RiskFactorKey> rawKeys_;
303 std::set<ore::analytics::RiskFactorKey> parKeys_;
304 // transposed inverse Jacobian, i.e. the matrix we use for the zero-par conversion effectively
305 QuantLib::SparseMatrix jacobi_transp_inv_;
306 //! Vector of absolute zero shift sizes
307 boost::numeric::ublas::vector<QuantLib::Real> zeroShifts_;
308 //! Vector of absolute par shift sizes
309 boost::numeric::ublas::vector<QuantLib::Real> parShifts_;
310};
