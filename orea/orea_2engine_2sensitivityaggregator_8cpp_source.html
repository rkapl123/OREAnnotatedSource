52 void SensitivityAggregator::aggregate(SensitivityStream& ss, const boost::shared_ptr<ScenarioFilter>& filter) {
53  // Ensure at start of stream
54  ss.reset();
55 
56  // Loop over stream's records
57  while (SensitivityRecord sr = ss.next()) {
58  // Skip this record if the risk factor is not in the filter
59  if (!sr.isCrossGamma() && !filter->allow(sr.key_1))
60  continue;
61  if (sr.isCrossGamma() && (!filter->allow(sr.key_1) || !filter->allow(sr.key_2)))
62  continue;
63 
64  // "Blank out" trade ID before adding
65  string tradeId = sr.tradeId;
66  sr.tradeId = "";
67 
68  // Update aggRecords_ for each category
69  for (const auto& kv : categories_) {
70  // Check if the sensitivity record's trade ID is in the category
71  if (kv.second(tradeId)) {
72  DLOG("Updating aggregated sensitivities for category " << kv.first << " with record: " << sr);
73  add(sr, aggRecords_[kv.first]);
74  }
75  }
76  }
77 }

103 void SensitivityAggregator::add(SensitivityRecord& sr, set<SensitivityRecord>& records) {
104  // Try to insert sr. This will only pass if sr is not there already.
105  auto p = records.insert(sr);
106  if (!p.second) {
107  // If sr is already in the set, update it.
108  p.first->baseNpv += sr.baseNpv;
109  p.first->delta += sr.delta;
110  p.first->gamma += sr.gamma;
111  }

114 bool SensitivityAggregator::inCategory(const string& tradeId, const string& category) const {
115  QL_REQUIRE(setCategories_.count(category), "The category " << category << " is not valid");
116  auto tradeIds = setCategories_.at(category);
117  for (auto it = tradeIds.begin(); it != tradeIds.end(); ++it) {
118  if (it->first == tradeId)
119  return true;
120  }
121  return false;
122 }
