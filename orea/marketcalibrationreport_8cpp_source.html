27 auto pos = spec.rfind("/");
28 if (pos != std::string::npos) {
29 return spec.substr(pos + 1);
30 } else {
31 return spec;

43void MarketCalibrationReport::populateReport(const boost::shared_ptr<ore::data::Market>& market,
44 const boost::shared_ptr<TodaysMarketParameters>& todaysMarketParams,
45 const std::string& label) {
46 initialise(label);
47 if (market == nullptr)
48 return;
49 auto t = boost::dynamic_pointer_cast<TodaysMarket>(market);
50 if (!t) {
51 DLOG("MarketCalibrationReport::populateReport() expected TodaysMarket");
52 return;
53 }
54
55 auto calibrationInfo = t->calibrationInfo();
56 if (calibrationFilters_.mdFilterCurves) {
57 // First cut at adding curves
58
59 // TODO simplify that and just loop over yield and dividend curve calibration info, the only change
60 // would be that we would not be able to set the discountCurve flag any more, not sure if that is very
61 // important? add to curves here Add discount curves first, so EUR-EONIA gets marked as a discount curve
62 // even if it is an IndexCurve too
63 if (todaysMarketParams->hasMarketObject(MarketObject::DiscountCurve)) {
64 for (auto it : todaysMarketParams->mapping(MarketObject::DiscountCurve, Market::defaultConfiguration)) {
65 auto yts = calibrationInfo->yieldCurveCalibrationInfo.find(it.second);
66 if (yts != calibrationInfo->yieldCurveCalibrationInfo.end())
67 addYieldCurve(calibrationInfo->asof, yts->second, getCurveName(it.second), true, label);
68 }
69 }
70 if (todaysMarketParams->hasMarketObject(MarketObject::YieldCurve)) {
71 for (auto it : todaysMarketParams->mapping(MarketObject::YieldCurve, Market::defaultConfiguration)) {
72 auto yts = calibrationInfo->yieldCurveCalibrationInfo.find(it.second);
73 if (yts != calibrationInfo->yieldCurveCalibrationInfo.end())
74 addYieldCurve(calibrationInfo->asof, yts->second, getCurveName(it.second), false, label);
75 }
76 }
77 if (todaysMarketParams->hasMarketObject(MarketObject::EquityCurve)) {
78 for (auto it : todaysMarketParams->mapping(MarketObject::EquityCurve, Market::defaultConfiguration)) {
79 auto yts = calibrationInfo->dividendCurveCalibrationInfo.find(it.second);
80 if (yts != calibrationInfo->dividendCurveCalibrationInfo.end())
81 addYieldCurve(calibrationInfo->asof, yts->second, getCurveName(it.second), false, label);
82 }
83 }
84 if (todaysMarketParams->hasMarketObject(MarketObject::IndexCurve)) {
85 for (auto it : todaysMarketParams->mapping(MarketObject::IndexCurve, Market::defaultConfiguration)) {
86 auto yts = calibrationInfo->yieldCurveCalibrationInfo.find(it.second);
87 if (yts != calibrationInfo->yieldCurveCalibrationInfo.end())
88 addYieldCurve(calibrationInfo->asof, yts->second, getCurveName(it.second), false, label);
89 }
90 }
91 }
92
93 if (calibrationFilters_.mdFilterInfCurves) {
94 // inflation curves
95 for (auto const& c : calibrationInfo->inflationCurveCalibrationInfo) {
96 addInflationCurve(calibrationInfo->asof, c.second, getCurveName(c.first), label);
97 }
98 }
99
100 if (calibrationFilters_.mdFilterCommCurves) {
101 // commodity curves
102 for (auto const& c : calibrationInfo->commodityCurveCalibrationInfo) {
103 addCommodityCurve(calibrationInfo->asof, c.second, getCurveName(c.first), label);
104 }
105 }
106
107 if (calibrationFilters_.mdFilterFxVols) {
108 // fx vol surfaces
109 for (auto const& c : calibrationInfo->fxVolCalibrationInfo) {
110 addFxVol(calibrationInfo->asof, c.second, getCurveName(c.first), label);
111 }
112 }
113
114 if (calibrationFilters_.mdFilterEqVols) {
115 // eq vol surfaces
116 for (auto const& c : calibrationInfo->eqVolCalibrationInfo) {
117 addEqVol(calibrationInfo->asof, c.second, getCurveName(c.first), label);
118 }
119 }
120
121 if (calibrationFilters_.mdFilterCommVols){
122 // commodity vols
123 for (auto const& c: calibrationInfo->commVolCalibrationInfo){
124 addCommVol(calibrationInfo->asof, c.second, getCurveName(c.first), label);
125 }
126 }
127
128 if (calibrationFilters_.mdFilterIrVols) {
129 // ir vol surfaces
130 for (auto const& c : calibrationInfo->irVolCalibrationInfo) {
131 addIrVol(calibrationInfo->asof, c.second, c.first, label);
132 }
133 }
134}
