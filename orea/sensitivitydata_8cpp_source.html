46 boost::shared_ptr<RiskFactorKey> parseRiskFactorKey(const std::string& str, std::vector<std::string>& addTokens) {
47  std::vector<std::string> tokens;
48  boost::split(tokens, str, boost::is_any_of("/"), boost::token_compress_off);
49  QL_REQUIRE(tokens.size() >= 3, "parseRiskFactorKey: at least 3 tokens required, string is \"" << str << "\"");
50  RiskFactorKey::KeyType type = parseRiskFactorKeyType(tokens[0]);
51  boost::shared_ptr<RiskFactorKey> key =
52  boost::make_shared<RiskFactorKey>(type, tokens[1], ore::data::parseInteger(tokens[2]));
53  for (Size i = 3; i < tokens.size(); ++i) {
54  addTokens.emplace_back(tokens[i]);
55  }
56  return key;
57 } // parseRiskFactorKey

110 void loadSensitivityDataFromCsv(SensitivityDataInMemory& data, const std::string& fileName, const char delim) {
111  LOG("Load Sensitivity Data from file " << fileName);
112  ore::data::CSVFileReader reader(fileName, true);
113  bool delta = reader.hasField("#TradeId") && reader.hasField("Factor") &&
114  (reader.hasField("Delta") || reader.hasField("ParDelta"));
115  bool gamma = reader.hasField("#TradeId") && reader.hasField("Factor 1") && reader.hasField("Factor 2") &&
116  (reader.hasField("CrossGamma") || reader.hasField("ParCrossGamma"));
117  QL_REQUIRE(delta || gamma,
118  "loadSensitivityDataFromCsv: file " << fileName << " not recognised as either delta or crossgamma file");
119  std::string valueField, valueField2;
120  if (delta) {
121  valueField = reader.hasField("ParDelta") ? "ParDelta" : "Delta";
122  valueField2 = reader.hasField("ParGamma") ? "ParGamma" : "Gamma";
123  } else
124  valueField = reader.hasField("ParCrossGamma") ? "ParCrossGamma" : "CrossGamma";
125  Size errorLines = 0, validLines = 0;
126  while (reader.next()) {
127  try {
128  if (delta) {
129  data.add(reader.get("#TradeId"), reader.get("Factor"), "", ore::data::parseReal(reader.get(valueField)),
130  ore::data::parseReal(reader.get(valueField2)));
131  } else if (gamma) {
132  data.add(reader.get("#TradeId"), reader.get("Factor 1"), reader.get("Factor 2"),
133  ore::data::parseReal(reader.get(valueField)), Null<Real>());
134  }
135  ++validLines;
136  } catch (const std::exception& e) {
137  ++errorLines;
138  WLOG("skipping line " << reader.currentLine() << ": " << e.what());
139  }
140  }
141  LOG("Read " << validLines << " valid data lines, skipped " << errorLines << " invalid data lines in file "
142  << fileName);
143 } // loadSensitivityDataFromCsv
