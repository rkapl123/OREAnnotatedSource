271 void ValueAdjustmentCalculator::build() {
272  const auto& numDates = dates().size();
273  const auto& today = asof();
274 
275  Handle<YieldTermStructure> borrowingCurve, lendingCurve, oisCurve;
276  if (baseCurrency_ != "")
277  oisCurve = market_->discountCurve(baseCurrency_, configuration_);
278 
279  string origDvaName = dvaName_;
280  // Trade XVA
281  for (Size i = 0; i < portfolio_->trades().size(); ++i) {
282  string tid = portfolio_->trades()[i]->id();
283  LOG("Update XVA for trade " << tid << (flipViewXVA_ ? ", inverted (flipViewXVA = Y)" : ", regular (flipViewXVA = N)"));
284  string cid;
285  if (flipViewXVA_) {
286  cid = origDvaName;
287  dvaName_ = portfolio_->trades()[i]->envelope().counterparty();
288  fvaBorrowingCurve_ = dvaName_ + flipViewBorrowingCurvePostfix_;
289  fvaLendingCurve_ = dvaName_ + flipViewLendingCurvePostfix_;
290  } else {
291  cid = portfolio_->trades()[i]->envelope().counterparty();
292  }
293  if (fvaBorrowingCurve_ != "")
294  borrowingCurve = market_->yieldCurve(fvaBorrowingCurve_, configuration_);
295  if (fvaLendingCurve_ != "")
296  lendingCurve = market_->yieldCurve(fvaLendingCurve_, configuration_);
297 
298  if (!borrowingCurve.empty() || !lendingCurve.empty()) {
299  QL_REQUIRE(baseCurrency_ != "", "baseCurrency required for FVA calculation");
300  }
301  string nid = portfolio_->trades()[i]->envelope().nettingSetId();
302  Real cvaRR = market_->recoveryRate(cid, configuration_)->value();
303  Real dvaRR = 0.0;
304  if (dvaName_ != "")
305  dvaRR = market_->recoveryRate(dvaName_, configuration_)->value();
306 
307  tradeCva_[tid] = 0.0;
308  tradeDva_[tid] = 0.0;
309  tradeFca_[tid] = 0.0;
310  tradeFca_exOwnSp_[tid] = 0.0;
311  tradeFca_exAllSp_[tid] = 0.0;
312  tradeFba_[tid] = 0.0;
313  tradeFba_exOwnSp_[tid] = 0.0;
314  tradeFba_exAllSp_[tid] = 0.0;
315  tradeMva_[tid] = 0.0;
316  for (Size j = 0; j < numDates; ++j) {
317 
318  // CVA / DVA
319  Date d0 = j == 0 ? today : dates()[j - 1];
320  Date d1 = dates()[j];
321  Real cvaIncrement = calculateCvaIncrement(tid, cid, d0, d1, cvaRR);
322  Real dvaIncrement = dvaName_ != "" ? calculateDvaIncrement(tid, d0, d1, dvaRR) : 0;
323  tradeCva_[tid] += cvaIncrement;
324  tradeDva_[tid] += dvaIncrement;
325 
326  // FCA
327  if (!borrowingCurve.empty()) {
328  Real dcf = borrowingCurve->discount(d0) / borrowingCurve->discount(d1) -
329  oisCurve->discount(d0) / oisCurve->discount(d1);
330  Real fcaIncrement = calculateFcaIncrement(tid, cid, dvaName_, d0, d1, dcf);
331  Real fcaIncrement_exOwnSP = calculateFcaIncrement(tid, cid, "", d0, d1, dcf);
332  Real fcaIncrement_exAllSP = calculateFcaIncrement(tid, "", "", d0, d1, dcf);
333  tradeFca_[tid] += fcaIncrement;
334  tradeFca_exOwnSp_[tid] += fcaIncrement_exOwnSP;
335  tradeFca_exAllSp_[tid] += fcaIncrement_exAllSP;
336 
337  }
338 
339  // FBA
340  if (!lendingCurve.empty()) {
341  Real dcf = lendingCurve->discount(d0) / lendingCurve->discount(d1) -
342  oisCurve->discount(d0) / oisCurve->discount(d1);
343  Real fbaIncrement = calculateFbaIncrement(tid, cid, dvaName_, d0, d1, dcf);
344  Real fbaIncrement_exOwnSP = calculateFbaIncrement(tid, cid, "", d0, d1, dcf);
345  Real fbaIncrement_exAllSP = calculateFbaIncrement(tid, "", "", d0, d1, dcf);
346  tradeFba_[tid] += fbaIncrement;
347  tradeFba_exOwnSp_[tid] += fbaIncrement_exOwnSP;
348  tradeFba_exAllSp_[tid] += fbaIncrement_exAllSP;
349  }
350  }
351  if (nettingSetSumCva_.find(nid) == nettingSetSumCva_.end()) {
352  nettingSetSumCva_[nid] = 0.0;
353  nettingSetSumDva_[nid] = 0.0;
354  }
355  nettingSetSumCva_[nid] += tradeCva_[tid];
356  nettingSetSumDva_[nid] += tradeDva_[tid];
357  }
358 
359  // Netting Set XVA
360  for (const auto& pair : nettingSetCpty_) {
361  string nid = pair.first;
362  LOG("Update XVA for netting set " << nid << (flipViewXVA_ ? ", inverted (flipViewXVA = Y)" : ", regular (flipViewXVA = N)"));
363  string cid;
364  if (flipViewXVA_) {
365  cid = origDvaName;
366  dvaName_ = pair.second;
367  fvaBorrowingCurve_ = dvaName_ + flipViewBorrowingCurvePostfix_;
368  fvaLendingCurve_ = dvaName_ + flipViewLendingCurvePostfix_;
369  } else {
370  cid = pair.second;
371  }
372  Real cvaRR = market_->recoveryRate(cid, configuration_)->value();
373  Real dvaRR = 0.0;
374  if (dvaName_ != "") {
375  dvaRR = market_->recoveryRate(dvaName_, configuration_)->value();
376  }
377  if (fvaBorrowingCurve_ != "")
378  borrowingCurve = market_->yieldCurve(fvaBorrowingCurve_, configuration_);
379  if (fvaLendingCurve_ != "")
380  lendingCurve = market_->yieldCurve(fvaLendingCurve_, configuration_);
381 
382  if (!borrowingCurve.empty() || !lendingCurve.empty()) {
383  QL_REQUIRE(baseCurrency_ != "", "baseCurrency required for FVA calculation");
384  }
385 
386  nettingSetCva_[nid] = 0.0;
387  nettingSetDva_[nid] = 0.0;
388  nettingSetFca_[nid] = 0.0;
389  nettingSetFca_exOwnSp_[nid] = 0.0;
390  nettingSetFca_exAllSp_[nid] = 0.0;
391  nettingSetFba_[nid] = 0.0;
392  nettingSetFba_exOwnSp_[nid] = 0.0;
393  nettingSetFba_exAllSp_[nid] = 0.0;
394  nettingSetMva_[nid] = 0.0;
395  for (Size j = 0; j < numDates; ++j) {
396  // CVA / DVA
397  Date d0 = j == 0 ? today : dates()[j - 1];
398  Date d1 = dates()[j];
399  Real cvaIncrement = calculateNettingSetCvaIncrement(nid, cid, d0, d1, cvaRR);
400  Real dvaIncrement = dvaName_ != "" ? calculateNettingSetDvaIncrement(nid, d0, d1, dvaRR) : 0;
401  nettingSetCva_[nid] += cvaIncrement;
402  nettingSetDva_[nid] += dvaIncrement;
403 
404  // FCA
405  if (!borrowingCurve.empty()) {
406  Real dcf = borrowingCurve->discount(d0) / borrowingCurve->discount(d1) -
407  oisCurve->discount(d0) / oisCurve->discount(d1);
408  Real fcaIncrement = calculateNettingSetFcaIncrement(nid, cid, dvaName_, d0, d1, dcf);
409  Real fcaIncrement_exOwnSP = calculateNettingSetFcaIncrement(nid, cid, "", d0, d1, dcf);
410  Real fcaIncrement_exAllSP = calculateNettingSetFcaIncrement(nid, "", "", d0, d1, dcf);
411  nettingSetFca_[nid] += fcaIncrement;
412  nettingSetFca_exOwnSp_[nid] += fcaIncrement_exOwnSP;
413  nettingSetFca_exAllSp_[nid] += fcaIncrement_exAllSP;
414 
415  // MVA
416  if (dimCalculator_) {
417  Real mvaIncrement = calculateNettingSetMvaIncrement(nid, cid, d0, d1, dcf);
418  nettingSetMva_[nid] += mvaIncrement;
419  }
420  }
421 
422  // FBA
423  if (!lendingCurve.empty()) {
424  Real dcf = lendingCurve->discount(d0) / lendingCurve->discount(d1) -
425  oisCurve->discount(d0) / oisCurve->discount(d1);
426  Real fbaIncrement = calculateNettingSetFbaIncrement(nid, cid, dvaName_, d0, d1, dcf);
427  Real fbaIncrement_exOwnSP = calculateNettingSetFbaIncrement(nid, cid, "", d0, d1, dcf);
428  Real fbaIncrement_exAllSP = calculateNettingSetFbaIncrement(nid, "", "", d0, d1, dcf);
429  nettingSetFba_[nid] += fbaIncrement;
430  nettingSetFba_exOwnSp_[nid] += fbaIncrement_exOwnSP;
431  nettingSetFba_exAllSp_[nid] += fbaIncrement_exAllSP;
432  }
433  }
434  }
