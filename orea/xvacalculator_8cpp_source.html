36ValueAdjustmentCalculator::ValueAdjustmentCalculator(
37 const boost::shared_ptr<Portfolio> portfolio,
38 const boost::shared_ptr<Market> market,
39 const string& configuration,
40 const string& baseCurrency,
41 const string& dvaName,
42 const string& fvaBorrowingCurve,
43 const string& fvaLendingCurve,
44 const bool applyDynamicInitialMargin,
45 const boost::shared_ptr<DynamicInitialMarginCalculator> dimCalculator,
46 const boost::shared_ptr<NPVCube> tradeExposureCube,
47 const boost::shared_ptr<NPVCube> nettingSetExposureCube,
48 const Size tradeEpeIndex, const Size tradeEneIndex,
49 const Size nettingSetEpeIndex, const Size nettingSetEneIndex,
50 const bool flipViewXVA, const string& flipViewBorrowingCurvePostfix, const string& flipViewLendingCurvePostfix)
51 : portfolio_(portfolio), market_(market), configuration_(configuration),
52 baseCurrency_(baseCurrency), dvaName_(dvaName),
53 fvaBorrowingCurve_(fvaBorrowingCurve), fvaLendingCurve_(fvaLendingCurve),
54 applyDynamicInitialMargin_(applyDynamicInitialMargin),
55 dimCalculator_(dimCalculator),
56 tradeExposureCube_(tradeExposureCube),
57 nettingSetExposureCube_(nettingSetExposureCube),
58 tradeEpeIndex_(tradeEpeIndex), tradeEneIndex_(tradeEneIndex), nettingSetEpeIndex_(nettingSetEpeIndex),
59 nettingSetEneIndex_(nettingSetEneIndex), flipViewXVA_(flipViewXVA),
60 flipViewBorrowingCurvePostfix_(flipViewBorrowingCurvePostfix), flipViewLendingCurvePostfix_(flipViewLendingCurvePostfix) {
61
62 QL_REQUIRE(portfolio_, "portfolio is null");
63
64 for (const auto& [tradeId, trade] : portfolio_->trades()) {
65 string nettingSetId = trade->envelope().nettingSetId();
66 if (nettingSetCpty_.find(nettingSetId) == nettingSetCpty_.end()) {
67 nettingSetCpty_[nettingSetId] = trade->envelope().counterparty();
68 }
69 }
70
71
72
73 QL_REQUIRE(tradeExposureCube_->numIds() == portfolio_->trades().size(),
74 "number of trades in tradeExposureCube and portfolio mismatch ("
75 << tradeExposureCube_->numIds() << " vs " << portfolio_->trades().size() << ")");
76
77 QL_REQUIRE(nettingSetExposureCube_->numIds() == nettingSetCpty_.size(),
78 "number of netting sets in nettingSetExposureCube and nettingSetCpty map mismatch ("
79 << nettingSetExposureCube_->numIds() << " vs " << nettingSetCpty_.size() << ")");
80
81 QL_REQUIRE(tradeExposureCube_->numDates() == nettingSetExposureCube_->numDates(),
82 "number of dates in tradeExposureCube and nettingSetExposureCube mismatch ("
83 << tradeExposureCube_->numDates() << " vs " << nettingSetExposureCube_->numDates() << ")");
84
85 for (Size i = 0; i < tradeExposureCube_->numDates(); i++) {
86 QL_REQUIRE(tradeExposureCube_->dates()[i] == nettingSetExposureCube_->dates()[i],
87 "date at " << i << " in tradeExposureCube and nettingSetExposureCube mismatch ("
88 << tradeExposureCube_->dates()[i] << " vs " << nettingSetExposureCube_->dates()[i] << ")");
89 }
90
91 QL_REQUIRE(tradeEpeIndex < tradeExposureCube_->depth(), "tradeEpeIndex("
92 << tradeEpeIndex << ") exceeds depth of tradeExposureCube("
93 << tradeExposureCube_->depth() << ")");
94
95 QL_REQUIRE(tradeEneIndex < tradeExposureCube_->depth(), "tradeEneIndex("
96 << tradeEneIndex << ") exceeds depth of tradeExposureCube("
97 << tradeExposureCube_->depth() << ")");
98
99 QL_REQUIRE(nettingSetEpeIndex < nettingSetExposureCube_->depth(), "nettingSetEpeIndex("
100 << nettingSetEpeIndex << ") exceeds depth of nettingSetExposureCube("
101 << nettingSetExposureCube_->depth() << ")");
102
103 QL_REQUIRE(nettingSetEneIndex < nettingSetExposureCube_->depth(), "nettingSetEneIndex("
104 << nettingSetEneIndex << ") exceeds depth of nettingSetExposureCube("
105 << nettingSetExposureCube_->depth() << ")");
106
107}

273void ValueAdjustmentCalculator::build() {
274 const auto& numDates = dates().size();
275 const auto& today = asof();
276
277 Handle<YieldTermStructure> borrowingCurve, lendingCurve, oisCurve;
278 if (baseCurrency_ != "")
279 oisCurve = market_->discountCurve(baseCurrency_, configuration_);
280
281 string origDvaName = dvaName_;
282 // Trade XVA
283 for (const auto& [tid, trade]: portfolio_->trades()) {
284 LOG("Update XVA for trade " << tid << (flipViewXVA_ ? ", inverted (flipViewXVA = Y)" : ", regular (flipViewXVA = N)"));
285 string cid;
286 if (flipViewXVA_) {
287 cid = origDvaName;
288 dvaName_ = trade->envelope().counterparty();
289 fvaBorrowingCurve_ = dvaName_ + flipViewBorrowingCurvePostfix_;
290 fvaLendingCurve_ = dvaName_ + flipViewLendingCurvePostfix_;
291 } else {
292 cid = trade->envelope().counterparty();
293 }
294 if (fvaBorrowingCurve_ != "")
295 borrowingCurve = market_->yieldCurve(fvaBorrowingCurve_, configuration_);
296 if (fvaLendingCurve_ != "")
297 lendingCurve = market_->yieldCurve(fvaLendingCurve_, configuration_);
298
299 if (!borrowingCurve.empty() || !lendingCurve.empty()) {
300 QL_REQUIRE(baseCurrency_ != "", "baseCurrency required for FVA calculation");
301 }
302 string nid = trade->envelope().nettingSetId();
303 Real cvaRR = market_->recoveryRate(cid, configuration_)->value();
304 Real dvaRR = 0.0;
305 if (dvaName_ != "")
306 dvaRR = market_->recoveryRate(dvaName_, configuration_)->value();
307
308 tradeCva_[tid] = 0.0;
309 tradeDva_[tid] = 0.0;
310 tradeFca_[tid] = 0.0;
311 tradeFca_exOwnSp_[tid] = 0.0;
312 tradeFca_exAllSp_[tid] = 0.0;
313 tradeFba_[tid] = 0.0;
314 tradeFba_exOwnSp_[tid] = 0.0;
315 tradeFba_exAllSp_[tid] = 0.0;
316 tradeMva_[tid] = 0.0;
317 for (Size j = 0; j < numDates; ++j) {
318
319 // CVA / DVA
320 Date d0 = j == 0 ? today : dates()[j - 1];
321 Date d1 = dates()[j];
322 Real cvaIncrement = calculateCvaIncrement(tid, cid, d0, d1, cvaRR);
323 Real dvaIncrement = dvaName_ != "" ? calculateDvaIncrement(tid, d0, d1, dvaRR) : 0;
324 tradeCva_[tid] += cvaIncrement;
325 tradeDva_[tid] += dvaIncrement;
326
327 // FCA
328 if (!borrowingCurve.empty()) {
329 Real dcf = borrowingCurve->discount(d0) / borrowingCurve->discount(d1) -
330 oisCurve->discount(d0) / oisCurve->discount(d1);
331 Real fcaIncrement = calculateFcaIncrement(tid, cid, dvaName_, d0, d1, dcf);
332 Real fcaIncrement_exOwnSP = calculateFcaIncrement(tid, cid, "", d0, d1, dcf);
333 Real fcaIncrement_exAllSP = calculateFcaIncrement(tid, "", "", d0, d1, dcf);
334 tradeFca_[tid] += fcaIncrement;
335 tradeFca_exOwnSp_[tid] += fcaIncrement_exOwnSP;
336 tradeFca_exAllSp_[tid] += fcaIncrement_exAllSP;
337
338 }
339
340 // FBA
341 if (!lendingCurve.empty()) {
342 Real dcf = lendingCurve->discount(d0) / lendingCurve->discount(d1) -
343 oisCurve->discount(d0) / oisCurve->discount(d1);
344 Real fbaIncrement = calculateFbaIncrement(tid, cid, dvaName_, d0, d1, dcf);
345 Real fbaIncrement_exOwnSP = calculateFbaIncrement(tid, cid, "", d0, d1, dcf);
346 Real fbaIncrement_exAllSP = calculateFbaIncrement(tid, "", "", d0, d1, dcf);
347 tradeFba_[tid] += fbaIncrement;
348 tradeFba_exOwnSp_[tid] += fbaIncrement_exOwnSP;
349 tradeFba_exAllSp_[tid] += fbaIncrement_exAllSP;
350 }
351 }
352 if (nettingSetSumCva_.find(nid) == nettingSetSumCva_.end()) {
353 nettingSetSumCva_[nid] = 0.0;
354 nettingSetSumDva_[nid] = 0.0;
355 }
356 nettingSetSumCva_[nid] += tradeCva_[tid];
357 nettingSetSumDva_[nid] += tradeDva_[tid];
358 }
359
360 // Netting Set XVA
361 for (const auto& pair : nettingSetCpty_) {
362 string nid = pair.first;
363 LOG("Update XVA for netting set " << nid << (flipViewXVA_ ? ", inverted (flipViewXVA = Y)" : ", regular (flipViewXVA = N)"));
364 string cid;
365 if (flipViewXVA_) {
366 cid = origDvaName;
367 dvaName_ = pair.second;
368 fvaBorrowingCurve_ = dvaName_ + flipViewBorrowingCurvePostfix_;
369 fvaLendingCurve_ = dvaName_ + flipViewLendingCurvePostfix_;
370 } else {
371 cid = pair.second;
372 }
373 Real cvaRR = market_->recoveryRate(cid, configuration_)->value();
374 Real dvaRR = 0.0;
375 if (dvaName_ != "") {
376 dvaRR = market_->recoveryRate(dvaName_, configuration_)->value();
377 }
378 if (fvaBorrowingCurve_ != "")
379 borrowingCurve = market_->yieldCurve(fvaBorrowingCurve_, configuration_);
380 if (fvaLendingCurve_ != "")
381 lendingCurve = market_->yieldCurve(fvaLendingCurve_, configuration_);
382
383 if (!borrowingCurve.empty() || !lendingCurve.empty()) {
384 QL_REQUIRE(baseCurrency_ != "", "baseCurrency required for FVA calculation");
385 }
386
387 nettingSetCva_[nid] = 0.0;
388 nettingSetDva_[nid] = 0.0;
389 nettingSetFca_[nid] = 0.0;
390 nettingSetFca_exOwnSp_[nid] = 0.0;
391 nettingSetFca_exAllSp_[nid] = 0.0;
392 nettingSetFba_[nid] = 0.0;
393 nettingSetFba_exOwnSp_[nid] = 0.0;
394 nettingSetFba_exAllSp_[nid] = 0.0;
395 nettingSetMva_[nid] = 0.0;
396 for (Size j = 0; j < numDates; ++j) {
397 // CVA / DVA
398 Date d0 = j == 0 ? today : dates()[j - 1];
399 Date d1 = dates()[j];
400 Real cvaIncrement = calculateNettingSetCvaIncrement(nid, cid, d0, d1, cvaRR);
401 Real dvaIncrement = dvaName_ != "" ? calculateNettingSetDvaIncrement(nid, d0, d1, dvaRR) : 0;
402 nettingSetCva_[nid] += cvaIncrement;
403 nettingSetDva_[nid] += dvaIncrement;
404
405 // FCA
406 if (!borrowingCurve.empty()) {
407 Real dcf = borrowingCurve->discount(d0) / borrowingCurve->discount(d1) -
408 oisCurve->discount(d0) / oisCurve->discount(d1);
409 Real fcaIncrement = calculateNettingSetFcaIncrement(nid, cid, dvaName_, d0, d1, dcf);
410 Real fcaIncrement_exOwnSP = calculateNettingSetFcaIncrement(nid, cid, "", d0, d1, dcf);
411 Real fcaIncrement_exAllSP = calculateNettingSetFcaIncrement(nid, "", "", d0, d1, dcf);
412 nettingSetFca_[nid] += fcaIncrement;
413 nettingSetFca_exOwnSp_[nid] += fcaIncrement_exOwnSP;
414 nettingSetFca_exAllSp_[nid] += fcaIncrement_exAllSP;
415
416 // MVA
417 if (dimCalculator_) {
418 Real mvaIncrement = calculateNettingSetMvaIncrement(nid, cid, d0, d1, dcf);
419 nettingSetMva_[nid] += mvaIncrement;
420 }
421 }
422
423 // FBA
424 if (!lendingCurve.empty()) {
425 Real dcf = lendingCurve->discount(d0) / lendingCurve->discount(d1) -
426 oisCurve->discount(d0) / oisCurve->discount(d1);
427 Real fbaIncrement = calculateNettingSetFbaIncrement(nid, cid, dvaName_, d0, d1, dcf);
428 Real fbaIncrement_exOwnSP = calculateNettingSetFbaIncrement(nid, cid, "", d0, d1, dcf);
429 Real fbaIncrement_exAllSP = calculateNettingSetFbaIncrement(nid, "", "", d0, d1, dcf);
430 nettingSetFba_[nid] += fbaIncrement;
431 nettingSetFba_exOwnSp_[nid] += fbaIncrement_exOwnSP;
432 nettingSetFba_exAllSp_[nid] += fbaIncrement_exAllSP;
433 }
434 }
435 }
