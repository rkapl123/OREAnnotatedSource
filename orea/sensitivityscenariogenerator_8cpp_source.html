286 void SensitivityScenarioGenerator::generateFxScenarios(bool up) {
287  Date asof = baseScenario_->asof();
288  // We can choose to shift fewer FX risk factors than listed in the market
289  // Is this too strict?
290  // - implemented to avoid cases where input cross FX rates are not consistent
291  // - Consider an example (baseCcy = EUR) of a GBPUSD FX trade - two separate routes to pricing
292  // - (a) call GBPUSD FX rate from sim market
293  // - (b) call GBPEUR and EURUSD FX rates, manually join them to obtain GBPUSD
294  // - now, if GBPUSD is an explicit risk factor in sim market, consider what happens
295  // - if we bump GBPUSD value and leave other FX rates unchanged (for e.g. a sensitivity analysis)
296  // - (a) the value of the trade changes
297  // - (b) the value of the GBPUSD trade stays the same
298  // - in light of the above we restrict the universe of FX pairs that we support here for the time being
299  string baseCcy = simMarketData_->baseCcy();
300  for (auto sensi_fx : sensitivityData_->fxShiftData()) {
301  string foreign = sensi_fx.first.substr(0, 3);
302  string domestic = sensi_fx.first.substr(3);
303  QL_REQUIRE((domestic == baseCcy) || (foreign == baseCcy),
304  "SensitivityScenarioGenerator does not support cross FX pairs("
305  << sensi_fx.first << ", but base currency is " << baseCcy << ")");
306  }
307  // Log an ALERT if some currencies in simmarket are excluded from the list
308  for (auto sim_fx : simMarketData_->fxCcyPairs()) {
309  if (sensitivityData_->fxShiftData().find(sim_fx) == sensitivityData_->fxShiftData().end()) {
310  WLOG("FX pair " << sim_fx << " in simmarket is not included in sensitivities analysis");
311  }
312  }
313  for (auto sensi_fx : sensitivityData_->fxShiftData()) {
314  string ccypair = sensi_fx.first; // foreign + domestic;
315  SensitivityScenarioData::SpotShiftData data = sensi_fx.second;
316  ShiftType type = parseShiftType(data.shiftType);
317  Real size = up ? data.shiftSize : -1.0 * data.shiftSize;
318  // QL_REQUIRE(type == SensitivityScenarioGenerator::ShiftType::Relative, "FX scenario type must be relative");
319  bool relShift = (type == SensitivityScenarioGenerator::ShiftType::Relative);
320 
321  Real rate;
322  RiskFactorKey key(RiskFactorKey::KeyType::FXSpot, ccypair);
323  if (!tryGetBaseScenarioValue(baseScenarioAbsolute_, key, rate, continueOnError_))
324  continue;
325 
326  boost::shared_ptr<Scenario> scenario = sensiScenarioFactory_->buildScenario(asof);
327 
328  scenarioDescriptions_.push_back(fxScenarioDescription(ccypair, up));
329 
330  Real newRate = relShift ? rate * (1.0 + size) : (rate + size);
331  // Real newRate = up ? rate * (1.0 + data.shiftSize) : rate * (1.0 - data.shiftSize);
332  scenario->add(key, newRate);
333 
334  // Store absolute shift size
335  if (up)
336  shiftSizes_[key] = newRate - rate;
337 
338  // Give the scenario a label
339  scenario->label(to_string(scenarioDescriptions_.back()));
340 
341  scenarios_.push_back(scenario);
342  DLOG("Sensitivity scenario # " << scenarios_.size() << ", label " << scenario->label()
343  << " created: " << newRate);
344  }
345  LOG("FX scenarios done");
346 }

407 void SensitivityScenarioGenerator::generateDiscountCurveScenarios(bool up) {
408  Date asof = baseScenario_->asof();
409  // Log an ALERT if some currencies in simmarket are excluded from the list
410  for (auto sim_ccy : simMarketData_->ccys()) {
411  if (sensitivityData_->discountCurveShiftData().find(sim_ccy) ==
412  sensitivityData_->discountCurveShiftData().end()) {
413  WLOG("Currency " << sim_ccy << " in simmarket is not included in sensitivities analysis");
414  }
415  }
416 
417  for (auto c : sensitivityData_->discountCurveShiftData()) {
418  string ccy = c.first;
419  Size n_ten = simMarketData_->yieldCurveTenors(ccy).size();
420  // original curves' buffer
421  std::vector<Real> zeros(n_ten);
422  std::vector<Real> times(n_ten);
423  // buffer for shifted zero curves
424  std::vector<Real> shiftedZeros(n_ten);
425  SensitivityScenarioData::CurveShiftData data = *c.second;
426  ShiftType shiftType = parseShiftType(data.shiftType);
427  DayCounter dc = Actual365Fixed();
428  try {
429  dc = simMarket_->discountCurve(ccy)->dayCounter();
430  } catch(const std::exception&) {
431  WLOG("Day counter lookup in simulation market failed for discount curve " << ccy << ", using default A365");
432  }
433 
434  Real quote = 0.0;
435  bool valid = true;
436  for (Size j = 0; j < n_ten; ++j) {
437  Date d = asof + simMarketData_->yieldCurveTenors(ccy)[j];
438  times[j] = dc.yearFraction(asof, d);
439 
440  RiskFactorKey key(RiskFactorKey::KeyType::DiscountCurve, ccy, j);
441  valid = valid && tryGetBaseScenarioValue(baseScenarioAbsolute_, key, quote, continueOnError_);
442  zeros[j] = -std::log(quote) / times[j];
443  }
444  if (!valid)
445  continue;
446 
447  std::vector<Period> shiftTenors = overrideTenors_ && simMarketData_->hasYieldCurveTenors(ccy)
448  ? simMarketData_->yieldCurveTenors(ccy)
449  : data.shiftTenors;
450  checkShiftTenors(shiftTenors, data.shiftTenors, "Discount Curve " + ccy, continueOnError_);
451  std::vector<Time> shiftTimes(shiftTenors.size());
452  for (Size j = 0; j < shiftTenors.size(); ++j)
453  shiftTimes[j] = dc.yearFraction(asof, asof + shiftTenors[j]);
454  Real shiftSize = data.shiftSize;
455  QL_REQUIRE(shiftTenors.size() > 0, "Discount shift tenors not specified");
456 
457  // Can we store a valid shift size?
458  bool validShiftSize = vectorEqual(times, shiftTimes);
459 
460  for (Size j = 0; j < shiftTenors.size(); ++j) {
461 
462  boost::shared_ptr<Scenario> scenario = sensiScenarioFactory_->buildScenario(asof);
463  scenarioDescriptions_.push_back(discountScenarioDescription(ccy, j, up));
464  DLOG("generate discount curve scenario, ccy " << ccy << ", bucket " << j << ", up " << up << ", desc "
465  << scenarioDescriptions_.back());
466 
467  // apply zero rate shift at tenor point j
468  applyShift(j, shiftSize, up, shiftType, shiftTimes, zeros, times, shiftedZeros, true);
469 
470  // store shifted discount curve in the scenario
471  for (Size k = 0; k < n_ten; ++k) {
472  RiskFactorKey key(RFType::DiscountCurve, ccy, k);
473  // FIXME why do we have that here, but not in generateIndexCurveScenarios?
474  if (!close_enough(shiftedZeros[k], zeros[k])) {
475  Real shiftedDiscount = exp(-shiftedZeros[k] * times[k]);
476  if (sensitivityData_->useSpreadedTermStructures()) {
477  Real discount = exp(-zeros[k] * times[k]);
478  scenario->add(key, shiftedDiscount / discount);
479  } else {
480  scenario->add(key, shiftedDiscount);
481  }
482  }
483 
484  // Possibly store valid shift size
485  if (validShiftSize && up && j == k) {
486  shiftSizes_[key] = shiftedZeros[k] - zeros[k];
487  }
488  }
489 
490  // Give the scenario a label
491  scenario->label(to_string(scenarioDescriptions_.back()));
492 
493  // add this scenario to the scenario vector
494  scenarios_.push_back(scenario);
495  DLOG("Sensitivity scenario # " << scenarios_.size() << ", label " << scenario->label() << " created");
496 
497  } // end of shift curve tenors
498  }
499  LOG("Discount curve scenarios done");
500 }
