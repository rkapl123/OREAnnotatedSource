Detailed Description

Sensitivity Analysis.

This class wraps functionality to perform a sensitivity analysis for a given portfolio. It comprises

  • building the "simulation" market to which sensitivity scenarios are applied
  • building the portfolio linked to this simulation market
  • generating sensitivity scenarios
  • running the scenario "engine" to apply these and compute the NPV impacts of all required shifts
  • compile first and second order sensitivities for all factors and all trades
  • fill result structures that can be queried

312 << feeCcySensi << ". (feeCcy, baseCcy, feeNpvDomCcy, " << pairDomBase

314 << baseCcy << "," << fee->NPV() << "," << shiftSize << "," << addMults[0] << ").");
315 }
316
317 if (feeCcy == domCcy) {
318 domCcySensi += feeCcySensi;
319 } else {
320 forCcySensi += feeCcySensi;
321 }
322 }
323 }
324 }
325

327 auto upFactors = sensiCube_->upFactors().left;
328 auto downFactors = sensiCube_->downFactors();

330
331 // Sensitivity to foreign currency.
332 if (forCcySensi != Null<Real>()) {
333 auto itu = upFactors.find(forCcyKey);
334 if (itu != upFactors.end()) {
335 TLOG("Adding analytic " << forCcy << " up sensitivity to cube.");
336 sensiCube_->npvCube()->set(baseNpv + forCcySensi, tradeIdx, itu->second.index);
337 }
338 auto itd = downFactors.find(forCcyKey);
339 if (itd != downFactors.end()) {
340 TLOG("Adding analytic " << forCcy << " down sensitivity to cube.");
341 sensiCube_->npvCube()->set(baseNpv - forCcySensi, tradeIdx, itd->second.index);
342 }
343 }
344
345 // Sensitivity to domestic currency.
346 if (domCcySensi != Null<Real>()) {
347 auto itu = upFactors.find(domCcyKey);
348 if (itu != upFactors.end()) {
349 TLOG("Adding analytic " << domCcy << " up sensitivity to cube.");
350 sensiCube_->npvCube()->set(baseNpv + domCcySensi, tradeIdx, itu->second.index);
351 }
352 auto itd = downFactors.find(domCcyKey);
353 if (itd != downFactors.end()) {
354 TLOG("Adding analytic " << domCcy << " down sensitivity to cube.");
355 sensiCube_->npvCube()->set(baseNpv - domCcySensi, tradeIdx, itd->second.index);
356 }
357 }
358 }
359 }

361 DLOG("addAnalyticFxSensitivities: finished overwriting FX sensitivities with analytic values.");
