34void SimmAnalyticImpl::runAnalytic(const boost::shared_ptr<ore::data::InMemoryLoader>& loader,
35 const std::set<std::string>& runTypes) {
36 if (!analytic()->match(runTypes))
37 return;
38
39 LOG("SimmAnalytic::runAnalytic called");
40
41 analytic()->buildMarket(loader, false);
42
43 auto simmAnalytic = static_cast<SimmAnalytic*>(analytic());
44 QL_REQUIRE(simmAnalytic, "Analytic must be of type SimmAnalytic");
45
46 LOG("Get CRIF records from CRIF loader and fill amountUSD");
47 simmAnalytic->loadCrifRecords(loader);
48
49 auto simmConfiguration = inputs_->crifLoader()->simmConfiguration();
50
51 if (analytic()->getWriteIntermediateReports()) {
52 boost::shared_ptr<InMemoryReport> crifReport = boost::make_shared<InMemoryReport>();
53 ReportWriter(inputs_->reportNaString())
54 .writeCrifReport(crifReport, *simmAnalytic->crifRecords());
55 analytic()->reports()[LABEL]["crif"] = crifReport;
56 LOG("CRIF report generated");
57
58 auto crifLoader = boost::make_shared<CrifLoader>(simmConfiguration, CrifRecord::additionalHeaders, true, true);
59 for (const auto& cr : *simmAnalytic->crifRecords())
60 crifLoader->add(cr);
61 boost::shared_ptr<InMemoryReport> simmDataReport = boost::make_shared<InMemoryReport>();
62 ReportWriter(inputs_->reportNaString())
63 .writeSIMMData(crifLoader->netRecords(true), simmDataReport);
64 analytic()->reports()[LABEL]["simm_data"] = simmDataReport;
65 LOG("SIMM data report generated");
66 }
67 MEM_LOG;
68
69 // Calculate SIMM
70 auto simm = boost::make_shared<SimmCalculator>(*simmAnalytic->crifRecords(),
71 simmConfiguration,
72 inputs_->simmCalculationCurrency(),
73 inputs_->simmResultCurrency(),
74 analytic()->market(),
75 simmAnalytic->determineWinningRegulations(),
76 inputs_->enforceIMRegulations());
77
78 Real fxSpot = 1.0;
79 if (!inputs_->simmReportingCurrency().empty()) {
80 fxSpot = analytic()->market()
81 ->fxRate(inputs_->simmResultCurrency() + inputs_->simmReportingCurrency())
82 ->value();
83 LOG("SIMM reporting currency is " << inputs_->simmReportingCurrency() << " with fxSpot " << fxSpot);
84 }
85
86 boost::shared_ptr<InMemoryReport> simmRegulationBreakdownReport = boost::make_shared<InMemoryReport>();
87 ReportWriter(inputs_->reportNaString())
88 .writeSIMMReport(simm->simmResults(), simmRegulationBreakdownReport, simmAnalytic->hasNettingSetDetails(),
89 inputs_->simmResultCurrency(), inputs_->simmReportingCurrency(), false, fxSpot);
90 LOG("SIMM regulation breakdown report generated");
91 analytic()->reports()[LABEL]["regulation_breakdown_simm"] = simmRegulationBreakdownReport;
92
93
94 boost::shared_ptr<InMemoryReport> simmReport = boost::make_shared<InMemoryReport>();
95 ReportWriter(inputs_->reportNaString())
96 .writeSIMMReport(simm->finalSimmResults(), simmReport, simmAnalytic->hasNettingSetDetails(),
97 inputs_->simmResultCurrency(), inputs_->simmReportingCurrency(), fxSpot);
98 analytic()->reports()[LABEL]["simm"] = simmReport;
99 LOG("SIMM report generated");
100 MEM_LOG;
101
102}

104void SimmAnalytic::loadCrifRecords(const boost::shared_ptr<ore::data::InMemoryLoader>& loader) {
105 QL_REQUIRE(inputs_, "Inputs not set");
106 QL_REQUIRE(inputs_->crifLoader(), "CRIF loader not set");
107 QL_REQUIRE(inputs_->crifLoader()->hasCrifRecords(), "CRIF loader does not contain any records");
108
109 inputs_->crifLoader()->fillAmountUsd(market());
110 crifRecords_ = boost::make_shared<SimmNetSensitivities>(inputs_->crifLoader()->netRecords(true));
111 hasNettingSetDetails_ = inputs_->crifLoader()->hasNettingSetDetails();
112}
