61 vector<string> ScenarioSimMarketParameters::paramsLookup(RiskFactorKey::KeyType kt) const {

71 bool ScenarioSimMarketParameters::hasParamsName(RiskFactorKey::KeyType kt, string name) const {
72  auto it = params_.find(kt);
73  if (it != params_.end()) {
74  return std::find(it->second.second.begin(), it->second.second.end(), name) == it->second.second.end() ? false
75  : true;
76  }

80 void ScenarioSimMarketParameters::addParamsName(RiskFactorKey::KeyType kt, vector<string> names) {
81  // check if key type exists - if doesn't exist set simulate to true first
82  if (names.size() > 0) {
83  auto it = params_.find(kt);
84  if (it == params_.end())
85  params_[kt].first = true;
86  for (auto name : names) {
87  if (!hasParamsName(kt, name))
88  params_[kt].second.insert(name);
89  }
90  }
91 }
