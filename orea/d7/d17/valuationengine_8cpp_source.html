64 void ValuationEngine::buildCube(const boost::shared_ptr<data::Portfolio>& portfolio,
65  boost::shared_ptr<analytics::NPVCube> outputCube,
66  vector<boost::shared_ptr<ValuationCalculator>> calculators, bool mporStickyDate,
67  boost::shared_ptr<analytics::NPVCube> outputCubeNettingSet,
68  boost::shared_ptr<analytics::NPVCube> outputCptyCube,
69  vector<boost::shared_ptr<CounterpartyCalculator>> cptyCalculators) {
70 
71  LOG("Build cube with mporStickyDate=" << mporStickyDate);
72 
73  QL_REQUIRE(portfolio->size() > 0, "ValuationEngine: Error portfolio is empty");
74 
75  QL_REQUIRE(outputCube->numIds() == portfolio->trades().size(),
76  "cube x dimension (" << outputCube->numIds() << ") "
77  << "different from portfolio size (" << portfolio->trades().size() << ")");
78 
79  QL_REQUIRE(outputCube->numDates() == dg_->valuationDates().size(),
80  "cube y dimension (" << outputCube->numDates() << ") "
81  << "different from number of valuation dates (" << dg_->valuationDates().size()
82  << ")");
83 
84  if (outputCptyCube) {
85  QL_REQUIRE(outputCptyCube->numIds() == portfolio->counterparties().size() + 1,
86  "cptyCube x dimension (" << outputCptyCube->numIds() << "minus 1) "
87  << "different from portfolio counterparty size ("
88  << portfolio->counterparties().size() << ")");
89 
90  QL_REQUIRE(outputCptyCube->numDates() == dg_->dates().size(), "outputCptyCube y dimension ("
91  << outputCptyCube->numDates() << ") "
92  << "different from number of time steps ("
93  << dg_->dates().size() << ")");
94  }
95 
96  LOG("Starting ValuationEngine for " << portfolio->size() << " trades, " << outputCube->samples() << " samples and "
97  << dg_->size() << " dates.");
98 
99  ObservationMode::Mode om = ObservationMode::instance().mode();
100  Real updateTime = 0.0;
101  Real pricingTime = 0.0;
102  Real fixingTime = 0.0;
103 
104  // Loop is Samples, Dates, Trades
105  const auto& dates = dg_->dates();
106  const auto& trades = portfolio->trades();
107  auto& counterparties = outputCptyCube ? outputCptyCube->ids() : vector<string>();
108  std::vector<bool> tradeHasError(trades.size(), false);
109  LOG("Initialise state objects...");
110  Size numFRC = 0;
111  // initialise state objects for each trade (required for path-dependent derivatives in particular)
112  for (Size i = 0; i < trades.size(); i++) {
113  QL_REQUIRE(trades[i]->npvCurrency() != "", "NPV currency not set for trade " << trades[i]->id());
114 
115  DLOG("Initialise wrapper for trade " << trades[i]->id());
116  trades[i]->instrument()->initialise(dates);
117 
118  recalibrateModels();
119 
120  // T0 values
121  try {
122  for (auto calc : calculators)
123  calc->calculateT0(trades[i], i, simMarket_, outputCube, outputCubeNettingSet);
124  } catch (const std::exception& e) {
125  string expMsg = string("T0 valuation error: ") + e.what();
126  ALOG(StructuredTradeErrorMessage(trades[i]->id(), trades[i]->tradeType(), "ScenarioValuation",
127  expMsg.c_str()));
128  tradeHasError[i] = true;
129  }
130 
131  if (om == ObservationMode::Mode::Unregister) {
132  for (const Leg& leg : trades[i]->legs()) {
133  for (Size n = 0; n < leg.size(); n++) {
134  boost::shared_ptr<FloatingRateCoupon> frc = boost::dynamic_pointer_cast<FloatingRateCoupon>(leg[n]);
135  if (frc) {
136  frc->unregisterWith(frc->index());
137  trades[i]->instrument()->qlInstrument()->unregisterWith(frc);
138  // Unregister with eval dates
139  frc->unregisterWith(Settings::instance().evaluationDate());
140  frc->index()->unregisterWith(Settings::instance().evaluationDate());
141  trades[i]->instrument()->qlInstrument()->unregisterWith(Settings::instance().evaluationDate());
142  }
143  }
144  }
145  }
146  }
147  LOG("Total number of swaps = " << portfolio->size());
148  LOG("Total number of FRC = " << numFRC);
149 
150  simMarket_->fixingManager()->initialise(portfolio, simMarket_);
151 
152  cpu_timer timer;
153  cpu_timer loopTimer;
154 
155  // We call Cube::samples() each time her to allow for dynamic stopping times
156  // e.g. MC convergence tests
157  for (Size sample = 0; sample < outputCube->samples(); ++sample) {
158  updateProgress(sample, outputCube->samples());
159 
160  for (auto& trade : trades)
161  trade->instrument()->reset();
162 
163  // loop over Dates, increase cubeDateIndex for each valuation date we hit
164  int cubeDateIndex = -1;
165  for (Size i = 0; i < dates.size(); ++i) {
166  Date d = dates[i];
167 
168  // Process auxiliary close-out dates first (may coincide with a valuation date, see below)
169  // Store result at same cubeDateIndex as the previous valuation date's result, but at different cube depth
170  // Differences to valuation date processing above:
171  // Update valuation date and fixings, trades exercisable depending on stickiness
172  bool scenarioUpdated = false;
173  if (dg_->isCloseOutDate()[i]) {
174  timer.start();
175 
176  // update market
177  simMarket_->preUpdate();
178  if (!mporStickyDate)
179  simMarket_->updateDate(d);
180  simMarket_->updateScenario(d);
181  scenarioUpdated = true;
182  simMarket_->postUpdate(d, !mporStickyDate); // with fixings only if not sticky
183 
184  recalibrateModels();
185 
186  timer.stop();
187  updateTime += timer.elapsed().wall * 1e-9;
188 
189  // loop over trades
190  timer.start();
191  if (mporStickyDate) // switch off if sticky
192  tradeExercisable(false, trades);
193  QL_REQUIRE(cubeDateIndex >= 0,
194  "negative cube date index, ensure that the date grid starts with a valuation date");
195  runCalculators(true, trades, tradeHasError, calculators, outputCube, outputCubeNettingSet, d,
196  cubeDateIndex, sample, simMarket_->label());
197  if (mporStickyDate) // switch on again, if sticky
198  tradeExercisable(true, trades);
199  timer.stop();
200  pricingTime += timer.elapsed().wall * 1e-9;
201  }
202 
203  // process a valuation date as usual
204  if (dg_->isValuationDate()[i]) {
205  timer.start();
206 
207  cubeDateIndex++;
208 
209  // All the steps below from preUpdate() to updateAsd(d) are combined in update(d), but we decompose as
210  // follows simMarket_->update(d);
211  simMarket_->preUpdate();
212  simMarket_->updateDate(d);
213  // We can skip this step, if we have done that above in the close-out date section
214  if (!scenarioUpdated)
215  simMarket_->updateScenario(d);
216  // Always with fixing update here, in contrast to the close-out date section
217  simMarket_->postUpdate(d, true);
218  // Aggregation scenario data update on valuation dates only
219  simMarket_->updateAsd(d);
220 
221  recalibrateModels();
222 
223  timer.stop();
224  updateTime += timer.elapsed().wall * 1e-9;
225 
226  timer.start();
227  // loop over trades
228  runCalculators(false, trades, tradeHasError, calculators, outputCube, outputCubeNettingSet, d,
229  cubeDateIndex, sample, simMarket_->label());
230  // loop over counterparty names
231  runCalculators(false, counterparties, cptyCalculators, outputCptyCube, d, cubeDateIndex, sample);
232  timer.stop();
233  pricingTime += timer.elapsed().wall * 1e-9;
234  }
235  }
236 
237  timer.start();
238  simMarket_->fixingManager()->reset();
239  fixingTime += timer.elapsed().wall * 1e-9;
240  }
241 
242  simMarket_->reset();
243  updateProgress(outputCube->samples(), outputCube->samples());
244  loopTimer.stop();
245  LOG("ValuationEngine completed: loop " << setprecision(2) << loopTimer.format(2, "%w") << " sec, "
246  << "pricing " << pricingTime << " sec, "
247  << "update " << updateTime << " sec "
248  << "fixing " << fixingTime);
249 
250  // for trades with errors set all output cube values to zero
251 
252  for (Size i = 0; i < trades.size(); ++i) {
253  if (tradeHasError[i]) {
254  ALOG("setting all results in output cube to zero for trade '"
255  << trades[i]->id() << "' since there was at least one error during simulation");
256  for (Size index = 0; index < outputCube->depth(); ++index) {
257  outputCube->setT0(0.0, i, index);
258  for (Size dateIndex = 0; dateIndex < dates.size(); ++dateIndex) {
259  for (Size sample = 0; sample < outputCube->samples(); ++sample) {
260  outputCube->set(0.0, i, dateIndex, sample, index);
261  }
262  }
263  }
264  }
265  }
266 }

268 void ValuationEngine::runCalculators(bool isCloseOutDate, const std::vector<boost::shared_ptr<Trade>>& trades,
269  std::vector<bool>& tradeHasError,
270  const std::vector<boost::shared_ptr<ValuationCalculator>>& calculators,
271  boost::shared_ptr<analytics::NPVCube>& outputCube,
272  boost::shared_ptr<analytics::NPVCube>& outputCubeNettingSet, const Date& d,
273  const Size cubeDateIndex, const Size sample, const string& label) {
274  ObservationMode::Mode om = ObservationMode::instance().mode();
275  // loop over trades
276  for (Size j = 0; j < trades.size(); ++j) {
277  if (tradeHasError[j])
278  continue;
279  auto trade = trades[j];
280  // We can avoid checking mode here and always call updateQlInstruments()
281  if (om == ObservationMode::Mode::Disable)
282  trade->instrument()->updateQlInstruments();
283  try {
284  for (auto calc : calculators)
285  calc->calculate(trade, j, simMarket_, outputCube, outputCubeNettingSet, d, cubeDateIndex, sample,
286  isCloseOutDate);
287  } catch (const std::exception& e) {
288  string expMsg = "date = " + ore::data::to_string(io::iso_date(d)) +
289  ", sample = " + ore::data::to_string(sample) + ", label = " + label + ": " + e.what();
290  ALOG(StructuredTradeErrorMessage(trade->id(), trade->tradeType(), "ScenarioValuation", expMsg.c_str()));
291  tradeHasError[j] = true;
292  }
293  }
294 }

297  const std::vector<boost::shared_ptr<CounterpartyCalculator>>& calculators,
298  boost::shared_ptr<analytics::NPVCube>& cptyCube, const Date& d,

305  }
