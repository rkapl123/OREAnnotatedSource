37Real fx(const std::vector<std::vector<std::vector<Real>>>& fxBuffer, const Size ccyIndex, const Size timeIndex,
38 const Size sample) {
39 if (ccyIndex == 0)
40 return 1.0;
41 return fxBuffer[ccyIndex - 1][timeIndex][sample];
42}
43
44Real state(const std::vector<std::vector<std::vector<Real>>>& irStateBuffer, const Size ccyIndex, const Size timeIndex,
45 const Size sample) {
46 return irStateBuffer[ccyIndex][timeIndex][sample];
47}
48
49Real numRatio(const boost::shared_ptr<CrossAssetModel>& model,
50 const std::vector<std::vector<std::vector<Real>>>& irStateBuffer, const Size ccyIndex,
51 const Size timeIndex, const Real time, const Size sample) {
52 if (ccyIndex == 0)
53 return 1.0;
54 Real state_base = state(irStateBuffer, 0, timeIndex, sample);
55 Real state_curr = state(irStateBuffer, ccyIndex, timeIndex, sample);
56 return model->numeraire(ccyIndex, time, state_curr) / model->numeraire(0, time, state_base);
57}
58
59Real num(const boost::shared_ptr<CrossAssetModel>& model,
60 const std::vector<std::vector<std::vector<Real>>>& irStateBuffer, const Size ccyIndex, const Size timeIndex,
61 const Real time, const Size sample) {
62 Real state_curr = state(irStateBuffer, ccyIndex, timeIndex, sample);
63 return model->numeraire(ccyIndex, time, state_curr);
64}

66Array simulatePathInterface1(const boost::shared_ptr<AmcCalculatorSinglePath>& amcCalc, const MultiPath& path,
67 const bool reuseLastEvents, const std::string& tradeLabel, const Size sample) {
68 try {
69 return amcCalc->simulatePath(path, reuseLastEvents);
70 } catch (const std::exception& e) {
71 ALOG("error for trade '" << tradeLabel << "' sample #" << sample << ": " << e.what());
72 return Array(path.pathSize(), 0.0);

91AMCValuationEngine::AMCValuationEngine(const boost::shared_ptr<QuantExt::CrossAssetModel>& model,
92 const boost::shared_ptr<ScenarioGeneratorData>& sgd,
93 const boost::shared_ptr<Market>& market,
94 const std::vector<string>& aggDataIndices,
95 const std::vector<string>& aggDataCurrencies)
96 : model_(model), sgd_(sgd), market_(market), aggDataIndices_(aggDataIndices),
97 aggDataCurrencies_(aggDataCurrencies) {
98
99 QL_REQUIRE((aggDataIndices.empty() && aggDataCurrencies.empty()) || market != nullptr,
100 "AMCValuationEngine: market is required for asd generation");
101 QL_REQUIRE(sgd->seed() != 0,
102 "AMCValuationEngine: path generation uses seed 0 - this might lead to inconsistent results to a classic "
103 "simulation run, if both are combined. Consider using a non-zero seed.");
104 QL_REQUIRE(
105 model->irlgm1f(0)->termStructure()->dayCounter() == sgd->getGrid()->dayCounter(),
106 "AMCValuationEngine: day counter in simulation parameters ("
107 << sgd->getGrid()->dayCounter() << ") is different from model day counter ("
108 << model->irlgm1f(0)->termStructure()->dayCounter()
109 << "), align these e.g. by setting the day counter in the simulation parameters to the model day counter");
110}

112MultiPath AMCValuationEngine::effectiveSimulationPath(const MultiPath& p, const bool processCloseOutDates) const {
113 QL_REQUIRE(sgd_->withCloseOutLag() && sgd_->withMporStickyDate(),
114 "effectiveSimulationPath(): expected grid with close-out lag and sticky date mpor mode");
115 MultiPath filteredPath(p.assetNumber(), sgd_->getGrid()->valuationTimeGrid());
116 Size filteredTimeIndex = 0;
117 for (Size i = 0; i < p.pathSize(); ++i) {
118 if (i == 0 || (i > 0 && (processCloseOutDates ? sgd_->getGrid()->isCloseOutDate()[i - 1]
119 : sgd_->getGrid()->isValuationDate()[i - 1]))) {
120 for (Size j = 0; j < p.assetNumber(); ++j) {
121 filteredPath[j][filteredTimeIndex] = p[j][i];
122 }
123 ++filteredTimeIndex;
124 }
125 }
126 return filteredPath;
127}

129void AMCValuationEngine::buildCube(const boost::shared_ptr<Portfolio>& portfolio,
130 boost::shared_ptr<NPVCube>& outputCube) {
131
132 LOG("Starting AMCValuationEngine for " << portfolio->size() << " trades, " << outputCube->samples()
133 << " samples and " << sgd_->getGrid()->size() << " dates.");
134
135 QL_REQUIRE(portfolio->size() > 0, "AMCValuationEngine::buildCube: empty portfolio");
136
137 QL_REQUIRE(outputCube->numIds() == portfolio->trades().size(),
138 "cube x dimension (" << outputCube->numIds() << ") "
139 << "different from portfolio size (" << portfolio->trades().size() << ")");
140
141 QL_REQUIRE(outputCube->numDates() == sgd_->getGrid()->valuationDates().size(),
142 "cube y dimension (" << outputCube->numDates() << ") "
143 << "different from number of valuation dates ("
144 << sgd_->getGrid()->valuationDates().size() << ")");
145
146 // base currency is the base currency of the cam
147
148 Currency baseCurrency = model_->irlgm1f(0)->currency();
149
150 // timings
151
152 boost::timer::cpu_timer timer, timerTotal;
153 Real calibrationTime = 0.0, valuationTime = 0.0, asdTime = 0.0, pathGenTime = 0.0, residualTime, totalTime;
154 timerTotal.start();
155
156 // prepare for asd writing
157
158 std::vector<Size> asdCurrencyIndex; // FX Spots
159 std::vector<string> asdCurrencyCode;
160 std::vector<boost::shared_ptr<LgmImpliedYtsFwdFwdCorrected>> asdIndexCurve; // Ibor Indices
161 std::vector<boost::shared_ptr<Index>> asdIndex;
162 std::vector<Size> asdIndexIndex;
163 std::vector<string> asdIndexName;
164 if (asd_ != nullptr) {
165 LOG("Collect information for aggregation scenario data...");
166 // fx spots
167 for (auto const& c : aggDataCurrencies_) {
168 Currency cur = parseCurrency(c);
169 if (cur == baseCurrency)
170 continue;
171 Size ccyIndex = model_->ccyIndex(cur);
172 asdCurrencyIndex.push_back(ccyIndex);
173 asdCurrencyCode.push_back(c);
174 }
175 // ibor indices
176 for (auto const& i : aggDataIndices_) {
177 boost::shared_ptr<IborIndex> tmp;
178 try {
179 tmp = *market_->iborIndex(i);
180 } catch (const std::exception& e) {
181 ALOG("index \"" << i << "\" not found in market, skipping. (" << e.what() << ")");
182 }
183 Size ccyIndex = model_->ccyIndex(tmp->currency());
184 asdIndexCurve.push_back(boost::make_shared<LgmImpliedYtsFwdFwdCorrected>(model_->lgm(ccyIndex),
185 tmp->forwardingTermStructure()));
186 asdIndex.push_back(tmp->clone(Handle<YieldTermStructure>(asdIndexCurve.back())));
187 asdIndexIndex.push_back(ccyIndex);
188 asdIndexName.push_back(i);
189 }
190 } else {
191 LOG("No asd object set, won't write aggregation scenario data...");
192 }
193
194 // extract AMC calculators and additional things we need from the ore wrapper
195
196 LOG("Extract AMC Calculators...");
197 std::vector<boost::shared_ptr<AmcCalculator>> amcCalculators;
198 std::vector<Size> tradeId;
199 std::vector<std::string> tradeLabel;
200 std::vector<Real> effectiveMultiplier;
201 std::vector<Size> currencyIndex;
202 timer.start();
203 for (Size i = 0; i < portfolio->size(); ++i) {
204 boost::shared_ptr<AmcCalculator> amcCalc;
205 const boost::shared_ptr<Trade>& t = portfolio->trades()[i];
206 try {
207 amcCalc = t->instrument()->qlInstrument(true)->result<boost::shared_ptr<AmcCalculator>>("amcCalculator");
208 LOG("AMCCalculator extracted for \"" << t->id() << "\"");
209 } catch (const std::exception& e) {
210 WLOG("could not extract AMCCalculator for trade \"" << t->id() << "\": " << e.what());
211 }
212 if (amcCalc != nullptr) {
213 amcCalculators.push_back(amcCalc);
214 Real effMult = t->instrument()->multiplier();
215 auto ow = boost::dynamic_pointer_cast<OptionWrapper>(t->instrument());
216 if (ow != nullptr) {
217 effMult *= ow->isLong() ? 1.0 : -1.0;
218 // we can ignore the underlying multiplier, since this is not involved in the AMC engine
219 }
220 effectiveMultiplier.push_back(effMult);
221 currencyIndex.push_back(model_->ccyIndex(amcCalc->npvCurrency()));
222 tradeId.push_back(i);
223 tradeLabel.push_back(t->id());
224 // TODO additional instruments (fees)
225 }
226 updateProgress(i, portfolio->size());
227 }
228 timer.stop();
229 calibrationTime += timer.elapsed().wall * 1e-9;
230 LOG("Extracted " << amcCalculators.size() << " AMCCalculators for " << portfolio->size() << " source trades");
231
232 // run the simulation and populate the cube with NPVs, and write aggregation scenario data
233
234 auto process = model_->stateProcess();
235
236 /* set up buffers for fx rates and ir states that we need below for the runs against interface 1 and 2
237 we set these buffers up on the full grid (i.e. valuation + close-out dates, also including the T0 date)
238 */
239
240 std::vector<std::vector<std::vector<Real>>> fxBuffer(
241 model_->components(CrossAssetModel::AssetType::FX),
242 std::vector<std::vector<Real>>(sgd_->getGrid()->dates().size() + 1, std::vector<Real>(outputCube->samples())));
243 std::vector<std::vector<std::vector<Real>>> irStateBuffer(
244 model_->components(CrossAssetModel::AssetType::IR),
245 std::vector<std::vector<Real>>(sgd_->getGrid()->dates().size() + 1, std::vector<Real>(outputCube->samples())));
246
247 /* set up cache for paths, this is used in interface 2 below */
248
249 Size nStates = process->size();
250 QL_REQUIRE(sgd_->getGrid()->timeGrid().size() > 0, "AMCValuationEngine: empty time grid given");
251 std::vector<Real> pathTimes(std::next(sgd_->getGrid()->timeGrid().begin(), 1), sgd_->getGrid()->timeGrid().end());
252 std::vector<std::vector<RandomVariable>> paths(
253 pathTimes.size(), std::vector<RandomVariable>(nStates, RandomVariable(outputCube->samples())));
254
255 // Run AmcCalculators implementing interface 1, write ASD and fill fxBuffer / irStateBuffer
256
257 // FIXME hardcoded ordering and directionIntegers here...
258 auto pathGenerator =
259 makeMultiPathGenerator(sgd_->sequenceType(), process, sgd_->getGrid()->timeGrid(), sgd_->seed());
260 LOG("Run simulation (amc calculators implementing interface 1, write ASD, fill internal fx and irState "
261 "buffers)...");
262 resetProgress();
263 for (Size i = 0; i < outputCube->samples(); ++i) {
264 timer.start();
265 const MultiPath& path = pathGenerator->next().value;
266 timer.stop();
267 pathGenTime += timer.elapsed().wall * 1e-9;
268
269 // populate fx and ir state buffers, populate cached paths for interface 2
270
271 for (Size k = 0; k < fxBuffer.size(); ++k) {
272 for (Size j = 0; j < sgd_->getGrid()->timeGrid().size(); ++j) {
273 fxBuffer[k][j][i] = std::exp(path[model_->pIdx(CrossAssetModel::AssetType::FX, k)][j]);
274 }
275 }
276 for (Size k = 0; k < irStateBuffer.size(); ++k) {
277 for (Size j = 0; j < sgd_->getGrid()->timeGrid().size(); ++j) {
278 irStateBuffer[k][j][i] = path[model_->pIdx(CrossAssetModel::AssetType::IR, k)][j];
279 }
280 }
281
282 // TODO we do not need this if there are no amc calculators implementing interface 2
283 for (Size k = 0; k < nStates; ++k) {
284 for (Size j = 0; j < pathTimes.size(); ++j) {
285 paths[j][k].set(i, path[k][j]);
286 }
287 }
288
289 // amc valuation and output to cube
290
291 timer.start();
292 for (Size j = 0; j < amcCalculators.size(); ++j) {
293 auto amcCalc = boost::dynamic_pointer_cast<AmcCalculatorSinglePath>(amcCalculators[j]);
294 if (amcCalc == nullptr)
295 continue;
296
297 if (!sgd_->withCloseOutLag()) {
298 // no close-out lag, fill depth 0 of cube with npvs on path
299 Array res = simulatePathInterface1(amcCalc, path, false, tradeLabel[j], i);
300 outputCube->setT0(res[0] * fx(fxBuffer, currencyIndex[j], 0, 0) *
301 numRatio(model_, irStateBuffer, currencyIndex[j], 0, 0.0, 0) *
302 effectiveMultiplier[j],
303 tradeId[j], 0);
304 int dateIndex = -1;
305 for (Size k = 1; k < res.size(); ++k) {
306 ++dateIndex;
307 Real t = sgd_->getGrid()->timeGrid()[k];
308 outputCube->set(res[k] * fx(fxBuffer, currencyIndex[j], k, i) *
309 numRatio(model_, irStateBuffer, currencyIndex[j], k, t, i) *
310 effectiveMultiplier[j],
311 tradeId[j], dateIndex, i, 0);
312 }
313 } else {
314 // with close-out lag, fille depth 0 with valuation date npvs, depth 1 with (inflated) close-out npvs
315 if (sgd_->withMporStickyDate()) {
316 // stikcy date mpor mode, simulate the valuation times and close out times separately
317 Array res =
318 simulatePathInterface1(amcCalc, effectiveSimulationPath(path, false), false, tradeLabel[j], i);
319 Array resCout =
320 simulatePathInterface1(amcCalc, effectiveSimulationPath(path, true), true, tradeLabel[j], i);
321 outputCube->setT0(res[0] * fx(fxBuffer, currencyIndex[j], 0, 0) *
322 numRatio(model_, irStateBuffer, currencyIndex[j], 0, 0.0, 0) *
323 effectiveMultiplier[j],
324 tradeId[j], 0);
325 int dateIndex = -1;
326 for (Size k = 0; k < sgd_->getGrid()->dates().size(); ++k) {
327 Real t = sgd_->getGrid()->timeGrid()[k + 1];
328 Real tm = sgd_->getGrid()->timeGrid()[k];
329 if (sgd_->getGrid()->isCloseOutDate()[k]) {
330 QL_REQUIRE(dateIndex >= 0, "first date in grid must be a valuation date");
331 outputCube->set(resCout[dateIndex + 1] * fx(fxBuffer, currencyIndex[j], k + 1, i) *
332 num(model_, irStateBuffer, currencyIndex[j], k + 1, tm, i) *
333 effectiveMultiplier[j],
334 tradeId[j], dateIndex, i, 1);
335 }
336 if (sgd_->getGrid()->isValuationDate()[k]) {
337 dateIndex++;
338 outputCube->set(res[dateIndex + 1] * fx(fxBuffer, currencyIndex[j], k + 1, i) *
339 numRatio(model_, irStateBuffer, currencyIndex[j], k + 1, t, i) *
340 effectiveMultiplier[j],
341 tradeId[j], dateIndex, i, 0);
342 }
343 }
344 } else {
345 // actual date mport mode: simulate all times in one go
346 Array res = simulatePathInterface1(amcCalc, path, false, tradeLabel[j], i);
347 outputCube->setT0(res[0] * fx(fxBuffer, currencyIndex[j], 0, 0) *
348 numRatio(model_, irStateBuffer, currencyIndex[j], 0, 0.0, 0) *
349 effectiveMultiplier[j],
350 tradeId[j], 0);
351 int dateIndex = -1;
352 for (Size k = 1; k < res.size(); ++k) {
353 Real t = sgd_->getGrid()->timeGrid()[k];
354 if (sgd_->getGrid()->isCloseOutDate()[k - 1]) {
355 QL_REQUIRE(dateIndex >= 0, "first date in grid must be a valuation date");
356 outputCube->set(res[k] * fx(fxBuffer, currencyIndex[j], k, i) *
357 num(model_, irStateBuffer, currencyIndex[j], k, t, i) *
358 effectiveMultiplier[j],
359 tradeId[j], dateIndex, i, 1);
360 }
361 if (sgd_->getGrid()->isValuationDate()[k - 1]) {
362 dateIndex++;
363 outputCube->set(res[k] * fx(fxBuffer, currencyIndex[j], k, i) *
364 numRatio(model_, irStateBuffer, currencyIndex[j], k, t, i) *
365 effectiveMultiplier[j],
366 tradeId[j], dateIndex, i, 0);
367 }
368 }
369 }
370 }
371 }
372 timer.stop();
373 valuationTime += timer.elapsed().wall * 1e-9;
374
375 // write aggregation scenario data, TODO this seems relatively slow, can we speed it up using LgmVectorised
376 // etc.?
377
378 if (asd_ != nullptr) {
379 timer.start();
380 Size dateIndex = 0;
381 for (Size k = 1; k < sgd_->getGrid()->timeGrid().size(); ++k) {
382 // only write asd on valuation dates
383 if (!sgd_->getGrid()->isValuationDate()[k - 1])
384 continue;
385 // set numeraire
386 asd_->set(dateIndex, i, model_->numeraire(0, path[0].time(k), path[0][k]),
387 AggregationScenarioDataType::Numeraire);
388 // set fx spots
389 for (Size j = 0; j < asdCurrencyIndex.size(); ++j) {
390 asd_->set(dateIndex, i, fx(fxBuffer, asdCurrencyIndex[j], k, i),
391 AggregationScenarioDataType::FXSpot, asdCurrencyCode[j]);
392 }
393 // set index fixings
394 Date d = sgd_->getGrid()->dates()[k - 1];
395 for (Size j = 0; j < asdIndex.size(); ++j) {
396 asdIndexCurve[j]->move(d, state(irStateBuffer, asdIndexIndex[j], k, i));
397 asd_->set(dateIndex, i, asdIndex[j]->fixing(d), AggregationScenarioDataType::IndexFixing,
398 asdIndexName[j]);
399 }
400 ++dateIndex;
401 }
402 timer.stop();
403 asdTime += timer.elapsed().wall * 1e-9;
404 }
405 updateProgress(i + 1, outputCube->samples());
406 }
407
408 // Run AmcCalculators implementing interface 2
409
410 LOG("Run simulation (amc calculators implementing interface 2)...");
411 resetProgress();
412
413 // set up vectors indicating valuation times, close-out times and all times
414
415 std::vector<bool> allTimes(pathTimes.size(), true);
416 std::vector<bool> valuationTimes(pathTimes.size()), closeOutTimes(pathTimes.size());
417 for (Size i = 0; i < pathTimes.size(); ++i) {
418 valuationTimes[i] = sgd_->getGrid()->isValuationDate()[i];
419 closeOutTimes[i] = sgd_->getGrid()->isCloseOutDate()[i];
420 }
421
422 // loop over amc calculators, get result and populate cube
423
424 timer.start();
425 for (Size j = 0; j < amcCalculators.size(); ++j) {
426 auto amcCalc = boost::dynamic_pointer_cast<AmcCalculatorMultiVariates>(amcCalculators[j]);
427 if (amcCalc == nullptr)
428 continue;
429 if (!sgd_->withCloseOutLag()) {
430 // no close-out lag, fill depth 0 with npv on path
431 auto res = simulatePathInterface2(amcCalc, pathTimes, paths, allTimes, false, tradeLabel[j]);
432 outputCube->setT0(res[0].at(0) * fx(fxBuffer, currencyIndex[j], 0, 0) *
433 numRatio(model_, irStateBuffer, currencyIndex[j], 0, 0.0, 0) * effectiveMultiplier[j],
434 tradeId[j], 0);
435 for (Size k = 1; k < res.size(); ++k) {
436 Real t = sgd_->getGrid()->timeGrid()[k];
437 for (Size i = 0; i < outputCube->samples(); ++i) {
438 outputCube->set(res[k][i] * fx(fxBuffer, currencyIndex[j], k, i) *
439 numRatio(model_, irStateBuffer, currencyIndex[j], k, t, i) *
440 effectiveMultiplier[j],
441 tradeId[j], k - 1, i, 0);
442 }
443 }
444 } else {
445 // with close-out lag, fill depth 0 with valuation date npvs, depth 1 with (inflated) close-out npvs
446 if (sgd_->withMporStickyDate()) {
447 // sticky date mpor mode. simulate the valuation times...
448 auto res = simulatePathInterface2(amcCalc, pathTimes, paths, valuationTimes, false, tradeLabel[j]);
449 // ... and then the close-out times, but times moved to the valuation times
450 auto resLag = simulatePathInterface2(amcCalc, pathTimes, paths, closeOutTimes, true, tradeLabel[j]);
451 outputCube->setT0(res[0].at(0) * fx(fxBuffer, currencyIndex[j], 0, 0) *
452 numRatio(model_, irStateBuffer, currencyIndex[j], 0, 0.0, 0) *
453 effectiveMultiplier[j],
454 tradeId[j], 0);
455 int dateIndex = -1;
456 for (Size k = 0; k < sgd_->getGrid()->dates().size(); ++k) {
457 Real t = sgd_->getGrid()->timeGrid()[k + 1];
458 Real tm = sgd_->getGrid()->timeGrid()[k];
459 if (sgd_->getGrid()->isCloseOutDate()[k]) {
460 QL_REQUIRE(dateIndex >= 0, "first date in grid must be a valuation date");
461 for (Size i = 0; i < outputCube->samples(); ++i) {
462 outputCube->set(resLag[dateIndex + 1][i] * fx(fxBuffer, currencyIndex[j], k + 1, i) *
463 num(model_, irStateBuffer, currencyIndex[j], k + 1, tm, i) *
464 effectiveMultiplier[j],
465 tradeId[j], dateIndex, i, 1);
466 }
467 }
468 if (sgd_->getGrid()->isValuationDate()[k]) {
469 ++dateIndex;
470 for (Size i = 0; i < outputCube->samples(); ++i) {
471 outputCube->set(res[dateIndex + 1][i] * fx(fxBuffer, currencyIndex[j], k + 1, i) *
472 numRatio(model_, irStateBuffer, currencyIndex[j], k + 1, t, i) *
473 effectiveMultiplier[j],
474 tradeId[j], dateIndex, i, 0);
475 }
476 }
477 }
478 } else {
479 // actual date mpor mode: simulate all times in one go
480 auto res = simulatePathInterface2(amcCalc, pathTimes, paths, allTimes, false, tradeLabel[j]);
481 outputCube->setT0(res[0].at(0) * fx(fxBuffer, currencyIndex[j], 0, 0) *
482 numRatio(model_, irStateBuffer, currencyIndex[j], 0, 0.0, 0) *
483 effectiveMultiplier[j],
484 tradeId[j], 0);
485 int dateIndex = -1;
486 for (Size k = 1; k < res.size(); ++k) {
487 Real t = sgd_->getGrid()->timeGrid()[k];
488 if (sgd_->getGrid()->isCloseOutDate()[k - 1]) {
489 QL_REQUIRE(dateIndex >= 0, "first date in grid must be a valuation date");
490 for (Size i = 0; i < outputCube->samples(); ++i) {
491 outputCube->set(res[k][i] * fx(fxBuffer, currencyIndex[j], k, i) *
492 num(model_, irStateBuffer, currencyIndex[j], k, t, i) *
493 effectiveMultiplier[j],
494 tradeId[j], dateIndex, i, 1);
495 }
496 }
497 if (sgd_->getGrid()->isValuationDate()[k - 1]) {
498 ++dateIndex;
499 for (Size i = 0; i < outputCube->samples(); ++i) {
500 outputCube->set(res[k][i] * fx(fxBuffer, currencyIndex[j], k, i) *
501 numRatio(model_, irStateBuffer, currencyIndex[j], k, t, i) *
502 effectiveMultiplier[j],
503 tradeId[j], dateIndex, i, 0);
504 }
505 }
506 }
507 }
508 }
509 updateProgress(j + 1, amcCalculators.size());
510 }
511 timer.stop();
512 valuationTime += timer.elapsed().wall * 1e-9;
513
514 totalTime = timerTotal.elapsed().wall * 1e-9;
515 residualTime = totalTime - (calibrationTime + pathGenTime + valuationTime + asdTime);
516 LOG("calibration time : " << calibrationTime << " sec");
517 LOG("path generation time : " << pathGenTime << " sec");
518 LOG("valuation time : " << valuationTime << " sec");
519 LOG("asd time : " << asdTime << " sec");
520 LOG("residual time : " << residualTime << " sec");
521 LOG("total time : " << totalTime << " sec");
522 LOG("AMCValuationEngine finished");
523}
