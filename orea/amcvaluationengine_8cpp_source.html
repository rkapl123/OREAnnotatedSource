53Real fx(const std::vector<std::vector<std::vector<Real>>>& fxBuffer, const Size ccyIndex, const Size timeIndex,
54 const Size sample) {
55 if (ccyIndex == 0)
56 return 1.0;
57 return fxBuffer[ccyIndex - 1][timeIndex][sample];
58}
59
60Real state(const std::vector<std::vector<std::vector<Real>>>& irStateBuffer, const Size ccyIndex, const Size timeIndex,
61 const Size sample) {
62 return irStateBuffer[ccyIndex][timeIndex][sample];
63}
64
65Real numRatio(const boost::shared_ptr<CrossAssetModel>& model,
66 const std::vector<std::vector<std::vector<Real>>>& irStateBuffer, const Size ccyIndex,
67 const Size timeIndex, const Real time, const Size sample) {
68 if (ccyIndex == 0)
69 return 1.0;
70 Real state_base = state(irStateBuffer, 0, timeIndex, sample);
71 Real state_curr = state(irStateBuffer, ccyIndex, timeIndex, sample);
72 return model->numeraire(ccyIndex, time, state_curr) / model->numeraire(0, time, state_base);
73}
74
75Real num(const boost::shared_ptr<CrossAssetModel>& model,
76 const std::vector<std::vector<std::vector<Real>>>& irStateBuffer, const Size ccyIndex, const Size timeIndex,
77 const Real time, const Size sample) {
78 Real state_curr = state(irStateBuffer, ccyIndex, timeIndex, sample);
79 return model->numeraire(ccyIndex, time, state_curr);
80}
81
82Real discount(const boost::shared_ptr<CrossAssetModel>& model,
83 const std::vector<std::vector<std::vector<Real>>>& irStateBuffer, const Size ccyIndex,
84 const Size timeIndex, const Real t, const Real T, const Size sample) {
85 Real state_curr = state(irStateBuffer, ccyIndex, timeIndex, sample);
86 return model->discountBond(ccyIndex, t, T, state_curr);
87}

89std::vector<QuantExt::RandomVariable>
90simulatePathInterface2(const boost::shared_ptr<AmcCalculator>& amcCalc, const std::vector<Real>& pathTimes,
91 std::vector<std::vector<RandomVariable>>& paths, const std::vector<bool>& isRelevantTime,
92 const bool moveStateToPreviousTime, const std::string& tradeLabel,
93 const std::string& tradeType) {
94 try {
95 return amcCalc->simulatePath(pathTimes, paths, isRelevantTime, moveStateToPreviousTime);
96 } catch (const std::exception& e) {
97 ALOG(StructuredTradeErrorMessage(tradeLabel, tradeType, "error during amc path simulation for trade.",
98 e.what()));
99 return std::vector<QuantExt::RandomVariable>(std::count(isRelevantTime.begin(), isRelevantTime.end(), true) + 1,
100 RandomVariable(paths.front().front().size()));
101 }
102}

104std::vector<QuantExt::RandomVariable>
105feeContributions(const Size j, const boost::shared_ptr<ScenarioGeneratorData>& sgd, const Date& asof,
106 const Size samples, const std::vector<std::vector<std::tuple<Size, Real, QuantLib::Date>>>& tradeFees,
107 const boost::shared_ptr<CrossAssetModel>& model,
108 const std::vector<std::vector<std::vector<Real>>>& fxBuffer,
109 const std::vector<std::vector<std::vector<Real>>>& irStateBuffer) {
110 std::vector<QuantExt::RandomVariable> result;
111 for (Size k = 0; k < sgd->getGrid()->timeGrid().size(); ++k) {
112 Date simDate = k == 0 ? asof : sgd->getGrid()->dates()[k - 1];
113 // slight approximation: we treat premiums as seen from the closeout date the same as if priced from the
114 // the valuation date in sticky date mode with mpor grid
115 if (k == 0 || !sgd->withCloseOutLag() || !sgd->withMporStickyDate() ||
116 sgd->getGrid()->isValuationDate()[k - 1]) {
117 result.push_back(RandomVariable(samples, 0.0));
118 if (tradeFees[j].empty())
119 continue;
120 for (Size i = 0; i < samples; ++i) {
121 Real tmp = 0.0;
122 for (Size f = 0; f < tradeFees[j].size(); ++f) {
123 if (std::get<2>(tradeFees[j][f]) > simDate) {
124 Real t = sgd->getGrid()->timeGrid()[k];
125 Real T = model->irModel(0)->termStructure()->timeFromReference(std::get<2>(tradeFees[j][f]));
126 tmp += std::get<1>(tradeFees[j][f]) * fx(fxBuffer, std::get<0>(tradeFees[j][f]), k, i) *
127 discount(model, irStateBuffer, std::get<0>(tradeFees[j][f]), k, t, T, i) *
128 num(model, irStateBuffer, 0, k, t, i);
129 }
130 }
131 result.back().set(i, tmp);
132 }
133 }
134 }
135 return result;
136}

138void runCoreEngine(const boost::shared_ptr<ore::data::Portfolio>& portfolio,
139 const boost::shared_ptr<QuantExt::CrossAssetModel>& model,
140 const boost::shared_ptr<ore::data::Market>& market,
141 const boost::shared_ptr<ore::analytics::ScenarioGeneratorData>& sgd,
142 const std::vector<string>& aggDataIndices, const std::vector<string>& aggDataCurrencies,
143 boost::shared_ptr<ore::analytics::AggregationScenarioData> asd,
144 boost::shared_ptr<NPVCube> outputCube, boost::shared_ptr<ProgressIndicator> progressIndicator) {
145
146 progressIndicator->updateProgress(0, portfolio->size() + 1);
147
148 // base currency is the base currency of the cam
149
150 Currency baseCurrency = model->irlgm1f(0)->currency();
151
152 // timings
153
154 boost::timer::cpu_timer timer, timerTotal;
155 Real calibrationTime = 0.0, valuationTime = 0.0, asdTime = 0.0, bufferTime = 0.0, pathGenTime = 0.0, residualTime,
156 totalTime;
157 timerTotal.start();
158
159 // prepare for asd writing
160
161 std::vector<Size> asdCurrencyIndex; // FX Spots
162 std::vector<string> asdCurrencyCode;
163 std::vector<boost::shared_ptr<LgmImpliedYtsFwdFwdCorrected>> asdIndexCurve; // Ibor Indices
164 std::vector<boost::shared_ptr<Index>> asdIndex;
165 std::vector<Size> asdIndexIndex;
166 std::vector<string> asdIndexName;
167 if (asd != nullptr) {
168 LOG("Collect information for aggregation scenario data...");
169 // fx spots
170 for (auto const& c : aggDataCurrencies) {
171 Currency cur = parseCurrency(c);
172 if (cur == baseCurrency)
173 continue;
174 Size ccyIndex = model->ccyIndex(cur);
175 asdCurrencyIndex.push_back(ccyIndex);
176 asdCurrencyCode.push_back(c);
177 }
178 // ibor indices
179 for (auto const& i : aggDataIndices) {
180 boost::shared_ptr<IborIndex> tmp;
181 try {
182 tmp = *market->iborIndex(i);
183 } catch (const std::exception& e) {
184 ALOG("index \"" << i << "\" not found in market, skipping. (" << e.what() << ")");
185 }
186 Size ccyIndex = model->ccyIndex(tmp->currency());
187 asdIndexCurve.push_back(
188 boost::make_shared<LgmImpliedYtsFwdFwdCorrected>(model->lgm(ccyIndex), tmp->forwardingTermStructure()));
189 asdIndex.push_back(tmp->clone(Handle<YieldTermStructure>(asdIndexCurve.back())));
190 asdIndexIndex.push_back(ccyIndex);
191 asdIndexName.push_back(i);
192 }
193 } else {
194 LOG("No asd object set, won't write aggregation scenario data...");
195 }
196
197 // extract AMC calculators, fees and some other infos we need from the ore wrapper
198
199 LOG("Extract AMC Calculators...");
200 std::vector<boost::shared_ptr<AmcCalculator>> amcCalculators;
201 std::vector<Size> tradeId;
202 std::vector<std::string> tradeLabel, tradeType;
203 std::vector<Real> effectiveMultiplier;
204 std::vector<Size> currencyIndex;
205 std::vector<std::vector<std::tuple<Size, Real, QuantLib::Date>>> tradeFees;
206 timer.start();
207 Size progressCounter = 0;
208
209 auto extractAmcCalculator = [&amcCalculators, &tradeId, &tradeLabel, &tradeType, &effectiveMultiplier,
210 &currencyIndex, &tradeFees, &model, &outputCube](
211 const std::pair<std::string, boost::shared_ptr<Trade>>& trade,
212 boost::shared_ptr<AmcCalculator> amcCalc, Real multiplier) {
213 LOG("AMCCalculator extracted for \"" << trade.first << "\"");
214 amcCalculators.push_back(amcCalc);
215 effectiveMultiplier.push_back(multiplier);
216 currencyIndex.push_back(model->ccyIndex(amcCalc->npvCurrency()));
217
218 if (auto id = outputCube->idsAndIndexes().find(trade.first); id != outputCube->idsAndIndexes().end()) {
219 tradeId.push_back(id->second);
220 } else {
221 QL_FAIL("AMCValuationEngine: trade id '" << trade.first
222 << "' is not present in output cube - internal error.");
223 }
224
225 tradeLabel.push_back(trade.first);
226 tradeType.push_back(trade.second->tradeType());
227 tradeFees.push_back({});
228 for (Size i = 0; i < trade.second->instrument()->additionalInstruments().size(); ++i) {
229 if (auto p = boost::dynamic_pointer_cast<QuantExt::Payment>(
230 trade.second->instrument()->additionalInstruments()[i])) {
231 tradeFees.back().push_back(std::make_tuple(model->ccyIndex(p->currency()), p->cashFlow()->amount(),
232 p->cashFlow()->date()));
233 } else {
234 ALOG(StructuredTradeErrorMessage(trade.second, "Additional instrument is ignored in AMC simulation",
235 "only QuantExt::Payment is handled as additional instrument."));
236 }
237 }
238 };
239 for (auto const& trade : portfolio->trades()) {
240 boost::shared_ptr<AmcCalculator> amcCalc;
241 try {
242 auto inst = trade.second->instrument()->qlInstrument(true);
243 Real multiplier = trade.second->instrument()->multiplier() *
244 trade.second->instrument()->multiplier2();
245 if (auto cInst = boost::dynamic_pointer_cast<CompositeInstrument>(inst)) {
246 auto addResults = cInst->additionalResults();
247 std::vector<Real> multipliers;
248 while(true) {
249 std::stringstream ss;
250 ss << multipliers.size() + 1 << "_multiplier";
251 if (addResults.find(ss.str()) == addResults.end())
252 break;
253 multipliers.push_back(inst->result<Real>(ss.str()));
254 }
255 for (Size cmpIdx = 0; cmpIdx < multipliers.size(); ++cmpIdx) {
256 std::stringstream ss;
257 ss << cmpIdx + 1 << "_amcCalculator";
258 if (addResults.find(ss.str()) != addResults.end()) {
259 amcCalc = inst->result<boost::shared_ptr<AmcCalculator>>(ss.str());
260 extractAmcCalculator(trade, amcCalc, multiplier * multipliers[cmpIdx]);
261 }
262 }
263 continue;
264 }
265 amcCalc = inst->result<boost::shared_ptr<AmcCalculator>>(
266 "amcCalculator");
267 extractAmcCalculator(trade, amcCalc, multiplier);
268 } catch (const std::exception& e) {
269 ALOG(StructuredTradeErrorMessage(trade.second, "Error building trade for AMC simulation", e.what()));
270 }
271 progressIndicator->updateProgress(++progressCounter, portfolio->size() + 1);
272 }
273 timer.stop();
274 calibrationTime += timer.elapsed().wall * 1e-9;
275 LOG("Extracted " << amcCalculators.size() << " AMCCalculators for " << portfolio->size() << " source trades");
276
277 // set up buffers for fx rates and ir states that we need below for the runs against interface 1 and 2
278 // we set these buffers up on the full grid (i.e. valuation + close-out dates, also including the T0 date)
279
280 std::vector<std::vector<std::vector<Real>>> fxBuffer(
281 model->components(CrossAssetModel::AssetType::FX),
282 std::vector<std::vector<Real>>(sgd->getGrid()->dates().size() + 1, std::vector<Real>(outputCube->samples())));
283 std::vector<std::vector<std::vector<Real>>> irStateBuffer(
284 model->components(CrossAssetModel::AssetType::IR),
285 std::vector<std::vector<Real>>(sgd->getGrid()->dates().size() + 1, std::vector<Real>(outputCube->samples())));
286
287 // set up cache for paths
288
289 auto process = model->stateProcess();
290 Size nStates = process->size();
291 QL_REQUIRE(sgd->getGrid()->timeGrid().size() > 0, "AMCValuationEngine: empty time grid given");
292 std::vector<Real> pathTimes(std::next(sgd->getGrid()->timeGrid().begin(), 1), sgd->getGrid()->timeGrid().end());
293 std::vector<std::vector<RandomVariable>> paths(
294 pathTimes.size(), std::vector<RandomVariable>(nStates, RandomVariable(outputCube->samples())));
295
296 // fill fx buffer, ir state buffer and write ASD
297
298 auto pathGenerator = makeMultiPathGenerator(sgd->sequenceType(), process, sgd->getGrid()->timeGrid(), sgd->seed(),
299 sgd->ordering(), sgd->directionIntegers());
300
301 LOG("Write ASD, fill internal fx and irState buffers...");
302
303 for (Size i = 0; i < outputCube->samples(); ++i) {
304 timer.start();
305 const auto& path = pathGenerator->next().value;
306 timer.stop();
307 pathGenTime += timer.elapsed().wall * 1e-9;
308
309 // populate fx and ir state buffers, populate cached paths for interface 2
310
311 timer.start();
312 for (Size k = 0; k < fxBuffer.size(); ++k) {
313 for (Size j = 0; j < sgd->getGrid()->timeGrid().size(); ++j) {
314 fxBuffer[k][j][i] = std::exp(path[model->pIdx(CrossAssetModel::AssetType::FX, k)][j]);
315 }
316 }
317 for (Size k = 0; k < irStateBuffer.size(); ++k) {
318 for (Size j = 0; j < sgd->getGrid()->timeGrid().size(); ++j) {
319 irStateBuffer[k][j][i] = path[model->pIdx(CrossAssetModel::AssetType::IR, k)][j];
320 }
321 }
322
323 for (Size k = 0; k < nStates; ++k) {
324 for (Size j = 0; j < pathTimes.size(); ++j) {
325 paths[j][k].set(i, path[k][j + 1]);
326 }
327 }
328 timer.stop();
329 bufferTime += timer.elapsed().wall * 1e-9;
330
331 // write aggregation scenario data, TODO this seems relatively slow, can we speed it up using LgmVectorised
332
333 if (asd != nullptr) {
334 timer.start();
335 Size dateIndex = 0;
336 for (Size k = 1; k < sgd->getGrid()->timeGrid().size(); ++k) {
337 // only write asd on valuation dates
338 if (!sgd->getGrid()->isValuationDate()[k - 1])
339 continue;
340 // set numeraire
341 asd->set(dateIndex, i, model->numeraire(0, path[0].time(k), path[0][k]),
342 AggregationScenarioDataType::Numeraire);
343 // set fx spots
344 for (Size j = 0; j < asdCurrencyIndex.size(); ++j) {
345 asd->set(dateIndex, i, fx(fxBuffer, asdCurrencyIndex[j], k, i), AggregationScenarioDataType::FXSpot,
346 asdCurrencyCode[j]);
347 }
348 // set index fixings
349 Date d = sgd->getGrid()->dates()[k - 1];
350 for (Size j = 0; j < asdIndex.size(); ++j) {
351 asdIndexCurve[j]->move(d, state(irStateBuffer, asdIndexIndex[j], k, i));
352 auto index = asdIndex[j];
353 if (auto fb = boost::dynamic_pointer_cast<FallbackIborIndex>(asdIndex[j])) {
354 // proxy fallback ibor index by its rfr index's fixing
355 index = fb->rfrIndex();
356 }
357 asd->set(dateIndex, i, index->fixing(index->fixingCalendar().adjust(d)),
358 AggregationScenarioDataType::IndexFixing, asdIndexName[j]);
359 }
360 ++dateIndex;
361 }
362 timer.stop();
363 asdTime += timer.elapsed().wall * 1e-9;
364 }
365 }
366
367 // Run AmcCalculators
368
369 LOG("Run simulation...");
370
371 // set up vectors indicating valuation times, close-out times and all times
372
373 std::vector<bool> allTimes(pathTimes.size(), true);
374 std::vector<bool> valuationTimes(pathTimes.size()), closeOutTimes(pathTimes.size());
375 for (Size i = 0; i < pathTimes.size(); ++i) {
376 valuationTimes[i] = sgd->getGrid()->isValuationDate()[i];
377 closeOutTimes[i] = sgd->getGrid()->isCloseOutDate()[i];
378 }
379
380 // loop over amc calculators, get result and populate cube
381
382 timer.start();
383 for (Size j = 0; j < amcCalculators.size(); ++j) {
384 auto resFee = feeContributions(j, sgd, model->irModel(0)->termStructure()->referenceDate(),
385 outputCube->samples(), tradeFees, model, fxBuffer, irStateBuffer);
386
387 if (!sgd->withCloseOutLag()) {
388 // no close-out lag, fill depth 0 with npv on path
389 auto res = simulatePathInterface2(amcCalculators[j], pathTimes, paths, allTimes, false, tradeLabel[j],
390 tradeType[j]);
391 Real v = outputCube->getT0(tradeId[j], 0);
392 outputCube->setT0(v + res[0].at(0) * fx(fxBuffer, currencyIndex[j], 0, 0) *
393 numRatio(model, irStateBuffer, currencyIndex[j], 0, 0.0, 0) *
394 effectiveMultiplier[j] +
395 resFee[0][0],
396 tradeId[j], 0);
397 for (Size k = 1; k < res.size(); ++k) {
398 Real t = sgd->getGrid()->timeGrid()[k];
399 for (Size i = 0; i < outputCube->samples(); ++i) {
400 Real v = outputCube->get(tradeId[j], k - 1, i, 0);
401 outputCube->set(v + res[k][i] * fx(fxBuffer, currencyIndex[j], k, i) *
402 numRatio(model, irStateBuffer, currencyIndex[j], k, t, i) *
403 effectiveMultiplier[j] +
404 resFee[k][i],
405 tradeId[j], k - 1, i, 0);
406 }
407 }
408 } else {
409 // with close-out lag, fill depth 0 with valuation date npvs, depth 1 with (inflated) close-out npvs
410 if (sgd->withMporStickyDate()) {
411 // sticky date mpor mode. simulate the valuation times...
412 auto res = simulatePathInterface2(amcCalculators[j], pathTimes, paths, valuationTimes, false,
413 tradeLabel[j], tradeType[j]);
414 // ... and then the close-out times, but times moved to the valuation times
415 auto resLag = simulatePathInterface2(amcCalculators[j], pathTimes, paths, closeOutTimes, true,
416 tradeLabel[j], tradeType[j]);
417 Real v = outputCube->getT0(tradeId[j], 0);
418 outputCube->setT0(v + res[0].at(0) * fx(fxBuffer, currencyIndex[j], 0, 0) *
419 numRatio(model, irStateBuffer, currencyIndex[j], 0, 0.0, 0) *
420 effectiveMultiplier[j] +
421 resFee[0][0],
422 tradeId[j], 0);
423 int dateIndex = -1;
424 for (Size k = 0; k < sgd->getGrid()->dates().size(); ++k) {
425 Real t = sgd->getGrid()->timeGrid()[k + 1];
426 Real tm = sgd->getGrid()->timeGrid()[k];
427 if (sgd->getGrid()->isCloseOutDate()[k]) {
428 QL_REQUIRE(dateIndex >= 0, "first date in grid must be a valuation date");
429 for (Size i = 0; i < outputCube->samples(); ++i) {
430 Real v = outputCube->get(tradeId[j], dateIndex, i, 1);
431 outputCube->set(v + resLag[dateIndex + 1][i] * fx(fxBuffer, currencyIndex[j], k + 1, i) *
432 num(model, irStateBuffer, currencyIndex[j], k + 1, tm, i) *
433 effectiveMultiplier[j] +
434 resFee[dateIndex + 1][i],
435 tradeId[j], dateIndex, i, 1);
436 }
437 }
438 if (sgd->getGrid()->isValuationDate()[k]) {
439 ++dateIndex;
440 for (Size i = 0; i < outputCube->samples(); ++i) {
441 Real v = outputCube->get(tradeId[j], dateIndex, i, 1);
442 outputCube->set(v + res[dateIndex + 1][i] * fx(fxBuffer, currencyIndex[j], k + 1, i) *
443 numRatio(model, irStateBuffer, currencyIndex[j], k + 1, t, i) *
444 effectiveMultiplier[j] +
445 resFee[dateIndex + 1][i],
446 tradeId[j], dateIndex, i, 0);
447 }
448 }
449 }
450 } else {
451 // actual date mpor mode: simulate all times in one go
452 auto res = simulatePathInterface2(amcCalculators[j], pathTimes, paths, allTimes, false, tradeLabel[j],
453 tradeType[j]);
454 Real v = outputCube->getT0(tradeId[j], 0);
455 outputCube->setT0(v + res[0].at(0) * fx(fxBuffer, currencyIndex[j], 0, 0) *
456 numRatio(model, irStateBuffer, currencyIndex[j], 0, 0.0, 0) *
457 effectiveMultiplier[j],
458 tradeId[j], 0);
459 int dateIndex = -1;
460 for (Size k = 1; k < res.size(); ++k) {
461 Real t = sgd->getGrid()->timeGrid()[k];
462 if (sgd->getGrid()->isCloseOutDate()[k - 1]) {
463 QL_REQUIRE(dateIndex >= 0, "first date in grid must be a valuation date");
464 for (Size i = 0; i < outputCube->samples(); ++i) {
465 Real v = outputCube->get(tradeId[j], dateIndex, i, 1);
466 outputCube->set(v + res[k][i] * fx(fxBuffer, currencyIndex[j], k, i) *
467 num(model, irStateBuffer, currencyIndex[j], k, t, i) *
468 effectiveMultiplier[j] +
469 resFee[k][i],
470 tradeId[j], dateIndex, i, 1);
471 }
472 }
473 if (sgd->getGrid()->isValuationDate()[k - 1]) {
474 ++dateIndex;
475 for (Size i = 0; i < outputCube->samples(); ++i) {
476 Real v = outputCube->get(tradeId[j], dateIndex, i, 0);
477 outputCube->set(v + res[k][i] * fx(fxBuffer, currencyIndex[j], k, i) *
478 numRatio(model, irStateBuffer, currencyIndex[j], k, t, i) *
479 effectiveMultiplier[j] +
480 resFee[k][i],
481 tradeId[j], dateIndex, i, 0);
482 }
483 }
484 }
485 }
486 }
487 progressIndicator->updateProgress(++progressCounter, portfolio->size() + 1);
488 }
489 timer.stop();
490 valuationTime += timer.elapsed().wall * 1e-9;
491
492 totalTime = timerTotal.elapsed().wall * 1e-9;
493 residualTime = totalTime - (calibrationTime + pathGenTime + valuationTime + asdTime + bufferTime);
494 LOG("calibration time : " << calibrationTime << " sec");
495 LOG("asd time : " << asdTime << " sec");
496 LOG("buffer time : " << bufferTime << " sec");
497 LOG("path generation time : " << pathGenTime << " sec");
498 LOG("valuation time : " << valuationTime << " sec");
499 LOG("residual time : " << residualTime << " sec");
500 LOG("total time : " << totalTime << " sec");
501 LOG("AMCValuationEngine finished for one of possibly multiple threads.");
502} // runCoreEngine()

506AMCValuationEngine::AMCValuationEngine(
507 const QuantLib::Size nThreads, const QuantLib::Date& today, const QuantLib::Size nSamples,
508 const boost::shared_ptr<ore::data::Loader>& loader,
509 const boost::shared_ptr<ScenarioGeneratorData>& scenarioGeneratorData, const std::vector<string>& aggDataIndices,
510 const std::vector<string>& aggDataCurrencies, const boost::shared_ptr<CrossAssetModelData>& crossAssetModelData,
511 const boost::shared_ptr<ore::data::EngineData>& engineData,
512 const boost::shared_ptr<ore::data::CurveConfigurations>& curveConfigs,
513 const boost::shared_ptr<ore::data::TodaysMarketParameters>& todaysMarketParams,
514 const std::string& configurationLgmCalibration, const std::string& configurationFxCalibration,
515 const std::string& configurationEqCalibration, const std::string& configurationInfCalibration,
516 const std::string& configurationCrCalibration, const std::string& configurationFinalModel,
517 const boost::shared_ptr<ore::data::ReferenceDataManager>& referenceData,
518 const ore::data::IborFallbackConfig& iborFallbackConfig, const bool handlePseudoCurrenciesTodaysMarket,
519 const std::function<boost::shared_ptr<ore::analytics::NPVCube>(const QuantLib::Date&, const std::set<std::string>&,

599void AMCValuationEngine::buildCube(const boost::shared_ptr<ore::data::Portfolio>& portfolio) {
600 LOG("Starting multi-threaded AMCValuationEngine for "
601 << portfolio->size() << " trades, " << nSamples_ << " samples and " << scenarioGeneratorData_->getGrid()->size()
602 << " dates.");
603
604 QL_REQUIRE(useMultithreading_, "AMCValuationEngine::buildCube() method was called with signature for "
605 "multi-threaded run, but engine was constructed for single-threaded runs");
606
607 QL_REQUIRE(portfolio->size() > 0, "AMCValuationEngine::buildCube: empty portfolio");
608
609 // split portfolio into nThreads parts (just distribute the trades assuming all are approximately expensive)
610
611 LOG("Splitting portfolio.");
612
613 Size eff_nThreads = std::min(portfolio->size(), nThreads_);
614
615 LOG("portfolio size = " << portfolio->size());
616 LOG("nThreads = " << nThreads_);
617 LOG("eff nThreads = " << eff_nThreads);
618
619 QL_REQUIRE(eff_nThreads > 0, "effective threads are zero, this is not allowed.");
620
621 std::vector<boost::shared_ptr<ore::data::Portfolio>> portfolios;
622 for (Size i = 0; i < eff_nThreads; ++i)
623 portfolios.push_back(boost::make_shared<ore::data::Portfolio>());
624
625 Size portfolioIndex = 0;
626 for (auto const& t : portfolio->trades()) {
627 portfolios[portfolioIndex]->add(t.second);
628 if (++portfolioIndex >= eff_nThreads)
629 portfolioIndex = 0;
630 }
631
632 // output the portfolios into strings so that the worker threads can load them from there
633
634 std::vector<std::string> portfoliosAsString;
635 for (auto const& p : portfolios) {
636 portfoliosAsString.emplace_back(p->toXMLString());
637 }
638
639 // log info on the portfolio split
640
641 for (Size i = 0; i < eff_nThreads; ++i) {
642 LOG("Portfolio #" << i << " number of trades : " << portfolios[i]->size());
643 }
644
645 // build loaders for each thread as clones of the original one
646
647 LOG("Cloning loaders for " << eff_nThreads << " threads...");
648 std::vector<boost::shared_ptr<ore::data::ClonedLoader>> loaders;
649 for (Size i = 0; i < eff_nThreads; ++i)
650 loaders.push_back(boost::make_shared<ore::data::ClonedLoader>(today_, loader_));
651
652 // build nThreads mini-cubes to which each thread writes its results
653
654 LOG("Build " << eff_nThreads << " mini result cubes...");
655 miniCubes_.clear();
656 for (Size i = 0; i < eff_nThreads; ++i) {
657 miniCubes_.push_back(
658 cubeFactory_(today_, portfolios[i]->ids(), scenarioGeneratorData_->getGrid()->valuationDates(), nSamples_));
659 }
660
661 // precompute sim dates
662
663 std::vector<Date> simDates =
664 scenarioGeneratorData_->withCloseOutLag() && !scenarioGeneratorData_->withMporStickyDate()
665 ? scenarioGeneratorData_->getGrid()->dates()
666 : scenarioGeneratorData_->getGrid()->valuationDates();
667
668 // build progress indicator consolidating the results from the threads
669
670 auto progressIndicator =
671 boost::make_shared<ore::analytics::MultiThreadedProgressIndicator>(this->progressIndicators());
672
673 // create the thread pool with eff_nThreads and queue size = eff_nThreads as well
674
675 // LOG("Create thread pool with " << eff_nThreads);
676 // ctpl::thread_pool threadPool(eff_nThreads);
677
678 // create the jobs and push them to the pool
679
680 using resultType = int;
681 std::vector<std::future<resultType>> results(eff_nThreads);
682
683 std::vector<std::thread> jobs; // not needed if thread pool is used
684
685 // get obs mode of main thread, so that we can set this mode in the worker threads below
686
687 ore::analytics::ObservationMode::Mode obsMode = ore::analytics::ObservationMode::instance().mode();
688
689 for (Size i = 0; i < eff_nThreads; ++i) {
690
691 auto job = [this, obsMode, &portfoliosAsString, &loaders, &simDates, &progressIndicator](int id) -> resultType {
692 // set thread local singletons
693
694 QuantLib::Settings::instance().evaluationDate() = today_;
695 ore::analytics::ObservationMode::instance().setMode(obsMode);
696
697 LOG("Start thread " << id);
698
699 int rc;
700
701 try {
702
703 // build todays market using cloned market data
704
705 boost::shared_ptr<ore::data::Market> initMarket = boost::make_shared<ore::data::TodaysMarket>(
706 today_, todaysMarketParams_, loaders[id], curveConfigs_, true, true, true, referenceData_, false,
707 iborFallbackConfig_, false, handlePseudoCurrenciesTodaysMarket_);
708
709 // build cam
710
711 ore::data::CrossAssetModelBuilder modelBuilder(
712 initMarket, crossAssetModelData_, configurationLgmCalibration_, configurationFxCalibration_,
713 configurationEqCalibration_, configurationInfCalibration_, configurationCrCalibration_,
714 configurationFinalModel_, false, true);
715
716 auto cam = *modelBuilder.model();
717
718 // build portfolio against init market
719
720 auto portfolio = boost::make_shared<ore::data::Portfolio>();
721 portfolio->fromXMLString(portfoliosAsString[id]);
722
723 boost::shared_ptr<EngineData> edCopy = boost::make_shared<EngineData>(*engineData_);
724 edCopy->globalParameters()["GenerateAdditionalResults"] = "false";
725 edCopy->globalParameters()["RunType"] = "NPV";
726 map<MarketContext, string> configurations{{MarketContext::irCalibration, configurationLgmCalibration_},
727 {MarketContext::fxCalibration, configurationFxCalibration_},
728 {MarketContext::pricing, configurationFinalModel_}};
729
730 auto engineFactory = boost::make_shared<EngineFactory>(
731 edCopy, initMarket, configurations, referenceData_, iborFallbackConfig_,
732 EngineBuilderFactory::instance().generateAmcEngineBuilders(cam, simDates), true);
733
734 portfolio->build(engineFactory, "amc-val-engine", true);
735
736 // run core engine code (asd is written for thread id 0 only)
737
738 runCoreEngine(portfolio, cam, initMarket, scenarioGeneratorData_, aggDataIndices_, aggDataCurrencies_,
739 id == 0 ? asd_ : nullptr, miniCubes_[id], progressIndicator);
740
741 // return code 0 = ok
742
743 LOG("Thread " << id << " successfully finished.");
744
745 rc = 0;
746
747 } catch (const std::exception& e) {
748
749 // log error and return code 1 = not ok
750
751 ALOG(ore::analytics::StructuredAnalyticsErrorMessage("AMC Valuation Engine (multithreaded mode)",
752 "",
753 e.what()));
754 rc = 1;
755 }
756
757 // exit
758
759 return rc;
760 };
761
762 // results[i] = threadPool.push(job);
763
764 // not needed if thread pool is used
765 std::packaged_task<resultType(int)> task(job);
766 results[i] = task.get_future();
767 std::thread thread(std::move(task), i);
768 jobs.emplace_back(std::move(thread));
769 }
770
771 // not needed if thread pool is used
772 for (auto& t : jobs)
773 t.join();
774
775 for (Size i = 0; i < results.size(); ++i) {
776 results[i].wait();
777 }
778
779 for (Size i = 0; i < results.size(); ++i) {
780 QL_REQUIRE(results[i].valid(), "internal error: did not get a valid result");
781 int rc = results[i].get();
782 QL_REQUIRE(rc == 0, "error: thread " << i << " exited with return code " << rc
783 << ". Check for structured errors from 'AMCValuationEngine'.");
784 }
785
786 // stop the thread pool, wait for unfinished jobs
787
788 // LOG("Stop thread pool");
789 // threadPool.stop(true);
790
791 LOG("Finished multi-threaded AMCValuationEngine run.");
792}
