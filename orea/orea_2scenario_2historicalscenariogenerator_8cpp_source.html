65Real ReturnConfiguration::returnValue(const RiskFactorKey& key, const Real v1, const Real v2, const QuantLib::Date& d1,
66 const QuantLib::Date& d2) const {
67
68 // Checks
69 check(key);
70
71 // Calculate the return
72 auto keyType = key.keytype;
73 switch (returnType_.at(keyType)) {
74 case ReturnConfiguration::ReturnType::Absolute:
75 return v2 - v1;
76 break;
77 case ReturnConfiguration::ReturnType::Relative:
78 if (!close(v1, 0)) {
79 return v2 / v1 - 1.0;
80 } else {
81 ALOG("Cannot calculate the relative return for key " << key << " so just returning 0: (" << d1 << "," << v1
82 << ") to (" << d2 << "," << v2 << ")");
83 return 0.0;
84 }
85 break;
86 case ReturnConfiguration::ReturnType::Log:
87 if (!close(v1, 0) && v2 / v1 > 0) {
88 return std::log(v2 / v1);
89 } else {
90 ALOG("Cannot calculate the relative return for key " << key << " so just returning 0: (" << d1 << "," << v1
91 << ") to (" << d2 << "," << v2 << ")");
92 return 0.0;
93 }
94 break;
95 default:
96 QL_FAIL("ReturnConfiguration: return type not covered for key " << key << ".");
97 }
98}

100Real ReturnConfiguration::applyReturn(const RiskFactorKey& key, const Real baseValue, const Real returnValue) const {
101
102 // Checks
103 check(key);
104
105 // Apply the return to the base value to generate the return value
106 auto keyType = key.keytype;
107 Real value;
108 switch (returnType_.at(keyType)) {
109 case ReturnConfiguration::ReturnType::Absolute:
110 value = baseValue + returnValue;
111 break;
112 case ReturnConfiguration::ReturnType::Relative:
113 value = baseValue * (1.0 + returnValue);
114 break;
115 case ReturnConfiguration::ReturnType::Log:
116 value = baseValue * std::exp(returnValue);
117 break;
118 default:
119 QL_FAIL("ReturnConfiguration: return type for key " << key << " not covered");
120 }
121
122 // Apply cap / floors to guarantee admissable values
123 if ((keyType == RiskFactorKey::KeyType::BaseCorrelation || keyType == RiskFactorKey::KeyType::Correlation) &&
124 (value > 1.0 || value < -1.0)) {
125 DLOG("Base correlation value, " << value << ", is not in range [-1.0, 1.0]");
126 value = std::max(std::min(value, 1.0), -1.0);
127 DLOG("Base correlation value amended to " << value);
128 }
129
130 if ((keyType == RiskFactorKey::KeyType::RecoveryRate || keyType == RiskFactorKey::KeyType::SurvivalProbability)
131 && (value > 1.0 || value < 0.0)) {
132 DLOG("Value of risk factor " << key << ", " << value << ", is not in range [0.0, 1.0]");
133 value = std::max(std::min(value, 1.0), 0.0);
134 DLOG("Value of risk factor " << key << " amended to " << value);
135 }
136
137 return value;
138}

165HistoricalScenarioGenerator::HistoricalScenarioGenerator(
166 const boost::shared_ptr<HistoricalScenarioLoader>& historicalScenarioLoader,
167 const boost::shared_ptr<ScenarioFactory>& scenarioFactory, const QuantLib::Calendar& cal,
168 const boost::shared_ptr<ore::data::AdjustmentFactors>& adjFactors, const Size mporDays,
169 const bool overlapping, const ReturnConfiguration& returnConfiguration,
170 const std::string& labelPrefix)
171 : i_(0), historicalScenarioLoader_(historicalScenarioLoader), scenarioFactory_(scenarioFactory),
172 cal_(cal), mporDays_(mporDays), adjFactors_(adjFactors), overlapping_(overlapping),
173 returnConfiguration_(returnConfiguration), labelPrefix_(labelPrefix) {
174
175 QL_REQUIRE(mporDays > 0, "Invalid mpor days of 0");
176
177 // check they are in order and all before the base scenario
178 // we do not make any assumptions here about the length
179 for (Size i = 1; i < historicalScenarioLoader_->numScenarios(); ++i) {
180 QL_REQUIRE(historicalScenarioLoader_->dates()[i] > historicalScenarioLoader_->dates()[i - 1],
181 "historical scenarios are not ordered");
182 }
183 setDates();
184}

186void HistoricalScenarioGenerator::setDates() {
187 // construct the vectors of start and end dates
188 for (Size i = 0; i < historicalScenarioLoader_->numScenarios();) {
189 Date sDate = historicalScenarioLoader_->dates()[i];
190 Date eDate = cal_.advance(sDate, mporDays_ * Days);
191 auto it =
192 std::find(historicalScenarioLoader_->dates().begin(), historicalScenarioLoader_->dates().end(), eDate);
193 if (it != historicalScenarioLoader_->dates().end()) {
194 startDates_.push_back(sDate);
195 endDates_.push_back(eDate);
196 }
197 if (overlapping_) {
198 ++i;
199 } else {
200 if (it != historicalScenarioLoader_->dates().end()) {
201 i = std::distance(historicalScenarioLoader_->dates().begin(), it);
202 } else {
203 i = std::distance(historicalScenarioLoader_->dates().begin(),
204 std::upper_bound(historicalScenarioLoader_->dates().begin(),
205 historicalScenarioLoader_->dates().end(), eDate));
206 }
207 }
208 }

211std::pair<boost::shared_ptr<Scenario>, boost::shared_ptr<Scenario>> HistoricalScenarioGenerator::scenarioPair() {
212 // Get the two historicals we are using
213 QL_REQUIRE(i_ < numScenarios(),
214 "Cannot generate any more scenarios (i=" << i_ << " numScenarios=" << numScenarios() << ")");
215 boost::shared_ptr<Scenario> s1 = historicalScenarioLoader_->getHistoricalScenario(startDates_[i_]);
216 boost::shared_ptr<Scenario> s2 = historicalScenarioLoader_->getHistoricalScenario(endDates_[i_]);
217 return std::pair<boost::shared_ptr<Scenario>, boost::shared_ptr<Scenario>>(s1, s2);

230boost::shared_ptr<Scenario> HistoricalScenarioGenerator::next(const Date& d) {
231
232 QL_REQUIRE(baseScenario_ != nullptr, "HistoricalScenarioGenerator: base scenario not set");
233
234 std::pair<boost::shared_ptr<Scenario>, boost::shared_ptr<Scenario>> scens = scenarioPair();
235 boost::shared_ptr<Scenario> s1 = scens.first;
236 boost::shared_ptr<Scenario> s2 = scens.second;
237
238 // build the scenarios
239 QL_REQUIRE(d >= baseScenario_->asof(), "Cannot generate a scenario in the past");
240 boost::shared_ptr<Scenario> scen = scenarioFactory_->buildScenario(d, "", 1.0);
241
242 // loop over all keys
243 calculationDetails_.resize(baseScenario_->keys().size());
244 Size calcDetailsCounter = 0;
245 for (auto const& key : baseScenario_->keys()) {
246 Real base = baseScenario_->get(key);
247 Real v1 = 1.0, v2 = 1.0;
248 if (!s1->has(key) || !s2->has(key)) {
249 DLOG("Missing key in historical scenario (" << io::iso_date(s1->asof()) << "," << io::iso_date(s2->asof())
250 << "): " << key << " => no move in this factor");
251 } else {
252 v1 = adjustedPrice(key, s1->asof(), s1->get(key));
253 v2 = adjustedPrice(key, s2->asof(), s2->get(key));
254 }
255 Real value = 0.0;
256
257 // Calculate the returned value
258 Real returnVal = returnConfiguration_.returnValue(key, v1, v2, s1->asof(), s2->asof());
259 // Adjust return for any scaling
260 Real scaling = this->scaling(key, returnVal);
261 returnVal = returnVal * scaling;
262 // Calculate the shifted value
263 value = returnConfiguration_.applyReturn(key, base, returnVal);
264 if (std::isinf(value)) {
265 ALOG("Value is inf for " << key << " from date " << s1->asof() << " to " << s2->asof());
266 }
267 // Add it
268 scen->add(key, value);
269 // Populate calculation details
270 calculationDetails_[calcDetailsCounter].scenarioDate1 = s1->asof();
271 calculationDetails_[calcDetailsCounter].scenarioDate2 = s2->asof();
272 calculationDetails_[calcDetailsCounter].key = key;
273 calculationDetails_[calcDetailsCounter].baseValue = base;
274 calculationDetails_[calcDetailsCounter].adjustmentFactor1 =
275 adjFactors_ ? adjFactors_->getFactor(key.name, s1->asof()) : 1.0;
276 calculationDetails_[calcDetailsCounter].adjustmentFactor2 =
277 adjFactors_ ? adjFactors_->getFactor(key.name, s2->asof()) : 1.0;
278 calculationDetails_[calcDetailsCounter].scenarioValue1 = v1;
279 calculationDetails_[calcDetailsCounter].scenarioValue2 = v2;
280 calculationDetails_[calcDetailsCounter].returnType = returnConfiguration_.returnTypes().at(key.keytype);
281 calculationDetails_[calcDetailsCounter].scaling = scaling;
282 calculationDetails_[calcDetailsCounter].returnValue = returnVal;
283 calculationDetails_[calcDetailsCounter].scenarioValue = value;
284 ++calcDetailsCounter;
285 }
286
287 // Label the scenario
288 string label = labelPrefix_ + ore::data::to_string(io::iso_date(s1->asof())) + "_" +
289 ore::data::to_string(io::iso_date(s2->asof()));
290 scen->label(label);
291
292 // return it.
293 ++i_;
294 return scen;
295}

317boost::shared_ptr<Scenario> HistoricalScenarioGeneratorRandom::next(const Date& d) {
318
319 QL_REQUIRE(baseScenario() != nullptr, "HistoricalScenarioGeneratorRandom: base scenario not set");
320
321 // build the scenarios
322 QL_REQUIRE(d >= baseScenario()->asof(),
323 "HistoricalScenarioGeneratorRandom: Cannot generate a scenario in the past");
324 boost::shared_ptr<Scenario> scen = scenarioFactory_->buildScenario(d, "", 1.0);
325
326 // loop over all keys
327 for (auto key : baseScenario()->keys()) {
328 Real base = baseScenario()->get(key);
329 Real value = 0.0;
330 switch (key.keytype) {
331 case RiskFactorKey::KeyType::DiscountCurve:
332 case RiskFactorKey::KeyType::DividendYield:
333 case RiskFactorKey::KeyType::YieldCurve:
334 case RiskFactorKey::KeyType::IndexCurve:
335 case RiskFactorKey::KeyType::SurvivalProbability:
336 if (close_enough(base, 0.0))
337 value = 0.0;
338 else
339 value = 1.0 - (1.0 - base) * (1.0 + normalrng_->next().value * 0.05);
340 break;
341 case RiskFactorKey::KeyType::ZeroInflationCurve:
342 case RiskFactorKey::KeyType::YoYInflationCurve:
343 value = base + (normalrng_->next().value * 0.0010);
344 break;
345 case RiskFactorKey::KeyType::FXSpot:
346 case RiskFactorKey::KeyType::EquitySpot:
347 case RiskFactorKey::KeyType::SwaptionVolatility:
348 case RiskFactorKey::KeyType::YieldVolatility:
349 case RiskFactorKey::KeyType::OptionletVolatility:
350 case RiskFactorKey::KeyType::CDSVolatility:
351 case RiskFactorKey::KeyType::FXVolatility:
352 case RiskFactorKey::KeyType::EquityVolatility:
353 case RiskFactorKey::KeyType::SecuritySpread:
354 value = base * (1.0 + normalrng_->next().value * 0.02);
355 break;
356 case RiskFactorKey::KeyType::BaseCorrelation:
357 value = base + (normalrng_->next().value * 0.05);
358 value = std::max(std::min(value, 0.9999), -0.9999);
359 break;
360 default:
361 QL_FAIL("HistoricalScenarioGeneratorRandom: unexpected key type in key " << key);
362 }
363 scen->add(key, value);
364 }
365
366 // return it.
367 ++i_;
368 return scen;
369}

376boost::shared_ptr<Scenario> HistoricalScenarioGeneratorTransform::next(const Date& d) {
377 boost::shared_ptr<Scenario> scenario = HistoricalScenarioGenerator::next(d)->clone();
378 const vector<RiskFactorKey>& keys = baseScenario()->keys();
379 Date asof = baseScenario()->asof();
380 vector<Period> tenors;
381 Date endDate;
382 DayCounter dc;
383
384 for (Size k = 0; k < keys.size(); ++k) {
385 if ((keys[k].keytype == RiskFactorKey::KeyType::DiscountCurve) ||
386 (keys[k].keytype == RiskFactorKey::KeyType::IndexCurve) ||
387 (keys[k].keytype == RiskFactorKey::KeyType::SurvivalProbability)) {
388 if ((keys[k].keytype == RiskFactorKey::KeyType::DiscountCurve)) {
389 dc = simMarket_->discountCurve(keys[k].name)->dayCounter();
390 tenors = simMarketConfig_->yieldCurveTenors(keys[k].name);
391 } else if (keys[k].keytype == RiskFactorKey::KeyType::IndexCurve) {
392 dc = simMarket_->iborIndex(keys[k].name)->dayCounter();
393 tenors = simMarketConfig_->yieldCurveTenors(keys[k].name);
394 } else if (keys[k].keytype == RiskFactorKey::KeyType::SurvivalProbability) {
395 dc = simMarket_->defaultCurve(keys[k].name)->curve()->dayCounter();
396 tenors = simMarketConfig_->defaultTenors(keys[k].name);
397 }
398
399 endDate = asof + tenors[keys[k].index];
400
401 auto toZero = [&dc, &asof, &endDate](double compound) {
402 return InterestRate::impliedRate(compound, dc, QuantLib::Compounding::Continuous,
403 QuantLib::Frequency::Annual, asof, endDate)
404 .rate();
405 };
406
407 Real zero = toZero(1.0 / scenario->get(keys[k]));
408 scenario->add(keys[k], zero);
409 // update calc details
410 calculationDetails_[k].baseValue = toZero(1.0 / calculationDetails_[k].baseValue);
411 calculationDetails_[k].scenarioValue1 = toZero(1.0 / calculationDetails_[k].scenarioValue1);
412 calculationDetails_[k].scenarioValue2 = toZero(1.0 / calculationDetails_[k].scenarioValue2);
413 calculationDetails_[k].returnType = ReturnConfiguration::ReturnType::Absolute;
414 calculationDetails_[k].returnValue =
415 calculationDetails_[k].scaling *
416 (calculationDetails_[k].scenarioValue2 - calculationDetails_[k].scenarioValue1);
417 calculationDetails_[k].scenarioValue = toZero(1.0 / calculationDetails_[k].scenarioValue);
418 }
419 }
420 return scenario;
421}

423HistoricalScenarioGeneratorWithFilteredDates::HistoricalScenarioGeneratorWithFilteredDates(
424 const std::vector<TimePeriod>& filter, const boost::shared_ptr<HistoricalScenarioGenerator>& gen)
425 : HistoricalScenarioGenerator(gen->scenarioLoader(), gen->scenarioFactory(), gen->cal(), gen->adjFactors(),
426 gen->mporDays(), gen->overlapping(), gen->returnConfiguration(), gen->labelPrefix()),
427 gen_(gen), i_orig_(0) {
428
429 // set base scenario
430
431 baseScenario_ = gen->baseScenario();
432
433 for (auto const& f : filter) {
434 // Check that backtest and benchmark periods are covered by the historical scenario generator
435 Date minDate = *std::min_element(f.startDates().begin(), f.startDates().end());
436 Date maxDate = *std::max_element(f.endDates().begin(), f.endDates().end());
437
438 QL_REQUIRE(startDates_.front() <= minDate && maxDate <= endDates_.back(),
439 "The backtesting period " << f << " is not covered by the historical scenario generator: Required dates = ["
440 << ore::data::to_string(minDate) << "," << ore::data::to_string(maxDate)
441 << "], Covered dates = [" << startDates_.front() << "," << endDates_.back() << "]");
442 }
443
444 // filter start / end dates on relevant scenarios
445
446 isRelevantScenario_ = std::vector<bool>(gen->numScenarios(), false);
447
448 std::vector<Date> newStartDates, newEndDates;
449
450 for (Size i = 0; i < startDates_.size(); ++i) {
451 isRelevantScenario_[i] = false;
452 for (auto const& f : filter) {
453 if (f.contains(startDates_[i]) && f.contains(endDates_[i]))
454 isRelevantScenario_[i] = true;
455 }
456 if (isRelevantScenario_[i]) {
457 newStartDates.push_back(startDates_[i]);
458 newEndDates.push_back(endDates_[i]);
459 }
460 }
461
462 startDates_ = newStartDates;
463 endDates_ = newEndDates;
464}

472boost::shared_ptr<Scenario> HistoricalScenarioGeneratorWithFilteredDates::next(const Date& d) {
473 while (i_orig_ < gen_->numScenarios() && !isRelevantScenario_[i_orig_]) {
474 gen_->next(d);
475 ++i_orig_;
476 }
477 QL_REQUIRE(i_orig_ < gen_->numScenarios(),
478 "HistoricalScenarioGeneratorWithFilteredDates:next(): no more scenarios available");
479 ++i_orig_;
