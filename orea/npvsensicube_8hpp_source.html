33 //! NPVSensiCube class stores NPVs resulting from risk factor shifts on an as of date.
34 /*! This class is a restriction of the NPVCube to a grid of values where the ids are
35  trade IDs and the samples are risk factor shifts
36 
37  \ingroup cube
38  */
39 class NPVSensiCube : public NPVCube {
40 public:
41  using NPVCube::get;
42  using NPVCube::set;
43 
44  //! Number of dates in the NPVSensiCube is exactly one i.e. the as of date
45  QuantLib::Size numDates() const override { return 1; }
46 
47  //! The depth in the NPVSensiCube is exactly one
48  QuantLib::Size depth() const override { return 1; }
49 
50  //! Convenience method to get a value from the cube using \p id and \p sample only
51  Real get(QuantLib::Size id, QuantLib::Size sample) const { return this->get(id, 0, sample, 0); }
52 
53  //! Convenience method to get a value from the cube using \p id and \p sample only
54  Real get(const std::string& id, QuantLib::Size sample) const { return this->get(id, asof(), sample, 0); };
55 
56  //! Convenience method to set a value in the cube using \p id and \p sample only
57  void set(QuantLib::Real value, QuantLib::Size id, QuantLib::Size sample) { this->set(value, id, 0, sample, 0); }
58 
59  //! Convenience method to set a value in the cube using \p id and \p sample only
60  void set(QuantLib::Real value, const std::string& id, QuantLib::Size sample) {
61  this->set(value, id, asof(), sample, 0);
62  }
63 
64  //! Return the index of the trade in the cube
65  Size getTradeIndex(const std::string& tradeId) const { return index(tradeId); }
66 
67  /*! Return a map for the trade ID at index \p tradeIdx where the map key is the index of the
68  risk factor shift and the map value is the NPV under that shift
69  */
70  // Returning a const reference to the map makes the derived class have it as a member
71  virtual const std::map<QuantLib::Size, QuantLib::Real>& getTradeNPVs(Size tradeIdx) const = 0;
72 
73  /*! Return a map for the \p tradeId where the map key is the index of the
74  risk factor shift and the map value is the NPV under that shift
75  */
76  const std::map<QuantLib::Size, QuantLib::Real>& getTradeNPVs(const std::string& tradeId) const {
77  return getTradeNPVs(index(tradeId));
78  }
79 };
