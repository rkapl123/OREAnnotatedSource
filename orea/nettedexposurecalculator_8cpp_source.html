62
63 vector<string> nettingSetIds;
64 for (auto nettingSet : nettingSetDefaultValue) {
65 nettingSetIds.push_back(nettingSet.first);
66 if (flipViewXVA_) {

69 }
70 }
71 }
72

74 market_->asofDate(), nettingSetIds, cube->dates(),

87void NettedExposureCalculator::build() {
88 LOG("Compute netting set exposure profiles");
89
90 const Date today = market_->asofDate();
91 const DayCounter dc = ActualActual(ActualActual::ISDA);
92
93 vector<Real> times = vector<Real>(cube_->dates().size(), 0.0);
94 for (Size i = 0; i < cube_->dates().size(); i++)
95 times[i] = dc.yearFraction(today, cube_->dates()[i]);
96
97 map<string, Real> nettingSetValueToday;
98 map<string, Date> nettingSetMaturity;
99 map<string, Size> nettingSetSize;
100 for (Size i = 0; i < portfolio_->trades().size(); ++i) {
101 const auto& trade = portfolio_->trades()[i];
102 string tradeId = trade->id();
103 string nettingSetId = trade->envelope().nettingSetId();
104 string cp = trade->envelope().counterparty();
105 if (counterpartyMap_.find(nettingSetId) == counterpartyMap_.end())
106 counterpartyMap_[nettingSetId] = trade->envelope().counterparty();
107 else {
108 QL_REQUIRE(counterpartyMap_[nettingSetId] == cp, "counterparty name is not unique within the netting set");
109 }
110 Real npv;
111 if (flipViewXVA_) {
112 npv = -cube_->getT0(i);
113 } else {
114 npv = cube_->getT0(i);
115 }
116
117 if (nettingSetValueToday.find(nettingSetId) == nettingSetValueToday.end()) {
118 nettingSetValueToday[nettingSetId] = 0.0;
119 nettingSetMaturity[nettingSetId] = today;
120 nettingSetSize[nettingSetId] = 0;
121 }
122
123 nettingSetValueToday[nettingSetId] += npv;
124
125 if (trade->maturity() > nettingSetMaturity[nettingSetId])
126 nettingSetMaturity[nettingSetId] = trade->maturity();
127 nettingSetSize[nettingSetId]++;
128 }
129
130 vector<vector<Real>> averagePositiveAllocation(portfolio_->size(), vector<Real>(cube_->dates().size(), 0.0));
131 vector<vector<Real>> averageNegativeAllocation(portfolio_->size(), vector<Real>(cube_->dates().size(), 0.0));
132
133 map<string, vector<vector<Real>>> nettingSetValue = (calcType_ == CollateralExposureHelper::CalculationType::NoLag
134 ? nettingSetCloseOutValue_
135 : nettingSetDefaultValue_);
136 Size nettingSetCount = 0;
137 for (auto n : nettingSetValue) {
138 string nettingSetId = n.first;
139
140 LOG("Aggregate exposure for netting set " << nettingSetId);
141 vector<vector<Real>> data = n.second;
142
143 // Get the collateral account balance paths for the netting set.
144 // The pointer may remain empty if there is no CSA or if it is inactive.
145 boost::shared_ptr<vector<boost::shared_ptr<CollateralAccount>>> collateral =
146 collateralPaths(nettingSetId,
147 nettingSetValueToday[nettingSetId],
148 nettingSetDefaultValue_[nettingSetId],
149 nettingSetMaturity[nettingSetId]);
150
151 // Get the CSA index for Eonia Floor calculation below
152 colva_[nettingSetId] = 0.0;
153 collateralFloor_[nettingSetId] = 0.0;
154 boost::shared_ptr<NettingSetDefinition> netting = nettingSetManager_->get(nettingSetId);
155 string csaIndexName;
156 Handle<IborIndex> csaIndex;
157 bool applyInitialMargin = false;
158 CSA::Type initialMarginType = CSA::Bilateral;
159 if (netting->activeCsaFlag()) {
160 csaIndexName = netting->csaDetails()->index();
161 if (csaIndexName != "") {
162 csaIndex = market_->iborIndex(csaIndexName);
163 QL_REQUIRE(scenarioData_->has(AggregationScenarioDataType::IndexFixing, csaIndexName),
164 "scenario data does not provide index values for " << csaIndexName);
165 }
166 QL_REQUIRE(netting->csaDetails(), "active CSA for netting set " << nettingSetId
167 << ", but CSA details not initialised");
168 applyInitialMargin = netting->csaDetails()->applyInitialMargin() && applyInitialMargin_;
169 initialMarginType = netting->csaDetails()->initialMarginType();
170 LOG("ApplyInitialMargin=" << applyInitialMargin << " for netting set " << nettingSetId
171 << ", CSA IM=" << netting->csaDetails()->applyInitialMargin()
172 << ", CSA IM Type=" << initialMarginType
173 << ", Analytics DIM=" << applyInitialMargin_);
174 if (applyInitialMargin_ && !netting->csaDetails()->applyInitialMargin())
175 ALOG("ApplyInitialMargin deactivated at netting set level " << nettingSetId);
176 if (!applyInitialMargin_ && netting->csaDetails()->applyInitialMargin())
177 ALOG("ApplyInitialMargin deactivated in analytics, but active at netting set level " << nettingSetId);
178 }
179
180 Handle<YieldTermStructure> curve = market_->discountCurve(baseCurrency_, configuration_);
181 vector<Real> epe(cube_->dates().size() + 1, 0.0);
182 vector<Real> ene(cube_->dates().size() + 1, 0.0);
183 vector<Real> ee_b(cube_->dates().size() + 1, 0.0);
184 vector<Real> eee_b(cube_->dates().size() + 1, 0.0);
185 vector<Real> eee_b_kva_1(cube_->dates().size() + 1, 0.0);
186 vector<Real> eee_b_kva_2(cube_->dates().size() + 1, 0.0);
187 vector<Real> eepe_b_kva_1(cube_->dates().size() + 1, 0.0);
188 vector<Real> eepe_b_kva_2(cube_->dates().size() + 1, 0.0);
189 vector<Real> eab(cube_->dates().size() + 1, 0.0);
190 vector<Real> pfe(cube_->dates().size() + 1, 0.0);
191 vector<Real> colvaInc(cube_->dates().size() + 1, 0.0);
192 vector<Real> eoniaFloorInc(cube_->dates().size() + 1, 0.0);
193 Real npv = nettingSetValueToday[nettingSetId];
194 if ((fullInitialCollateralisation_) & (netting->activeCsaFlag())) {
195 // This assumes that the collateral at t=0 is the same as the npv at t=0.
196 epe[0] = 0;
197 ene[0] = 0;
198 pfe[0] = 0;
199 } else {
200 epe[0] = std::max(npv, 0.0);
201 ene[0] = std::max(-npv, 0.0);
202 pfe[0] = std::max(npv, 0.0);
203 }
204 // The fullInitialCollateralisation flag doesn't affect the eab, which feeds into the "ExpectedCollateral"
205 // column of the 'exposure_nettingset_*' reports. We always assume the full collateral here.
206 eab[0] = -npv;
207 ee_b[0] = epe[0];
208 eee_b[0] = ee_b[0];
209 nettedCube_->setT0(npv, nettingSetCount);
210 exposureCube_->setT0(epe[0], nettingSetCount, ExposureIndex::EPE);
211 exposureCube_->setT0(ene[0], nettingSetCount, ExposureIndex::ENE);
212
213 for (Size j = 0; j < cube_->dates().size(); ++j) {
214
215 Date date = cube_->dates()[j];
216 Date prevDate = j > 0 ? cube_->dates()[j - 1] : today;
217 vector<Real> distribution(cube_->samples(), 0.0);
218 for (Size k = 0; k < cube_->samples(); ++k) {
219 Real balance = 0.0;
220 if (collateral) {
221 balance = collateral->at(k)->accountBalance(date);
222 if (netting->csaDetails()->csaCurrency() != baseCurrency_) {
223 // Convert from CSACurrency to baseCurrency
224 double fxRate = scenarioData_->get(j, k, AggregationScenarioDataType::FXSpot,
225 netting->csaDetails()->csaCurrency());
226 balance *= fxRate;
227 }
228 }
229 eab[j + 1] += balance / cube_->samples();
230 Real exposure = data[j][k] - balance;
231 Real dim = 0.0;
232 if (applyInitialMargin && collateral) { // don't apply initial margin without VM, i.e. inactive CSA
233 // Initial Margin
234 // Use IM to reduce exposure
235 // Size dimIndex = j == 0 ? 0 : j - 1;
236 Size dimIndex = j;
237 dim = dimCalculator_->dynamicIM(nettingSetId)[dimIndex][k];
238 QL_REQUIRE(dim >= 0, "negative DIM for set " << nettingSetId << ", date " << j << ", sample " << k
239 << ": " << dim);
240 }
241 Real dim_epe = 0;
242 Real dim_ene = 0;
243 if (initialMarginType != CSA::Type::PostOnly)
244 dim_epe = dim;
245 if (initialMarginType != CSA::Type::CallOnly)
246 dim_ene = dim;
247 epe[j + 1] += std::max(exposure - dim_epe, 0.0) /
248 cube_->samples(); // dim here represents the held IM, and is expressed as a positive number
249 ene[j + 1] += std::max(-exposure - dim_ene, 0.0) /
250 cube_->samples(); // dim here represents the posted IM, and is expressed as a positive number
251 distribution[k] = exposure;
252 nettedCube_->set(exposure, nettingSetCount, j, k);
253 if (multiPath_) {
254 exposureCube_->set(std::max(exposure - dim_epe, 0.0), nettingSetCount, j, k, ExposureIndex::EPE);
255 exposureCube_->set(std::max(-exposure - dim_ene, 0.0), nettingSetCount, j, k, ExposureIndex::ENE);
256 }
257
258 if (netting->activeCsaFlag()) {
259 Real indexValue = 0.0;
260 DayCounter dc = ActualActual(ActualActual::ISDA);
261 if (csaIndexName != "") {
262 indexValue = scenarioData_->get(j, k, AggregationScenarioDataType::IndexFixing, csaIndexName);
263 dc = csaIndex->dayCounter();
264 }
265 Real dcf = dc.yearFraction(prevDate, date);
266 Real collateralSpread = (balance >= 0.0 ? netting->csaDetails()->collatSpreadRcv() : netting->csaDetails()->collatSpreadPay());
267 Real numeraire = scenarioData_->get(j, k, AggregationScenarioDataType::Numeraire);
268 Real colvaDelta = -balance * collateralSpread * dcf / numeraire / cube_->samples();
269 // intuitive floorDelta including collateralSpread would be:
270 // -balance * (max(indexValue - collateralSpread,0) - (indexValue - collateralSpread)) * dcf /
271 // samples
272 Real floorDelta = -balance * std::max(-(indexValue - collateralSpread), 0.0) * dcf / numeraire / cube_->samples();
273 colvaInc[j + 1] += colvaDelta;
274 colva_[nettingSetId] += colvaDelta;
275 eoniaFloorInc[j + 1] += floorDelta;
276 collateralFloor_[nettingSetId] += floorDelta;
277 }
278
279 if (marginalAllocation_) {
280 for (Size i = 0; i < portfolio_->trades().size(); ++i) {
281 string nid = portfolio_->trades()[i]->envelope().nettingSetId();
282 if (nid != nettingSetId)
283 continue;
284 string tid = portfolio_->trades()[i]->id();
285 Real allocation = 0.0;
286 if (balance == 0.0)
287 allocation = cubeInterpretation_->getDefaultNpv(cube_, i, j, k);
288 // else if (data[j][k] == 0.0)
289 else if (fabs(data[j][k]) <= marginalAllocationLimit_)
290 allocation = exposure / nettingSetSize[nid];
291 else
292 allocation = exposure * cubeInterpretation_->getDefaultNpv(cube_, i, j, k) / data[j][k];
293
294 if (multiPath_) {
295 if (exposure > 0.0)
296 tradeExposureCube_->set(allocation, i, j, k, allocatedEpeIndex_);
297 else
298 tradeExposureCube_->set(-allocation, i, j, k, allocatedEneIndex_);
299 } else {
300 if (exposure > 0.0)
301 averagePositiveAllocation[i][j] += allocation / cube_->samples();
302 else
303 averageNegativeAllocation[i][j] -= allocation / cube_->samples();
304 }
305 }
306 }
307 }
308 if (!multiPath_) {
309 exposureCube_->set(epe[j + 1], nettingSetCount, j, 0, ExposureIndex::EPE);
310 exposureCube_->set(ene[j + 1], nettingSetCount, j, 0, ExposureIndex::ENE);
311 }
312 ee_b[j + 1] = epe[j + 1] / curve->discount(cube_->dates()[j]);
313 eee_b[j + 1] = std::max(eee_b[j], ee_b[j + 1]);
314 std::sort(distribution.begin(), distribution.end());
315 Size index = Size(floor(quantile_ * (cube_->samples() - 1) + 0.5));
316 pfe[j + 1] = std::max(distribution[index], 0.0);
317 }
318 ee_b_[nettingSetId] = ee_b;
319 eee_b_[nettingSetId] = eee_b;
320 pfe_[nettingSetId] = pfe;
321 expectedCollateral_[nettingSetId] = eab;
322 colvaInc_[nettingSetId] = colvaInc;
323 eoniaFloorInc_[nettingSetId] = eoniaFloorInc;
324
325 nettingSetCount++;
326
327 Real epe_b = 0;
328 Real eepe_b = 0;
329
330 Size t = 0;
331 Calendar cal = WeekendsOnly();
332 Date maturity = std::min(cal.adjust(today + 1 * Years + 4 * Days), nettingSetMaturity[nettingSetId]);
333 QuantLib::Real maturityTime = dc.yearFraction(today, maturity);
334
335 while (t < cube_->dates().size() && times[t] <= maturityTime)
336 ++t;
337
338 if (t > 0) {
339 vector<double> weights(t);
340 weights[0] = times[0];
341 for (Size k = 1; k < t; k++)
342 weights[k] = times[k] - times[k - 1];
343 double totalWeights = std::accumulate(weights.begin(), weights.end(), 0.0);
344 for (Size k = 0; k < t; k++)
345 weights[k] /= totalWeights;
346
347 for (Size k = 0; k < t; k++) {
348 epe_b += ee_b[k] * weights[k];
349 eepe_b += eee_b[k] * weights[k];
350 }
351 }
352 epe_b_[nettingSetId] = epe_b;
353 eepe_b_[nettingSetId] = eepe_b;
354 }
355
356 if (marginalAllocation_ && !multiPath_) {
357 for (Size i = 0; i < portfolio_->trades().size(); ++i) {
358 for (Size j = 0; j < cube_->dates().size(); ++j) {
359 tradeExposureCube_->set(averagePositiveAllocation[i][j], i, j, 0, allocatedEpeIndex_);
360 tradeExposureCube_->set(averageNegativeAllocation[i][j], i, j, 0, allocatedEneIndex_);
361 }
362 }
363 }

366boost::shared_ptr<vector<boost::shared_ptr<CollateralAccount>>>
367NettedExposureCalculator::collateralPaths(
368 const string& nettingSetId,
369 const Real& nettingSetValueToday,
370 const vector<vector<Real>>& nettingSetValue,
371 const Date& nettingSetMaturity) {
372
373 boost::shared_ptr<vector<boost::shared_ptr<CollateralAccount>>> collateral;
374
375 if (!nettingSetManager_->has(nettingSetId) || !nettingSetManager_->get(nettingSetId)->activeCsaFlag()) {
376 LOG("CSA missing or inactive for netting set " << nettingSetId);
377 return collateral;
378 }
379
380 LOG("Build collateral account balance paths for netting set " << nettingSetId);
381 boost::shared_ptr<NettingSetDefinition> netting = nettingSetManager_->get(nettingSetId);
382 string csaFxPair = netting->csaDetails()->csaCurrency() + baseCurrency_;
383 Real csaFxRateToday = 1.0;
384 if (netting->csaDetails()->csaCurrency() != baseCurrency_)
385 csaFxRateToday = market_->fxRate(csaFxPair, configuration_)->value();
386 LOG("CSA FX rate for pair " << csaFxPair << " = " << csaFxRateToday);
387
388 // Don't use Settings::instance().evaluationDate() here, this has moved to simulation end date.
389 Date today = market_->asofDate();
390 string csaIndexName = netting->csaDetails()->index();
391 // avoid thrown errors of the index fixing here on holidays of the index, instead take the preceding date then.
392 if (!market_->iborIndex(csaIndexName, configuration_)->isValidFixingDate(today)) {
393 today = market_->iborIndex(csaIndexName, configuration_)->fixingCalendar().adjust(today, Preceding);
394 }
395 Real csaRateToday = market_->iborIndex(csaIndexName, configuration_)->fixing(today);
396 LOG("CSA compounding rate for index " << csaIndexName << " = " << setprecision(8) << csaRateToday << " as of " << today);
397
398 // Copy scenario data to keep the collateral exposure helper unchanged
399 vector<vector<Real>> csaScenFxRates(cube_->dates().size(), vector<Real>(cube_->samples(), 0.0));
400 vector<vector<Real>> csaScenRates(cube_->dates().size(), vector<Real>(cube_->samples(), 0.0));
401 if (netting->csaDetails()->csaCurrency() != baseCurrency_) {
402 QL_REQUIRE(scenarioData_->has(AggregationScenarioDataType::FXSpot, netting->csaDetails()->csaCurrency()),
403 "scenario data does not provide FX rates for " << csaFxPair);
404 }
405 if (csaIndexName != "") {
406 QL_REQUIRE(scenarioData_->has(AggregationScenarioDataType::IndexFixing, csaIndexName),
407 "scenario data does not provide index values for " << csaIndexName);
408 }
409 for (Size j = 0; j < cube_->dates().size(); ++j) {
410 for (Size k = 0; k < cube_->samples(); ++k) {
411 if (netting->csaDetails()->csaCurrency() != baseCurrency_)
412 csaScenFxRates[j][k] = cubeInterpretation_->getDefaultAggrionScenarioData(
413 scenarioData_, AggregationScenarioDataType::FXSpot, j, k, netting->csaDetails()->csaCurrency());
414 else
415 csaScenFxRates[j][k] = 1.0;
416 if (csaIndexName != "") {
417 csaScenRates[j][k] = cubeInterpretation_->getDefaultAggrionScenarioData(
418 scenarioData_, AggregationScenarioDataType::IndexFixing, j, k, csaIndexName);
419 }
420 }
421 }
422
423 collateral = CollateralExposureHelper::collateralBalancePaths(
424 netting, // this netting set's definition
425 nettingSetValueToday, // today's netting set NPV
426 market_->asofDate(), // original evaluation date
427 nettingSetValue, // matrix of netting set values by date and sample
428 nettingSetMaturity, // netting set's maximum maturity date
429 cube_->dates(), // vector of future evaluation dates
430 csaFxRateToday, // today's FX rate for CSA to base currency, possibly 1
431 csaScenFxRates, // matrix of fx rates by date and sample, possibly 1
432 csaRateToday, // today's collateral compounding rate in CSA currency
433 csaScenRates, // matrix of CSA ccy short rates by date and sample
434 calcType_);
435 LOG("Collateral account balance paths for netting set " << nettingSetId << " done");
436
437 return collateral;
438}

440vector<Real> NettedExposureCalculator::getMeanExposure(const string& tid, ExposureIndex index) {
441 vector<Real> exp(cube_->dates().size() + 1, 0.0);
442 exp[0] = exposureCube_->getT0(tid, index);
443 for (Size i = 0; i < cube_->dates().size(); i++) {
444 if (multiPath_) {
445 for (Size k = 0; k < exposureCube_->samples(); k++) {
446 exp[i + 1] += exposureCube_->get(tid, cube_->dates()[i], k, index);
447 }
448 exp[i + 1] /= exposureCube_->samples();
449 }
450 else {
451 exp[i + 1] = exposureCube_->get(tid, cube_->dates()[i], 0, index);
452 }
453 }
454 return exp;
455}
