115// Utility function that is in catch blocks below
116void processException(bool continueOnError, const std::exception& e, const std::string& curveId = "",
117 ore::analytics::RiskFactorKey::KeyType keyType = ore::analytics::RiskFactorKey::KeyType::None,
118 const bool simDataWritten = false) {
119 string curve;
120 if (keyType != ore::analytics::RiskFactorKey::KeyType::None)
121 curve = to_string(keyType) + "/";
122 curve += curveId;
123
124 std::string message = "skipping this object in scenario sim market";
125 if (!curve.empty()) {
126 message += " (scenario data was ";
127 if (!simDataWritten)
128 message += "not ";
129 message += "written for this object.)";
130 }
131 if (continueOnError) {
132 std::string exceptionMessage = e.what();
133 /* We do not log a structured curve error message, if the exception message indicates that the problem
134 already occurred in the inti market. In this case we have already logged a structured error there. */
135 if (boost::starts_with(exceptionMessage, "did not find object ")) {
136 ALOG("CurveID: " << curve << ": " << message << ": " << exceptionMessage);
137 } else {
138 ALOG(StructuredCurveErrorMessage(curve, message, exceptionMessage));
139 }
140 } else {
141 QL_FAIL("Object with CurveID '" << curve << "' failed to build in scenario sim market: " << e.what());
142 }
143}

149RiskFactorKey::KeyType yieldCurveRiskFactor(const ore::data::YieldCurveType y) {
150
151 if (y == ore::data::YieldCurveType::Discount) {
152 return RiskFactorKey::KeyType::DiscountCurve;
153 } else if (y == ore::data::YieldCurveType::Yield) {
154 return RiskFactorKey::KeyType::YieldCurve;
155 } else if (y == ore::data::YieldCurveType::EquityDividend) {
156 return RiskFactorKey::KeyType::DividendYield;
157 } else {
158 QL_FAIL("yieldCurveType not supported");
159 }
160}

188void checkDayCounterConsistency(const std::string& curveId, const DayCounter& initCurveDayCounter,
189 const DayCounter& simCurveDayCounter) {
190 if (initCurveDayCounter != simCurveDayCounter) {
191 std::string initDcName = initCurveDayCounter.empty() ? "(empty)" : initCurveDayCounter.name();
192 std::string ssmDcName = simCurveDayCounter.empty() ? "(empty)" : simCurveDayCounter.name();

196 "), continuing anyway, please consider fixing this in either the initial market or ssm "
197 "configuration");
198 }

201boost::shared_ptr<YieldTermStructure>
202makeYieldCurve(const std::string& curveId, const bool spreaded, const Handle<YieldTermStructure>& initMarketTs,
203 const std::vector<Real>& yieldCurveTimes, const std::vector<Handle<Quote>>& quotes, const DayCounter& dc,
204 const Calendar& cal, const std::string& interpolation, const std::string& extrapolation) {
205 if (ObservationMode::instance().mode() == ObservationMode::Mode::Unregister && !spreaded) {
206 return boost::shared_ptr<YieldTermStructure>(boost::make_shared<QuantExt::InterpolatedDiscountCurve>(
207 yieldCurveTimes, quotes, 0, cal, dc,
208 interpolation == "LogLinear" ? QuantExt::InterpolatedDiscountCurve::Interpolation::logLinear
209 : QuantExt::InterpolatedDiscountCurve::Interpolation::linearZero,
210 extrapolation == "FlatZero" ? QuantExt::InterpolatedDiscountCurve::Extrapolation::flatZero
211 : QuantExt::InterpolatedDiscountCurve::Extrapolation::flatFwd));
212 } else {
213 if (spreaded) {
214 checkDayCounterConsistency(curveId, initMarketTs->dayCounter(), dc);
215 return boost::make_shared<QuantExt::SpreadedDiscountCurve>(
216 initMarketTs, yieldCurveTimes, quotes,
217 interpolation == "LogLinear" ? QuantExt::SpreadedDiscountCurve::Interpolation::logLinear
218 : QuantExt::SpreadedDiscountCurve::Interpolation::linearZero,
219 extrapolation == "FlatZero" ? SpreadedDiscountCurve::Extrapolation::flatZero
220 : SpreadedDiscountCurve::Extrapolation::flatFwd);
221 } else {
222 auto idc = boost::make_shared<QuantExt::InterpolatedDiscountCurve2>(
223 yieldCurveTimes, quotes, dc,
224 interpolation == "LogLinear" ? QuantExt::InterpolatedDiscountCurve2::Interpolation::logLinear
225 : QuantExt::InterpolatedDiscountCurve2::Interpolation::linearZero,
226 extrapolation == "FlatZero" ? InterpolatedDiscountCurve2::Extrapolation::flatZero
227 : InterpolatedDiscountCurve2::Extrapolation::flatFwd);
228 idc->setAdjustReferenceDate(false);
229 return idc;
230 }

244void ScenarioSimMarket::addYieldCurve(const boost::shared_ptr<Market>& initMarket, const std::string& configuration,
245 const RiskFactorKey::KeyType rf, const string& key, const vector<Period>& tenors,
246 bool& simDataWritten, bool simulate, bool spreaded) {
247 Handle<YieldTermStructure> wrapper = (riskFactorYieldCurve(rf) == ore::data::YieldCurveType::Discount)
248 ? initMarket->discountCurve(key, configuration)
249 : initMarket->yieldCurve(riskFactorYieldCurve(rf), key, configuration);
250 QL_REQUIRE(!wrapper.empty(), "yield curve not provided for " << key);
251 QL_REQUIRE(tenors.front() > 0 * Days, "yield curve tenors must not include t=0");
252 // include today
253
254 // constructing yield curves
255 DayCounter dc = wrapper->dayCounter();
256 vector<Time> yieldCurveTimes(1, 0.0); // include today
257 vector<Date> yieldCurveDates(1, asof_);
258 for (auto& tenor : tenors) {
259 yieldCurveTimes.push_back(dc.yearFraction(asof_, asof_ + tenor));
260 yieldCurveDates.push_back(asof_ + tenor);
261 }
262
263 vector<Handle<Quote>> quotes;
264 boost::shared_ptr<SimpleQuote> q(new SimpleQuote(1.0));
265 quotes.push_back(Handle<Quote>(q));
266 vector<Real> discounts(yieldCurveTimes.size());
267 std::map<RiskFactorKey, boost::shared_ptr<SimpleQuote>> simDataTmp;
268 std::map<RiskFactorKey, Real> absoluteSimDataTmp;
269 for (Size i = 0; i < yieldCurveTimes.size() - 1; i++) {
270 Real val = wrapper->discount(yieldCurveDates[i + 1]);
271 DLOG("ScenarioSimMarket yield curve " << rf << " " << key << " discount[" << i << "]=" << val);
272 boost::shared_ptr<SimpleQuote> q(new SimpleQuote(spreaded ? 1.0 : val));
273 Handle<Quote> qh(q);
274 quotes.push_back(qh);
275
276 // Check if the risk factor is simulated before adding it
277 if (simulate) {
278 simDataTmp.emplace(std::piecewise_construct, std::forward_as_tuple(rf, key, i), std::forward_as_tuple(q));
279 // if generating spreaded scenarios, add the absolute value as well
280 if (spreaded) {
281 absoluteSimDataTmp.emplace(std::piecewise_construct, std::forward_as_tuple(rf, key, i),
282 std::forward_as_tuple(val));
283 }
284 }
285 }
286
287 writeSimData(simDataTmp, absoluteSimDataTmp);
288 simDataWritten = true;
289
290 boost::shared_ptr<YieldTermStructure> yieldCurve =
291 makeYieldCurve(key, spreaded, wrapper, yieldCurveTimes, quotes, dc, TARGET(), parameters_->interpolation(),
292 parameters_->extrapolation());
293
294 Handle<YieldTermStructure> ych(yieldCurve);
295 if (wrapper->allowsExtrapolation())
296 ych->enableExtrapolation();
297 yieldCurves_.insert(pair<tuple<string, ore::data::YieldCurveType, string>, Handle<YieldTermStructure>>(
298 make_tuple(Market::defaultConfiguration, riskFactorYieldCurve(rf), key), ych));
299}

312ScenarioSimMarket::ScenarioSimMarket(
313 const boost::shared_ptr<Market>& initMarket, const boost::shared_ptr<ScenarioSimMarketParameters>& parameters,
314 const boost::shared_ptr<FixingManager>& fixingManager, const std::string& configuration,
315 const ore::data::CurveConfigurations& curveConfigs, const ore::data::TodaysMarketParameters& todaysMarketParams,
316 const bool continueOnError, const bool useSpreadedTermStructures, const bool cacheSimData,
317 const bool allowPartialScenarios, const IborFallbackConfig& iborFallbackConfig, const bool handlePseudoCurrencies)
318 : SimMarket(handlePseudoCurrencies), parameters_(parameters), fixingManager_(fixingManager),
319 filter_(boost::make_shared<ScenarioFilter>()), useSpreadedTermStructures_(useSpreadedTermStructures),
320 cacheSimData_(cacheSimData), allowPartialScenarios_(allowPartialScenarios),
321 iborFallbackConfig_(iborFallbackConfig) {
322
323 LOG("building ScenarioSimMarket...");
324 asof_ = initMarket->asofDate();
325 DLOG("AsOf " << QuantLib::io::iso_date(asof_));
326
327 // check ssm parameters
328 QL_REQUIRE(parameters_->interpolation() == "LogLinear" || parameters_->interpolation() == "LinearZero",
329 "ScenarioSimMarket: Interpolation (" << parameters_->interpolation()
330 << ") must be set to 'LogLinear' or 'LinearZero'");
331 QL_REQUIRE(parameters_->extrapolation() == "FlatZero" || parameters_->extrapolation() == "FlatFwd",
332 "ScenarioSimMarket: YieldCurves / Extrapolation ('" << parameters_->extrapolation()
333 << "') must be set to 'FlatZero' or 'FlatFwd'");
334 QL_REQUIRE(parameters_->defaultCurveExtrapolation() == "FlatZero" ||
335 parameters_->defaultCurveExtrapolation() == "FlatFwd",
336 "ScenarioSimMarket: DefaultCurves / Extrapolation ('" << parameters_->extrapolation()
337 << "') must be set to 'FlatZero' or 'FlatFwd'");
338
339 for (const auto& param : parameters->parameters()) {
340 try {
341 // we populate the temp containers for each curve and write the result to the global
342 // containers only if the set of data points is complete for this curve
343 std::map<RiskFactorKey, boost::shared_ptr<SimpleQuote>> simDataTmp;
344 std::map<RiskFactorKey, Real> absoluteSimDataTmp;
345
346 boost::timer::cpu_timer timer;
347
348 switch (param.first) {
349 case RiskFactorKey::KeyType::FXSpot: {
350 std::map<std::string, Handle<Quote>> fxQuotes;
351 for (const auto& name : param.second.second) {
352 bool simDataWritten = false;
353 try {
354 // constructing fxSpots_
355 DLOG("adding " << name << " FX rates");
356 auto q = boost::make_shared<SimpleQuote>(initMarket->fxSpot(name, configuration)->value());
357 auto qh = Handle<Quote>(q);
358 fxQuotes[name] = qh;
359 // Check if the risk factor is simulated before adding it
360 if (param.second.first) {
361 simDataTmp.emplace(std::piecewise_construct, std::forward_as_tuple(param.first, name),
362 std::forward_as_tuple(q));
363 }
364 writeSimData(simDataTmp, absoluteSimDataTmp);
365 simDataWritten = true;
366 } catch (const std::exception& e) {
367 processException(continueOnError, e, name, param.first, simDataWritten);
368 }
369 }
370 fx_ = boost::make_shared<FXTriangulation>(fxQuotes);
371 break;
372 }
373
374 case RiskFactorKey::KeyType::DiscountCurve:
375 case RiskFactorKey::KeyType::YieldCurve:
376 for (const auto& name : param.second.second) {
377 bool simDataWritten = false;
378 try {
379 DLOG("building " << name << " yield curve..");
380 vector<Period> tenors = parameters->yieldCurveTenors(name);
381 addYieldCurve(initMarket, configuration, param.first, name, tenors, simDataWritten,
382 param.second.first, useSpreadedTermStructures_);
383 DLOG("building " << name << " yield curve done");
384 } catch (const std::exception& e) {
385 processException(continueOnError, e, name, param.first, simDataWritten);
386 }
387 }
388 break;
389
390 case RiskFactorKey::KeyType::IndexCurve: {
391 // make sure we built overnight indices first, so that we can build ibor fallback indices
392 // that depend on them
393 std::vector<std::string> indices;
394 for (auto const& i : param.second.second) {
395 bool isOn = false;
396 try {
397 isOn = boost::dynamic_pointer_cast<OvernightIndex>(*initMarket->iborIndex(i, configuration)) !=
398 nullptr;
399 } catch (...) {
400 }
401 if (isOn)
402 indices.insert(indices.begin(), i);
403 else
404 indices.push_back(i);
405 }
406 // loop over sorted indices and build them
407 for (const auto& name : indices) {
408 bool simDataWritten = false;
409 try {
410 DLOG("building " << name << " index curve");
411 std::vector<string> indexTokens;
412 split(indexTokens, name, boost::is_any_of("-"));
413 Handle<IborIndex> index;
414 if (indexTokens[1] == "GENERIC") {
415 // If we have a generic curve build the index using the index currency's discount curve
416 // no need to check for a convention based ibor index in this case
417 index = Handle<IborIndex>(
418 parseIborIndex(name, initMarket->discountCurve(indexTokens[0], configuration)));
419 } else {
420 index = initMarket->iborIndex(name, configuration);
421 }
422 QL_REQUIRE(!index.empty(), "index object for " << name << " not provided");
423 Handle<YieldTermStructure> wrapperIndex = index->forwardingTermStructure();
424 QL_REQUIRE(!wrapperIndex.empty(), "no termstructure for index " << name);
425 vector<string> keys(parameters->yieldCurveTenors(name).size());
426
427 DayCounter dc = wrapperIndex->dayCounter();
428 vector<Time> yieldCurveTimes(1, 0.0); // include today
429 vector<Date> yieldCurveDates(1, asof_);
430 QL_REQUIRE(parameters->yieldCurveTenors(name).front() > 0 * Days,
431 "yield curve tenors must not include t=0");
432 for (auto& tenor : parameters->yieldCurveTenors(name)) {
433 yieldCurveTimes.push_back(dc.yearFraction(asof_, asof_ + tenor));
434 yieldCurveDates.push_back(asof_ + tenor);
435 }
436
437 // include today
438 vector<Handle<Quote>> quotes;
439 boost::shared_ptr<SimpleQuote> q(new SimpleQuote(1.0));
440 quotes.push_back(Handle<Quote>(q));
441
442 for (Size i = 0; i < yieldCurveTimes.size() - 1; i++) {
443 Real val = wrapperIndex->discount(yieldCurveDates[i + 1]);
444 boost::shared_ptr<SimpleQuote> q(new SimpleQuote(useSpreadedTermStructures_ ? 1.0 : val));
445 Handle<Quote> qh(q);
446 quotes.push_back(qh);
447
448 simDataTmp.emplace(std::piecewise_construct, std::forward_as_tuple(param.first, name, i),
449 std::forward_as_tuple(q));
450 if (useSpreadedTermStructures_) {
451 absoluteSimDataTmp.emplace(std::piecewise_construct,
452 std::forward_as_tuple(param.first, name, i),
453 std::forward_as_tuple(val));
454 }
455 // FIXME where do we check whether the risk factor is simulated?
456 DLOG("ScenarioSimMarket index curve " << name << " discount[" << i << "]=" << val);
457 }
458
459 writeSimData(simDataTmp, absoluteSimDataTmp);
460 simDataWritten = true;
461
462 boost::shared_ptr<YieldTermStructure> indexCurve = makeYieldCurve(
463 name, useSpreadedTermStructures_, wrapperIndex, yieldCurveTimes, quotes, dc,
464 index->fixingCalendar(), parameters_->interpolation(), parameters_->extrapolation());
465
466 Handle<YieldTermStructure> ich(indexCurve);
467 if (wrapperIndex->allowsExtrapolation())
468 ich->enableExtrapolation();
469
470 boost::shared_ptr<IborIndex> i = index->clone(ich);
471 if (iborFallbackConfig_.isIndexReplaced(name, asof_)) {
472 // handle ibor fallback indices
473 auto fallbackData = iborFallbackConfig_.fallbackData(name);
474 auto f = iborIndices_.find(make_pair(Market::defaultConfiguration, fallbackData.rfrIndex));
475 QL_REQUIRE(f != iborIndices_.end(),
476 "Could not build ibor fallback index '"
477 << name << "', because rfr index '" << fallbackData.rfrIndex
478 << "' is not present in scenario sim market, is the rfr index in the "
479 "scenario sim market parameters?");
480 auto rfrInd = boost::dynamic_pointer_cast<OvernightIndex>(*f->second);
481 QL_REQUIRE(rfrInd != nullptr,
482 "Could not cast '"
483 << fallbackData.rfrIndex
484 << "' to overnight index when building the ibor fallback index '" << name
485 << "'");
486 if (auto original = boost::dynamic_pointer_cast<OvernightIndex>(i))
487 i = boost::make_shared<QuantExt::FallbackOvernightIndex>(
488 original, rfrInd, fallbackData.spread, fallbackData.switchDate,
489 iborFallbackConfig_.useRfrCurveInSimulationMarket());
490 else
491 i = boost::make_shared<QuantExt::FallbackIborIndex>(
492 i, rfrInd, fallbackData.spread, fallbackData.switchDate,
493 iborFallbackConfig_.useRfrCurveInSimulationMarket());
494 DLOG("built ibor fall back index '"
495 << name << "' with rfr index '" << fallbackData.rfrIndex << "', spread "
496 << fallbackData.spread << ", use rfr curve in scen sim market: " << std::boolalpha
497 << iborFallbackConfig_.useRfrCurveInSimulationMarket());
498 }
499 iborIndices_.insert(
500 make_pair(make_pair(Market::defaultConfiguration, name), Handle<IborIndex>(i)));
501 DLOG("building " << name << " index curve done");
502 } catch (const std::exception& e) {
503 processException(continueOnError, e, name, param.first, simDataWritten);
504 }
505 }
506 break;
507 }
508
509 case RiskFactorKey::KeyType::EquitySpot:
510 for (const auto& name : param.second.second) {
511 bool simDataWritten = false;
512 try {
513 // building equity spots
514 DLOG("adding " << name << " equity spot...");
515 Real spotVal = initMarket->equitySpot(name, configuration)->value();
516 boost::shared_ptr<SimpleQuote> q(new SimpleQuote(spotVal));

796 if (isAtm) {
797 svp = atm;

800 // if we have a cube, we keep the vol spreads constant under scenarios

802 // which means that we keep the smiles constant in terms of vol spreads when

821 }

849 svp->enableExtrapolation(); // FIXME

867 processException(continueOnError, e, name, param.first, simDataWritten);

887 // Check if the risk factor is simulated before adding it

1012 Leg capFloor =

1041 << std::setprecision(4) << strike << "] is "

1044 boost::make_shared<SimpleQuote>(useSpreadedTermStructures_ ? 0.0 : vol);
1045 Size index = i * strikes.size() + j;
1046 simDataTmp.emplace(std::piecewise_construct,
1047 std::forward_as_tuple(param.first, name, index),
1048 std::forward_as_tuple(q));

2776
2777 currentScenario_ = scenario;
2778
2779 // 1 handle delta scenario
2780
2781 auto deltaScenario = boost::dynamic_pointer_cast<DeltaScenario>(scenario);
2782
2783 /*! our assumption is that either all or none of the scenarios we apply are
2784 delta scenarios or the base scenario */
2785
2786 if (deltaScenario != nullptr) {
2787 for (auto const& key : diffToBaseKeys_) {
2788 auto it = simData_.find(key);
2789 if (it != simData_.end()) {
2790 it->second->setValue(baseScenario_->get(key));
2791 }
2792 }
2793 diffToBaseKeys_.clear();
2794 auto delta = deltaScenario->delta();
2795 bool missingPoint = false;
2796 for (auto const& key : delta->keys()) {
2797 auto it = simData_.find(key);
2798 if (it == simData_.end()) {
2799 ALOG("simulation data point missing for key " << key);
2800 missingPoint = true;
2801 } else {
2802 if (filter_->allow(key)) {
2803 it->second->setValue(delta->get(key));
2804 diffToBaseKeys_.insert(key);
2805 }
2806 }
2807 }
2808 QL_REQUIRE(!missingPoint, "simulation data points missing from scenario, exit.");
2809
2810 return;
2811 }
2812
2813 // 2 apply scenario based on cached indices for simData_ for a SimpleScenario
2814 // this assumes that all scenarios have an identical key structure in their data map
2815
2816 if (cacheSimData_) {
2817 if (auto s = boost::dynamic_pointer_cast<SimpleScenario>(scenario)) {
2818
2819 // fill cache
2820
2821 if (cachedSimData_.empty()) {
2822 Size count = 0;
2823 for (auto const& d : s->data()) {
2824 auto it = simData_.find(d.first);
2825 if (it == simData_.end()) {
2826 WLOG("simulation data point missing for key " << d.first);
2827 cachedSimData_.push_back(boost::shared_ptr<SimpleQuote>());
2828 cachedSimDataActive_.push_back(false);
2829 } else {
2830 ++count;
2831 cachedSimData_.push_back(it->second);
2832 cachedSimDataActive_.push_back(filter_->allow(d.first));
2833 }
2834 }
2835 if (count != simData_.size() && !allowPartialScenarios_) {
2836 ALOG("mismatch between scenario and sim data size, " << count << " vs " << simData_.size());
2837 for (auto it : simData_) {
2838 if (!scenario->has(it.first))
2839 WLOG("Key " << it.first << " missing in scenario");
2840 }
2841 QL_FAIL("mismatch between scenario and sim data size, exit.");
2842 }
2843 }
2844
2845 // apply scenario data according to cached indices
2846
2847 Size i = 0;
2848 for (auto const& q : s->data()) {
2849 if (cachedSimDataActive_[i])
2850 cachedSimData_[i]->setValue(q.second);
2851 ++i;
2852 }
2853
2854 return;
2855 }
2856 }
2857
2858 // 3 all other cases
2859
2860 const vector<RiskFactorKey>& keys = scenario->keys();
2861
2862 Size count = 0;
2863 for (const auto& key : keys) {
2864 // Loop through the scenario keys and check which keys are present in simData_,
2865 // adding to the count when a match is identified
2866 // Then check that the count=simData_.size - this ensures that simData_ is a valid
2867 // subset of the scenario - fails is a member of simData is not present in the
2868 // scenario
2869 auto it = simData_.find(key);
2870 if (it == simData_.end()) {
2871 WLOG("simulation data point missing for key " << key);
2872 } else {
2873 if (filter_->allow(key)) {
2874 it->second->setValue(scenario->get(key));
2875 }
2876 count++;
2877 }
2878 }
2879
2880 if (count != simData_.size() && !allowPartialScenarios_) {
2881 ALOG("mismatch between scenario and sim data size, " << count << " vs " << simData_.size());
2882 for (auto it : simData_) {
2883 if (!scenario->has(it.first))
2884 ALOG("Key " << it.first << " missing in scenario");
2885 }
2886 QL_FAIL("mismatch between scenario and sim data size, exit.");
2887 }
