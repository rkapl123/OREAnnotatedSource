101 void processException(bool continueOnError, const std::exception& e, const std::string& curveId = "",
102  ore::analytics::RiskFactorKey::KeyType keyType = ore::analytics::RiskFactorKey::KeyType::None,
103  const bool simDataWritten = false) {
104  string curve;
105  if (keyType != ore::analytics::RiskFactorKey::KeyType::None)
106  curve = to_string(keyType) + "/";
107  curve += curveId;
108 
109  std::string message = "skipping this object in scenario sim market";
110  if (!curve.empty()) {
111  message += " (scenario data was ";
112  if (!simDataWritten)
113  message += "not ";
114  message += "written for this object.)";
115  }
116  if (continueOnError) {
117  std::string exceptionMessage = e.what();
118  /* We do not log a structured curve error message, if the exception message indicates that the problem
119  already occurred in the inti market. In this case we have already logged a structured error there. */
120  if (boost::starts_with(exceptionMessage, "did not find object ")) {
121  ALOG("CurveID: " << curve << ": " << message << ": " << exceptionMessage);
122  } else {
123  ALOG(StructuredCurveErrorMessage(curve, message, exceptionMessage));
124  }
125  } else {
126  QL_FAIL("Object with CurveID '" << curve << "' failed to build in scenario sim market: " << e.what());
127  }

186 boost::shared_ptr<YieldTermStructure>
187 makeYieldCurve(const std::string& curveId, const bool spreaded, const Handle<YieldTermStructure>& initMarketTs,
188  const std::vector<Real>& yieldCurveTimes, const std::vector<Handle<Quote>>& quotes, const DayCounter& dc,
189  const Calendar& cal, const std::string& interpolation, const std::string& extrapolation) {
190  if (ObservationMode::instance().mode() == ObservationMode::Mode::Unregister && !spreaded) {
191  return boost::shared_ptr<YieldTermStructure>(boost::make_shared<QuantExt::InterpolatedDiscountCurve>(
192  yieldCurveTimes, quotes, 0, cal, dc,
193  interpolation == "LogLinear" ? QuantExt::InterpolatedDiscountCurve::Interpolation::logLinear
194  : QuantExt::InterpolatedDiscountCurve::Interpolation::linearZero,
195  extrapolation == "FlatZero" ? QuantExt::InterpolatedDiscountCurve::Extrapolation::flatZero
196  : QuantExt::InterpolatedDiscountCurve::Extrapolation::flatFwd));
197  } else {
198  if (spreaded) {
199  checkDayCounterConsistency(curveId, initMarketTs->dayCounter(), dc);
200  return boost::make_shared<QuantExt::SpreadedDiscountCurve>(
201  initMarketTs, yieldCurveTimes, quotes,
202  interpolation == "LogLinear" ? QuantExt::SpreadedDiscountCurve::Interpolation::logLinear
203  : QuantExt::SpreadedDiscountCurve::Interpolation::linearZero,
204  extrapolation == "FlatZero" ? SpreadedDiscountCurve::Extrapolation::flatZero
205  : SpreadedDiscountCurve::Extrapolation::flatFwd);
206  } else {
207  return boost::make_shared<QuantExt::InterpolatedDiscountCurve2>(
208  yieldCurveTimes, quotes, dc,
209  interpolation == "LogLinear" ? QuantExt::InterpolatedDiscountCurve2::Interpolation::logLinear
210  : QuantExt::InterpolatedDiscountCurve2::Interpolation::linearZero,
211  extrapolation == "FlatZero" ? InterpolatedDiscountCurve2::Extrapolation::flatZero
212  : InterpolatedDiscountCurve2::Extrapolation::flatFwd);
213  }

227 void ScenarioSimMarket::addYieldCurve(const boost::shared_ptr<Market>& initMarket, const std::string& configuration,
228  const RiskFactorKey::KeyType rf, const string& key, const vector<Period>& tenors,
229  bool& simDataWritten, bool simulate, bool spreaded) {
230  Handle<YieldTermStructure> wrapper = (riskFactorYieldCurve(rf) == ore::data::YieldCurveType::Discount)
231  ? initMarket->discountCurve(key, configuration)
232  : initMarket->yieldCurve(riskFactorYieldCurve(rf), key, configuration);
233  QL_REQUIRE(!wrapper.empty(), "yield curve not provided for " << key);
234  QL_REQUIRE(tenors.front() > 0 * Days, "yield curve tenors must not include t=0");
235  // include today
236 
237  // constructing yield curves
238  DayCounter dc = wrapper->dayCounter();
239  vector<Time> yieldCurveTimes(1, 0.0); // include today
240  vector<Date> yieldCurveDates(1, asof_);
241  for (auto& tenor : tenors) {
242  yieldCurveTimes.push_back(dc.yearFraction(asof_, asof_ + tenor));
243  yieldCurveDates.push_back(asof_ + tenor);
244  }
245 
246  vector<Handle<Quote>> quotes;
247  boost::shared_ptr<SimpleQuote> q(new SimpleQuote(1.0));
248  quotes.push_back(Handle<Quote>(q));
249  vector<Real> discounts(yieldCurveTimes.size());
250  std::map<RiskFactorKey, boost::shared_ptr<SimpleQuote>> simDataTmp;
251  std::map<RiskFactorKey, Real> absoluteSimDataTmp;
252  for (Size i = 0; i < yieldCurveTimes.size() - 1; i++) {
253  Real val = wrapper->discount(yieldCurveDates[i + 1]);
254  DLOG("ScenarioSimMarket yield curve " << rf << " " << key << " discount[" << i << "]=" << val);
255  boost::shared_ptr<SimpleQuote> q(new SimpleQuote(spreaded ? 1.0 : val));
256  Handle<Quote> qh(q);
257  quotes.push_back(qh);
258 
259  // Check if the risk factor is simulated before adding it
260  if (simulate) {
261  simDataTmp.emplace(std::piecewise_construct, std::forward_as_tuple(rf, key, i), std::forward_as_tuple(q));
262  // if generating spreaded scenarios, add the absolute value as well
263  if (spreaded) {
264  absoluteSimDataTmp.emplace(std::piecewise_construct, std::forward_as_tuple(rf, key, i),
265  std::forward_as_tuple(val));
266  }
267  }
268  }
269 
270  writeSimData(simDataTmp, absoluteSimDataTmp);
271  simDataWritten = true;
272 
273  boost::shared_ptr<YieldTermStructure> yieldCurve =
274  makeYieldCurve(key, spreaded, wrapper, yieldCurveTimes, quotes, dc, TARGET(), parameters_->interpolation(),
275  parameters_->extrapolation());
276 
277  Handle<YieldTermStructure> ych(yieldCurve);
278  if (wrapper->allowsExtrapolation())
279  ych->enableExtrapolation();
280  yieldCurves_.insert(pair<tuple<string, ore::data::YieldCurveType, string>, Handle<YieldTermStructure>>(
281  make_tuple(Market::defaultConfiguration, riskFactorYieldCurve(rf), key), ych));
282 }

295 ScenarioSimMarket::ScenarioSimMarket(
296  const boost::shared_ptr<Market>& initMarket, const boost::shared_ptr<ScenarioSimMarketParameters>& parameters,
297  const boost::shared_ptr<FixingManager>& fixingManager,
298  const std::string& configuration, const ore::data::CurveConfigurations& curveConfigs,
299  const ore::data::TodaysMarketParameters& todaysMarketParams, const bool continueOnError,
300  const bool useSpreadedTermStructures, const bool cacheSimData, const bool allowPartialScenarios,
301  const IborFallbackConfig& iborFallbackConfig)
302  : SimMarket(), parameters_(parameters), fixingManager_(fixingManager),
303  filter_(boost::make_shared<ScenarioFilter>()), useSpreadedTermStructures_(useSpreadedTermStructures),
304  cacheSimData_(cacheSimData), allowPartialScenarios_(allowPartialScenarios),
305  iborFallbackConfig_(iborFallbackConfig) {
306 
307  LOG("building ScenarioSimMarket...");
308  asof_ = initMarket->asofDate();
309  DLOG("AsOf " << QuantLib::io::iso_date(asof_));
310 
311  // check ssm parameters
312  QL_REQUIRE(parameters_->interpolation() == "LogLinear" || parameters_->interpolation() == "LinearZero",
313  "ScenarioSimMarket: Interpolation (" << parameters_->interpolation()
314  << ") must be set to 'LogLinear' or 'LinearZero'");
315  QL_REQUIRE(parameters_->extrapolation() == "FlatZero" || parameters_->extrapolation() == "FlatFwd",
316  "ScenarioSimMarket: YieldCurves / Extrapolation ('" << parameters_->extrapolation()
317  << "') must be set to 'FlatZero' or 'FlatFwd'");
318  QL_REQUIRE(parameters_->defaultCurveExtrapolation() == "FlatZero" ||
319  parameters_->defaultCurveExtrapolation() == "FlatFwd",
320  "ScenarioSimMarket: DefaultCurves / Extrapolation ('" << parameters_->extrapolation()
321  << "') must be set to 'FlatZero' or 'FlatFwd'");
322 
323  const SmileDynamicsConfig& smileDynamics = curveConfigs.smileDynamicsConfig();
324 
325  for (const auto& param : parameters->parameters()) {
326  try {
327  // we populate the temp containers for each curve and write the result to the global
328  // containers only if the set of data points is complete for this curve
329  std::map<RiskFactorKey, boost::shared_ptr<SimpleQuote>> simDataTmp;
330  std::map<RiskFactorKey, Real> absoluteSimDataTmp;
331 
332  boost::timer::cpu_timer timer;
333 
334  switch (param.first) {
335  case RiskFactorKey::KeyType::FXSpot: {
336  std::map<std::string, Handle<Quote>> fxQuotes;
337  for (const auto& name : param.second.second) {
338  bool simDataWritten = false;
339  try {
340  // constructing fxSpots_
341  DLOG("adding " << name << " FX rates");
342  auto q = boost::make_shared<SimpleQuote>(initMarket->fxSpot(name, configuration)->value());
343  auto qh = Handle<Quote>(q);
344  fxQuotes[name] = qh;
345  // Check if the risk factor is simulated before adding it
346  if (param.second.first) {
347  simDataTmp.emplace(std::piecewise_construct, std::forward_as_tuple(param.first, name),
348  std::forward_as_tuple(q));
349  }
350  writeSimData(simDataTmp, absoluteSimDataTmp);
351  simDataWritten = true;
352  } catch (const std::exception& e) {
353  processException(continueOnError, e, name, param.first, simDataWritten);
354  }
355  }
356  fx_ = boost::make_shared<FXTriangulation>(fxQuotes);
357  break;
358  }
359 
360  case RiskFactorKey::KeyType::DiscountCurve:
361  case RiskFactorKey::KeyType::YieldCurve:
362  for (const auto& name : param.second.second) {
363  bool simDataWritten = false;
364  try {
365  DLOG("building " << name << " yield curve..");
366  vector<Period> tenors = parameters->yieldCurveTenors(name);
367  addYieldCurve(initMarket, configuration, param.first, name, tenors, simDataWritten,
368  param.second.first, useSpreadedTermStructures_);
369  DLOG("building " << name << " yield curve done");
370  } catch (const std::exception& e) {
371  processException(continueOnError, e, name, param.first, simDataWritten);
372  }
373  }
374  break;
375 
376  case RiskFactorKey::KeyType::IndexCurve: {
377  // make sure we built overnight indices first, so that we can build ibor fallback indices
378  // that depend on them
379  std::vector<std::string> indices;
380  for (auto const& i : param.second.second) {
381  bool isOn = false;
382  try {
383  isOn = boost::dynamic_pointer_cast<OvernightIndex>(*initMarket->iborIndex(i, configuration)) !=
384  nullptr;
385  } catch (...) {
386  }
387  if (isOn)
388  indices.insert(indices.begin(), i);
389  else
390  indices.push_back(i);
391  }
392  // loop over sorted indices and build them
393  for (const auto& name : indices) {
394  bool simDataWritten = false;
395  try {
396  DLOG("building " << name << " index curve");
397  std::vector<string> indexTokens;
398  split(indexTokens, name, boost::is_any_of("-"));
399  Handle<IborIndex> index;
400  if (indexTokens[1] == "GENERIC") {
401  // If we have a generic curve build the index using the index currency's discount curve
402  // no need to check for a convention based ibor index in this case
403  index = Handle<IborIndex>(
404  parseIborIndex(name, initMarket->discountCurve(indexTokens[0], configuration)));
405  } else {
406  index = initMarket->iborIndex(name, configuration);
407  }
408  QL_REQUIRE(!index.empty(), "index object for " << name << " not provided");
409  Handle<YieldTermStructure> wrapperIndex = index->forwardingTermStructure();
410  QL_REQUIRE(!wrapperIndex.empty(), "no termstructure for index " << name);
411  vector<string> keys(parameters->yieldCurveTenors(name).size());
412 
413  DayCounter dc = wrapperIndex->dayCounter();
414  vector<Time> yieldCurveTimes(1, 0.0); // include today
415  vector<Date> yieldCurveDates(1, asof_);
416  QL_REQUIRE(parameters->yieldCurveTenors(name).front() > 0 * Days,
417  "yield curve tenors must not include t=0");
418  for (auto& tenor : parameters->yieldCurveTenors(name)) {
419  yieldCurveTimes.push_back(dc.yearFraction(asof_, asof_ + tenor));
420  yieldCurveDates.push_back(asof_ + tenor);
421  }
422 
423  // include today
424  vector<Handle<Quote>> quotes;
425  boost::shared_ptr<SimpleQuote> q(new SimpleQuote(1.0));
426  quotes.push_back(Handle<Quote>(q));
427 
428  for (Size i = 0; i < yieldCurveTimes.size() - 1; i++) {
429  Real val = wrapperIndex->discount(yieldCurveDates[i + 1]);
430  boost::shared_ptr<SimpleQuote> q(new SimpleQuote(useSpreadedTermStructures_ ? 1.0 : val));
431  Handle<Quote> qh(q);
432  quotes.push_back(qh);
433 
434  simDataTmp.emplace(std::piecewise_construct, std::forward_as_tuple(param.first, name, i),
435  std::forward_as_tuple(q));
436  if (useSpreadedTermStructures_) {
437  absoluteSimDataTmp.emplace(std::piecewise_construct,
438  std::forward_as_tuple(param.first, name, i),
439  std::forward_as_tuple(val));
440  }
441  // FIXME where do we check whether the risk factor is simulated?
442  DLOG("ScenarioSimMarket index curve " << name << " discount[" << i << "]=" << val);
443  }
444 
445  writeSimData(simDataTmp, absoluteSimDataTmp);
446  simDataWritten = true;
447 
448  boost::shared_ptr<YieldTermStructure> indexCurve = makeYieldCurve(
449  name, useSpreadedTermStructures_, wrapperIndex, yieldCurveTimes, quotes, dc,
450  index->fixingCalendar(), parameters_->interpolation(), parameters_->extrapolation());
451 
452  Handle<YieldTermStructure> ich(indexCurve);
453  if (wrapperIndex->allowsExtrapolation())
454  ich->enableExtrapolation();
455 
456  boost::shared_ptr<IborIndex> i = index->clone(ich);
457  if (iborFallbackConfig_.isIndexReplaced(name, asof_)) {
458  // handle ibor fallback indices
459  auto fallbackData = iborFallbackConfig_.fallbackData(name);
460  auto f = iborIndices_.find(make_pair(Market::defaultConfiguration, fallbackData.rfrIndex));
461  QL_REQUIRE(f != iborIndices_.end(),
462  "Could not build ibor fallback index '"
463  << name << "', because rfr index '" << fallbackData.rfrIndex
464  << "' is not present in scenario sim market, is the rfr index in the "
465  "scenario sim market parameters?");
466  auto rfrInd = boost::dynamic_pointer_cast<OvernightIndex>(*f->second);
467  QL_REQUIRE(rfrInd != nullptr,
468  "Could not cast '"
469  << fallbackData.rfrIndex
470  << "' to overnight index when building the ibor fallback index '" << name
471  << "'");
472  if (auto original = boost::dynamic_pointer_cast<OvernightIndex>(i))
473  i = boost::make_shared<QuantExt::FallbackOvernightIndex>(
474  original, rfrInd, fallbackData.spread, fallbackData.switchDate,
475  iborFallbackConfig_.useRfrCurveInSimulationMarket());
476  else
477  i = boost::make_shared<QuantExt::FallbackIborIndex>(
478  i, rfrInd, fallbackData.spread, fallbackData.switchDate,
479  iborFallbackConfig_.useRfrCurveInSimulationMarket());
480  DLOG("built ibor fall back index '"
481  << name << "' with rfr index '" << fallbackData.rfrIndex << "', spread "
482  << fallbackData.spread << ", use rfr curve in scen sim market: " << std::boolalpha
483  << iborFallbackConfig_.useRfrCurveInSimulationMarket());
484  }
485  iborIndices_.insert(
486  make_pair(make_pair(Market::defaultConfiguration, name), Handle<IborIndex>(i)));
487  DLOG("building " << name << " index curve done");
