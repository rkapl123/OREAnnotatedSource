132 void ScenarioSimMarket::addYieldCurve(const boost::shared_ptr<Market>& initMarket, const std::string& configuration,
133  const RiskFactorKey::KeyType rf, const string& key, const vector<Period>& tenors,
134  const string& dayCounter, bool simulate) {
135  Handle<YieldTermStructure> wrapper = initMarket->yieldCurve(riskFactorYieldCurve(rf), key, configuration);
136  QL_REQUIRE(!wrapper.empty(), "yield curve not provided for " << key);
137  QL_REQUIRE(tenors.front() > 0 * Days, "yield curve tenors must not include t=0");
138  // include today
139 
140  // constructing yield curves
141  DayCounter dc = ore::data::parseDayCounter(dayCounter); // used to convert YieldCurve Periods to Times
142  vector<Time> yieldCurveTimes(1, 0.0); // include today
143  vector<Date> yieldCurveDates(1, asof_);
144  for (auto& tenor : tenors) {
145  yieldCurveTimes.push_back(dc.yearFraction(asof_, asof_ + tenor));
146  yieldCurveDates.push_back(asof_ + tenor);
147  }
148 
149  vector<Handle<Quote>> quotes;
150  boost::shared_ptr<SimpleQuote> q(new SimpleQuote(1.0));
151  quotes.push_back(Handle<Quote>(q));
152  vector<Real> discounts(yieldCurveTimes.size());
153  std::map<RiskFactorKey, boost::shared_ptr<SimpleQuote>> simDataTmp;
154  for (Size i = 0; i < yieldCurveTimes.size() - 1; i++) {
155  boost::shared_ptr<SimpleQuote> q(new SimpleQuote(wrapper->discount(yieldCurveDates[i + 1])));
156  Handle<Quote> qh(q);
157  quotes.push_back(qh);
158 
159  // Check if the risk factor is simulated before adding it
160  if (simulate) {
161  simDataTmp.emplace(std::piecewise_construct, std::forward_as_tuple(rf, key, i), std::forward_as_tuple(q));
162  DLOG("ScenarioSimMarket yield curve " << key << " discount[" << i << "]=" << q->value());
163  }
164  }
165 
166  boost::shared_ptr<YieldTermStructure> yieldCurve;
167 
168  if (ObservationMode::instance().mode() == ObservationMode::Mode::Unregister) {
169  yieldCurve = boost::shared_ptr<YieldTermStructure>(
170  new QuantExt::InterpolatedDiscountCurve(yieldCurveTimes, quotes, 0, TARGET(), dc));
171  } else {
172  yieldCurve = boost::shared_ptr<YieldTermStructure>(
173  new QuantExt::InterpolatedDiscountCurve2(yieldCurveTimes, quotes, dc));
174  }
175 
176  Handle<YieldTermStructure> ych(yieldCurve);
177  if (wrapper->allowsExtrapolation())
178  ych->enableExtrapolation();
179  yieldCurves_.insert(pair<tuple<string, ore::data::YieldCurveType, string>, Handle<YieldTermStructure>>(
180  make_tuple(Market::defaultConfiguration, riskFactorYieldCurve(rf), key), ych));
181  simData_.insert(simDataTmp.begin(), simDataTmp.end());
182 }

192 ScenarioSimMarket::ScenarioSimMarket(
193  const boost::shared_ptr<Market>& initMarket, const boost::shared_ptr<ScenarioSimMarketParameters>& parameters,
194  const Conventions& conventions, const boost::shared_ptr<FixingManager>& fixingManager,
195  const std::string& configuration, const ore::data::CurveConfigurations& curveConfigs,
196  const ore::data::TodaysMarketParameters& todaysMarketParams, const bool continueOnError)
197  : SimMarket(conventions), parameters_(parameters), fixingManager_(fixingManager),
198  filter_(boost::make_shared<ScenarioFilter>()) {
199 
200  LOG("building ScenarioSimMarket...");
201  asof_ = initMarket->asofDate();
202  LOG("AsOf " << QuantLib::io::iso_date(asof_));
203 
204  // Sort parameters so they get processed in correct order
205  map<RiskFactorKey::KeyType, pair<bool, set<string>>> params;
206  params.insert(parameters->parameters().begin(), parameters->parameters().end());
207 
208  for (const auto& param : params) {
209  try {
210  std::map<RiskFactorKey, boost::shared_ptr<SimpleQuote>> simDataTmp;
211 
212  switch (param.first) {
213  case RiskFactorKey::KeyType::FXSpot:
214  for (const auto& name : param.second.second) {
215  try {
216  // constructing fxSpots_
217  LOG("adding " << name << " FX rates");
218  boost::shared_ptr<SimpleQuote> q(new SimpleQuote(initMarket->fxSpot(name, configuration)->value()));
219  Handle<Quote> qh(q);
220  fxSpots_[Market::defaultConfiguration].addQuote(name, qh);
221  // Check if the risk factor is simulated before adding it
222  if (param.second.first) {
223  simDataTmp.emplace(std::piecewise_construct, std::forward_as_tuple(param.first, name),
224  std::forward_as_tuple(q));
225  }
226  } catch (const std::exception& e) {
227  processException(continueOnError, e);
228  }
229  }
230  break;
231 
232  case RiskFactorKey::KeyType::DiscountCurve:
233  case RiskFactorKey::KeyType::YieldCurve:
234  for (const auto& name : param.second.second) {
235  try {
236  LOG("building " << name << " yield curve..");
237  vector<Period> tenors = parameters->yieldCurveTenors(name);
238  addYieldCurve(initMarket, configuration, param.first, name, tenors,
239  parameters->yieldCurveDayCounter(name), param.second.first);
240  LOG("building " << name << " yield curve done");
241  } catch (const std::exception& e) {
242  processException(continueOnError, e);
243  }
244  }
245  break;
246 
247  case RiskFactorKey::KeyType::IndexCurve:
248  for (const auto& name : param.second.second) {
249  try {
250  LOG("building " << name << " index curve");
251  std::vector<string> indexTokens;
252  split(indexTokens, name, boost::is_any_of("-"));
253  Handle<IborIndex> index;
254  if (indexTokens[1] == "GENERIC") {
255  // If we have a generic curve build the index using the index currency's discount curve
256  index = Handle<IborIndex>(
257  parseIborIndex(name, initMarket->discountCurve(indexTokens[0], configuration)));
258  } else {
259  index = initMarket->iborIndex(name, configuration);
260  }
261  QL_REQUIRE(!index.empty(), "index object for " << name << " not provided");
262  Handle<YieldTermStructure> wrapperIndex = index->forwardingTermStructure();
263  QL_REQUIRE(!wrapperIndex.empty(), "no termstructure for index " << name);
264  vector<string> keys(parameters->yieldCurveTenors(name).size());
265 
266  DayCounter dc = ore::data::parseDayCounter(
267  parameters->yieldCurveDayCounter(name)); // used to convert YieldCurve Periods to Times
268  vector<Time> yieldCurveTimes(1, 0.0); // include today
269  vector<Date> yieldCurveDates(1, asof_);
270  QL_REQUIRE(parameters->yieldCurveTenors(name).front() > 0 * Days,
271  "yield curve tenors must not include t=0");
272  for (auto& tenor : parameters->yieldCurveTenors(name)) {
273  yieldCurveTimes.push_back(dc.yearFraction(asof_, asof_ + tenor));
274  yieldCurveDates.push_back(asof_ + tenor);
275  }
276 
277  // include today
278  vector<Handle<Quote>> quotes;
279  boost::shared_ptr<SimpleQuote> q(new SimpleQuote(1.0));
280  quotes.push_back(Handle<Quote>(q));
281 
282  for (Size i = 0; i < yieldCurveTimes.size() - 1; i++) {
283  boost::shared_ptr<SimpleQuote> q(
284  new SimpleQuote(wrapperIndex->discount(yieldCurveDates[i + 1])));
285  Handle<Quote> qh(q);
286  quotes.push_back(qh);
287 
288  simDataTmp.emplace(std::piecewise_construct, std::forward_as_tuple(param.first, name, i),
289  std::forward_as_tuple(q));
290 
291  DLOG("ScenarioSimMarket index curve " << name << " discount[" << i << "]=" << q->value());
292  }
293  // FIXME interpolation fixed to linear, added to xml??
294  boost::shared_ptr<YieldTermStructure> indexCurve;
295  if (ObservationMode::instance().mode() == ObservationMode::Mode::Unregister) {
296  indexCurve = boost::shared_ptr<YieldTermStructure>(new QuantExt::InterpolatedDiscountCurve(
297  yieldCurveTimes, quotes, 0, index->fixingCalendar(), dc));
298  } else {
299  indexCurve = boost::shared_ptr<YieldTermStructure>(
300  new QuantExt::InterpolatedDiscountCurve2(yieldCurveTimes, quotes, dc));
301  }
302 
303  // wrapped curve, is slower than a native curve
304  // boost::shared_ptr<YieldTermStructure> correctedIndexCurve(
305  // new StaticallyCorrectedYieldTermStructure(
306  // discountCurves_[ccy], initMarket->discountCurve(ccy, configuration),
307  // wrapperIndex));
308 
309  Handle<YieldTermStructure> ich(indexCurve);
310  // Handle<YieldTermStructure> ich(correctedIndexCurve);
311  if (wrapperIndex->allowsExtrapolation())
312  ich->enableExtrapolation();
313 
314  boost::shared_ptr<IborIndex> i(index->clone(ich));
315  Handle<IborIndex> ih(i);
316  iborIndices_.insert(pair<pair<string, string>, Handle<IborIndex>>(
317  make_pair(Market::defaultConfiguration, name), ih));
318  LOG("building " << name << " index curve done");
319  } catch (const std::exception& e) {
320  processException(continueOnError, e);
321  }
322  }
323  break;
324 
325  case RiskFactorKey::KeyType::EquitySpot:
326  for (const auto& name : param.second.second) {
327  try {
328  // building equity spots
329  LOG("adding " << name << " equity spot...");
330  Real spotVal = initMarket->equitySpot(name, configuration)->value();
331  boost::shared_ptr<SimpleQuote> q(new SimpleQuote(spotVal));
332  Handle<Quote> qh(q);
333  equitySpots_.insert(
334  pair<pair<string, string>, Handle<Quote>>(make_pair(Market::defaultConfiguration, name), qh));
335  simDataTmp.emplace(std::piecewise_construct, std::forward_as_tuple(param.first, name),
336  std::forward_as_tuple(q));
337  LOG("adding " << name << " equity spot done");
338  } catch (const std::exception& e) {
339  processException(continueOnError, e);
340  }
341  }
342  break;
343 
344  case RiskFactorKey::KeyType::DividendYield:
345  for (const auto& name : param.second.second) {
346  try {
347  LOG("building " << name << " equity dividend yield curve..");
348  vector<Period> tenors = parameters->equityDividendTenors(name);
349  addYieldCurve(initMarket, configuration, param.first, name, tenors,
350  parameters->yieldCurveDayCounter(name), param.second.first);
351  LOG("building " << name << " equity dividend yield curve done");
352 
353  // Equity spots and Yield/Index curves added first so we can now build equity index
354  // First get Forecast Curve
355  string forecastCurve;
356  if (curveConfigs.hasEquityCurveConfig(name)) {
357  // From the equity config, get the currency and forecast curve of the equity
358  auto eqVolConfig = curveConfigs.equityCurveConfig(name);
359  string forecastName = eqVolConfig->forecastingCurve();
360  string eqCcy = eqVolConfig->currency();
361  // Build a YieldCurveSpec and extract the yieldCurveSpec name
362  YieldCurveSpec ycspec(eqCcy, forecastName);
363  forecastCurve = ycspec.name();
364  TLOG("Got forecast curve '" << forecastCurve << "' from equity curve config for " << name);
365  }
366 
367  // Get the nominal term structure from this scenario simulation market
368  Handle<YieldTermStructure> forecastTs = getYieldCurve(forecastCurve, todaysMarketParams, Market::defaultConfiguration);
369  Handle<EquityIndex> curve = initMarket->equityCurve(name, configuration);
370 
371  // If forecast term structure is empty, fall back on this scenario simulation market's discount curve
372  if (forecastTs.empty()) {
373  string ccy = curve->currency().code();
374  TLOG("Falling back on the discount curve for currency '" << ccy << "', the currency of inflation index '" << name << "'");
375  forecastTs = discountCurve(ccy);
376  }
377  boost::shared_ptr<EquityIndex> ei(
378  curve->clone(equitySpot(name, configuration), forecastTs,
379  yieldCurve(YieldCurveType::EquityDividend, name, configuration)));
380  Handle<EquityIndex> eh(ei);
381  equityCurves_.insert(pair<pair<string, string>, Handle<EquityIndex>>(
382  make_pair(Market::defaultConfiguration, name), eh));
383  } catch (const std::exception& e) {
384  processException(continueOnError, e);
385  }
386  }
387  break;
388 
389  case RiskFactorKey::KeyType::SecuritySpread:
390  for (const auto& name : param.second.second) {
391  try {
392  DLOG("Adding security spread " << name << " from configuration " << configuration);
393  // we have a security spread for each security, so no try-catch block required
394  boost::shared_ptr<SimpleQuote> spreadQuote(
395  new SimpleQuote(initMarket->securitySpread(name, configuration)->value()));
396  if (param.second.first) {
397  simDataTmp.emplace(std::piecewise_construct, std::forward_as_tuple(param.first, name),
398  std::forward_as_tuple(spreadQuote));
399  }
400  securitySpreads_.insert(pair<pair<string, string>, Handle<Quote>>(
401  make_pair(Market::defaultConfiguration, name), Handle<Quote>(spreadQuote)));
402 
403  DLOG("Adding security recovery rate " << name << " from configuration " << configuration);
404  // security recovery rates are optional, so we need a try-catch block
405  try {
406  boost::shared_ptr<SimpleQuote> recoveryQuote(
407  new SimpleQuote(initMarket->recoveryRate(name, configuration)->value()));
408  // TODO this comes from the default curves section in the parameters,
409  // do we want to specify the simulation of security recovery rates separately?
410  if (parameters->simulateRecoveryRates()) {
411  simDataTmp.emplace(std::piecewise_construct,
412  std::forward_as_tuple(RiskFactorKey::KeyType::RecoveryRate, name),
413  std::forward_as_tuple(recoveryQuote));
414  }
415  recoveryRates_.insert(pair<pair<string, string>, Handle<Quote>>(
416  make_pair(Market::defaultConfiguration, name), Handle<Quote>(recoveryQuote)));
417  } catch (const std::exception& e) {
418  // security recovery rates are optional, therefore we never throw
419  ALOG("skipping this object: " << e.what());
420  }
421  } catch (const std::exception& e) {
422  processException(continueOnError, e);
423  }
424  }
425  break;
426 
427  case RiskFactorKey::KeyType::SwaptionVolatility:
428  case RiskFactorKey::KeyType::YieldVolatility:
429  for (const auto& name : param.second.second) {
430  try {
431  // set parameters for swaption resp. yield vols
432  RelinkableHandle<SwaptionVolatilityStructure> wrapper;
433  vector<Period> optionTenors, underlyingTenors;
434  vector<Real> strikeSpreads;
435  string shortSwapIndexBase = "", swapIndexBase = "";
436  bool isCube, atmOnly;
437  if (param.first == RiskFactorKey::KeyType::SwaptionVolatility) {
438  LOG("building " << name << " swaption volatility curve...");
439  wrapper.linkTo(*initMarket->swaptionVol(name, configuration));
440  shortSwapIndexBase = initMarket->shortSwapIndexBase(name, configuration);
441  swapIndexBase = initMarket->swapIndexBase(name, configuration);
442  isCube = parameters->swapVolIsCube();
443  optionTenors = parameters->swapVolExpiries();
444  underlyingTenors = parameters->swapVolTerms();
445  strikeSpreads = parameters->swapVolStrikeSpreads();
446  atmOnly = parameters->simulateSwapVolATMOnly();
447  } else {
448  LOG("building " << name << " yield volatility curve...");
449  wrapper.linkTo(*initMarket->yieldVol(name, configuration));
450  isCube = false;
451  optionTenors = parameters->yieldVolExpiries();
452  underlyingTenors = parameters->yieldVolTerms();
453  strikeSpreads = {0.0};
454  atmOnly = true;
455  }
456  LOG("Initial market " << name << " yield volatility type = " << wrapper->volatilityType());
457 
458  // If volatility type is not Normal and we have swaptions, convert to Normal for the simulation
459  // Notice that this is not possible for yield volatilities, since the ATM level is not known
460  if (wrapper->volatilityType() != Normal &&
461  param.first == RiskFactorKey::KeyType::SwaptionVolatility) {
462  // FIXME we can not convert constant swaption vol structures yet
463  if (boost::dynamic_pointer_cast<ConstantSwaptionVolatility>(*wrapper) != nullptr) {
464  ALOG("Constant swaption volatility found in configuration "
465  << configuration << " for currency " << name
466  << " will not be converted to normal");
467  } else {
468  // Get swap index associated with this volatility structure
469  Handle<SwapIndex> swapIndex = initMarket->swapIndex(swapIndexBase, configuration);
470  Handle<SwapIndex> shortSwapIndex =
471  initMarket->swapIndex(shortSwapIndexBase, configuration);
472  // Set up swaption volatility converter
473  SwaptionVolatilityConverter converter(asof_, *wrapper, *swapIndex, *shortSwapIndex,
474  Normal);
475  wrapper.linkTo(converter.convert());
476  LOG("Converting swaption volatilities in configuration "
477  << configuration << " with currency " << name
478  << " to normal swaption volatilities");
479  }
480  }
481  Handle<SwaptionVolatilityStructure> svp;
482  if (param.second.first) {
483  LOG("Simulating (" << wrapper->volatilityType() << ") yield vols for ccy " << name);
484  DLOG("YieldVol atmOnly : " << (atmOnly ? "True" : "False"));
485  DLOG("YieldVol isCube : " << (isCube ? "True" : "False"));
486  if (atmOnly) {
487  QL_REQUIRE(strikeSpreads.size() == 1 && close_enough(strikeSpreads[0], 0),
488  "for atmOnly strikeSpreads must be {0.0}");
489  } else {
490  QL_REQUIRE(isCube, "Only atmOnly simulation supported for swaption vol surfaces");
491  }
492  boost::shared_ptr<QuantLib::SwaptionVolatilityCube> cube;
493  if (isCube) {
494  boost::shared_ptr<SwaptionVolCubeWithATM> tmp =
495  boost::dynamic_pointer_cast<SwaptionVolCubeWithATM>(*wrapper);
496  QL_REQUIRE(tmp, "swaption cube missing")
497  cube = tmp->cube();
498  }
499  vector<vector<Handle<Quote>>> quotes, atmQuotes;
500  quotes.resize(optionTenors.size() * underlyingTenors.size(),
501  vector<Handle<Quote>>(strikeSpreads.size(), Handle<Quote>()));
502  atmQuotes.resize(optionTenors.size(),
503  std::vector<Handle<Quote>>(underlyingTenors.size(), Handle<Quote>()));
504  vector<vector<Real>> shift(optionTenors.size(), vector<Real>(underlyingTenors.size(), 0.0));
505  Size atmSlice = std::find_if(strikeSpreads.begin(), strikeSpreads.end(),
506  [](const Real s) { return close_enough(s, 0.0); }) -
507  strikeSpreads.begin();
508  QL_REQUIRE(atmSlice < strikeSpreads.size(),
509  "could not find atm slice (strikeSpreads do not contain 0.0)");
510  for (Size k = 0; k < strikeSpreads.size(); ++k) {
511  for (Size i = 0; i < optionTenors.size(); ++i) {
512  for (Size j = 0; j < underlyingTenors.size(); ++j) {
513  Real strike = Null<Real>();
514  if (!atmOnly && cube)
515  strike = cube->atmStrike(optionTenors[i], underlyingTenors[j]) +
