28void DeltaScenario::add(const ore::analytics::RiskFactorKey& key, Real value) {
29 QL_REQUIRE(baseScenario_->has(key), "base scenario must also possess key");

34Real DeltaScenario::get(const ore::analytics::RiskFactorKey& key) const {
