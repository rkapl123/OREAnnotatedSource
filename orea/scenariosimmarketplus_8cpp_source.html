27void ScenarioSimMarketPlus::applyScenario(const boost::shared_ptr<ore::analytics::Scenario>& scenario) {
28
29 bool missingPoint = false;
30 auto deltaScenario = boost::dynamic_pointer_cast<DeltaScenario>(scenario);
31
32 /*! our assumption is that either all or none of the scenarios we apply are
33 delta scenarios! */
34
35 if (deltaScenario != nullptr) {
36 for (auto const& key : diffToBaseKeys_) {
37 auto it = simData_.find(key);
38 if (it != simData_.end()) {
39 it->second->setValue(baseScenario_->get(key));
40 }
41 }
42 diffToBaseKeys_.clear();
43 auto delta = deltaScenario->delta();
44 for (auto const& key : delta->keys()) {
45 auto it = simData_.find(key);
46 if (it == simData_.end()) {
47 ALOG("simulation data point missing for key " << key);
48 missingPoint = true;
49 } else {
50 if (filter_->allow(key)) {
51 it->second->setValue(delta->get(key));
52 diffToBaseKeys_.insert(key);
53 }
54 }
55 }
56 QL_REQUIRE(!missingPoint, "simulation data points missing from scenario, exit.");
57 asof_ = scenario->asof();
58 } else {
59 ScenarioSimMarket::applyScenario(scenario);
60 }
61}
