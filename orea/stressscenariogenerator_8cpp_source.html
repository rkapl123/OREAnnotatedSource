43 void StressScenarioGenerator::generateScenarios() {
44  Date asof = baseScenario_->asof();
45  for (Size i = 0; i < stressData_->data().size(); ++i) {
46  StressTestScenarioData::StressTestData data = stressData_->data().at(i);
47  boost::shared_ptr<Scenario> scenario = stressScenarioFactory_->buildScenario(asof, data.label);
48 
49  if (simMarketData_->simulateFxSpots())
50  addFxShifts(data, scenario);
51  addEquityShifts(data, scenario);
52  addDiscountCurveShifts(data, scenario);
53  addIndexCurveShifts(data, scenario);
54  addYieldCurveShifts(data, scenario);
55  if (simMarketData_->simulateFXVols())
56  addFxVolShifts(data, scenario);
57  if (simMarketData_->simulateEquityVols())
58  addEquityVolShifts(data, scenario);
59  if (simMarketData_->simulateSwapVols())
60  addSwaptionVolShifts(data, scenario);
61  if (simMarketData_->simulateCapFloorVols())
62  addCapFloorVolShifts(data, scenario);
63  if (simMarketData_->securitySpreadsSimulate())
64  addSecuritySpreadShifts(data, scenario);
65  if (simMarketData_->simulateRecoveryRates())
66  addRecoveryRateShifts(data, scenario);
67  if (simMarketData_->simulateSurvivalProbabilities())
68  addSurvivalProbabilityShifts(data, scenario);
69 
70  scenarios_.push_back(scenario);
71  }
72 
73  LOG("stress scenario generator initialised");
74 }

76 void StressScenarioGenerator::addFxShifts(StressTestScenarioData::StressTestData& std,
77  boost::shared_ptr<Scenario>& scenario) {
78  for (auto d : std.fxShifts) {
79  string ccypair = d.first; // foreign + domestic;
80 
81  // Is this too strict?
82  // - implemented to avoid cases where input cross FX rates are not consistent
83  // - Consider an example (baseCcy = EUR) of a GBPUSD FX trade - two separate routes to pricing
84  // - (a) call GBPUSD FX rate from sim market
85  // - (b) call GBPEUR and EURUSD FX rates, manually join them to obtain GBPUSD
86  // - now, if GBPUSD is an explicit risk factor in sim market, consider what happens
87  // - if we bump GBPUSD value and leave other FX rates unchanged (for e.g. a sensitivity analysis)
88  // - (a) the value of the trade changes
89  // - (b) the value of the GBPUSD trade stays the same
90  // in light of the above we restrict the universe of FX pairs that we support here for the time being
91  string baseCcy = simMarketData_->baseCcy();
92  string foreign = ccypair.substr(0, 3);
93  string domestic = ccypair.substr(3);
94  QL_REQUIRE((domestic == baseCcy) || (foreign == baseCcy),
95  "SensitivityScenarioGenerator does not support cross FX pairs("
96  << ccypair << ", but base currency is " << baseCcy << ")");
97 
98  LOG("Apply stress scenario to fx " << ccypair);
99 
100  StressTestScenarioData::SpotShiftData data = d.second;
101  ShiftType type = parseShiftType(data.shiftType);
102  bool relShift = (type == ShiftType::Relative);
103  // QL_REQUIRE(type == ShiftType::Relative, "FX scenario type must be relative");
104  Real size = data.shiftSize;
105 
106  RiskFactorKey key(RiskFactorKey::KeyType::FXSpot, ccypair);
107  Real rate = scenario->get(key);
108  Real newRate = relShift ? rate * (1.0 + size) : (rate + size);
109  scenario->add(RiskFactorKey(RiskFactorKey::KeyType::FXSpot, ccypair), newRate);
110  }
111  LOG("FX scenarios done");
112 }

133 void StressScenarioGenerator::addDiscountCurveShifts(StressTestScenarioData::StressTestData& std,
134  boost::shared_ptr<Scenario>& scenario) {
135  Date asof = baseScenario_->asof();
136 
137  for (auto d : std.discountCurveShifts) {
138  string ccy = d.first;
139  LOG("Apply stress scenario to discount curve " << ccy);
140 
141  Size n_ten = simMarketData_->yieldCurveTenors(ccy).size();
142  // original curves' buffer
143  std::vector<Real> zeros(n_ten);
144  std::vector<Real> times(n_ten);
145  // buffer for shifted zero curves
146  std::vector<Real> shiftedZeros(n_ten);
147 
148  StressTestScenarioData::CurveShiftData data = d.second;
149  ShiftType shiftType = parseShiftType(data.shiftType);
150  //DayCounter dc = parseDayCounter(simMarketData_->yieldCurveDayCounter(ccy));
151  DayCounter dc = simMarket_->discountCurve(ccy)->dayCounter();
152 
153  for (Size j = 0; j < n_ten; ++j) {
154  Date d = asof + simMarketData_->yieldCurveTenors(ccy)[j];
155  times[j] = dc.yearFraction(asof, d);
156  RiskFactorKey key(RiskFactorKey::KeyType::DiscountCurve, ccy, j);
157  Real quote = baseScenario_->get(key);
158  zeros[j] = -std::log(quote) / times[j];
159  }
160 
161  std::vector<Period> shiftTenors = data.shiftTenors;
162  QL_REQUIRE(shiftTenors.size() > 0, "Discount shift tenors not specified");
163  std::vector<Real> shifts = data.shifts;
164  QL_REQUIRE(shiftTenors.size() == shifts.size(), "shift tenor and shift size vectors do not match");
165  std::vector<Time> shiftTimes(shiftTenors.size());
166  for (Size j = 0; j < shiftTenors.size(); ++j)
167  shiftTimes[j] = dc.yearFraction(asof, asof + shiftTenors[j]);
168 
169  // apply zero rate shift at tenor point j
170  for (Size j = 0; j < shiftTenors.size(); ++j)
171  applyShift(j, shifts[j], true, shiftType, shiftTimes, zeros, times, shiftedZeros, j == 0 ? true : false);
172 
173  // store shifted discount curve in the scenario
174  for (Size k = 0; k < n_ten; ++k) {
175  Real shiftedDiscount = exp(-shiftedZeros[k] * times[k]);
176  scenario->add(RiskFactorKey(RiskFactorKey::KeyType::DiscountCurve, ccy, k), shiftedDiscount);
177  }
178  }
179  LOG("Discount curve stress scenarios done");
180 }

439 void StressScenarioGenerator::addSwaptionVolShifts(StressTestScenarioData::StressTestData& std,
440  boost::shared_ptr<Scenario>& scenario) {
441  Date asof = baseScenario_->asof();
442 
443  for (auto d : std.swaptionVolShifts) {
444  std::string key = d.first;
445  LOG("Apply stress scenario to swaption vol structure '" << key << "'");
446 
447  Size n_swvol_term = simMarketData_->swapVolTerms(key).size();
448  Size n_swvol_exp = simMarketData_->swapVolExpiries(key).size();
449 
450  vector<vector<Real>> volData(n_swvol_exp, vector<Real>(n_swvol_term, 0.0));
451  vector<Real> volExpiryTimes(n_swvol_exp, 0.0);
452  vector<Real> volTermTimes(n_swvol_term, 0.0);
453  vector<vector<Real>> shiftedVolData(n_swvol_exp, vector<Real>(n_swvol_term, 0.0));
454 
455  StressTestScenarioData::SwaptionVolShiftData data = d.second;
456  ShiftType shiftType = parseShiftType(data.shiftType);
457  map<pair<Period, Period>, Real> shifts = data.shifts;
458 
459  vector<Real> shiftExpiryTimes(data.shiftExpiries.size(), 0.0);
460  vector<Real> shiftTermTimes(data.shiftTerms.size(), 0.0);
461 
462  DayCounter dc = simMarket_->swaptionVol(key)->dayCounter();
463 
464  // cache original vol data
465  for (Size j = 0; j < n_swvol_exp; ++j) {
466  Date expiry = asof + simMarketData_->swapVolExpiries(key)[j];
467  volExpiryTimes[j] = dc.yearFraction(asof, expiry);
468  }
469  for (Size j = 0; j < n_swvol_term; ++j) {
470  Date term = asof + simMarketData_->swapVolTerms(key)[j];
471  volTermTimes[j] = dc.yearFraction(asof, term);
472  }
473  for (Size j = 0; j < n_swvol_exp; ++j) {
474  for (Size k = 0; k < n_swvol_term; ++k) {
475  Size idx = j * n_swvol_term + k;
476 
477  RiskFactorKey rf(RiskFactorKey::KeyType::SwaptionVolatility, key, idx);
478  volData[j][k] = baseScenario_->get(rf);
479  }
480  }
481 
482  // cache tenor times
483  for (Size j = 0; j < shiftExpiryTimes.size(); ++j)
484  shiftExpiryTimes[j] = dc.yearFraction(asof, asof + data.shiftExpiries[j]);
485  for (Size j = 0; j < shiftTermTimes.size(); ++j)
486  shiftTermTimes[j] = dc.yearFraction(asof, asof + data.shiftTerms[j]);
487 
488  // loop over shift expiries and terms
489  // FIXME: apply same shifts to all strikes when present
490  for (Size j = 0; j < shiftExpiryTimes.size(); ++j) {
491  for (Size k = 0; k < shiftTermTimes.size(); ++k) {
492  // Size strikeBucket = 0; // FIXME
493  Real shift = 0.0;
494  pair<Period, Period> key(data.shiftExpiries[j], data.shiftTerms[k]);
495  if (shifts.size() == 0)
496  shift = data.parallelShiftSize;
497  else {
498  QL_REQUIRE(shifts.find(key) != shifts.end(), "swaption vol shift not found for expiry "
499  << data.shiftExpiries[j] << " and term "
500  << data.shiftTerms[k]);
501  shift = shifts[key];
502  }
503  applyShift(j, k, shift, true, shiftType, shiftExpiryTimes, shiftTermTimes, volExpiryTimes, volTermTimes,
504  volData, shiftedVolData, j == 0 && k == 0);
505  }
506  }
507 
508  // add shifted vol data to the scenario
509  for (Size jj = 0; jj < n_swvol_exp; ++jj) {
510  for (Size kk = 0; kk < n_swvol_term; ++kk) {
511  Size idx = jj * n_swvol_term + kk;
512  scenario->add(RiskFactorKey(RiskFactorKey::KeyType::SwaptionVolatility, key, idx),
513  shiftedVolData[jj][kk]);
514  }
515  }
516  }

520 void StressScenarioGenerator::addCapFloorVolShifts(StressTestScenarioData::StressTestData& std,
521  boost::shared_ptr<Scenario>& scenario) {
522  Date asof = baseScenario_->asof();
523 
524  for (auto d : std.capVolShifts) {
525  std::string key = d.first;
526  LOG("Apply stress scenario to cap/floor vol structure " << key);
527 
528  vector<Real> volStrikes = simMarketData_->capFloorVolStrikes(key);
529  // Strikes may be empty which indicates that the optionlet structure in the simulation market is an ATM curve
530  if (volStrikes.empty()) {
531  volStrikes = {0.0};
532  }
533  Size n_cfvol_strikes = volStrikes.size();
534 
535  Size n_cfvol_exp = simMarketData_->capFloorVolExpiries(key).size();
536  vector<vector<Real>> volData(n_cfvol_exp, vector<Real>(n_cfvol_strikes, 0.0));
537  vector<Real> volExpiryTimes(n_cfvol_exp, 0.0);
538  vector<vector<Real>> shiftedVolData(n_cfvol_exp, vector<Real>(n_cfvol_strikes, 0.0));
539 
540  StressTestScenarioData::CapFloorVolShiftData data = d.second;
541 
542  ShiftType shiftType = parseShiftType(data.shiftType);
543  vector<Real> shifts = data.shifts;
544  vector<Real> shiftExpiryTimes(data.shiftExpiries.size(), 0.0);
545 
546  //DayCounter dc = parseDayCounter(simMarketData_->capFloorVolDayCounter(key));
547  DayCounter dc = simMarket_->capFloorVol(key)->dayCounter();
548 
549  // cache original vol data
550  for (Size j = 0; j < n_cfvol_exp; ++j) {
551  Date expiry = asof + simMarketData_->capFloorVolExpiries(key)[j];
552  volExpiryTimes[j] = dc.yearFraction(asof, expiry);
553  }
554  for (Size j = 0; j < n_cfvol_exp; ++j) {
555  for (Size k = 0; k < n_cfvol_strikes; ++k) {
556  Size idx = j * n_cfvol_strikes + k;
557  volData[j][k] =
558  baseScenario_->get(RiskFactorKey(RiskFactorKey::KeyType::OptionletVolatility, key, idx));
559  }
560  }
561 
562  // cache tenor times
563  for (Size j = 0; j < shiftExpiryTimes.size(); ++j)
564  shiftExpiryTimes[j] = dc.yearFraction(asof, asof + data.shiftExpiries[j]);
565 
566  // loop over shift expiries, apply same shifts across all strikes
567  vector<Real> shiftStrikes = volStrikes;
568  for (Size j = 0; j < shiftExpiryTimes.size(); ++j)
569  for (Size k = 0; k < shiftStrikes.size(); ++k)
570  applyShift(j, k, shifts[j], true, shiftType, shiftExpiryTimes, shiftStrikes, volExpiryTimes, volStrikes,
571  volData, shiftedVolData, j == 0 && k == 0);
572 
573  // add shifted vol data to the scenario
574  for (Size jj = 0; jj < n_cfvol_exp; ++jj) {
575  for (Size kk = 0; kk < n_cfvol_strikes; ++kk) {
576  Size idx = jj * n_cfvol_strikes + kk;
577  scenario->add(RiskFactorKey(RiskFactorKey::KeyType::OptionletVolatility, key, idx),
578  shiftedVolData[jj][kk]);
579  }
580  }
581  }
582  LOG("Optionlet vol scenarios done");
583 }
