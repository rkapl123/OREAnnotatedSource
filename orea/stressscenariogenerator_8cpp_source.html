30StressScenarioGenerator::StressScenarioGenerator(const boost::shared_ptr<StressTestScenarioData>& stressData,
31 const boost::shared_ptr<Scenario>& baseScenario,
32 const boost::shared_ptr<ScenarioSimMarketParameters>& simMarketData,

43void StressScenarioGenerator::generateScenarios() {
44 Date asof = baseScenario_->asof();
45 for (Size i = 0; i < stressData_->data().size(); ++i) {
46 StressTestScenarioData::StressTestData data = stressData_->data().at(i);
47 boost::shared_ptr<Scenario> scenario = stressScenarioFactory_->buildScenario(asof, data.label);
48
49 if (simMarketData_->simulateFxSpots())
50 addFxShifts(data, scenario);
51 addEquityShifts(data, scenario);
52 addDiscountCurveShifts(data, scenario);
53 addIndexCurveShifts(data, scenario);
54 addYieldCurveShifts(data, scenario);
55 if (simMarketData_->simulateFXVols())
56 addFxVolShifts(data, scenario);
57 if (simMarketData_->simulateEquityVols())
58 addEquityVolShifts(data, scenario);
59 if (simMarketData_->simulateSwapVols())
60 addSwaptionVolShifts(data, scenario);
61 if (simMarketData_->simulateCapFloorVols())
62 addCapFloorVolShifts(data, scenario);
63 if (simMarketData_->securitySpreadsSimulate())
64 addSecuritySpreadShifts(data, scenario);
65 if (simMarketData_->simulateRecoveryRates())
66 addRecoveryRateShifts(data, scenario);
67 if (simMarketData_->simulateSurvivalProbabilities())
68 addSurvivalProbabilityShifts(data, scenario);
69
70 scenarios_.push_back(scenario);
71 }
72
73 LOG("stress scenario generator initialised");
74}

76void StressScenarioGenerator::addFxShifts(StressTestScenarioData::StressTestData& std,
77 boost::shared_ptr<Scenario>& scenario) {
78 for (auto d : std.fxShifts) {
79 string ccypair = d.first; // foreign + domestic;
80
81 // Is this too strict?
82 // - implemented to avoid cases where input cross FX rates are not consistent
83 // - Consider an example (baseCcy = EUR) of a GBPUSD FX trade - two separate routes to pricing
84 // - (a) call GBPUSD FX rate from sim market
85 // - (b) call GBPEUR and EURUSD FX rates, manually join them to obtain GBPUSD
86 // - now, if GBPUSD is an explicit risk factor in sim market, consider what happens
87 // - if we bump GBPUSD value and leave other FX rates unchanged (for e.g. a sensitivity analysis)
88 // - (a) the value of the trade changes
89 // - (b) the value of the GBPUSD trade stays the same
90 // in light of the above we restrict the universe of FX pairs that we support here for the time being
91 string baseCcy = simMarketData_->baseCcy();
92 string foreign = ccypair.substr(0, 3);
93 string domestic = ccypair.substr(3);
94 QL_REQUIRE((domestic == baseCcy) || (foreign == baseCcy),
95 "SensitivityScenarioGenerator does not support cross FX pairs("
96 << ccypair << ", but base currency is " << baseCcy << ")");
97
98 LOG("Apply stress scenario to fx " << ccypair);
99
100 StressTestScenarioData::SpotShiftData data = d.second;
101 ShiftType type = parseShiftType(data.shiftType);
102 bool relShift = (type == ShiftType::Relative);
103 // QL_REQUIRE(type == ShiftType::Relative, "FX scenario type must be relative");
104 Real size = data.shiftSize;
105
106 RiskFactorKey key(RiskFactorKey::KeyType::FXSpot, ccypair);
107 Real rate = scenario->get(key);
108 Real newRate = relShift ? rate * (1.0 + size) : (rate + size);
109 scenario->add(RiskFactorKey(RiskFactorKey::KeyType::FXSpot, ccypair), newRate);
110 }
111 LOG("FX scenarios done");
112}

133void StressScenarioGenerator::addDiscountCurveShifts(StressTestScenarioData::StressTestData& std,
134 boost::shared_ptr<Scenario>& scenario) {
135 Date asof = baseScenario_->asof();
136
137 for (auto d : std.discountCurveShifts) {
138 string ccy = d.first;
139 LOG("Apply stress scenario to discount curve " << ccy);
140
141 Size n_ten = simMarketData_->yieldCurveTenors(ccy).size();
142 // original curves' buffer
143 std::vector<Real> zeros(n_ten);
144 std::vector<Real> times(n_ten);
145 // buffer for shifted zero curves
146 std::vector<Real> shiftedZeros(n_ten);
147
148 StressTestScenarioData::CurveShiftData data = d.second;
149 ShiftType shiftType = parseShiftType(data.shiftType);
150 //DayCounter dc = parseDayCounter(simMarketData_->yieldCurveDayCounter(ccy));
151 DayCounter dc;
152 if(auto s = simMarket_.lock()) {
153 dc = s->discountCurve(ccy)->dayCounter();
154 } else {
155 QL_FAIL("Internal error: could not lock simMarket. Contact dev.");
156 }
157
158 for (Size j = 0; j < n_ten; ++j) {
159 Date d = asof + simMarketData_->yieldCurveTenors(ccy)[j];
160 times[j] = dc.yearFraction(asof, d);
161 RiskFactorKey key(RiskFactorKey::KeyType::DiscountCurve, ccy, j);
162 Real quote = baseScenario_->get(key);
163 zeros[j] = -std::log(quote) / times[j];
164 }
165
166 std::vector<Period> shiftTenors = data.shiftTenors;
167 QL_REQUIRE(shiftTenors.size() > 0, "Discount shift tenors not specified");
168 std::vector<Real> shifts = data.shifts;
169 QL_REQUIRE(shiftTenors.size() == shifts.size(), "shift tenor and shift size vectors do not match");
170 std::vector<Time> shiftTimes(shiftTenors.size());
171 for (Size j = 0; j < shiftTenors.size(); ++j)
172 shiftTimes[j] = dc.yearFraction(asof, asof + shiftTenors[j]);
173
174 // apply zero rate shift at tenor point j
175 for (Size j = 0; j < shiftTenors.size(); ++j)
176 applyShift(j, shifts[j], true, shiftType, shiftTimes, zeros, times, shiftedZeros, j == 0 ? true : false);
177
178 // store shifted discount curve in the scenario
179 for (Size k = 0; k < n_ten; ++k) {
180 Real shiftedDiscount = exp(-shiftedZeros[k] * times[k]);
181 scenario->add(RiskFactorKey(RiskFactorKey::KeyType::DiscountCurve, ccy, k), shiftedDiscount);
182 }
183 }
184 LOG("Discount curve stress scenarios done");
185}

469void StressScenarioGenerator::addSwaptionVolShifts(StressTestScenarioData::StressTestData& std,
470 boost::shared_ptr<Scenario>& scenario) {
471 Date asof = baseScenario_->asof();
472
473 for (auto d : std.swaptionVolShifts) {
474 std::string key = d.first;
475 LOG("Apply stress scenario to swaption vol structure '" << key << "'");
476
477 Size n_swvol_term = simMarketData_->swapVolTerms(key).size();
478 Size n_swvol_exp = simMarketData_->swapVolExpiries(key).size();
479
480 vector<vector<Real>> volData(n_swvol_exp, vector<Real>(n_swvol_term, 0.0));
481 vector<Real> volExpiryTimes(n_swvol_exp, 0.0);
482 vector<Real> volTermTimes(n_swvol_term, 0.0);
483 vector<vector<Real>> shiftedVolData(n_swvol_exp, vector<Real>(n_swvol_term, 0.0));
484
485 StressTestScenarioData::SwaptionVolShiftData data = d.second;
486 ShiftType shiftType = parseShiftType(data.shiftType);
487 map<pair<Period, Period>, Real> shifts = data.shifts;
488
489 vector<Real> shiftExpiryTimes(data.shiftExpiries.size(), 0.0);
490 vector<Real> shiftTermTimes(data.shiftTerms.size(), 0.0);
491
492 DayCounter dc;
493 if(auto s = simMarket_.lock()) {
494 dc = s->swaptionVol(key)->dayCounter();
495 } else {
496 QL_FAIL("Internal error: could not lock simMarket. Contact dev.");
497 }
498
499 // cache original vol data
500 for (Size j = 0; j < n_swvol_exp; ++j) {
501 Date expiry = asof + simMarketData_->swapVolExpiries(key)[j];
502 volExpiryTimes[j] = dc.yearFraction(asof, expiry);
503 }
504 for (Size j = 0; j < n_swvol_term; ++j) {
505 Date term = asof + simMarketData_->swapVolTerms(key)[j];
506 volTermTimes[j] = dc.yearFraction(asof, term);
507 }
508 for (Size j = 0; j < n_swvol_exp; ++j) {
509 for (Size k = 0; k < n_swvol_term; ++k) {
510 Size idx = j * n_swvol_term + k;
511
512 RiskFactorKey rf(RiskFactorKey::KeyType::SwaptionVolatility, key, idx);
513 volData[j][k] = baseScenario_->get(rf);
514 }
515 }
516
517 // cache tenor times
518 for (Size j = 0; j < shiftExpiryTimes.size(); ++j)
519 shiftExpiryTimes[j] = dc.yearFraction(asof, asof + data.shiftExpiries[j]);
520 for (Size j = 0; j < shiftTermTimes.size(); ++j)
521 shiftTermTimes[j] = dc.yearFraction(asof, asof + data.shiftTerms[j]);
522
523 // loop over shift expiries and terms
524 // FIXME: apply same shifts to all strikes when present
525 for (Size j = 0; j < shiftExpiryTimes.size(); ++j) {
526 for (Size k = 0; k < shiftTermTimes.size(); ++k) {
527 // Size strikeBucket = 0; // FIXME
528 Real shift = 0.0;
529 pair<Period, Period> key(data.shiftExpiries[j], data.shiftTerms[k]);
530 if (shifts.size() == 0)
531 shift = data.parallelShiftSize;
532 else {
533 QL_REQUIRE(shifts.find(key) != shifts.end(), "swaption vol shift not found for expiry "
534 << data.shiftExpiries[j] << " and term "
535 << data.shiftTerms[k]);
536 shift = shifts[key];
537 }
538 applyShift(j, k, shift, true, shiftType, shiftExpiryTimes, shiftTermTimes, volExpiryTimes, volTermTimes,
539 volData, shiftedVolData, j == 0 && k == 0);
540 }
541 }
542
543 // add shifted vol data to the scenario
544 for (Size jj = 0; jj < n_swvol_exp; ++jj) {
545 for (Size kk = 0; kk < n_swvol_term; ++kk) {
546 Size idx = jj * n_swvol_term + kk;
547 scenario->add(RiskFactorKey(RiskFactorKey::KeyType::SwaptionVolatility, key, idx),
548 shiftedVolData[jj][kk]);
549 }
550 }
551 }
552 LOG("Swaption vol scenarios done");
553}

555void StressScenarioGenerator::addCapFloorVolShifts(StressTestScenarioData::StressTestData& std,
556 boost::shared_ptr<Scenario>& scenario) {
557 Date asof = baseScenario_->asof();
558
559 for (auto d : std.capVolShifts) {
560 std::string key = d.first;
561 LOG("Apply stress scenario to cap/floor vol structure " << key);
562
563 vector<Real> volStrikes = simMarketData_->capFloorVolStrikes(key);
564 // Strikes may be empty which indicates that the optionlet structure in the simulation market is an ATM curve
565 if (volStrikes.empty()) {
566 volStrikes = {0.0};
567 }
568 Size n_cfvol_strikes = volStrikes.size();
569
570 Size n_cfvol_exp = simMarketData_->capFloorVolExpiries(key).size();
571 vector<vector<Real>> volData(n_cfvol_exp, vector<Real>(n_cfvol_strikes, 0.0));
572 vector<Real> volExpiryTimes(n_cfvol_exp, 0.0);
573 vector<vector<Real>> shiftedVolData(n_cfvol_exp, vector<Real>(n_cfvol_strikes, 0.0));
574
575 StressTestScenarioData::CapFloorVolShiftData data = d.second;
576
577 ShiftType shiftType = parseShiftType(data.shiftType);
578 vector<Real> shifts = data.shifts;
579 vector<Real> shiftExpiryTimes(data.shiftExpiries.size(), 0.0);
580
581 //DayCounter dc = parseDayCounter(simMarketData_->capFloorVolDayCounter(key));
582 DayCounter dc;
583 if (auto s = simMarket_.lock()) {
584 dc = s->capFloorVol(key)->dayCounter();
585 } else {
586 QL_FAIL("Internal error: could not lock simMarket. Contact dev.");
587 }
588
589 // cache original vol data
590 for (Size j = 0; j < n_cfvol_exp; ++j) {
591 Date expiry = asof + simMarketData_->capFloorVolExpiries(key)[j];
592 volExpiryTimes[j] = dc.yearFraction(asof, expiry);
593 }
594 for (Size j = 0; j < n_cfvol_exp; ++j) {
595 for (Size k = 0; k < n_cfvol_strikes; ++k) {
596 Size idx = j * n_cfvol_strikes + k;
597 volData[j][k] =
598 baseScenario_->get(RiskFactorKey(RiskFactorKey::KeyType::OptionletVolatility, key, idx));
599 }
600 }
601
602 // cache tenor times
603 for (Size j = 0; j < shiftExpiryTimes.size(); ++j)
604 shiftExpiryTimes[j] = dc.yearFraction(asof, asof + data.shiftExpiries[j]);
605
606 // loop over shift expiries, apply same shifts across all strikes
607 vector<Real> shiftStrikes = volStrikes;
608 for (Size j = 0; j < shiftExpiryTimes.size(); ++j)
609 for (Size k = 0; k < shiftStrikes.size(); ++k)
610 applyShift(j, k, shifts[j], true, shiftType, shiftExpiryTimes, shiftStrikes, volExpiryTimes, volStrikes,
611 volData, shiftedVolData, j == 0 && k == 0);
612
613 // add shifted vol data to the scenario
614 for (Size jj = 0; jj < n_cfvol_exp; ++jj) {
615 for (Size kk = 0; kk < n_cfvol_strikes; ++kk) {
616 Size idx = jj * n_cfvol_strikes + kk;
617 scenario->add(RiskFactorKey(RiskFactorKey::KeyType::OptionletVolatility, key, idx),
618 shiftedVolData[jj][kk]);
619 }
620 }
621 }
622 LOG("Optionlet vol scenarios done");
623}

625void StressScenarioGenerator::addSecuritySpreadShifts(StressTestScenarioData::StressTestData& std,
626 boost::shared_ptr<Scenario>& scenario) {
627 for (auto d : std.securitySpreadShifts) {
628 string bond = d.first;
629 StressTestScenarioData::SpotShiftData data = d.second;

632 Real size = data.shiftSize;
633
634 RiskFactorKey key(RiskFactorKey::KeyType::SecuritySpread, bond);
635 Real base_spread = baseScenario_->get(key);
636
637 Real newSpread = relShift ? base_spread * (1.0 + size) : (base_spread + size);
638 scenario->add(RiskFactorKey(RiskFactorKey::KeyType::SecuritySpread, bond), newSpread);
639 }
640 LOG("Security spread scenarios done");
641}
