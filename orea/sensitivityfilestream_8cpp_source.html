54SensitivityRecord SensitivityFileStream::next() {
55 // Just in case
56 QL_REQUIRE(file_.is_open(), "The file stream is not open.");
57
58 // Get the next valid SensitivityRecord
59 string line;
60 while (getline(file_, line)) {
61 // Update the current line number
62 ++lineNo_;
63
64 // Trim leading and trailing space
65 boost::trim(line);
66
67 // If line is empty or a comment line, skip to next
68 if (line.empty() || boost::starts_with(line, comment_))
69 continue;
70
71 // Try to parse line in to a SensitivityRecord
72 DLOG("Processing line number " << lineNo_ << ": " << line);
73 vector<string> entries;
74 boost::split(
75 entries, line, [this](char c) { return c == delim_; }, boost::token_compress_off);
76 return processRecord(entries);
77 }
78
79 // If we get to here, no more lines to process so return empty record
80 return SensitivityRecord();
81}

90SensitivityRecord SensitivityFileStream::processRecord(const vector<string>& entries) const {
91
92 QL_REQUIRE(entries.size() == 10, "On line number " << lineNo_ << ": A sensitivity record needs 10 entries");
93
94 SensitivityRecord sr;
95 sr.tradeId = entries[0];
96 sr.isPar = parseBool(entries[1]);
97
98 auto p = deconstructFactor(entries[2]);
99 sr.key_1 = p.first;
100 sr.desc_1 = p.second;
101 tryParseReal(entries[3], sr.shift_1);
102
103 p = deconstructFactor(entries[4]);
104 sr.key_2 = p.first;
105 sr.desc_2 = p.second;
106 tryParseReal(entries[5], sr.shift_2);
107
108 sr.currency = entries[6];
109 sr.baseNpv = parseReal(entries[7]);
110 sr.delta = parseReal(entries[8]);
111 tryParseReal(entries[9], sr.gamma); // might be #N/A, if not computed
112
113 return sr;
114}
