95bool Analytic::match(const std::set<std::string>& runTypes) {
96 if (runTypes.size() == 0)
97 return true;
98
99 for (const auto& rt : runTypes) {
100 if (types_.find(rt) != types_.end()) {
101 LOG("Requested analytics " << to_string(runTypes) << " match analytic class " << label());
102 return true;
103 }
104 }
105 WLOG("None of the requested analytics " << to_string(runTypes) << " are covered by the analytic class " << label());
106 return false;
107}

112 if (configurations().todaysMarketParams)
113 tmps.push_back(configurations().todaysMarketParams);

117 tmps.insert(end(tmps), begin(ctmps), end(ctmps));

123boost::shared_ptr<EngineFactory> Analytic::Impl::engineFactory() {
124 LOG("Analytic::engineFactory() called");

140void Analytic::buildMarket(const boost::shared_ptr<ore::data::InMemoryLoader>& loader,
141 const bool marketRequired) {
142 LOG("Analytic::buildMarket called");
143 cpu_timer mtimer;
144
145 QL_REQUIRE(loader, "market data loader not set");
146 QL_REQUIRE(configurations().curveConfig, "curve configurations not set");
147
148 // first build the market if we have a todaysMarketParams
149 if (configurations().todaysMarketParams) {
150 try {
151 // Note: we usually update the loader with implied data, but we simply use the provided loader here
152 loader_ = loader;
153 // Check that the loader has quotes
154 QL_REQUIRE( loader_->hasQuotes(inputs()->asof()),
155 "There are no quotes available for date " << inputs()->asof());
156 // Build the market
157 market_ = boost::make_shared<TodaysMarket>(inputs()->asof(), configurations().todaysMarketParams, loader_,
158 configurations().curveConfig, inputs()->continueOnError(),
159 true, inputs()->lazyMarketBuilding(), inputs()->refDataManager(),
160 false, *inputs()->iborFallbackConfig());
161 // Note: we usually wrap the market into a PC market, but skip this step here
162 } catch (const std::exception& e) {
163 if (marketRequired)
164 QL_FAIL("Failed to build market: " << e.what());
165 else
166 WLOG("Failed to build market: " << e.what());
167 }
168 } else {
169 ALOG("Skip building the market due to missing today's market parameters in configurations");
170 }
171 mtimer.stop();
172 LOG("Market Build time " << setprecision(2) << mtimer.format(default_places, "%w") << " sec");
173}

180void Analytic::buildPortfolio() {
181 // create a new empty portfolio
182 portfolio_ = boost::make_shared<Portfolio>(inputs()->buildFailedTrades());
183
184 inputs()->portfolio()->reset();
185 // populate with trades
186 for (const auto& [tradeId, trade] : inputs()->portfolio()->trades())
187 portfolio()->add(trade);
188
189 if (market_) {
190 replaceTrades();
191
192 LOG("Build the portfolio");
193 boost::shared_ptr<EngineFactory> factory = impl()->engineFactory();
194 portfolio()->build(factory, "analytic/" + label());
195
196 // remove dates that will have matured
197 Date maturityDate = inputs()->asof();
198 if (inputs()->portfolioFilterDate() != Null<Date>())
199 maturityDate = inputs()->portfolioFilterDate();
200
201 LOG("Filter trades that expire before " << maturityDate);
202 portfolio()->removeMatured(maturityDate);
203 } else {
204 ALOG("Skip building the portfolio, because market not set");
205 }
206}

216void MarketDataAnalyticImpl::runAnalytic(
217 const boost::shared_ptr<ore::data::InMemoryLoader>& loader,
218 const std::set<std::string>& runTypes) {
219
220 Settings::instance().evaluationDate() = inputs_->asof();
221 ObservationMode::instance().setMode(inputs_->observationModel());
222
223 CONSOLEW("Build Market");
224 analytic()->buildMarket(loader);
225 CONSOLE("OK");
226
227 if (inputs_->outputTodaysMarketCalibration()) {
228 CONSOLEW("Market Calibration");
229 LOG("Write todays market calibration report");
230 auto t = boost::dynamic_pointer_cast<TodaysMarket>(analytic()->market());
231 QL_REQUIRE(t != nullptr, "expected todays market instance");
232 boost::shared_ptr<InMemoryReport> mktReport = boost::make_shared<InMemoryReport>();
233 ore::analytics::ReportWriter(inputs_->reportNaString())
234 .writeTodaysMarketCalibrationReport(*mktReport, t->calibrationInfo());
235 analytic()->reports()["MARKET"]["todaysmarketcalibration"] = mktReport;
236 CONSOLE("OK");
237 }
238}
