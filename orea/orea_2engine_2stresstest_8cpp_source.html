52StressTest::StressTest(const boost::shared_ptr<ore::data::Portfolio>& portfolio,
53 const boost::shared_ptr<ore::data::Market>& market, const string& marketConfiguration,
54 const boost::shared_ptr<ore::data::EngineData>& engineData,
55 const boost::shared_ptr<ScenarioSimMarketParameters>& simMarketData,
56 const boost::shared_ptr<StressTestScenarioData>& stressData,
57 const CurveConfigurations& curveConfigs, const TodaysMarketParameters& todaysMarketParams,
58 boost::shared_ptr<ScenarioFactory> scenarioFactory,
59 const boost::shared_ptr<ReferenceDataManager>& referenceData,
60 const IborFallbackConfig& iborFallbackConfig, bool continueOnError) {
61
62 LOG("Build Simulation Market");
63 boost::shared_ptr<ScenarioSimMarket> simMarket = boost::make_shared<ScenarioSimMarket>(
64 market, simMarketData, Market::defaultConfiguration, curveConfigs, todaysMarketParams,
65 continueOnError, false, false, false, iborFallbackConfig);
66
67 LOG("Build Stress Scenario Generator");
68 Date asof = market->asofDate();
69 boost::shared_ptr<Scenario> baseScenario = simMarket->baseScenario();
70 scenarioFactory = scenarioFactory ? scenarioFactory : boost::make_shared<CloneScenarioFactory>(baseScenario);
71 boost::shared_ptr<StressScenarioGenerator> scenarioGenerator =
72 boost::make_shared<StressScenarioGenerator>(stressData, baseScenario, simMarketData, simMarket, scenarioFactory);
73 simMarket->scenarioGenerator() = scenarioGenerator;
74
75 LOG("Build Engine Factory");
76 map<MarketContext, string> configurations;
77 configurations[MarketContext::pricing] = marketConfiguration;
78 auto ed = boost::make_shared<EngineData>(*engineData);
79 ed->globalParameters()["RunType"] = "Stress";
80 boost::shared_ptr<EngineFactory> factory =
81 boost::make_shared<EngineFactory>(ed, simMarket, configurations, referenceData, iborFallbackConfig);
82
83 LOG("Reset and Build Portfolio");
84 portfolio->reset();
85 portfolio->build(factory, "stress analysis");
86
87 LOG("Build the cube object to store sensitivities");
88 boost::shared_ptr<NPVCube> cube = boost::make_shared<DoublePrecisionInMemoryCube>(
89 asof, portfolio->ids(), vector<Date>(1, asof), scenarioGenerator->samples());
90
91 boost::shared_ptr<DateGrid> dg = boost::make_shared<DateGrid>(
92 "1,0W"); // TODO - extend the DateGrid interface so that it can actually take a vector of dates as input
93 vector<boost::shared_ptr<ValuationCalculator>> calculators;
94 calculators.push_back(boost::make_shared<NPVCalculator>(simMarketData->baseCcy()));
95 ValuationEngine engine(asof, dg, simMarket, factory->modelBuilders());
96 LOG("Run Stress Scenarios");
97 /*ostringstream o;
98 o.str("");
99 o << "Stress scenarios " << portfolio->size() << " x " << dg->size() << " x " << scenarioGenerator->samples() <<
100 "... ";
101 auto progressBar = boost::make_shared<SimpleProgressBar>(o.str());
102 auto progressLog = boost::make_shared<ProgressLog>("Building scenarios...");
103 engine.registerProgressIndicator(progressBar);
104 engine.registerProgressIndicator(progressLog);*/
105 engine.buildCube(portfolio, cube, calculators);
106
107 /*****************
108 * Collect results
109 */
110 baseNPV_.clear();
111 shiftedNPV_.clear();
112 delta_.clear();
113 labels_.clear();
114 size_t i = 0;
115 for (auto tradeIt = portfolio->trades().begin(); tradeIt != portfolio->trades().end(); ++tradeIt, ++i) {
116 string tradeId = tradeIt->first;
117 Real npv0 = cube->getT0(i, 0);
118 trades_.insert(tradeId);
119 baseNPV_[tradeId] = npv0;
120 for (Size j = 0; j < scenarioGenerator->samples(); ++j) {
121 string label = scenarioGenerator->scenarios()[j]->label();
122 Real npv = cube->get(i, 0, j, 0);
123 pair<string, string> p(tradeId, label);
124 shiftedNPV_[p] = npv;
125 delta_[p] = npv - npv0;
126 labels_.insert(label);
127 }
128 }
129 LOG("Stress testing done");
130}

132void StressTest::writeReport(const boost::shared_ptr<ore::data::Report>& report, Real outputThreshold) {
133
134 report->addColumn("TradeId", string());
135 report->addColumn("ScenarioLabel", string());
136 report->addColumn("Base NPV", double(), 2);
137 report->addColumn("Scenario NPV", double(), 2);
138 report->addColumn("Sensitivity", double(), 2);
139
140 for (auto data : shiftedNPV_) {
141 string id = data.first.first;
142 string factor = data.first.second;
143 Real npv = data.second;
144 Real base = baseNPV_[id];
145 Real sensi = npv - base;
146 if (fabs(sensi) > outputThreshold) {
147 report->next();
148 report->add(id);
149 report->add(factor);
150 report->add(base);
151 report->add(npv);
152 report->add(sensi);
153 }
154 }
155
156 report->end();
157}
