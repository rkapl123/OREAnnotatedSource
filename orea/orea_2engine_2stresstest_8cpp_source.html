63 
64  LOG("Build Simulation Market");
65  boost::shared_ptr<ScenarioSimMarket> simMarket = boost::make_shared<ScenarioSimMarket>(
66  market, simMarketData, Market::defaultConfiguration, curveConfigs, todaysMarketParams,
67  continueOnError, false, false, false, iborFallbackConfig);
68 
69  LOG("Build Stress Scenario Generator");
70  Date asof = market->asofDate();
71  boost::shared_ptr<Scenario> baseScenario = simMarket->baseScenario();
72  scenarioFactory = scenarioFactory ? scenarioFactory : boost::make_shared<CloneScenarioFactory>(baseScenario);
73  boost::shared_ptr<StressScenarioGenerator> scenarioGenerator =
74  boost::make_shared<StressScenarioGenerator>(stressData, baseScenario, simMarketData, simMarket, scenarioFactory);
75  simMarket->scenarioGenerator() = scenarioGenerator;
76 
77  LOG("Build Engine Factory");
78  map<MarketContext, string> configurations;
79  configurations[MarketContext::pricing] = marketConfiguration;
80  auto ed = boost::make_shared<EngineData>(*engineData);
81  ed->globalParameters()["RunType"] = "Stress";
82  boost::shared_ptr<EngineFactory> factory = boost::make_shared<EngineFactory>(
83  ed, simMarket, configurations, extraEngineBuilders, extraLegBuilders, referenceData, iborFallbackConfig);
84 
85  LOG("Reset and Build Portfolio");
86  portfolio->reset();
87  portfolio->build(factory, "stress analysis");
88 
89  LOG("Build the cube object to store sensitivities");
90  boost::shared_ptr<NPVCube> cube = boost::make_shared<DoublePrecisionInMemoryCube>(
91  asof, portfolio->ids(), vector<Date>(1, asof), scenarioGenerator->samples());
92 
93  boost::shared_ptr<DateGrid> dg = boost::make_shared<DateGrid>(
94  "1,0W"); // TODO - extend the DateGrid interface so that it can actually take a vector of dates as input
95  vector<boost::shared_ptr<ValuationCalculator>> calculators;
96  calculators.push_back(boost::make_shared<NPVCalculator>(simMarketData->baseCcy()));
97  ValuationEngine engine(asof, dg, simMarket, factory->modelBuilders());
98  LOG("Run Stress Scenarios");
99  /*ostringstream o;
100  o.str("");
101  o << "Stress scenarios " << portfolio->size() << " x " << dg->size() << " x " << scenarioGenerator->samples() <<
102  "... ";
103  auto progressBar = boost::make_shared<SimpleProgressBar>(o.str());
104  auto progressLog = boost::make_shared<ProgressLog>("Building scenarios...");
105  engine.registerProgressIndicator(progressBar);
106  engine.registerProgressIndicator(progressLog);*/
107  engine.buildCube(portfolio, cube, calculators);
108 
109  /*****************
110  * Collect results
111  */
112  baseNPV_.clear();
113  shiftedNPV_.clear();
114  delta_.clear();
115  labels_.clear();
116  for (Size i = 0; i < portfolio->size(); ++i) {
117  Real npv0 = cube->getT0(i, 0);
118  string id = portfolio->trades()[i]->id();
119  trades_.insert(id);
120  baseNPV_[id] = npv0;
121  for (Size j = 0; j < scenarioGenerator->samples(); ++j) {
122  string label = scenarioGenerator->scenarios()[j]->label();
123  Real npv = cube->get(i, 0, j, 0);
124  pair<string, string> p(id, label);
125  shiftedNPV_[p] = npv;
126  delta_[p] = npv - npv0;
127  labels_.insert(label);
128  }
129  }
130  LOG("Stress testing done");

133 void StressTest::writeReport(const boost::shared_ptr<ore::data::Report>& report, Real outputThreshold) {
134 
135  report->addColumn("TradeId", string());
136  report->addColumn("ScenarioLabel", string());
137  report->addColumn("Base NPV", double(), 2);
138  report->addColumn("Scenario NPV", double(), 2);
139  report->addColumn("Sensitivity", double(), 2);
140 
141  for (auto data : shiftedNPV_) {
142  string id = data.first.first;
143  string factor = data.first.second;
144  Real npv = data.second;
145  Real base = baseNPV_[id];
146  Real sensi = npv - base;
147  if (fabs(sensi) > outputThreshold) {
148  report->next();
149  report->add(id);
150  report->add(factor);
151  report->add(base);
152  report->add(npv);
153  report->add(sensi);
154  }
155  }
156 
157  report->end();
158 }
