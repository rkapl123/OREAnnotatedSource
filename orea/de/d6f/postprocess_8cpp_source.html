304 void PostProcess::updateNettingSetKVA() {
305 
306  // Loop over all netting sets
307  for (auto nettingSetId : nettingSetIds()) {
308  // Init results
309  ourNettingSetKVACCR_[nettingSetId] = 0.0;
310  theirNettingSetKVACCR_[nettingSetId] = 0.0;
311  ourNettingSetKVACVA_[nettingSetId] = 0.0;
312  theirNettingSetKVACVA_[nettingSetId] = 0.0;
313  }
314 
315  if (!analytics_["kva"])
316  return;
317 
318  LOG("Update netting set KVA");
319 
320  vector<Date> dateVector = cube_->dates();
321  Size dates = dateVector.size();
322  Date today = market_->asofDate();
323  Handle<YieldTermStructure> discountCurve = market_->discountCurve(baseCurrency_, configuration_);
324  DayCounter dc = ActualActual();
325 
326  // Loop over all netting sets
327  for (auto nettingSetId : nettingSetIds()) {
328  string cid;
329  if (analytics_["flipViewXVA"]) {
330  cid = dvaName_;
331  } else {
332  cid = nettedExposureCalculator_->counterparty(nettingSetId);
333  }
334  LOG("KVA for netting set " << nettingSetId);
335 
336  // Main input are the EPE and ENE profiles, previously computed
337  vector<Real> epe = netEPE_[nettingSetId];
338  vector<Real> ene = netENE_[nettingSetId];
339 
340  // PD from counterparty Dts, floored to avoid 0 ...
341  // Today changed to today+1Y to get the one-year PD
342  Handle<DefaultProbabilityTermStructure> cvaDts = market_->defaultCurve(cid, configuration_);
343  QL_REQUIRE(!cvaDts.empty(), "Default curve missing for counterparty " << cid);
344  Real cvaRR = market_->recoveryRate(cid, configuration_)->value();
345  Real PD1 = std::max(cvaDts->defaultProbability(today + 1 * Years), 0.000000000001);
346  Real LGD1 = (1 - cvaRR);
347 
348  // FIXME: if flipViewXVA is sufficient, then all code for their KVA-CCR could be discarded here...
349  Handle<DefaultProbabilityTermStructure> dvaDts;
350  Real dvaRR = 0.0;
351  Real PD2 = 0;
352  if (analytics_["flipViewXVA"]) {
353  dvaName_ = nettedExposureCalculator_->counterparty(nettingSetId);
354  }
355  if (dvaName_ != "") {
356  dvaDts = market_->defaultCurve(dvaName_, configuration_);
357  dvaRR = market_->recoveryRate(dvaName_, configuration_)->value();
358  PD2 = std::max(dvaDts->defaultProbability(today + 1 * Years), 0.000000000001);
359  } else {
360  ALOG("dvaName not specified, own PD set to zero for their KVA calculation");
361  }
362  Real LGD2 = (1 - dvaRR);
363 
364  // Granularity adjustment, Gordy (2004):
365  Real rho1 = 0.12 * (1 - std::exp(-50 * PD1)) / (1 - std::exp(-50)) +
366  0.24 * (1 - (1 - std::exp(-50 * PD1)) / (1 - std::exp(-50)));
367  Real rho2 = 0.12 * (1 - std::exp(-50 * PD2)) / (1 - std::exp(-50)) +
368  0.24 * (1 - (1 - std::exp(-50 * PD2)) / (1 - std::exp(-50)));
369 
370  // Basel II internal rating based (IRB) estimate of worst case PD:
371  // Large homogeneous pool (LHP) approximation of Vasicek (1997)
372  InverseCumulativeNormal icn;
373  CumulativeNormalDistribution cnd;
374  Real PD99_1 = cnd((icn(PD1) + std::sqrt(rho1) * icn(0.999)) / (std::sqrt(1 - rho1))) - PD1;
375  Real PD99_2 = cnd((icn(PD2) + std::sqrt(rho2) * icn(0.999)) / (std::sqrt(1 - rho2))) - PD2;
376 
377  // KVA regulatory PD, worst case PD, floored at 0.03 for corporates and banks, not floored for sovereigns
378  Real kva99PD1 = std::max(PD99_1, kvaTheirPdFloor_);
379  Real kva99PD2 = std::max(PD99_2, kvaOurPdFloor_);
380 
381  // Factor B(PD) for the maturity adjustment factor, B(PD) = (0.11852 - 0.05478 * ln(PD)) ^ 2
382  Real kvaMatAdjB1 = std::pow((0.11852 - 0.05478 * std::log(PD1)), 2.0);
383  Real kvaMatAdjB2 = std::pow((0.11852 - 0.05478 * std::log(PD2)), 2.0);
384 
385  DLOG("Our KVA-CCR " << nettingSetId << ": PD=" << PD1);
386  DLOG("Our KVA-CCR " << nettingSetId << ": LGD=" << LGD1);
387  DLOG("Our KVA-CCR " << nettingSetId << ": rho=" << rho1);
388  DLOG("Our KVA-CCR " << nettingSetId << ": PD99=" << PD99_1);
389  DLOG("Our KVA-CCR " << nettingSetId << ": PD Floor=" << kvaTheirPdFloor_);
390  DLOG("Our KVA-CCR " << nettingSetId << ": Floored PD99=" << kva99PD1);
391  DLOG("Our KVA-CCR " << nettingSetId << ": B(PD)=" << kvaMatAdjB1);
392 
393  DLOG("Their KVA-CCR " << nettingSetId << ": PD=" << PD2);
394  DLOG("Their KVA-CCR " << nettingSetId << ": LGD=" << LGD2);
395  DLOG("Their KVA-CCR " << nettingSetId << ": rho=" << rho2);
396  DLOG("Their KVA-CCR " << nettingSetId << ": PD99=" << PD99_2);
397  DLOG("Their KVA-CCR " << nettingSetId << ": PD Floor=" << kvaOurPdFloor_);
398  DLOG("Their KVA-CCR " << nettingSetId << ": Floored PD99=" << kva99PD2);
399  DLOG("Their KVA-CCR " << nettingSetId << ": B(PD)=" << kvaMatAdjB2);
400 
401  for (Size j = 0; j < dates; ++j) {
402  Date d0 = j == 0 ? today : cube_->dates()[j - 1];
403  Date d1 = cube_->dates()[j];
404 
405  // Preprocess:
406  // 1) Effective maturity from effective expected exposure as of time j
407  // Index _1 corresponds to our perspective, index _2 to their perspective.
408  // 2) Basel EEPE as of time j, i.e. as time averge over EEE, starting at time j
409  // More accuracy may be achieved here by using a Longstaff-Schwartz method / regression
410  Real eee_kva_1 = 0.0, eee_kva_2 = 0.0;
411  Real effMatNumer1 = 0.0, effMatNumer2 = 0.0;
412  Real effMatDenom1 = 0.0, effMatDenom2 = 0.0;
413  Real eepe_kva_1 = 0, eepe_kva_2 = 0.0;
414  Size kmax = j, count = 0;
415  // Cut off index for EEPE/EENE calculation: One year ahead
416  while (dateVector[kmax] < dateVector[j] + 1 * Years + 4 * Days && kmax < dates - 1)
417  kmax++;
418  Real sumdt = 0.0, eee1_b = 0.0, eee2_b = 0.0;
419  for (Size k = j; k < dates; ++k) {
420  Date d2 = cube_->dates()[k];
421  Date prevDate = k == 0 ? today : dateVector[k - 1];
422 
423  eee_kva_1 = std::max(eee_kva_1, epe[k + 1]);
424  eee_kva_2 = std::max(eee_kva_2, ene[k + 1]);
425 
426  // Components of the KVA maturity adjustment MA as of time j
427  if (dc.yearFraction(d1, d2) > 1.0) {
428  effMatNumer1 += epe[k + 1] * dc.yearFraction(prevDate, d2);
429  effMatNumer2 += ene[k + 1] * dc.yearFraction(prevDate, d2);
430  }
431  if (dc.yearFraction(d1, d2) <= 1.0) {
432  effMatDenom1 += eee_kva_1 * dc.yearFraction(prevDate, d2);
433  effMatDenom2 += eee_kva_2 * dc.yearFraction(prevDate, d2);
434  }
435 
436  if (k < kmax) {
437  Real dt = dc.yearFraction(cube_->dates()[k], cube_->dates()[k + 1]);
438  sumdt += dt;
439  Real epe_b = epe[k + 1] / discountCurve->discount(dateVector[k]);
440  Real ene_b = ene[k + 1] / discountCurve->discount(dateVector[k]);
441  eee1_b = std::max(epe_b, eee1_b);
442  eee2_b = std::max(ene_b, eee2_b);
443  eepe_kva_1 += eee1_b * dt;
444  eepe_kva_2 += eee2_b * dt;
445  count++;
446  }
447  }
448 
449  // Normalize EEPE/EENE calculation
450  eepe_kva_1 = count > 0 ? eepe_kva_1 / sumdt : 0.0;
451  eepe_kva_2 = count > 0 ? eepe_kva_2 / sumdt : 0.0;
452 
453  // KVA CCR using the IRB risk weighted asset method and IMM:
454  // KVA effective maturity of the nettingSet, capped at 5
455  Real kvaNWMaturity1 = std::min(1.0 + (effMatDenom1 == 0.0 ? 0.0 : effMatNumer1 / effMatDenom1), 5.0);
456  Real kvaNWMaturity2 = std::min(1.0 + (effMatDenom2 == 0.0 ? 0.0 : effMatNumer2 / effMatDenom2), 5.0);
457 
458  // Maturity adjustment factor for the RWA method:
459  // MA(PD, M) = (1 + (M - 2.5) * B(PD)) / (1 - 1.5 * B(PD)), capped at 5, floored at 1, M = effective
460  // maturity
461  Real kvaMatAdj1 =
462  std::max(std::min((1.0 + (kvaNWMaturity1 - 2.5) * kvaMatAdjB1) / (1.0 - 1.5 * kvaMatAdjB1), 5.0), 1.0);
463  Real kvaMatAdj2 =
464  std::max(std::min((1.0 + (kvaNWMaturity2 - 2.5) * kvaMatAdjB2) / (1.0 - 1.5 * kvaMatAdjB2), 5.0), 1.0);
465 
466  // CCR Capital: RC = EAD x LGD x PD99.9 x MA(PD, M); EAD = alpha x EEPE(t) (approximated by EPE here);
467  Real kvaRC1 = kvaAlpha_ * eepe_kva_1 * LGD1 * kva99PD1 * kvaMatAdj1;
468  Real kvaRC2 = kvaAlpha_ * eepe_kva_2 * LGD2 * kva99PD2 * kvaMatAdj2;
469 
470  // Expected risk capital discounted at capital discount rate
471  Real kvaCapitalDiscount = 1 / std::pow(1 + kvaCapitalDiscountRate_, dc.yearFraction(today, d0));
472  Real kvaCCRIncrement1 =
473  kvaRC1 * kvaCapitalDiscount * dc.yearFraction(d0, d1) * kvaCapitalHurdle_ * kvaRegAdjustment_;
474  Real kvaCCRIncrement2 =
475  kvaRC2 * kvaCapitalDiscount * dc.yearFraction(d0, d1) * kvaCapitalHurdle_ * kvaRegAdjustment_;
476 
477  ourNettingSetKVACCR_[nettingSetId] += kvaCCRIncrement1;
478  theirNettingSetKVACCR_[nettingSetId] += kvaCCRIncrement2;
479 
480  DLOG("Our KVA-CCR for " << nettingSetId << ": " << j << " EEPE=" << setprecision(2) << eepe_kva_1
481  << " EPE=" << epe[j] << " RC=" << kvaRC1 << " M=" << setprecision(6)
482  << kvaNWMaturity1 << " MA=" << kvaMatAdj1 << " Cost=" << setprecision(2)
483  << kvaCCRIncrement1 << " KVA=" << ourNettingSetKVACCR_[nettingSetId]);
484  DLOG("Their KVA-CCR for " << nettingSetId << ": " << j << " EENE=" << eepe_kva_2 << " ENE=" << ene[j]
485  << " RC=" << kvaRC2 << " M=" << setprecision(6) << kvaNWMaturity2
486  << " MA=" << kvaMatAdj2 << " Cost=" << setprecision(2) << kvaCCRIncrement2
487  << " KVA=" << theirNettingSetKVACCR_[nettingSetId]);
488 
489  // CVA Capital
490  // effective maturity without cap at 5, DF set to 1 for IMM banks
491  // TODO: Set MA in CCR capital calculation to 1
492  Real kvaCvaMaturity1 = 1.0 + (effMatDenom1 == 0.0 ? 0.0 : effMatNumer1 / effMatDenom1);
493  Real kvaCvaMaturity2 = 1.0 + (effMatDenom2 == 0.0 ? 0.0 : effMatNumer2 / effMatDenom2);
494  Real scva1 = kvaTheirCvaRiskWeight_ * kvaCvaMaturity1 * eepe_kva_1;
495  Real scva2 = kvaOurCvaRiskWeight_ * kvaCvaMaturity2 * eepe_kva_2;
496  Real kvaCVAIncrement1 =
497  scva1 * kvaCapitalDiscount * dc.yearFraction(d0, d1) * kvaCapitalHurdle_ * kvaRegAdjustment_;
498  Real kvaCVAIncrement2 =
499  scva2 * kvaCapitalDiscount * dc.yearFraction(d0, d1) * kvaCapitalHurdle_ * kvaRegAdjustment_;
500 
501  DLOG("Our KVA-CVA for " << nettingSetId << ": " << j << " EEPE=" << eepe_kva_1 << " SCVA=" << scva1
502  << " Cost=" << kvaCVAIncrement1);
503  DLOG("Their KVA-CVA for " << nettingSetId << ": " << j << " EENE=" << eepe_kva_2 << " SCVA=" << scva2
504  << " Cost=" << kvaCVAIncrement2);
505 
506  ourNettingSetKVACVA_[nettingSetId] += kvaCVAIncrement1;
507  theirNettingSetKVACVA_[nettingSetId] += kvaCVAIncrement2;
508  }
509  }
510 
511  LOG("Update netting set KVA done");

514 void PostProcess::updateNettingSetCvaSensitivity() {
515 
516  if (!analytics_["cvaSensi"])
517  return;
518 
519  LOG("Update netting set CVA sensitvities");
520 
521  Handle<YieldTermStructure> discountCurve = market_->discountCurve(baseCurrency_, configuration_);
522 
523  for (auto n : netEPE_) {
524  string nettingSetId = n.first;
525  vector<Real> epe = netEPE_[nettingSetId];
526  string cid;
527  Handle<DefaultProbabilityTermStructure> cvaDts;
528  Real cvaRR;
529  if (analytics_["flipViewXVA"]) {
530  cid = dvaName_;
531  } else {
532  cid = nettedExposureCalculator_->counterparty(nettingSetId);
533  }
534  cvaDts = market_->defaultCurve(cid);
535  QL_REQUIRE(!cvaDts.empty(), "Default curve missing for counterparty " << cid);
536  cvaRR = market_->recoveryRate(cid, configuration_)->value();
537 
538  bool cvaSensi = analytics_["cvaSensi"];
539  LOG("CVA Sensitivity: " << cvaSensi);
540  if (cvaSensi) {
541  boost::shared_ptr<CVASpreadSensitivityCalculator> cvaSensiCalculator = boost::make_shared<CVASpreadSensitivityCalculator>(
542  nettingSetId, market_->asofDate(), epe, cube_->dates(), cvaDts, cvaRR, discountCurve, cvaSpreadSensiGrid_, cvaSpreadSensiShiftSize_);
543 
544  for (Size i = 0; i < cvaSensiCalculator->shiftTimes().size(); ++i) {
545  DLOG("CVA Sensi Calculator: t=" << cvaSensiCalculator->shiftTimes()[i]
546  << " h=" << cvaSensiCalculator->hazardRateSensitivities()[i]
547  << " s=" << cvaSensiCalculator->cdsSpreadSensitivities()[i]);
548  }
549 
550  netCvaHazardRateSensi_[nettingSetId] = cvaSensiCalculator->hazardRateSensitivities();
551  netCvaSpreadSensi_[nettingSetId] = cvaSensiCalculator->cdsSpreadSensitivities();
552  cvaSpreadSensiTimes_ = cvaSensiCalculator->shiftTimes();
553  } else {
554  cvaSpreadSensiTimes_ = vector<Real>();
555  netCvaHazardRateSensi_[nettingSetId] = vector<Real>();
556  netCvaSpreadSensi_[nettingSetId] = vector<Real>();
557  }
558  }
559 
560  LOG("Update netting set CVA sensitivities done");
561 }
