41  SensitivityCube(const boost::shared_ptr<NPVCube>& cube,
42  boost::shared_ptr<std::vector<ShiftScenarioGenerator::ScenarioDescription>>& scenarioDescriptions,
43  const boost::shared_ptr<ore::data::Portfolio>& portfolio)
44  : cube_(cube), scenarioDescriptions_(scenarioDescriptions) {
45  for (Size i = 0; i < scenarioDescriptions->size(); i++) {
46  ShiftScenarioGenerator::ScenarioDescription desc = (*scenarioDescriptions)[i];
47 
48  if (desc.type() == ShiftScenarioGenerator::ScenarioDescription::Type::Up) {
49  upFactorIndices_[desc.factor1()] = i;
50  } else if (desc.type() == ShiftScenarioGenerator::ScenarioDescription::Type::Down) {
51  downFactorIndices_[desc.factor1()] = i;
52  } else if (desc.type() == ShiftScenarioGenerator::ScenarioDescription::Type::Cross) {
53  std::pair<string, string> pair(desc.factor1(), desc.factor2());
54  crossFactorIndices_[pair] = i;
55  } else if (desc.type() == ShiftScenarioGenerator::ScenarioDescription::Type::Base) {
56  baseFactorIndices_[desc.factor1()] = i;
57  }
58  }
59  for (Size i = 0; i < portfolio->size(); i++) {
60  tradeIndices_[portfolio->trades()[i]->id()] = i;
61  }
62  }

84  // the scenarioDescriptions_ indices match the cube_ indices for a given type/factor
85  Size getFactorIndex(ShiftScenarioGenerator::ScenarioDescription::Type type, string factor1, string factor2) const {
86  if (type == ShiftScenarioGenerator::ScenarioDescription::Type::Up) {
87  return upFactorIndices_.find(factor1)->second;
88  } else if (type == ShiftScenarioGenerator::ScenarioDescription::Type::Down) {
89  return downFactorIndices_.find(factor1)->second;
90  } else if (type == ShiftScenarioGenerator::ScenarioDescription::Type::Cross) {
91  std::pair<string, string> pair(factor1, factor2);
92  return crossFactorIndices_.find(pair)->second;
93  } else if (type == ShiftScenarioGenerator::ScenarioDescription::Type::Base) {
94  return baseFactorIndices_.find(factor1)->second;
95  } else {
96  QL_FAIL("Invalid type");
97  }
98  }
