70void ScenarioWriter::writeScenario(boost::shared_ptr<Scenario>& s, const bool writeHeader) {
71 const Date d = s->asof();
72 // take a copy of the keys here to ensure the order is preserved
73 keys_ = s->keys();
74 std::sort(keys_.begin(), keys_.end());
75 if (fp_) {
76 if (writeHeader) {
77 QL_REQUIRE(keys_.size() > 0, "No keys in scenario");
78 fprintf(fp_, "Date%cScenario%cNumeraire%c%s", sep_, sep_, sep_, to_string(keys_[0]).c_str());
79 for (Size i = 1; i < keys_.size(); i++)
80 fprintf(fp_, "%c%s", sep_, to_string(keys_[i]).c_str());
81 fprintf(fp_, "\n");
82
83 // set the first date, this will bump i_ to 1 below
84 firstDate_ = d;
85 }
86 if (d == firstDate_)
87 i_++;
88
89 fprintf(fp_, "%s%c%zu%c%.8f", to_string(d).c_str(), sep_, i_, sep_, s->getNumeraire());
90 for (auto k : keys_)
91 fprintf(fp_, "%c%.8f", sep_, s->get(k));
92 fprintf(fp_, "\n");
93 fflush(fp_);
94 }
95
96 if (report_) {
97 if (writeHeader) {
98 QL_REQUIRE(keys_.size() > 0, "No keys in scenario");
99 report_->addColumn("Date", string());
100 report_->addColumn("Scenario", Size());
101 report_->addColumn("Numeraire", double(), 8);
102 for (Size i = 0; i < keys_.size(); i++)
103 report_->addColumn(to_string(keys_[i]), double(), 8);
104 // set the first date, this will bump i_ to 1 below
105 firstDate_ = d;
106 }
107 if (d == firstDate_)
108 i_++;
109 report_->next();
110 report_->add(to_string(d));
111 report_->add(i_);
112 report_->add(s->getNumeraire());
113 for (auto k : keys_)
114 report_->add(s->get(k));
115 }
116}
