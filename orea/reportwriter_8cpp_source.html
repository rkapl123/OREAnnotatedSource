43 void ReportWriter::writeNpv(ore::data::Report& report, const std::string& baseCurrency,
44  boost::shared_ptr<Market> market, const std::string& configuration,
45  boost::shared_ptr<Portfolio> portfolio) {
46  LOG("portfolio valuation");
47  DayCounter dc = ActualActual();
48  Date today = Settings::instance().evaluationDate();
49  report.addColumn("TradeId", string())
50  .addColumn("TradeType", string())
51  .addColumn("Maturity", Date())
52  .addColumn("MaturityTime", double(), 6)
53  .addColumn("NPV", double(), 6)
54  .addColumn("NpvCurrency", string())
55  .addColumn("NPV(Base)", double(), 6)
56  .addColumn("BaseCurrency", string())
57  .addColumn("Notional", double(), 2)
58  .addColumn("Notional(Base)", double(), 2)
59  .addColumn("NettingSet", string())
60  .addColumn("CounterParty", string());
61  for (auto trade : portfolio->trades()) {
62  string npvCcy = trade->npvCurrency();
63  Real fx = 1.0;
64  if (npvCcy != baseCurrency)
65  fx = market->fxSpot(npvCcy + baseCurrency, configuration)->value();
66  try {
67  Real npv = trade->instrument()->NPV();
68  QL_REQUIRE(std::isfinite(npv), "npv is not finite (" << npv << ")");
69  Date maturity = trade->maturity();
70  report.next()
71  .add(trade->id())
72  .add(trade->tradeType())
73  .add(maturity)
74  .add(maturity == QuantLib::Null<Date>() ? Null<Real>() : dc.yearFraction(today, maturity))
75  .add(npv)
76  .add(npvCcy)
77  .add(npv * fx)
78  .add(baseCurrency)
79  .add(trade->notional())
80  .add(trade->notional() * fx)
81  .add(trade->envelope().nettingSetId())
82  .add(trade->envelope().counterparty());
83  } catch (std::exception& e) {
84  ALOG(StructuredTradeErrorMessage(trade->id(), trade->tradeType(), "Error during trade pricing", e.what()));
85  Date maturity = trade->maturity();
86  report.next()
87  .add(trade->id())
88  .add(trade->tradeType())
89  .add(maturity)
90  .add(maturity == QuantLib::Null<Date>() ? Null<Real>() : dc.yearFraction(today, maturity))
91  .add(Null<Real>())
92  .add("#NA")
93  .add(Null<Real>())
94  .add("#NA")
95  .add(Null<Real>())
96  .add(Null<Real>())
97  .add("#NA")
98  .add("#NA");
99  }
100  }
101  report.end();
102  LOG("NPV file written");
103 }

105 void ReportWriter::writeCashflow(ore::data::Report& report, boost::shared_ptr<ore::data::Portfolio> portfolio,
106  boost::shared_ptr<ore::data::Market> market, const std::string& configuration) {
107  Date asof = Settings::instance().evaluationDate();
108  bool write_discount_factor = market ? true : false;
109  LOG("Writing cashflow report for " << asof);
110  report.addColumn("TradeId", string())
111  .addColumn("Type", string())
112  .addColumn("CashflowNo", Size())
113  .addColumn("LegNo", Size())
114  .addColumn("PayDate", Date())
115  .addColumn("FlowType", string())
116  .addColumn("Amount", double(), 4)
117  .addColumn("Currency", string())
118  .addColumn("Coupon", double(), 10)
119  .addColumn("Accrual", double(), 10)
120  .addColumn("fixingDate", Date())
121  .addColumn("fixingValue", double(), 10)
122  .addColumn("Notional", double(), 4);
123 
124  if (write_discount_factor) {
125  report.addColumn("DiscountFactor", double(), 10);
126  report.addColumn("PresentValue", double(), 10);
127  }
128  const vector<boost::shared_ptr<Trade>>& trades = portfolio->trades();
129 
130  for (Size k = 0; k < trades.size(); k++) {
131  if (trades[k]->tradeType() == "Swaption" || trades[k]->tradeType() == "CapFloor") {
132  WLOG("cashflow for " << trades[k]->tradeType() << " " << trades[k]->id() << " skipped");
133  continue;
134  }
135  try {
136  const vector<Leg>& legs = trades[k]->legs();
137  for (size_t i = 0; i < legs.size(); i++) {
138  const QuantLib::Leg& leg = legs[i];
139  bool payer = trades[k]->legPayers()[i];
140  string ccy = trades[k]->legCurrencies()[i];
141  Handle<YieldTermStructure> discountCurve;
142  if (write_discount_factor)
143  discountCurve = market->discountCurve(ccy, configuration);
144  for (size_t j = 0; j < leg.size(); j++) {
145  boost::shared_ptr<QuantLib::CashFlow> ptrFlow = leg[j];
146  Date payDate = ptrFlow->date();
147  if (payDate >= asof) {
148  Real amount = ptrFlow->amount();
149  string flowType = "";
150  if (payer)
151  amount *= -1.0;
152  std::string ccy = trades[k]->legCurrencies()[i];
153  boost::shared_ptr<QuantLib::Coupon> ptrCoupon =
154  boost::dynamic_pointer_cast<QuantLib::Coupon>(ptrFlow);
155  Real coupon;
156  Real accrual;
157  Real notional;
158  if (ptrCoupon) {
159  coupon = ptrCoupon->rate();
160  accrual = ptrCoupon->accrualPeriod();
161  notional = ptrCoupon->nominal();
162  flowType = "Interest";
163  } else {
164  coupon = Null<Real>();
165  accrual = Null<Real>();
166  notional = Null<Real>();
167  flowType = "Notional";
168  }
169  // This BMA part here (and below) is necessary because the fixingDay() method of
170  // AverageBMACoupon returns an exception rather than the last fixing day of the period.
171  boost::shared_ptr<AverageBMACoupon> ptrBMA =
172  boost::dynamic_pointer_cast<QuantLib::AverageBMACoupon>(ptrFlow);
173  boost::shared_ptr<QuantLib::FloatingRateCoupon> ptrFloat =
174  boost::dynamic_pointer_cast<QuantLib::FloatingRateCoupon>(ptrFlow);
175  boost::shared_ptr<QuantLib::InflationCoupon> ptrInfl =
176  boost::dynamic_pointer_cast<QuantLib::InflationCoupon>(ptrFlow);
177  boost::shared_ptr<QuantLib::IndexedCashFlow> ptrIndCf =
178  boost::dynamic_pointer_cast<QuantLib::IndexedCashFlow>(ptrFlow);
179  boost::shared_ptr<QuantExt::FXLinkedCashFlow> ptrFxlCf =
180  boost::dynamic_pointer_cast<QuantExt::FXLinkedCashFlow>(ptrFlow);
181  Date fixingDate;
182  Real fixingValue;
183  if (ptrBMA) {
184  // We return the last fixing inside the coupon period
185  fixingDate = ptrBMA->fixingDates().end()[-2];
186  fixingValue = ptrBMA->pricer()->swapletRate();
187  if (ptrBMA->index()->pastFixing(fixingDate) == Null<Real>())
188  flowType = "BMAaverage";
189  } else if (ptrFloat) {
190  fixingDate = ptrFloat->fixingDate();
191  fixingValue = ptrFloat->index()->fixing(fixingDate);
192  if (ptrFloat->index()->pastFixing(fixingDate) == Null<Real>())
193  flowType = "InterestProjected";
194  } else if (ptrInfl) {
195  fixingDate = ptrInfl->fixingDate();
196  fixingValue = ptrInfl->index()->fixing(fixingDate);
197  flowType = "Inflation";
198  } else if (ptrIndCf) {
199  fixingDate = ptrIndCf->fixingDate();
200  fixingValue = ptrIndCf->index()->fixing(fixingDate);
201  flowType = "Index";
202  } else if (ptrFxlCf) {
203  fixingDate = ptrFxlCf->fxFixingDate();
204  fixingValue = ptrFxlCf->fxRate();
205  } else {
206  fixingDate = Null<Date>();
207  fixingValue = Null<Real>();
208  }
209  report.next()
210  .add(trades[k]->id())
211  .add(trades[k]->tradeType())
212  .add(j + 1)
213  .add(i)
214  .add(payDate)
215  .add(flowType)
216  .add(amount)
217  .add(ccy)
218  .add(coupon)
219  .add(accrual)
220  .add(fixingDate)
221  .add(fixingValue)
222  .add(notional);
223 
224  if (write_discount_factor) {
225  Real discountFactor = discountCurve->discount(payDate);
226  report.add(discountFactor);
227  Real presentValue = discountFactor * amount;
228  report.add(presentValue);
229  }
230  }
231  }
232  }
233  } catch (std::exception& e) {
234  LOG("Exception writing cashflow report : " << e.what());
235  } catch (...) {
236  LOG("Exception writing cashflow report : Unkown Exception");
237  }
238  }
239  report.end();
240  LOG("Cashflow report written");
241 }

243 void ReportWriter::writeCurves(ore::data::Report& report, const std::string& configID, const DateGrid& grid,
244  const TodaysMarketParameters& marketConfig, const boost::shared_ptr<Market>& market,
245  const bool continueOnError) {
246  LOG("Write curves... ");
247 
248  QL_REQUIRE(marketConfig.hasConfiguration(configID), "curve configuration " << configID << " not found");
249 
250  map<string, string> discountCurves = marketConfig.mapping(MarketObject::DiscountCurve, configID);
251  map<string, string> YieldCurves = marketConfig.mapping(MarketObject::YieldCurve, configID);
252  map<string, string> indexCurves = marketConfig.mapping(MarketObject::IndexCurve, configID);
253  map<string, string> zeroInflationIndices, defaultCurves;
254  if (marketConfig.hasMarketObject(MarketObject::ZeroInflationCurve))
255  zeroInflationIndices = marketConfig.mapping(MarketObject::ZeroInflationCurve, configID);
256  if (marketConfig.hasMarketObject(MarketObject::DefaultCurve))
257  defaultCurves = marketConfig.mapping(MarketObject::DefaultCurve, configID);
258 
259  vector<Handle<YieldTermStructure>> yieldCurves;
260  vector<Handle<ZeroInflationIndex>> zeroInflationFixings;
261  vector<Handle<DefaultProbabilityTermStructure>> probabilityCurves;
262 
263  report.addColumn("Tenor", Period()).addColumn("Date", Date());
264 
265  for (auto it : discountCurves) {
266  DLOG("discount curve - " << it.first);
267  try {
268  yieldCurves.push_back(market->discountCurve(it.first, configID));
269  report.addColumn(it.first, double(), 15);
270  } catch (const std::exception& e) {
271  if (continueOnError) {
272  WLOG("skip this curve: " << e.what());
273  } else {
274  QL_FAIL(e.what());
275  }
276  }
277  }
278  for (auto it : YieldCurves) {
279  DLOG("yield curve - " << it.first);
280  try {
281  yieldCurves.push_back(market->yieldCurve(it.first, configID));
282  report.addColumn(it.first, double(), 15);
283  } catch (const std::exception& e) {
284  if (continueOnError) {
285  WLOG("skip this curve: " << e.what());
286  } else {
287  QL_FAIL(e.what());
288  }
289  }
290  }
291  for (auto it : indexCurves) {
292  DLOG("index curve - " << it.first);
293  try {
294  yieldCurves.push_back(market->iborIndex(it.first, configID)->forwardingTermStructure());
295  report.addColumn(it.first, double(), 15);
296  } catch (const std::exception& e) {
297  if (continueOnError) {
298  WLOG("skip this curve: " << e.what());
299  } else {
300  QL_FAIL(e.what());
301  }
302  }
303  }
304  for (auto it : zeroInflationIndices) {
305  DLOG("inflation curve - " << it.first);
306  try {
307  zeroInflationFixings.push_back(market->zeroInflationIndex(it.first, configID));
308  report.addColumn(it.first, double(), 15);
309  } catch (const std::exception& e) {
310  if (continueOnError) {
311  WLOG("skip this curve: " << e.what());
312  } else {
313  QL_FAIL(e.what());
314  }
315  }
316  }
317  for (auto it : defaultCurves) {
318  DLOG("default curve - " << it.first);
319  try {
320  probabilityCurves.push_back(market->defaultCurve(it.first, configID));
321  report.addColumn(it.first, double(), 15);
322  } catch (const std::exception& e) {
323  if (continueOnError) {
324  WLOG("skip this curve: " << e.what());
325  } else {
326  QL_FAIL(e.what());
327  }
328  }
329  }
330 
331  for (Size j = 0; j < grid.size(); ++j) {
332  Date date = grid[j];
333  report.next().add(grid.tenors()[j]).add(date);
334  for (Size i = 0; i < yieldCurves.size(); ++i)
335  report.add(yieldCurves[i]->discount(date));
336  for (Size i = 0; i < zeroInflationFixings.size(); ++i)
337  report.add(zeroInflationFixings[i]->fixing(date));
338  for (Size i = 0; i < probabilityCurves.size(); ++i)
339  report.add(probabilityCurves[i]->survivalProbability(date));
340  }
341  report.end();
342 }

527 void ReportWriter::writeNettingSetColva(ore::data::Report& report, boost::shared_ptr<PostProcess> postProcess,

671  } else if (!std::isfinite(sr.delta) || !std::isfinite(sr.gamma)) {
