58 void ReportWriter::writeNpv(ore::data::Report& report, const std::string& baseCurrency,
59  boost::shared_ptr<Market> market, const std::string& configuration,
60  boost::shared_ptr<Portfolio> portfolio) {
61  LOG("portfolio valuation");
62  DayCounter dc = ActualActual(ActualActual::ISDA);
63  Date today = Settings::instance().evaluationDate();
64  report.addColumn("TradeId", string())
65  .addColumn("TradeType", string())
66  .addColumn("Maturity", Date())
67  .addColumn("MaturityTime", double(), 6)
68  .addColumn("NPV", double(), 6)
69  .addColumn("NpvCurrency", string())
70  .addColumn("NPV(Base)", double(), 6)
71  .addColumn("BaseCurrency", string())
72  .addColumn("Notional", double(), 2)
73  .addColumn("NotionalCurrency", string())
74  .addColumn("Notional(Base)", double(), 2)
75  .addColumn("NettingSet", string())
76  .addColumn("CounterParty", string());
77  for (auto trade : portfolio->trades()) {
78  try {
79  string npvCcy = trade->npvCurrency();
80  Real fx = 1.0, fxNotional = 1.0;
81  if (npvCcy != baseCurrency)
82  fx = market->fxRate(npvCcy + baseCurrency, configuration)->value();
83  if (trade->notionalCurrency() != "" && trade->notionalCurrency() != baseCurrency)
84  fxNotional = market->fxRate(trade->notionalCurrency() + baseCurrency, configuration)->value();
85  Real npv = trade->instrument()->NPV();
86  QL_REQUIRE(std::isfinite(npv), "npv is not finite (" << npv << ")");
87  Date maturity = trade->maturity();
88  report.next()
89  .add(trade->id())
90  .add(trade->tradeType())
91  .add(maturity)
92  .add(maturity == QuantLib::Null<Date>() ? Null<Real>() : dc.yearFraction(today, maturity))
93  .add(npv)
94  .add(npvCcy)
95  .add(npv * fx)
96  .add(baseCurrency)
97  .add(trade->notional())
98  .add(trade->notionalCurrency() == "" ? nullString_ : trade->notionalCurrency())
99  .add(trade->notional() == Null<Real>() || trade->notionalCurrency() == ""
100  ? Null<Real>()
101  : trade->notional() * fxNotional)
102  .add(trade->envelope().nettingSetId())
103  .add(trade->envelope().counterparty());
104  } catch (std::exception& e) {
105  ALOG(StructuredTradeErrorMessage(trade->id(), trade->tradeType(), "Error during trade pricing", e.what()));
106  Date maturity = trade->maturity();
107  report.next()
108  .add(trade->id())
109  .add(trade->tradeType())
110  .add(maturity)
111  .add(maturity == QuantLib::Null<Date>() ? Null<Real>() : dc.yearFraction(today, maturity))
112  .add(Null<Real>())
113  .add(nullString_)
114  .add(Null<Real>())
115  .add(nullString_)
116  .add(Null<Real>())
117  .add(nullString_)
118  .add(Null<Real>())
119  .add(nullString_)
120  .add(nullString_);
121  }
122  }
123  report.end();
124  LOG("NPV file written");
125 }

189  string ccy = trades[k]->legCurrencies()[i];

194  boost::shared_ptr<QuantLib::CashFlow> ptrFlow = leg[j];
195  Date payDate = ptrFlow->date();
196  if (!ptrFlow->hasOccurred(asof) || includePastCashflows) {
197  Real amount = ptrFlow->amount();
198  string flowType = "";
199  if (payer)
200  amount *= -1.0;
201  std::string ccy = trades[k]->legCurrencies()[i];
202  boost::shared_ptr<QuantLib::Coupon> ptrCoupon =
203  boost::dynamic_pointer_cast<QuantLib::Coupon>(ptrFlow);
204  boost::shared_ptr<QuantExt::CommodityIndexedCashFlow> ptrCommCf =
205  boost::dynamic_pointer_cast<QuantExt::CommodityIndexedCashFlow>(ptrFlow);

209  Date accrualStartDate, accrualEndDate;

212  coupon = ptrCoupon->rate();
213  accrual = ptrCoupon->accrualPeriod();

215  accrualStartDate = ptrCoupon->accrualStartDate();
216  accrualEndDate = ptrCoupon->accrualEndDate();

221  } else if (ptrCommCf) {

223  accrual = Null<Real>();

226  accruedAmount = Null<Real>();

228  } else {
229  coupon = Null<Real>();
230  accrual = Null<Real>();
231  notional = Null<Real>();
232  accrualStartDate = accrualEndDate = Null<Date>();
233  accruedAmount = Null<Real>();
234  flowType = "Notional";
235  }
236  // This BMA part here (and below) is necessary because the fixingDay() method of
237  // AverageBMACoupon returns an exception rather than the last fixing day of the period.
238 
239  boost::shared_ptr<QuantLib::Coupon> cpn =
240  boost::dynamic_pointer_cast<QuantLib::Coupon>(ptrFlow);
241  if (cpn) {
242  ptrFlow = unpackIndexedCoupon(cpn);
243  }

245  boost::dynamic_pointer_cast<QuantLib::AverageBMACoupon>(ptrFlow);
246  boost::shared_ptr<QuantLib::FloatingRateCoupon> ptrFloat =
247  boost::dynamic_pointer_cast<QuantLib::FloatingRateCoupon>(ptrFlow);
248  boost::shared_ptr<QuantLib::InflationCoupon> ptrInfl =
249  boost::dynamic_pointer_cast<QuantLib::InflationCoupon>(ptrFlow);

251  boost::dynamic_pointer_cast<QuantLib::IndexedCashFlow>(ptrFlow);
252  boost::shared_ptr<QuantExt::FXLinkedCashFlow> ptrFxlCf =
253  boost::dynamic_pointer_cast<QuantExt::FXLinkedCashFlow>(ptrFlow);
254  boost::shared_ptr<QuantExt::EquityCoupon> ptrEqCp =
255  boost::dynamic_pointer_cast<QuantExt::EquityCoupon>(ptrFlow);
256  Date fixingDate;
257  Real fixingValue;
258  if (ptrBMA) {
259  // We return the last fixing inside the coupon period
260  fixingDate = ptrBMA->fixingDates().end()[-2];
261  fixingValue = ptrBMA->pricer()->swapletRate();
262  if (fixingDate > asof)
263  flowType = "BMAaverage";
264  } else if (ptrFloat) {
265  fixingDate = ptrFloat->fixingDate();
266  fixingValue = ptrFloat->index()->fixing(fixingDate);
267  if (fixingDate > asof)

269  // for ON coupons the fixing value is the compounded / averaged rate, not the last

271  if (auto on = boost::dynamic_pointer_cast<QuantExt::AverageONIndexedCoupon>(ptrFloat)) {
272  fixingValue = (on->rate() - on->spread()) / on->gearing();
273  } else if (auto on = boost::dynamic_pointer_cast<QuantExt::OvernightIndexedCoupon>(
274  ptrFloat)) {
275  fixingValue = (on->rate() - on->effectiveSpread()) / on->gearing();
276  } else if (auto c = boost::dynamic_pointer_cast<
277  QuantExt::CappedFlooredAverageONIndexedCoupon>(ptrFloat)) {
278  fixingValue = (c->underlying()->rate() - c->underlying()->spread()) /
279  c->underlying()->gearing();
280  } else if (auto c = boost::dynamic_pointer_cast<
281  QuantExt::CappedFlooredOvernightIndexedCoupon>(ptrFloat)) {
282  fixingValue = (c->underlying()->rate() - c->underlying()->effectiveSpread()) /
283  c->underlying()->gearing();
284  }
285  } else if (ptrInfl) {
286  fixingDate = ptrInfl->fixingDate();
287  fixingValue = ptrInfl->indexFixing();
288  flowType = "Inflation";
289  } else if (ptrIndCf) {
290  fixingDate = ptrIndCf->fixingDate();
291  fixingValue = ptrIndCf->indexFixing();
292  flowType = "Index";
293  } else if (ptrFxlCf) {
294  fixingDate = ptrFxlCf->fxFixingDate();
295  fixingValue = ptrFxlCf->fxRate();
296  } else if (ptrEqCp) {
297  fixingDate = ptrEqCp->fixingEndDate();
298  fixingValue = ptrEqCp->equityCurve()->fixing(fixingDate);
299  } else if (ptrCommCf) {
300  fixingDate = ptrCommCf->date();
301  fixingValue = ptrCommCf->index()->fixing(ptrCommCf->pricingDate());
302  } else {
303  fixingDate = Null<Date>();
304  fixingValue = Null<Real>();
305  }
306 
307  Real effectiveAmount = Null<Real>();
308  Real discountFactor = Null<Real>();
309  Real presentValue = Null<Real>();
310  Real floorStrike = Null<Real>();
311  Real capStrike = Null<Real>();
312  Real floorVolatility = Null<Real>();
313  Real capVolatility = Null<Real>();
314 
315  if (amount != Null<Real>())
316  effectiveAmount = amount * multiplier;
317 
318  if (market) {
319  discountFactor = ptrFlow->hasOccurred(asof) ? 0.0 : discountCurve->discount(payDate);
320  if(effectiveAmount != Null<Real>())
321  presentValue = discountFactor * effectiveAmount;
322 
323  // scan for known capped / floored coupons and extract cap / floor strike and fixing
324  // date
325 
326  // unpack stripped cap/floor coupon
327  boost::shared_ptr<CashFlow> c = ptrFlow;
328  if (auto tmp = boost::dynamic_pointer_cast<StrippedCappedFlooredCoupon>(ptrFlow)) {
329  c = tmp->underlying();
330  }

332  std::string qlIndexName; // index used to retrieve vol
333  bool usesCapVol = false, usesSwaptionVol = false;
334  Period swaptionTenor;
335  if (auto tmp = boost::dynamic_pointer_cast<CappedFlooredCoupon>(c)) {
336  floorStrike = tmp->effectiveFloor();
337  capStrike = tmp->effectiveCap();
338  volFixingDate = tmp->fixingDate();
339  qlIndexName = tmp->index()->name();
340  if (auto cms = boost::dynamic_pointer_cast<CmsCoupon>(tmp->underlying())) {
341  swaptionTenor = cms->swapIndex()->tenor();
342  qlIndexName = cms->swapIndex()->iborIndex()->name();
343  usesSwaptionVol = true;
344  } else if (auto ibor = boost::dynamic_pointer_cast<IborCoupon>(tmp->underlying())) {
345  qlIndexName = ibor->index()->name();
346  usesCapVol = true;
347  }
348  } else if (auto tmp =
349  boost::dynamic_pointer_cast<CappedFlooredOvernightIndexedCoupon>(c)) {
350  floorStrike = tmp->effectiveFloor();
351  capStrike = tmp->effectiveCap();
352  volFixingDate = tmp->underlying()->fixingDates().front();
353  qlIndexName = tmp->index()->name();
354  usesCapVol = true;
355  } else if (auto tmp =
356  boost::dynamic_pointer_cast<CappedFlooredAverageONIndexedCoupon>(c)) {
357  floorStrike = tmp->effectiveFloor();
358  capStrike = tmp->effectiveCap();
359  volFixingDate = tmp->underlying()->fixingDates().front();
360  qlIndexName = tmp->index()->name();
361  usesCapVol = true;
362  }
363 
364  // get market volaility for cap / floor
365 
366  if (volFixingDate != Date() && fixingDate > market->asofDate()) {
367  volFixingDate = std::max(volFixingDate, market->asofDate() + 1);
368  if (floorStrike != Null<Real>()) {
369  if (usesSwaptionVol) {
370  floorVolatility =
371  market
372  ->swaptionVol(IndexNameTranslator::instance().oreName(qlIndexName),
373  configuration)
374  ->volatility(volFixingDate, swaptionTenor, floorStrike);
375  } else if (usesCapVol) {
376  floorVolatility =
377  market
378  ->capFloorVol(IndexNameTranslator::instance().oreName(qlIndexName),
379  configuration)
380  ->volatility(volFixingDate, floorStrike);
381  }
382  }
383  if (capStrike != Null<Real>()) {
384  if (usesSwaptionVol) {
385  capVolatility =
386  market
387  ->swaptionVol(IndexNameTranslator::instance().oreName(qlIndexName),
388  configuration)
389  ->volatility(volFixingDate, swaptionTenor, capStrike);
390  } else if (usesCapVol) {
391  capVolatility =
392  market
393  ->capFloorVol(IndexNameTranslator::instance().oreName(qlIndexName),
394  configuration)
395  ->volatility(volFixingDate, capStrike);
396  }
397  }
398  }
399  }
400 
401  report.next()
402  .add(trades[k]->id())
403  .add(trades[k]->tradeType())
404  .add(j + 1)
405  .add(i)
406  .add(payDate)
407  .add(flowType)
408  .add(effectiveAmount)
409  .add(ccy)
410  .add(coupon)
411  .add(accrual)
412  .add(accrualStartDate)
413  .add(accrualEndDate)
414  .add(accruedAmount * (accruedAmount == Null<Real>() ? 1.0 : multiplier))
415  .add(fixingDate)
416  .add(fixingValue)
417  .add(notional * (notional == Null<Real>() ? 1.0 : multiplier))
418  .add(discountFactor)
419  .add(presentValue)
420  .add(floorStrike)
421  .add(capStrike)
422  .add(floorVolatility)
423  .add(capVolatility);
424  }
425  }
426  }
427 
428  } else {
429 
430  // additional result based cashflow reporting
431 
432  auto tmp = trades[k]->instrument()->additionalResults().find("cashFlowResults");
433  QL_REQUIRE(
434  tmp != trades[k]->instrument()->additionalResults().end(),
435  "internal error: expected cashFlowResults in additional results when writing cashflow report");

440  for (auto const& cf : cfResults) {
441  string ccy = "";
442  if (!cf.currency.empty()) {
443  ccy = cf.currency;
444  } else if (trades[k]->legCurrencies().size() > cf.legNumber) {
445  ccy = trades[k]->legCurrencies()[cf.legNumber];
446  } else {
447  ccy = trades[k]->npvCurrency();
448  }
449 
450  Real effectiveAmount = Null<Real>();
451  Real discountFactor = Null<Real>();
452  Real presentValue = Null<Real>();
453  Real floorStrike = Null<Real>();
454  Real capStrike = Null<Real>();
455  Real floorVolatility = Null<Real>();
456  Real capVolatility = Null<Real>();

459  effectiveAmount = cf.amount * multiplier;
460  if (cf.discountFactor != Null<Real>())
461  discountFactor = cf.discountFactor;
462  else if (!cf.currency.empty() && cf.payDate != Null<Date>() && market) {
463  discountFactor =
464  cf.payDate < asof ? 0.0 : market->discountCurve(cf.currency)->discount(cf.payDate);
465  }
466  if (cf.presentValue != Null<Real>()) {
467  presentValue = cf.presentValue * multiplier;
468  } else if (effectiveAmount != Null<Real>() && discountFactor != Null<Real>()) {
469  presentValue = effectiveAmount * discountFactor;
470  }
471  if (cf.floorStrike != Null<Real>())
472  floorStrike = cf.floorStrike;
473  if (cf.capStrike != Null<Real>())
474  capStrike = cf.capStrike;
475  if (cf.floorVolatility != Null<Real>())
476  floorVolatility = cf.floorVolatility;
477  if (cf.capVolatility != Null<Real>())
478  capVolatility = cf.capVolatility;
479

481  .add(trades[k]->id())
482  .add(trades[k]->tradeType())
483  .add(++cashflowNumber[cf.legNumber])
484  .add(cf.legNumber)
485  .add(cf.payDate)
486  .add(cf.type)
487  .add(effectiveAmount)
488  .add(ccy)
489  .add(cf.rate)
490  .add(cf.accrualPeriod)
491  .add(cf.accrualStartDate)
492  .add(cf.accrualEndDate)
493  .add(cf.accruedAmount * (cf.accruedAmount == Null<Real>() ? 1.0 : multiplier))
494  .add(cf.fixingDate)
495  .add(cf.fixingValue)
496  .add(cf.notional * (cf.notional == Null<Real>() ? 1.0 : multiplier))
497  .add(discountFactor)
498  .add(presentValue)
499  .add(floorStrike)
500  .add(capStrike)
501  .add(floorVolatility)
502  .add(capVolatility);
503  }
504  }
505 
506  } catch (std::exception& e) {
507  ALOG("Exception writing cashflow report : " << e.what());
508  } catch (...) {
509  ALOG("Exception writing cashflow report : Unknown Exception");
510  }
511  }
512  report.end();
513  LOG("Cashflow report written");
514 }

516 void ReportWriter::writeCashflowNpv(ore::data::Report& report,
517  const ore::data::InMemoryReport& cashflowReport,
518  boost::shared_ptr<ore::data::Market> market,
519  const std::string& configuration,
520  const std::string& baseCcy,
521  const Date& horizon) {
522  // Pick the following fields form the in memory report:
523  // - tradeId
524  // - payment date
525  // - currency
526  // - present value
527  // Then convert PVs into base currency, aggrate per trade if payment date is within the horizon
528  // Write the resulting aggregate PV per trade into the report.
529 
530  Size tradeIdColumn = 0;
531  Size tradeTypeColumn = 1;
532  Size payDateColumn = 4;
533  Size ccyColumn = 7;
534  Size pvColumn = 17;
535  QL_REQUIRE(cashflowReport.header(tradeIdColumn) == "TradeId", "incorrect trade id column " << tradeIdColumn);
536  QL_REQUIRE(cashflowReport.header(tradeTypeColumn) == "Type", "incorrect trade type column " << tradeTypeColumn);
537  QL_REQUIRE(cashflowReport.header(payDateColumn) == "PayDate", "incorrect payment date column " << payDateColumn);
538  QL_REQUIRE(cashflowReport.header(ccyColumn) == "Currency", "incorrect currency column " << ccyColumn);
539  QL_REQUIRE(cashflowReport.header(pvColumn) == "PresentValue", "incorrect pv column " << pvColumn);
540 
541  map<string, Real> npvMap;
542  Date asof = Settings::instance().evaluationDate();
543  for (Size i = 0; i < cashflowReport.rows(); ++i) {
544  string tradeId = boost::get<string>(cashflowReport.data(tradeIdColumn).at(i));
545  string tradeType = boost::get<string>(cashflowReport.data(tradeTypeColumn).at(i));
546  Date payDate = boost::get<Date>(cashflowReport.data(payDateColumn).at(i));
547  string ccy = boost::get<string>(cashflowReport.data(ccyColumn).at(i));
548  Real pv = boost::get<Real>(cashflowReport.data(pvColumn).at(i));
549  Real fx = 1.0;
550  // There shouldn't be entries in the cf report without ccy. We assume ccy = baseCcy in this case and log an error.
551  if (ccy.empty()) {
552  ALOG(StructuredTradeErrorMessage(tradeId, tradeType, "Error during CashflowNpv calculation.",
553  "Cashflow in row " + std::to_string(i) +
554  " has no ccy. Assuming ccy = baseCcy = " + baseCcy + "."));
555  }
556  if (!ccy.empty() && ccy != baseCcy)
557  fx = market->fxSpot(ccy + baseCcy, configuration)->value();
558  if (npvMap.find(tradeId) == npvMap.end())
559  npvMap[tradeId] = 0.0;
560  if (payDate > asof && payDate <= horizon) {
