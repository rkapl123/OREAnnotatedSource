41/*! A container for holding single CRIF records or aggregated CRIF records.
42 A CRIF record is a row of the CRIF file outlined in the document:
43 <em>ISDA SIMM Methodology, Risk Data Standards. Version 1.36: 1 February 2017.</em>
44 or an updated version thereof.
45*/
46struct CrifRecord {
47
48 // required data
49 std::string tradeId;
50 std::string portfolioId;
51 SimmConfiguration::ProductClass productClass = SimmConfiguration::ProductClass::Empty;
52 SimmConfiguration::RiskType riskType = SimmConfiguration::RiskType::Notional;
53 std::string qualifier;
54 std::string bucket;
55 std::string label1;
56 std::string label2;
57 std::string amountCurrency;
58 mutable QuantLib::Real amount = QuantLib::Null<QuantLib::Real>();
59 mutable QuantLib::Real amountUsd = QuantLib::Null<QuantLib::Real>();
60
61 // optional data
62 std::string tradeType;
63 std::string agreementType;
64 std::string callType;
65 std::string initialMarginType;
66 std::string legalEntityId;
67 NettingSetDetails nettingSetDetails; // consists of the above: agreementType ... legalEntityId
68 mutable std::string imModel;
69 mutable std::string collectRegulations;
70 mutable std::string postRegulations;
71 std::string endDate;
72
73 // additional data
74 std::map<std::string, std::string> additionalFields;
75
76 // Default Constructor
77 CrifRecord() {}
78
79 CrifRecord(std::string tradeId, std::string tradeType, NettingSetDetails nettingSetDetails,
80 SimmConfiguration::ProductClass productClass, SimmConfiguration::RiskType riskType,
81 std::string qualifier, std::string bucket, std::string label1, std::string label2,
82 std::string amountCurrency, QuantLib::Real amount, QuantLib::Real amountUsd, std::string imModel = "",
83 std::string collectRegulations = "", std::string postRegulations = "", std::string endDate = "",
84 std::map<std::string, std::string> additionalFields = {})
85 : tradeId(tradeId), portfolioId(nettingSetDetails.nettingSetId()),
86 productClass(productClass), riskType(riskType), qualifier(qualifier),
87 bucket(bucket), label1(label1), label2(label2), amountCurrency(amountCurrency), amount(amount),
88 amountUsd(amountUsd), tradeType(tradeType), nettingSetDetails(nettingSetDetails), imModel(imModel),
89 collectRegulations(collectRegulations), postRegulations(postRegulations), endDate(endDate),
90 additionalFields(additionalFields) {}
91
92 CrifRecord(std::string tradeId, std::string tradeType, std::string portfolioId,
93 SimmConfiguration::ProductClass productClass, SimmConfiguration::RiskType riskType,
94 std::string qualifier, std::string bucket, std::string label1, std::string label2,
95 std::string amountCurrency, QuantLib::Real amount, QuantLib::Real amountUsd, std::string imModel = "",
96 std::string collectRegulations = "", std::string postRegulations = "", std::string endDate = "",
97 std::map<std::string, std::string> additionalFields = {})
98 : CrifRecord(tradeId, tradeType, NettingSetDetails(portfolioId), productClass, riskType, qualifier,
99 bucket, label1, label2, amountCurrency, amount, amountUsd, imModel,
100 collectRegulations, postRegulations, endDate, additionalFields) {}
101
102 bool hasAmountCcy() const { return !amountCurrency.empty(); }
103 bool hasAmount() const { return amount != QuantLib::Null<QuantLib::Real>(); }
104 bool hasAmountUsd() const { return amountUsd != QuantLib::Null<QuantLib::Real>(); }
105
106 // We use (and require) amountUsd for all risk types except for SIMM parameters AddOnNotionalFactor and
107 // ProductClassMultiplier as these are multipliers and not amounts denominated in the amountCurrency
108 bool requiresAmountUsd() const {
109 return riskType != SimmConfiguration::RiskType::AddOnNotionalFactor &&
110 riskType != SimmConfiguration::RiskType::ProductClassMultiplier;
111 }
112
113 bool isSimmParameter() const {
114 return riskType == SimmConfiguration::RiskType::AddOnFixedAmount ||
115 riskType == SimmConfiguration::RiskType::AddOnNotionalFactor ||
116 riskType == SimmConfiguration::RiskType::ProductClassMultiplier;
117 }
118
119 //! Define how CRIF records are compared
120 bool operator<(const CrifRecord& cr) const {
121 return std::tie(tradeId, nettingSetDetails, productClass, riskType, qualifier, bucket, label1, label2,
122 amountCurrency, collectRegulations, postRegulations) <
123 std::tie(cr.tradeId, cr.nettingSetDetails, cr.productClass, cr.riskType, cr.qualifier, cr.bucket,
124 cr.label1, cr.label2, cr.amountCurrency, cr.collectRegulations, cr.postRegulations);
125 }
126 static bool amountCcyLTCompare(const CrifRecord& cr1, const CrifRecord& cr2) {
127 return std::tie(cr1.tradeId, cr1.nettingSetDetails, cr1.productClass, cr1.riskType, cr1.qualifier, cr1.bucket,
128 cr1.label1, cr1.label2, cr1.collectRegulations, cr1.postRegulations) <
129 std::tie(cr2.tradeId, cr2.nettingSetDetails, cr2.productClass, cr2.riskType, cr2.qualifier, cr2.bucket,
130 cr2.label1, cr2.label2, cr2.collectRegulations, cr2.postRegulations);
131 }
132 bool operator==(const CrifRecord& cr) const {
133 return std::tie(tradeId, nettingSetDetails, productClass, riskType, qualifier, bucket, label1, label2,
134 amountCurrency, collectRegulations, postRegulations) ==
135 std::tie(cr.tradeId, cr.nettingSetDetails, cr.productClass, cr.riskType, cr.qualifier, cr.bucket,
136 cr.label1, cr.label2, cr.amountCurrency, cr.collectRegulations, cr.postRegulations);
137 }
138 static bool amountCcyEQCompare(const CrifRecord& cr1, const CrifRecord& cr2) {
139 return std::tie(cr1.tradeId, cr1.nettingSetDetails, cr1.productClass, cr1.riskType, cr1.qualifier, cr1.bucket,
140 cr1.label1, cr1.label2, cr1.collectRegulations, cr1.postRegulations) ==
141 std::tie(cr2.tradeId, cr2.nettingSetDetails, cr2.productClass, cr2.riskType, cr2.qualifier, cr2.bucket,
142 cr2.label1, cr2.label2, cr2.collectRegulations, cr2.postRegulations);
143 }
144
145 static std::map<QuantLib::Size, std::set<std::string>> additionalHeaders;
146
147};
