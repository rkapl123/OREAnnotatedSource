37//! Simple Scenario class
38/*! This implementation stores the data in several maps within in each scenario class instance.
39 This contains redundant information (keys) across scenario instances which can cause issues
40 when many scenario instances are managed in memory.
41
42 \ingroup scenario
43*/
44class SimpleScenario : public Scenario {
45public:
46 //! Constructor
47 SimpleScenario() {}
48 //! Constructor
49 SimpleScenario(Date asof, const std::string& label = "", Real numeraire = 0)
50 : asof_(asof), numeraire_(numeraire), label_(label) {}
51
52 //! Return the scenario asof date
53 const Date& asof() const override { return asof_; }
54
55 //! Return the scenario label
56 const std::string& label() const override { return label_; }
57 //! set the label
58 void label(const string& s) override { label_ = s; }
59
60 //! Get Numeraire ratio n = N(t) / N(0) so that Price(0) = N(0) * E [Price(t) / N(t) ]
61 Real getNumeraire() const override { return numeraire_; }
62 //! Set the Numeraire ratio n = N(t) / N(0) so that Price(0) = N(0) * E [Price(t) / N(t) ]
63 void setNumeraire(Real n) override { numeraire_ = n; }
64
65 //! Check, get, add a single market point
66 bool has(const RiskFactorKey& key) const override;
67 const std::vector<RiskFactorKey>& keys() const override { return keys_; }
68 void add(const RiskFactorKey& key, Real value) override;
69 Real get(const RiskFactorKey& key) const override;
70
71 boost::shared_ptr<Scenario> clone() const override;
72
73 //! get data map
74 const std::map<RiskFactorKey, Real> data() const { return data_; }
75
76private:
77 friend class boost::serialization::access;
78 template <class Archive> void serialize(Archive& ar, const unsigned int) {
79 ar& boost::serialization::base_object<Scenario>(*this);
80 ar& asof_;
81 ar& numeraire_;
82 ar& data_;
83 ar& keys_;
84 ar& label_;
85 }
86 Date asof_;
87 Real numeraire_;
88 std::map<RiskFactorKey, Real> data_;
89 std::vector<RiskFactorKey> keys_;
90 std::string label_;
91};
