39Size matches(const std::set<std::string>& requested, const std::set<std::string>& available) {
40 Size count = 0;
41 for (auto r : requested) {
42 if (available.find(r) != available.end())
43 count++;
44 }
45 return count;

65void AnalyticsManager::addAnalytic(const std::string& label, const boost::shared_ptr<Analytic>& analytic) {
66 // Allow overriding, but warn
67 if (analytics_.find(label) != analytics_.end()) {
68 WLOG("Overwriting analytic with label " << label);
69 }
70
71 // Label is not necessarily a valid analytics type
72 // Get the latter via analytic->analyticTypes()
73 LOG("register analytic with label '" << label << "' and sub-analytics " << to_string(analytic->analyticTypes()));
74 analytics_[label] = analytic;
75 // This forces an update of valid analytics vector with the next call to validAnalytics()
76 validAnalytics_.clear();
77}

107void AnalyticsManager::runAnalytics(const std::set<std::string>& analyticTypes,
108 const boost::shared_ptr<MarketCalibrationReport>& marketCalibrationReport) {
109
110 requestedAnalytics_ = analyticTypes;
111
112 if (analytics_.size() == 0)
113 return;
114
115 std::vector<boost::shared_ptr<ore::data::TodaysMarketParameters>> tmps;
116 std::set<Date> marketDates;
117 for (const auto& a : analytics_) {
118 std::vector<boost::shared_ptr<ore::data::TodaysMarketParameters>> atmps = a.second->todaysMarketParams();
119 tmps.insert(end(tmps), begin(atmps), end(atmps));
120 auto mdates = a.second->marketDates();
121 marketDates.insert(mdates.begin(), mdates.end());
122 }
123
124 // Do we need market data
125 bool requireMarketData = false;
126 for (const auto& tmp : tmps) {
127 if (!tmp->empty())
128 requireMarketData = true;
129 }
130
131 LOG("AnalyticsManager::runAnalytics: requireMarketData " << (requireMarketData ? "Y" : "N"));
132
133 if (requireMarketData) {
134 // load the market data
135 if (tmps.size() > 0) {
136 LOG("AnalyticsManager::runAnalytics: populate loader");
137 marketDataLoader_->populateLoader(tmps, marketDates);
138 }
139
140 boost::shared_ptr<InMemoryReport> mdReport = boost::make_shared<InMemoryReport>();
141 boost::shared_ptr<InMemoryReport> fixingReport = boost::make_shared<InMemoryReport>();
142 boost::shared_ptr<InMemoryReport> dividendReport = boost::make_shared<InMemoryReport>();
143
144 ore::analytics::ReportWriter(inputs_->reportNaString())
145 .writeMarketData(*mdReport, marketDataLoader_->loader(), inputs_->asof(),
146 marketDataLoader_->quotes()[inputs_->asof()],
147 !inputs_->entireMarket());
148 ore::analytics::ReportWriter(inputs_->reportNaString())
149 .writeFixings(*fixingReport, marketDataLoader_->loader());
150 ore::analytics::ReportWriter(inputs_->reportNaString())
151 .writeDividends(*dividendReport, marketDataLoader_->loader());
152
153 reports_["MARKETDATA"]["marketdata"] = mdReport;
154 reports_["FIXINGS"]["fixings"] = fixingReport;
155 reports_["DIVIDENDS"]["dividends"] = dividendReport;
156 }
157
158 // run requested analytics
159 for (auto a : analytics_) {
160 if (matches(analyticTypes, a.second->analyticTypes()) > 0) {
161 LOG("run analytic with label '" << a.first << "'");
162 a.second->runAnalytic(marketDataLoader_->loader(), analyticTypes);
163 LOG("run analytic with label '" << a.first << "' finished.");
164 // then populate the market calibration report if required
165 if (marketCalibrationReport)
166 a.second->marketCalibration(marketCalibrationReport);
167 }
168 }
169
170 if (inputs_->portfolio()) {
171 auto pricingStatsReport = boost::make_shared<InMemoryReport>();
172 ReportWriter(inputs_->reportNaString())
173 .writePricingStats(*pricingStatsReport, inputs_->portfolio());
174 reports_["STATS"]["pricingstats"] = pricingStatsReport;
175 }
176
177 if (marketCalibrationReport)
178 marketCalibrationReport->outputCalibrationReport();
179}

208std::map<std::string, Size> checkReportNames(const ore::analytics::Analytic::analytic_reports& rpts) {
209 std::map<std::string, Size> m;
210 for (const auto& rep : rpts) {
211 for (auto b : rep.second) {
212 string reportName = b.first;
213 auto it = m.find(reportName);
214 if (it == m.end())
215 m[reportName] = 1;
216 else
217 m[reportName] ++;
218 }
219 }
220 for (auto r : m) {
221 LOG("report name " << r.first << " occurs " << r.second << " times");
222 }
223 return m;
224}

226bool endsWith(const std::string& name, const std::string& suffix) {
227 if (suffix.size() > name.size())

229 else

233void AnalyticsManager::toFile(const ore::analytics::Analytic::analytic_reports& rpts, const std::string& outputPath,
234 const std::map<std::string, std::string>& reportNames, const char sep,
235 const bool commentCharacter, char quoteChar, const string& nullString,
236 const std::set<std::string>& lowerHeaderReportNames) {
237 std::map<std::string, Size> hits = checkReportNames(rpts);
238 for (const auto& rep : rpts) {
239 string analytic = rep.first;
240 for (auto b : rep.second) {
241 string reportName = b.first;
242 boost::shared_ptr<InMemoryReport> report = b.second;
243 string fileName;
244 auto it = hits.find(reportName);
245 QL_REQUIRE(it != hits.end(), "something wrong here");
246 if (it->second == 1) {
247 // The report name is unique, check whether we want to rename it or use the standard name
248 auto it2 = reportNames.find(reportName);
249 fileName = it2 != reportNames.end() ? it2->second : reportName;
250 }
251 else {
252 ALOG("Report " << reportName << " occurs " << it->second << " times, fix report naming");
253 fileName = analytic + "_" + reportName + "_" + to_string(hits[fileName]);
254 }
255
256 // attach a suffix only if it does not have one already
257 string suffix = "";
258 if (!endsWith(fileName,".csv") && !endsWith(fileName, ".txt"))
259 suffix = ".csv";
260 std::string fullFileName = outputPath + "/" + fileName + suffix;
261
262 report->toFile(fullFileName, sep, commentCharacter, quoteChar, nullString,
263 lowerHeaderReportNames.find(reportName) != lowerHeaderReportNames.end());
264 LOG("report " << reportName << " written to " << fullFileName);
265 }
266 }
267}
