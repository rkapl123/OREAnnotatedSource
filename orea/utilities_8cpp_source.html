54std::vector<std::string> loadFactorList(const std::string& inputFileName, const char delim) {
55 LOG("Load factor list from file " << inputFileName);
56 std::ifstream file;
57 file.open(inputFileName);
58 QL_REQUIRE(file.is_open(), "error opening file " << inputFileName);
59 std::vector<string> result;
60 try {
61 while (!file.eof()) {
62 std::string line;
63 getline(file, line, delim);
64 if (line.size() == 0)
65 continue;
66 result.push_back(line);
67 }
68 } catch (std::exception& e) {
69 file.close();
70 QL_FAIL("error during reading file: " << e.what());
71 }
72
73 LOG("Loaded factor list of size " << result.size());
74 return result;
75} // loadFactorList

77std::vector<std::vector<double>> loadScenarios(const std::string& inputFileName, const char delim) {
78 LOG("Load scenarios from file " << inputFileName);
79 std::ifstream file;
80 file.open(inputFileName);
81 QL_REQUIRE(file.is_open(), "error opening file " << inputFileName);
82 std::vector<std::vector<double>> result;
83 try {
84 while (!file.eof()) {
85 Size currentScenario = Null<Size>(); //, currentFactor;// = Null<Size>();
86 std::string line;
87 getline(file, line, delim);
88 if (line.size() == 0)
89 continue;
90 std::vector<string> tokens;
91 boost::trim(line);
92 boost::split(tokens, line, boost::is_any_of(",;\t "), boost::token_compress_off);
93 QL_REQUIRE(tokens.size() == 3, "loadScenarios, expected 3 tokens in line: " << line);
94 Size tmpSc = parseInteger(tokens[0]);
95 // Size tmpFc = parseInteger(tokens[1]);
96 if (currentScenario != tmpSc) {
97 result.push_back(std::vector<double>());
98 currentScenario = tmpSc;
99 }
100 // currentFactor = tmpFc;
101 double move = parseReal(tokens[2]);
102 result.back().push_back(move);
103 }
104 } catch (std::exception& e) {
105 file.close();
106 QL_FAIL("error during reading file: " << e.what());
107 }
108
109 LOG("Loaded " << result.size() << " scenarios, first entry contains " << result.front().size() << " factors");
110 return result;
111} // loadScenarios

113Matrix loadCovarianceMatrix(const std::string& inputFileName, const char delim) {
114 LOG("Load covariance matrix from file " << inputFileName);
115 std::ifstream file;
116 file.open(inputFileName);
117 QL_REQUIRE(file.is_open(), "error opening file " << inputFileName);
118 std::vector<std::pair<Size, Size>> pos;
119 std::vector<double> val;
120 Size maxI = 0, maxJ = 0;
121 try {
122 while (!file.eof()) {
123 std::string line;
124 getline(file, line, delim);
125 if (line.size() == 0)
126 continue;
127 std::vector<string> tokens;
128 boost::trim(line);
129 boost::split(tokens, line, boost::is_any_of(",;\t "), boost::token_compress_off);
130 QL_REQUIRE(tokens.size() == 3, "loadCovarianceMatrix, expected 3 tokens in line: " << line);
131 Size i = parseInteger(tokens[0]);
132 Size j = parseInteger(tokens[1]);
133 double tmp = parseReal(tokens[2]);
134 pos.push_back(std::make_pair(i, j));
135 val.push_back(tmp);
136 maxI = std::max(maxI, i);
137 maxJ = std::max(maxJ, j);
138 }
139 } catch (std::exception& e) {
140 file.close();
141 QL_FAIL("error during reading file: " << e.what());
142 }
143
144 LOG("Loaded " << val.size() << " data points, dimension of matrix is " << maxI + 1 << "x" << maxJ + 1);
145 QL_REQUIRE(maxI == maxJ, "Expected quadratic matrix");
146
147 Matrix result(maxI + 1, maxI + 1);
148 for (Size i = 0; i < pos.size(); ++i) {
149 result[pos[i].first][pos[i].second] = val[i];
150 result[pos[i].second][pos[i].first] = val[i];
151 }
152
153 // log the eigenvalues
154 QuantLib::SymmetricSchurDecomposition ssd(result);
155 for (Size i = 0; i < ssd.eigenvalues().size(); ++i) {
156 LOG("Eigenvalue " << i << " = " << ssd.eigenvalues()[i]);
157 }
158 return result;
159} // load CovarianceMatrix

161SimmVersion parseSimmVersion(const string& version) {
162 static map<string, SimmVersion> versionMap = {{"1.0", SimmVersion::V1_0},
163 {"1.1", SimmVersion::V1_1},

190boost::shared_ptr<SimmConfiguration> buildSimmConfiguration(const string& simmVersion,
191 const boost::shared_ptr<SimmBucketMapper>& simmBucketMapper,
192 const Size& mporDays) {
193
194 auto version = parseSimmVersion(simmVersion);
195
196 switch (version) {
197 case SimmVersion::V1_0:
198 return boost::make_shared<SimmConfiguration_ISDA_V1_0>(simmBucketMapper);
199 break;
200 case SimmVersion::V1_3:
201 return boost::make_shared<SimmConfiguration_ISDA_V1_3>(simmBucketMapper);
202 break;
203 case SimmVersion::V1_3_38:
204 return boost::make_shared<SimmConfiguration_ISDA_V1_3_38>(simmBucketMapper);
205 break;
206 case SimmVersion::V2_0:
207 return boost::make_shared<SimmConfiguration_ISDA_V2_0>(simmBucketMapper);
208 break;
209 case SimmVersion::V2_1:
210 return boost::make_shared<SimmConfiguration_ISDA_V2_1>(simmBucketMapper);
211 break;
212 case SimmVersion::V2_2:
213 return boost::make_shared<SimmConfiguration_ISDA_V2_2>(simmBucketMapper, mporDays);
214 break;
215 case SimmVersion::V2_3:
216 return boost::make_shared<SimmConfiguration_ISDA_V2_3>(simmBucketMapper, mporDays);
217 break;
218 case SimmVersion::V2_3_8:
219 return boost::make_shared<SimmConfiguration_ISDA_V2_3_8>(simmBucketMapper, mporDays);
220 break;
221 case SimmVersion::V2_5:
222 return boost::make_shared<SimmConfiguration_ISDA_V2_5>(simmBucketMapper, mporDays);
223 break;
224 case SimmVersion::V2_5A:
225 return boost::make_shared<SimmConfiguration_ISDA_V2_5A>(simmBucketMapper, mporDays);
226 break;
227 case SimmVersion::V2_6:
228 return boost::make_shared<SimmConfiguration_ISDA_V2_6>(simmBucketMapper, mporDays);
229 break;
230 default:
231 break;
232 }
233
234 QL_FAIL("SIMM configuration for version '" << simmVersion << "' cannot be built");
235}

237std::string escapeCommaSeparatedList(const std::string& str, const char& csvQuoteChar) {
238 std::string result = str;
239 if (result.find(',') != string::npos && csvQuoteChar == '\0') {
240 result = '\"' + result + '\"';
241 }
242 return result;
243}
