48 struct ShiftData {
49 virtual ~ShiftData() {}
50 ShiftData() : shiftSize(0.0) {}
51 string shiftType;
52 Real shiftSize;
53 };
54
55 struct CurveShiftData : ShiftData {
56 virtual ~CurveShiftData() {}
57 vector<Period> shiftTenors;
58 };
59
60 struct SpotShiftData : ShiftData {};
61
62 struct CdsVolShiftData : ShiftData {
63 string ccy;
64 vector<Period> shiftExpiries;
65 };
66
67 struct BaseCorrelationShiftData : ShiftData {

70 string indexName;
71 };
72
73 struct VolShiftData : ShiftData {
74 VolShiftData() : shiftStrikes({0.0}), isRelative(false) {}
75 vector<Period> shiftExpiries;
76 vector<Real> shiftStrikes;
77 bool isRelative;
78 };
79
80 struct CapFloorVolShiftData : VolShiftData {
81 string indexName;
82 };
83
84 struct GenericYieldVolShiftData : VolShiftData {
85 vector<Period> shiftTerms;
86 };
87
88 struct CurveShiftParData : SensitivityScenarioData::CurveShiftData {
89 CurveShiftParData(SensitivityScenarioData::CurveShiftData c) {
90 shiftType = c.shiftType;
91 shiftSize = c.shiftSize;
92 shiftTenors = c.shiftTenors;
93 }
94 CurveShiftParData() {}
95 vector<string> parInstruments;
96 bool parInstrumentSingleCurve = true;
97
98 // Allows direct specification of discount curve if parInstrumentSingleCurve is false
99 // If not given, we default to old behaviour of using the market's discount curve for
100 // that currency. This string will be an index name that is searched for in the market.
101 std::string discountCurve;
102
103 // Allow specification of the other currency when creating cross currency basis swap
104 // par instruments for discount curves. If left empty, we take the base currency of the
105 // run as the other leg's currency.
106 std::string otherCurrency;
107
108 map<string, string> parInstrumentConventions;
109 };
110
111 struct CapFloorVolShiftParData : SensitivityScenarioData::CapFloorVolShiftData {
112 CapFloorVolShiftParData(SensitivityScenarioData::CapFloorVolShiftData c) {
113 shiftType = c.shiftType;

115 shiftExpiries = c.shiftExpiries;
116 shiftStrikes = c.shiftStrikes;
117 indexName = c.indexName;
118 }
119 CapFloorVolShiftParData() {}
120 vector<string> parInstruments;
121 bool parInstrumentSingleCurve = true;
122
123 // Allows direct specification of discount curve

125 // that currency. This string will be an index name that is searched for in the market.
