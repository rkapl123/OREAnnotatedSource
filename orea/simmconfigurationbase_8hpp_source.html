76 /*! Return the SIMM <em>risk weight</em> for the given risk type \p rt with
77 the given \p qualifier and the given \p label_1. Three possibilities:
78 -# there is a flat risk weight for the risk factor's RiskType so only need \p rt
79 -# there is a qualifier-dependent risk weight for the risk factor's RiskType
80 so need \p rt and \p qualifier
81 -# there is a qualifier-dependent and label1-dependent risk weight for the risk
82 factor's RiskType so need all three parameters
83 */
84 QuantLib::Real weight(const RiskType& rt, boost::optional<std::string> qualifier = boost::none,
85 boost::optional<std::string> label_1 = boost::none,
86 const std::string& calculationCurrency = "") const override;
87
88 /*! Gives back the value of the scaling function used in the calculation of curvature risk
89 for the risk type \p rt with SIMM <em>Label1</em> value \p label_1. The scaling function is:
90 \f[
91 SF(t) = 0.5 \times \min \left(1, \frac{14}{t} \right)
92 \f]
93 where \f$t\f$ is given in days.
94
95 \warning An error is thrown if there is no curvature from the risk type \p rt
96 */
97 QuantLib::Real curvatureWeight(const RiskType& rt, const std::string& label_1) const override;
98
99 /*! Give back the SIMM <em>historical volatility ratio</em> for the risk type \p rt
100
101 \remark 1.0 is returned if there is no historical volatility ratio for \p rt
102 */
103 QuantLib::Real historicalVolatilityRatio(const RiskType& rt) const override;
104
105 /*! Give back the value of \f$\sigma_{kj}\f$ from the SIMM docs for risk type \p rt.
106 In general, \p rt is a volatility risk type and the method returns:
107 \f[
108 \sigma_{kj} = \frac{RW_k \sqrt{\frac{365}{14}}}{\Phi^{-1}(0.99)}
109 \f]
110 where \f$RW_k\f$ is the corresponding delta risk weight and \f$\Phi(z)\f$ is
111 the cumulative standard normal distribution.
112
113 \remark For convenience, returns 1.0 if not applicable for risk type \p rt
114 */
115 QuantLib::Real sigma(const RiskType& rt, boost::optional<std::string> qualifier = boost::none,
116 boost::optional<std::string> label_1 = boost::none,
117 const std::string& calculationCurrency = "") const override;
118
119 /*! Give back the scaling factor for the Interest Rate curvature margin
120 */
121 QuantLib::Real curvatureMarginScaling() const override { return 2.3; }
122
123 /*! Give back the SIMM <em>concentration threshold</em> for the risk type \p rt and the
124 SIMM \p qualifier
125 */
126 QuantLib::Real concentrationThreshold(const RiskType& rt, const std::string& qualifier) const override {
127 return simmConcentration_->threshold(rt, qualifier);
128 }
129
130 /*! Return true if \p rt is a valid SIMM <em>RiskType</em> under the current configuration.
131 Otherwise, return false.
132 */
133 bool isValidRiskType(const RiskType& rt) const override { return validRiskTypes_.count(rt) > 0; }
134
135 //! Return the correlation between SIMM risk classes \p rc_1 and \p rc_2
136 QuantLib::Real correlationRiskClasses(const RiskClass& rc_1, const RiskClass& rc_2) const override;
137
138 /*! Return the correlation between the \p firstQualifier with risk type \p firstRt,
139 Label1 value of \p firstLabel_1 and Label2 value of \p firstLabel_2 *and* the
140 \p secondQualifier with risk type \p secondRt, Label1 value of \p secondLabel_1
141 and Label2 value of \p secondLabel_2
142
143 \remark if not using \p firstLabel_1 and \p secondLabel_1, just enter an empty
144 string for both. Similarly for \p firstLabel_2 and \p secondLabel_2.
145
146 \warning Returns 0 if no correlation found
147
148 \todo test if the default return value of 0 makes sense
149 */
150 QuantLib::Real correlation(const RiskType& firstRt, const std::string& firstQualifier,
151 const std::string& firstLabel_1, const std::string& firstLabel_2,
152 const RiskType& secondRt, const std::string& secondQualifier,
153 const std::string& secondLabel_1, const std::string& secondLabel_2,
154 const std::string& calculationCurrency = "") const override;
155
156 //! MPOR in days
157 QuantLib::Size mporDays() const { return mporDays_; }
158
159private:
160 //! Name of the SIMM configuration
161 std::string name_;
162 //! SIMM configuration version

164 //! Calculate variable for use in sigma method
165 QuantLib::Real sigmaMultiplier() const;
166

170 const std::string version, QuantLib::Size mporDays = 10);

206 std::map<RiskType, QuantLib::Real> rwRiskType_;
207 std::map<RiskType, std::vector<QuantLib::Real>> rwBucket_;
208 std::map<std::pair<RiskType, std::string>, std::vector<QuantLib::Real>> rwLabel_1_;
209

214 std::map<RiskType, std::vector<QuantLib::Real>> curvatureWeights_;
