626boost::shared_ptr<Portfolio> buildPortfolio(Size portfolioSize, bool swapsOnly = false,
627 boost::shared_ptr<EngineFactory> factory = {}) {
628
629 boost::shared_ptr<Portfolio> portfolio(new Portfolio());
630
631 vector<string> ccys = {"EUR", "USD", "GBP", "JPY", "CHF"};
632
633 map<string, vector<string>> indices = {{"EUR", {"EUR-EURIBOR-6M"}},
634 {"USD", {"USD-LIBOR-3M"}},
635 {"GBP", {"GBP-LIBOR-6M"}},
636 {"CHF", {"CHF-LIBOR-6M"}},
637 {"JPY", {"JPY-LIBOR-6M"}}};
638
639 vector<string> fixedTenors = {"6M", "1Y"};
640
641 Size minStart = 0;
642 Size maxStart = 5;
643 Size minTerm = 2;
644 Size maxTerm = 30;
645
646 Size minFixedBps = 10;
647 Size maxFixedBps = 400;
648
649 Size seed = 5; // keep this constant to ensure portfolio doesn't change
650 MersenneTwisterUniformRng rng(seed);
651
652 Calendar cal = TARGET();
653 string fixDC = "30/360";
654 string floatDC = "ACT/365";
655
656 Real notional = 1000000;
657 Real spread = 0.0;
658
659 for (Size i = 0; i < portfolioSize; i++) {
660
661 // ccy + index
662 string ccy = portfolioSize == 1 ? "EUR" : randString(rng, ccys);
663 string index = portfolioSize == 1 ? "EUR-EURIBOR-6M" : randString(rng, indices[ccy]);
664 string floatFreq = portfolioSize == 1 ? "6M" : index.substr(index.find('-', 4) + 1);
665
666 // fixed details
667 Real fixedRate = portfolioSize == 1 ? 0.02 : randInt(rng, minFixedBps, maxFixedBps) / 100.0;
668 string fixFreq = portfolioSize == 1 ? "1Y" : randString(rng, fixedTenors);
669
670 bool isPayer = randBoolean(rng);
671
672 // id
673 std::ostringstream oss;
674 oss.clear();
675 oss.str("");
676 oss << "Trade_" << i + 1;
677 string id = oss.str();
678
679 if (i % 2 == 0 && !swapsOnly) {
680 int start = randInt(rng, minTerm, maxTerm);
681 Size term = portfolioSize == 1 ? 20 : randInt(rng, minTerm, maxTerm);
682 string longShort = randBoolean(rng) ? "Long" : "Short";
683 portfolio->add(testsuite::buildEuropeanSwaption(id, longShort, ccy, isPayer, notional,
684 start, term, fixedRate, spread, fixFreq,
685 fixDC, floatFreq, floatDC, index));
686 } else {
687 int start = randInt(rng, minStart, maxStart);
688 Size end = randInt(rng, minTerm, maxTerm);
689 portfolio->add(testsuite::buildSwap(id, ccy, isPayer, notional, start, end, fixedRate,
690 spread, fixFreq, fixDC, floatFreq, floatDC, index));
691 }
692 }
693 if (factory)
694 portfolio->build(factory);
695
696 BOOST_CHECK_MESSAGE(portfolio->size() == portfolioSize,
697 "Failed to build portfolio (got " << portfolio->size() << " expected " << portfolioSize << ")");
698
699 return portfolio;
700}
