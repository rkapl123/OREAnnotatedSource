39//! Abstract base class defining the interface for a SIMM configuration

44 //! Enum indicating the relevant side of the SIMM calculation

47 /*! Risk class types in SIMM plus an All type for convenience
48
49 \warning The ordering here matters. It is used in indexing in to
50 correlation matrices for the correlation between risk classes.
51
52 \warning Internal methods rely on the last element being 'All'
53 */
54 enum class RiskClass { InterestRate, CreditQualifying, CreditNonQualifying, Equity, Commodity, FX, All };

56 /*! Risk types plus an All type for convenience
57 Internal methods rely on the last element being 'All'

151 //! Define ordering for ProductClass according to a waterfall:
152 // Empty < RatesFX < Equity < Commodity < Credit
153 // All is unhandled
154 // Would use operator< but there is a bug in the visual studio compiler.
155 static bool less_than(const ProductClass& lhs, const ProductClass& rhs);
156 static bool greater_than(const ProductClass& lhs, const ProductClass& rhs);
157 static bool less_than_or_equal_to(const ProductClass& lhs, const ProductClass& rhs);
158 static bool greater_than_or_equal_to(const ProductClass& lhs, const ProductClass& rhs);
159
160 //! Return the "worse" ProductClass using a waterfall logic:
161 // RatesFX <
162 static ProductClass maxProductClass(ProductClass pc1, ProductClass pc2) {
163 QL_REQUIRE(pc1 != ProductClass::All && pc2 != ProductClass::All,
164 "Cannot define worse product type if even one of the product classes is indeterminate.");
165 return (less_than(pc1, pc2) ? pc2 : pc1);
166 }

210 /*! Add SIMM <em>Label2</em> values under certain circumstances.
211
212 For example, in v338 and later, CreditQ label2 should have the payment
213 currency if sensitivty is not the result of a securitisation and
214 "payment currency,Sec" if sensitivty is the result of a securitisation.
215 Adding to label2 in the configuration means you do not have to have an
216 exhaustive list up front.
217 */
218 virtual void addLabels2(const RiskType& rt, const std::string& label_2) = 0;
219
220 /*! Return the SIMM <em>risk weight</em> for the given risk type \p rt with
221 the given \p qualifier and the given \p label_1. Three possibilities:
222 -# there is a flat risk weight for the risk factor's RiskType so only need \p rt
223 -# there is a qualifier-dependent risk weight for the risk factor's RiskType
224 so need \p rt and \p qualifier
225 -# there is a qualifier-dependent and label1-dependent risk weight for the risk
226 factor's RiskType so need all three parameters
227 */
228 virtual QuantLib::Real weight(const RiskType& rt, boost::optional<std::string> qualifier = boost::none,
229 boost::optional<std::string> label_1 = boost::none,
230 const std::string& calculationCurrency = "") const = 0;
231
232 /*! Gives back the value of the scaling function used in the calculation of curvature risk
233 for the risk type \p rt with SIMM <em>Label1</em> value \p label_1. The scaling function is:
234 \f[
235 SF(t) = 0.5 \times \min \left(1, \frac{14}{t} \right)
236 \f]
237 where \f$t\f$ is given in days.
238 */
239 virtual QuantLib::Real curvatureWeight(const RiskType& rt, const std::string& label_1) const = 0;
240
241 /*! Give back the SIMM <em>historical volatility ratio</em> for the risk type \p rt
242 */
243 virtual QuantLib::Real historicalVolatilityRatio(const RiskType& rt) const = 0;
244
245 /*! Give back the value of \f$\sigma_{kj}\f$ from the SIMM docs for risk type \p rt.
246 In general, \p rt is a volatility risk type and the method returns:
247 \f[
248 \sigma_{kj} = \frac{RW_k \sqrt{\frac{365}{14}}}{\Phi^{-1}(0.99)}
249 \f]
250 where \f$RW_k\f$ is the corresponding delta risk weight and \f$\Phi(z)\f$ is
251 the cumulative standard normal distribution.
252
253 \remark For convenience, returns 1.0 if not applicable for risk type \p rt
254 */
255 virtual QuantLib::Real sigma(const RiskType& rt, boost::optional<std::string> qualifier = boost::none,
256 boost::optional<std::string> label_1 = boost::none,
257 const std::string& calculationCurrency = "") const = 0;
258
259 /*! Give back the scaling factor for the Interest Rate curvature margin
260 */
261 virtual QuantLib::Real curvatureMarginScaling() const = 0;
262
263 /*! Give back the SIMM <em>concentration threshold</em> for the risk type \p rt and the
264 SIMM \p qualifier

266 virtual QuantLib::Real concentrationThreshold(const RiskType& rt, const std::string& qualifier) const = 0;

268 /*! Return true if \p rt is a valid SIMM <em>RiskType</em> under the current configuration.

271 virtual bool isValidRiskType(const RiskType& rt) const = 0;

273 //! Return the correlation between SIMM risk classes \p rc_1 and \p rc_2
274 virtual QuantLib::Real correlationRiskClasses(const RiskClass& rc_1, const RiskClass& rc_2) const = 0;
275
276 /*! Return the correlation between the \p firstQualifier with risk type \p firstRt,
277 Label1 value of \p firstLabel_1 and Label2 value of \p firstLabel_2 *and* the
278 \p secondQualifier with risk type \p secondRt, Label1 value of \p secondLabel_1
279 and Label2 value of \p secondLabel_2

288 const std::string& secondLabel_1, const std::string& secondLabel_2,
289 const std::string& calculationCurrency = "") const = 0;
