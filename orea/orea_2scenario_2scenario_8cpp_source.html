91std::ostream& operator<<(std::ostream& out, const RiskFactorKey& key) {
92 // If empty key just return empty string (not "?//0")
93 if (key == RiskFactorKey()) {
94 return out << "";
95 }
96
97 string keyStr = key.name;
98 size_t index = 0;
99 while (true) {
100 /* Locate the substring to replace. */
101 index = keyStr.find("/", index);
102 if (index == std::string::npos)
103 break;
104
105 keyStr.replace(index, 1, "\\/");
106 index += 2;
107 }
108
109 // If not empty key
110 return out << key.keytype << "/" << keyStr << "/" << key.index;
111}

168RiskFactorKey parseRiskFactorKey(const string& str) {
169 boost::escaped_list_separator<char> sep('\\', '/', '\"');
170 boost::tokenizer<boost::escaped_list_separator<char> > tokenSplit(str, sep);
171 vector<string> tokens(tokenSplit.begin(), tokenSplit.end());
172
173 QL_REQUIRE(tokens.size() == 3, "Could not parse key " << str);
174 RiskFactorKey rfk(parseRiskFactorKeyType(tokens[0]), tokens[1], parseInteger(tokens[2]));
175 return rfk;
176}
