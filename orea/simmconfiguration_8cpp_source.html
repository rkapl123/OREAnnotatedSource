44// Following bimaps ease the conversion from enum to string and back
45// It also puts in one place a mapping from string to enum value

47// custom comparator for bimaps
48struct string_cmp {
49 bool operator()(const string& lhs, const string& rhs) const {
50 return ((boost::to_lower_copy(lhs)) < (boost::to_lower_copy(rhs)));
51 }
52};

333set<SimmConfiguration::RiskClass> SimmConfiguration::riskClasses(bool includeAll) {
334
335 // This only works if 'All' is the last enum value
336 Size bound = includeAll ? numberOfRiskClasses : numberOfRiskClasses - 1;
337
338 // Return the set of values
339 set<RiskClass> result;
340 for (Size i = 0; i < bound; ++i) {
341 result.insert(RiskClass(i));
342 }
343 return result;
344}

439SimmConfiguration::SimmSide parseSimmSide(const string& side) {
440 if (side == "Call") {
441 return SimmConfiguration::SimmSide::Call;
442 } else if (side == "Post") {
443 return SimmConfiguration::SimmSide::Post;
444 } else {
445 QL_FAIL("Could not parse the string '" << side << "' to a SimmSide");
446 }
447}

499set<string> parseRegulationString(const string& regsString, const set<string>& valueIfEmpty) {
500 set<string> uniqueRegNames;
501
502 // "," is a delimiter; "[" and "]" are possible characters, but are not needed in processing
503 vector<string> regNames;
504 boost::split(regNames, regsString, boost::is_any_of(",[] "));
505
506 // Remove any resultant blank strings
507 regNames.erase(std::remove(regNames.begin(), regNames.end(), ""), regNames.end());
508
509 // One last trim for good measure
510 for (string& r : regNames)
511 boost::trim(r);
512
513 // Sort the elements to prevent different permutations of the same regulations being treated differently
514 // e.g. "APRA,USPR" and "USPR,APRA" should ultimately be treated as the same.
515 std::sort(regNames.begin(), regNames.end());
516
517 uniqueRegNames = set<string>(regNames.begin(), regNames.end());
518
519 // If no (valid) regulations provided, we consider the regulation to be unspecified
520 if (uniqueRegNames.empty())
521 return valueIfEmpty;
522 else
523 return uniqueRegNames;
524}

527 // This method will sort the regulations
528 set<string> uniqueRegNames = parseRegulationString(regsString);
529
530 if (uniqueRegNames.size() == 0 ||
531 (uniqueRegNames.size() == 1 && uniqueRegNames.count("Unspecified") > 0)) {
532 return string();
533 } else {
534 return boost::algorithm::join(uniqueRegNames, ",");
535 }

538string removeRegulations(const string& regsString, const vector<string>& regsToRemove) {
539 set<string> uniqueRegNames = parseRegulationString(regsString);
540
541 for (const string& r : regsToRemove) {
542 auto it = uniqueRegNames.find(r);
543 if (it != uniqueRegNames.end())
544 uniqueRegNames.erase(it);
545 }
546
547 if (!uniqueRegNames.empty())
548 return boost::algorithm::join(uniqueRegNames, ",");
549 else
550 return string();
551}

553string filterRegulations(const string& regsString, const vector<string>& regsToFilter) {
554 set<string> uniqueRegNames = parseRegulationString(regsString);
555 set<string> filteredRegs;
556
557 for (const string& r : uniqueRegNames) {
558 auto it = std::find(regsToFilter.begin(), regsToFilter.end(), r);
559 if (it != regsToFilter.end())
560 filteredRegs.insert(*it);
561 }
562
563 if (!filteredRegs.empty())
564 return boost::algorithm::join(filteredRegs, ",");
565 else
566 return string();
567}

569SimmConfiguration::Regulation getWinningRegulation(const std::vector<string>& winningRegulations) {
570
571 vector<SimmConfiguration::Regulation> mappedRegulations;
572
573 for (const string& reg : winningRegulations)
574 mappedRegulations.push_back(parseRegulation(reg));
575
576 SimmConfiguration::Regulation winningRegulation = mappedRegulations.front();
577 for (const auto& reg : mappedRegulations) {
578 if (reg < winningRegulation)
579 winningRegulation = reg;
580 }
581
582 return winningRegulation;
583}

585//! Define ordering for ProductClass
586bool SimmConfiguration::less_than(const SimmConfiguration::ProductClass& lhs,
587 const SimmConfiguration::ProductClass& rhs) {
588 QL_REQUIRE(lhs != SimmConfiguration::ProductClass::All && rhs != SimmConfiguration::ProductClass::All,
589 "Cannot compare the \"All\" ProductClass.");
590 QL_REQUIRE(static_cast<int>(SimmConfiguration::ProductClass::All) == 10,
591 "Number of SIMM Product classes is not 11. Some order comparisons are not handled.");
592
593 // all branches end in returns so no breaks are included.
594 switch (lhs) {
595 case SimmConfiguration::ProductClass::AddOnFixedAmount:
596 case SimmConfiguration::ProductClass::AddOnNotionalFactor:
597 case SimmConfiguration::ProductClass::Empty:
598 case SimmConfiguration::ProductClass::Other:
599 switch (rhs) {
600 case SimmConfiguration::ProductClass::AddOnFixedAmount:
601 case SimmConfiguration::ProductClass::AddOnNotionalFactor:
602 case SimmConfiguration::ProductClass::Empty:
603 case SimmConfiguration::ProductClass::Other:
604 return false;
605 default:
606 return true;
607 }
608 case SimmConfiguration::ProductClass::RatesFX:
609 case SimmConfiguration::ProductClass::Rates:
610 case SimmConfiguration::ProductClass::FX:
611 switch (rhs) {
612 case SimmConfiguration::ProductClass::Empty:
613 case SimmConfiguration::ProductClass::Other:
614 case SimmConfiguration::ProductClass::RatesFX:
615 case SimmConfiguration::ProductClass::Rates:
616 case SimmConfiguration::ProductClass::FX:
617 return false;
618 default:
619 return true;
620 }
621 case SimmConfiguration::ProductClass::Equity:
622 switch (rhs) {
623 case SimmConfiguration::ProductClass::Empty:
624 case SimmConfiguration::ProductClass::Other:
625 case SimmConfiguration::ProductClass::RatesFX:
626 case SimmConfiguration::ProductClass::Rates:
627 case SimmConfiguration::ProductClass::FX:
628 case SimmConfiguration::ProductClass::Equity:
629 return false;
630 default:
631 return true;
632 }
633 case SimmConfiguration::ProductClass::Commodity:
634 if (rhs != SimmConfiguration::ProductClass::Credit) {
635 return false;
636 } else {
637 return true;
638 }
639 case SimmConfiguration::ProductClass::Credit:
640 return false;
641 case SimmConfiguration::ProductClass::All:
642 // not handled, fall through to failure
643 break;
644 }
645 QL_FAIL("Unhandled SIMM Product class in waterfall logic.");
646}
