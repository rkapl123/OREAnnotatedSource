62string periodToLabels2(const QuantLib::Period& p) {
63 if ((p.units() == Months && p.length() == 3) || (p.units() == Weeks && p.length() == 13)) {
64 return "Libor3m";
65 } else if ((p.units() == Months && p.length() == 6) || (p.units() == Weeks && p.length() == 26)) {
66 return "Libor6m";
67 } else if ((p.units() == Days && p.length() == 1) || p == 1 * Weeks) {
68 // 7 days here is based on ISDA SIMM FAQ and Implementation Questions, Sep 4, 2019 Section E.9
69 // Sub curve to be used for CNY seven-day repo rate (closest is OIS).
70 return "OIS";
71 } else if ((p.units() == Months && p.length() == 1) || (p.units() == Weeks && p.length() == 2) ||
72 (p.units() == Weeks && p.length() == 4) || (p.units() == Days && p.length() >= 28 && p.length() <= 31)) {
73 // 2 weeks here is based on ISDA SIMM Methodology paragraph 14:
74 // "Any sub curve not given on the above list should be mapped to its closest equivalent."
75 // A 2 week rate is more like sub-period than OIS.
76 return "Libor1m";
77 } else if ((p.units() == Months && p.length() == 12) || (p.units() == Years && p.length() == 1) ||
78 (p.units() == Weeks && p.length() == 52)) {
79 return "Libor12m";
80 } else {
81 return "";
82 }
83}

126 // There was no municipal until later so override this in

141QuantLib::Real SimmConfigurationBase::weight(const RiskType& rt, boost::optional<string> qualifier,
142 boost::optional<std::string> label_1, const std::string&) const {
143
144 QL_REQUIRE(isValidRiskType(rt),
145 "The risk type " << rt << " is not valid for SIMM configuration with name" << name_);
146
147 // If the risk type has flat risk weights, ignore last 2 parameters
148 if (rwRiskType_.count(rt) > 0) {
149 return rwRiskType_.at(rt);
150 }
151
152 // We now at least have bucket dependent risk weights so check qualifier and buckets
153 QL_REQUIRE(qualifier, "Need a valid qualifier to return a risk weight because the risk type "
154 << rt << " has bucket dependent risk weights");
155 QL_REQUIRE(!buckets(rt).empty(), "Could not find any buckets for risk type " << rt);
156 string bucket = simmBucketMapper_->bucket(rt, *qualifier);
157
158 // If risk weight for this risk type is bucket-dependent
159 if (rwBucket_.count(rt) > 0) {
160 auto idx = labelIndex(bucket, buckets(rt));
161
162 return rwBucket_.at(rt)[idx];
163 }
164
165 // If we get to here, risk weight must depend on risk type, bucket and Label1
166 if (rwLabel_1_.count({rt, bucket}) > 0) {
167 QL_REQUIRE(label_1, "Need a valid Label1 value to return a risk weight because the risk type "
168 << rt << " has bucket and Label1 dependent risk weights");
169 QL_REQUIRE(!labels1(rt).empty(), "Could not find any Label1 values for risk type " << rt);
170 auto idx = labelIndex(*label_1, labels1(rt));
171
172 return rwLabel_1_.at({rt, bucket})[idx];
173 }
174
175 // If we get to here, we have failed to get a risk weight
176 QL_FAIL("Could not find a risk weight for (risk type, qualifier, Label1) = (" << rt << "," << qualifier << ","
177 << label_1 << ")");
178}

204Real SimmConfigurationBase::sigma(const RiskType& rt, boost::optional<std::string> qualifier,
205 boost::optional<std::string> label_1, const std::string& calculationCurrency) const {
206
207 Real sigmaMultiplier = SimmConfigurationBase::sigmaMultiplier();
208
209 Real result = 1.0;
210 Real deltaRiskWeight;
211 std::string ccy1, ccy2;
212 switch (rt) {
213 case RiskType::CommodityVol:
214 deltaRiskWeight = weight(RiskType::Commodity, qualifier, label_1);
215 result = sigmaMultiplier * deltaRiskWeight;
216 break;
217 case RiskType::EquityVol:
218 deltaRiskWeight = weight(RiskType::Equity, qualifier, label_1);
219 result = sigmaMultiplier * deltaRiskWeight;
220 break;
221 case RiskType::FXVol:
222 /*For FX vega (which depends on a pair of currencies), the risk weight to use here is
223 the FX delta risk weight value. Before ISDA 2.2 this value is common for all currency pairs.
224 For ISDA 2.2 the FX delta is from the sensitivity table, given explicitly
225 in section I of the ISDA document, whose row is the FX volatility group of the first currency and
226 whose column is the FX volatility group of the second currency.
227 */
228 ccy1 = (*qualifier).substr(0, 3);
229 ccy2 = (*qualifier).substr(3, 3);
230 // make sure they parse as currencies
231 ore::data::parseCurrency(ccy1);
232 ore::data::parseCurrency(ccy2);
233 deltaRiskWeight = weight(RiskType::FX, ccy1, label_1, ccy2);
234 result = sigmaMultiplier * deltaRiskWeight;
235 break;
236 default:
237 break;
238 }
239
240 return result;
241}

243QuantLib::Real SimmConfigurationBase::correlation(const RiskType& firstRt, const string& firstQualifier,
244 const string& firstLabel_1, const string& firstLabel_2,
245 const RiskType& secondRt, const string& secondQualifier,
246 const string& secondLabel_1, const string& secondLabel_2,
247 const std::string&) const {
248
249 // First check that we have valid risk types for the configuration in question
250 QL_REQUIRE(isValidRiskType(firstRt),
251 "The risk type " << firstRt << " is not valid for SIMM configuration with name" << name());
252 QL_REQUIRE(isValidRiskType(secondRt),
253 "The risk type " << secondRt << " is not valid for SIMM configuration with name" << name());
254
255 // Deal with trivial case of everything equal
256 if (firstRt == secondRt && firstQualifier == secondQualifier && firstLabel_1 == secondLabel_1 &&
257 firstLabel_2 == secondLabel_2) {
258 return 1.0;
259 }
260
261 // Deal with case of different risk types
262 if ((firstRt != secondRt) && (firstQualifier == secondQualifier)) {
263 if (((firstRt == RiskType::IRCurve || firstRt == RiskType::Inflation) && secondRt == RiskType::XCcyBasis) ||
264 (firstRt == RiskType::XCcyBasis && (secondRt == RiskType::IRCurve || secondRt == RiskType::Inflation))) {
265 // Between xccy basis and any yield or inflation in same currency
266 return xccyCorr_;
267 }
268 if ((firstRt == RiskType::IRCurve && secondRt == RiskType::Inflation) ||
269 (firstRt == RiskType::Inflation && secondRt == RiskType::IRCurve)) {
270 // Between any yield and inflation in same currency
271 return infCorr_;
272 }
273 if ((firstRt == RiskType::IRVol && secondRt == RiskType::InflationVol) ||
274 (firstRt == RiskType::InflationVol && secondRt == RiskType::IRVol)) {
275 // Between any yield volatility and inflation volatility in same currency
276 return infVolCorr_;
277 }
278 }
279
280 // Deal with IRCurve and IRVol correlations
281 if ((firstRt == RiskType::IRCurve && secondRt == RiskType::IRCurve) ||
282 (firstRt == RiskType::IRVol && secondRt == RiskType::IRVol)) {
283
284 // If the qualifiers, i.e. currencies, are the same
285 if (firstQualifier == secondQualifier) {
286 // Label2 level, i.e. sub-curve, correlations
287 if (firstLabel_2 != secondLabel_2) {
288 QL_REQUIRE(
289 firstLabel_1 == "" && secondLabel_1 == "",
290 "When asking for Label2 level correlations, "
291 << "the Label1 level values should both contain the default parameter i.e. empty string");
292 QL_REQUIRE(firstRt != RiskType::IRVol, "There is no correlation at the Label2 level for Risk_IRVol");
293 return irSubCurveCorr_;
294 }
295
296 // Label1 level, i.e. tenor, correlations
297 auto idx_1 = labelIndex(firstLabel_1, labels1(firstRt));
298 auto idx_2 = labelIndex(secondLabel_1, labels1(secondRt));
299 return irTenorCorrelation_[idx_1][idx_2];
300 } else {
301 // If the qualifiers, i.e. currencies, are not the same
302 return irInterCurrencyCorr_;
303 }
304 }
305
306 // Deal with inflation volatility correlations
307 if (firstRt == RiskType::InflationVol && secondRt == RiskType::InflationVol) {
308 return 1.0;
309 }
310
311 // Deal with CreditQ correlations
312 if ((firstRt == RiskType::CreditQ && secondRt == RiskType::CreditQ) ||
313 (firstRt == RiskType::CreditVol && secondRt == RiskType::CreditVol)) {
314
315 // Get the bucket of each qualifier
316 string bucket_1 = simmBucketMapper_->bucket(firstRt, firstQualifier);
317 string bucket_2 = simmBucketMapper_->bucket(secondRt, secondQualifier);
318
319 // Residual is special
320 if (bucket_1 == "Residual" || bucket_2 == "Residual") {
321 if (bucket_1 == bucket_2) {
322 // Both Residual
323 return crqResidualIntraCorr_;
324 } else {
325 // One is a residual bucket and the other is not
326 return 0.0;
327 }
328 }
329
330 // Non-residual
331 if (bucket_1 == bucket_2) {
332 // If same bucket
333 if (firstQualifier != secondQualifier) {
334 // If different qualifier (i.e. here issuer/seniority)
335 return crqDiffIntraCorr_;
336 } else {
337 // If same qualifier (i.e. here issuer/seniority)
338 return crqSameIntraCorr_;
339 }
340 } else {
341 // Get the bucket indices of each qualifier for reading the matrix
342 Integer b_1 = parseInteger(bucket_1) - 1;
343 Integer b_2 = parseInteger(bucket_2) - 1;
344 // If different buckets, return the non-residual inter-bucket correlation
345 return interBucketCorrelation_.at(RiskType::CreditQ)[b_1][b_2];
346 }
347 }
348
349 // Deal with CreditNonQ correlations
350 if ((firstRt == RiskType::CreditNonQ && secondRt == RiskType::CreditNonQ) ||
351 (firstRt == RiskType::CreditVolNonQ && secondRt == RiskType::CreditVolNonQ)) {
352
353 // Get the bucket of each qualifier
354 string bucket_1 = simmBucketMapper_->bucket(firstRt, firstQualifier);
355 string bucket_2 = simmBucketMapper_->bucket(secondRt, secondQualifier);
356
357 // Residual is special
358 if (bucket_1 == "Residual" || bucket_2 == "Residual") {
359 if (bucket_1 == bucket_2) {
360 // Both Residual
361 return crnqResidualIntraCorr_;
362 } else {
363 // One is a residual bucket and the other is not
364 return 0.0;
365 }
366 }
367
368 // Non-residual
369 if (bucket_1 == bucket_2) {
370 SimmVersion version = parseSimmVersion(version_);
371 SimmVersion thresholdVersion = SimmVersion::V2_2;
372 if (version >= thresholdVersion) {
373 // In ISDA SIMM version 2.2 or greater, the CRNQ correlations differ depending on whether or not
374 // the entities have the same group name i.e. CMBX.
375 return firstLabel_2 == secondLabel_2 ? crnqSameIntraCorr_ : crnqDiffIntraCorr_;
376 }
377 // TODO:
378 // If same bucket. For ISDA SIMM < 2.2 there is a section in the documentation where
379 // you choose between a correlation if the underlying names are the same and another
380 // correlation if the underlying names are different. The underlying names being the
381 // same is defined in terms of an overlap of 80% in notional terms in underlying names.
382 // Don't know how to pass this down yet so we just go on qualifiers here.
383 return firstQualifier == secondQualifier ? crnqSameIntraCorr_ : crnqDiffIntraCorr_;
384 } else {
385 // If different buckets, return the inter-bucket correlation
386 return crnqInterCorr_;
387 }
388 }
389
390 // Deal with Equity correlations
391 if ((firstRt == RiskType::Equity && secondRt == RiskType::Equity) ||
392 (firstRt == RiskType::EquityVol && secondRt == RiskType::EquityVol)) {
393
394 // Get the bucket of each qualifier
395 string bucket_1 = simmBucketMapper_->bucket(firstRt, firstQualifier);
396 string bucket_2 = simmBucketMapper_->bucket(secondRt, secondQualifier);
397
398 // Residual is special, 0 correlation inter and intra except if same qualifier
399 if (bucket_1 == "Residual" || bucket_2 == "Residual") {
400 return firstQualifier == secondQualifier ? 1.0 : 0.0;
401 }
402
403 // Non-residual
404 // Get the bucket index of each qualifier
405 Integer b_1 = parseInteger(bucket_1) - 1;
406 Integer b_2 = parseInteger(bucket_2) - 1;
407
408 if (b_1 == b_2) {
409 // If same bucket, return the intra-bucket correlation
410 return firstQualifier == secondQualifier ? 1.0 : intraBucketCorrelation_.at(RiskType::Equity)[b_1];
411 } else {
412 // If different buckets, return the inter-bucket correlation
413 return interBucketCorrelation_.at(RiskType::Equity)[b_1][b_2];
414 }
415 }
416
417 // Deal with Commodity correlations
418 if ((firstRt == RiskType::Commodity && secondRt == RiskType::Commodity) ||
419 (firstRt == RiskType::CommodityVol && secondRt == RiskType::CommodityVol)) {
420
421 // Get the bucket index of each qualifier
422 Integer b_1 = parseInteger(simmBucketMapper_->bucket(firstRt, firstQualifier)) - 1;
423 Integer b_2 = parseInteger(simmBucketMapper_->bucket(secondRt, secondQualifier)) - 1;
424
425 if (b_1 == b_2) {
426 // If same bucket, return the intra-bucket correlation
427 return firstQualifier == secondQualifier ? 1.0 : intraBucketCorrelation_.at(RiskType::Commodity)[b_1];
428 } else {
429 // If different buckets, return the inter-bucket correlation
430 return interBucketCorrelation_.at(RiskType::Commodity)[b_1][b_2];
431 }
432 }
433
434 // Deal with FX correlations
435 // TODO:
436 // For FXVol, qualifier is a currency pair. Is it possible to get here
437 // and to have something like secondQualifier = USDJPY and firstQualifier =
438 // JPYUSD an to give back 0.5
439 if ((firstRt == RiskType::FX && secondRt == RiskType::FX) ||
440 (firstRt == RiskType::FXVol && secondRt == RiskType::FXVol)) {
441 return firstQualifier == secondQualifier ? 1.0 : fxCorr_;
442 }
443
444 // Both risk types Base correlation
445 if (firstRt == RiskType::BaseCorr && secondRt == RiskType::BaseCorr) {
446 return basecorrCorr_;
447 }
448
449 // If we get to here
450 return 0.0;
451}

453Real SimmConfigurationBase::sigmaMultiplier() const {
454 // return 2.19486471232815;
455 // Use boost inverse normal here as opposed to QL. Using QL inverse normal
456 // will cause the ISDA SIMM unit tests to fail
457
458 /* We write sqrt(365.0 / (1.4 * mporDays_)) so that this function is
459 sqrt(365.0 / 14) for MPOR = 10 as expected and sqrt(365.0 / 1.4) for MPOR = 1.
460 This is described in SIMM:Technical Paper (Version 10), Section I.2 */
461 return sqrt(365.0 / (1.4 * mporDays_)) / boost::math::quantile(boost::math::normal(), 0.99);
462}

464QuantLib::Real SimmConfigurationBase::correlationRiskClasses(const RiskClass& rc_1, const RiskClass& rc_2) const {
465
466 // Get the row and column index in the matrix of the risk classes
467 Size index_1 = static_cast<Size>(rc_1);
468 QL_REQUIRE(index_1 < riskClassCorrelation_.rows(),
469 "The risk class " << rc_1 << "(" << index_1
470 << ") is not consistent with the number of rows in the correlation matrix ("
471 << riskClassCorrelation_.rows() << ")");
472 Size index_2 = static_cast<Size>(rc_2);
473 QL_REQUIRE(index_2 < riskClassCorrelation_.columns(),
474 "The risk class " << rc_2 << "(" << index_2
475 << ") is not consistent with the number of columns in the correlation matrix ("
476 << riskClassCorrelation_.columns() << ")");
477
478 return riskClassCorrelation_(index_1, index_2);
479}

488void SimmConfigurationBase::addLabels2Impl(const RiskType& rt, const string& label_2) {
489 // Only currently need this for risk type CreditQ
490 QL_REQUIRE(rt == RiskType::CreditQ, "addLabels2 only supported for RiskType_CreditQ");

502 }

504 QL_FAIL("Label2 passed to addLabels2 is unusable for RiskType " << rt);
