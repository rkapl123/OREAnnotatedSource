38//! Delta Scenario class
39/*! This implementation stores a pointer to a "base scenario", as well as a smaller "delta" scenario,
40 which stores values for any keys where the value is different to base.
41 This is intended as an efficient storage mechanism for e.g. sensitivity scenarios,
42 where many scenario instances are managed in memory but the actual differences are minor.
43
44 \ingroup scenario
45*/
46class DeltaScenario : public virtual ore::analytics::Scenario {
47public:
48 //! Constructor
49 DeltaScenario() {}
50 //! Constructor
51 DeltaScenario(const boost::shared_ptr<ore::analytics::Scenario>& baseScenario,
52 const boost::shared_ptr<ore::analytics::Scenario>& incrementalScenario)
53 : baseScenario_(baseScenario), delta_(incrementalScenario) {}
54
55 //! Return the scenario asof date
56 const Date& asof() const override { return delta_->asof(); }
57
58 //! Return the scenario label
59 const std::string& label() const override { return delta_->label(); }
60 //! set the label
61 void label(const string& s) override { delta_->label(s); }
62
63 //! Get Numeraire ratio n = N(t) / N(0) so that Price(0) = N(0) * E [Price(t) / N(t) ]
64 Real getNumeraire() const override;
65 //! Set the Numeraire ratio n = N(t) / N(0) so that Price(0) = N(0) * E [Price(t) / N(t) ]
66 void setNumeraire(Real n) override { delta_->setNumeraire(n); }
67
68 //! Check, get, add a single market point
69 bool has(const ore::analytics::RiskFactorKey& key) const override { return baseScenario_->has(key); }
70 const std::vector<ore::analytics::RiskFactorKey>& keys() const override { return baseScenario_->keys(); }
71 void add(const ore::analytics::RiskFactorKey& key, Real value) override;
72 Real get(const ore::analytics::RiskFactorKey& key) const override;
73
74 //! Get delta
75 boost::shared_ptr<Scenario> delta() const { return delta_; }
76
77 boost::shared_ptr<Scenario> clone() const override;
78
79protected:
80 boost::shared_ptr<Scenario> baseScenario_;
81 boost::shared_ptr<Scenario> delta_;
82
83private:
84 friend class boost::serialization::access;
85 template <class Archive> void serialize(Archive& ar, const unsigned int) {
86 ar& boost::serialization::base_object<Scenario>(*this);
87 ar& baseScenario_;
88 ar& delta_;
89 }
90};
91} // namespace sensitivity
92} // namespace ore
