35std::map<RiskFactorKey, std::string> getScenarioDescriptions(boost::shared_ptr<ScenarioGenerator> scenGen) {
36 std::map<RiskFactorKey, std::string> descriptions;
37 auto sensiScenGen = boost::dynamic_pointer_cast<SensitivityScenarioGenerator>(scenGen);
38 if (sensiScenGen) {
39 for (const auto& desc : sensiScenGen->scenarioDescriptions()) {
40 descriptions[desc.key1()] = desc.indexDesc1();
41 }
42 }
43 return descriptions;
44}

54void ParConversionAnalyticImpl::runAnalytic(const boost::shared_ptr<ore::data::InMemoryLoader>& loader,
55 const std::set<std::string>& runTypes) {
56 if (!analytic()->match(runTypes))
57 return;
58
59 LOG("ParConversionAnalytic::runAnalytic called");
60
61 analytic()->buildMarket(loader, false);
62
63 auto parConversionAnalytic = static_cast<ParConversionAnalytic*>(analytic());
64
65 QL_REQUIRE(parConversionAnalytic, "ParConversionAnalyticImpl internal error, can not convert analytic() to ParConversionAnalytic");
66
67 auto zeroSensis = parConversionAnalytic->loadZeroSensitivities();
68
69 if (!zeroSensis.empty()) {
70 set<RiskFactorKey::KeyType> typesDisabled{RiskFactorKey::KeyType::OptionletVolatility};
71
72 auto parAnalysis = boost::make_shared<ParSensitivityAnalysis>(
73 inputs_->asof(), analytic()->configurations().simMarketParams,
74 *analytic()->configurations().sensiScenarioData, "", true, typesDisabled);
75
76 if (inputs_->parConversionAlignPillars()) {
77 LOG("Sensi analysis - align pillars (for the par conversion or because alignPillars is enabled)");
78 parAnalysis->alignPillars();
79 } else {
80 LOG("Sensi analysis - skip aligning pillars");
81 }
82
83 auto& configs = analytic()->configurations();
84
85 auto simMarket = buildScenarioSimMarketForSensitivityAnalysis(
86 analytic()->market(), configs.simMarketParams, configs.sensiScenarioData, configs.curveConfig,
87 configs.todaysMarketParams, nullptr, inputs_->marketConfig("pricing"), true, false,
88 *inputs_->iborFallbackConfig());
89
90 parAnalysis->computeParInstrumentSensitivities(simMarket);
91
92 boost::shared_ptr<ParSensitivityConverter> parConverter =
93 boost::make_shared<ParSensitivityConverter>(parAnalysis->parSensitivities(), parAnalysis->shiftSizes());
94
95 map<RiskFactorKey, Size> factorToIndex;
96
97 auto shiftSizes = parAnalysis->shiftSizes();
98
99 Size counter = 0;
100 for (auto const& k : parConverter->rawKeys()) {
101 factorToIndex[k] = counter++;
102 }
103
104 std::vector<SensitivityRecord> results;
105 std::map<RiskFactorKey, std::string> descriptions = getScenarioDescriptions(simMarket->scenarioGenerator());
106
107 for (const auto& [id, sensis] : zeroSensis) {
108 boost::numeric::ublas::vector<Real> zeroDeltas(parConverter->rawKeys().size(), 0.0);
109 std::vector<SensitivityRecord> excludedDeltas;
110 bool valid = true;
111 for (const auto& zero : sensis) {
112 if (zero.currency != configs.simMarketParams->baseCcy()) {
113 valid = false;
114 ALOG("Currency in the sensitivity input and config aren't consistent. Skip trade " << id);
115 break;
116 }
117 auto [rf, desc] = deconstructFactor(zero.riskFactor);
118 if (rf.keytype != RiskFactorKey::KeyType::None) {
119 auto it = factorToIndex.find(rf);
120 if (it == factorToIndex.end()) {
121 if (ParSensitivityAnalysis::isParType(rf.keytype) && typesDisabled.count(rf.keytype) != 1) {
122
123 ALOG(StructuredAnalyticsErrorMessage("Par conversion", "",
124 "Par factor " + ore::data::to_string(rf) +
125 " not found in factorToIndex map"));
126 } else {
127 SensitivityRecord sr;
128 sr.tradeId = id;
129 sr.isPar = true;
130 sr.key_1 = rf;
131 sr.desc_1 = desc;
132 sr.delta = zero.delta;
133 sr.baseNpv = zero.baseNpv;
134 sr.currency = zero.currency;
135 sr.shift_1 = zero.shiftSize;
136 sr.gamma = QuantLib::Null<QuantLib::Real>();
137 excludedDeltas.push_back(sr);
138 }
139 } else {
140 auto shiftSize = shiftSizes.find(rf);
141 if (shiftSize == shiftSizes.end() || !close_enough(shiftSize->second.first, zero.shiftSize)) {
142 valid = false;
143 ALOG("Shift sizes in the sensitivity input and config aren't consistent. Skip trade "
144 << id);
145 break;
146 }
147
148 zeroDeltas[it->second] = zero.delta;
149 }
150 }
151 }
152 if (!sensis.empty() && valid) {
153 boost::numeric::ublas::vector<Real> parDeltas = parConverter->convertSensitivity(zeroDeltas);
154 Size counter = 0;
155 for (const auto& key : parConverter->parKeys()) {
156 if (!close(parDeltas[counter], 0.0)) {
157 SensitivityRecord sr;
158 sr.tradeId = id;
159 sr.isPar = true;
160 sr.key_1 = key;
161 sr.desc_1 = descriptions[key];
162 sr.delta = parDeltas[counter];
163 sr.baseNpv = sensis.begin()->baseNpv;
164 sr.currency = sensis.begin()->currency;
165 sr.shift_1 = shiftSizes[key].second;
166 sr.gamma = QuantLib::Null<QuantLib::Real>();
167 results.push_back(sr);
168 }
169 counter++;
170 }
171 results.insert(results.end(), excludedDeltas.begin(), excludedDeltas.end());
172 }
173 }
174
175 auto ss = boost::make_shared<SensitivityInMemoryStream>(results.begin(), results.end());
176 boost::shared_ptr<InMemoryReport> report = boost::make_shared<InMemoryReport>();
177 ReportWriter(inputs_->reportNaString()).writeSensitivityReport(*report, ss, inputs_->parConversionThreshold());
178 analytic()->reports()["PARCONVERSION"]["parConversionSensitivity"] = report;
179
180 if (inputs_->parConversionOutputJacobi()) {
181 boost::shared_ptr<InMemoryReport> jacobiReport = boost::make_shared<InMemoryReport>();
182 writeParConversionMatrix(parAnalysis->parSensitivities(), *jacobiReport);
183 analytic()->reports()["PARCONVERSION"]["parConversionJacobi"] = jacobiReport;
184
185 boost::shared_ptr<InMemoryReport> jacobiInverseReport = boost::make_shared<InMemoryReport>();
186 parConverter->writeConversionMatrix(*jacobiInverseReport);
187 analytic()->reports()["PARCONVERSION"]["parConversionJacobi_inverse"] = jacobiInverseReport;
188 }
189 }
190 LOG("Sensi Analysis - Completed");
191 CONSOLE("OK");
192}
