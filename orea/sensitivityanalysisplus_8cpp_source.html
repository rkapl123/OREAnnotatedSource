36void SensitivityAnalysisPlus::initializeSimMarket(boost::shared_ptr<ore::analytics::ScenarioFactory> scenFact) {
37
38 LOG("Initialise sim market for sensitivity analysis (continueOnError=" << std::boolalpha << continueOnError_
39 << ")");
40 simMarket_ = boost::make_shared<ScenarioSimMarketPlus>(
41 market_, simMarketData_, marketConfiguration_,
42 curveConfigs_ ? *curveConfigs_ : ore::data::CurveConfigurations(),
43 todaysMarketParams_ ? *todaysMarketParams_ : ore::data::TodaysMarketParameters(), continueOnError_,
44 sensitivityData_->useSpreadedTermStructures(), false, false, iborFallbackConfig_);
45 LOG("Sim market initialised for sensitivity analysis");
46
47 LOG("Create scenario factory for sensitivity analysis");
48 boost::shared_ptr<ScenarioFactory> scenarioFactory;
49 if (scenFact) {
50 scenarioFactory = scenFact;
51 } else {
52 scenarioFactory = boost::make_shared<DeltaScenarioFactory>(simMarket_->baseScenario());
53 LOG("DeltaScenario factory created for sensitivity analysis");
54 }
55
56 LOG("Create scenario generator for sensitivity analysis (continueOnError=" << std::boolalpha << continueOnError_
57 << ")");
58 scenarioGenerator_ = boost::make_shared<SensitivityScenarioGenerator>(
59 sensitivityData_, simMarket_->baseScenario(), simMarketData_, simMarket_, scenarioFactory, overrideTenors_,
60 continueOnError_, simMarket_->baseScenarioAbsolute());
61 LOG("Scenario generator created for sensitivity analysis");
62
63 // Set simulation market's scenario generator
64 simMarket_->scenarioGenerator() = scenarioGenerator_;
65}

67void SensitivityAnalysisPlus::initialize(boost::shared_ptr<ore::analytics::NPVSensiCube>& cube) {
68 LOG("Build Sensitivity Scenario Generator and Simulation Market");
69 initializeSimMarket();
70
71 LOG("Build Engine Factory and rebuild portfolio");
72 boost::shared_ptr<EngineFactory> factory = buildFactory();
73 resetPortfolio(factory);
74 if (recalibrateModels_)
75 modelBuilders_ = factory->modelBuilders();
76 else
77 modelBuilders_.clear();
78
79 if (!cube) {
80 LOG("Build the cube object to store sensitivities");
81 initializeCube(cube);
82 }
83
84 sensiCube_ =
85 boost::make_shared<SensitivityCube>(cube, scenarioGenerator_->scenarioDescriptions(),
86 scenarioGenerator_->shiftSizes(), sensitivityData_->twoSidedDeltas());
87
88 initialized_ = true;
89}

102void SensitivityAnalysisPlus::generateSensitivities(boost::shared_ptr<NPVSensiCube> cube) {
103
104 QL_REQUIRE(useSingleThreadedEngine_ || cube == nullptr,
105 "SensitivityAnalysis::generateSensitivities(): when using multi-threaded engine no NPVSensiCube should "
106 "be specified, it is built automatically");
107
108 QL_REQUIRE(useSingleThreadedEngine_ || !nonShiftedBaseCurrencyConversion_,
109 "SensitivityAnalysis::generateSensitivities(): multi-threaded engine does not support non-shifted base "
110 "ccy conversion currently. This requires a a small code extension. Contact Dev.");
111
112 QL_REQUIRE(useSingleThreadedEngine_ || recalibrateModels_,
113 "SensitivityAnalysis::generateSensitivities(): multi-threaded engine does not support recalibrateModels "
114 "= false. This requires a small code extension. Contact Dev.");
115
116 if (useSingleThreadedEngine_) {
117 ore::analytics::SensitivityAnalysis::generateSensitivities(cube);
118 return;
119 }
120
121 // handle request to use multi-threaded engine
122
123 LOG("SensitivitiyAnalysis::generateSensitivities(): use multi-threaded engine to generate sensi cube.");
124
125 market_ =
126 boost::make_shared<ore::data::TodaysMarket>(asof_, todaysMarketParams_, loader_, curveConfigs_, true, true,
127 false, referenceData_, false, iborFallbackConfig_, false);
128
129 simMarket_ = boost::make_shared<ScenarioSimMarketPlus>(
130 market_, simMarketData_, marketConfiguration_,
131 curveConfigs_ ? *curveConfigs_ : ore::data::CurveConfigurations(),
132 todaysMarketParams_ ? *todaysMarketParams_ : ore::data::TodaysMarketParameters(), continueOnError_,
133 sensitivityData_->useSpreadedTermStructures(), false, false, iborFallbackConfig_);
134
135 scenarioGenerator_ = boost::make_shared<SensitivityScenarioGenerator>(
136 sensitivityData_, simMarket_->baseScenario(), simMarketData_, simMarket_,
137 boost::make_shared<DeltaScenarioFactory>(simMarket_->baseScenario()), overrideTenors_, continueOnError_,
138 simMarket_->baseScenarioAbsolute());
139
140 simMarket_->scenarioGenerator() = scenarioGenerator_;
141
142 auto ed = boost::make_shared<EngineData>(*engineData_);
143 ed->globalParameters()["RunType"] =
144 std::string("Sensitivity") + (sensitivityData_->computeGamma() ? "DeltaGamma" : "Delta");
145
146 MultiThreadedValuationEngine engine(
147 nThreads_, asof_, boost::make_shared<ore::analytics::DateGrid>(), scenarioGenerator_->numScenarios(), loader_,
148 scenarioGenerator_, ed, curveConfigs_, todaysMarketParams_, marketConfiguration_, simMarketData_,
149 sensitivityData_->useSpreadedTermStructures(), false, boost::make_shared<ore::analytics::ScenarioFilter>(),
150 referenceData_, iborFallbackConfig_, true, true,
151 [](const QuantLib::Date& asof, const std::set<std::string>& ids, const std::vector<QuantLib::Date>&,
152 const QuantLib::Size samples) {
153 return boost::make_shared<ore::analytics::DoublePrecisionSensiCube>(ids, asof, samples);
154 },
155 {}, {}, context_);
156 for (auto const& i : this->progressIndicators())
157 engine.registerProgressIndicator(i);
158
159 auto baseCcy = simMarketData_->baseCcy();
160 engine.buildCube(
161 portfolio_,
162 [&baseCcy]() -> std::vector<boost::shared_ptr<ValuationCalculator>> {
163 return {boost::make_shared<NPVCalculator>(baseCcy)};
164 },
165 {}, true, dryRun_);
166 std::vector<boost::shared_ptr<NPVSensiCube>> miniCubes;
167 for (auto const& c : engine.outputCubes()) {
168 miniCubes.push_back(boost::dynamic_pointer_cast<NPVSensiCube>(c));
169 QL_REQUIRE(miniCubes.back() != nullptr,
170 "SensitivityAnalysis::generateSensitivities(): internal error, could not cast to NPVSensiCube.");
171 }
172 cube = boost::make_shared<JointNPVSensiCube>(miniCubes, portfolio_->ids());
173
174 sensiCube_ =
175 boost::make_shared<SensitivityCube>(cube, scenarioGenerator_->scenarioDescriptions(),
176 scenarioGenerator_->shiftSizes(), sensitivityData_->twoSidedDeltas());
177
178 initialized_ = true;
179}
