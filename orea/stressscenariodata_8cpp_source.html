34 
35  for (XMLNode* testCase = XMLUtils::getChildNode(node, "StressTest"); testCase;
36  testCase = XMLUtils::getNextSibling(testCase)) {
37 
38  StressTestData test;
39  test.label = XMLUtils::getAttribute(testCase, "id");
40  // XMLUtils::getChildValue(testCase, "Label", true);
41 
42  LOG("Load stress test label " << test.label);
43 
44  LOG("Get recovery rate shift parameters");
45  XMLNode* recoveryRates = XMLUtils::getChildNode(testCase, "RecoveryRates");
46  test.recoveryRateShifts.clear();
47  for (XMLNode* child = XMLUtils::getChildNode(recoveryRates, "RecoveryRate"); child;
48  child = XMLUtils::getNextSibling(child)) {
49  string isin = XMLUtils::getAttribute(child, "id");

55  }

57  LOG("Get survival probability shift parameters");
58  XMLNode* survivalProbability = XMLUtils::getChildNode(testCase, "SurvivalProbabilities");
59  QL_REQUIRE(survivalProbability, "Survival Probabilities node not found");
60  test.survivalProbabilityShifts.clear();
61  for (XMLNode* child = XMLUtils::getChildNode(survivalProbability, "SurvivalProbability"); child;
62  child = XMLUtils::getNextSibling(child)) {
63  string name = XMLUtils::getAttribute(child, "name");
64  LOG("Loading stress parameters for survival probability for " << name);
65  CurveShiftData data;
66  data.shiftType = XMLUtils::getChildValue(child, "ShiftType", true);
67  data.shifts = XMLUtils::getChildrenValuesAsDoublesCompact(child, "Shifts", true);
68  data.shiftTenors = XMLUtils::getChildrenValuesAsPeriods(child, "ShiftTenors", true);
69  QL_REQUIRE(data.shifts.size() == data.shiftTenors.size(),
70  "number of tenors and shifts does not match in survival probability stress data");
71  QL_REQUIRE(data.shifts.size() > 0, "no shifts provided in survival probability stress data");
72  test.survivalProbabilityShifts[name] = data;
73  }
74 
75  LOG("Get discount curve shift parameters");
76  XMLNode* discountCurves = XMLUtils::getChildNode(testCase, "DiscountCurves");
77  QL_REQUIRE(discountCurves, "DiscountCurves node not found");
78  test.discountCurveShifts.clear();
79  for (XMLNode* child = XMLUtils::getChildNode(discountCurves, "DiscountCurve"); child;
80  child = XMLUtils::getNextSibling(child)) {
81  string ccy = XMLUtils::getAttribute(child, "ccy");
82  LOG("Loading stress parameters for discount curve for ccy " << ccy);
83  CurveShiftData data;
84  data.shiftType = XMLUtils::getChildValue(child, "ShiftType", true);
85  data.shifts = XMLUtils::getChildrenValuesAsDoublesCompact(child, "Shifts", true);
86  data.shiftTenors = XMLUtils::getChildrenValuesAsPeriods(child, "ShiftTenors", true);
87  QL_REQUIRE(data.shifts.size() == data.shiftTenors.size(),
88  "number of tenors and shifts does not match in discount curve stress data");

90  test.discountCurveShifts[ccy] = data;
91  }
92 
93  LOG("Get index curve stress parameters");
94  XMLNode* indexCurves = XMLUtils::getChildNode(testCase, "IndexCurves");
95  QL_REQUIRE(indexCurves, "IndexCurves node not found");
96  test.indexCurveShifts.clear();

98  child = XMLUtils::getNextSibling(child)) {
99  string index = XMLUtils::getAttribute(child, "index");

102  CurveShiftData data;
103  data.shiftType = XMLUtils::getChildValue(child, "ShiftType", true);
104  data.shifts = XMLUtils::getChildrenValuesAsDoublesCompact(child, "Shifts", true);

189  LOG("Get swaption vol stress parameters");
190  XMLNode* swaptionVols = XMLUtils::getChildNode(testCase, "SwaptionVolatilities");
191  QL_REQUIRE(swaptionVols, "SwaptionVols node not found");
192  test.swaptionVolShifts.clear();
193  for (XMLNode* child = XMLUtils::getChildNode(swaptionVols, "SwaptionVolatility"); child;
194  child = XMLUtils::getNextSibling(child)) {
195  string ccy = XMLUtils::getAttribute(child, "ccy");
196  LOG("Loading stress parameters for swaption vols " << ccy);
197  SwaptionVolShiftData data;
198  data.shiftType = XMLUtils::getChildValue(child, "ShiftType", true);
199  data.shiftTerms = XMLUtils::getChildrenValuesAsPeriods(child, "ShiftTerms", true);
200  data.shiftExpiries = XMLUtils::getChildrenValuesAsPeriods(child, "ShiftExpiries", true);
201  XMLNode* shiftSizes = XMLUtils::getChildNode(child, "Shifts");
202  data.parallelShiftSize = 0.0;
203  for (XMLNode* child2 = XMLUtils::getChildNode(shiftSizes, "Shift"); child2;
204  child2 = XMLUtils::getNextSibling(child2)) {
205  string expiry = XMLUtils::getAttribute(child2, "expiry");
206  string term = XMLUtils::getAttribute(child2, "term");
207  if (expiry == "" && term == "")
208  data.parallelShiftSize = ore::data::parseReal(XMLUtils::getNodeValue(child2));
209  else {
210  QL_REQUIRE(expiry != "" && term != "", "expiry and term attributes required on shift size nodes");
211  Period e = ore::data::parsePeriod(expiry);
212  Period t = ore::data::parsePeriod(term);
213  Real value = ore::data::parseReal(XMLUtils::getNodeValue(child2));
214  pair<Period, Period> key(e, t);
215  data.shifts[key] = value;
216  }
217  }
218  test.swaptionVolShifts[ccy] = data;
219  }
220 
221  LOG("Get cap/floor vol stress parameters");
222  XMLNode* capVols = XMLUtils::getChildNode(testCase, "CapFloorVolatilities");
223  QL_REQUIRE(capVols, "CapVols node not found");
224  for (XMLNode* child = XMLUtils::getChildNode(capVols, "CapFloorVolatility"); child;
225  child = XMLUtils::getNextSibling(child)) {
226  string key = XMLUtils::getAttribute(child, "key");
227  if(key.empty()) {
228  string ccyAttr = XMLUtils::getAttribute(child, "ccy");
229  if(!ccyAttr.empty()) {
230  key = ccyAttr;
231  ALOG("StressScenarioData: 'ccy' is deprecated as an attribute for CapFloorVolatilities, use 'key' instead.");
232  }
233  }
234  CapFloorVolShiftData data;
235  data.shiftType = XMLUtils::getChildValue(child, "ShiftType", true);
236  data.shiftExpiries = XMLUtils::getChildrenValuesAsPeriods(child, "ShiftExpiries", true);
237  data.shifts = XMLUtils::getChildrenValuesAsDoublesCompact(child, "Shifts", true);
238  test.capVolShifts[key] = data;
239  }
240 
241  LOG("Get Security spread stress parameters");
242  XMLNode* securitySpreads = XMLUtils::getChildNode(testCase, "SecuritySpreads");
243  QL_REQUIRE(securitySpreads, "SecuritySpreads node not found");
244  test.securitySpreadShifts.clear();
245  for (XMLNode* child = XMLUtils::getChildNode(securitySpreads, "SecuritySpread"); child;
246  child = XMLUtils::getNextSibling(child)) {
247  string bond = XMLUtils::getAttribute(child, "security");
248  LOG("Loading stress parameters for Security spreads " << bond);
249  SpotShiftData data;
250  data.shiftType = XMLUtils::getChildValue(child, "ShiftType", true);
251  data.shiftSize = XMLUtils::getChildValueAsDouble(child, "ShiftSize", true);
252  test.securitySpreadShifts[bond] = data;
253  }
254 
255  data_.push_back(test);
256 
257  LOG("Loading stress test label " << test.label << " done");
258  }
259 
260  LOG("Loading stress tests done");
