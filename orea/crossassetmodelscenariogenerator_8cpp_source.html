37 CrossAssetModelScenarioGenerator::CrossAssetModelScenarioGenerator(
38  boost::shared_ptr<QuantExt::CrossAssetModel> model,
39  boost::shared_ptr<QuantExt::MultiPathGeneratorBase> pathGenerator,
40  boost::shared_ptr<ScenarioFactory> scenarioFactory, boost::shared_ptr<ScenarioSimMarketParameters> simMarketConfig,
41  Date today, boost::shared_ptr<ore::analytics::DateGrid> grid, boost::shared_ptr<ore::data::Market> initMarket,
42  const std::string& configuration)
43  : ScenarioPathGenerator(today, grid->dates(), grid->timeGrid()), model_(model), pathGenerator_(pathGenerator),
44  scenarioFactory_(scenarioFactory), simMarketConfig_(simMarketConfig), initMarket_(initMarket),
45  configuration_(configuration) {
46 
47  QL_REQUIRE(initMarket != NULL, "CrossAssetScenarioGenerator: initMarket is null");
48  QL_REQUIRE(timeGrid_.size() == dates_.size() + 1, "date/time grid size mismatch");
49 
50  // TODO, curve tenors might be overwritten by dates in simMarketConfig_, here we just take the tenors
51 
52  // Cache yield curve keys
53  Size n_ccy = model_->components(IR);
54  discountCurveKeys_.reserve(n_ccy * simMarketConfig_->yieldCurveTenors("").size());
55  for (Size j = 0; j < model_->components(IR); j++) {
56  std::string ccy = model_->irlgm1f(j)->currency().code();
57  ten_dsc_.push_back(simMarketConfig_->yieldCurveTenors(ccy));
58  Size n_ten = ten_dsc_.back().size();
59  for (Size k = 0; k < n_ten; k++)
60  discountCurveKeys_.emplace_back(RiskFactorKey::KeyType::DiscountCurve, ccy, k); // j * n_ten + k
61  }
62 
63  // Cache index curve keys
64  Size n_indices = simMarketConfig_->indices().size();
65  indexCurveKeys_.reserve(n_indices * simMarketConfig_->yieldCurveTenors("").size());
66  for (Size j = 0; j < n_indices; ++j) {
67  ten_idx_.push_back(simMarketConfig_->yieldCurveTenors(simMarketConfig_->indices()[j]));
68  Size n_ten = ten_idx_.back().size();
69  for (Size k = 0; k < n_ten; ++k) {
70  indexCurveKeys_.emplace_back(RiskFactorKey::KeyType::IndexCurve, simMarketConfig_->indices()[j], k);
71  }
72  }
73 
74  // Cache yield curve keys
75  Size n_curves = simMarketConfig_->yieldCurveNames().size();
76  yieldCurveKeys_.reserve(n_curves * simMarketConfig_->yieldCurveTenors("").size());
77  for (Size j = 0; j < n_curves; ++j) {
78  ten_yc_.push_back(simMarketConfig_->yieldCurveTenors(simMarketConfig_->yieldCurveNames()[j]));
79  Size n_ten = ten_yc_.back().size();
80  for (Size k = 0; k < n_ten; ++k) {
81  yieldCurveKeys_.emplace_back(RiskFactorKey::KeyType::YieldCurve, simMarketConfig_->yieldCurveNames()[j], k);
82  }
83  }
84 
85  // Cache FX rate keys
86  fxKeys_.reserve(n_ccy - 1);
87  for (Size k = 0; k < n_ccy - 1; k++) {
88  const string& foreign = model_->irlgm1f(k + 1)->currency().code(); // Is this safe?
89  const string& domestic = model_->irlgm1f(0)->currency().code();
90  fxKeys_.emplace_back(RiskFactorKey::KeyType::FXSpot, foreign + domestic); // k
91  }
92 
93  // set up CrossAssetModelImpliedFxVolTermStructures
94  if (simMarketConfig_->simulateFXVols()) {
95  LOG("CrossAssetModel is simulating FX vols");
96  for (Size k = 0; k < simMarketConfig_->fxVolCcyPairs().size(); k++) {
97  // Calculating the index is messy
98  const string& pair = simMarketConfig_->fxVolCcyPairs()[k];
99  LOG("Set up CrossAssetModelImpliedFxVolTermStructures for " << pair);
100  QL_REQUIRE(pair.size() == 6, "Invalid ccypair " << pair);
101  const string& domestic = pair.substr(0, 3);
102  const string& foreign = pair.substr(3);
103  QL_REQUIRE(domestic == model_->irlgm1f(0)->currency().code(), "Only DOM-FOR fx vols supported");
104  Size index = model->ccyIndex(parseCurrency(foreign)); // will throw if foreign not there
105  QL_REQUIRE(index > 0, "Invalid index for ccy " << foreign << " should be > 0");
106  // fxVols_ are indexed by ccyPairs
107  LOG("Pair " << pair << " index " << index);
108  // index - 1 to convert "IR" index into an "FX" index
109  fxVols_.push_back(boost::make_shared<CrossAssetModelImpliedFxVolTermStructure>(model_, index - 1));
110  LOG("Set up CrossAssetModelImpliedFxVolTermStructures for " << pair << " done");
111  }
112  }
113 
114  // Cache EQ rate keys
115  Size n_eq = model_->components(EQ);
116  eqKeys_.reserve(n_eq);
117  for (Size k = 0; k < n_eq; k++) {
118  const string& eqName = model_->eqbs(k)->eqName();
119  eqKeys_.emplace_back(RiskFactorKey::KeyType::EquitySpot, eqName);
120  }
121 
122  // equity vols
123  if (simMarketConfig_->equityVolNames().size() > 0 && simMarketConfig_->simulateEquityVols()) {
124  LOG("CrossAssetModel is simulating EQ vols");
125  for (Size k = 0; k < simMarketConfig_->equityVolNames().size(); k++) {
126  // Calculating the index is messy
127  const string& equityName = simMarketConfig_->equityVolNames()[k];
128  LOG("Set up CrossAssetModelImpliedEqVolTermStructures for " << equityName);
129  Size eqIndex = model->eqIndex(equityName);
130  // eqVols_ are indexed by ccyPairs
131  LOG("EQ Vol Name = " << equityName << ", index = " << eqIndex);
132  // index - 1 to convert "IR" index into an "FX" index
133  eqVols_.push_back(boost::make_shared<CrossAssetModelImpliedEqVolTermStructure>(model_, eqIndex));
134  LOG("Set up CrossAssetModelImpliedEqVolTermStructures for " << equityName << " done");
135  }
136  }
137 
138  // Cache INF rate keys0
139  Size n_inf = model_->components(INF);
140  if (n_inf > 0) {
141  cpiKeys_.reserve(n_inf);
142  for (Size j = 0; j < n_inf; ++j) {
143  cpiKeys_.emplace_back(RiskFactorKey::KeyType::CPIIndex, model->infdk(j)->name());
144  }
145 
146  Size n_zeroinf = simMarketConfig_->zeroInflationIndices().size();
147  if (n_zeroinf > 0) {
148  zeroInflationKeys_.reserve(n_zeroinf * simMarketConfig_->zeroInflationTenors("").size());
149  for (Size j = 0; j < n_zeroinf; ++j) {
150  ten_zinf_.push_back(simMarketConfig_->zeroInflationTenors(simMarketConfig_->zeroInflationIndices()[j]));
151  Size n_ten = ten_zinf_.back().size();
152  for (Size k = 0; k < n_ten; ++k) {
153  zeroInflationKeys_.emplace_back(RiskFactorKey::KeyType::ZeroInflationCurve,
154  simMarketConfig_->zeroInflationIndices()[j], k);
155  }
156  }
157  }
158 
159  Size n_yoyinf = simMarketConfig_->yoyInflationIndices().size();
160  if (n_yoyinf > 0) {
161  yoyInflationKeys_.reserve(n_yoyinf * simMarketConfig_->yoyInflationTenors("").size());
162  for (Size j = 0; j < n_yoyinf; ++j) {
163  ten_yinf_.push_back(simMarketConfig_->yoyInflationTenors(simMarketConfig_->yoyInflationIndices()[j]));
164  Size n_ten = ten_yinf_.back().size();
165  for (Size k = 0; k < n_ten; ++k) {
166  yoyInflationKeys_.emplace_back(RiskFactorKey::KeyType::YoYInflationCurve,
167  simMarketConfig_->yoyInflationIndices()[j], k);
168  }
169  }
170  }

175 std::vector<boost::shared_ptr<Scenario>> CrossAssetModelScenarioGenerator::nextPath() {
176  std::vector<boost::shared_ptr<Scenario>> scenarios(dates_.size());
177  Sample<MultiPath> sample = pathGenerator_->next();
178  Size n_ccy = model_->components(IR);
179  Size n_eq = model_->components(EQ);
180  Size n_inf = model_->components(INF);
181  Size n_indices = simMarketConfig_->indices().size();
182  Size n_curves = simMarketConfig_->yieldCurveNames().size();
183  Size n_zeroinf = simMarketConfig_->zeroInflationIndices().size();
184  Size n_yoyinf = simMarketConfig_->yoyInflationIndices().size();
185  vector<boost::shared_ptr<QuantExt::LgmImpliedYieldTermStructure>> curves, fwdCurves, yieldCurves;
186  vector<boost::shared_ptr<QuantExt::DkImpliedZeroInflationTermStructure>> zeroInfCurves;
187  vector<boost::shared_ptr<QuantExt::DkImpliedYoYInflationTermStructure>> yoyInfCurves;
188  vector<boost::shared_ptr<IborIndex>> indices;
189  vector<Currency> yieldCurveCurrency;
190  vector<boost::shared_ptr<QuantExt::LgmImpliedYieldTermStructure>> equityForecastCurves;
191  vector<Currency> equityForecastCurrency;
192  vector<string> zeroInflationIndex, yoyInflationIndex;
193 
194  DayCounter dc = model_->irlgm1f(0)->termStructure()->dayCounter();
195 
196  for (Size j = 0; j < n_ccy; ++j) {
197  curves.push_back(boost::make_shared<QuantExt::LgmImpliedYieldTermStructure>(model_->lgm(j), dc, true));
198  }
199 
200  for (Size j = 0; j < n_indices; ++j) {
201  std::string indexName = simMarketConfig_->indices()[j];
202  boost::shared_ptr<IborIndex> index = *initMarket_->iborIndex(indexName, configuration_);
203  Handle<YieldTermStructure> fts = index->forwardingTermStructure();
204  auto impliedFwdCurve = boost::make_shared<LgmImpliedYtsFwdFwdCorrected>(
205  model_->lgm(model_->ccyIndex(index->currency())), fts, dc, false);
206  fwdCurves.push_back(impliedFwdCurve);
207  indices.push_back(index->clone(Handle<YieldTermStructure>(impliedFwdCurve)));
208  }
209 
210  for (Size j = 0; j < n_curves; ++j) {
211  std::string curveName = simMarketConfig_->yieldCurveNames()[j];
212  Currency ccy = ore::data::parseCurrency(simMarketConfig_->yieldCurveCurrencies().at(curveName));
213  Handle<YieldTermStructure> yts = initMarket_->yieldCurve(curveName, configuration_);
214  auto impliedYieldCurve =
215  boost::make_shared<LgmImpliedYtsFwdFwdCorrected>(model_->lgm(model_->ccyIndex(ccy)), yts, dc, false);
216  yieldCurves.push_back(impliedYieldCurve);
217  yieldCurveCurrency.push_back(ccy);
218  }
219 
220  for (Size j = 0; j < n_zeroinf; ++j) {
221  zeroInfCurves.push_back(boost::make_shared<QuantExt::DkImpliedZeroInflationTermStructure>(model_, j));
222  }
223 
224  for (Size j = 0; j < n_yoyinf; ++j) {
225  yoyInfCurves.push_back(boost::make_shared<QuantExt::DkImpliedYoYInflationTermStructure>(model_, j));
226  }
227 
228  for (Size j = 0; j < n_eq; ++j) {
229  std::string curveName = simMarketConfig_->equityNames()[j];
230  Currency ccy = model_->eqbs(j)->currency();
231  equityForecastCurrency.push_back(ccy);
232  }
233 
234  for (Size i = 0; i < dates_.size(); i++) {
235  Real t = timeGrid_[i + 1]; // recall: time grid has inserted t=0
236 
237  scenarios[i] = scenarioFactory_->buildScenario(dates_[i]);
238 
239  // Set numeraire, numeraire currency and the (deterministic) domestic discount
240  // Asset index 0 in sample.value[0][i+1] refers to the domestic currency process.
241  Real z0 = sample.value[0][i + 1]; // domestic LGM factor, second index = 0 holds initial values
242  scenarios[i]->setNumeraire(model_->numeraire(0, t, z0));
243 
244  // Discount curves
245  for (Size j = 0; j < n_ccy; j++) {
246  // LGM factor value, second index = 0 holds initial values
247  Real z = sample.value[model_->pIdx(IR, j)][i + 1];
248  curves[j]->move(t, z);
249  for (Size k = 0; k < ten_dsc_[j].size(); k++) {
250  Date d = dates_[i] + ten_dsc_[j][k];
251  Time T = dc.yearFraction(dates_[i], d);
252  Real discount = std::max(curves[j]->discount(T), 0.00001);
253  scenarios[i]->add(discountCurveKeys_[j * ten_dsc_[j].size() + k], discount);
254  }
255  }
256 
257  // Index curves and Index fixings
258  for (Size j = 0; j < n_indices; ++j) {
259  Real z = sample.value[model_->pIdx(IR, model_->ccyIndex(indices[j]->currency()))][i + 1];
260  fwdCurves[j]->move(dates_[i], z);
261  for (Size k = 0; k < ten_idx_[j].size(); ++k) {
262  Date d = dates_[i] + ten_idx_[j][k];
263  Time T = dc.yearFraction(dates_[i], d);
264  Real discount = std::max(fwdCurves[j]->discount(T), 0.00001);
265  scenarios[i]->add(indexCurveKeys_[j * ten_idx_[j].size() + k], discount);
266  }
267  }
268 
269  // Yield curves
270  for (Size j = 0; j < n_curves; ++j) {
271  Real z = sample.value[model_->pIdx(IR, model_->ccyIndex(yieldCurveCurrency[j]))][i + 1];
272  yieldCurves[j]->move(dates_[i], z);
273  for (Size k = 0; k < ten_yc_[j].size(); ++k) {
274  Date d = dates_[i] + ten_yc_[j][k];
275  Time T = dc.yearFraction(dates_[i], d);
276  Real discount = std::max(yieldCurves[j]->discount(T), 0.00001);
277  scenarios[i]->add(yieldCurveKeys_[j * ten_yc_[j].size() + k], discount);
278  }
279  }
280 
281  // FX rates
282  for (Size k = 0; k < n_ccy - 1; k++) {
283  // if (i == 0) { // construct the labels at the first scenario date only
284  // const string& foreign = model_->irlgm1f(k+1)->currency().code(); // Is this safe?
285  // const string& domestic = model_->irlgm1f(0)->currency().code();
286  // ccyPairs[k] = foreign + domestic; // x USDEUR means 1 USD is worth x EUR
287  // }
288  Real fx = std::exp(sample.value[model_->pIdx(FX, k)][i + 1]); // multiplies USD amount to get EUR
289  scenarios[i]->add(fxKeys_[k], fx);
290  }
291 
292  // FX vols
293  if (simMarketConfig_->simulateFXVols()) {
294  const vector<Period>& expires = simMarketConfig_->fxVolExpiries();
295  for (Size k = 0; k < simMarketConfig_->fxVolCcyPairs().size(); k++) {
296  const string& ccyPair = simMarketConfig_->fxVolCcyPairs()[k];
297 
298  Size fxIndex = fxVols_[k]->fxIndex();
299  Real zFor = sample.value[fxIndex + 1][i + 1];
300  Real logFx = sample.value[n_ccy + fxIndex][i + 1]; // multiplies USD amount to get EUR
301  fxVols_[k]->move(dates_[i], z0, zFor, logFx);
302 
303  for (Size j = 0; j < expires.size(); j++) {
304  Real vol = fxVols_[k]->blackVol(dates_[i] + expires[j], Null<Real>(), true);
305  scenarios[i]->add(RiskFactorKey(RiskFactorKey::KeyType::FXVolatility, ccyPair, j), vol);
306  }
307  }
308  }
309 
310  // Equity spots
311  for (Size k = 0; k < n_eq; k++) {
312  Real eqSpot = std::exp(sample.value[model_->pIdx(EQ, k)][i + 1]);
313  scenarios[i]->add(eqKeys_[k], eqSpot);
314  }
315 
316  // Equity vols
317  if (simMarketConfig_->simulateEquityVols()) {
318  const vector<Period>& expiries = simMarketConfig_->equityVolExpiries();
319  for (Size k = 0; k < simMarketConfig_->equityVolNames().size(); k++) {
320  const string& equityName = simMarketConfig_->equityVolNames()[k];
321 
322  Size eqIndex = eqVols_[k]->equityIndex();
323  Size eqCcyIdx = eqVols_[k]->eqCcyIndex();
324  Real z_eqIr = sample.value[eqCcyIdx][i + 1];
325  Real logEq = sample.value[eqIndex][i + 1];
326  eqVols_[k]->move(dates_[i], z_eqIr, logEq);
327 
328  for (Size j = 0; j < expiries.size(); j++) {
329  Real vol = eqVols_[k]->blackVol(dates_[i] + expiries[j], Null<Real>(), true);
330  scenarios[i]->add(RiskFactorKey(RiskFactorKey::KeyType::EquityVolatility, equityName, j), vol);
331  }
332  }
333  }
334 
335  // Inflation curves
336  for (Size j = 0; j < n_inf; j++) {
337  // LGM factor value, second index = 0 holds initial values
338  Real z = sample.value[model_->pIdx(INF, j, 0)][i + 1];
339  Real y = sample.value[model_->pIdx(INF, j, 1)][i + 1];
340 
341  // update fixing manage with fixing for base date
342  boost::shared_ptr<ZeroInflationIndex> index = *initMarket_->zeroInflationIndex(model_->infdk(j)->name());
343  Date baseDate = index->zeroInflationTermStructure()->baseDate();
344  Time relativeTime =
345  inflationYearFraction(index->zeroInflationTermStructure()->frequency(),
346  index->zeroInflationTermStructure()->indexIsInterpolated(),
347  index->zeroInflationTermStructure()->dayCounter(), baseDate,
348  dates_[i] - index->zeroInflationTermStructure()->observationLag());
349  std::pair<Real, Real> ii = model_->infdkI(j, relativeTime, relativeTime, z, y);
350 
351  Real baseCPI = index->fixing(baseDate);
352  scenarios[i]->add(cpiKeys_[j], baseCPI * ii.first);
353  }
354 
355  for (Size j = 0; j < n_zeroinf; ++j) {
356  std::string indexName = simMarketConfig_->zeroInflationIndices()[j];
357  Real z = sample.value[model_->pIdx(INF, model_->infIndex(indexName), 0)][i + 1];
358  Real y = sample.value[model_->pIdx(INF, model_->infIndex(indexName), 1)][i + 1];
359  zeroInfCurves[j]->move(dates_[i], z, y);
360  for (Size k = 0; k < ten_zinf_[j].size(); k++) {
361  Date d = dates_[i] + ten_zinf_[j][k];
362  Time T = dc.yearFraction(dates_[i], d);
363  Real zero = zeroInfCurves[j]->zeroRate(T);
364  scenarios[i]->add(zeroInflationKeys_[j * ten_zinf_[j].size() + k], zero);
365  }
366  }
367 
368  for (Size j = 0; j < n_yoyinf; ++j) {
369  std::string indexName = simMarketConfig_->yoyInflationIndices()[j];
370  Size ccy = model_->ccyIndex(model_->infdk(j)->currency());
371  Real z = sample.value[model_->pIdx(INF, model_->infIndex(indexName), 0)][i + 1];
372  Real y = sample.value[model_->pIdx(INF, model_->infIndex(indexName), 1)][i + 1];
373  Real ir_z = sample.value[model_->pIdx(IR, ccy)][i + 1];
374  yoyInfCurves[j]->move(dates_[i], z, y, ir_z);
375  vector<Date> d_yinf;
376  for (Size k = 0; k < ten_yinf_[j].size(); k++)
377  d_yinf.push_back(dates_[i] + ten_yinf_[j][k]);
378  map<Date, Real> yoyRates = yoyInfCurves[j]->yoyRates(d_yinf);
379  for (Size l = 0; l < d_yinf.size(); l++) {
380  scenarios[i]->add(yoyInflationKeys_[j * ten_yinf_[j].size() + l], yoyRates[d_yinf[l]]);
381  }
382  }
383 
384  // TODO: Further risk factor classes are added here
385  }
386  return scenarios;
