31BOOST_AUTO_TEST_CASE(testAbsoluteDiscountShift) {
32
33 BOOST_TEST_MESSAGE("Testing absolute shift in a discount curve");
34
35 // Set up
36 auto ssd = boost::make_shared<SensitivityScenarioData>();
37 auto ssp = boost::make_shared<ScenarioSimMarketParameters>();
38
39 // Discount curve sensitivity set up to have 1bp absolute shift
40 ssd->discountCurveShiftData()["EUR"] = boost::make_shared<CurveShiftData>();
41 ssd->discountCurveShiftData()["EUR"]->shiftSize = 0.0001;
42 ssd->discountCurveShiftData()["EUR"]->shiftType = "Absolute";
43
44 ssp->setYieldCurveTenors("EUR", {3 * Months, 6 * Months});
45
46 // Pick out the 6M discount from the scenarios
47 RiskFactorKey rf(RFType::DiscountCurve, "EUR", 1);
48
49 SimpleScenario scen_1(asof);
50 DiscountFactor v_1 = 0.995;
51 scen_1.add(rf, v_1);
52
53 SimpleScenario scen_2(asof);
54 DiscountFactor v_2 = 0.990;
55 scen_2.add(rf, v_2);
56
57 // Set up the calculator
58 ScenarioShiftCalculator ssc(ssd, ssp);
59
60 // Test the result
61 Real cal = ssc.shift(rf, scen_1, scen_2);
62 Real exp = 100.480591307889;
63 BOOST_CHECK_CLOSE(cal, exp, tol);
64
65 // Scale the shift size and the calculated shift multiple should scale
66 Real factor = 2.5;
67 ssd->discountCurveShiftData()["EUR"]->shiftSize *= factor;
68 cal = ssc.shift(rf, scen_1, scen_2);
69 exp /= factor;
70 BOOST_CHECK_CLOSE(cal, exp, tol);
71}
