43 CrossAssetStateProcess::discretization parseDiscretization(const string& s) {

66 void ScenarioGeneratorData::fromXML(XMLNode* root) {
67  XMLNode* sim = XMLUtils::locateNode(root, "Simulation");
68  XMLNode* node = XMLUtils::getChildNode(sim, "Parameters");
69  XMLUtils::checkNode(node, "Parameters");
70 
71  std::string discString = XMLUtils::getChildValue(node, "Discretization", true); // mandatory
72  discretization_ = parseDiscretization(discString);
73  LOG("ScenarioGeneratorData discretization = " << discString);
74 
75  std::string calString = XMLUtils::getChildValue(node, "Calendar", true);
76  Calendar cal = parseCalendar(calString);
77 
78  std::string gridString = XMLUtils::getChildValue(node, "Grid", true);
79  std::vector<std::string> tokens;
80  boost::split(tokens, gridString, boost::is_any_of(","));
81  if (tokens.size() <= 2) {
82  grid_ = boost::make_shared<ore::analytics::DateGrid>(gridString, cal);
83  } else {
84  std::vector<Period> gridTenors = XMLUtils::getChildrenValuesAsPeriods(node, "Grid", true);
85  grid_ = boost::make_shared<ore::analytics::DateGrid>(gridTenors, cal);
86  }
87  LOG("ScenarioGeneratorData grid points size = " << grid_->size());
88 
89  std::string sequenceTypeString = XMLUtils::getChildValue(node, "Sequence", true);
90  sequenceType_ = parseSequenceType(sequenceTypeString);
91  LOG("ScenarioGeneratorData sequence type = " << sequenceTypeString);
92 
93  seed_ = XMLUtils::getChildValueAsInt(node, "Seed", true);
94  LOG("ScenarioGeneratorData seed = " << seed_);
95 
96  samples_ = XMLUtils::getChildValueAsInt(node, "Samples", true);
97  LOG("ScenarioGeneratorData samples = " << samples_);
98 
99  if (auto n = XMLUtils::getChildNode(node, "Ordering"))
100  ordering_ = parseSobolBrownianGeneratorOrdering(XMLUtils::getNodeValue(n));
101  else
102  ordering_ = SobolBrownianGenerator::Steps;
103 
104  if (auto n = XMLUtils::getChildNode(node, "DirectionIntegers"))
105  directionIntegers_ = parseSobolRsgDirectionIntegers(XMLUtils::getNodeValue(n));
106  else
107  directionIntegers_ = SobolRsg::JoeKuoD7;
108 
109  LOG("ScenarioGeneratorData done.");
110 }
