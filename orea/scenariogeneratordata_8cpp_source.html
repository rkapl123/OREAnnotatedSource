44void ScenarioGeneratorData::setGrid(boost::shared_ptr<DateGrid> grid) {
45 grid_ = grid;
46
47 std::ostringstream oss;
48 if (grid->tenors().size() == 0) {
49 oss << "";
50 } else {
51 oss << grid->tenors()[0];
52 for (Size i = 1; i < grid->tenors().size(); i++) {
53 oss << ", " << grid->tenors()[i];
54 }
55 }
56
57 gridString_ = oss.str();
58}

60void ScenarioGeneratorData::fromXML(XMLNode* root) {
61 XMLNode* sim = XMLUtils::locateNode(root, "Simulation");
62 XMLNode* node = XMLUtils::getChildNode(sim, "Parameters");
63 XMLUtils::checkNode(node, "Parameters");
64
65 std::string calString = XMLUtils::getChildValue(node, "Calendar", true);
66 Calendar cal = parseCalendar(calString);
67
68 std::string dcString = XMLUtils::getChildValue(node, "DayCounter", false);
69 DayCounter dc = dcString.empty() ? ActualActual(ActualActual::ISDA) : parseDayCounter(dcString);
70
71 gridString_ = XMLUtils::getChildValue(node, "Grid", true);
72 std::vector<std::string> tokens;
73 boost::split(tokens, gridString_, boost::is_any_of(","));
74 if (tokens.size() <= 2) {
75 grid_ = boost::make_shared<DateGrid>(gridString_, cal, dc);
76 } else {
77 std::vector<Period> gridTenors = XMLUtils::getChildrenValuesAsPeriods(node, "Grid", true);
78 grid_ = boost::make_shared<DateGrid>(gridTenors, cal, dc);
79 }
80 LOG("ScenarioGeneratorData grid points size = " << grid_->size());
81
82 std::string sequenceTypeString = XMLUtils::getChildValue(node, "Sequence", true);
83 sequenceType_ = parseSequenceType(sequenceTypeString);
84 LOG("ScenarioGeneratorData sequence type = " << sequenceTypeString);
85
86 seed_ = XMLUtils::getChildValueAsInt(node, "Seed", true);
87 LOG("ScenarioGeneratorData seed = " << seed_);
88
89 samples_ = XMLUtils::getChildValueAsInt(node, "Samples", true);
90 LOG("ScenarioGeneratorData samples = " << samples_);
91
92 // overwrite samples with environment variable OVERWRITE_SCENARIOGENERATOR_SAMPLES
93 if (auto c = getenv("OVERWRITE_SCENARIOGENERATOR_SAMPLES")) {
94 samples_ = std::stol(c);
95 LOG("Overwrite samples with " << samples_ << " from environment variable OVERWRITE_SCENARIOGENERATOR_SAMPLES");
96 }
97
98 if (auto n = XMLUtils::getChildNode(node, "Ordering"))
99 ordering_ = parseSobolBrownianGeneratorOrdering(XMLUtils::getNodeValue(n));
100 else
101 ordering_ = SobolBrownianGenerator::Steps;
102
103 if (auto n = XMLUtils::getChildNode(node, "DirectionIntegers"))
104 directionIntegers_ = parseSobolRsgDirectionIntegers(XMLUtils::getNodeValue(n));
105 else
106 directionIntegers_ = SobolRsg::JoeKuoD7;
107
108 withCloseOutLag_ = false;
109 if (XMLUtils::getChildNode(node, "CloseOutLag") != NULL) {
110 withCloseOutLag_ = true;
111 closeOutLag_ = parsePeriod(XMLUtils::getChildValue(node, "CloseOutLag", true));
112 grid_->addCloseOutDates(closeOutLag_);
113 LOG("Use lagged close out grid, lag period is " << closeOutLag_);
114 }
115 withMporStickyDate_ = false;
116 if (XMLUtils::getChildNode(node, "MporMode") != NULL) {
117 string mporMode = XMLUtils::getChildValue(node, "MporMode", true);
118 if (mporMode == "StickyDate") {
119 withMporStickyDate_ = true;
120 LOG("Use Mpor sticky date mode");
121 } else if (mporMode == "ActualDate") {
122 withMporStickyDate_ = false;
123 LOG("Use Mpor actual date mode");
124 } else {
125 QL_FAIL("MporMode " << mporMode << " not recognised");
126 }
127 }
128
129 LOG("ScenarioGeneratorData done.");
130}

132XMLNode* ScenarioGeneratorData::toXML(XMLDocument& doc) {
133 XMLNode* node = doc.allocNode("Simulation");
134 XMLNode* pNode = XMLUtils::addChild(doc, node, "Parameters");
135
136 if (grid_) {
137 XMLUtils::addChild(doc, pNode, "Calendar", grid_->calendar().name());
138 XMLUtils::addChild(doc, pNode, "DayCounter", grid_->dayCounter().name());
139 if (!gridString_.empty()) {
140 XMLUtils::addChild(doc, pNode, "Grid", gridString_);
141 } else {
142 XMLUtils::addGenericChildAsList(doc, pNode, "Grid", grid_->tenors());
143 }
144 }
145
146 XMLUtils::addChild(doc, pNode, "Sequence", ore::data::to_string( sequenceType_));
147 XMLUtils::addChild(doc, pNode, "Seed", to_string(seed_));
148 XMLUtils::addChild(doc, pNode, "Samples", to_string(samples_));
149
150 XMLUtils::addChild(doc, pNode, "Ordering", ore::data::to_string((SobolBrownianGenerator::Ordering) ordering_) );
151 XMLUtils::addChild(doc, pNode, "DirectionIntegers", ore::data::to_string(directionIntegers_));
152
153 if (withCloseOutLag_) {
154 XMLUtils::addChild(doc, pNode, "CloseOutLag", closeOutLag_);
155 }
156 if (withMporStickyDate_) {
157 XMLUtils::addChild(doc, pNode, "MporMode", "StickyDate");
158 } else {
159 XMLUtils::addChild(doc, pNode, "MporMode", "ActualDate");
160 }
161
162 return node;
163}
