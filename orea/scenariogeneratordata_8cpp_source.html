44void ScenarioGeneratorData::setGrid(boost::shared_ptr<DateGrid> grid) {
45 grid_ = grid;

60void ScenarioGeneratorData::fromXML(XMLNode* root) {
61 XMLNode* sim = XMLUtils::locateNode(root, "Simulation");
62 XMLNode* node = XMLUtils::getChildNode(sim, "Parameters");
63 XMLUtils::checkNode(node, "Parameters");
64
65 std::string calString = XMLUtils::getChildValue(node, "Calendar", true);
66 Calendar cal = parseCalendar(calString);
67
68 std::string dcString = XMLUtils::getChildValue(node, "DayCounter", false);
69 DayCounter dc = dcString.empty() ? ActualActual(ActualActual::ISDA) : parseDayCounter(dcString);
70
71 gridString_ = XMLUtils::getChildValue(node, "Grid", true);
72 std::vector<std::string> tokens;
73 boost::split(tokens, gridString_, boost::is_any_of(","));
74 if (tokens.size() <= 2) {
75 grid_ = boost::make_shared<DateGrid>(gridString_, cal, dc);
76 } else {
77 std::vector<Period> gridTenors = XMLUtils::getChildrenValuesAsPeriods(node, "Grid", true);
78 grid_ = boost::make_shared<DateGrid>(gridTenors, cal, dc);
79 }
80 LOG("ScenarioGeneratorData grid points size = " << grid_->size());
81
82 std::string sequenceTypeString = XMLUtils::getChildValue(node, "Sequence", true);
83 sequenceType_ = parseSequenceType(sequenceTypeString);
84 LOG("ScenarioGeneratorData sequence type = " << sequenceTypeString);
85
86 seed_ = XMLUtils::getChildValueAsInt(node, "Seed", true);
87 LOG("ScenarioGeneratorData seed = " << seed_);
88
89 samples_ = XMLUtils::getChildValueAsInt(node, "Samples", true);
90 LOG("ScenarioGeneratorData samples = " << samples_);
91
92 // overwrite samples with environment variable OVERWRITE_SCENARIOGENERATOR_SAMPLES
93 if (auto c = getenv("OVERWRITE_SCENARIOGENERATOR_SAMPLES")) {
94 try {
95 samples_ = std::stol(c);
96 } catch (const std::exception& e) {
97 WLOG("enviroment variable OVERWRITE_SCENARIOGENERATOR_SAMPLES is set ("
98 << c << ") but can not be parsed to a number - ignoring.");
99 }
100 LOG("Overwrite samples with " << samples_ << " from environment variable OVERWRITE_SCENARIOGENERATOR_SAMPLES")
101 }
102
103 if (auto n = XMLUtils::getChildNode(node, "Ordering"))
104 ordering_ = parseSobolBrownianGeneratorOrdering(XMLUtils::getNodeValue(n));
105 else
106 ordering_ = SobolBrownianGenerator::Steps;
107
108 if (auto n = XMLUtils::getChildNode(node, "DirectionIntegers"))
109 directionIntegers_ = parseSobolRsgDirectionIntegers(XMLUtils::getNodeValue(n));
110 else
111 directionIntegers_ = SobolRsg::JoeKuoD7;
112
113 withCloseOutLag_ = false;
114 if (XMLUtils::getChildNode(node, "CloseOutLag") != NULL) {
115 withCloseOutLag_ = true;
116 closeOutLag_ = parsePeriod(XMLUtils::getChildValue(node, "CloseOutLag", true));
117 grid_->addCloseOutDates(closeOutLag_);
118 LOG("Use lagged close out grid, lag period is " << closeOutLag_);
119 }
120 withMporStickyDate_ = false;
121 if (XMLUtils::getChildNode(node, "MporMode") != NULL) {
122 string mporMode = XMLUtils::getChildValue(node, "MporMode", true);
123 if (mporMode == "StickyDate") {
124 withMporStickyDate_ = true;
125 LOG("Use Mpor sticky date mode");
126 } else if (mporMode == "ActualDate") {
127 withMporStickyDate_ = false;
128 LOG("Use Mpor actual date mode");
129 } else {
130 QL_FAIL("MporMode " << mporMode << " not recognised");
131 }
132 }
133
134 if (XMLUtils::getChildNode(node, "MporCashFlowMode") != NULL) {
135 string mporCashFlowModeString = XMLUtils::getChildValue(node, "MporCashFlowMode", true);
136 mporCashFlowMode_ = parseMporCashFlowMode(mporCashFlowModeString);
137 LOG("Mpor cash flow mode is" << mporCashFlowModeString);
138 } else {
139 if (withMporStickyDate_)
140 mporCashFlowMode_ = ScenarioGeneratorData::MporCashFlowMode::NonePay;
141 else
142 mporCashFlowMode_ = ScenarioGeneratorData::MporCashFlowMode::BothPay;
143 }
144
145 LOG("ScenarioGeneratorData done.");
146}

148XMLNode* ScenarioGeneratorData::toXML(XMLDocument& doc) {
149 XMLNode* node = doc.allocNode("Simulation");
150 XMLNode* pNode = XMLUtils::addChild(doc, node, "Parameters");
151
152 if (grid_) {
153 XMLUtils::addChild(doc, pNode, "Calendar", grid_->calendar().name());
154 XMLUtils::addChild(doc, pNode, "DayCounter", grid_->dayCounter().name());
155 if (!gridString_.empty()) {
156 XMLUtils::addChild(doc, pNode, "Grid", gridString_);
157 } else {
158 XMLUtils::addGenericChildAsList(doc, pNode, "Grid", grid_->tenors());
159 }
160 }
161
162 XMLUtils::addChild(doc, pNode, "Sequence", ore::data::to_string( sequenceType_));
163 XMLUtils::addChild(doc, pNode, "Seed", to_string(seed_));
164 XMLUtils::addChild(doc, pNode, "Samples", to_string(samples_));
165
166 XMLUtils::addChild(doc, pNode, "Ordering", ore::data::to_string((SobolBrownianGenerator::Ordering) ordering_) );
167 XMLUtils::addChild(doc, pNode, "DirectionIntegers", ore::data::to_string(directionIntegers_));
168
169 if (withCloseOutLag_) {
170 XMLUtils::addChild(doc, pNode, "CloseOutLag", closeOutLag_);
171 }
172 if (withMporStickyDate_) {
173 XMLUtils::addChild(doc, pNode, "MporMode", "StickyDate");
174 } else {
175 XMLUtils::addChild(doc, pNode, "MporMode", "ActualDate");
176 }
177
178 return node;
179}

181ScenarioGeneratorData::MporCashFlowMode parseMporCashFlowMode(const string& s) {
182 static map<string, ScenarioGeneratorData::MporCashFlowMode> m = {
183 {"NonePay", ScenarioGeneratorData::MporCashFlowMode::NonePay},
184 {"BothPay", ScenarioGeneratorData::MporCashFlowMode::BothPay},
185 {"WePay", ScenarioGeneratorData::MporCashFlowMode::WePay},
186 {"TheyPay", ScenarioGeneratorData::MporCashFlowMode::TheyPay}
187 };
188 auto it = m.find(s);
189 if (it != m.end()) {
190 return it->second;
191 } else {
192 QL_FAIL("Mpor cash flow mode \"" << s << "\" not recognized");
193 }
194}

196std::ostream& operator<<(std::ostream& out, ScenarioGeneratorData::MporCashFlowMode t) {
197 if (t == ScenarioGeneratorData::MporCashFlowMode::NonePay)
198 out << "NonePay";
199 else if (t == ScenarioGeneratorData::MporCashFlowMode::BothPay)
200 out << "BothPay";
201 else if (t == ScenarioGeneratorData::MporCashFlowMode::WePay)
202 out << "WePay";
203 else if (t == ScenarioGeneratorData::MporCashFlowMode::TheyPay)
204 out << "TheyPay";
205 else
206 QL_FAIL("Mpor cash flow mode not covered");
207 return out;
208}
