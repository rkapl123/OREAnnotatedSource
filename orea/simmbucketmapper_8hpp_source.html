40 //! Destructor

43 /*! Return the SIMM <em>bucket</em> for a given SIMM <em>RiskType</em> and
44 SIMM <em>Qualifier</em> (using valid mappings only)
45
46 \warning An error is thrown if there is no <em>bucket</em>
47 for the combination.
48 */
49 virtual std::string bucket(const SimmConfiguration::RiskType& riskType, const std::string& qualifier) const = 0;

54 //! Check if the given \p riskType and \p qualifier has a mapping (which is valid, and matches the fallback flag if given)
55 virtual bool has(const SimmConfiguration::RiskType& riskType, const std::string& qualifier, boost::optional<bool> fallback = boost::none) const = 0;
56
57 //! Add a single \p bucket mapping for \p qualifier with risk type \p riskType

61 */
62 virtual void addMapping(const SimmConfiguration::RiskType& riskType, const std::string& qualifier,
63 const std::string& bucket, const std::string& validFrom = "", const std::string& validTo = "", bool fallback = false) = 0;
64
65 virtual const std::set<FailedMapping>& failedMappings() const = 0;
66};

68inline bool operator<(const SimmBucketMapper::FailedMapping& a, const SimmBucketMapper::FailedMapping& b) {
69 return std::make_tuple(a.name, a.lookupName, a.riskType, a.lookupRiskType) <
70 std::make_tuple(b.name, b.lookupName, b.riskType, b.lookupRiskType);
