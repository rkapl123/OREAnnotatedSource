30RiskFilter::RiskFilter(const Size riskClassIndex, const Size riskTypeIndex)
31 : riskClassIndex_(riskClassIndex), riskTypeIndex_(riskTypeIndex) {
32
33 QL_REQUIRE(riskClassIndex_ < riskClassLabel_.size(),
34 "RiskFilter: riskClassIndex " << riskClassIndex_ << " not allowed.");
35 QL_REQUIRE(riskTypeIndex_ < riskTypeLabel_.size(),
36 "RiskFilter: riskTypeIndex " << riskTypeIndex_ << " not allowed.");
37
38 static const std::set<RiskFactorKey::KeyType> all = {RiskFactorKey::KeyType::DiscountCurve,
39 RiskFactorKey::KeyType::YieldCurve,
40 RiskFactorKey::KeyType::IndexCurve,
41 RiskFactorKey::KeyType::SwaptionVolatility,
42 RiskFactorKey::KeyType::OptionletVolatility,
43 RiskFactorKey::KeyType::FXSpot,
44 RiskFactorKey::KeyType::FXVolatility,
45 RiskFactorKey::KeyType::EquitySpot,
46 RiskFactorKey::KeyType::EquityVolatility,
47 RiskFactorKey::KeyType::DividendYield,
48 RiskFactorKey::KeyType::SurvivalProbability,
49 RiskFactorKey::KeyType::RecoveryRate,
50 RiskFactorKey::KeyType::CDSVolatility,
51 RiskFactorKey::KeyType::BaseCorrelation,
52 RiskFactorKey::KeyType::CPIIndex,
53 RiskFactorKey::KeyType::ZeroInflationCurve,
54 RiskFactorKey::KeyType::YoYInflationCurve,
55 RiskFactorKey::KeyType::YoYInflationCapFloorVolatility,
56 RiskFactorKey::KeyType::SecuritySpread,
57 RiskFactorKey::KeyType::YieldVolatility};
58
59 std::set<RiskFactorKey::KeyType> allowed_type, allowed;
60
61 if (riskTypeIndex_ == 0) {
62 allowed_type = all;
63 } else {
64 switch (riskTypeIndex_) {
65 case 1:
66 allowed_type = {RiskFactorKey::KeyType::DiscountCurve,
67 RiskFactorKey::KeyType::YieldCurve,
68 RiskFactorKey::KeyType::IndexCurve,
69 RiskFactorKey::KeyType::FXSpot,
70 RiskFactorKey::KeyType::EquitySpot,
71 RiskFactorKey::KeyType::DividendYield,
72 RiskFactorKey::KeyType::SurvivalProbability,
73 RiskFactorKey::KeyType::RecoveryRate,
74 RiskFactorKey::KeyType::CPIIndex,
75 RiskFactorKey::KeyType::ZeroInflationCurve,
76 RiskFactorKey::KeyType::YoYInflationCurve,
77 RiskFactorKey::KeyType::SecuritySpread};
78 break;
79 case 2:
80 allowed_type = {RiskFactorKey::KeyType::SwaptionVolatility,
81 RiskFactorKey::KeyType::OptionletVolatility,
82 RiskFactorKey::KeyType::FXVolatility,
83 RiskFactorKey::KeyType::EquityVolatility,
84 RiskFactorKey::KeyType::CDSVolatility,
85 RiskFactorKey::KeyType::YieldVolatility,
86 RiskFactorKey::KeyType::YoYInflationCapFloorVolatility};
87 break;
88 case 3:
89 allowed_type = {RiskFactorKey::KeyType::BaseCorrelation};
90 break;
91 default:
92 QL_FAIL("unexpected riskTypeIndex " << riskTypeIndex_);
93 }
94 }
95
96 if (riskClassIndex_ == 0) {
97 allowed = allowed_type;
98 } else {
99 std::set<RiskFactorKey::KeyType> allowed_class;
100 switch (riskClassIndex_) {
101 case 1:
102 allowed_class = {
103 RiskFactorKey::KeyType::DiscountCurve, RiskFactorKey::KeyType::YieldCurve,
104 RiskFactorKey::KeyType::IndexCurve, RiskFactorKey::KeyType::SwaptionVolatility,
105 RiskFactorKey::KeyType::OptionletVolatility, RiskFactorKey::KeyType::SecuritySpread,
106 RiskFactorKey::KeyType::YieldVolatility, RiskFactorKey::KeyType::YoYInflationCapFloorVolatility};
107 break;
108 case 2:
109 allowed_class = {RiskFactorKey::KeyType::CPIIndex, RiskFactorKey::KeyType::ZeroInflationCurve,
110 RiskFactorKey::KeyType::YoYInflationCurve};
111 break;
112 case 3:
113 allowed_class = {RiskFactorKey::KeyType::SurvivalProbability, RiskFactorKey::KeyType::RecoveryRate,
114 RiskFactorKey::KeyType::CDSVolatility, RiskFactorKey::KeyType::BaseCorrelation};
115 break;
116 case 4:
117 allowed_class = {RiskFactorKey::KeyType::EquitySpot, RiskFactorKey::KeyType::EquityVolatility,
118 RiskFactorKey::KeyType::DividendYield};
119 break;
120 case 5:
121 allowed_class = {RiskFactorKey::KeyType::FXSpot, RiskFactorKey::KeyType::FXVolatility};
122 break;
123 default:
124 QL_FAIL("unexpected riskClassIndex " << riskClassIndex_);
125 }
126 std::set_intersection(allowed_type.begin(), allowed_type.end(), allowed_class.begin(), allowed_class.end(),
127 std::inserter(allowed, allowed.begin()));
128 }
129 // use the complement if smaller, the check is called frequently
130 if (allowed.size() > all.size() / 2) {
131 std::set_difference(all.begin(), all.end(), allowed.begin(), allowed.end(),
132 std::inserter(allowed_, allowed_.begin()));
133 neg_ = true;
134 } else {
135 allowed_ = allowed;
136 neg_ = false;
137 }
138}
139
140bool RiskFilter::allow(const RiskFactorKey& t) const {
141 bool found = std::find(allowed_.begin(), allowed_.end(), t.keytype) != allowed_.end();
142 return neg_ ? !found : found;
143}
