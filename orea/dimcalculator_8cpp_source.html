48 DynamicInitialMarginCalculator::DynamicInitialMarginCalculator(
49  const boost::shared_ptr<Portfolio>& portfolio, const boost::shared_ptr<NPVCube>& cube,
50  const boost::shared_ptr<CubeInterpretation>& cubeInterpretation,
51  const boost::shared_ptr<AggregationScenarioData>& scenarioData, Real quantile, Size horizonCalendarDays,
52  const std::map<std::string, Real>& currentIM)
53  : portfolio_(portfolio), cube_(cube), cubeInterpretation_(cubeInterpretation), scenarioData_(scenarioData),
54  quantile_(quantile), horizonCalendarDays_(horizonCalendarDays), currentIM_(currentIM) {
55 
56  QL_REQUIRE(portfolio_, "portfolio is null");
57  QL_REQUIRE(cube_, "cube is null");
58  QL_REQUIRE(cubeInterpretation_, "cube interpretation is null");
59  QL_REQUIRE(scenarioData_, "aggregation scenario data is null");
60 
61  boost::shared_ptr<RegularCubeInterpretation> regCubeInt =
62  boost::dynamic_pointer_cast<RegularCubeInterpretation>(cubeInterpretation_);
63  cubeIsRegular_ = (regCubeInt != NULL);
64  datesLoopSize_ = cubeIsRegular_ ? cube_->dates().size() - 1 : cube_->dates().size();
65 
66  Size dates = cube_->dates().size();
67  Size samples = cube_->samples();
68 
69  if (!cubeInterpretation_->hasMporFlows(cube_)) {
70  WLOG("cube holds no mpor flows, will assume no flows in the dim calculation");
71  }
72 
73  // initialise aggregate NPV and Flow by date and scenario
74  set<string> nettingSets;
75  for (Size i = 0; i < portfolio_->size(); ++i) {
76  string tradeId = portfolio_->trades()[i]->id();
77  string nettingSetId = portfolio_->trades()[i]->envelope().nettingSetId();
78  if (nettingSets.find(nettingSetId) == nettingSets.end()) {
79  nettingSets.insert(nettingSetId);
80  nettingSetNPV_[nettingSetId] = vector<vector<Real>>(dates, vector<Real>(samples, 0.0));
81  nettingSetCloseOutNPV_[nettingSetId] = vector<vector<Real>>(dates, vector<Real>(samples, 0.0));
82  nettingSetFLOW_[nettingSetId] = vector<vector<Real>>(dates, vector<Real>(samples, 0.0));
83  nettingSetDeltaNPV_[nettingSetId] = vector<vector<Real>>(dates, vector<Real>(samples, 0.0));
84  nettingSetDIM_[nettingSetId] = vector<vector<Real>>(dates, vector<Real>(samples, 0.0));
85  nettingSetExpectedDIM_[nettingSetId] = vector<Real>(dates, 0.0);
86  }
87  for (Size j = 0; j < datesLoopSize_; ++j) {
88  for (Size k = 0; k < samples; ++k) {
89  Real defaultNpv = cubeInterpretation_->getDefaultNpv(cube_, i, j, k);
90  Real closeOutNpv = cubeInterpretation_->getCloseOutNpv(cube_, i, j, k);
91  Real mporFlow =
92  cubeInterpretation_->hasMporFlows(cube_) ? cubeInterpretation_->getMporFlows(cube_, i, j, k) : 0.0;
93  nettingSetNPV_[nettingSetId][j][k] += defaultNpv;
94  nettingSetCloseOutNPV_[nettingSetId][j][k] += closeOutNpv;
95  nettingSetFLOW_[nettingSetId][j][k] += mporFlow;
96  }
97  }
98  }
99 
100  for (auto n : nettingSets)
101  nettingSetIds_.push_back(n);
102 
103  dimCube_ = boost::make_shared<SinglePrecisionInMemoryCube>(cube_->asof(), nettingSetIds_, cube_->dates(),
104  cube_->samples());
105 }

128 void DynamicInitialMarginCalculator::exportDimEvolution(ore::data::Report& dimEvolutionReport) {
129 
130  Size samples = dimCube_->samples();
131  Size stopDatesLoop = datesLoopSize_;
132  Date asof = cube_->asof();
133 
134  dimEvolutionReport.addColumn("TimeStep", Size())
135  .addColumn("Date", Date())
136  .addColumn("DaysInPeriod", Size())
137  .addColumn("AverageDIM", Real(), 6)
138  .addColumn("AverageFLOW", Real(), 6)
139  .addColumn("NettingSet", string())
140  .addColumn("Time", Real(), 6);
141 
142  for (auto nettingSet : dimCube_->ids()) {
143 
144  LOG("Export DIM evolution for netting set " << nettingSet);
145  for (Size i = 0; i < stopDatesLoop; ++i) {
146  Real expectedFlow = 0.0;
147  for (Size j = 0; j < samples; ++j) {
148  expectedFlow += nettingSetFLOW_[nettingSet][i][j] / samples;
149  }
150 
151  Date defaultDate = dimCube_->dates()[i];
152  Time t = ActualActual(ActualActual::ISDA).yearFraction(asof, defaultDate);
153  Size days = cubeInterpretation_->getMporCalendarDays(dimCube_, i);
154  dimEvolutionReport.next()
155  .add(i)
156  .add(defaultDate)
157  .add(days)
158  .add(nettingSetExpectedDIM_[nettingSet][i])
159  .add(expectedFlow)
160  .add(nettingSet)
161  .add(t);
162  }
163  }
164  dimEvolutionReport.end();
165  LOG("Exporting expected DIM through time done");
166 }
