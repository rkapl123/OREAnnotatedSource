43 //! InMemoryCube stores the cube in memory using nested STL vectors
44 /*! InMemoryCube stores the cube in memory using nested STL vectors, this class is a template
45  * to allow both single and double precision implementations.
46  *
47  * The use of nested STL vectors is adds a small memory overhead (~ 1 to 2%)
48 
49  \ingroup cube
50  */
51 template <typename T> class InMemoryCubeBase : public NPVCube {
52 public:
53  //! default ctor
54  InMemoryCubeBase(const Date& asof, const vector<std::string>& ids, const vector<Date>& dates, Size samples,
55  const T& t = T())
56  : asof_(asof), ids_(ids), dates_(dates), samples_(samples), t0Data_(ids.size(), t),
57  data_(ids.size(), vector<vector<T>>(dates.size(), vector<T>(samples, t))) {
58  QL_REQUIRE(ids.size() > 0, "InMemoryCube::InMemoryCube no ids specified");
59  QL_REQUIRE(dates.size() > 0, "InMemoryCube::InMemoryCube no dates specified");
60  QL_REQUIRE(samples > 0, "InMemoryCube::InMemoryCube samples must be > 0");
61  }
62  //! construct from file
63  InMemoryCubeBase(const std::string& fileName) {
64  load(fileName);
65  QL_REQUIRE(numIds() > 0 && numDates() > 0 && samples() > 0,
66  "InMemoryCube::InMemoryCube failed to load from file " << fileName);
67  }
68 
69  //! default constructor
70  InMemoryCubeBase() {}
71 
72  //! load cube from an archive
73  void load(const std::string& fileName) override {
74  std::ifstream ifs(fileName.c_str(), std::fstream::binary);
75  QL_REQUIRE(ifs.is_open(), "error opening file " << fileName);
76  boost::archive::binary_iarchive ia(ifs);
77  ia >> *this;
78  }
79 
80  //! write cube to an archive
81  void save(const std::string& fileName) const override {
82  std::ofstream ofs(fileName.c_str(), std::fstream::binary);
83  QL_REQUIRE(ofs.is_open(), "error opening file " << fileName);
84  boost::archive::binary_oarchive oa(ofs);
85  oa << *this;
86  }
87 
88  //! Return the length of each dimension
89  Size numIds() const override { return ids_.size(); }
90  Size numDates() const override { return dates_.size(); }
91  virtual Size samples() const override { return samples_; }
92 
93  //! Get the vector of ids for this cube
94  const std::vector<std::string>& ids() const override { return ids_; }
95  //! Get the vector of dates for this cube
96  const std::vector<QuantLib::Date>& dates() const override { return dates_; }
97 
98  //! Return the asof date (T0 date)
99  QuantLib::Date asof() const override { return asof_; }
100 
101 protected:
102  void check(Size i, Size j, Size k, Size d) const {
103  QL_REQUIRE(i < numIds(), "Out of bounds on ids (i=" << i << ", numIds=" << numIds() << ")");
104  QL_REQUIRE(j < numDates(), "Out of bounds on dates (j=" << j << ", numDates=" << numDates() << ")");
105  QL_REQUIRE(k < samples(), "Out of bounds on samples (k=" << k << ", samples=" << samples() << ")");
106  QL_REQUIRE(d < depth(), "Out of bounds on depth (d=" << d << ", depth=" << depth() << ")");
107  }
108 
109 private:
110  friend class boost::serialization::access;
111  template <class Archive> void serialize(Archive& ar, const unsigned int) {
112  ar& asof_;
113  ar& ids_;
114  ar& dates_;
115  ar& samples_;
116  ar& t0Data_;
117  ar& data_;
118  }
119 
120 private:
121  QuantLib::Date asof_;
122  vector<std::string> ids_;
123  vector<QuantLib::Date> dates_;
124  Size samples_;
125 
126 protected:
127  vector<T> t0Data_;
128  vector<vector<vector<T>>> data_;
129 };

131 //! InMemoryCube of fixed depth 1
132 /*! This implementation stores the type directly in an InMemoryCubeBase to avoid us having
133  * nested vectors all of length 1
134  */
135 template <typename T> class InMemoryCube1 : public InMemoryCubeBase<T> {
136 public:
137  //! ctor
138  InMemoryCube1(const Date& asof, const vector<std::string>& ids, const vector<Date>& dates, Size samples,
139  const T& t = T())
140  : InMemoryCubeBase<T>(asof, ids, dates, samples, t) {}
141 
142  //! default
143  InMemoryCube1() {}
144 
145  //! Fixed depth
146  Size depth() const override { return 1; }
147 
148  //! Get a T0 value from the cube
149  virtual Real getT0(Size i, Size d) const override {
150  this->check(i, 0, 0, d);
151  return this->t0Data_[i];
152  }
153 
154  //! Set a value in the cube
155  virtual void setT0(Real value, Size i, Size d) override {
156  this->check(i, 0, 0, d);
157  this->t0Data_[i] = static_cast<T>(value);
158  }
159 
160  //! Get a value from the cube
161  Real get(Size i, Size j, Size k, Size d) const override {
162  this->check(i, j, k, d);
163  return this->data_[i][j][k];
164  }
165 
166  //! Set a value in the cube
167  void set(Real value, Size i, Size j, Size k, Size d) override {
168  this->check(i, j, k, d);
169  this->data_[i][j][k] = static_cast<T>(value);
170  }
171 };
