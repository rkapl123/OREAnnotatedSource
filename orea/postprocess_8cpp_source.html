328void PostProcess::updateNettingSetKVA() {
329
330 // Loop over all netting sets
331 for (const auto& [nettingSetId,pos] : nettingSetIds()) {
332 // Init results
333 ourNettingSetKVACCR_[nettingSetId] = 0.0;
334 theirNettingSetKVACCR_[nettingSetId] = 0.0;
335 ourNettingSetKVACVA_[nettingSetId] = 0.0;
336 theirNettingSetKVACVA_[nettingSetId] = 0.0;
337 }
338
339 if (!analytics_["kva"])
340 return;
341
342 LOG("Update netting set KVA");
343
344 vector<Date> dateVector = cube_->dates();
345 Size dates = dateVector.size();
346 Date today = market_->asofDate();
347 Handle<YieldTermStructure> discountCurve = market_->discountCurve(baseCurrency_, configuration_);
348 DayCounter dc = ActualActual(ActualActual::ISDA);
349
350 // Loop over all netting sets
351 for (const auto& [nettingSetId, pos] : nettingSetIds()) {
352 string cid;
353 if (analytics_["flipViewXVA"]) {
354 cid = dvaName_;
355 } else {
356 cid = nettedExposureCalculator_->counterparty(nettingSetId);
357 }
358 LOG("KVA for netting set " << nettingSetId);
359
360 // Main input are the EPE and ENE profiles, previously computed
361 vector<Real> epe = netEPE_[nettingSetId];
362 vector<Real> ene = netENE_[nettingSetId];
363
364 // PD from counterparty Dts, floored to avoid 0 ...
365 // Today changed to today+1Y to get the one-year PD
366 Handle<DefaultProbabilityTermStructure> cvaDts = market_->defaultCurve(cid, configuration_)->curve();
367 QL_REQUIRE(!cvaDts.empty(), "Default curve missing for counterparty " << cid);
368 Real cvaRR = market_->recoveryRate(cid, configuration_)->value();
369 Real PD1 = std::max(cvaDts->defaultProbability(today + 1 * Years), 0.000000000001);
370 Real LGD1 = (1 - cvaRR);
371
372 // FIXME: if flipViewXVA is sufficient, then all code for their KVA-CCR could be discarded here...
373 Handle<DefaultProbabilityTermStructure> dvaDts;
374 Real dvaRR = 0.0;
375 Real PD2 = 0;
376 if (analytics_["flipViewXVA"]) {
377 dvaName_ = nettedExposureCalculator_->counterparty(nettingSetId);
378 }
379 if (dvaName_ != "") {
380 dvaDts = market_->defaultCurve(dvaName_, configuration_)->curve();
381 dvaRR = market_->recoveryRate(dvaName_, configuration_)->value();
382 PD2 = std::max(dvaDts->defaultProbability(today + 1 * Years), 0.000000000001);
383 } else {
384 ALOG("dvaName not specified, own PD set to zero for their KVA calculation");
385 }
386 Real LGD2 = (1 - dvaRR);
387
388 // Granularity adjustment, Gordy (2004):
389 Real rho1 = 0.12 * (1 - std::exp(-50 * PD1)) / (1 - std::exp(-50)) +
390 0.24 * (1 - (1 - std::exp(-50 * PD1)) / (1 - std::exp(-50)));
391 Real rho2 = 0.12 * (1 - std::exp(-50 * PD2)) / (1 - std::exp(-50)) +
392 0.24 * (1 - (1 - std::exp(-50 * PD2)) / (1 - std::exp(-50)));
393
394 // Basel II internal rating based (IRB) estimate of worst case PD:
395 // Large homogeneous pool (LHP) approximation of Vasicek (1997)
396 InverseCumulativeNormal icn;
397 CumulativeNormalDistribution cnd;
398 Real PD99_1 = cnd((icn(PD1) + std::sqrt(rho1) * icn(0.999)) / (std::sqrt(1 - rho1))) - PD1;
399 Real PD99_2 = cnd((icn(PD2) + std::sqrt(rho2) * icn(0.999)) / (std::sqrt(1 - rho2))) - PD2;
400
401 // KVA regulatory PD, worst case PD, floored at 0.03 for corporates and banks, not floored for sovereigns
402 Real kva99PD1 = std::max(PD99_1, kvaTheirPdFloor_);
403 Real kva99PD2 = std::max(PD99_2, kvaOurPdFloor_);
404
405 // Factor B(PD) for the maturity adjustment factor, B(PD) = (0.11852 - 0.05478 * ln(PD)) ^ 2
406 Real kvaMatAdjB1 = std::pow((0.11852 - 0.05478 * std::log(PD1)), 2.0);
407 Real kvaMatAdjB2 = std::pow((0.11852 - 0.05478 * std::log(PD2)), 2.0);
408
409 DLOG("Our KVA-CCR " << nettingSetId << ": PD=" << PD1);
410 DLOG("Our KVA-CCR " << nettingSetId << ": LGD=" << LGD1);
411 DLOG("Our KVA-CCR " << nettingSetId << ": rho=" << rho1);
412 DLOG("Our KVA-CCR " << nettingSetId << ": PD99=" << PD99_1);
413 DLOG("Our KVA-CCR " << nettingSetId << ": PD Floor=" << kvaTheirPdFloor_);
414 DLOG("Our KVA-CCR " << nettingSetId << ": Floored PD99=" << kva99PD1);
415 DLOG("Our KVA-CCR " << nettingSetId << ": B(PD)=" << kvaMatAdjB1);
416
417 DLOG("Their KVA-CCR " << nettingSetId << ": PD=" << PD2);
418 DLOG("Their KVA-CCR " << nettingSetId << ": LGD=" << LGD2);
419 DLOG("Their KVA-CCR " << nettingSetId << ": rho=" << rho2);
420 DLOG("Their KVA-CCR " << nettingSetId << ": PD99=" << PD99_2);
421 DLOG("Their KVA-CCR " << nettingSetId << ": PD Floor=" << kvaOurPdFloor_);
422 DLOG("Their KVA-CCR " << nettingSetId << ": Floored PD99=" << kva99PD2);
423 DLOG("Their KVA-CCR " << nettingSetId << ": B(PD)=" << kvaMatAdjB2);
424
425 for (Size j = 0; j < dates; ++j) {
426 Date d0 = j == 0 ? today : cube_->dates()[j - 1];
427 Date d1 = cube_->dates()[j];
428
429 // Preprocess:
430 // 1) Effective maturity from effective expected exposure as of time j
431 // Index _1 corresponds to our perspective, index _2 to their perspective.
432 // 2) Basel EEPE as of time j, i.e. as time average over EEE, starting at time j
433 // More accuracy may be achieved here by using a Longstaff-Schwartz method / regression
434 Real eee_kva_1 = 0.0, eee_kva_2 = 0.0;
435 Real effMatNumer1 = 0.0, effMatNumer2 = 0.0;
436 Real effMatDenom1 = 0.0, effMatDenom2 = 0.0;
437 Real eepe_kva_1 = 0, eepe_kva_2 = 0.0;
438 Size kmax = j, count = 0;
439 // Cut off index for EEPE/EENE calculation: One year ahead
440 while (dateVector[kmax] < dateVector[j] + 1 * Years + 4 * Days && kmax < dates - 1)
441 kmax++;
442 Real sumdt = 0.0, eee1_b = 0.0, eee2_b = 0.0;
443 for (Size k = j; k < dates; ++k) {
444 Date d2 = cube_->dates()[k];
445 Date prevDate = k == 0 ? today : dateVector[k - 1];
446
447 eee_kva_1 = std::max(eee_kva_1, epe[k + 1]);
448 eee_kva_2 = std::max(eee_kva_2, ene[k + 1]);
449
450 // Components of the KVA maturity adjustment MA as of time j
451 if (dc.yearFraction(d1, d2) > 1.0) {
452 effMatNumer1 += epe[k + 1] * dc.yearFraction(prevDate, d2);
453 effMatNumer2 += ene[k + 1] * dc.yearFraction(prevDate, d2);
454 }
455 if (dc.yearFraction(d1, d2) <= 1.0) {
456 effMatDenom1 += eee_kva_1 * dc.yearFraction(prevDate, d2);
457 effMatDenom2 += eee_kva_2 * dc.yearFraction(prevDate, d2);
458 }
459
460 if (k < kmax) {
461 Real dt = dc.yearFraction(cube_->dates()[k], cube_->dates()[k + 1]);
462 sumdt += dt;
463 Real epe_b = epe[k + 1] / discountCurve->discount(dateVector[k]);
464 Real ene_b = ene[k + 1] / discountCurve->discount(dateVector[k]);
465 eee1_b = std::max(epe_b, eee1_b);
466 eee2_b = std::max(ene_b, eee2_b);
467 eepe_kva_1 += eee1_b * dt;
468 eepe_kva_2 += eee2_b * dt;
469 count++;
470 }
471 }
472
473 // Normalize EEPE/EENE calculation
474 eepe_kva_1 = count > 0 ? eepe_kva_1 / sumdt : 0.0;
475 eepe_kva_2 = count > 0 ? eepe_kva_2 / sumdt : 0.0;
476
477 // KVA CCR using the IRB risk weighted asset method and IMM:
478 // KVA effective maturity of the nettingSet, capped at 5
479 Real kvaNWMaturity1 = std::min(1.0 + (effMatDenom1 == 0.0 ? 0.0 : effMatNumer1 / effMatDenom1), 5.0);
480 Real kvaNWMaturity2 = std::min(1.0 + (effMatDenom2 == 0.0 ? 0.0 : effMatNumer2 / effMatDenom2), 5.0);
481
482 // Maturity adjustment factor for the RWA method:
483 // MA(PD, M) = (1 + (M - 2.5) * B(PD)) / (1 - 1.5 * B(PD)), capped at 5, floored at 1, M = effective
484 // maturity
485 Real kvaMatAdj1 =
486 std::max(std::min((1.0 + (kvaNWMaturity1 - 2.5) * kvaMatAdjB1) / (1.0 - 1.5 * kvaMatAdjB1), 5.0), 1.0);
487 Real kvaMatAdj2 =
488 std::max(std::min((1.0 + (kvaNWMaturity2 - 2.5) * kvaMatAdjB2) / (1.0 - 1.5 * kvaMatAdjB2), 5.0), 1.0);
489
490 // CCR Capital: RC = EAD x LGD x PD99.9 x MA(PD, M); EAD = alpha x EEPE(t) (approximated by EPE here);
491 Real kvaRC1 = kvaAlpha_ * eepe_kva_1 * LGD1 * kva99PD1 * kvaMatAdj1;
492 Real kvaRC2 = kvaAlpha_ * eepe_kva_2 * LGD2 * kva99PD2 * kvaMatAdj2;
493
494 // Expected risk capital discounted at capital discount rate
495 Real kvaCapitalDiscount = 1 / std::pow(1 + kvaCapitalDiscountRate_, dc.yearFraction(today, d0));
496 Real kvaCCRIncrement1 =
497 kvaRC1 * kvaCapitalDiscount * dc.yearFraction(d0, d1) * kvaCapitalHurdle_ * kvaRegAdjustment_;
498 Real kvaCCRIncrement2 =
499 kvaRC2 * kvaCapitalDiscount * dc.yearFraction(d0, d1) * kvaCapitalHurdle_ * kvaRegAdjustment_;
500
501 ourNettingSetKVACCR_[nettingSetId] += kvaCCRIncrement1;
502 theirNettingSetKVACCR_[nettingSetId] += kvaCCRIncrement2;
503
504 DLOG("Our KVA-CCR for " << nettingSetId << ": " << j << " EEPE=" << setprecision(2) << eepe_kva_1
505 << " EPE=" << epe[j] << " RC=" << kvaRC1 << " M=" << setprecision(6)
506 << kvaNWMaturity1 << " MA=" << kvaMatAdj1 << " Cost=" << setprecision(2)
507 << kvaCCRIncrement1 << " KVA=" << ourNettingSetKVACCR_[nettingSetId]);
508 DLOG("Their KVA-CCR for " << nettingSetId << ": " << j << " EENE=" << eepe_kva_2 << " ENE=" << ene[j]
509 << " RC=" << kvaRC2 << " M=" << setprecision(6) << kvaNWMaturity2
510 << " MA=" << kvaMatAdj2 << " Cost=" << setprecision(2) << kvaCCRIncrement2
511 << " KVA=" << theirNettingSetKVACCR_[nettingSetId]);
512
513 // CVA Capital
514 // effective maturity without cap at 5, DF set to 1 for IMM banks
515 // TODO: Set MA in CCR capital calculation to 1
516 Real kvaCvaMaturity1 = 1.0 + (effMatDenom1 == 0.0 ? 0.0 : effMatNumer1 / effMatDenom1);
517 Real kvaCvaMaturity2 = 1.0 + (effMatDenom2 == 0.0 ? 0.0 : effMatNumer2 / effMatDenom2);
518 Real scva1 = kvaTheirCvaRiskWeight_ * kvaCvaMaturity1 * eepe_kva_1;
519 Real scva2 = kvaOurCvaRiskWeight_ * kvaCvaMaturity2 * eepe_kva_2;
520 Real kvaCVAIncrement1 =
521 scva1 * kvaCapitalDiscount * dc.yearFraction(d0, d1) * kvaCapitalHurdle_ * kvaRegAdjustment_;
522 Real kvaCVAIncrement2 =
523 scva2 * kvaCapitalDiscount * dc.yearFraction(d0, d1) * kvaCapitalHurdle_ * kvaRegAdjustment_;
524
525 DLOG("Our KVA-CVA for " << nettingSetId << ": " << j << " EEPE=" << eepe_kva_1 << " SCVA=" << scva1
526 << " Cost=" << kvaCVAIncrement1);
527 DLOG("Their KVA-CVA for " << nettingSetId << ": " << j << " EENE=" << eepe_kva_2 << " SCVA=" << scva2
528 << " Cost=" << kvaCVAIncrement2);
529
530 ourNettingSetKVACVA_[nettingSetId] += kvaCVAIncrement1;
531 theirNettingSetKVACVA_[nettingSetId] += kvaCVAIncrement2;
532 }
533 }

535 LOG("Update netting set KVA done");

538void PostProcess::updateNettingSetCvaSensitivity() {
539
540 if (!analytics_["cvaSensi"])
541 return;
542
543 LOG("Update netting set CVA sensitivities");
544
545 Handle<YieldTermStructure> discountCurve = market_->discountCurve(baseCurrency_, configuration_);
546
547 for (auto n : netEPE_) {

549 vector<Real> epe = netEPE_[nettingSetId];
550 string cid;

552 Real cvaRR;
553 if (analytics_["flipViewXVA"]) {

555 } else {
556 cid = nettedExposureCalculator_->counterparty(nettingSetId);

558 cvaDts = market_->defaultCurve(cid)->curve();
559 QL_REQUIRE(!cvaDts.empty(), "Default curve missing for counterparty " << cid);

561
562 bool cvaSensi = analytics_["cvaSensi"];
563 LOG("CVA Sensitivity: " << cvaSensi);
564 if (cvaSensi) {
565 boost::shared_ptr<CVASpreadSensitivityCalculator> cvaSensiCalculator = boost::make_shared<CVASpreadSensitivityCalculator>(
566 nettingSetId, market_->asofDate(), epe, cube_->dates(), cvaDts, cvaRR, discountCurve, cvaSpreadSensiGrid_, cvaSpreadSensiShiftSize_);
567
568 for (Size i = 0; i < cvaSensiCalculator->shiftTimes().size(); ++i) {
569 DLOG("CVA Sensi Calculator: t=" << cvaSensiCalculator->shiftTimes()[i]
570 << " h=" << cvaSensiCalculator->hazardRateSensitivities()[i]
571 << " s=" << cvaSensiCalculator->cdsSpreadSensitivities()[i]);
572 }
573
574 netCvaHazardRateSensi_[nettingSetId] = cvaSensiCalculator->hazardRateSensitivities();
575 netCvaSpreadSensi_[nettingSetId] = cvaSensiCalculator->cdsSpreadSensitivities();
576 cvaSpreadSensiTimes_ = cvaSensiCalculator->shiftTimes();
577 } else {
578 cvaSpreadSensiTimes_ = vector<Real>();
579 netCvaHazardRateSensi_[nettingSetId] = vector<Real>();
580 netCvaSpreadSensi_[nettingSetId] = vector<Real>();
581 }
582 }
583
584 LOG("Update netting set CVA sensitivities done");
