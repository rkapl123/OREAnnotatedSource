47 AllocationMethod parseAllocationMethod(const string& s) {
48  static map<string, AllocationMethod> m = {
49  {"None", AllocationMethod::None},
50  {"Marginal", AllocationMethod::Marginal},
51  {"RelativeFairValueGross", AllocationMethod::RelativeFairValueGross},
52  {"RelativeFairValueNet", AllocationMethod::RelativeFairValueNet},
53  {"RelativeXVA", AllocationMethod::RelativeXVA},
54  };
55 
56  auto it = m.find(s);
57  if (it != m.end()) {
58  return it->second;
59  } else {
60  QL_FAIL("AllocationMethod \"" << s << "\" not recognized");
61  }
62 }
63 
64 std::ostream& operator<<(std::ostream& out, AllocationMethod m) {
65  if (m == AllocationMethod::None)
66  out << "None";
67  else if (m == AllocationMethod::Marginal)
68  out << "Marginal";

72  out << "RelativeFairValueNet";
73  else if (m == AllocationMethod::RelativeXVA)
74  out << "RelativeXVA";
75  else
76  QL_FAIL("Allocation method not covered");
77  return out;
78 }

527  QL_REQUIRE(nettingSetNegativeValueToday[nid] > 0.0, "non-zero negative NPV expected");

536  netEPE_[nid][j] * tradeValueToday[tid] / nettingSetValueToday[nid];
537  allocatedTradeENE_[tid][j + 1] =
538  netENE_[nid][j] * tradeValueToday[tid] / nettingSetValueToday[nid];
539  } else if (allocationMethod == AllocationMethod::RelativeXVA) {

564 void PostProcess::updateNettingSetKVA() {
565  // Loop over all netting sets
566  for (auto n : netEPE_) {
567  string nettingSetId = n.first;
568  // Init results
569  ourNettingSetKVACCR_[nettingSetId] = 0.0;
570  theirNettingSetKVACCR_[nettingSetId] = 0.0;
571  ourNettingSetKVACVA_[nettingSetId] = 0.0;
572  theirNettingSetKVACVA_[nettingSetId] = 0.0;
573  }
574 
575  if (!analytics_["kva"])
576  return;
577 
578  vector<Date> dateVector = cube_->dates();
579  Size dates = dateVector.size();
580  Date today = market_->asofDate();
581  Handle<YieldTermStructure> discountCurve = market_->discountCurve(baseCurrency_, configuration_);
582  DayCounter dc = ActualActual();
583 
584  // Loop over all netting sets
585  for (auto n : netEPE_) {
586  string nettingSetId = n.first;
587  string cid = counterpartyId_[nettingSetId];
588  LOG("KVA for netting set " << nettingSetId);
589 
590  // Main input are the EPE and ENE profiles, previously computed
591  vector<Real> epe = netEPE_[nettingSetId];
592  vector<Real> ene = netENE_[nettingSetId];
593 
594  // PD from counterparty Dts, floored to avoid 0 ...
595  // Today changed to today+1Y to get the one-year PD
596  Handle<DefaultProbabilityTermStructure> cvaDts = market_->defaultCurve(cid, configuration_);
597  QL_REQUIRE(!cvaDts.empty(), "Default curve missing for counterparty " << cid);
598  Real cvaRR = market_->recoveryRate(cid, configuration_)->value();
599  Real PD1 = std::max(cvaDts->defaultProbability(today + 1 * Years), 0.000000000001);
600  Real LGD1 = (1 - cvaRR);
601 
602  Handle<DefaultProbabilityTermStructure> dvaDts;
603  Real dvaRR = 0.0;
604  Real PD2 = 0;
605  if (dvaName_ != "") {
606  dvaDts = market_->defaultCurve(dvaName_, configuration_);
607  dvaRR = market_->recoveryRate(dvaName_, configuration_)->value();
608  PD2 = std::max(dvaDts->defaultProbability(today + 1 * Years), 0.000000000001);
609  }
610  else {
611  ALOG("dvaName not specified, own PD set to zero for their KVA calculation");
612  }
613  Real LGD2 = (1 - dvaRR);
614 
615  // Granularity adjustment, Gordy (2004):
616  Real rho1 = 0.12 * (1 - std::exp(-50 * PD1)) / (1 - std::exp(-50)) +
617  0.24 * (1 - (1 - std::exp(-50 * PD1)) / (1 - std::exp(-50)));
618  Real rho2 = 0.12 * (1 - std::exp(-50 * PD2)) / (1 - std::exp(-50)) +
619  0.24 * (1 - (1 - std::exp(-50 * PD2)) / (1 - std::exp(-50)));
620 
621  // Basel II internal rating based (IRB) estimate of worst case PD:
622  // Large homogeneous pool (LHP) approximation of Vasicek (1997)
623  InverseCumulativeNormal icn;
624  CumulativeNormalDistribution cnd;
625  Real PD99_1 = cnd((icn(PD1) + std::sqrt(rho1) * icn(0.999)) / (std::sqrt(1 - rho1))) - PD1;
626  Real PD99_2 = cnd((icn(PD2) + std::sqrt(rho2) * icn(0.999)) / (std::sqrt(1 - rho2))) - PD2;
627 
628  // KVA regulatory PD, worst case PD, floored at 0.03 for corporates and banks, not floored for sovereigns
629  Real kva99PD1 = std::max(PD99_1, kvaTheirPdFloor_);
630  Real kva99PD2 = std::max(PD99_2, kvaOurPdFloor_);
631 
632  // Factor B(PD) for the maturity adjustment factor, B(PD) = (0.11852 - 0.05478 * ln(PD)) ^ 2
633  Real kvaMatAdjB1 = std::pow((0.11852 - 0.05478 * std::log(PD1)), 2.0);
634  Real kvaMatAdjB2 = std::pow((0.11852 - 0.05478 * std::log(PD2)), 2.0);
635 
636  DLOG("Our KVA-CCR " << nettingSetId << ": PD=" << PD1);
637  DLOG("Our KVA-CCR " << nettingSetId << ": LGD=" << LGD1);
638  DLOG("Our KVA-CCR " << nettingSetId << ": rho=" << rho1);
639  DLOG("Our KVA-CCR " << nettingSetId << ": PD99=" << PD99_1);
640  DLOG("Our KVA-CCR " << nettingSetId << ": PD Floor=" << kvaTheirPdFloor_);
641  DLOG("Our KVA-CCR " << nettingSetId << ": Floored PD99=" << kva99PD1);
642  DLOG("Our KVA-CCR " << nettingSetId << ": B(PD)=" << kvaMatAdjB1);
643 
644  DLOG("Their KVA-CCR " << nettingSetId << ": PD=" << PD2);
645  DLOG("Their KVA-CCR " << nettingSetId << ": LGD=" << LGD2);
646  DLOG("Their KVA-CCR " << nettingSetId << ": rho=" << rho2);
647  DLOG("Their KVA-CCR " << nettingSetId << ": PD99=" << PD99_2);
648  DLOG("Their KVA-CCR " << nettingSetId << ": PD Floor=" << kvaOurPdFloor_);
649  DLOG("Their KVA-CCR " << nettingSetId << ": Floored PD99=" << kva99PD2);
650  DLOG("Their KVA-CCR " << nettingSetId << ": B(PD)=" << kvaMatAdjB2);
651 
652  for (Size j = 0; j < dates; ++j) {
653  Date d0 = j == 0 ? today : cube_->dates()[j - 1];
654  Date d1 = cube_->dates()[j];
655 
656  // Preprocess:
657  // 1) Effective maturity from effective expected exposure as of time j
658  // Index _1 corresponds to our perspective, index _2 to their perspective.
659  // 2) Basel EEPE as of time j, i.e. as time averge over EEE, starting at time j
660  // More accuracy may be achieved here by using a Longstaff-Schwartz method / regression
661  Real eee_kva_1 = 0.0, eee_kva_2 = 0.0;
662  Real effMatNumer1 = 0.0, effMatNumer2 = 0.0;
663  Real effMatDenom1 = 0.0, effMatDenom2 = 0.0;
664  Real eepe_kva_1 = 0, eepe_kva_2 = 0.0;
665  Size kmax = j, count = 0;
666  // Cut off index for EEPE/EENE calculation: One year ahead
667  while (dateVector[kmax] < dateVector[j] + 1 * Years + 4 * Days && kmax < dates - 1)
668  kmax ++;
669  Real sumdt = 0.0, eee1_b = 0.0, eee2_b = 0.0;
670  for (Size k = j; k < dates; ++k) {
671  Date d2 = cube_->dates()[k];
672  Date prevDate = k == 0 ? today : dateVector[k - 1];
673 
674  eee_kva_1 = std::max(eee_kva_1, epe[k + 1]);
675  eee_kva_2 = std::max(eee_kva_2, ene[k + 1]);
676 
677  // Components of the KVA maturity adjustment MA as of time j
678  if (dc.yearFraction(d1, d2) > 1.0) {
679  effMatNumer1 += epe[k + 1] * dc.yearFraction(prevDate, d2);
680  effMatNumer2 += ene[k + 1] * dc.yearFraction(prevDate, d2);
681  }
682  if (dc.yearFraction(d1, d2) <= 1.0) {
683  effMatDenom1 += eee_kva_1 * dc.yearFraction(prevDate, d2);
684  effMatDenom2 += eee_kva_2 * dc.yearFraction(prevDate, d2);
685  }
686 
687  if (k < kmax) {
688  Real dt = dc.yearFraction(cube_->dates()[k], cube_->dates()[k+1]);
689  sumdt += dt;
690  Real epe_b = epe[k + 1] / discountCurve->discount(dateVector[k]);
691  Real ene_b = ene[k + 1] / discountCurve->discount(dateVector[k]);
692  eee1_b = std::max(epe_b, eee1_b);
693  eee2_b = std::max(ene_b, eee2_b);
694  eepe_kva_1 += eee1_b * dt;
695  eepe_kva_2 += eee2_b * dt;
696  count ++;
697  }
698  }
699 
700  // Normalize EEPE/EENE calculation
701  eepe_kva_1 = count > 0 ? eepe_kva_1/sumdt : 0.0;
702  eepe_kva_2 = count > 0 ? eepe_kva_2/sumdt : 0.0;
703 
704  // KVA CCR using the IRB risk weighted asset method and IMM:
705  // KVA effective maturity of the nettingSet, capped at 5
706  Real kvaNWMaturity1 = std::min(1.0 + (effMatDenom1 == 0.0 ? 0.0 : effMatNumer1 / effMatDenom1), 5.0);
707  Real kvaNWMaturity2 = std::min(1.0 + (effMatDenom2 == 0.0 ? 0.0 : effMatNumer2 / effMatDenom2), 5.0);
708 
709  // Maturity adjustment factor for the RWA method:
710  // MA(PD, M) = (1 + (M - 2.5) * B(PD)) / (1 - 1.5 * B(PD)), capped at 5, floored at 1, M = effective maturity
711  Real kvaMatAdj1 =
712  std::max(std::min((1.0 + (kvaNWMaturity1 - 2.5) * kvaMatAdjB1) / (1.0 - 1.5 * kvaMatAdjB1), 5.0), 1.0);
713  Real kvaMatAdj2 =
714  std::max(std::min((1.0 + (kvaNWMaturity2 - 2.5) * kvaMatAdjB2) / (1.0 - 1.5 * kvaMatAdjB2), 5.0), 1.0);
715 
716  // CCR Capital: RC = EAD x LGD x PD99.9 x MA(PD, M); EAD = alpha x EEPE(t) (approximated by EPE here);
717  Real kvaRC1 = kvaAlpha_ * eepe_kva_1 * LGD1 * kva99PD1 * kvaMatAdj1;
718  Real kvaRC2 = kvaAlpha_ * eepe_kva_2 * LGD2 * kva99PD2 * kvaMatAdj2;
719 
720  // Expected risk capital discounted at capital discount rate
721  Real kvaCapitalDiscount = 1 / std::pow(1 + kvaCapitalDiscountRate_, dc.yearFraction(today, d0));
722  Real kvaCCRIncrement1 = kvaRC1 * kvaCapitalDiscount * dc.yearFraction(d0, d1) * kvaCapitalHurdle_ * kvaRegAdjustment_;
723  Real kvaCCRIncrement2 = kvaRC2 * kvaCapitalDiscount * dc.yearFraction(d0, d1) * kvaCapitalHurdle_ * kvaRegAdjustment_;
724 
725  ourNettingSetKVACCR_[nettingSetId] += kvaCCRIncrement1;
726  theirNettingSetKVACCR_[nettingSetId] += kvaCCRIncrement2;
727 
728  DLOG("Our KVA-CCR for " << nettingSetId << ": " << j
729  << " EEPE=" << setprecision(2) << eepe_kva_1
730  << " EPE=" << epe[j]
731  << " RC=" << kvaRC1
732  << " M=" << kvaNWMaturity1
733  << " MA=" << kvaMatAdj1
734  << " Cost=" << kvaCCRIncrement1
735  << " KVA=" << ourNettingSetKVACCR_[nettingSetId]);
736  DLOG("Their KVA-CCR for " << nettingSetId << ": " << j
737  << " EENE=" << eepe_kva_2
738  << " ENE=" << ene[j]
739  << " RC=" << kvaRC2
740  << " M=" << kvaNWMaturity1
741  << " MA=" << kvaMatAdj1
742  << " Cost=" << kvaCCRIncrement2
743  << " KVA=" << theirNettingSetKVACCR_[nettingSetId]);
744 
745  // CVA Capital
746  // effective maturity without cap at 5, DF set to 1 for IMM banks
747  // TODO: Set MA in CCR capital calculation to 1
748  Real kvaCvaMaturity1 = 1.0 + (effMatDenom1 == 0.0 ? 0.0 : effMatNumer1 / effMatDenom1);
749  Real kvaCvaMaturity2 = 1.0 + (effMatDenom2 == 0.0 ? 0.0 : effMatNumer2 / effMatDenom2);
750  Real scva1 = kvaTheirCvaRiskWeight_ * kvaCvaMaturity1 * eepe_kva_1;
751  Real scva2 = kvaOurCvaRiskWeight_ * kvaCvaMaturity2 * eepe_kva_2;
752  Real kvaCVAIncrement1 = scva1 * kvaCapitalDiscount * dc.yearFraction(d0, d1) * kvaCapitalHurdle_ * kvaRegAdjustment_;
753  Real kvaCVAIncrement2 = scva2 * kvaCapitalDiscount * dc.yearFraction(d0, d1) * kvaCapitalHurdle_ * kvaRegAdjustment_;
754 
755  DLOG("Our KVA-CVA for " << nettingSetId << ": " << j
756  << " EEPE=" << eepe_kva_1
757  << " SCVA=" << scva1
758  << " Cost=" << kvaCVAIncrement1);
759  DLOG("Their KVA-CVA for " << nettingSetId << ": " << j
760  << " EENE=" << eepe_kva_2
761  << " SCVA=" << scva2
762  << " Cost=" << kvaCVAIncrement2);
763 
764  ourNettingSetKVACVA_[nettingSetId] += kvaCVAIncrement1;
765  theirNettingSetKVACVA_[nettingSetId] += kvaCVAIncrement2;
766  }
767  }

939  if (fvaLendingCurve_ != "")

965  Real fcaIncrement_exOwnSP = cvaS0 * borrowingSpreadDcf * epe[j + 1];
966  Real fcaIncrement_exAllSP = borrowingSpreadDcf * epe[j + 1];

974  oisCurve->discount(d0) / oisCurve->discount(d1);
975  Real fbaIncrement = cvaS0 * dvaS0 * lendingSpreadDcf * ene[j + 1];
976  Real fbaIncrement_exOwnSP = cvaS0 * lendingSpreadDcf * ene[j + 1];
977  Real fbaIncrement_exAllSP = lendingSpreadDcf * ene[j + 1];

1046 void PostProcess::dynamicInitialMargin() {
1047  LOG("DIM Analysis by regression");
1048 
1049  Date today = market_->asofDate();
1050  Size dates = cube_->dates().size();
1051  Size samples = cube_->samples();
1052  map<string, Size> nettingSetSize;
1053  set<string> nettingSets;
1054 
1055  // initialise aggregate NPV and Flow by date and scenario
1056  for (Size i = 0; i < portfolio_->size(); ++i) {
1057  string tradeId = portfolio_->trades()[i]->id();
1058  string nettingSetId = portfolio_->trades()[i]->envelope().nettingSetId();
1059  LOG("Aggregate exposure for trade " << tradeId);
1060  if (nettingSets.find(nettingSetId) == nettingSets.end()) {
1061  nettingSetNPV_[nettingSetId] = vector<vector<Real>>(dates, vector<Real>(samples, 0.0));
1062  nettingSetFLOW_[nettingSetId] = vector<vector<Real>>(dates, vector<Real>(samples, 0.0));
1063  nettingSetDIM_[nettingSetId] = vector<vector<Real>>(dates, vector<Real>(samples, 0.0));
1064  nettingSetDeltaNPV_[nettingSetId] = vector<vector<Real>>(dates, vector<Real>(samples, 0.0));
1065  regressorArray_[nettingSetId] = vector<vector<Array>>(dates, vector<Array>(samples));
1066  nettingSetLocalDIM_[nettingSetId] = vector<vector<Real>>(dates, vector<Real>(samples, 0.0));
1067  nettingSetExpectedDIM_[nettingSetId] = vector<Real>(dates, 0.0);
1068  nettingSetZeroOrderDIM_[nettingSetId] = vector<Real>(dates, 0.0);
1069  nettingSetSimpleDIMh_[nettingSetId] = vector<Real>(dates, 0.0);
1070  nettingSetSimpleDIMp_[nettingSetId] = vector<Real>(dates, 0.0);
1071  nettingSets.insert(nettingSetId);
1072  nettingSetSize[nettingSetId] = 0;
1073  }
1074  nettingSetSize[nettingSetId]++;
1075 
1076  for (Size j = 0; j < dates; ++j) {
1077  for (Size k = 0; k < samples; ++k) {
1078  Real npv = cube_->get(i, j, k);
1079  QL_REQUIRE(cube_->depth() > 1, "cube depth > 1 expected for DIM, found depth " << cube_->depth());
1080  Real flow = cube_->get(i, j, k, 1);
1081  nettingSetNPV_[nettingSetId][j][k] += npv;
1082  nettingSetFLOW_[nettingSetId][j][k] += flow;
1083  }
1084  }
1085  }
1086 
1087  vector<string> nettingSetIds;
1088  for (auto n : nettingSets)
1089  nettingSetIds.push_back(n);
1090 
1091  // Perform the T0 calculation
1092  performT0DimCalc();
1093 
1094  // This is allocated here and not outside the post processor because we determine the dimension (netting sets) here
1095  dimCube_ = boost::make_shared<SinglePrecisionInMemoryCube>(today, nettingSetIds, cube_->dates(), samples);
1096 
1097  Size polynomOrder = dimRegressionOrder_;
1098  LOG("DIM regression polynom order = " << dimRegressionOrder_);
1099  LsmBasisSystem::PolynomType polynomType = LsmBasisSystem::Monomial;
1100  Size regressionDimension = dimRegressors_.empty() ? 1 : dimRegressors_.size();
1101  LOG("DIM regression dimension = " << regressionDimension);
1102 #if QL_HEX_VERSION > 0x01150000
1103  std::vector<ext::function<Real(Array)>> v(
1104 #else // QL 1.14 and below
1105  std::vector<boost::function1<Real, Array>> v(
1106 #endif
1107  LsmBasisSystem::multiPathBasisSystem(regressionDimension, polynomOrder, polynomType));
1108  Real confidenceLevel = QuantLib::InverseCumulativeNormal()(dimQuantile_);
1109  LOG("DIM confidence level " << confidenceLevel);
1110 
1111  Size simple_dim_index_h = Size(floor(dimQuantile_ * (samples - 1) + 0.5));
1112  Size simple_dim_index_p = Size(floor((1.0 - dimQuantile_) * (samples - 1) + 0.5));
1113 
1114  Size nettingSetCount = 0;
1115  for (auto n : nettingSets) {
1116  LOG("Process netting set " << n);
1117  // Set last date's IM to zero for all samples
1118  for (Size k = 0; k < samples; ++k) {
1119  nettingSetDIM_[n][dates - 1][k] = 0.0;
1120  nettingSetLocalDIM_[n][dates - 1][k] = 0.0;
1121  nettingSetDeltaNPV_[n][dates - 1][k] = 0.0;
1122  }
1123  for (Size j = 0; j < dates - 1; ++j) {
1124  accumulator_set<double, stats<tag::mean, tag::variance>> accDiff;
1125  accumulator_set<double, stats<tag::mean>> accOneOverNumeraire;
1126  for (Size k = 0; k < samples; ++k) {
1127  Real num1 = scenarioData_->get(j, k, AggregationScenarioDataType::Numeraire);
1128  Real num2 = scenarioData_->get(j + 1, k, AggregationScenarioDataType::Numeraire);
1129  Real npv1 = nettingSetNPV_[n][j][k];
1130  Real flow = nettingSetFLOW_[n][j][k];
1131  Real npv2 = nettingSetNPV_[n][j + 1][k];
1132  accDiff(npv2 * num2 + flow * num1 - npv1 * num1);
1133  accOneOverNumeraire(1.0 / num1);
1134  }
1135 
1136  Date d1 = cube_->dates()[j];
1137  Date d2 = cube_->dates()[j + 1];
1138  Real horizonScaling = sqrt(1.0 * dimHorizonCalendarDays_ / (d2 - d1));
1139  Real stdevDiff = sqrt(variance(accDiff));
1140  Real E_OneOverNumeraire =
1141  mean(accOneOverNumeraire); // "re-discount" (the stdev is calculated on non-discounted deltaNPVs)
1142 
1143  nettingSetZeroOrderDIM_[n][j] = stdevDiff * horizonScaling * confidenceLevel;
1144  nettingSetZeroOrderDIM_[n][j] *= E_OneOverNumeraire;
1145 
1146  vector<Real> rx0(samples, 0.0);
1147  vector<Array> rx(samples, Array());
1148  vector<Real> ry1(samples, 0.0);
1149  vector<Real> ry2(samples, 0.0);
1150  for (Size k = 0; k < samples; ++k) {
1151  Real num1 = scenarioData_->get(j, k, AggregationScenarioDataType::Numeraire);
1152  Real num2 = scenarioData_->get(j + 1, k, AggregationScenarioDataType::Numeraire);
1153  Real x = nettingSetNPV_[n][j][k] * num1;
1154  Real f = nettingSetFLOW_[n][j][k] * num1;
1155  Real y = nettingSetNPV_[n][j + 1][k] * num2;
1156  Real z = (y + f - x);
1157  rx[k] = dimRegressors_.empty() ? Array(1, nettingSetNPV_[n][j][k]) : regressorArray(n, j, k);
1158  rx0[k] = rx[k][0];
1159  ry1[k] = z; // for local regression
1160  ry2[k] = z * z; // for least squares regression
1161  nettingSetDeltaNPV_[n][j][k] = z;
1162  regressorArray_[n][j][k] = rx[k];
1163  }
1164  vector<Real> delNpvVec_copy = nettingSetDeltaNPV_[n][j];
1165  sort(delNpvVec_copy.begin(), delNpvVec_copy.end());
1166  Real simpleDim_h = delNpvVec_copy[simple_dim_index_h];
1167  Real simpleDim_p = delNpvVec_copy[simple_dim_index_p];
1168  simpleDim_h *= horizonScaling; // the usual scaling factors
1169  simpleDim_p *= horizonScaling; // the usual scaling factors
1170  nettingSetSimpleDIMh_[n][j] = simpleDim_h * E_OneOverNumeraire; // discounted DIM
1171  nettingSetSimpleDIMp_[n][j] = simpleDim_p * E_OneOverNumeraire; // discounted DIM
1172 
1173  QL_REQUIRE(rx.size() > v.size(), "not enough points for regression with polynom order " << polynomOrder);
1174  if (close_enough(stdevDiff, 0.0)) {
1175  LOG("DIM: Zero std dev estimation at step " << j);
1176  // Skip IM calculation if all samples have zero NPV (e.g. after latest maturity)
1177  for (Size k = 0; k < samples; ++k) {
1178  nettingSetDIM_[n][j][k] = 0.0;
1179  nettingSetLocalDIM_[n][j][k] = 0.0;
1180  }
1181  } else {
1182  // Least squares polynomial regression with specified polynom order
1183  QuantExt::StabilisedGLLS ls(rx, ry2, v, QuantExt::StabilisedGLLS::MeanStdDev);
1184  LOG("DIM data normalisation at time step "
1185  << j << ": " << scientific << setprecision(6) << " x-shift = " << ls.xShift() << " x-multiplier = "
1186  << ls.xMultiplier() << " y-shift = " << ls.yShift() << " y-multiplier = " << ls.yMultiplier());
1187  LOG("DIM regression coefficients at time step " << j << ": " << fixed << setprecision(6)
1188  << ls.transformedCoefficients());
1189 
1190  // Local regression versus first regression variable (i.e. we do not perform a
1191  // multidimensional local regression):
1192  // We evaluate this at a limited number of samples only for validation purposes.
1193  // Note that computational effort scales quadratically with number of samples.
1194  // NadarayaWatson needs a large number of samples for good results.
1195  QuantExt::NadarayaWatson lr(rx0.begin(), rx0.end(), ry1.begin(),
1196  GaussianKernel(0.0, dimLocalRegressionBandwidth_));
1197  Size localRegressionSamples = samples;
1198  if (dimLocalRegressionEvaluations_ > 0)
1199  localRegressionSamples = Size(floor(1.0 * samples / dimLocalRegressionEvaluations_ + .5));
1200 
1201  // Evaluate regression function to compute DIM for each scenario
1202  for (Size k = 0; k < samples; ++k) {
1203  Real num1 = scenarioData_->get(j, k, AggregationScenarioDataType::Numeraire);
1204  Array regressor =
1205  dimRegressors_.empty() ? Array(1, nettingSetNPV_[n][j][k]) : regressorArray(n, j, k);
1206  Real e = ls.eval(regressor, v);
1207  if (e < 0.0)
1208  LOG("Negative variance regression for date " << j << ", sample " << k
1209  << ", regressor = " << regressor);
1210 
1211  // Note:
1212  // 1) We assume vanishing mean of "z", because the drift over a MPOR is usually small,
1213  // and to avoid a second regression for the conditional mean
1214  // 2) In particular the linear regression function can yield negative variance values in
1215  // extreme scenarios where an exact analytical or delta VaR calculation would yield a
1216  // variance aproaching zero. We correct this here by taking the positive part.
1217  Real std = sqrt(std::max(e, 0.0));
1218  Real scalingFactor = horizonScaling * confidenceLevel * dimScaling_;
1219  Real dim = std * scalingFactor / num1;
1220  dimCube_->set(dim, nettingSetCount, j, k);
1221  nettingSetDIM_[n][j][k] = dim;
1222  nettingSetExpectedDIM_[n][j] += dim / samples;
1223 
1224  // Evaluate the Kernel regression for a subset of the samples only (performance)
1225  if (k % localRegressionSamples == 0)
1226  nettingSetLocalDIM_[n][j][k] = lr.standardDeviation(regressor[0]) * scalingFactor / num1;
1227  else
1228  nettingSetLocalDIM_[n][j][k] = 0.0;
1229  }
1230  }
1231  }
1232 
1233  nettingSetCount++;
1234  }
1235  LOG("DIM by regression done");
1236 }

1492 void PostProcess::performT0DimCalc() {
1493  // In this function we proxy the model-implied T0 IM by looking at the
1494  // cube grid horizon lying closest to t0+mpor. We measure diffs relative
1495  // to the mean of the distribution at this same time horizon, thus avoiding
1496  // any cashflow-specific jumps
1497 
1498  Date today = market_->asofDate();
1499  Size relevantDateIdx = 0;
1500  Real sqrtTimeScaling = 1.0;
1501  for (Size i = 0; i < cube_->dates().size(); ++i) {
1502  Size daysFromT0 = (cube_->dates()[i] - today);
1503  if (daysFromT0 < dimHorizonCalendarDays_) {
1504  // iterate until we straddle t0+mpor
1505  continue;
1506  } else if (daysFromT0 == dimHorizonCalendarDays_) {
1507  // this date corresponds to t0+mpor, so use it
1508  relevantDateIdx = i;
1509  sqrtTimeScaling = 1.0;
1510  break;
1511  } else if (daysFromT0 > dimHorizonCalendarDays_) {
1512  // the first date greater than t0+MPOR, check if it is closest
1513  Size lastIdx = (i == 0) ? 0 : (i - 1);
1514  Size lastDaysFromT0 = (cube_->dates()[lastIdx] - today);
1515  int daysFromT0CloseOut = daysFromT0 - dimHorizonCalendarDays_;
1516  int prevDaysFromT0CloseOut = lastDaysFromT0 - dimHorizonCalendarDays_;
1517  if (std::abs(daysFromT0CloseOut) <= std::abs(prevDaysFromT0CloseOut)) {
1518  relevantDateIdx = i;
1519  sqrtTimeScaling = std::sqrt(Real(dimHorizonCalendarDays_) / Real(daysFromT0));
1520  } else {
1521  relevantDateIdx = lastIdx;
1522  sqrtTimeScaling = std::sqrt(Real(dimHorizonCalendarDays_) / Real(lastDaysFromT0));
1523  }
1524  break;
1525  }
1526  }
1527  // set some reasonable bounds on the sqrt time scaling, so that we are not looking at a ridiculous time horizon
1528  QL_REQUIRE((sqrtTimeScaling * sqrtTimeScaling >= 0.5) && (sqrtTimeScaling * sqrtTimeScaling <= 2.0),
1529  "T0 IM Estimation - The estimation time horizon from grid "
1530  << "is not sufficiently close to t0+MPOR - "
1531  << QuantLib::io::iso_date(cube_->dates()[relevantDateIdx]));
1532 
1533  // TODO: Ensure that the simulation containers read-from below are indeed populated
1534 
1535  Real confidenceLevel = QuantLib::InverseCumulativeNormal()(dimQuantile_);
1536  Size simple_dim_index_h = Size(floor(dimQuantile_ * (cube_->samples() - 1) + 0.5));
1537  for (auto it_map = nettingSetNPV_.begin(); it_map != nettingSetNPV_.end(); ++it_map) {
1538  string key = it_map->first;
1539  boost::shared_ptr<NettingSetDefinition> nettingObj = nettingSetManager_->get(key);
1540  vector<Real> t0_dist = it_map->second[relevantDateIdx];
1541  Size dist_size = t0_dist.size();
1542  QL_REQUIRE(dist_size == cube_->samples(),
1543  "T0 IM - cube samples size mismatch - " << dist_size << ", " << cube_->samples());
1544  Real mean_t0_dist = std::accumulate(t0_dist.begin(), t0_dist.end(), 0.0);
1545  mean_t0_dist /= dist_size;
1546  vector<Real> t0_delMtM_dist(dist_size, 0.0);
1547  accumulator_set<double, stats<tag::mean, tag::variance>> acc_delMtm;
1548  accumulator_set<double, stats<tag::mean>> acc_OneOverNum;
1549  for (Size i = 0; i < dist_size; ++i) {
1550  Real numeraire = scenarioData_->get(relevantDateIdx, i, AggregationScenarioDataType::Numeraire);
1551  Real deltaMtmFromMean = numeraire * (t0_dist[i] - mean_t0_dist) * sqrtTimeScaling;
1552  t0_delMtM_dist[i] = deltaMtmFromMean;
1553  acc_delMtm(deltaMtmFromMean);
1554  acc_OneOverNum(1.0 / numeraire);
1555  }
1556  Real E_OneOverNumeraire = mean(acc_OneOverNum);
1557  Real variance_t0 = variance(acc_delMtm);
1558  Real sqrt_t0 = sqrt(variance_t0);
1559  net_t0_im_reg_h_[key] = (sqrt_t0 * confidenceLevel * E_OneOverNumeraire);
1560  std::sort(t0_delMtM_dist.begin(), t0_delMtM_dist.end());
1561  net_t0_im_simple_h_[key] = (t0_delMtM_dist[simple_dim_index_h] * E_OneOverNumeraire);
1562 
1563  LOG("T0 IM (Reg) - {" << key << "} = " << net_t0_im_reg_h_[key]);
1564  LOG("T0 IM (Simple) - {" << key << "} = " << net_t0_im_simple_h_[key]);
1565  }
1566  LOG("T0 IM Calculations Completed");
1567 }
