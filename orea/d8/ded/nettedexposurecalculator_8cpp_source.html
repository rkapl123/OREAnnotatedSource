66

72  market_->asofDate(), nettingSetIds, cube->dates(),
73  cube->samples(), EXPOSURE_CUBE_DEPTH); // EPE, ENE
74  } else {
75  exposureCube_ = boost::make_shared<DoublePrecisionInMemoryCubeN>(
76  market_->asofDate(), nettingSetIds, cube->dates(),

78  }

81 void NettedExposureCalculator::build() {
82  LOG("Compute netting set exposure profiles");
83 
84  const Date today = market_->asofDate();
85  const DayCounter dc = ActualActual();
86 
87  vector<Real> times = vector<Real>(cube_->dates().size(), 0.0);
88  for (Size i = 0; i < cube_->dates().size(); i++)
89  times[i] = dc.yearFraction(today, cube_->dates()[i]);
90 
91  map<string, Real> nettingSetValueToday;
92  map<string, Date> nettingSetMaturity;
93  map<string, Size> nettingSetSize;
94  for (Size i = 0; i < portfolio_->trades().size(); ++i) {
95  const auto& trade = portfolio_->trades()[i];
96  string tradeId = trade->id();
97  string nettingSetId = trade->envelope().nettingSetId();
98  string cp = trade->envelope().counterparty();
99  if (counterpartyMap_.find(nettingSetId) == counterpartyMap_.end())
100  counterpartyMap_[nettingSetId] = trade->envelope().counterparty();
101  else {
102  QL_REQUIRE(counterpartyMap_[nettingSetId] == cp, "counterparty name is not unique withint the netting set");
103  }
104  Real npv;
105  if (flipViewXVA_) {
106  npv = -cube_->getT0(i);
107  } else {
108  npv = cube_->getT0(i);
109  }
110 
111  if (nettingSetValueToday.find(nettingSetId) == nettingSetValueToday.end()) {
112  nettingSetValueToday[nettingSetId] = 0.0;
113  nettingSetMaturity[nettingSetId] = today;
114  nettingSetSize[nettingSetId] = 0;
115  }
116 
117  nettingSetValueToday[nettingSetId] += npv;
118 
119  if (trade->maturity() > nettingSetMaturity[nettingSetId])
120  nettingSetMaturity[nettingSetId] = trade->maturity();
121  nettingSetSize[nettingSetId]++;
122  }
123 
124  vector<vector<Real>> averagePositiveAllocation(portfolio_->size(), vector<Real>(cube_->dates().size(), 0.0));
125  vector<vector<Real>> averageNegativeAllocation(portfolio_->size(), vector<Real>(cube_->dates().size(), 0.0));
126 
127  map<string, vector<vector<Real>>> nettingSetValue = (calcType_ == CollateralExposureHelper::CalculationType::NoLag
128  ? nettingSetCloseOutValue_
129  : nettingSetDefaultValue_);
130  Size nettingSetCount = 0;
131  for (auto n : nettingSetValue) {
132  string nettingSetId = n.first;
133 
134  LOG("Aggregate exposure for netting set " << nettingSetId);
135  vector<vector<Real>> data = n.second;
136 
137  // Get the collateral account balance paths for the netting set.
138  // The pointer may remain empty if there is no CSA or if it is inactive.
139  boost::shared_ptr<vector<boost::shared_ptr<CollateralAccount>>> collateral =
140  collateralPaths(nettingSetId,
141  nettingSetValueToday[nettingSetId],
142  nettingSetDefaultValue_[nettingSetId],
143  nettingSetMaturity[nettingSetId]);
144 
145  // Get the CSA index for Eonia Floor calculation below
146  colva_[nettingSetId] = 0.0;
147  collateralFloor_[nettingSetId] = 0.0;
148  boost::shared_ptr<NettingSetDefinition> netting = nettingSetManager_->get(nettingSetId);
149  string csaIndexName;
150  Handle<IborIndex> csaIndex;
151  bool applyInitialMargin = false;
152  CSA::Type initialMarginType = CSA::Bilateral;
153  if (netting->activeCsaFlag()) {
154  csaIndexName = netting->csaDetails()->index();
155  if (csaIndexName != "") {
156  csaIndex = market_->iborIndex(csaIndexName);
157  QL_REQUIRE(scenarioData_->has(AggregationScenarioDataType::IndexFixing, csaIndexName),
158  "scenario data does not provide index values for " << csaIndexName);
159  }
160  QL_REQUIRE(netting->csaDetails(), "active CSA for netting set " << nettingSetId
161  << ", but CSA detailsd not initialised");
162  applyInitialMargin = netting->csaDetails()->applyInitialMargin() && applyInitialMargin_;
163  initialMarginType = netting->csaDetails()->initialMarginType();
164  LOG("ApplyInitialMargin=" << applyInitialMargin << " for netting set " << nettingSetId
165  << ", CSA IM=" << netting->csaDetails()->applyInitialMargin()
166  << ", CSA IM Type=" << initialMarginType
167  << ", Analytics DIM=" << applyInitialMargin_);
168  if (applyInitialMargin_ && !netting->csaDetails()->applyInitialMargin())
169  ALOG("ApplyInitialMargin deactivated at netting set level " << nettingSetId);
170  if (!applyInitialMargin_ && netting->csaDetails()->applyInitialMargin())
171  ALOG("ApplyInitialMargin deactivated in analytics, but active at netting set level " << nettingSetId);
172  }
173 
174  Handle<YieldTermStructure> curve = market_->discountCurve(baseCurrency_, configuration_);
175  vector<Real> epe(cube_->dates().size() + 1, 0.0);
176  vector<Real> ene(cube_->dates().size() + 1, 0.0);
177  vector<Real> ee_b(cube_->dates().size() + 1, 0.0);
178  vector<Real> eee_b(cube_->dates().size() + 1, 0.0);
179  vector<Real> eee_b_kva_1(cube_->dates().size() + 1, 0.0);
180  vector<Real> eee_b_kva_2(cube_->dates().size() + 1, 0.0);
181  vector<Real> eepe_b_kva_1(cube_->dates().size() + 1, 0.0);
182  vector<Real> eepe_b_kva_2(cube_->dates().size() + 1, 0.0);
183  vector<Real> eab(cube_->dates().size() + 1, 0.0);
184  vector<Real> pfe(cube_->dates().size() + 1, 0.0);
185  vector<Real> colvaInc(cube_->dates().size() + 1, 0.0);
186  vector<Real> eoniaFloorInc(cube_->dates().size() + 1, 0.0);
187  Real npv = nettingSetValueToday[nettingSetId];
188  if ((fullInitialCollateralisation_) & (netting->activeCsaFlag())) {
189  // This assumes that the collateral at t=0 is the same as the npv at t=0.
190  epe[0] = 0;
191  ene[0] = 0;
192  pfe[0] = 0;
193  } else {
194  epe[0] = std::max(npv, 0.0);
195  ene[0] = std::max(-npv, 0.0);
196  pfe[0] = std::max(npv, 0.0);
197  }
198  // The fullInitialCollateralisation flag doesn't affect the eab, which feeds into the "ExpectedCollateral"
199  // column of the 'exposure_nettingset_*' reports. We always assume the full collateral here.
200  eab[0] = -npv;
201  ee_b[0] = epe[0];
202  eee_b[0] = ee_b[0];
203  nettedCube_->setT0(npv, nettingSetCount);
204  exposureCube_->setT0(epe[0], nettingSetCount, ExposureIndex::EPE);
205  exposureCube_->setT0(ene[0], nettingSetCount, ExposureIndex::ENE);
206 
207  for (Size j = 0; j < cube_->dates().size(); ++j) {
208 
209  Date date = cube_->dates()[j];
210  Date prevDate = j > 0 ? cube_->dates()[j - 1] : today;
211  vector<Real> distribution(cube_->samples(), 0.0);
212  for (Size k = 0; k < cube_->samples(); ++k) {
213  Real balance = 0.0;
214  if (collateral) {
215  balance = collateral->at(k)->accountBalance(date);
216  if (netting->csaDetails()->csaCurrency() != baseCurrency_) {
217  // Convert from CSACurrency to baseCurrency
218  double fxRate = scenarioData_->get(j, k, AggregationScenarioDataType::FXSpot,
219  netting->csaDetails()->csaCurrency());
220  balance *= fxRate;
221  }
222  }
223  eab[j + 1] += balance / cube_->samples();
224  Real exposure = data[j][k] - balance;
225  Real dim = 0.0;
226  if (applyInitialMargin && collateral) { // don't apply initial margin without VM, i.e. inactive CSA
227  // Initial Margin
228  // Use IM to reduce exposure
229  // Size dimIndex = j == 0 ? 0 : j - 1;
230  Size dimIndex = j;
231  dim = dimCalculator_->dynamicIM(nettingSetId)[dimIndex][k];
232  QL_REQUIRE(dim >= 0, "negative DIM for set " << nettingSetId << ", date " << j << ", sample " << k
233  << ": " << dim);
234  }
235  Real dim_epe = 0;
236  Real dim_ene = 0;
237  if (initialMarginType != CSA::Type::PostOnly)
238  dim_epe = dim;
239  if (initialMarginType != CSA::Type::CallOnly)
240  dim_ene = dim;
241  epe[j + 1] += std::max(exposure - dim_epe, 0.0) /
242  cube_->samples(); // dim here represents the held IM, and is expressed as a positive number
243  ene[j + 1] += std::max(-exposure - dim_ene, 0.0) /
244  cube_->samples(); // dim here represents the posted IM, and is expressed as a positive number
245  distribution[k] = exposure;
246  nettedCube_->set(exposure, nettingSetCount, j, k);
247  if (multiPath_) {
248  exposureCube_->set(std::max(exposure - dim_epe, 0.0), nettingSetCount, j, k, ExposureIndex::EPE);
249  exposureCube_->set(std::max(-exposure - dim_ene, 0.0), nettingSetCount, j, k, ExposureIndex::ENE);
250  }
251 
252  if (netting->activeCsaFlag()) {
253  Real indexValue = 0.0;
254  DayCounter dc = ActualActual();
255  if (csaIndexName != "") {
256  indexValue = scenarioData_->get(j, k, AggregationScenarioDataType::IndexFixing, csaIndexName);
257  dc = csaIndex->dayCounter();
258  }
259  Real dcf = dc.yearFraction(prevDate, date);
260  Real collateralSpread = (balance >= 0.0 ? netting->csaDetails()->collatSpreadRcv() : netting->csaDetails()->collatSpreadPay());
261  Real numeraire = scenarioData_->get(j, k, AggregationScenarioDataType::Numeraire);
262  Real colvaDelta = -balance * collateralSpread * dcf / numeraire / cube_->samples();
263  // inutuitive floorDelta including collateralSpread would be:
264  // -balance * (max(indexValue - collateralSpread,0) - (indexValue - collateralSpread)) * dcf /
265  // samples
266  Real floorDelta = -balance * std::max(-(indexValue - collateralSpread), 0.0) * dcf / numeraire / cube_->samples();
267  colvaInc[j + 1] += colvaDelta;
268  colva_[nettingSetId] += colvaDelta;
269  eoniaFloorInc[j + 1] += floorDelta;
270  collateralFloor_[nettingSetId] += floorDelta;
271  }
272 
273  if (marginalAllocation_) {
274  for (Size i = 0; i < portfolio_->trades().size(); ++i) {
275  string nid = portfolio_->trades()[i]->envelope().nettingSetId();
276  if (nid != nettingSetId)
277  continue;
278  string tid = portfolio_->trades()[i]->id();
279  Real allocation = 0.0;
280  if (balance == 0.0)
281  allocation = cubeInterpretation_->getDefaultNpv(cube_, i, j, k);
282  // else if (data[j][k] == 0.0)
283  else if (fabs(data[j][k]) <= marginalAllocationLimit_)
284  allocation = exposure / nettingSetSize[nid];
285  else
286  allocation = exposure * cubeInterpretation_->getDefaultNpv(cube_, i, j, k) / data[j][k];
287 
288  if (multiPath_) {
289  if (exposure > 0.0)
290  tradeExposureCube_->set(allocation, i, j, k, allocatedEpeIndex_);
291  else
292  tradeExposureCube_->set(-allocation, i, j, k, allocatedEneIndex_);
293  } else {
294  if (exposure > 0.0)
295  averagePositiveAllocation[i][j] += allocation / cube_->samples();
296  else
297  averageNegativeAllocation[i][j] -= allocation / cube_->samples();
298  }
299  }
300  }
301  }
302  if (!multiPath_) {
303  exposureCube_->set(epe[j + 1], nettingSetCount, j, 0, ExposureIndex::EPE);
304  exposureCube_->set(ene[j + 1], nettingSetCount, j, 0, ExposureIndex::ENE);
305  }
306  ee_b[j + 1] = epe[j + 1] / curve->discount(cube_->dates()[j]);
307  eee_b[j + 1] = std::max(eee_b[j], ee_b[j + 1]);
308  std::sort(distribution.begin(), distribution.end());
309  Size index = Size(floor(quantile_ * (cube_->samples() - 1) + 0.5));
310  pfe[j + 1] = std::max(distribution[index], 0.0);
311  }
312  ee_b_[nettingSetId] = ee_b;
313  eee_b_[nettingSetId] = eee_b;
314  pfe_[nettingSetId] = pfe;
315  expectedCollateral_[nettingSetId] = eab;
316  colvaInc_[nettingSetId] = colvaInc;
317  eoniaFloorInc_[nettingSetId] = eoniaFloorInc;
318 
319  nettingSetCount++;
320 
321  Real epe_b = 0;
322  Real eepe_b = 0;
323 
324  Size t = 0;
325  Calendar cal = WeekendsOnly();
326  Date maturity = std::min(cal.adjust(today + 1 * Years + 4 * Days), nettingSetMaturity[nettingSetId]);
327  QuantLib::Real maturityTime = dc.yearFraction(today, maturity);
328 
329  while (t < cube_->dates().size() && times[t] <= maturityTime)
330  ++t;
331 
332  if (t > 0) {
333  vector<double> weights(t);
334  weights[0] = times[0];
335  for (Size k = 1; k < t; k++)
336  weights[k] = times[k] - times[k - 1];
337  double totalWeights = std::accumulate(weights.begin(), weights.end(), 0.0);
338  for (Size k = 0; k < t; k++)
339  weights[k] /= totalWeights;
340 
341  for (Size k = 0; k < t; k++) {
342  epe_b += ee_b[k] * weights[k];
343  eepe_b += eee_b[k] * weights[k];
344  }
345  }
346  epe_b_[nettingSetId] = epe_b;
347  eepe_b_[nettingSetId] = eepe_b;
348  }
349 
350  if (marginalAllocation_ && !multiPath_) {
351  for (Size i = 0; i < portfolio_->trades().size(); ++i) {
352  for (Size j = 0; j < cube_->dates().size(); ++j) {
353  tradeExposureCube_->set(averagePositiveAllocation[i][j], i, j, 0, allocatedEpeIndex_);
354  tradeExposureCube_->set(averageNegativeAllocation[i][j], i, j, 0, allocatedEneIndex_);
355  }
356  }
357  }
358 }

360 boost::shared_ptr<vector<boost::shared_ptr<CollateralAccount>>>
361 NettedExposureCalculator::collateralPaths(
362  const string& nettingSetId,
363  const Real& nettingSetValueToday,
364  const vector<vector<Real>>& nettingSetValue,
365  const Date& nettingSetMaturity) {
366 
367  boost::shared_ptr<vector<boost::shared_ptr<CollateralAccount>>> collateral;
368 
369  if (!nettingSetManager_->has(nettingSetId) || !nettingSetManager_->get(nettingSetId)->activeCsaFlag()) {
370  LOG("CSA missing or inactive for netting set " << nettingSetId);
371  return collateral;
372  }
373 
374  LOG("Build collateral account balance paths for netting set " << nettingSetId);
375  boost::shared_ptr<NettingSetDefinition> netting = nettingSetManager_->get(nettingSetId);
376  string csaFxPair = netting->csaDetails()->csaCurrency() + baseCurrency_;
377  Real csaFxRateToday = 1.0;
378  if (netting->csaDetails()->csaCurrency() != baseCurrency_)
379  csaFxRateToday = market_->fxSpot(csaFxPair, configuration_)->value();
380  LOG("CSA FX rate for pair " << csaFxPair << " = " << csaFxRateToday);
381 
382  // Don't use Settings::instance().evaluationDate() here, this has moved to simulation end date.
383  Date today = market_->asofDate();
384  string csaIndexName = netting->csaDetails()->index();
385  Real csaRateToday = market_->iborIndex(csaIndexName, configuration_)->fixing(today);
386  LOG("CSA compounding rate for index " << csaIndexName << " = " << csaRateToday);
387 
388  // Copy scenario data to keep the collateral exposure helper unchanged
389  vector<vector<Real>> csaScenFxRates(cube_->dates().size(), vector<Real>(cube_->samples(), 0.0));
390  vector<vector<Real>> csaScenRates(cube_->dates().size(), vector<Real>(cube_->samples(), 0.0));
391  if (netting->csaDetails()->csaCurrency() != baseCurrency_) {
392  QL_REQUIRE(scenarioData_->has(AggregationScenarioDataType::FXSpot, netting->csaDetails()->csaCurrency()),
393  "scenario data does not provide FX rates for " << csaFxPair);
394  }
395  if (csaIndexName != "") {
396  QL_REQUIRE(scenarioData_->has(AggregationScenarioDataType::IndexFixing, csaIndexName),
397  "scenario data does not provide index values for " << csaIndexName);
398  }
399  for (Size j = 0; j < cube_->dates().size(); ++j) {
400  for (Size k = 0; k < cube_->samples(); ++k) {
401  if (netting->csaDetails()->csaCurrency() != baseCurrency_)
402  csaScenFxRates[j][k] = cubeInterpretation_->getDefaultAggrionScenarioData(
403  scenarioData_, AggregationScenarioDataType::FXSpot, j, k, netting->csaDetails()->csaCurrency());
404  else
405  csaScenFxRates[j][k] = 1.0;
406  if (csaIndexName != "") {
407  csaScenRates[j][k] = cubeInterpretation_->getDefaultAggrionScenarioData(
408  scenarioData_, AggregationScenarioDataType::IndexFixing, j, k, csaIndexName);
409  }
410  }
411  }
412 
413  collateral = CollateralExposureHelper::collateralBalancePaths(
414  netting, // this netting set's definition
415  nettingSetValueToday, // today's netting set NPV
416  market_->asofDate(), // original evaluation date
417  nettingSetValue, // matrix of netting set values by date and sample
418  nettingSetMaturity, // netting set's maximum maturity date
419  cube_->dates(), // vector of future evaluation dates
420  csaFxRateToday, // today's FX rate for CSA to base currency, possibly 1
421  csaScenFxRates, // matrix of fx rates by date and sample, possibly 1
422  csaRateToday, // today's collateral compounding rate in CSA currency
423  csaScenRates, // matrix of CSA ccy short rates by date and sample
424  calcType_);
425  LOG("Collateral account balance paths for netting set " << nettingSetId << " done");
426 
427  return collateral;
428 }

430 vector<Real> NettedExposureCalculator::getMeanExposure(const string& tid, ExposureIndex index) {
431  vector<Real> exp(cube_->dates().size() + 1, 0.0);
432  exp[0] = exposureCube_->getT0(tid, index);
433  for (Size i = 0; i < cube_->dates().size(); i++) {
434  if (multiPath_) {
435  for (Size k = 0; k < exposureCube_->samples(); k++) {
436  exp[i + 1] += exposureCube_->get(tid, cube_->dates()[i], k, index);
437  }
438  exp[i + 1] /= exposureCube_->samples();
439  }
440  else {
441  exp[i + 1] = exposureCube_->get(tid, cube_->dates()[i], 0, index);
442  }
443  }
444  return exp;
445 }
