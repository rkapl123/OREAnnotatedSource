50// Search for a valid fixing date maximum gap days larger than d, the only relevant case for this so far is BMA/SIFMA
51Date nextValidFixingDate(Date d, const boost::shared_ptr<Index>& index, Size gap = 7) {
52 Date adjusted = d;
53 for (Size i = 0; i <= gap; ++i) {
54 adjusted = d + i;
55 if (index->isValidFixingDate(adjusted))
56 return adjusted;
57 }
58 QL_FAIL("no valid fixing date found for index " << index->name() << " within gap from " << io::iso_date(d));
59}

63//! Initialise the manager-

65void FixingManager::initialise(const boost::shared_ptr<Portfolio>& portfolio, const boost::shared_ptr<Market>& market,
66 const std::string& configuration) {
67
68 // populate the map "Index -> set of required fixing dates", where the index on the LHS is linked to curves
69 for (auto const& [tradeId,t] : portfolio->trades()) {
70 auto r = t->requiredFixings();
71 r.unsetPayDates();
72 for (auto const& [name, dates] : r.fixingDatesIndices(QuantLib::Date::maxDate())) {
73 try {
74 auto rawIndex = parseIndex(name);
75 if (auto index = boost::dynamic_pointer_cast<EquityIndex2>(rawIndex)) {
76 fixingMap_[*market->equityCurve(index->familyName(), configuration)].insert(dates.begin(),
77 dates.end());
78 } else if (auto index = boost::dynamic_pointer_cast<BondIndex>(rawIndex)) {
79 QL_FAIL("BondIndex not handled");
80 } else if (auto index = boost::dynamic_pointer_cast<CommodityIndex>(rawIndex)) {
81 // for comm indices with non-daily expiries the expiry date's day of month is 1 always
82 Date safeExpiryDate = index->expiryDate();
83 if (safeExpiryDate != Date() && !index->keepDays()) {
84 safeExpiryDate = Date::endOfMonth(safeExpiryDate);
85 }
86 fixingMap_[index->clone(safeExpiryDate,
87 market->commodityPriceCurve(index->underlyingName(), configuration))]
88 .insert(dates.begin(), dates.end());
89 } else if (auto index = boost::dynamic_pointer_cast<FxIndex>(rawIndex)) {
90 fixingMap_[*market->fxIndex(index->oreName(), configuration)].insert(dates.begin(), dates.end());
91 } else if (auto index = boost::dynamic_pointer_cast<GenericIndex>(rawIndex)) {
92 QL_FAIL("GenericIndex not handled");
93 } else if (auto index = boost::dynamic_pointer_cast<ConstantMaturityBondIndex>(rawIndex)) {
94 QL_FAIL("ConstantMaturityBondIndex not handled");
95 } else if (auto index = boost::dynamic_pointer_cast<IborIndex>(rawIndex)) {
96 fixingMap_[*market->iborIndex(name, configuration)].insert(dates.begin(), dates.end());
97 } else if (auto index = boost::dynamic_pointer_cast<SwapIndex>(rawIndex)) {
98 fixingMap_[*market->swapIndex(name, configuration)].insert(dates.begin(), dates.end());
99 } else if (auto index = boost::dynamic_pointer_cast<ZeroInflationIndex>(rawIndex)) {
100 fixingMap_[*market->zeroInflationIndex(name, configuration)].insert(dates.begin(), dates.end());
101 }
102 } catch (const std::exception& e) {
103 ALOG("FixingManager: error " << e.what() << " - no fixings are added for '" << name << "'");
104 }
105 TLOG("Added " << dates.size() << " fixing dates for '" << name << "'");
106 }
107 }
108
109 // Now cache the original fixings so we can re-write on reset()
110 for (auto const& m : fixingMap_) {
111 fixingCache_[m.first] = IndexManager::instance().getHistory(m.first->name());
112 }
113}

115//! Update fixings to date d
116void FixingManager::update(Date d) {
117 if (!fixingMap_.empty()) {
118 QL_REQUIRE(d >= fixingsEnd_, "Can't go back in time, fixings must be reset."
119 " Update date "
120 << d << " but current fixings go to " << fixingsEnd_);
121 if (d > fixingsEnd_)
122 applyFixings(fixingsEnd_, d);
123 }
124 fixingsEnd_ = d;
125}

127//! Reset fixings to t0 (today)
128void FixingManager::reset() {
129 if (modifiedFixingHistory_) {
130 for (auto& kv : fixingCache_)
131 IndexManager::instance().setHistory(kv.first->name(), kv.second);
132 modifiedFixingHistory_ = false;
133 }
134 fixingsEnd_ = today_;
135}

137void FixingManager::applyFixings(Date start, Date end) {
138 // Loop over all indices
139 for (auto const& m : fixingMap_) {
140 Date fixStart = start;
141 Date fixEnd = end;
142 Date currentFixingDate;
143 if (auto zii = boost::dynamic_pointer_cast<ZeroInflationIndex>(m.first)) {
144 fixStart =
145 inflationPeriod(fixStart - zii->zeroInflationTermStructure()->observationLag(), zii->frequency()).first;
146 fixEnd =
147 inflationPeriod(fixEnd - zii->zeroInflationTermStructure()->observationLag(), zii->frequency()).first +
148 1;
149 currentFixingDate = fixEnd;
150 } else if (auto yii = boost::dynamic_pointer_cast<YoYInflationIndex>(m.first)) {
151 fixStart =
152 inflationPeriod(fixStart - yii->yoyInflationTermStructure()->observationLag(), yii->frequency()).first;
153 fixEnd =
154 inflationPeriod(fixEnd - yii->yoyInflationTermStructure()->observationLag(), yii->frequency()).first +
155 1;
156 currentFixingDate = fixEnd;
157 } else {
158 currentFixingDate = m.first->fixingCalendar().adjust(fixEnd, Following);
159 // This date is a business day but may not be a valid fixing date in case of BMA/SIFMA
160 if (!m.first->isValidFixingDate(currentFixingDate))
161 currentFixingDate = nextValidFixingDate(currentFixingDate, m.first);
162 }
163
164 // Add we have a coupon between start and asof.
165 bool needFixings = false;
166 for (auto const& d : m.second) {
167 if (d >= fixStart && d < fixEnd) {
168 needFixings = true;
169 break;
170 }
171 }
172
173 if (needFixings) {
174 Rate currentFixing;
175 if (auto comm = boost::dynamic_pointer_cast<QuantExt::CommodityIndex>(m.first);
176 comm != nullptr && comm->expiryDate() < currentFixingDate) {
177 currentFixing = comm->priceCurve()->price(currentFixingDate);
178 } else {
179 currentFixing = m.first->fixing(currentFixingDate);
180 }
181 // if we read the fixing from an inverted FxIndex we have to undo the inversion
182 TimeSeries<Real> history;
183 for (auto const& d : m.second) {
184 if (d >= fixStart && d < fixEnd) {
185 // Fixing dates include the valuation grid dates which might not be valid fixing dates (BMA/SIFMA)
186 bool valid = m.first->isValidFixingDate(d);
187 if (valid) {
188 history[d] = currentFixing;
189 modifiedFixingHistory_ = true;
190 }
191 }
192 if (d >= fixEnd)
193 break;
194 }
195 m.first->addFixings(history, true);
196 }
197 }
198}
