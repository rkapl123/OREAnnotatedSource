65 void FixingManager::processCashFlows(const boost::shared_ptr<QuantLib::CashFlow> cf) {
66 
67  // For any coupon type that requires fixings, it must be handled here
68  // Most coupons are based off a floating rate coupon and their single index
69  // will be captured in section A.
70  //
71  // Other more exotic coupons (inflation, CMS spreads, etc) are captured on a
72  // case by case basis in section B.
73  //
74  // In all cases we want to add dates to the fixingMap_ map.
75 
76  // A floating rate coupons
77 
78  // extract underlying from cap/floored coupons
79  boost::shared_ptr<FloatingRateCoupon> frc;
80  auto cfCpn = boost::dynamic_pointer_cast<CappedFlooredCoupon>(cf);
81  if (cfCpn)
82  frc = cfCpn->underlying();
83  else
84  frc = boost::dynamic_pointer_cast<FloatingRateCoupon>(cf);
85 
86  if (frc) {
87  // A1 indices with fixings derived from underlying indices
88  auto cmssp = boost::dynamic_pointer_cast<CmsSpreadCoupon>(frc);
89  if (cmssp) {
90  fixingMap_[cmssp->swapSpreadIndex()->swapIndex1()].insert(frc->fixingDate());
91  fixingMap_[cmssp->swapSpreadIndex()->swapIndex2()].insert(frc->fixingDate());
92  return;
93  }
94  auto dcmssp = boost::dynamic_pointer_cast<DigitalCmsSpreadCoupon>(frc);
95  if (dcmssp) {
96  fixingMap_
97  [boost::dynamic_pointer_cast<CmsSpreadCoupon>(dcmssp->underlying())->swapSpreadIndex()->swapIndex1()]
98  .insert(frc->fixingDate());
99  fixingMap_
100  [boost::dynamic_pointer_cast<CmsSpreadCoupon>(dcmssp->underlying())->swapSpreadIndex()->swapIndex2()]
101  .insert(frc->fixingDate());
102  return;
103  }
104 
105  // A2 indices with native fixings, but no only on the standard fixing date
106  auto on = boost::dynamic_pointer_cast<OvernightIndexedCoupon>(frc);
107  if (on) {
108  for (auto const& d : on->fixingDates())
109  fixingMap_[on->index()].insert(d);
110  return;
111  }
112  auto avon = boost::dynamic_pointer_cast<AverageONIndexedCoupon>(frc);
113  if (avon) {
114  for (auto const& d : avon->fixingDates())
115  fixingMap_[avon->index()].insert(d);
116  return;
117  }
118  auto bma = boost::dynamic_pointer_cast<AverageBMACoupon>(frc);
119  if (bma) {
120  for (auto const& d : bma->fixingDates())
121  fixingMap_[bma->index()].insert(d);
122  return;
123  }
124 
125  // A3 standard case
126  fixingMap_[frc->index()].insert(frc->fixingDate());
127  }
128 
129  // B other coupon types
130 
131  boost::shared_ptr<FloatingRateFXLinkedNotionalCoupon> fc =
132  boost::dynamic_pointer_cast<FloatingRateFXLinkedNotionalCoupon>(cf);
133  if (fc) {
134  fixingMap_[fc->index()].insert(fc->fixingDate());
135  fixingMap_[fc->fxIndex()].insert(fc->fxFixingDate());
136  return;
137  }
138 
139  boost::shared_ptr<FXLinkedCashFlow> flcf = boost::dynamic_pointer_cast<FXLinkedCashFlow>(cf);
140  if (flcf) {
141  fixingMap_[flcf->fxIndex()].insert(flcf->fxFixingDate());
142  return;
143  }
144 
145  boost::shared_ptr<CPICoupon> cpc = boost::dynamic_pointer_cast<CPICoupon>(cf);
146  if (cpc) {
147  fixingMap_[cpc->index()].insert(cpc->fixingDate());
148  return;
149  }
150 
151  boost::shared_ptr<InflationCoupon> ic = boost::dynamic_pointer_cast<InflationCoupon>(cf);
152  if (ic) {
153  fixingMap_[ic->index()].insert(ic->fixingDate());
154  return;
155  }
156 
157  boost::shared_ptr<CPICashFlow> cpcf = boost::dynamic_pointer_cast<CPICashFlow>(cf);
158  if (cpcf) {
159  fixingMap_[cpcf->index()].insert(cpcf->fixingDate());
160  return;
161  }
162 }

164 //! Update fixings to date d
165 void FixingManager::update(Date d) {
166  if (!fixingMap_.empty()) {
167  QL_REQUIRE(d >= fixingsEnd_, "Can't go back in time, fixings must be reset."
168  " Update date "
169  << d << " but current fixings go to " << fixingsEnd_);
170  applyFixings(fixingsEnd_, d);
171  }
172  fixingsEnd_ = d;
173 }

175 //! Reset fixings to t0 (today)
176 void FixingManager::reset() {
177  if (modifiedFixingHistory_) {
178  for (auto& kv : fixingCache_)
179  IndexManager::instance().setHistory(kv.first->name(), kv.second);
180  modifiedFixingHistory_ = false;
181  }
182  fixingsEnd_ = today_;
183 }

185 void FixingManager::applyFixings(Date start, Date end) {
186  // Loop over all indices
187  for (auto const& m : fixingMap_) {
188  boost::shared_ptr<ZeroInflationIndex> zii = boost::dynamic_pointer_cast<ZeroInflationIndex>(m.first);
189  boost::shared_ptr<YoYInflationIndex> yii = boost::dynamic_pointer_cast<YoYInflationIndex>(m.first);
190  set<Date> fixingDates;
191  Date currentFixingDate;
192  Date fixStart = start;
193  Date fixEnd = end;
194  if (zii) { // for inflation indices we just only add a fixing for the first date in the month
195  fixStart =
196  inflationPeriod(fixStart - zii->zeroInflationTermStructure()->observationLag(), zii->frequency()).first;
197  fixEnd =
198  inflationPeriod(fixEnd - zii->zeroInflationTermStructure()->observationLag(), zii->frequency()).first +
199  1;
200  currentFixingDate = fixEnd;
201  for (auto const& d : m.second) {
202  fixingDates.insert(inflationPeriod(d, zii->frequency()).first);
203  }
204  } else if (yii) {
205  fixStart =
206  inflationPeriod(fixStart - yii->yoyInflationTermStructure()->observationLag(), yii->frequency()).first;
207  fixEnd =
208  inflationPeriod(fixEnd - yii->yoyInflationTermStructure()->observationLag(), yii->frequency()).first +
209  1;
210  currentFixingDate = fixEnd;
211  for (auto const& d : m.second) {
212  fixingDates.insert(inflationPeriod(d, yii->frequency()).first);
213  }
214  } else {
215  currentFixingDate = m.first->fixingCalendar().adjust(fixEnd, Following);
216  fixingDates = m.second;
217  }
218 
219  // Add we have a coupon between start and asof.
220  bool needFixings = false;
221  for (auto const& d : fixingDates) {
222  if (d >= fixStart && d < fixEnd) {
223  needFixings = true;
224  break;
225  }
226  }
227 
228  if (needFixings) {
229  Rate currentFixing = m.first->fixing(currentFixingDate);
230  TimeSeries<Real> history;
231  for (auto const& d : fixingDates) {
232  if (d >= fixStart && d < fixEnd) {
233  history[d] = currentFixing;
234  modifiedFixingHistory_ = true;
235  }
236  if (d >= fixEnd)
237  break;
238  }
239  m.first->addFixings(history, true);
240  }
241  }
242 }
