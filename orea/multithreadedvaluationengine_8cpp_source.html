104void MultiThreadedValuationEngine::buildCube(
105 const boost::shared_ptr<ore::data::Portfolio>& portfolio,
106 const std::function<std::vector<boost::shared_ptr<ore::analytics::ValuationCalculator>>()>& calculators,
107 const std::function<std::vector<boost::shared_ptr<ore::analytics::CounterpartyCalculator>>()>& cptyCalculators,
108 bool mporStickyDate, bool dryRun) {
109
110 boost::timer::cpu_timer timer;
111
112 LOG("MultiThreadedValuationEngine::buildCube() was called");
113
114 // extract pricing stats accumulated so far and clear them
115
116 LOG("Extract pricing stats and clear them in the current portfolio");
117
118 std::map<std::string, std::pair<std::size_t, boost::timer::nanosecond_type>> pricingStats;
119 for (auto const& [tid, t] : portfolio->trades())
120 pricingStats[tid] = std::make_pair(t->getNumberOfPricings(), t->getCumulativePricingTime());
121
122 // build portfolio against init market and trigger single pricing to generate pricing stats
123
124 LOG("Reset and build portfolio against init market to produce pricing stats from a single pricing.");
125
126 boost::shared_ptr<ore::data::Market> initMarket = boost::make_shared<ore::data::TodaysMarket>(
127 today_, todaysMarketParams_, loader_, curveConfigs_, true, true, true, referenceData_, false,
128 iborFallbackConfig_, false, handlePseudoCurrenciesTodaysMarket_);
129
130 auto engineFactory = boost::make_shared<ore::data::EngineFactory>(
131 engineData_, initMarket,
132 std::map<ore::data::MarketContext, string>{{ore::data::MarketContext::pricing, configuration_}}, referenceData_,
133 iborFallbackConfig_);
134
135 portfolio->build(engineFactory, context_, true);
136
137 for (auto const& [tid, t] : portfolio->trades()) {
138 TLOG("got npv for " << tid << ": " << std::setprecision(12) << t->instrument()->NPV() << " "
139 << t->npvCurrency());
140 }
141
142 // split portfolio into nThreads parts such that each part has an approximately similar total avg pricing time
143
144 Size eff_nThreads = std::min(portfolio->size(), nThreads_);
145
146 LOG("Splitting portfolio.");
147
148 LOG("portfolio size = " << portfolio->size());
149 LOG("nThreads = " << nThreads_);
150 LOG("eff nThreads = " << eff_nThreads);
151
152 QL_REQUIRE(eff_nThreads > 0, "effective threads are zero, this is not allowed.");
153
154 std::vector<boost::shared_ptr<ore::data::Portfolio>> portfolios;
155 for (Size i = 0; i < eff_nThreads; ++i)
156 portfolios.push_back(boost::make_shared<ore::data::Portfolio>());
157
158 double totalAvgPricingTime = 0.0;
159 std::vector<std::pair<std::string, double>> timings;
160 for (auto const& [tid, t] : portfolio->trades()) {
161 if (t->getNumberOfPricings() != 0) {
162 double dt = t->getCumulativePricingTime() / static_cast<double>(t->getNumberOfPricings());
163 timings.push_back(std::make_pair(tid, dt));
164 totalAvgPricingTime += dt;
165 } else {
166 // trade might be a failed trade
167 timings.push_back(std::make_pair(tid, 0.0));
168 }
169 }
170
171 std::sort(timings.begin(), timings.end(),
172 [](const std::pair<std::string, double>& p1, const std::pair<std::string, double> p2) {
173 if (p1.second == p2.second)
174 return p1.first < p2.first;
175 else
176 return p1.second > p2.second;
177 });
178
179 std::vector<double> portfolioTotalAvgPricingTime(portfolios.size());
180 Size portfolioIndex = 0;
181 for (auto const& t : timings) {
182 portfolios[portfolioIndex]->add(portfolio->get(t.first));
183 portfolioTotalAvgPricingTime[portfolioIndex] += t.second;
184 if (++portfolioIndex >= eff_nThreads)
185 portfolioIndex = 0;
186 }
187
188 // output the portfolios into strings so that the worker threads can load them from there
189
190 std::vector<std::string> portfoliosAsString;
191 for (auto const& p : portfolios) {
192 portfoliosAsString.emplace_back(p->toXMLString());
193 }
194
195 // log info on the portfolio split
196
197 LOG("Total avg pricing time : " << totalAvgPricingTime / 1E6 << " ms");
198 for (Size i = 0; i < eff_nThreads; ++i) {
199 LOG("Portfolio #" << i << " number of trades : " << portfolios[i]->size());
200 LOG("Portfolio #" << i << " total avg pricing time : " << portfolioTotalAvgPricingTime[i] / 1E6 << " ms");
201 }
202
203 // build scenario generators for each thread as clones of the original one
204
205 LOG("Cloning scenario generators for " << eff_nThreads << " threads...");
206 std::vector<boost::shared_ptr<ore::analytics::ScenarioGenerator>> scenarioGenerators;
207 auto tmp =
208 boost::make_shared<ore::analytics::ClonedScenarioGenerator>(scenarioGenerator_, dateGrid_->dates(), nSamples_);
209 scenarioGenerators.push_back(tmp);
210 DLOG("generator for thread 1 cloned.");
211 for (Size i = 1; i < eff_nThreads; ++i) {
212 scenarioGenerators.push_back(boost::make_shared<ore::analytics::ClonedScenarioGenerator>(*tmp));
213 DLOG("generator for thread " << (i + 1) << " cloned.");
214 }
215
216 // build loaders for each thread as clones of the original one
217
218 LOG("Cloning loaders for " << eff_nThreads << " threads...");
219 std::vector<boost::shared_ptr<ore::data::ClonedLoader>> loaders;
220 for (Size i = 0; i < eff_nThreads; ++i)
221 loaders.push_back(boost::make_shared<ore::data::ClonedLoader>(today_, loader_));
222
223 // build nThreads mini-cubes to which each thread writes its results
224
225 LOG("Build " << eff_nThreads << " mini result cubes...");
226 miniCubes_.clear();
227 miniNettingSetCubes_.clear();
228 miniCptyCubes_.clear();
229 for (Size i = 0; i < eff_nThreads; ++i) {
230 miniCubes_.push_back(cubeFactory_(today_, portfolios[i]->ids(), dateGrid_->dates(), nSamples_));
231 miniNettingSetCubes_.push_back(nettingSetCubeFactory_(today_, dateGrid_->dates(), nSamples_));
232 miniCptyCubes_.push_back(
233 cptyCubeFactory_(today_, portfolios[i]->counterparties(), dateGrid_->dates(), nSamples_));
234 }
235
236 // build progress indicator consolidating the results from the threads
237
238 auto progressIndicator =
239 boost::make_shared<ore::analytics::MultiThreadedProgressIndicator>(this->progressIndicators());
240
241 // create the thread pool with eff_nThreads and queue size = eff_nThreads as well
242
243 // LOG("Create thread pool with " << eff_nThreads);
244 // ctpl::thread_pool threadPool(eff_nThreads);
245
246 // create the jobs and push them to the pool
247
248 using resultType = int;
249 std::vector<std::future<resultType>> results(eff_nThreads);
250
251 std::vector<std::thread> jobs; // not needed if thread pool is used
252
253 // pricing stats accumulated in worker threads
254 std::vector<std::map<std::string, std::pair<std::size_t, boost::timer::nanosecond_type>>> workerPricingStats(
255 eff_nThreads);
256
257 // get obs mode of main thread, so that we can set this mode in the worker threads below
258 ore::analytics::ObservationMode::Mode obsMode = ore::analytics::ObservationMode::instance().mode();
259
260 for (Size i = 0; i < eff_nThreads; ++i) {
261
262 auto job = [this, obsMode, dryRun, &calculators, &cptyCalculators, mporStickyDate, &portfoliosAsString,
263 &scenarioGenerators, &loaders, &workerPricingStats, &progressIndicator](int id) -> resultType {
264 // set thread local singletons
265
266 QuantLib::Settings::instance().evaluationDate() = today_;
267 ore::analytics::ObservationMode::instance().setMode(obsMode);
268
269 LOG("Start thread " << id);
270
271 int rc;
272
273 try {
274
275 // build todays market using cloned market data
276
277 boost::shared_ptr<ore::data::Market> initMarket = boost::make_shared<ore::data::TodaysMarket>(
278 today_, todaysMarketParams_, loaders[id], curveConfigs_, true, true, true, referenceData_, false,
279 iborFallbackConfig_, false, handlePseudoCurrenciesTodaysMarket_);
280
281 // build sim market
282
283 boost::shared_ptr<ore::analytics::ScenarioSimMarket> simMarket =
284 boost::make_shared<ore::analytics::ScenarioSimMarketPlus>(
285 initMarket, simMarketData_, configuration_, *curveConfigs_, *todaysMarketParams_, true,
286 useSpreadedTermStructures_, cacheSimData_, false, iborFallbackConfig_,
287 handlePseudoCurrenciesSimMarket_);
288
289 // set aggregation scenario data, but only in one of the sim markets, that's sufficient to populate it
290
291 if (id == 0 && aggregationScenarioData_ != nullptr)
292 simMarket->aggregationScenarioData() = aggregationScenarioData_;
293
294 // link scenario generator to sim market
295
296 simMarket->scenarioGenerator() = scenarioGenerators[id];
297
298 // set scenario filter
299
300 if (scenarioFilter_)
301 simMarket->filter() = scenarioFilter_;
302
303 // build portfolio against sim market
304
305 auto portfolio = boost::make_shared<ore::data::Portfolio>();
306 portfolio->fromXMLString(portfoliosAsString[id]);
307 auto engineFactory = boost::make_shared<ore::data::EngineFactory>(
308 engineData_, simMarket, std::map<ore::data::MarketContext, string>(), referenceData_,
309 iborFallbackConfig_);
310
311 portfolio->build(engineFactory, context_, true);
312
313 // build valuation engine
314
315 auto valEngine = boost::make_shared<ore::analytics::ValuationEngine>(today_, dateGrid_, simMarket,
316 engineFactory->modelBuilders());
317 valEngine->registerProgressIndicator(progressIndicator);
318
319 // build mini-cube
320
321 valEngine->buildCube(portfolio, miniCubes_[id], calculators(), mporStickyDate, miniNettingSetCubes_[id],
322 miniCptyCubes_[id],
323 cptyCalculators ? cptyCalculators()
324 : std::vector<boost::shared_ptr<CounterpartyCalculator>>(),
325 dryRun);
326
327 // set pricing stats for val engine run
328
329 for (auto const& [tid, t] : portfolio->trades())
330 workerPricingStats[id][tid] =
331 std::make_pair(t->getNumberOfPricings(), t->getCumulativePricingTime());
332
333 // return code 0 = ok
334
335 LOG("Thread " << id << " successfully finished.");
336
337 rc = 0;
338
339 } catch (const std::exception& e) {
340
341 // log error and return code 1 = not ok
342
343 ALOG(ore::analytics::StructuredAnalyticsErrorMessage("Multithreaded Valuation Engine", "", e.what()));
344 rc = 1;
345 }
346
347 // exit
348
349 return rc;
350 };
351
352 // results[i] = threadPool.push(job);
353
354 // not needed if thread pool is used
355 std::packaged_task<resultType(int)> task(job);
356 results[i] = task.get_future();
357 std::thread thread(std::move(task), i);
358 jobs.emplace_back(std::move(thread));
359 }
360
361 // check return codes from jobs
362
363 // not needed if thread pool is used
364 for (auto& t : jobs)
365 t.join();
366
367 for (Size i = 0; i < results.size(); ++i) {
368 results[i].wait();
369 }
370
371 for (Size i = 0; i < results.size(); ++i) {
372 QL_REQUIRE(results[i].valid(), "internal error: did not get a valid result");
373 int rc = results[i].get();
374 QL_REQUIRE(rc == 0, "error: thread " << i << " exited with return code " << rc
375 << ". Check for structured errors from 'MultiThreaded Valuation Engine'.");
376 }
377
378 // stop the thread pool, wait for unfinished jobs
379
380 // LOG("Stop thread pool");
381 // threadPool.stop(true);
382
383 // set updated pricing stats in original portfolio
384
385 LOG("Update pricing stats of trades.");
386
387 for (auto const& [tid, t] : portfolio->trades()) {
388 auto p = pricingStats[tid];
389 std::size_t n = p.first;
390 boost::timer::nanosecond_type d = p.second;
391 for (auto const& w : workerPricingStats) {
392 auto p = w.find(tid);
393 if (p != w.end()) {
394 n += p->second.first;
395 d += p->second.second;
396 }
397 }
398 t->resetPricingStats(n, d);
399 }
400
401 // log timings and return the result mini-cubes
402
403 LOG("MultiThreadedValuationEngine::buildCube() successfully finished, timings: "
404 << static_cast<double>(timer.elapsed().wall) / 1.0E9 << "s Wall, "
405 << static_cast<double>(timer.elapsed().user) / 1.0E9 << "s User, "
406 << static_cast<double>(timer.elapsed().system) / 1.0E9 << "s System.");
407}
