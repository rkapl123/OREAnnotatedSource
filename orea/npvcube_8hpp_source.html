36//! NPV Cube class stores both future and current NPV values.
37/*! The cube class stores futures NPV values in a 3-D array, i.e. each side can be of a different
38 * length (so a cuboid).
39 *
40 * This abstract base class is just used for the storage of a cube.
41 * This class also stores the tradeIds, dates and vector of T0 NPVs
42 \ingroup cube
43 */
44class NPVCube {
45public:
46 //! default ctor
47 NPVCube() {}
48
49 //! Do not allow cube copying
50 NPVCube(NPVCube&) = delete;
51 NPVCube& operator=(NPVCube const&) = delete;
52
53 //! dtor
54 virtual ~NPVCube() {}
55
56 //! Return the length of each dimension
57 virtual Size numIds() const = 0;
58 virtual Size numDates() const = 0;
59 virtual Size samples() const = 0;
60 virtual Size depth() const = 0;
61
62 //! Get the vector of ids for this cube
63 virtual const std::vector<std::string>& ids() const = 0;
64 //! Get the vector of dates for this cube
65 virtual const std::vector<QuantLib::Date>& dates() const = 0; // TODO: Should this be the full date grid?
66
67 //! Return the asof date (T0 date)
68 virtual QuantLib::Date asof() const = 0;
69 //! Get a T0 value from the cube using index
70 virtual Real getT0(Size id, Size depth = 0) const = 0;
71 //! Get a T0 value from the cube using trade id
72 virtual Real getT0(const std::string& id, Size depth = 0) const { return getT0(index(id), depth); };
73 //! Set a value in the cube using index
74 virtual void setT0(Real value, Size id, Size depth = 0) = 0;
75 //! Set a value in the cube using trade id
76 virtual void setT0(Real value, const std::string& id, Size depth = 0) { setT0(value, index(id), depth); };
77
78 //! Get a value from the cube using index
79 virtual Real get(Size id, Size date, Size sample, Size depth = 0) const = 0;
80 //! Set a value in the cube using index
81 virtual void set(Real value, Size id, Size date, Size sample, Size depth = 0) = 0;
82
83 //! Get a value from the cube using trade id and date
84 virtual Real get(const std::string& id, const QuantLib::Date& date, Size sample, Size depth = 0) const {
85 return get(index(id), index(date), sample, depth);
86 };
87 //! Set a value in the cube using trade id and date
88 virtual void set(Real value, const std::string& id, const QuantLib::Date& date, Size sample, Size depth = 0) {
89 set(value, index(id), index(date), sample, depth);
90 }
91
92 //! Load cube contents from disk
93 virtual void load(const std::string& fileName) = 0;
94 //! Persist cube contents to disk
95 virtual void save(const std::string& fileName) const = 0;
96
97protected:
98 virtual Size index(const std::string& id) const {
99 auto it = std::find(ids().begin(), ids().end(), id);
100 QL_REQUIRE(it != ids().end(), "NPVCube can't find an index for id " << id);
101 return std::distance(ids().begin(), it);
102 };
103
104 virtual Size index(const QuantLib::Date& date) const {
105 auto it = std::find(dates().begin(), dates().end(), date);
106 QL_REQUIRE(it != dates().end(), "NPVCube can't find an index for date " << date);
107 return std::distance(dates().begin(), it);
108 };
109};
