36 //! NPV Cube class stores both future and current NPV values.
37 /*! The cube class stores futures NPV values in a 3-D array, i.e. each side can be of a different
38  * length (so a cuboid).
39  *
40  * This abstract base class is just used for the storage of a cube.
41  * This class also stores the tradeIds, dates and vector of T0 NPVs
42  \ingroup cube
43  */

98  virtual Size index(const std::string& id) const {
99  auto it = std::find(ids().begin(), ids().end(), id);
100  QL_REQUIRE(it != ids().end(), "NPVCube can't find an index for id " << id);
101  return std::distance(ids().begin(), it);
102  };
103 
104  virtual Size index(const QuantLib::Date& date) const {
105  auto it = std::find(dates().begin(), dates().end(), date);

107  return std::distance(dates().begin(), it);
