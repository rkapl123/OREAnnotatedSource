37Real SimmConcentrationBase::thresholdImpl(const boost::shared_ptr<SimmBucketMapper>& simmBucketMapper,
38 const RiskType& riskType, const string& qualifier) const {
39
40 // Deal with some specific cases first
41 if (riskType == RiskType::IRCurve || riskType == RiskType::IRVol || riskType == RiskType::FX) {
42
43 QL_REQUIRE(qualifier.size() == 3, "Expect the qualifier, " << qualifier << ", to be a valid currency code");
44 QL_REQUIRE(checkCurrency(qualifier),
45 "The qualifier, " << qualifier << ", is not a supported currency code");
46 auto cat = riskType == RiskType::FX ? category(qualifier, fxCategories_) : category(qualifier, irCategories_);
47
48 return bucketedThresholds_.at(riskType).at(cat) * units_;
49 }
50
51 if (riskType == RiskType::FXVol) {
52 return fxVolThreshold(qualifier) * units_;
53 }
54
55 // Check if the risk type's threshold is not bucket dependent and if so return the threshold
56 if (flatThresholds_.count(riskType) > 0) {
57 return flatThresholds_.at(riskType) * units_;
58 }
59
60 // Check if the risk type's threshold is bucket dependent and if so, find the bucket and return the threshold
61 if (bucketedThresholds_.count(riskType) > 0) {
62 string bucket = simmBucketMapper->bucket(riskType, qualifier);
63 bucket = bucket == "residual" ? "Residual" : bucket;
64 auto const& m = bucketedThresholds_.at(riskType);
65 auto const& t = m.find(bucket);
66 QL_REQUIRE(t != m.end(), "SimmConcentrationBase::thresholdImpl(): bucket '"
67 << bucket << "' not found in bucketedThresholds for qualifier '" << qualifier
68 << "' and risk type '" << riskType << "'");
69 return t->second * units_;
70 }
71
72 // If we get to here, no threshold
73 return QL_MAX_REAL;
74}

76string SimmConcentrationBase::category(const string& qualifier, const map<string, set<string>>& categories) const {
77 string result;
78 for (const auto& kv : categories) {
79 if (kv.second.empty()) {
80 result = kv.first;
81 } else {
82 if (kv.second.count(qualifier) > 0) {
83 // Found qualifier in category so return it
84 return kv.first;
85 }
86 }
87 }
88
89 // If we get here, result should hold category with empty set
90 return result;
91}

93Real SimmConcentrationBase::fxVolThreshold(const string& fxPair) const {
94
95 QL_REQUIRE(fxPair.size() == 6, "Expected '" << fxPair << "' to be a currency pair so it should be of length 6.");
96
97 string ccy_1 = fxPair.substr(0, 3);
98 QL_REQUIRE(checkCurrency(ccy_1),
99 "First currency in pair " << fxPair << " (" << ccy_1 << ") is not a supported currency code");
100 string ccy_2 = fxPair.substr(3);
101 QL_REQUIRE(checkCurrency(ccy_2),
102 "Second currency in pair " << fxPair << " (" << ccy_2 << ") is not a supported currency code");
103
104 // Find category of currency 1 and currency 2
105 string category_1 = category(ccy_1, fxCategories_);
106 string category_2 = category(ccy_2, fxCategories_);
107
108 if (category_1 == "1" && category_2 == "1") {
109 // Both currencies in FX category 1
110 return bucketedThresholds_.at(RiskType::FXVol).at("1");
111 } else if ((category_1 == "1" && category_2 == "2") || (category_1 == "2" && category_2 == "1")) {
112 // One currency in FX category 1 and the other in FX category 2
113 return bucketedThresholds_.at(RiskType::FXVol).at("2");
114 } else if ((category_1 == "1" && category_2 == "3") || (category_1 == "3" && category_2 == "1")) {
115 // One currency in FX category 1 and the other in FX category 3
116 return bucketedThresholds_.at(RiskType::FXVol).at("3");
117 } else if (category_1 == "2" && category_2 == "2") {
118 // Both currencies in FX category 2
119 return bucketedThresholds_.at(RiskType::FXVol).at("4");
120 } else if ((category_1 == "2" && category_2 == "3") || (category_1 == "3" && category_2 == "2")) {
121 // One currency in FX category 2 and the other in FX category 3
122 return bucketedThresholds_.at(RiskType::FXVol).at("5");
123 } else {
124 // Both currencies in FX category 3
125 return bucketedThresholds_.at(RiskType::FXVol).at("6");
126 }
127}
