124//-----------------------------------------------------------------------------------------------
125//! Scenario Base Class
126/*! A scenario contains a single cross asset model sample in terms of
127 yield curves by currency, FX rates, etc.
128
129 This base class provides the interface to add and retrieve data to and from a scenario.
130 Concrete simple and memory optimized "compact" scenario classes are derived from this.
131
132 \ingroup scenario
133*/
134class Scenario {
135public:
136 //! Destructor
137 virtual ~Scenario() {}
138
139 //! Return the scenario asof date
140 virtual const Date& asof() const = 0;
141
142 //! Get the scenario label
143 virtual const string& label() const = 0;
144 //! Set the scenario label
145 virtual void label(const string&) = 0;
146
147 //! Get Numeraire ratio n = N(t) / N(0) so that Price(0) = N(0) * E [Price(t) / N(t) ]
148 virtual Real getNumeraire() const = 0;
149 //! Set the Numeraire ratio n = N(t) / N(0) so that Price(0) = N(0) * E [Price(t) / N(t) ]
150 virtual void setNumeraire(Real n) = 0;
151
152 //! Check whether this scenario provides the data for the given key
153 virtual bool has(const RiskFactorKey& key) const = 0;
154 //! Risk factor keys for which this scenario provides data
155 virtual const std::vector<RiskFactorKey>& keys() const = 0;
156 //! Add an element to the scenario
157 virtual void add(const RiskFactorKey& key, Real value) = 0;
158 //! Get an element from the scenario
159 virtual Real get(const RiskFactorKey& key) const = 0;
160
161 //! clones a scenario and returns a pointer to the new object
162 virtual boost::shared_ptr<Scenario> clone() const = 0;
163
164private:
165 friend class boost::serialization::access;
166 template <class Archive> void serialize(Archive&, const unsigned int) {}
167};
