51 void ValuationEngine::buildCube(const boost::shared_ptr<data::Portfolio>& portfolio,
52  boost::shared_ptr<analytics::NPVCube> outputCube,
53  vector<boost::shared_ptr<ValuationCalculator>> calculators) {
54 
55  QL_REQUIRE(portfolio->size() > 0, "ValuationEngine: Error portfolio is empty");
56 
57  QL_REQUIRE(outputCube->numIds() == portfolio->trades().size(),
58  "cube x dimension (" << outputCube->numIds() << ") "
59  << "different from portfolio size (" << portfolio->trades().size() << ")");
60 
61  QL_REQUIRE(outputCube->numDates() == dg_->dates().size(),
62  "cube y dimension (" << outputCube->numDates() << ") "
63  << "different from number of time steps (" << dg_->dates().size() << ")");
64 
65  LOG("Starting ValuationEngine for " << portfolio->size() << " trades, " << outputCube->samples() << " samples and "
66  << dg_->size() << " dates.");
67 
68  ObservationMode::Mode om = ObservationMode::instance().mode();
69  Real updateTime = 0.0;
70  Real pricingTime = 0.0;
71  Real fixingTime = 0.0;
72 
73  // Loop is Samples, Dates, Trades
74  const auto& dates = dg_->dates();
75  const auto& trades = portfolio->trades();
76 
77  LOG("Initialise state objects...");
78  Size numFRC = 0;
79  // initialise state objects for each trade (required for path-dependent derivatives in particular)
80  for (Size i = 0; i < trades.size(); i++) {
81  QL_REQUIRE(trades[i]->npvCurrency() != "", "NPV currency not set for trade " << trades[i]->id());
82 
83  DLOG("Initialise wrapper for trade " << trades[i]->id());
84  trades[i]->instrument()->initialise(dates);
85 
86  // T0 values
87  for (auto calc : calculators)
88  calc->calculateT0(trades[i], i, simMarket_, outputCube);
89 
90  if (om == ObservationMode::Mode::Unregister) {
91  for (const Leg& leg : trades[i]->legs()) {
92  for (Size n = 0; n < leg.size(); n++) {
93  boost::shared_ptr<FloatingRateCoupon> frc = boost::dynamic_pointer_cast<FloatingRateCoupon>(leg[n]);
94  if (frc) {
95  frc->unregisterWith(frc->index());
96  trades[i]->instrument()->qlInstrument()->unregisterWith(frc);
97  // Unregister with eval dates
98  frc->unregisterWith(Settings::instance().evaluationDate());
99  frc->index()->unregisterWith(Settings::instance().evaluationDate());
100  trades[i]->instrument()->qlInstrument()->unregisterWith(Settings::instance().evaluationDate());
101  }
102  }
103  }
104  }
105  }
106  LOG("Total number of swaps = " << portfolio->size());
107  LOG("Total number of FRC = " << numFRC);
108 
109  simMarket_->fixingManager()->initialise(portfolio);
110 
111  boost::timer timer;
112  boost::timer loopTimer;
113 
114  // We call Cube::samples() each time her to allow for dynamic stopping times
115  // e.g. MC convergence tests
116  for (Size sample = 0; sample < outputCube->samples(); ++sample) {
117  updateProgress(sample, outputCube->samples());
118 
119  for (auto& trade : trades)
120  trade->instrument()->reset();
121 
122  // loop over Dates
123  for (Size i = 0; i < dates.size(); ++i) {
124  Date d = dates[i];
125 
126  timer.restart();
127 
128  simMarket_->update(d);
129 
130  // recalibrate models
131  for (auto const& b : modelBuilders_) {
132  if (om == ObservationMode::Mode::Disable)
133  b.second->recalculate();
134  b.second->recalibrate();
135  }
136 
137  updateTime += timer.elapsed();
138 
139  // loop over trades
140  timer.restart();
141  for (Size j = 0; j < trades.size(); ++j) {
142  auto trade = trades[j];
143 
144  // We can avoid checking mode here and always call updateQlInstruments()
145  if (om == ObservationMode::Mode::Disable)
146  trade->instrument()->updateQlInstruments();
147 
148  for (auto calc : calculators)
149  calc->calculate(trade, j, simMarket_, outputCube, d, i, sample);
150  }
151  pricingTime += timer.elapsed();
152  }
153 
154  timer.restart();
155  simMarket_->fixingManager()->reset();
156  fixingTime += timer.elapsed();
157  }
158 
159  simMarket_->reset();
160  updateProgress(outputCube->samples(), outputCube->samples());
161  LOG("ValuationEngine completed: loop " << setprecision(2) << loopTimer.elapsed() << " sec, "
162  << "pricing " << pricingTime << " sec, "
163  << "update " << updateTime << " sec "
164  << "fixing " << fixingTime);
165 }
