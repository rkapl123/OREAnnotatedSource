43ValuationEngine::ValuationEngine(const Date& today, const boost::shared_ptr<DateGrid>& dg,
44 const boost::shared_ptr<SimMarket>& simMarket,
45 const set<std::pair<string, boost::shared_ptr<ModelBuilder>>>& modelBuilders)
46 : today_(today), dg_(dg), simMarket_(simMarket), modelBuilders_(modelBuilders) {
47
48 QL_REQUIRE(dg_->size() > 0, "Error, DateGrid size must be > 0");
49 QL_REQUIRE(today <= dg_->dates().front(), "ValuationEngine: Error today ("
50 << today << ") must not be later than first DateGrid date "
51 << dg_->dates().front());
52 QL_REQUIRE(simMarket_, "ValuationEngine: Error, Null SimMarket");
53}

55void ValuationEngine::recalibrateModels() {
56 ObservationMode::Mode om = ObservationMode::instance().mode();
57 for (auto const& b : modelBuilders_) {
58 if (om == ObservationMode::Mode::Disable)
59 b.second->forceRecalculate();
60 b.second->recalibrate();
61 }

64void ValuationEngine::buildCube(const boost::shared_ptr<data::Portfolio>& portfolio,
65 boost::shared_ptr<analytics::NPVCube> outputCube,
66 vector<boost::shared_ptr<ValuationCalculator>> calculators, bool mporStickyDate,
67 boost::shared_ptr<analytics::NPVCube> outputCubeNettingSet,
68 boost::shared_ptr<analytics::NPVCube> outputCptyCube,
69 vector<boost::shared_ptr<CounterpartyCalculator>> cptyCalculators, bool dryRun) {
70
71 LOG("Build cube with mporStickyDate=" << mporStickyDate << ", dryRun=" << std::boolalpha << dryRun);
72
73 QL_REQUIRE(portfolio->size() > 0, "ValuationEngine: Error portfolio is empty");
74
75 QL_REQUIRE(outputCube->numIds() == portfolio->trades().size(),
76 "cube x dimension (" << outputCube->numIds() << ") "
77 << "different from portfolio size (" << portfolio->trades().size() << ")");
78
79 QL_REQUIRE(outputCube->numDates() == dg_->valuationDates().size(),
80 "cube y dimension (" << outputCube->numDates() << ") "
81 << "different from number of valuation dates (" << dg_->valuationDates().size()
82 << ")");
83
84 if (outputCptyCube) {
85 QL_REQUIRE(outputCptyCube->numIds() == portfolio->counterparties().size() + 1,
86 "cptyCube x dimension (" << outputCptyCube->numIds() << "minus 1) "
87 << "different from portfolio counterparty size ("
88 << portfolio->counterparties().size() << ")");
89
90 QL_REQUIRE(outputCptyCube->numDates() == dg_->dates().size(), "outputCptyCube y dimension ("
91 << outputCptyCube->numDates() << ") "
92 << "different from number of time steps ("
93 << dg_->dates().size() << ")");
94 }
95
96 LOG("Starting ValuationEngine for " << portfolio->size() << " trades, " << outputCube->samples() << " samples and "
97 << dg_->size() << " dates.");
98
99 ObservationMode::Mode om = ObservationMode::instance().mode();
100 Real updateTime = 0.0;
101 Real pricingTime = 0.0;
102 Real fixingTime = 0.0;
103
104 LOG("Initialise " << calculators.size() << " valuation calculators");
105 for (auto const& c : calculators) {
106 c->init(portfolio, simMarket_);
107 c->initScenario();
108 }
109
110 // Loop is Samples, Dates, Trades
111 const auto& dates = dg_->dates();
112 const auto& trades = portfolio->trades();
113 auto& counterparties = outputCptyCube ? outputCptyCube->ids() : vector<string>();
114 std::vector<bool> tradeHasError(trades.size(), false);
115 LOG("Initialise state objects...");
116 // initialise state objects for each trade (required for path-dependent derivatives in particular)
117 for (Size i = 0; i < trades.size(); i++) {
118 QL_REQUIRE(!trades[i]->npvCurrency().empty(), "NPV currency not set for trade " << trades[i]->id());
119
120 DLOG("Initialise wrapper for trade " << trades[i]->id());
121 trades[i]->instrument()->initialise(dates);
122
123 recalibrateModels();
124
125 // T0 values
126 try {
127 for (auto& calc : calculators)
128 calc->calculateT0(trades[i], i, simMarket_, outputCube, outputCubeNettingSet);
129 } catch (const std::exception& e) {
130 string expMsg = string("T0 valuation error: ") + e.what();
131 ALOG(StructuredTradeErrorMessage(trades[i]->id(), trades[i]->tradeType(), "ScenarioValuation",
132 expMsg.c_str()));
133 tradeHasError[i] = true;
134 }
135
136 if (om == ObservationMode::Mode::Unregister) {
137 for (const Leg& leg : trades[i]->legs()) {
138 for (Size n = 0; n < leg.size(); n++) {
139 boost::shared_ptr<FloatingRateCoupon> frc = boost::dynamic_pointer_cast<FloatingRateCoupon>(leg[n]);
140 if (frc) {
141 frc->unregisterWith(frc->index());
142 trades[i]->instrument()->qlInstrument()->unregisterWith(frc);
143 // Unregister with eval dates
144 frc->unregisterWith(Settings::instance().evaluationDate());
145 frc->index()->unregisterWith(Settings::instance().evaluationDate());
146 trades[i]->instrument()->qlInstrument()->unregisterWith(Settings::instance().evaluationDate());
147 }
148 }
149 }
150 }
151 }
152 LOG("Total number of swaps = " << portfolio->size());
153
154 if (dates.size() > 1) {
155 // only need to init the fixing manager if there is more than one sim date
156 simMarket_->fixingManager()->initialise(portfolio, simMarket_);
157 }
158
159 cpu_timer timer;
160 cpu_timer loopTimer;
161
162 // We call Cube::samples() each time her to allow for dynamic stopping times
163 // e.g. MC convergence tests
164 for (Size sample = 0; sample < (dryRun ? std::min<Size>(1, outputCube->samples()) : outputCube->samples());
165 ++sample) {
166 updateProgress(sample, outputCube->samples());
167
168 for (auto& trade : trades)
169 trade->instrument()->reset();
170
171 // loop over Dates, increase cubeDateIndex for each valuation date we hit
172 int cubeDateIndex = -1;
173 for (Size i = 0; i < dates.size(); ++i) {
174 Date d = dates[i];
175
176 // Process auxiliary close-out dates first (may coincide with a valuation date, see below)
177 // Store result at same cubeDateIndex as the previous valuation date's result, but at different cube depth
178 // Differences to valuation date processing above:
179 // Update valuation date and fixings, trades exercisable depending on stickiness
180 bool scenarioUpdated = false;
181 if (dg_->isCloseOutDate()[i]) {
182 timer.start();
183
184 // update market
185 simMarket_->preUpdate();
186 if (!mporStickyDate)
187 simMarket_->updateDate(d);
188 simMarket_->updateScenario(d);
189 scenarioUpdated = true;
190 simMarket_->postUpdate(d, !mporStickyDate); // with fixings only if not sticky
191
192 recalibrateModels();
193
194 timer.stop();
195 updateTime += timer.elapsed().wall * 1e-9;
196
197 // loop over trades
198 timer.start();
199 if (mporStickyDate) // switch off if sticky
200 tradeExercisable(false, trades);
201 QL_REQUIRE(cubeDateIndex >= 0,
202 "negative cube date index, ensure that the date grid starts with a valuation date");
203 runCalculators(true, trades, tradeHasError, calculators, outputCube, outputCubeNettingSet, d,
204 cubeDateIndex, sample, simMarket_->label());
205 if (mporStickyDate) // switch on again, if sticky
206 tradeExercisable(true, trades);
207 timer.stop();
208 pricingTime += timer.elapsed().wall * 1e-9;
209 }
210
211 // process a valuation date as usual
212 if (dg_->isValuationDate()[i]) {
213 timer.start();
214
215 cubeDateIndex++;
216
217 // All the steps below from preUpdate() to updateAsd(d) are combined in update(d), but we decompose as
218 // follows simMarket_->update(d);
219 simMarket_->preUpdate();
220 simMarket_->updateDate(d);
221 // We can skip this step, if we have done that above in the close-out date section
222 if (!scenarioUpdated)
223 simMarket_->updateScenario(d);
224 // Always with fixing update here, in contrast to the close-out date section
225 simMarket_->postUpdate(d, true);
226 // Aggregation scenario data update on valuation dates only
227 simMarket_->updateAsd(d);
228
229 recalibrateModels();
230
231 timer.stop();
232 updateTime += timer.elapsed().wall * 1e-9;
233
234 timer.start();
235 // loop over trades
236 runCalculators(false, trades, tradeHasError, calculators, outputCube, outputCubeNettingSet, d,
237 cubeDateIndex, sample, simMarket_->label());
238 // loop over counterparty names
239 runCalculators(false, counterparties, cptyCalculators, outputCptyCube, d, cubeDateIndex, sample);
240 timer.stop();
241 pricingTime += timer.elapsed().wall * 1e-9;
242 }
243 }
244
245 timer.start();
246 simMarket_->fixingManager()->reset();
247 fixingTime += timer.elapsed().wall * 1e-9;
248 }
249
250 if (dryRun) {
251 LOG("Doing a dry run - fill remaining cube with random values.");
252 for (Size sample = 1; sample < outputCube->samples(); ++sample) {
253 for (Size i = 0; i < dates.size(); ++i) {
254 for (Size j = 0; j < trades.size(); ++j) {
255 for (Size d = 0; d < outputCube->depth(); ++d) {
256 // add some noise, but only for the first few samples, so that e.g.
257 // a sensi run is not polluted with too many sensis for each trade
258 Real noise = sample < 10 ? static_cast<Real>(i + j + d + sample) : 0.0;
259 outputCube->set(outputCube->getT0(j, d) + noise, j, i, sample, d);
260 }
261 }
262 }
263 }
264 }
265
266 simMarket_->reset();
267 updateProgress(outputCube->samples(), outputCube->samples());
268 loopTimer.stop();
269 LOG("ValuationEngine completed: loop " << setprecision(2) << loopTimer.format(2, "%w") << " sec, "
270 << "pricing " << pricingTime << " sec, "
271 << "update " << updateTime << " sec "
272 << "fixing " << fixingTime);
273
274 // for trades with errors set all output cube values to zero
275
276 for (Size i = 0; i < trades.size(); ++i) {
277 if (tradeHasError[i]) {
278 ALOG("setting all results in output cube to zero for trade '"
279 << trades[i]->id() << "' since there was at least one error during simulation");
280 for (Size index = 0; index < outputCube->depth(); ++index) {
281 outputCube->setT0(0.0, i, index);
282 for (Size dateIndex = 0; dateIndex < outputCube->numDates(); ++dateIndex) {
283 for (Size sample = 0; sample < outputCube->samples(); ++sample) {
284 outputCube->set(0.0, i, dateIndex, sample, index);
285 }
286 }
287 }
288 }
289 }
290}

292void ValuationEngine::runCalculators(bool isCloseOutDate, const std::vector<boost::shared_ptr<Trade>>& trades,
293 std::vector<bool>& tradeHasError,
294 const std::vector<boost::shared_ptr<ValuationCalculator>>& calculators,
295 boost::shared_ptr<analytics::NPVCube>& outputCube,
296 boost::shared_ptr<analytics::NPVCube>& outputCubeNettingSet, const Date& d,
297 const Size cubeDateIndex, const Size sample, const string& label) {
298 ObservationMode::Mode om = ObservationMode::instance().mode();
299 for(auto& calc: calculators)
300 calc->initScenario();
301 // loop over trades
302 for (Size j = 0; j < trades.size(); ++j) {
303 if (tradeHasError[j])
304 continue;
305 auto trade = trades[j];
306 // We can avoid checking mode here and always call updateQlInstruments()
307 if (om == ObservationMode::Mode::Disable || om == ObservationMode::Mode::Unregister)
308 trade->instrument()->updateQlInstruments();
309 try {
310 for (auto& calc : calculators)
311 calc->calculate(trade, j, simMarket_, outputCube, outputCubeNettingSet, d, cubeDateIndex, sample,
312 isCloseOutDate);
313 } catch (const std::exception& e) {
314 string expMsg = "date = " + ore::data::to_string(io::iso_date(d)) +
315 ", sample = " + ore::data::to_string(sample) + ", label = " + label + ": " + e.what();
316 ALOG(StructuredTradeErrorMessage(trade->id(), trade->tradeType(), "ScenarioValuation", expMsg.c_str()));
317 tradeHasError[j] = true;
318 }
319 }
320}

322void ValuationEngine::runCalculators(bool isCloseOutDate, const std::vector<std::string>& counterparties,
323 const std::vector<boost::shared_ptr<CounterpartyCalculator>>& calculators,
324 boost::shared_ptr<analytics::NPVCube>& cptyCube, const Date& d,
325 const Size cubeDateIndex, const Size sample) {
326 // loop over counterparties
327 for (Size j = 0; j < counterparties.size(); ++j) {
328 auto counterparty = counterparties[j];
329 for (auto& calc : calculators)
330 calc->calculate(counterparty, j, simMarket_, cptyCube, d, cubeDateIndex, sample, isCloseOutDate);
331 }
332}

334void ValuationEngine::tradeExercisable(bool enable, const std::vector<boost::shared_ptr<Trade>>& trades) {
335 for (Size j = 0; j < trades.size(); ++j) {
336 auto t = boost::dynamic_pointer_cast<OptionWrapper>(trades[j]->instrument());

339 t->enableExercise();

342 }
