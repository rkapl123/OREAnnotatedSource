98 boost::shared_ptr<Portfolio> buildPortfolio(Size portfolioSize, boost::shared_ptr<EngineFactory>& factory) {
99 
100  boost::shared_ptr<Portfolio> portfolio(new Portfolio());

102  vector<string> ccys = {"EUR", "USD", "GBP", "JPY", "CHF"};
103 
104  map<string, vector<string>> indices = {{"EUR", {"EUR-EURIBOR-6M"}},
105  {"USD", {"USD-LIBOR-3M"}},
106  {"GBP", {"GBP-LIBOR-6M"}},

108  {"JPY", {"JPY-LIBOR-6M"}}};
109 
110  vector<string> fixedTenors = {"6M", "1Y"};
111 
112  Size minTerm = 2;
113  Size maxTerm = 30;
114 
115  Size minFixedBps = 10;
116  Size maxFixedBps = 400;
117 
118  Size seed = 5; // keep this constant to ensure portfolio doesn't change
119  MersenneTwisterUniformRng rng(seed);
120 
121  Date today = Settings::instance().evaluationDate();

123  string calStr = "TARGET";
124  string conv = "MF";
125  string rule = "Forward";
126  Size days = 2;
127  string fixDC = "30/360";
128  string floatDC = "ACT/365";
129 
130  vector<double> notional(1, 1000000);
131  vector<double> spread(1, 0);
132 
133  for (Size i = 0; i < portfolioSize; i++) {
134  Size term = portfolioSize == 1 ? 20 : randInt(rng, minTerm, maxTerm);
135 
136  // Start today +/- 1 Year
137  Date startDate = portfolioSize == 1 ? cal.adjust(today) : cal.adjust(today - 365 + randInt(rng, 0, 730));
138  Date endDate = cal.adjust(startDate + term * Years);
139 
140  // date 2 string
141  ostringstream oss;
142  oss << io::iso_date(startDate);
143  string start(oss.str());
144  oss.str("");
145  oss.clear();
146  oss << io::iso_date(endDate);
147  string end(oss.str());
148 
149  // ccy + index
150  string ccy = portfolioSize == 1 ? "EUR" : randString(rng, ccys);
151  string index = portfolioSize == 1 ? "EUR-EURIBOR-6M" : randString(rng, indices[ccy]);
152  string floatFreq = portfolioSize == 1 ? "6M" : index.substr(index.find('-', 4) + 1);
153 
154  // This variable is not used and only here to ensure that the random numbers generated in
155  // subsequent blocks produce a swap portfolio, which is compatible with the archived values.
156  string fixedTenor = portfolioSize == 1 ? "1Y" : randString(rng, fixedTenors);
157  fixedTenor = fixedTenor + "_";
158 
159  // fixed details
160  Real fixedRate = portfolioSize == 1 ? 0.02 : randInt(rng, minFixedBps, maxFixedBps) / 100.0;
161  string fixFreq = portfolioSize == 1 ? "1Y" : randString(rng, fixedTenors);
162 
163  // envelope
164  Envelope env("CP");
165 
166  // Schedules
167  ScheduleData floatSchedule(ScheduleRules(start, end, floatFreq, calStr, conv, conv, rule));
168  ScheduleData fixedSchedule(ScheduleRules(start, end, fixFreq, calStr, conv, conv, rule));
169 
170  bool isPayer = randBoolean(rng);
171 
172  // fixed Leg - with dummy rate
173  LegData fixedLeg(boost::make_shared<FixedLegData>(vector<double>(1, fixedRate)), isPayer, ccy, fixedSchedule,
174  fixDC, notional);
175 
176  // float Leg
177  vector<double> spreads(1, 0);
178  LegData floatingLeg(boost::make_shared<FloatingLegData>(index, days, false, spread), !isPayer, ccy,
179  floatSchedule, floatDC, notional);
180 
181  boost::shared_ptr<Trade> swap(new data::Swap(env, floatingLeg, fixedLeg));
182 
183  // id
184  oss.clear();
185  oss.str("");
186  oss << "Trade_" << i + 1;
187  swap->id() = oss.str();
188 
189  portfolio->add(swap);
190  }
191  // portfolio->save("port.xml");
192 
193  portfolio->build(factory);
194 
195  if (portfolio->size() != portfolioSize)
196  BOOST_ERROR("Failed to build portfolio (got " << portfolio->size() << " expected " << portfolioSize << ")");
197 
198  // Dump stats about portfolio
199  Time maturity = 0;
200  DayCounter dc = ActualActual(ActualActual::ISDA);
201  map<string, Size> fixedFreqs;
202  map<string, Size> floatFreqs;
203  for (Size i = 0; i < portfolioSize; ++i) {
204  boost::shared_ptr<Trade> trade = portfolio->trades()[i];
205  maturity += dc.yearFraction(today, trade->maturity());
206 
207  // fixed Freq
208  boost::shared_ptr<data::Swap> swap = boost::dynamic_pointer_cast<data::Swap>(trade);
209  string floatFreq = swap->legData()[0].schedule().rules().front().tenor();
210  string fixFreq = swap->legData()[1].schedule().rules().front().tenor();
211  QL_REQUIRE(swap->legData()[0].legType() == "Floating" && swap->legData()[1].legType() == "Fixed", "Leg mixup");
212  if (fixedFreqs.find(fixFreq) == fixedFreqs.end())
213  fixedFreqs[fixFreq] = 1;
214  else
215  fixedFreqs[fixFreq]++;
216  if (floatFreqs.find(floatFreq) == floatFreqs.end())
217  floatFreqs[floatFreq] = 1;
218  else
219  floatFreqs[floatFreq]++;
220  }
221  maturity /= portfolioSize;
222  BOOST_TEST_MESSAGE("Portfolio Size : " << portfolioSize);
223  BOOST_TEST_MESSAGE("Average Maturity : " << maturity);
224  ostringstream oss;
225  for (Size i = 0; i < ccys.size(); i++)
226  oss << ccys[i] << " ";
227  BOOST_TEST_MESSAGE("Currencies : " << oss.str());
228  // dump % breakdown of tenors
229  map<string, Size>::iterator it;
230  BOOST_TEST_MESSAGE("Fixed Tenors : ");
231  for (it = fixedFreqs.begin(); it != fixedFreqs.end(); ++it) {
232  Real perc = 100 * it->second / (Real)portfolioSize;
233  BOOST_TEST_MESSAGE(" " << it->first << " " << perc << " %");
234  }
235  BOOST_TEST_MESSAGE("Floating Tenors : ");
236  for (it = floatFreqs.begin(); it != floatFreqs.end(); ++it) {
237  Real perc = 100 * it->second / (Real)portfolioSize;
238  BOOST_TEST_MESSAGE(" " << it->first << " " << perc << " %");
239  }
240 
241  return portfolio;
242 }
