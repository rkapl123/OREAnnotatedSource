90 
91  string outputPath = params_->get("setup", "outputPath");
92  Real sensiThreshold = parseReal(params_->get("sensitivity", "outputSensitivityThreshold"));
93 
94  string outputFile = outputPath + "/" + params_->get("sensitivity", "scenarioOutputFile");

96  ReportWriter().writeScenarioReport(scenReport, sensiAnalysis->sensiCube(), sensiThreshold);
97 
98  // Create a stream from the sensitivity cube

101 
102  outputFile = outputPath + "/" + params_->get("sensitivity", "sensitivityOutputFile");
