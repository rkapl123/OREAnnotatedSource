44 CrossAssetStateProcess::discretization parseDiscretization(const string& s) {
45  static map<string, QuantExt::CrossAssetStateProcess::discretization> m = {
46  {"Exact", QuantExt::CrossAssetStateProcess::exact}, {"Euler", QuantExt::CrossAssetStateProcess::euler}};
47 
48  auto it = m.find(s);
49  if (it != m.end()) {
50  return it->second;
51  } else {
52  QL_FAIL("Cannot convert \"" << s << "\" to QuantExt::CrossAssetStateProcess::discretization");
53  }

72 void ScenarioGeneratorData::fromXML(XMLNode* root) {
73  XMLNode* sim = XMLUtils::locateNode(root, "Simulation");
74  XMLNode* node = XMLUtils::getChildNode(sim, "Parameters");
75  XMLUtils::checkNode(node, "Parameters");
76 
77  std::string discString = XMLUtils::getChildValue(node, "Discretization", true); // mandatory
78  discretization_ = parseDiscretization(discString);
79  LOG("ScenarioGeneratorData discretization = " << discString);
80 
81  std::string calString = XMLUtils::getChildValue(node, "Calendar", true);
82  Calendar cal = parseCalendar(calString);
83 
84  std::string dcString = XMLUtils::getChildValue(node, "DayCounter", false);
85  DayCounter dc = dcString.empty() ? ActualActual() : parseDayCounter(dcString);
86 
87  gridString_ = XMLUtils::getChildValue(node, "Grid", true);
88  std::vector<std::string> tokens;
89  boost::split(tokens, gridString_, boost::is_any_of(","));
90  if (tokens.size() <= 2) {
91  grid_ = boost::make_shared<DateGrid>(gridString_, cal, dc);
92  } else {
93  std::vector<Period> gridTenors = XMLUtils::getChildrenValuesAsPeriods(node, "Grid", true);
94  grid_ = boost::make_shared<DateGrid>(gridTenors, cal, dc);
95  }
96  LOG("ScenarioGeneratorData grid points size = " << grid_->size());
97 
98  std::string sequenceTypeString = XMLUtils::getChildValue(node, "Sequence", true);
99  sequenceType_ = parseSequenceType(sequenceTypeString);
100  LOG("ScenarioGeneratorData sequence type = " << sequenceTypeString);
101 
102  seed_ = XMLUtils::getChildValueAsInt(node, "Seed", true);
103  LOG("ScenarioGeneratorData seed = " << seed_);
104 
105  samples_ = XMLUtils::getChildValueAsInt(node, "Samples", true);
106  LOG("ScenarioGeneratorData samples = " << samples_);
107 
108  // overwrite samples with enviroment variable OVERWRITE_SCENARIOGENERATOR_SAMPLES
109  if (auto c = getenv("OVERWRITE_SCENARIOGENERATOR_SAMPLES")) {
110  samples_ = std::stol(c);
111  LOG("Overwrite samples with " << samples_ << " from enviroment variable OVERWRITE_SCENARIOGENERATOR_SAMPLES");
112  }
113 
114  if (auto n = XMLUtils::getChildNode(node, "Ordering"))
115  ordering_ = parseSobolBrownianGeneratorOrdering(XMLUtils::getNodeValue(n));
116  else
117  ordering_ = SobolBrownianGenerator::Steps;
118 
119  if (auto n = XMLUtils::getChildNode(node, "DirectionIntegers"))
120  directionIntegers_ = parseSobolRsgDirectionIntegers(XMLUtils::getNodeValue(n));
121  else
122  directionIntegers_ = SobolRsg::JoeKuoD7;
123 
124  withCloseOutLag_ = false;
125  if (XMLUtils::getChildNode(node, "CloseOutLag") != NULL) {
126  withCloseOutLag_ = true;
127  closeOutLag_ = parsePeriod(XMLUtils::getChildValue(node, "CloseOutLag", true));
128  grid_->addCloseOutDates(closeOutLag_);
129  LOG("Use lagged close out grid, lag period is " << closeOutLag_);
130  }
131  withMporStickyDate_ = false;
132  if (XMLUtils::getChildNode(node, "MporMode") != NULL) {
133  string mporMode = XMLUtils::getChildValue(node, "MporMode", true);
134  if (mporMode == "StickyDate") {
135  withMporStickyDate_ = true;
136  LOG("Use Mpor sticky date mode");
137  } else if (mporMode == "ActualDate") {
138  withMporStickyDate_ = false;
139  LOG("Use Mpor actual date mode");
140  } else {
141  QL_FAIL("MporMode " << mporMode << " not recognised");
142  }
143  }
144 
145  LOG("ScenarioGeneratorData done.");
146 }

148 XMLNode* ScenarioGeneratorData::toXML(XMLDocument& doc) {
149  XMLNode* node = doc.allocNode("Simulation");
150  XMLNode* pNode = XMLUtils::addChild(doc, node, "Parameters");
151 
152  XMLUtils::addChild(doc, pNode, "Discretization", ore::data::to_string((QuantExt::CrossAssetStateProcess::discretization) discretization_));
153 
154  if (grid_) {
155  XMLUtils::addChild(doc, pNode, "Calendar", grid_->calendar().name());
156  XMLUtils::addChild(doc, pNode, "DayCounter", grid_->dayCounter().name());
157  if (gridString_ != "") {
158  XMLUtils::addChild(doc, node, "Grid", gridString_);
159  } else {
160  XMLUtils::addGenericChildAsList(doc, node, "Grid", grid_->tenors());
161  }
162  }
163 
164 
165  XMLUtils::addChild(doc, pNode, "Sequence", ore::data::to_string( sequenceType_));
166  XMLUtils::addChild(doc, pNode, "Seed", to_string(seed_));
167  XMLUtils::addChild(doc, pNode, "Samples", to_string(samples_));
168 
169  XMLUtils::addChild(doc, pNode, "Ordering", ore::data::to_string((SobolBrownianGenerator::Ordering) ordering_) );
170  XMLUtils::addChild(doc, pNode, "DirectionIntegers", ore::data::to_string(directionIntegers_));
171 
172  if (withCloseOutLag_) {
173  XMLUtils::addChild(doc, pNode, "CloseOutLag", closeOutLag_);
174  }
175  if (withMporStickyDate_) {
176  XMLUtils::addChild(doc, pNode, "MporMode", "StickyDate");
177  } else {
178  XMLUtils::addChild(doc, pNode, "MporMode", "ActualDate");
179  }
180 
181  return node;
182 }
