79const ShiftData& SensitivityScenarioData::shiftData(const RiskFactorKey::KeyType& keyType, const string& name) const {
80 // Not nice but not spending time refactoring the class now.
81 switch (keyType) {
82 case RFType::DiscountCurve:
83 return *discountCurveShiftData().at(name);
84 case RFType::IndexCurve:
85 return *indexCurveShiftData().at(name);
86 case RFType::YieldCurve:
87 return *yieldCurveShiftData().at(name);
88 case RFType::FXSpot:
89 return fxShiftData().at(name);
90 case RFType::SwaptionVolatility:
91 return swaptionVolShiftData().at(name);
92 case RFType::YieldVolatility:
93 return yieldVolShiftData().at(name);
94 case RFType::OptionletVolatility:
95 return *capFloorVolShiftData().at(name);
96 case RFType::FXVolatility:
97 return fxVolShiftData().at(name);
98 case RFType::CDSVolatility:
99 return cdsVolShiftData().at(name);
100 case RFType::BaseCorrelation:
101 return baseCorrelationShiftData().at(name);
102 case RFType::ZeroInflationCurve:
103 return *zeroInflationCurveShiftData().at(name);
104 case RFType::SurvivalProbability:
105 return *creditCurveShiftData().at(name);
106 case RFType::YoYInflationCurve:
107 return *yoyInflationCurveShiftData().at(name);
108 case RFType::YoYInflationCapFloorVolatility:
109 return *yoyInflationCapFloorVolShiftData().at(name);
110 case RFType::ZeroInflationCapFloorVolatility:
111 return *zeroInflationCapFloorVolShiftData().at(name);
112 case RFType::EquitySpot:
113 return equityShiftData().at(name);
114 case RFType::EquityVolatility:
115 return equityVolShiftData().at(name);
116 case RFType::DividendYield:
117 return *dividendYieldShiftData().at(name);
118 case RFType::CommodityCurve:
119 return *commodityCurveShiftData().at(name);
120 case RFType::CommodityVolatility:
121 return commodityVolShiftData().at(name);
122 case RFType::SecuritySpread:
123 return securityShiftData().at(name);
124 case RFType::Correlation:
125 return correlationShiftData().at(name);
126 default:
127 QL_FAIL("Cannot return shift data for key type: " << keyType);
128 }
129}
