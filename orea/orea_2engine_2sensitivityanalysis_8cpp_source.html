69 std::vector<boost::shared_ptr<ValuationCalculator>> SensitivityAnalysis::buildValuationCalculators() const {
70  vector<boost::shared_ptr<ValuationCalculator>> calculators;
71  if (nonShiftedBaseCurrencyConversion_) // use "original" FX rates to convert sensi to base currency
72  calculators.push_back(boost::make_shared<NPVCalculatorFXT0>(simMarketData_->baseCcy(), market_));
73  else // use the scenario FX rate when converting sensi to base currency
74  calculators.push_back(boost::make_shared<NPVCalculator>(simMarketData_->baseCcy()));
75  return calculators;
76 }

78 void SensitivityAnalysis::initialize(boost::shared_ptr<NPVSensiCube>& cube) {
79  LOG("Build Sensitivity Scenario Generator and Simulation Market");
80  initializeSimMarket();
81 
82  LOG("Build Engine Factory and rebuild portfolio");
83  boost::shared_ptr<EngineFactory> factory = buildFactory(extraEngineBuilders_, extraLegBuilders_);
84  resetPortfolio(factory);
85  if (recalibrateModels_)
86  modelBuilders_ = factory->modelBuilders();
87  else
88  modelBuilders_.clear();
89 
90  if (!cube) {
91  LOG("Build the cube object to store sensitivities");
92  initializeCube(cube);
93  }
94 
95  sensiCube_ = boost::make_shared<SensitivityCube>(cube, scenarioGenerator_->scenarioDescriptions(),
96  scenarioGenerator_->shiftSizes());
97  initialized_ = true;
98 }

100 void SensitivityAnalysis::generateSensitivities(boost::shared_ptr<NPVSensiCube> cube) {
101 
102  QL_REQUIRE(!initialized_, "unexpected state of SensitivitiesAnalysis object");
103 
104  // initialize the helper member objects
105  initialize(cube);
106  QL_REQUIRE(initialized_, "SensitivitiesAnalysis member objects not correctly initialized");
107  boost::shared_ptr<DateGrid> dg = boost::make_shared<DateGrid>("1,0W", NullCalendar());
108  vector<boost::shared_ptr<ValuationCalculator>> calculators = buildValuationCalculators();
109  ValuationEngine engine(asof_, dg, simMarket_, modelBuilders_);
110  for (auto const& i : this->progressIndicators())
111  engine.registerProgressIndicator(i);
112  LOG("Run Sensitivity Scenarios");
113  engine.buildCube(portfolio_, cube, calculators);
114 
115  computed_ = true;
116  LOG("Sensitivity analysis completed");
117 }

119 void SensitivityAnalysis::initializeSimMarket(boost::shared_ptr<ScenarioFactory> scenFact) {

128  boost::shared_ptr<Scenario> baseScenario = simMarket_->baseScenario();

132 
133  LOG("Create scenario generator for sensitivity analysis (continueOnError=" << std::boolalpha << continueOnError_

136  sensitivityData_, baseScenario, simMarketData_, scenarioFactory, overrideTenors_, continueOnError_);
137  LOG("Scenario generator created for sensitivity analysis");
138 
139  // Set simulation market's scenario generator
140  simMarket_->scenarioGenerator() = scenarioGenerator_;

162 Real getShiftSize(const RiskFactorKey& key, const SensitivityScenarioData& sensiParams,
163  const boost::shared_ptr<ScenarioSimMarket>& simMarket, const string& marketConfiguration) {
164 
165  Date asof = simMarket->asofDate();
166  RiskFactorKey::KeyType keytype = key.keytype;
167  string keylabel = key.name;
168  Real shiftSize = 0.0;
169  Real shiftMult = 1.0;
170  switch (keytype) {
171  case RiskFactorKey::KeyType::FXSpot: {
172  auto itr = sensiParams.fxShiftData().find(keylabel);
173  QL_REQUIRE(itr != sensiParams.fxShiftData().end(), "shiftData not found for " << keylabel);
174  shiftSize = itr->second.shiftSize;
175  if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
176  shiftMult = simMarket->fxSpot(keylabel, marketConfiguration)->value();
177  }
178  } break;
179  case RiskFactorKey::KeyType::EquitySpot: {
180  auto itr = sensiParams.equityShiftData().find(keylabel);
181  QL_REQUIRE(itr != sensiParams.equityShiftData().end(), "shiftData not found for " << keylabel);
182  shiftSize = itr->second.shiftSize;
183  if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
184  shiftMult = simMarket->equitySpot(keylabel, marketConfiguration)->value();
185  }
186  } break;
187  case RiskFactorKey::KeyType::DiscountCurve: {
188  string ccy = keylabel;
189  auto itr = sensiParams.discountCurveShiftData().find(ccy);
190  QL_REQUIRE(itr != sensiParams.discountCurveShiftData().end(), "shiftData not found for " << ccy);
191  shiftSize = itr->second->shiftSize;
192  if (parseShiftType(itr->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
193  Size keyIdx = key.index;
194  Period p = itr->second->shiftTenors[keyIdx];
195  Handle<YieldTermStructure> yts = simMarket->discountCurve(ccy, marketConfiguration);
196  Time t = yts->dayCounter().yearFraction(asof, asof + p);
197  Real zeroRate = yts->zeroRate(t, Continuous);
198  shiftMult = zeroRate;
199  }
200  } break;
201  case RiskFactorKey::KeyType::IndexCurve: {
202  string idx = keylabel;
203  auto itr = sensiParams.indexCurveShiftData().find(idx);
204  QL_REQUIRE(itr != sensiParams.indexCurveShiftData().end(), "shiftData not found for " << idx);
205  shiftSize = itr->second->shiftSize;
206  if (parseShiftType(itr->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
207  Size keyIdx = key.index;
208  Period p = itr->second->shiftTenors[keyIdx];
209  Handle<YieldTermStructure> yts = simMarket->iborIndex(idx, marketConfiguration)->forwardingTermStructure();
210  Time t = yts->dayCounter().yearFraction(asof, asof + p);
211  Real zeroRate = yts->zeroRate(t, Continuous);
212  shiftMult = zeroRate;
213  }
214  } break;
215  case RiskFactorKey::KeyType::YieldCurve: {
216  string yc = keylabel;
217  auto itr = sensiParams.yieldCurveShiftData().find(yc);
218  QL_REQUIRE(itr != sensiParams.yieldCurveShiftData().end(), "shiftData not found for " << yc);
219  shiftSize = itr->second->shiftSize;
220  if (parseShiftType(itr->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
221  Size keyIdx = key.index;
222  Period p = itr->second->shiftTenors[keyIdx];
223  Handle<YieldTermStructure> yts = simMarket->yieldCurve(yc, marketConfiguration);
224  Time t = yts->dayCounter().yearFraction(asof, asof + p);
225  Real zeroRate = yts->zeroRate(t, Continuous);
226  shiftMult = zeroRate;
227  }
228  } break;
229  case RiskFactorKey::KeyType::DividendYield: {
230  string eq = keylabel;
231  auto itr = sensiParams.dividendYieldShiftData().find(eq);
232  QL_REQUIRE(itr != sensiParams.dividendYieldShiftData().end(), "shiftData not found for " << eq);
233  shiftSize = itr->second->shiftSize;
234 
235  if (parseShiftType(itr->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
236  Size keyIdx = key.index;
237  Period p = itr->second->shiftTenors[keyIdx];
238  Handle<YieldTermStructure> ts = simMarket->equityDividendCurve(eq, marketConfiguration);
239  Time t = ts->dayCounter().yearFraction(asof, asof + p);
240  Real zeroRate = ts->zeroRate(t, Continuous);
241  shiftMult = zeroRate;
242  }
243  } break;
244  case RiskFactorKey::KeyType::FXVolatility: {
245  string pair = keylabel;
246  auto itr = sensiParams.fxVolShiftData().find(pair);
247  QL_REQUIRE(itr != sensiParams.fxVolShiftData().end(), "shiftData not found for " << pair);
248  shiftSize = itr->second.shiftSize;
249  if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
250  vector<Real> strikes = sensiParams.fxVolShiftData().at(pair).shiftStrikes;
251  QL_REQUIRE(strikes.size() == 1 && close_enough(strikes[0], 0), "Only ATM FX vols supported");
252  Real atmFwd = 0.0; // hardcoded, since only ATM supported
253  Size keyIdx = key.index;
254  Period p = itr->second.shiftExpiries[keyIdx];
255  Handle<BlackVolTermStructure> vts = simMarket->fxVol(pair, marketConfiguration);
256  Time t = vts->dayCounter().yearFraction(asof, asof + p);
257  Real atmVol = vts->blackVol(t, atmFwd);
258  shiftMult = atmVol;
259  }
260  } break;
261  case RiskFactorKey::KeyType::EquityVolatility: {
262  string pair = keylabel;
263  auto itr = sensiParams.equityVolShiftData().find(pair);
264  QL_REQUIRE(itr != sensiParams.equityVolShiftData().end(), "shiftData not found for " << pair);
265  shiftSize = itr->second.shiftSize;
266  if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
267  Size keyIdx = key.index;
268  Period p = itr->second.shiftExpiries[keyIdx];
269  Handle<BlackVolTermStructure> vts = simMarket->equityVol(pair, marketConfiguration);
270  Time t = vts->dayCounter().yearFraction(asof, asof + p);
271  Real atmVol = vts->blackVol(t, Null<Real>());
272  shiftMult = atmVol;
273  }
274  } break;
275  case RiskFactorKey::KeyType::SwaptionVolatility: {
276  string ccy = keylabel;
277  auto itr = sensiParams.swaptionVolShiftData().find(ccy);
278  QL_REQUIRE(itr != sensiParams.swaptionVolShiftData().end(), "shiftData not found for " << ccy);
279  shiftSize = itr->second.shiftSize;
280  if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
281  vector<Real> strikes = itr->second.shiftStrikes;
282  vector<Period> tenors = itr->second.shiftTerms;
283  vector<Period> expiries = itr->second.shiftExpiries;
284  Size keyIdx = key.index;
285  Size expIdx = keyIdx / (tenors.size() * strikes.size());
286  Period p_exp = expiries[expIdx];
287  Size tenIdx = keyIdx % tenors.size();
288  Period p_ten = tenors[tenIdx];
289  Real strike = Null<Real>(); // for cubes this will be ATM
290  Handle<SwaptionVolatilityStructure> vts = simMarket->swaptionVol(ccy, marketConfiguration);
291  // Time t_exp = vts->dayCounter().yearFraction(asof, asof + p_exp);
292  // Time t_ten = vts->dayCounter().yearFraction(asof, asof + p_ten);
293  // Real atmVol = vts->volatility(t_exp, t_ten, Null<Real>());
294  Real vol = vts->volatility(p_exp, p_ten, strike);
295  shiftMult = vol;
296  }
297  } break;
298  case RiskFactorKey::KeyType::YieldVolatility: {
299  string securityId = keylabel;
300  auto itr = sensiParams.yieldVolShiftData().find(securityId);
301  QL_REQUIRE(itr != sensiParams.yieldVolShiftData().end(), "shiftData not found for " << securityId);
302  shiftSize = itr->second.shiftSize;
303  if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
304  vector<Real> strikes = itr->second.shiftStrikes;
305  QL_REQUIRE(strikes.size() == 1 && close_enough(strikes[0], 0.0),
306  "shift strikes should be {0.0} for yield volatilities");
307  vector<Period> tenors = itr->second.shiftTerms;
308  vector<Period> expiries = itr->second.shiftExpiries;
309  Size keyIdx = key.index;
310  Size expIdx = keyIdx / (tenors.size() * strikes.size());
311  Period p_exp = expiries[expIdx];
312  Size tenIdx = keyIdx % tenors.size();
313  Period p_ten = tenors[tenIdx];
314  Real strike = Null<Real>(); // for cubes this will be ATM
315  Handle<SwaptionVolatilityStructure> vts = simMarket->yieldVol(securityId, marketConfiguration);
316  Real vol = vts->volatility(p_exp, p_ten, strike);
317  shiftMult = vol;
318  }
319  } break;
320  case RiskFactorKey::KeyType::OptionletVolatility: {
321  string ccy = keylabel;
322  auto itr = sensiParams.capFloorVolShiftData().find(ccy);
323  QL_REQUIRE(itr != sensiParams.capFloorVolShiftData().end(), "shiftData not found for " << ccy);
324  shiftSize = itr->second.shiftSize;
325  if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
326  vector<Real> strikes = itr->second.shiftStrikes;
327  vector<Period> expiries = itr->second.shiftExpiries;
328  QL_REQUIRE(strikes.size() > 0, "Only strike capfloor vols supported");
329  Size keyIdx = key.index;
330  Size expIdx = keyIdx / strikes.size();
331  Period p_exp = expiries[expIdx];
332  Size strIdx = keyIdx % strikes.size();
333  Real strike = strikes[strIdx];
334  Handle<OptionletVolatilityStructure> vts = simMarket->capFloorVol(ccy, marketConfiguration);
335  Time t_exp = vts->dayCounter().yearFraction(asof, asof + p_exp);
336  Real vol = vts->volatility(t_exp, strike);
337  shiftMult = vol;
338  }
339  } break;
340  case RiskFactorKey::KeyType::CDSVolatility: {
341  string name = keylabel;
342  auto itr = sensiParams.cdsVolShiftData().find(name);
343  QL_REQUIRE(itr != sensiParams.cdsVolShiftData().end(), "shiftData not found for " << name);
344  shiftSize = itr->second.shiftSize;
345  if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
346  vector<Period> expiries = itr->second.shiftExpiries;
347  Size keyIdx = key.index;
348  Size expIdx = keyIdx;
349  Period p_exp = expiries[expIdx];
350  Handle<BlackVolTermStructure> vts = simMarket->cdsVol(name, marketConfiguration);
351  Time t_exp = vts->dayCounter().yearFraction(asof, asof + p_exp);
352  Real strike = 0.0; // FIXME
353  Real atmVol = vts->blackVol(t_exp, strike);
354  shiftMult = atmVol;
355  }
356  } break;
357  case RiskFactorKey::KeyType::SurvivalProbability: {
358  string name = keylabel;
359  auto itr = sensiParams.creditCurveShiftData().find(name);
360  QL_REQUIRE(itr != sensiParams.creditCurveShiftData().end(), "shiftData not found for " << name);
361  shiftSize = itr->second->shiftSize;
362  if (parseShiftType(itr->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
363  Size keyIdx = key.index;
364  Period p = itr->second->shiftTenors[keyIdx];

366  Time t = ts->dayCounter().yearFraction(asof, asof + p);
367  Real prob = ts->survivalProbability(t);
368  shiftMult = -std::log(prob) / t;
369  }
370  } break;
371  case RiskFactorKey::KeyType::BaseCorrelation: {
372  string name = keylabel;
373  auto itr = sensiParams.baseCorrelationShiftData().find(name);
374  QL_REQUIRE(itr != sensiParams.baseCorrelationShiftData().end(), "shiftData not found for " << name);

376  if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
377  vector<Real> lossLevels = itr->second.shiftLossLevels;

379  Size keyIdx = key.index;

382  Size termIdx = keyIdx % terms.size();
383  Period term = terms[termIdx];
384  Handle<BilinearBaseCorrelationTermStructure> ts = simMarket->baseCorrelation(name, marketConfiguration);
385  Real bc = ts->correlation(asof + term, lossLevel, true); // extrapolate
386  shiftMult = bc;

414  Time t = yts->dayCounter().yearFraction(asof, asof + p);

418  } break;
419  case RiskFactorKey::KeyType::YoYInflationCapFloorVolatility: {
420  string name = keylabel;
421  auto itr = sensiParams.yoyInflationCapFloorVolShiftData().find(name);
422  QL_REQUIRE(itr != sensiParams.yoyInflationCapFloorVolShiftData().end(), "shiftData not found for " << name);
423  shiftSize = itr->second.shiftSize;
424  if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
425  vector<Real> strikes = itr->second.shiftStrikes;
426  vector<Period> expiries = itr->second.shiftExpiries;
427  QL_REQUIRE(strikes.size() > 0, "Only strike yoy inflation capfloor vols supported");
428  Size keyIdx = key.index;

430  Period p_exp = expiries[expIdx];
431  Size strIdx = keyIdx % strikes.size();
432  Real strike = strikes[strIdx];

434  Real vol = vts->volatility(p_exp, strike, vts->observationLag());
435  shiftMult = vol;
436  }
437  } break;
438  case RiskFactorKey::KeyType::CommodityCurve: {
439  auto it = sensiParams.commodityCurveShiftData().find(keylabel);
440  QL_REQUIRE(it != sensiParams.commodityCurveShiftData().end(), "shiftData not found for " << keylabel);
441  shiftSize = it->second->shiftSize;
442  if (parseShiftType(it->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
443  Period p = it->second->shiftTenors[key.index];
444  Handle<PriceTermStructure> priceCurve = simMarket->commodityPriceCurve(keylabel, marketConfiguration);
445  Time t = priceCurve->dayCounter().yearFraction(asof, asof + p);
446  shiftMult = priceCurve->price(t);
447  }
448  } break;
449  case RiskFactorKey::KeyType::CommodityVolatility: {
450  auto it = sensiParams.commodityVolShiftData().find(keylabel);
451  QL_REQUIRE(it != sensiParams.commodityVolShiftData().end(), "shiftData not found for " << keylabel);
452 
453  shiftSize = it->second.shiftSize;
454  if (parseShiftType(it->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
455  Size moneynessIndex = key.index / it->second.shiftExpiries.size();
456  Size expiryIndex = key.index % it->second.shiftExpiries.size();
457  Real moneyness = it->second.shiftStrikes[moneynessIndex];
458  Period expiry = it->second.shiftExpiries[expiryIndex];
459  Real spotValue = simMarket->commoditySpot(keylabel, marketConfiguration)->value();
460  Handle<BlackVolTermStructure> vts = simMarket->commodityVolatility(keylabel, marketConfiguration);
461  Time t = vts->dayCounter().yearFraction(asof, asof + expiry);
462  Real vol = vts->blackVol(t, moneyness * spotValue);
463  shiftMult = vol;
464  }
465  } break;
466  case RiskFactorKey::KeyType::SecuritySpread: {
467  auto itr = sensiParams.securityShiftData().find(keylabel);
468  QL_REQUIRE(itr != sensiParams.securityShiftData().end(), "shiftData not found for " << keylabel);
469  shiftSize = itr->second.shiftSize;
470  if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
471  shiftMult = simMarket->securitySpread(keylabel, marketConfiguration)->value();
472  }
473  } break;
474  default:
475  // KeyType::CPIIndex does not get shifted
476  QL_FAIL("KeyType not supported yet - " << keytype);
477  }
478  Real realShift = shiftSize * shiftMult;
479  return realShift;
480 }
