79std::vector<boost::shared_ptr<ValuationCalculator>> SensitivityAnalysis::buildValuationCalculators() const {
80 vector<boost::shared_ptr<ValuationCalculator>> calculators;
81 if (nonShiftedBaseCurrencyConversion_) // use "original" FX rates to convert sensi to base currency
82 calculators.push_back(boost::make_shared<NPVCalculatorFXT0>(simMarketData_->baseCcy(), market_));
83 else // use the scenario FX rate when converting sensi to base currency
84 calculators.push_back(boost::make_shared<NPVCalculator>(simMarketData_->baseCcy()));
85 return calculators;
86}

88void SensitivityAnalysis::initialize(boost::shared_ptr<NPVSensiCube>& cube) {
89 LOG("Build Sensitivity Scenario Generator and Simulation Market");
90 initializeSimMarket();
91
92 LOG("Build Engine Factory and rebuild portfolio");
93 boost::shared_ptr<EngineFactory> factory = buildFactory();
94 resetPortfolio(factory);
95 if (recalibrateModels_)
96 modelBuilders_ = factory->modelBuilders();
97 else
98 modelBuilders_.clear();
99
100 if (!cube) {
101 LOG("Build the cube object to store sensitivities");
102 initializeCube(cube);
103 }
104
105 sensiCube_ =
106 boost::make_shared<SensitivityCube>(cube, scenarioGenerator_->scenarioDescriptions(),
107 scenarioGenerator_->shiftSizes(), sensitivityData_->twoSidedDeltas());
108
109 initialized_ = true;
110}

112void SensitivityAnalysis::generateSensitivities(boost::shared_ptr<NPVSensiCube> cube) {
113
114 QL_REQUIRE(useSingleThreadedEngine_ || cube == nullptr,
115 "SensitivityAnalysis::generateSensitivities(): when using multi-threaded engine no NPVSensiCube should "
116 "be specified, it is built automatically");
117
118 QL_REQUIRE(useSingleThreadedEngine_ || !nonShiftedBaseCurrencyConversion_,
119 "SensitivityAnalysis::generateSensitivities(): multi-threaded engine does not support non-shifted base "
120 "ccy conversion currently. This requires a a small code extension. Contact Dev.");
121
122 QL_REQUIRE(useSingleThreadedEngine_ || recalibrateModels_,
123 "SensitivityAnalysis::generateSensitivities(): multi-threaded engine does not support recalibrateModels "
124 "= false. This requires a small code extension. Contact Dev.");
125
126 if (useSingleThreadedEngine_) {
127
128 // handle single threaded sensi analysis
129
130 QL_REQUIRE(!initialized_, "unexpected state of SensitivitiesAnalysis object (it is already initialized)");
131 initialize(cube);
132 QL_REQUIRE(initialized_, "SensitivitiesAnalysis member objects not correctly initialized");
133 boost::shared_ptr<DateGrid> dg = boost::make_shared<DateGrid>("1,0W", NullCalendar());
134 vector<boost::shared_ptr<ValuationCalculator>> calculators = buildValuationCalculators();
135 ValuationEngine engine(asof_, dg, simMarket_, modelBuilders_);
136 for (auto const& i : this->progressIndicators())
137 engine.registerProgressIndicator(i);
138 LOG("Run Sensitivity Scenarios");
139 engine.buildCube(portfolio_, cube, calculators, true, nullptr, nullptr, {}, dryRun_);
140
141 } else {
142
143 // handle request to use multi-threaded engine
144
145 LOG("SensitivitiyAnalysis::generateSensitivities(): use multi-threaded engine to generate sensi cube. Using "
146 "configuration '"
147 << marketConfiguration_ << "'");
148
149 market_ =
150 boost::make_shared<ore::data::TodaysMarket>(asof_, todaysMarketParams_, loader_, curveConfigs_, true, true,
151 false, referenceData_, false, iborFallbackConfig_, false);
152
153 simMarket_ = boost::make_shared<ScenarioSimMarket>(
154 market_, simMarketData_, marketConfiguration_,
155 curveConfigs_ ? *curveConfigs_ : ore::data::CurveConfigurations(),
156 todaysMarketParams_ ? *todaysMarketParams_ : ore::data::TodaysMarketParameters(), continueOnError_,
157 sensitivityData_->useSpreadedTermStructures(), false, false, iborFallbackConfig_);
158
159 scenarioGenerator_ = boost::make_shared<SensitivityScenarioGenerator>(
160 sensitivityData_, simMarket_->baseScenario(), simMarketData_, simMarket_,
161 boost::make_shared<DeltaScenarioFactory>(simMarket_->baseScenario()), overrideTenors_, continueOnError_,
162 simMarket_->baseScenarioAbsolute());
163
164 simMarket_->scenarioGenerator() = scenarioGenerator_;
165
166 auto ed = boost::make_shared<EngineData>(*engineData_);
167 ed->globalParameters()["RunType"] =
168 std::string("Sensitivity") + (sensitivityData_->computeGamma() ? "DeltaGamma" : "Delta");
169
170 MultiThreadedValuationEngine engine(
171 nThreads_, asof_, boost::make_shared<ore::analytics::DateGrid>(), scenarioGenerator_->numScenarios(),
172 loader_, scenarioGenerator_, ed, curveConfigs_, todaysMarketParams_, marketConfiguration_, simMarketData_,
173 sensitivityData_->useSpreadedTermStructures(), false, boost::make_shared<ore::analytics::ScenarioFilter>(),
174 referenceData_, iborFallbackConfig_, true, true,
175 [](const QuantLib::Date& asof, const std::set<std::string>& ids, const std::vector<QuantLib::Date>&,
176 const QuantLib::Size samples) {
177 return boost::make_shared<ore::analytics::DoublePrecisionSensiCube>(ids, asof, samples);
178 },
179 {}, {}, context_);
180 for (auto const& i : this->progressIndicators())
181 engine.registerProgressIndicator(i);
182
183 auto baseCcy = simMarketData_->baseCcy();
184 engine.buildCube(
185 portfolio_,
186 [&baseCcy]() -> std::vector<boost::shared_ptr<ValuationCalculator>> {
187 return {boost::make_shared<NPVCalculator>(baseCcy)};
188 },
189 {}, true, dryRun_);
190 std::vector<boost::shared_ptr<NPVSensiCube>> miniCubes;
191 for (auto const& c : engine.outputCubes()) {
192 miniCubes.push_back(boost::dynamic_pointer_cast<NPVSensiCube>(c));
193 QL_REQUIRE(miniCubes.back() != nullptr,
194 "SensitivityAnalysis::generateSensitivities(): internal error, could not cast to NPVSensiCube.");
195 }
196 cube = boost::make_shared<JointNPVSensiCube>(miniCubes, portfolio_->ids());
197
198 sensiCube_ =
199 boost::make_shared<SensitivityCube>(cube, scenarioGenerator_->scenarioDescriptions(),
200 scenarioGenerator_->shiftSizes(), sensitivityData_->twoSidedDeltas());
201
202 initialized_ = true;
203 }
204
205 LOG("Sensitivity analysis completed");
206}

217 configurations[MarketContext::pricing] = marketConfiguration_;

235Real getShiftSize(const RiskFactorKey& key, const SensitivityScenarioData& sensiParams,
236 const boost::shared_ptr<ScenarioSimMarket>& simMarket, const string& marketConfiguration) {
237
238 Date asof = simMarket->asofDate();
239 RiskFactorKey::KeyType keytype = key.keytype;
240 string keylabel = key.name;
241 Real shiftSize = 0.0;
242 Real shiftMult = 1.0;
243 switch (keytype) {
244 case RiskFactorKey::KeyType::FXSpot: {
245 auto itr = sensiParams.fxShiftData().find(keylabel);
246 QL_REQUIRE(itr != sensiParams.fxShiftData().end(), "shiftData not found for " << keylabel);
247 shiftSize = itr->second.shiftSize;
248 if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
249 shiftMult = simMarket->fxRate(keylabel, marketConfiguration)->value();
250 }
251 } break;
252 case RiskFactorKey::KeyType::EquitySpot: {
253 auto itr = sensiParams.equityShiftData().find(keylabel);
254 QL_REQUIRE(itr != sensiParams.equityShiftData().end(), "shiftData not found for " << keylabel);
255 shiftSize = itr->second.shiftSize;
256 if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
257 shiftMult = simMarket->equitySpot(keylabel, marketConfiguration)->value();
258 }
259 } break;
260 case RiskFactorKey::KeyType::DiscountCurve: {
261 string ccy = keylabel;
262 auto itr = sensiParams.discountCurveShiftData().find(ccy);
263 QL_REQUIRE(itr != sensiParams.discountCurveShiftData().end(), "shiftData not found for " << ccy);
264 shiftSize = itr->second->shiftSize;
265 if (parseShiftType(itr->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
266 Size keyIdx = key.index;
267 Period p = itr->second->shiftTenors[keyIdx];
268 Handle<YieldTermStructure> yts = simMarket->discountCurve(ccy, marketConfiguration);
269 Time t = yts->dayCounter().yearFraction(asof, asof + p);
270 Real zeroRate = yts->zeroRate(t, Continuous);
271 shiftMult = zeroRate;
272 }
273 } break;
274 case RiskFactorKey::KeyType::IndexCurve: {
275 string idx = keylabel;
276 auto itr = sensiParams.indexCurveShiftData().find(idx);
277 QL_REQUIRE(itr != sensiParams.indexCurveShiftData().end(), "shiftData not found for " << idx);
278 shiftSize = itr->second->shiftSize;
279 if (parseShiftType(itr->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
280 Size keyIdx = key.index;
281 Period p = itr->second->shiftTenors[keyIdx];
282 Handle<YieldTermStructure> yts = simMarket->iborIndex(idx, marketConfiguration)->forwardingTermStructure();
283 Time t = yts->dayCounter().yearFraction(asof, asof + p);
284 Real zeroRate = yts->zeroRate(t, Continuous);
285 shiftMult = zeroRate;
286 }
287 } break;
288 case RiskFactorKey::KeyType::YieldCurve: {
289 string yc = keylabel;
290 auto itr = sensiParams.yieldCurveShiftData().find(yc);
291 QL_REQUIRE(itr != sensiParams.yieldCurveShiftData().end(), "shiftData not found for " << yc);
292 shiftSize = itr->second->shiftSize;
293 if (parseShiftType(itr->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
294 Size keyIdx = key.index;
295 Period p = itr->second->shiftTenors[keyIdx];
296 Handle<YieldTermStructure> yts = simMarket->yieldCurve(yc, marketConfiguration);
297 Time t = yts->dayCounter().yearFraction(asof, asof + p);
298 Real zeroRate = yts->zeroRate(t, Continuous);
299 shiftMult = zeroRate;
300 }
301 } break;
302 case RiskFactorKey::KeyType::DividendYield: {
303 string eq = keylabel;
304 auto itr = sensiParams.dividendYieldShiftData().find(eq);
305 QL_REQUIRE(itr != sensiParams.dividendYieldShiftData().end(), "shiftData not found for " << eq);
306 shiftSize = itr->second->shiftSize;
307
308 if (parseShiftType(itr->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
309 Size keyIdx = key.index;
310 Period p = itr->second->shiftTenors[keyIdx];
311 Handle<YieldTermStructure> ts = simMarket->equityDividendCurve(eq, marketConfiguration);
312 Time t = ts->dayCounter().yearFraction(asof, asof + p);
313 Real zeroRate = ts->zeroRate(t, Continuous);
314 shiftMult = zeroRate;
315 }
316 } break;
317 case RiskFactorKey::KeyType::FXVolatility: {
318 string pair = keylabel;
319 auto itr = sensiParams.fxVolShiftData().find(pair);
320 QL_REQUIRE(itr != sensiParams.fxVolShiftData().end(), "shiftData not found for " << pair);
321 shiftSize = itr->second.shiftSize;
322 if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
323 vector<Real> strikes = sensiParams.fxVolShiftData().at(pair).shiftStrikes;
324 QL_REQUIRE(strikes.size() == 1 && close_enough(strikes[0], 0), "Only ATM FX vols supported");
325 Real atmFwd = 0.0; // hardcoded, since only ATM supported
326 Size keyIdx = key.index;
327 Period p = itr->second.shiftExpiries[keyIdx];
328 Handle<BlackVolTermStructure> vts = simMarket->fxVol(pair, marketConfiguration);
329 Time t = vts->dayCounter().yearFraction(asof, asof + p);
330 Real atmVol = vts->blackVol(t, atmFwd);
331 shiftMult = atmVol;
332 }
333 } break;
334 case RiskFactorKey::KeyType::EquityVolatility: {
335 string pair = keylabel;
336 auto itr = sensiParams.equityVolShiftData().find(pair);
337 QL_REQUIRE(itr != sensiParams.equityVolShiftData().end(), "shiftData not found for " << pair);
338 shiftSize = itr->second.shiftSize;
339 if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
340 Size keyIdx = key.index;
341 Period p = itr->second.shiftExpiries[keyIdx];
342 Handle<BlackVolTermStructure> vts = simMarket->equityVol(pair, marketConfiguration);
343 Time t = vts->dayCounter().yearFraction(asof, asof + p);
344 Real atmVol = vts->blackVol(t, Null<Real>());
345 shiftMult = atmVol;
346 }
347 } break;
348 case RiskFactorKey::KeyType::SwaptionVolatility: {
349 auto itr = sensiParams.swaptionVolShiftData().find(keylabel);
350 QL_REQUIRE(itr != sensiParams.swaptionVolShiftData().end(), "shiftData not found for " << keylabel);
351 shiftSize = itr->second.shiftSize;
352 if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
353 vector<Real> strikes = itr->second.shiftStrikes;
354 vector<Period> tenors = itr->second.shiftTerms;
355 vector<Period> expiries = itr->second.shiftExpiries;
356 Size keyIdx = key.index;
357 Size expIdx = keyIdx / (tenors.size() * strikes.size());
358 Period p_exp = expiries[expIdx];
359 Size tenIdx = keyIdx % tenors.size();
360 Period p_ten = tenors[tenIdx];
361 Real strike = Null<Real>(); // for cubes this will be ATM
362 Handle<SwaptionVolatilityStructure> vts = simMarket->swaptionVol(keylabel, marketConfiguration);
363 Real vol = vts->volatility(p_exp, p_ten, strike);
364 shiftMult = vol;
365 }
366 } break;
367 case RiskFactorKey::KeyType::YieldVolatility: {
368 string securityId = keylabel;
369 auto itr = sensiParams.yieldVolShiftData().find(securityId);
370 QL_REQUIRE(itr != sensiParams.yieldVolShiftData().end(), "shiftData not found for " << securityId);
371 shiftSize = itr->second.shiftSize;
372 if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
373 vector<Real> strikes = itr->second.shiftStrikes;
374 QL_REQUIRE(strikes.size() == 1 && close_enough(strikes[0], 0.0),
375 "shift strikes should be {0.0} for yield volatilities");
376 vector<Period> tenors = itr->second.shiftTerms;
377 vector<Period> expiries = itr->second.shiftExpiries;
378 Size keyIdx = key.index;
379 Size expIdx = keyIdx / (tenors.size() * strikes.size());
380 Period p_exp = expiries[expIdx];
381 Size tenIdx = keyIdx % tenors.size();
382 Period p_ten = tenors[tenIdx];
383 Real strike = Null<Real>(); // for cubes this will be ATM
384 Handle<SwaptionVolatilityStructure> vts = simMarket->yieldVol(securityId, marketConfiguration);
385 Real vol = vts->volatility(p_exp, p_ten, strike);
386 shiftMult = vol;
387 }
388 } break;
389 case RiskFactorKey::KeyType::OptionletVolatility: {
390 string ccy = keylabel;
391 auto itr = sensiParams.capFloorVolShiftData().find(ccy);
392 QL_REQUIRE(itr != sensiParams.capFloorVolShiftData().end(), "shiftData not found for " << ccy);
393 shiftSize = itr->second->shiftSize;
394 if (parseShiftType(itr->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
395 vector<Real> strikes = itr->second->shiftStrikes;
396 vector<Period> expiries = itr->second->shiftExpiries;
397 QL_REQUIRE(strikes.size() > 0, "Only strike capfloor vols supported");
398 shiftMult = simMarket->baseScenario()->get(
399 RiskFactorKey(RiskFactorKey::KeyType::OptionletVolatility, ccy, key.index));
400 }
401 } break;
402 case RiskFactorKey::KeyType::CDSVolatility: {
403 string name = keylabel;
404 auto itr = sensiParams.cdsVolShiftData().find(name);
405 QL_REQUIRE(itr != sensiParams.cdsVolShiftData().end(), "shiftData not found for " << name);
406 shiftSize = itr->second.shiftSize;
407 if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
408 vector<Period> expiries = itr->second.shiftExpiries;
409 Size keyIdx = key.index;
410 Size expIdx = keyIdx;
411 Period p_exp = expiries[expIdx];
412 Handle<CreditVolCurve> vts = simMarket->cdsVol(name, marketConfiguration);
413 // hardcoded 5y term
414 Real atmVol = vts->volatility(asof + p_exp, 5.0, Null<Real>(), vts->type());
415 shiftMult = atmVol;
416 }
417 } break;
418 case RiskFactorKey::KeyType::SurvivalProbability: {
419 string name = keylabel;
420 auto itr = sensiParams.creditCurveShiftData().find(name);
421 QL_REQUIRE(itr != sensiParams.creditCurveShiftData().end(), "shiftData not found for " << name);
422 shiftSize = itr->second->shiftSize;
423 if (parseShiftType(itr->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
424 Size keyIdx = key.index;
425 Period p = itr->second->shiftTenors[keyIdx];
426 Handle<DefaultProbabilityTermStructure> ts = simMarket->defaultCurve(name, marketConfiguration)->curve();
427 Time t = ts->dayCounter().yearFraction(asof, asof + p);
428 Real prob = ts->survivalProbability(t);
429 shiftMult = -std::log(prob) / t;
430 }
431 } break;
432 case RiskFactorKey::KeyType::BaseCorrelation: {
433 string name = keylabel;
434 auto itr = sensiParams.baseCorrelationShiftData().find(name);
435 QL_REQUIRE(itr != sensiParams.baseCorrelationShiftData().end(), "shiftData not found for " << name);
436 shiftSize = itr->second.shiftSize;
437 if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
438 vector<Real> lossLevels = itr->second.shiftLossLevels;
439 vector<Period> terms = itr->second.shiftTerms;
440 Size keyIdx = key.index;
441 Size lossLevelIdx = keyIdx / terms.size();
442 Real lossLevel = lossLevels[lossLevelIdx];
443 Size termIdx = keyIdx % terms.size();
444 Period term = terms[termIdx];
445 Handle<QuantExt::BaseCorrelationTermStructure> ts = simMarket->baseCorrelation(name, marketConfiguration);
446 Real bc = ts->correlation(asof + term, lossLevel, true); // extrapolate
447 shiftMult = bc;
448 }
449 } break;
450 case RiskFactorKey::KeyType::ZeroInflationCurve: {
451 string idx = keylabel;
452 auto itr = sensiParams.zeroInflationCurveShiftData().find(idx);
453 QL_REQUIRE(itr != sensiParams.zeroInflationCurveShiftData().end(), "shiftData not found for " << idx);
454 shiftSize = itr->second->shiftSize;
455 if (parseShiftType(itr->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
456 Size keyIdx = key.index;
457 Period p = itr->second->shiftTenors[keyIdx];
458 Handle<ZeroInflationTermStructure> yts =
459 simMarket->zeroInflationIndex(idx, marketConfiguration)->zeroInflationTermStructure();
460 Time t = yts->dayCounter().yearFraction(asof, asof + p);
461 Real zeroRate = yts->zeroRate(t);
462 shiftMult = zeroRate;
463 }
464 } break;
465 case RiskFactorKey::KeyType::YoYInflationCurve: {
466 string idx = keylabel;
467 auto itr = sensiParams.yoyInflationCurveShiftData().find(idx);
468 QL_REQUIRE(itr != sensiParams.yoyInflationCurveShiftData().end(), "shiftData not found for " << idx);
469 shiftSize = itr->second->shiftSize;
470 if (parseShiftType(itr->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
471 Size keyIdx = key.index;
472 Period p = sensiParams.yoyInflationCurveShiftData().at(idx)->shiftTenors[keyIdx];
473 Handle<YoYInflationTermStructure> yts =
474 simMarket->yoyInflationIndex(idx, marketConfiguration)->yoyInflationTermStructure();
475 Time t = yts->dayCounter().yearFraction(asof, asof + p);
476 Real yoyRate = yts->yoyRate(t);
477 shiftMult = yoyRate;
478 }
479 } break;
480 case RiskFactorKey::KeyType::YoYInflationCapFloorVolatility: {
481 string name = keylabel;
482 auto itr = sensiParams.yoyInflationCapFloorVolShiftData().find(name);
483 QL_REQUIRE(itr != sensiParams.yoyInflationCapFloorVolShiftData().end(), "shiftData not found for " << name);
484 shiftSize = itr->second->shiftSize;
485 if (parseShiftType(itr->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
486 vector<Real> strikes = itr->second->shiftStrikes;
487 vector<Period> expiries = itr->second->shiftExpiries;
488 QL_REQUIRE(strikes.size() > 0, "Only strike yoy inflation capfloor vols supported");
489 Size keyIdx = key.index;
490 Size expIdx = keyIdx / strikes.size();
491 Period p_exp = expiries[expIdx];
492 Size strIdx = keyIdx % strikes.size();
493 Real strike = strikes[strIdx];
494 Handle<QuantExt::YoYOptionletVolatilitySurface> vts = simMarket->yoyCapFloorVol(name, marketConfiguration);
495 Real vol = vts->volatility(p_exp, strike, vts->observationLag());
496 shiftMult = vol;
497 }
498 } break;
499 case RiskFactorKey::KeyType::ZeroInflationCapFloorVolatility: {
500 string name = keylabel;
501 auto itr = sensiParams.zeroInflationCapFloorVolShiftData().find(name);
502 QL_REQUIRE(itr != sensiParams.zeroInflationCapFloorVolShiftData().end(), "shiftData not found for " << name);
503 shiftSize = itr->second->shiftSize;
504 if (parseShiftType(itr->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
505 vector<Real> strikes = itr->second->shiftStrikes;
506 vector<Period> expiries = itr->second->shiftExpiries;
507 QL_REQUIRE(strikes.size() > 0, "Only strike zc inflation capfloor vols supported");
508 Size keyIdx = key.index;
509 Size expIdx = keyIdx / strikes.size();
510 Period p_exp = expiries[expIdx];
511 Size strIdx = keyIdx % strikes.size();
512 Real strike = strikes[strIdx];
513 Handle<CPIVolatilitySurface> vts =
514 simMarket->cpiInflationCapFloorVolatilitySurface(name, marketConfiguration);
515 Real vol = vts->volatility(p_exp, strike, vts->observationLag());
516 shiftMult = vol;
517 }
518 } break;
519 case RiskFactorKey::KeyType::CommodityCurve: {
520 auto it = sensiParams.commodityCurveShiftData().find(keylabel);
521 QL_REQUIRE(it != sensiParams.commodityCurveShiftData().end(), "shiftData not found for " << keylabel);
522 shiftSize = it->second->shiftSize;
523 if (parseShiftType(it->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
524 Period p = it->second->shiftTenors[key.index];
525 Handle<PriceTermStructure> priceCurve = simMarket->commodityPriceCurve(keylabel, marketConfiguration);
526 Time t = priceCurve->dayCounter().yearFraction(asof, asof + p);
527 shiftMult = priceCurve->price(t);
528 }
529 } break;
530 case RiskFactorKey::KeyType::CommodityVolatility: {
531 auto it = sensiParams.commodityVolShiftData().find(keylabel);
532 QL_REQUIRE(it != sensiParams.commodityVolShiftData().end(), "shiftData not found for " << keylabel);
533
534 shiftSize = it->second.shiftSize;
535 if (parseShiftType(it->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
536 Size moneynessIndex = key.index / it->second.shiftExpiries.size();
537 Size expiryIndex = key.index % it->second.shiftExpiries.size();
538 Real moneyness = it->second.shiftStrikes[moneynessIndex];
539 Period expiry = it->second.shiftExpiries[expiryIndex];
540 Real spotValue = simMarket->commodityPriceCurve(keylabel, marketConfiguration)->price(0);
541 Handle<BlackVolTermStructure> vts = simMarket->commodityVolatility(keylabel, marketConfiguration);
542 Time t = vts->dayCounter().yearFraction(asof, asof + expiry);
543 Real vol = vts->blackVol(t, moneyness * spotValue);
544 shiftMult = vol;
545 }
546 } break;
547 case RiskFactorKey::KeyType::SecuritySpread: {
548 auto itr = sensiParams.securityShiftData().find(keylabel);
549 QL_REQUIRE(itr != sensiParams.securityShiftData().end(), "shiftData not found for " << keylabel);
550 shiftSize = itr->second.shiftSize;
551 if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
552 shiftMult = 1.0;
553 try {
554 shiftMult = simMarket->securitySpread(keylabel, marketConfiguration)->value();
555 } catch (...) {
556 // if there is no spread given for a security, we return 1.0, this is not relevant anyway,
557 // because there will be no scenario generated for this risk factor
558 }
559 }
560 } break;
561 default:
562 // KeyType::CPIIndex does not get shifted
563 QL_FAIL("KeyType not supported yet - " << keytype);
564 }
565 Real realShift = shiftSize * shiftMult;
566 return realShift;
567}
