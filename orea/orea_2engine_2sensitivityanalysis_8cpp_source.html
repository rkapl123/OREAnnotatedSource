72std::vector<boost::shared_ptr<ValuationCalculator>> SensitivityAnalysis::buildValuationCalculators() const {
73 vector<boost::shared_ptr<ValuationCalculator>> calculators;
74 if (nonShiftedBaseCurrencyConversion_) // use "original" FX rates to convert sensi to base currency
75 calculators.push_back(boost::make_shared<NPVCalculatorFXT0>(simMarketData_->baseCcy(), market_));
76 else // use the scenario FX rate when converting sensi to base currency
77 calculators.push_back(boost::make_shared<NPVCalculator>(simMarketData_->baseCcy()));
78 return calculators;
79}

81void SensitivityAnalysis::initialize(boost::shared_ptr<NPVSensiCube>& cube) {
82 LOG("Build Sensitivity Scenario Generator and Simulation Market");
83 initializeSimMarket();
84
85 LOG("Build Engine Factory and rebuild portfolio");
86 boost::shared_ptr<EngineFactory> factory = buildFactory();
87 resetPortfolio(factory);
88 if (recalibrateModels_)
89 modelBuilders_ = factory->modelBuilders();
90 else
91 modelBuilders_.clear();
92
93 if (!cube) {
94 LOG("Build the cube object to store sensitivities");
95 initializeCube(cube);
96 }
97
98 sensiCube_ =
99 boost::make_shared<SensitivityCube>(cube, scenarioGenerator_->scenarioDescriptions(),
100 scenarioGenerator_->shiftSizes(), sensitivityData_->twoSidedDeltas());
101
102 initialized_ = true;
103}

105void SensitivityAnalysis::generateSensitivities(boost::shared_ptr<NPVSensiCube> cube) {
106
107 QL_REQUIRE(!initialized_, "unexpected state of SensitivitiesAnalysis object");
108
109 // initialize the helper member objects
110 initialize(cube);
111 QL_REQUIRE(initialized_, "SensitivitiesAnalysis member objects not correctly initialized");
112 boost::shared_ptr<DateGrid> dg = boost::make_shared<DateGrid>("1,0W", NullCalendar());
113 vector<boost::shared_ptr<ValuationCalculator>> calculators = buildValuationCalculators();
114 ValuationEngine engine(asof_, dg, simMarket_, modelBuilders_);
115 for (auto const& i : this->progressIndicators())
116 engine.registerProgressIndicator(i);
117 LOG("Run Sensitivity Scenarios");
118 engine.buildCube(portfolio_, cube, calculators, true, nullptr, nullptr, {}, dryRun_);
119
120 computed_ = true;
121
122 LOG("Sensitivity analysis completed");
123}

125void SensitivityAnalysis::initializeSimMarket(boost::shared_ptr<ScenarioFactory> scenFact) {
126
127 LOG("Initialise sim market for sensitivity analysis (continueOnError=" << std::boolalpha << continueOnError_
128 << ")");
129 simMarket_ = boost::make_shared<ScenarioSimMarket>(
130 market_, simMarketData_, marketConfiguration_,
131 curveConfigs_ ? *curveConfigs_ : ore::data::CurveConfigurations(),
132 todaysMarketParams_ ? *todaysMarketParams_ : TodaysMarketParameters(), continueOnError_,
133 sensitivityData_->useSpreadedTermStructures(), false, false, iborFallbackConfig_);
134
135 LOG("Sim market initialised for sensitivity analysis");
136
137 LOG("Create scenario factory for sensitivity analysis");
138 boost::shared_ptr<Scenario> baseScenario = simMarket_->baseScenario();
139 boost::shared_ptr<ScenarioFactory> scenarioFactory =
140 scenFact ? scenFact : boost::make_shared<CloneScenarioFactory>(baseScenario);
141 LOG("Scenario factory created for sensitivity analysis");
142
143 LOG("Create scenario generator for sensitivity analysis (continueOnError=" << std::boolalpha << continueOnError_
144 << ")");
145 scenarioGenerator_ = boost::make_shared<SensitivityScenarioGenerator>(
146 sensitivityData_, baseScenario, simMarketData_, simMarket_, scenarioFactory, overrideTenors_, continueOnError_);
147 LOG("Scenario generator created for sensitivity analysis");
148
149 // Set simulation market's scenario generator
150 simMarket_->scenarioGenerator() = scenarioGenerator_;
151}

153boost::shared_ptr<EngineFactory> SensitivityAnalysis::buildFactory() const {
154 map<MarketContext, string> configurations;
155 configurations[MarketContext::pricing] = marketConfiguration_;
156 auto ed = boost::make_shared<EngineData>(*engineData_);
157 ed->globalParameters()["RunType"] =
158 std::string("Sensitivity") + (sensitivityData_->computeGamma() ? "DeltaGamma" : "Delta");
159 boost::shared_ptr<EngineFactory> factory =
160 boost::make_shared<EngineFactory>(ed, simMarket_, configurations, referenceData_, iborFallbackConfig_);
161 return factory;
162}

173Real getShiftSize(const RiskFactorKey& key, const SensitivityScenarioData& sensiParams,
174 const boost::shared_ptr<ScenarioSimMarket>& simMarket, const string& marketConfiguration) {
175
176 Date asof = simMarket->asofDate();
177 RiskFactorKey::KeyType keytype = key.keytype;
178 string keylabel = key.name;
179 Real shiftSize = 0.0;
180 Real shiftMult = 1.0;
181 switch (keytype) {
182 case RiskFactorKey::KeyType::FXSpot: {
183 auto itr = sensiParams.fxShiftData().find(keylabel);
184 QL_REQUIRE(itr != sensiParams.fxShiftData().end(), "shiftData not found for " << keylabel);
185 shiftSize = itr->second.shiftSize;
186 if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
187 shiftMult = simMarket->fxRate(keylabel, marketConfiguration)->value();
188 }
189 } break;
190 case RiskFactorKey::KeyType::EquitySpot: {
191 auto itr = sensiParams.equityShiftData().find(keylabel);
192 QL_REQUIRE(itr != sensiParams.equityShiftData().end(), "shiftData not found for " << keylabel);
193 shiftSize = itr->second.shiftSize;
194 if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
195 shiftMult = simMarket->equitySpot(keylabel, marketConfiguration)->value();
196 }
197 } break;
198 case RiskFactorKey::KeyType::DiscountCurve: {
199 string ccy = keylabel;
200 auto itr = sensiParams.discountCurveShiftData().find(ccy);
201 QL_REQUIRE(itr != sensiParams.discountCurveShiftData().end(), "shiftData not found for " << ccy);
202 shiftSize = itr->second->shiftSize;
203 if (parseShiftType(itr->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
204 Size keyIdx = key.index;
205 Period p = itr->second->shiftTenors[keyIdx];
206 Handle<YieldTermStructure> yts = simMarket->discountCurve(ccy, marketConfiguration);
207 Time t = yts->dayCounter().yearFraction(asof, asof + p);
208 Real zeroRate = yts->zeroRate(t, Continuous);
209 shiftMult = zeroRate;
210 }
211 } break;
212 case RiskFactorKey::KeyType::IndexCurve: {
213 string idx = keylabel;
214 auto itr = sensiParams.indexCurveShiftData().find(idx);
215 QL_REQUIRE(itr != sensiParams.indexCurveShiftData().end(), "shiftData not found for " << idx);
216 shiftSize = itr->second->shiftSize;
217 if (parseShiftType(itr->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
218 Size keyIdx = key.index;
219 Period p = itr->second->shiftTenors[keyIdx];
220 Handle<YieldTermStructure> yts = simMarket->iborIndex(idx, marketConfiguration)->forwardingTermStructure();
221 Time t = yts->dayCounter().yearFraction(asof, asof + p);
222 Real zeroRate = yts->zeroRate(t, Continuous);
223 shiftMult = zeroRate;
224 }
225 } break;
226 case RiskFactorKey::KeyType::YieldCurve: {
227 string yc = keylabel;
228 auto itr = sensiParams.yieldCurveShiftData().find(yc);
229 QL_REQUIRE(itr != sensiParams.yieldCurveShiftData().end(), "shiftData not found for " << yc);
230 shiftSize = itr->second->shiftSize;
231 if (parseShiftType(itr->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
232 Size keyIdx = key.index;
233 Period p = itr->second->shiftTenors[keyIdx];
234 Handle<YieldTermStructure> yts = simMarket->yieldCurve(yc, marketConfiguration);
235 Time t = yts->dayCounter().yearFraction(asof, asof + p);
236 Real zeroRate = yts->zeroRate(t, Continuous);
237 shiftMult = zeroRate;
238 }
239 } break;
240 case RiskFactorKey::KeyType::DividendYield: {
241 string eq = keylabel;
242 auto itr = sensiParams.dividendYieldShiftData().find(eq);
243 QL_REQUIRE(itr != sensiParams.dividendYieldShiftData().end(), "shiftData not found for " << eq);
244 shiftSize = itr->second->shiftSize;
245
246 if (parseShiftType(itr->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
247 Size keyIdx = key.index;
248 Period p = itr->second->shiftTenors[keyIdx];
249 Handle<YieldTermStructure> ts = simMarket->equityDividendCurve(eq, marketConfiguration);
250 Time t = ts->dayCounter().yearFraction(asof, asof + p);
251 Real zeroRate = ts->zeroRate(t, Continuous);
252 shiftMult = zeroRate;
253 }
254 } break;
255 case RiskFactorKey::KeyType::FXVolatility: {
256 string pair = keylabel;
257 auto itr = sensiParams.fxVolShiftData().find(pair);
258 QL_REQUIRE(itr != sensiParams.fxVolShiftData().end(), "shiftData not found for " << pair);
259 shiftSize = itr->second.shiftSize;
260 if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
261 vector<Real> strikes = sensiParams.fxVolShiftData().at(pair).shiftStrikes;
262 QL_REQUIRE(strikes.size() == 1 && close_enough(strikes[0], 0), "Only ATM FX vols supported");
263 Real atmFwd = 0.0; // hardcoded, since only ATM supported
264 Size keyIdx = key.index;
265 Period p = itr->second.shiftExpiries[keyIdx];
266 Handle<BlackVolTermStructure> vts = simMarket->fxVol(pair, marketConfiguration);
267 Time t = vts->dayCounter().yearFraction(asof, asof + p);
268 Real atmVol = vts->blackVol(t, atmFwd);
269 shiftMult = atmVol;
270 }
271 } break;
272 case RiskFactorKey::KeyType::EquityVolatility: {
273 string pair = keylabel;
274 auto itr = sensiParams.equityVolShiftData().find(pair);
275 QL_REQUIRE(itr != sensiParams.equityVolShiftData().end(), "shiftData not found for " << pair);
276 shiftSize = itr->second.shiftSize;
277 if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
278 Size keyIdx = key.index;
279 Period p = itr->second.shiftExpiries[keyIdx];
280 Handle<BlackVolTermStructure> vts = simMarket->equityVol(pair, marketConfiguration);
281 Time t = vts->dayCounter().yearFraction(asof, asof + p);
282 Real atmVol = vts->blackVol(t, Null<Real>());
283 shiftMult = atmVol;
284 }
285 } break;
286 case RiskFactorKey::KeyType::SwaptionVolatility: {
287 auto itr = sensiParams.swaptionVolShiftData().find(keylabel);
288 QL_REQUIRE(itr != sensiParams.swaptionVolShiftData().end(), "shiftData not found for " << keylabel);
289 shiftSize = itr->second.shiftSize;
290 if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
291 vector<Real> strikes = itr->second.shiftStrikes;
292 vector<Period> tenors = itr->second.shiftTerms;
293 vector<Period> expiries = itr->second.shiftExpiries;
294 Size keyIdx = key.index;
295 Size expIdx = keyIdx / (tenors.size() * strikes.size());
296 Period p_exp = expiries[expIdx];
297 Size tenIdx = keyIdx % tenors.size();
298 Period p_ten = tenors[tenIdx];
299 Real strike = Null<Real>(); // for cubes this will be ATM
300 Handle<SwaptionVolatilityStructure> vts = simMarket->swaptionVol(keylabel, marketConfiguration);
301 Real vol = vts->volatility(p_exp, p_ten, strike);
302 shiftMult = vol;
303 }
304 } break;
305 case RiskFactorKey::KeyType::YieldVolatility: {
306 string securityId = keylabel;
307 auto itr = sensiParams.yieldVolShiftData().find(securityId);
308 QL_REQUIRE(itr != sensiParams.yieldVolShiftData().end(), "shiftData not found for " << securityId);
309 shiftSize = itr->second.shiftSize;
310 if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
311 vector<Real> strikes = itr->second.shiftStrikes;
312 QL_REQUIRE(strikes.size() == 1 && close_enough(strikes[0], 0.0),
313 "shift strikes should be {0.0} for yield volatilities");
314 vector<Period> tenors = itr->second.shiftTerms;
315 vector<Period> expiries = itr->second.shiftExpiries;
316 Size keyIdx = key.index;
317 Size expIdx = keyIdx / (tenors.size() * strikes.size());
318 Period p_exp = expiries[expIdx];
319 Size tenIdx = keyIdx % tenors.size();
320 Period p_ten = tenors[tenIdx];
321 Real strike = Null<Real>(); // for cubes this will be ATM
322 Handle<SwaptionVolatilityStructure> vts = simMarket->yieldVol(securityId, marketConfiguration);
323 Real vol = vts->volatility(p_exp, p_ten, strike);
324 shiftMult = vol;
325 }
326 } break;
327 case RiskFactorKey::KeyType::OptionletVolatility: {
328 string ccy = keylabel;
329 auto itr = sensiParams.capFloorVolShiftData().find(ccy);
330 QL_REQUIRE(itr != sensiParams.capFloorVolShiftData().end(), "shiftData not found for " << ccy);
331 shiftSize = itr->second->shiftSize;
332 if (parseShiftType(itr->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
333 vector<Real> strikes = itr->second->shiftStrikes;
334 vector<Period> expiries = itr->second->shiftExpiries;
335 QL_REQUIRE(strikes.size() > 0, "Only strike capfloor vols supported");
336 shiftMult = simMarket->baseScenario()->get(
337 RiskFactorKey(RiskFactorKey::KeyType::OptionletVolatility, ccy, key.index));
338 }
339 } break;
340 case RiskFactorKey::KeyType::CDSVolatility: {
341 string name = keylabel;
342 auto itr = sensiParams.cdsVolShiftData().find(name);
343 QL_REQUIRE(itr != sensiParams.cdsVolShiftData().end(), "shiftData not found for " << name);
344 shiftSize = itr->second.shiftSize;
345 if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
346 vector<Period> expiries = itr->second.shiftExpiries;
347 Size keyIdx = key.index;
348 Size expIdx = keyIdx;
349 Period p_exp = expiries[expIdx];
350 Handle<CreditVolCurve> vts = simMarket->cdsVol(name, marketConfiguration);
351 // hardcoded 5y term
352 Real atmVol = vts->volatility(asof + p_exp, 5.0, Null<Real>(), vts->type());
353 shiftMult = atmVol;
354 }
355 } break;
356 case RiskFactorKey::KeyType::SurvivalProbability: {
357 string name = keylabel;
358 auto itr = sensiParams.creditCurveShiftData().find(name);
359 QL_REQUIRE(itr != sensiParams.creditCurveShiftData().end(), "shiftData not found for " << name);
360 shiftSize = itr->second->shiftSize;
361 if (parseShiftType(itr->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
362 Size keyIdx = key.index;
363 Period p = itr->second->shiftTenors[keyIdx];
364 Handle<DefaultProbabilityTermStructure> ts = simMarket->defaultCurve(name, marketConfiguration)->curve();
365 Time t = ts->dayCounter().yearFraction(asof, asof + p);
366 Real prob = ts->survivalProbability(t);
367 shiftMult = -std::log(prob) / t;
368 }
369 } break;
370 case RiskFactorKey::KeyType::BaseCorrelation: {
371 string name = keylabel;
372 auto itr = sensiParams.baseCorrelationShiftData().find(name);
373 QL_REQUIRE(itr != sensiParams.baseCorrelationShiftData().end(), "shiftData not found for " << name);
374 shiftSize = itr->second.shiftSize;
375 if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
376 vector<Real> lossLevels = itr->second.shiftLossLevels;
377 vector<Period> terms = itr->second.shiftTerms;
378 Size keyIdx = key.index;
379 Size lossLevelIdx = keyIdx / terms.size();
380 Real lossLevel = lossLevels[lossLevelIdx];
381 Size termIdx = keyIdx % terms.size();
382 Period term = terms[termIdx];
383 Handle<QuantExt::BaseCorrelationTermStructure> ts = simMarket->baseCorrelation(name, marketConfiguration);
384 Real bc = ts->correlation(asof + term, lossLevel, true); // extrapolate
385 shiftMult = bc;
386 }
387 } break;
388 case RiskFactorKey::KeyType::ZeroInflationCurve: {
389 string idx = keylabel;
390 auto itr = sensiParams.zeroInflationCurveShiftData().find(idx);
391 QL_REQUIRE(itr != sensiParams.zeroInflationCurveShiftData().end(), "shiftData not found for " << idx);
392 shiftSize = itr->second->shiftSize;
393 if (parseShiftType(itr->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
394 Size keyIdx = key.index;
395 Period p = itr->second->shiftTenors[keyIdx];
396 Handle<ZeroInflationTermStructure> yts =
397 simMarket->zeroInflationIndex(idx, marketConfiguration)->zeroInflationTermStructure();
398 Time t = yts->dayCounter().yearFraction(asof, asof + p);
399 Real zeroRate = yts->zeroRate(t);
400 shiftMult = zeroRate;
401 }
402 } break;
403 case RiskFactorKey::KeyType::YoYInflationCurve: {
404 string idx = keylabel;
405 auto itr = sensiParams.yoyInflationCurveShiftData().find(idx);
406 QL_REQUIRE(itr != sensiParams.yoyInflationCurveShiftData().end(), "shiftData not found for " << idx);
407 shiftSize = itr->second->shiftSize;
408 if (parseShiftType(itr->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
409 Size keyIdx = key.index;
410 Period p = sensiParams.yoyInflationCurveShiftData().at(idx)->shiftTenors[keyIdx];
411 Handle<YoYInflationTermStructure> yts =
412 simMarket->yoyInflationIndex(idx, marketConfiguration)->yoyInflationTermStructure();
413 Time t = yts->dayCounter().yearFraction(asof, asof + p);
414 Real yoyRate = yts->yoyRate(t);
415 shiftMult = yoyRate;
416 }
417 } break;
418 case RiskFactorKey::KeyType::YoYInflationCapFloorVolatility: {
419 string name = keylabel;
420 auto itr = sensiParams.yoyInflationCapFloorVolShiftData().find(name);
421 QL_REQUIRE(itr != sensiParams.yoyInflationCapFloorVolShiftData().end(), "shiftData not found for " << name);
422 shiftSize = itr->second->shiftSize;
423 if (parseShiftType(itr->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
424 vector<Real> strikes = itr->second->shiftStrikes;
425 vector<Period> expiries = itr->second->shiftExpiries;
426 QL_REQUIRE(strikes.size() > 0, "Only strike yoy inflation capfloor vols supported");
427 Size keyIdx = key.index;
428 Size expIdx = keyIdx / strikes.size();
429 Period p_exp = expiries[expIdx];
430 Size strIdx = keyIdx % strikes.size();
431 Real strike = strikes[strIdx];
432 Handle<QuantExt::YoYOptionletVolatilitySurface> vts = simMarket->yoyCapFloorVol(name, marketConfiguration);
433 Real vol = vts->volatility(p_exp, strike, vts->observationLag());
434 shiftMult = vol;
435 }
436 } break;
437 case RiskFactorKey::KeyType::ZeroInflationCapFloorVolatility: {
438 string name = keylabel;
439 auto itr = sensiParams.zeroInflationCapFloorVolShiftData().find(name);
440 QL_REQUIRE(itr != sensiParams.zeroInflationCapFloorVolShiftData().end(), "shiftData not found for " << name);
441 shiftSize = itr->second->shiftSize;
442 if (parseShiftType(itr->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
443 vector<Real> strikes = itr->second->shiftStrikes;
444 vector<Period> expiries = itr->second->shiftExpiries;
445 QL_REQUIRE(strikes.size() > 0, "Only strike zc inflation capfloor vols supported");
446 Size keyIdx = key.index;
447 Size expIdx = keyIdx / strikes.size();
448 Period p_exp = expiries[expIdx];
449 Size strIdx = keyIdx % strikes.size();
450 Real strike = strikes[strIdx];
451 Handle<CPIVolatilitySurface> vts =
452 simMarket->cpiInflationCapFloorVolatilitySurface(name, marketConfiguration);
453 Real vol = vts->volatility(p_exp, strike, vts->observationLag());
454 shiftMult = vol;
455 }
456 } break;
457 case RiskFactorKey::KeyType::CommodityCurve: {
458 auto it = sensiParams.commodityCurveShiftData().find(keylabel);
459 QL_REQUIRE(it != sensiParams.commodityCurveShiftData().end(), "shiftData not found for " << keylabel);
460 shiftSize = it->second->shiftSize;
461 if (parseShiftType(it->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
462 Period p = it->second->shiftTenors[key.index];
463 Handle<PriceTermStructure> priceCurve = simMarket->commodityPriceCurve(keylabel, marketConfiguration);
464 Time t = priceCurve->dayCounter().yearFraction(asof, asof + p);
465 shiftMult = priceCurve->price(t);
466 }
467 } break;
468 case RiskFactorKey::KeyType::CommodityVolatility: {
469 auto it = sensiParams.commodityVolShiftData().find(keylabel);
470 QL_REQUIRE(it != sensiParams.commodityVolShiftData().end(), "shiftData not found for " << keylabel);
471
472 shiftSize = it->second.shiftSize;
473 if (parseShiftType(it->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
474 Size moneynessIndex = key.index / it->second.shiftExpiries.size();
475 Size expiryIndex = key.index % it->second.shiftExpiries.size();
476 Real moneyness = it->second.shiftStrikes[moneynessIndex];
477 Period expiry = it->second.shiftExpiries[expiryIndex];
478 Real spotValue = simMarket->commodityPriceCurve(keylabel, marketConfiguration)->price(0);
479 Handle<BlackVolTermStructure> vts = simMarket->commodityVolatility(keylabel, marketConfiguration);
480 Time t = vts->dayCounter().yearFraction(asof, asof + expiry);
481 Real vol = vts->blackVol(t, moneyness * spotValue);
482 shiftMult = vol;
483 }
484 } break;
485 case RiskFactorKey::KeyType::SecuritySpread: {
486 auto itr = sensiParams.securityShiftData().find(keylabel);
487 QL_REQUIRE(itr != sensiParams.securityShiftData().end(), "shiftData not found for " << keylabel);
488 shiftSize = itr->second.shiftSize;
489 if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
490 shiftMult = 1.0;
491 try {
492 shiftMult = simMarket->securitySpread(keylabel, marketConfiguration)->value();
493 } catch (...) {
494 // if there is no spread given for a security, we return 1.0, this is not relevant anyway,
495 // because there will be no scenario generated for this risk factor
496 }
497 }
498 } break;
499 default:
500 // KeyType::CPIIndex does not get shifted
501 QL_FAIL("KeyType not supported yet - " << keytype);
502 }
503 Real realShift = shiftSize * shiftMult;
504 return realShift;
