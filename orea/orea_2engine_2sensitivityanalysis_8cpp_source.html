76 std::vector<boost::shared_ptr<ValuationCalculator>> SensitivityAnalysis::buildValuationCalculators() const {
77  vector<boost::shared_ptr<ValuationCalculator>> calculators;
78  if (nonShiftedBaseCurrencyConversion_) // use "original" FX rates to convert sensi to base currency
79  calculators.push_back(boost::make_shared<NPVCalculatorFXT0>(simMarketData_->baseCcy(), market_));

109 void SensitivityAnalysis::generateSensitivities(boost::shared_ptr<NPVSensiCube> cube) {
110 
111  QL_REQUIRE(!initialized_, "unexpected state of SensitivitiesAnalysis object");
112 
113  // initialize the helper member objects
114  initialize(cube);
115  QL_REQUIRE(initialized_, "SensitivitiesAnalysis member objects not correctly initialized");
116  boost::shared_ptr<DateGrid> dg = boost::make_shared<DateGrid>("1,0W", NullCalendar());
117  vector<boost::shared_ptr<ValuationCalculator>> calculators = buildValuationCalculators();
118  ValuationEngine engine(asof_, dg, simMarket_, modelBuilders_);
119  for (auto const& i : this->progressIndicators())
120  engine.registerProgressIndicator(i);
121  LOG("Run Sensitivity Scenarios");
122  engine.buildCube(portfolio_, cube, calculators, true, nullptr, nullptr, {}, dryRun_);
123 
124  addAnalyticFxSensitivities();
125 
126  computed_ = true;
127 
128  LOG("Sensitivity analysis completed");
129 }

181 void SensitivityAnalysis::addAnalyticFxSensitivities() {
182 
183  if (!analyticFxSensis_) {
184  return;
185  }
186 
187  DLOG("addAnalyticFxSensitivities: start overwriting FX sensitivities with analytic values.");
188 
189  // We want simulation market in its initial state for calculating analytic values.
190  simMarket_->reset();
191 
192  // The currency of the sensitivities.
193  string baseCcy = simMarketData_->baseCcy();
194 
195  for (const auto& trade : portfolio_->trades()) {
196 
197  if (auto fxo = boost::dynamic_pointer_cast<FxOption>(trade)) {
198 
199  DLOG("addAnalyticFxSensitivities: trade id " << trade->id() << " is an FX option so include.");
200 
201  // Skip this option if it has matured.
202  if (QuantLib::detail::simple_event(fxo->maturity()).hasOccurred()) {
203  DLOG("addAnalyticFxSensitivities: skipping FX option with id "
204  << trade->id() << " as it has matured. Maturity is " << io::iso_date(fxo->maturity()) << ".");
205  continue;
206  }
207 
208  Size tradeIdx = sensiCube_->tradeIdx().at(trade->id());
209 
210  const auto& domCcy = fxo->soldCurrency();
211  if (isMetal(parseCurrency(domCcy))) {
212  DLOG("addAnalyticFxSensitivities: FX option " << trade->id() << " references the price of a metal, "
213  << domCcy << ", so skipping it.");
214  continue;
215  }
216  string pairDomBase = domCcy + baseCcy;
217  RiskFactorKey domCcyKey(RiskFactorKey::KeyType::FXSpot, pairDomBase);
218  Real domCcySensi = Null<Real>();
219 
220  const auto& forCcy = fxo->boughtCurrency();
221  if (isMetal(parseCurrency(forCcy))) {
222  DLOG("addAnalyticFxSensitivities: FX option " << trade->id() << " references the price of a metal, "
223  << forCcy << ", so skipping it.");
224  continue;
225  }
226  string pairForBase = forCcy + baseCcy;
227  RiskFactorKey forCcyKey(RiskFactorKey::KeyType::FXSpot, pairForBase);
228  Real forCcySensi = Null<Real>();
229 
230  const auto& instWrapper = fxo->instrument();
231  auto qlInst = instWrapper->qlInstrument();
232  auto qlMult = instWrapper->multiplier();
233 
234  if (auto vo = boost::dynamic_pointer_cast<VanillaOption>(qlInst)) {
235 
236  // Some engines don't populate delta, e.g. AnalyticEuropeanEngineDeltaGamma, so need a try catch here.
237  Real delta = Null<Real>();
238  try {
239  delta = vo->delta();
240  } catch (...) {
241  DLOG("addAnalyticFxSensitivities: underlying engine for trade id "
242  << trade->id() << " does not provide delta so skipping this trade.");
243  continue;
244  }
245 
246  // Delta may possibly still not populated at this point in some cases but unlikely.
247  if (delta == Null<Real>()) {
248  DLOG("addAnalyticFxSensitivities: underlying engine for trade id "
249  << trade->id() << " does not populate delta so skipping this trade.");
250  continue;
251  }
252 
253  if (forCcy != baseCcy) {
254  Real shiftSize = getShiftSize(forCcyKey, *sensitivityData_, simMarket_);
255  forCcySensi = shiftSize * delta * qlMult;
256  TLOG("addAnalyticFxSensitivities: currency "
257  << forCcy << " sensitivity " << forCcySensi << ". (" << pairForBase
258  << " shift, delta, mult) = (" << std::fixed << std::setprecision(9) << shiftSize << ","
259  << delta << "," << qlMult << ").");
260  }
261 
262  if (domCcy != baseCcy) {
263  string fxPair = forCcy + domCcy;
264  Real spot = simMarket_->fxRate(fxPair)->value();
265  Real altDelta = vo->NPV() - spot * delta;
266  Real shiftSize = getShiftSize(domCcyKey, *sensitivityData_, simMarket_);
267  domCcySensi = shiftSize * altDelta * qlMult;
268  TLOG("addAnalyticFxSensitivities: currency "
269  << domCcy << " sensitivity " << domCcySensi << ". (" << domCcy << " npv, " << fxPair << ", "
270  << pairDomBase << " shift, delta, mult) = (" << std::fixed << std::setprecision(9) << vo->NPV()
271  << "," << spot << "," << shiftSize << "," << altDelta << "," << qlMult << ").");
272  }
273  }
274 
275  // If we have nothing by now continue skip to next trade.
276  if (domCcySensi == Null<Real>() && forCcySensi == Null<Real>()) {
277  DLOG("addAnalyticFxSensitivities: could not populate an analytical FX delta for trade id "
278  << trade->id() << " so skipping this trade.");
279  continue;
280  }
281 
282  // Need to include the premium also. Expect one additional "instrument" for this.
283  // If greater than one, abandon attempt to add analytical FX sensitivities.
284  // We can leave the original sensitivity if the premium currency is not equal to dom or for currency.
285  const auto& addInsts = instWrapper->additionalInstruments();
286  const auto& addMults = instWrapper->additionalMultipliers();
287  if (addInsts.size() > 1) {
288  DLOG("addAnalyticFxSensitivities: more than one additional instrument so skip.");
289  continue;
290  } else if (addInsts.size() == 1) {
291  QL_REQUIRE(addMults.size() == 1, "addAnalyticFxSensitivities: number of additional multipliers "
292  << "should always equal the number of additional instruments.");
293  if (auto fee = boost::dynamic_pointer_cast<QuantExt::Payment>(addInsts[0])) {
294  string feeCcy = fee->currency().code();
295  if (feeCcy != baseCcy && (feeCcy == domCcy || feeCcy == forCcy)) {
296 
297  Real feeCcySensi;
298  if (domCcy == baseCcy) {
299  Real spot = simMarket_->fxRate(pairForBase)->value();
300  Real feeNpvForCcy = fee->NPV() / spot;
301  Real shiftSize = getShiftSize(forCcyKey, *sensitivityData_, simMarket_);
302  feeCcySensi = addMults[0] * shiftSize * feeNpvForCcy;
303  TLOG("addAnalyticFxSensitivities: fee sensitivity "
304  << feeCcySensi << ". (feeCcy, baseCcy, feeNpvForCcy, " << pairForBase << ", "
305  << pairForBase << " shift, mult) = (" << std::fixed << std::setprecision(9) << feeCcy
306  << "," << baseCcy << "," << feeNpvForCcy << "," << spot << "," << shiftSize << ","
307  << addMults[0] << ").");
308  } else {
309  Real shiftSize = getShiftSize(domCcyKey, *sensitivityData_, simMarket_);
310  feeCcySensi = addMults[0] * shiftSize * fee->NPV();
311  TLOG("addAnalyticFxSensitivities: fee sensitivity "
312  << feeCcySensi << ". (feeCcy, baseCcy, feeNpvDomCcy, " << pairDomBase
313  << " shift, mult) = (" << std::fixed << std::setprecision(9) << feeCcy << ","
314  << baseCcy << "," << fee->NPV() << "," << shiftSize << "," << addMults[0] << ").");
315  }
316 
317  if (feeCcy == domCcy) {
318  domCcySensi += feeCcySensi;
319  } else {
320  forCcySensi += feeCcySensi;
321  }
322  }
323  }
324  }
325 
326  // Overwrite the sensitivity cube entries.
327  auto upFactors = sensiCube_->upFactors().left;
328  auto downFactors = sensiCube_->downFactors();
329  auto baseNpv = sensiCube_->npvCube()->getT0(tradeIdx);
330 
331  // Sensitivity to foreign currency.
332  if (forCcySensi != Null<Real>()) {
333  auto itu = upFactors.find(forCcyKey);
334  if (itu != upFactors.end()) {
335  TLOG("Adding analytic " << forCcy << " up sensitivity to cube.");
336  sensiCube_->npvCube()->set(baseNpv + forCcySensi, tradeIdx, itu->second.index);
337  }
338  auto itd = downFactors.find(forCcyKey);
339  if (itd != downFactors.end()) {
340  TLOG("Adding analytic " << forCcy << " down sensitivity to cube.");
341  sensiCube_->npvCube()->set(baseNpv - forCcySensi, tradeIdx, itd->second.index);
342  }
343  }
344 
345  // Sensitivity to domestic currency.
346  if (domCcySensi != Null<Real>()) {
347  auto itu = upFactors.find(domCcyKey);
348  if (itu != upFactors.end()) {
349  TLOG("Adding analytic " << domCcy << " up sensitivity to cube.");
350  sensiCube_->npvCube()->set(baseNpv + domCcySensi, tradeIdx, itu->second.index);
351  }
352  auto itd = downFactors.find(domCcyKey);
353  if (itd != downFactors.end()) {
354  TLOG("Adding analytic " << domCcy << " down sensitivity to cube.");
355  sensiCube_->npvCube()->set(baseNpv - domCcySensi, tradeIdx, itd->second.index);
356  }
357  }
358  }
359  }
360 
361  DLOG("addAnalyticFxSensitivities: finished overwriting FX sensitivities with analytic values.");
362 }

364 Real getShiftSize(const RiskFactorKey& key, const SensitivityScenarioData& sensiParams,
365  const boost::shared_ptr<ScenarioSimMarket>& simMarket, const string& marketConfiguration) {
366 
367  Date asof = simMarket->asofDate();
368  RiskFactorKey::KeyType keytype = key.keytype;
369  string keylabel = key.name;
370  Real shiftSize = 0.0;
371  Real shiftMult = 1.0;
372  switch (keytype) {
373  case RiskFactorKey::KeyType::FXSpot: {
374  auto itr = sensiParams.fxShiftData().find(keylabel);
375  QL_REQUIRE(itr != sensiParams.fxShiftData().end(), "shiftData not found for " << keylabel);
376  shiftSize = itr->second.shiftSize;
377  if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
378  shiftMult = simMarket->fxRate(keylabel, marketConfiguration)->value();
379  }
380  } break;
381  case RiskFactorKey::KeyType::EquitySpot: {
382  auto itr = sensiParams.equityShiftData().find(keylabel);
383  QL_REQUIRE(itr != sensiParams.equityShiftData().end(), "shiftData not found for " << keylabel);
384  shiftSize = itr->second.shiftSize;
385  if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
386  shiftMult = simMarket->equitySpot(keylabel, marketConfiguration)->value();
387  }
388  } break;
389  case RiskFactorKey::KeyType::DiscountCurve: {
390  string ccy = keylabel;
391  auto itr = sensiParams.discountCurveShiftData().find(ccy);
392  QL_REQUIRE(itr != sensiParams.discountCurveShiftData().end(), "shiftData not found for " << ccy);
393  shiftSize = itr->second->shiftSize;
394  if (parseShiftType(itr->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
395  Size keyIdx = key.index;
396  Period p = itr->second->shiftTenors[keyIdx];
397  Handle<YieldTermStructure> yts = simMarket->discountCurve(ccy, marketConfiguration);
398  Time t = yts->dayCounter().yearFraction(asof, asof + p);
399  Real zeroRate = yts->zeroRate(t, Continuous);
400  shiftMult = zeroRate;
401  }
402  } break;
403  case RiskFactorKey::KeyType::IndexCurve: {
404  string idx = keylabel;
405  auto itr = sensiParams.indexCurveShiftData().find(idx);
406  QL_REQUIRE(itr != sensiParams.indexCurveShiftData().end(), "shiftData not found for " << idx);
407  shiftSize = itr->second->shiftSize;
408  if (parseShiftType(itr->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
409  Size keyIdx = key.index;
410  Period p = itr->second->shiftTenors[keyIdx];
411  Handle<YieldTermStructure> yts = simMarket->iborIndex(idx, marketConfiguration)->forwardingTermStructure();
412  Time t = yts->dayCounter().yearFraction(asof, asof + p);
413  Real zeroRate = yts->zeroRate(t, Continuous);
414  shiftMult = zeroRate;
415  }
416  } break;
417  case RiskFactorKey::KeyType::YieldCurve: {
418  string yc = keylabel;
419  auto itr = sensiParams.yieldCurveShiftData().find(yc);
420  QL_REQUIRE(itr != sensiParams.yieldCurveShiftData().end(), "shiftData not found for " << yc);
421  shiftSize = itr->second->shiftSize;
422  if (parseShiftType(itr->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
423  Size keyIdx = key.index;
424  Period p = itr->second->shiftTenors[keyIdx];
425  Handle<YieldTermStructure> yts = simMarket->yieldCurve(yc, marketConfiguration);
426  Time t = yts->dayCounter().yearFraction(asof, asof + p);
427  Real zeroRate = yts->zeroRate(t, Continuous);
428  shiftMult = zeroRate;
429  }
430  } break;
431  case RiskFactorKey::KeyType::DividendYield: {
432  string eq = keylabel;
433  auto itr = sensiParams.dividendYieldShiftData().find(eq);
434  QL_REQUIRE(itr != sensiParams.dividendYieldShiftData().end(), "shiftData not found for " << eq);
435  shiftSize = itr->second->shiftSize;
436 
437  if (parseShiftType(itr->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
438  Size keyIdx = key.index;
439  Period p = itr->second->shiftTenors[keyIdx];
440  Handle<YieldTermStructure> ts = simMarket->equityDividendCurve(eq, marketConfiguration);
441  Time t = ts->dayCounter().yearFraction(asof, asof + p);
442  Real zeroRate = ts->zeroRate(t, Continuous);
443  shiftMult = zeroRate;
444  }
445  } break;
446  case RiskFactorKey::KeyType::FXVolatility: {
447  string pair = keylabel;
448  auto itr = sensiParams.fxVolShiftData().find(pair);
449  QL_REQUIRE(itr != sensiParams.fxVolShiftData().end(), "shiftData not found for " << pair);
450  shiftSize = itr->second.shiftSize;
451  if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
452  vector<Real> strikes = sensiParams.fxVolShiftData().at(pair).shiftStrikes;
453  QL_REQUIRE(strikes.size() == 1 && close_enough(strikes[0], 0), "Only ATM FX vols supported");
454  Real atmFwd = 0.0; // hardcoded, since only ATM supported
455  Size keyIdx = key.index;
456  Period p = itr->second.shiftExpiries[keyIdx];
457  Handle<BlackVolTermStructure> vts = simMarket->fxVol(pair, marketConfiguration);
458  Time t = vts->dayCounter().yearFraction(asof, asof + p);
459  Real atmVol = vts->blackVol(t, atmFwd);
460  shiftMult = atmVol;
461  }
462  } break;
463  case RiskFactorKey::KeyType::EquityVolatility: {
464  string pair = keylabel;
465  auto itr = sensiParams.equityVolShiftData().find(pair);
466  QL_REQUIRE(itr != sensiParams.equityVolShiftData().end(), "shiftData not found for " << pair);
467  shiftSize = itr->second.shiftSize;
468  if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
469  Size keyIdx = key.index;
470  Period p = itr->second.shiftExpiries[keyIdx];
471  Handle<BlackVolTermStructure> vts = simMarket->equityVol(pair, marketConfiguration);
472  Time t = vts->dayCounter().yearFraction(asof, asof + p);
473  Real atmVol = vts->blackVol(t, Null<Real>());
474  shiftMult = atmVol;
475  }
476  } break;
477  case RiskFactorKey::KeyType::SwaptionVolatility: {
478  auto itr = sensiParams.swaptionVolShiftData().find(keylabel);
479  QL_REQUIRE(itr != sensiParams.swaptionVolShiftData().end(), "shiftData not found for " << keylabel);
480  shiftSize = itr->second.shiftSize;
481  if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
482  vector<Real> strikes = itr->second.shiftStrikes;
483  vector<Period> tenors = itr->second.shiftTerms;
484  vector<Period> expiries = itr->second.shiftExpiries;
485  Size keyIdx = key.index;
486  Size expIdx = keyIdx / (tenors.size() * strikes.size());
487  Period p_exp = expiries[expIdx];
488  Size tenIdx = keyIdx % tenors.size();
489  Period p_ten = tenors[tenIdx];
490  Real strike = Null<Real>(); // for cubes this will be ATM
491  Handle<SwaptionVolatilityStructure> vts = simMarket->swaptionVol(keylabel, marketConfiguration);
492  Real vol = vts->volatility(p_exp, p_ten, strike);
493  shiftMult = vol;
494  }
495  } break;
496  case RiskFactorKey::KeyType::YieldVolatility: {
497  string securityId = keylabel;
498  auto itr = sensiParams.yieldVolShiftData().find(securityId);
499  QL_REQUIRE(itr != sensiParams.yieldVolShiftData().end(), "shiftData not found for " << securityId);
500  shiftSize = itr->second.shiftSize;
501  if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
502  vector<Real> strikes = itr->second.shiftStrikes;
503  QL_REQUIRE(strikes.size() == 1 && close_enough(strikes[0], 0.0),
504  "shift strikes should be {0.0} for yield volatilities");
505  vector<Period> tenors = itr->second.shiftTerms;
506  vector<Period> expiries = itr->second.shiftExpiries;
507  Size keyIdx = key.index;
508  Size expIdx = keyIdx / (tenors.size() * strikes.size());
509  Period p_exp = expiries[expIdx];
510  Size tenIdx = keyIdx % tenors.size();
511  Period p_ten = tenors[tenIdx];
512  Real strike = Null<Real>(); // for cubes this will be ATM
513  Handle<SwaptionVolatilityStructure> vts = simMarket->yieldVol(securityId, marketConfiguration);
514  Real vol = vts->volatility(p_exp, p_ten, strike);
515  shiftMult = vol;
516  }
517  } break;
518  case RiskFactorKey::KeyType::OptionletVolatility: {
519  string ccy = keylabel;
520  auto itr = sensiParams.capFloorVolShiftData().find(ccy);
521  QL_REQUIRE(itr != sensiParams.capFloorVolShiftData().end(), "shiftData not found for " << ccy);
522  shiftSize = itr->second->shiftSize;
523  if (parseShiftType(itr->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
524  vector<Real> strikes = itr->second->shiftStrikes;
525  vector<Period> expiries = itr->second->shiftExpiries;
526  QL_REQUIRE(strikes.size() > 0, "Only strike capfloor vols supported");
527  shiftMult = simMarket->baseScenario()->get(
528  RiskFactorKey(RiskFactorKey::KeyType::OptionletVolatility, ccy, key.index));
529  }
530  } break;
531  case RiskFactorKey::KeyType::CDSVolatility: {
532  string name = keylabel;
533  auto itr = sensiParams.cdsVolShiftData().find(name);
534  QL_REQUIRE(itr != sensiParams.cdsVolShiftData().end(), "shiftData not found for " << name);
535  shiftSize = itr->second.shiftSize;
536  if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
537  vector<Period> expiries = itr->second.shiftExpiries;
538  Size keyIdx = key.index;
539  Size expIdx = keyIdx;
540  Period p_exp = expiries[expIdx];
541  Handle<CreditVolCurve> vts = simMarket->cdsVol(name, marketConfiguration);
542  // hardcoded 5y term
543  Real atmVol = vts->volatility(asof + p_exp, 5.0, Null<Real>(), vts->type());
544  shiftMult = atmVol;
545  }
546  } break;
547  case RiskFactorKey::KeyType::SurvivalProbability: {
548  string name = keylabel;
549  auto itr = sensiParams.creditCurveShiftData().find(name);
550  QL_REQUIRE(itr != sensiParams.creditCurveShiftData().end(), "shiftData not found for " << name);
551  shiftSize = itr->second->shiftSize;
552  if (parseShiftType(itr->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
553  Size keyIdx = key.index;
554  Period p = itr->second->shiftTenors[keyIdx];
555  Handle<DefaultProbabilityTermStructure> ts = simMarket->defaultCurve(name, marketConfiguration)->curve();
556  Time t = ts->dayCounter().yearFraction(asof, asof + p);
557  Real prob = ts->survivalProbability(t);
558  shiftMult = -std::log(prob) / t;
559  }
560  } break;
561  case RiskFactorKey::KeyType::BaseCorrelation: {
562  string name = keylabel;
563  auto itr = sensiParams.baseCorrelationShiftData().find(name);
564  QL_REQUIRE(itr != sensiParams.baseCorrelationShiftData().end(), "shiftData not found for " << name);
565  shiftSize = itr->second.shiftSize;
566  if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
567  vector<Real> lossLevels = itr->second.shiftLossLevels;
568  vector<Period> terms = itr->second.shiftTerms;
569  Size keyIdx = key.index;
570  Size lossLevelIdx = keyIdx / terms.size();
571  Real lossLevel = lossLevels[lossLevelIdx];
572  Size termIdx = keyIdx % terms.size();
573  Period term = terms[termIdx];
574  Handle<QuantExt::BaseCorrelationTermStructure> ts = simMarket->baseCorrelation(name, marketConfiguration);
575  Real bc = ts->correlation(asof + term, lossLevel, true); // extrapolate
576  shiftMult = bc;
577  }
578  } break;
579  case RiskFactorKey::KeyType::ZeroInflationCurve: {
580  string idx = keylabel;
581  auto itr = sensiParams.zeroInflationCurveShiftData().find(idx);
582  QL_REQUIRE(itr != sensiParams.zeroInflationCurveShiftData().end(), "shiftData not found for " << idx);
583  shiftSize = itr->second->shiftSize;
584  if (parseShiftType(itr->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
585  Size keyIdx = key.index;
586  Period p = itr->second->shiftTenors[keyIdx];
587  Handle<ZeroInflationTermStructure> yts =
588  simMarket->zeroInflationIndex(idx, marketConfiguration)->zeroInflationTermStructure();
589  Time t = yts->dayCounter().yearFraction(asof, asof + p);
590  Real zeroRate = yts->zeroRate(t);
591  shiftMult = zeroRate;
592  }
593  } break;
594  case RiskFactorKey::KeyType::YoYInflationCurve: {
595  string idx = keylabel;
596  auto itr = sensiParams.yoyInflationCurveShiftData().find(idx);
597  QL_REQUIRE(itr != sensiParams.yoyInflationCurveShiftData().end(), "shiftData not found for " << idx);
598  shiftSize = itr->second->shiftSize;
599  if (parseShiftType(itr->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
600  Size keyIdx = key.index;
601  Period p = sensiParams.yoyInflationCurveShiftData().at(idx)->shiftTenors[keyIdx];
602  Handle<YoYInflationTermStructure> yts =
603  simMarket->yoyInflationIndex(idx, marketConfiguration)->yoyInflationTermStructure();
604  Time t = yts->dayCounter().yearFraction(asof, asof + p);
605  Real yoyRate = yts->yoyRate(t);
606  shiftMult = yoyRate;
607  }
608  } break;
609  case RiskFactorKey::KeyType::YoYInflationCapFloorVolatility: {
610  string name = keylabel;
611  auto itr = sensiParams.yoyInflationCapFloorVolShiftData().find(name);
612  QL_REQUIRE(itr != sensiParams.yoyInflationCapFloorVolShiftData().end(), "shiftData not found for " << name);
613  shiftSize = itr->second->shiftSize;
614  if (parseShiftType(itr->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
615  vector<Real> strikes = itr->second->shiftStrikes;
616  vector<Period> expiries = itr->second->shiftExpiries;
617  QL_REQUIRE(strikes.size() > 0, "Only strike yoy inflation capfloor vols supported");
618  Size keyIdx = key.index;
619  Size expIdx = keyIdx / strikes.size();
620  Period p_exp = expiries[expIdx];
621  Size strIdx = keyIdx % strikes.size();
622  Real strike = strikes[strIdx];
623  Handle<QuantExt::YoYOptionletVolatilitySurface> vts = simMarket->yoyCapFloorVol(name, marketConfiguration);
624  Real vol = vts->volatility(p_exp, strike, vts->observationLag());
625  shiftMult = vol;
626  }
627  } break;
628  case RiskFactorKey::KeyType::ZeroInflationCapFloorVolatility: {
629  string name = keylabel;
630  auto itr = sensiParams.zeroInflationCapFloorVolShiftData().find(name);
631  QL_REQUIRE(itr != sensiParams.zeroInflationCapFloorVolShiftData().end(), "shiftData not found for " << name);
632  shiftSize = itr->second->shiftSize;
633  if (parseShiftType(itr->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
634  vector<Real> strikes = itr->second->shiftStrikes;
635  vector<Period> expiries = itr->second->shiftExpiries;
636  QL_REQUIRE(strikes.size() > 0, "Only strike zc inflation capfloor vols supported");
637  Size keyIdx = key.index;
638  Size expIdx = keyIdx / strikes.size();
639  Period p_exp = expiries[expIdx];
640  Size strIdx = keyIdx % strikes.size();
641  Real strike = strikes[strIdx];
642  Handle<CPIVolatilitySurface> vts =
643  simMarket->cpiInflationCapFloorVolatilitySurface(name, marketConfiguration);
644  Real vol = vts->volatility(p_exp, strike, vts->observationLag());
645  shiftMult = vol;
646  }
647  } break;
648  case RiskFactorKey::KeyType::CommodityCurve: {
649  auto it = sensiParams.commodityCurveShiftData().find(keylabel);
650  QL_REQUIRE(it != sensiParams.commodityCurveShiftData().end(), "shiftData not found for " << keylabel);
651  shiftSize = it->second->shiftSize;
652  if (parseShiftType(it->second->shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
653  Period p = it->second->shiftTenors[key.index];
654  Handle<PriceTermStructure> priceCurve = simMarket->commodityPriceCurve(keylabel, marketConfiguration);
655  Time t = priceCurve->dayCounter().yearFraction(asof, asof + p);
656  shiftMult = priceCurve->price(t);
657  }
658  } break;
659  case RiskFactorKey::KeyType::CommodityVolatility: {
660  auto it = sensiParams.commodityVolShiftData().find(keylabel);
661  QL_REQUIRE(it != sensiParams.commodityVolShiftData().end(), "shiftData not found for " << keylabel);
662 
663  shiftSize = it->second.shiftSize;
664  if (parseShiftType(it->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
665  Size moneynessIndex = key.index / it->second.shiftExpiries.size();
666  Size expiryIndex = key.index % it->second.shiftExpiries.size();
667  Real moneyness = it->second.shiftStrikes[moneynessIndex];
668  Period expiry = it->second.shiftExpiries[expiryIndex];
669  Real spotValue = simMarket->commodityPriceCurve(keylabel, marketConfiguration)->price(0);
670  Handle<BlackVolTermStructure> vts = simMarket->commodityVolatility(keylabel, marketConfiguration);
671  Time t = vts->dayCounter().yearFraction(asof, asof + expiry);
672  Real vol = vts->blackVol(t, moneyness * spotValue);
673  shiftMult = vol;
674  }
675  } break;
676  case RiskFactorKey::KeyType::SecuritySpread: {
677  auto itr = sensiParams.securityShiftData().find(keylabel);
678  QL_REQUIRE(itr != sensiParams.securityShiftData().end(), "shiftData not found for " << keylabel);
679  shiftSize = itr->second.shiftSize;
680  if (parseShiftType(itr->second.shiftType) == SensitivityScenarioGenerator::ShiftType::Relative) {
681  shiftMult = 1.0;
682  try {
683  shiftMult = simMarket->securitySpread(keylabel, marketConfiguration)->value();
684  } catch (...) {
685  // if there is no spread given for a security, we return 1.0, this is not relevant anyway,
686  // because there will be no scenario generated for this risk factor
687  }
688  }
689  } break;
690  default:
691  // KeyType::CPIIndex does not get shifted
692  QL_FAIL("KeyType not supported yet - " << keytype);
693  }
694  Real realShift = shiftSize * shiftMult;
695  return realShift;
696 }
