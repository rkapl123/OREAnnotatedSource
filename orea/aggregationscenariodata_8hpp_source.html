48 //! Container for storing simulated market data
49 /*! The indexes for dates and samples are (by convention) the
50  same as in the npv cube
51 
52  \ingroup scenario
53 */
54 class AggregationScenarioData {
55 public:
56  //! default ctor
57  AggregationScenarioData() : dIndex_(0), sIndex_(0) {}
58  //! Do not allow copying
59  AggregationScenarioData(AggregationScenarioData&) = delete;
60  AggregationScenarioData& operator=(AggregationScenarioData const&) = delete;
61  //! dtor
62  virtual ~AggregationScenarioData() {}
63 
64  //! Return the length of each dimension
65  virtual Size dimDates() const = 0;
66  virtual Size dimSamples() const = 0;
67 
68  //! Check whether data is available for the given type
69  virtual bool has(const AggregationScenarioDataType& type, const string& qualifier = "") const = 0;
70 
71  //! Get a value from the cube
72  virtual Real get(Size dateIndex, Size sampleIndex, const AggregationScenarioDataType& type,
73  const string& qualifier = "") const = 0;
74  //! Set a value in the cube
75  virtual void set(Size dateIndex, Size sampleIndex, Real value, const AggregationScenarioDataType& type,
76  const string& qualifier = "") = 0;
77 
78  // Get available keys (type, qualifier)
79  virtual std::vector<std::pair<AggregationScenarioDataType, std::string>> keys() const = 0;
80 
81  //! Load cube contents from disk
82  virtual void load(const std::string&) {}
83  //! Persist cube contents to disk
84  virtual void save(const std::string&) const {}
85 
86  //! Set a value in the cube, assumes normal traversal of the cube (dates then samples)
87  virtual void set(Real value, const AggregationScenarioDataType& type, const string& qualifier = "") {
88  set(dIndex_, sIndex_, value, type, qualifier);
89  }
90  //! Go to the next point on the cube
91  /*! Go to the next point on the cube, assumes we do date, then samples
92  */
93  virtual void next() {
94  if (++dIndex_ == dimDates()) {
95  dIndex_ = 0;
96  sIndex_++;
97  }
98  }
99 
100 private:
101  Size dIndex_, sIndex_;
102 };

104 //! A concrete in memory implementation of AggregationScenarioData
105 /*! \ingroup scenario
106  */
107 class InMemoryAggregationScenarioData : public AggregationScenarioData {
108 public:
109  InMemoryAggregationScenarioData() : AggregationScenarioData(), dimDates_(0), dimSamples_(0) {}
110  InMemoryAggregationScenarioData(Size dimDates, Size dimSamples)
111  : AggregationScenarioData(), dimDates_(dimDates), dimSamples_(dimSamples) {}
112  Size dimDates() const override { return dimDates_; }
113  Size dimSamples() const override { return dimSamples_; }
114 
115  bool has(const AggregationScenarioDataType& type, const string& qualifier = "") const override {
116  return data_.find(std::make_pair(type, qualifier)) != data_.end();
117  }
118 
119  /*! might throw (the underlying's container exception),
120  if type is not known */
121  Real get(Size dateIndex, Size sampleIndex, const AggregationScenarioDataType& type,
122  const string& qualifier = "") const override {
123  check(dateIndex, sampleIndex, type, qualifier);
124  return data_.at(std::make_pair(type, qualifier))[dateIndex][sampleIndex];
125  }
126 
127  std::vector<std::pair<AggregationScenarioDataType, std::string>> keys() const override {
128  std::vector<std::pair<AggregationScenarioDataType, std::string>> res;
129  for (auto const& k : data_)
130  res.push_back(k.first);
131  return res;
132  }
133 
134  void set(Size dateIndex, Size sampleIndex, Real value, const AggregationScenarioDataType& type,
135  const string& qualifier = "") override {
136  check(dateIndex, sampleIndex, type, qualifier);
137  auto key = std::make_pair(type, qualifier);
138  auto it = data_.find(key);
139  if (it == data_.end()) {
140  data_.insert(make_pair(key, vector<vector<Real>>(dimDates_, vector<Real>(dimSamples_, 0.0))));
141  }
142  data_[key][dateIndex][sampleIndex] = value;
143  }
144 
145  void load(const std::string& fileName) override {
146  std::ifstream ifs(fileName.c_str(), std::fstream::binary);
147  QL_REQUIRE(ifs.is_open(), "error opening file " << fileName);
148  boost::archive::binary_iarchive ia(ifs);
149  ia&* this;
150  }
151 
152  void save(const std::string& fileName) const override {
153  std::ofstream ofs(fileName.c_str(), std::fstream::binary);
154  QL_REQUIRE(ofs.is_open(), "error opening file " << fileName);
155  boost::archive::binary_oarchive oa(ofs);
156  oa&* this;
157  }
158 
159 private:
160  friend class boost::serialization::access;
161  template <class Archive> void serialize(Archive& ar, const unsigned int) {
162  ar& dimDates_;
163  ar& dimSamples_;
164  ar& data_;
165  }
166 
167  void check(Size dateIndex, Size sampleIndex, const AggregationScenarioDataType& type,
168  const string& qualifier) const {
169  QL_REQUIRE(dateIndex < dimDates_, "dateIndex (" << dateIndex << ") out of range 0..." << dimDates_ - 1);
170  QL_REQUIRE(sampleIndex < dimSamples_,
171  "sampleIndex (" << sampleIndex << ") out of range 0..." << dimSamples_ - 1);
172  return;
173  }
174  Size dimDates_, dimSamples_;
175  map<std::pair<AggregationScenarioDataType, string>, vector<vector<Real>>> data_;
176 };
