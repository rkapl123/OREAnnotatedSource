98 void RegressionDynamicInitialMarginCalculator::build() {
99  LOG("DIM Analysis by polynomial regression");
100 
101  map<string, Real> currentDim = unscaledCurrentDIM();
102 
103  Size stopDatesLoop = datesLoopSize_;
104  Size samples = cube_->samples();
105 
106  Size polynomOrder = regressionOrder_;
107  LOG("DIM regression polynom order = " << regressionOrder_);
108  LsmBasisSystem::PolynomType polynomType = LsmBasisSystem::Monomial;
109  Size regressionDimension = regressors_.empty() ? 1 : regressors_.size();
110  LOG("DIM regression dimension = " << regressionDimension);
111 #if QL_HEX_VERSION > 0x01150000
112  std::vector<ext::function<Real(Array)>> v(
113  LsmBasisSystem::multiPathBasisSystem(regressionDimension, polynomOrder, polynomType));
114 #else // QL 1.14 and below
115  std::vector<boost::function1<Real, Array>> v(
116  LsmBasisSystem::multiPathBasisSystem(regressionDimension, polynomOrder, polynomType));
117 #endif
118  Real confidenceLevel = QuantLib::InverseCumulativeNormal()(quantile_);
119  LOG("DIM confidence level " << confidenceLevel);
120 
121  Size simple_dim_index_h = Size(floor(quantile_ * (samples - 1) + 0.5));
122  Size simple_dim_index_p = Size(floor((1.0 - quantile_) * (samples - 1) + 0.5));
123 
124  Size nettingSetCount = 0;
125  for (auto n : nettingSetIds_) {
126  LOG("Process netting set " << n);
127 
128  if (currentIM_.find(n) != currentIM_.end()) {
129  Real t0im = currentIM_[n];
130  QL_REQUIRE(currentDim.find(n) != currentDim.end(), "current DIM not found for netting set " << n);
131  Real t0dim = currentDim[n];
132  Real t0scaling = t0im / t0dim;
133  LOG("t0 scaling for netting set " << n << ": t0im" << t0im << " t0dim=" << t0dim
134  << " t0scaling=" << t0scaling);
135  nettingSetScaling_[n] = t0scaling;
136  }
137 
138  Real nettingSetDimScaling =
139  nettingSetScaling_.find(n) == nettingSetScaling_.end() ? 1.0 : nettingSetScaling_[n];
140  LOG("Netting set DIM scaling factor: " << nettingSetDimScaling);
141 
142  for (Size j = 0; j < stopDatesLoop; ++j) {
143  accumulator_set<double, stats<tag::mean, tag::variance>> accDiff;
144  accumulator_set<double, stats<tag::mean>> accOneOverNumeraire;
145  for (Size k = 0; k < samples; ++k) {
146  Real numDefault = cubeInterpretation_->getDefaultAggrionScenarioData(
147  scenarioData_, AggregationScenarioDataType::Numeraire, j, k);
148  Real numCloseOut = cubeInterpretation_->getCloseOutAggrionScenarioData(
149  scenarioData_, AggregationScenarioDataType::Numeraire, j, k);
150  Real npvDefault = nettingSetNPV_[n][j][k];
151  Real flow = nettingSetFLOW_[n][j][k];
152  Real npvCloseOut = nettingSetCloseOutNPV_[n][j][k];
153  accDiff((npvCloseOut * numCloseOut) + (flow * numDefault) - (npvDefault * numDefault));
154  accOneOverNumeraire(1.0 / numDefault);
155  }
156 
157  Size mporCalendarDays = cubeInterpretation_->getMporCalendarDays(cube_, j);
158  Real horizonScaling = sqrt(1.0 * horizonCalendarDays_ / mporCalendarDays);
159 
160  Real stdevDiff = sqrt(variance(accDiff));
161  Real E_OneOverNumeraire =
162  mean(accOneOverNumeraire); // "re-discount" (the stdev is calculated on non-discounted deltaNPVs)
163 
164  nettingSetZeroOrderDIM_[n][j] = stdevDiff * horizonScaling * confidenceLevel;
165  nettingSetZeroOrderDIM_[n][j] *= E_OneOverNumeraire;
166 
167  vector<Real> rx0(samples, 0.0);
168  vector<Array> rx(samples, Array());
169  vector<Real> ry1(samples, 0.0);
170  vector<Real> ry2(samples, 0.0);
171  for (Size k = 0; k < samples; ++k) {
172  Real numDefault = cubeInterpretation_->getDefaultAggrionScenarioData(
173  scenarioData_, AggregationScenarioDataType::Numeraire, j, k);
174  Real numCloseOut = cubeInterpretation_->getCloseOutAggrionScenarioData(
175  scenarioData_, AggregationScenarioDataType::Numeraire, j, k);
176  Real x = nettingSetNPV_[n][j][k] * numDefault;
177  Real f = nettingSetFLOW_[n][j][k] * numDefault;
178  Real y = nettingSetCloseOutNPV_[n][j][k] * numCloseOut;
179  Real z = (y + f - x);
180  rx[k] = regressors_.empty() ? Array(1, nettingSetNPV_[n][j][k]) : regressorArray(n, j, k);
181  rx0[k] = rx[k][0];
182  ry1[k] = z; // for local regression
183  ry2[k] = z * z; // for least squares regression
184  nettingSetDeltaNPV_[n][j][k] = z;
185  regressorArray_[n][j][k] = rx[k];
186  }
187  vector<Real> delNpvVec_copy = nettingSetDeltaNPV_[n][j];
188  sort(delNpvVec_copy.begin(), delNpvVec_copy.end());
189  Real simpleDim_h = delNpvVec_copy[simple_dim_index_h];
190  Real simpleDim_p = delNpvVec_copy[simple_dim_index_p];
191  simpleDim_h *= horizonScaling; // the usual scaling factors
192  simpleDim_p *= horizonScaling; // the usual scaling factors
193  nettingSetSimpleDIMh_[n][j] = simpleDim_h * E_OneOverNumeraire; // discounted DIM
194  nettingSetSimpleDIMp_[n][j] = simpleDim_p * E_OneOverNumeraire; // discounted DIM
195 
196  QL_REQUIRE(rx.size() > v.size(), "not enough points for regression with polynom order " << polynomOrder);
197  if (close_enough(stdevDiff, 0.0)) {
198  LOG("DIM: Zero std dev estimation at step " << j);
199  // Skip IM calculation if all samples have zero NPV (e.g. after latest maturity)
200  for (Size k = 0; k < samples; ++k) {
201  nettingSetDIM_[n][j][k] = 0.0;
202  nettingSetLocalDIM_[n][j][k] = 0.0;
203  }
204  } else {
205  // Least squares polynomial regression with specified polynom order
206  QuantExt::StabilisedGLLS ls(rx, ry2, v, QuantExt::StabilisedGLLS::MeanStdDev);
207  LOG("DIM data normalisation at time step "
208  << j << ": " << scientific << setprecision(6) << " x-shift = " << ls.xShift() << " x-multiplier = "
209  << ls.xMultiplier() << " y-shift = " << ls.yShift() << " y-multiplier = " << ls.yMultiplier());
210  LOG("DIM regression coefficients at time step " << j << ": " << fixed << setprecision(6)
211  << ls.transformedCoefficients());
212 
213  // Local regression versus first regression variable (i.e. we do not perform a
214  // multidimensional local regression):
215  // We evaluate this at a limited number of samples only for validation purposes.
216  // Note that computational effort scales quadratically with number of samples.
217  // NadarayaWatson needs a large number of samples for good results.
218  QuantExt::NadarayaWatson lr(rx0.begin(), rx0.end(), ry1.begin(),
219  GaussianKernel(0.0, localRegressionBandWidth_));
220  Size localRegressionSamples = samples;
221  if (localRegressionEvaluations_ > 0)
222  localRegressionSamples = Size(floor(1.0 * samples / localRegressionEvaluations_ + .5));
223 
224  // Evaluate regression function to compute DIM for each scenario
225  for (Size k = 0; k < samples; ++k) {
226  // Real num1 = scenarioData_->get(j, k, AggregationScenarioDataType::Numeraire);
227  Real numDefault = cubeInterpretation_->getDefaultAggrionScenarioData(
228  scenarioData_, AggregationScenarioDataType::Numeraire, j, k);
229  Array regressor = regressors_.empty() ? Array(1, nettingSetNPV_[n][j][k]) : regressorArray(n, j, k);
230  Real e = ls.eval(regressor, v);
231  if (e < 0.0)
232  LOG("Negative variance regression for date " << j << ", sample " << k
233  << ", regressor = " << regressor);
234 
235  // Note:
236  // 1) We assume vanishing mean of "z", because the drift over a MPOR is usually small,
237  // and to avoid a second regression for the conditional mean
238  // 2) In particular the linear regression function can yield negative variance values in
239  // extreme scenarios where an exact analytical or delta VaR calculation would yield a
240  // variance aproaching zero. We correct this here by taking the positive part.
241  Real std = sqrt(std::max(e, 0.0));
242  Real scalingFactor = horizonScaling * confidenceLevel * nettingSetDimScaling;
243  // Real dim = std * scalingFactor / num1;
244  Real dim = std * scalingFactor / numDefault;
245  dimCube_->set(dim, nettingSetCount, j, k);
246  nettingSetDIM_[n][j][k] = dim;
247  nettingSetExpectedDIM_[n][j] += dim / samples;
248 
249  // Evaluate the Kernel regression for a subset of the samples only (performance)
250  if (localRegressionEvaluations_ > 0 && (k % localRegressionSamples == 0))
251  // nettingSetLocalDIM_[n][j][k] = lr.standardDeviation(regressor[0]) * scalingFactor / num1;
252  nettingSetLocalDIM_[n][j][k] = lr.standardDeviation(regressor[0]) * scalingFactor / numDefault;
253  else
254  nettingSetLocalDIM_[n][j][k] = 0.0;
255  }
256  }
257  }
258 
259  nettingSetCount++;
260  }
261  LOG("DIM by polynomial regression done");
262 }

264 Disposable<Array> RegressionDynamicInitialMarginCalculator::regressorArray(string nettingSet, Size dateIndex,
265  Size sampleIndex) {
266  Array a(regressors_.size());
267  for (Size i = 0; i < regressors_.size(); ++i) {
268  string variable = regressors_[i];
269  if (boost::to_upper_copy(variable) ==
270  "NPV") // this allows possibility to include NPV as a regressor alongside more fundamental risk factors
271  a[i] = nettingSetNPV_[nettingSet][dateIndex][sampleIndex];
272  else if (scenarioData_->has(AggregationScenarioDataType::IndexFixing, variable))
273  a[i] = cubeInterpretation_->getDefaultAggrionScenarioData(
274  scenarioData_, AggregationScenarioDataType::IndexFixing, dateIndex, sampleIndex, variable);
275  else if (scenarioData_->has(AggregationScenarioDataType::FXSpot, variable))
276  a[i] = cubeInterpretation_->getDefaultAggrionScenarioData(
277  scenarioData_, AggregationScenarioDataType::FXSpot, dateIndex, sampleIndex, variable);
278  else if (scenarioData_->has(AggregationScenarioDataType::Generic, variable))
279  a[i] = cubeInterpretation_->getDefaultAggrionScenarioData(
280  scenarioData_, AggregationScenarioDataType::Generic, dateIndex, sampleIndex, variable);
281  else
282  QL_FAIL("scenario data does not provide data for " << variable);
283  }
284  return a;
285 }

287 map<string, Real> RegressionDynamicInitialMarginCalculator::unscaledCurrentDIM() {
288  // In this function we proxy the model-implied T0 IM by looking at the
289  // cube grid horizon lying closest to t0+mpor. We measure diffs relative
290  // to the mean of the distribution at this same time horizon, thus avoiding
291  // any cashflow-specific jumps
292 
293  Date today = cube_->asof();
294  Size relevantDateIdx = 0;
295  Real sqrtTimeScaling = 1.0;
296  for (Size i = 0; i < cube_->dates().size(); ++i) {
297  Size daysFromT0 = (cube_->dates()[i] - today);
298  if (daysFromT0 < horizonCalendarDays_) {
299  // iterate until we straddle t0+mpor
300  continue;
301  } else if (daysFromT0 == horizonCalendarDays_) {
302  // this date corresponds to t0+mpor, so use it
303  relevantDateIdx = i;
304  sqrtTimeScaling = 1.0;
305  break;
306  } else if (daysFromT0 > horizonCalendarDays_) {
307  // the first date greater than t0+MPOR, check if it is closest
308  Size lastIdx = (i == 0) ? 0 : (i - 1);
309  Size lastDaysFromT0 = (cube_->dates()[lastIdx] - today);
310  int daysFromT0CloseOut = daysFromT0 - horizonCalendarDays_;
311  int prevDaysFromT0CloseOut = lastDaysFromT0 - horizonCalendarDays_;
312  if (std::abs(daysFromT0CloseOut) <= std::abs(prevDaysFromT0CloseOut)) {
313  relevantDateIdx = i;
314  sqrtTimeScaling = std::sqrt(Real(horizonCalendarDays_) / Real(daysFromT0));
315  } else {
316  relevantDateIdx = lastIdx;
317  sqrtTimeScaling = std::sqrt(Real(horizonCalendarDays_) / Real(lastDaysFromT0));
318  }
319  break;
320  }
321  }
322  // set some reasonable bounds on the sqrt time scaling, so that we are not looking at a ridiculous time horizon
323  if (sqrtTimeScaling < std::sqrt(0.5) || sqrtTimeScaling > std::sqrt(2.0)) {
324  WLOG("T0 IM Estimation - The estimation time horizon from grid is not sufficiently close to t0+MPOR - "
325  << QuantLib::io::iso_date(cube_->dates()[relevantDateIdx])
326  << ", the T0 IM estimate might be inaccurate. Consider inserting a first grid tenor closer to the dim "
327  "horizon");
328  }
329 
330  // TODO: Ensure that the simulation containers read-from below are indeed populated
331 
332  Real confidenceLevel = QuantLib::InverseCumulativeNormal()(quantile_);
333  Size simple_dim_index_h = Size(floor(quantile_ * (cube_->samples() - 1) + 0.5));
334  map<string, Real> t0dimReg, t0dimSimple;
335  for (auto it_map = nettingSetNPV_.begin(); it_map != nettingSetNPV_.end(); ++it_map) {
336  string key = it_map->first;
337  vector<Real> t0_dist = it_map->second[relevantDateIdx];
338  Size dist_size = t0_dist.size();
339  QL_REQUIRE(dist_size == cube_->samples(),
340  "T0 IM - cube samples size mismatch - " << dist_size << ", " << cube_->samples());
341  Real mean_t0_dist = std::accumulate(t0_dist.begin(), t0_dist.end(), 0.0);
342  mean_t0_dist /= dist_size;
343  vector<Real> t0_delMtM_dist(dist_size, 0.0);
344  accumulator_set<double, stats<tag::mean, tag::variance>> acc_delMtm;
345  accumulator_set<double, stats<tag::mean>> acc_OneOverNum;
346  for (Size i = 0; i < dist_size; ++i) {
347  Real numeraire = scenarioData_->get(relevantDateIdx, i, AggregationScenarioDataType::Numeraire);
348  Real deltaMtmFromMean = numeraire * (t0_dist[i] - mean_t0_dist) * sqrtTimeScaling;
349  t0_delMtM_dist[i] = deltaMtmFromMean;
350  acc_delMtm(deltaMtmFromMean);
351  acc_OneOverNum(1.0 / numeraire);
352  }
353  Real E_OneOverNumeraire = mean(acc_OneOverNum);
354  Real variance_t0 = variance(acc_delMtm);
355  Real sqrt_t0 = sqrt(variance_t0);
356  t0dimReg[key] = (sqrt_t0 * confidenceLevel * E_OneOverNumeraire);
357  std::sort(t0_delMtM_dist.begin(), t0_delMtM_dist.end());
358  t0dimSimple[key] = (t0_delMtM_dist[simple_dim_index_h] * E_OneOverNumeraire);
359 
360  LOG("T0 IM (Reg) - {" << key << "} = " << t0dimReg[key]);
361  LOG("T0 IM (Simple) - {" << key << "} = " << t0dimSimple[key]);
362  }
363  LOG("T0 IM Calculations Completed");
364 
365  return t0dimReg;
366 }

368 void RegressionDynamicInitialMarginCalculator::exportDimEvolution(ore::data::Report& dimEvolutionReport) {
369 
370  Size samples = dimCube_->samples();
371  Size stopDatesLoop = datesLoopSize_;
372  Date asof = cube_->asof();
373 
374  dimEvolutionReport.addColumn("TimeStep", Size())
375  .addColumn("Date", Date())
376  .addColumn("DaysInPeriod", Size())
377  .addColumn("ZeroOrderDIM", Real(), 6)
378  .addColumn("AverageDIM", Real(), 6)
379  .addColumn("AverageFLOW", Real(), 6)
380  .addColumn("SimpleDIM", Real(), 6)
381  .addColumn("NettingSet", string())
382  .addColumn("Time", Real(), 6);
383 
384  for (auto nettingSet : dimCube_->ids()) {
385 
386  LOG("Export DIM evolution for netting set " << nettingSet);
387  for (Size i = 0; i < stopDatesLoop; ++i) {
388  Real expectedFlow = 0.0;
389  for (Size j = 0; j < samples; ++j) {
390  expectedFlow += nettingSetFLOW_[nettingSet][i][j] / samples;
391  }
392 
393  Date defaultDate = dimCube_->dates()[i];
394  Time t = ActualActual().yearFraction(asof, defaultDate);
395  Size days = cubeInterpretation_->getMporCalendarDays(dimCube_, i);
396  dimEvolutionReport.next()
397  .add(i)
398  .add(defaultDate)
399  .add(days)
400  .add(nettingSetZeroOrderDIM_[nettingSet][i])
401  .add(nettingSetExpectedDIM_[nettingSet][i])
402  .add(expectedFlow)
403  .add(nettingSetSimpleDIMh_[nettingSet][i])
404  .add(nettingSet)
405  .add(t);
406  }
407  }
408  dimEvolutionReport.end();
409  LOG("Exporting expected DIM through time done");
410 }

412 void RegressionDynamicInitialMarginCalculator::exportDimRegression(
413  const std::string& nettingSet, const std::vector<Size>& timeSteps,
414  const std::vector<boost::shared_ptr<ore::data::Report>>& dimRegReports) {
415 
416  QL_REQUIRE(dimRegReports.size() == timeSteps.size(),
417  "number of file names (" << dimRegReports.size() << ") does not match number of time steps ("
418  << timeSteps.size() << ")");
419  for (Size ii = 0; ii < timeSteps.size(); ++ii) {
420  Size timeStep = timeSteps[ii];
421  LOG("Export DIM by sample for netting set " << nettingSet << " and time step " << timeStep);
422 
423  Size dates = dimCube_->dates().size();
424  const std::vector<std::string>& ids = dimCube_->ids();
425 
426  int index = -1;
427  for (Size i = 0; i < ids.size(); ++i) {
428  if (ids[i] == nettingSet) {
429  index = i;
430  break;
431  }
432  }
433  QL_REQUIRE(index >= 0, "netting set " << nettingSet << " not found in DIM cube");
434 
435  QL_REQUIRE(timeStep < dates - 1, "selected time step " << timeStep << " out of range [0, " << dates - 1 << "]");
436 
437  Size samples = cube_->samples();
438  vector<Real> numeraires(samples, 0.0);
439  for (Size k = 0; k < samples; ++k)
440  // numeraires[k] = scenarioData_->get(timeStep, k, AggregationScenarioDataType::Numeraire);
441  numeraires[k] = cubeInterpretation_->getDefaultAggrionScenarioData(
442  scenarioData_, AggregationScenarioDataType::Numeraire, timeStep, k);
443 
444  auto p = sort_permutation(regressorArray_[nettingSet][timeStep], lessThan);
445  vector<Array> reg = apply_permutation(regressorArray_[nettingSet][timeStep], p);
446  vector<Real> dim = apply_permutation(nettingSetDIM_[nettingSet][timeStep], p);
447  vector<Real> ldim = apply_permutation(nettingSetLocalDIM_[nettingSet][timeStep], p);
448  vector<Real> delta = apply_permutation(nettingSetDeltaNPV_[nettingSet][timeStep], p);
449  vector<Real> num = apply_permutation(numeraires, p);
450 
451  boost::shared_ptr<ore::data::Report> regReport = dimRegReports[ii];
452  regReport->addColumn("Sample", Size());
453  for (Size k = 0; k < reg[0].size(); ++k) {
454  ostringstream o;
455  o << "Regressor_" << k << "_";
456  o << (regressors_.empty() ? "NPV" : regressors_[k]);
457  regReport->addColumn(o.str(), Real(), 6);
458  }
459  regReport->addColumn("RegressionDIM", Real(), 6)
460  .addColumn("LocalDIM", Real(), 6)
461  .addColumn("ExpectedDIM", Real(), 6)
462  .addColumn("ZeroOrderDIM", Real(), 6)
463  .addColumn("DeltaNPV", Real(), 6)
464  .addColumn("SimpleDIM", Real(), 6);
465 
466  // Note that RegressionDIM, LocalDIM, DeltaNPV are _not_ reduced by the numeraire in this output,
467  // but ExpectedDIM, ZeroOrderDIM and SimpleDIM _are_ reduced by the numeraire.
468  // This is so that the regression formula can be manually validated
469 
470  for (Size j = 0; j < reg.size(); ++j) {
471  regReport->next().add(j);

473  regReport->add(reg[j][k]);
