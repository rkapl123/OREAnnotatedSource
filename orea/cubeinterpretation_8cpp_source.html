26CubeInterpretation::CubeInterpretation(const bool storeFlows, const bool withCloseOutLag,
27 const QuantLib::Handle<AggregationScenarioData>& aggregationScenarioData,
28 const boost::shared_ptr<DateGrid>& dateGrid, const Size storeCreditStateNPVs,
29 const bool flipViewXVA)
30 : storeFlows_(storeFlows), withCloseOutLag_(withCloseOutLag), aggregationScenarioData_(aggregationScenarioData),
31 dateGrid_(dateGrid), storeCreditStateNPVs_(storeCreditStateNPVs), flipViewXVA_(flipViewXVA) {
32
33 // determine required cube depth and layout
34
35 requiredCubeDepth_ = 1;
36 defaultDateNpvIndex_ = 0;
37
38 if (withCloseOutLag_) {
39 closeOutDateNpvIndex_ = requiredCubeDepth_++;
40 QL_REQUIRE(dateGrid_ != nullptr, "CubeInterpretation: dateGrid is required when withCloseOutLag is true");
41 }
42
43 if (storeFlows_) {
44 mporFlowsIndex_ = requiredCubeDepth_;
45 requiredCubeDepth_ += 2;
46 }
47
48 if (storeCreditStateNPVs_ > 0) {
49 creditStateNPVsIndex_ = requiredCubeDepth_;
50 requiredCubeDepth_ += storeCreditStateNPVs_;
51 }
52}

75Real CubeInterpretation::getGenericValue(const boost::shared_ptr<NPVCube>& cube, Size tradeIdx, Size dateIdx,
76 Size sampleIdx, Size depth) const {
77 if (flipViewXVA_) {
78 return -cube->get(tradeIdx, dateIdx, sampleIdx, depth);
79 } else {
80 return cube->get(tradeIdx, dateIdx, sampleIdx, depth);
81 }

89Real CubeInterpretation::getCloseOutNpv(const boost::shared_ptr<NPVCube>& cube, Size tradeIdx, Size dateIdx,
90 Size sampleIdx) const {
91 if (withCloseOutLag_)
92 return getGenericValue(cube, tradeIdx, dateIdx, sampleIdx, closeOutDateNpvIndex_) /
93 getCloseOutAggregationScenarioData(AggregationScenarioDataType::Numeraire, dateIdx, sampleIdx);
94 else
95 return getGenericValue(cube, tradeIdx, dateIdx + 1, sampleIdx, defaultDateNpvIndex_);
96}

98Real CubeInterpretation::getMporPositiveFlows(const boost::shared_ptr<NPVCube>& cube, Size tradeIdx, Size dateIdx,
99 Size sampleIdx) const {
100 Real aggMporFlowsVal = 0.0;
101 try {
102 aggMporFlowsVal = getGenericValue(cube, tradeIdx, dateIdx, sampleIdx, mporFlowsIndex_);
103 } catch (std::exception& e) {
104 DLOG("Unable to retrieve MPOR flows for trade " << tradeIdx << ", date " << dateIdx << ", sample " << sampleIdx
105 << "; " << e.what());
106 }
107 return aggMporFlowsVal;
108}

134Real CubeInterpretation::getCloseOutAggregationScenarioData(const AggregationScenarioDataType& dataType, Size dateIdx,
135 Size sampleIdx, const std::string& qualifier) const {
136 if (withCloseOutLag_) {
137 QL_REQUIRE(dataType == AggregationScenarioDataType::Numeraire,
138 "close out aggr scen data only available for numeraire");
139 // this is an approximation
140 return getDefaultAggregationScenarioData(dataType, dateIdx, sampleIdx, qualifier);
141 } else {
142 QL_REQUIRE(!aggregationScenarioData_.empty(), "CubeInterpretation::getCloseOutAggregationScenarioData(): no "
143 "aggregation scenario data given");
144 return aggregationScenarioData_->get(dateIdx + 1, sampleIdx, dataType, qualifier);
145 }
146}

148Size CubeInterpretation::getMporCalendarDays(const boost::shared_ptr<NPVCube>& cube, Size dateIdx) const {
149 if (withCloseOutLag_) {
150 QuantLib::Date dd = dateGrid_->valuationDates()[dateIdx];
151 QuantLib::Date cd = dateGrid_->closeOutDates()[dateIdx];
152 QL_REQUIRE(cd > dd, "close-out date (" << cd << ") must be greater than default date (" << dd << ") at index "
153 << dateIdx);
154 return (cd - dd);
155 } else {
156 return cube->dates()[dateIdx + 1] - cube->dates()[dateIdx];
157 }
158}
