130 Size MporGridCubeInterpretation::getMporCalendarDays(const boost::shared_ptr<NPVCube>& cube, Size dateIdx) const {
131  QuantLib::Date dd = dateGrid_->valuationDates()[dateIdx];
132  QuantLib::Date cd = dateGrid_->closeOutDates()[dateIdx];
133  QL_REQUIRE(cd > dd,
134  "close-out date (" << cd << ") must be greater than default date (" << dd << ") at index " << dateIdx);
135  return (cd - dd);
136 }
