53void XvaAnalyticImpl::checkConfigurations(const boost::shared_ptr<Portfolio>& portfolio) {
54 //find the unique nettingset keys in portfolio
55 std::map<std::string, std::string> nettingSetMap = portfolio->nettingSetMap();
56 std::vector<std::string> nettingSetKeys;
57 for(std::map<std::string, std::string>::iterator it = nettingSetMap.begin(); it != nettingSetMap.end(); ++it)
58 nettingSetKeys.push_back(it->second);
59 //unique nettingset keys
60 sort(nettingSetKeys.begin(), nettingSetKeys.end());
61 nettingSetKeys.erase(unique( nettingSetKeys.begin(), nettingSetKeys.end() ), nettingSetKeys.end());
62 //controls on calcType and grid type, if netting-set has an active CSA in place
63 for(auto const& key : nettingSetKeys){
64 LOG("For netting-set "<<key<<"CSA flag is "<<inputs_->nettingSetManager()->get(key)->activeCsaFlag());
65 if (inputs_->nettingSetManager()->get(key)->activeCsaFlag()){
66 string calculationType = inputs_->collateralCalculationType();
67 if (analytic()->configurations().scenarioGeneratorData->withCloseOutLag()){
68 QL_REQUIRE(calculationType == "NoLag", "For nettingSetID "<<key<< ", CSA is active and a close-out grid is configured in the simulation.xml. Therefore, calculation type "<<calculationType<<" is not admissable. It must be set to NoLag!");
69 LOG("For netting-set "<<key<<", calculation type is "<<calculationType);
70 }
71 else{
72 QL_REQUIRE(calculationType != "NoLag", "For nettingSetID "<<key<< ", CSA is active and a close-out grid is not configured in the simulation.xml. Therefore, calculation type " <<calculationType<<" is not admissable. It must be set to either Symmetric or AsymmerticCVA or AsymmetricDVA!" );
73 LOG("For netting-set "<<key<<", calculation type is "<<calculationType);
74 }
75 if (analytic()->configurations().scenarioGeneratorData->withCloseOutLag() && analytic()->configurations().scenarioGeneratorData->closeOutLag() != 0*Days){
76 Period mpor_simulation = analytic()->configurations().scenarioGeneratorData->closeOutLag();
77 Period mpor_netting = inputs_->nettingSetManager()->get(key)->csaDetails()->marginPeriodOfRisk();
78 if (mpor_simulation != mpor_netting)
79 WLOG(StructuredAnalyticsWarningMessage(
80 "XvaAnalytic", "Inconsistent MPoR period",
81 "For netting set " + key +", close-out lag is not consistent with the netting-set's mpor "));
82 }
83 }
84 }

188void XvaAnalyticImpl::initClassicRun(const boost::shared_ptr<Portfolio>& portfolio) {
189
190 LOG("XVA: initClassicRun");
191
192 initCubeDepth();
193
194 // May have been set already
195 if (scenarioData_.empty()) {
196 LOG("XVA: Create asd " << grid_->valuationDates().size() << " x " << samples_);
197 scenarioData_.linkTo(
198 boost::make_shared<InMemoryAggregationScenarioData>(grid_->valuationDates().size(), samples_));
199 simMarket_->aggregationScenarioData() = *scenarioData_;
200
201 }
202
203 // We can skip the cube initialization if the mt val engine is used, since it builds its own cubes
204 if (inputs_->nThreads() == 1) {
205 if (portfolio->size() > 0)
206 initCube(cube_, portfolio->ids(), cubeDepth_);
207 // not required by any calculators in ore at the moment
208 nettingSetCube_ = nullptr;
209 // Init counterparty cube for the storage of survival probabilities
210 if (inputs_->storeSurvivalProbabilities()) {
211 // Use full list of counterparties, not just those in the sub-portflio
212 auto counterparties = inputs_->portfolio()->counterparties();
213 counterparties.insert(inputs_->dvaName());
214 initCube(cptyCube_, counterparties, 1);
215 } else {
216 cptyCube_ = nullptr;
217 }
218 }
219
220 LOG("XVA: initClassicRun completed");
221}

224boost::shared_ptr<Portfolio> XvaAnalyticImpl::classicRun(const boost::shared_ptr<Portfolio>& portfolio) {
225 LOG("XVA: classicRun");
226
227
228 Size n = portfolio->size();
229 // Create a new empty portfolio, fill it and link it to the simulation market
230 // We don't use Analytic::buildPortfolio() here because we are possibly dealing with a sub-portfolio only.
231 LOG("XVA: Build classic portfolio of size " << n << " linked to the simulation market");
232 CONSOLEW("XVA: Build Portfolio");
233 classicPortfolio_ = boost::make_shared<Portfolio>(inputs_->buildFailedTrades());
234 portfolio->reset();
235 for (const auto& [tradeId, trade] : portfolio->trades())
236 classicPortfolio_->add(trade);
237 QL_REQUIRE(analytic()->market(), "today's market not set");
238 boost::shared_ptr<EngineFactory> factory = engineFactory();
239 classicPortfolio_->build(factory, "analytic/" + label());
240 Date maturityDate = inputs_->asof();
241 if (inputs_->portfolioFilterDate() != Null<Date>())
242 maturityDate = inputs_->portfolioFilterDate();
243 LOG("Filter trades that expire before " << maturityDate);
244 classicPortfolio_->removeMatured(maturityDate);
245 CONSOLE("OK");
246
247 // Allocate cubes for the sub-portfolio we are processing here
248 initClassicRun(classicPortfolio_);
249
250 // This is where the valuation work is done
251 buildClassicCube(classicPortfolio_);
252
253 LOG("XVA: classicRun completed");
254
255 return classicPortfolio_;
256}

259void XvaAnalyticImpl::buildClassicCube(const boost::shared_ptr<Portfolio>& portfolio) {
260
261 LOG("XVA::buildCube");
262
263 // set up valuation calculator factory
264 auto calculators = [this]() {
265 vector<boost::shared_ptr<ValuationCalculator>> calculators;
266 if (analytic()->configurations().scenarioGeneratorData->withCloseOutLag()) {
267 boost::shared_ptr<NPVCalculator> npvCalc =
268 boost::make_shared<NPVCalculator>(inputs_->exposureBaseCurrency());
269 calculators.push_back(boost::make_shared<MPORCalculator>(npvCalc, cubeInterpreter_->defaultDateNpvIndex(),
270 cubeInterpreter_->closeOutDateNpvIndex()));
271 } else
272 calculators.push_back(boost::make_shared<NPVCalculator>(inputs_->exposureBaseCurrency()));
273 if (inputs_->storeFlows())
274 calculators.push_back(boost::make_shared<CashflowCalculator>(
275 inputs_->exposureBaseCurrency(), inputs_->asof(), grid_, cubeInterpreter_->mporFlowsIndex()));
276 if(inputs_->storeCreditStateNPVs() > 0) {
277 calculators.push_back(boost::make_shared<MultiStateNPVCalculator>(inputs_->exposureBaseCurrency(),
278 cubeInterpreter_->creditStateNPVsIndex(),
279 inputs_->storeCreditStateNPVs()));
280 }
281 return calculators;
282 };
283
284 // set up cpty calculator factory
285
286 auto cptyCalculators = [this]() {
287 vector<boost::shared_ptr<CounterpartyCalculator>> cptyCalculators;
288 if (inputs_->storeSurvivalProbabilities()) {
289 string configuration = inputs_->marketConfig("simulation");
290 cptyCalculators.push_back(boost::make_shared<SurvivalProbabilityCalculator>(configuration));
291 }
292 return cptyCalculators;
293 };
294
295 // log message
296
297 ostringstream o;
298 o << "XVA: Build Cube " << portfolio->size() << " x " << grid_->valuationDates().size() << " x " << samples_;
299 CONSOLEW(o.str());
300 LOG(o.str());
301
302 // set up progress indicators
303
304 auto progressBar = boost::make_shared<SimpleProgressBar>(o.str(), ConsoleLog::instance().width(), ConsoleLog::instance().progressBarWidth());
305 auto progressLog = boost::make_shared<ProgressLog>("Building cube", 100, ORE_NOTICE);
306
307 if(inputs_->nThreads() == 1) {
308
309 // single-threaded engine run
310
311 ValuationEngine engine(inputs_->asof(), grid_, simMarket_);
312 engine.registerProgressIndicator(progressBar);
313 engine.registerProgressIndicator(progressLog);
314 engine.buildCube(portfolio, cube_, calculators(), analytic()->configurations().scenarioGeneratorData->withMporStickyDate(),
315 nettingSetCube_, cptyCube_, cptyCalculators());
316 } else {
317
318 // multi-threaded engine run
319
320 /* TODO we assume no netting output cube is needed. Currently there are no valuation calculators in ore that require this cube. */
321
322 auto cubeFactory = [this](const QuantLib::Date& asof, const std::set<std::string>& ids,
323 const std::vector<QuantLib::Date>& dates,
324 const Size samples) -> boost::shared_ptr<NPVCube> {
325 if (cubeDepth_ == 1)
326 return boost::make_shared<SinglePrecisionInMemoryCube>(asof, ids, dates, samples, 0.0f);
327 else
328 return boost::make_shared<SinglePrecisionInMemoryCubeN>(asof, ids, dates, samples,
329 cubeDepth_, 0.0f);
330 };
331
332 std::function<boost::shared_ptr<NPVCube>(const QuantLib::Date&, const std::set<std::string>&,
333 const std::vector<QuantLib::Date>&,
334 const QuantLib::Size)>
335 cptyCubeFactory;
336 if (inputs_->storeSurvivalProbabilities()) {
337 cptyCubeFactory = [](const QuantLib::Date& asof, const std::set<std::string>& ids,
338 const std::vector<QuantLib::Date>& dates,
339 const Size samples) -> boost::shared_ptr<NPVCube> {
340 return boost::make_shared<SinglePrecisionInMemoryCube>(asof, ids, dates, samples, 0.0f);
341 };
342 } else {
343 cptyCubeFactory = [](const QuantLib::Date& asof, const std::set<std::string>& ids,
344 const std::vector<QuantLib::Date>& dates,
345 const Size samples) -> boost::shared_ptr<NPVCube> { return nullptr; };
346 }
347
348 MultiThreadedValuationEngine engine(
349 inputs_->nThreads(), inputs_->asof(), grid_, samples_, analytic()->loader(), scenarioGenerator_,
350 inputs_->simulationPricingEngine(), inputs_->curveConfigs().get(), analytic()->configurations().todaysMarketParams,
351 inputs_->marketConfig("simulation"), analytic()->configurations().simMarketParams, false, false,
352 boost::make_shared<ScenarioFilter>(), inputs_->refDataManager(),
353 *inputs_->iborFallbackConfig(), true, false, cubeFactory, {}, cptyCubeFactory, "xva-simulation");
354
355 engine.registerProgressIndicator(progressBar);
356 engine.registerProgressIndicator(progressLog);
357
358 engine.buildCube(portfolio, calculators, cptyCalculators,
359 analytic()->configurations().scenarioGeneratorData->withMporStickyDate());
360
361 cube_ = boost::make_shared<JointNPVCube>(engine.outputCubes(), portfolio->ids());
362
363 if (inputs_->storeSurvivalProbabilities())
364 cptyCube_ = boost::make_shared<JointNPVCube>(
365 engine.outputCptyCubes(), portfolio->counterparties(), false,
366 [](Real a, Real x) { return std::max(a, x); }, 0.0);
367 }
368
369 CONSOLE("OK");
370
371 LOG("XVA::buildCube done");
372
373 Settings::instance().evaluationDate() = inputs_->asof();
374}

432
433 LOG("XVA: amcRun");
434
435 if (scenarioData_.empty()) {
436 LOG("XVA: Create asd " << grid_->valuationDates().size() << " x " << samples_);
437 scenarioData_.linkTo(

448
449 if (inputs_->nThreads() == 1) {
450 initCube(amcCube_, amcPortfolio_->ids(), cubeDepth_);
451 AMCValuationEngine amcEngine(model_, inputs_->scenarioGeneratorData(), analytic()->market(),
452 inputs_->exposureSimMarketParams()->additionalScenarioDataIndices(),
453 inputs_->exposureSimMarketParams()->additionalScenarioDataCcys());
454 amcEngine.registerProgressIndicator(progressBar);
455 amcEngine.registerProgressIndicator(progressLog);
456 // We only need to generate asd, if this does not happen in the classic run
457 if (!doClassicRun)
458 amcEngine.aggregationScenarioData() = *scenarioData_;
459 amcEngine.buildCube(amcPortfolio_, amcCube_);
460 } else {
461 auto cubeFactory = [this](const QuantLib::Date& asof, const std::set<std::string>& ids,
462 const std::vector<QuantLib::Date>& dates,
463 const Size samples) -> boost::shared_ptr<NPVCube> {
464 if (cubeDepth_ == 1)
465 return boost::make_shared<SinglePrecisionInMemoryCube>(asof, ids, dates, samples, 0.0f);
466 else
467 return boost::make_shared<SinglePrecisionInMemoryCubeN>(asof, ids, dates, samples,
468 cubeDepth_, 0.0f);
469 };
470 AMCValuationEngine amcEngine(
471 inputs_->nThreads(), inputs_->asof(), samples_, analytic()->loader(), inputs_->scenarioGeneratorData(),
472 inputs_->exposureSimMarketParams()->additionalScenarioDataIndices(),
473 inputs_->exposureSimMarketParams()->additionalScenarioDataCcys(), inputs_->crossAssetModelData(),
474 inputs_->amcPricingEngine(), inputs_->curveConfigs().get(), analytic()->configurations().todaysMarketParams,
475 inputs_->marketConfig("lgmcalibration"), inputs_->marketConfig("fxcalibration"),
476 inputs_->marketConfig("eqcalibration"), inputs_->marketConfig("infcalibration"),
477 inputs_->marketConfig("crcalibration"), inputs_->marketConfig("simulation"), inputs_->refDataManager(),
478 *inputs_->iborFallbackConfig(), true, cubeFactory);
479
480 amcEngine.registerProgressIndicator(progressBar);
481 amcEngine.registerProgressIndicator(progressLog);
482 // as for the single-threaded case, we only need to generate asd, if this does not happen in the classic run
483 if (!doClassicRun)
484 amcEngine.aggregationScenarioData() = *scenarioData_;
485 amcEngine.buildCube(amcPortfolio_);
486 amcCube_ = boost::make_shared<JointNPVCube>(amcEngine.outputCubes());
487 }
488
489 CONSOLE("OK");
490
491 LOG("XVA: amcRun completed");
492}

575void XvaAnalyticImpl::runAnalytic(const boost::shared_ptr<ore::data::InMemoryLoader>& loader,
576 const std::set<std::string>& runTypes) {
577
578 LOG("XVA analytic called with asof " << io::iso_date(inputs_->asof()));
579
580 if (runTypes.find("EXPOSURE") != runTypes.end() || runTypes.empty())
581 runSimulation_ = true;
582
583 if (runTypes.find("XVA") != runTypes.end() || runTypes.empty())
584 runXva_ = true;
585
586 Settings::instance().evaluationDate() = inputs_->asof();
587 ObservationMode::instance().setMode(inputs_->exposureObservationModel());
588
589 LOG("XVA: Build Today's Market");
590 CONSOLEW("XVA: Build Market");
591 analytic()->buildMarket(loader);
592 CONSOLE("OK");
593
594 grid_ = analytic()->configurations().scenarioGeneratorData->getGrid();
595 cubeInterpreter_ = boost::make_shared<CubeInterpretation>(
596 inputs_->storeFlows(), analytic()->configurations().scenarioGeneratorData->withCloseOutLag(), scenarioData_,
597 grid_, inputs_->storeCreditStateNPVs(), inputs_->flipViewXVA());
598
599 if (runSimulation_) {
600
601 LOG("XVA: Build simulation market");
602 buildScenarioSimMarket();
603
604 LOG("XVA: Build Scenario Generator");
605 auto globalParams = inputs_->simulationPricingEngine()->globalParameters();
606 auto continueOnCalErr = globalParams.find("ContinueOnCalibrationError");
607 bool continueOnErr = (continueOnCalErr != globalParams.end()) && parseBool(continueOnCalErr->second);
608 buildScenarioGenerator(continueOnErr);
609
610 LOG("XVA: Attach Scenario Generator to ScenarioSimMarket");
611 simMarket_->scenarioGenerator() = scenarioGenerator_;
612
613 // We may have to build two cubes below for complementary sub-portfolios, a classical cube and an AMC cube
614 bool doClassicRun = true;
615 bool doAmcRun = false;
616
617 // Initialize the residual "classical" portfolio that we do not process using AMC
618 auto residualPortfolio = boost::make_shared<Portfolio>(inputs_->buildFailedTrades());
619
620 if (inputs_->amc()) {
621 // Build a separate sub-portfolio for the AMC cube generation and perform its training
622 buildAmcPortfolio();
623
624 // Build the residual portfolio for the classic cube generation, i.e. strip out the AMC part
625 for (auto const& [tradeId, trade] : inputs_->portfolio()->trades()) {
626 if (inputs_->amcTradeTypes().find(trade->tradeType()) == inputs_->amcTradeTypes().end())
627 residualPortfolio->add(trade);
628 }
629
630 LOG("AMC portfolio size " << amcPortfolio_->size());
631 LOG("Residual portfolio size " << residualPortfolio->size());
632
633 doAmcRun = !amcPortfolio_->trades().empty();
634 doClassicRun = !residualPortfolio->trades().empty();
635 } else {
636 for (const auto& [tradeId, trade] : inputs_->portfolio()->trades())
637 residualPortfolio->add(trade);
638 }
639
640 /********************************************************************************
641 * This is where we build cubes and the "classic" valuation work is done
642 * The bulk of the AMC work is done before in the AMC portfolio building/training
643 ********************************************************************************/
644
645 if (doAmcRun)
646 amcRun(doClassicRun);
647 else
648 amcPortfolio_ = boost::make_shared<Portfolio>(inputs_->buildFailedTrades());
649
650 if (doClassicRun)
651 classicPortfolio_ = classicRun(residualPortfolio);
652 else
653 classicPortfolio_ = boost::make_shared<Portfolio>(inputs_->buildFailedTrades());
654
655 /***************************************************
656 * We may have two cubes now that need to be merged
657 ***************************************************/
658
659 if (doClassicRun && doAmcRun) {
660 LOG("Joining classical and AMC cube");
661 cube_ = boost::make_shared<JointNPVCube>(cube_, amcCube_);
662 } else if (!doClassicRun && doAmcRun) {
663 LOG("We have generated an AMC cube only");
664 cube_ = amcCube_;
665 } else {
666 WLOG("We have generated a classic cube only");
667 }
668
669 LOG("NPV cube generation completed");
670
671 /***********************************************************************
672 * We may have two non-empty portfolios to be merged for post processing
673 ***********************************************************************/
674
675 LOG("Classic portfolio size " << classicPortfolio_->size());
676 LOG("AMC portfolio size " << amcPortfolio_->size());
677 auto newPortfolio = boost::make_shared<Portfolio>();
678 for (const auto& [tradeId, trade] : classicPortfolio_->trades())
679 newPortfolio->add(trade);
680 for (const auto& [tradeId, trade] : amcPortfolio_->trades())
681 newPortfolio->add(trade);
682 LOG("Total portfolio size " << newPortfolio->size());
683 if (newPortfolio->size() < inputs_->portfolio()->size()) {
684 ALOG("input portfolio size is " << inputs_->portfolio()->size() <<
685 ", but we have built only " << newPortfolio->size() << " trades");
686 }
687 analytic()->setPortfolio(newPortfolio);
688 } else { // runSimulation_
689
690 // build the portfolio linked to today's market
691 analytic()->buildPortfolio();
692
693 // ... and load a pre-built cube for post-processing
694
695 LOG("Skip cube generation, load input cubes for XVA");
696 CONSOLEW("XVA: Load Cubes");
697 QL_REQUIRE(inputs_->cube(), "XVA without EXPOSURE requires an NPV cube as input");
698 cube_= inputs_->cube();
699 QL_REQUIRE(inputs_->mktCube(), "XVA without EXPOSURE requires a market cube as input");
700 scenarioData_.linkTo(inputs_->mktCube());
701 if (inputs_->nettingSetCube())
702 nettingSetCube_= inputs_->nettingSetCube();
703 if (inputs_->cptyCube())
704 cptyCube_ = inputs_->cptyCube();
705 CONSOLE("OK");
706 }
707
708 MEM_LOG;
709
710 // Return the cubes to serialalize
711 if (inputs_->writeCube()) {
712 analytic()->npvCubes()["XVA"]["cube"] = cube_;
713 analytic()->mktCubes()["XVA"]["scenariodata"] = *scenarioData_;
714 if (nettingSetCube_) {
715 analytic()->npvCubes()["XVA"]["nettingsetcube"] = nettingSetCube_;
716 }
717 if (cptyCube_) {
718 analytic()->npvCubes()["XVA"]["cptycube"] = cptyCube_;
719 }
720 }
721
722 // Generate cube reports to inspect
723 if (inputs_->rawCubeOutput()) {
724 map<string, string> nettingSetMap = analytic()->portfolio()->nettingSetMap();
725 auto report = boost::make_shared<InMemoryReport>();
726 ReportWriter(inputs_->reportNaString()).writeCube(*report, cube_, nettingSetMap);
727 analytic()->reports()["XVA"]["rawcube"] = report;
728 }
729
730 if (runXva_) {
731
732 /*********************************************************************
733 * This is where the aggregation work is done: call the post-processor
734 *********************************************************************/
735
736 CONSOLEW("XVA: Aggregation");
737 runPostProcessor();
738 CONSOLE("OK");
739
740 /******************************************************
741 * Finally generate various (in-memory) reports/outputs
742 ******************************************************/
743
744 CONSOLEW("XVA: Reports");
745 LOG("Generating XVA reports and cube outputs");
746
747 if (inputs_->exposureProfilesByTrade()) {
748 for (const auto& [tradeId, tradeIdCubePos] : postProcess_->tradeIds()) {
749 auto report = boost::make_shared<InMemoryReport>();
750 ReportWriter(inputs_->reportNaString())
751 .writeTradeExposures(*report, postProcess_, tradeId);
752 analytic()->reports()["XVA"]["exposure_trade_" + tradeId] = report;
753 }
754 }
755
756 if (inputs_->exposureProfiles()) {
757 for (auto [nettingSet, nettingSetPosInCube] : postProcess_->nettingSetIds()) {
758 auto exposureReport = boost::make_shared<InMemoryReport>();
759 ReportWriter(inputs_->reportNaString())
760 .writeNettingSetExposures(*exposureReport, postProcess_, nettingSet);
761 analytic()->reports()["XVA"]["exposure_nettingset_" + nettingSet] = exposureReport;
762
763 auto colvaReport = boost::make_shared<InMemoryReport>();
764 ReportWriter(inputs_->reportNaString())
765 .writeNettingSetColva(*colvaReport, postProcess_, nettingSet);
766 analytic()->reports()["XVA"]["colva_nettingset_" + nettingSet] = colvaReport;
767
768 auto cvaSensiReport = boost::make_shared<InMemoryReport>();
769 ReportWriter(inputs_->reportNaString())
770 .writeNettingSetCvaSensitivities(*cvaSensiReport, postProcess_, nettingSet);
771 analytic()->reports()["XVA"]["cva_sensitivity_nettingset_" + nettingSet] = cvaSensiReport;
772 }
773 }
774
775 auto xvaReport = boost::make_shared<InMemoryReport>();
776 ReportWriter(inputs_->reportNaString())
777 .writeXVA(*xvaReport, inputs_->exposureAllocationMethod(), analytic()->portfolio(), postProcess_);
778 analytic()->reports()["XVA"]["xva"] = xvaReport;
779
780 if (inputs_->netCubeOutput()) {
781 auto report = boost::make_shared<InMemoryReport>();
782 ReportWriter(inputs_->reportNaString()).writeCube(*report, postProcess_->netCube());
783 analytic()->reports()["XVA"]["netcube"] = report;
784 }
785
786 if (inputs_->dimAnalytic() || inputs_->mvaAnalytic()) {
787 // Generate DIM evolution report
788 auto dimEvolutionReport = boost::make_shared<InMemoryReport>();
789 postProcess_->exportDimEvolution(*dimEvolutionReport);
790 analytic()->reports()["XVA"]["dim_evolution"] = dimEvolutionReport;
791
792 // Generate DIM regression reports
793 vector<boost::shared_ptr<ore::data::Report>> dimRegReports;
794 for (Size i = 0; i < inputs_->dimOutputGridPoints().size(); ++i) {
795 auto rep = boost::make_shared<InMemoryReport>();
796 dimRegReports.push_back(rep);
797 analytic()->reports()["XVA"]["dim_regression_" + to_string(i)] = rep;
798 }
799 postProcess_->exportDimRegression(inputs_->dimOutputNettingSet(), inputs_->dimOutputGridPoints(),
800 dimRegReports);
801 }
802
803 if (inputs_->creditMigrationAnalytic()) {
804 QL_REQUIRE(
805 postProcess_->creditMigrationPdf().size() == inputs_->creditMigrationTimeSteps().size(),
806 "XvaAnalyticImpl::runAnalytic(): inconsistent post process results for credit migration pdf / cdf ("
807 << postProcess_->creditMigrationPdf().size() << ") and input credit migration time steps ("
808 << inputs_->creditMigrationTimeSteps().size() << ")");
809 for (Size i = 0; i < postProcess_->creditMigrationPdf().size(); ++i) {
810 auto rep = boost::make_shared<InMemoryReport>();
811 analytic()->reports()["XVA"]["credit_migration_" + to_string(inputs_->creditMigrationTimeSteps()[i])] =
812 rep;
813 (*rep)
814 .addColumn("upperBucketBound", double(), 6)
815 .addColumn("pdf", double(), 8)
816 .addColumn("cdf", double(), 8);
817 for (Size j = 0; j < postProcess_->creditMigrationPdf()[i].size(); ++j) {
818 (*rep)
819 .next()
820 .add(postProcess_->creditMigrationUpperBucketBounds()[j])
821 .add(postProcess_->creditMigrationPdf()[i][j])
822 .add(postProcess_->creditMigrationCdf()[i][j]);
823 }
824 rep->end();
825 }
826 }
827
828 CONSOLE("OK");
829 }
830
831 // reset that mode
832 ObservationMode::instance().setMode(inputs_->observationModel());
833}
