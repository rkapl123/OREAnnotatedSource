68//! Historical Scenario Generator
69/*! A Scenario Generator that takes historical scenarios and builds new scenarios by applying historical shifts to the
70 * base scenario (which typically comes from todays market).
71 *
72 * This generator can be used for historical VAR and backtesting calculations, unlike a PRNG based generator it is
73 * limited in the number of scenarios it can generate.
74 *
75 * The scenarios generated are based on the scenario differences between t and t+mpor, these differences are typically
76 * a relative change and this change is then applied to the baseScenario to give a new scenario which is asof Today or
77 * Today+mpor.
78 */
79class HistoricalScenarioGenerator : public ScenarioGenerator {
80public:
81 //! Default constructor
82 HistoricalScenarioGenerator(
83 //! Historical Scenario Loader containing all scenarios
84 const boost::shared_ptr<HistoricalScenarioLoader>& historicalScenarioLoader,
85 //! Scenario factory to use
86 const boost::shared_ptr<ScenarioFactory>& scenarioFactory,
87 //! Calendar to use
88 const QuantLib::Calendar& cal,
89 //! Mpor days or step size
90 const Size mporDays = 10,
91 //! overlapping scenarios
92 const bool overlapping = true,
93 //! return configuration
94 const ReturnConfiguration& returnConfiguration = ReturnConfiguration(),
95 //! optional adjustment factors for stock splits etc
96 const boost::shared_ptr<ore::data::AdjustmentFactors>& adjFactors = nullptr,
97 //! string prepended to label of all scenarios generated
98 const std::string& labelPrefix = "");
99
100 //! Set base scenario, this also defines the asof date
101 boost::shared_ptr<Scenario>& baseScenario() { return baseScenario_; }
102 //! Get base scenario
103 const boost::shared_ptr<Scenario>& baseScenario() const { return baseScenario_; }
104
105 //! Get calendar
106 const QuantLib::Calendar& cal() const { return cal_; }
107 //! Get mpor days
108 QuantLib::Size mporDays() const { return mporDays_; }
109 //! Are scenarios overlapping
110 bool overlapping() const { return overlapping_; }
111
112 //! Return configuration
113 const ReturnConfiguration& returnConfiguration() const { return returnConfiguration_; }
114
115 //! Scaling
116 virtual QuantLib::Real scaling(const RiskFactorKey& key, const QuantLib::Real& keyReturn) { return 1; };
117
118 //! Return the next scenario for the given date.
119 /*! Date should be asof or asof+mporDays, this class only checks that date is >= the asof date.
120 Whatever you put in here will be in the returned scenario.
121
122 Generator returns scenarios in order of shifts and throws if we have run out of historicals
123
124 If Mpor > 1 than the scenarios will overlap.
125 */
126 boost::shared_ptr<Scenario> next(const QuantLib::Date& d) override;
127
128 //! Reset the generator so calls to next() return the first scenario.
129 /*! This allows re-generation of scenarios if required. */
130 void reset() override { i_ = 0; }
131
132 //! Number of scenarios this generator can generate
133 virtual QuantLib::Size numScenarios() const;
134
135 //! set the start and end dates, can be overwritten in derived class
136 virtual void setDates();
137
138 //! start dates from which the scenarios were generated from
139 const std::vector<QuantLib::Date>& startDates() const { return startDates_; }
140 //! end dates from which the scenarios were generated from
141 const std::vector<QuantLib::Date>& endDates() const { return endDates_; }
142
143 //! Get the HistoricalScenarioLoader
144 const boost::shared_ptr<HistoricalScenarioLoader>& scenarioLoader() const { return historicalScenarioLoader_; }
145
146 //! Get the ScenarioFactory
147 const boost::shared_ptr<ScenarioFactory>& scenarioFactory() const { return scenarioFactory_; }
148
149 //! Get the adj factors
150 const boost::shared_ptr<ore::data::AdjustmentFactors>& adjFactors() const { return adjFactors_; }
151
152 //! Get (start, end) scenario date pairs filtered on the given period
153 std::vector<std::pair<QuantLib::Date, QuantLib::Date>> filteredScenarioDates(const ore::data::TimePeriod& period) const;
154
155 //! Get the scenario label prefix
156 const std::string& labelPrefix() const { return labelPrefix_; }
157
158protected:
159 // to be managed in derived classes, if next is overwritten
160 Size i_;
161
162 boost::shared_ptr<HistoricalScenarioLoader> historicalScenarioLoader_;
163 std::vector<QuantLib::Date> startDates_, endDates_;
164
165 boost::shared_ptr<ScenarioFactory> scenarioFactory_;
166 boost::shared_ptr<Scenario> baseScenario_;
167
168 //! The Scenario Pairs for a given index
169 std::pair<boost::shared_ptr<Scenario>, boost::shared_ptr<Scenario>> scenarioPair();
170
171 //! Returns the adjusted price
172 /*! Scenarios may contian unadjusted market prices e.g equity spot prices,
173 apply adjustment factors to ensure no jumps between 2 scenarios
174 Only handles equity spot adjustments at the moment */
175 QuantLib::Real adjustedPrice(RiskFactorKey key, QuantLib::Date d, QuantLib::Real price);
176
177private:
178 QuantLib::Calendar cal_;
179 QuantLib::Size mporDays_;
180 bool overlapping_;
181 ReturnConfiguration returnConfiguration_;
182 boost::shared_ptr<ore::data::AdjustmentFactors> adjFactors_;
183 std::string labelPrefix_;
184};
