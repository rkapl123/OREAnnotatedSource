73//! Historical Scenario Generator
74/*! A Scenario Generator that takes historical scenarios and builds new scenarios by applying historical shifts to the
75 * base scenario (which typically comes from todays market).
76 *
77 * This generator can be used for historical VAR and backtesting calculations, unlike a PRNG based generator it is
78 * limited in the number of scenarios it can generate.
79 *
80 * The scenarios generated are based on the scenario differences between t and t+mpor, these differences are typically
81 * a relative change and this change is then applied to the baseScenario to give a new scenario which is asof Today or
82 * Today+mpor.
83 */
84class HistoricalScenarioGenerator : public ScenarioGenerator {
85public:
86 struct HistoricalScenarioCalculationDetails {
87 Date scenarioDate1;
88 Date scenarioDate2;
89 RiskFactorKey key;
90 double baseValue;
91 double adjustmentFactor1;
92 double adjustmentFactor2;
93 double scenarioValue1;
94 double scenarioValue2;
95 ReturnConfiguration::ReturnType returnType;
96 double scaling;
97 double returnValue;
98 double scenarioValue;
99 };
100
101 //! Default constructor
102 HistoricalScenarioGenerator(
103 //! Historical Scenario Loader containing all scenarios
104 const boost::shared_ptr<HistoricalScenarioLoader>& historicalScenarioLoader,
105 //! Scenario factory to use
106 const boost::shared_ptr<ScenarioFactory>& scenarioFactory,
107 //! Calendar to use
108 const QuantLib::Calendar& cal,
109 //! optional adjustment factors for stock splits etc
110 const boost::shared_ptr<ore::data::AdjustmentFactors>& adjFactors = nullptr,
111 //! Mpor days or step size
112 const Size mporDays = 10,
113 //! overlapping scenarios
114 const bool overlapping = true,
115 //! return configuration
116 const ReturnConfiguration& returnConfiguration = ReturnConfiguration(),
117 //! string prepended to label of all scenarios generated
118 const std::string& labelPrefix = "");
119
120 //! Set base scenario, this also defines the asof date
121 boost::shared_ptr<Scenario>& baseScenario() { return baseScenario_; }
122 //! Get base scenario
123 const boost::shared_ptr<Scenario>& baseScenario() const { return baseScenario_; }
124
125 //! Get calendar
126 const QuantLib::Calendar& cal() const { return cal_; }
127 //! Get mpor days
128 QuantLib::Size mporDays() const { return mporDays_; }
129 //! Are scenarios overlapping
130 bool overlapping() const { return overlapping_; }
131
132 //! Return configuration
133 const ReturnConfiguration& returnConfiguration() const { return returnConfiguration_; }
134
135 //! Scaling
136 virtual QuantLib::Real scaling(const RiskFactorKey& key, const QuantLib::Real& keyReturn) { return 1; };
137
138 //! Return the next scenario for the given date.
139 /*! Date should be asof or asof+mporDays, this class only checks that date is >= the asof date.
140 Whatever you put in here will be in the returned scenario.
141
142 Generator returns scenarios in order of shifts and throws if we have run out of historicals
143
144 If Mpor > 1 than the scenarios will overlap.
145 */
146 boost::shared_ptr<Scenario> next(const QuantLib::Date& d) override;
147
148 //! Return the calculation details of the last generated scenario */
149 const std::vector<HistoricalScenarioCalculationDetails>& lastHistoricalScenarioCalculationDetails() const;
150
151 //! Reset the generator so calls to next() return the first scenario.
152 /*! This allows re-generation of scenarios if required. */
153 void reset() override { i_ = 0; }
154
155 //! Number of scenarios this generator can generate
156 virtual QuantLib::Size numScenarios() const;
157
158 //! set the start and end dates, can be overwritten in derived class
159 virtual void setDates();
160
161 //! start dates from which the scenarios were generated from
162 const std::vector<QuantLib::Date>& startDates() const { return startDates_; }
163 //! end dates from which the scenarios were generated from
164 const std::vector<QuantLib::Date>& endDates() const { return endDates_; }
165
166 //! Get the HistoricalScenarioLoader
167 const boost::shared_ptr<HistoricalScenarioLoader>& scenarioLoader() const { return historicalScenarioLoader_; }
168
169 //! Get the ScenarioFactory
170 const boost::shared_ptr<ScenarioFactory>& scenarioFactory() const { return scenarioFactory_; }
171
172 //! Get the adj factors
173 const boost::shared_ptr<ore::data::AdjustmentFactors>& adjFactors() const { return adjFactors_; }
174
175 //! Get (start, end) scenario date pairs filtered on the given period
176 std::vector<std::pair<QuantLib::Date, QuantLib::Date>> filteredScenarioDates(const ore::data::TimePeriod& period) const;
177
178 //! Get the scenario label prefix
179 const std::string& labelPrefix() const { return labelPrefix_; }
180
181protected:
182 // to be managed in derived classes, if next is overwritten
183 Size i_;
184
185 boost::shared_ptr<HistoricalScenarioLoader> historicalScenarioLoader_;
186 std::vector<QuantLib::Date> startDates_, endDates_;
187
188 boost::shared_ptr<ScenarioFactory> scenarioFactory_;
189 boost::shared_ptr<Scenario> baseScenario_;
190
191 //! The Scenario Pairs for a given index
192 std::pair<boost::shared_ptr<Scenario>, boost::shared_ptr<Scenario>> scenarioPair();
193
194 //! Returns the adjusted price
195 /*! Scenarios may contian unadjusted market prices e.g equity spot prices,
196 apply adjustment factors to ensure no jumps between 2 scenarios
197 Only handles equity spot adjustments at the moment */
198 QuantLib::Real adjustedPrice(RiskFactorKey key, QuantLib::Date d, QuantLib::Real price);
199
200 // details on the last generated scenario
201 std::vector<HistoricalScenarioCalculationDetails> calculationDetails_;
202
203protected:
204 QuantLib::Calendar cal_;
205 QuantLib::Size mporDays_;
206
207private:
208 boost::shared_ptr<ore::data::AdjustmentFactors> adjFactors_;
209 bool overlapping_;
210 ReturnConfiguration returnConfiguration_;
211 std::string labelPrefix_;
212};

244//! Historical scenario generator transform
245/*! This class allows scenarios within a historical scenario generator to be transformed from discount rates to
246 * zero rates and survival probabilities to hazard rates
247 * WARNING: This transform function should only be used for backtesting statistics reports requiring the transforms
248 * listed above.
249 */
250class HistoricalScenarioGeneratorTransform : public HistoricalScenarioGenerator {
251public:
252 HistoricalScenarioGeneratorTransform(boost::shared_ptr<HistoricalScenarioGenerator>& hsg,
253 const boost::shared_ptr<ScenarioSimMarket>& simMarket,
254 const boost::shared_ptr<ScenarioSimMarketParameters>& simMarketConfig)
255 : HistoricalScenarioGenerator(hsg->scenarioLoader(), hsg->scenarioFactory(), hsg->cal(), hsg->adjFactors(),
256 hsg->mporDays(), hsg->overlapping(), hsg->returnConfiguration(),
257 hsg->labelPrefix()),
258 simMarket_(simMarket), simMarketConfig_(simMarketConfig) {
259 baseScenario_ = hsg->baseScenario();
260 }
261
262 boost::shared_ptr<Scenario> next(const QuantLib::Date& d) override;
263
264private:
265 boost::shared_ptr<ScenarioSimMarket> simMarket_;
266 boost::shared_ptr<ScenarioSimMarketParameters> simMarketConfig_;
267};
