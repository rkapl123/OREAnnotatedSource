32void initCube(NPVCube& cube) {
33 // Set every (i,j,k,d) node to be i*1000000 + j + (k/1000000) + d*3
34 for (Size i = 0; i < cube.numIds(); ++i) {
35 for (Size j = 0; j < cube.numDates(); ++j) {
36 for (Size k = 0; k < cube.samples(); ++k) {
37 for (Size d = 0; d < cube.depth(); ++d) {
38 cube.set(i * 1000000.0 + j + k / 1000000.0 + d * 3, i, j, k, d);
39 }
40 }
41 }
42 }
43}

45void checkCube(NPVCube& cube, Real tolerance) {
46 // Now check each value
47 for (Size i = 0; i < cube.numIds(); ++i) {
48 for (Size j = 0; j < cube.numDates(); ++j) {
49 for (Size k = 0; k < cube.samples(); ++k) {
50 for (Size d = 0; d < cube.depth(); ++d) {
51 Real expected = i * 1000000.0 + j + k / 1000000.0 + d * 3;
52 Real actual = cube.get(i, j, k, d);
53 BOOST_CHECK_CLOSE(expected, actual, tolerance);
54 }
55 }
56 }
57 }
58}

60void testCube(NPVCube& cube, const std::string& cubeName, Real tolerance) {
61 BOOST_TEST_MESSAGE("Testing cube " << cubeName);
62
63 initCube(cube);
64
65 // Check we can't set anything out of bounds
66 BOOST_CHECK_THROW(cube.set(1.0, cube.numIds(), 0, 0), std::exception);
67 BOOST_CHECK_THROW(cube.set(1.0, 0, cube.numDates(), 0), std::exception);
68 BOOST_CHECK_THROW(cube.set(1.0, 0, 0, cube.samples()), std::exception);
69 BOOST_CHECK_THROW(cube.set(1.0, "test_id", Date::todaysDate(), 0), std::exception);
70
71 // Check we can't get anything out of bounds
72 BOOST_CHECK_THROW(cube.get(cube.numIds(), 0, 0), std::exception);
73 BOOST_CHECK_THROW(cube.get(0, cube.numDates(), 0), std::exception);
74 BOOST_CHECK_THROW(cube.get(0, 0, cube.samples()), std::exception);
75 BOOST_CHECK_THROW(cube.get("test_id", Date::todaysDate(), 0), std::exception);
76
77 checkCube(cube, tolerance);
78 // All done
79}

81template <class T> void testCubeFileIO(NPVCube& cube, const std::string& cubeName, Real tolerance) {
82
83 initCube(cube);
84
85 // get a random filename
86 string filename = boost::filesystem::unique_path().string();
87 BOOST_TEST_MESSAGE("Saving cube " << cubeName << " to file " << filename);
88 cube.save(filename);
89
90 // Create a new Cube and load it
91 T* cube2 = new T();
92 BOOST_TEST_MESSAGE("Loading from file " << filename);
93 cube2->load(filename);
94 BOOST_TEST_MESSAGE("Cube " << cubeName << " loaded from file.");
95
96 // Delete the file to make sure all reads are from memory
97 boost::filesystem::remove(filename);
98
99 // Check dimensions match
100 BOOST_CHECK_EQUAL(cube.numIds(), cube2->numIds());
101 BOOST_CHECK_EQUAL(cube.numDates(), cube2->numDates());
102 BOOST_CHECK_EQUAL(cube.samples(), cube2->samples());
103 BOOST_CHECK_EQUAL(cube.depth(), cube2->depth());
104
105 // check all values
106 checkCube(*cube2, tolerance);
107 // All done
108 delete cube2;
109}
