46 vector<string> getFilenames(const string& fileString, const string& path) {
47  vector<string> fileNames;
48  boost::split(fileNames, fileString, boost::is_any_of(",;"), boost::token_compress_on);
49  for (auto it = fileNames.begin(); it < fileNames.end(); it++) {
50  boost::trim(*it);
51  *it = path + "/" + *it;
52  }
53  return fileNames;
54 }

79 int OREApp::run() {
80 
81  boost::timer timer;
82 
83  try {
84  out_ << "ORE starting" << std::endl;
85  LOG("ORE starting");
86  // readSetup();
87 
88  /*********
89  * Build Markets
90  */
91  out_ << setw(tab_) << left << "Market... " << flush;
92  buildMarket();
93  out_ << "OK" << endl;
94 
95  /************************
96  *Build Pricing Engine Factory
97  */
98  out_ << setw(tab_) << left << "Engine factory... " << flush;
99  engineFactory_ = buildEngineFactory(market_);
100  out_ << "OK" << endl;
101 
102  /******************************
103  * Load and Build the Portfolio
104  */
105  out_ << setw(tab_) << left << "Portfolio... " << flush;
106  portfolio_ = buildPortfolio(engineFactory_);
107  out_ << "OK" << endl;
108 
109  /******************************
110  * Write initial reports
111  */
112  out_ << setw(tab_) << left << "Write Reports... " << flush;
113  writeInitialReports();
114  out_ << "OK" << endl;
115 
116  /**************************
117  * Write base scenario file
118  */
119  out_ << setw(tab_) << left << "Write Base Scenario... " << flush;
120  if (writeBaseScenario_) {
121  writeBaseScenario();
122  out_ << "OK" << endl;
123  } else {
124  LOG("skip base scenario");
125  out_ << "SKIP" << endl;
126  }
127 
128  /**********************
129  * Sensitivity analysis
130  */
131  if (sensitivity_) {
132  out_ << setw(tab_) << left << "Sensitivity Report... " << flush;
133 
134  // We reset this here because the date grid building in sensitivity analysis depends on it.
135  Settings::instance().evaluationDate() = asof_;
136  getSensitivityRunner()->runSensitivityAnalysis(market_, conventions_, curveConfigs_, marketParameters_);
137  out_ << "OK" << endl;
138  } else {
139  LOG("skip sensitivity analysis");
140  out_ << setw(tab_) << left << "Sensitivity... ";
141  out_ << "SKIP" << endl;
142  }
143 
144  /****************
145  * Stress testing
146  */
147  if (stress_) {
148  runStressTest();
149  } else {
150  LOG("skip stress test");
151  out_ << setw(tab_) << left << "Stress testing... ";
152  out_ << "SKIP" << endl;
153  }
154 
155  /****************
156  * Parametric VaR
157  */
158  if (parametricVar_) {
159  runParametricVar();
160  } else {
161  LOG("skip parametric var");
162  out_ << setw(tab_) << left << "Parametric VaR... ";
163  out_ << "SKIP" << endl;
164  }
165 
166  /******************************************
167  * Simulation: Scenario and Cube Generation
168  */
169 
170  if (simulate_) {
171  generateNPVCube();
172  } else {
173  LOG("skip simulation");
174  out_ << setw(tab_) << left << "Simulation... ";
175  out_ << "SKIP" << endl;
176  }
177 
178  /*****************************
179  * Aggregation and XVA Reports
180  */
181  out_ << setw(tab_) << left << "Aggregation and XVA Reports... " << flush;
182  if (xva_) {
183 
184  // We reset this here because the date grid building below depends on it.
185  Settings::instance().evaluationDate() = asof_;
186 
187  // Use pre-generated cube
188  if (!cube_)
189  loadCube();
190 
191  QL_REQUIRE(cube_->numIds() == portfolio_->size(),
192  "cube x dimension (" << cube_->numIds() << ") does not match portfolio size ("
193  << portfolio_->size() << ")");
194 
195  // Use pre-generared scenarios
196  if (!scenarioData_)
197  loadScenarioData();
198 
199  QL_REQUIRE(scenarioData_->dimDates() == cube_->dates().size(),
200  "scenario dates do not match cube grid size");
201  QL_REQUIRE(scenarioData_->dimSamples() == cube_->samples(),
202  "scenario sample size does not match cube sample size");
203 
204  runPostProcessor();
205  out_ << "OK" << endl;
206  out_ << setw(tab_) << left << "Write Reports... " << flush;
207  writeXVAReports();
208  if (writeDIMReport_)
209  writeDIMReport();
210  out_ << "OK" << endl;
211  } else {
212  LOG("skip XVA reports");
213  out_ << "SKIP" << endl;
214  }
215 
216  } catch (std::exception& e) {
217  ALOG("Error: " << e.what());
218  out_ << "Error: " << e.what() << endl;
219  return 1;
220  }
221 
222  out_ << "run time: " << setprecision(2) << timer.elapsed() << " sec" << endl;
223  out_ << "ORE done." << endl;
224 
225  LOG("ORE done.");
226  return 0;
227 }

362 boost::shared_ptr<ScenarioGenerator>
363 OREApp::buildScenarioGenerator(boost::shared_ptr<Market> market,
364  boost::shared_ptr<ScenarioSimMarketParameters> simMarketData,
365  boost::shared_ptr<ScenarioGeneratorData> sgd) {
366  LOG("Build Simulation Model");
367  string simulationConfigFile = inputPath_ + "/" + params_->get("simulation", "simulationConfigFile");
368  LOG("Load simulation model data from file: " << simulationConfigFile);
369  boost::shared_ptr<CrossAssetModelData> modelData = boost::make_shared<CrossAssetModelData>();
370  modelData->fromFile(simulationConfigFile);
371  string lgmCalibrationMarketStr = Market::defaultConfiguration;
372  if (params_->has("markets", "lgmcalibration"))
373  lgmCalibrationMarketStr = params_->get("markets", "lgmcalibration");
374  string fxCalibrationMarketStr = Market::defaultConfiguration;
375  if (params_->has("markets", "fxcalibration"))
376  fxCalibrationMarketStr = params_->get("markets", "fxcalibration");
377  string eqCalibrationMarketStr = Market::defaultConfiguration;
378  if (params_->has("markets", "eqcalibration"))
379  eqCalibrationMarketStr = params_->get("markets", "eqcalibration");
380  string infCalibrationMarketStr = Market::defaultConfiguration;
381  if (params_->has("markets", "infcalibration"))
382  infCalibrationMarketStr = params_->get("markets", "infcalibration");
383  string simulationMarketStr = Market::defaultConfiguration;
384  if (params_->has("markets", "simulation"))
385  simulationMarketStr = params_->get("markets", "simulation");
386 
387  CrossAssetModelBuilder modelBuilder(market, lgmCalibrationMarketStr, fxCalibrationMarketStr, eqCalibrationMarketStr,
388  infCalibrationMarketStr, simulationMarketStr);
389  boost::shared_ptr<QuantExt::CrossAssetModel> model = modelBuilder.build(modelData);
390 
391  LOG("Load Simulation Parameters");
392  ScenarioGeneratorBuilder sgb(sgd);
393  boost::shared_ptr<ScenarioFactory> sf = boost::make_shared<SimpleScenarioFactory>();
394  boost::shared_ptr<ScenarioGenerator> sg = sgb.build(
395  model, sf, simMarketData, asof_, market, params_->get("markets", "simulation")); // pricing or simulation?
396  // Optionally write out scenarios
397  if (params_->has("simulation", "scenariodump")) {
398  string filename = outputPath_ + "/" + params_->get("simulation", "scenariodump");
399  sg = boost::make_shared<ScenarioWriter>(sg, filename);
400  }
401  return sg;
402 }

514 void OREApp::runParametricVar() {
515 
516  MEM_LOG;
517  LOG("Running parametric VaR");
518 
519  out_ << setw(tab_) << left << "Parametric VaR Report... " << flush;
520 
521  LOG("Get sensitivity data");
522  string sensiFile = inputPath_ + "/" + params_->get("parametricVar", "sensitivityInputFile");
523  auto ss = boost::make_shared<SensitivityFileStream>(sensiFile);
524 
525  LOG("Build trade to portfolio id mapping");
526  map<string, set<string>> tradePortfolio;
527  for (auto const& t : portfolio_->trades()) {
528  tradePortfolio[t->id()].insert(t->portfolioIds().begin(), t->portfolioIds().end());
529  }
530 
531  LOG("Load covariance matrix data");
532  map<std::pair<RiskFactorKey, RiskFactorKey>, Real> covarData;
533  loadCovarianceDataFromCsv(covarData, inputPath_ + "/" + params_->get("parametricVar", "covarianceInputFile"));
534 
535  Size mcSamples = Null<Size>(), mcSeed = Null<Size>();
536  string method = params_->get("parametricVar", "method");
537  if (method == "MonteCarlo") {
538  mcSamples = parseInteger(params_->get("parametricVar", "mcSamples"));
539  mcSeed = parseInteger(params_->get("parametricVar", "mcSeed"));
540  }
541 
542  string portfolioFilter =
543  params_->has("parametricVar", "portfolioFilter") ? params_->get("parametricVar", "portfolioFilter") : "";
544 
545  LOG("Build parametric var report");
546  auto calc =
547  buildParametricVarCalculator(tradePortfolio, portfolioFilter, ss, covarData,
548  parseListOfValues<Real>(params_->get("parametricVar", "quantiles"), &parseReal),
549  method, mcSamples, mcSeed, parseBool(params_->get("parametricVar", "breakdown")),
550  parseBool(params_->get("parametricVar", "salvageCovarianceMatrix")));
551 
552  CSVFileReport report(outputPath_ + "/" + params_->get("parametricVar", "outputFile"));
553  calc->calculate(report);
554  out_ << "OK" << endl;
555 
556  LOG("Parametric VaR completed");
557  MEM_LOG;
558 }

578  LOG("Get Market Configuration");

580 
581  LOG("Get Simulation Market Parameters");
