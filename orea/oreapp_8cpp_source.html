51vector<string> getFilenames(const string& fileString, const string& path) {
52 vector<string> fileNames;
53 boost::split(fileNames, fileString, boost::is_any_of(",;"), boost::token_compress_on);
54 for (auto it = fileNames.begin(); it < fileNames.end(); it++) {
55 boost::trim(*it);
56 *it = path + "/" + *it;
57 }
58 return fileNames;
59}

131 out_ << "OK" << endl;

133 /******************************
134 * Write initial reports
135 */

139 * Output pricing stats
140 */
141

146 */

158 */

175 */

224

226 boost::shared_ptr<XvaRunner> xva = getXvaRunner();
227 xva->runXva(market_, true);
228 postProcess_ = xva->postProcess();
229 out_ << "OK" << endl;
230

232 writeXVAReports();

234 writeDIMReport();
235 out_ << "OK" << endl;
236

238

252 */

255 // We reset this here because the date grid building below depends on it.

257

259 if (!cube_)

261
262 QL_REQUIRE(cube_->numIds() == portfolio_->size(),
263 "cube x dimension (" << cube_->numIds() << ") does not match portfolio size ("
264 << portfolio_->size() << ")");
265

267 if (!scenarioData_)
268 loadScenarioData();
269
270 QL_REQUIRE(scenarioData_->dimDates() == cube_->dates().size(),

272 QL_REQUIRE(scenarioData_->dimSamples() == cube_->samples(),
273 "scenario sample size does not match cube sample size");
274

276 out_ << "OK" << endl;
277 out_ << setw(tab_) << left << "Write Reports... " << flush;

280 writeDIMReport();
281 out_ << "OK" << endl;
282 } else {
283 LOG("skip XVA reports");
284 out_ << "SKIP" << endl;
285 }
286 }
287
288 } catch (std::exception& e) {
289 ALOG("Error: " << e.what());
290 out_ << "Error: " << e.what() << endl;
291 return 1;
292 }
293
294 timer.stop();
295 out_ << "run time: " << setprecision(2) << timer.format(default_places, "%w") << " sec" << endl;
296 out_ << "ORE done." << endl;
297
298 LOG("ORE done.");
299 return 0;
