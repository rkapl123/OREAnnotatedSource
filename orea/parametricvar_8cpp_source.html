49void ParametricVarCalculator::calculate(ore::data::Report& report) {
50 LOG("Parametric VaR calculation started...");
51
52 // prepare report
53 report.addColumn("Portfolio", string()).addColumn("RiskClass", string()).addColumn("RiskType", string());
54 for (Size i = 0; i < p_.size(); ++i)
55 report.addColumn("Quantile_" + std::to_string(p_[i]), double(), 6);
56
57 // build portfolio filter, if given
58 bool hasFilter = false;
59 boost::regex filter;
60 if (portfolioFilter_ != "") {
61 hasFilter = true;
62 filter = boost::regex(portfolioFilter_);
63 LOG("Portfolio filter: " << portfolioFilter_);
64 } else {
65 LOG("No portfolio filter will be applied.");
66 }
67
68 // read sensitivities and preaggregate them per portfolio
69 LOG("Preaggregate sensitivities per portfolio");
70 std::set<RiskFactorKey> sensiKeysTmp;
71 std::set<std::string> portfoliosTmp;
72 std::map<std::string, std::map<std::pair<RiskFactorKey, RiskFactorKey>, Real>> value1, value2;
73 std::map<std::pair<RiskFactorKey, RiskFactorKey>, Real> value1All, value2All;
74 while (SensitivityRecord sr = sensitivities_->next()) {
75 std::set<std::string> portfolios;
76 auto pn = tradePortfolios_.find(sr.tradeId);
77 if (pn != tradePortfolios_.end()) {
78 if (pn->second.empty())
79 portfolios = {"(empty)"};
80 else
81 portfolios = pn->second;
82 } else
83 portfolios = {"(unknown)"};
84 RiskFactorKey k1 = sr.key_1;
85 RiskFactorKey k2 = sr.key_2;
86 auto key = std::make_pair(k1, k2);
87 if (k1 != RiskFactorKey())
88 sensiKeysTmp.insert(k1);
89 if (k2 != RiskFactorKey())
90 sensiKeysTmp.insert(k2);
91 bool relevant = false;
92 for (auto const& p : portfolios) {
93 if (!hasFilter || boost::regex_match(p, filter)) {
94 relevant = true;
95 portfoliosTmp.insert(p);
96 if (sr.isCrossGamma()) {
97 value1[p][key] += sr.gamma;
98 } else {
99 value1[p][key] += sr.delta;
100 }
101 if (!sr.isCrossGamma()) {
102 value2[p][key] += sr.gamma;
103 }
104 }
105 }
106 if (relevant) {
107 if (sr.isCrossGamma()) {
108 value1All[key] += sr.gamma;
109 } else {
110 value1All[key] += sr.delta;
111 }
112 if (!sr.isCrossGamma()) {
113 value2All[key] += sr.gamma;
114 }
115 }
116 }
117 std::vector<RiskFactorKey> sensiKeys(sensiKeysTmp.begin(), sensiKeysTmp.end());
118 std::vector<bool> sensiKeyHasNonZeroVariance(sensiKeys.size(), false);
119 std::vector<std::string> portfolios(portfoliosTmp.begin(), portfoliosTmp.end());
120 LOG("Have " << sensiKeys.size() << " sensitivity keys in " << portfolios.size() << " portfolios");
121
122 // build global covariance matrix
123 Matrix omega(sensiKeys.size(), sensiKeys.size(), 0.0);
124 Size unusedCovariance = 0;
125 for (const auto& c : covariance_) {
126 auto k1 = std::find(sensiKeys.begin(), sensiKeys.end(), c.first.first);
127 auto k2 = std::find(sensiKeys.begin(), sensiKeys.end(), c.first.second);
128 if (k1 != sensiKeys.end() && k2 != sensiKeys.end()) {
129 omega(k1 - sensiKeys.begin(), k2 - sensiKeys.begin()) = c.second;
130 if (k1 == k2)
131 sensiKeyHasNonZeroVariance[k1 - sensiKeys.begin()] = true;
132 } else {
133 ++unusedCovariance;
134 }
135 }
136 LOG("Found " << covariance_.size() << " covariance matrix entries, " << unusedCovariance
137 << " do not match a portfolio sensitivity and will not be used.");
138 for (Size i = 0; i < sensiKeyHasNonZeroVariance.size(); ++i) {
139 if (!sensiKeyHasNonZeroVariance[i]) {
140 WLOG("Zero variance assigned to sensitivity key " << sensiKeys[i]);
141 }
142 }
143
144 // make covariance matrix positive semi-definite
145 boost::shared_ptr<QuantExt::CovarianceSalvage> covarianceSalvage;
146 LOG("Covariance matrix has dimension " << sensiKeys.size() << " x " << sensiKeys.size());
147 if (salvageCovarianceMatrix_) {
148 LOG("Make covariance matrix positive semi-definite using spectral method");
149 covarianceSalvage = boost::make_shared<QuantExt::SpectralCovarianceSalvage>();
150 } else {
151 LOG("Covariance matrix is no salvaged, check for positive semi-definiteness");
152 SymmetricSchurDecomposition ssd(omega);
153 Real evMin = ssd.eigenvalues().back();
154 QL_REQUIRE(evMin > 0.0 || close_enough(evMin, 0.0),
155 "ParametricVar: input covariance matrix is not positive semi-definite, smallest eigenvalue is "
156 << evMin);
157 LOG("Smallest eigenvalue is " << evMin);
158 covarianceSalvage = boost::make_shared<QuantExt::NoCovarianceSalvage>();
159 }
160 LOG("Done.");
161
162 // loop over portfolios (index 0 = all portfolios)
163 for (Size i = 0; i <= (!breakdown_ || portfolios.size() == 1 ? 0 : portfolios.size()); ++i) {
164 std::string portfolioName = i == 0 ? (portfolios.size() > 1 ? "(all)" : portfolios.front()) : portfolios[i - 1];
165 // build delta and gamma for given portfolio
166 const auto& val1 = (i == 0 ? value1All : value1[portfolios[i - 1]]);
167 const auto& val2 = (i == 0 ? value2All : value2[portfolios[i - 1]]);
168 Array delta(sensiKeys.size(), 0.0);
169 Matrix gamma(sensiKeys.size(), sensiKeys.size(), 0.0);
170 for (auto const& p : val1) {
171 auto k1 = p.first.first;
172 auto k2 = p.first.second;
173 Size idx1 = std::find(sensiKeys.begin(), sensiKeys.end(), k1) - sensiKeys.begin();
174 QL_REQUIRE(idx1 < sensiKeys.size(), "ParametricVarCalculator::computeVar: key1 \""
175 << k1 << "\" in value1 not found, this is unexpected.");
176 if (k2 == RiskFactorKey()) {
177 // delta
178 delta[idx1] += p.second;
179 } else {
180 // cross gamma
181 Size idx2 = std::find(sensiKeys.begin(), sensiKeys.end(), k2) - sensiKeys.begin();
182 QL_REQUIRE(idx2 < sensiKeys.size(), "ParametricVarCalculator::computeVar: key2 \""
183 << k2 << "\" in value1 not found, this is unexpected.");
184 gamma[idx1][idx2] = gamma[idx2][idx1] = p.second;
185 }
186 }
187 for (auto const& p : val2) {
188 // diagonal gamma
189 auto k1 = p.first.first;
190 Size idx1 = std::find(sensiKeys.begin(), sensiKeys.end(), k1) - sensiKeys.begin();
191 QL_REQUIRE(idx1 < sensiKeys.size(), "ParametricVarCalculator::computeVar: key1 \""
192 << k1 << "\" in value2 not found, this is unexpected.");
193 gamma[idx1][idx1] = p.second;
194 }
195 // loop over risk class and type filters (index 0 == all risk types)
196 for (Size j = 0; j < (breakdown_ ? RiskFilter::numberOfRiskClasses() : 1); ++j) {
197 for (Size k = 0; k < (breakdown_ ? RiskFilter::numberOfRiskTypes() : 1); ++k) {
198 // TODO should we rather project on the set of indices with non-zero sensis
199 // instead of copying the initial sensis and setting the non-relevant entries
200 // to zero?
201 Array deltaFiltered(delta);
202 Matrix gammaFiltered(gamma);
203 RiskFilter rf(j, k);
204 LOG("Compute parametric var for portfolio \"" << portfolioName << "\""
205 << ", risk class " << rf.riskClassLabel()
206 << ", risk type " << rf.riskTypeLabel());
207 // set sensis which do not belong to risk type filter to zero
208 for (Size idx = 0; idx < sensiKeys.size(); ++idx) {
209 if (!rf.allowed(sensiKeys[idx].keytype)) {
210 deltaFiltered[idx] = 0.0;
211 for (Size ii = 0; ii < sensiKeys.size(); ++ii) {
212 gammaFiltered[idx][ii] = gammaFiltered[ii][idx] = 0.0;
213 }
214 }
215 }
216 // are all sensis zero, then skip the computation
217 bool zeroSensis = close_enough(QuantExt::detail::absMax(deltaFiltered), 0.0) &&
218 close_enough(QuantExt::detail::absMax(gammaFiltered), 0.0);
219 // compute var and write to report
220 std::vector<Real> var = zeroSensis
221 ? std::vector<Real>(p_.size(), 0.0)
222 : computeVar(omega, deltaFiltered, gammaFiltered, p_, *covarianceSalvage);
223 if (!close_enough(QuantExt::detail::absMax(var), 0.0)) {
224 report.next();
225 report.add(portfolioName);
226 report.add(rf.riskClassLabel());
227 report.add(rf.riskTypeLabel());
228 for (auto const& v : var)
229 report.add(v);
230 }
231 } // for k (risk types)
232 } // for j (risk classes)
233 } // for i (portfolios)
234 LOG("parametric var computation done.");
235 report.end();
236
237} // calculate

239std::vector<Real> ParametricVarCalculator::computeVar(const Matrix& omega, const Array& delta, const Matrix& gamma,
240 const std::vector<Real>& p,
241 const QuantExt::CovarianceSalvage& covarianceSalvage) {
242 if (method_ == "Delta") {
243 std::vector<Real> res(p.size());
244 for (Size i = 0; i < p.size(); ++i) {
245 res[i] = QuantExt::deltaVar(omega, delta, p[i], covarianceSalvage);
246 }
247 return res;
248 } else if (method_ == "DeltaGammaNormal") {
249 std::vector<Real> res(p.size());
250 for (Size i = 0; i < p.size(); ++i) {
251 res[i] = QuantExt::deltaGammaVarNormal(omega, delta, gamma, p[i], covarianceSalvage);
252 }
253 return res;
254 } else if (method_ == "MonteCarlo") {
255 QL_REQUIRE(mcSamples_ != Null<Size>(),
256 "ParametricVarCalculator::computeVar(): method MonteCarlo requires mcSamples");
257 QL_REQUIRE(mcSeed_ != Null<Size>(),
258 "ParametricVarCalculator::computeVar(): method MonteCarlo requires mcSamples");
259 return QuantExt::deltaGammaVarMc<PseudoRandom>(omega, delta, gamma, p, mcSamples_, mcSeed_, covarianceSalvage);
260 } else {
261 QL_FAIL("ParametricVarCalculator::computeVar(): method " << method_ << " not known.");
262 }
263}

265void loadCovarianceDataFromCsv(std::map<std::pair<RiskFactorKey, RiskFactorKey>, Real>& data,
266 const std::string& fileName, const char delim) {
267 LOG("Load Covariance Data from file " << fileName);
268 ore::data::CSVFileReader reader(fileName, false);
269 std::vector<std::string> dummy;
270 while (reader.next()) {
271 data[std::make_pair(*parseRiskFactorKey(reader.get(0), dummy), *parseRiskFactorKey(reader.get(1), dummy))] =
272 ore::data::parseReal(reader.get(2));
273 }
274 LOG("Read " << data.size() << " valid data lines from file " << fileName);
275}
