45ParametricVarCalculator::ParametricVarParams::Method parseParametricVarMethod(const std::string& s) {
46 static map<std::string, ParametricVarCalculator::ParametricVarParams::Method> m = {
47 {"Delta", ParametricVarCalculator::ParametricVarParams::Method::Delta},
48 {"DeltaGammaNormal", ParametricVarCalculator::ParametricVarParams::Method::DeltaGammaNormal},
49 {"MonteCarlo", ParametricVarCalculator::ParametricVarParams::Method::MonteCarlo},
50 {"Cornish-Fisher", ParametricVarCalculator::ParametricVarParams::Method::CornishFisher},
51 {"Saddlepoint", ParametricVarCalculator::ParametricVarParams::Method::Saddlepoint}
52 };
53
54 auto it = m.find(s);
55 if (it != m.end())
56 return it->second;
57 else
58 QL_FAIL("ParametricVarParams Method \"" << s << "\" not recognized");
59}

78QuantLib::Real ParametricVarCalculator::var(QuantLib::Real confidence, const bool isCall,
79 const set<pair<string, Size>>& tradeIds) {
80 Real factor = isCall ? 1.0 : -1.0;
81
82 Array delta(deltas_.size(), 0.0);
83 Matrix gamma(deltas_.size(), deltas_.size(), 0.0);
84
85 if (includeDeltaMargin_) {
86 Size counter = 0;
87 for (auto it = deltas_.begin(); it != deltas_.end(); it++)
88 delta[counter++] = factor * it->second;
89 }
90
91 if (includeGammaMargin_) {
92 Size outerIdx = 0;
93 for (auto ito = deltas_.begin(); ito != deltas_.end(); ito++) {
94 Size innerIdx = 0;
95 // Error if no diagonal element
96 gamma[outerIdx][outerIdx] = factor * gammas_.at(std::make_pair(ito->first, ito->first));
97 for (auto iti = deltas_.begin(); iti != ito; iti++) {
98 auto it = gammas_.find(std::make_pair(iti->first, ito->first));
99 if (it != gammas_.end()) {
100 gamma[innerIdx][outerIdx] = factor * it->second;
101 gamma[outerIdx][innerIdx] = factor * it->second;
102 }
103 innerIdx++;
104 }
105 outerIdx++;
106 }
107 }
108
109 if (parametricVarParams_.method == ParametricVarCalculator::ParametricVarParams::Method::Delta)
110 return QuantExt::deltaVar(omega_, delta, confidence, *covarianceSalvage_);
111 else if (parametricVarParams_.method ==
112 ParametricVarCalculator::ParametricVarParams::Method::DeltaGammaNormal)
113 return QuantExt::deltaGammaVarNormal(omega_, delta, gamma, confidence, *covarianceSalvage_);
114 else if (parametricVarParams_.method == ParametricVarCalculator::ParametricVarParams::Method::MonteCarlo) {
115 QL_REQUIRE(parametricVarParams_.samples != Null<Size>(),
116 "ParametricVarCalculator::computeVar(): method MonteCarlo requires mcSamples");
117 QL_REQUIRE(parametricVarParams_.seed != Null<Size>(),
118 "ParametricVarCalculator::computeVar(): method MonteCarlo requires mcSamples");
119 return QuantExt::deltaGammaVarMc<PseudoRandom>(omega_, delta, gamma, confidence, parametricVarParams_.samples,
120 parametricVarParams_.seed, *covarianceSalvage_);
121 } else if (parametricVarParams_.method == ParametricVarCalculator::ParametricVarParams::Method::CornishFisher)
122 return QuantExt::deltaGammaVarCornishFisher(omega_, delta, gamma, confidence, *covarianceSalvage_);
123 else if (parametricVarParams_.method == ParametricVarCalculator::ParametricVarParams::Method::Saddlepoint) {
124 Real res;
125 try {
126 res = QuantExt::deltaGammaVarSaddlepoint(omega_, delta, gamma, confidence, *covarianceSalvage_);
127 } catch (const std::exception& e) {
128 ALOG("Saddlepoint VaR computation exited with an error: " << e.what()
129 << ", falling back on Monte-Carlo");
130 res = QuantExt::deltaGammaVarMc<PseudoRandom>(omega_, delta, gamma, confidence,
131 parametricVarParams_.samples, parametricVarParams_.seed, *covarianceSalvage_);
132 }
133 return res;
134 } else
135 QL_FAIL("ParametricVarCalculator::computeVar(): method " << parametricVarParams_.method << " not known.");
136}

165void ParametricVarReport::calculate(ore::data::Report& report) {
166 // prepare report
167 report.addColumn("Portfolio", string()).addColumn("RiskClass", string()).addColumn("RiskType", string());
168 for (Size i = 0; i < p_.size(); ++i)
169 report.addColumn("Quantile_" + std::to_string(p_[i]), double(), 6);
170
171 // build portfolio filter, if given
172 bool hasFilter = false;
173 boost::regex filter;
174 if (portfolioFilter_ != "") {
175 hasFilter = true;
176 filter = boost::regex(portfolioFilter_);
177 LOG("Portfolio filter: " << portfolioFilter_);
178 }
179
180 vector<string> portfolioIds;
181 for (const auto& [portfolioId, portfolio] : tradePortfolios_)
182 portfolioIds.push_back(portfolioId);
183
184 string allStr = "(all)";
185 if (breakdown_ && tradePortfolios_.size() > 1) {
186 // Create a summary portfolio
187 portfolioIds.insert(portfolioIds.begin(), allStr);
188 set<pair<string, Size>> allTradeIds;
189 for (const auto& [portfolioId, portfolio] : tradePortfolios_) {
190 if (!hasFilter || boost::regex_match(portfolioId, filter)) {
191 for (const auto& trade : portfolio)
192 allTradeIds.insert(make_pair(trade.first, Null<Size>()));
193 }
194 }
195 tradePortfolios_[allStr] = allTradeIds;
196 }
197
198 SensitivityAggregator sensiAgg(tradePortfolios_);
199
200 // create a calculator for the VAR
201
202 Matrix covariance;
203 map<RiskFactorKey, Real> deltas;
204 map<pair<RiskFactorKey, RiskFactorKey>, Real> gammas;
205 boost::shared_ptr<QuantExt::CovarianceSalvage> covarianceSalvage =
206 boost::make_shared<QuantExt::SpectralCovarianceSalvage>();
207 bool includeGammaMargin = true;
208 bool includeDeltaMargin = true;
209 ParametricVarCalculator calculator(parametricVarParams_, covariance, deltas, gammas, covarianceSalvage,
210 includeGammaMargin, includeDeltaMargin);
211
212 // loop over risk class and type filters (index 0 == all risk types)
213 for (Size j = 0; j < (breakdown_ ? RiskFilter::numberOfRiskClasses() : 1); ++j) {
214 for (Size k = 0; k < (breakdown_ ? RiskFilter::numberOfRiskTypes() : 1); ++k) {
215 ext::shared_ptr<RiskFilter> rf = ext::make_shared<RiskFilter>(j, k);
216 sensiAgg.aggregate(*sensitivities_, rf);
217
218 for (const auto& portfolioId : portfolioIds) {
219 if (!hasFilter || portfolioId == allStr || boost::regex_match(portfolioId, filter)) {
220
221 set<SensitivityRecord> srs = sensiAgg.sensitivities(portfolioId);
222 deltas.clear();
223 gammas.clear();
224 // Populate the deltas and gammas for a parametric VAR benchmark calculation
225 sensiAgg.generateDeltaGamma(portfolioId, deltas, gammas);
226
227 std::vector<Real> var;
228 if (deltas.size() > 0) {
229 vector<RiskFactorKey> keys;
230 transform(deltas.begin(), deltas.end(), back_inserter(keys),
231 [](const pair<RiskFactorKey, Real>& kv) { return kv.first; });
232
233 // If covariance is provided use it, otherwise generate
234 if (covariance_.size() > 0) {
235 std::vector<bool> sensiKeyHasNonZeroVariance(keys.size(), false);
236
237 // build global covariance matrix
238 covariance = Matrix(keys.size(), keys.size(), 0.0);
239 Size unusedCovariance = 0;
240 for (const auto& c : covariance_) {
241 auto k1 = std::find(keys.begin(), keys.end(), c.first.first);
242 auto k2 = std::find(keys.begin(), keys.end(), c.first.second);
243 if (k1 != keys.end() && k2 != keys.end()) {
244 covariance(k1 - keys.begin(), k2 - keys.begin()) = c.second;
245 if (k1 == k2)
246 sensiKeyHasNonZeroVariance[k1 - keys.begin()] = true;
247 } else
248 ++unusedCovariance;
249 }
250 DLOG("Found " << covariance_.size() << " covariance matrix entries, " << unusedCovariance
251 << " do not match a portfolio sensitivity and will not be used.");
252 for (Size i = 0; i < sensiKeyHasNonZeroVariance.size(); ++i) {
253 if (!sensiKeyHasNonZeroVariance[i])
254 WLOG("Zero variance assigned to sensitivity key " << keys[i]);
255 }
256 } else {
257 QL_REQUIRE(benchmarkPeriod_, "No benchmark period provided to generate covariance matrix");
258 ext::shared_ptr<CovarianceCalculator> covCalculator =
259 QuantLib::ext::make_shared<CovarianceCalculator>(benchmarkPeriod_.value());
260 ext::shared_ptr<ScenarioShiftCalculator> shiftCalculator =
261 ext::make_shared<ScenarioShiftCalculator>(sensitivityConfig_, simMarketConfig_);
262 ext::shared_ptr<HistoricalSensiPnlCalculator> sensiPnlCalculator =
263 ext::make_shared<HistoricalSensiPnlCalculator>(hisScenGen_, sensitivities_);
264
265 ext::shared_ptr<NPVCube> npvCube;
266
267 sensiPnlCalculator->populateSensiShifts(npvCube, keys, shiftCalculator);
268 sensiPnlCalculator->calculateSensiPnl(srs, keys, npvCube, {}, covCalculator);
269
270 covariance = covCalculator->covariance();
271 }
272
273 // make covariance matrix positive semi-definite
274 DLOG("Covariance matrix has dimension " << keys.size() << " x " << keys.size());
275 if (!salvageCovarianceMatrix_) {
276 LOG("Covariance matrix is no salvaged, check for positive semi-definiteness");
277 SymmetricSchurDecomposition ssd(covariance);
278 Real evMin = ssd.eigenvalues().back();
279 QL_REQUIRE(evMin > 0.0 || close_enough(evMin, 0.0),
280 "ParametricVar: input covariance matrix is not positive semi-definite, smallest "
281 "eigenvalue is "
282 << evMin);
283 DLOG("Smallest eigenvalue is " << evMin);
284 covarianceSalvage = boost::make_shared<QuantExt::NoCovarianceSalvage>();
285 }
286 DLOG("Covariance matrix salvage complete.");
287 // compute var and write to report
288 for (Size i = 0; i < p_.size(); i++)
289 var.push_back(calculator.var(p_[i]));
290 } else
291 var = std::vector<Real>(p_.size(), 0.0);
292
293 if (!close_enough(QuantExt::detail::absMax(var), 0.0)) {
294 report.next();
295 report.add(portfolioId);
296 report.add(rf->riskClassLabel());
297 report.add(rf->riskTypeLabel());
298 for (auto const& v : var)
299 report.add(v);
300 }
301 }
302 }
303 sensiAgg.reset();
304 }
305 }
306}
