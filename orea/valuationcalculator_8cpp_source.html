31void NPVCalculator::init(const boost::shared_ptr<Portfolio>& portfolio, const boost::shared_ptr<SimMarket>& simMarket) {
32 DLOG("init NPVCalculator");
33 tradeCcyIndex_.resize(portfolio->size());
34 std::set<std::string> ccys;
35 for (auto const& t : portfolio->trades())
36 ccys.insert(t->npvCurrency());

42 fxRates_.resize(ccys.size());
43}

64Real NPVCalculator::npv(Size tradeIndex, const boost::shared_ptr<Trade>& trade,
65 const boost::shared_ptr<SimMarket>& simMarket) {
66 Real npv = trade->instrument()->NPV();
67 if (close_enough(npv, 0.0))
68 return npv;
69 Real fx = fxRates_[tradeCcyIndex_[tradeIndex]];
70 Real numeraire = simMarket->numeraire();
71 return npv * fx / numeraire;
72}

74void CashflowCalculator::init(const boost::shared_ptr<Portfolio>& portfolio,
75 const boost::shared_ptr<SimMarket>& simMarket) {
76 DLOG("init CashflowCalculator");
77 tradeAndLegCcyIndex_.clear();
78 std::set<std::string> ccys;
79 for (auto const& t : portfolio->trades()) {
80 tradeAndLegCcyIndex_.push_back(std::vector<Size>(t->legs().size()));
81 for (auto const& l : t->legCurrencies()) {
82 ccys.insert(l);
83 }
84 }
85 for (Size i = 0; i < portfolio->trades().size(); ++i) {
86 for (Size j = 0; j < portfolio->trades()[i]->legs().size(); ++j) {
87 tradeAndLegCcyIndex_[i][j] =
88 std::distance(ccys.begin(), ccys.find(portfolio->trades()[i]->legCurrencies()[j]));
89 }
90 }
91 ccyQuotes_.resize(ccys.size());
92 for (Size i = 0; i < ccys.size(); ++i)
93 ccyQuotes_[i] = (simMarket->fxRate(*std::next(ccys.begin(), i) + baseCcyCode_));
94 fxRates_.resize(ccys.size());
95}

102void CashflowCalculator::calculate(const boost::shared_ptr<Trade>& trade, Size tradeIndex,
103 const boost::shared_ptr<SimMarket>& simMarket,
104 boost::shared_ptr<NPVCube>& outputCube,
105 boost::shared_ptr<NPVCube>& outputCubeNettingSet, const Date& date, Size dateIndex,
106 Size sample, bool isCloseOut) {
107 if (isCloseOut)
108 return;
109
110 Real netFlow = 0;
111
112 QL_REQUIRE(dateGrid_->dates()[dateIndex] == date, "Date mixup, date is " << date << " but grid index is "
113 << dateIndex << ", grid(dateIndex) is "
114 << dateGrid_->dates()[dateIndex]);
115 // Date startDate = dateIndex == 0 ? t0Date_ : dateGrid_->dates()[dateIndex - 1];
116 Date startDate = date;
117 Date endDate = date == dateGrid_->dates().back() ? date : dateGrid_->dates()[dateIndex + 1];
118
119 bool isOption = trade->instrument()->isOption();
120 bool isExercised = false;
121 bool isPhysical = false;
122 Real longShort = 1.0;
123 if (isOption) {
124 boost::shared_ptr<data::OptionWrapper> wrapper =
125 boost::dynamic_pointer_cast<data::OptionWrapper>(trade->instrument());
126 isExercised = wrapper->isExercised();
127 longShort = wrapper->isLong() ? 1.0 : -1.0;
128 isPhysical = wrapper->isPhysicalDelivery();
129 }
130
131 try {
132 if (!isOption || (isExercised && isPhysical)) {
133 for (Size i = 0; i < trade->legs().size(); i++) {
134 const Leg& leg = trade->legs()[i];
135 Real legFlow = 0;
136 for (auto flow : leg) {
137 // Take flows in (t, t+1]
138 if (startDate < flow->date() && flow->date() <= endDate)
139 legFlow += flow->amount();
140 }
141 if (legFlow != 0) {
142 // Do FX conversion and add to netFlow
143 Real fx = fxRates_[tradeAndLegCcyIndex_[tradeIndex][i]];
144 Real direction = trade->legPayers()[i] ? -1.0 : 1.0;
145 netFlow += legFlow * direction * longShort * fx;
146 }
147 }
148 }
149 } catch (std::exception& e) {
150 ALOG("Failed to calculate cashflows for trade " << trade->id() << " : " << e.what());
151 netFlow = 0;
152 } catch (...) {
153 ALOG("Failed to calculate cashflows for trade " << trade->id() << " : Unhandled Exception");
154 netFlow = 0;
155 }
156
157 Real numeraire = simMarket->numeraire();
158
159 outputCube->set(netFlow / numeraire, tradeIndex, dateIndex, sample, index_);
160}
