42 Real NPVCalculator::npv(const boost::shared_ptr<Trade>& trade, const boost::shared_ptr<SimMarket>& simMarket) {
43  Real npv = 0;
44  try {
45  Real fx = simMarket->fxSpot(trade->npvCurrency() + baseCcyCode_)->value();
46  Real numeraire = simMarket->numeraire();
47 
48  npv = trade->instrument()->NPV() * fx / numeraire;
49 
50  } catch (std::exception& e) {
51  ALOG("Failed to price trade " << trade->id() << " : " << e.what());
52  npv = 0;
53  } catch (...) {
54  ALOG("Failed to price trade " << trade->id() << " : Unhandled Exception");
55  npv = 0;
56  }
57  return npv;
58 }

60 void CashflowCalculator::calculate(const boost::shared_ptr<Trade>& trade, Size tradeIndex,
61  const boost::shared_ptr<SimMarket>& simMarket,
62  boost::shared_ptr<NPVCube>& outputCube, const Date& date, Size dateIndex,
63  Size sample) {
64 
65  Real netFlow = 0;
66 
67  QL_REQUIRE(dateGrid_->dates()[dateIndex] == date,
68  "Date mixup, date is " << date << " but grid index is " << dateIndex);
69  // Date startDate = dateIndex == 0 ? t0Date_ : dateGrid_->dates()[dateIndex - 1];
70  Date startDate = date;
71  Date endDate = date == dateGrid_->dates().back() ? date : dateGrid_->dates()[dateIndex + 1];
72 
73  bool isOption = trade->instrument()->isOption();
74  bool isExercised = false;
75  bool isPhysical = false;
76  Real longShort = 1.0;
77  if (isOption) {
78  boost::shared_ptr<data::OptionWrapper> wrapper =
79  boost::dynamic_pointer_cast<data::OptionWrapper>(trade->instrument());
80  isExercised = wrapper->isExercised();
81  longShort = wrapper->isLong() ? 1.0 : -1.0;
82  isPhysical = wrapper->isPhysicalDelivery();
83  }
84 
85  try {
86  if (!isOption || (isExercised && isPhysical)) {
87  for (Size i = 0; i < trade->legs().size(); i++) {
88  const Leg& leg = trade->legs()[i];
89  Real legFlow = 0;
90  for (auto flow : leg) {
91  // Take flows in (t, t+1]
92  if (startDate < flow->date() && flow->date() <= endDate)
93  legFlow += flow->amount();
94  }
95  if (legFlow != 0) {
96  // Do FX conversion and add to netFlow
97  Real fx = simMarket->fxSpot(trade->legCurrencies()[i] + baseCcyCode_)->value();
98  Real direction = trade->legPayers()[i] ? -1.0 : 1.0;
99  netFlow += legFlow * direction * longShort * fx;
100  }
101  }
102  }
103  } catch (std::exception& e) {
104  ALOG("Failed to calculate cashflows for trade " << trade->id() << " : " << e.what());
105  netFlow = 0;
106  } catch (...) {
107  ALOG("Failed to calculate cashflows for trade " << trade->id() << " : Unhandled Exception");
108  netFlow = 0;
109  }
110 
111  Real numeraire = simMarket->numeraire();
112 
113  outputCube->set(netFlow / numeraire, tradeIndex, dateIndex, sample, index_);
114 }
