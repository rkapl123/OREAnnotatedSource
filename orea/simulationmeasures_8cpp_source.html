195 void test_measure(std::string measureName, Real shiftHorizon, std::string discName) {
196 
197  BOOST_TEST_MESSAGE("Testing market simulation, measure " << measureName << ", horizon " << shiftHorizon
198  << ", discretization " << discName);
199 
200  TestData d(measureName, shiftHorizon);
201 
202  CrossAssetStateProcess::discretization discretization;
203  if (discName == "exact")
204  discretization = CrossAssetStateProcess::exact;
205  else if (discName == "euler")
206  discretization = CrossAssetStateProcess::euler;
207  else {
208  QL_FAIL("discretization " << discName << " not recognized");
209  }
210 
211  // Simulation date grid
212  Date today = d.referenceDate;
213  std::vector<Period> tenorGrid;
214  if (discName == "exact")
215  tenorGrid = {1 * Years, 2 * Years, 3 * Years, 5 * Years, 7 * Years, 10 * Years};
216  else {
217  for (Size i = 1; i <= 120; ++i)
218  tenorGrid.push_back(i * Months);
219  }
220  boost::shared_ptr<DateGrid> grid = boost::make_shared<DateGrid>(tenorGrid);
221 
222  // Model
223  boost::shared_ptr<QuantExt::CrossAssetModel> model = d.ccLgm;
224 
225  // Simulation market parameters, we just need the yield curve structure here
226  boost::shared_ptr<ScenarioSimMarketParameters> simMarketConfig(new ScenarioSimMarketParameters);
227  simMarketConfig->setYieldCurveTenors("", {3 * Months, 6 * Months, 1 * Years, 2 * Years, 3 * Years, 4 * Years,
228  5 * Years, 7 * Years, 10 * Years, 12 * Years, 15 * Years, 20 * Years,
229  30 * Years, 40 * Years, 50 * Years});
230  simMarketConfig->setSimulateFXVols(false);
231  simMarketConfig->setSimulateEquityVols(false);
232 
233  simMarketConfig->baseCcy() = "EUR";
234  simMarketConfig->setDiscountCurveNames({"EUR", "USD", "GBP"});
235  simMarketConfig->setIndices({"EUR-EURIBOR-6M", "USD-LIBOR-3M", "GBP-LIBOR-6M"});
236  simMarketConfig->interpolation() = "LogLinear";
237  simMarketConfig->setSwapVolExpiries("", {6 * Months, 1 * Years, 2 * Years, 3 * Years, 5 * Years, 10 * Years});
238  simMarketConfig->setSwapVolTerms("", {1 * Years, 2 * Years, 3 * Years, 5 * Years, 7 * Years, 10 * Years});
239  simMarketConfig->setFxCcyPairs({"USDEUR", "GBPEUR"});
240 
241  boost::shared_ptr<ScenarioGeneratorData> sgd(new ScenarioGeneratorData);
242  sgd->discretization() = discretization;
243  sgd->sequenceType() = Sobol;
244  sgd->seed() = 42;
245  sgd->setGrid(grid);
246 
247  ScenarioGeneratorBuilder sgb(sgd);
248  boost::shared_ptr<ScenarioFactory> sf = boost::make_shared<SimpleScenarioFactory>();
249  boost::shared_ptr<ScenarioGenerator> sg = sgb.build(model, sf, simMarketConfig, today, d.market);
250 
251  convs();
252  auto simMarket = boost::make_shared<ScenarioSimMarket>(d.market, simMarketConfig);
253  simMarket->scenarioGenerator() = sg;
254 
255  // Basic Martingale tests
256  Size samples = 10000;
257  Real eur = 0.0, usd = 0.0, gbp = 0.0, eur2 = 0.0, usd2 = 0.0, gbp2 = 0.0;
258  Real eur3 = 0.0, usd3 = 0.0, gbp3 = 0.0;
259  int horizon = 10;
260 
261  Date d1 = grid->dates().back();
262  Date d2 = d1 + horizon * Years;
263  Real relTolerance = 0.003;
264  Real eurExpected = d.market->discountCurve("EUR")->discount(d2);
265  Real eurExpected2 = d.market->discountCurve("EUR")->discount(d1);
266  Real gbpExpected = d.market->fxRate("GBPEUR")->value() * d.market->discountCurve("GBP")->discount(d2);
267  Real gbpExpected2 = d.market->fxRate("GBPEUR")->value() * d.market->discountCurve("GBP")->discount(d1);
268  Real usdExpected = d.market->fxRate("USDEUR")->value() * d.market->discountCurve("USD")->discount(d2);
269  Real usdExpected2 = d.market->fxRate("USDEUR")->value() * d.market->discountCurve("USD")->discount(d1);
270 
271  cpu_timer timer, timer2;
272  BOOST_TEST_MESSAGE("running " << samples << " samples simulation over " << grid->dates().size() << " time steps");
273  for (Size i = 0; i < samples; i++) {
274  for (Date d : grid->dates()) {
275  timer.resume();
276  simMarket->update(d);
277  timer.stop();
278  if (d == grid->dates().back()) {
279  Real numeraire = simMarket->numeraire();
280  Real usdeurFX = simMarket->fxRate("USDEUR")->value();
281  Real gbpeurFX = simMarket->fxRate("GBPEUR")->value();
282  Real eurDiscount = simMarket->discountCurve("EUR")->discount(1.0 * horizon);
283  Real gbpDiscount = simMarket->discountCurve("GBP")->discount(1.0 * horizon);
284  ;
285  Real usdDiscount = simMarket->discountCurve("USD")->discount(1.0 * horizon);
286  ;
287  Real eurIndex =
288  simMarket->iborIndex("EUR-EURIBOR-6M")->forwardingTermStructure()->discount(1.0 * horizon);
289  Real gbpIndex =
290  simMarket->iborIndex("GBP-LIBOR-6M")->forwardingTermStructure()->discount(1.0 * horizon);
291  ;
292  Real usdIndex =
293  simMarket->iborIndex("USD-LIBOR-3M")->forwardingTermStructure()->discount(1.0 * horizon);
294  ;
295  eur += eurDiscount / numeraire;
296  gbp += gbpDiscount * gbpeurFX / numeraire;
297  usd += usdDiscount * usdeurFX / numeraire;
298  eur2 += 1.0 / numeraire;
299  gbp2 += gbpeurFX / numeraire;
300  usd2 += usdeurFX / numeraire;
301  eur3 += eurIndex / numeraire;
302  gbp3 += gbpIndex * gbpeurFX / numeraire;
303  usd3 += usdIndex * usdeurFX / numeraire;
304  }
305  }
306  }
307 
308  timer2.stop();
309 
310  eur /= samples;
311  gbp /= samples;
312  usd /= samples;
313  eur2 /= samples;
314  gbp2 /= samples;
315  usd2 /= samples;
316  eur3 /= samples;
317  gbp3 /= samples;
318  usd3 /= samples;
319 
320  Real eurDiff = fabs(eur - eurExpected) / eurExpected;
321  BOOST_CHECK_MESSAGE(eurDiff < relTolerance, "EUR 20Y Discount mismatch: " << eur << " vs " << eurExpected);
322 
323  Real gbpDiff = fabs(gbp - gbpExpected) / gbpExpected;
324  BOOST_CHECK_MESSAGE(gbpDiff < relTolerance,
325  "GBP 20Y Discount mismatch: " << gbp << " vs " << gbpExpected << " (" << gbpDiff << ")");
326 
327  Real usdDiff = fabs(usd - usdExpected) / usdExpected;
328  BOOST_CHECK_MESSAGE(usdDiff < relTolerance, "USD 20Y Discount mismatch: " << usd << " vs " << usdExpected);
329 
330  Real eur3Diff = fabs(eur3 - eurExpected) / eurExpected;
331  BOOST_CHECK_MESSAGE(eur3Diff < relTolerance, "EUR 20Y Index Discount mismatch: " << eur3 << " vs " << eurExpected);
332 
333  Real gbp3Diff = fabs(gbp3 - gbpExpected) / gbpExpected;
334  BOOST_CHECK_MESSAGE(gbp3Diff < relTolerance, "GBP 20Y Index Discount mismatch: " << gbp3 << " vs " << gbpExpected);
335 
336  Real usd3Diff = fabs(usd3 - usdExpected) / usdExpected;
337  BOOST_CHECK_MESSAGE(usd3Diff < relTolerance, "USD 20Y Index Discount mismatch: " << usd3 << " vs " << usdExpected);
338 
339  Real eur2Diff = fabs(eur2 - eurExpected2) / eurExpected2;
340  BOOST_CHECK_MESSAGE(eur2Diff < relTolerance, "EUR 10Y Discount mismatch: " << eur2 << " vs " << eurExpected2);
341 
342  Real gbp2Diff = fabs(gbp2 - gbpExpected2) / gbpExpected2;
343  BOOST_CHECK_MESSAGE(gbp2Diff < relTolerance, "GBP 10Y Discount mismatch: " << gbp2 << " vs " << gbpExpected2);
344 
345  Real usd2Diff = fabs(usd2 - usdExpected2) / usdExpected2;
346  BOOST_CHECK_MESSAGE(usd2Diff < relTolerance, "USD 10Y Discount mismatch: " << usd2 << " vs " << usdExpected2);
347 
348  BOOST_TEST_MESSAGE("CrossAssetModel via ScenarioSimMarket");
349  BOOST_TEST_MESSAGE("EUR " << QuantLib::io::iso_date(d2) << " Discount: " << eur << " vs " << eurExpected
350  << " (" << eurDiff << ")");
351  BOOST_TEST_MESSAGE("GBP " << QuantLib::io::iso_date(d2) << " Discount in EUR: " << gbp << " vs " << gbpExpected
352  << " (" << gbpDiff << ")");
353  BOOST_TEST_MESSAGE("USD " << QuantLib::io::iso_date(d2) << " Discount in EUR: " << usd << " vs " << usdExpected
354  << " (" << usdDiff << ")");
355  BOOST_TEST_MESSAGE("EUR " << QuantLib::io::iso_date(d1) << " Discount: " << eur2 << " vs " << eurExpected2
356  << " (" << eur2Diff << ")");
357  BOOST_TEST_MESSAGE("GBP " << QuantLib::io::iso_date(d1) << " Discount in EUR: " << gbp2 << " vs " << gbpExpected2
358  << " (" << gbp2Diff << ")");
359  BOOST_TEST_MESSAGE("USD " << QuantLib::io::iso_date(d1) << " Discount in EUR: " << usd2 << " vs " << usdExpected2
360  << " (" << usd2Diff << ")");
361  BOOST_TEST_MESSAGE("Simulation time " << timer.format(default_places, "%w") << ", total "
362  << timer2.format(default_places, "%w"));
363 }
