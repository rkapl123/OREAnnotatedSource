256Handle<ZeroInflationIndex> TestMarket::makeZeroInflationIndex(string index, vector<Date> dates, vector<Rate> rates,
257 boost::shared_ptr<ZeroInflationIndex> ii,
258 Handle<YieldTermStructure> yts) {
259 // build UKRPI index
260
261 boost::shared_ptr<ZeroInflationTermStructure> cpiTS;
262 // now build the helpers ...
263 vector<boost::shared_ptr<BootstrapHelper<ZeroInflationTermStructure>>> instruments;
264 for (Size i = 0; i < dates.size(); i++) {
265 Handle<Quote> quote(boost::shared_ptr<Quote>(new SimpleQuote(rates[i] / 100.0)));
266 boost::shared_ptr<BootstrapHelper<ZeroInflationTermStructure>> anInstrument(new ZeroCouponInflationSwapHelper(
267 quote, Period(2, Months), dates[i], TARGET(), ModifiedFollowing, ActualActual(ActualActual::ISDA), ii, CPI::AsIndex, yts));
268 anInstrument->unregisterWith(Settings::instance().evaluationDate());
269 instruments.push_back(anInstrument);
270 };
271 // we can use historical or first ZCIIS for this
272 // we know historical is WAY off market-implied, so use market implied flat.
273 Rate baseZeroRate = rates[0] / 100.0;
274 boost::shared_ptr<PiecewiseZeroInflationCurve<Linear>> pCPIts(
275 new PiecewiseZeroInflationCurve<Linear>(asof_, TARGET(), ActualActual(ActualActual::ISDA), Period(2, Months), ii->frequency(),
276 baseZeroRate, instruments));
277 pCPIts->recalculate();
278 cpiTS = boost::dynamic_pointer_cast<ZeroInflationTermStructure>(pCPIts);
279 cpiTS->enableExtrapolation(true);
280 cpiTS->unregisterWith(Settings::instance().evaluationDate());
281 return Handle<ZeroInflationIndex>(parseZeroInflationIndex(index, false, Handle<ZeroInflationTermStructure>(cpiTS)));
282}

284Handle<YoYInflationIndex> TestMarket::makeYoYInflationIndex(string index, vector<Date> dates, vector<Rate> rates,
285 boost::shared_ptr<YoYInflationIndex> ii,
286 Handle<YieldTermStructure> yts) {
287 // build UKRPI index
288
289 boost::shared_ptr<YoYInflationTermStructure> yoyTS;
290 // now build the helpers ...
291 vector<boost::shared_ptr<BootstrapHelper<YoYInflationTermStructure>>> instruments;
292 for (Size i = 0; i < dates.size(); i++) {
293 Handle<Quote> quote(boost::shared_ptr<Quote>(new SimpleQuote(rates[i] / 100.0)));
294 boost::shared_ptr<BootstrapHelper<YoYInflationTermStructure>> anInstrument(new YearOnYearInflationSwapHelper(
295 quote, Period(2, Months), dates[i], TARGET(), ModifiedFollowing, ActualActual(ActualActual::ISDA), ii, yts));
296 instruments.push_back(anInstrument);
297 };
298 // we can use historical or first ZCIIS for this
299 // we know historical is WAY off market-implied, so use market implied flat.
300 Rate baseZeroRate = rates[0] / 100.0;
301 boost::shared_ptr<PiecewiseYoYInflationCurve<Linear>> pYoYts(
302 new PiecewiseYoYInflationCurve<Linear>(asof_, TARGET(), ActualActual(ActualActual::ISDA), Period(2, Months), ii->frequency(),
303 ii->interpolated(), baseZeroRate, instruments));
304 pYoYts->recalculate();
305 yoyTS = boost::dynamic_pointer_cast<YoYInflationTermStructure>(pYoYts);
306 return Handle<YoYInflationIndex>(boost::make_shared<QuantExt::YoYInflationIndexWrapper>(
307 parseZeroInflationIndex(index, false), false, Handle<YoYInflationTermStructure>(pYoYts)));
308}
