64vector<boost::shared_ptr<RateHelper>>
65parRateCurveHelpers(const string& ccy, const vector<string>& parInst, const vector<Period>& parTenor,
66 const vector<Handle<Quote>>& parVal,
67 const Handle<YieldTermStructure> exDiscount = Handle<YieldTermStructure>(),
68 // fx spot to base ccy
69 const Handle<Quote>& fxSpot = Handle<Quote>(),
70 // base currency discount for xccy par instruments
71 const Handle<YieldTermStructure>& fgnDiscount = Handle<YieldTermStructure>(),
72 // pointer to the market that is being built
73 ore::data::Market* market = NULL) {
74
75 BOOST_ASSERT(parInst.size() == parTenor.size());
76 BOOST_ASSERT(parInst.size() == parVal.size());
77 boost::shared_ptr<Conventions> conventions = InstrumentConventions::instance().conventions();
78 vector<boost::shared_ptr<RateHelper>> instruments;
79 for (Size i = 0; i < parInst.size(); ++i) {
80 Handle<Quote> parRateQuote = parVal[i];
81 Period tenor = parTenor[i];
82 boost::shared_ptr<RateHelper> rateHelper;
83 if (parInst[i] == "DEP") {
84 boost::shared_ptr<Convention> conv = conventions->get(ccy + "-DEP-CONVENTIONS");
85 boost::shared_ptr<DepositConvention> depConv = boost::dynamic_pointer_cast<DepositConvention>(conv);
86 std::ostringstream o;
87 o << depConv->index() << "-" << io::short_period(tenor);
88 string indexName = o.str();
89 boost::shared_ptr<IborIndex> index = parseIborIndex(indexName);
90 rateHelper = boost::make_shared<DepositRateHelper>(parRateQuote, index);
91 } else if (parInst[i] == "FRA") {
92 boost::shared_ptr<Convention> conv = conventions->get(ccy + "-FRA-CONVENTIONS");
93 boost::shared_ptr<FraConvention> fraConv = boost::dynamic_pointer_cast<FraConvention>(conv);
94 BOOST_ASSERT(fraConv);
95 BOOST_ASSERT(tenor > fraConv->index()->tenor());
96 Period startTenor = tenor - fraConv->index()->tenor();
97 rateHelper = boost::make_shared<FraRateHelper>(parRateQuote, startTenor, fraConv->index());
98 } else if (parInst[i] == "IRS") {
99 boost::shared_ptr<Convention> conv = conventions->get(ccy + "-6M-SWAP-CONVENTIONS");
100 boost::shared_ptr<IRSwapConvention> swapConv = boost::dynamic_pointer_cast<IRSwapConvention>(conv);
101 BOOST_ASSERT(swapConv);
102 rateHelper = boost::make_shared<SwapRateHelper>(
103 parRateQuote, tenor, swapConv->fixedCalendar(), swapConv->fixedFrequency(), swapConv->fixedConvention(),
104 swapConv->fixedDayCounter(), swapConv->index(), Handle<Quote>(), 0 * Days, exDiscount);
105 } else if (parInst[i] == "OIS") {
106 boost::shared_ptr<Convention> conv = conventions->get(ccy + "-OIS-CONVENTIONS");
107 boost::shared_ptr<OisConvention> oisConv = boost::dynamic_pointer_cast<OisConvention>(conv);
108 BOOST_ASSERT(oisConv);
109 rateHelper = boost::make_shared<QuantExt::OISRateHelper>(
110 oisConv->spotLag(), tenor, parRateQuote, oisConv->index(), oisConv->fixedDayCounter(),
111 oisConv->fixedCalendar(), oisConv->paymentLag(), oisConv->eom(), oisConv->fixedFrequency(),
112 oisConv->fixedConvention(), oisConv->fixedPaymentConvention(), oisConv->rule(), exDiscount, true);
113 } else if (parInst[i] == "FXF") {
114 boost::shared_ptr<Convention> conv = conventions->get(ccy + "-FX-CONVENTIONS");
115 boost::shared_ptr<FXConvention> fxConv = boost::dynamic_pointer_cast<FXConvention>(conv);
116 BOOST_ASSERT(fxConv);
117 boost::dynamic_pointer_cast<SimpleQuote>(*parRateQuote)
118 ->setValue(0.0); // set the fwd and basis points to zero for these tests
119 bool isFxBaseCurrencyCollateralCurrency = false;
120 // the fx swap rate helper interprets the fxSpot as of the spot date, our fx spot here
121 // is as of today, therefore we set up the fx spot helper with zero settlement days
122 // and compute the tenor such that the correct maturity date is still matched
123 Date today = Settings::instance().evaluationDate();
124 Date spotDate = fxConv->advanceCalendar().advance(today, fxConv->spotDays() * Days);
125 Date endDate = fxConv->advanceCalendar().advance(spotDate, tenor);
126 rateHelper = boost::make_shared<FxSwapRateHelper>(parRateQuote, fxSpot, (endDate - today) * Days, 0,
127 NullCalendar(), Unadjusted, false,
128 isFxBaseCurrencyCollateralCurrency, fgnDiscount);
129 } else if (parInst[i] == "XBS") {
130 boost::shared_ptr<Convention> conv = conventions->get(ccy + "-XCCY-BASIS-CONVENTIONS");
131 boost::shared_ptr<CrossCcyBasisSwapConvention> basisConv =
132 boost::dynamic_pointer_cast<CrossCcyBasisSwapConvention>(conv);
133 BOOST_ASSERT(market);
134 boost::shared_ptr<IborIndex> flatIndex =
135 *market->iborIndex(basisConv->flatIndexName(), Market::defaultConfiguration);
136 boost::shared_ptr<IborIndex> spreadIndex =
137 *market->iborIndex(basisConv->spreadIndexName(), Market::defaultConfiguration);
138 BOOST_ASSERT(basisConv);
139 BOOST_ASSERT(flatIndex);
140 BOOST_ASSERT(*flatIndex->forwardingTermStructure());
141 BOOST_ASSERT(spreadIndex);
142 BOOST_ASSERT(*spreadIndex->forwardingTermStructure());
143 BOOST_ASSERT(*fgnDiscount);
144 boost::dynamic_pointer_cast<SimpleQuote>(*parRateQuote)
145 ->setValue(0.0); // set the fwd and basis points to zero for these tests
146 bool flatIsDomestic = true; // assumes fxSpot is in form 1*BaseCcy = X*Ccy
147 rateHelper = boost::make_shared<CrossCcyBasisSwapHelper>(
148 parRateQuote, fxSpot, basisConv->settlementDays(), basisConv->settlementCalendar(), tenor,
149 basisConv->rollConvention(), flatIndex, spreadIndex, fgnDiscount, exDiscount, basisConv->eom(),
150 flatIsDomestic);
151 } else {
152 BOOST_ERROR("Unrecognised par rate instrument in curve construction - " << i);
153 }
154 instruments.push_back(rateHelper);
155 }
156 return instruments;
157}

159vector<boost::shared_ptr<DefaultProbabilityHelper>>
160parRateCurveHelpers(const string& name, const vector<Period>& parTenor, const vector<Handle<Quote>>& parVal,
161 const Handle<YieldTermStructure> exDiscount = Handle<YieldTermStructure>(),
162 // pointer to the market that is being built
163 ore::data::Market* market = NULL) {
164 BOOST_ASSERT(parTenor.size() == parVal.size());
165 boost::shared_ptr<Conventions> conventions = InstrumentConventions::instance().conventions();
166 vector<boost::shared_ptr<DefaultProbabilityHelper>> instruments;
167 Date today = Settings::instance().evaluationDate();
168 for (Size i = 0; i < parTenor.size(); ++i) {
169 Handle<Quote> parRateQuote = parVal[i];
170 Period tenor = parTenor[i];
171 boost::shared_ptr<QuantExt::CdsHelper> rateHelper;
172 boost::shared_ptr<Convention> conv = conventions->get("CDS-STANDARD-CONVENTIONS");
173 boost::shared_ptr<CdsConvention> cdsConv = boost::dynamic_pointer_cast<CdsConvention>(conv);
174 BOOST_ASSERT(cdsConv);
175 BOOST_ASSERT(market);
176 Real recoveryRate = market->recoveryRate(name, Market::defaultConfiguration)->value();
177
178 BOOST_ASSERT(*exDiscount);
179 rateHelper = boost::make_shared<QuantExt::SpreadCdsHelper>(
180 parRateQuote, tenor, cdsConv->settlementDays(), cdsConv->calendar(), cdsConv->frequency(),
181 cdsConv->paymentConvention(), cdsConv->rule(), cdsConv->dayCounter(), recoveryRate, exDiscount,
182 true, QuantExt::CreditDefaultSwap::ProtectionPaymentTime::atDefault, today + cdsConv->settlementDays());
183 instruments.push_back(rateHelper);
184 }
185 return instruments;
186}

560Handle<ZeroInflationIndex> TestMarket::makeZeroInflationIndex(string index, vector<Date> dates, vector<Rate> rates,
561 boost::shared_ptr<ZeroInflationIndex> ii,
562 Handle<YieldTermStructure> yts) {
563 // build UKRPI index
564 boost::shared_ptr<ZeroInflationTermStructure> cpiTS;
565 // now build the helpers ...
566 vector<boost::shared_ptr<BootstrapHelper<ZeroInflationTermStructure>>> instruments;
567 for (Size i = 0; i < dates.size(); i++) {
568 Handle<Quote> quote(boost::shared_ptr<Quote>(new SimpleQuote(rates[i] / 100.0)));
569 boost::shared_ptr<BootstrapHelper<ZeroInflationTermStructure>> anInstrument(new ZeroCouponInflationSwapHelper(
570 quote, Period(2, Months), dates[i], TARGET(), ModifiedFollowing, ActualActual(ActualActual::ISDA), ii, CPI::AsIndex, yts));
571 anInstrument->unregisterWith(Settings::instance().evaluationDate());
572 instruments.push_back(anInstrument);
573 };
574 // we can use historical or first ZCIIS for this
575 // we know historical is WAY off market-implied, so use market implied flat.
576 Rate baseZeroRate = rates[0] / 100.0;
577 boost::shared_ptr<PiecewiseZeroInflationCurve<Linear>> pCPIts(
578 new PiecewiseZeroInflationCurve<Linear>(asof_, TARGET(), ActualActual(ActualActual::ISDA), Period(2, Months), ii->frequency(),
579 baseZeroRate, instruments));
580 pCPIts->recalculate();
581 cpiTS = boost::dynamic_pointer_cast<ZeroInflationTermStructure>(pCPIts);
582 cpiTS->enableExtrapolation(true);
583 cpiTS->unregisterWith(Settings::instance().evaluationDate());
584 return Handle<ZeroInflationIndex>(parseZeroInflationIndex(index, Handle<ZeroInflationTermStructure>(cpiTS)));
585}

1033void TestMarketParCurves::createSwaptionVolCurve(const string& name, const vector<Period>& optionTenors,
1034 const vector<Period>& swapTenors, const vector<Real>& strikeSpreads,
1035 const vector<Real>& parRates) {
1036 DayCounter dc = Actual365Fixed();
1037 Calendar cal = TARGET();
1038 BusinessDayConvention bdc = Following;
1039 swaptionVolRateHelperTenorsMap_[name] = optionTenors;
1040 swaptionVolRateHelperSwapTenorsMap_[name] = swapTenors;
1041
1042 vector<vector<Handle<Quote>>> parQuotes(optionTenors.size(), vector<Handle<Quote>>(swapTenors.size()));
1043 vector<vector<Handle<Quote>>> cubeQuotes(optionTenors.size() * swapTenors.size(),
1044 vector<Handle<Quote>>(strikeSpreads.size(), Handle<Quote>()));
1045 vector<Handle<Quote>> allQuotes(parRates.size());
1046
1047 vector<vector<Real>> shift(optionTenors.size(), vector<Real>(swapTenors.size(), 0.0));
1048 for (Size i = 0; i < optionTenors.size(); ++i) {
1049 for (Size j = 0; j < swapTenors.size(); ++j) {
1050 for (Size k = 0; k < strikeSpreads.size(); ++k) {

1053 if (close_enough(strikeSpreads[k], 0.0))
1054 parQuotes[i][j] = quote;
1055
1056 cubeQuotes[i * swapTenors.size() + j][k] = quote;
1057 allQuotes[l] = quote;
1058 }
1059 }
1060 }
1061 swaptionVolRateHelperValuesMap_[name] = allQuotes;
1062 boost::shared_ptr<SwaptionVolatilityStructure> atm(new SwaptionVolatilityMatrix(
1063 asof_, cal, bdc, optionTenors, swapTenors, parQuotes, dc, true, QuantLib::Normal, shift));
1064
1065 Handle<SwaptionVolatilityStructure> hATM(atm);
1066 Handle<SwapIndex> si = swapIndex(swapIndexBase(name));
1067 Handle<SwapIndex> ssi = swapIndex(shortSwapIndexBase(name));
1068
1069 boost::shared_ptr<SwaptionVolatilityCube> tmp(new QuantExt::SwaptionVolCube2(
1070 hATM, optionTenors, swapTenors, strikeSpreads, cubeQuotes, *si, *ssi, false, true, false));
1071 tmp->enableExtrapolation();
1072
1073 Handle<SwaptionVolatilityStructure> svp =
1074 Handle<SwaptionVolatilityStructure>(boost::make_shared<SwaptionVolCubeWithATM>(tmp));
1075
1076 swaptionCurves_[make_pair(Market::defaultConfiguration, name)] = svp;
1077}
