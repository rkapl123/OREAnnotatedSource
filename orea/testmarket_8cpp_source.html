307 Handle<ZeroInflationIndex> TestMarket::makeZeroInflationIndex(string index, vector<Date> dates, vector<Rate> rates,
308  boost::shared_ptr<ZeroInflationIndex> ii,
309  Handle<YieldTermStructure> yts) {
310  // build UKRPI index
311 
312  boost::shared_ptr<ZeroInflationTermStructure> cpiTS;
313  // now build the helpers ...
314  vector<boost::shared_ptr<BootstrapHelper<ZeroInflationTermStructure>>> instruments;
315  for (Size i = 0; i < dates.size(); i++) {
316  Handle<Quote> quote(boost::shared_ptr<Quote>(new SimpleQuote(rates[i] / 100.0)));
317  boost::shared_ptr<BootstrapHelper<ZeroInflationTermStructure>> anInstrument(new ZeroCouponInflationSwapHelper(
318  quote, Period(2, Months), dates[i], TARGET(), ModifiedFollowing, ActualActual(), ii));
319  anInstrument->unregisterWith(Settings::instance().evaluationDate());
320  instruments.push_back(anInstrument);
321  };
322  // we can use historical or first ZCIIS for this
323  // we know historical is WAY off market-implied, so use market implied flat.
324  Rate baseZeroRate = rates[0] / 100.0;
325  boost::shared_ptr<PiecewiseZeroInflationCurve<Linear>> pCPIts(
326  new PiecewiseZeroInflationCurve<Linear>(asof_, TARGET(), ActualActual(), Period(2, Months), ii->frequency(),
327  ii->interpolated(), baseZeroRate, yts, instruments));
328  pCPIts->recalculate();
329  cpiTS = boost::dynamic_pointer_cast<ZeroInflationTermStructure>(pCPIts);
330  cpiTS->enableExtrapolation(true);
331  cpiTS->unregisterWith(Settings::instance().evaluationDate());
332  return Handle<ZeroInflationIndex>(parseZeroInflationIndex(index, false, Handle<ZeroInflationTermStructure>(cpiTS)));
333 }

335 Handle<YoYInflationIndex> TestMarket::makeYoYInflationIndex(string index, vector<Date> dates, vector<Rate> rates,
336  boost::shared_ptr<YoYInflationIndex> ii,
337  Handle<YieldTermStructure> yts) {
338  // build UKRPI index
339 
340  boost::shared_ptr<YoYInflationTermStructure> yoyTS;
341  // now build the helpers ...
342  vector<boost::shared_ptr<BootstrapHelper<YoYInflationTermStructure>>> instruments;
343  for (Size i = 0; i < dates.size(); i++) {
344  Handle<Quote> quote(boost::shared_ptr<Quote>(new SimpleQuote(rates[i] / 100.0)));
345  boost::shared_ptr<BootstrapHelper<YoYInflationTermStructure>> anInstrument(new YearOnYearInflationSwapHelper(
346  quote, Period(2, Months), dates[i], TARGET(), ModifiedFollowing, ActualActual(), ii));
347  instruments.push_back(anInstrument);
348  };
349  // we can use historical or first ZCIIS for this
350  // we know historical is WAY off market-implied, so use market implied flat.
351  Rate baseZeroRate = rates[0] / 100.0;
352  boost::shared_ptr<PiecewiseYoYInflationCurve<Linear>> pYoYts(
353  new PiecewiseYoYInflationCurve<Linear>(asof_, TARGET(), ActualActual(), Period(2, Months), ii->frequency(),
354  ii->interpolated(), baseZeroRate, yts, instruments));
355  pYoYts->recalculate();
356  yoyTS = boost::dynamic_pointer_cast<YoYInflationTermStructure>(pYoYts);
357  return Handle<YoYInflationIndex>(boost::make_shared<QuantExt::YoYInflationIndexWrapper>(
