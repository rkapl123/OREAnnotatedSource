152 RiskFactorKey parseRiskFactorKey(const string& str) {
153  std::vector<string> tokens;
154  boost::split(tokens, str, boost::is_any_of("/"), boost::token_compress_on);
155  QL_REQUIRE(tokens.size() == 3, "Could not parse key " << str);
156  RiskFactorKey rfk(parseRiskFactorKeyType(tokens[0]), tokens[1], parseInteger(tokens[2]));
157  return rfk;
158 }
