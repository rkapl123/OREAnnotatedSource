34string ShiftScenarioGenerator::ScenarioDescription::keyName(RiskFactorKey key) const {
35 string keyName;
36 RiskFactorKey::KeyType keyType = key.keytype;
37 if (keyType != RiskFactorKey::KeyType::IndexCurve)
38 keyName = key.name;
39 else {
40 std::vector<string> tokens;
41 boost::split(tokens, key.name, boost::is_any_of("-"));
42 keyName = tokens[0];
43 }
44
45 boost::replace_all(keyName, "/", "\\/");
46
47 std::ostringstream o;
48 o << keyType << "/" << keyName;
49 return o.str();
50}
51ShiftScenarioGenerator::ScenarioDescription::ScenarioDescription(const string& description) {
52
53 // Expect string to be possibly three pieces delimited by ":"
54 vector<string> tokens;
55 boost::split(tokens, description, boost::is_any_of(":"));
56
57 if (tokens.size() == 1 && tokens[0] == "Base") {
58 type_ = ScenarioDescription::Type::Base;
59
60 key1_ = RiskFactorKey();
61 indexDesc1_ = "";
62
63 key2_ = RiskFactorKey();
64 indexDesc2_ = "";
65
66 } else if (tokens.size() == 2 && (tokens[0] == "Up" || tokens[0] == "Down")) {
67 type_ = tokens[0] == "Up" ? ScenarioDescription::Type::Up : ScenarioDescription::Type::Down;
68
69 auto temp = deconstructFactor(tokens[1]);
70 key1_ = temp.first;
71 indexDesc1_ = temp.second;
72
73 key2_ = RiskFactorKey();
74 indexDesc2_ = "";
75
76 } else if (tokens.size() == 3 && tokens[0] == "Cross") {
77 type_ = ScenarioDescription::Type::Cross;
78
79 auto temp = deconstructFactor(tokens[1]);
80 key1_ = temp.first;
81 indexDesc1_ = temp.second;
82
83 temp = deconstructFactor(tokens[2]);
84 key2_ = temp.first;
85 indexDesc2_ = temp.second;
86
87 } else {
88 QL_FAIL("Could not construct ScenarioDescription from string '" << description << "'");
89 }
90}

175pair<RiskFactorKey, string> deconstructFactor(const string& factor) {
176
177 // If string is empty
178 if (factor.empty()) {
179 return make_pair(RiskFactorKey(), "");
180 }
181
182 boost::escaped_list_separator<char> sep('\\', '/', '\"');
183 boost::tokenizer<boost::escaped_list_separator<char> > tokenSplit(factor, sep);
184
185 vector<string> tokens(tokenSplit.begin(), tokenSplit.end());
186
187 // first 3 tokens are the risk factor key, the remainder are the description
188 ostringstream o;
189 if (tokens.size() > 3) {
190 o << tokens[3];
191 Size i = 4;
192 while (i < tokens.size()) {
193 o << "/" << tokens[i];
194 i++;
195 }
196 }
197
198 return make_pair(RiskFactorKey(parseRiskFactorKeyType(tokens[0]), tokens[1], parseInteger(tokens[2])), o.str());
199}

211boost::shared_ptr<RiskFactorKey> parseRiskFactorKey(const string& str, vector<string>& addTokens) {
212 // Use deconstructFactor to split in to pair: [key, additional token str]
213 auto p = deconstructFactor(str);
214
215 // The additional tokens
216 boost::escaped_list_separator<char> sep('\\', '/', '\"');
217 boost::tokenizer<boost::escaped_list_separator<char> > tokenSplit(p.second, sep);
218
219 vector<string> tokens(tokenSplit.begin(), tokenSplit.end());
220 addTokens = tokens;
221
222 // Return the key value
223 return boost::make_shared<RiskFactorKey>(p.first.keytype, p.first.name, p.first.index);
224}

226void ShiftScenarioGenerator::applyShift(Size j, Real shiftSize, bool up, ShiftType shiftType,
227 const vector<Time>& tenors, const vector<Real>& values,
228 const vector<Real>& times, vector<Real>& shiftedValues, bool initialise) {
229
230 QL_REQUIRE(j < tenors.size(), "index j out of range");
231 QL_REQUIRE(times.size() == values.size(), "vector size mismatch");
232 QL_REQUIRE(shiftedValues.size() == values.size(), "shifted values vector size does not match input");
233
234 Time t1 = tenors[j];
235
236 if (initialise) {
237 for (Size i = 0; i < values.size(); ++i)
238 shiftedValues[i] = values[i];
239 }
240
241 if (tenors.size() == 1) { // single shift tenor means parallel shift
242 Real w = up ? 1.0 : -1.0;
243 for (Size k = 0; k < times.size(); k++) {
244 if (shiftType == ShiftType::Absolute)
245 shiftedValues[k] += w * shiftSize;
246 else
247 shiftedValues[k] *= (1.0 + w * shiftSize);
248 }
249 } else if (j == 0) { // first shift tenor, flat extrapolation to the left
250 Time t2 = tenors[j + 1];
251 for (Size k = 0; k < times.size(); k++) {
252 Real w = 0.0;
253 if (times[k] <= t1) // full shift
254 w = 1.0;
255 else if (times[k] <= t2) // linear interpolation in t1 < times[k] < t2
256 w = (t2 - times[k]) / (t2 - t1);
257 if (!up)
258 w *= -1.0;
259 if (shiftType == ShiftType::Absolute)
260 shiftedValues[k] += w * shiftSize;
261 else
262 shiftedValues[k] *= (1.0 + w * shiftSize);
263 }
264 } else if (j == tenors.size() - 1) { // last shift tenor, flat extrapolation to the right
265 Time t0 = tenors[j - 1];
266 for (Size k = 0; k < times.size(); k++) {
267 Real w = 0.0;
268 if (times[k] >= t0 && times[k] <= t1) // linear interpolation in t0 < times[k] < t1
269 w = (times[k] - t0) / (t1 - t0);
270 else if (times[k] > t1) // full shift
271 w = 1.0;
272 if (!up)
273 w *= -1.0;
274 if (shiftType == ShiftType::Absolute)
275 shiftedValues[k] += w * shiftSize;
276 else
277 shiftedValues[k] *= (1.0 + w * shiftSize);
278 }
279 } else { // intermediate shift tenor
280 Time t0 = tenors[j - 1];
281 Time t2 = tenors[j + 1];
282 for (Size k = 0; k < times.size(); k++) {
283 Real w = 0.0;
284 if (times[k] >= t0 && times[k] <= t1) // linear interpolation in t0 < times[k] < t1
285 w = (times[k] - t0) / (t1 - t0);
286 else if (times[k] > t1 && times[k] <= t2) // linear interpolation in t1 < times[k] < t2
287 w = (t2 - times[k]) / (t2 - t1);
288 if (!up)
289 w *= -1.0;
290 if (shiftType == ShiftType::Absolute)
291 shiftedValues[k] += w * shiftSize;
292 else
293 shiftedValues[k] *= (1.0 + w * shiftSize);
294 }
295 }
296}

298void ShiftScenarioGenerator::applyShift(Size i, Size j, Real shiftSize, bool up, ShiftType shiftType,
299 const vector<Time>& shiftX, const vector<Time>& shiftY,
300 const vector<Time>& dataX, const vector<Time>& dataY,
301 const vector<vector<Real>>& data, vector<vector<Real>>& shiftedData,
302 bool initialise) {
303 QL_REQUIRE(shiftX.size() >= 1 && shiftY.size() >= 1, "shift vector size >= 1 required");
304 QL_REQUIRE(i < shiftX.size(), "index i out of range");
305 QL_REQUIRE(j < shiftY.size(), "index j out of range");
306
307 // initialise the shifted data
308 if (initialise) {
309 for (Size k = 0; k < dataX.size(); ++k) {
310 for (Size l = 0; l < dataY.size(); ++l)
311 shiftedData[k][l] = data[k][l];
312 }
313 }
314
315 // single shift point means parallel shift
316 if (shiftX.size() == 1 && shiftY.size() == 1) {
317 Real w = up ? 1.0 : -1.0;
318 for (Size k = 0; k < dataX.size(); ++k) {
319 for (Size l = 0; l < dataY.size(); ++l) {
320 if (shiftType == ShiftType::Absolute)
321 shiftedData[k][l] += w * shiftSize;
322 else
323 shiftedData[k][l] *= (1.0 + w * shiftSize);
324 }
325 }
326 return;
327 }
328
329 Size iMax = shiftX.size() - 1;
330 Size jMax = shiftY.size() - 1;
331 Real tx = shiftX[i];
332 Real ty = shiftY[j];
333 Real tx1 = i > 0 ? shiftX[i - 1] : QL_MAX_REAL;
334 Real ty1 = j > 0 ? shiftY[j - 1] : QL_MAX_REAL;
335 Real tx2 = i < iMax ? shiftX[i + 1] : -QL_MAX_REAL;
336 Real ty2 = j < jMax ? shiftY[j + 1] : -QL_MAX_REAL;
337
338 for (Size ix = 0; ix < dataX.size(); ++ix) {
339 Real x = dataX[ix];
340 for (Size iy = 0; iy < dataY.size(); ++iy) {
341 Real y = dataY[iy];
342 Real wx = 0.0;
343 Real wy = 0.0;
344 if (x >= tx && x <= tx2 && y >= ty && y <= ty2) {
345 wx = (tx2 - x) / (tx2 - tx);
346 wy = (ty2 - y) / (ty2 - ty);
347 } else if (x >= tx && x <= tx2 && y >= ty1 && y <= ty) {
348 wx = (tx2 - x) / (tx2 - tx);
349 wy = (y - ty1) / (ty - ty1);
350 } else if (x >= tx1 && x <= tx && y >= ty1 && y <= ty) {
351 wx = (x - tx1) / (tx - tx1);
352 wy = (y - ty1) / (ty - ty1);
353 } else if (x >= tx1 && x <= tx && y >= ty && y <= ty2) {
354 wx = (x - tx1) / (tx - tx1);
355 wy = (ty2 - y) / (ty2 - ty);
356 } else if ((x <= tx && i == 0 && y < ty && j == 0) || (x <= tx && i == 0 && y >= ty && j == jMax) ||
357 (x >= tx && i == iMax && y >= ty && j == jMax) || (x >= tx && i == iMax && y < ty && j == 0)) {
358 wx = 1.0;
359 wy = 1.0;
360 } else if (((x <= tx && i == 0) || (x >= tx && i == iMax)) && y >= ty1 && y <= ty) {
361 wx = 1.0;
362 wy = (y - ty1) / (ty - ty1);
363 } else if (((x <= tx && i == 0) || (x >= tx && i == iMax)) && y >= ty && y <= ty2) {
364 wx = 1.0;
365 wy = (ty2 - y) / (ty2 - ty);
366 } else if (x >= tx1 && x <= tx && ((y < ty && j == 0) || (y >= ty && j == jMax))) {
367 wx = (x - tx1) / (tx - tx1);
368 wy = 1.0;
369 } else if (x >= tx && x <= tx2 && ((y < ty && j == 0) || (y >= ty && j == jMax))) {
370 wx = (tx2 - x) / (tx2 - tx);
371 wy = 1.0;
372 }
373 QL_REQUIRE(wx >= 0.0 && wx <= 1.0, "wx out of range");
374 QL_REQUIRE(wy >= 0.0 && wy <= 1.0, "wy out of range");
375
376 Real w = up ? 1.0 : -1.0;
377 if (shiftType == ShiftType::Absolute)
378 shiftedData[ix][iy] += w * wx * wy * shiftSize;
379 else
380 shiftedData[ix][iy] *= (1.0 + w * wx * wy * shiftSize);
381 }
382 }
