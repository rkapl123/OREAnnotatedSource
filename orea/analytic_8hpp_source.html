236template <class T> inline QuantLib::ext::shared_ptr<T> Analytic::dependentAnalytic(const std::string& key) const {
237 auto it = dependentAnalytics_.find(key);
238 QL_REQUIRE(it != dependentAnalytics_.end(), "Could not find dependent Analytic " << key);
239 boost::shared_ptr<T> analytic = boost::dynamic_pointer_cast<T>(it->second);
240 QL_REQUIRE(analytic, "Could not cast analytic for key " << key);
241 return analytic;
242}
