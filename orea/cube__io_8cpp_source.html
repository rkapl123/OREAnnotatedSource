38bool use_compression(const std::string& filename) {
39#ifdef ORE_USE_ZLIB
40 // assume compression for all filenames that do not end with csv or txt
41
42 std::string extension = boost::filesystem::path(filename).extension().string();
43 return extension != ".csv" && extension != ".txt";
44#else
45 return false;
46#endif
47}

49std::string getMetaData(const std::string& line, const std::string& tag) {
50
51 // assuming a fixed width format "# tag : <value>"
52
53 QL_REQUIRE(line.substr(0, 1) == "#",
54 "internal error: getMetaData(" << line << ", " << tag << "): line does not start with #");
55 QL_REQUIRE(line.substr(2, tag.size()) == tag,
56 "internal error: getMetaData(" << line << ", " << tag << ") failed, tag is not matched.");
57 return line.substr(15);
58}

62boost::shared_ptr<NPVCube> loadCube(const std::string& filename, const bool doublePrecision) {
63
64 // open file
65
66 bool gzip = use_compression(filename);
67 std::ifstream in1(filename, gzip ? (std::ios::binary | std::ios::in) : std::ios::in);
68 boost::iostreams::filtering_stream<boost::iostreams::input> in;
69#ifdef ORE_USE_ZLIB
70 if (gzip)
71 in.push(boost::iostreams::gzip_decompressor());
72#endif
73 in.push(in1);
74
75 // read meta data
76
77 std::string line;
78
79 std::getline(in, line);
80 QuantLib::Date asof = ore::data::parseDate(getMetaData(line, "asof"));
81 std::getline(in, line);
82 Size numIds = ore::data::parseInteger(getMetaData(line, "numIds"));
83 std::getline(in, line);
84 Size numDates = ore::data::parseInteger(getMetaData(line, "numDates"));
85 std::getline(in, line);
86 Size samples = ore::data::parseInteger(getMetaData(line, "samples"));
87 std::getline(in, line);
88 Size depth = ore::data::parseInteger(getMetaData(line, "depth"));
89
90 std::getline(in, line);
91 getMetaData(line, "dates");
92 std::vector<QuantLib::Date> dates;
93 for (Size i = 0; i < numDates; ++i) {
94 std::getline(in, line);
95 dates.push_back(ore::data::parseDate(line.substr(2)));
96 }
97
98 std::getline(in, line);
99 getMetaData(line, "ids");
100 std::set<std::string> ids;
101 for (Size i = 0; i < numIds; ++i) {
102 std::getline(in, line);
103 ids.insert(line.substr(2));
104 }
105
106 boost::shared_ptr<NPVCube> result;
107 if (doublePrecision && depth <= 1) {
108 result = boost::make_shared<DoublePrecisionInMemoryCube>(asof, ids, dates, samples, 0.0);
109 } else if (doublePrecision && depth > 1) {
110 result = boost::make_shared<DoublePrecisionInMemoryCubeN>(asof, ids, dates, samples, depth, 0.0);
111 } else if (!doublePrecision && depth <= 1) {
112 result = boost::make_shared<SinglePrecisionInMemoryCube>(asof, ids, dates, samples, 0.0f);
113 } else if (!doublePrecision && depth > 1) {
114 result = boost::make_shared<SinglePrecisionInMemoryCubeN>(asof, ids, dates, samples, depth, 0.0f);
115 }
116
117 std::getline(in, line); // header line for data
118
119 vector<string> tokens;
120 Size nData = 0;
121 while (!in.eof()) {
122 std::getline(in, line);
123 if (line.empty())
124 continue;
125 boost::split(tokens, line, [](char c) { return c == ','; });
126 QL_REQUIRE(tokens.size() == 5, "loadCube(): invalid data line '" << line << "', expected 5 tokens");
127 Size id = ore::data::parseInteger(tokens[0]);
128 Size date = ore::data::parseInteger(tokens[1]);
129 Size sample = ore::data::parseInteger(tokens[2]);
130 Size depth = ore::data::parseInteger(tokens[3]);
131 double value = ore::data::parseReal(tokens[4]);
132 if (date == 0)
133 result->setT0(value, id, depth);
134 else
135 result->set(value, id, date - 1, sample, depth);
136 ++nData;
137 }
138
139 LOG("loaded cube from " << filename << ": asof = " << asof << ", dim = " << numIds << " x " << numDates << " x "
140 << samples << " x " << depth << ", " << nData << " data lines read.");
141
142 return result;
143}

145void saveCube(const std::string& filename, const NPVCube& cube, const bool doublePrecision) {
146
147 // open file
148
149 bool gzip = use_compression(filename);
150 std::ofstream out1(filename, gzip ? (std::ios::binary | std::ios::out) : std::ios::out);
151 boost::iostreams::filtering_stream<boost::iostreams::output> out;
152#ifdef ORE_USE_ZLIB
153 if (gzip)
154 out.push(boost::iostreams::gzip_compressor(/*boost::iostreams::gzip_params(9)*/));
155#endif
156 out.push(out1);
157
158 // write meta data (tag width is hardcoded and used in getMetaData())
159
160 out << "# asof : " << ore::data::to_string(cube.asof()) << "\n";
161 out << "# numIds : " << std::to_string(cube.numIds()) << "\n";
162 out << "# numDates : " << std::to_string(cube.numDates()) << "\n";
163 out << "# samples : " << ore::data::to_string(cube.samples()) << "\n";
164 out << "# depth : " << ore::data::to_string(cube.depth()) << "\n";
165 out << "# dates : \n";
166 for (auto const& d : cube.dates())
167 out << "# " << ore::data::to_string(d) << "\n";
168
169 out << "# ids : \n";
170 std::map<Size, std::string> ids;
171 for (auto const& d : cube.idsAndIndexes()) {
172 ids[d.second] = d.first;
173 }
174 for (auto const& d : ids) {
175 out << "# " << d.second << "\n";
176 }
177
178 // set precision
179
180 out << std::setprecision(doublePrecision ? std::numeric_limits<double>::max_digits10
181 : std::numeric_limits<float>::max_digits10);
182
183 // write cube data
184
185 out << "#id,date,sample,depth,value\n";
186 for (Size i = 0; i < cube.numIds(); ++i) {
187 out << i << ",0,0,0," << cube.getT0(i) << "\n";
188 for (Size j = 0; j < cube.numDates(); ++j) {
189 for (Size k = 0; k < cube.samples(); ++k) {
190 for (Size d = 0; d < cube.depth(); ++d) {
191 double value = cube.get(i, j, k, d);
192 if (value != 0.0)
193 out << i << "," << (j + 1) << "," << k << "," << d << "," << value << "\n";
194 }
195 }
196 }
197 }

200boost::shared_ptr<AggregationScenarioData> loadAggregationScenarioData(const std::string& filename) {
201
202 // open file
203
204 bool gzip = use_compression(filename);
205 std::ifstream in1(filename, gzip ? (std::ios::binary | std::ios::in) : std::ios::in);
206 boost::iostreams::filtering_stream<boost::iostreams::input> in;
207#ifdef ORE_USE_ZLIB
208 if (gzip)
209 in.push(boost::iostreams::gzip_decompressor());
210#endif
211 in.push(in1);
212
213 // read meta data
214
215 std::string line;
216
217 std::getline(in, line);
218 Size dimDates = ore::data::parseInteger(getMetaData(line, "dimDates"));
219 std::getline(in, line);
220 Size dimSamples = ore::data::parseInteger(getMetaData(line, "dimSamples"));
221
222 vector<string> tokens;
223
224 std::getline(in, line);
225 Size numKeys = ore::data::parseInteger(getMetaData(line, "keys"));
226 std::vector<std::pair<AggregationScenarioDataType, std::string>> keys;
227 for (Size i = 0; i < numKeys; ++i) {
228 std::getline(in, line);
229 boost::split(tokens, line.substr(2), [](char c) { return c == ','; });
230 QL_REQUIRE(tokens.size() == 2,
231 "loadAggregationScenarioData(): invalid data line '" << line << "', expected 2 tokens");
232 keys.push_back(std::make_pair(AggregationScenarioDataType(ore::data::parseInteger(tokens[0])), tokens[1]));
233 }
234
235 std::getline(in, line); // header line for data
236
237 boost::shared_ptr<InMemoryAggregationScenarioData> result =
238 boost::make_shared<InMemoryAggregationScenarioData>(dimDates, dimSamples);
239
240 std::getline(in, line); // header line for data
241
242 Size nData = 0;
243 while (!in.eof()) {
244 std::getline(in, line);
245 if (line.empty())
246 continue;
247 boost::split(tokens, line, [](char c) { return c == ','; });
248 QL_REQUIRE(tokens.size() == 4,
249 "loadAggregationScenarioData(): invalid data line '" << line << "', expected 4 tokens");
250 Size date = ore::data::parseInteger(tokens[0]);
251 Size sample = ore::data::parseInteger(tokens[1]);
252 Size key = ore::data::parseInteger(tokens[2]);
253 double value = ore::data::parseReal(tokens[3]);
254 QL_REQUIRE(key < keys.size(), "loadAggregationScenarioData(): invalid data line '" << line << "', key (" << key
255 << ") is out of range 0..."
256 << (keys.size() - 1));
257 result->set(date - 1, sample, value, keys[key].first, keys[key].second);
258 ++nData;
259 }
260
261 LOG("loaded aggregation scenario data from " << filename << ": dimDates = " << dimDates
262 << ", dimSamples = " << dimSamples << ", keys = " << keys.size()
263 << ", " << nData << " data lines read.");
264
265 return result;
266}

268void saveAggregationScenarioData(const std::string& filename, const AggregationScenarioData& cube) {
269
270 // open file
271
272 bool gzip = use_compression(filename);
273 std::ofstream out1(filename, gzip ? (std::ios::binary | std::ios::out) : std::ios::out);
274 boost::iostreams::filtering_stream<boost::iostreams::output> out;
275#ifdef ORE_USE_ZLIB
276 if (gzip)
277 out.push(boost::iostreams::gzip_compressor(/*boost::iostreams::gzip_params(9)*/));
278#endif
279 out.push(out1);
280
281 // write meta data (tag width is hardcoded and used in getMetaData())
282
283 out << "# dimDates : " << std::to_string(cube.dimDates()) << "\n";
284 out << "# dimSamples : " << std::to_string(cube.dimSamples()) << "\n";
285
286 auto keys = cube.keys();
287
288 out << "# keys : " << std::to_string(keys.size()) << "\n";
289 for (auto const& k : keys) {
290 out << "# " << (unsigned int)k.first << "," << k.second << "\n";
291 }
292
293 // set precision
294
295 out << std::setprecision(std::numeric_limits<double>::max_digits10);
296
297 // write data
298
299 out << "#date,sample,key,value\n";
300 for (Size i = 0; i < cube.dimDates(); ++i) {
301 for (Size j = 0; j < cube.dimSamples(); ++j) {
302 for (Size k = 0; k < keys.size(); ++k) {
303 out << (i + 1) << "," << j << "," << k << "," << cube.get(i, j, keys[k].first, keys[k].second) << "\n";
304 }
305 }
306 }
