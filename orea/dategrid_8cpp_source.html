36 DateGrid::DateGrid(const string& grid, const QuantLib::Calendar& gridCalendar, const QuantLib::DayCounter& dayCounter) {
37 
38  if (grid == "ALPHA") {
39  // ALPHA is
40  // quarterly up to 10Y,
41  // annual up to 30Y,
42  // quinquennial up to 100Y
43  for (Size i = 1; i < 40; i++) { // 3M up to 39*3M = 117M = 9Y9M
44  Period p(i * 3, Months);
45  p.normalize();
46  tenors_.push_back(p);
47  }
48  for (Size i = 10; i < 30; i++) // 10Y up to 29Y
49  tenors_.push_back(Period(i, Years));
50  for (Size i = 30; i < 105; i += 5) // 30Y up to 100Y
51  tenors_.push_back(Period(i, Years));
52  } else if (grid == "BETA") {
53  // BETA is
54  // monthly up to 10Y,
55  // quarterly up to 20Y,
56  // annually up to 50Y,
57  // quinquennial up to 100Y
58  for (Size i = 1; i < 119; i++) {
59  Period p = i * Months;
60  p.normalize();
61  tenors_.push_back(p);
62  }
63  for (Size i = 40; i < 80; i++) {
64  Period p = i * 3 * Months;
65  p.normalize();
66  tenors_.push_back(p);
67  }
68  for (Size i = 20; i < 50; i++)
69  tenors_.push_back(i * Years);
70  for (Size i = 50; i <= 100; i += 5)
71  tenors_.push_back(i * Years);
72  } else { // uniform grid of format "numPillars,spacing" (e.g. 40,1M)
73  vector<string> tokens;
74  boost::split(tokens, grid, boost::is_any_of(","));
75  if (tokens.size() <= 2) {
76  // uniform grid of format "numPillars,spacing" (e.g. 40,1M)
77  Period gridTenor = 1 * Years; // default
78  Size gridSize = atoi(tokens[0].c_str());
79  QL_REQUIRE(gridSize > 0, "Invalid DateGrid string " << grid);
80  if (tokens.size() == 2)
81  gridTenor = data::parsePeriod(tokens[1]);
82  if (gridTenor == Period(1, Days)) {
83  // we have a daily grid. Period and Calendar are not consistant with
84  // working & actual days, so we set the tenor grid
85  Date today = Settings::instance().evaluationDate();
86  Date d = today;
87  for (Size i = 0; i < gridSize; i++) {
88  d = gridCalendar.advance(d, Period(1, Days), Following); // next working day
89  Size n = d - today;
90  tenors_.push_back(Period(n, Days));
91  }
92  } else {
93  for (Size i = 0; i < gridSize; i++)
94  tenors_.push_back((i + 1) * gridTenor);
95  }
96  } else {
97  // New style : 1D,2D,1W,2W,3Y,5Y,....
98  for (Size i = 0; i < tokens.size(); i++)
99  tenors_.push_back(data::parsePeriod(tokens[i]));
100  }
101  }
102  buildDates(gridCalendar, dayCounter);
103 }

114 DateGrid::DateGrid(const vector<Date>& dates, const DayCounter& dayCounter) : dates_(dates) {
115  QL_REQUIRE(!dates_.empty(), "Construction of DateGrid requires a non-empty vector of dates");
116  QL_REQUIRE(is_sorted(dates_.begin(), dates_.end()),
117  "Construction of DateGrid requires a sorted vector of unique dates");
118  Date today = Settings::instance().evaluationDate();
119  QL_REQUIRE(today < dates_.front(),
120  "Construction of DateGrid requires first element to be strictly greater than today");
121 
122  // Populate the tenors, times and timegrid
123  tenors_.resize(dates_.size());
124  times_.resize(dates_.size());
125  for (Size i = 0; i < dates_.size(); i++) {
126  tenors_[i] = (dates_[i] - today) * Days;
127  times_[i] = dayCounter.yearFraction(today, dates_[i]);
128  }
129  timeGrid_ = TimeGrid(times_.begin(), times_.end());
130 
131  // Log the date grid
132  log();
133 }

135 void DateGrid::buildDates(const QuantLib::Calendar& cal, const QuantLib::DayCounter& dc) {
136  // build dates from tenors
137  // this is called by both constructors
138  dates_.resize(tenors_.size());
139  Date today = Settings::instance().evaluationDate();
140  for (Size i = 0; i < tenors_.size(); i++) {
141  if (tenors_[i].units() == Days)
142  dates_[i] = cal.adjust(today + tenors_[i]);
143  else
144  dates_[i] = cal.advance(today, tenors_[i], Following, true);
145  if (i > 0) {
146  QL_REQUIRE(dates_[i] >= dates_[i - 1], "DateGrid::buildDates(): tenors must be monotonic");
147  if (dates_[i] == dates_[i - 1]) {
148  dates_.erase(std::next(dates_.begin(), i));
149  tenors_.erase(std::next(tenors_.begin(), i));
150  --i;
151  }
152  }
153  }
154 
155  // Build times
156  times_.resize(dates_.size());
157  for (Size i = 0; i < dates_.size(); i++)
158  times_[i] = dc.yearFraction(today, dates_[i]);
159 
160  timeGrid_ = TimeGrid(times_.begin(), times_.end());
161 
162  // Log the date grid
163  log();
164 }

172 void DateGrid::truncate(const Date& d, bool overrun) {
173  if (d >= dates_.back())
174  return; // no need for any truncation
175  DLOG("Truncating DateGrid beyond " << QuantLib::io::iso_date(d));
176  vector<Date>::iterator it = std::upper_bound(dates_.begin(), dates_.end(), d);
177  if (overrun)
178  ++it;
179  dates_.erase(it, dates_.end());
180  tenors_.resize(dates_.size());
181  times_.resize(dates_.size());
182  timeGrid_ = TimeGrid(times_.begin(), times_.end());
183  DLOG("DateGrid size now " << dates_.size());
184 }

186 void DateGrid::truncate(Size len) {
187  // Truncate grid up length len
188  if (dates_.size() > len) {
189  DLOG("Truncating DateGrid, removing elements " << dates_[len] << " to " << dates_.back());
190  dates_.resize(len);
191  tenors_.resize(len);
192  times_.resize(len);
193  timeGrid_ = TimeGrid(times_.begin(), times_.end());
194  DLOG("DateGrid size now " << dates_.size());
195  }
