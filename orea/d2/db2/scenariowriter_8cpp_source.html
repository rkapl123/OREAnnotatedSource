65 void ScenarioWriter::writeScenario(boost::shared_ptr<Scenario>& s, const bool writeHeader) {
66  if (fp_) {
67  const Date d = s->asof();
68  // take a copy of the keys here to ensure the order is preseved
69  keys_ = s->keys();
70  std::sort(keys_.begin(), keys_.end());
71  if (writeHeader) {
72  QL_REQUIRE(keys_.size() > 0, "No keys in scenario");
73  fprintf(fp_, "Date%cScenario%cNumeraire%c%s", sep_, sep_, sep_, to_string(keys_[0]).c_str());
74  for (Size i = 1; i < keys_.size(); i++)
75  fprintf(fp_, "%c%s", sep_, to_string(keys_[i]).c_str());
76  fprintf(fp_, "\n");
77 
78  // set the first date, this will bump i_ to 1 below
79  firstDate_ = d;
80  }
81  if (d == firstDate_)
82  i_++;
83 
84  fprintf(fp_, "%s%c%zu%c%.8f", to_string(d).c_str(), sep_, i_, sep_, s->getNumeraire());
85  for (auto k : keys_)
86  fprintf(fp_, "%c%.8f", sep_, s->get(k));
87  fprintf(fp_, "\n");
88  fflush(fp_);
89  }
90 }
