84 void SensitivityCube::initialise() {
85  QL_REQUIRE(scenarioDescriptions_[0].type() == ShiftScenarioDescription::Type::Base,
86  "Expected the first scenario in the sensitivity cube to be of type 'Base'");
87 
88  // Populate the trade ID lookup map
89  for (Size i = 0; i < cube_->numIds(); i++) {
90  tradeIdx_[cube_->ids()[i]] = i;
91  }
92 
93  // Populate the scenario lookup map
94  crossPair factorPair;
95  std::map<crossPair, Size> crossFactors;
96  for (Size i = 0; i < scenarioDescriptions_.size(); i++) {
97  auto des = scenarioDescriptions_[i];
98  FactorData fd;
99  // Don't info add for base - missing from maps
100  if (des.type() != ShiftScenarioDescription::Type::Base) {
101  fd.index = i;
102  fd.shiftSize = shiftSize(des.key1());
103  fd.factorDesc = deconstructFactor(des.factor1()).second;
104  }
105  scenarioIdx_[des] = i;
106 
107  // Populate factors_ = list of factors for which we can calculate a delta/gamma
108  switch (des.type()) {
109  case ShiftScenarioDescription::Type::Up:
110  QL_REQUIRE(upFactors_.left.count(des.key1()) == 0, "Cannot have multiple up factors with "
111  "the same risk factor key["
112  << des.key1() << "]");
113  factors_.insert(des.key1());
114  upFactors_.insert(bm_type::value_type(des.key1(), fd));
115  break;
116  case ShiftScenarioDescription::Type::Down:
117  QL_REQUIRE(downFactors_.count(des.key1()) == 0, "Cannot have multiple down factors with "
118  "the same risk factor key ["
119  << des.key1() << "]");
120  downFactors_[des.key1()] = fd;
121  break;
122  case ShiftScenarioDescription::Type::Cross:
123  factorPair = make_pair(des.key1(), des.key2());
124  QL_REQUIRE(crossFactors.count(factorPair) == 0, "Cannot have multiple cross factors with "
125  "the same risk factor key pair ["
126  << des.key1() << ", " << des.key2() << "]");
127  crossFactors[factorPair] = i;
128  break;
129  default:
130  // Do nothing
131  break;
132  }
133  }
134 
135  // Add each cross factor to crossFactors with index of the two contributing factors
136  for (auto cf : crossFactors) {
137  FactorData id_1 = index(cf.first.first, upFactors_.left);
138  FactorData id_2 = index(cf.first.second, upFactors_.left);
139  crossFactors_[cf.first] = make_tuple(id_1, id_2, cf.second);
140  }
141 
142  // Log warnings if each factor does not have a shift size entry and that it is not a Null<Real>()
143  if (upFactors_.size() != shiftSizes_.size()) {
144  WLOG("The number of 'Up' shifts (" << upFactors_.size() << ") does not equal "
145  << "the number of shift sizes (" << shiftSizes_.size() << ") supplied");
146  }
147 
148  for (auto const& kv : upFactors_.left) {
149  auto it = shiftSizes_.find(kv.first);
150  if (it == shiftSizes_.end()) {
151  WLOG("No entry for risk factor " << kv.first << " in shift sizes.");
152  }
153  if (it->second == Null<Real>()) {
154  WLOG("The shift size for risk factor " << kv.first << " is not valid.")
155  }
156  }
157 }

213 Real SensitivityCube::delta(const string& tradeId, const RiskFactorKey& riskFactorKey) const {
214  Size scenarioIdx = index(riskFactorKey, upFactors_.left).index;
215  Size tradeIdx = cube_->getTradeIndex(tradeId);
216  if (!twoSidedDelta(riskFactorKey.keytype)) {
217  return delta(tradeIdx, scenarioIdx);
218  } else {
219  Size downIdx = index(riskFactorKey, downFactors_).index;
220  return delta(tradeIdx, scenarioIdx, downIdx);

224 Real SensitivityCube::gamma(Size id, Size upScenarioIdx, Size downScenarioIdx) const {
225 
226  Real baseNpv = cube_->getT0(id, 0);
227  Real upNpv = cube_->get(id, upScenarioIdx);

241 Real SensitivityCube::crossGamma(Size id, Size upIdx_1, Size upIdx_2, Size crossIdx) const {
242  // Approximate f_{xy}|(x,y) by
243  // ([f_{x}|(x,y + dy)] - [f_{x}|(x,y)]) / dy
244  // ([f(x + dx,y + dy) - f(x, y + dy)] - [f(x + dx,y) - f(x,y)]) / (dx dy)
245  Real baseNpv = cube_->getT0(id, 0);
246  Real upNpv_1 = cube_->get(id, upIdx_1);
247  Real upNpv_2 = cube_->get(id, upIdx_2);
248  Real crossNpv = cube_->get(id, crossIdx);
249 
250  return crossNpv - upNpv_1 - upNpv_2 + baseNpv;
251 }

263 Real SensitivityCube::crossGamma(const string& tradeId, const crossPair& riskFactorKeyPair) const {
264  FactorData upFd_1, upFd_2;
265  Size upIdx_1, upIdx_2, crossIdx, tradeIdx;
266  std::tie(upFd_1, upFd_2, crossIdx) = index(riskFactorKeyPair, crossFactors_);
267  upIdx_1 = upFd_1.index;
268  upIdx_2 = upFd_2.index;
269  tradeIdx = cube_->getTradeIndex(tradeId);
270 
271  return crossGamma(tradeIdx, upIdx_1, upIdx_2, crossIdx);
272 }
