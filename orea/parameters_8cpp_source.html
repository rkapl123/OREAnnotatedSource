57 void Parameters::fromXML(XMLNode* node) {
58  XMLUtils::checkNode(node, "ORE");
59 
60  XMLNode* setupNode = XMLUtils::getChildNode(node, "Setup");
61  QL_REQUIRE(setupNode, "node Setup not found in parameter file");
62  map<string, string> setupMap;
63  for (XMLNode* child = XMLUtils::getChildNode(setupNode); child; child = XMLUtils::getNextSibling(child)) {
64  string key = XMLUtils::getAttribute(child, "name");
65  string value = XMLUtils::getNodeValue(child);
66  setupMap[key] = value;
67  }
68  data_["setup"] = setupMap;
69 
70  XMLNode* marketsNode = XMLUtils::getChildNode(node, "Markets");
71  if (marketsNode) {
72  map<string, string> marketsMap;
73  for (XMLNode* child = XMLUtils::getChildNode(marketsNode); child; child = XMLUtils::getNextSibling(child)) {
74  string key = XMLUtils::getAttribute(child, "name");
75  string value = XMLUtils::getNodeValue(child);
76  marketsMap[key] = value;
77  }
78  data_["markets"] = marketsMap;
79  }
80 
81  XMLNode* analyticsNode = XMLUtils::getChildNode(node, "Analytics");
82  if (analyticsNode) {
83  for (XMLNode* child = XMLUtils::getChildNode(analyticsNode); child; child = XMLUtils::getNextSibling(child)) {
84  string groupName = XMLUtils::getAttribute(child, "type");
85  map<string, string> analyticsMap;
86  for (XMLNode* paramNode = XMLUtils::getChildNode(child); paramNode;
87  paramNode = XMLUtils::getNextSibling(paramNode)) {
88  string key = XMLUtils::getAttribute(paramNode, "name");
89  string value = XMLUtils::getNodeValue(paramNode);
90  analyticsMap[key] = value;
91  }
92  data_[groupName] = analyticsMap;
93  }
94  }
95 }
