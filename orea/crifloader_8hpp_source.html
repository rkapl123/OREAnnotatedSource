37/*! A class for loading CRIF records. The records are aggregated and stored in a
38 SimmNetSensitivities object so that they can later be used in a SIMM calculation
39*/
40class CrifLoader {
41public:
42 /*! Constructor
43 We set the trade ID to an empty string if we are going to be netting at portfolio level.
44 This is the default. To override this the flag \p keepTradeId may be set to true.
45 */
46 CrifLoader(const boost::shared_ptr<SimmConfiguration>& configuration,
47 std::map<QuantLib::Size, std::set<std::string>> additionalHeaders = std::map<QuantLib::Size, std::set<std::string>>(),
48 bool updateMapper = false, bool aggregateTrades = true)
49 : configuration_(configuration), additionalHeaders_(additionalHeaders),
50 updateMapper_(updateMapper), aggregateTrades_(aggregateTrades) {}
51
52 /*! Destructor */
53 virtual ~CrifLoader() {}
54
55 //! Add a single CRIF record
56 virtual void add(CrifRecord crifRecord, const bool onDiffAmountCcy = false);
57
58 /*! Load CRIF records from a CRIF file.
59
60 The Risk Data Standards document outlines what character should be used to separate columns,
61 \p delim, and end lines, \p eol. These parameters have the correct defaults here but can be changed.
62
63 The \p quoteChar allows one specify a character that can be used to enclose strings in the CRIF file.
64 If this character is not `\0`, then an attempt is made to strip this character from the each column.
65 */
66 void loadFromFile(const std::string& fileName, char eol = '\n', char delim = '\t', char quoteChar = '\0',
67 char escapeChar = '\\');
68 //! Load CRIF records in string format
69 void loadFromString(const std::string& csvBuffer, char eol = '\n', char delim = '\t', char quoteChar = '\0',
70 char escapeChar = '\\');
71 //! Core CRIF loader from generic istream
72 void loadFromStream(std::istream& stream, char eol = '\n', char delim = '\t', char quoteChar = '\0',
73 char escapeChar = '\\');
74
75 //! Return the netted CRIF records for use in a SIMM calculation
76 const SimmNetSensitivities netRecords(const bool includeSimmParams = false) const;
77
78 //! Return the SIMM parameters for use in calculating additional margin in SIMM
79 const SimmNetSensitivities& simmParameters() const { return simmParameters_; }
80
81 const bool hasCrifRecords() const { return !crifRecords_.empty(); }
82 const bool hasSimmParameters() const { return !simmParameters_.empty(); }
83
84 void setCrifRecords(const SimmNetSensitivities& crifRecords) { crifRecords_ = crifRecords; }
85 void setSimmParameters(const SimmNetSensitivities& simmParameters) { simmParameters_ = simmParameters; }
86
87 //! Give back the set of portfolio IDs that have been loaded
88 const std::set<std::string>& portfolioIds() const;
89 const std::set<ore::data::NettingSetDetails>& nettingSetDetails() const;
90
91 //! Reset loader to initial state
92 void clear();
93
94 //! Map giving required CRIF file headers and their allowable alternatives
95 static std::map<QuantLib::Size, std::set<std::string>> requiredHeaders;
96
97 //! Map giving optional CRIF file headers and their allowable alternatives
98 static std::map<QuantLib::Size, std::set<std::string>> optionalHeaders;
99
100 //! Check if netting set details are used anywhere, instead of just the netting set ID
101 bool hasNettingSetDetails() const;
102
103 //! Aggregate all existing records
104 void aggregate();
105
106 //! For each CRIF record in netRecords_, checks if amountCurrency and amount are
107 //! defined and uses these to populate the record's amountUsd
108 void fillAmountUsd(const boost::shared_ptr<ore::data::Market> market);
109
110 //! SIMM configuration getter
111 const boost::shared_ptr<SimmConfiguration>& simmConfiguration() { return configuration_; }
112
113protected:
114 //! Simm configuration that is used during loading of CRIF records
115 boost::shared_ptr<SimmConfiguration> configuration_;
116
117 //! Defines accepted column headers, beyond required and optional headers, see crifloader.cpp
118 std::map<QuantLib::Size, std::set<std::string>> additionalHeaders_;
119
120 /*! If true, the SIMM configuration's bucket mapper is updated during the
121 CRIF loading with the mapping from SIMM qualifier to SIMM bucket. This is
122 useful when consuming CRIF files from elsewhere in that it allows for
123 using the mapping that is already present in the external file.
124 */
125 bool updateMapper_;
126
127 /*! If true, aggregate over trade ids */
128 bool aggregateTrades_;
129
130 //! Netted CRIF records that can subsequently be used in a SIMM calculation
131 SimmNetSensitivities crifRecords_;
132
133 //! SIMM parameters for additional margin, provided in the same format as CRIF records
134 SimmNetSensitivities simmParameters_;
135
136 //! Set of portfolio IDs that have been loaded
137 std::set<std::string> portfolioIds_;
138 std::set<ore::data::NettingSetDetails> nettingSetDetails_;
139
140 /*! Internal map from known index of CRIF record member to file column
141 For example, give trade ID an index of 0 and find the column index of
142 trade ID in the CRIF file e.g. n. The map entry would be [0, n]
143 */
144 std::map<QuantLib::Size, QuantLib::Size> columnIndex_;
145
146 //! Process the elements of a header line of a CRIF file
147 virtual void processHeader(const std::vector<std::string>& headers);
148
149 /*! Process a line of a CRIF file and return true if valid line
150 or false if an invalid line
151 */
152 virtual bool process(const std::vector<std::string>& entries, QuantLib::Size maxIndex, QuantLib::Size currentLine);
153};
