99void RegressionDynamicInitialMarginCalculator::build() {
100 LOG("DIM Analysis by polynomial regression");
101
102 map<string, Real> currentDim = unscaledCurrentDIM();
103
104 Size stopDatesLoop = datesLoopSize_;
105 Size samples = cube_->samples();
106
107 Size polynomOrder = regressionOrder_;
108 LOG("DIM regression polynom order = " << regressionOrder_);
109 LsmBasisSystem::PolynomialType polynomType = LsmBasisSystem::Monomial;
110 Size regressionDimension = regressors_.empty() ? 1 : regressors_.size();
111 LOG("DIM regression dimension = " << regressionDimension);
112#if QL_HEX_VERSION > 0x01150000
113 std::vector<ext::function<Real(Array)>> v(
114 LsmBasisSystem::multiPathBasisSystem(regressionDimension, polynomOrder, polynomType));
115#else // QL 1.14 and below
116 std::vector<boost::function1<Real, Array>> v(
117 LsmBasisSystem::multiPathBasisSystem(regressionDimension, polynomOrder, polynomType));
118#endif
119 Real confidenceLevel = QuantLib::InverseCumulativeNormal()(quantile_);
120 LOG("DIM confidence level " << confidenceLevel);
121
122 Size simple_dim_index_h = Size(floor(quantile_ * (samples - 1) + 0.5));
123 Size simple_dim_index_p = Size(floor((1.0 - quantile_) * (samples - 1) + 0.5));
124
125 Size nettingSetCount = 0;
126 for (auto n : nettingSetIds_) {
127 LOG("Process netting set " << n);
128
129 if (inputs_) {
130 // Check whether a deterministic IM evolution was provided for this netting set
131 // If found then we use this external data to overwrite the following
132 // - nettingSetExpectedDIM_ by nettingSet and date
133 // - nettingSetZeroOrderDIM_ by nettingSet and date
134 // - nettingSetSimpleDIMh_ by nettingSet and date
135 // - nettingSetSimpleDIMp_ by nettingSet and date
136 // - nettingSetDIM_ by nettingSet, date and sample
137 // - dimCube_ by nettingSet, date and sample
138 TimeSeries<Real> im = inputs_->deterministicInitialMargin(n);
139 LOG("External IM evolution for netting set " << n << " has size " << im.size());
140 if (im.size() > 0) {
141 WLOG("Try overriding DIM with externally provided IM evolution for netting set " << n);
142 for (Size j = 0; j < stopDatesLoop; ++j) {
143 Date d = cube_->dates()[j];
144 try {
145 Real value = im[d];
146 nettingSetExpectedDIM_[n][j] = value;
147 nettingSetZeroOrderDIM_[n][j] = value;
148 nettingSetSimpleDIMh_[n][j] = value;
149 nettingSetSimpleDIMp_[n][j] = value;
150 for (Size k = 0; k < samples; ++k) {
151 dimCube_->set(value, nettingSetCount, j, k);
152 nettingSetDIM_[n][j][k] = value;
153 }
154 } catch(std::exception& e) {
155 QL_FAIL("Failed to lookup external IM for netting set " << n
156 << " at date " << io::iso_date(d));
157 }
158 }
159 WLOG("Overriding DIM for netting set " << n << " succeeded");
160 // continue to the next netting set
161 continue;
162 }
163 }
164
165 if (currentIM_.find(n) != currentIM_.end()) {
166 Real t0im = currentIM_[n];
167 QL_REQUIRE(currentDim.find(n) != currentDim.end(), "current DIM not found for netting set " << n);
168 Real t0dim = currentDim[n];
169 Real t0scaling = t0im / t0dim;
170 LOG("t0 scaling for netting set " << n << ": t0im" << t0im << " t0dim=" << t0dim
171 << " t0scaling=" << t0scaling);
172 nettingSetScaling_[n] = t0scaling;
173 }
174
175 Real nettingSetDimScaling =
176 nettingSetScaling_.find(n) == nettingSetScaling_.end() ? 1.0 : nettingSetScaling_[n];
177 LOG("Netting set DIM scaling factor: " << nettingSetDimScaling);
178
179 for (Size j = 0; j < stopDatesLoop; ++j) {
180 accumulator_set<double, stats<boost::accumulators::tag::mean, boost::accumulators::tag::variance>> accDiff;
181 accumulator_set<double, stats<boost::accumulators::tag::mean>> accOneOverNumeraire;
182 for (Size k = 0; k < samples; ++k) {
183 Real numDefault =
184 cubeInterpretation_->getDefaultAggregationScenarioData(AggregationScenarioDataType::Numeraire, j, k);
185 Real numCloseOut =
186 cubeInterpretation_->getCloseOutAggregationScenarioData(AggregationScenarioDataType::Numeraire, j, k);
187 Real npvDefault = nettingSetNPV_[n][j][k];
188 Real flow = nettingSetFLOW_[n][j][k];
189 Real npvCloseOut = nettingSetCloseOutNPV_[n][j][k];
190 accDiff((npvCloseOut * numCloseOut) + (flow * numDefault) - (npvDefault * numDefault));
191 accOneOverNumeraire(1.0 / numDefault);
192 }
193
194 Size mporCalendarDays = cubeInterpretation_->getMporCalendarDays(cube_, j);
195 Real horizonScaling = sqrt(1.0 * horizonCalendarDays_ / mporCalendarDays);
196
197 Real stdevDiff = sqrt(variance(accDiff));
198 Real E_OneOverNumeraire =
199 mean(accOneOverNumeraire); // "re-discount" (the stdev is calculated on non-discounted deltaNPVs)
200
201 nettingSetZeroOrderDIM_[n][j] = stdevDiff * horizonScaling * confidenceLevel;
202 nettingSetZeroOrderDIM_[n][j] *= E_OneOverNumeraire;
203
204 vector<Real> rx0(samples, 0.0);
205 vector<Array> rx(samples, Array());
206 vector<Real> ry1(samples, 0.0);
207 vector<Real> ry2(samples, 0.0);
208 for (Size k = 0; k < samples; ++k) {
209 Real numDefault =
210 cubeInterpretation_->getDefaultAggregationScenarioData(AggregationScenarioDataType::Numeraire, j, k);
211 Real numCloseOut =
212 cubeInterpretation_->getCloseOutAggregationScenarioData(AggregationScenarioDataType::Numeraire, j, k);
213 Real x = nettingSetNPV_[n][j][k] * numDefault;
214 Real f = nettingSetFLOW_[n][j][k] * numDefault;
215 Real y = nettingSetCloseOutNPV_[n][j][k] * numCloseOut;
216 Real z = (y + f - x);
217 rx[k] = regressors_.empty() ? Array(1, nettingSetNPV_[n][j][k]) : regressorArray(n, j, k);
218 rx0[k] = rx[k][0];
219 ry1[k] = z; // for local regression
220 ry2[k] = z * z; // for least squares regression
221 nettingSetDeltaNPV_[n][j][k] = z;
222 regressorArray_[n][j][k] = rx[k];
223 }
224 vector<Real> delNpvVec_copy = nettingSetDeltaNPV_[n][j];
225 sort(delNpvVec_copy.begin(), delNpvVec_copy.end());
226 Real simpleDim_h = delNpvVec_copy[simple_dim_index_h];
227 Real simpleDim_p = delNpvVec_copy[simple_dim_index_p];
228 simpleDim_h *= horizonScaling; // the usual scaling factors
229 simpleDim_p *= horizonScaling; // the usual scaling factors
230 nettingSetSimpleDIMh_[n][j] = simpleDim_h * E_OneOverNumeraire; // discounted DIM
231 nettingSetSimpleDIMp_[n][j] = simpleDim_p * E_OneOverNumeraire; // discounted DIM
232
233 QL_REQUIRE(rx.size() > v.size(), "not enough points for regression with polynom order " << polynomOrder);
234 if (close_enough(stdevDiff, 0.0)) {
235 LOG("DIM: Zero std dev estimation at step " << j);
236 // Skip IM calculation if all samples have zero NPV (e.g. after latest maturity)
237 for (Size k = 0; k < samples; ++k) {
238 nettingSetDIM_[n][j][k] = 0.0;
239 nettingSetLocalDIM_[n][j][k] = 0.0;
240 }
241 } else {
242 // Least squares polynomial regression with specified polynom order
243 QuantExt::StabilisedGLLS ls(rx, ry2, v, QuantExt::StabilisedGLLS::MeanStdDev);
244 LOG("DIM data normalisation at time step "
245 << j << ": " << scientific << setprecision(6) << " x-shift = " << ls.xShift() << " x-multiplier = "
246 << ls.xMultiplier() << " y-shift = " << ls.yShift() << " y-multiplier = " << ls.yMultiplier());
247 LOG("DIM regression coefficients at time step " << j << ": " << fixed << setprecision(6)
248 << ls.transformedCoefficients());
249
250 // Local regression versus first regression variable (i.e. we do not perform a
251 // multidimensional local regression):
252 // We evaluate this at a limited number of samples only for validation purposes.
253 // Note that computational effort scales quadratically with number of samples.
254 // NadarayaWatson needs a large number of samples for good results.
255 QuantExt::NadarayaWatson lr(rx0.begin(), rx0.end(), ry1.begin(),
256 GaussianKernel(0.0, localRegressionBandWidth_));
257 Size localRegressionSamples = samples;
258 if (localRegressionEvaluations_ > 0)
259 localRegressionSamples = Size(floor(1.0 * samples / localRegressionEvaluations_ + .5));
260
261 // Evaluate regression function to compute DIM for each scenario
262 for (Size k = 0; k < samples; ++k) {
263 // Real num1 = scenarioData_->get(j, k, AggregationScenarioDataType::Numeraire);
264 Real numDefault = cubeInterpretation_->getDefaultAggregationScenarioData(
265 AggregationScenarioDataType::Numeraire, j, k);
266 Array regressor = regressors_.empty() ? Array(1, nettingSetNPV_[n][j][k]) : regressorArray(n, j, k);
267 Real e = ls.eval(regressor, v);
268 if (e < 0.0)
269 LOG("Negative variance regression for date " << j << ", sample " << k
270 << ", regressor = " << regressor);
271
272 // Note:
273 // 1) We assume vanishing mean of "z", because the drift over a MPOR is usually small,
274 // and to avoid a second regression for the conditional mean
275 // 2) In particular the linear regression function can yield negative variance values in
276 // extreme scenarios where an exact analytical or delta VaR calculation would yield a
277 // variance approaching zero. We correct this here by taking the positive part.
278 Real std = sqrt(std::max(e, 0.0));
279 Real scalingFactor = horizonScaling * confidenceLevel * nettingSetDimScaling;
280 // Real dim = std * scalingFactor / num1;
281 Real dim = std * scalingFactor / numDefault;
282 dimCube_->set(dim, nettingSetCount, j, k);
283 nettingSetDIM_[n][j][k] = dim;
284 nettingSetExpectedDIM_[n][j] += dim / samples;
285
286 // Evaluate the Kernel regression for a subset of the samples only (performance)
287 if (localRegressionEvaluations_ > 0 && (k % localRegressionSamples == 0))
288 // nettingSetLocalDIM_[n][j][k] = lr.standardDeviation(regressor[0]) * scalingFactor / num1;
289 nettingSetLocalDIM_[n][j][k] = lr.standardDeviation(regressor[0]) * scalingFactor / numDefault;
290 else
291 nettingSetLocalDIM_[n][j][k] = 0.0;
292 }
293 }
294 }
295
296 nettingSetCount++;
297 }
298 LOG("DIM by polynomial regression done");
299}

301Array RegressionDynamicInitialMarginCalculator::regressorArray(string nettingSet, Size dateIndex,
302 Size sampleIndex) {
303 Array a(regressors_.size());
304 for (Size i = 0; i < regressors_.size(); ++i) {
305 string variable = regressors_[i];
306 if (boost::to_upper_copy(variable) ==
307 "NPV") // this allows possibility to include NPV as a regressor alongside more fundamental risk factors
308 a[i] = nettingSetNPV_[nettingSet][dateIndex][sampleIndex];
309 else if (scenarioData_->has(AggregationScenarioDataType::IndexFixing, variable))
310 a[i] = cubeInterpretation_->getDefaultAggregationScenarioData(AggregationScenarioDataType::IndexFixing,
311 dateIndex, sampleIndex, variable);
312 else if (scenarioData_->has(AggregationScenarioDataType::FXSpot, variable))
313 a[i] = cubeInterpretation_->getDefaultAggregationScenarioData(AggregationScenarioDataType::FXSpot, dateIndex,
314 sampleIndex, variable);
315 else if (scenarioData_->has(AggregationScenarioDataType::Generic, variable))
316 a[i] = cubeInterpretation_->getDefaultAggregationScenarioData(AggregationScenarioDataType::Generic, dateIndex,
317 sampleIndex, variable);
318 else
319 QL_FAIL("scenario data does not provide data for " << variable);
320 }
321 return a;
322}

324map<string, Real> RegressionDynamicInitialMarginCalculator::unscaledCurrentDIM() {
325 // In this function we proxy the model-implied T0 IM by looking at the
326 // cube grid horizon lying closest to t0+mpor. We measure diffs relative
327 // to the mean of the distribution at this same time horizon, thus avoiding
328 // any cashflow-specific jumps
329
330 Date today = cube_->asof();
331 Size relevantDateIdx = 0;
332 Real sqrtTimeScaling = 1.0;
333 for (Size i = 0; i < cube_->dates().size(); ++i) {
334 Size daysFromT0 = (cube_->dates()[i] - today);
335 if (daysFromT0 < horizonCalendarDays_) {
336 // iterate until we straddle t0+mpor
337 continue;
338 } else if (daysFromT0 == horizonCalendarDays_) {
339 // this date corresponds to t0+mpor, so use it
340 relevantDateIdx = i;
341 sqrtTimeScaling = 1.0;
342 break;
343 } else if (daysFromT0 > horizonCalendarDays_) {
344 // the first date greater than t0+MPOR, check if it is closest
345 Size lastIdx = (i == 0) ? 0 : (i - 1);
346 Size lastDaysFromT0 = (cube_->dates()[lastIdx] - today);
347 int daysFromT0CloseOut = daysFromT0 - horizonCalendarDays_;
348 int prevDaysFromT0CloseOut = lastDaysFromT0 - horizonCalendarDays_;
349 if (std::abs(daysFromT0CloseOut) <= std::abs(prevDaysFromT0CloseOut)) {
350 relevantDateIdx = i;
351 sqrtTimeScaling = std::sqrt(Real(horizonCalendarDays_) / Real(daysFromT0));
352 } else {
353 relevantDateIdx = lastIdx;
354 sqrtTimeScaling = std::sqrt(Real(horizonCalendarDays_) / Real(lastDaysFromT0));
355 }
356 break;
357 }
358 }
359 // set some reasonable bounds on the sqrt time scaling, so that we are not looking at a ridiculous time horizon
360 if (sqrtTimeScaling < std::sqrt(0.5) || sqrtTimeScaling > std::sqrt(2.0)) {
361 WLOG("T0 IM Estimation - The estimation time horizon from grid is not sufficiently close to t0+MPOR - "
362 << QuantLib::io::iso_date(cube_->dates()[relevantDateIdx])
363 << ", the T0 IM estimate might be inaccurate. Consider inserting a first grid tenor closer to the dim "
364 "horizon");
365 }
366
367 // TODO: Ensure that the simulation containers read-from below are indeed populated
368
369 Real confidenceLevel = QuantLib::InverseCumulativeNormal()(quantile_);
370 Size simple_dim_index_h = Size(floor(quantile_ * (cube_->samples() - 1) + 0.5));
371 map<string, Real> t0dimReg, t0dimSimple;
372 for (auto it_map = nettingSetNPV_.begin(); it_map != nettingSetNPV_.end(); ++it_map) {
373 string key = it_map->first;
374 vector<Real> t0_dist = it_map->second[relevantDateIdx];
375 Size dist_size = t0_dist.size();
376 QL_REQUIRE(dist_size == cube_->samples(),
377 "T0 IM - cube samples size mismatch - " << dist_size << ", " << cube_->samples());
378 Real mean_t0_dist = std::accumulate(t0_dist.begin(), t0_dist.end(), 0.0);
379 mean_t0_dist /= dist_size;
380 vector<Real> t0_delMtM_dist(dist_size, 0.0);
381 accumulator_set<double, stats<boost::accumulators::tag::mean, boost::accumulators::tag::variance>> acc_delMtm;
382 accumulator_set<double, stats<boost::accumulators::tag::mean>> acc_OneOverNum;
383 for (Size i = 0; i < dist_size; ++i) {
384 Real numeraire = scenarioData_->get(relevantDateIdx, i, AggregationScenarioDataType::Numeraire);
385 Real deltaMtmFromMean = numeraire * (t0_dist[i] - mean_t0_dist) * sqrtTimeScaling;
386 t0_delMtM_dist[i] = deltaMtmFromMean;
387 acc_delMtm(deltaMtmFromMean);
388 acc_OneOverNum(1.0 / numeraire);
389 }
390 Real E_OneOverNumeraire = mean(acc_OneOverNum);
391 Real variance_t0 = variance(acc_delMtm);
392 Real sqrt_t0 = sqrt(variance_t0);
393 t0dimReg[key] = (sqrt_t0 * confidenceLevel * E_OneOverNumeraire);
394 std::sort(t0_delMtM_dist.begin(), t0_delMtM_dist.end());
395 t0dimSimple[key] = (t0_delMtM_dist[simple_dim_index_h] * E_OneOverNumeraire);
396
397 LOG("T0 IM (Reg) - {" << key << "} = " << t0dimReg[key]);
398 LOG("T0 IM (Simple) - {" << key << "} = " << t0dimSimple[key]);
399 }
400 LOG("T0 IM Calculations Completed");
401
402 return t0dimReg;
403}

405void RegressionDynamicInitialMarginCalculator::exportDimEvolution(ore::data::Report& dimEvolutionReport) {
406
407 Size samples = dimCube_->samples();
408 Size stopDatesLoop = datesLoopSize_;
409 Date asof = cube_->asof();
410
411 dimEvolutionReport.addColumn("TimeStep", Size())
412 .addColumn("Date", Date())
413 .addColumn("DaysInPeriod", Size())
414 .addColumn("ZeroOrderDIM", Real(), 6)
415 .addColumn("AverageDIM", Real(), 6)
416 .addColumn("AverageFLOW", Real(), 6)
417 .addColumn("SimpleDIM", Real(), 6)
418 .addColumn("NettingSet", string())
419 .addColumn("Time", Real(), 6);
420
421 for (const auto& [nettingSet, _] : dimCube_->idsAndIndexes()) {
422
423 LOG("Export DIM evolution for netting set " << nettingSet);
424 for (Size i = 0; i < stopDatesLoop; ++i) {
425 Real expectedFlow = 0.0;
426 for (Size j = 0; j < samples; ++j) {
427 expectedFlow += nettingSetFLOW_[nettingSet][i][j] / samples;
428 }
429
430 Date defaultDate = dimCube_->dates()[i];
431 Time t = ActualActual(ActualActual::ISDA).yearFraction(asof, defaultDate);
432 Size days = cubeInterpretation_->getMporCalendarDays(dimCube_, i);
433 dimEvolutionReport.next()
434 .add(i)
435 .add(defaultDate)
436 .add(days)
437 .add(nettingSetZeroOrderDIM_[nettingSet][i])
438 .add(nettingSetExpectedDIM_[nettingSet][i])
439 .add(expectedFlow)
440 .add(nettingSetSimpleDIMh_[nettingSet][i])
441 .add(nettingSet)
442 .add(t);
443 }
444 }
445 dimEvolutionReport.end();
446 LOG("Exporting expected DIM through time done");
447}

449void RegressionDynamicInitialMarginCalculator::exportDimRegression(
450 const std::string& nettingSet, const std::vector<Size>& timeSteps,
451 const std::vector<boost::shared_ptr<ore::data::Report>>& dimRegReports) {
452
453 QL_REQUIRE(dimRegReports.size() == timeSteps.size(),
454 "number of file names (" << dimRegReports.size() << ") does not match number of time steps ("
455 << timeSteps.size() << ")");
456 for (Size ii = 0; ii < timeSteps.size(); ++ii) {
457 Size timeStep = timeSteps[ii];
458 LOG("Export DIM by sample for netting set " << nettingSet << " and time step " << timeStep);
459
460 Size dates = dimCube_->dates().size();
461 const std::map<std::string, Size>& ids = dimCube_->idsAndIndexes();
462
463 auto it = ids.find(nettingSet);
464
465 QL_REQUIRE(it != ids.end(), "netting set " << nettingSet << " not found in DIM cube");
466
467 QL_REQUIRE(timeStep < dates - 1, "selected time step " << timeStep << " out of range [0, " << dates - 1 << "]");
468
469 Size samples = cube_->samples();
470 vector<Real> numeraires(samples, 0.0);
471 for (Size k = 0; k < samples; ++k)
472 // numeraires[k] = scenarioData_->get(timeStep, k, AggregationScenarioDataType::Numeraire);
473 numeraires[k] =
474 cubeInterpretation_->getDefaultAggregationScenarioData(AggregationScenarioDataType::Numeraire, timeStep, k);
475
476 auto p = sort_permutation(regressorArray_[nettingSet][timeStep], lessThan);
477 vector<Array> reg = apply_permutation(regressorArray_[nettingSet][timeStep], p);
478 vector<Real> dim = apply_permutation(nettingSetDIM_[nettingSet][timeStep], p);
479 vector<Real> ldim = apply_permutation(nettingSetLocalDIM_[nettingSet][timeStep], p);
480 vector<Real> delta = apply_permutation(nettingSetDeltaNPV_[nettingSet][timeStep], p);
481 vector<Real> num = apply_permutation(numeraires, p);
482
483 boost::shared_ptr<ore::data::Report> regReport = dimRegReports[ii];
484 regReport->addColumn("Sample", Size());
485 for (Size k = 0; k < reg[0].size(); ++k) {
486 ostringstream o;
487 o << "Regressor_" << k << "_";
488 o << (regressors_.empty() ? "NPV" : regressors_[k]);
489 regReport->addColumn(o.str(), Real(), 6);
490 }
491 regReport->addColumn("RegressionDIM", Real(), 6)
492 .addColumn("LocalDIM", Real(), 6)
493 .addColumn("ExpectedDIM", Real(), 6)
494 .addColumn("ZeroOrderDIM", Real(), 6)
495 .addColumn("DeltaNPV", Real(), 6)
496 .addColumn("SimpleDIM", Real(), 6);
497
498 // Note that RegressionDIM, LocalDIM, DeltaNPV are _not_ reduced by the numeraire in this output,

500 // This is so that the regression formula can be manually validated
501

503 regReport->next().add(j);
504 for (Size k = 0; k < reg[j].size(); ++k)
505 regReport->add(reg[j][k]);

509 .add(nettingSetZeroOrderDIM_[nettingSet][timeStep])
510 .add(delta[j])
511 .add(nettingSetSimpleDIMh_[nettingSet][timeStep]);
512 }
513 regReport->end();
514 LOG("Exporting DIM by Sample done for");
515 }
