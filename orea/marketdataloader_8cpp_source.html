32// Additional quotes for fx fixings, add fixing quotes with USD and EUR to the list of fixings
33// requested in order to triangulate missing fixings
34void additional_fx_fixings(const string& fixingId, const set<Date>& fixingDates, FixingMap& relevantFixings) {
35 std::vector<std::string> tokens;
36 boost::split(tokens, fixingId, boost::is_any_of("-"));
37 QL_REQUIRE(tokens.size() == 4, "MarketDataLoader::additional_fx_fixings: Invalid fixing id, "
38 << "must be of form FX-TYPE-CCY1-CCY, e.g FX-ECB-EUR-GBP");
39
40 // add fixings on inverted ccy pair
41 relevantFixings[tokens[0] + "-" + tokens[1] + "-" + tokens[3] + "-" + tokens[2]].insert(fixingDates.begin(),
42 fixingDates.end());
43
44 vector<string> baseCcys = {"USD", "EUR"};
45
46 for (auto ccy : baseCcys) {
47
48 string fixingType = tokens[0] + "-" + tokens[1] + "-";
49 if (tokens[2] != ccy) {
50 string fix = fixingType + ccy + "-" + tokens[2];
51 relevantFixings[fix].insert(fixingDates.begin(), fixingDates.end());
52
53 if (tokens[3] != ccy) {
54 fix = fixingType + tokens[2] + "-" + ccy;
55 relevantFixings[fix].insert(fixingDates.begin(), fixingDates.end());
56 }
57 }
58
59 if (tokens[3] != ccy) {
60 string fix = fixingType + ccy + "-" + tokens[3];
61 relevantFixings[fix].insert(fixingDates.begin(), fixingDates.end());
62
63 if (tokens[2] != ccy) {
64 fix = fixingType + tokens[3] + "-" + ccy;
65 relevantFixings[fix].insert(fixingDates.begin(), fixingDates.end());
66 }
67 }
68 }
69}

71// Additional quotes for commodity fixings
72void additional_commodity_fixings(const string& fixingId, const set<Date>& fixingDates, FixingMap& fixings,
73 map<pair<string, Date>, set<Date>>& commodityMap) {
74
75 boost::shared_ptr<Conventions> conventions = InstrumentConventions::instance().conventions();
76
77 auto ind = parseCommodityIndex(fixingId);
78 string commName = ind->underlyingName();
79
80 boost::shared_ptr<CommodityFutureConvention> cfc;
81 if (conventions->has(commName)) {
82 cfc = boost::dynamic_pointer_cast<CommodityFutureConvention>(conventions->get(commName));
83 }
84
85 if (cfc) {
86 // Add historical fixings for daily and monthly expiring contracts
87 // TODO: potentially need to add for OffPeakPowerIndex commodities too.
88 if (cfc->contractFrequency() == Daily) {
89 for (const auto& fd : fixingDates) {
90 // Add 1 week lookback for each date for daily expiry
91 set<Date> dates;
92 Date wLookback = fd - Period(1, Weeks);
93 do {
94 dates.insert(wLookback++);
95 } while (wLookback <= fd);
96
97 TLOG("Adding (date, id) = (" << io::iso_date(fd) << "," << fixingId << ")");
98 // Add to the fixings so a fixing is requested for all dates, and also to the commodityMap
99 // so we can map a fixing to the correct date required
100 fixings[fixingId].insert(dates.begin(), dates.end());
101 commodityMap[pair(fixingId, fd)].insert(dates.begin(), dates.end());
102 }
103 } else {
104 // for monthly expiries add fixings for the last 45 days
105 for (const auto& fd : fixingDates) {
106 set<Date> dates;
107 Date wLookback = fd - Period(45, Days);
108 do {
109 dates.insert(wLookback++);
110 } while (wLookback <= fd);
111
112 TLOG("Adding (date, id) = (" << io::iso_date(fd) << "," << fixingId << ")");
113 // Add to the fixings so a fixing is requested for all dates, and also to the commodityMap
114 // so we can map a fixing to the correct date required
115 fixings[fixingId].insert(dates.begin(), dates.end());
116 commodityMap[pair(fixingId, fd)].insert(dates.begin(), dates.end());
117 }
118 }
119 }
120}

122// Additional fixings for equity index decomposition
123void additional_equity_fixings(map<string, set<Date>>& fixings, const TodaysMarketParameters& mktParams,
124 const boost::shared_ptr<ReferenceDataManager> refData,
125 const boost::shared_ptr<CurveConfigurations>& curveConfigs) {
126 std::string configuration = Market::defaultConfiguration;
127 Date asof = Settings::instance().evaluationDate();
128 boost::shared_ptr<CurrencyHedgedEquityIndexReferenceDatum> currencyHedgedIndexData;
129 if (mktParams.hasMarketObject(MarketObject::EquityCurve)) {
130 for (const auto& [equityName, _] : mktParams.mapping(MarketObject::EquityCurve, configuration)) {
131 try {
132 auto indexDecomposition = loadCurrencyHedgedIndexDecomposition(equityName, refData, curveConfigs);
133 if (indexDecomposition) {
134 indexDecomposition->addAdditionalFixingsForEquityIndexDecomposition(asof, fixings);
135 }
136 } catch (const std::exception& e) {
137 ALOG("adding addtional equity fixing failed, " << e.what());
138 }
139 }
140 }

148void MarketDataLoader::addRelevantFixings(
149 const std::pair<std::string, std::set<QuantLib::Date>>& fixing,
150 std::map<std::pair<std::string, QuantLib::Date>, std::set<QuantLib::Date>>& lastAvailableFixingLookupMap) {
151 if (isFxIndex(fixing.first)) {
152 // for FX fixings we want to add additional fixings to allow triangulation in case of missing
153 // fixings if we need EUR/GBP fixing but it is not available, we can imply from EUR/USD and GBP/USD
154 additional_fx_fixings(fixing.first, fixing.second, fixings_);
155 }
156 if (isCommodityIndex(fixing.first)) {
157 additional_commodity_fixings(fixing.first, fixing.second, fixings_, lastAvailableFixingLookupMap);
158 }
159 fixings_[fixing.first].insert(fixing.second.begin(), fixing.second.end());
160 // also add to portfolioFixings_, used for alerting missing required fixings
161 portfolioFixings_[fixing.first].insert(fixing.second.begin(), fixing.second.end());
162}

164void MarketDataLoader::populateFixings(
165 const std::vector<boost::shared_ptr<ore::data::TodaysMarketParameters>>& todaysMarketParameters,
166 const std::set<QuantLib::Date>& loaderDates) {
167 if (inputs_->allFixings()) {
168 impl()->retrieveFixings(loader_);
169 } else {
170 LOG("Asking portfolio for its required fixings");
171 FixingMap portfolioFixings;
172 std::map<std::pair<std::string, QuantLib::Date>, std::set<QuantLib::Date>> lastAvailableFixingLookupMap;
173
174 // portfolio fixings will warn if missinbg
175 if (inputs_->portfolio()) {
176 portfolioFixings = inputs_->portfolio()->fixings();
177 LOG("The portfolio depends on fixings from " << portfolioFixings.size() << " indices");
178 for (const auto& it : portfolioFixings)
179 addRelevantFixings(it, lastAvailableFixingLookupMap);
180 }
181
182 LOG("Add fixings possibly required for bootstrapping TodaysMarket");
183 for (const auto& tmp : todaysMarketParameters) {
184 for (const auto d : loaderDates)
185 addMarketFixingDates(d, fixings_, *tmp);
186 LOG("Add fixing possibly required for equity index delta risk decomposition")
187 additional_equity_fixings(fixings_, *tmp, inputs_->refDataManager(),
188 inputs_->curveConfigs().get());
189 }
190
191 if (inputs_->eomInflationFixings()) {
192 LOG("Adjust inflation fixing dates to the end of the month before the request");
193 amendInflationFixingDates(fixings_);
194 amendInflationFixingDates(portfolioFixings_);
195 }
196
197 if (fixings_.size() > 0)
198 impl()->retrieveFixings(loader_, fixings_, lastAvailableFixingLookupMap);
199
200 // apply all fixings now
201 applyFixings(loader_->loadFixings());
202
203 // check and warn any missing fixings - only warn for portfolio fixings
204 for (const auto& f : portfolioFixings_) {
205 for (const auto& d : f.second) {
206 if (!loader_->hasFixing(f.first, d)) {
207 string fixingErr = "";
208 if (isFxIndex(f.first)) {
209 auto fxInd = parseFxIndex(f.first);
210 try {
211 if(fxInd->fixingCalendar().isBusinessDay(d))
212 fxInd->fixing(d);
213 break;
214 } catch (const std::exception& e) {
215 fixingErr = ", error: " + ore::data::to_string(e.what());
216 }
217 }
218 WLOG(StructuredFixingWarningMessage(f.first, d, "Missing fixing",
219 "Could not find required fixing ID."));
220 }
221 }
222 }
223 }

226void MarketDataLoader::populateLoader(
227 const std::vector<boost::shared_ptr<ore::data::TodaysMarketParameters>>& todaysMarketParameters,
228 const std::set<QuantLib::Date>& loaderDates) {
229
230 // create a loader if doesn't already exist
231 if (!loader_)
232 loader_ = boost::make_shared<InMemoryLoader>();
233 else
234 loader_->reset(); // can only populate once to avoid duplicates
235
236 // check input data
237 QL_REQUIRE(!inputs_->curveConfigs().empty(), "Need at least one curve configuration to populate loader.");
238 QL_REQUIRE(todaysMarketParameters.size() > 0, "No todaysMarketParams provided to populate market data loader.");
239
240 // for equitites check if we have corporate action data
241 std::map<std::string, std::string> equities;
242 for (const auto& tmp : todaysMarketParameters) {
243 if (tmp->hasMarketObject(MarketObject::EquityCurve)) {
244 auto eqMap = tmp->mapping(MarketObject::EquityCurve, Market::defaultConfiguration);
245 for (auto eq : eqMap) {
246 if (inputs_->refDataManager() && inputs_->refDataManager()->hasData("Equity", eq.first)) {
247 auto refData = boost::dynamic_pointer_cast<EquityReferenceDatum>(
248 inputs_->refDataManager()->getData("Equity", eq.first));
249 auto erData = refData->equityData();
250 equities[eq.first] = erData.equityId;
251 } else
252 equities[eq.first] = eq.first;
253 }
254 }
255 }
256 if (equities.size() > 0)
257 impl()->loadCorporateActionData(loader_, equities);
258
259 // apply dividends now
260 applyDividends(loader_->loadDividends());
261
262 populateFixings(todaysMarketParameters, loaderDates);
263
264 LOG("Adding the loaded fixings to the IndexManager");
265 applyFixings(loader_->loadFixings());
266
267 // Get set of quotes we need
268 LOG("Generating market datum set");
269 set<string> quotes;
270 for (const auto& tmp : todaysMarketParameters) {
271 // Find all configurations in this todaysMarket
272 std::set<std::string> configurations;
273 for (auto c : tmp->configurations())
274 configurations.insert(c.first);
275
276 for (const auto& [_,curveConfig] : inputs_->curveConfigs().curveConfigurations()) {
277 auto qs = curveConfig->quotes(tmp, configurations);
278 quotes.insert(qs.begin(), qs.end());
279 }
280 }
281
282 for (const auto& d : loaderDates) {
283 QuoteMap quoteMap;
284 quoteMap[d] = quotes;
285
286 LOG("CurveConfigs require " << quotes.size() << " quotes");
287
288 // Get the relevant market data loader for the pricing call
289 impl()->retrieveMarketData(loader_, quoteMap, d);
290
291 quotes_[d] = quotes;
292 }
293 LOG("Got market data");
294}
