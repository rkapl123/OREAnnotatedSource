50map<RiskFactorKey, Real> ZeroToParCube::parDeltas(QuantLib::Size tradeIdx) const {
51
52 DLOG("Calculating par deltas for trade index " << tradeIdx);
53
54 map<RiskFactorKey, Real> result;
55
56 // Get the "par-convertible" zero deltas
57 boost::numeric::ublas::vector<Real> zeroDeltas(parConverter_->rawKeys().size(), 0.0);
58 const boost::shared_ptr<NPVSensiCube>& sensiCube = zeroCube_->npvCube();
59
60 for (auto const& kv : sensiCube->getTradeNPVs(tradeIdx)) {
61 auto factor = zeroCube_->upFactor(kv.first);
62 // index might not belong to an up/down scenario
63 if (factor.keytype != RiskFactorKey::KeyType::None) {
64 auto it = factorToIndex_.find(factor);
65 if (it == factorToIndex_.end()) {
66 if (ParSensitivityAnalysis::isParType(factor.keytype) && typesDisabled_.count(factor.keytype) != 1) {
67 if (continueOnError_) {
68 ALOG(StructuredAnalyticsErrorMessage("Par conversion", "", "Par factor " + ore::data::to_string(factor) + " not found in factorToIndex map"));
69 } else {
70 QL_REQUIRE(!ParSensitivityAnalysis::isParType(factor.keytype) ||
71 typesDisabled_.count(factor.keytype) == 1,
72 "ZeroToParCube::parDeltas(): par factor " << factor << " not found in factorToIndex map");
73 }
74 }
75 } else if (!zeroCube_->twoSidedDelta(factor.keytype)) {
76 zeroDeltas[it->second] = zeroCube_->delta(tradeIdx, kv.first);
77 } else {
78 Size downIdx = zeroCube_->downFactors().at(factor).index;
79 zeroDeltas[it->second] = zeroCube_->delta(tradeIdx, kv.first, downIdx);
80 }
81 }
82 }
83
84 // Convert the zero deltas to par deltas
85 boost::numeric::ublas::vector<Real> parDeltas = parConverter_->convertSensitivity(zeroDeltas);
86 Size counter = 0;
87 for (const auto& key : parConverter_->parKeys()) {
88 if (!close(parDeltas[counter], 0.0)) {
89 result[key] = parDeltas[counter];
90 }
91 counter++;
92 }
93
94 // Add non-zero deltas that do not need to be converted from underlying zero cube
95 for (const auto& key : zeroCube_->upFactors()) {
96 if (!ParSensitivityAnalysis::isParType(key.first.keytype) || typesDisabled_.count(key.first.keytype) == 1) {
97 Real delta = 0.0;
98 if (!zeroCube_->twoSidedDelta(key.first.keytype)) {
99 delta = zeroCube_->delta(tradeIdx, key.second.index);
100 } else {
101 Size downIdx = zeroCube_->downFactors().at(key.first).index;
102 delta = zeroCube_->delta(tradeIdx, key.second.index, downIdx);
103 }
104 if (!close(delta, 0.0)) {
105 result[key.first] = delta;
106 }
107 }
108 }
109
110 DLOG("Finished calculating par deltas for trade index " << tradeIdx);
111
112 return result;
113}

115map<RiskFactorKey, Real> ZeroToParCube::parDeltas(const string& tradeId) const {
116
117 DLOG("Calculating par deltas for trade " << tradeId);
118 map<RiskFactorKey, Real> result;
119 result = parDeltas(zeroCube_->npvCube()->getTradeIndex(tradeId));
120
121 DLOG("Finished calculating par deltas for trade " << tradeId);
122
123 return result;
124}
