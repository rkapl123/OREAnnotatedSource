96 void XvaRunner::bufferSimulationPaths() {
97 
98  LOG("XvaRunner::bufferSimulationPaths() called");
99 
100  auto stateProcess = model_->stateProcess(scenarioGeneratorData_->discretization());
101  auto pathGen = MultiPathGeneratorFactory().build(scenarioGeneratorData_->sequenceType(), stateProcess,
102  scenarioGeneratorData_->getGrid()->timeGrid(),
103  scenarioGeneratorData_->seed(), scenarioGeneratorData_->ordering(),
104  scenarioGeneratorData_->directionIntegers());
105 
106  if (!bufferedPaths_) {
107  bufferedPaths_ = boost::make_shared<std::vector<std::vector<Path>>>(
108  scenarioGeneratorData_->samples(), std::vector<Path>(stateProcess->size(), TimeGrid()));
109  }
110 
111  for (Size p = 0; p < scenarioGeneratorData_->samples(); ++p) {
112  const MultiPath& path = pathGen->next().value;
113  for (Size d = 0; d < stateProcess->size(); ++d) {
114  (*bufferedPaths_)[p][d] = path[d];
115  }
116  }
117 
118  LOG("XvaRunner::bufferSimulationPaths() finished");
119 }

121 void XvaRunner::buildSimMarket(const boost::shared_ptr<ore::data::Market>& market,
122  const boost::optional<std::set<std::string>>& currencyFilter, const bool continueOnErr) {
123  LOG("XvaRunner::buildSimMarket() called");
124 
125  Settings::instance().evaluationDate() = asof_;
126 
127  boost::shared_ptr<ScenarioSimMarketParameters> projectedSsmData;
128  if (currencyFilter) {
129  projectedSsmData = projectSsmData(*currencyFilter);
130  }
131  else {
132  projectedSsmData = simMarketData_;
133  }
134 
135  boost::shared_ptr<ScenarioFactory> sf = boost::make_shared<SimpleScenarioFactory>();
136  boost::shared_ptr<ScenarioGenerator> sg =
137  getProjectedScenarioGenerator(currencyFilter, market, projectedSsmData, sf, continueOnErr);
138  simMarket_ = boost::make_shared<ScenarioSimMarket>(
139  market, projectedSsmData, *conventions_, Market::defaultConfiguration, *curveConfigs_, *todaysMarketParams_,
140  true, false, true, false, iborFallbackConfig_);
141  simMarket_->scenarioGenerator() = sg;
142 
143  for (auto b : extraEngineBuilders_)
144  b->reset();
145 
146  DLOG("build scenario data");
147 
148  scenarioData_ = boost::make_shared<InMemoryAggregationScenarioData>(
149  scenarioGeneratorData_->getGrid()->valuationDates().size(), scenarioGeneratorData_->samples());
150  simMarket_->aggregationScenarioData() = scenarioData_;
151 
152  simFactory_ =
153  boost::make_shared<EngineFactory>(engineData_, simMarket_, map<MarketContext, string>(), extraEngineBuilders_,
154  extraLegBuilders_, referenceData_, iborFallbackConfig_);
155 }

157 void XvaRunner::buildCube(const boost::optional<std::set<std::string>>& tradeIds, const bool continueOnErr) {
158 
159  LOG("XvaRunner::buildCube called");
160 
161  Settings::instance().evaluationDate() = asof_;
162 
163  boost::shared_ptr<Portfolio> portfolio = boost::make_shared<Portfolio>();
164  if (tradeIds) {
165  for (auto const& t : *tradeIds) {
166  QL_REQUIRE(portfolio_->has(t), "XvaRunner::buildCube(): portfolio does not contain trade with id '"
167  << t << "' specified in the filter");
168  portfolio->add(portfolio_->get(t));
169  }
170  } else {
171  portfolio = portfolio_;
172  }
173 
174  DLOG("build portfolio");
175 
176  // FIXME why do we need this? portfolio_->reset() is not sufficient to ensure XVA simulation run fast (and this is
177  // called before)
178  for (auto const& t : portfolio_->trades()) {
179  try {
180  t->build(simFactory_);
181  } catch (...) {
182  // we don't care, this is just to reset the portfolio, the real build is below
183  }
184  }
185 
186  portfolio->build(simFactory_);
187 
188  DLOG("build calculators");
189 
190  std::vector<boost::shared_ptr<ValuationCalculator>> calculators;
191  boost::shared_ptr<NPVCalculator> npvCalculator = boost::make_shared<NPVCalculator>(baseCurrency_);
192  if (scenarioGeneratorData_->withCloseOutLag()) {
193  // depth 2: NPV and close-out NPV
194  cube_ = getNpvCube(asof_, portfolio->ids(), scenarioGeneratorData_->getGrid()->valuationDates(),
195  scenarioGeneratorData_->samples(), 2);
196  cubeInterpreter_ = boost::make_shared<MporGridCubeInterpretation>(scenarioGeneratorData_->getGrid());
197  // default date value stored at index 0, close-out value at index 1
198  calculators.push_back(boost::make_shared<MPORCalculator>(npvCalculator, 0, 1));
199  calculationType_ = "NoLag";
200  if (calculationType_ != inputCalculationType_) {
201  ALOG("Forcing calculation type " << calculationType_ << " for simulations with close-out grid");
202  }
203  } else {
204  if (storeFlows_) {
205  // regular, depth 2: NPV and cash flow
206  cube_ = getNpvCube(asof_, portfolio->ids(), scenarioGeneratorData_->getGrid()->dates(),
207  scenarioGeneratorData_->samples(), 2);
208  calculators.push_back(
209  boost::make_shared<CashflowCalculator>(baseCurrency_, asof_, scenarioGeneratorData_->getGrid(), 1));
210  } else
211  // regular, depth 1
212  cube_ = getNpvCube(asof_, portfolio->ids(), scenarioGeneratorData_->getGrid()->dates(),
213  scenarioGeneratorData_->samples(), 1);
214 
215  cubeInterpreter_ = boost::make_shared<RegularCubeInterpretation>();
216  calculators.push_back(npvCalculator);
217  calculationType_ = inputCalculationType_;
218  }
219 
220  DLOG("get netting cube");
221 
222  nettingCube_ = getNettingSetCube(calculators, portfolio);
223 
224  DLOG("run valuation engine");
225 
226  ValuationEngine engine(asof_, scenarioGeneratorData_->getGrid(), simMarket_);
227  engine.buildCube(portfolio, cube_, calculators, scenarioGeneratorData_->withMporStickyDate(), nettingCube_);
228 }

231  const boost::shared_ptr<NPVCube>& npvCube,
