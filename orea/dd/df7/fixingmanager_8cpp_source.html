48 // Search for a valid fixing date maximum gap days larger than d, the only relevant case for this so far is BMA/SIFMA
49 Date nextValidFixingDate(Date d, const boost::shared_ptr<Index>& index, Size gap = 7) {
50  Date adjusted = d;
51  for (Size i = 0; i <= gap; ++i) {
52  adjusted = d + i;
53  if (index->isValidFixingDate(adjusted))
54  return adjusted;
55  }
56  QL_FAIL("no valid fixing date found for index " << index->name() << " within gap from " << io::iso_date(d));
57 }

63 void FixingManager::initialise(const boost::shared_ptr<Portfolio>& portfolio, const boost::shared_ptr<Market>& market,
64  const std::string& configuration) {
65 
66  // loop over all cashflows, populate index map
67 
68  auto standardCashflowHandler = boost::make_shared<StandardFixingManagerCashflowHandler>();
69 
70  for (auto trade : portfolio->trades()) {
71  for (auto leg : trade->legs()) {
72  for (auto cf : leg) {
73  bool done = false;
74  for (auto& h : FixingManagerCashflowHandlerFactory::instance().handlers()) {
75  if (done)
76  break;
77  done = done || h->processCashflow(cf, fixingMap_);
78  }
79  if (!done)
80  standardCashflowHandler->processCashflow(cf, fixingMap_);
81  }
82  }
83  // some trades require fixings, but don't have legs - actually we might switch to the
84  // logic here for all trade types eventually, see ore ticket 1157
85  if (trade->tradeType() == "ForwardRateAgreement") {
86  for (auto const& m : trade->fixings(QuantLib::Date::maxDate())) {
87  auto index = market->iborIndex(m.first, configuration);
88  for (auto const& d : m.second) {
89  fixingMap_[*index].insert(d);
90  }
91  }
92  }
93  }
94 
95  // Now cache the original fixings so we can re-write on reset()
96  for (auto m : fixingMap_) {
97  fixingCache_[m.first] = IndexManager::instance().getHistory(m.first->name());
98  }
99 }

102 void FixingManager::update(Date d) {
103  if (!fixingMap_.empty()) {
104  QL_REQUIRE(d >= fixingsEnd_, "Can't go back in time, fixings must be reset."
105  " Update date "
106  << d << " but current fixings go to " << fixingsEnd_);
107  if (d > fixingsEnd_)
108  applyFixings(fixingsEnd_, d);
109  }
110  fixingsEnd_ = d;
111 }

114 void FixingManager::reset() {

118  modifiedFixingHistory_ = false;
119  }
120  fixingsEnd_ = today_;

123 void FixingManager::applyFixings(Date start, Date end) {
124  // Loop over all indices
125  for (auto const& m : fixingMap_) {
126  boost::shared_ptr<ZeroInflationIndex> zii = boost::dynamic_pointer_cast<ZeroInflationIndex>(m.first);
127  boost::shared_ptr<YoYInflationIndex> yii = boost::dynamic_pointer_cast<YoYInflationIndex>(m.first);
128  set<Date> fixingDates;
129  Date currentFixingDate;
130  Date fixStart = start;
131  Date fixEnd = end;
132  if (zii) { // for inflation indices we just only add a fixing for the first date in the month
133  fixStart =
134  inflationPeriod(fixStart - zii->zeroInflationTermStructure()->observationLag(), zii->frequency()).first;
135  fixEnd =
136  inflationPeriod(fixEnd - zii->zeroInflationTermStructure()->observationLag(), zii->frequency()).first +
137  1;
138  currentFixingDate = fixEnd;
139  for (auto const& d : m.second) {
140  fixingDates.insert(inflationPeriod(d, zii->frequency()).first);
141  }
142  } else if (yii) {
143  fixStart =
144  inflationPeriod(fixStart - yii->yoyInflationTermStructure()->observationLag(), yii->frequency()).first;
145  fixEnd =
146  inflationPeriod(fixEnd - yii->yoyInflationTermStructure()->observationLag(), yii->frequency()).first +
147  1;
148  currentFixingDate = fixEnd;
149  for (auto const& d : m.second) {
150  fixingDates.insert(inflationPeriod(d, yii->frequency()).first);
151  }
152  } else {
153  currentFixingDate = m.first->fixingCalendar().adjust(fixEnd, Following);
154  // This date is a business day but may not be a valid fixing date in case of BMA/SIFMA
155  if (!m.first->isValidFixingDate(currentFixingDate))
156  currentFixingDate = nextValidFixingDate(currentFixingDate, m.first);
157  fixingDates = m.second;
158  }
159 
160  // Add we have a coupon between start and asof.
161  bool needFixings = false;
162  for (auto const& d : fixingDates) {
163  if (d >= fixStart && d < fixEnd) {
164  needFixings = true;
165  break;
166  }
167  }
168 
169  if (needFixings) {
170  Rate currentFixing = m.first->fixing(currentFixingDate);
171  // if we read the fixing from an inverted FxIndex we have to undo the inversion
172  if (auto f = boost::dynamic_pointer_cast<FxIndex>(m.first)) {
173  if (f->inverseIndex())
174  currentFixing = 1.0 / currentFixing;
175  }
176  TimeSeries<Real> history;
177  for (auto const& d : fixingDates) {
178  if (d >= fixStart && d < fixEnd) {
179  // Fixing dates include the valuation grid dates which might not be valid fixing dates (BMA/SIFMA)
180  bool valid = m.first->isValidFixingDate(d);
181  if (valid) {
182  history[d] = currentFixing;
183  modifiedFixingHistory_ = true;
184  }
185  }
186  if (d >= fixEnd)
187  break;
188  }
189  m.first->addFixings(history, true);
190  }
191  }
192 }

194 bool StandardFixingManagerCashflowHandler::processCashflow(const boost::shared_ptr<QuantLib::CashFlow>& cf,
195  FixingManager::FixingMap& fixingMap) {
196 
197  // For any coupon type that requires fixings, it must be handled here
198  // Most coupons are based off a floating rate coupon and their single index
199  // will be captured in section A.
200  //
201  // Other more exotic coupons (inflation, CMS spreads, etc) are captured on a
202  // case by case basis in section B.
203  //
204  // In all cases we want to add dates to the fixingMap_ map.
205 
206  // A floating rate coupons
207 
208  // extract underlying from cap/floored coupons
209  boost::shared_ptr<FloatingRateCoupon> frc;
210  auto cfCpn = boost::dynamic_pointer_cast<CappedFlooredCoupon>(cf);
211  if (cfCpn)
212  frc = cfCpn->underlying();
213  else
214  frc = boost::dynamic_pointer_cast<FloatingRateCoupon>(cf);
215 
216  if (frc) {
217  // A1 indices with fixings derived from underlying indices
218  auto cmssp = boost::dynamic_pointer_cast<CmsSpreadCoupon>(frc);
219  if (cmssp) {
220  fixingMap[cmssp->swapSpreadIndex()->swapIndex1()].insert(frc->fixingDate());
221  fixingMap[cmssp->swapSpreadIndex()->swapIndex2()].insert(frc->fixingDate());
222  return true;
223  }
224  auto dcmssp = boost::dynamic_pointer_cast<DigitalCmsSpreadCoupon>(frc);
225  if (dcmssp) {
226  fixingMap
227  [boost::dynamic_pointer_cast<CmsSpreadCoupon>(dcmssp->underlying())->swapSpreadIndex()->swapIndex1()]
228  .insert(frc->fixingDate());
229  fixingMap
230  [boost::dynamic_pointer_cast<CmsSpreadCoupon>(dcmssp->underlying())->swapSpreadIndex()->swapIndex2()]
231  .insert(frc->fixingDate());
232  return true;
233  }
234 
235  // A2 indices with native fixings, but no only on the standard fixing date
236  auto on = boost::dynamic_pointer_cast<QuantExt::OvernightIndexedCoupon>(frc);
237  if (on) {
238  for (auto const& d : on->fixingDates()) {
239  fixingMap[on->index()].insert(d);
240  }
241  return true;
242  }
243  auto avon = boost::dynamic_pointer_cast<AverageONIndexedCoupon>(frc);
244  if (avon) {
245  for (auto const& d : avon->fixingDates())
246  fixingMap[avon->index()].insert(d);
247  return true;
248  }
249  auto bma = boost::dynamic_pointer_cast<AverageBMACoupon>(frc);
250  if (bma) {
251  for (auto const& d : bma->fixingDates())
252  fixingMap[bma->index()].insert(d);
253  return true;
254  }
255 
256  // A3 standard case
257  auto fb = boost::dynamic_pointer_cast<FallbackIborIndex>(frc->index());
258  if (fb != nullptr && frc->fixingDate() >= fb->switchDate()) {
259  // handle ibor fallback
260  return processCashflow(fb->onCoupon(frc->fixingDate()), fixingMap);
261  } else {
262  // handle other cases
263  fixingMap[frc->index()].insert(frc->fixingDate());
264  }
265  }
266 
267  // B other coupon types
268 
269  boost::shared_ptr<FloatingRateFXLinkedNotionalCoupon> fc =
270  boost::dynamic_pointer_cast<FloatingRateFXLinkedNotionalCoupon>(cf);
271  if (fc) {
272  fixingMap[fc->fxIndex()].insert(fc->fxFixingDate());
273  return processCashflow(fc->underlying(), fixingMap);
274  }
275 
276  boost::shared_ptr<FXLinkedCashFlow> flcf = boost::dynamic_pointer_cast<FXLinkedCashFlow>(cf);
277  if (flcf) {
278  fixingMap[flcf->fxIndex()].insert(flcf->fxFixingDate());
279  return true;
280  }
281 
282  boost::shared_ptr<CPICoupon> cpc = boost::dynamic_pointer_cast<CPICoupon>(cf);
283  if (cpc) {
284  fixingMap[cpc->index()].insert(cpc->fixingDate());
285  return true;
286  }
287 
288  boost::shared_ptr<InflationCoupon> ic = boost::dynamic_pointer_cast<InflationCoupon>(cf);
289  if (ic) {
290  fixingMap[ic->index()].insert(ic->fixingDate());
291  return true;
292  }
293 
294  boost::shared_ptr<CPICashFlow> cpcf = boost::dynamic_pointer_cast<CPICashFlow>(cf);
295  if (cpcf) {
296  fixingMap[cpcf->index()].insert(cpcf->fixingDate());
297  return true;
298  }
299 
300  boost::shared_ptr<EquityCoupon> ec = boost::dynamic_pointer_cast<EquityCoupon>(cf);
301  if (ec) {
302  for (auto const& f : ec->fixingDates())
303  fixingMap[ec->equityCurve()].insert(f);
304  if (ec->fxIndex() != nullptr) {
305  fixingMap[ec->fxIndex()].insert(ec->fixingStartDate());
306  }
307  }
308 
309  return true;
310 }
