38void VarAnalyticImpl::runAnalytic(const boost::shared_ptr<ore::data::InMemoryLoader>& loader,
39 const std::set<std::string>& runTypes) {
40 MEM_LOG;
41 LOG("Running parametric VaR");
42
43 Settings::instance().evaluationDate() = inputs_->asof();
44 ObservationMode::instance().setMode(inputs_->observationModel());
45
46 LOG("VAR: Build Market");
47 CONSOLEW("Risk: Build Market for VaR");
48 analytic()->buildMarket(loader);
49 CONSOLE("OK");
50
51 CONSOLEW("Risk: Build Portfolio for VaR");
52 analytic()->buildPortfolio();
53 CONSOLE("OK");
54
55 LOG("Build trade to portfolio id mapping");
56 map<string, set<pair<string, Size>>> tradePortfolio;
57 for (auto const& [tradeId, trade] : analytic()->portfolio()->trades()) {
58 for (auto const& pId : trade->portfolioIds())
59 tradePortfolio[pId].insert(make_pair(tradeId, Null<Size>()));
60 }
61
62 ParametricVarCalculator::ParametricVarParams varParams(inputs_->varMethod(), inputs_->mcVarSamples(),
63 inputs_->mcVarSeed());
64
65 LOG("Build VaR calculator");
66 auto calc = boost::make_shared<ParametricVarReport>(tradePortfolio, inputs_->portfolioFilter(),
67 inputs_->sensitivityStream(), inputs_->covarianceData(), inputs_->varQuantiles(),
68 varParams, inputs_->varBreakDown(), inputs_->salvageCovariance());
69
70 boost::shared_ptr<InMemoryReport> report = boost::make_shared<InMemoryReport>();
71 analytic()->reports()["VAR"]["var"] = report;
72
73 LOG("Call VaR calculation");
74 CONSOLEW("Risk: VaR Calculation");
75 calc->calculate(*report);
76 CONSOLE("OK");
77
78 LOG("Parametric VaR completed");
79 MEM_LOG;
80}
