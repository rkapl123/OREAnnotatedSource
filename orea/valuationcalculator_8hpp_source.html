39class ValuationCalculator {

43 virtual void calculate(
44 //! The trade
45 const boost::shared_ptr<Trade>& trade,
46 //! Trade index for writing to the cube
47 Size tradeIndex,
48 //! The market
49 const boost::shared_ptr<SimMarket>& simMarket,

54 //! The date

58 //! Sample
59 Size sample,
60 //! isCloseOut

62
63 virtual void calculateT0(
64 //! The trade

66 //! Trade index for writing to the cube
67 Size tradeIndex,
68 //! The market

70 //! The cube
71 boost::shared_ptr<NPVCube>& outputCube,
72 //! The cube
73 boost::shared_ptr<NPVCube>& outputCubeNettingSet) = 0;
74
75 // called once before the valuation engine run
76 virtual void init(const boost::shared_ptr<Portfolio>& portfolio, const boost::shared_ptr<SimMarket>& simMarket) = 0;
77

116//! CashflowCalculator
117/*! Calculates the cashflow, converted to base ccy, from t to t+1, this interval is defined by the provided dategrid
118 * The interval is (t, t+1], i.e. we exclude todays flows and include flows that fall exactly on t+1.
119 * For t0 we do nothing (and so the cube will have a 0 value)
120 */
121class CashflowCalculator : public ValuationCalculator {
122public:
123 //! Constructor takes the base ccy, date grid and index of cube to write to.
124 CashflowCalculator(const std::string& baseCcyCode, const Date& t0Date, const boost::shared_ptr<DateGrid>& dateGrid,
125 Size index)
126 : baseCcyCode_(baseCcyCode), t0Date_(t0Date), dateGrid_(dateGrid), index_(index) {}
127
128 virtual void calculate(const boost::shared_ptr<Trade>& trade, Size tradeIndex,
129 const boost::shared_ptr<SimMarket>& simMarket, boost::shared_ptr<NPVCube>& outputCube,
130 boost::shared_ptr<NPVCube>& outputCubeNettingSet, const Date& date, Size dateIndex,
131 Size sample, bool isCloseOut = false) override;
132
133 virtual void calculateT0(const boost::shared_ptr<Trade>& trade, Size tradeIndex,
134 const boost::shared_ptr<SimMarket>& simMarket, boost::shared_ptr<NPVCube>& outputCube,
135 boost::shared_ptr<NPVCube>& outputCubeNettingSet) override {}
136
137 void init(const boost::shared_ptr<Portfolio>& portfolio, const boost::shared_ptr<SimMarket>& simMarket) override;
138 void initScenario() override;
139
140private:
141 std::string baseCcyCode_;
142 Date t0Date_;
143 boost::shared_ptr<DateGrid> dateGrid_;
144 Size index_;
145
146 std::vector<Handle<Quote>> ccyQuotes_;
147 std::vector<double> fxRates_;
148 std::vector<std::vector<Size>> tradeAndLegCcyIndex_;
149};
