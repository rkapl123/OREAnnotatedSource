158 template <class Interpolator2D> void StrippedCPIVolatilitySurface<Interpolator2D>::performCalculations() const {
159  strikes_ = priceSurface_->strikes();
160  maturities_ = priceSurface_->maturities();
161  volData_ = QuantLib::Matrix(strikes_.size(), maturities_.size(), QuantLib::Null<QuantLib::Real>());
162  QuantLib::Brent solver;
163  QuantLib::Real guess = (upperVolBound_ + lowerVolBound_) / 2.0;
164  QuantLib::Date startDate = QuantLib::Settings::instance().evaluationDate();
165  for (QuantLib::Size i = 0; i < strikes_.size(); i++) {
166  for (QuantLib::Size j = 0; j < maturities_.size(); j++) {
167  QuantLib::Date maturityDate = optionDateFromTenor(maturities_[j]);
168  QuantLib::Real atmRate = index_->zeroInflationTermStructure()->zeroRate(maturityDate);
169  bool useFloor = chooseFloor(strikes_[i], atmRate);
170 
171  // FIXME: Do we need an effective maturity here ?
172  QuantLib::Real priceToMatch = useFloor ? priceSurface_->floorPrice(maturities_[j], strikes_[i])
173  : priceSurface_->capPrice(maturities_[j], strikes_[i]);
174  try {
175  ObjectiveFunction func(priceToMatch, useFloor, strikes_[i], startDate, maturityDate, baseCPI_,
176  priceSurface_, index_, engine_);
177  QuantLib::Real found = solver.solve(func, solverTolerance_, guess, lowerVolBound_, upperVolBound_);
178  volData_[i][j] = found;
179  } catch (std::exception& e) {
180  QL_FAIL("failed to find implied vol for "
181  << (useFloor ? "Floor" : "Cap") << " with strike " << strikes_[i] << " and maturity "
182  << maturities_[j] << ", because: " << e.what() << " "
183  << QuantLib::io::iso_date(startDate + maturities_[j]) << " " << maturityDate);
184  }
185  }
186  }
187 
188  maturityTimes_.clear();
189  for (QuantLib::Size i = 0; i < maturities_.size(); i++) {
190  QuantLib::Date d = optionDateFromTenor(maturities_[i]);
191  maturityTimes_.push_back(inflationTime(d));
192  }
193 
194  vols_ = interpolator2d_.interpolate(maturityTimes_.begin(), maturityTimes_.end(), strikes_.begin(), strikes_.end(),
195  volData_);
196  vols_.enableExtrapolation();
197 }

199 template <class Interpolator2D>
200 bool StrippedCPIVolatilitySurface<Interpolator2D>::chooseFloor(QuantLib::Real strike, QuantLib::Real atmRate) const {
201  if (preference_ == Floor) {
202  if (strike <= priceSurface_->floorStrikes().back())
203  return true;
204  else
205  return false;
206  }
207 
208  if (preference_ == Cap) {
209  if (strike < priceSurface_->capStrikes().front())
210  return true;
211  else
212  return false;
213  }
214 
215  // else: Use floors where we have floor quotes only, caps where we have cap quotes only,
216  // and decide based on ATM where we have both cap and floor quotes
217 
218  // 1) strike < maxFloorStrike < minCapStrike: Floor!
219  if (strike <= priceSurface_->floorStrikes().back() && strike < priceSurface_->capStrikes().front())
220  return true;
221  // 2) strike > maxFloorStrike and strike >= minCapStrike: Cap!
222  else if (strike > priceSurface_->floorStrikes().back() && strike >= priceSurface_->capStrikes().front())
223  return false;
224  // 3) Overlap, maxFloorStrike > minCapStrike and strike in between: Depends on atmRate which surface we pick
225  else if (strike <= priceSurface_->floorStrikes().back() && strike >= priceSurface_->capStrikes().front()) {
226  // we have overlapping strikes, decide depending on atm level
227  if (strike < atmRate)
228  return true;
229  else
230  return false;
231  // 4) Gap, maxFloorStrike < minCapStrike and strike in the gap: Depends on atmRate which surface we
232  // extrapolate
233  } else if (strike > priceSurface_->floorStrikes().back() && strike < priceSurface_->capStrikes().front()) {
234  // there is a gap between flor end and caps begin, decise again depending on strike level
235  if (strike < atmRate)
236  return true;
237  else
238  return false;
239  } else {
240  QL_FAIL("case not covered in StrippedCPIVolatilitySurface: strike="
241  << strike << " maxFloorStrike=" << priceSurface_->floorStrikes().back()
242  << " minCapStrike=" << priceSurface_->capStrikes().front() << " atm=" << atmRate);
243  return false;
244  }
245 }
