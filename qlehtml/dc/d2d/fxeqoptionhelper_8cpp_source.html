49 void FxEqOptionHelper::performCalculations() const {
50  if (hasMaturity_)
51  exerciseDate_ = calendar_.advance(termStructure_->referenceDate(), maturity_);
52  tau_ = termStructure_->timeFromReference(exerciseDate_);
53  atm_ = spot_->value() * foreignYield_->discount(tau_) / termStructure_->discount(tau_);
54  effStrike_ = strike_;
55  if (effStrike_ == Null<Real>())
56  effStrike_ = atm_;
57  type_ = effStrike_ >= atm_ ? Option::Call : Option::Put;
58  boost::shared_ptr<StrikedTypePayoff> payoff(new PlainVanillaPayoff(type_, effStrike_));
59  boost::shared_ptr<Exercise> exercise = boost::make_shared<EuropeanExercise>(exerciseDate_);
60  option_ = boost::shared_ptr<VanillaOption>(new VanillaOption(payoff, exercise));
61  BlackCalibrationHelper::performCalculations();
62 }
