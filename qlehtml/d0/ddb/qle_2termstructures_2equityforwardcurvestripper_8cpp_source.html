51 Real PriceError::operator()(Volatility x) const {
52  vol_.setValue(x);
53  Real npv;
54  // Barone Adesi Whaley fails for very small variance, so wrap in a try/catch
55  try {
56  npv = option_.NPV();
57  } catch (...) {
58  npv = 0.0;
59  }
60  return npv - targetValue_;
61 }

93 void EquityForwardCurveStripper::performCalculations() const {
94 
95  vector<vector<Real> > allStrikes = callSurface_->strikes();
96  forwards_.resize(callSurface_->expiries().size());
97 
98  // at each option expiry time we calulate a forward
99  for (Size i = 0; i < expiries().size(); i++) {
100  Date expiry = expiries()[i];
101  // get the relevant strikes at this expiry
102  vector<Real> strikes = allStrikes[i];
103  QL_REQUIRE(strikes.size() > 0, "No strikes for expiry " << expiry);
104 
105  // if we only have one strike we just use that to get the forward
106  if (strikes.size() == 1) {
107  forwards_[i] = forwardFromPutCallParity(expiry, strikes.front(), *callSurface_, *putSurface_);
108  continue;
109  }
110 
111  // we make a first guess at the forward price
112  // strikes are ordered, lowest to highest, we take the first guess as midpoint of 2 strikes
113  // where (C-P) goes from positive to negative
114  Real forward = strikes.back();
115  for (Size k = 0; k < strikes.size(); k++) {
116  if (callSurface_->price(expiry, strikes[k]) <= putSurface_->price(expiry, strikes[k])) {
117  if (k == 0)
118  forward = strikes.front();
119  else
120  forward = (strikes[k] + strikes[k - 1]) / 2;
121  break;
122  }
123  }
124 
125  // call and put surface to be used to find forward - updated for American
126  auto callSurface = callSurface_;
127  auto putSurface = putSurface_;
128 
129  Size maxIter = 100;
130  Size j = 0;
131  bool isForward = false;
132  while (!isForward && j < maxIter) {
133 
134  if (type_ == Exercise::American) {
135 
136  // we take the strike either side of our forward guess, this is sufficient because
137  // we construct new price surfaces but only ask for the price at the forward from these
138  vector<Real> amerStrikes(2);
139  auto it_lower = std::lower_bound(strikes.begin(), strikes.end(), forward);
140  if (it_lower == strikes.end())
141  it_lower = std::prev(it_lower);
142  amerStrikes[1] = *it_lower;
143  amerStrikes[0] = (it_lower == strikes.begin()) ? *it_lower : *std::prev(it_lower);
144 
145  // for American options we first get the implied vol from the American premiums
146  // we use these to construct the European prices in order to apply put call parity
147 
148  // get date and daycounter from the prics surface
149  Date asof = callSurface_->referenceDate();
150  DayCounter dc = callSurface_->dayCounter();
151  Calendar cal = callSurface_->calendar();
152  Time t = dc.yearFraction(asof, expiry);
153 
154  // dividend rate from S_t = S * exp((r - q) * t)
155  Real q = forecastCurve_->zeroRate(t, Continuous) - log(forward / equitySpot_->value()) / t;
156 
157  // term structures needed to get implied vol
158  boost::shared_ptr<SimpleQuote> volQuote = boost::make_shared<SimpleQuote>(0.1);
159  Handle<BlackVolTermStructure> volTs(
160  boost::make_shared<BlackConstantVol>(asof, cal, Handle<Quote>(volQuote), dc));
161  Handle<YieldTermStructure> divTs(boost::make_shared<FlatForward>(asof, q, dc));
162 
163  // a black scholes process
164  boost::shared_ptr<GeneralizedBlackScholesProcess> gbsp =
165  boost::make_shared<BlackScholesMertonProcess>(equitySpot_, divTs, forecastCurve_, volTs);
166  boost::shared_ptr<PricingEngine> engine =
167  boost::make_shared<QuantExt::BaroneAdesiWhaleyApproximationEngine>(gbsp);
168 
169  vector<vector<Volatility> > vols(2, vector<Volatility>(amerStrikes.size()));
170  vector<Option::Type> types;
171  types.push_back(Option::Call);
172  types.push_back(Option::Put);
173 
174  for (Size l = 0; l < types.size(); l++) {
175  for (Size k = 0; k < amerStrikes.size(); k++) {
176  // create an american option for current strike/expiry and type
177  boost::shared_ptr<StrikedTypePayoff> payoff(new PlainVanillaPayoff(types[l], amerStrikes[k]));
178  boost::shared_ptr<Exercise> exercise = boost::make_shared<AmericanExercise>(expiry);
179  VanillaOption option(payoff, exercise);
180  option.setPricingEngine(engine);
181 
182  // option.setPricingEngine(engine);
183  Real targetPrice = types[l] == Option::Call ? callSurface_->price(expiry, amerStrikes[k])
184  : putSurface_->price(expiry, amerStrikes[k]);
185 
186  // calculate the implied volatility using a solver
187  try {
188  PriceError f(option, *volQuote, targetPrice);
189  Brent solver;
190  solver.setMaxEvaluations(100);
191  solver.setLowerBound(0.0001);
192  vols[l][k] = solver.solve(f, 0.0001, 0.2, 0.01);
193  } catch (...) {
194  vols[l][k] = 0.0;
195  }
196  }
197  }
198 
199  vector<Real> newStrikes;
200  vector<Date> dates;
201  vector<Real> callPremiums, putPremiums;
202 
203  for (Size k = 0; k < amerStrikes.size(); k++) {
204  if (vols[0][k] != 0.0 && vols[1][k] != 0.0) {
205  // get the european option prices for each strike
206  Real call = blackFormula(Option::Call, amerStrikes[k], forward, vols[0][k] * sqrt(t),
207  forecastCurve_->discount(t));
208  Real put = blackFormula(Option::Put, amerStrikes[k], forward, vols[1][k] * sqrt(t),
209  forecastCurve_->discount(t));
210 
211  if (call && put) {
212  newStrikes.push_back(amerStrikes[k]);
213  dates.push_back(expiry);
214  callPremiums.push_back(call);
215  putPremiums.push_back(put);
216  }
217  }
218  }
219  // throw away any strikes where the vol is zero for either put or call
220  // must have at least one new strike otherwise continue with currenct price surfaces
221  if (newStrikes.size() > 0) {
222  // build call/put price surfaces with the new European prices
223  callSurface = boost::make_shared<OptionPriceSurface>(asof, dates, newStrikes, callPremiums, dc);
224  putSurface = boost::make_shared<OptionPriceSurface>(asof, dates, newStrikes, putPremiums, dc);
225  }
226  }
227 
228  Real newForward = 0.0;
229  // if our guess is below the first strike or after the last strike we just take the relevant strike
230  if (forward <= strikes.front()) {
231  newForward = forwardFromPutCallParity(expiry, strikes.front(), *callSurface, *putSurface);
232  // if forward is still less than first strike we accept this
233  isForward = newForward <= strikes.front();
234  } else if (forward >= strikes.back()) {
235  newForward = forwardFromPutCallParity(expiry, strikes.back(), *callSurface, *putSurface);
236  // if forward is still greater than last strike we accept this
237  isForward = newForward >= strikes.back();
238  } else {
239  newForward = forwardFromPutCallParity(expiry, forward, *callSurface, *putSurface);
240 
241  // check - has it moved by less that 0.1%
242  isForward = fabs((newForward - forward) / forward) < 0.001;
243  }
244  forward = newForward;
245  j++;
246  }
247  forwards_[i] = forward;
248  }
249 }

251 Real EquityForwardCurveStripper::forwardFromPutCallParity(Date d, Real strike, const OptionPriceSurface& callSurface,
252  const OptionPriceSurface& putSurface) const {
253  Real C = callSurface.price(d, strike);
254  Real P = putSurface.price(d, strike);
255  Real D = forecastCurve_->discount(d);
256 
257  return strike + (C - P) / D;
258 }
