38 void OvernightIndexedCrossCcyBasisSwap::initialize() {
39  legs_[0] = OvernightLeg(paySchedule_, payIndex_).withNotionals(payNominal_).withSpreads(paySpread_);
40  legs_[0].insert(legs_[0].begin(),
41  boost::shared_ptr<CashFlow>(new SimpleCashFlow(-payNominal_, paySchedule_.dates().front())));
42  legs_[0].push_back(boost::shared_ptr<CashFlow>(new SimpleCashFlow(payNominal_, paySchedule_.dates().back())));
43 
44  legs_[1] = OvernightLeg(recSchedule_, recIndex_).withNotionals(recNominal_).withSpreads(recSpread_);
45  legs_[1].insert(legs_[1].begin(),
46  boost::shared_ptr<CashFlow>(new SimpleCashFlow(-recNominal_, recSchedule_.dates().front())));
47  legs_[1].push_back(boost::shared_ptr<CashFlow>(new SimpleCashFlow(recNominal_, recSchedule_.dates().back())));
48 
49  for (Size j = 0; j < 2; ++j) {
50  for (Leg::iterator i = legs_[j].begin(); i != legs_[j].end(); ++i)
51  registerWith(*i);
52  }
53 
54  payer_[0] = -1.0;
55  payer_[1] = +1.0;
56 
57  currency_[0] = payCurrency_;
58  currency_[1] = recCurrency_;
59 }
