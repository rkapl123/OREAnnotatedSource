140 inline Real ForwardBondTypePayoff::operator()(Real price) const {
141  switch (type_) {
142  case Position::Long:
143  return (price - strike_);
144  case Position::Short:
145  return (strike_ - price);
146  default:
147  QL_FAIL("unknown/illegal position type");
