158 template <class TimeInterpolator, class SmileInterpolator>
159 inline QuantLib::Rate StrippedOptionletAdapter<TimeInterpolator, SmileInterpolator>::minStrike() const {
160 
161  // If only one strike
162  if (oneStrike_) {
163  if (volatilityType() == QuantLib::ShiftedLognormal) {
164  return displacement() > 0.0 ? -displacement() : 0.0;
165  } else {
166  return QL_MIN_REAL;
167  }
168  }
169 
170  // Return the minimum strike over all optionlet tenors
171  QuantLib::Rate minStrike = optionletBase_->optionletStrikes(0).front();
172  for (QuantLib::Size i = 1; i < optionletBase_->optionletMaturities(); ++i) {
173  minStrike = std::min(optionletBase_->optionletStrikes(i).front(), minStrike);
174  }
175  return minStrike;
176 }

178 template <class TimeInterpolator, class SmileInterpolator>
179 inline QuantLib::Rate StrippedOptionletAdapter<TimeInterpolator, SmileInterpolator>::maxStrike() const {
180 
181  // If only one strike, there is no max strike
182  if (oneStrike_) {
183  return QL_MAX_REAL;

187  QuantLib::Rate maxStrike = optionletBase_->optionletStrikes(0).back();
188  for (QuantLib::Size i = 1; i < optionletBase_->optionletMaturities(); ++i) {
189  maxStrike = std::max(optionletBase_->optionletStrikes(i).back(), maxStrike);
190  }
191  return maxStrike;

215 template <class TimeInterpolator, class SmileInterpolator>
216 inline void StrippedOptionletAdapter<TimeInterpolator, SmileInterpolator>::performCalculations() const {
217 
218  // Some localised typedefs and using declarations to make the code more readable
219  using QuantLib::Rate;
220  using QuantLib::Size;
221  using QuantLib::Volatility;
222  using std::vector;
223 
224  // If only one strike, no strike section to update
225  if (oneStrike_) {
226  return;
227  }
228 
229  for (Size i = 0; i < optionletBase_->optionletMaturities(); ++i) {
230  const vector<Rate>& strikes = optionletBase_->optionletStrikes(i);
231  const vector<Volatility>& vols = optionletBase_->optionletVolatilities(i);
232  strikeSections_[i] = si_.interpolate(strikes.begin(), strikes.end(), vols.begin());
233  // Extrapolation can be enabled here. Range checks are performed in the volatility methods.
234  strikeSections_[i].enableExtrapolation();
235  }

250 template <class TimeInterpolator, class SmileInterpolator>
251 inline boost::shared_ptr<QuantLib::SmileSection>
252 StrippedOptionletAdapter<TimeInterpolator, SmileInterpolator>::smileSectionImpl(QuantLib::Time optionTime) const {
253 
254  // Some localised typedefs and using declarations to make the code more readable
255  using boost::lambda::_1;
256  using boost::lambda::_2;
257  using QuantLib::close;
258  using QuantLib::Null;
259  using QuantLib::Rate;
260  using QuantLib::Real;
261  using QuantLib::Size;
262  using QuantLib::Volatility;
263  using QuantLib::io::ordinal;
264  using std::equal;
265  using std::sqrt;
266  using std::vector;
267 
268  // Leave ATM rate as Null<Real>() for now but could interpolate optionletBase_->atmOptionletRates()?
269  Real atmRate = Null<Real>();
270 
271  // If one strike, return a flat smile section
272  if (oneStrike_) {
273  Volatility vol = volatility(optionTime, optionletBase_->optionletStrikes(0)[0], true);
274  return boost::make_shared<QuantLib::FlatSmileSection>(optionTime, vol, optionletBase_->dayCounter(), atmRate,
275  volatilityType(), displacement());
276  }
277 
278  // This method can only return a valid value if the strikes are the same for all optionlet dates
279  const vector<Rate>& strikes = optionletBase_->optionletStrikes(0);
280  for (Size i = 1; i < optionletBase_->optionletMaturities(); ++i) {
281  const vector<Rate>& compStrikes = optionletBase_->optionletStrikes(i);
282  QL_REQUIRE(equal(strikes.begin(), strikes.end(), compStrikes.begin(), boost::bind(close, _1, _2)),
283  "The strikes at the " << ordinal(i)
284  << " optionlet date do not equal those at the first optionlet date");
285  }
286 
287  // Store standard deviation at each strike
288  vector<Real> stdDevs;
289  for (Size i = 0; i < strikes.size(); i++) {
290  stdDevs.push_back(sqrt(blackVariance(optionTime, strikes[i], true)));
291  }
292 
293  // Return the smile section.
294  return boost::make_shared<QuantLib::InterpolatedSmileSection<SmileInterpolator> >(
295  optionTime, strikes, stdDevs, atmRate, si_, optionletBase_->dayCounter(), volatilityType(), displacement());
296 }

299 inline QuantLib::Volatility
300 StrippedOptionletAdapter<TimeInterpolator, SmileInterpolator>::volatilityImpl(QuantLib::Time optionTime,
301  QuantLib::Rate strike) const {
302 
303  // Some localised typedefs and using declarations to make the code more readable
304  using QuantLib::Interpolation;
305  using QuantLib::Size;
306  using QuantLib::Time;
307  using QuantLib::Volatility;
308  using std::vector;
309 
310  calculate();
311 
312  vector<Volatility> vols(optionletBase_->optionletMaturities());
313  for (Size i = 0; i < optionletBase_->optionletMaturities(); ++i) {
314  vols[i] = oneStrike_ ? optionletBase_->optionletVolatilities(i)[0] : strikeSections_[i](strike);
315  }
316 
317  vector<Time> fixingTimes = optionletBase_->optionletFixingTimes();
318  Interpolation ti = ti_.interpolate(fixingTimes.begin(), fixingTimes.end(), vols.begin());
319 
320  // Extrapolation can be enabled at this level. The range checks will have already been performed in
321  // the public OptionletVolatilityStructure::volatility method that calls this `Impl`
322  ti.enableExtrapolation();
323 
324  return ti(optionTime);
325 }
