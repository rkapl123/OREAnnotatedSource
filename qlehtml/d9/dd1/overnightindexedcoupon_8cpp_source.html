55 class OvernightIndexedCouponPricer : public FloatingRateCouponPricer {
56 public:
57  void initialize(const FloatingRateCoupon& coupon) override {
58  coupon_ = dynamic_cast<const OvernightIndexedCoupon*>(&coupon);
59  QL_ENSURE(coupon_, "wrong coupon type");
60  }
61 
62  void compute() const {
63  ext::shared_ptr<OvernightIndex> index = ext::dynamic_pointer_cast<OvernightIndex>(coupon_->index());
64 
65  const vector<Date>& fixingDates = coupon_->fixingDates();
66  const vector<Time>& dt = coupon_->dt();
67 
68  Size n = dt.size();
69  Size i = 0;
70  QL_REQUIRE(coupon_->rateCutoff() < n, "rate cutoff (" << coupon_->rateCutoff()
71  << ") must be less than number of fixings in period ("
72  << n << ")");
73  Size nCutoff = n - coupon_->rateCutoff();
74 
75  Real compoundFactor = 1.0, compoundFactorWithoutSpread = 1.0;
76 
77  // already fixed part
78  Date today = Settings::instance().evaluationDate();
79  while (i < n && fixingDates[std::min(i, nCutoff)] < today) {
80  // rate must have been fixed
81  Rate pastFixing = IndexManager::instance().getHistory(index->name())[fixingDates[std::min(i, nCutoff)]];
82  QL_REQUIRE(pastFixing != Null<Real>(),
83  "Missing " << index->name() << " fixing for " << fixingDates[std::min(i, nCutoff)]);
84  if (coupon_->includeSpread()) {
85  compoundFactorWithoutSpread *= (1.0 + pastFixing * dt[i]);
86  pastFixing += coupon_->spread();
87  }
88  compoundFactor *= (1.0 + pastFixing * dt[i]);
89  ++i;
90  }
91 
92  // today is a border case
93  if (i < n && fixingDates[std::min(i, nCutoff)] == today) {
94  // might have been fixed
95  try {
96  Rate pastFixing = IndexManager::instance().getHistory(index->name())[fixingDates[std::min(i, nCutoff)]];
97  if (pastFixing != Null<Real>()) {
98  if (coupon_->includeSpread()) {
99  compoundFactorWithoutSpread *= (1.0 + pastFixing * dt[i]);
100  pastFixing += coupon_->spread();
101  }
102  compoundFactor *= (1.0 + pastFixing * dt[i]);
103  ++i;
104  } else {
105  ; // fall through and forecast
106  }
107  } catch (Error&) {
108  ; // fall through and forecast
109  }
110  }
111 
112  // forward part using telescopic property in order
113  // to avoid the evaluation of multiple forward fixings
114  const vector<Date>& dates = coupon_->valueDates();
115  if (i < n) {
116  Handle<YieldTermStructure> curve = index->forwardingTermStructure();
117  QL_REQUIRE(!curve.empty(), "null term structure set to this instance of " << index->name());
118 
119  // handle the part until the rate cutoff (might be empty, i.e. startDiscount = endDiscount)
120  DiscountFactor startDiscount = curve->discount(dates[i]);
121  DiscountFactor endDiscount = curve->discount(dates[std::max(nCutoff, i)]);
122 
123  // handle the rate cutoff period (if there is any, i.e. if nCutoff < n)
124  if (nCutoff < n) {
125  // forward discount factor for one calendar day on the cutoff date
126  DiscountFactor discountCutoffDate =
127  curve->discount(dates[nCutoff] + 1) / curve->discount(dates[nCutoff]);
128  // keep the above forward discount factor constant during the cutoff period
129  endDiscount *= std::pow(discountCutoffDate, dates[n] - dates[nCutoff]);
130  }
131 
132  compoundFactor *= startDiscount / endDiscount;
133 
134  if (coupon_->includeSpread()) {
135  compoundFactorWithoutSpread *= startDiscount / endDiscount;
136  // this is an approximation, see "Ester / Daily Spread Curve Setup in ORE":
137  // set tau to an average value
138  Real tau = coupon_->dayCounter().yearFraction(dates[i], dates.back()) / (dates.back() - dates[i]);
139  // now use formula (4) from the paper
140  compoundFactor *= std::pow(1.0 + tau * coupon_->spread(), static_cast<int>(dates.back() - dates[i]));
141  }
142  }
143 
144  Rate tau = coupon_->lookback() == 0 * Days ? coupon_->accrualPeriod()
145  : coupon_->dayCounter().yearFraction(dates.front(), dates.back());
146  Rate rate = (compoundFactor - 1.0) / tau;
147  swapletRate_ = coupon_->gearing() * rate;
148  if (!coupon_->includeSpread()) {
149  swapletRate_ += coupon_->spread();
150  effectiveSpread_ = coupon_->spread();
151  effectiveIndexFixing_ = rate;
152  } else {
153  effectiveSpread_ = rate - (compoundFactorWithoutSpread - 1.0) / tau;
154  effectiveIndexFixing_ = rate - effectiveSpread_;
155  }
156  }
157 
158  Rate swapletRate() const override {
159  compute();
160  return swapletRate_;
161  }
162 
163  Rate effectiveSpread() const {
164  compute();
165  return effectiveSpread_;
166  }
167 
168  Rate effectiveIndexFixing() const {
169  compute();
170  return effectiveIndexFixing_;
171  }
172 
173  Real swapletPrice() const override { QL_FAIL("swapletPrice not available"); }
174  Real capletPrice(Rate) const override { QL_FAIL("capletPrice not available"); }
175  Rate capletRate(Rate) const override { QL_FAIL("capletRate not available"); }
176  Real floorletPrice(Rate) const override { QL_FAIL("floorletPrice not available"); }
177  Rate floorletRate(Rate) const override { QL_FAIL("floorletRate not available"); }
178 
179 protected:
180  const OvernightIndexedCoupon* coupon_;
181  mutable Real swapletRate_, effectiveSpread_, effectiveIndexFixing_;
182 };

187 OvernightIndexedCoupon::OvernightIndexedCoupon(const Date& paymentDate, Real nominal, const Date& startDate,
188  const Date& endDate,
189  const ext::shared_ptr<OvernightIndex>& overnightIndex, Real gearing,
190  Spread spread, const Date& refPeriodStart, const Date& refPeriodEnd,
191  const DayCounter& dayCounter, bool telescopicValueDates,
192  bool includeSpread, const Period& lookback, const Natural rateCutoff,
193  const Natural fixingDays, const Date& rateComputationStartDate,
194  const Date& rateComputationEndDate)
195  : FloatingRateCoupon(paymentDate, nominal, startDate, endDate, fixingDays, overnightIndex, gearing, spread,
196  refPeriodStart, refPeriodEnd, dayCounter, false),
197  includeSpread_(includeSpread), lookback_(lookback), rateCutoff_(rateCutoff),
198  rateComputationStartDate_(rateComputationStartDate), rateComputationEndDate_(rateComputationEndDate) {
199 
200  Date valueStart = rateComputationStartDate_ == Null<Date>() ? startDate : rateComputationStartDate_;
201  Date valueEnd = rateComputationEndDate_ == Null<Date>() ? endDate : rateComputationEndDate_;
202  if (lookback != 0 * Days) {
203  BusinessDayConvention bdc = lookback.length() > 0 ? Preceding : Following;
204  valueStart = overnightIndex->fixingCalendar().advance(valueStart, -lookback, bdc);
205  valueEnd = overnightIndex->fixingCalendar().advance(valueEnd, -lookback, bdc);
206  }
207 
208  // value dates
209  Date tmpEndDate = valueEnd;
210 
211  /* For the coupon's valuation only the first and last future valuation
212  dates matter, therefore we can avoid to construct the whole series
213  of valuation dates, a front and back stub will do. However notice
214  that if the global evaluation date moves forward it might run past
215  the front stub of valuation dates we build here (which incorporates
216  a grace period of 7 business after the evluation date). This will
217  lead to false coupon projections (see the warning the class header). */
218 
219  if (telescopicValueDates) {
220  // build optimised value dates schedule: front stub goes
221  // from start date to max(evalDate,valueStart) + 7bd
222  Date evalDate = Settings::instance().evaluationDate();
223  tmpEndDate = overnightIndex->fixingCalendar().advance(std::max(valueStart, evalDate), 7, Days, Following);
224  tmpEndDate = std::min(tmpEndDate, valueEnd);
225  }
226  Schedule sch = MakeSchedule()
227  .from(valueStart)
228  // .to(valueEnd)
229  .to(tmpEndDate)
230  .withTenor(1 * Days)
231  .withCalendar(overnightIndex->fixingCalendar())
232  .withConvention(overnightIndex->businessDayConvention())
233  .backwards();
234  valueDates_ = sch.dates();
235 
236  if (telescopicValueDates) {
237  // build optimised value dates schedule: back stub
238  // contains at least two dates
239  Date tmp = overnightIndex->fixingCalendar().advance(valueEnd, -1, Days, Preceding);
240  if (tmp != valueDates_.back())
241  valueDates_.push_back(tmp);
242  tmp = overnightIndex->fixingCalendar().adjust(valueEnd, overnightIndex->businessDayConvention());
243  if (tmp != valueDates_.back())
244  valueDates_.push_back(tmp);
245  }
246 
247  QL_ENSURE(valueDates_.size() >= 2 + rateCutoff_, "degenerate schedule");
248 
249  // fixing dates
250  n_ = valueDates_.size() - 1;
251  if (FloatingRateCoupon::fixingDays() == 0) {
252  fixingDates_ = vector<Date>(valueDates_.begin(), valueDates_.end() - 1);
253  } else {
254  fixingDates_.resize(n_);
255  for (Size i = 0; i < n_; ++i)
256  fixingDates_[i] = overnightIndex->fixingCalendar().advance(
257  valueDates_[i], -static_cast<Integer>(FloatingRateCoupon::fixingDays()), Days, Preceding);
258  }
259 
260  // accrual (compounding) periods
261  dt_.resize(n_);
262  const DayCounter& dc = overnightIndex->dayCounter();
263  for (Size i = 0; i < n_; ++i)
264  dt_[i] = dc.yearFraction(valueDates_[i], valueDates_[i + 1]);
265 
266  setPricer(ext::shared_ptr<FloatingRateCouponPricer>(new OvernightIndexedCouponPricer));
267 
268  // check that rate cutoff is < number of fixing dates
269  QL_REQUIRE(rateCutoff_ < n_,
270  "rate cutoff (" << rateCutoff_ << ") must be less than number of fixings in period (" << n_ << ")");
271 }

273 const vector<Rate>& OvernightIndexedCoupon::indexFixings() const {
274  fixings_.resize(n_);
275  Size i;
276  for (i = 0; i < n_ - rateCutoff_; ++i) {
277  fixings_[i] = index_->fixing(fixingDates_[i]);
278  }
279  Rate cutoffFixing = fixings_[i - 1];
280  while (i < n_) {
281  fixings_[i] = cutoffFixing;
282  i++;
283  }
284  return fixings_;
285 }

349 Rate CappedFlooredOvernightIndexedCoupon::rate() const {
350  QL_REQUIRE(underlying_->pricer(), "pricer not set");
351  Rate swapletRate = nakedOption_ ? 0.0 : underlying_->rate();
352  if (floor_ != Null<Real>() || cap_ != Null<Real>())
353  pricer()->initialize(*this);
354  Rate floorletRate = 0.;
355  if (floor_ != Null<Real>())
356  floorletRate = pricer()->floorletRate(effectiveFloor());
357  Rate capletRate = 0.;
358  if (cap_ != Null<Real>())
359  capletRate = (nakedOption_ && floor_ == Null<Real>() ? -1.0 : 1.0) * pricer()->capletRate(effectiveCap());
360  return swapletRate + floorletRate - capletRate;
361 }

365 Rate CappedFlooredOvernightIndexedCoupon::effectiveCap() const {
366  if (cap_ == Null<Real>())
367  return Null<Real>();
368  /* We have four cases dependent on localCapFloor_ and includeSpread. Notation in the formulas:
369  g gearing,
370  s spread,
371  A coupon amount,
372  f_i daily fixings,
373  \tau_i daily accrual fractions,
374  \tau coupon accrual fraction,
375  C cap rate
376  F floor rate
377  */
378  if (localCapFloor_) {
379  if (underlying_->includeSpread()) {
380  // A = g \cdot \frac{\prod (1 + \tau_i \min ( \max ( f_i + s , F), C)) - 1}{\tau}
381  return cap_ - underlying_->spread();
382  } else {
383  // A = g \cdot \frac{\prod (1 + \tau_i \min ( \max ( f_i , F), C)) - 1}{\tau} + s
384  return cap_;
385  }
386  } else {
387  if (underlying_->includeSpread()) {
388  // A = \min \left( \max \left( g \cdot \frac{\prod (1 + \tau_i(f_i + s)) - 1}{\tau}, F \right), C \right)
389  return (cap_ / gearing() - underlying_->effectiveSpread());
390  } else {
391  // A = \min \left( \max \left( g \cdot \frac{\prod (1 + \tau_i f_i) - 1}{\tau} + s, F \right), C \right)
392  return (cap_ - underlying_->effectiveSpread()) / gearing();
393  }
394  }
395 }

397 Rate CappedFlooredOvernightIndexedCoupon::effectiveFloor() const {
398  if (floor_ == Null<Real>())
399  return Null<Real>();
400  if (localCapFloor_) {
401  if (underlying_->includeSpread()) {
402  return floor_ - underlying_->spread();
403  } else {
404  return floor_;
405  }
406  } else {
407  if (underlying_->includeSpread()) {
408  return (floor_ - underlying_->effectiveSpread());
409  } else {
410  return (floor_ - underlying_->effectiveSpread()) / gearing();
411  }
412  }

564 OvernightLeg::operator Leg() const {
565 
566  QL_REQUIRE(!notionals_.empty(), "no notional given for compounding overnight leg");
567 
568  Leg cashflows;
569 
570  Calendar calendar = schedule_.calendar();
571  Calendar paymentCalendar = paymentCalendar_;
572 
573  if (calendar.empty())
574  calendar = paymentCalendar;
575  if (calendar.empty())
576  calendar = WeekendsOnly();
577  if (paymentCalendar.empty())
578  paymentCalendar = calendar;
579 
580  Date refStart, start, refEnd, end;
581  Date paymentDate;
582 
583  Size n = schedule_.size() - 1;
584  for (Size i = 0; i < n; ++i) {
585  refStart = start = schedule_.date(i);
586  refEnd = end = schedule_.date(i + 1);
587  paymentDate = paymentCalendar.advance(end, paymentLag_, Days, paymentAdjustment_);
588 
589  // determine refStart and refEnd
590 
591  if (i == 0 && schedule_.hasIsRegular() && !schedule_.isRegular(i + 1))
592  refStart = calendar.adjust(end - schedule_.tenor(), paymentAdjustment_);
593  if (i == n - 1 && schedule_.hasIsRegular() && !schedule_.isRegular(i + 1))
594  refEnd = calendar.adjust(start + schedule_.tenor(), paymentAdjustment_);
595 
596  // Determine the rate computation start and end date as
597  // - the coupon start and end date, if in arrears, and
598  // - the previous coupon start and end date, if in advance.
599  // In addition, adjust the start date, if a last recent period is given.
600 
601  Date rateComputationStartDate, rateComputationEndDate;
602  if (inArrears_) {
603  // in arrears fixing (i.e. the "classic" case)
604  rateComputationStartDate = start;
605  rateComputationEndDate = end;
606  } else {
607  // handle in advance fixing
608  if (i > 0) {
609  // if there is a previous period, we take that
610  rateComputationStartDate = schedule_.date(i - 1);
611  rateComputationEndDate = schedule_.date(i);
612  } else {
613  // otherwise we construct the previous period
614  rateComputationEndDate = start;
615  if (schedule_.hasTenor())
616  rateComputationStartDate = calendar.adjust(start - schedule_.tenor(), Preceding);
617  else
618  rateComputationStartDate = calendar.adjust(start - (end - start), Preceding);
619  }
620  }
621 
622  if (lastRecentPeriod_) {
623  rateComputationStartDate = (lastRecentPeriodCalendar_.empty() ? calendar : lastRecentPeriodCalendar_)
624  .advance(rateComputationEndDate, -*lastRecentPeriod_);
625  }
626 
627  // build coupon
628 
629  if (close_enough(detail::get(gearings_, i, 1.0), 0.0)) {
630  // fixed coupon
631  cashflows.push_back(boost::make_shared<FixedRateCoupon>(
632  paymentDate, detail::get(notionals_, i, 1.0), detail::effectiveFixedRate(spreads_, caps_, floors_, i),
633  paymentDayCounter_, start, end, refStart, refEnd));
634  } else {
635  // floating coupon
636  auto cpn = ext::make_shared<OvernightIndexedCoupon>(
637  paymentDate, detail::get(notionals_, i, 1.0), start, end, overnightIndex_,
638  detail::get(gearings_, i, 1.0), detail::get(spreads_, i, 0.0), refStart, refEnd, paymentDayCounter_,
639  telescopicValueDates_, includeSpread_, lookback_, rateCutoff_, fixingDays_, rateComputationStartDate,
640  rateComputationEndDate);
641  Real cap = detail::get(caps_, i, Null<Real>());
642  Real floor = detail::get(floors_, i, Null<Real>());
643  if (cap == Null<Real>() && floor == Null<Real>()) {
644  cashflows.push_back(cpn);
645  } else {
646  cashflows.push_back(ext::make_shared<CappedFlooredOvernightIndexedCoupon>(cpn, cap, floor, nakedOption_,
647  localCapFloor_));
648  }
649  }
650  }
651  return cashflows;
652 }
