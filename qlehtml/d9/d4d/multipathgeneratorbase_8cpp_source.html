66 const Sample<MultiPath>& MultiPathGeneratorSobolBrownianBridge::next() const {
67  Array asset = process_->initialValues();
68  MultiPath& path = next_.value;
69  for (Size j = 0; j < asset.size(); ++j) {
70  path[j].front() = asset[j];
71  }
72  next_.weight = gen_->nextPath();
73  std::vector<Real> output(asset.size());
74  for (Size i = 1; i < grid_.size(); ++i) {
75  Real t = grid_[i - 1];
76  Real dt = grid_.dt(i - 1);
77  gen_->nextStep(output);
78  Array tmp(output.begin(), output.end());
79  asset = process_->evolve(t, asset, dt, tmp);
80  for (Size j = 0; j < asset.size(); ++j) {
81  path[j][i] = asset[j];
82  }
83  }
84  return next_;
85 }
