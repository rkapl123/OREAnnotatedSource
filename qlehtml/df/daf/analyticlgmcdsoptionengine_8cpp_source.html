36 void AnalyticLgmCdsOptionEngine::calculate() const {
37 
38  QL_REQUIRE(arguments_.swap->protectionPaymentTime() == CreditDefaultSwap::ProtectionPaymentTime::atDefault,
39  "AnalyticLgmCdsOptionEngine: protection payment time must be atDefault");
40 
41  Real w = (arguments_.side == Protection::Buyer) ? -1.0 : 1.0;
42  Rate swapSpread = arguments_.swap->runningSpread();
43  const Handle<YieldTermStructure>& yts =
44  termStructure_.empty() ? model_->irlgm1f(0)->termStructure() : termStructure_;
45 
46  Real riskyAnnuity = std::fabs(arguments_.swap->couponLegNPV() / swapSpread);
47  results_.riskyAnnuity = riskyAnnuity;
48 
49  // incorporate upfront amount
50 
51  swapSpread -= w * arguments_.swap->upfrontNPV() / riskyAnnuity;
52 
53  Size n = arguments_.swap->coupons().size();
54  t_ = Array(n + 1, 0.0);
55  G_ = Array(n + 1, 0.0);
56  Array C(n, 0.0), D(n, 0.0);
57 
58  if (arguments_.exercise->date(0) <= yts->referenceDate()) {
59  results_.value = 0.0;
60  return;
61  }
62 
63  tex_ = yts->timeFromReference(arguments_.exercise->date(0));
64  t_[0] = std::max(tex_, yts->timeFromReference(arguments_.swap->protectionStartDate()));
65 
66  Real accrualSettlementAmount = 0.0;
67  for (Size i = 0; i < n; ++i) {
68  boost::shared_ptr<FixedRateCoupon> cpn =
69  boost::dynamic_pointer_cast<FixedRateCoupon>(arguments_.swap->coupons()[i]);
70  QL_REQUIRE(cpn != NULL, "AnalyticLgmCdsOptionEngine: expected fixed rate coupon");
71  t_[i + 1] = yts->timeFromReference(cpn->date());
72  Real mid = (t_[i] + t_[i + 1]) / 2.0;
73  if (arguments_.swap->settlesAccrual()) {
74  Real accStartTime = i == 0 ? yts->timeFromReference(cpn->accrualStartDate()) : t_[i];
75  // mid > accStartTime practically always the case?
76  accrualSettlementAmount = mid > accStartTime ? swapSpread * cpn->accrualPeriod() * (mid - accStartTime) /
77  (t_[i + 1] - accStartTime)
78  : 0.0;
79  }
80  C[i] = ((1.0 - recoveryRate_) - accrualSettlementAmount) * yts->discount(mid) / yts->discount(tex_);
81  D[i] = swapSpread * cpn->accrualPeriod() * yts->discount(t_[i + 1]) / yts->discount(tex_);
82  }
83  G_[0] = -C[0];
84  for (Size i = 0; i < n - 1; ++i) {
85  G_[i + 1] = C[i] + D[i] - C[i + 1];
86  }
87  G_[n] = C[n - 1] + D[n - 1];
88 
89  // if a non knock-out payer option, add front end protection value
90  Real frontEndProtection = 0.0;
91  if (arguments_.side == Protection::Buyer && !arguments_.knocksOut) {
92  frontEndProtection = arguments_.swap->notional() * (1. - recoveryRate_) *
93  model_->crlgm1f(index_)->termStructure()->defaultProbability(tex_) * yts->discount(tex_);
94  }
95 
96  Brent b;
97  Real lambdaStar;
98  try {
99  lambdaStar = b.solve(boost::bind(&AnalyticLgmCdsOptionEngine::lambdaStarHelper, this, _1), 1.0E-6, 0.0, 0.01);
100  } catch (const std::exception& e) {
101  QL_FAIL("AnalyticLgmCdsOptionEngine, failed to compute lambdaStar, " << e.what());
102  }
103 
104  Real sum = 0.0;
105  for (Size i = 1; i < G_.size(); ++i) {
106  Real strike = model_->crlgm1fS(index_, ccy_, tex_, t_[i], lambdaStar, 0.0).second /
107  model_->crlgm1fS(index_, ccy_, tex_, t_[0], lambdaStar, 0.0).second;
108  sum += G_[i] * Ei(w, strike, i) * yts->discount(tex_);
109  }
110 
111  results_.value = arguments_.swap->notional() * sum + frontEndProtection;
112 
113 } // calculate

115 Real AnalyticLgmCdsOptionEngine::Ei(const Real w, const Real strike, const Size i) const {
116  Real pS = model_->crlgm1f(index_)->termStructure()->survivalProbability(t_[0]);
117  Real pT = model_->crlgm1f(index_)->termStructure()->survivalProbability(t_[i]);
118  // slight generalization of Lichters, Stamm, Gallagher 11.2.1
119  // with t < S, SSRN: https://ssrn.com/abstract=2246054
120  Real sigma = sqrt(model_->crlgm1f(index_)->zeta(tex_)) *
121  (model_->crlgm1f(index_)->H(t_[i]) - model_->crlgm1f(index_)->H(t_[0]));
122  Real dp = (std::log(pT / (strike * pS)) / sigma + 0.5 * sigma);
123  Real dm = dp - sigma;
124  CumulativeNormalDistribution N;
125  return w * (pT * N(w * dp) - pS * strike * N(w * dm));
126 }

128 Real AnalyticLgmCdsOptionEngine::lambdaStarHelper(const Real lambda) const {
129  Real sum = 0.0;
130  for (Size i = 0; i < G_.size(); ++i) {
131  Real S = model_->crlgm1fS(index_, ccy_, tex_, t_[i], lambda, 0.0).second /
132  model_->crlgm1fS(index_, ccy_, tex_, t_[0], lambda, 0.0).second;
133  sum += G_[i] * S;
134  }
135  return sum;
136 }
