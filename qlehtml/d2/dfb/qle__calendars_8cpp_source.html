43 void checkCalendars(const std::vector<Date>& expectedHolidays, const std::vector<Date>& testHolidays) {
44 
45  for (Size i = 0; i < expectedHolidays.size(); i++) {
46  if (testHolidays[i] != expectedHolidays[i])
47  BOOST_FAIL("expected holiday was " << expectedHolidays[i] << " while calculated holiday is "
48  << testHolidays[i]);
49  }
50 }
51 } // namespace check
