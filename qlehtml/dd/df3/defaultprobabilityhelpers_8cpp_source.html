97 void CdsHelper::initializeDates() {
98 
99  // For CDS, the standard day counter is Actual/360 and the final period coupon accrual includes the maturity date.
100  // If the main day counter is Act/360 and no lastPeriodDayCounter_ is given, default to Act/360 including last.
101  Actual360 standardDayCounter;
102  if (lastPeriodDayCounter_.empty()) {
103  lastPeriodDayCounter_ = dayCounter_ == standardDayCounter ? Actual360(true) : dayCounter_;
104  }
105 
106  protectionStart_ = evaluationDate_ + settlementDays_;
107 
108  Date startDate = startDate_ == Date() ? protectionStart_ : startDate_;
109  // Only adjust start date if rule is not CDS or CDS2015. Unsure about OldCDS.
110  if (rule_ != DateGeneration::CDS && rule_ != DateGeneration::CDS2015) {
111  startDate = calendar_.adjust(startDate, paymentConvention_);
112  }
113 
114  Date endDate;
115  if (rule_ == DateGeneration::CDS2015 || rule_ == DateGeneration::CDS || rule_ == DateGeneration::OldCDS) {
116  Date refDate = startDate_ == Date() ? evaluationDate_ : startDate_;
117  endDate = cdsMaturity(refDate, tenor_, rule_);
118  } else {
119  // Keep the old logic here
120  Date refDate = startDate_ == Date() ? protectionStart_ : startDate_ + settlementDays_;
121  endDate = refDate + tenor_;
122  }
123 
124  schedule_ = MakeSchedule()
125  .from(startDate)
126  .to(endDate)
127  .withFrequency(frequency_)
128  .withCalendar(calendar_)
129  .withConvention(paymentConvention_)
130  .withTerminationDateConvention(Unadjusted)
131  .withRule(rule_);
132  earliestDate_ = schedule_.dates().front();
133  latestDate_ = calendar_.adjust(schedule_.dates().back(), paymentConvention_);
134 }
