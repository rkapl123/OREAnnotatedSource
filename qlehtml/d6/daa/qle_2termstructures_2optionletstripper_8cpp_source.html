29 OptionletStripper::OptionletStripper(const ext::shared_ptr<QuantExt::CapFloorTermVolSurface>& termVolSurface,
30  const ext::shared_ptr<IborIndex>& iborIndex,
31  const Handle<YieldTermStructure>& discount, const VolatilityType type,
32  const Real displacement)
33  : termVolSurface_(termVolSurface), iborIndex_(iborIndex), discount_(discount),
34  nStrikes_(termVolSurface->strikes().size()), volatilityType_(type), displacement_(displacement) {
35 
36  if (volatilityType_ == Normal) {
37  QL_REQUIRE(displacement_ == 0.0, "non-null displacement is not allowed with Normal model");
38  }
39 
40  registerWith(termVolSurface);
41  registerWith(iborIndex_);
42  registerWith(discount_);
43  registerWith(Settings::instance().evaluationDate());
44 
45  Period indexTenor = iborIndex_->tenor();
46  QL_REQUIRE(termVolSurface->optionTenors().size() > 0, "OptionletStripper: No OptionTenors provided.");
47  Period maxCapFloorTenor = termVolSurface->optionTenors().back();
48 
49  // optionlet tenors and capFloor lengths
50  optionletTenors_.push_back(indexTenor);
51  capFloorLengths_.push_back(optionletTenors_.back() + indexTenor);
52  QL_REQUIRE(maxCapFloorTenor >= capFloorLengths_.back(),
53  "too short (" << maxCapFloorTenor << ") capfloor term vol termVolSurface");
54  Period nextCapFloorLength = capFloorLengths_.back() + indexTenor;
55  while (nextCapFloorLength <= maxCapFloorTenor) {
56  optionletTenors_.push_back(capFloorLengths_.back());
57  capFloorLengths_.push_back(nextCapFloorLength);
58  nextCapFloorLength += indexTenor;
59  }
60  nOptionletTenors_ = optionletTenors_.size();
61 
62  optionletVolatilities_ = vector<vector<Volatility> >(nOptionletTenors_, vector<Volatility>(nStrikes_));
63  optionletStrikes_ = vector<vector<Rate> >(nOptionletTenors_, termVolSurface->strikes());
64  optionletDates_ = vector<Date>(nOptionletTenors_);
65  optionletTimes_ = vector<Time>(nOptionletTenors_);
66  atmOptionletRate_ = vector<Rate>(nOptionletTenors_);
67  optionletPaymentDates_ = vector<Date>(nOptionletTenors_);
68  optionletAccrualPeriods_ = vector<Time>(nOptionletTenors_);
69 }

132 void OptionletStripper::populateDates() const {
133 
134  Date referenceDate = termVolSurface_->referenceDate();
135  DayCounter dc = termVolSurface_->dayCounter();
136  boost::shared_ptr<BlackCapFloorEngine> dummyEngine =
137  boost::make_shared<BlackCapFloorEngine>(iborIndex_->forwardingTermStructure(), 0.20, dc);
138 
139  for (Size i = 0; i < nOptionletTenors_; ++i) {
140  CapFloor dummyCap =
141  MakeCapFloor(CapFloor::Cap, capFloorLengths_[i], iborIndex_, 0.04, 0 * Days).withPricingEngine(dummyEngine);
142  boost::shared_ptr<FloatingRateCoupon> lastCoupon = dummyCap.lastFloatingRateCoupon();
143  optionletDates_[i] = lastCoupon->fixingDate();
144  optionletPaymentDates_[i] = lastCoupon->date();
145  optionletAccrualPeriods_[i] = lastCoupon->accrualPeriod();
146  optionletTimes_[i] = dc.yearFraction(referenceDate, optionletDates_[i]);
147  atmOptionletRate_[i] = lastCoupon->indexFixing();
148  }
149 }
