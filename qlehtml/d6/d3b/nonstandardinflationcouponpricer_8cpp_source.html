72 Real NonStandardYoYInflationCouponPricer::optionletRate(Option::Type optionType, Real effStrike) const {
73  Date fixingDate = coupon_->fixingDate();
74  if (fixingDate <= Settings::instance().evaluationDate()) {
75  // the amount is determined
76  Real a, b;
77  if (optionType == Option::Call) {
78  a = coupon_->indexFixing();
79  b = effStrike;
80  } else {
81  a = effStrike;
82  b = coupon_->indexFixing();
83  }
84  return std::max(a - b, 0.0);
85  } else {
86  // not yet determined, use Black/DD1/Bachelier/whatever from Impl
87  QL_REQUIRE(!capletVolatility().empty(), "missing optionlet volatility");
88  Real stdDev = std::sqrt(capletVolatility()->totalVariance(fixingDate, effStrike));
89  return optionletPriceImp(optionType, effStrike, adjustedFixing(), stdDev);
90  }
91 }

102 void NonStandardYoYInflationCouponPricer::initialize(const InflationCoupon& coupon) {
103  coupon_ = dynamic_cast<const NonStandardYoYInflationCoupon*>(&coupon);
104  QL_REQUIRE(coupon_, "year-on-year inflation coupon needed");
105  gearing_ = coupon_->gearing();
106  spread_ = coupon_->spread();
107  paymentDate_ = coupon_->date();
108  rateCurve_ = !nominalTermStructure_.empty() ? nominalTermStructure_
109  : ext::dynamic_pointer_cast<ZeroInflationIndex>(coupon.index())
110  ->zeroInflationTermStructure()
111  ->nominalTermStructure();
112 
113  // past or future fixing is managed in YoYInflationIndex::fixing()
114  // use yield curve from index (which sets discount)
115 
116  discount_ = 1.0;
117  if (paymentDate_ > rateCurve_->referenceDate()) {
118  if (rateCurve_.empty()) {
119  // allow to extract rates, but mark the discount as invalid for prices
120  discount_ = Null<Real>();
121  } else {
122  discount_ = rateCurve_->discount(paymentDate_);
123  }
124  }
125 }

132 Rate NonStandardYoYInflationCouponPricer::swapletRate() const {
133  // This way we do not require the index to have
134  // a yield curve, i.e. we do not get the problem
135  // that a discounting-instrument-pricer is used
136  // with a different yield curve
137  return gearing_ * adjustedFixing() + spread_;
138 }
