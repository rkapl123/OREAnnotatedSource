30 InterpolatedSmileSection::InterpolatedSmileSection(Real spot, Real rd, Real rf, Time t,
31  const std::vector<Real>& strikes,
32  const std::vector<Volatility>& vols, InterpolationMethod method,
33  bool flatExtrapolation)
34  : FxSmileSection(spot, rd, rf, t), strikes_(strikes), vols_(vols), flatExtrapolation_(flatExtrapolation) {
35 
36  if (method == InterpolationMethod::Linear)
37  interpolator_ = Linear().interpolate(strikes_.begin(), strikes_.end(), vols_.begin());
38  else if (method == InterpolationMethod::NaturalCubic)
39  interpolator_ =
40  Cubic(CubicInterpolation::Kruger, true).interpolate(strikes_.begin(), strikes_.end(), vols_.begin());
41  else if (method == InterpolationMethod::FinancialCubic)
42  interpolator_ = Cubic(CubicInterpolation::Kruger, true, CubicInterpolation::SecondDerivative, 0.0,
43  CubicInterpolation::FirstDerivative)
44  .interpolate(strikes_.begin(), strikes_.end(), vols_.begin());
45  else {
46  QL_FAIL("Invalid method " << (int)method);
47  }
48 }

50 Volatility InterpolatedSmileSection::volatility(Real strike) const {
51  if (flatExtrapolation_) {
52  if (strike < strikes_.front())
53  return vols_.front();
54  else if (strike > strikes_.back())
55  return vols_.back();
56  }
57  // and then call the interpolator with extrapolation on
58  return interpolator_(strike, true);
59 }

61 BlackVolatilitySurfaceDelta::BlackVolatilitySurfaceDelta(
62  Date referenceDate, const std::vector<Date>& dates, const std::vector<Real>& putDeltas,
63  const std::vector<Real>& callDeltas, bool hasAtm, const Matrix& blackVolMatrix, const DayCounter& dayCounter,
64  const Calendar& cal, const Handle<Quote>& spot, const Handle<YieldTermStructure>& domesticTS,
65  const Handle<YieldTermStructure>& foreignTS, DeltaVolQuote::DeltaType dt, DeltaVolQuote::AtmType at,
66  boost::optional<DeltaVolQuote::DeltaType> atmDeltaType, const Period& switchTenor, DeltaVolQuote::DeltaType ltdt,
67  DeltaVolQuote::AtmType ltat, boost::optional<QuantLib::DeltaVolQuote::DeltaType> longTermAtmDeltaType,
68  InterpolatedSmileSection::InterpolationMethod im, bool flatExtrapolation)
69  : BlackVolatilityTermStructure(referenceDate, cal, Following, dayCounter), dates_(dates), times_(dates.size(), 0),
70  putDeltas_(putDeltas), callDeltas_(callDeltas), hasAtm_(hasAtm), spot_(spot), domesticTS_(domesticTS),
71  foreignTS_(foreignTS), dt_(dt), at_(at), atmDeltaType_(atmDeltaType), switchTenor_(switchTenor), ltdt_(ltdt),
72  ltat_(ltat), longTermAtmDeltaType_(longTermAtmDeltaType), interpolationMethod_(im),
73  flatExtrapolation_(flatExtrapolation) {
74 
75  // If ATM delta type is not given, set it to dt
76  if (!atmDeltaType_)
77  atmDeltaType_ = dt_;
78  if (!longTermAtmDeltaType_)
79  longTermAtmDeltaType_ = ltdt_;
80 
81  // set switch time
82  switchTime_ = switchTenor_ == 0 * Days ? QL_MAX_REAL : timeFromReference(optionDateFromTenor(switchTenor));
83 
84  QL_REQUIRE(dates.size() > 1, "at least 1 date required");
85  // this has already been done for dates
86  for (Size i = 0; i < dates.size(); i++) {
87  QL_REQUIRE(referenceDate < dates[i], "Dates must be greater than reference date");
88  times_[i] = timeFromReference(dates[i]);
89  if (i > 0) {
90  QL_REQUIRE(times_[i] > times_[i - 1], "dates must be sorted unique!");
91  }
92  }
93 
94  // check size of matrix
95  Size n = putDeltas.size() + (hasAtm ? 1 : 0) + callDeltas.size();
96  QL_REQUIRE(n > 0, "Need at least one delta");
97  QL_REQUIRE(blackVolMatrix.columns() == n, "Invalid number of columns in blackVolMatrix, got "
98  << blackVolMatrix.columns() << " but have " << n << " deltas");
99  QL_REQUIRE(blackVolMatrix.rows() == dates.size(), "Invalid number of rows in blackVolMatrix, got "
100  << blackVolMatrix.rows() << " but have " << dates.size()
101  << " dates");
102 
103  // build interpolators for each delta
104  // TODO: template this so it can be changed
105  bool forceMonotoneVariance = false;
106  for (Size i = 0; i < n; i++) {
107  vector<Volatility> vols(dates.size());
108  for (Size j = 0; j < dates.size(); j++) {
109  vols[j] = blackVolMatrix[j][i];
110  }
111 
112  // BlackVarianceCurve will make a local copy of vols and dates
113  interpolators_.push_back(
114  boost::make_shared<BlackVarianceCurve>(referenceDate, dates, vols, dayCounter, forceMonotoneVariance));
115  }
116 
117  // register
118  registerWith(spot_);
119  registerWith(domesticTS_);
120  registerWith(foreignTS_);
121 }

123 boost::shared_ptr<FxSmileSection> BlackVolatilitySurfaceDelta::blackVolSmile(Time t) const {
124 
125  Real spot = spot_->value();
126  DiscountFactor dDiscount = domesticTS_->discount(t);
127  DiscountFactor fDiscount = foreignTS_->discount(t);
128 
129  DeltaVolQuote::AtmType at;
130  DeltaVolQuote::DeltaType dt;
131  DeltaVolQuote::DeltaType atmDt;
132  if (t < switchTime_ && !close_enough(t, switchTime_)) {
133  at = at_;
134  dt = dt_;
135  atmDt = *atmDeltaType_;
136  } else {
137  at = ltat_;
138  dt = ltdt_;
139  atmDt = *longTermAtmDeltaType_;
140  }
141 
142  // Store smile section in map. Use strikes as key and vols as values for automatic sorting by strike.
143  // If we have already have a strike from a previous delta, we do not overwrite it.
144  auto comp = [](Real a, Real b) { return !close(a, b) && a < b; };
145  map<Real, Real, decltype(comp)> smileSection(comp);
146  Size i = 0;
147  for (Real delta : putDeltas_) {
148  Real vol = interpolators_.at(i)->blackVol(t, 1, true);
149  try {
150  BlackDeltaCalculator bdc(Option::Put, dt, spot, dDiscount, fDiscount, vol * sqrt(t));
151  Real strike = bdc.strikeFromDelta(delta);
152  if (smileSection.count(strike) == 0)
153  smileSection[strike] = vol;
154  } catch (const std::exception& e) {
155  QL_FAIL("BlackVolatilitySurfaceDelta: Error during calculating put strike at delta " << delta << ": "
156  << e.what());
157  }
158  i++;
159  }
160  if (hasAtm_) {
161  Real vol = interpolators_.at(i)->blackVol(t, 1, true);
162  try {
163  BlackDeltaCalculator bdc(Option::Put, atmDt, spot, dDiscount, fDiscount, vol * sqrt(t));
164  Real strike = bdc.atmStrike(at);
165  if (smileSection.count(strike) == 0)
166  smileSection[strike] = vol;
167  } catch (const std::exception& e) {
168  QL_FAIL("BlackVolatilitySurfaceDelta: Error during calculating atm strike: " << e.what());
169  }
170  i++;
171  }
172  for (Real delta : callDeltas_) {
173  Real vol = interpolators_.at(i)->blackVol(t, 1, true);
174  try {
175  BlackDeltaCalculator bdc(Option::Call, dt, spot, dDiscount, fDiscount, vol * sqrt(t));
176  Real strike = bdc.strikeFromDelta(delta);
177  if (smileSection.count(strike) == 0)
178  smileSection[strike] = vol;
179  } catch (const std::exception& e) {
180  QL_FAIL("BlackVolatilitySurfaceDelta: Error during calculating call strike at delta " << delta << ": "
181  << e.what());
182  }
183  i++;
184  }
185 
186  // sort and extract to vectors
187  vector<Real> strikes;
188  strikes.reserve(smileSection.size());
189  vector<Real> vols;
190  vols.reserve(smileSection.size());
191  for (const auto& kv : smileSection) {
192  strikes.push_back(kv.first);
193  vols.push_back(kv.second);
194  }
195 
196  // now build smile from strikes and vols
197  QL_REQUIRE(!vols.empty(),
198  "BlackVolatilitySurfaceDelta::blackVolSmile(" << t << "): no strikes given, this is unexpected.");
199  if (vols.size() == 1) {
200  // handle the situation that we only have one strike (might occur for e.g. t=0)
201  return boost::make_shared<ConstantSmileSection>(vols.front());
202  } else {
203  // we have at least two strikes
204  return boost::make_shared<InterpolatedSmileSection>(spot, dDiscount, fDiscount, t, strikes, vols,
205  interpolationMethod_, flatExtrapolation_);
206  }
207 }

217 Volatility BlackVolatilitySurfaceDelta::blackVolImpl(Time t, Real strike) const {
218  // If asked for strike == 0, just return the ATM value.
219  if (strike == 0 || strike == Null<Real>()) {
220  if (hasAtm_) {
221  // ask the ATM interpolator directly
222  return interpolators_[putDeltas_.size()]->blackVol(t, Null<Real>(), true);
223  } else {
224  // set strike to be fwd and we will return ATMF
225  strike = forward(t);
226  }
227  }
228  return blackVolSmile(t)->volatility(strike);
229 }
