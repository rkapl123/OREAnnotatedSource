41 void ExposureAllocator::build() {
42  LOG("Compute allocated trade exposures");
43 
44  for (string nettingSetId : nettedExposureCube_->ids()) {
45 
46  for (auto trade : portfolio_->trades()) {
47  string nid = trade->envelope().nettingSetId();
48  if (nid != nettingSetId)
49  continue;
50  string tid = trade->id();
51 
52  for (Date date : tradeExposureCube_->dates()) {
53  for (Size k = 0; k < tradeExposureCube_->samples(); ++k) {
54  tradeExposureCube_->set(calculateAllocatedEpe(tid, nid, date, k),
55  tid, date, k, allocatedTradeEpeIndex_);
56  tradeExposureCube_->set(calculateAllocatedEne(tid, nid, date, k),
57  tid, date, k, allocatedTradeEneIndex_);
58  }
59  }
60  }
61  }
62  LOG("Completed calculating allocated trade exposures");
63 }

65 RelativeFairValueNetExposureAllocator::RelativeFairValueNetExposureAllocator(
66  const boost::shared_ptr<Portfolio>& portfolio,
67  const boost::shared_ptr<NPVCube>& tradeExposureCube,
68  const boost::shared_ptr<NPVCube>& nettedExposureCube,
69  const boost::shared_ptr<NPVCube>& npvCube,
70  const Size allocatedTradeEpeIndex, const Size allocatedTradeEneIndex,
71  const Size tradeEpeIndex, const Size tradeEneIndex,
72  const Size nettingSetEpeIndex, const Size nettingSetEneIndex)
73  : ExposureAllocator(portfolio, tradeExposureCube, nettedExposureCube,
74  allocatedTradeEpeIndex, allocatedTradeEneIndex,
75  tradeEpeIndex, tradeEneIndex,
76  nettingSetEpeIndex, nettingSetEneIndex) {
77  for (Size i = 0; i < portfolio->ids().size(); ++i) {
78  string tradeId = portfolio_->ids()[i];
79  string nettingSetId = portfolio->trades()[i]->envelope().nettingSetId();
80  if (nettingSetPositiveValueToday_.find(nettingSetId) == nettingSetPositiveValueToday_.end()) {
81  nettingSetPositiveValueToday_[nettingSetId] = 0.0;
82  nettingSetNegativeValueToday_[nettingSetId] = 0.0;
83  }
84  Real npv = npvCube->getT0(i);
85  tradeValueToday_[tradeId] = npv;
86  if (npv > 0)
87  nettingSetPositiveValueToday_[nettingSetId] += npv;
88  else
89  nettingSetNegativeValueToday_[nettingSetId] += npv;
90  }
91 }

93 Real RelativeFairValueNetExposureAllocator::calculateAllocatedEpe(const string& tid, const string& nid,
94  const Date& date, const Size sample) {
95  // FIXME: What to do when either the pos. or neg. netting set value is zero?
96  QL_REQUIRE(nettingSetPositiveValueToday_[nid] > 0.0, "non-zero positive NPV expected");
97  Real netEPE = nettedExposureCube_->get(nid, date, sample, nettingSetEpeIndex_);
98  return netEPE * std::max(tradeValueToday_[tid], 0.0) / nettingSetPositiveValueToday_[nid];
99 }

101 Real RelativeFairValueNetExposureAllocator::calculateAllocatedEne(const string& tid, const string& nid,
102  const Date& date, const Size sample) {
103  // FIXME: What to do when either the pos. or neg. netting set value is zero?
104  QL_REQUIRE(nettingSetNegativeValueToday_[nid] > 0.0, "non-zero negative NPV expected");
105  Real netENE = nettedExposureCube_->get(nid, date, sample, nettingSetEneIndex_);
106  return netENE * -std::max(-tradeValueToday_[tid], 0.0) / nettingSetPositiveValueToday_[nid];
107 }

193 ExposureAllocator::AllocationMethod parseAllocationMethod(const string& s) {
194  static map<string, ExposureAllocator::AllocationMethod> m = {
195  {"None", ExposureAllocator::AllocationMethod::None},
196  {"Marginal", ExposureAllocator::AllocationMethod::Marginal},
197  {"RelativeFairValueGross", ExposureAllocator::AllocationMethod::RelativeFairValueGross},
198  {"RelativeFairValueNet", ExposureAllocator::AllocationMethod::RelativeFairValueNet},
199  {"RelativeXVA", ExposureAllocator::AllocationMethod::RelativeXVA},
200  };
201 
202  auto it = m.find(s);
203  if (it != m.end()) {
204  return it->second;
205  } else {
206  QL_FAIL("AllocationMethod \"" << s << "\" not recognized");
207  }
208 }
209 
210 std::ostream& operator<<(std::ostream& out, ExposureAllocator::AllocationMethod m) {
211  if (m == ExposureAllocator::AllocationMethod::None)
212  out << "None";
213  else if (m == ExposureAllocator::AllocationMethod::Marginal)
214  out << "Marginal";
215  else if (m == ExposureAllocator::AllocationMethod::RelativeFairValueGross)
216  out << "RelativeFairValueGross";
217  else if (m == ExposureAllocator::AllocationMethod::RelativeFairValueNet)
218  out << "RelativeFairValueNet";
219  else if (m == ExposureAllocator::AllocationMethod::RelativeXVA)
220  out << "RelativeXVA";
221  else
222  QL_FAIL("Allocation method not covered");
223  return out;
224 }
