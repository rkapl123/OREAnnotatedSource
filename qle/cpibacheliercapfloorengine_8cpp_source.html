44 void CPIBachelierCapFloorEngine::calculate() const {
45  Date maturity = arguments_.payDate;
46 
47  auto index = arguments_.infIndex;
48  DiscountFactor d = arguments_.nominal * discountCurve_->discount(maturity);
49 
50  bool isInterpolated = arguments_.observationInterpolation == CPI::Linear ||
51  (arguments_.observationInterpolation == CPI::AsIndex && arguments_.infIndex->interpolated());
52 
53  Date optionObservationDate = arguments_.payDate - arguments_.observationLag;
54 
55  Date optionBaseDate = arguments_.startDate - arguments_.observationLag;
56 
57  Real optionBaseFixing = arguments_.baseCPI == Null<Real>()
58  ? ZeroInflation::cpiFixing(arguments_.infIndex.currentLink(), arguments_.startDate,
59  arguments_.observationLag, isInterpolated)
60  : arguments_.baseCPI;
61 
62  Real atmCPIFixing = ZeroInflation::cpiFixing(arguments_.infIndex.currentLink(), maturity, arguments_.observationLag,
63  isInterpolated);
64 
65  Time timeToMaturityFromInception =
66  inflationYearFraction(index->frequency(), isInterpolated, index->zeroInflationTermStructure()->dayCounter(),
67  optionBaseDate, optionObservationDate);
68 
69  Real atmGrowth = atmCPIFixing / optionBaseFixing;
70  Real strike = std::pow(1.0 + arguments_.strike, timeToMaturityFromInception);
71 
72  auto lastKnownFixingDate =
73  ZeroInflation::lastAvailableFixing(*index.currentLink(), volatilitySurface_->referenceDate());
74  auto observationPeriod = inflationPeriod(optionObservationDate, index->frequency());
75  auto requiredFixing = isInterpolated ? observationPeriod.first : observationPeriod.second + 1 * Days;
76 
77  // if time from base <= 0 the fixing is already known and stdDev is zero, return the intrinsic value
78  Real stdDev = 0.0;
79  if (requiredFixing > lastKnownFixingDate) {
80  // For reading volatility in the current market volatiltiy structure
81  // baseFixingSwap(T0) * pow(1 + strikeRate(T0), T-T0) = StrikeIndex = baseFixing(t) * pow(1 + strikeRate(t),
82  // T-t), solve for strikeRate(t):
83  auto surfaceBaseFixing = ZeroInflation::cpiFixing(index.currentLink(), volatilitySurface_->baseDate(), 0 * Days,
84  volatilitySurface_->indexIsInterpolated());
85  auto ttmFromSurfaceBaseDate = inflationYearFraction(
86  volatilitySurface_->frequency(), volatilitySurface_->indexIsInterpolated(),
87  index->zeroInflationTermStructure()->dayCounter(), volatilitySurface_->baseDate(), optionObservationDate);
88  Real strikeZeroRate = pow(optionBaseFixing / surfaceBaseFixing * strike, 1.0 / ttmFromSurfaceBaseDate) - 1.0;
89  if (measureTimeFromLastAvailableFixing_) {
90  auto vol = volatilitySurface_->volatility(optionObservationDate, strikeZeroRate, 0 * Days);
91  auto ttm =
92  inflationYearFraction(volatilitySurface_->frequency(), volatilitySurface_->indexIsInterpolated(),
93  volatilitySurface_->dayCounter(), lastKnownFixingDate, optionObservationDate);
94  stdDev = std::sqrt(ttm * vol * vol);
95  } else {
96  stdDev = std::sqrt(volatilitySurface_->totalVariance(optionObservationDate, strikeZeroRate, 0 * Days));
97  }
98  }
99  results_.value = bachelierBlackFormula(arguments_.type, strike, atmGrowth, stdDev, d);
100 }
