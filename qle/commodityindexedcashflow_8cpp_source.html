25 CommodityIndexedCashFlow::CommodityIndexedCashFlow(
26  Real quantity, const Date& startDate, const Date& endDate, const ext::shared_ptr<CommodityIndex>& index,
27  Natural paymentLag, const Calendar& paymentCalendar, BusinessDayConvention paymentConvention, Natural pricingLag,
28  const Calendar& pricingLagCalendar, Real spread, Real gearing, bool payInAdvance, bool isInArrears,
29  bool useFuturePrice, bool useFutureExpiryDate, Natural futureMonthOffset,
30  const ext::shared_ptr<FutureExpiryCalculator>& calc, const QuantLib::Date& paymentDateOverride,
31  const QuantLib::Date& pricingDateOverride)
32  : quantity_(quantity), pricingDate_(pricingDateOverride), paymentDate_(paymentDateOverride), index_(index),
33  spread_(spread), gearing_(gearing), useFuturePrice_(useFuturePrice), useFutureExpiryDate_(useFutureExpiryDate),
34  futureMonthOffset_(futureMonthOffset), periodQuantity_(quantity) {
35 
36  // Derive the pricing date if an explicit override has not been provided
37  if (pricingDate_ == Date()) {
38  pricingDate_ = isInArrears ? endDate : startDate;
39  if (!useFuturePrice_ || !useFutureExpiryDate_) {
40  // We just use the pricing date rules to get the pricing date
41  pricingDate_ = pricingLagCalendar.advance(pricingDate_, -static_cast<Integer>(pricingLag), Days, Preceding);
42  } else {
43  // We need to use the expiry date of the future contract
44  QL_REQUIRE(calc, "CommodityIndexedCashFlow needs a valid future "
45  << "expiry calculator when using first future");
46  pricingDate_ = calc->expiryDate(pricingDate_, futureMonthOffset_);
47  }
48  }
49 
50  // Derive the payment date if an explicit override has not been provided
51  if (paymentDate_ == Date()) {
52  paymentDate_ = payInAdvance ? startDate : endDate;
53  paymentDate_ = paymentCalendar.advance(paymentDate_, paymentLag, Days, paymentConvention);
54  }
55 
56  // the pricing date has to lie on or before the payment date
57  pricingDate_ = index_->fixingCalendar().adjust(std::min(paymentDate_, pricingDate_), Preceding);
58 
59  // We may not need the month and year if we are not using a future settlement price but get them
60  // and pass them here in any case to init
61  Date ref = isInArrears ? endDate : startDate;
62 
63  init(calc, ref);
64 }

83 void CommodityIndexedCashFlow::init(const ext::shared_ptr<FutureExpiryCalculator>& calc, const Date& contractDate) {
84 
85  QL_REQUIRE(paymentDate_ >= pricingDate_, "Expected that the payment date ("
86  << io::iso_date(paymentDate_)
87  << ") would be on or after the pricing date ("
88  << io::iso_date(pricingDate_) << ")");
89  QL_REQUIRE(index_->isValidFixingDate(pricingDate_), "Pricing date " << io::iso_date(pricingDate_)
90  << " is not valid for commodity "
91  << index_->underlyingName());
92 
93  // If we are using the future settlement price as the reference price, then we need to create the
94  // relevant "future index" here and update the cashflow's index with it.
95  if (useFuturePrice_) {
96  QL_REQUIRE(calc, "CommodityIndexedCashFlow needs a valid future expiry calculator when using "
97  << "the future settlement price as reference price");
98  Date expiry = calc->expiryDate(contractDate, futureMonthOffset_);
99  QL_REQUIRE(expiry >= pricingDate_, "Expected that the expiry date (" << io::iso_date(expiry) <<
100  ") for commodity " << index_->underlyingName() << " future with contract date of " <<
101  io::iso_date(contractDate) << " and with month offset of " << futureMonthOffset_ <<
102  " would be on or after the pricing date (" << io::iso_date(pricingDate_) << ")");
103  index_ = index_->clone(expiry);
104  }
105 
106  registerWith(index_);
107 }

216 CommodityIndexedLeg::operator Leg() const {
217 
218  // Number of commodity indexed cashflows
219  Size numberCashflows = schedule_.size() - 1;
220 
221  // Initial consistency checks
222  QL_REQUIRE(!quantities_.empty(), "No quantities given");
223  QL_REQUIRE(quantities_.size() <= numberCashflows,
224  "Too many quantities (" << quantities_.size() << "), only " << numberCashflows << " required");
225  if (useFuturePrice_) {
226  QL_REQUIRE(calc_, "CommodityIndexedCashFlow needs a valid future expiry calculator when using first future");
227  }
228  if (!pricingDates_.empty()) {
229  QL_REQUIRE(pricingDates_.size() == numberCashflows, "Expected the number of explicit pricing dates ("
230  << pricingDates_.size()
231  << ") to equal the number of calculation periods ("
232  << numberCashflows << ")");
233  }
234  if (!paymentDates_.empty()) {
235  QL_REQUIRE(paymentDates_.size() == numberCashflows, "Expected the number of explicit payment dates ("
236  << paymentDates_.size()
237  << ") to equal the number of calculation periods ("
238  << numberCashflows << ")");
239  }
240 
241  // If pay at maturity, populate payment date.
242  Date paymentDate;
243  if (payAtMaturity_) {
244  paymentDate = paymentCalendar_.advance(schedule_.dates().back(), paymentLag_ * Days, paymentConvention_);
245  }
246 
247  // Leg to hold the result
248  Leg leg;
249  leg.reserve(numberCashflows);
250  for (Size i = 0; i < numberCashflows; ++i) {
251 
252  Date start = schedule_.date(i);
253  Date end = schedule_.date(i + 1);
254  Real quantity = detail::get(quantities_, i, 1.0);
255  Real spread = detail::get(spreads_, i, 0.0);
256  Real gearing = detail::get(gearings_, i, 1.0);
257  Date pricingDate = detail::get(pricingDates_, i, Date());
258 
259  // If explicit payment dates provided, use them.
260  if (!paymentDates_.empty()) {
261  paymentDate = paymentDates_[i];
262  }
263 
264  leg.push_back(ext::make_shared<CommodityIndexedCashFlow>(
265  quantity, start, end, index_, paymentLag_, paymentCalendar_, paymentConvention_, pricingLag_,
266  pricingLagCalendar_, spread, gearing, payInAdvance_, inArrears_, useFuturePrice_, useFutureExpiryDate_,
267  futureMonthOffset_, calc_, paymentDate, pricingDate));
268  }
269 
270  return leg;
271 }
