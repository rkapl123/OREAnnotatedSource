26CommodityIndexedCashFlow::CommodityIndexedCashFlow(
27 Real quantity, const Date& startDate, const Date& endDate, const ext::shared_ptr<CommodityIndex>& index,
28 Natural paymentLag, const Calendar& paymentCalendar, BusinessDayConvention paymentConvention, Natural pricingLag,
29 const Calendar& pricingLagCalendar, Real spread, Real gearing, PaymentTiming paymentTiming, bool isInArrears,
30 bool useFuturePrice, bool useFutureExpiryDate, Natural futureMonthOffset,
31 const ext::shared_ptr<FutureExpiryCalculator>& calc, const QuantLib::Date& paymentDateOverride,
32 const QuantLib::Date& pricingDateOverride, QuantLib::Natural dailyExpiryOffset)
33 : CommodityCashFlow(quantity, spread, gearing, useFuturePrice, index), pricingDate_(pricingDateOverride),
34 paymentDate_(paymentDateOverride), useFutureExpiryDate_(useFutureExpiryDate),
35 futureMonthOffset_(futureMonthOffset), periodQuantity_(quantity), dailyExpiryOffset_(dailyExpiryOffset) {
36
37 // Derive the pricing date if an explicit override has not been provided
38 if (pricingDate_ == Date()) {
39 pricingDate_ = isInArrears ? endDate : startDate;
40 if (!useFuturePrice_ || !useFutureExpiryDate_) {
41 // We just use the pricing date rules to get the pricing date
42 pricingDate_ = pricingLagCalendar.advance(pricingDate_, -static_cast<Integer>(pricingLag), Days, Preceding);
43 } else {
44 // We need to use the expiry date of the future contract
45 QL_REQUIRE(calc, "CommodityIndexedCashFlow needs a valid future "
46 << "expiry calculator when using first future");
47 Date expiry = calc->expiryDate(pricingDate_, futureMonthOffset_);
48 if (dailyExpiryOffset_ != Null<Natural>()) {
49 expiry = index_->fixingCalendar().advance(expiry, dailyExpiryOffset_ * Days);
50 }
51 pricingDate_ = expiry;
52 }
53 }
54
55 // We may not need the month and year if we are not using a future settlement price but get them
56 // and pass them here in any case to init
57 Date ref = isInArrears ? endDate : startDate;
58
59 init(calc, ref, paymentTiming, startDate, endDate, paymentLag, paymentConvention, paymentCalendar);
60}

77void CommodityIndexedCashFlow::init(const ext::shared_ptr<FutureExpiryCalculator>& calc, const Date& contractDate,
78 const PaymentTiming paymentTiming, const Date& startDate, const Date& endDate,
79 const Natural paymentLag, const BusinessDayConvention paymentConvention,
80 const Calendar& paymentCalendar) {
81
82 pricingDate_ = index_->fixingCalendar().adjust(pricingDate_, Preceding);
83
84 // If we are using the future settlement price as the reference price, then we need to create the
85 // relevant "future index" here and update the cashflow's index with it.
86 Date expiry;
87 if (useFuturePrice_) {
88 QL_REQUIRE(calc, "CommodityIndexedCashFlow needs a valid future expiry calculator when using "
89 << "the future settlement price as reference price");
90 expiry = calc->expiryDate(contractDate, futureMonthOffset_);
91 if (dailyExpiryOffset_ != Null<Natural>()) {
92 expiry = index_->fixingCalendar().advance(expiry, dailyExpiryOffset_ * Days);
93 }
94 index_ = index_->clone(expiry);
95 }
96
97 // Derive the payment date if an explicit override has not been provided
98 if (paymentDate_ == Date()) {
99 if (paymentTiming == PaymentTiming::InAdvance) {
100 QL_REQUIRE(startDate != Date(), "CommodityIndexedCashFlow: startDate is null, can not derive paymentDate.");
101 paymentDate_ = startDate;
102 } else if (paymentTiming == PaymentTiming::InArrears) {
103 QL_REQUIRE(endDate != Date(), "CommodityIndexedCashFlow: endDate is null, can not derive paymentDate.");
104 paymentDate_ = endDate;
105 } else if (paymentTiming == PaymentTiming::RelativeToExpiry) {
106 QL_REQUIRE(
107 expiry != Date(),
108 "CommodityIndexedCashFlow: payment relative to expiry date only possibly when future price is used.");
109 paymentDate_ = expiry;
110 }
111 paymentDate_ = paymentCalendar.advance(paymentDate_, paymentLag, Days, paymentConvention);
112 }
113
114 // the pricing date has to lie on or before the payment date
115 pricingDate_ = index_->fixingCalendar().adjust(std::min(paymentDate_, pricingDate_), Preceding);
116
117 registerWith(index_);
118}

232CommodityIndexedLeg::operator Leg() const {
233
234 // Number of commodity indexed cashflows
235 Size numberCashflows = schedule_.size() - 1;
236
237 // Initial consistency checks
238 QL_REQUIRE(!quantities_.empty(), "No quantities given");
239 QL_REQUIRE(quantities_.size() <= numberCashflows,
240 "Too many quantities (" << quantities_.size() << "), only " << numberCashflows << " required");
241 if (useFuturePrice_) {
242 QL_REQUIRE(calc_, "CommodityIndexedCashFlow needs a valid future expiry calculator when using first future");
243 }
244 if (!pricingDates_.empty()) {
245 QL_REQUIRE(pricingDates_.size() == numberCashflows, "Expected the number of explicit pricing dates ("
246 << pricingDates_.size()
247 << ") to equal the number of calculation periods ("
248 << numberCashflows << ")");
249 }
250 if (!paymentDates_.empty()) {
251 QL_REQUIRE(paymentDates_.size() == numberCashflows, "Expected the number of explicit payment dates ("
252 << paymentDates_.size()
253 << ") to equal the number of calculation periods ("
254 << numberCashflows << ")");
255 }
256
257 // If pay at maturity, populate payment date.
258 Date paymentDate;
259 if (payAtMaturity_) {
260 paymentDate = paymentCalendar_.advance(schedule_.dates().back(), paymentLag_ * Days, paymentConvention_);
261 }
262
263 // Leg to hold the result
264 Leg leg;
265 leg.reserve(numberCashflows);
266 for (Size i = 0; i < numberCashflows; ++i) {
267
268 Date start = schedule_.date(i);
269 Date end = schedule_.date(i + 1);
270 Real quantity = detail::get(quantities_, i, 1.0);
271 Real spread = detail::get(spreads_, i, 0.0);
272 Real gearing = detail::get(gearings_, i, 1.0);
273 Date pricingDate = detail::get(pricingDates_, i, Date());
274
275 // If explicit payment dates provided, use them.
276 if (!paymentDates_.empty()) {
277 paymentDate = paymentDates_[i];
278 }
279
280 leg.push_back(ext::make_shared<CommodityIndexedCashFlow>(
281 quantity, start, end, index_, paymentLag_, paymentCalendar_, paymentConvention_, pricingLag_,
282 pricingLagCalendar_, spread, gearing, paymentTiming_, inArrears_, useFuturePrice_, useFutureExpiryDate_,
283 futureMonthOffset_, calc_, paymentDate, pricingDate, dailyExpiryOffset_));
284 }
285
286 return leg;
287}
