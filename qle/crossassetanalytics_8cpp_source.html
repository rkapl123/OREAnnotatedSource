45 pair<Real, Real> inf_jy_expectation_1(const CrossAssetModel* x, Size i, Time t0, Real dt) {
46 
47  QL_REQUIRE(x->modelType(INF, i) == JY, "inf_jy_expectation_1: should only be used for JY CAM inflation component.");
48 
49  auto res = make_pair(0.0, 0.0);
50 
51  // 1) Real rate process drift
52  res.first = -integral(x, P(Hy(i), ay(i), ay(i)), t0, t0 + dt) +
53  integral(x, P(rzy(0, i), Hz(0), az(0), ay(i)), t0, t0 + dt) -
54  integral(x, P(ryy(i, i, 0, 1), ay(i), sy(i)), t0, t0 + dt);
55 
56  // i_i - index of i-th inflation component's currency.
57  Size i_i = x->ccyIndex(x->infjy(i)->currency());
58  if (i_i > 0) {
59  res.first -= integral(x, P(rxy(i_i - 1, i, 0), ay(i), sx(i_i - 1)), t0, t0 + dt);
60  }
61 
62  // 2) Inflation index process drift
63  const auto& zts = x->infjy(i)->realRate()->termStructure();
64  bool indexIsInterpolated = true; // FIXME, see also crossassetmodel.cpp line 706ff
65  res.second = std::log(inflationGrowth(zts, t0 + dt, indexIsInterpolated) / inflationGrowth(zts, t0, indexIsInterpolated));
66 
67  res.second -= 0.5 * (vy(i).eval(x, t0 + dt) - vy(i).eval(x, t0));
68 
69  // Final _s means start of period i.e. t_0 and _e means end of period i.e. t_0 + dt
70  Real Hi_i_s = Hz(i_i).eval(x, t0);
71  Real Hi_s = Hy(i).eval(x, t0);
72  Real Hi_i_e = Hz(i_i).eval(x, t0 + dt);
73  Real Hi_e = Hy(i).eval(x, t0 + dt);
74  Real zetai_i_s = zetaz(i_i).eval(x, t0);
75  Real zetai_s = zetay(i).eval(x, t0);
76  Real zetai_i_e = zetaz(i_i).eval(x, t0 + dt);
77  Real zetai_e = zetay(i).eval(x, t0 + dt);
78  res.second += 0.5 * (Hi_i_e * Hi_i_e * zetai_i_e - Hi_i_s * Hi_i_s * zetai_i_s);
79  res.second -= 0.5 * integral(x, P(Hz(i_i), Hz(i_i), az(i_i), az(i_i)), t0, t0 + dt);
80  res.second -= 0.5 * (Hi_e * Hi_e * zetai_e - Hi_s * Hi_s * zetai_s);
81  res.second += 0.5 * integral(x, P(Hy(i), Hy(i), ay(i), ay(i)), t0, t0 + dt);
82 
83  res.second += integral(x, P(rzy(0, i, 1), Hz(0), az(0), sy(i)), t0, t0 + dt);
84 
85  res.second -=
86  integral(x,
87  P(LC(Hi_e, -1.0, Hy(i)), LC(0.0, -1.0, P(Hy(i), ay(i), ay(i)), 1.0, P(Hz(0), az(0), ay(i), rzy(0, i)),
88  -1.0, P(ryy(i, i, 0, 1), ay(i), sy(i)))),
89  t0, t0 + dt);
90 
91  if (i_i > 0) {
92  res.second += integral(x,
93  P(LC(Hi_i_e, -1.0, Hz(i_i)),
94  LC(0.0, -1.0, P(Hz(i_i), az(i_i), az(i_i)), 1.0, P(Hz(0), az(0), az(i_i), rzz(0, i_i)),
95  -1.0, P(rzx(i_i, i_i - 1), az(i_i), sx(i_i - 1)))),
96  t0, t0 + dt);
97  res.second -= integral(x, P(rxy(i_i - 1, i, 1), sy(i), sx(i_i - 1)), t0, t0 + dt);
98  res.second += integral(x, P(LC(Hi_e, -1.0, Hy(i)), ay(i), sx(i_i - 1), rxy(i_i - 1, i)), t0, t0 + dt);
99  }
100 
101  return res;
102 }

121 Real fx_expectation_1(const CrossAssetModel* x, const Size i, const Time t0, const Real dt) {
122  bool bam = (x->measure() == Measure::BA);
123  Real H0_a = Hz(0).eval(x, t0);
124  Real Hi_a = Hz(i + 1).eval(x, t0);
125  Real H0_b = Hz(0).eval(x, t0 + dt);
126  Real Hi_b = Hz(i + 1).eval(x, t0 + dt);
127  Real zeta0_a = zetaz(0).eval(x, t0);
128  Real zetai_a = zetaz(i + 1).eval(x, t0);
129  Real zeta0_b = zetaz(0).eval(x, t0 + dt);
130  Real zetai_b = zetaz(i + 1).eval(x, t0 + dt);
131  Real res = std::log(
132  x->irlgm1f(i + 1)->termStructure()->discount(t0 + dt) / x->irlgm1f(i + 1)->termStructure()->discount(t0) *
133  x->irlgm1f(0)->termStructure()->discount(t0) / x->irlgm1f(0)->termStructure()->discount(t0 + dt));
134  res -= 0.5 * (vx(i).eval(x, t0 + dt) - vx(i).eval(x, t0));
135  res +=
136  0.5 * (H0_b * H0_b * zeta0_b - H0_a * H0_a * zeta0_a - integral(x, P(Hz(0), Hz(0), az(0), az(0)), t0, t0 + dt));
137  res -= 0.5 * (Hi_b * Hi_b * zetai_b - Hi_a * Hi_a * zetai_a -
138  integral(x, P(Hz(i + 1), Hz(i + 1), az(i + 1), az(i + 1)), t0, t0 + dt));
139  res += (bam ? 0.0 : integral(x, P(Hz(0), az(0), sx(i), rzx(0, i)), t0, t0 + dt));
140  res -= Hi_b * (-integral(x, P(Hz(i + 1), az(i + 1), az(i + 1)), t0, t0 + dt) +
141  (bam ? 0.0 : integral(x, P(Hz(0), az(0), az(i + 1), rzz(0, i + 1)), t0, t0 + dt)) -
142  integral(x, P(az(i + 1), sx(i), rzx(i + 1, i)), t0, t0 + dt));
143  res += -integral(x, P(Hz(i + 1), Hz(i + 1), az(i + 1), az(i + 1)), t0, t0 + dt) +
144  (bam ? 0.0 : integral(x, P(Hz(0), Hz(i + 1), az(0), az(i + 1), rzz(0, i + 1)), t0, t0 + dt)) -
145  integral(x, P(Hz(i + 1), az(i + 1), sx(i), rzx(i + 1, i)), t0, t0 + dt);
146  if (bam) {
147  res -= H0_b * integral(x, P(Hz(0), az(0), az(0)), t0, t0 + dt);
148  res += integral(x, P(Hz(0), Hz(0), az(0), az(0)), t0, t0 + dt);
149  }
150  return res;
151 }

160 Real eq_expectation_1(const CrossAssetModel* x, const Size k, const Time t0, const Real dt) {
161  Size i = x->ccyIndex(x->eqbs(k)->currency());
162  Size eps_i = (i == 0) ? 0 : 1;
163  Real Hi_a = Hz(i).eval(x, t0);
164  Real Hi_b = Hz(i).eval(x, t0 + dt);
165  Real zetai_a = zetaz(i).eval(x, t0);
166  Real zetai_b = zetaz(i).eval(x, t0 + dt);
167  Real res =
168  std::log(x->eqbs(k)->equityDivYieldCurveToday()->discount(t0 + dt) /
169  x->eqbs(k)->equityDivYieldCurveToday()->discount(t0) * x->eqbs(k)->equityIrCurveToday()->discount(t0) /
170  x->eqbs(k)->equityIrCurveToday()->discount(t0 + dt));
171  res -= 0.5 * (vs(k).eval(x, t0 + dt) - vs(k).eval(x, t0));
172  res +=
173  0.5 * (Hi_b * Hi_b * zetai_b - Hi_a * Hi_a * zetai_a - integral(x, P(Hz(i), Hz(i), az(i), az(i)), t0, t0 + dt));
174  res += integral(x, P(rzs(0, k), Hz(0), az(0), ss(k)), t0, t0 + dt);
175  if (eps_i > 0) {
176  res -= integral(x, P(rxs(i - 1, k), sx(i - 1), ss(k)), t0, t0 + dt);
177  }
178  // expand gamma term
179  if (eps_i > 0) {
180  res += Hi_b * (-integral(x, P(Hz(i), az(i), az(i)), t0, t0 + dt) -
181  integral(x, P(rzx(i, i - 1), sx(i - 1), az(i)), t0, t0 + dt) +
182  integral(x, P(rzz(0, i), az(i), az(0), Hz(0)), t0, t0 + dt));
183  res -= (-integral(x, P(Hz(i), Hz(i), az(i), az(i)), t0, t0 + dt) -
184  integral(x, P(Hz(i), rzx(i, i - 1), sx(i - 1), az(i)), t0, t0 + dt) +
185  integral(x, P(Hz(i), rzz(0, i), az(i), az(0), Hz(0)), t0, t0 + dt));
186  }
187  return res;
188 }

215  Real Hi = Hz(i + 1).eval(x, t0 + dt);
216  Real Hj = Hz(j + 1).eval(x, t0 + dt);
217  Real res =
218  // row 1
219  H0 * H0 * (zetaz(0).eval(x, t0 + dt) - zetaz(0).eval(x, t0)) -
220  2.0 * H0 * integral(x, P(Hz(0), az(0), az(0)), t0, t0 + dt) +
221  integral(x, P(Hz(0), Hz(0), az(0), az(0)), t0, t0 + dt) -
222  // row 2

227  // row 3

235  // row 5

237  integral(x, P(Hz(0), az(0), sx(i), rzx(0, i)), t0, t0 + dt) -
238  // row 6

240  integral(x, P(Hz(i + 1), az(i + 1), sx(j), rzx(i + 1, j)), t0, t0 + dt) -

282 Real infy_infy_covariance(const CrossAssetModel* x, const Size i, const Size j, const Time t0, const Time dt) {
283 
284  Real res = 0.0;
285 
286  // Assumption that INF is either JY or DK. Four possibilities.
287  auto mti = x->modelType(INF, i);
288  auto mtj = x->modelType(INF, j);
289  if (mti == DK && mtj == DK) {
290  res = integral(x, P(ryy(i, j), Hy(i), ay(i), Hy(j), ay(j)), t0, t0 + dt);
291  } else if (mti == JY && mtj == DK) {
292 
293  // i_i - index of i-th inflation component's currency.
294  Size i_i = x->ccyIndex(x->infjy(i)->currency());
295  // H_{i_i}^{z}(t_0 + dt)
296  Real Hi_i = Hz(i_i).eval(x, t0 + dt);
297  // H_{i}^{y}(t_0 + dt)
298  Real Hi = Hy(i).eval(x, t0 + dt);
299 
300  // 3 terms in the covariance.
301  res = integral(x, P(rzy(i_i, j), Hy(j), ay(j), az(i_i), LC(Hi_i, -1.0, Hz(i_i))), t0, t0 + dt);
302  res -= integral(x, P(ryy(i, j, 0, 0), Hy(j), ay(j), ay(i), LC(Hi, -1.0, Hy(i))), t0, t0 + dt);
303  res += integral(x, P(ryy(i, j, 1, 0), Hy(j), ay(j), sy(i)), t0, t0 + dt);
304 
305  } else if (mti == DK && mtj == JY) {
306 
307  // i_j - index of j-th inflation component's currency.
308  Size i_j = x->ccyIndex(x->infjy(j)->currency());
309  // H_{i_j}^{z}(t_0 + dt)
310  Real Hi_j = Hz(i_j).eval(x, t0 + dt);
311  // H_{j}^{y}(t_0 + dt)
312  Real Hj = Hy(j).eval(x, t0 + dt);
313 
314  // 3 terms in the covariance.
315  res = integral(x, P(rzy(i_j, i), Hy(i), ay(i), az(i_j), LC(Hi_j, -1.0, Hz(i_j))), t0, t0 + dt);
316  res -= integral(x, P(ryy(i, j, 0, 0), Hy(i), ay(i), ay(j), LC(Hj, -1.0, Hy(j))), t0, t0 + dt);
317  res += integral(x, P(ryy(i, j, 0, 1), Hy(i), ay(i), sy(j)), t0, t0 + dt);
318 
319  } else {
320 
321  // index of each inflation component's currency.
322  Size i_i = x->ccyIndex(x->infjy(i)->currency());
323  Size i_j = x->ccyIndex(x->infjy(j)->currency());
324  // H_{i_\cdot}^{z}(t_0 + dt)
325  Real Hi_i = Hz(i_i).eval(x, t0 + dt);
326  Real Hi_j = Hz(i_j).eval(x, t0 + dt);
327  // H_{\cdot}^{y}(t_0 + dt)
328  Real Hi = Hy(i).eval(x, t0 + dt);
329  Real Hj = Hy(j).eval(x, t0 + dt);
330 
331  res = integral(x, P(rzz(i_i, i_j), az(i_i), LC(Hi_i, -1.0, Hz(i_i)), az(i_j), LC(Hi_j, -1.0, Hz(i_j))), t0,
332  t0 + dt);
333  res -=
334  integral(x, P(rzy(i_i, j, 0), az(i_i), LC(Hi_i, -1.0, Hz(i_i)), ay(j), LC(Hj, -1.0, Hy(j))), t0, t0 + dt);
335  res += integral(x, P(rzy(i_i, j, 1), az(i_i), LC(Hi_i, -1.0, Hz(i_i)), sy(j)), t0, t0 + dt);
336  res -=
337  integral(x, P(rzy(i_j, i, 0), ay(i), LC(Hi, -1.0, Hy(i)), az(i_j), LC(Hi_j, -1.0, Hz(i_j))), t0, t0 + dt);
338  res += integral(x, P(ryy(i, j), ay(i), LC(Hi, -1.0, Hy(i)), ay(j), LC(Hj, -1.0, Hy(j))), t0, t0 + dt);
339  res -= integral(x, P(ryy(i, j, 0, 1), ay(i), LC(Hi, -1.0, Hy(i)), sy(j)), t0, t0 + dt);
340  res += integral(x, P(rzy(i_j, i, 1), sy(i), az(i_j), LC(Hi_j, -1.0, Hz(i_j))), t0, t0 + dt);
341  res -= integral(x, P(ryy(i, j, 1, 0), sy(i), ay(j), LC(Hj, -1.0, Hy(j))), t0, t0 + dt);
342  res += integral(x, P(ryy(i, j, 1, 1), sy(i), sy(j)), t0, t0 + dt);
343  }
