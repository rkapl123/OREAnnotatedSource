46std::size_t ComputationGraph::insert(const std::string& label) {
47 std::size_t node = predecessors_.size();
48 predecessors_.push_back(std::vector<std::size_t>());
49 opId_.push_back(0);
50 maxNodeRequiringArg_.push_back(0);
51 isConstant_.push_back(false);
52 constantValue_.push_back(0.0);
53 if (!label.empty())
54 labels_[node].insert(label);
55 return node;
56}

58std::size_t ComputationGraph::insert(const std::vector<std::size_t>& predecessors, const std::size_t opId,
59 const std::string& label) {
60 std::size_t node = predecessors_.size();
61 predecessors_.push_back(predecessors);
62 opId_.push_back(opId);
63 for (auto const& p : predecessors) {
64 maxNodeRequiringArg_[p] = node;
65 }
66 maxNodeRequiringArg_.push_back(0);
67 isConstant_.push_back(false);
68 constantValue_.push_back(0.0);
69 if (!label.empty())
70 labels_[node].insert(label);
71 return node;
72}

82std::size_t ComputationGraph::constant(const double x) {
83 auto c = constants_.find(x);
84 if (c != constants_.end())
85 return c->second;
86 else {
87 std::size_t node = predecessors_.size();
88 constants_.insert(std::make_pair(x, node));
89 predecessors_.push_back(std::vector<std::size_t>());
90 opId_.push_back(0);
91 maxNodeRequiringArg_.push_back(0);
92 isConstant_.push_back(true);
93 constantValue_.push_back(x);
94 labels_[node].insert(std::to_string(x));
95 return node;
96 }
97}

101std::size_t ComputationGraph::variable(const std::string& name, const bool createIfNotExists) {
102 auto c = variables_.find(name);
103 if (c != variables_.end())
104 return c->second;
105 else if (createIfNotExists) {
106 std::size_t node = predecessors_.size();
107 variables_.insert(std::make_pair(name, node));
108 variableVersion_[name] = 0;
109 labels_[node].insert(name + "(v" + std::to_string(++variableVersion_[name]) + ")");
110 predecessors_.push_back(std::vector<std::size_t>());
111 opId_.push_back(0);
112 maxNodeRequiringArg_.push_back(0);
113 isConstant_.push_back(false);
114 constantValue_.push_back(0.0);
115 return node;
116 } else {
117 QL_FAIL("ComputationGraph::variable(" << name << ") not found.");
118 }
119}

123void ComputationGraph::setVariable(const std::string& name, const std::size_t node) {
124 auto v = variables_.find(name);
125 if (v != variables_.end()) {
126 if (v->second != node) {
127 labels_[node].insert(name + "(v" + std::to_string(++variableVersion_[name]) + ")");
128 v->second = node;
129 }
130 } else {
131 variableVersion_[name] = 0;
132 labels_[node].insert(name + "(v" + std::to_string(++variableVersion_[name]) + ")");
133 variables_[name] = node;
134 }
135}

149std::size_t cg_add(ComputationGraph& g, const std::size_t a, const std::size_t b, const std::string& label) {
150 if (g.isConstant(a) && g.isConstant(b))
151 return cg_const(g, g.constantValue(a) + g.constantValue(b));
152 if (g.isConstant(a) && QuantLib::close_enough(g.constantValue(a), 0.0))
153 return b;
154 if (g.isConstant(b) && QuantLib::close_enough(g.constantValue(b), 0.0))
155 return a;
156 return g.insert({a, b}, RandomVariableOpCode::Add, label);
157}

159std::size_t cg_subtract(ComputationGraph& g, const std::size_t a, const std::size_t b, const std::string& label) {
160 if (a == b)
161 return cg_const(g, 0.0);
162 if (g.isConstant(a) && g.isConstant(b))
163 return cg_const(g, g.constantValue(a) - g.constantValue(b));
164 if (g.isConstant(a) && QuantLib::close_enough(g.constantValue(a), 0.0))
165 return cg_negative(g, b);
166 if (g.isConstant(b) && QuantLib::close_enough(g.constantValue(b), 0.0))
167 return a;
168 return g.insert({a, b}, RandomVariableOpCode::Subtract, label);
169}

177std::size_t cg_mult(ComputationGraph& g, const std::size_t a, const std::size_t b, const std::string& label) {
178 if (g.isConstant(a) && g.isConstant(b))
179 return cg_const(g, g.constantValue(a) * g.constantValue(b));
180 if (g.isConstant(a) && QuantLib::close_enough(g.constantValue(a), 1.0))
181 return b;
182 if (g.isConstant(b) && QuantLib::close_enough(g.constantValue(b), 1.0))
183 return a;
184 if ((g.isConstant(a) && QuantLib::close_enough(g.constantValue(a), 0.0)) ||
185 (g.isConstant(b) && QuantLib::close_enough(g.constantValue(b), 0.0)))
186 return cg_const(g, 0.0);
187 return g.insert({a, b}, RandomVariableOpCode::Mult, label);
188}

190std::size_t cg_div(ComputationGraph& g, const std::size_t a, const std::size_t b, const std::string& label) {
191 if (a == b)
192 return cg_const(g, 1.0);
193 if (g.isConstant(a) && g.isConstant(b))
194 return cg_const(g, g.constantValue(a) / g.constantValue(b));
195 if (g.isConstant(b) && QuantLib::close_enough(g.constantValue(b), 1.0))
196 return a;
197 if (g.isConstant(a) && QuantLib::close_enough(g.constantValue(a), 0.0))
198 return cg_const(g, 0.0);
199 return g.insert({a, b}, RandomVariableOpCode::Div, label);
200}

202std::size_t cg_conditionalExpectation(ComputationGraph& g, const std::size_t regressand,
203 const std::vector<std::size_t>& regressor, const std::size_t filter,
204 const std::string& label) {
205 if (g.isConstant(regressand))
206 return regressand;
207 std::vector<size_t> args;
208 args.push_back(regressand);
209 args.push_back(filter);
210 args.insert(args.end(), regressor.begin(), regressor.end());
211 return g.insert(args, RandomVariableOpCode::ConditionalExpectation, label);
212}
