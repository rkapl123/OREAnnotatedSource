38 BRLCdiSwap::BRLCdiSwap(Type type, Real nominal, const Date& startDate, const Date& endDate, Rate fixedRate,
39  const boost::shared_ptr<BRLCdi>& overnightIndex, Spread spread, bool telescopicValueDates)
40  : OvernightIndexedSwap(type, nominal,
41  Schedule(list_of(startDate)(endDate).convert_to_container<vector<Date> >(), NullCalendar(),
42  QuantLib::Unadjusted, QuantLib::Unadjusted, 100 * Years),
43  fixedRate, overnightIndex->dayCounter(), overnightIndex, spread, 0, ModifiedFollowing,
44  overnightIndex->fixingCalendar(), telescopicValueDates),
45  startDate_(startDate), endDate_(endDate), index_(overnightIndex) {
46 
47  // Need to overwrite the fixed leg with the correct fixed leg for a standard BRL CDI swap
48  // Fixed leg is of the form: N [(1 + k) ^ \delta - 1]
49  // where \delta is the number of BRL business days in the period divided by 252 i.e.
50  // the day count fraction for the period on a Business252 basis.
51  Time dcf = index_->dayCounter().yearFraction(startDate_, endDate_);
52  Real fixedLegPayment = nominal * (pow(1.0 + fixedRate, dcf) - 1.0);
53  Date paymentDate = legs_[0].back()->date();
54  boost::shared_ptr<CashFlow> fixedCashflow = boost::make_shared<SimpleCashFlow>(fixedLegPayment, paymentDate);
55  legs_[0].clear();
56  legs_[0].push_back(fixedCashflow);
57  registerWith(fixedCashflow);
58 
59  // Set the pricer on the BRL CDI coupon
60  QL_REQUIRE(legs_[1].size() == 1, "BRLCdiSwap expected exactly one overnight coupon");
61  boost::shared_ptr<QuantLib::OvernightIndexedCoupon> coupon =
62  boost::dynamic_pointer_cast<QuantLib::OvernightIndexedCoupon>(legs_[1][0]);
63  QL_REQUIRE(coupon, "BRLCdiSwap: expected QuantLib::OvernightIndexedCoupon");
64  coupon->setPricer(boost::make_shared<BRLCdiCouponPricer>());
65 }

67 Real BRLCdiSwap::fixedLegBPS() const {
68 
69  calculate();
70 
71  static Spread basisPoint = 1.0e-4;
72  if (!close(endDiscounts_[0], 0.0) && endDiscounts_[0] != Null<DiscountFactor>()) {
73  DiscountFactor df = endDiscounts_[0];
74  Time dcf = index_->dayCounter().yearFraction(startDate_, endDate_);
75  legBPS_[0] = df * nominal() * (pow(1.0 + fixedRate() + basisPoint, dcf) - pow(1.0 + fixedRate(), dcf));
76  return legBPS_[0];
77  }
78 
79  QL_FAIL("BRLCdiSwap cannot calculate fixed leg BPS because end discount is not populated");
80 }

82 Real BRLCdiSwap::fairRate() const {
83 
84  calculate();
85 
86  if (!close(endDiscounts_[0], 0.0) && endDiscounts_[0] != Null<DiscountFactor>()) {
87  DiscountFactor df = endDiscounts_[0];
88  Time dcf = index_->dayCounter().yearFraction(startDate_, endDate_);
89  return pow(overnightLegNPV() / (nominal() * df) + 1.0, 1.0 / dcf) - 1.0;
90  }

92  QL_FAIL("BRLCdiSwap cannot calculate fair rate because end discount is not populated");
93 }
