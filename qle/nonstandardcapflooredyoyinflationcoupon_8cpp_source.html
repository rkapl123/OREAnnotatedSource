27void NonStandardCappedFlooredYoYInflationCoupon::setCommon(Rate cap, Rate floor) {
28
29 isCapped_ = false;
30 isFloored_ = false;
31
32 if (gearing_ > 0) {
33 if (cap != Null<Rate>()) {
34 isCapped_ = true;
35 cap_ = cap;
36 }
37 if (floor != Null<Rate>()) {
38 floor_ = floor;
39 isFloored_ = true;
40 }
41 } else {
42 if (cap != Null<Rate>()) {
43 floor_ = cap;
44 isFloored_ = true;
45 }
46 if (floor != Null<Rate>()) {
47 isCapped_ = true;
48 cap_ = floor;
49 }
50 }
51 if (isCapped_ && isFloored_) {
52 QL_REQUIRE(cap >= floor, "cap level (" << cap << ") less than floor level (" << floor << ")");
53 }
54}

89Rate NonStandardCappedFlooredYoYInflationCoupon::rate() const {
90 Rate swapletRate = underlying_ ? underlying_->rate() : NonStandardYoYInflationCoupon::rate();
91
92 if (isFloored_ || isCapped_) {
93 if (underlying_) {
94 QL_REQUIRE(underlying_->pricer(), "pricer not set");
95 } else {
96 QL_REQUIRE(pricer_, "pricer not set");
97 }
98 }
99
100 Rate floorletRate = 0.;
101 if (isFloored_) {
102 floorletRate = underlying_ ? underlying_->pricer()->floorletRate(effectiveFloor())
103 : pricer()->floorletRate(effectiveFloor());
104 }
105 Rate capletRate = 0.;
106 if (isCapped_) {
107 capletRate =
108 underlying_ ? underlying_->pricer()->capletRate(effectiveCap()) : pricer()->capletRate(effectiveCap());
109 }
110
111 return swapletRate + floorletRate - capletRate;
112}

236NonStandardYoYInflationLeg::operator Leg() const {
237
238 Size n = schedule_.size() - 1;
239 QL_REQUIRE(!notionals_.empty(), "no notional given");
240 QL_REQUIRE(notionals_.size() <= n, "too many nominals (" << notionals_.size() << "), only " << n << " required");
241 QL_REQUIRE(gearings_.size() <= n, "too many gearings (" << gearings_.size() << "), only " << n << " required");
242 QL_REQUIRE(spreads_.size() <= n, "too many spreads (" << spreads_.size() << "), only " << n << " required");
243 QL_REQUIRE(caps_.size() <= n, "too many caps (" << caps_.size() << "), only " << n << " required");
244 QL_REQUIRE(floors_.size() <= n, "too many floors (" << floors_.size() << "), only " << n << " required");
245
246 Leg leg;
247 leg.reserve(n);
248
249 Calendar calendar = paymentCalendar_;
250
251 Date refStart, start, refEnd, end;
252
253 for (Size i = 0; i < n; ++i) {
254 refStart = start = schedule_.date(i);
255 refEnd = end = schedule_.date(i + 1);
256 Date paymentDate = calendar.adjust(end, paymentAdjustment_);
257 if (i == 0 && schedule_.hasIsRegular() && !schedule_.isRegular(i + 1)) {
258 BusinessDayConvention bdc = schedule_.businessDayConvention();
259 refStart = schedule_.calendar().adjust(end - schedule_.tenor(), bdc);
260 }
261 if (i == n - 1 && schedule_.hasIsRegular() && !schedule_.isRegular(i + 1)) {
262 BusinessDayConvention bdc = schedule_.businessDayConvention();
263 refEnd = schedule_.calendar().adjust(start + schedule_.tenor(), bdc);
264 }
265 if (detail::get(gearings_, i, 1.0) == 0.0) { // fixed coupon
266 leg.push_back(ext::shared_ptr<CashFlow>(new FixedRateCoupon(
267 paymentDate, detail::get(notionals_, i, 1.0), detail::effectiveFixedRate(spreads_, caps_, floors_, i),
268 paymentDayCounter_, start, end, refStart, refEnd)));
269 } else { // yoy inflation coupon
270 if (detail::noOption(caps_, floors_, i)) { // just swaplet
271 ext::shared_ptr<NonStandardYoYInflationCoupon> coup(new NonStandardYoYInflationCoupon(
272 paymentDate, detail::get(notionals_, i, 1.0), start, end, detail::get(fixingDays_, i, 0), index_,
273 observationLag_, paymentDayCounter_, detail::get(gearings_, i, 1.0), detail::get(spreads_, i, 0.0),
274 refStart, refEnd, addInflationNotional_));
275
276 // in this case you can set a pricer
277 // straight away because it only provides computation - not data
278 ext::shared_ptr<NonStandardYoYInflationCouponPricer> pricer(
279 new NonStandardYoYInflationCouponPricer(rateCurve_));
280 coup->setPricer(pricer);
281 leg.push_back(ext::dynamic_pointer_cast<CashFlow>(coup));
282
283 } else { // cap/floorlet
284 leg.push_back(ext::shared_ptr<CashFlow>(new NonStandardCappedFlooredYoYInflationCoupon(
285 paymentDate, detail::get(notionals_, i, 1.0), start, end, detail::get(fixingDays_, i, 0), index_,
286 observationLag_, paymentDayCounter_, detail::get(gearings_, i, 1.0), detail::get(spreads_, i, 0.0),
287 detail::get(caps_, i, Null<Rate>()), detail::get(floors_, i, Null<Rate>()), refStart, refEnd,
288 addInflationNotional_)));
289 }
290 }
291 }
292
293 return leg;
294}
