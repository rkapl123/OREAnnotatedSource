52Real BondTRSCashFlow::bondStart() const {

70BondTRSLeg::operator Leg() const {
71 Leg leg;
72 Real initialPrice;
73
74 for (Size i = 0; i < valuationDates_.size() - 1; ++i) {
75 initialPrice = i == 0 ? initialPrice_ : Null<Real>();
76 leg.push_back(boost::make_shared<BondTRSCashFlow>(paymentDates_[i], valuationDates_[i], valuationDates_[i + 1],
77 bondNotional_, bondIndex_, initialPrice, fxIndex_));
78 }
79 return leg;
80}
