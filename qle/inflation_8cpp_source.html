42// Throws an error if index doesn't have a historical fixing for fixingDate
43void throwExceptionIfHistoricalFixingMissing(const Date& fixingDate, const QuantLib::ZeroInflationIndex& index) {
44 QL_REQUIRE(index.hasHistoricalFixing(fixingDate),
45 "Historical fixing missing for index " << index.name() << " on " << fixingDate);

48// Throws an error if any fixing required to compute a cpi fixing is missing
49void checkIfFixingAvailable(const Date& maturity, const Period obsLag, bool interpolated,
50 const QuantLib::ZeroInflationIndex& index) {
51
52 auto fixingPeriod = inflationPeriod(maturity - obsLag, index.frequency());
53 throwExceptionIfHistoricalFixingMissing(fixingPeriod.first, index);
54 if (interpolated)
55 throwExceptionIfHistoricalFixingMissing(fixingPeriod.second + 1 * Days, index);
56}

83Real inflationLinkedBondQuoteFactor(const boost::shared_ptr<QuantLib::Bond>& bond) {
84 QuantLib::Real inflFactor = 1;
85 for (auto& cf : bond->cashflows()) {
86 if (auto inflCpn = boost::dynamic_pointer_cast<QuantLib::CPICoupon>(cf)) {
87 const auto& inflationIndex = boost::dynamic_pointer_cast<QuantLib::ZeroInflationIndex>(inflCpn->index());
88 Date settlementDate = bond->settlementDate();
89 std::pair<Date, Date> currentInflationPeriod = inflationPeriod(settlementDate, inflationIndex->frequency());
90 std::pair<Date, Date> settlementFixingPeriod = inflationPeriod(settlementDate - inflCpn->observationLag(), inflationIndex->frequency());
91 Date curveBaseDate = settlementFixingPeriod.first;
92 //Date curveBaseDate = inflationIndex->zeroInflationTermStructure()->baseDate();
93 Real todaysCPI = inflationIndex->fixing(curveBaseDate);
94 if (inflCpn->observationInterpolation() == QuantLib::CPI::Linear) {
95
96 std::pair<Date, Date> observationPeriod = inflationPeriod(curveBaseDate, inflationIndex->frequency());
97
98 Real indexStart = inflationIndex->fixing(observationPeriod.first);
99 Real indexEnd = inflationIndex->fixing(observationPeriod.second + 1 * QuantLib::Days);
100
101 todaysCPI = indexStart + (settlementDate - currentInflationPeriod.first) *
102 (indexEnd - indexStart) /
103 (Real)(currentInflationPeriod.second - currentInflationPeriod.first);
104 }
105 QuantLib::Rate baseCPI = inflCpn->baseCPI();
106 if (baseCPI == QuantLib::Null<QuantLib::Rate>()) {
107 baseCPI =
108 QuantLib::CPI::laggedFixing(inflCpn->cpiIndex(), inflCpn->baseDate() + inflCpn->observationLag(),
109 inflCpn->observationLag(),
110 inflCpn->observationInterpolation());
111 }
112 inflFactor = todaysCPI / baseCPI;
113 break;
114 }
115 }
116 return inflFactor;
117}

120QuantLib::Date lastAvailableFixing(const QuantLib::ZeroInflationIndex& index, const QuantLib::Date& asof) {
121 Date availbilityLagFixingDate = inflationPeriod(asof - index.availabilityLag(), index.frequency()).first;
122 if (index.hasHistoricalFixing(availbilityLagFixingDate)) {
123 return availbilityLagFixingDate;
124 } else {
125 // The fixing for the inflation period before the availability lag should be there
126 return inflationPeriod(availbilityLagFixingDate - 1 * QuantLib::Days, index.frequency()).first;
127 }
128}

136QuantLib::Date curveBaseDate(const bool baseDateLastKnownFixing, const QuantLib::Date& refDate,
137 const QuantLib::Period obsLagCurve, const QuantLib::Frequency curveFreq,
138 const boost::shared_ptr<QuantLib::ZeroInflationIndex>& index) {
139 if (baseDateLastKnownFixing) {
140 QL_REQUIRE(index, "can not compute curve base date based on the last known index fixing if no index provided");
141 return lastAvailableFixing(*index, refDate);
142 } else {
143 return QuantLib::inflationPeriod(refDate - obsLagCurve, curveFreq).first;
144 }

147QuantLib::Date fixingDate(const QuantLib::Date& d, const QuantLib::Period obsLag, const QuantLib::Frequency freq,
148 bool interpolated) {
149 Date obsDate = d - obsLag;
150 if (!interpolated)
151 obsDate = QuantLib::inflationPeriod(obsDate, freq).first;
152 return obsDate;

155QuantLib::Rate guessCurveBaseRate(const bool baseDateLastKnownFixing, const QuantLib::Date& swapStart,
156 const QuantLib::Date& asof,
157 const QuantLib::Period& swapTenor, const QuantLib::DayCounter& swapZCLegDayCounter,
158 const QuantLib::Period& swapObsLag, const QuantLib::Rate zeroCouponRate,
159 const QuantLib::Period& curveObsLag, const QuantLib::DayCounter& curveDayCounter,
160 const boost::shared_ptr<QuantLib::ZeroInflationIndex>& index, const bool interpolated,
161 const boost::shared_ptr<QuantLib::Seasonality>& seasonality) {
162 boost::shared_ptr<QuantLib::MultiplicativePriceSeasonality> multiplicativeSeasonality =
163 seasonality ? boost::dynamic_pointer_cast<QuantLib::MultiplicativePriceSeasonality>(seasonality) : nullptr;
164
165 QL_REQUIRE(seasonality == nullptr || multiplicativeSeasonality,
166 "Only multiplicative seasonality supported at the moment");
167
168 Date swapBaseDate = ZeroInflation::fixingDate(swapStart, swapObsLag, index->frequency(), interpolated);
169
170 Date curveBaseDate =
171 ZeroInflation::curveBaseDate(baseDateLastKnownFixing, asof, curveObsLag, index->frequency(), index);
172
173 // If the baseDate in Curve is today - obsLag then the quoted zeroRate should be used
174 if (!baseDateLastKnownFixing && swapBaseDate == curveBaseDate)
175 return zeroCouponRate;
176
177 QL_REQUIRE(index, "can not compute base cpi of the zero coupon swap");
178 // Otherwise we need to take into account the accrued inflation between rate helper base date and curve base date
179 // Check if historical fixings available
180 try {
181 checkIfFixingAvailable(swapStart, swapObsLag, interpolated, *index);
182 } catch (const std::exception& e) {
183 QL_FAIL("Can not estimate the curve base date, got " << e.what());
184 }
185
186 Date swapMaturity = swapStart + swapTenor;
187 Date swapObservationDate = ZeroInflation::fixingDate(swapMaturity, swapObsLag, index->frequency(), interpolated);
188
189 // TODO check historical fixings available
190 Rate instrumentBaseCPI = cpiFixing(index, swapStart, swapObsLag, interpolated);
191 Time timeFromSwapBase =
192 inflationYearFraction(index->frequency(), interpolated, swapZCLegDayCounter, swapBaseDate, swapObservationDate);
193
194 auto fwdCPI = instrumentBaseCPI * std::pow(1 + zeroCouponRate, timeFromSwapBase);
195
196 double curveBaseFixing = index->fixing(curveBaseDate);
197
198 if (!interpolated) {
199 Time timeFromCurveBase = inflationYearFraction(index->frequency(), interpolated, curveDayCounter, curveBaseDate,
200 swapObservationDate);
201 double rateWithSeasonality = std::pow(fwdCPI / curveBaseFixing, 1.0 / timeFromCurveBase) - 1.0;
202
203 if (multiplicativeSeasonality) {
204 double factorAt = multiplicativeSeasonality->seasonalityFactor(swapObservationDate);
205 double factorBase = multiplicativeSeasonality->seasonalityFactor(curveBaseDate);
206 double seasonalityFactor = std::pow(factorAt / factorBase, 1.0 / timeFromCurveBase);
207 return (rateWithSeasonality + 1) / seasonalityFactor - 1;
208 } else {
209 return rateWithSeasonality;
210 }
211 } else {
212 // Compute the interpolated fixing of the ZCIIS at maturity
213 auto fixingPeriod = inflationPeriod(swapObservationDate, index->frequency());
214 auto paymentPeriod = inflationPeriod(swapMaturity, index->frequency());
215 // Fixing times from curve base date
216 Time timeToFixing1 =
217 inflationYearFraction(index->frequency(), false, curveDayCounter, curveBaseDate, fixingPeriod.first);
218 Time timeToFixing2 = inflationYearFraction(index->frequency(), false, curveDayCounter, curveBaseDate,
219 fixingPeriod.second + 1 * Days);
220
221 // Time interpolation
222 Time timeToPayment =
223 inflationYearFraction(index->frequency(), true, curveDayCounter, curveBaseDate, swapMaturity);
224 Time timeToStartPaymentPeriod =
225 inflationYearFraction(index->frequency(), false, curveDayCounter, curveBaseDate, paymentPeriod.first);
226 Time timeToEndPaymenetPeriod = inflationYearFraction(index->frequency(), false, curveDayCounter, curveBaseDate,
227 paymentPeriod.second + 1 * Days);
228 Time interpolationFactor =
229 (timeToPayment - timeToStartPaymentPeriod) / (timeToEndPaymenetPeriod - timeToStartPaymentPeriod);
230
231 // Root search for a constant rate that the interpolation of both cpi matches the forward cpi
232 Real target = fwdCPI / curveBaseFixing;
233
234 Real seasonalityFactor1 = 1.0;
235 Real seasonalityFactor2 = 1.0;
236
237 if (multiplicativeSeasonality) {
238 double factorAt1 = multiplicativeSeasonality->seasonalityFactor(fixingPeriod.first);
239 double factorAt2 = multiplicativeSeasonality->seasonalityFactor(fixingPeriod.second + 1 * Days);
240 double factorBase = multiplicativeSeasonality->seasonalityFactor(curveBaseDate);
241 seasonalityFactor1 = factorAt1 / factorBase;
242 seasonalityFactor2 = factorAt2 / factorBase;
243 }
244
245
246 std::function<double(double)> objectiveFunction = [&timeToFixing1, &timeToFixing2, &interpolationFactor,
247 &target, &seasonalityFactor1, &seasonalityFactor2](Rate r) {
248 return target - (std::pow(1 + r, timeToFixing1) * seasonalityFactor1 +
249 (std::pow(1 + r, timeToFixing2) * seasonalityFactor2 -
250 std::pow(1 + r, timeToFixing1) * seasonalityFactor1) *
251 interpolationFactor);
252 };
253
254 Rate guess = std::pow(fwdCPI / curveBaseFixing, 1.0 / timeToFixing2) - 1.0;
255 Rate r = guess;
256 try {
257 QuantLib::Brent solver;
258 r = solver.solve(objectiveFunction, 1e-8, guess, -0.1, 0.2);
259 } catch (...) {
260 r = guess;
261 }
262 return r;
263 }
