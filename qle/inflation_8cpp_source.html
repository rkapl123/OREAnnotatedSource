82 Real inflationLinkedBondQuoteFactor(const boost::shared_ptr<QuantLib::Bond>& bond) {
83  QuantLib::Real inflFactor = 1;
84  for (auto& cf : bond->cashflows()) {
85  if (auto inflCpn = boost::dynamic_pointer_cast<QuantLib::CPICoupon>(cf)) {
86  const auto& inflationIndex = boost::dynamic_pointer_cast<QuantLib::ZeroInflationIndex>(inflCpn->index());
87  Date settlementDate = bond->settlementDate();
88  std::pair<Date, Date> currentInflationPeriod = inflationPeriod(settlementDate, inflationIndex->frequency());
89  std::pair<Date, Date> settlementFixingPeriod = inflationPeriod(settlementDate - inflCpn->observationLag(), inflationIndex->frequency());
90  Date curveBaseDate = settlementFixingPeriod.first;
91  //Date curveBaseDate = inflationIndex->zeroInflationTermStructure()->baseDate();
92  Real todaysCPI = inflationIndex->fixing(curveBaseDate);
93  if (inflCpn->observationInterpolation() == QuantLib::CPI::Linear) {
94 
95  std::pair<Date, Date> observationPeriod = inflationPeriod(curveBaseDate, inflationIndex->frequency());
96 
97  Real indexStart = inflationIndex->fixing(observationPeriod.first);
98  Real indexEnd = inflationIndex->fixing(observationPeriod.second + 1 * QuantLib::Days);
99 
100  todaysCPI = indexStart + (settlementDate - currentInflationPeriod.first) *
101  (indexEnd - indexStart) /
102  (Real)(currentInflationPeriod.second - currentInflationPeriod.first);
103  }
104  QuantLib::Rate baseCPI = inflCpn->baseCPI();
105  if (baseCPI == QuantLib::Null<QuantLib::Rate>()) {
106  baseCPI =
107  QuantLib::CPI::laggedFixing(inflCpn->cpiIndex(), inflCpn->baseDate() + inflCpn->observationLag(),
108  inflCpn->observationLag(),
109  inflCpn->observationInterpolation());
110  }
111  inflFactor = todaysCPI / baseCPI;
112  break;
113  }
114  }
115  return inflFactor;
116 }

119 QuantLib::Date lastAvailableFixing(const QuantLib::ZeroInflationIndex& index, const QuantLib::Date& asof) {
120  Date availbilityLagFixingDate = inflationPeriod(asof - index.availabilityLag(), index.frequency()).first;
121  if (index.hasHistoricalFixing(availbilityLagFixingDate)) {
122  return availbilityLagFixingDate;
123  } else {
124  // The fixing for the inflation period before the availability lag should be there
125  return inflationPeriod(availbilityLagFixingDate - 1 * QuantLib::Days, index.frequency()).first;
126  }

154 QuantLib::Rate guessCurveBaseRate(const bool baseDateLastKnownFixing, const QuantLib::Date& swapStart,
155  const QuantLib::Period& swapTenor, const QuantLib::DayCounter& swapZCLegDayCounter,
156  const QuantLib::Period& swapObsLag, const QuantLib::Rate zeroCouponRate,
157  const QuantLib::Period& curveObsLag, const QuantLib::DayCounter& curveDayCounter,
158  const boost::shared_ptr<QuantLib::ZeroInflationIndex>& index, const bool interpolated) {
159 
160  // If the baseDate in Curve is today - obsLag then the quoted zeroRate should be used
161  if (!baseDateLastKnownFixing && swapObsLag == curveObsLag)
162  return zeroCouponRate;
163 
164  QL_REQUIRE(index, "can not compute base cpi of the zero coupon swap");
165  // Otherwise we need to take into account the accrued inflation between rate helper base date and curve base date
166  // Check if historical fixings available
167  try {
168  checkIfFixingAvailable(swapStart, swapObsLag, interpolated, *index);
169  } catch (const std::exception& e) {
170  QL_FAIL("Can not estimate the curve base date, got " << e.what());
171  }
172 
173  Date swapBaseDate = swapStart - swapObsLag;
174  Date swapMaturity = swapStart + swapTenor;
175  Date swapObservationDate = swapMaturity - swapObsLag;
176 
177  // TODO check historical fixings available
178  Rate instrumentBaseCPI = cpiFixing(index, swapStart, swapObsLag, interpolated);
179  Time timeFromSwapBase =
180  inflationYearFraction(index->frequency(), interpolated, swapZCLegDayCounter, swapBaseDate, swapObservationDate);
181 
182  auto fwdCPI = instrumentBaseCPI * std::pow(1 + zeroCouponRate, timeFromSwapBase);
183 
184  Date curveBaseDt = curveBaseDate(baseDateLastKnownFixing, QuantLib::Settings::instance().evaluationDate(), curveObsLag,
185  index->frequency(), index);
186 
187  double curveBaseFixing = index->fixing(curveBaseDt);
188 
189  if (!interpolated) {
190  Time timeFromCurveBase =
191  inflationYearFraction(index->frequency(), interpolated, curveDayCounter, curveBaseDt,
192  swapObservationDate);
193 
194  return std::pow(fwdCPI / curveBaseFixing, 1.0 / timeFromCurveBase) - 1.0;
195  } else {
196  // Compute the interpolated fixing of the ZCIIS at maturity
197  auto fixingPeriod = inflationPeriod(swapObservationDate, index->frequency());
198  auto paymentPeriod = inflationPeriod(swapMaturity, index->frequency());
199  // Fixing times from curve base date
200  Time timeToFixing1 =
201  inflationYearFraction(index->frequency(), false, curveDayCounter, curveBaseDt, fixingPeriod.first);
202  Time timeToFixing2 = inflationYearFraction(index->frequency(), false, curveDayCounter, curveBaseDt,
203  fixingPeriod.second + 1 * Days);
204 
205  // Time interpolation
206  Time timeToPayment = inflationYearFraction(index->frequency(), true, curveDayCounter, curveBaseDt, swapMaturity);
207  Time timeToStartPaymentPeriod =
208  inflationYearFraction(index->frequency(), false, curveDayCounter, curveBaseDt, paymentPeriod.first);
209  Time timeToEndPaymenetPeriod = inflationYearFraction(index->frequency(), false, curveDayCounter, curveBaseDt,
210  paymentPeriod.second + 1 * Days);
211  Time interpolationFactor =
212  (timeToPayment - timeToStartPaymentPeriod) / (timeToEndPaymenetPeriod - timeToStartPaymentPeriod);
213 
214  // Root search for a constant rate that the interpolation of both cpi matches the forward cpi
215  Real target = fwdCPI / curveBaseFixing;
216 
217  std::function<double(double)> objectiveFunction = [&timeToFixing1, &timeToFixing2, &interpolationFactor,
218  &target](Rate r) {
219  return target - (std::pow(1 + r, timeToFixing1) +
220  (std::pow(1 + r, timeToFixing2) - std::pow(1 + r, timeToFixing1)) * interpolationFactor);
221  };
222 
223  Rate guess = std::pow(fwdCPI / curveBaseFixing, 1.0 / timeToFixing2) - 1.0;
224  Rate r = guess;
225  try {
226  QuantLib::Brent solver;
227  r = solver.solve(objectiveFunction, 1e-8, guess, -0.1, 0.2);
228  } catch (...) {
229  r = guess;
230  }
231  return r;
232  }
233 }
