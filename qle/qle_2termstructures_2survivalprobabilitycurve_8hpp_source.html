103 template <class T> Probability SurvivalProbabilityCurve<T>::survivalProbabilityImpl(Time t) const {
104  calculate();
105  if (t <= this->times_.back())
106  return this->interpolation_(t, true);
107 
108  // flat hazard rate extrapolation
109  Time tMax = this->times_.back();
110  Probability sMax = this->data_.back();
111  Rate hazardMax = -this->interpolation_.derivative(tMax) / sMax;
112  return sMax * std::exp(-hazardMax * (t - tMax));
113 }
114 
115 template <class T> Real SurvivalProbabilityCurve<T>::defaultDensityImpl(Time t) const {
116  calculate();
117  if (t <= this->times_.back())
118  return -this->interpolation_.derivative(t, true);
119 
120  // flat hazard rate extrapolation
121  Time tMax = this->times_.back();
122  Probability sMax = this->data_.back();
123  Rate hazardMax = -this->interpolation_.derivative(tMax) / sMax;
124  return sMax * hazardMax * std::exp(-hazardMax * (t - tMax));
125 }

156 template <class T> void SurvivalProbabilityCurve<T>::initialize() {
157  QL_REQUIRE(dates_.size() >= T::requiredPoints, "not enough input dates given");
158  QL_REQUIRE(quotes_.size() == dates_.size(), "dates/data count mismatch");
159 
160  this->times_.resize(dates_.size());
161  this->data_.resize(dates_.size());
162  this->times_[0] = 0.0;
163  for (Size i = 1; i < dates_.size(); ++i) {
164  QL_REQUIRE(dates_[i] > dates_[i - 1], "invalid date (" << dates_[i] << ", vs " << dates_[i - 1] << ")");
165  this->times_[i] = dayCounter().yearFraction(dates_[0], dates_[i]);
166  QL_REQUIRE(!close(this->times_[i], this->times_[i - 1]), "two dates correspond to the same time "
167  "under this curve's day count convention");
168  // QL_REQUIRE(this->data_[i] > 0.0, "negative probability");
169  // QL_REQUIRE(this->data_[i] <= this->data_[i-1],
170  // "negative hazard rate implied by the survival "
171  // "probability " <<
172  // this->data_[i] << " at " << dates_[i] <<

174  // "probability " <<
