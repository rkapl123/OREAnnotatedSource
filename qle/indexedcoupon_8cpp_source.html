156 IndexedCouponLeg::operator Leg() const {
157  Leg resultLeg;
158  resultLeg.reserve(underlyingLeg_.size());
159 
160  for (Size i = 0; i < underlyingLeg_.size(); ++i) {
161  bool firstValuationDate = (i == 0);
162  Date fixingDate;
163 
164  if (auto cpn = boost::dynamic_pointer_cast<Coupon>(underlyingLeg_[i])) {
165  if (valuationSchedule_.empty()) {
166  fixingDate = inArrearsFixing_ ? cpn->accrualEndDate() : cpn->accrualStartDate();
167  } else {
168  if (valuationSchedule_.size() == underlyingLeg_.size() + 1) {
169  // valuation schedule corresponds one to one to underlying schedule
170  fixingDate = inArrearsFixing_ ? valuationSchedule_.date(i + 1) : valuationSchedule_.date(i);
171  } else {
172  // look for the latest valuation date less or equal to the underlying accrual start date (if
173  // the indexing is using in advance fixing) resp. accrual end date (for in arrears fixing)
174  auto valDates = valuationSchedule_.dates();
175  Date refDate = inArrearsFixing_ ? cpn->accrualEndDate() : cpn->accrualStartDate();
176  Size index =
177  std::distance(valDates.begin(), std::upper_bound(valDates.begin(), valDates.end(), refDate));
178  QL_REQUIRE(index > 0, "IndexedCouponLeg: First valuation date ("
179  << valDates.front() << ") must be leq accrual "
180  << (inArrearsFixing_ ? "end" : "start") << " date ("
181  << cpn->accrualStartDate() << ") of the " << (i + 1)
182  << "th coupon in the underlying leg");
183  fixingDate = valuationSchedule_.date(index - 1);
184  firstValuationDate = (index == 1);
185  }
186  }
187 
188  fixingDate = fixingCalendar_.advance(fixingDate, -static_cast<int>(fixingDays_), Days, fixingConvention_);
189  if (index_ != nullptr)
190  fixingDate = index_->fixingCalendar().adjust(fixingDate, Preceding);
191 
192  if (firstValuationDate && initialFixing_ != Null<Real>()) {
193  resultLeg.push_back(boost::make_shared<IndexedCoupon>(cpn, qty_, initialFixing_));
194  } else {
195  resultLeg.push_back(boost::make_shared<IndexedCoupon>(cpn, qty_, index_, fixingDate));
196  }
197  } else if (auto csf = boost::dynamic_pointer_cast<CashFlow>(underlyingLeg_[i])) {
198  fixingDate = fixingCalendar_.advance(csf->date(), -static_cast<int>(fixingDays_), Days, fixingConvention_);
199  // use the initial fixing if the cashflow date equals the first date in the schedule
200  if (!valuationSchedule_.empty() && valuationSchedule_.dates().size() > 0 &&
201  valuationSchedule_.date(0) == csf->date() && initialFixing_ != Null<Real>()) {
202  resultLeg.push_back(boost::make_shared<IndexWrappedCashFlow>(csf, qty_, initialFixing_));
203  } else {
204  resultLeg.push_back(boost::make_shared<IndexWrappedCashFlow>(csf, qty_, index_, fixingDate));
205  }
206  } else {
207  QL_FAIL("IndexedCouponLeg: coupon or cashflow required");
208  }
209  }
210 
211  return resultLeg;
212 }

237 Real getIndexedCouponOrCashFlowMultiplier(const boost::shared_ptr<CashFlow>& c) {
238  if (auto cpn = boost::dynamic_pointer_cast<IndexedCoupon>(c)) {
239  return cpn->multiplier() * getIndexedCouponOrCashFlowMultiplier(cpn->underlying());
240  } else if (auto cf = boost::dynamic_pointer_cast<IndexWrappedCashFlow>(c)) {
241  return cf->multiplier() * getIndexedCouponOrCashFlowMultiplier(cf->underlying());
242  } else {
243  return 1.0;
244  }
