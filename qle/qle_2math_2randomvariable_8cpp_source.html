155 RandomVariable::RandomVariable(const Filter& f, const Real valueTrue, const Real valueFalse, const Real time) {
156  if (!f.initialised()) {
157  clear();
158  return;
159  }
160  n_ = f.size();
161  if (f.deterministic())
162  setAll(f.at(0) ? valueTrue : valueFalse);

164  deterministic_ = false;
165  data_.resize(n_);

167  set(i, f[i] ? valueTrue : valueFalse);

619  QL_REQUIRE(x.size() == y.size(),
620  "RandomVariable: x >= y: x size (" << x.size() << ") must be equal to y size (" << y.size() << ")");
621  checkTimeConsistency(x, y);
622  if (x.deterministic_ && y.deterministic_) {
623  return Filter(x.size(),
624  x.data_.front() > y.data_.front() || QuantLib::close_enough(x.data_.front(), y.data_.front()));
625  }
626  Filter result(x.size(), false);
627  for (Size i = 0; i < x.size(); ++i) {
628  result.set(i, x[i] > y[i] || QuantLib::close_enough(x[i], y[i]));
629  }
630  return result;
631 }

679 Array regressionCoefficients(
680  RandomVariable r, const std::vector<const RandomVariable*>& regressor,
681  const std::vector<std::function<RandomVariable(const std::vector<const RandomVariable*>&)>>& basisFn,
682  const Filter& filter) {
683  for (auto const reg : regressor) {
684  QL_REQUIRE(reg->size() == r.size(),
685  "regressor size (" << reg->size() << ") must match regressand size (" << r.size() << ")");
686  }
687  QL_REQUIRE(filter.size() == 0 || filter.size() == r.size(),
688  "filter size (" << filter.size() << ") must match regressand size (" << r.size() << ")");
689  Matrix A(r.size(), basisFn.size());
690  for (Size j = 0; j < basisFn.size(); ++j) {
691  RandomVariable a = basisFn[j](regressor);
692  if (filter.initialised()) {
693  a = applyFilter(a, filter);
694  }
695  if (a.deterministic())
696  std::fill(A.column_begin(j), A.column_end(j), a[0]);
697  else
698  std::copy(a.begin(), a.end(), A.column_begin(j));
699  }
700 
701  if (filter.size() > 0) {
702  r = applyFilter(r, filter);
703  }
704  Array b(r.size());
705  if (r.deterministic())
706  std::fill(b.begin(), b.end(), r[0]);
707  else
708  std::copy(r.begin(), r.end(), b.begin());
709  Array res = qrSolve(A, b);
710  return res;
711 }

713 RandomVariable conditionalExpectation(
714  const std::vector<const RandomVariable*>& regressor,
715  const std::vector<std::function<RandomVariable(const std::vector<const RandomVariable*>&)>>& basisFn,
716  const Array& coefficients) {
717  QL_REQUIRE(!regressor.empty(), "regressor vector is empty");
718  Size n = regressor.front()->size();
719  for (Size i = 1; i < regressor.size(); ++i) {
720  QL_REQUIRE(n == regressor[i]->size(), "regressor #" << i << " size (" << regressor[i]->size()
721  << ") must match regressor #0 size (" << n << ")");
722  }
723  QL_REQUIRE(basisFn.size() == coefficients.size(),
724  "basisFn size (" << basisFn.size() << ") must match coefficients size (" << coefficients.size() << ")");
725  RandomVariable r(n, 0.0);
726  for (Size i = 0; i < coefficients.size(); ++i) {
727  r = r + RandomVariable(n, coefficients[i]) * basisFn[i](regressor);
728  }
729  return r;
730 }

749 RandomVariable black(const RandomVariable& omega, const RandomVariable& t, const RandomVariable& strike,
750  const RandomVariable& forward, const RandomVariable& impliedVol) {
751  Filter zeroStrike = close_enough(strike, RandomVariable(omega.size(), 0.0));
752  Filter call = omega > RandomVariable(omega.size(), 0.0);
753  RandomVariable stdDev = impliedVol * sqrt(t);
754  RandomVariable d1 = log(forward / strike) / stdDev + RandomVariable(omega.size(), 0.5) * stdDev;
755  RandomVariable d2 = d1 - stdDev;
756  return applyFilter(forward, zeroStrike && call) +
757  applyInverseFilter(omega * (forward * normalCdf(omega * d1) - strike * normalCdf(omega * d2)), zeroStrike);
758 }

760 RandomVariable indicatorDerivative(const RandomVariable& x, const double eps) {
761  RandomVariable tmp(x.size(), 0.0);
762 
763  // We follow section 4, eq 10 in
764  // Fries, 2017: Automatic Backward Differentiation for American Monte-Carlo Algorithms -
765  // ADD for Conditional Expectations and Indicator Functions
766 
767  if (QuantLib::close_enough(eps, 0.0) || x.deterministic())
768  return tmp;
769 
770  Real sum = 0.0;
771  for (Size i = 0; i < x.size(); ++i) {
772  sum += x[i] * x[i];
773  }
774 
775  Real delta = std::sqrt(sum / static_cast<Real>(x.size())) * eps / 2.0;
776 
777  if (QuantLib::close_enough(delta, 0.0))
778  return tmp;
779 
780  // compute derivative
781 
782  for (Size i = 0; i < tmp.size(); ++i) {
783  Real ax = std::abs(x[i]);
784 
785  // linear approximation of step
786  // if (ax < delta) {
787  // tmp.set(i, 1.0 / (2.0 * delta));
788  // }
789 
790  // logistic function
791  tmp.set(i, std::exp(-1.0 / delta * ax) / (delta * std::pow(1.0 + std::exp(-1.0 / delta * ax), 2.0)));
792  }
793 
794  return tmp;
795 }
