32 RandomVariable LgmVectorised::numeraire(const Time t, const RandomVariable& x,
33  const Handle<YieldTermStructure> discountCurve) const {
34  QL_REQUIRE(t >= 0.0, "t (" << t << ") >= 0 required in LGM::numeraire");
35  RandomVariable Ht(x.size(), p_->H(t));
36  return exp(Ht * x + RandomVariable(x.size(), 0.5 * p_->zeta(t)) * Ht * Ht) /
37  RandomVariable(x.size(),
38  (discountCurve.empty() ? p_->termStructure()->discount(t) : discountCurve->discount(t)));
39 }

41 RandomVariable LgmVectorised::discountBond(const Time t, const Time T, const RandomVariable& x,
42  Handle<YieldTermStructure> discountCurve) const {
43  if (QuantLib::close_enough(t, T))
44  return RandomVariable(x.size(), 1.0);
45  QL_REQUIRE(T >= t && t >= 0.0, "T(" << T << ") >= t(" << t << ") >= 0 required in LGM::discountBond");
46  RandomVariable Ht(x.size(), p_->H(t));
47  RandomVariable HT(x.size(), p_->H(T));
48  return RandomVariable(x.size(),
49  (discountCurve.empty() ? p_->termStructure()->discount(T) / p_->termStructure()->discount(t)
50  : discountCurve->discount(T) / discountCurve->discount(t))) *
51  exp(-(HT - Ht) * x - RandomVariable(x.size(), 0.5 * p_->zeta(t)) * (HT * HT - Ht * Ht));
52 }

54 RandomVariable LgmVectorised::reducedDiscountBond(const Time t, const Time T, const RandomVariable& x,
55  const Handle<YieldTermStructure> discountCurve) const {
56  if (QuantLib::close_enough(t, T))
57  return RandomVariable(x.size(), 1.0) / numeraire(t, x, discountCurve);
58  QL_REQUIRE(T >= t && t >= 0.0, "T(" << T << ") >= t(" << t << ") >= 0 required in LGM::reducedDiscountBond");
59  RandomVariable HT(x.size(), p_->H(T));
60  return RandomVariable(x.size(),
61  (discountCurve.empty() ? p_->termStructure()->discount(T) : discountCurve->discount(T))) *
62  exp(-HT * x - RandomVariable(x.size(), 0.5 * p_->zeta(t)) * HT * HT);
63 }

65 RandomVariable LgmVectorised::fixing(const boost::shared_ptr<InterestRateIndex>& index, const Date& fixingDate,
66  const Time t, const RandomVariable& x) const {
67 
68  // handle case where fixing is deterministic
69  Date today = Settings::instance().evaluationDate();
70  if (fixingDate <= today)
71  return RandomVariable(x.size(), index->fixing(fixingDate));
72 
73  // handle stochastic fixing
74  if (auto ibor = boost::dynamic_pointer_cast<IborIndex>(index)) {
75  Date d1 = ibor->valueDate(fixingDate);
76  Date d2 = ibor->maturityDate(d1);
77  Time T1 = p_->termStructure()->timeFromReference(d1);
78  Time T2 = p_->termStructure()->timeFromReference(d2);
79  Time dt = ibor->dayCounter().yearFraction(d1, d2);
80  // reducedDiscountBond is faster to compute, there we use this here instead of discountBond
81  RandomVariable disc1 = reducedDiscountBond(t, T1, x, ibor->forwardingTermStructure());
82  RandomVariable disc2 = reducedDiscountBond(t, T2, x, ibor->forwardingTermStructure());
83  return (disc1 / disc2 - RandomVariable(x.size(), 1.0)) / RandomVariable(x.size(), dt);
84  } else if (auto swap = boost::dynamic_pointer_cast<SwapIndex>(index)) {
85  auto swapDiscountCurve =
86  swap->exogenousDiscount() ? swap->discountingTermStructure() : swap->forwardingTermStructure();
87  Leg floatingLeg, fixedLeg;
88  if (auto ois = boost::dynamic_pointer_cast<OvernightIndexedSwapIndex>(index)) {
89  auto underlying = ois->underlyingSwap(fixingDate);
90  floatingLeg = underlying->overnightLeg();
91  fixedLeg = underlying->fixedLeg();
92  } else {
93  auto underlying = swap->underlyingSwap(fixingDate);
94  floatingLeg = underlying->floatingLeg();
95  fixedLeg = underlying->fixedLeg();
96  }
97  RandomVariable numerator(x.size(), 0.0), denominator(x.size(), 0.0);
98  for (auto const& c : floatingLeg) {
99  if (auto cpn = boost::dynamic_pointer_cast<IborCoupon>(c)) {
100  Date fixingValueDate = swap->iborIndex()->fixingCalendar().advance(
101  cpn->fixingDate(), swap->iborIndex()->fixingDays(), Days);
102  Date fixingEndDate = cpn->fixingEndDate();
103  Time T1 = p_->termStructure()->timeFromReference(fixingValueDate);
104  Time T2 = p_->termStructure()->timeFromReference(fixingEndDate); // accounts for QL_INDEXED_COUPON
105  Time T3 = p_->termStructure()->timeFromReference(cpn->date());
106  RandomVariable disc1 = reducedDiscountBond(t, T1, x, swap->forwardingTermStructure());
107  RandomVariable disc2 = reducedDiscountBond(t, T2, x, swap->forwardingTermStructure());
108  Real adjFactor =
109  cpn->dayCounter().yearFraction(cpn->accrualStartDate(), cpn->accrualEndDate(),
110  cpn->referencePeriodStart(), cpn->referencePeriodEnd()) /
111  swap->iborIndex()->dayCounter().yearFraction(fixingValueDate, fixingEndDate);
112  RandomVariable tmp = disc1 / disc2 - RandomVariable(x.size(), 1.0);
113  if (!QuantLib::close_enough(adjFactor, 1.0)) {
114  tmp *= RandomVariable(x.size(), adjFactor);
115  }
116  numerator += tmp * reducedDiscountBond(t, T3, x, swapDiscountCurve);
117  } else if (auto cpn = boost::dynamic_pointer_cast<OvernightIndexedCoupon>(c)) {
118  Date start = cpn->valueDates().front();
119  Date end = cpn->valueDates().back();
120  Time T1 = p_->termStructure()->timeFromReference(start);
121  Time T2 = p_->termStructure()->timeFromReference(end);
122  Time T3 = p_->termStructure()->timeFromReference(cpn->date());
123  RandomVariable disc1 = reducedDiscountBond(t, T1, x, swap->forwardingTermStructure());
124  RandomVariable disc2 = reducedDiscountBond(t, T2, x, swap->forwardingTermStructure());
125  Real adjFactor =
126  cpn->dayCounter().yearFraction(cpn->accrualStartDate(), cpn->accrualEndDate(),
127  cpn->referencePeriodStart(), cpn->referencePeriodEnd()) /
128  swap->iborIndex()->dayCounter().yearFraction(start, end);
129  RandomVariable tmp;
130  if (cpn->averagingMethod() == RateAveraging::Compound) {
131  tmp = disc1 / disc2 - RandomVariable(x.size(), 1.0);
132  } else if (cpn->averagingMethod() == RateAveraging::Simple) {
133  tmp = QuantExt::log(disc1 / disc2);
134  } else {
135  QL_FAIL("LgmVectorised::fixing(): RateAveraging '"
136  << static_cast<int>(cpn->averagingMethod())
137  << "' not handled - internal error, contact dev.");
138  }
139  if (!QuantLib::close_enough(adjFactor, 1.0)) {
140  tmp *= RandomVariable(x.size(), adjFactor);
141  }
142  numerator += tmp * reducedDiscountBond(t, T3, x, swapDiscountCurve);
143  } else {
144  QL_FAIL("LgmVectorised::fixing(): excepted ibor coupon");
145  }
146  }
147  for (auto const& c : fixedLeg) {
148  auto cpn = boost::dynamic_pointer_cast<FixedRateCoupon>(c);
149  QL_REQUIRE(cpn, "LgmVectorised::fixing(): excepted fixed coupon");
150  Date d = cpn->date();
151  Time T = p_->termStructure()->timeFromReference(d);
152  denominator +=
153  reducedDiscountBond(t, T, x, swapDiscountCurve) * RandomVariable(x.size(), cpn->accrualPeriod());
154  }
155  return numerator / denominator;
156  } else {
157  QL_FAIL("LgmVectorised::fixing(): index ('" << index->name() << "') must be ibor or swap index");
158  }
159 }

161 RandomVariable LgmVectorised::compoundedOnRate(const boost::shared_ptr<OvernightIndex>& index,
162  const std::vector<Date>& fixingDates,
163  const std::vector<Date>& valueDates, const std::vector<Real>& dt,
164  const Natural rateCutoff, const bool includeSpread, const Real spread,
165  const Real gearing, const Period lookback,
166  const DayCounter& accrualDayCounter, const Real cap, const Real floor,
167  const bool localCapFloor, const bool nakedOption, const Time t,
168  const RandomVariable& x) const {
169 
170  QL_REQUIRE(!localCapFloor, "LgmVectorised::compoundedOnRate(): localCapFloor = true is not supported");
171  QL_REQUIRE(!nakedOption, "LgmVectorised::compoundedOnRate(): nakedOption = true is not supported");
172 
173  QL_REQUIRE(!includeSpread || QuantLib::close_enough(gearing, 1.0),
174  "LgmVectorised::compoundedOnRate(): if include spread = true, only a gearing 1.0 is allowed - scale "
175  "the notional in this case instead.");
176 
177  QL_REQUIRE(rateCutoff < dt.size(), "LgmVectorised::compoundedOnRate(): rate cutoff ("
178  << rateCutoff << ") must be less than number of fixings in period ("
179  << dt.size() << ")");
180 
181  /* We allow the observation time t to be later than the value dates for which to project ON fixings.
182  In this case we project the period from the first (future) value date to the last value date starting
183  from t, but use the actual portion of the underlying curve.
184  As a refinement, we might consider to scale x down to the volatility corresponding to the first future
185  value date as well (TODO) - this is all experimental and an approximation to meet the requirements of
186  an 1D backward solver, i.e. to be able to price e.g. Bermudan OIS swaptions in an efficient way. */
187 
188  // the following is similar to the code in the overnight index coupon pricer
189 
190  Size i = 0, n = dt.size();
191  Size nCutoff = n - rateCutoff;
192  Real compoundFactor = 1.0, compoundFactorWithoutSpread = 1.0;
193 
194  Date today = Settings::instance().evaluationDate();
195 
196  while (i < n && fixingDates[std::min(i, nCutoff)] < today) {
197  Rate pastFixing = IndexManager::instance().getHistory(index->name())[fixingDates[std::min(i, nCutoff)]];
198  QL_REQUIRE(pastFixing != Null<Real>(), "LgmVectorised::compoundedOnRate(): Missing "
199  << index->name() << " fixing for "
200  << fixingDates[std::min(i, nCutoff)]);
201  if (includeSpread) {
202  compoundFactorWithoutSpread *= (1.0 + pastFixing * dt[i]);
203  pastFixing += spread;
204  }
205  compoundFactor *= (1.0 + pastFixing * dt[i]);
206  ++i;
207  }
208 
209  if (i < n && fixingDates[std::min(i, nCutoff)] == today) {
210  Rate pastFixing = IndexManager::instance().getHistory(index->name())[fixingDates[std::min(i, nCutoff)]];
211  if (pastFixing != Null<Real>()) {
212  if (includeSpread) {
213  compoundFactorWithoutSpread *= (1.0 + pastFixing * dt[i]);
214  pastFixing += spread;
215  }
216  compoundFactor *= (1.0 + pastFixing * dt[i]);
217  ++i;
218  }
219  }
220 
221  RandomVariable compoundFactorLgm(x.size(), compoundFactor),
222  compoundFactorWithoutSpreadLgm(x.size(), compoundFactorWithoutSpread);
223 
224  if (i < n) {
225  Handle<YieldTermStructure> curve = index->forwardingTermStructure();
226  QL_REQUIRE(!curve.empty(),
227  "LgmVectorised::compoundedOnRate(): null term structure set to this instance of " << index->name());
228 
229  DiscountFactor startDiscount = curve->discount(valueDates[i]);
230  DiscountFactor endDiscount = curve->discount(valueDates[std::max(nCutoff, i)]);
231 
232  if (nCutoff < n) {
233  DiscountFactor discountCutoffDate =
234  curve->discount(valueDates[nCutoff] + 1) / curve->discount(valueDates[nCutoff]);
235  endDiscount *= std::pow(discountCutoffDate, valueDates[n] - valueDates[nCutoff]);
236  }
237 
238  // the times associated to the projection on the T0 curve
239 
240  Real T1 = p_->termStructure()->timeFromReference(valueDates[i]);
241  Real T2 = p_->termStructure()->timeFromReference(valueDates[n]);
242 
243  // the times we use for the projection in the LGM model, if t > T1 they are displaced by (t-T1)
244 
245  Real T1_lgm = T1, T2_lgm = T2;
246  if (t > T1) {
247  T1_lgm += t - T1;
248  T2_lgm += t - T1;
249  }
250 
251  // the discount factors estimated in the lgm model
252 
253  RandomVariable disc1 = reducedDiscountBond(t, T1_lgm, x, curve);
254  RandomVariable disc2 = reducedDiscountBond(t, T2_lgm, x, curve);
255 
256  // apply a correction to the discount factors
257 
258  disc1 *= RandomVariable(x.size(), startDiscount / curve->discount(T1_lgm));
259  disc2 *= RandomVariable(x.size(), endDiscount / curve->discount(T2_lgm));
260 
261  // continue with the usual computation
262 
263  compoundFactorLgm *= disc1 / disc2;
264 
265  if (includeSpread) {
266  compoundFactorWithoutSpreadLgm *= disc1 / disc2;
267  Real tau =
268  accrualDayCounter.yearFraction(valueDates[i], valueDates.back()) / (valueDates.back() - valueDates[i]);
269  compoundFactorLgm *= RandomVariable(
270  x.size(), std::pow(1.0 + tau * spread, static_cast<int>(valueDates.back() - valueDates[i])));
271  }
272  }
273 
274  Rate tau = accrualDayCounter.yearFraction(valueDates.front(), valueDates.back());
275  RandomVariable rate = (compoundFactorLgm - RandomVariable(x.size(), 1.0)) / RandomVariable(x.size(), tau);
276  RandomVariable swapletRate = RandomVariable(x.size(), gearing) * rate;
277  // RandomVariable effectiveSpread, effectiveIndexFixing;
278  if (!includeSpread) {
279  swapletRate += RandomVariable(x.size(), spread);
280  // effectiveSpread = RandomVariable(x.size(), spread);
281  // effectiveIndexFixing = RandomVaraible(x, rate);
282  } else {
283  // effectiveSpread =
284  // rate - (compoundFactorWithoutSpreadLgm - RandomVariable(x.size() 1.0)) / RandomVariable(x.size(), tau);
285  // effectiveIndexFixing = rate - effectiveSpread_;
286  }
287 
288  RandomVariable floorVal(x.size(), floor == Null<Real>() ? -QL_MAX_REAL : floor);
289  RandomVariable capVal(x.size(), cap == Null<Real>() ? QL_MAX_REAL : cap);
290 
291  return max(floorVal, min(capVal, swapletRate));
292 }

294 RandomVariable LgmVectorised::averagedOnRate(const boost::shared_ptr<OvernightIndex>& index,
295  const std::vector<Date>& fixingDates, const std::vector<Date>& valueDates,
296  const std::vector<Real>& dt, const Natural rateCutoff,
297  const bool includeSpread, const Real spread, const Real gearing,
298  const Period lookback, const DayCounter& accrualDayCounter, const Real cap,
299  const Real floor, const bool localCapFloor, const bool nakedOption,
300  const Time t, const RandomVariable& x) const {
301 
302  QL_REQUIRE(!localCapFloor, "LgmVectorised::averageOnRate(): localCapFloor = true is not supported");
303  QL_REQUIRE(!nakedOption, "LgmVectorised::averagedOnRate(): nakedOption = true is not supported");
304 
305  QL_REQUIRE(!includeSpread || QuantLib::close_enough(gearing, 1.0),
306  "LgmVectorised::averageOnRate(): if include spread = true, only a gearing 1.0 is allowed - scale "
307  "the notional in this case instead.");
308 
309  QL_REQUIRE(rateCutoff < dt.size(), "LgmVectorised::averageOnRate(): rate cutoff ("
310  << rateCutoff << ") must be less than number of fixings in period ("
311  << dt.size() << ")");
312 
313  /* Same comment on t as in compoundedOnRate() above applies here */
314 
315  // the following is similar to the code in the overnight index coupon pricer
316 
317  Size i = 0, n = dt.size();
318  Size nCutoff = n - rateCutoff;
319  Real accumulatedRate = 0.0;
320 
321  Date today = Settings::instance().evaluationDate();
322 
323  while (i < n && fixingDates[std::min(i, nCutoff)] < today) {
324  Rate pastFixing = IndexManager::instance().getHistory(index->name())[fixingDates[std::min(i, nCutoff)]];
325  QL_REQUIRE(pastFixing != Null<Real>(), "LgmVectorised::averageOnRate(): Missing "
326  << index->name() << " fixing for "
327  << fixingDates[std::min(i, nCutoff)]);
328  accumulatedRate += pastFixing * dt[i];
329  ++i;
330  }
331 
332  if (i < n && fixingDates[std::min(i, nCutoff)] == today) {
333  Rate pastFixing = IndexManager::instance().getHistory(index->name())[fixingDates[std::min(i, nCutoff)]];
334  if (pastFixing != Null<Real>()) {
335  accumulatedRate += pastFixing * dt[i];
336  ++i;
337  }
338  }
339 
340  RandomVariable accumulatedRateLgm(x.size(), accumulatedRate);
341 
342  if (i < n) {
343  Handle<YieldTermStructure> curve = index->forwardingTermStructure();
344  QL_REQUIRE(!curve.empty(),
345  "LgmVectorised::averageOnRate(): null term structure set to this instance of " << index->name());
346 
347  DiscountFactor startDiscount = curve->discount(valueDates[i]);
348  DiscountFactor endDiscount = curve->discount(valueDates[std::max(nCutoff, i)]);
349 
350  if (nCutoff < n) {
351  DiscountFactor discountCutoffDate =
352  curve->discount(valueDates[nCutoff] + 1) / curve->discount(valueDates[nCutoff]);
353  endDiscount *= std::pow(discountCutoffDate, valueDates[n] - valueDates[nCutoff]);
354  }
355 
356  // the times associated to the projection on the T0 curve
357 
358  Real T1 = p_->termStructure()->timeFromReference(valueDates[i]);
359  Real T2 = p_->termStructure()->timeFromReference(valueDates[n]);
360 
361  // the times we use for the projection in the LGM model, if t > T1 they are displaced by (t-T1)
362 
363  Real T1_lgm = T1, T2_lgm = T2;
364  if (t > T1) {
365  T1_lgm += t - T1;
366  T2_lgm += t - T1;
367  }
368 
369  // the discount factors estimated in the lgm model
370 
371  RandomVariable disc1 = reducedDiscountBond(t, T1_lgm, x, curve);
372  RandomVariable disc2 = reducedDiscountBond(t, T2_lgm, x, curve);
373 
374  // apply a correction to the discount factors
375 
376  disc1 *= RandomVariable(x.size(), startDiscount / curve->discount(T1_lgm));
377  disc2 *= RandomVariable(x.size(), endDiscount / curve->discount(T2_lgm));
378 
379  // continue with the usual computation
380 
381  accumulatedRateLgm += log(disc1 / disc2);
382  }
383 
384  Rate tau = accrualDayCounter.yearFraction(valueDates.front(), valueDates.back());
385  RandomVariable rate =
386  RandomVariable(x.size(), gearing / tau) * accumulatedRateLgm + RandomVariable(x.size(), spread);
387  RandomVariable floorVal(x.size(), floor == Null<Real>() ? -QL_MAX_REAL : floor);
388  RandomVariable capVal(x.size(), cap == Null<Real>() ? QL_MAX_REAL : cap);
389  return max(floorVal, min(capVal, rate));
390 }

392 RandomVariable LgmVectorised::averagedBmaRate(const boost::shared_ptr<BMAIndex>& index,
393  const std::vector<Date>& fixingDates, const Date& accrualStartDate,
394  const Date& accrualEndDate, const Time t, const RandomVariable& x) const {
395 
396  // similar to AverageBMACouponPricer
397 
398  Natural cutoffDays = 0;
399  Date startDate = accrualStartDate - cutoffDays;
400  Date endDate = accrualEndDate - cutoffDays;
401  Date d1 = startDate, d2 = startDate;
402 
403  QL_REQUIRE(!fixingDates.empty(), "LgmVectorised::averagedBmaRate(): fixing date list empty");
404  QL_REQUIRE(index->valueDate(fixingDates.front()) <= startDate,
405  "LgmVectorised::averagedBmaRate(): first fixing date valid after period start");
406  QL_REQUIRE(index->valueDate(fixingDates.back()) >= endDate,
407  "LgmVectorised::averagedBmaRate(): last fixing date valid before period end");
408 
409  Handle<YieldTermStructure> curve = index->forwardingTermStructure();
410  QL_REQUIRE(!curve.empty(),
411  "LgmVectorised::averagedBmaRate(): null term structure set to this instance of " << index->name());
412 
413  Date today = Settings::instance().evaluationDate();
414 
415  RandomVariable avgBMA(x.size(), 0.0);
416 
417  for (Size i = 0; i < fixingDates.size() - 1; ++i) {
418  Date valueDate = index->valueDate(fixingDates[i]);
419  Date nextValueDate = index->valueDate(fixingDates[i + 1]);
420  if (fixingDates[i] >= endDate || valueDate >= endDate)
421  break;
422  if (fixingDates[i + 1] < startDate || nextValueDate <= startDate)
423  continue;
424 
425  d2 = std::min(nextValueDate, endDate);
426  RandomVariable fixing;
427  if (fixingDates[i] <= today) {
428  // past fixing or forecast today's fixing on T0 curve (which is ok, since model independent)
429  fixing = RandomVariable(x.size(), index->fixing(fixingDates[i]));
430  } else {
431 
432  Date start = index->fixingCalendar().advance(fixingDates[i], 1, Days);
433  Date end = index->maturityDate(start);
434  DiscountFactor startDiscount = curve->discount(start);
435  DiscountFactor endDiscount = curve->discount(end);
436 
437  // the times associated to the projection on the T0 curve
438 
439  Real T1 = p_->termStructure()->timeFromReference(start);
440  Real T2 = p_->termStructure()->timeFromReference(end);
441 
442  // the times we use for the projection in the LGM model, if t > T1 they are displaced by (t-T1)
443 
444  Real T1_lgm = T1, T2_lgm = T2;
445  if (t > T1) {
446  T1_lgm += t - T1;
447  T2_lgm += t - T1;
448  }
449 
450  // the discount factors estimated in the lgm model
451 
452  RandomVariable disc1 = reducedDiscountBond(t, T1_lgm, x, curve);
453  RandomVariable disc2 = reducedDiscountBond(t, T2_lgm, x, curve);
454 
455  // apply a correction to the discount factors
456 
457  disc1 *= RandomVariable(x.size(), startDiscount / curve->discount(T1_lgm));
458  disc2 *= RandomVariable(x.size(), endDiscount / curve->discount(T2_lgm));
459 
460  // estimate the fixing
461 
462  fixing = (disc1 / disc2 - RandomVariable(x.size(), 1.0)) /
463  RandomVariable(x.size(), index->dayCounter().yearFraction(start, end));
464  }
465 
466  avgBMA += fixing * RandomVariable(x.size(), (d2 - d1));
467  d1 = d2;
468  }
469 
470  avgBMA /= RandomVariable(x.size(), (endDate - startDate));
471 
472  return avgBMA;
473 }
