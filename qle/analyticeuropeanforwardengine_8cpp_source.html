74 void AnalyticEuropeanForwardEngine::calculate() const {
75
76 // if the discount curve is not specified, we default to the
77 // risk free rate curve embedded within the GBM process
78 boost::shared_ptr<YieldTermStructure> discountPtr =
79 discountCurve_.empty() ?
80 process_->riskFreeRate().currentLink() :
81 discountCurve_.currentLink();
82
83 QL_REQUIRE(arguments_.exercise->type() == Exercise::European,
84 "not an European option");
85
86 boost::shared_ptr<StrikedTypePayoff> payoff =
87 boost::dynamic_pointer_cast<StrikedTypePayoff>(arguments_.payoff);
88 QL_REQUIRE(payoff, "non-striked payoff given");
89
90 Real variance =
91 process_->blackVolatility()->blackVariance(
92 arguments_.exercise->lastDate(),
93 payoff->strike());
94 DiscountFactor dividendDiscount =
95 process_->dividendYield()->discount(
96 arguments_.forwardDate);
97 DiscountFactor df = discountPtr->discount(arguments_.exercise->lastDate());
98 DiscountFactor riskFreeDiscountForFwdEstimation =
99 process_->riskFreeRate()->discount(arguments_.forwardDate);
100 Real spot = process_->stateVariable()->value();
101 QL_REQUIRE(spot > 0.0, "negative or null underlying given");
102 Real forwardPrice = spot * dividendDiscount / riskFreeDiscountForFwdEstimation;
103
104 QuantLib::BlackCalculator black(payoff, forwardPrice, std::sqrt(variance),df);
105
106
107 results_.value = black.value();
108 results_.delta = black.delta(spot);
109 results_.deltaForward = black.deltaForward();
110 results_.elasticity = black.elasticity(spot);
111 results_.gamma = black.gamma(spot);
112
113 DayCounter rfdc = discountPtr->dayCounter();
114 DayCounter divdc = process_->dividendYield()->dayCounter();
115 DayCounter voldc = process_->blackVolatility()->dayCounter();
116 Time t = rfdc.yearFraction(process_->riskFreeRate()->referenceDate(),
117 arguments_.exercise->lastDate());
118 results_.rho = black.rho(t);
119
120 t = divdc.yearFraction(process_->dividendYield()->referenceDate(),
121 arguments_.exercise->lastDate());
122 results_.dividendRho = black.dividendRho(t);
123
124 t = voldc.yearFraction(process_->blackVolatility()->referenceDate(),
125 arguments_.exercise->lastDate());
126 results_.vega = black.vega(t);
127 try {
128 results_.theta = black.theta(spot, t);
129 results_.thetaPerDay =
130 black.thetaPerDay(spot, t);
131 } catch (Error&) {
132 results_.theta = Null<Real>();
133 results_.thetaPerDay = Null<Real>();
134 }
135
136 results_.strikeSensitivity = black.strikeSensitivity();
137 results_.itmCashProbability = black.itmCashProbability();
138
139 Real tte = process_->blackVolatility()->timeFromReference(arguments_.exercise->lastDate());
140 results_.additionalResults["spot"] = spot;
141 results_.additionalResults["dividendDiscount"] = dividendDiscount;
142 results_.additionalResults["riskFreeDiscount"] = riskFreeDiscountForFwdEstimation;
143 results_.additionalResults["forward"] = forwardPrice;
144 results_.additionalResults["strike"] = payoff->strike();
145 results_.additionalResults["volatility"] = std::sqrt(variance / tte);
146 results_.additionalResults["timeToExpiry"] = tte;
147 results_.additionalResults["discountFactor"] = df;
148 }
