98BOOST_DATA_TEST_CASE(testMartingaleProperty,
99 bdata::make(driftFreeState) * bdata::make(steps),
100 driftFreeState, steps) {
101
102 //BOOST_AUTO_TEST_CASE(testMartingaleProperty) {
103
104 BOOST_TEST_MESSAGE("Testing martingale property in the COM Schwartz model ...");
105
106 CommoditySchwartzModelTestData data(driftFreeState);
107 boost::shared_ptr<StochasticProcess> process = data.model->stateProcess();
108 QL_REQUIRE(process != NULL, "process has null pointer!");
109
110 Size n = 100000; // number of paths
111 Size seed = 42; // rng seed
112 Time t = 10.0; // simulation horizon
113 Time T = 20.0; // forward price maturity
114 //Size steps = 1;
115
116 TimeGrid grid(t, steps);
117 LowDiscrepancy::rsg_type sg = LowDiscrepancy::make_sequence_generator(steps, seed);
118 MultiPathGenerator<LowDiscrepancy::rsg_type> pg(process, grid, sg, false);
119 //MultiPathGeneratorMersenneTwister pg(process, grid, seed, true);
120
121 accumulator_set<double, stats<tag::mean, tag::variance, tag::error_of<tag::mean>>> acc_price, acc_state;
122
123 Array state(1);
124 for (Size j = 0; j < n; ++j) {
125 Sample<MultiPath> path = pg.next();
126 Size l = path.value[0].length() - 1;
127 state[0] = path.value[0][l];
128 Real price = data.model->forwardPrice(t, T, state);
129 acc_price(price);
130 acc_state(state[0]);
131 }
132
133 BOOST_TEST_MESSAGE("sigma = " << data.model->parametrization()->sigmaParameter());
134 BOOST_TEST_MESSAGE("kappa = " << data.model->parametrization()->kappaParameter());
135 BOOST_TEST_MESSAGE("samples = " << n);
136 BOOST_TEST_MESSAGE("steps = " << steps);
137 BOOST_TEST_MESSAGE("t = " << t);
138 BOOST_TEST_MESSAGE("T = " << T);
139
140 // Martingale test for F(t,T)
141 {
142 Real found = mean(acc_price);
143 Real expected = data.parametrization->priceCurve()->price(T);
144 Real error = error_of<tag::mean>(acc_price);
145
146 BOOST_TEST_MESSAGE("Check that E[F(t,T)] = F(0,T)");
147 BOOST_TEST_MESSAGE("Avg = " << found
148 << " +- " << error
149 << " vs expected " << expected);
150 BOOST_TEST(fabs(found - expected) < error, "Martingale test failed for F(t,T) evolution, found " << found << ", expected " << expected);
151 }
152
153 // Martingale test for the state variable
154 {
155 Real found = mean(acc_state);
156 Real expected = 0;
157 Real error = error_of<tag::mean>(acc_state);
158
159 BOOST_TEST_MESSAGE("Check that the mean of the state variable is zero");
160 BOOST_TEST_MESSAGE("Avg = " << found
161 << " +- " << error_of<tag::mean>(acc_state)
162 << " vs expected " << expected);
163 BOOST_TEST(fabs(found - expected) < error, "Martingale test failed for the state variable, found " << found << ", expected " << expected);
164 }
165
166 // Variance test for the state variable, implicit in the first test above
167 {
168 Real found = variance(acc_state);
169 Real expected = data.parametrization->variance(t);
170 // FIXME: What's the MC error here? I assume that the mean error is smaller than the error of variance estimate, so re-using it.
171 Real error = error_of<tag::mean>(acc_state);
172
173 boost::shared_ptr<CommoditySchwartzStateProcess> stateProcess = boost::dynamic_pointer_cast<CommoditySchwartzStateProcess>(process);
174 QL_REQUIRE(stateProcess, "state process is null");
175 Real expected2 = stateProcess->variance(0.0, 0.0, t);
176
177 // consistency check, should be identical
178 BOOST_TEST(fabs(expected - expected2) < 1e-10, "Inconsistent state variable variance " << expected << " vs " << expected2);
179
180 BOOST_TEST_MESSAGE("Check that the variance of the state variable matches expectation");
181 BOOST_TEST_MESSAGE("Var = " << found
182 << " +- " << error
183 << " vs expected " << expected);
184 BOOST_TEST(fabs(found - expected2) < error,
185 "Simulated variance of the state variable does match expectation, found " << found << ", expected " << expected);
186
187 }
