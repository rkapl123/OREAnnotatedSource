133inline const Sample<MultiPath>& MultiPathGeneratorMersenneTwister::next() const {
134 if (antitheticSampling_) {
135 antitheticVariate_ = !antitheticVariate_;
136 return antitheticVariate_ ? pg_->antithetic() : pg_->next();
137 } else {
138 return pg_->next();
139 }
