105Date FdConvertibleBondEvents::nextExerciseDate(const Date& d,
106 const std::vector<ConvertibleBond2::CallabilityData>& data) const {
107 Date result = Date::maxDate();
108 for (auto const& x : data) {
109 if (x.exerciseDate > d)
110 result = std::min(result, x.exerciseDate);
111 }
112 if (result == Date::maxDate())
113 return Null<Date>();
114 return result;
115}

120 if (x.exerciseDate > d)

122 }
123 if (result == Date::maxDate())
124 return Null<Date>();

128void FdConvertibleBondEvents::processBondCashflows() {
129 lastRedemptionDate_ = Date::minDate();
130 for (auto const& c : registeredBondCashflows_) {
131 if (boost::dynamic_pointer_cast<Coupon>(c) == nullptr)
132 lastRedemptionDate_ = std::max(lastRedemptionDate_, c->date());
133 }
134 for (auto const& d : registeredBondCashflows_) {
135 bool isRedemption = boost::dynamic_pointer_cast<Coupon>(d) == nullptr;
136 Size index = grid_.index(time(d->date()));
137 hasBondCashflow_[index] = true;
138 associatedDate_[index] = d->date();
139 if (isRedemption && d->date() == lastRedemptionDate_)
140 bondFinalRedemption_[index] += d->amount();
141 else
142 bondCashflow_[index] += d->amount();
143 }

146void FdConvertibleBondEvents::processExerciseData(const std::vector<ConvertibleBond2::CallabilityData>& sourceData,
147 std::vector<bool>& targetFlags, std::vector<CallData>& targetData) {
148 for (auto const& c : sourceData) {
149 if (c.exerciseDate <= today_ && c.exerciseType == ConvertibleBond2::CallabilityData::ExerciseType::OnThisDate)
150 continue;
151 Size indexStart = grid_.index(time(std::max(c.exerciseDate, today_)));
152 Size indexEnd;
153 associatedDate_[indexStart] = std::max(c.exerciseDate, today_);
154 if (c.exerciseType == ConvertibleBond2::CallabilityData::ExerciseType::OnThisDate) {
155 indexEnd = indexStart;
156 } else if (c.exerciseType == ConvertibleBond2::CallabilityData::ExerciseType::FromThisDateOn) {
157 Date nextDate = nextExerciseDate(c.exerciseDate, sourceData);
158 QL_REQUIRE(nextDate != Null<Date>(),
159 "FdConvertibleBondEvents::processExerciseData(): internal error: did not find a next exercise "
160 "date after "
161 << c.exerciseDate
162 << ", the last exercise date should not have exercise type FromThisDateOn");
163 indexEnd = grid_.index(time(nextDate)) - 1;
164 } else {
165 QL_FAIL("FdConvertibleBondEvents: internal error, exercise type not "
166 "recognized");
167 }
168 for (Size i = indexStart; i <= indexEnd; ++i) {
169 targetFlags[i] = true;
170 targetData[i] = CallData{c.price, c.priceType, c.includeAccrual,
171 c.isSoft, c.softTriggerRatio, std::function<Real(Real, Real)>()};
172 }
173 }

176void FdConvertibleBondEvents::processMakeWholeData() {
177 if (registeredMakeWholeData_.crIncreaseData) {
178
179 // init and checks
180
181 auto const& stockPrices = (*registeredMakeWholeData_.crIncreaseData).stockPrices;
182 auto const& effDates = (*registeredMakeWholeData_.crIncreaseData).effectiveDates;
183 auto const& crInc = (*registeredMakeWholeData_.crIncreaseData).crIncrease;
184 QL_REQUIRE(stockPrices.size() >= 2,
185 "FdConvertibleBondEvents::processMakeWholeData(): at least two stock prices required (cr increase)");
186 QL_REQUIRE(
187 effDates.size() >= 2,
188 "FdConvertibleBondEvents::processMakeWholeData(): at least two effective dates required (cr increase)");
189 QL_REQUIRE(effDates.size() == crInc.size(), "FdConvertibleBondEvents::processMakeWholeData(): effective dates ("
190 << effDates.size() << ") must match cr increase rows ("
191 << crInc.size() << ") (cr increase)");
192 for (auto const& c : crInc) {
193 QL_REQUIRE(c.size() == stockPrices.size(),
194 "FdConvertibleBondEvents::processMakeWholeData(): stock prices size ("
195 << stockPrices.size() << ") must match cr increase coluns (" << c.size() << ")");
196 }
197
198 // build interpolation
199
200 mw_cr_inc_x_ = Array(stockPrices.begin(), stockPrices.end());
201 mw_cr_inc_y_ = Array(effDates.size());
202 mw_cr_inc_z_ = Matrix(mw_cr_inc_y_.size(), mw_cr_inc_x_.size());
203
204 for (Size i = 0; i < effDates.size(); ++i) {
205 mw_cr_inc_y_[i] = time(effDates[i]);
206 for (Size j = 0; j < stockPrices.size(); ++j) {
207 mw_cr_inc_z_(i, j) = crInc[i][j];
208 }
209 }
210
211 auto interpolation = boost::make_shared<BilinearInterpolation>(
212 mw_cr_inc_x_.begin(), mw_cr_inc_x_.end(), mw_cr_inc_y_.begin(), mw_cr_inc_y_.end(), mw_cr_inc_z_);
213
214 // init cap (infty if not given)
215
216 Real cap = QL_MAX_REAL;
217 if ((*registeredMakeWholeData_.crIncreaseData).cap != Null<Real>())
218 cap = (*registeredMakeWholeData_.crIncreaseData).cap;
219
220 // set effetive mw cr functions on the call data
221
222 for (Size i = 0; i < grid_.size(); ++i) {
223 if (hasCall_[i]) {
224 Real t = grid_[i];
225 callData_[i].mwCr = [interpolation, t, cap](const Real S, const Real cr) {
226 if ((S < interpolation->xMin() && !close_enough(S, interpolation->xMin())) ||
227 (S > interpolation->xMax() && !close_enough(S, interpolation->xMax())) ||
228 (t < interpolation->yMin() && !close_enough(t, interpolation->yMin())) ||
229 (t > interpolation->yMax() && !close_enough(t, interpolation->yMax()))) {
230 return cr;
231 } else {
232 Real tmp = interpolation->operator()(S, t);
233 // apply cap, but if initial cr was already greater than result, keep that
234 return std::max(cr, std::min(cr + tmp, cap));
235 }
236 };
237 }
238 }
239 }
240}

242void FdConvertibleBondEvents::processConversionAndDivProtData() {
243
244 // set the initial conversion ratio
245
246 std::set<std::pair<Date, Real>> crSchedule;
247 for (const auto& d : registeredConversionRatioData_) {
248 crSchedule.insert(std::make_pair(d.fromDate, d.conversionRatio));
249 }
250 if (!crSchedule.empty()) {
251 initialConversionRatio_ = crSchedule.begin()->second;
252 }
253 std::fill(currentConversionRatio_.begin(), currentConversionRatio_.end(), initialConversionRatio_);
254 additionalResults_["historicEvents.initialConversionRatio"] = initialConversionRatio_;
255
256 // collect all relevant conversion events
257 // - cd: cr reset event
258 // - dd: dp event (with cr adjustment or pass-through)
259 // - vd: voluntary conversion (with coco possibly)
260 // - newCr: cr changed (or initially set to) specific value
261
262 struct AdjEvent {
263 ConvertibleBond2::ConversionResetData* cd = nullptr;
264 ConvertibleBond2::DividendProtectionData* dd = nullptr;
265 ConvertibleBond2::ConversionData* vd = nullptr;
266 Real newCr = Null<Real>(); // set cr to specific value on event date
267 };
268
269 std::map<Date, AdjEvent> adjEvents;
270
271 for (auto& d : registeredConversionResetData_) {
272 adjEvents[d.resetDate].cd = &d;
273 }
274
275 for (auto& d : registeredDividendProtectionData_) {
276 adjEvents[d.protectionDate].dd = &d;
277 }
278
279 for (auto& d : registeredConversionRatioData_) {
280 adjEvents[d.fromDate].newCr = d.conversionRatio;
281 }
282
283 for (auto& d : registeredConversionData_) {
284 adjEvents[d.exerciseDate].vd = &d;
285 }
286
287 // step through the events and process them (historical and future events)
288
289 Real currentCr = initialConversionRatio_;
290 Size lastDividendProtectionTimeIndex = Null<Size>();
291 bool haveStochasticCr = false;
292
293 for (auto const& event : adjEvents) {
294
295 if (event.second.cd != nullptr) {
296 const ConvertibleBond2::ConversionResetData& c = *event.second.cd;
297 if (c.resetDate <= today_) {
298
299 // historical conversion ratio reset event
300
301 Real S = equity_->fixing(equity_->fixingCalendar().adjust(c.resetDate, Preceding));
302 Real fx = 1.0;
303 if (fxConversion_ != nullptr) {
304 fx = fxConversion_->fixing(JointCalendar(equity_->fixingCalendar(), fxConversion_->fixingCalendar())
305 .adjust(c.resetDate, Preceding));
306 }
307 Real cr = c.referenceType == ConvertibleBond2::ConversionResetData::ReferenceType::CurrentCP
308 ? currentCr
309 : initialConversionRatio_;
310 Real referenceCP = N0_ / cr;
311 additionalResults_["historicEvents.crReset_" + getDateStr(event.first) + "_S"] = S;
312 additionalResults_["historicEvents.crReset_" + getDateStr(event.first) + "_threshold"] = c.threshold;
313 additionalResults_["historicEvents.crReset_" + getDateStr(event.first) + "_referenceCP"] = referenceCP;
314 additionalResults_["historicEvents.crReset_" + getDateStr(event.first) + "_gearing"] = c.gearing;
315 additionalResults_["historicEvents.crReset_" + getDateStr(event.first) + "_floor"] = c.floor;
316 additionalResults_["historicEvents.crReset_" + getDateStr(event.first) + "_globalFloor"] =
317 c.globalFloor * fx;
318 additionalResults_["historicEvents.crReset_" + getDateStr(event.first) + "_fxConversion"] = fx;
319 additionalResults_["historicEvents.crReset_" + getDateStr(event.first) + "_currentCr"] = currentCr;
320 if (!close_enough(cr, 0.0) && S < c.threshold * referenceCP) {
321 Real adjustedConversionRatio = QL_MAX_REAL;
322 if (!close_enough(c.gearing, 0.0)) {
323 adjustedConversionRatio = std::min(adjustedConversionRatio, N0_ / (c.gearing * S));
324 }
325 if (!close_enough(c.floor, 0.0)) {
326 adjustedConversionRatio = std::min(adjustedConversionRatio, N0_ / (c.floor * referenceCP));
327 }
328 if (!close_enough(c.globalFloor * fx, 0.0)) {
329 adjustedConversionRatio = std::min(adjustedConversionRatio, N0_ / (c.globalFloor * fx));
330 }
331 adjustedConversionRatio = std::max(
332 currentCr, adjustedConversionRatio != QL_MAX_REAL ? adjustedConversionRatio : -QL_MAX_REAL);
333 currentCr = std::max(currentCr, adjustedConversionRatio);
334 }
335 additionalResults_["historicEvents.crReset_" + getDateStr(event.first) + "_adjustedCr"] = currentCr;
336 } else {
337
338 // future conversion ratio reset event
339
340 Size index = grid_.index(time(event.first));
341 associatedDate_[index] = event.first;
342 hasConversionReset_[index] = true;
343 const ConvertibleBond2::ConversionResetData& c = *event.second.cd;
344 conversionResetData_[index].resetActive = true;
345 conversionResetData_[index].reference = c.referenceType;
346 conversionResetData_[index].gearing = c.gearing;
347 conversionResetData_[index].floor = c.floor;
348 conversionResetData_[index].globalFloor = c.globalFloor * currentFxConversion_[index];
349 conversionResetData_[index].threshold = c.threshold;
350 for (Size i = index + 1; i < grid_.size(); ++i) {
351 stochasticConversionRatio_[i] = true;
352 }
353 haveStochasticCr = true;
354 }
355 }
356
357 if (event.second.dd != nullptr) {
358 const ConvertibleBond2::DividendProtectionData& c = *event.second.dd;
359 if (c.protectionDate <= today_) {
360 if (c.adjustmentStyle == ConvertibleBond2::DividendProtectionData::AdjustmentStyle::CrUpOnly ||
361 c.adjustmentStyle == ConvertibleBond2::DividendProtectionData::AdjustmentStyle::CrUpDown ||
362 c.adjustmentStyle == ConvertibleBond2::DividendProtectionData::AdjustmentStyle::CrUpOnly2 ||
363 c.adjustmentStyle == ConvertibleBond2::DividendProtectionData::AdjustmentStyle::CrUpDown2) {
364
365 // historic dividend protection event with conversion ratio adjustment
366
367 Real fx = 1.0;
368 if (fxConversion_ != nullptr) {
369 fx = fxConversion_->fixing(
370 JointCalendar(equity_->fixingCalendar(), fxConversion_->fixingCalendar())
371 .adjust(c.protectionDate, Preceding));
372 }
373 Real S = equity_->fixing(equity_->fixingCalendar().adjust(c.protectionDate, Preceding));
374 Real D = equity_->dividendsBetweenDates(c.startDate, c.protectionDate);
375 Real H = c.threshold * fx;
376
377 additionalResults_["historicEvents.crReset_DP_" + getDateStr(event.first) + "_div_" +
378 getDateStr(c.startDate) + "_" + getDateStr(c.protectionDate)] = D;
379 additionalResults_["historicEvents.crReset_DP_" + getDateStr(event.first) + "_S"] = S;
380 additionalResults_["historicEvents.crReset_DP_" + getDateStr(event.first) + "_threshold"] = H;
381 additionalResults_["historicEvents.crReset_DP_" + getDateStr(event.first) + "_fxConversion"] = fx;
382 additionalResults_["historicEvents.crReset_DP_" + getDateStr(event.first) + "_currentCr"] =
383 currentCr;
384
385 if (c.adjustmentStyle == ConvertibleBond2::DividendProtectionData::AdjustmentStyle::CrUpOnly ||
386 c.adjustmentStyle == ConvertibleBond2::DividendProtectionData::AdjustmentStyle::CrUpDown) {
387 bool absolute =
388 c.dividendType == ConvertibleBond2::DividendProtectionData::DividendType::Absolute;
389 Real d = absolute ? D : D / S;
390 Real C =
391 c.adjustmentStyle == ConvertibleBond2::DividendProtectionData::AdjustmentStyle::CrUpOnly
392 ? std::max(d - H, 0.0)
393 : d - H;
394 currentCr *= absolute ? S / std::max(S - C, 1E-4) : (1.0 + C);
395 } else {
396 Real f = std::max(S - H, 0.0) / std::max(S - D, 1E-4);
397 if (c.adjustmentStyle == ConvertibleBond2::DividendProtectionData::AdjustmentStyle::CrUpOnly2)
398 f = std::max(f, 1.0);
399 currentCr *= f;
400 }
401
402 additionalResults_["historicEvents.crReset_DP_" + getDateStr(event.first) + "_adjustedCr"] =
403 currentCr;
404 }
405 } else {
406
407 if (c.adjustmentStyle == ConvertibleBond2::DividendProtectionData::AdjustmentStyle::CrUpOnly ||
408 c.adjustmentStyle == ConvertibleBond2::DividendProtectionData::AdjustmentStyle::CrUpDown ||
409 c.adjustmentStyle == ConvertibleBond2::DividendProtectionData::AdjustmentStyle::CrUpOnly2 ||
410 c.adjustmentStyle == ConvertibleBond2::DividendProtectionData::AdjustmentStyle::CrUpDown2) {
411
412 // future dividend protection event with conversion ratio adjustment
413
414 Size index = grid_.index(time(event.first));
415 associatedDate_[index] = event.first;
416 hasConversionReset_[index] = true;
417 const ConvertibleBond2::DividendProtectionData& c = *event.second.dd;
418 conversionResetData_[index].divProtActive = true;
419 conversionResetData_[index].adjustmentStyle = c.adjustmentStyle;
420 conversionResetData_[index].dividendType = c.dividendType;
421 conversionResetData_[index].divThreshold = c.threshold * currentFxConversion_[index];
422 if (lastDividendProtectionTimeIndex == Null<Size>()) {
423 conversionResetData_[index].lastDividendProtectionTimeIndex = 0;
424 conversionResetData_[index].accruedHistoricalDividends =
425 equity_->dividendsBetweenDates(c.startDate, today_);
426 additionalResults_["historicEvents.accruedDividends_" + getDateStr(c.startDate) + "_" +
427 getDateStr(today_)] = conversionResetData_[index].accruedHistoricalDividends;
428 } else {
429 conversionResetData_[index].lastDividendProtectionTimeIndex = lastDividendProtectionTimeIndex;
430 conversionResetData_[index].accruedHistoricalDividends = 0.0;
431 }
432 lastDividendProtectionTimeIndex = index;
433 for (Size i = index + 1; i < grid_.size(); ++i) {
434 stochasticConversionRatio_[i] = true;
435 }
436 haveStochasticCr = true;
437
438 } else {
439
440 // future dividend pass through event
441 Size index = grid_.index(time(c.protectionDate));
442 associatedDate_[index] = c.protectionDate;
443 hasDividendPassThrough_[index] = true;
444 dividendPassThroughData_[index].adjustmentStyle = c.adjustmentStyle;
445 dividendPassThroughData_[index].dividendType = c.dividendType;
446 dividendPassThroughData_[index].divThreshold = c.threshold;
447 if (lastDividendProtectionTimeIndex == Null<Size>()) {
448 dividendPassThroughData_[index].lastDividendProtectionTimeIndex = 0;
449 dividendPassThroughData_[index].accruedHistoricalDividends =
450 equity_->dividendsBetweenDates(c.startDate, today_);
451 additionalResults_["historicEvents.accruedDividends_" + getDateStr(c.startDate) + "_" +
452 getDateStr(today_)] =
453 dividendPassThroughData_[index].accruedHistoricalDividends;
454 } else {
455 dividendPassThroughData_[index].lastDividendProtectionTimeIndex =
456 lastDividendProtectionTimeIndex;
457 dividendPassThroughData_[index].accruedHistoricalDividends = 0.0;
458 }
459 lastDividendProtectionTimeIndex = index;
460 }
461 }
462 }
463
464 if (event.second.vd != nullptr) {
465
466 // voluntary conversion data (possibly with coco)
467
468 const ConvertibleBond2::ConversionData& vd = *event.second.vd;
469
470 Date nextConvDate = nextConversionDate(vd.exerciseDate);
471 if (nextConvDate > today_) {
472 bool conversionIsProhibited = false;
473 Size indexStart = grid_.index(time(std::max(vd.exerciseDate, today_)));
474 Size indexEnd;
475 associatedDate_[indexStart] = std::max(vd.exerciseDate, today_);
476 if (vd.exerciseType == ConvertibleBond2::ConversionData::ExerciseType::OnThisDate) {
477 indexEnd = indexStart;
478 } else if (vd.exerciseType == ConvertibleBond2::ConversionData::ExerciseType::FromThisDateOn) {
479 QL_REQUIRE(nextConvDate != Null<Date>(),
480 "FdConvertibleBondEvents::processConversionData(): internal error: did not find a next "
481 "conversion "
482 "date after "
483 << vd.exerciseDate
484 << ", the last conversione date should not have exercise type FromThisDateOn");
485 indexEnd = grid_.index(time(nextConvDate));
486 // check whether conversion is prohibited for the future due to past coco condition check
487 if (vd.cocoType == ConvertibleBond2::ConversionData::CocoType::StartOfPeriod &&
488 vd.exerciseDate <= today_) {
489 Real S = equity_->fixing(equity_->fixingCalendar().adjust(vd.exerciseDate, Preceding));
490 conversionIsProhibited = S * currentCr <= vd.cocoBarrier;
491 additionalResults_["historicEvents.coco_" + getDateStr(vd.exerciseDate) + "_S"] = S;
492 additionalResults_["historicEvents.coco_" + getDateStr(vd.exerciseDate) + "_currentCr"] =
493 currentCr;
494 additionalResults_["historicEvents.coco_" + getDateStr(vd.exerciseDate) + "_cocoBarrier"] =
495 vd.cocoBarrier;
496 additionalResults_["historicEvents.coco_" + getDateStr(vd.exerciseDate) + "_triggered"] =
497 !conversionIsProhibited;
498 }
499 } else {
500 QL_FAIL("FdConvertibleBondEvents: internal error, exercise type not recognized");
501 }
502 // update the grid info
503 for (Size i = indexStart; i <= indexEnd; ++i) {
504 // if we have set conversion information already and the current is the last conversion
505 // date, we do not modify the exisiting information, because then the current date is the
506 // end date of the last american period.
507 if (nextConvDate == Null<Date>() && hasConversionInfoSet_[i])
508 continue;
509 // otherwise mark the index as set
510 hasConversionInfoSet_[i] = true;
511 // if the conversion is not allowed, do not set the hasConversion flag
512 if (conversionIsProhibited)
513 continue;
514 // if the conversion is allowed, set the conversion flag ...
515 hasConversion_[i] = true;
516 // ... and the conversion data
517 conversionData_[i] = ConversionData{vd.cocoBarrier};
518 // a start of period coco check should only be done, if the date is > today, otherwise it's
519 // done already
520 if (vd.cocoType == ConvertibleBond2::ConversionData::CocoType::Spot) {
521 hasContingentConversion_[i] = true;
522 } else if (vd.exerciseDate > today_ &&
523 vd.cocoType == ConvertibleBond2::ConversionData::CocoType::StartOfPeriod) {
524 hasContingentConversion_[i] = true;
525 hasNoConversionPlane_[i] = i > indexStart;
526 }
527 }
528 }
529 }
530
531 if (event.second.newCr != Null<Real>()) {
532 // update current cr
533 currentCr = event.second.newCr;
534 if (event.first >= today_) {
535 associatedDate_[grid_.index(time(event.first))] = event.first;
536 if (haveStochasticCr) {
537 // we know event.first >= today if haveStochasticCr is true
538 Size index = grid_.index(time(event.first));
539 hasConversionReset_[index] = true;
540 conversionResetData_[index].resetToSpecificValue = true;
541 conversionResetData_[index].newCr = currentCr;
542 }
543 }
544 }
545
546 // update current cr on grid
547
548 for (Size i = grid_.index(time(std::max(event.first, today_))); i < grid_.size(); ++i) {
549 currentConversionRatio_[i] = currentCr;
550 }
551
552 } // for adjEvents
553
554} // processConversionData()

556void FdConvertibleBondEvents::processMandatoryConversionData() {
557 for (auto const& d : registeredMandatoryConversionData_) {
558 if (d.exerciseDate <= today_)
559 continue;
560 Size index = grid_.index(time(d.exerciseDate));
561 associatedDate_[index] = d.exerciseDate;
562 hasMandatoryConversion_[index] = true;
563 mandatoryConversionData_[index] = {d.pepsUpperBarrier * currentFxConversion_[index],
564 d.pepsLowerBarrier * currentFxConversion_[index], d.pepsUpperConversionRatio,
565 d.pepsLowerConversionRatio};
566 }

569void FdConvertibleBondEvents::finalise(const TimeGrid& grid) {
570 QL_REQUIRE(!finalised_, "FdConvertibleBondEvents: internal error, events already finalised");
571 finalised_ = true;
572 grid_ = grid;
573
574 hasBondCashflow_.resize(grid.size(), false);
575 hasCall_.resize(grid.size(), false);
576 hasPut_.resize(grid.size(), false);
577 hasConversion_.resize(grid.size(), false);
578 hasMandatoryConversion_.resize(grid.size(), false);
579 hasContingentConversion_.resize(grid.size(), false);
580 hasConversionInfoSet_.resize(grid.size(), false);
581 hasNoConversionPlane_.resize(grid.size(), false);
582 hasConversionReset_.resize(grid.size(), false);
583 hasDividendPassThrough_.resize(grid.size(), false);
584
585 bondCashflow_.resize(grid.size(), 0.0);
586 bondFinalRedemption_.resize(grid.size(), 0.0);
587 callData_.resize(grid.size());
588 putData_.resize(grid.size());
589 conversionData_.resize(grid.size());
590 conversionResetData_.resize(grid.size());
591 dividendPassThroughData_.resize(grid.size());
592 mandatoryConversionData_.resize(grid.size());
593
594 stochasticConversionRatio_.resize(grid.size(), false);
595 currentConversionRatio_.resize(grid.size(), 0.0);
596 currentFxConversion_.resize(grid.size(), 1.0);
597 associatedDate_.resize(grid.size(), Null<Date>());
598
599 initialConversionRatio_ = 0.0;
600
601 // fill fx conversion rate on grid
602
603 if (fxConversion_ != nullptr) {
604 auto source = fxConversion_->sourceCurve();
605 auto target = fxConversion_->targetCurve();
606 Real spot = fxConversion_->fixing(today_);
607 for (Size i = 0; i < grid_.size(); ++i) {
608 Real fx = spot * source->discount(grid_[i]) / target->discount(grid_[i]);
609 currentFxConversion_[i] = fx;
610 }
611 }
612
613 // process data
614
615 processBondCashflows();
616 processExerciseData(registeredCallData_, hasCall_, callData_);
617 processExerciseData(registeredPutData_, hasPut_, putData_);
618 processMakeWholeData();
619 processConversionAndDivProtData();
620 processMandatoryConversionData();
621
622 // checks
623
624 Size lastRedemptionIndex = grid_.index(time(lastRedemptionDate_));
625 for (Size k = lastRedemptionIndex + 1; k < grid_.size(); ++k) {
626 QL_REQUIRE(!hasConversion(k) && !hasMandatoryConversion(k),
627 "FdConvertibleBondEvents: conversion right after last bond redemption flow not allowed");
628 }
629}
