55 void CrossCcyFixFloatMtMResetSwapHelper::initializeDates() {
56 
57  // Swap start and end
58  Date referenceDate = evaluationDate_ = Settings::instance().evaluationDate();
59  referenceDate = paymentCalendar_.adjust(referenceDate);
60  Date start = paymentCalendar_.advance(referenceDate, settlementDays_ * Days);
61  Date end = start + tenor_;
62 
63  // Fixed schedule
64  Schedule fixedSchedule(start, end, Period(fixedFrequency_), paymentCalendar_, fixedConvention_, fixedConvention_,
65  DateGeneration::Backward, endOfMonth_);
66 
67  // Float schedule
68  Schedule floatSchedule(start, end, index_->tenor(), paymentCalendar_, paymentConvention_, paymentConvention_,
69  DateGeneration::Backward, endOfMonth_);
70 
71  Real nominal = 1.0;
72 
73  // build an FX index for forward rate projection (TODO - review settlement and calendar)
74  Natural paymentLag = 0;
75  Spread floatSpread = spread_.empty() ? 0.0 : spread_->value();
76  boost::shared_ptr<FxIndex> fxIdx;
77  if (resetsOnFloatLeg_) {
78  fxIdx = boost::make_shared<FxIndex>("dummy", settlementDays_, fixedCurrency_, index_->currency(), paymentCalendar_,
79  spotFx_, termStructureHandle_, floatDiscount_);
80  } else {
81  fxIdx = boost::make_shared<FxIndex>("dummy", settlementDays_, index_->currency(), fixedCurrency_, paymentCalendar_,
82  spotFx_, floatDiscount_, termStructureHandle_);
83  }
84 
85  swap_ = boost::make_shared<CrossCcyFixFloatMtMResetSwap>(nominal, fixedCurrency_, fixedSchedule, 0.0, fixedDayCount_, paymentConvention_,
86  paymentLag, paymentCalendar_, index_->currency(), floatSchedule, index_, floatSpread, paymentConvention_,
87  paymentLag, paymentCalendar_, fxIdx, resetsOnFloatLeg_);
88 
89  // Attach engine
90  boost::shared_ptr<PricingEngine> engine = boost::make_shared<CrossCcySwapEngine>(
91  fixedCurrency_, termStructureHandle_, index_->currency(), floatDiscount_, spotFx_);
92  swap_->setPricingEngine(engine);
93 
94  earliestDate_ = swap_->startDate();
95  latestDate_ = swap_->maturityDate();
96 
97  /* May need to adjust latestDate_ if you are projecting libor based
98  on tenor length rather than from accrual date to accrual date. */
99 #ifdef QL_USE_INDEXED_COUPON
100  Size numCashflows = swap_->leg(1).size();
101  Date endDate = latestDate_;
102  if (numCashflows > 0) {
103  for (Size i = numCashflows - 1; i >= 0; i--) {
104  boost::shared_ptr<FloatingRateCoupon> lastFloating =
105  boost::dynamic_pointer_cast<FloatingRateCoupon>(swap_->leg(1)[i]);
106  if (!lastFloating)
107  continue;
108  else {
109  Date fixingValueDate = domesticCcyIndex_->valueDate(lastFloating->fixingDate());
110  endDate = domesticCcyIndex_->maturityDate(fixingValueDate);
111  Date endValueDate = domesticCcyIndex_->maturityDate(fixingValueDate);
112  latestDate_ = std::max(latestDate_, endValueDate);
113  break;
114  }
115  }
116  }
117 #endif
118 }

125 void CrossCcyFixFloatMtMResetSwapHelper::update() {
126  // Maybe FX spot quote or spread quote changed
127  if (!close(spotFx_->value(), swap_->nominal()) ||
128  (!spread_.empty() && !close(spread_->value(), swap_->floatSpread()))) {
129  initializeDates();
130  }
131 
132  // Maybe evaluation date changed. RelativeDateRateHelper will take care of this
133  // Note: if initializeDates() was called in above if statement, it will not be called
134  // again in RelativeDateRateHelper::update() because evaluationDate_ is set in
135  // initializeDates(). So, redundant instrument builds are avoided.
136  RelativeDateRateHelper::update();
137 }
