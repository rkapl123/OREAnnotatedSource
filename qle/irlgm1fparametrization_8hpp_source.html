112 template <class TS>
113 inline Real Lgm1fParametrization<TS>::zetan(const Size n, const Time t,
114  const boost::shared_ptr<Integrator>& integrator) {
115  auto z = zetan_cached_.find(std::make_pair(n, t));
116  if (z == zetan_cached_.end()) {
117  std::vector<Real> times;
118  for (Size i = 0; i < numberOfParameters(); ++i)
119  times.insert(times.end(), parameterTimes(i).begin(), parameterTimes(i).end());
120  PiecewiseIntegral pwint(integrator, times, true);
121  Real v = pwint([this, n](Real s) { return std::pow(this->alpha(s), 2) * std::pow(this->H(s), n); }, 0.0, t);
122  zetan_cached_[std::make_pair(n, t)] = v;
123  return v;
124  } else {
125  return z->second;
126  }
127 }
