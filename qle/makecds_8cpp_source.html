62 MakeCreditDefaultSwap::operator boost::shared_ptr<CreditDefaultSwap>() const {
63 
64  Date tradeDate = Settings::instance().evaluationDate();
65  Date upfrontDate = WeekendsOnly().advance(tradeDate, cashSettlementDays_, Days);
66 
67  Date protectionStart;
68  if (rule_ == DateGeneration::CDS2015 || rule_ == DateGeneration::CDS) {
69  protectionStart = tradeDate;
70  } else {
71  protectionStart = tradeDate + 1;
72  }
73 
74  Date end;
75  if (tenor_) {
76  if (rule_ == DateGeneration::CDS2015 || rule_ == DateGeneration::CDS || rule_ == DateGeneration::OldCDS) {
77  end = cdsMaturity(tradeDate, *tenor_, rule_);
78  } else {
79  end = tradeDate + *tenor_;
80  }
81  } else {
82  end = *termDate_;
83  }
84 
85  Schedule schedule(protectionStart, end, couponTenor_, WeekendsOnly(), Following, Unadjusted, rule_, false);
86 
87  boost::shared_ptr<CreditDefaultSwap> cds = boost::make_shared<CreditDefaultSwap>(
88  side_, nominal_, upfrontRate_, couponRate_, schedule, Following, dayCounter_, true,
89  CreditDefaultSwap::atDefault, protectionStart, upfrontDate, boost::shared_ptr<Claim>(),
90  lastPeriodDayCounter_, tradeDate, cashSettlementDays_);
91 
92  cds->setPricingEngine(engine_);
93  return cds;
94 }
