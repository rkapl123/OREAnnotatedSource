15set<Date> pricingDates(const Date& s, const Date& e, const Calendar& pricingCalendar,
16 bool excludeStart, bool includeEnd, bool useBusinessDays) {
17
18 // If start date is after end date, return no dates.
19 if (s > e)
20 return set<Date>();
21
22 Date start = s;
23 Date end = e;
24
25 // Cover the possible exclusion of the start date
26 if ((useBusinessDays && pricingCalendar.isBusinessDay(start)) && excludeStart) {
27 start = pricingCalendar.advance(start, 1, Days);
28 }
29
30 if ((!useBusinessDays && pricingCalendar.isHoliday(start)) && excludeStart) {
31 while (pricingCalendar.isHoliday(start) && start <= end)
32 start++;
33 }
34
35 // Cover the possible exclusion of the end date
36 if ((useBusinessDays && pricingCalendar.isBusinessDay(end)) && !includeEnd) {
37 end = pricingCalendar.advance(end, -1, Days);
38 }
39
40 if ((!useBusinessDays && pricingCalendar.isHoliday(end)) && !includeEnd) {
41 while (pricingCalendar.isHoliday(end) && start <= end)
42 end--;
43 }
44
45 // Create the set of dates, which may be empty.
46 set<Date> res;
47 for (; start <= end; start++) {
48 if (isPricingDate(start, pricingCalendar, useBusinessDays))
49 res.insert(start);
50 }
51
52 return res;
53}
54
55bool isPricingDate(const Date& d, const Calendar& pricingCalendar, bool useBusinessDays) {
56 return ((useBusinessDays && pricingCalendar.isBusinessDay(d)) ||
57 (!useBusinessDays && pricingCalendar.isHoliday(d)));
58}
