39void AnalyticJyCpiCapFloorEngine::calculate() const {
40
41 // If the pay date has occurred, nothing to do.
42 SimpleCashFlow cf(0.0, arguments_.payDate);
43 if (cf.hasOccurred()) {
44 results_.value = 0.0;
45 return;
46 }
47
48 // Discount factor to pay date will be needed below.
49 Size irIdx = model_->ccyIndex(model_->infjy(index_)->currency());
50 DiscountFactor df = model_->irlgm1f(irIdx)->termStructure()->discount(arguments_.payDate);
51
52 QL_DEPRECATED_DISABLE_WARNING
53 bool interpolate = arguments_.observationInterpolation == CPI::Linear ||
54 (arguments_.observationInterpolation == CPI::AsIndex && arguments_.index->interpolated());
55 QL_DEPRECATED_ENABLE_WARNING
56 // Get the time to expiry. This determines if we use the JY model or look for an inflation index fixing.
57 auto zts = model_->infjy(index_)->realRate()->termStructure();
58 Real t = inflationYearFraction(arguments_.index->frequency(), interpolate,
59 zts->dayCounter(), zts->baseDate(), arguments_.fixDate);
60
61 // If time to expiry is non-positive, we return the discounted value of the settled amount.
62 // CPICapFloor should really have its own day counter for going from strike rate to k. We use t here.
63 Real k = pow(1.0 + arguments_.strike, t);
64 if (t <= 0.0) {
65 auto cpiAtExpiry = arguments_.index->fixing(arguments_.fixDate);
66 if (arguments_.type == Option::Call) {
67 results_.value = max(cpiAtExpiry / arguments_.baseCPI - k, 0.0);
68 } else {
69 results_.value = max(k - cpiAtExpiry / arguments_.baseCPI, 0.0);
70 }
71 results_.value *= arguments_.nominal * df;
72 return;
73 }
74
75 // Section 13, Book. ZCII Cap. Note that there is a difference between the base CPI value associated with the
76 // inflation term structures and used as a starting point in the CAM evolution vs. the contractual base CPI in
77 // the CPICapFloor instrument. The former is curveBaseCpi below and the latter is arguments_.baseCPI.
78 const CrossAssetModel* x = model_.get();
79
80 // Calculate the variance, \Sigma^2_I in the book.
81 Real H_n_t = Hz(irIdx).eval(x, t);
82 Real H_r_t = Hy(index_).eval(x, t);
83 Real v = integral(x, P(LC(H_n_t, -1.0, Hz(irIdx)), LC(H_n_t, -1.0, Hz(irIdx)), az(irIdx), az(irIdx)), 0.0, t);
84 v += integral(x, P(LC(H_r_t, -1.0, Hy(index_)), LC(H_r_t, -1.0, Hy(index_)), ay(index_), ay(index_)), 0.0, t);
85 v += integral(x, P(sy(index_), sy(index_)), 0.0, t);
86 v -= 2 * integral(x, P(rzy(irIdx, index_), LC(H_r_t, -1.0, Hy(index_)), LC(H_n_t, -1.0, Hz(irIdx)),
87 ay(index_), az(irIdx)), 0.0, t);
88 v += 2 * integral(x, P(rzy(irIdx, index_, 1), LC(H_n_t, -1.0, Hz(irIdx)), az(irIdx), sy(index_)), 0.0, t);
89 v -= 2 * integral(x, P(ryy(index_, index_, 0, 1), LC(H_r_t, -1.0, Hy(index_)), ay(index_), sy(index_)), 0.0, t);
90
91 // Calculate the forward CPI, F_I(0,T) in the book.
92 Real fwd = arguments_.index->fixing(arguments_.fixDate);
93
94 // Get adjusted nominal and strike, \tilde{N} and \tilde{K} from the book.
95 Real adjNominal = arguments_.nominal / arguments_.baseCPI;
96 Real adjStrike = k * arguments_.baseCPI;
97
98 results_.value = adjNominal * blackFormula(arguments_.type, adjStrike, fwd, sqrt(v), df);
99}
