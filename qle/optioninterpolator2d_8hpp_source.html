166 template <class IS, class IE>
167 void OptionInterpolator2d<IS, IE>::initialise(const std::vector<QuantLib::Date>& dates,
168  const std::vector<QuantLib::Real>& strikes,
169  const std::vector<QuantLib::Real>& values) {
170  using QuantLib::Date;
171  using QuantLib::Interpolation;
172  using QuantLib::Real;
173  using QuantLib::Size;
174  using QuantLib::Time;
175  using std::pair;
176  using std::set;
177  using std::vector;
178  QL_REQUIRE((strikes.size() == dates.size()) && (dates.size() == values.size()),
179  "dates, strikes and values vectors not of equal size.");
180 
181  // first get uniques dates and sort
182  set<Date> datesSet(dates.begin(), dates.end());
183  expiries_ = vector<Date>(datesSet.begin(), datesSet.end());
184  vector<bool> dateDone(expiries_.size(), false);
185  times_ = vector<Time>(expiries_.size());
186  interpolations_ = vector<Interpolation>(expiries_.size());
187  values_ = vector<vector<Real> >(expiries_.size());
188  strikes_ = vector<vector<Real> >(expiries_.size());
189 
190  // Populate expiry info. (Except interpolation obj members)
191  for (Size i = 0; i < dates.size(); i++) {
192  vector<Date>::iterator found = find(expiries_.begin(), expiries_.end(), dates[i]);
193  QL_REQUIRE(found != expiries_.end(), "Date should already be loaded" << dates[i]);
194  Size ii;
195  ii = distance(expiries_.begin(), found); // index of expiry
196 
197  if (!dateDone[ii]) {
198  // add expiry data if not found
199  QL_REQUIRE(dates[i] >= referenceDate_,
200  "Expiry date:" << dates[i] << " before asof date: " << referenceDate_);
201  times_[ii] = dayCounter_.yearFraction(referenceDate_, dates[i]);
202  strikes_[ii].push_back(strikes[i]);
203  values_[ii].push_back(values[i]);
204  dateDone[ii] = true;
205  } else {
206  // expiry found => add if strike not found
207  Real tmpStrike = strikes[i];
208  vector<Real>::iterator fnd =
209  find_if(strikes_[ii].begin(), strikes_[ii].end(), CloseEnoughComparator(tmpStrike));
210  if (fnd == strikes_[ii].end()) {
211  // add strike/var pairs if strike not found for this expiry
212  strikes_[ii].push_back(strikes[i]);
213  values_[ii].push_back(values[i]);
214  }
215  }
216  }
217 
218  // check on strikes and values size
219  for (Size i = 0; i < expiries_.size(); i++)
220  QL_REQUIRE(strikes_[i].size() == values_[i].size(),
221  "different number of variances and strikes for date: " << expiries_[i]);
222 
223  // set expiries' interpolations.
224  for (Size i = 0; i < expiries_.size(); i++) {
225  // sort strikes within this expiry
226  vector<pair<Real, Real> > tmpPairs(strikes_[i].size());
227  vector<Real> sortedStrikes;
228  vector<Real> sortedVals;
229  for (Size j = 0; j < strikes_[i].size(); j++) {
230  tmpPairs[j] = pair<Real, Real>(strikes_[i][j], values_[i][j]);
231  }
232  sort(tmpPairs.begin(), tmpPairs.end()); // sorts according to first. (strikes)
233  for (vector<pair<Real, Real> >::iterator it = tmpPairs.begin(); it != tmpPairs.end(); it++) {
234  sortedStrikes.push_back(it->first);
235  sortedVals.push_back(it->second);
236  }
237  strikes_[i] = sortedStrikes;
238  values_[i] = sortedVals;
239 
240  // set interpolation
241  if (strikes_[i].size() == 1) {
242  interpolations_[i] = Constant().interpolate(values_[i].front());
243  } else {
244  interpolations_[i] =
245  interpolatorStrike_.interpolate(strikes_[i].begin(), strikes_[i].end(), values_[i].begin());
246  }
247  }
248  initialised_ = true;
249 }

265 template <class IS, class IE>
266 QuantLib::Real OptionInterpolator2d<IS, IE>::getValueForStrike(QuantLib::Real strike,
267  const std::vector<QuantLib::Real>& strks,
268  const std::vector<QuantLib::Real>& vars,
269  const QuantLib::Interpolation& intrp) const {
270 
271  QuantLib::Real retVar;
272  if (strike > strks.back() && upperStrikeConstExtrap_) {
273  retVar = vars.back(); // force flat extrapolate far strike if requested
274  } else if (strike < strks.front() && lowerStrikeConstExtrap_) {
275  retVar = vars.front(); // force flat extrapolate near strike if requested
276  } else {
277  retVar = intrp(strike); // interpolate between strikes or extrap with interpolator default
278  }
279  return retVar;
280 }

282 template <class IS, class IE>
283 QuantLib::Real OptionInterpolator2d<IS, IE>::getValue(QuantLib::Time t, QuantLib::Real strike) const {
284  using QuantLib::Interpolation;
285  using QuantLib::Real;
286  using QuantLib::Size;
287  using QuantLib::Time;
288  using std::vector;
289  QL_REQUIRE(initialised_, "No data provided to OptionInterpolator2d");
290  QL_REQUIRE(t >= 0, "Variance requested for date before reference date: " << referenceDate_);
291  Real varReturn;
292  if (t == 0.0) {
293  // requested at reference date
294  varReturn = values_[0][0];
295  } else {
296  // ind1 and ind2 two expiries on either side of requested time.
297  Size ind1, ind2;
298  if (t <= times_.front()) {
299  // near end of expiries, use first 2 strikes
300  ind1 = 0;
301  ind2 = 1;
302  } else if (t > times_.back()) {
303  // far end of expiries, use last 2
304  ind1 = times_.size() - 2;
305  ind2 = times_.size() - 1;
306  } else {
307  // requested between existing expiries (interpolate between expiries)
308  ind2 = distance(times_.begin(), lower_bound(times_.begin(), times_.end(), t));
309  ind1 = (ind2 != 0) ? ind2 - 1 : 0;
310  }
311  // interpolate between expiries
312  vector<Real> tmpVars(2);
313  vector<Time> xAxis;
314  xAxis.push_back(times_[ind1]);
315  xAxis.push_back(times_[ind2]);
316 
317  tmpVars[0] = getValueForStrike(strike, strikes_[ind1], values_[ind1], interpolations_[ind1]);
318  tmpVars[1] = getValueForStrike(strike, strikes_[ind2], values_[ind2], interpolations_[ind2]);
319  Interpolation interp = interpolatorExpiry_.interpolate(xAxis.begin(), xAxis.end(), tmpVars.begin());
320  // linear extrapolation of expiries in case t > time_.back() above.
321  interp.enableExtrapolation(true);
322  varReturn = interp(t);
323  }
324  return varReturn;
325 }

327 template <class IS, class IE>
328 QuantLib::Real OptionInterpolator2d<IS, IE>::getValue(QuantLib::Date d, QuantLib::Real strike) const {
329  QL_REQUIRE(initialised_, "No data provided to OptionInterpolator2d");
330  QL_REQUIRE(d >= referenceDate_, "Variance requested for date before reference date: " << referenceDate_);
331  QuantLib::Real valueReturn;
332 
333  std::vector<QuantLib::Date>::const_iterator it = find(expiries_.begin(), expiries_.end(), d);
334  // if the date matches one of the expiries get the value on that day, otherwise use time interpolation
335  if (it != expiries_.end()) {
336  QuantLib::Size dis = distance(expiries_.begin(), it);
337  valueReturn = getValueForStrike(strike, strikes_[dis], values_[dis], interpolations_[dis]);
338  } else {
339  QuantLib::Time t = dayCounter_.yearFraction(referenceDate_, d);
340  valueReturn = getValue(t, strike);
341  }
342  return valueReturn;
343 }
