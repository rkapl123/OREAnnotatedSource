151 inline Real LinearGaussMarkovModel::numeraire(const Time t, const Real x,
152  const Handle<YieldTermStructure> discountCurve) const {
153  QL_REQUIRE(t >= 0.0, "t (" << t << ") >= 0 required in LGM::numeraire");
154  Real Ht = parametrization_->H(t);
155  return std::exp(Ht * x + 0.5 * Ht * Ht * parametrization_->zeta(t)) /
156  (discountCurve.empty() ? parametrization_->termStructure()->discount(t) : discountCurve->discount(t));
157 }

159 inline Real LinearGaussMarkovModel::discountBond(const Time t, const Time T, const Real x,
160  const Handle<YieldTermStructure> discountCurve) const {
161  if (QuantLib::close_enough(t, T))
162  return 1.0;
163  QL_REQUIRE(T >= t && t >= 0.0, "T(" << T << ") >= t(" << t << ") >= 0 required in LGM::discountBond");
164  Real Ht = parametrization_->H(t);
165  Real HT = parametrization_->H(T);
166  return (discountCurve.empty()
167  ? parametrization_->termStructure()->discount(T) / parametrization_->termStructure()->discount(t)
168  : discountCurve->discount(T) / discountCurve->discount(t)) *
169  std::exp(-(HT - Ht) * x - 0.5 * (HT * HT - Ht * Ht) * parametrization_->zeta(t));
170 }

172 inline Real LinearGaussMarkovModel::reducedDiscountBond(const Time t, const Time T, const Real x,
173  const Handle<YieldTermStructure> discountCurve) const {
174  if (QuantLib::close_enough(t, T))
175  return 1.0 / numeraire(t, x, discountCurve);
176  QL_REQUIRE(T >= t && t >= 0.0, "T(" << T << ") >= t(" << t << ") >= 0 required in LGM::reducedDiscountBond");
177  Real HT = parametrization_->H(T);
178  return (discountCurve.empty() ? parametrization_->termStructure()->discount(T) : discountCurve->discount(T)) *
179  std::exp(-HT * x - 0.5 * HT * HT * parametrization_->zeta(t));
180 }

182 inline Real LinearGaussMarkovModel::discountBondOption(Option::Type type, const Real K, const Time t, const Time S,
183  const Time T,
184  const Handle<YieldTermStructure> discountCurve) const {
185  QL_REQUIRE(T > S && S >= t && t >= 0.0,
186  "T(" << T << ") > S(" << S << ") >= t(" << t << ") >= 0 required in LGM::discountBondOption");
187  Real w = (type == Option::Call ? 1.0 : -1.0);
188  Real pS = discountCurve.empty() ? parametrization_->termStructure()->discount(S) : discountCurve->discount(S);
189  Real pT = discountCurve.empty() ? parametrization_->termStructure()->discount(T) : discountCurve->discount(T);
190  // slight generalization of Lichters, Stamm, Gallagher 11.2.1
191  // with t < S, SSRN: https://ssrn.com/abstract=2246054
192  Real sigma = std::sqrt(parametrization_->zeta(t)) * (parametrization_->H(T) - parametrization_->H(S));
193  Real dp = (std::log(pT / (K * pS)) / sigma + 0.5 * sigma);
194  Real dm = dp - sigma;
195  CumulativeNormalDistribution N;
196  return w * (pT * N(w * dp) - pS * K * N(w * dm));
197 }
