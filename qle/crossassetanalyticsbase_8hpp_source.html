48/*! product expression, 2 factors */
49template <typename E1, typename E2> struct P2_ {
50 P2_(const E1& e1, const E2& e2) : e1_(e1), e2_(e2) {}
51 Real eval(const CrossAssetModel* x, const Real t) const { return e1_.eval(x, t) * e2_.eval(x, t); }
52 const E1& e1_;
53 const E2& e2_;
54};

56/*! product expression, 3 factors */
57template <typename E1, typename E2, typename E3> struct P3_ {
58 P3_(const E1& e1, const E2& e2, const E3& e3) : e1_(e1), e2_(e2), e3_(e3) {}
59 Real eval(const CrossAssetModel* x, const Real t) const { return e1_.eval(x, t) * e2_.eval(x, t) * e3_.eval(x, t); }
60 const E1& e1_;
61 const E2& e2_;
62 const E3& e3_;
63};

111/*! linear combination, 1 factor i.e. c + c1 * factor1. */
112template <typename E1> struct LC1_ {
113
114 LC1_(QuantLib::Real c, QuantLib::Real c1, const E1& e1) : c_(c), c1_(c1), e1_(e1) {}
