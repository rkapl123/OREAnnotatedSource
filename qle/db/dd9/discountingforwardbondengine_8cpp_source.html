58 void DiscountingForwardBondEngine::calculate() const {
59  // Do some checks on data
60  QL_REQUIRE(!discountCurve_.empty(), "discounting term structure handle is empty");
61  QL_REQUIRE(!incomeCurve_.empty(), "income term structure handle is empty");
62  QL_REQUIRE(!bondReferenceYieldCurve_.empty(), "bond rerference term structure handle is empty");
63 
64  Date npvDate = npvDate_; // this is today when the valuation occurs
65  if (npvDate == Null<Date>()) {
66  npvDate = (*discountCurve_)->referenceDate();
67  }
68  Date settlementDate = settlementDate_; //
69  if (settlementDate == Null<Date>()) {
70  settlementDate = (*discountCurve_)->referenceDate();
71  }
72 
73  Date maturityDate =
74  arguments_.fwdMaturityDate; // this is the date when the forward is executed, i.e. cash and bond change hands
75 
76  Real cmpPayment = arguments_.compensationPayment;
77  if (cmpPayment == Null<Real>()) {
78  cmpPayment = 0.0;
79  }
80  Date cmpPaymentDate = arguments_.compensationPaymentDate;
81  if (cmpPaymentDate == Null<Date>()) {
82  cmpPaymentDate = npvDate;
83  }
84 
85  // in case that the premium payment has occured in the past, we set the amount to 0. the date itself is set to the
86  // npvDate to have a valid date for "discounting"
87  Date cmpPaymentDate_use = cmpPaymentDate >= npvDate ? cmpPaymentDate : maturityDate;
88  cmpPayment = cmpPaymentDate >= npvDate ? cmpPayment : 0.0; // premium cashflow is not relevant for npv if in the
89  // past
90 
91  // initialize
92  results_.value = 0.0; // this is today's npv of the forward contract
93  results_.underlyingSpotValue = 0.0; // this is today's value of the "restricted bond". Restricted means that only
94  // cashflows after maturity are taken into account.
95  results_.forwardValue = 0.0; // this the value of the forward contract just before maturity
96 
97  bool dirty = arguments_.settlementDirty;
98 
99  results_.underlyingSpotValue = calculateBondNpv(npvDate, maturityDate); // cashflows before maturity will be ignored
100 
101  boost::tie(results_.forwardValue, results_.value) = calculateForwardContractPresentValue(
102  results_.underlyingSpotValue, cmpPayment, npvDate, maturityDate, arguments_.fwdSettlementDate,
103  !arguments_.isPhysicallySettled, cmpPaymentDate_use, dirty);
104 }

106 Real DiscountingForwardBondEngine::calculateBondNpv(Date npvDate, Date computeDate) const {
107  Real npvValue = 0.0;
108  Size numCoupons = 0;
109  bool hasLiveCashFlow = false;
110 
111  // handle case where we wish to price simply with benchmark curve and scalar security spread
112  // i.e. credit curve term structure (and recovery) have not been specified
113  // we set the default probability and recovery rate to zero in this instance (issuer credit worthiness already
114  // captured within security spread)
115  boost::shared_ptr<DefaultProbabilityTermStructure> creditCurvePtr =
116  bondDefaultCurve_.empty()
117  ? boost::make_shared<QuantLib::FlatHazardRate>(npvDate, 0.0, bondReferenceYieldCurve_->dayCounter())
118  : bondDefaultCurve_.currentLink();
119  Rate recoveryVal = bondRecoveryRate_.empty() ? 0.0 : bondRecoveryRate_->value(); // setup default bond recovery rate
120 
121  std::vector<Date> bondCashflowPayDates;
122  std::vector<Real> bondCashflows, bondCashflowSurvivalProbabilities, bondCashflowDiscountFactors;
123 
124  // load the shared pointer into bd
125  boost::shared_ptr<Bond> bd = arguments_.underlying;
126  for (Size i = 0; i < bd->cashflows().size(); i++) {
127  // Recovery amount is computed over the whole time interval (npvDate,maturityOfBond)
128 
129  if (bd->cashflows()[i]->hasOccurred(
130  computeDate, includeSettlementDateFlows_)) // Cashflows before computeDate not relevant for npv
131  continue;
132 
133  /* The amount recovered in the case of default is the recoveryrate*Notional*Probability of
134  Default; this is added to the NPV value. For coupon bonds the coupon periods are taken
135  as the timesteps for integrating over the probability of default.
136  */
137  boost::shared_ptr<Coupon> coupon = boost::dynamic_pointer_cast<Coupon>(bd->cashflows()[i]);
138 
139  if (coupon) {
140  numCoupons++;
141  Date startDate = coupon->accrualStartDate();
142  Date endDate = coupon->accrualEndDate();
143  Date effectiveStartDate = (startDate <= computeDate && computeDate <= endDate) ? computeDate : startDate;
144  Date defaultDate = effectiveStartDate + (endDate - effectiveStartDate) / 2;
145  Probability P = creditCurvePtr->defaultProbability(effectiveStartDate, endDate);
146 
147  npvValue += coupon->nominal() * recoveryVal * P * bondReferenceYieldCurve_->discount(defaultDate);
148  }
149 
150  hasLiveCashFlow = true; // check if a cashflow is available after the date of valuation.
151 
152  // Coupon value is discounted future payment times the survival probability
153  Probability S = creditCurvePtr->survivalProbability(bd->cashflows()[i]->date()) /
154  creditCurvePtr->survivalProbability(computeDate);
155  npvValue += bd->cashflows()[i]->amount() * S * bondReferenceYieldCurve_->discount(bd->cashflows()[i]->date());
156 
157  bondCashflows.push_back(bd->cashflows()[i]->amount());
158  bondCashflowPayDates.push_back(bd->cashflows()[i]->date());
159  bondCashflowSurvivalProbabilities.push_back(S);
160  bondCashflowDiscountFactors.push_back(bondReferenceYieldCurve_->discount(bd->cashflows()[i]->date()));
161  }
162 
163  // the ql instrument might not yet be expired and still have not anything to value if
164  // the computeDate > evaluation date
165  if (!hasLiveCashFlow)
166  return 0.0;
167 
168  if (bd->cashflows().size() > 1 && numCoupons == 0) {
169  QL_FAIL("DiscountingForwardBondEngine does not support bonds with multiple cashflows but no coupons");
170  }
171 
172  Real bondRecovery = 0;
173  boost::shared_ptr<Coupon> firstCoupon = boost::dynamic_pointer_cast<Coupon>(bd->cashflows()[0]);
174  if (firstCoupon) {
175  Date startDate = computeDate; // face value recovery starting with computeDate
176  while (startDate < bd->cashflows()[0]->date()) {
177  Date stepDate = startDate + timestepPeriod_;
178  Date endDate = (stepDate > bd->cashflows()[0]->date()) ? bd->cashflows()[0]->date() : stepDate;
179  Date defaultDate = startDate + (endDate - startDate) / 2;
180  Probability P = creditCurvePtr->defaultProbability(startDate, endDate);
181 
182  bondRecovery += firstCoupon->nominal() * recoveryVal * P * bondReferenceYieldCurve_->discount(defaultDate);
183  startDate = stepDate;
184  }
185  }
186  /* If there are no coupon, as in a Zero Bond, we must integrate over the entire period from npv date to
187  maturity. The timestepPeriod specified is used as provide the steps for the integration. This only applies
188  to bonds with 1 cashflow, identified as a final redemption payment.
189  */
190  if (bd->cashflows().size() == 1) {
191  boost::shared_ptr<Redemption> redemption = boost::dynamic_pointer_cast<Redemption>(bd->cashflows()[0]);
192  if (redemption) {
193  Date startDate = computeDate;
194  while (startDate < redemption->date()) {
195  Date stepDate = startDate + timestepPeriod_;
196  Date endDate = (stepDate > redemption->date()) ? redemption->date() : stepDate;
197  Date defaultDate = startDate + (endDate - startDate) / 2;
198  Probability P = creditCurvePtr->defaultProbability(startDate, endDate);
199 
200  bondRecovery +=
201  redemption->amount() * recoveryVal * P * bondReferenceYieldCurve_->discount(defaultDate);
202  startDate = stepDate;
203  }
204  }
205  }
206 
207  npvValue += bondRecovery;
208 
209  results_.additionalResults["bondCashflow"] = bondCashflows;
210  results_.additionalResults["bondCashflowPayDates"] = bondCashflowPayDates;
211  results_.additionalResults["bondCashflowSurvivalProbabilities"] = bondCashflowSurvivalProbabilities;
212  results_.additionalResults["bondCashflowDiscountFactors"] = bondCashflowDiscountFactors;
213  results_.additionalResults["bondRecovery"] = bondRecovery;
214 
215  return npvValue * arguments_.bondNotional;
216 }

218 boost::tuple<Real, Real> DiscountingForwardBondEngine::calculateForwardContractPresentValue(
219  Real spotValue, Real cmpPayment, Date npvDate, Date computeDate, Date settlementDate, bool cashSettlement,
220  Date cmpPaymentDate, bool dirty) const {
221 
222  // here we go with the true forward computation
223  Real forwardBondValue = 0.0;
224  Real forwardContractPresentValue = 0.0;
225  Real forwardContractForwardValue = 0.0;
226 
227  std::vector<Date> fwdBondCashflowPayDates;
228  std::vector<Real> fwdBondCashflows, fwdBondCashflowSurvivalProbabilities, fwdBondCashflowDiscountFactors;
229 
230  // handle case where we wish to price simply with benchmark curve and scalar security spread
231  // i.e. credit curve term structure (and recovery) have not been specified
232  // we set the default probability and recovery rate to zero in this instance (issuer credit worthiness already
233  // captured within security spread)
234  boost::shared_ptr<DefaultProbabilityTermStructure> creditCurvePtr =
235  bondDefaultCurve_.empty()
236  ? boost::make_shared<QuantLib::FlatHazardRate>(npvDate, 0.0, bondReferenceYieldCurve_->dayCounter())
237  : bondDefaultCurve_.currentLink();
238  Rate recoveryVal = bondRecoveryRate_.empty() ? 0.0 : bondRecoveryRate_->value(); // setup default bond recovery rate
239  // load the shared pointer into bd
240  boost::shared_ptr<Bond> bd = arguments_.underlying;
241 
242  // the case of dirty strike corresponds here to an accrual of 0.0. This will be convenient in the code.
243  Date bondSettlementDate = bd->settlementDate(computeDate);
244  Real accruedAmount = dirty ? 0.0
245  : bd->accruedAmount(bondSettlementDate) * bd->notional(bondSettlementDate) / 100.0 *
246  arguments_.bondNotional;
247 
248  /* Discounting and compounding, taking account of possible bond default before delivery*/
249 
250  if (cashSettlement) {
251  forwardBondValue = spotValue / (incomeCurve_->discount(bondSettlementDate));
252  } else {
253  forwardBondValue = spotValue / (incomeCurve_->discount(settlementDate));
254  }
255 
256  // Subtract strike at maturity. Regarding accrual (i.e. strike is given clean vs dirty) there are two
257  // cases: long or short.
258 
259  // Long: forwardBondValue - strike_dirt = (forwardBondValue - accrual) - strike_clean
260  // Short: strike_dirt - forwardBondValue = strike_clean - (forwardBondValue - accrual)
261  // In total:
262  boost::shared_ptr<Payoff> effectivePayoff;
263  if (arguments_.payoff) {
264  // vanilla forward bond calculation
265  forwardContractForwardValue = (*arguments_.payoff)(forwardBondValue - accruedAmount);
266  effectivePayoff = arguments_.payoff;
267  } else if (arguments_.lockRate != Null<Real>()) {
268  // lock rate specified forward bond calculation, use hardcoded conventions (compounded / semi annual) here, from
269  // treasury bonds
270  Real price = forwardBondValue / arguments_.bondNotional / bd->notional(bondSettlementDate) * 100.0;
271  Real yield = BondFunctions::yield(*bd, price, arguments_.lockRateDayCounter, Compounded, Semiannual,
272  bondSettlementDate, 1E-10, 100, 0.05, Bond::Price::Dirty);
273  Real dv01 = price / 100.0 *
274  BondFunctions::duration(*bd, yield, arguments_.lockRateDayCounter, Compounded, Semiannual,
275  Duration::Modified, bondSettlementDate);
276  QL_REQUIRE(arguments_.longInForward, "DiscountingForwardBondEngine: internal error, longInForward must be "
277  "populated if payoff is specified via lock-rate");
278  Real multiplier = (*arguments_.longInForward) ? 1.0 : -1.0;
279  forwardContractForwardValue = multiplier * (yield - arguments_.lockRate) * dv01 * arguments_.bondNotional *
280  bd->notional(bondSettlementDate);
281  effectivePayoff = boost::make_shared<ForwardBondTypePayoff>(
282  (*arguments_.longInForward) ? Position::Long : Position::Short,
283  arguments_.lockRate * dv01 * arguments_.bondNotional * bd->notional(bondSettlementDate));
284  } else {
285  QL_FAIL("DiscountingForwardBondEngine: internal error, no payoff and no lock rate given, expected exactly one "
286  "of them to be populated.");
287  }
288 
289  // forwardContractPresentValue adjusted for potential default before computeDate:
290  forwardContractPresentValue =
291  forwardContractForwardValue * (discountCurve_->discount(settlementDate)) *
292  creditCurvePtr->survivalProbability(computeDate) -
293  cmpPayment *
294  (discountCurve_->discount(cmpPaymentDate)); // The forward is a derivative. We use "OIS curve" to discount.
295  // We subtract the potential payment due to Premium.
296 
297  fwdBondCashflows.push_back(forwardContractForwardValue);
298  fwdBondCashflowPayDates.push_back(computeDate);
299  fwdBondCashflowSurvivalProbabilities.push_back(creditCurvePtr->survivalProbability(computeDate));
300  fwdBondCashflowDiscountFactors.push_back(discountCurve_->discount(computeDate));
301 
302  fwdBondCashflows.push_back(-1 * cmpPayment);
303  fwdBondCashflowPayDates.push_back(cmpPaymentDate);
304  fwdBondCashflowSurvivalProbabilities.push_back(1);
305  fwdBondCashflowDiscountFactors.push_back(discountCurve_->discount(cmpPaymentDate));
306 
307  Real fwdBondRecovery = 0;
308  // Take account of face value recovery:
309  // A) Recovery for time period when coupons are present
310  for (Size i = 0; i < bd->cashflows().size(); i++) {
311  if (bd->cashflows()[i]->hasOccurred(
312  npvDate, includeSettlementDateFlows_)) // Cashflows before npvDate not relevant for npv
313  continue;
314  if (bd->cashflows()[i]->date() >=
315  computeDate) // Cashflows after computeDate do not fall into the forward period
316  continue;
317  boost::shared_ptr<Coupon> coupon = boost::dynamic_pointer_cast<Coupon>(bd->cashflows()[i]);
318 
319  if (coupon) {
320  Date startDate = coupon->accrualStartDate();
321  Date endDate = coupon->accrualEndDate();
322  Date effectiveStartDate = (startDate <= npvDate && npvDate <= endDate) ? npvDate : startDate;
323  Date effectiveEndDate = (startDate <= computeDate && computeDate <= endDate) ? computeDate : endDate;
324  Date defaultDate = effectiveStartDate + (effectiveEndDate - effectiveStartDate) / 2;
325  Probability P = creditCurvePtr->defaultProbability(effectiveStartDate, effectiveEndDate);
326 
327  fwdBondRecovery +=
328  (*effectivePayoff)(coupon->nominal() * arguments_.bondNotional * recoveryVal - accruedAmount) * P *
329  (discountCurve_->discount(defaultDate));
330  }
331  }
332 
333  // B) Recovery for time period before coupons are present
334  boost::shared_ptr<Coupon> firstCoupon = boost::dynamic_pointer_cast<Coupon>(bd->cashflows()[0]);
335  if (firstCoupon) {
336  Date startDate = npvDate; // face value recovery starting with npvDate
337  Date stopDate = std::min(bd->cashflows()[0]->date(), computeDate);
338  while (startDate < stopDate) {
339  Date stepDate = startDate + timestepPeriod_;
340  Date endDate = (stepDate > stopDate) ? stopDate : stepDate;
341  Date defaultDate = startDate + (endDate - startDate) / 2;
342  Probability P = creditCurvePtr->defaultProbability(startDate, endDate);
343 
344  fwdBondRecovery +=
345  (*effectivePayoff)(firstCoupon->nominal() * arguments_.bondNotional * recoveryVal - accruedAmount) * P *
346  (discountCurve_->discount(defaultDate));
347  startDate = stepDate;
348  }
349  }
350  // C) ZCB
351  /* If there are no coupon, as in a Zero Bond, we must integrate over the entire period from npv date to
352  maturity. The timestepPeriod specified is used as provide the steps for the integration. This only applies
353  to bonds with 1 cashflow, identified as a final redemption payment.
354  */
355  if (bd->cashflows().size() == 1) {
356  boost::shared_ptr<Redemption> redemption = boost::dynamic_pointer_cast<Redemption>(bd->cashflows()[0]);
357  if (redemption) {
358  Date startDate = npvDate;
359  while (startDate < redemption->date()) {
360  Date stepDate = startDate + timestepPeriod_;
361  Date endDate = (stepDate > redemption->date()) ? redemption->date() : stepDate;
362  Date defaultDate = startDate + (endDate - startDate) / 2;
363  Probability P = creditCurvePtr->defaultProbability(startDate, endDate);
364 
365  fwdBondRecovery +=
366  (*effectivePayoff)(redemption->amount() * arguments_.bondNotional * recoveryVal - accruedAmount) *
367  P * (discountCurve_->discount(defaultDate));
368  startDate = stepDate;
369  }
370  }
371  }
372 
373  results_.additionalResults["forwardBondCashflow"] = fwdBondCashflows;
374  results_.additionalResults["forwardBondCashflowPayDates"] = fwdBondCashflowPayDates;
375  results_.additionalResults["forwardBondCashflowSurvivalProbabilities"] = fwdBondCashflowSurvivalProbabilities;
376  results_.additionalResults["forwardBondCashflowDiscountFactors"] = fwdBondCashflowDiscountFactors;
377  results_.additionalResults["forwardBondRecovery"] = fwdBondRecovery;
378 
379  forwardContractPresentValue += fwdBondRecovery;
380 
381  return boost::make_tuple(forwardContractForwardValue, forwardContractPresentValue);
382 }
