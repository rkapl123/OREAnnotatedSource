60void CrossCcyBasisMtMResetSwap::initialize() {
61 // Pay (foreign) leg
62 if (auto on = boost::dynamic_pointer_cast<QuantLib::OvernightIndex>(foreignIndex_)) {
63 // ON leg
64 if (foreignIsAveraged_ && *foreignIsAveraged_) {
65 legs_[0] = QuantExt::AverageONLeg(foreignSchedule_, on)
66 .withNotional(foreignNominal_)
67 .withSpread(foreignSpread_)
68 .withPaymentLag(foreignPaymentLag_)
69 .withLookback(foreignLookback_ ? *foreignLookback_ : 0 * Days)
70 .withFixingDays(foreignFixingDays_ ? *foreignFixingDays_ : 0)
71 .withRateCutoff(foreignRateCutoff_ ? *foreignRateCutoff_ : 0)
72 .withTelescopicValueDates(telescopicValueDates_);
73 } else {
74 legs_[0] = QuantExt::OvernightLeg(foreignSchedule_, on)
75 .withNotionals(foreignNominal_)
76 .withSpreads(foreignSpread_)
77 .withPaymentLag(foreignPaymentLag_)
78 .includeSpread(foreignIncludeSpread_ ? *foreignIncludeSpread_ : false)
79 .withLookback(foreignLookback_ ? *foreignLookback_ : 0 * Days)
80 .withFixingDays(foreignFixingDays_ ? *foreignFixingDays_ : 0)
81 .withRateCutoff(foreignRateCutoff_ ? *foreignRateCutoff_ : 0)
82 .withTelescopicValueDates(telescopicValueDates_);
83 }
84 } else {
85 // Ibor leg
86 legs_[0] = IborLeg(foreignSchedule_, foreignIndex_)
87 .withNotionals(foreignNominal_)
88 .withSpreads(foreignSpread_)
89 .withPaymentLag(foreignPaymentLag_);
90 }
91 receiveDomestic_ ? payer_[0] = -1.0 : payer_[0] = +1.0;
92
93 currencies_[0] = foreignCurrency_;
94 // Pay leg notional exchange at start.
95 Date initialPayDate = foreignSchedule_.dates().front();
96 boost::shared_ptr<CashFlow> initialPayCF(new SimpleCashFlow(-foreignNominal_, initialPayDate));
97 legs_[0].insert(legs_[0].begin(), initialPayCF);
98 // Pay leg notional exchange at end.
99 Date finalPayDate = foreignSchedule_.dates().back();
100 boost::shared_ptr<CashFlow> finalPayCF(new SimpleCashFlow(foreignNominal_, finalPayDate));
101 legs_[0].push_back(finalPayCF);
102
103 // Receive (domestic/resettable) leg
104 // start by creating a dummy vanilla floating leg
105 if (auto on = boost::dynamic_pointer_cast<QuantLib::OvernightIndex>(domesticIndex_)) {
106 // ON leg
107 if (domesticIsAveraged_ && *domesticIsAveraged_) {
108 legs_[1] = QuantExt::AverageONLeg(domesticSchedule_, on)
109 .withNotional(0.0)
110 .withSpread(domesticSpread_)
111 .withPaymentLag(domesticPaymentLag_)
112 .withLookback(domesticLookback_ ? *domesticLookback_ : 0 * Days)
113 .withFixingDays(domesticFixingDays_ ? *domesticFixingDays_ : 0)
114 .withRateCutoff(domesticRateCutoff_ ? *domesticRateCutoff_ : 0)
115 .withTelescopicValueDates(telescopicValueDates_);
116 } else {
117 legs_[1] = QuantExt::OvernightLeg(domesticSchedule_, on)
118 .withNotionals(0.0)
119 .withSpreads(domesticSpread_)
120 .withPaymentLag(domesticPaymentLag_)
121 .includeSpread(domesticIncludeSpread_ ? *domesticIncludeSpread_ : false)
122 .withLookback(domesticLookback_ ? *domesticLookback_ : 0 * Days)
123 .withFixingDays(domesticFixingDays_ ? *domesticFixingDays_ : 0)
124 .withRateCutoff(domesticRateCutoff_ ? *domesticRateCutoff_ : 0)
125 .withTelescopicValueDates(telescopicValueDates_);
126 }
127 } else {
128 // Ibor leg
129 legs_[1] = IborLeg(domesticSchedule_, domesticIndex_)
130 .withNotionals(0.0)
131 .withSpreads(domesticSpread_)
132 .withPaymentLag(domesticPaymentLag_);
133 }
134 receiveDomestic_ ? payer_[1] = +1.0 : payer_[1] = -1.0;
135 currencies_[1] = domesticCurrency_;
136 // replace the coupons with a FloatingRateFXLinkedNotionalCoupon
137 // (skip the first coupon as it has a fixed notional)
138 for (Size j = 0; j < legs_[1].size(); ++j) {
139 boost::shared_ptr<FloatingRateCoupon> coupon = boost::dynamic_pointer_cast<FloatingRateCoupon>(legs_[1][j]);
140 Date fixingDate = fxIndex_->fixingCalendar().advance(coupon->accrualStartDate(),
141 -static_cast<Integer>(fxIndex_->fixingDays()), Days);
142 boost::shared_ptr<FloatingRateFXLinkedNotionalCoupon> fxLinkedCoupon(
143 new FloatingRateFXLinkedNotionalCoupon(fixingDate, foreignNominal_, fxIndex_, coupon));
144 legs_[1][j] = fxLinkedCoupon;
145 }
146 // now build a separate leg to store the domestic (resetting) notionals
147 receiveDomestic_ ? payer_[2] = +1.0 : payer_[2] = -1.0;
148 currencies_[2] = domesticCurrency_;
149 for (Size j = 0; j < legs_[1].size(); j++) {
150 boost::shared_ptr<Coupon> c = boost::dynamic_pointer_cast<Coupon>(legs_[1][j]);
151 QL_REQUIRE(c, "Resetting XCCY - expected Coupon"); // TODO: fixed fx resettable?
152 // build a pair of notional flows, one at the start and one at the end of
153 // the accrual period. Both with the same FX fixing date
154 Date fixingDate = fxIndex_->fixingCalendar().advance(c->accrualStartDate(),
155 -static_cast<Integer>(fxIndex_->fixingDays()), Days);
156 legs_[2].push_back(boost::shared_ptr<CashFlow>(
157 new FXLinkedCashFlow(c->accrualStartDate(), fixingDate, -foreignNominal_, fxIndex_)));
158 legs_[2].push_back(boost::shared_ptr<CashFlow>(
159 new FXLinkedCashFlow(c->accrualEndDate(), fixingDate, foreignNominal_, fxIndex_)));
160 }
161
162 // Register the instrument with all cashflows on each leg.
163 for (Size legNo = 0; legNo < legs_.size(); legNo++) {
164 Leg::iterator it;
165 for (it = legs_[legNo].begin(); it != legs_[legNo].end(); ++it) {
166 registerWith(*it);
167 }
168 }
169}

186void CrossCcyBasisMtMResetSwap::fetchResults(const PricingEngine::results* r) const {
187
188 CrossCcySwap::fetchResults(r);
189
190 const CrossCcyBasisMtMResetSwap::results* results = dynamic_cast<const CrossCcyBasisMtMResetSwap::results*>(r);
191 if (results) {
192 /* If PricingEngine::results are of type
193 CrossCcyBasisSwap::results */
194 fairForeignSpread_ = results->fairForeignSpread;
195 fairDomesticSpread_ = results->fairDomesticSpread;
196 } else {
197 /* If not, e.g. if the engine is a CrossCcySwap::engine */
198 fairForeignSpread_ = Null<Spread>();
199 fairDomesticSpread_ = Null<Spread>();
200 }
201
202 /* Calculate the fair pay and receive spreads if they are null */
203 static Spread basisPoint = 1.0e-4;
204 if (fairForeignSpread_ == Null<Spread>()) {
205 if (legBPS_[0] != Null<Real>())
206 fairForeignSpread_ = foreignSpread_ - NPV_ / (legBPS_[0] / basisPoint);
207 }
208 if (fairDomesticSpread_ == Null<Spread>()) {
209 if (legBPS_[1] != Null<Real>())
210 fairDomesticSpread_ = domesticSpread_ - NPV_ / (legBPS_[1] / basisPoint);
211 }
212}
