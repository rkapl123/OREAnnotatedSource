120 inline void CirppImpliedDefaultTermStructure::update() {
121  if (!purelyTimeBased_) {
122  relativeTime_ = dayCounter().yearFraction(model_->defaultCurve()->referenceDate(), referenceDate_);
123  }
124  notifyObservers();
125 }

127 inline Real CirppImpliedDefaultTermStructure::survivalProbabilityImpl(Time t) const {
128  QL_REQUIRE(t >= 0.0, "negative time (" << t << ") given");
129  if (QuantLib::close_enough(t, 0))
130  return 1.0;
131  return model_->survivalProbability(relativeTime_, relativeTime_+ t, y_);
132 }
