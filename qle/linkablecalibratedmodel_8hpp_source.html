106 bool test(const Array& params) const override {
107 Size k = 0;
108 for (Size i = 0; i < arguments_.size(); i++) {
109 Size size = arguments_[i]->size();
110 Array testParams(size);
111 for (Size j = 0; j < size; j++, k++)
112 testParams[j] = params[k];
113 if (!arguments_[i]->testParams(testParams))
114 return false;
115 }
116 return true;
117 }
118
119 Array upperBound(const Array& params) const override {
120 Size k = 0, k2 = 0;
121 Size totalSize = 0;
122 for (Size i = 0; i < arguments_.size(); i++) {
123 totalSize += arguments_[i]->size();
124 }
125 Array result(totalSize);
126 for (Size i = 0; i < arguments_.size(); i++) {
127 Size size = arguments_[i]->size();
128 Array partialParams(size);
129 for (Size j = 0; j < size; j++, k++)
130 partialParams[j] = params[k];
131 Array tmpBound = arguments_[i]->constraint().upperBound(partialParams);
132 for (Size j = 0; j < size; j++, k2++)
133 result[k2] = tmpBound[j];
134 }
135 return result;
136 }
