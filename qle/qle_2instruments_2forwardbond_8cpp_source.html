80void ForwardBond::arguments::validate() const {
81 QL_REQUIRE(underlying, "bond pointer is null");
82 QL_REQUIRE((payoff != nullptr && lockRate == Null<Real>()) || (payoff == nullptr && lockRate != Null<Real>()),
83 "exactly one of payoff or lockRate must be filled");
84 QL_REQUIRE(lockRate == Null<Real>() || longInForward, "if lockRate is given, longInForward must be given as well");
85}
