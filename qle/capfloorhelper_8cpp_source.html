36// The argument to RelativeDateBootstrapHelper below is not simply the `quote`. Instead we create a DerivedQuote that,
37// each time it is asked for its value, it returns a premium by calling CapFloorHelper::npv with the `quote` value. In
38// this way, the `BootstrapHelper<T>::quoteError()` is always based on the cap floor premium and we do not have imply
39// the volatility. This leads to all kinds of issues when deciding on max and min values in the iterative bootstrap
40// where the quote volatility type input is of one type and the optionlet structure that we are trying to build is of
41// another different type.
42CapFloorHelper::CapFloorHelper(Type type, const Period& tenor, Rate strike, const Handle<Quote>& quote,
43 const boost::shared_ptr<IborIndex>& iborIndex,
44 const Handle<YieldTermStructure>& discountingCurve, bool moving,
45 const QuantLib::Date& effectiveDate, QuoteType quoteType,
46 QuantLib::VolatilityType quoteVolatilityType, QuantLib::Real quoteDisplacement,
47 bool endOfMonth, bool firstCapletExcluded)
48 : RelativeDateBootstrapHelper<OptionletVolatilityStructure>(
49 Handle<Quote>(boost::make_shared<DerivedQuote<boost::function<Real(Real)> > >(
50 quote, boost::bind(&CapFloorHelper::npv, this, _1)))),
51 type_(type), tenor_(tenor), strike_(strike), iborIndex_(iborIndex), discountHandle_(discountingCurve),
52 moving_(moving), effectiveDate_(effectiveDate), quoteType_(quoteType), quoteVolatilityType_(quoteVolatilityType),
53 quoteDisplacement_(quoteDisplacement), endOfMonth_(endOfMonth), firstCapletExcluded_(firstCapletExcluded),
54 rawQuote_(quote), initialised_(false) {
55
56 if (quoteType_ == Premium) {
57 QL_REQUIRE(type_ != Automatic, "Cannot have CapFloorHelper type 'Automatic' with quote type of Premium");
58 }
59
60 QL_REQUIRE(!(moving_ && effectiveDate_ != Date()),
61 "A fixed effective date does not make sense for a moving helper");
62
63 registerWith(iborIndex_);
64 registerWith(discountHandle_);
65
66 initializeDates();
67 initialised_ = true;
68}

70void CapFloorHelper::initializeDates() {
71
72 if (!initialised_ || moving_) {
73 CapFloor::Type capFloorType = CapFloor::Cap;
74 if (type_ == CapFloorHelper::Floor) {
75 capFloorType = CapFloor::Floor;
76 }
77
78 // Initialise the instrument and a copy
79 // The strike can be Null<Real>() to indicate an ATM cap floor helper
80 Rate dummyStrike = strike_ == Null<Real>() ? 0.01 : strike_;
81 capFloor_ = MakeCapFloor(capFloorType, tenor_, iborIndex_, dummyStrike, 0 * Days)
82 .withEndOfMonth(endOfMonth_)
83 .withEffectiveDate(effectiveDate_, firstCapletExcluded_);
84 capFloorCopy_ = MakeCapFloor(capFloorType, tenor_, iborIndex_, dummyStrike, 0 * Days)
85 .withEndOfMonth(endOfMonth_)
86 .withEffectiveDate(effectiveDate_, firstCapletExcluded_);
87
88 // Maturity date is just the maturity date of the cap floor
89 maturityDate_ = capFloor_->maturityDate();
90
91 // We need the leg underlying the cap floor to determine the remaining date members
92 const Leg& leg = capFloor_->floatingLeg();
93
94 // Earliest date is the first optionlet fixing date
95 boost::shared_ptr<CashFlow> cf = leg.front();
96 boost::shared_ptr<FloatingRateCoupon> frc = boost::dynamic_pointer_cast<FloatingRateCoupon>(cf);
97 QL_REQUIRE(frc, "Expected the first cashflow on the cap floor instrument to be a FloatingRateCoupon");
98 earliestDate_ = frc->fixingDate();
99
100 // Remaining dates are each equal to the fixing date on the final optionlet
101 cf = leg.back();
102 frc = boost::dynamic_pointer_cast<FloatingRateCoupon>(cf);
103 QL_REQUIRE(frc, "Expected the final cashflow on the cap floor instrument to be a FloatingRateCoupon");
104 pillarDate_ = latestDate_ = latestRelevantDate_ = frc->fixingDate();
105 }
106}

108void CapFloorHelper::setTermStructure(OptionletVolatilityStructure* ovts) {
109
110 if (strike_ == Null<Real>()) {
111 // If the strike is Null<Real>(), we want an ATM helper
112 Rate atm = capFloor_->atmRate(**discountHandle_);
113 capFloor_ = MakeCapFloor(capFloor_->type(), tenor_, iborIndex_, atm, 0 * Days)
114 .withEndOfMonth(endOfMonth_)
115 .withEffectiveDate(effectiveDate_, firstCapletExcluded_);
116 capFloorCopy_ = MakeCapFloor(capFloor_->type(), tenor_, iborIndex_, atm, 0 * Days)
117 .withEndOfMonth(endOfMonth_)
118 .withEffectiveDate(effectiveDate_, firstCapletExcluded_);
119
120 } else if (type_ == CapFloorHelper::Automatic && quoteType_ != Premium) {
121 // If the helper is set to automatically choose the underlying instrument type, do it now based on the ATM rate
122 Rate atm = capFloor_->atmRate(**discountHandle_);
123 CapFloor::Type capFloorType = atm > strike_ ? CapFloor::Floor : CapFloor::Cap;
124 if (capFloor_->type() != capFloorType) {
125 capFloor_ = MakeCapFloor(capFloorType, tenor_, iborIndex_, strike_, 0 * Days)
126 .withEndOfMonth(endOfMonth_)
127 .withEffectiveDate(effectiveDate_, firstCapletExcluded_);
128 capFloorCopy_ = MakeCapFloor(capFloorType, tenor_, iborIndex_, strike_, 0 * Days)
129 .withEndOfMonth(endOfMonth_)
130 .withEffectiveDate(effectiveDate_, firstCapletExcluded_);
131 }
132 }
133
134 // Set this helper's optionlet volatility structure
135 boost::shared_ptr<OptionletVolatilityStructure> temp(ovts, no_deletion);
136 ovtsHandle_.linkTo(temp, false);
137
138 // Set the term structure pointer member variable in the base class
139 RelativeDateBootstrapHelper<OptionletVolatilityStructure>::setTermStructure(ovts);
140
141 // Set this helper's pricing engine depending on the type of the optionlet volatilities
142 if (ovts->volatilityType() == ShiftedLognormal) {
143 capFloor_->setPricingEngine(boost::make_shared<BlackCapFloorEngine>(discountHandle_, ovtsHandle_));
144 } else {
145 capFloor_->setPricingEngine(boost::make_shared<BachelierCapFloorEngine>(discountHandle_, ovtsHandle_));
146 }
147
148 // If the quote type is not a premium, we will need to use capFloorCopy_ to return the premium from the volatility
149 // quote
150 if (quoteType_ != Premium) {
151 if (quoteVolatilityType_ == ShiftedLognormal) {
152 capFloorCopy_->setPricingEngine(boost::make_shared<BlackCapFloorEngine>(
153 discountHandle_, rawQuote_, ovtsHandle_->dayCounter(), quoteDisplacement_));
154 } else {
155 capFloorCopy_->setPricingEngine(
156 boost::make_shared<BachelierCapFloorEngine>(discountHandle_, rawQuote_, ovtsHandle_->dayCounter()));
157 }
158 }
159}

161Real CapFloorHelper::impliedQuote() const {
162 QL_REQUIRE(termStructure_ != 0, "CapFloorHelper's optionlet volatility term structure has not been set");
163 capFloor_->deepUpdate();
164 return capFloor_->NPV();
165}

167void CapFloorHelper::accept(AcyclicVisitor& v) {
168 if (Visitor<CapFloorHelper>* v1 = dynamic_cast<Visitor<CapFloorHelper>*>(&v))
169 v1->visit(*this);
170 else
171 RelativeDateBootstrapHelper<OptionletVolatilityStructure>::accept(v);
172}

174Real CapFloorHelper::npv(Real quoteValue) {
175 if (quoteType_ == Premium) {
176 return quoteValue;
177 } else {
178 // If the quote value is a volatility, return the premium
179 return capFloorCopy_->NPV();
180 }
181}
