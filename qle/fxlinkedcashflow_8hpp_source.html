73 //! FX Linked cash-flow
74 /*!
75  * Cashflow of Domestic currency where the amount is fx linked
76  * to some fixed foreign amount.
77  *
78  * For example: a JPY flow based off 1M USD, if the USDJPY FX rate
79  * is 123.45 then the JPY amount is 123.45 M JPY.
80  *
81  * FXLinkedCashFlow checks the FX fixing date against the eval date
82  *
83  * For future fixings (date > eval) this class calculates the FX Fwd
84  * rate (using the provided FX Spot rate and FOR and DOM yield curves)
85  *
86  * For todays fixing (date = eval) this class converts the foreign
87  * amount using the provided FX Spot rate.
88  *
89  * For previous fixings (date < eval) this class checks the QuantLib
90  * IndexManager to get the FX fixing at which the foreign rate should be
91  * converted at. The name of the index is a parameter to the constructor.
92  *
93  * This is not a lazy object.
94 
95  \ingroup cashflows
96  */
97 class FXLinkedCashFlow : public CashFlow, public FXLinked, public Observer {
98 public:
99  FXLinkedCashFlow(const Date& cashFlowDate, const Date& fixingDate, Real foreignAmount,
100  boost::shared_ptr<FxIndex> fxIndex);
101 
102  //! \name CashFlow interface
103  //@{
104  Date date() const override { return cashFlowDate_; }
105  Real amount() const override { return foreignAmount() * fxRate(); }
106  //@}
107 
108  //! \name Visitability

110  void accept(AcyclicVisitor&) override;
111  //@}
112
