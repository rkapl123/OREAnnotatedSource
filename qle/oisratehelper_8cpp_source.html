58 void OISRateHelper::initializeDates() {
59 
60  Calendar paymentCalendar_ = overnightIndex_->fixingCalendar();
61 
62  swap_ = MakeOIS(swapTenor_, overnightIndex_, 0.0)
63  .withSettlementDays(settlementDays_)
64  .withFixedLegDayCount(fixedDayCounter_)
65  .withEndOfMonth(endOfMonth_)
66  .withPaymentFrequency(paymentFrequency_)
67  .withRule(rule_)
68  .withPaymentCalendar(paymentCalendar_)
69  .withPaymentAdjustment(paymentAdjustment_)
70  .withPaymentLag(paymentLag_)
71  .withDiscountingTermStructure(discountRelinkableHandle_)
72  .withTelescopicValueDates(telescopicValueDates_);
73  // TODO: patch QL?
74  //.withFixedAccrualConvention(fixedConvention_)
75 
76  earliestDate_ = swap_->startDate();
77  latestDate_ = swap_->maturityDate();
78 
79  // Latest Date may need to be updated due to payment lag.
80  Date date;
81  if (paymentLag_ != 0) {
82  date = paymentCalendar_.advance(latestDate_, paymentLag_, Days, paymentAdjustment_, false);
83  latestDate_ = std::max(date, latestDate_);
84  }
85 }

87 void OISRateHelper::setTermStructure(YieldTermStructure* t) {
88  // do not set the relinkable handle as an observer
89  // force recalculation when needed
90  bool observer = false;
91 
92  boost::shared_ptr<YieldTermStructure> temp(t, no_deletion);
93  termStructureHandle_.linkTo(temp, observer);
94 
95  if (discountHandle_.empty())
96  discountRelinkableHandle_.linkTo(temp, observer);
97  else
98  discountRelinkableHandle_.linkTo(*discountHandle_, observer);
99 
100  RelativeDateRateHelper::setTermStructure(t);
