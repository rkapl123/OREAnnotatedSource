38struct sort_by_cost {

44template <class I> void randomize(I begin, I end, const MersenneTwisterUniformRng& rng) {
45 Size n = static_cast<Size>(end - begin);
46 for (Size i = n - 1; i > 0; --i) {
47 std::swap(begin[i], begin[rng.nextInt32() % (i + 1)]);
48 }
49}

53bool DifferentialEvolution_MT::checkMaxTime() const {
54 if (maxTime_.empty())
55 return false;
56 QL_REQUIRE(maxTime_.length() == 15, "maxTime (" << maxTime_ << ") must have format YYYYMMDDTHHMMSS");
57 std::string ts = to_iso_string(boost::posix_time::microsec_clock::local_time()).substr(0, 15);
58 return ts > maxTime_;

61EndCriteria::Type DifferentialEvolution_MT::minimize(Problem_MT& p, const EndCriteria& endCriteria) {
62 EndCriteria::Type ecType;
63
64 if (configuration().upperBound.empty()) {
65 upperBound_ = p.constraint().upperBound(p.currentValue());
66 } else {
67 QL_REQUIRE(configuration().upperBound.size() == p.currentValue().size(),
68 "wrong upper bound size in differential evolution configuration");
69 upperBound_ = configuration().upperBound;
70 }
71 if (configuration().lowerBound.empty()) {
72 lowerBound_ = p.constraint().lowerBound(p.currentValue());
73 } else {
74 QL_REQUIRE(configuration().lowerBound.size() == p.currentValue().size(),
75 "wrong lower bound size in differential evolution configuration");
76 lowerBound_ = configuration().lowerBound;
77 }
78
79 currGenSizeWeights_ = Array(configuration().populationMembers, configuration().stepsizeWeight);
80 currGenCrossover_ = Array(configuration().populationMembers, configuration().crossoverProbability);
81
82 std::vector<Candidate> population;
83 if (!configuration().initialPopulation.empty()) {
84 population.resize(configuration().initialPopulation.size());
85 for (Size i = 0; i < population.size(); ++i) {
86 population[i].values = configuration().initialPopulation[i];
87 QL_REQUIRE(population[i].values.size() == p.currentValue().size(),
88 "wrong values size in initial population");
89 }
90 } else {
91 population = std::vector<Candidate>(configuration().populationMembers, Candidate(p.currentValue().size()));
92 fillInitialPopulation(population, p);
93 }
94
95 updateCost(population, p);
96
97 std::partial_sort(population.begin(), population.begin() + 1, population.end(), sort_by_cost());
98 bestMemberEver_ = population.front();
99 Real fxOld = population.front().cost;
100 Size iteration = 0, stationaryPointIteration = 0;
101
102 // main loop - calculate consecutive emerging populations
103 while (!endCriteria.checkMaxIterations(iteration++, ecType) && !checkMaxTime()) {
104 calculateNextGeneration(population, p);
105 std::partial_sort(population.begin(), population.begin() + 1, population.end(), sort_by_cost());
106 if (population.front().cost < bestMemberEver_.cost)
107 bestMemberEver_ = population.front();
108 Real fxNew = population.front().cost;
109 if (endCriteria.checkStationaryFunctionValue(fxOld, fxNew, stationaryPointIteration, ecType))
110 break;
111 fxOld = fxNew;
112 };
113 p.setCurrentValue(bestMemberEver_.values);
114 p.setFunctionValue(bestMemberEver_.cost);
115 if (checkMaxTime())
116 ecType = EndCriteria::Type::Unknown;
117 return ecType;
118}

120void DifferentialEvolution_MT::calculateNextGeneration(std::vector<Candidate>& population, Problem_MT& p) const {
121
122 std::vector<Candidate> mirrorPopulation;
123 std::vector<Candidate> oldPopulation = population;
124
125 switch (configuration().strategy) {
126
127 case QuantLib::DifferentialEvolution::Rand1Standard: {
128 randomize(population.begin(), population.end(), rng_);
129 std::vector<Candidate> shuffledPop1 = population;
130 randomize(population.begin(), population.end(), rng_);
131 std::vector<Candidate> shuffledPop2 = population;
132 randomize(population.begin(), population.end(), rng_);
133 mirrorPopulation = shuffledPop1;
134
135 for (Size popIter = 0; popIter < population.size(); popIter++) {
136 population[popIter].values =
137 population[popIter].values +
138 configuration().stepsizeWeight * (shuffledPop1[popIter].values - shuffledPop2[popIter].values);
139 }
140 } break;
141
142 case QuantLib::DifferentialEvolution::BestMemberWithJitter: {
143 randomize(population.begin(), population.end(), rng_);
144 std::vector<Candidate> shuffledPop1 = population;
145 randomize(population.begin(), population.end(), rng_);
146 Array jitter(population[0].values.size(), 0.0);
147
148 for (Size popIter = 0; popIter < population.size(); popIter++) {
149 for (double& jitterIter : jitter) {
150 jitterIter = rng_.nextReal();
151 }
152 population[popIter].values =
153 bestMemberEver_.values + (shuffledPop1[popIter].values - population[popIter].values) *
154 (0.0001 * jitter + configuration().stepsizeWeight);
155 }
156 mirrorPopulation = std::vector<Candidate>(population.size(), bestMemberEver_);
157 } break;
158
159 case QuantLib::DifferentialEvolution::CurrentToBest2Diffs: {
160 randomize(population.begin(), population.end(), rng_);
161 std::vector<Candidate> shuffledPop1 = population;
162 randomize(population.begin(), population.end(), rng_);
163
164 for (Size popIter = 0; popIter < population.size(); popIter++) {
165 population[popIter].values =
166 oldPopulation[popIter].values +
167 configuration().stepsizeWeight * (bestMemberEver_.values - oldPopulation[popIter].values) +
168 configuration().stepsizeWeight * (population[popIter].values - shuffledPop1[popIter].values);
169 }
170 mirrorPopulation = shuffledPop1;
171 } break;
172
173 case QuantLib::DifferentialEvolution::Rand1DiffWithPerVectorDither: {
174 randomize(population.begin(), population.end(), rng_);
175 std::vector<Candidate> shuffledPop1 = population;
176 randomize(population.begin(), population.end(), rng_);
177 std::vector<Candidate> shuffledPop2 = population;
178 randomize(population.begin(), population.end(), rng_);
179 mirrorPopulation = shuffledPop1;
180 Array FWeight = Array(population.front().values.size(), 0.0);
181 for (double& fwIter : FWeight)
182 fwIter = (1.0 - configuration().stepsizeWeight) * rng_.nextReal() + configuration().stepsizeWeight;
183 for (Size popIter = 0; popIter < population.size(); popIter++) {
184 population[popIter].values =
185 population[popIter].values + FWeight * (shuffledPop1[popIter].values - shuffledPop2[popIter].values);
186 }
187 } break;
188
189 case QuantLib::DifferentialEvolution::Rand1DiffWithDither: {
190 randomize(population.begin(), population.end(), rng_);
191 std::vector<Candidate> shuffledPop1 = population;
192 randomize(population.begin(), population.end(), rng_);
193 std::vector<Candidate> shuffledPop2 = population;
194 randomize(population.begin(), population.end(), rng_);
195 mirrorPopulation = shuffledPop1;
196 Real FWeight = (1.0 - configuration().stepsizeWeight) * rng_.nextReal() + configuration().stepsizeWeight;
197 for (Size popIter = 0; popIter < population.size(); popIter++) {
198 population[popIter].values =
199 population[popIter].values + FWeight * (shuffledPop1[popIter].values - shuffledPop2[popIter].values);
200 }
201 } break;
202
203 case QuantLib::DifferentialEvolution::EitherOrWithOptimalRecombination: {
204 randomize(population.begin(), population.end(), rng_);
205 std::vector<Candidate> shuffledPop1 = population;
206 randomize(population.begin(), population.end(), rng_);
207 std::vector<Candidate> shuffledPop2 = population;
208 randomize(population.begin(), population.end(), rng_);
209 mirrorPopulation = shuffledPop1;
210 Real probFWeight = 0.5;
211 if (rng_.nextReal() < probFWeight) {
212 for (Size popIter = 0; popIter < population.size(); popIter++) {
213 population[popIter].values =
214 oldPopulation[popIter].values +
215 configuration().stepsizeWeight * (shuffledPop1[popIter].values - shuffledPop2[popIter].values);
216 }
217 } else {
218 Real K = 0.5 * (configuration().stepsizeWeight + 1); // invariant with respect to probFWeight used
219 for (Size popIter = 0; popIter < population.size(); popIter++) {
220 population[popIter].values =
221 oldPopulation[popIter].values + K * (shuffledPop1[popIter].values - shuffledPop2[popIter].values -
222 2.0 * population[popIter].values);
223 }
224 }
225 } break;
226
227 case QuantLib::DifferentialEvolution::Rand1SelfadaptiveWithRotation: {
228 randomize(population.begin(), population.end(), rng_);
229 std::vector<Candidate> shuffledPop1 = population;
230 randomize(population.begin(), population.end(), rng_);
231 std::vector<Candidate> shuffledPop2 = population;
232 randomize(population.begin(), population.end(), rng_);
233 mirrorPopulation = shuffledPop1;
234
235 adaptSizeWeights();
236
237 for (Size popIter = 0; popIter < population.size(); popIter++) {
238 if (rng_.nextReal() < 0.1) {
239 population[popIter].values = rotateArray(bestMemberEver_.values);
240 } else {
241 population[popIter].values =
242 bestMemberEver_.values +
243 currGenSizeWeights_[popIter] * (shuffledPop1[popIter].values - shuffledPop2[popIter].values);
244 }
245 }
246 } break;
247
248 default:
249 QL_FAIL("Unknown strategy (" << Integer(configuration().strategy) << ")");
250 }
251 // in order to avoid unnecessary copying we use the same population object for mutants
252 crossover(oldPopulation, population, population, mirrorPopulation, p);
253}

255void DifferentialEvolution_MT::crossover(const std::vector<Candidate>& oldPopulation,
256 std::vector<Candidate>& population,
257 const std::vector<Candidate>& mutantPopulation,
258 const std::vector<Candidate>& mirrorPopulation, Problem_MT& p) const {
259
260 if (configuration().crossoverIsAdaptive) {
261 adaptCrossover();
262 }
263
264 Array mutationProbabilities = getMutationProbabilities(population);
265
266 std::vector<Array> crossoverMask(population.size(), Array(population.front().values.size(), 1.0));
267 std::vector<Array> invCrossoverMask = crossoverMask;
268 getCrossoverMask(crossoverMask, invCrossoverMask, mutationProbabilities);
269
270 // crossover of the old and mutant population
271
272 for (Size popIter = 0; popIter < population.size(); popIter++) {
273 population[popIter].values = oldPopulation[popIter].values * invCrossoverMask[popIter] +
274 mutantPopulation[popIter].values * crossoverMask[popIter];
275 // immediately apply bounds if specified
276 if (configuration().applyBounds) {
277 for (Size memIter = 0; memIter < population[popIter].values.size(); memIter++) {
278 if (population[popIter].values[memIter] > upperBound_[memIter])
279 population[popIter].values[memIter] =
280 upperBound_[memIter] +
281 rng_.nextReal() * (mirrorPopulation[popIter].values[memIter] - upperBound_[memIter]);
282 if (population[popIter].values[memIter] < lowerBound_[memIter])
283 population[popIter].values[memIter] =
284 lowerBound_[memIter] +
285 rng_.nextReal() * (mirrorPopulation[popIter].values[memIter] - lowerBound_[memIter]);
286 }
287 }
288 }
289
290 updateCost(population, p);
291}

293void DifferentialEvolution_MT::updateCost(std::vector<Candidate>& population, Problem_MT& p) const {
294 struct Worker {
295 Worker(std::vector<Candidate>& p, const Size s, const Size e, const boost::shared_ptr<CostFunction> c)
296 : population(p), start(s), end(e), costFunction(c) {}
297 void operator()() {
298 for (Size popIter = start; popIter < end; popIter++) {
299 // evaluate objective function as soon as possible to avoid unnecessary loops
300 try {
301 population[popIter].cost = costFunction->value(population[popIter].values);
302 } catch (Error&) {
303 population[popIter].cost = QL_MAX_REAL;
304 }
305 if (!std::isfinite(population[popIter].cost))
306 population[popIter].cost = QL_MAX_REAL;
307 }
308 }
309 std::vector<Candidate>& population;
310 const Size start, end;
311 const boost::shared_ptr<CostFunction> costFunction;
312 };
313
314 Size threads = p.costFunctions().size();
315 QL_REQUIRE(threads > 0, "DifferentialEvolution_MT: number of available threads is zero");
316
317 std::vector<Size> chunks(threads, std::max<Size>(population.size() / threads, 1));
318 int rest = population.size() - threads * chunks[0];
319 do {
320 Size i = 0;
321 while (i < chunks.size() && rest > 0) {
322 chunks[i]++;
323 rest--;
324 ++i;
325 }
326 } while (rest > 0);
327
328 std::vector<boost::shared_ptr<std::thread>> workers(threads);
329
330 Size start = 0, end;
331 for (Size thread = 0; thread < threads; ++thread) {
332 end = std::min(start + chunks[thread], population.size());
333 Worker worker(population, start, end, p.costFunctions()[thread]);
334 workers[thread] = boost::make_shared<std::thread>(worker);
335 start = end;
336 }
337
338 for (Size thread = 0; thread < threads; ++thread) {
339 workers[thread]->join();
340 }
341}

343void DifferentialEvolution_MT::getCrossoverMask(std::vector<Array>& crossoverMask, std::vector<Array>& invCrossoverMask,
344 const Array& mutationProbabilities) const {
345 for (Size cmIter = 0; cmIter < crossoverMask.size(); cmIter++) {
346 for (Size memIter = 0; memIter < crossoverMask[cmIter].size(); memIter++) {
347 if (rng_.nextReal() < mutationProbabilities[cmIter]) {
348 invCrossoverMask[cmIter][memIter] = 0.0;
349 } else {
350 crossoverMask[cmIter][memIter] = 0.0;
351 }
352 }
353 }
354}

356Array DifferentialEvolution_MT::getMutationProbabilities(const std::vector<Candidate>& population) const {
357 Array mutationProbabilities = currGenCrossover_;
358 switch (configuration().crossoverType) {
359 case DifferentialEvolution::Normal:
360 break;
361 case DifferentialEvolution::Binomial:
362 mutationProbabilities =
363 currGenCrossover_ * (1.0 - 1.0 / population.front().values.size()) + 1.0 / population.front().values.size();
364 break;
365 case DifferentialEvolution::Exponential:
366 for (Size coIter = 0; coIter < currGenCrossover_.size(); coIter++) {
367 mutationProbabilities[coIter] =
368 (1.0 - std::pow(currGenCrossover_[coIter], (int)population.front().values.size())) /
369 (population.front().values.size() * (1.0 - currGenCrossover_[coIter]));
370 }
371 break;
372 default:
373 QL_FAIL("Unknown crossover type (" << Integer(configuration().crossoverType) << ")");
374 break;
375 }
376 return mutationProbabilities;
377}

384void DifferentialEvolution_MT::adaptSizeWeights() const {
385 // [=Fl & =Fu] respectively see Brest, J. et al., 2006,
386 // "Self-Adapting Control Parameters in Differential
387 // Evolution"
388 Real sizeWeightLowerBound = 0.1, sizeWeightUpperBound = 0.9;
389 // [=tau1] A Comparative Study on Numerical Benchmark
390 // Problems." page 649 for reference
391 Real sizeWeightChangeProb = 0.1;
392 for (Size coIter = 0; coIter < currGenSizeWeights_.size(); coIter++) {
393 if (rng_.nextReal() < sizeWeightChangeProb)
394 currGenSizeWeights_[coIter] = sizeWeightLowerBound + rng_.nextReal() * sizeWeightUpperBound;
395 }
396}

398void DifferentialEvolution_MT::adaptCrossover() const {
399 Real crossoverChangeProb = 0.1; // [=tau2]
400 for (Size coIter = 0; coIter < currGenCrossover_.size(); coIter++) {
401 if (rng_.nextReal() < crossoverChangeProb)
402 currGenCrossover_[coIter] = rng_.nextReal();
403 }

406void DifferentialEvolution_MT::fillInitialPopulation(std::vector<Candidate>& population, const Problem_MT& p) const {
407
408 // use initial values provided by the user
409 population.front().values = p.currentValue();
410 // rest of the initial population is random
411 for (Size j = 1; j < population.size(); ++j) {
412 for (Size i = 0; i < p.currentValue().size(); ++i) {
413 Real l = lowerBound_[i], u = upperBound_[i];
414 population[j].values[i] = l + (u - l) * rng_.nextReal();
415 }
416 }
417}
