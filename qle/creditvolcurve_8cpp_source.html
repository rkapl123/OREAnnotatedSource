51 void CreditVolCurve::init() {
52  QL_REQUIRE(terms_.size() == termCurves_.size(), "CreditVolCurve: terms size (" << terms_.size()
53  << ") must match termCurves size ("
54  << termCurves_.size());
55  // sort terms and curves
56 
57  std::vector<Size> p(terms_.size());
58  std::iota(p.begin(), p.end(), 0);
59  std::sort(p.begin(), p.end(), [this](Size i, Size j) { return this->terms_[i] < this->terms_[j]; });
60 
61  std::vector<Period> sortedTerms(terms_.size());
62  std::vector<Handle<CreditCurve>> sortedCurves(terms_.size());
63  std::transform(p.begin(), p.end(), sortedTerms.begin(), [this](Size i) { return this->terms_[i]; });
64  std::transform(p.begin(), p.end(), sortedCurves.begin(), [this](Size i) { return this->termCurves_[i]; });
65 
66  terms_ = sortedTerms;
67  termCurves_ = sortedCurves;
68 
69  // register with curves
70 
71  for (const auto& c : termCurves_)
72  registerWith(c);

80 Real CreditVolCurve::volatility(const Real exerciseTime, const Real underlyingLength, const Real strike,
81  const CreditVolCurve::Type& targetType) const {
82  Date d = lowerDate(exerciseTime, referenceDate(), dayCounter());
83  Real t1 = timeFromReference(d);
84  Real t2 = timeFromReference(d + 1);
85  Real alpha = (t2 - exerciseTime) / (t2 - t1);
86  Real v1 = volatility(d, underlyingLength, strike, targetType);
87  if (close_enough(alpha, 1.0))
88  return v1;
89  return alpha * v1 + (1.0 - alpha) * volatility(d + 1, underlyingLength, strike, targetType);
90 }

104 Real CreditVolCurve::moneyness(const Real strike, const Real atmStrike) const {
105  if (type() == Type::Spread) {
106  return strike - atmStrike;
107  } else if (type() == Type::Price) {
108  return std::log(strike / atmStrike);
109  } else {
110  QL_FAIL("InterpolatingCreditVolCurve::moneyness(): internal error, type not handled");
111  }

114 Real CreditVolCurve::strike(const Real moneyness, const Real atmStrike) const {
115  if (type() == Type::Spread) {

118  return atmStrike * std::exp(moneyness);

128 Real CreditVolCurve::atmStrike(const Date& expiry, const Real underlyingLength) const {
129 
130  // do we have the desired value in the cache?
131 
132  calculate();
133  auto v = atmStrikeCache_.find(std::make_pair(expiry, underlyingLength));
134  if (v != atmStrikeCache_.end())
135  return v->second;
136 
137  /* we need at least on term curve to compute the atm strike properly - we set it to 0 / 1 (spread, price) if
138  we don't have a term, so that we can build strike-independent curves without curves. It is the user's
139  responsibility to provide terms for strike-dependent curves. */
140  if (terms().empty())
141  return type() == Type::Price ? 1.0 : 0.0;
142 
143  // interpolate in term length
144 
145  std::vector<Real> termLengths;
146  for (auto const& p : terms())
147  termLengths.push_back(periodToTime(p));
148 
149  Size termIndex_m, termIndex_p;
150  Real term_alpha;
151  std::tie(termIndex_m, termIndex_p, term_alpha) = interpolationIndices(termLengths, underlyingLength);
152 
153  // construct cds underlying(s) for the terms we will use for the interpolation
154 
155  std::map<Period, boost::shared_ptr<QuantExt::CreditDefaultSwap>> underlyings;
156 
157  const CreditCurve::RefData& refData = termCurves_[termIndex_m]->refData();
158  QL_REQUIRE(refData.runningSpread != Null<Real>(), "CreditVolCurve: need runningSpread for ATM strike computation. "
159  "Is the running spread in the term curve configuration?");
160 
161  /* Use index maturity date based on index start date. If no start date is given, assume this is a single name option
162  running from today. */
163  Date mat = cdsMaturity(refData.startDate != Null<Date>() ? refData.startDate : referenceDate(), terms_[termIndex_m],
164  refData.rule);
165  Date effExp = std::min(mat - 1, expiry);
166  Schedule schedule(effExp, mat, refData.tenor, refData.calendar, refData.convention, refData.termConvention,
167  refData.rule, refData.endOfMonth);
168  Date protectionStartDate = (refData.rule == DateGeneration::CDS || refData.rule == DateGeneration::CDS2015)
169  ? effExp
170  : schedule.dates().front();
171  DayCounter lastPeriodDayCounter = refData.lastPeriodDayCounter.empty() && refData.dayCounter == Actual360()
172  ? Actual360(true)
173  : refData.lastPeriodDayCounter;
174  underlyings[terms_[termIndex_m]] = boost::make_shared<CreditDefaultSwap>(
175  Protection::Buyer, 1.0, refData.runningSpread, schedule, refData.payConvention, refData.dayCounter, true,
176  CreditDefaultSwap::atDefault, protectionStartDate, boost::shared_ptr<Claim>(), lastPeriodDayCounter, true,
177  effExp, refData.cashSettlementDays);
178 
179  QL_REQUIRE(!termCurves_[termIndex_m]->rateCurve().empty() && !termCurves_[termIndex_p]->rateCurve().empty(),
180  "CreditVolCurve: need discounting rate curve of index for ATM strike computation.");
181  QL_REQUIRE(!termCurves_[termIndex_m]->recovery().empty() && !termCurves_[termIndex_p]->recovery().empty(),
182  "CreditVolCurve: need recovery rate of index for ATM strike computation.");
183 
184  auto engine = boost::make_shared<MidPointCdsEngine>(termCurves_[termIndex_m]->curve(),
185  termCurves_[termIndex_m]->recovery()->value(),
186  termCurves_[termIndex_m]->rateCurve());
187 
188  underlyings[terms_[termIndex_m]]->setPricingEngine(engine);
189 
190  // compute (interpolated) ATM strike
191 
192  Real atmStrike;
193  Real fep1, fep2, fairSpread1, fairSpread2, rpv01_1, rpv01_2, adjFairSpread1 = 0.0, adjFairSpread2 = 0.0,
194  discToExercise = 1.0, forwardPrice1, forwardPrice2,
195  adjForwardPrice1 = 1.0, adjForwardPrice2 = 1.0;
196 
197  fep1 = (1.0 - termCurves_[termIndex_m]->recovery()->value()) *
198  termCurves_[termIndex_m]->curve()->defaultProbability(effExp);
199  if (type() == Type::Spread) {
200  fairSpread1 = underlyings[terms_[termIndex_m]]->fairSpreadClean();
201  rpv01_1 = std::abs(underlyings[terms_[termIndex_m]]->couponLegNPV() +
202  underlyings[terms_[termIndex_m]]->accrualRebateNPV()) /
203  underlyings[terms_[termIndex_m]]->runningSpread();
204  adjFairSpread1 = fairSpread1 + fep1 / rpv01_1;
205  atmStrike = adjFairSpread1;
206  } else if (type() == Type::Price) {
207  discToExercise = termCurves_[termIndex_m]->rateCurve()->discount(effExp);
208  forwardPrice1 = (1.0 - underlyings[terms_[termIndex_m]]->NPV()) / discToExercise;
209  adjForwardPrice1 = forwardPrice1 - fep1 / discToExercise;
210  atmStrike = adjForwardPrice1;
211  } else {
212  QL_FAIL("InterpolatingCreditVolCurve::atmStrike(): internal error, type not handled");
213  }
214 
215  if (!close_enough(term_alpha, 1.0)) {
216  Date mat = cdsMaturity(refData.startDate, terms_[termIndex_p], refData.rule);
217  Date effExp = std::min(mat - 1, expiry);
218  Schedule schedule(effExp, mat, refData.tenor, refData.calendar, refData.convention, refData.termConvention,
219  refData.rule, refData.endOfMonth);
220  Date protectionStartDate = (refData.rule == DateGeneration::CDS || refData.rule == DateGeneration::CDS2015)
221  ? effExp
222  : schedule.dates().front();
223  underlyings[terms_[termIndex_p]] = boost::make_shared<CreditDefaultSwap>(
224  Protection::Buyer, 1.0, refData.runningSpread, schedule, refData.payConvention, refData.dayCounter, true,
225  CreditDefaultSwap::atDefault, protectionStartDate, boost::shared_ptr<Claim>(), lastPeriodDayCounter,
226  true, effExp, refData.cashSettlementDays);
227  auto engine = boost::make_shared<MidPointCdsEngine>(termCurves_[termIndex_p]->curve(),
228  termCurves_[termIndex_p]->recovery()->value(),
229  termCurves_[termIndex_p]->rateCurve());
230  underlyings[terms_[termIndex_p]]->setPricingEngine(engine);
231 
232  fep2 = (1.0 - termCurves_[termIndex_p]->recovery()->value()) *
233  termCurves_[termIndex_p]->curve()->defaultProbability(effExp);
234  if (type() == Type::Spread) {
235  fairSpread2 = underlyings[terms_[termIndex_p]]->fairSpreadClean();
236  rpv01_2 = std::abs(underlyings[terms_[termIndex_p]]->couponLegNPV() +
237  underlyings[terms_[termIndex_p]]->accrualRebateNPV()) /
238  underlyings[terms_[termIndex_p]]->runningSpread();
239  adjFairSpread2 = fairSpread2 + fep2 / rpv01_2;
240  atmStrike = term_alpha * adjFairSpread1 + (1.0 - term_alpha * adjFairSpread2);
241  } else if (type() == Type::Price) {
242  forwardPrice2 = (1.0 - underlyings[terms_[termIndex_p]]->NPV()) / discToExercise;
243  adjForwardPrice2 = forwardPrice2 - fep2 / discToExercise;
244  atmStrike = term_alpha * adjForwardPrice1 + (1.0 - term_alpha) * adjForwardPrice2;
245  } else {
246  QL_FAIL("InterpolatingCreditVolCurve::atmStrike(): internal error, type not handled");
247  }
248  }
249 
250  // add the result to the cache and return it
251 
252  atmStrikeCache_[std::make_pair(expiry, underlyingLength)] = atmStrike;
253  return atmStrike;
254 }

279 Real InterpolatingCreditVolCurve::volatility(const Date& expiry, const Real underlyingLength, const Real strike,
280  const Type& targetType) const {
281 
282  calculate();
283 
284  QL_REQUIRE(targetType == type(), "InterpolatingCreditVolCurve: Vol type conversion between strike types 'Price' "
285  "and 'Spread' is not supported. The vol "
286  "surface used to price an option must have the same strike type as the option.");
287 
288  Real effStrike = strike == Null<Real>() ? atmStrike(expiry, underlyingLength) : strike;
289 
290  // term interpolation
291 
292  Size termIndex_m, termIndex_p;
293  Real term_alpha;
294  std::tie(termIndex_m, termIndex_p, term_alpha) = interpolationIndices(smileTermLengths_, underlyingLength);
295 
296  // expiry interpolation
297 
298  Size expiryIndex_m, expiryIndex_p;
299  Real expiry_alpha;
300  Real t = timeFromReference(expiry);
301  std::tie(expiryIndex_m, expiryIndex_p, expiry_alpha) = interpolationIndices(smileExpiryTimes_, t);
302 
303  // smiles by expiry / term
304 
305  const Smile& smile_1_1 = smiles_[std::make_pair(smileExpiries_[expiryIndex_m], smileTerms_[termIndex_m])];
306  const Smile& smile_1_2 = smiles_[std::make_pair(smileExpiries_[expiryIndex_m], smileTerms_[termIndex_p])];
307  const Smile& smile_2_1 = smiles_[std::make_pair(smileExpiries_[expiryIndex_p], smileTerms_[termIndex_m])];
308  const Smile& smile_2_2 = smiles_[std::make_pair(smileExpiries_[expiryIndex_p], smileTerms_[termIndex_p])];
309 
310  // atm levels by expiry / term
311 
312  Real atm_1_1 = smile_1_1.first;
313  Real atm_1_2 = smile_1_2.first;
314  Real atm_2_1 = smile_2_1.first;
315  Real atm_2_2 = smile_2_2.first;
316 
317  // vols at target moneyness
318 
319  Real m = moneyness(effStrike, atmStrike(expiry, underlyingLength));
320  Real vol_1_1 = smile_1_1.second->operator()(this->strike(m, atm_1_1));
321  Real vol_1_2 = smile_1_2.second->operator()(this->strike(m, atm_1_2));
322  Real vol_2_1 = smile_2_1.second->operator()(this->strike(m, atm_2_1));
323  Real vol_2_2 = smile_2_2.second->operator()(this->strike(m, atm_2_2));
324 
325  // interpolate in term direction
326 
327  Real vol_1 = term_alpha * vol_1_1 + (1.0 - term_alpha) * vol_1_2;
328  Real vol_2 = term_alpha * vol_2_1 + (1.0 - term_alpha) * vol_2_2;
329 
330  // interpolate in expiry direction
331 
332  return std::sqrt((expiry_alpha * (vol_1 * vol_1 * smileExpiryTimes_[expiryIndex_m]) +
333  (1.0 - expiry_alpha) * (vol_2 * vol_2 * smileExpiryTimes_[expiryIndex_p])) /
334  t);
335 }

337 void InterpolatingCreditVolCurve::performCalculations() const {
338 
339  CreditVolCurve::performCalculations();
340 
341  QL_REQUIRE(!quotes_.empty(), "InterpolatingCreditVolCurve: no quotes given, can not build a volatility curve.");
342 
343  // For each term and option expiry create an interpolation object representing a vol smile.
344 
345  smileTerms_.clear();
346  smileExpiries_.clear();
347  smileTermLengths_.clear();
348  smileExpiryTimes_.clear();
349  strikes_.clear();
350  vols_.clear();
351  smiles_.clear();
352 
353  Period currentTerm = 0 * Days;
354  Date currentExpiry = Null<Date>();
355  std::vector<Real> currentStrikes;
356  std::vector<Real> currentVols;
357  Date expiry;
358  Period term;
359  Real strike;
360  Real vol;
361 
362  auto q = quotes_.begin();
363  for (Size i = 0; i <= quotes_.size(); ++i) {
364  if (i < quotes_.size()) {
365  expiry = std::get<0>(q->first);
366  term = std::get<1>(q->first);
367  strike = std::get<2>(q->first);
368  vol = q->second->value();
369  ++q;
370  } else {
371  currentTerm = term;
372  currentExpiry = expiry;
373  }
374  if (term != currentTerm || expiry != currentExpiry || i == quotes_.size()) {
375  if (!currentStrikes.empty()) {
376  if (currentStrikes.size() == 1) {
377  currentStrikes.push_back(currentStrikes.back() + 0.01);
378  currentVols.push_back(currentVols.back());
379  }
380  auto key = std::make_pair(currentExpiry, currentTerm);
381  auto s = strikes_.insert(std::make_pair(key, currentStrikes)).first;
382  auto v = vols_.insert(std::make_pair(key, currentVols)).first;
383  auto tmp = boost::make_shared<FlatExtrapolation>(
384  boost::make_shared<LinearInterpolation>(s->second.begin(), s->second.end(), v->second.begin()));
385  tmp->enableExtrapolation();
386  smiles_[key] = std::make_pair(atmStrike(currentExpiry, currentTerm), tmp);
387  currentStrikes.clear();
388  currentVols.clear();
389  smileTerms_.push_back(currentTerm);
390  smileExpiries_.push_back(currentExpiry);
391  }
392  currentTerm = term;
393  currentExpiry = expiry;
394  }
395  if (i < quotes_.size()) {
396  currentStrikes.push_back(strike);
397  currentVols.push_back(vol);
398  }
399  }
400 
401  // populate times vectors
402 
403  for (auto const& p : smileTerms_)
404  smileTermLengths_.push_back(periodToTime(p));
405  for (auto const& d : smileExpiries_)
406  smileExpiryTimes_.push_back(timeFromReference(d));
407 
408  /* For each term, add missing option expiries that we saw for other terms by creating an interpolated smile.
409  We interpolate in terms of
410  - absolute moneyness (Type = Spread)
411  - log-moneyness (Type = Price)
412  */
413 
414  for (auto const& term : smileTerms_) {
415  for (auto const& expiry : smileExpiries_) {
416  auto key = std::make_pair(expiry, term);
417  if (smiles_.find(key) == smiles_.end()) {
418 
419  // search neighboured expiries for same term
420 
421  Date expiry_m = Null<Date>();
422  Date expiry_p = Null<Date>();
423  for (auto const& s : smiles_) {
424  if (s.first.second != term)
425  continue;
426  if (s.first.first >= expiry) {
427  expiry_p = s.first.first;
428  break;
429  }
430  expiry_m = s.first.first;
431  }
432 
433  // if we have a smile for the expiry and term already, there is nothing to do
434 
435  if (expiry_m == expiry || expiry_p == expiry)
436  continue;
437 
438  // otherwise build an interpolated / extrapoalted smile
439 
440  if (expiry_m == Null<Date>() && expiry_p != Null<Date>()) {
441  // expiry <= smallest expiry for that term = expiry_p
442  createSmile(expiry, term, expiry_p, expiry_m);
443  } else if (expiry_m != Null<Date>() && expiry_p != Null<Date>()) {
444  // expiry_m < expiry < expiry_p
445  createSmile(expiry, term, expiry_m, expiry_p);
446  } else if (expiry_m != Null<Date>() && expiry_p == Null<Date>()) {
447  // expiry >= largest expiry for that term = expiry_m
448  createSmile(expiry, term, expiry_p, expiry_m);
449  } else {
450  QL_FAIL("InterpolatingCreditVolCurve: internal error, expiry_m = expiry_p = null, i.e. there are "
451  "no smiles for term "
452  << term);
453  }
454  }
455  }
456  }

465 void InterpolatingCreditVolCurve::createSmile(const Date& expiry, const Period& term, const Date& expiry_m,
466  const Date& expiry_p) const {
467  Real thisAtm = atmStrike(expiry, term);
468  if (expiry_p == Null<Date>()) {
469  auto key = std::make_pair(expiry_m, term);
470  const Smile& smile = smiles_[key];
471  std::vector<Real> strikes;
472  std::vector<Real> vols;
473  for (auto const& k : strikes_[key]) {
474  strikes.push_back(strike(moneyness(k, smile.first), thisAtm));
475  }
476  for (auto const& k : strikes)
477  vols.push_back(smile.second->operator()(k));
478  auto s = strikes_.insert(std::make_pair(key, strikes));
479  auto v = vols_.insert(std::make_pair(key, vols));
480  auto tmp = boost::make_shared<FlatExtrapolation>(boost::make_shared<LinearInterpolation>(
481  s.first->second.begin(), s.first->second.end(), v.first->second.begin()));
482  tmp->enableExtrapolation();
483  smiles_[std::make_pair(expiry, term)] = std::make_pair(thisAtm, tmp);
484  } else if (expiry_m == Null<Date>()) {
485  auto key = std::make_pair(expiry_p, term);
486  const Smile& smile = smiles_[key];
487  std::vector<Real> strikes;
488  std::vector<Real> vols;
489  for (auto const& k : strikes_[key]) {
490  strikes.push_back(strike(moneyness(k, smile.first), thisAtm));
491  }
492  for (auto const& k : strikes)
493  vols.push_back(smile.second->operator()(k));
494  auto s = strikes_.insert(std::make_pair(key, strikes));
495  auto v = vols_.insert(std::make_pair(key, vols));
496  auto tmp = boost::make_shared<FlatExtrapolation>(boost::make_shared<LinearInterpolation>(
497  s.first->second.begin(), s.first->second.end(), v.first->second.begin()));
498  tmp->enableExtrapolation();
499  smiles_[std::make_pair(expiry, term)] = std::make_pair(thisAtm, tmp);
500  } else {
501  auto key_m = std::make_pair(expiry_m, term);
502  auto key_p = std::make_pair(expiry_p, term);
503  const Smile& smile_m = smiles_[key_m];
504  const Smile& smile_p = smiles_[key_p];
505  std::set<Real, CompClose> strikes_set;
506  for (auto const& k : strikes_[key_m]) {
507  strikes_set.insert(strike(moneyness(k, smile_m.first), thisAtm));
508  }
509  for (auto const& k : strikes_[key_p]) {
510  strikes_set.insert(strike(moneyness(k, smile_p.first), thisAtm));
511  }
512  std::vector<Real> strikes(strikes_set.begin(), strikes_set.end());
513  std::vector<Real> vols;
514  Real t = timeFromReference(expiry);
515  Real t_m = timeFromReference(expiry_m);
516  Real t_p = timeFromReference(expiry_m);
517  Real alpha = (t_p - t) / (t_p - t_m);
518  for (auto const& k : strikes) {
519  Real vol_m = smile_m.second->operator()(k);
520  Real vol_p = smile_p.second->operator()(k);
521  vols.push_back(std::sqrt((alpha * (vol_m * vol_m * t_m) + (1.0 - alpha) * (vol_p * vol_p * t_p)) / t));
522  }
523  auto s = strikes_.insert(std::make_pair(std::make_pair(expiry, term), strikes));
524  auto v = vols_.insert(std::make_pair(std::make_pair(expiry, term), vols));
525  auto tmp = boost::make_shared<FlatExtrapolation>(boost::make_shared<LinearInterpolation>(
526  s.first->second.begin(), s.first->second.end(), v.first->second.begin()));
527  tmp->enableExtrapolation();
528  smiles_[std::make_pair(expiry, term)] = std::make_pair(thisAtm, tmp);
529  }
530 }

559 Real SpreadedCreditVolCurve::volatility(const Date& exerciseDate, const Real underlyingLength, const Real strike,
560  const Type& targetType) const {
561  calculate();
562  Real thisAtm =
563  (strike == Null<Real>() || stickyMoneyness_) ? this->atmStrike(exerciseDate, underlyingLength) : Null<Real>();
564  Real effStrike = strike == Null<Real>() ? thisAtm : strike;
565  Real adj = stickyMoneyness_ ? baseCurve_->atmStrike(exerciseDate, underlyingLength) - thisAtm : 0.0;
566  Real base = baseCurve_->volatility(exerciseDate, underlyingLength, effStrike + adj, targetType);
567  Real spread = interpolatedSpreads_->operator()(timeFromReference(exerciseDate));
568  return base + spread;
