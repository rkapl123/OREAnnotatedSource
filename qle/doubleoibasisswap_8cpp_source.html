48void DoubleOvernightIndexedBasisSwap::initialize() {
49 legs_[0] = OvernightLeg(paySchedule_, payIndex_)
50 .withNotionals(nominals_)
51 .withSpreads(paySpread_)
52 .withTelescopicValueDates(telescopicValueDates_);
53
54 legs_[1] = OvernightLeg(recSchedule_, recIndex_)
55 .withNotionals(nominals_)
56 .withSpreads(recSpread_)
57 .withTelescopicValueDates(telescopicValueDates_);
58
59 for (Size j = 0; j < 2; ++j) {
60 for (Leg::iterator i = legs_[j].begin(); i != legs_[j].end(); ++i)
61 registerWith(*i);
62 }
63
64 payer_[0] = -1.0;
65 payer_[1] = +1.0;
66}
