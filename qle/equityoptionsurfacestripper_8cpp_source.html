48 Real PriceError::operator()(Volatility x) const {
49  vol_.setValue(x);
50  Real npv;
51  // Barone Adesi Whaley fails for very small variance, so wrap in a try/catch
52  try {
53  npv = option_.NPV();
54  } catch (...) {
55  npv = 0.0;
56  }
57  return npv - targetValue_;
58 }

84 void EquityOptionSurfaceStripper::performCalculations() const {
85  // create a set of all dates
86  std::set<Date> allExpiries;
87  std::vector<Date> callExpiries = callSurface_->expiries();
88  std::vector<Date> putExpiries = putSurface_->expiries();
89  for (auto expiry : callExpiries)
90  allExpiries.insert(expiry);
91  for (auto expiry : putExpiries)
92  allExpiries.insert(expiry);
93 
94  boost::shared_ptr<SimpleQuote> volQuote;
95  boost::shared_ptr<PricingEngine> engine;
96 
97  boost::shared_ptr<BlackVarianceSurfaceSparse> callVolSurface, putVolSurface;
98 
99  // if the surface is a price surface then we have premiums
100  bool premiumSurfaces = boost::dynamic_pointer_cast<OptionPriceSurface>(callSurface_) != NULL;
101  if (premiumSurfaces) {
102  // first also check the put surface
103  QL_REQUIRE(boost::dynamic_pointer_cast<OptionPriceSurface>(putSurface_) != NULL,
104  "Call price surface provided, but no put price surface");
105 
106  // Set up the engine for implying the vols
107  // term structures needed to get implied vol
108  volQuote = boost::make_shared<SimpleQuote>(0.1);
109  Handle<BlackVolTermStructure> volTs(boost::make_shared<BlackConstantVol>(
110  callSurface_->referenceDate(), calendar_, Handle<Quote>(volQuote), dayCounter_));
111 
112  // a black scholes process
113  boost::shared_ptr<GeneralizedBlackScholesProcess> gbsp = boost::make_shared<BlackScholesMertonProcess>(
114  eqIndex_->equitySpot(), eqIndex_->equityDividendCurve(), eqIndex_->equityForecastCurve(), volTs);
115 
116  // hard code the engines here
117  // BaroneAdesiWhaley for American options - much faster than alternatives
118  // Black for European
119  if (type_ == Exercise::American) {
120  engine = boost::make_shared<QuantExt::BaroneAdesiWhaleyApproximationEngine>(gbsp);
121  } else if (type_ == Exercise::European) {
122  engine = boost::make_shared<QuantExt::AnalyticEuropeanEngine>(gbsp);
123  } else {
124  QL_FAIL("Unsupported exercise type for option stripping");
125  }
126  } else {
127  // we have variance surfaces, explicitly cast so we can look up vol later
128  callVolSurface = boost::dynamic_pointer_cast<BlackVarianceSurfaceSparse>(callSurface_);
129  putVolSurface = boost::dynamic_pointer_cast<BlackVarianceSurfaceSparse>(putSurface_);
130  }
131 
132  vector<Real> volStrikes;
133  vector<Real> volData;
134  vector<Date> volExpiries;
135 
136  // loop over each expiry
137  for (auto exp : allExpiries) {
138  // get the forward price at time
139  Real forward = eqIndex_->fixing(exp);
140 
141  vector<Real> callStrikes, putStrikes;
142  auto itc = std::find(callExpiries.begin(), callExpiries.end(), exp);
143  if (itc != callExpiries.end()) {
144  auto pos = std::distance(callExpiries.begin(), itc);
145  callStrikes = callSurface_->strikes().at(pos);
146  }
147  auto itp = std::find(putExpiries.begin(), putExpiries.end(), exp);
148  if (itp != putExpiries.end()) {
149  auto pos = std::distance(putExpiries.begin(), itp);
150  putStrikes = putSurface_->strikes().at(pos);
151  }
152 
153  // We want a set of prices both sides of ATM forward
154  // We take calls where strike < atm, and puts where strike > atm
155 
156  bool haveCalls = false, havePuts = false;
157  // check if calls in the correct range
158  if (callStrikes.size() > 0 && callStrikes.front() < forward)
159  haveCalls = true;
160 
161  // check if puts in the correct range
162  if (putStrikes.size() > 0 && putStrikes.back() > forward)
163  havePuts = true;
164 
165  for (auto cs : callStrikes) {
166  if (!havePuts || cs < forward) {
167  volStrikes.push_back(cs);
168  volExpiries.push_back(exp);
169  if (premiumSurfaces) {
170  volData.push_back(implyVol(exp, cs, Option::Call, engine, volQuote));
171  } else {
172  volData.push_back(callVolSurface->blackVol(exp, cs));
173  }
174  }
175  }
176  for (auto ps : putStrikes) {
177  if (!haveCalls || ps > forward) {
178  volStrikes.push_back(ps);
179  volExpiries.push_back(exp);
180  if (premiumSurfaces) {
181  volData.push_back(implyVol(exp, ps, Option::Put, engine, volQuote));
182  } else {
183  volData.push_back(putVolSurface->blackVol(exp, ps));
184  }
185  }
186  }
187  }
188  volSurface_ = boost::make_shared<BlackVarianceSurfaceSparse>(
189  callSurface_->referenceDate(), calendar_, volExpiries, volStrikes, volData, dayCounter_,
190  lowerStrikeConstExtrap_, upperStrikeConstExtrap_, timeFlatExtrapolation_);
191 }

193 Real EquityOptionSurfaceStripper::implyVol(Date expiry, Real strike, Option::Type type,
194  boost::shared_ptr<PricingEngine> engine,
195  boost::shared_ptr<SimpleQuote> volQuote) const {
196 
197  // create an american option for current strike/expiry and type
198  boost::shared_ptr<StrikedTypePayoff> payoff(new PlainVanillaPayoff(type, strike));
199  boost::shared_ptr<Exercise> exercise;
200  if (type_ == Exercise::American) {
201  exercise = boost::make_shared<AmericanExercise>(expiry);
202  } else if (type_ == Exercise::European) {
203  exercise = boost::make_shared<EuropeanExercise>(expiry);
204  } else {
205  QL_FAIL("Unsupported exercise type for option stripping");
206  }
207  VanillaOption option(payoff, exercise);
208  option.setPricingEngine(engine);
209 
210  // option.setPricingEngine(engine);
211  Real targetPrice =
212  type == Option::Call ? callSurface_->getValue(expiry, strike) : putSurface_->getValue(expiry, strike);
213 
214  // calculate the implied volatility using a solver
215  Real vol;
216  try {
217  PriceError f(option, *volQuote, targetPrice);
218  Brent solver;
219  solver.setMaxEvaluations(100);
220  solver.setLowerBound(0.0001);
221  vol = solver.solve(f, 0.0001, 0.2, 0.01);
222  } catch (...) {
223  vol = 0.0;
224  }
225  return vol;
226 }
