39 //! Commodity average basis price curve.
40 /*! Class representing an outright commodity price curve created from a base price curve and a collection of basis
41  quotes that are added to or subtracted from the base curve. This class is intended to be used only for commodity

124 template <class Interpolator>
125 CommodityAverageBasisPriceCurve<Interpolator>::CommodityAverageBasisPriceCurve(
126  const QuantLib::Date& referenceDate, const std::map<QuantLib::Date, QuantLib::Handle<QuantLib::Quote> >& basisData,
127  const boost::shared_ptr<FutureExpiryCalculator>& basisFec, const boost::shared_ptr<CommodityIndex>& index,
128  const QuantLib::Handle<PriceTermStructure>& basePts, const boost::shared_ptr<FutureExpiryCalculator>& baseFec,
129  bool addBasis, const Interpolator& interpolator)
130  : PriceTermStructure(referenceDate, QuantLib::NullCalendar(), basePts->dayCounter()),
131  QuantLib::InterpolatedCurve<Interpolator>(interpolator), basisData_(basisData), basisFec_(basisFec),
132  index_(index), basePts_(basePts), baseFec_(baseFec), addBasis_(addBasis) {
133 
134  using QuantLib::Date;
135  using QuantLib::Schedule;
136  using QuantLib::io::iso_date;
137  using QuantLib::io::ordinal;
138  using std::distance;
139  using std::find;
140  using std::max;
141  using std::sort;
142  using std::vector;
143 
144  registerWith(basePts_);
145 
146  // Observe the quotes
147  for (auto it = basisData_.cbegin(); it != basisData_.cend();) {
148  if (it->first < referenceDate) {
149  // If the basis quote's date is before the reference date, remove it.
150  basisData_.erase(it++);
151  } else {
152  // Otherwise, process the basis quote
153  dates_.push_back(it->first);
154  basisTimes_.push_back(timeFromReference(it->first));
155  if (addBasis_)
156  basisValues_.push_back(it->second->value());
157  else
158  basisValues_.push_back(-it->second->value());
159  registerWith(it->second);
160  ++it;
161  }
162  }
163 
164  // Set up the interpolation to be used on the basis
165  basisInterpolation_ = interpolator.interpolate(basisTimes_.begin(), basisTimes_.end(), basisValues_.begin());
166 
167  // Initialise this curve's times with the basis pillars. We will add more pillars below.
168  this->times_ = basisTimes_;
169 
170  // Get the first basis contract expiry date strictly prior to the curve reference date.
171  Date start = basisFec_->priorExpiry(false, referenceDate);
172 
173  // Get the first basis contract expiry date on or after the max date. Here, max date is defined as the maximum of
174  // 1) last pillar date of base price curve and 2) basis curve data.
175  Date maxDate = max(basePts_->maxDate(), basisData_.rbegin()->first);
176  Date end = basisFec_->nextExpiry(true, maxDate);
177 
178  // Create the leg schedule using a vector of dates which are the successive basis contract expiry dates
179  QL_REQUIRE(start < end, "Expected that the start date, " << io::iso_date(start)
180  << ", would be strictly less than the end date, "
181  << io::iso_date(end) << ".");
182  vector<Date> expiries{ start + 1 * Days };
183  vector<Time> scheduleTimes;
184  while (start < end) {
185  start = basisFec_->nextExpiry(true, start + 1 * Days);
186  expiries.push_back(start);
187  Time t = timeFromReference(start);
188  // Only add to this->times_ if it is not already there. We can use dates_ for this check.
189  if (find(dates_.begin(), dates_.end(), start) == dates_.end()) {
190  this->times_.push_back(t);
191  dates_.push_back(start);
192  }
193  scheduleTimes.push_back(t);
194  }
195  QL_REQUIRE(start == end, "Expected that the start date, " << io::iso_date(start) << ", to equal the end date, "
196  << io::iso_date(end)
197  << ", after creating the sequence of expiry dates.");
198 
199  // Sort the times and dates vector and ensure no duplicates in the times vector.
200  sort(this->times_.begin(), this->times_.end());
201  sort(dates_.begin(), dates_.end());
202  auto it = unique(this->times_.begin(), this->times_.end(), [](double s, double t) { return close(s, t); });
203  QL_REQUIRE(it == this->times_.end(), "Unexpected duplicate time, " << *it << ", in the times vector.");
204  this->data_.resize(this->times_.size());
205 
206  // Populate the leg of cashflows.
207  baseLeg_ = CommodityIndexedAverageLeg(Schedule(expiries), index_)
208  .withFutureExpiryCalculator(baseFec_)
209  .useFuturePrice(true)
210  .withQuantities(1.0);
211  QL_REQUIRE(baseLeg_.size() == scheduleTimes.size(), "Unexpected number of averaging cashflows in the leg: "
212  << "got " << baseLeg_.size() << " but expected "
213  << scheduleTimes.size());
214 
215  // Populate the legIndexMap_
216  for (Size i = 0; i < this->times_.size(); i++) {
217  for (Size j = 0; j < scheduleTimes.size(); j++) {
218 
219  if (this->times_[i] < scheduleTimes[j] || close(this->times_[i], scheduleTimes[j])) {
220  QL_REQUIRE(legIndexMap_.find(i) == legIndexMap_.end(),
221  "Should not already have a mapping for the " << ordinal(i) << " time.");
222  legIndexMap_[i] = j;
223  break;
224  }
225 
226  if (j == scheduleTimes.size()) {
227  QL_FAIL("Could not map the " << ordinal(i) << " time, " << this->times_[i] << ", to a cashflow.");
228  }
229  }
230  }

233  QuantLib::InterpolatedCurve<Interpolator>::setupInterpolation();
234 }

240 template <class Interpolator> void CommodityAverageBasisPriceCurve<Interpolator>::performCalculations() const {
241 
242  // Update the basis interpolation object
243  Size basisIdx = 0;
244  for (const auto& kv : basisData_) {
245  basisValues_[basisIdx] = addBasis_ ? kv.second->value() : -kv.second->value();
246  basisIdx++;
247  }
248  basisInterpolation_.update();
249 
250  // Update this curve's interpolation
251  for (Size i = 0; i < this->times_.size(); i++) {
252 
253  Real baseValue = baseLeg_[legIndexMap_.at(i)]->amount();
254 
255  // Get the basis with flat extrapolation
256  Real basis = 0.0;
257  if (this->times_[i] < basisTimes_.front()) {
258  basis = basisValues_.front();
259  } else if (this->times_[i] > basisTimes_.back()) {
260  basis = basisValues_.back();
261  } else {
262  basis = basisInterpolation_(this->times_[i], true);
263  }
264 
265  // Update the outright value
266  this->data_[i] = baseValue + basis;
267  }
268  this->interpolation_.update();
269 }
