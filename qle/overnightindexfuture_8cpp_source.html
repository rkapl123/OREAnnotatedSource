49 Real OvernightIndexFuture::spotValue() const {
50  Date today = Settings::instance().evaluationDate();
51  Real prod = 1;
52  if (today > valueDate_) {
53  // can't value on a weekend inside reference period because we
54  // won't know the reset rate until start of next business day.
55  // user can supply an estimate if they really want to do this
56  today = calendar_.adjust(today, businessDayConvention_);
57  // for valuations inside the reference period, index quotes
58  // must have been populated in the history
59  const TimeSeries<Real>& history = IndexManager::instance().getHistory(overnightIndex_->name());
60  Date d1 = valueDate_;
61  while (d1 < today) {
62  Real r = history[d1];
63  QL_REQUIRE(r != Null<Real>(), "missing rate on " << d1 << " for index " << overnightIndex_->name());
64  Date d2 = calendar_.advance(d1, 1, Days);
65  prod *= 1 + r * dayCounter_.yearFraction(d1, d2);
66  d1 = d2;
67  }
68  }
69  DiscountFactor forwardDiscount = discountCurve_->discount(maturityDate_);
70  if (valueDate_ > today) {
71  forwardDiscount /= discountCurve_->discount(valueDate_);
72  }
73  prod /= forwardDiscount;
74  Real convAdj = convexityAdjustment_.empty() ? 0.0 : convexityAdjustment_->value();
75  Real R = convAdj + (prod - 1) / dayCounter_.yearFraction(valueDate_, maturityDate_);
76  underlyingSpotValue_ = 100.0 * (1.0 - R);
77  return underlyingSpotValue_;
78 }
