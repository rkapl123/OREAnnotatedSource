45  //! Constrained optimization problem
46  /*! \warning The passed Constraint instances are
47  stored by reference. The user of this class must
48  make sure that they are not destroyed before the
49  Problem instance.
50  */
51  class Problem_MT {
52  public:
53  /*! The requirement is that the cost functions can be evaluated
54  from multiple threads without data races, which can e.g. be done
55  by complete separation of the underlying data. Note that some
56  methods in this class are *not* thread safe, namely
57  reset, setCurrentValue, setFunctionValue, setGradientNormValue,
58  i.e. those can not be used from several threads in a mt optimizer */
59  Problem_MT(const std::vector<boost::shared_ptr<CostFunction>>& costFunctions,
60  Constraint& constraint,
61  const Array& initialValue = Array())
62  : costFunctions_(costFunctions), constraint_(constraint),
63  currentValue_(initialValue), functionEvaluation_(costFunctions.size(),0),
64  gradientEvaluation_(costFunctions.size(),0) {}

159  inline void Problem_MT::reset() {
160  for(Size i=0;i<costFunctions_.size();++i) {
161  functionEvaluation_.at(i) = gradientEvaluation_.at(i) = 0;
162  }
163  functionValue_ = squaredNorm_ = Null<Real>();
164  }
165 
166  inline Integer Problem_MT::functionEvaluation() const {
167  Integer tmp = 0;
168  for (Size i = 0; i < costFunctions_.size(); ++i)
169  tmp += functionEvaluation_.at(i);
170  return tmp;
171  }
172 
173  inline Integer Problem_MT::gradientEvaluation() const {
174  Integer tmp = 0;
175  for (Size i = 0; i < costFunctions_.size(); ++i)
176  tmp += gradientEvaluation_.at(i);
177  return tmp;
