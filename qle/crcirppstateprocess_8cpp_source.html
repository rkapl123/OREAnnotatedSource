37 Disposable<Array> CrCirppStateProcess::evolve(Time t0, const Array& x0, Time dt, const Array& dw) const {
38  Array retVal(size());
39  Real kappa, theta, sigma, y0;
40  kappa = model_->parametrization()->kappa(t0);
41  theta = model_->parametrization()->theta(t0);
42  sigma = model_->parametrization()->sigma(t0);
43  y0 = model_->parametrization()->y0(t0);
44 
45  const Real sdt = std::sqrt(dt);
46  switch (discretization_) {
47 
48  case BrigoAlfonsi: {
49  // see D. Brigo and F. Mercurio. Interest Rate Models: Theory and Practice, 2nd
50  // Edition. Springer, 2006.
51  // Ensures non-negative values for \sigma^2 \leq 2*\kappa*\theta
52  Real temp = (1 - kappa / 2.0 * dt);
53  Real temp2 = temp * std::sqrt(x0[0]) + sigma * sdt * dw[0] / (2.0 * temp);
54  retVal[0] = temp2 * temp2 + (kappa * theta - sigma * sigma / 4) * dt;
55  break;
56  }
57  default:
58  QL_FAIL("unknown discretization schema");
59  }
60 
61  // second element is S(0,i)
62  Real SM_ti = model_->defaultCurve()->survivalProbability(t0 + dt);
63  Real SM_ti_pre = model_->defaultCurve()->survivalProbability(t0);
64  Real Pcir_ti = model_->zeroBond(0, t0 + dt, y0);
65  Real Pcir_ti_pre = model_->zeroBond(0, t0, y0);
66  retVal[1] = x0[1] * SM_ti / SM_ti_pre * Pcir_ti_pre / Pcir_ti * exp(-x0[0] * dt);
67 
68  return retVal;
69 }
