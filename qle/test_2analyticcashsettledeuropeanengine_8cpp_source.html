95 // Check option values on or after expiry date
96 void checkOptionValues(CashSettledEuropeanOption& option, Rate r, Real exercisePrice, Real tolerance = 1e-12) {
97 
98  // Discount factor from payment and time to payment date.
99  auto yts = flatYts(r);
100  DiscountFactor df_tp = yts->discount(option.paymentDate());
101  Time tp = yts->timeFromReference(option.paymentDate());
102  BOOST_TEST_MESSAGE("Discount factor from payment is: " << fixed << setprecision(12) << df_tp);
103 
104  // Value at exercise
105  Real valueAtExpiry = (*option.payoff())(exercisePrice);
106 
107  // Check the results after manual exercise.
108  map<string, Real> cashSettledResults = results(option);
109  BOOST_REQUIRE(cashSettledResults.count("npv") == 1);
110  for (const auto& kv : cashSettledResults) {
111  BOOST_TEST_CONTEXT("With result " << kv.first) {
112  BOOST_TEST_MESSAGE("Value for " << kv.first << " with cash settlement is: " << fixed << setprecision(12)
113  << kv.second);
114  if (kv.first == "npv") {
115  BOOST_CHECK_SMALL(kv.second - df_tp * valueAtExpiry, tolerance);
116  } else if (kv.first == "rho") {
117  BOOST_CHECK_SMALL(kv.second + tp * cashSettledResults.at("npv"), tolerance);
118  } else if (kv.first == "theta") {
119  if (tp > 0.0 && !close(tp, 0.0)) {
120  BOOST_CHECK_SMALL(kv.second + std::log(df_tp) / tp * cashSettledResults.at("npv"), tolerance);
121  } else {
122  BOOST_CHECK(close(kv.second, 0.0));
123  }
124  } else if (kv.first == "thetaPerDay") {
125  if (cashSettledResults.count("theta") == 0) {
126  BOOST_ERROR("Expected results to contain a value for theta");
127  continue;
128  } else {
129  BOOST_CHECK_SMALL(kv.second - cashSettledResults.at("theta") / 365.0, tolerance);
130  }
131  } else {
132  BOOST_CHECK(close(kv.second, 0.0));
133  }
134  }
135  }
136 }

147 BOOST_DATA_TEST_CASE(testOptionBeforeExpiry, bdata::make(strikes) * bdata::make(optionTypes), strike, optionType) {
148 
149  BOOST_TEST_MESSAGE("Testing cash settled option pricing before expiry...");
150 
151  Settings::instance().evaluationDate() = Date(3, Jun, 2020);
152 
153  // Create cash settled option instrument
154  Date expiry(3, Sep, 2020);
155  Date payment(7, Sep, 2020);
156  bool automaticExercise = false;
157  CashSettledEuropeanOption option(optionType, strike, expiry, payment, automaticExercise);
158 
159  // Create engine that accounts for cash settlement.
160  Rate spot = 60.00;
161  Volatility vol = 0.30;
162  Rate r = 0.02;
163  Rate q = 0.01;
164  boost::shared_ptr<PricingEngine> engine =
165  boost::make_shared<AnalyticCashSettledEuropeanEngine>(getProcess(spot, vol, r, q));
166 
167  // Value the option accounting for cash settlement and store all results
168  option.setPricingEngine(engine);
169  map<string, Real> cashSettledResults = results(option);
170 
171  // Value the option ignoring cash settlement
172  engine = boost::make_shared<AnalyticEuropeanEngine>(getProcess(spot, vol, r, q));
173  option.setPricingEngine(engine);
174  map<string, Real> theoreticalResults = results(option);
175 
176  // Discount factor from payment to expiry.
177  auto yts = flatYts(r);
178  DiscountFactor df_te_tp = yts->discount(payment) / yts->discount(expiry);
179  BOOST_TEST_MESSAGE("Discount factor from payment to expiry is: " << fixed << setprecision(12) << df_te_tp);
180 
181  // Check the results
182  Real tolerance = 1e-12;
183  BOOST_REQUIRE_EQUAL(cashSettledResults.size(), theoreticalResults.size());
184  BOOST_REQUIRE(cashSettledResults.count("npv") == 1);
185  BOOST_REQUIRE(theoreticalResults.count("npv") == 1);
186  for (const auto& kv : cashSettledResults) {
187 
188  BOOST_TEST_CONTEXT("With result " << kv.first) {
189 
190  auto it = theoreticalResults.find(kv.first);
191  BOOST_CHECK(it != theoreticalResults.end());
192 
193  // If there is no matching result in theoreticalResults, skip to next result.
194  if (it == theoreticalResults.end())
195  continue;
196 
197  // Check the result is as expected by comparing to the results that do not account for delayed payment.
198  Real theorResult = it->second;
199  BOOST_TEST_MESSAGE("Value for " << kv.first << " with cash settlement is: " << fixed << setprecision(12)
200  << kv.second);
201  BOOST_TEST_MESSAGE("Value for " << kv.first << " ignoring cash settlement is: " << fixed << setprecision(12)
202  << theorResult);
203 
204  // Most results should be of the form cashSettledResult = DF(t_e, t_p) * theoreticalResult.
205  // There are some exceptions dealt with below.
206  if (close(theorResult, 0.0)) {
207  BOOST_CHECK(close(kv.second, 0.0));
208  } else if (kv.first == "elasticity" || kv.first == "itmCashProbability") {
209  BOOST_CHECK(close(kv.second, theorResult));
210  } else if (kv.first == "rho") {
211  Time delta_te_tp = yts->timeFromReference(payment) - yts->timeFromReference(expiry);
212  Real expRho = df_te_tp * (theorResult - delta_te_tp * theoreticalResults.at("npv"));
213  BOOST_TEST_MESSAGE("Value for expected rho is: " << fixed << setprecision(12) << expRho);
214  BOOST_CHECK(close(kv.second, expRho));
215  } else {
216  BOOST_CHECK_SMALL(kv.second / theorResult - df_te_tp, tolerance);
217  }
218  }
219  }
220 }
