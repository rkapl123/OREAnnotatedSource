18 //! Black volatility surface that implies an ATM vol based on triangulation
19 /*! This class is used when one wants to proxy a volatility like XAU/EUR using
20  * XAU/USD, EUR/USD and a correlation. It uses the cosing rule.
21  * The correlation can be implied from a vol (if you had XAU/EUR) or historically estimated.
22  * This class is just ATM, otherwise there is a degree of freedom in selecting strikes.
23  *
24  * One application of this is SIMM sensis, where the vol for XAU/EUR must be broken down
25  * into XAU/USD and EUR/USD.
26  *
27  * Other methods for building a full surface exist, but to keep things simple we just do ATM
28  *
29  * \ingroup termstructures
30  */
31 class BlackTriangulationATMVolTermStructure : public BlackVolatilityTermStructure {
32 public:
33  //! Constructor takes two BlackVolTermStructure and a correlation
34  /*! Attributes like referenceDate, settlementDays, Calendar, etc are taken from vol1
35  */
36  BlackTriangulationATMVolTermStructure(const Handle<BlackVolTermStructure>& vol1,
37  const Handle<BlackVolTermStructure>& vol2,
38  const Handle<CorrelationTermStructure>& rho)
39  : BlackVolatilityTermStructure(vol1->businessDayConvention(), vol1->dayCounter()), vol1_(vol1), vol2_(vol2),
40  rho_(rho) {
41  registerWith(vol1_);
42  registerWith(vol2_);
43  registerWith(rho_);
44  enableExtrapolation(vol1_->allowsExtrapolation() && vol2_->allowsExtrapolation());
45  }
46  //! \name TermStructure interface
47  //@{
48  const Date& referenceDate() const override { return vol1_->referenceDate(); }
49  Date maxDate() const override { return std::min(vol1_->maxDate(), vol2_->maxDate()); }
50  Natural settlementDays() const override { return vol1_->settlementDays(); }
51  Calendar calendar() const override { return vol1_->calendar(); }
52  //! \name Observer interface
53  //@{
54  void update() override { notifyObservers(); }
55  //@}
56  //! \name VolatilityTermStructure interface
57  //@{
58  Real minStrike() const override { return 0; }
59  Real maxStrike() const override { return QL_MAX_REAL; }
60  //@}
61  //! \name Visitability
62  //@{
63  virtual void accept(AcyclicVisitor&) override;
64  //@}
65 protected:
66  virtual Volatility blackVolImpl(Time t, Real) const override {
67  // get ATM vols and correlation
68  Volatility v1 = vol1_->blackVol(t, Null<Real>());
69  Volatility v2 = vol2_->blackVol(t, Null<Real>());
70  Real c = rho_->correlation(t);
71 
72  // get vol^2
73  Volatility volSquared = v1 * v1 + v2 * v2 - 2.0 * c * v1 * v2;
74  if (volSquared <= 0)
75  return 0;
76  return std::sqrt(volSquared);
77  }
78 
79 private:
80  Handle<BlackVolTermStructure> vol1_;
81  Handle<BlackVolTermStructure> vol2_;
82  Handle<CorrelationTermStructure> rho_;
83 };
