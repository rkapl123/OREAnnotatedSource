35 SimpleDeltaInterpolatedSmile::SimpleDeltaInterpolatedSmile(
36  const Real spot, const Real domDisc, const Real forDisc, const Real expiryTime, const std::vector<Real>& deltas,
37  const std::vector<Real>& putVols, const std::vector<Real>& callVols, const Real atmVol,
38  const DeltaVolQuote::DeltaType dt, const DeltaVolQuote::AtmType at,
39  const BlackVolatilitySurfaceBFRR::SmileInterpolation smileInterpolation, const Real accuracy,
40  const Size maxIterations)
41  : spot_(spot), domDisc_(domDisc), forDisc_(forDisc), expiryTime_(expiryTime), deltas_(deltas), putVols_(putVols),
42  callVols_(callVols), atmVol_(atmVol), dt_(dt), at_(at), smileInterpolation_(smileInterpolation),
43  accuracy_(accuracy), maxIterations_(maxIterations) {
44 
45  forward_ = spot_ / domDisc_ * forDisc_;
46 
47  std::vector<Real> x, y;
48 
49  /* Convert the puts, atm, calls to strikes using the given conventions and then to simple delta
50  using the given conventions. Store the simple deltas as x-values for the interpolation and
51  the log of the vols as y-values */
52 
53  for (Size i = 0; i < deltas_.size(); ++i) {
54  try {
55  BlackDeltaCalculator c(Option::Put, dt_, spot_, domDisc_, forDisc_, putVols_[i] * std::sqrt(expiryTime_));
56  x.push_back(simpleDeltaFromStrike(c.strikeFromDelta(-deltas[i])));
57  } catch (const std::exception& e) {
58  QL_FAIL("SimpleDeltaInterpolatedSmile: strikeFromDelta("
59  << -deltas[i] << ") could not be computed for spot=" << spot_
60  << ", forward=" << spot_ / domDisc_ * forDisc_ << " (domRate=" << -std::log(domDisc_) / expiryTime_
61  << ", forRate=" << -std::log(forDisc) / expiryTime_ << "), putVol=" << putVols_[i]
62  << ", expiry=" << expiryTime_ << ": " << e.what());
63  }
64  y.push_back(transformVol(putVols_[i]));
65  }
66 
67  try {
68  BlackDeltaCalculator c(Option::Call, dt_, spot_, domDisc_, forDisc_, atmVol * std::sqrt(expiryTime_));
69  x.push_back(simpleDeltaFromStrike(c.atmStrike(at_)));
70  } catch (const std::exception& e) {
71  QL_FAIL("SimpleDeltaIinterpolatedSmile: atmStrike could not be computed for spot="
72  << spot_ << ", forward=" << spot_ / domDisc_ * forDisc_
73  << " (domRate=" << -std::log(domDisc_) / expiryTime_ << ", forRate=" << -std::log(forDisc) / expiryTime_
74  << "), atmVol=" << atmVol << ", expiry=" << expiryTime_ << ": " << e.what());
75  }
76  y.push_back(transformVol(atmVol_));
77 
78  for (Size i = deltas_.size(); i > 0; --i) {
79  try {
80  BlackDeltaCalculator c(Option::Call, dt_, spot_, domDisc_, forDisc_,
81  callVols_[i - 1] * std::sqrt(expiryTime_));
82  x.push_back(simpleDeltaFromStrike(c.strikeFromDelta(deltas[i - 1])));
83  y.push_back(transformVol(callVols_[i - 1]));
84  } catch (const std::exception& e) {
85  QL_FAIL("SimpleDeltaInterpolatedSmile: strikeFromDelta("
86  << deltas[i - 1] << ") could not be computed for spot=" << spot_
87  << ", forward=" << spot_ / domDisc_ * forDisc_ << " (domRate=" << -std::log(domDisc_) / expiryTime_
88  << ", forRate=" << -std::log(forDisc) / expiryTime_ << "), callVol=" << callVols_[i - 1]
89  << ", expiry=" << expiryTime_ << ": " << e.what());
90  }
91  }
92 
93  /* sort the strikes */
94 
95  std::vector<Size> perm(x.size());
96  std::iota(perm.begin(), perm.end(), 0);
97  std::sort(perm.begin(), perm.end(), [&x](Size a, Size b) { return x[a] < x[b]; });
98  for (Size i = 0; i < perm.size(); ++i) {
99  x_.push_back(x[perm[i]]);
100  y_.push_back(y[perm[i]]);
101  }
102 
103  /* check the strikes are not (numerically) identical */
104 
105  for (Size i = 0; i < x_.size() - 1; ++i) {
106  QL_REQUIRE(!close_enough(x_[i], x_[i + 1]), "SmileDeltaInterpolatedSmile: interpolation points x["
107  << i << "] = x[" << (i + 1) << "] = " << x_[i]
108  << " are numerically identical.");
109  }
110 
111  /* Create the interpolation object */
112 
113  if (smileInterpolation_ == BlackVolatilitySurfaceBFRR::SmileInterpolation::Linear) {
114  interpolation_ = boost::make_shared<LinearInterpolation>(x_.begin(), x_.end(), y_.begin());
115  } else if (smileInterpolation_ == BlackVolatilitySurfaceBFRR::SmileInterpolation::Cubic) {
116  interpolation_ = boost::make_shared<CubicInterpolation>(
117  x_.begin(), x_.end(), y_.begin(), CubicInterpolation::Spline, false, CubicInterpolation::SecondDerivative,
118  0.0, CubicInterpolation::SecondDerivative, 0.0);
119  } else {
120  QL_FAIL("invalid interpolation, this is unexpected");
121  }
122 
123  interpolation_->enableExtrapolation();
124 }

126 Real SimpleDeltaInterpolatedSmile::strikeFromDelta(const Option::Type type, const Real delta,
127  const DeltaVolQuote::DeltaType dt) {
128  Real result = forward_, lastResult;
129  Size iterations = 0;
130  do {
131  Real stddev = std::sqrt(expiryTime_) * volatility(result);
132  try {
133  BlackDeltaCalculator c(type, dt, spot_, domDisc_, forDisc_, stddev);
134  lastResult = result;
135  result = c.strikeFromDelta((type == Option::Call ? 1.0 : -1.0) * delta);
136  } catch (const std::exception& e) {
137  QL_FAIL("SimpleDeltaInterpolatedSmile::strikeFromDelta("
138  << (type == Option::Call ? 1.0 : -1.0) * delta << ") could not be computed for spot=" << spot_
139  << ", forward=" << spot_ / domDisc_ * forDisc_ << " (domRate=" << -std::log(domDisc_) / expiryTime_
140  << ", forRate=" << -std::log(forDisc_) / expiryTime_ << "), vol=" << stddev / std::sqrt(expiryTime_)
141  << ", expiry=" << expiryTime_ << ": " << e.what());
142  }
143  } while (std::abs((result - lastResult) / lastResult) > accuracy_ && ++iterations < maxIterations_);
144  QL_REQUIRE(iterations < maxIterations_,
145  "SmileDeltaInterpolatedSmile::strikeFromDelta("
146  << (type == Option::Call ? 1.0 : -1.0) * delta << "): max iterations (" << maxIterations_
147  << "), no solution found for accuracy " << accuracy_ << ", last iterations: " << lastResult << "/"
148  << result << ", spot=" << spot_ << ", forward=" << spot_ / domDisc_ * forDisc_
149  << " (domRate=" << -std::log(domDisc_) / expiryTime_
150  << ", forRate=" << -std::log(forDisc_) / expiryTime_ << "), expiry=" << expiryTime_);
151  return result;
152 }

156  Size iterations = 0;
157  do {

166  << -std::log(domDisc_) / expiryTime_ << ", forRate=" << -std::log(forDisc_) / expiryTime_
167  << "), vol=" << stddev / std::sqrt(expiryTime_) << ", expiry=" << expiryTime_ << ": " << e.what());
168  }
169  } while (std::abs((result - lastResult) / lastResult) > accuracy_ && ++iterations < maxIterations_);
170  QL_REQUIRE(iterations < maxIterations_,

198 Real SimpleDeltaInterpolatedSmile::simpleDeltaFromStrike(const Real strike) const {
199  if (close_enough(strike, 0.0))
200  return 0.0;
201  CumulativeNormalDistribution Phi;
202  return Phi(std::log(strike / forward_) / (atmVol_ * std::sqrt(expiryTime_)));

205 boost::shared_ptr<SimpleDeltaInterpolatedSmile>
206 createSmile(const Real spot, const Real domDisc, const Real forDisc, const Real expiryTime,
207  const std::vector<Real>& deltas, const std::vector<Real>& bfQuotes, const std::vector<Real>& rrQuotes,
208  const Real atmVol, const DeltaVolQuote::DeltaType dt, const DeltaVolQuote::AtmType at,
209  const Option::Type riskReversalInFavorOf, const bool butterflyIsBrokerStyle,
210  const BlackVolatilitySurfaceBFRR::SmileInterpolation smileInterpolation) {
211 
212  Real phirr = riskReversalInFavorOf == Option::Call ? 1.0 : -1.0;
213 
214  if (!butterflyIsBrokerStyle) {
215 
216  // butterfly is not broker style: we can directly compute the call/put vols ...
217 
218  std::vector<Real> vol_p, vol_c;
219 
220  for (Size i = 0; i < deltas.size(); ++i) {
221  QL_REQUIRE(atmVol + bfQuotes[i] - 0.5 * std::abs(rrQuotes[i]) > 0.0,
222  "createSmile: atmVol ("
223  << atmVol << ") + bf (" << bfQuotes[i] << ") - rr (" << rrQuotes[i]
224  << ") must be positive when creating smile from smile bf quotes, tte=" << expiryTime);
225  vol_p.push_back(atmVol + bfQuotes[i] - 0.5 * phirr * rrQuotes[i]);
226  vol_c.push_back(atmVol + bfQuotes[i] + 0.5 * phirr * rrQuotes[i]);
227  }
228 
229  // ... and set up the interpolated smile
230 
231  return boost::make_shared<SimpleDeltaInterpolatedSmile>(spot, domDisc, forDisc, expiryTime, deltas, vol_p,
232  vol_c, atmVol, dt, at, smileInterpolation);
233 
234  } else {
235 
236  Real forward = spot / domDisc * forDisc;
237 
238  /* handle broker style butterflys: first determine the strikes and the (non-discounted) premiums
239  of the broker butterflies */
240 
241  std::vector<Real> kb_c, kb_p, vb;
242 
243  for (Size i = 0; i < deltas.size(); ++i) {
244  Real stddevb = (atmVol + bfQuotes[i]) * std::sqrt(expiryTime);
245  QL_REQUIRE(stddevb > 0.0,
246  "createSmile: atmVol ("
247  << atmVol << ") + bf (" << bfQuotes[i]
248  << ") must be positive when creating smile from broker bf quotes, tte=" << expiryTime);
249  BlackDeltaCalculator cp(Option::Type::Put, dt, spot, domDisc, forDisc, stddevb);
250  BlackDeltaCalculator cc(Option::Type::Call, dt, spot, domDisc, forDisc, stddevb);
251  kb_p.push_back(cp.strikeFromDelta(-deltas[i]));
252  kb_c.push_back(cc.strikeFromDelta(deltas[i]));
253  vb.push_back(blackFormula(Option::Put, kb_p.back(), forward, stddevb) +
254  blackFormula(Option::Call, kb_c.back(), forward, stddevb));
255  }
256 
257  /* set initial guess for smile butterfly vol := broker butterfly vol
258  we optimise in z = log( atmVol + bf - 0.5 * abs(rr) */
259 
260  Array guess(deltas.size());
261  for (Size i = 0; i < deltas.size(); ++i) {
262  guess[i] = std::log(std::max(0.0001, bfQuotes[i] + atmVol - 0.5 * rrQuotes[i]));
263  }

267  struct TargetFunction : public QuantLib::CostFunction {
268  TargetFunction(Real atmVol, Real phirr, Real spot, Real domDisc, Real forDisc, Real forward,
269  Real expiryTime, DeltaVolQuote::DeltaType dt, DeltaVolQuote::AtmType at,
270  const std::vector<Real>& rrQuotes, const std::vector<Real>& deltas,
271  const std::vector<Real>& kb_p, const std::vector<Real>& kb_c, const std::vector<Real>& vb,
272  BlackVolatilitySurfaceBFRR::SmileInterpolation smileInterpolation)
273  : atmVol(atmVol), phirr(phirr), spot(spot), domDisc(domDisc), forDisc(forDisc), forward(forward),
274  expiryTime(expiryTime), dt(dt), at(at), rrQuotes(rrQuotes), deltas(deltas), kb_p(kb_p), kb_c(kb_c),
275  vb(vb), smileInterpolation(smileInterpolation) {}
276 
277  Real atmVol, phirr, spot, domDisc, forDisc, forward, expiryTime;
278  DeltaVolQuote::DeltaType dt;
279  DeltaVolQuote::AtmType at;
280  const std::vector<Real>&rrQuotes, deltas, kb_p, kb_c, vb;
281  BlackVolatilitySurfaceBFRR::SmileInterpolation smileInterpolation;
282 
283  mutable Real bestValue = QL_MAX_REAL;
284  mutable boost::shared_ptr<SimpleDeltaInterpolatedSmile> bestSmile;
285 
286  Array values(const Array& x) const override {
287 
288  Array rrTmp(rrQuotes.begin(), rrQuotes.end());
289  Array smileBfVol = Exp(x) - atmVol + 0.5 * Abs(rrTmp);
290 
291  // compute the call/put vols ....
292 
293  std::vector<Real> vol_c, vol_p;
294 
295  for (Size i = 0; i < deltas.size(); ++i) {
296  vol_p.push_back(atmVol + smileBfVol[i] - 0.5 * phirr * rrQuotes[i]);
297  vol_c.push_back(atmVol + smileBfVol[i] + 0.5 * phirr * rrQuotes[i]);
298  QL_REQUIRE(vol_p.back() > 0.0, " createSmile: internal error: put vol = "
299  << vol_p.back() << " during broker bf fitting");
300  QL_REQUIRE(vol_c.back() > 0.0, " createSmile: internal error: call vol = "
301  << vol_c.back() << " during broker bf fitting");
302  }
303 
304  // ... set up the interpolated smile ...
305 
306  auto tmpSmile = boost::make_shared<SimpleDeltaInterpolatedSmile>(
307  spot, domDisc, forDisc, expiryTime, deltas, vol_p, vol_c, atmVol, dt, at, smileInterpolation);
308 
309  // ... and price the market butterfly on the constructed smile
310 
311  std::vector<Real> vs;
312  for (Size i = 0; i < deltas.size(); ++i) {
313  vs.push_back(blackFormula(Option::Put, kb_p[i], forward,
314  tmpSmile->volatility(kb_p[i]) * std::sqrt(expiryTime)) +
315  blackFormula(Option::Call, kb_c[i], forward,
316  tmpSmile->volatility(kb_c[i]) * std::sqrt(expiryTime)));
317  }
318 
319  // now set the target function to the relative difference of smile vs. market price
320 
321  Array result(deltas.size());
322  for (Size i = 0; i < deltas.size(); ++i) {
323  result[i] = (vs[i] - vb[i]) / vb[i];
324  }
325 
326  Real value =
327  std::accumulate(result.begin(), result.end(), 0.0, [](Real acc, Real x) { return acc + x * x; });
328 
329  if (value < bestValue) {
330  bestValue = value;
331  bestSmile = tmpSmile;
332  }
333 
334  return result;
335  }
336  }; // TargetFunction declaration;
337 
338  TargetFunction targetFunction{atmVol, phirr, spot, domDisc, forDisc, forward, expiryTime, dt,
339  at, rrQuotes, deltas, kb_p, kb_c, vb, smileInterpolation};
340  NoConstraint noConstraint;
341  LevenbergMarquardt lm;
342  EndCriteria endCriteria(100, 10, 1E-8, 1E-8, 1E-8);
343  Problem problem(targetFunction, noConstraint, guess);
344  lm.minimize(problem, endCriteria);
345 
346  QL_REQUIRE(problem.functionValue() < 0.01, "createSmile at expiry "
347  << expiryTime << " failed: target function value ("
348  << problem.functionValue() << ") not close to zero");
349 
350  return targetFunction.bestSmile;
351  }
352 }

458 Volatility BlackVolatilitySurfaceBFRR::blackVolImpl(Time t, Real strike) const {
459 
460  calculate();
461 
462  /* minimum supported time is 1D, i.e. if t is smaller, we return the vol at 1D */
463 
464  t = std::max(t, 1.0 / 365.0);
465 
466  t = t <= expiryTimes_.back() ? t : expiryTimes_.back();
467 
468  /* if we have cached the interpolated smile at t, we use that */
469 
470  auto s = cachedInterpolatedSmiles_.find(t);
471  if (s != cachedInterpolatedSmiles_.end()) {
472  return s->second->volatility(strike);
473  }
474 
475  /* find the indices ip and im such that t_im <= t < t_ip, im will be null if t < first expiry,
476  ip will be null if t >= last expiry */
477 
478  Size index_p = std::upper_bound(expiryTimes_.begin(), expiryTimes_.end(), t) - expiryTimes_.begin();
479  Size index_m = index_p == 0 ? Null<Size>() : index_p - 1;
480  if (index_p == expiryTimes_.size())
481  index_p = Null<Size>();
482 
483  /* the smiles at the indices might have errors, then go to the next ones without errors */
484 
485  while (index_m != Null<Size>() && index_m > 0 && smileHasError_[index_m])
486  --index_m;
487 
488  while (index_p != Null<Size>() && index_p < expiryTimes_.size() - 1 && smileHasError_[index_p])
489  ++index_p;
490 
491  if (index_m != Null<Size>() && smileHasError_[index_m])
492  index_m = Null<Size>();
493 
494  if (index_p != Null<Size>() && smileHasError_[index_p])
495  index_p = Null<Size>();
496 
497  QL_REQUIRE(index_m != Null<Size>() || index_p != Null<Size>(),
498  "BlackVolatilitySurfaceBFRR::blackVolImpl(" << t << "," << strike
499  << "): no valid smiles, check the market data input.");
500 
501  /* build the smiles on the indices, if we do not have them yet */
502 
503  if (index_m != Null<Size>() && smiles_[index_m] == nullptr) {
504  DeltaVolQuote::AtmType at;
505  DeltaVolQuote::DeltaType dt;
506  if (expiryTimes_[index_m] < switchTime_ && !close_enough(switchTime_, expiryTimes_[index_m])) {
507  at = at_;
508  dt = dt_;
509  } else {
510  at = ltat_;
511  dt = ltdt_;
512  }
513  try {
514  smiles_[index_m] = detail::createSmile(
515  spot_->value(), domesticTS_->discount(settlementDates_[index_m]) / settlDomDisc_,
516  foreignTS_->discount(settlementDates_[index_m]) / settlForDisc_, expiryTimes_[index_m], deltas_,
517  bfQuotes_[index_m], rrQuotes_[index_m], atmQuotes_[index_m], dt, at, riskReversalInFavorOf_,
518  butterflyIsBrokerStyle_, smileInterpolation_);
519  } catch (const std::exception& e) {
520  smileHasError_[index_m] = true;
521  smileErrorMessage_[index_m] = e.what();
522  return blackVolImpl(t, strike);
523  }
524  }
525 
526  if (index_p != Null<Size>() && smiles_[index_p] == nullptr) {
527  DeltaVolQuote::AtmType at;
528  DeltaVolQuote::DeltaType dt;
529  if (expiryTimes_[index_p] < switchTime_ && !close_enough(switchTime_, expiryTimes_[index_p])) {
530  at = at_;
531  dt = dt_;
532  } else {
533  at = ltat_;
534  dt = ltdt_;
535  }
536  try {
537  smiles_[index_p] = detail::createSmile(
538  spot_->value(), domesticTS_->discount(settlementDates_[index_p]) / settlDomDisc_,
539  foreignTS_->discount(settlementDates_[index_p]) / settlForDisc_, expiryTimes_[index_p], deltas_,
540  bfQuotes_[index_p], rrQuotes_[index_p], atmQuotes_[index_p], dt, at, riskReversalInFavorOf_,
541  butterflyIsBrokerStyle_, smileInterpolation_);
542  } catch (const std::exception& e) {
543  smileHasError_[index_p] = true;
544  smileErrorMessage_[index_p] = e.what();
545  return blackVolImpl(t, strike);
546  }
547  }
548 
549  /* Choose a consistent set of smile conventions for all maturities, we follow Clark, 4.2.3 and set
550  delta type = forward delta, with pa if the original (short term) delta convention has pa.
551  The atm type ist set to delta neutral. */
552 
553  DeltaVolQuote::DeltaType dt_c =
554  dt_ == (DeltaVolQuote::Spot || dt_ == DeltaVolQuote::Fwd) ? DeltaVolQuote::Fwd : DeltaVolQuote::PaFwd;
555  DeltaVolQuote::AtmType at_c = DeltaVolQuote::AtmDeltaNeutral;
556 
557  /* find the vols on both smiles for the artificial smile conventions */
558 
559  Real atmVol_m = 0.0, atmVol_p = 0.0;
560  std::vector<Real> putVols_m, callVols_m, putVols_p, callVols_p;
561 
562  if (index_m != Null<Size>()) {
563  try {
564  atmVol_m = smiles_[index_m]->volatility(smiles_[index_m]->atmStrike(dt_c, at_c));
565  for (auto const& d : deltas_) {
566  putVols_m.push_back(
567  smiles_[index_m]->volatility(smiles_[index_m]->strikeFromDelta(Option::Put, d, dt_c)));
568  callVols_m.push_back(
569  smiles_[index_m]->volatility(smiles_[index_m]->strikeFromDelta(Option::Call, d, dt_c)));
570  }
571  } catch (const std::exception& e) {
572  smileHasError_[index_m] = true;
573  smileErrorMessage_[index_m] = e.what();
574  return blackVolImpl(t, strike);
575  }
576  }
577 
578  if (index_p != Null<Size>()) {
579  try {
580  atmVol_p = smiles_[index_p]->volatility(smiles_[index_p]->atmStrike(dt_c, at_c));
581  for (auto const& d : deltas_) {
582  putVols_p.push_back(
583  smiles_[index_p]->volatility(smiles_[index_p]->strikeFromDelta(Option::Put, d, dt_c)));
584  callVols_p.push_back(
585  smiles_[index_p]->volatility(smiles_[index_p]->strikeFromDelta(Option::Call, d, dt_c)));
586  }
587  } catch (const std::exception& e) {
588  smileHasError_[index_p] = true;
589  smileErrorMessage_[index_p] = e.what();
590  return blackVolImpl(t, strike);
591  }
592  }
593 
594  /* find the interpolated vols */
595 
596  Real atmVol_i;
597  std::vector<Real> putVols_i, callVols_i;
598 
599  if (index_p == Null<Size>()) {
600  // extrapolate beyond last expiry
601  atmVol_i = atmVol_m;
602  putVols_i = putVols_m;
603  callVols_i = callVols_m;
604  QL_REQUIRE(atmVol_i > 0.0, "BlackVolatilitySurfaceBFRR: negative front-extrapolated atm vol " << atmVol_i);
605  for (Size i = 0; i < deltas_.size(); ++i) {
606  QL_REQUIRE(putVols_i[i] > 0.0,
607  "BlackVolatilitySurfaceBFRR: negative front-extrapolated put vol " << putVols_i[i]);
608  QL_REQUIRE(callVols_i[i] > 0.0,
609  "BlackVolatilitySurfaceBFRR: negative front-extrapolated call vol " << callVols_i[i]);
610  }
611  } else if (index_m == Null<Size>()) {
612  // extrapolate before first expiry
613  atmVol_i = atmVol_p;
614  putVols_i = putVols_p;
615  callVols_i = callVols_p;
616  QL_REQUIRE(atmVol_i > 0.0, "BlackVolatilitySurfaceBFRR: negative back-extrapolated atm vol " << atmVol_i);
617  for (Size i = 0; i < deltas_.size(); ++i) {
618  QL_REQUIRE(putVols_i[i] > 0.0,
619  "BlackVolatilitySurfaceBFRR: negative back-extrapolated put vol " << putVols_i[i]);
620  QL_REQUIRE(callVols_i[i] > 0.0,
621  "BlackVolatilitySurfaceBFRR: negative back-extrapolated call vol " << callVols_i[i]);
622  }
623  } else {
624  // interpolate between two expiries
625  Real a = (t - expiryTimes_[index_m]) / (expiryTimes_[index_p] - expiryTimes_[index_m]);
626  atmVol_i = (1.0 - a) * atmVol_m + a * atmVol_p;
627  QL_REQUIRE(atmVol_i > 0.0, "BlackVolatilitySurfaceBFRR: negative atm vol "
628  << atmVol_i << " = " << (1.0 - a) << " * " << atmVol_m << " + " << a << " * "
629  << atmVol_p);
630  for (Size i = 0; i < deltas_.size(); ++i) {
631  putVols_i.push_back((1.0 - a) * putVols_m[i] + a * putVols_p[i]);
632  callVols_i.push_back((1.0 - a) * callVols_m[i] + a * callVols_p[i]);
633  QL_REQUIRE(putVols_i.back() > 0.0, "BlackVolatilitySurfaceBFRR: negative put vol for delta="
634  << deltas_[i] << ", " << putVols_i.back() << " = " << (1.0 - a)
635  << " * " << putVols_m[i] << " + " << a << " * " << putVols_p[i]);
636  QL_REQUIRE(callVols_i.back() > 0.0, "BlackVolatilitySurfaceBFRR: negative call vol for delta="
637  << deltas_[i] << ", " << callVols_i.back() << " = " << (1.0 - a)
638  << " * " << callVols_m[i] << " + " << a << " * " << callVols_p[i]);
639  }
640  }
641 
642  /* build a new smile using the interpolated vols and artificial conventions
643  (querying the dom / for TS at t + settlLag_ is not entirely correct, because
644  - of possibly different dcs in the curves and the vol ts and
645  - because settLag_ is the time from today
646  to today's settlement date,
647  but the best we can realistically do in this context, because we don't have a date
648  corresponding to t) */
649 
650  boost::shared_ptr<detail::SimpleDeltaInterpolatedSmile> smile;
651 
652  try {
653  smile = boost::make_shared<detail::SimpleDeltaInterpolatedSmile>(
654  spot_->value(), domesticTS_->discount(t + settlLag_) / settlDomDisc_,
655  foreignTS_->discount(t + settlLag_) / settlForDisc_, t, deltas_, putVols_i, callVols_i, atmVol_i, dt_c,
656  at_c, smileInterpolation_);
657  } catch (const std::exception& e) {
658  // the interpolated smile failed to build => mark the "m" smile as a failure if available and retry, otherwise
659  // market the "p" smile as a failure and retry
660  Size failureIndex = index_m != Null<Size>() ? index_m : index_p;
661  smileHasError_[failureIndex] = true;
662  smileErrorMessage_[failureIndex] = e.what();
663  return blackVolImpl(t, strike);
664  }
665 
666  /* store the new smile in the cache */
667 
668  cachedInterpolatedSmiles_[t] = smile;
669 
670  /* return the volatility from the smile we just created */
671 
672  return smile->volatility(strike);
673 }
