35SimpleDeltaInterpolatedSmile::SimpleDeltaInterpolatedSmile(
36 const Real spot, const Real domDisc, const Real forDisc, const Real expiryTime, const std::vector<Real>& deltas,
37 const std::vector<Real>& putVols, const std::vector<Real>& callVols, const Real atmVol,
38 const DeltaVolQuote::DeltaType dt, const DeltaVolQuote::AtmType at,
39 const BlackVolatilitySurfaceBFRR::SmileInterpolation smileInterpolation, const Real accuracy,
40 const Size maxIterations)
41 : spot_(spot), domDisc_(domDisc), forDisc_(forDisc), expiryTime_(expiryTime), deltas_(deltas), putVols_(putVols),
42 callVols_(callVols), atmVol_(atmVol), dt_(dt), at_(at), smileInterpolation_(smileInterpolation),
43 accuracy_(accuracy), maxIterations_(maxIterations) {
44
45 forward_ = spot_ / domDisc_ * forDisc_;
46
47 std::vector<Real> x, y;
48
49 /* Convert the puts, atm, calls to strikes using the given conventions and then to simple delta
50 using the given conventions. Store the simple deltas as x-values for the interpolation and
51 the log of the vols as y-values */
52
53 for (Size i = 0; i < deltas_.size(); ++i) {
54 try {
55 BlackDeltaCalculator c(Option::Put, dt_, spot_, domDisc_, forDisc_, putVols_[i] * std::sqrt(expiryTime_));
56 x.push_back(simpleDeltaFromStrike(c.strikeFromDelta(-deltas[i])));
57 y.push_back(transformVol(putVols_[i]));
58 } catch (const std::exception& e) {
59 QL_FAIL("SimpleDeltaInterpolatedSmile: strikeFromDelta("
60 << -deltas[i] << ") could not be computed for spot=" << spot_
61 << ", forward=" << spot_ / domDisc_ * forDisc_ << " (domRate=" << -std::log(domDisc_) / expiryTime_
62 << ", forRate=" << -std::log(forDisc) / expiryTime_ << "), putVol=" << putVols_[i]
63 << ", expiry=" << expiryTime_ << ": " << e.what());
64 }
65 }
66
67 try {
68 BlackDeltaCalculator c(Option::Call, dt_, spot_, domDisc_, forDisc_, atmVol * std::sqrt(expiryTime_));
69 x.push_back(simpleDeltaFromStrike(c.atmStrike(at_)));
70 y.push_back(transformVol(atmVol_));
71 } catch (const std::exception& e) {
72 QL_FAIL("SimpleDeltaIinterpolatedSmile: atmStrike could not be computed for spot="
73 << spot_ << ", forward=" << spot_ / domDisc_ * forDisc_
74 << " (domRate=" << -std::log(domDisc_) / expiryTime_ << ", forRate=" << -std::log(forDisc) / expiryTime_
75 << "), atmVol=" << atmVol << ", expiry=" << expiryTime_ << ": " << e.what());
76 }
77
78 for (Size i = deltas_.size(); i > 0; --i) {
79 try {
80 BlackDeltaCalculator c(Option::Call, dt_, spot_, domDisc_, forDisc_,
81 callVols_[i - 1] * std::sqrt(expiryTime_));
82 x.push_back(simpleDeltaFromStrike(c.strikeFromDelta(deltas[i - 1])));
83 y.push_back(transformVol(callVols_[i - 1]));
84 } catch (const std::exception& e) {
85 QL_FAIL("SimpleDeltaInterpolatedSmile: strikeFromDelta("
86 << deltas[i - 1] << ") could not be computed for spot=" << spot_
87 << ", forward=" << spot_ / domDisc_ * forDisc_ << " (domRate=" << -std::log(domDisc_) / expiryTime_
88 << ", forRate=" << -std::log(forDisc) / expiryTime_ << "), callVol=" << callVols_[i - 1]
89 << ", expiry=" << expiryTime_ << ": " << e.what());
90 }
91 }
92
93 /* sort the strikes */
94
95 std::vector<Size> perm(x.size());
96 std::iota(perm.begin(), perm.end(), 0);
97 std::sort(perm.begin(), perm.end(), [&x](Size a, Size b) { return x[a] < x[b]; });
98 for (Size i = 0; i < perm.size(); ++i) {
99 x_.push_back(x[perm[i]]);
100 y_.push_back(y[perm[i]]);
101 }
102
103 /* check the strikes are not (numerically) identical */
104
105 for (Size i = 0; i < x_.size() - 1; ++i) {
106 QL_REQUIRE(!close_enough(x_[i], x_[i + 1]), "SmileDeltaInterpolatedSmile: interpolation points x["
107 << i << "] = x[" << (i + 1) << "] = " << x_[i]
108 << " are numerically identical.");
109 }
110
111 /* Create the interpolation object */
112
113 if (smileInterpolation_ == BlackVolatilitySurfaceBFRR::SmileInterpolation::Linear) {
114 interpolation_ = boost::make_shared<LinearInterpolation>(x_.begin(), x_.end(), y_.begin());
115 } else if (smileInterpolation_ == BlackVolatilitySurfaceBFRR::SmileInterpolation::Cubic) {
116 interpolation_ = boost::make_shared<CubicInterpolation>(
117 x_.begin(), x_.end(), y_.begin(), CubicInterpolation::Spline, false, CubicInterpolation::SecondDerivative,
118 0.0, CubicInterpolation::SecondDerivative, 0.0);
119 } else {
120 QL_FAIL("invalid interpolation, this is unexpected");
121 }
122
123 interpolation_->enableExtrapolation();
124}

126Real SimpleDeltaInterpolatedSmile::strikeFromDelta(const Option::Type type, const Real delta,
127 const DeltaVolQuote::DeltaType dt) {
128 Real result = forward_, lastResult;
129 Size iterations = 0;
130 do {
131 Real stddev = std::sqrt(expiryTime_) * volatility(result);
132 try {
133 BlackDeltaCalculator c(type, dt, spot_, domDisc_, forDisc_, stddev);
134 lastResult = result;
135 result = c.strikeFromDelta((type == Option::Call ? 1.0 : -1.0) * delta);
136 } catch (const std::exception& e) {
137 QL_FAIL("SimpleDeltaInterpolatedSmile::strikeFromDelta("
138 << (type == Option::Call ? 1.0 : -1.0) * delta << ") could not be computed for spot=" << spot_
139 << ", forward=" << spot_ / domDisc_ * forDisc_ << " (domRate=" << -std::log(domDisc_) / expiryTime_
140 << ", forRate=" << -std::log(forDisc_) / expiryTime_ << "), vol=" << stddev / std::sqrt(expiryTime_)
141 << ", expiry=" << expiryTime_ << ": " << e.what());
142 }
143 } while (std::abs((result - lastResult) / lastResult) > accuracy_ && ++iterations < maxIterations_);
144 QL_REQUIRE(iterations < maxIterations_,
145 "SmileDeltaInterpolatedSmile::strikeFromDelta("
146 << (type == Option::Call ? 1.0 : -1.0) * delta << "): max iterations (" << maxIterations_
147 << "), no solution found for accuracy " << accuracy_ << ", last iterations: " << lastResult << "/"
148 << result << ", spot=" << spot_ << ", forward=" << spot_ / domDisc_ * forDisc_
149 << " (domRate=" << -std::log(domDisc_) / expiryTime_
150 << ", forRate=" << -std::log(forDisc_) / expiryTime_ << "), expiry=" << expiryTime_);
151 return result;
152}

154Real SimpleDeltaInterpolatedSmile::atmStrike(const DeltaVolQuote::DeltaType dt, const DeltaVolQuote::AtmType at) {
155 Real result = forward_, lastResult;
156 Size iterations = 0;
157 do {
158 Real stddev = std::sqrt(expiryTime_) * volatility(result);
159 try {
160 BlackDeltaCalculator c(Option::Call, dt, spot_, domDisc_, forDisc_, stddev);
161 lastResult = result;
162 result = c.atmStrike(at);
163 } catch (const std::exception& e) {
164 QL_FAIL("SimpleDeltaInterpolatedSmile::atmStrike() could not be computed for spot="
165 << spot_ << ", forward=" << spot_ / domDisc_ * forDisc_ << " (domRate="
166 << -std::log(domDisc_) / expiryTime_ << ", forRate=" << -std::log(forDisc_) / expiryTime_
167 << "), vol=" << stddev / std::sqrt(expiryTime_) << ", expiry=" << expiryTime_ << ": " << e.what());
168 }
169 } while (std::abs((result - lastResult) / lastResult) > accuracy_ && ++iterations < maxIterations_);
170 QL_REQUIRE(iterations < maxIterations_,
171 "SmileDeltaInterpolatedSmile::atmStrike(): max iterations ("
172 << maxIterations_ << "), no solution found for accuracy " << accuracy_
173 << ", last iterations: " << lastResult << "/" << result << ", spot=" << spot_
174 << ", forward=" << spot_ / domDisc_ * forDisc_ << " (domRate=" << -std::log(domDisc_) / expiryTime_
175 << ", forRate=" << -std::log(forDisc_) / expiryTime_ << "), expiry=" << expiryTime_);
176 return result;
177}

183Real SimpleDeltaInterpolatedSmile::volatility(const Real strike) {
184 Real tmp = untransformVol((*interpolation_)(simpleDeltaFromStrike(strike)));
185 if (!std::isfinite(tmp)) {
186 std::ostringstream os;
187 for (Size i = 0; i < x_.size(); ++i) {
188 os << "(" << x_[i] << "," << y_[i] << ")";
189 }
190 QL_FAIL("SimpleDeltaInterpolatedSmile::volatility() non-finite result ("
191 << tmp << ") for strike " << strike << ", simple delta is " << simpleDeltaFromStrike(strike)
192 << ", interpolated value is " << (*interpolation_)(simpleDeltaFromStrike(strike))
193 << ", interpolation data point are " << os.str());
194 }
195 return tmp;
196}

198Real SimpleDeltaInterpolatedSmile::simpleDeltaFromStrike(const Real strike) const {
199 if (close_enough(strike, 0.0))
200 return 0.0;
201 CumulativeNormalDistribution Phi;
202 return Phi(std::log(strike / forward_) / (atmVol_ * std::sqrt(expiryTime_)));
203}

205boost::shared_ptr<SimpleDeltaInterpolatedSmile>
206createSmile(const Real spot, const Real domDisc, const Real forDisc, const Real expiryTime,
207 const std::vector<Real>& deltas, const std::vector<Real>& bfQuotes, const std::vector<Real>& rrQuotes,
208 const Real atmVol, const DeltaVolQuote::DeltaType dt, const DeltaVolQuote::AtmType at,
209 const Option::Type riskReversalInFavorOf, const bool butterflyIsBrokerStyle,
210 const BlackVolatilitySurfaceBFRR::SmileInterpolation smileInterpolation) {
211
212 Real phirr = riskReversalInFavorOf == Option::Call ? 1.0 : -1.0;
213
214 if (!butterflyIsBrokerStyle) {
215
216 // butterfly is not broker style: we can directly compute the call/put vols ...
217
218 std::vector<Real> vol_p, vol_c;
219
220 for (Size i = 0; i < deltas.size(); ++i) {
221 QL_REQUIRE(atmVol + bfQuotes[i] - 0.5 * std::abs(rrQuotes[i]) > 0.0,
222 "createSmile: atmVol ("
223 << atmVol << ") + bf (" << bfQuotes[i] << ") - rr (" << rrQuotes[i]
224 << ") must be positive when creating smile from smile bf quotes, tte=" << expiryTime);
225 vol_p.push_back(atmVol + bfQuotes[i] - 0.5 * phirr * rrQuotes[i]);
226 vol_c.push_back(atmVol + bfQuotes[i] + 0.5 * phirr * rrQuotes[i]);
227 }
228
229 // ... and set up the interpolated smile
230
231 return boost::make_shared<SimpleDeltaInterpolatedSmile>(spot, domDisc, forDisc, expiryTime, deltas, vol_p,
232 vol_c, atmVol, dt, at, smileInterpolation);
233
234 } else {
235
236 Real forward = spot / domDisc * forDisc;
237
238 /* handle broker style butterflys: first determine the strikes and the (non-discounted) premiums
239 of the broker butterflies */
240
241 std::vector<Real> kb_c, kb_p, vb;
242
243 for (Size i = 0; i < deltas.size(); ++i) {
244 Real stddevb = (atmVol + bfQuotes[i]) * std::sqrt(expiryTime);
245 QL_REQUIRE(stddevb > 0.0,
246 "createSmile: atmVol ("
247 << atmVol << ") + bf (" << bfQuotes[i]
248 << ") must be positive when creating smile from broker bf quotes, tte=" << expiryTime);
249 BlackDeltaCalculator cp(Option::Type::Put, dt, spot, domDisc, forDisc, stddevb);
250 BlackDeltaCalculator cc(Option::Type::Call, dt, spot, domDisc, forDisc, stddevb);
251 kb_p.push_back(cp.strikeFromDelta(-deltas[i]));
252 kb_c.push_back(cc.strikeFromDelta(deltas[i]));
253 vb.push_back(blackFormula(Option::Put, kb_p.back(), forward, stddevb) +
254 blackFormula(Option::Call, kb_c.back(), forward, stddevb));
255 }
256
257 /* set initial guess for smile butterfly vol := broker butterfly vol
258 we optimise in z = log( bf - 0.5 * abs(rr) + atmVol ) */
259
260 Array guess(deltas.size());
261 for (Size i = 0; i < deltas.size(); ++i) {
262 guess[i] = std::log(std::max(0.0001, bfQuotes[i] - 0.5 * std::abs(rrQuotes[i]) + atmVol));
263 }
264
265 /* define the target function to match up the butterfly market values and smile values */
266
267 struct TargetFunction : public QuantLib::CostFunction {
268 TargetFunction(Real atmVol, Real phirr, Real spot, Real domDisc, Real forDisc, Real forward,
269 Real expiryTime, DeltaVolQuote::DeltaType dt, DeltaVolQuote::AtmType at,
270 const std::vector<Real>& rrQuotes, const std::vector<Real>& deltas,
271 const std::vector<Real>& kb_p, const std::vector<Real>& kb_c, const std::vector<Real>& vb,
272 BlackVolatilitySurfaceBFRR::SmileInterpolation smileInterpolation)
273 : atmVol(atmVol), phirr(phirr), spot(spot), domDisc(domDisc), forDisc(forDisc), forward(forward),
274 expiryTime(expiryTime), dt(dt), at(at), rrQuotes(rrQuotes), deltas(deltas), kb_p(kb_p), kb_c(kb_c),
275 vb(vb), smileInterpolation(smileInterpolation) {}
276
277 Real atmVol, phirr, spot, domDisc, forDisc, forward, expiryTime;
278 DeltaVolQuote::DeltaType dt;
279 DeltaVolQuote::AtmType at;
280 const std::vector<Real>&rrQuotes, deltas, kb_p, kb_c, vb;
281 BlackVolatilitySurfaceBFRR::SmileInterpolation smileInterpolation;
282
283 mutable Real bestValue = QL_MAX_REAL;
284 mutable boost::shared_ptr<SimpleDeltaInterpolatedSmile> bestSmile;
285
286 Array values(const Array& x) const override {
287
288 constexpr Real large_error = 1E6;
289
290 Array smileBfVol(x.size());
291 for (Size i = 0; i < x.size(); ++i)
292 smileBfVol[i] = std::exp(x[i]) + 0.5 * std::abs(rrQuotes[i]) - atmVol;
293
294 // compute the call/put vols ....
295
296 std::vector<Real> vol_c, vol_p;
297
298 for (Size i = 0; i < deltas.size(); ++i) {
299 vol_p.push_back(atmVol + smileBfVol[i] - 0.5 * phirr * rrQuotes[i]);
300 vol_c.push_back(atmVol + smileBfVol[i] + 0.5 * phirr * rrQuotes[i]);
301 QL_REQUIRE(vol_p.back() > 0.0, " createSmile: internal error: put vol = "
302 << vol_p.back() << " during broker bf fitting");
303 QL_REQUIRE(vol_c.back() > 0.0, " createSmile: internal error: call vol = "
304 << vol_c.back() << " during broker bf fitting");
305 }
306
307 // ... set up the interpolated smile ...
308
309 boost::shared_ptr<SimpleDeltaInterpolatedSmile> tmpSmile;
310 try {
311 tmpSmile = boost::make_shared<SimpleDeltaInterpolatedSmile>(
312 spot, domDisc, forDisc, expiryTime, deltas, vol_p, vol_c, atmVol, dt, at, smileInterpolation);
313 } catch (...) {
314 // if we run into a problem we return max error and continue with the optimization
315 return Array(deltas.size(), large_error);
316 }
317
318 // ... and price the market butterfly on the constructed smile
319
320 std::vector<Real> vs;
321 for (Size i = 0; i < deltas.size(); ++i) {
322 Real pvol, cvol;
323 try {
324 pvol = tmpSmile->volatility(kb_p[i]);
325 cvol = tmpSmile->volatility(kb_c[i]);
326 } catch (...) {
327 // as above, if there is a problem, return max error
328 return Array(deltas.size(), large_error);
329 }
330 vs.push_back(blackFormula(Option::Put, kb_p[i], forward, pvol * std::sqrt(expiryTime)) +
331 blackFormula(Option::Call, kb_c[i], forward, cvol * std::sqrt(expiryTime)));
332 }
333
334 // now set the target function to the relative difference of smile vs. market price
335
336 Array result(deltas.size());
337 for (Size i = 0; i < deltas.size(); ++i) {
338 result[i] = (vs[i] - vb[i]) / vb[i];
339 if (!std::isfinite(result[i]))
340 result[i] = large_error;
341 }
342
343 Real value = std::sqrt(std::accumulate(result.begin(), result.end(), 0.0,
344 [](Real acc, Real x) { return acc + x * x; })) /
345 result.size();
346
347 if (value < bestValue) {
348 bestValue = value;
349 bestSmile = tmpSmile;
350 }
351
352 return result;
353 }
354 }; // TargetFunction declaration;
355
356 TargetFunction targetFunction{atmVol, phirr, spot, domDisc, forDisc, forward, expiryTime, dt,
357 at, rrQuotes, deltas, kb_p, kb_c, vb, smileInterpolation};
358 NoConstraint noConstraint;
359 LevenbergMarquardt lm;
360 EndCriteria endCriteria(100, 10, 1E-8, 1E-8, 1E-8);
361 Problem problem(targetFunction, noConstraint, guess);
362 lm.minimize(problem, endCriteria);
363
364 QL_REQUIRE(targetFunction.bestValue < 0.01, "createSmile at expiry "
365 << expiryTime << " failed: target function value ("
366 << problem.functionValue() << ") not close to zero");
367
368 return targetFunction.bestSmile;
369 }
370}

407 << atmQuotes_.size() << ") mismatch with expiry dates ("
408 << dates_.size() << ")");

410 QL_REQUIRE(q.size() == deltas_.size(), "BlackVolatilitySurfaceBFRR: bfQuotes inner vector ("

412 }
413 for (auto const& q : rrQuotes) {
414 QL_REQUIRE(q.size() == deltas_.size(), "BlackVolatilitySurfaceBFRR: bfQuotes inner vector ("
415 << q.size() << ") mismatch with deltas (" << deltas_.size() << ")");
416 }
417
418 // register with observables
419
420 registerWith(spot_);
421 registerWith(domesticTS_);
422 registerWith(foreignTS_);
423}

435void BlackVolatilitySurfaceBFRR::performCalculations() const {
436
437 // calculate switch time
438
439 switchTime_ = switchTenor_ == 0 * Days ? QL_MAX_REAL : timeFromReference(optionDateFromTenor(switchTenor_));
440
441 // calculate times associated to expiry dates
442
443 expiryTimes_.clear();
444 settlementDates_.clear();
445 for (auto const& d : dates_) {
446 expiryTimes_.push_back(timeFromReference(d));
447 settlementDates_.push_back(spotCalendar_.advance(d, spotDays_ * Days));
448 }
449
450 // resize cache for smiles on expiry dates
451
452 smiles_.resize(expiryTimes_.size());
453 smileHasError_.resize(expiryTimes_.size());
454 smileErrorMessage_.resize(expiryTimes_.size());
455
456 // calculate discount factors for spot settlement date and the settlement lag
457
458 Date settlDate = spotCalendar_.advance(referenceDate(), spotDays_ * Days);
459 settlDomDisc_ = domesticTS_->discount(settlDate);
460 settlForDisc_ = foreignTS_->discount(settlDate);
461 settlLag_ = timeFromReference(settlDate);
462
463 // clear caches
464
465 clearCaches();
466
467 // reset deltas to use
468
469 currentDeltas_ = deltas_;
470}

484Volatility BlackVolatilitySurfaceBFRR::blackVolImpl(Time t, Real strike) const {
485
486 calculate();
487
488 /* minimum supported time is 1D, i.e. if t is smaller, we return the vol at 1D */
489
490 t = std::max(t, 1.0 / 365.0);
491
492 t = t <= expiryTimes_.back() ? t : expiryTimes_.back();
493
494 /* if we have cached the interpolated smile at t, we use that */
495
496 auto s = cachedInterpolatedSmiles_.find(t);
497 if (s != cachedInterpolatedSmiles_.end()) {
498 return s->second->volatility(strike);
499 }
500
501 /* find the indices ip and im such that t_im <= t < t_ip, im will be null if t < first expiry,
502 ip will be null if t >= last expiry */
503
504 Size index_p = std::upper_bound(expiryTimes_.begin(), expiryTimes_.end(), t) - expiryTimes_.begin();
505 Size index_m = index_p == 0 ? Null<Size>() : index_p - 1;
506 if (index_p == expiryTimes_.size())
507 index_p = Null<Size>();
508
509 /* the smiles at the indices might have errors, then go to the next ones without errors */
510
511 while (index_m != Null<Size>() && index_m > 0 && smileHasError_[index_m])
512 --index_m;
513
514 while (index_p != Null<Size>() && index_p < expiryTimes_.size() - 1 && smileHasError_[index_p])
515 ++index_p;
516
517 if (index_m != Null<Size>() && smileHasError_[index_m])
518 index_m = Null<Size>();
519
520 if (index_p != Null<Size>() && smileHasError_[index_p])
521 index_p = Null<Size>();
522
523 if (index_m == Null<Size>() && index_p == Null<Size>()) {
524
525 /* no valid smiles, try to remove the smallest delta until there is only one delta left, then throw an error */
526
527 if (currentDeltas_.size() <= 1) {
528 QL_FAIL("BlackVolatilitySurfaceBFRR::blackVolImpl(" << t << "," << strike
529 << "): no valid smiles, check the market data input.");
530 }
531
532 currentDeltas_.erase(currentDeltas_.begin());
533 clearCaches();
534 return blackVolImpl(t, strike);
535 }
536
537 /* build the smiles on the indices, if we do not have them yet */
538
539 if (index_m != Null<Size>() && smiles_[index_m] == nullptr) {
540 DeltaVolQuote::AtmType at;
541 DeltaVolQuote::DeltaType dt;
542 if (expiryTimes_[index_m] < switchTime_ && !close_enough(switchTime_, expiryTimes_[index_m])) {
543 at = at_;
544 dt = dt_;
545 } else {
546 at = ltat_;
547 dt = ltdt_;
548 }
549 try {
550 smiles_[index_m] = detail::createSmile(
551 spot_->value(), domesticTS_->discount(settlementDates_[index_m]) / settlDomDisc_,
552 foreignTS_->discount(settlementDates_[index_m]) / settlForDisc_, expiryTimes_[index_m], currentDeltas_,
553 bfQuotes_[index_m], rrQuotes_[index_m], atmQuotes_[index_m], dt, at, riskReversalInFavorOf_,
554 butterflyIsBrokerStyle_, smileInterpolation_);
555 } catch (const std::exception& e) {
556 smileHasError_[index_m] = true;
557 smileErrorMessage_[index_m] = e.what();
558 return blackVolImpl(t, strike);
559 }
560 }
561
562 if (index_p != Null<Size>() && smiles_[index_p] == nullptr) {
563 DeltaVolQuote::AtmType at;
564 DeltaVolQuote::DeltaType dt;
565 if (expiryTimes_[index_p] < switchTime_ && !close_enough(switchTime_, expiryTimes_[index_p])) {
566 at = at_;
567 dt = dt_;
568 } else {
569 at = ltat_;
570 dt = ltdt_;
571 }
572 try {
573 smiles_[index_p] = detail::createSmile(
574 spot_->value(), domesticTS_->discount(settlementDates_[index_p]) / settlDomDisc_,
575 foreignTS_->discount(settlementDates_[index_p]) / settlForDisc_, expiryTimes_[index_p], currentDeltas_,
576 bfQuotes_[index_p], rrQuotes_[index_p], atmQuotes_[index_p], dt, at, riskReversalInFavorOf_,
577 butterflyIsBrokerStyle_, smileInterpolation_);
578 } catch (const std::exception& e) {
579 smileHasError_[index_p] = true;
580 smileErrorMessage_[index_p] = e.what();
581 return blackVolImpl(t, strike);
582 }
583 }
584
585 /* Choose a consistent set of smile conventions for all maturities, we follow Clark, 4.2.3 and set
586 delta type = forward delta, with pa if the original (short term) delta convention has pa.
587 The atm type ist set to delta neutral. */
588
589 DeltaVolQuote::DeltaType dt_c =
590 dt_ == (DeltaVolQuote::Spot || dt_ == DeltaVolQuote::Fwd) ? DeltaVolQuote::Fwd : DeltaVolQuote::PaFwd;
591 DeltaVolQuote::AtmType at_c = DeltaVolQuote::AtmDeltaNeutral;
592
593 /* find the vols on both smiles for the artificial smile conventions */
594
595 Real atmVol_m = 0.0, atmVol_p = 0.0;
596 std::vector<Real> putVols_m, callVols_m, putVols_p, callVols_p;
597
598 if (index_m != Null<Size>()) {
599 try {
600 atmVol_m = smiles_[index_m]->volatility(smiles_[index_m]->atmStrike(dt_c, at_c));
601 for (auto const& d : currentDeltas_) {
602 putVols_m.push_back(
603 smiles_[index_m]->volatility(smiles_[index_m]->strikeFromDelta(Option::Put, d, dt_c)));
604 callVols_m.push_back(
605 smiles_[index_m]->volatility(smiles_[index_m]->strikeFromDelta(Option::Call, d, dt_c)));
606 }
607 } catch (const std::exception& e) {
608 smileHasError_[index_m] = true;
609 smileErrorMessage_[index_m] = e.what();
610 return blackVolImpl(t, strike);
611 }
612 }
613
614 if (index_p != Null<Size>()) {
615 try {
616 atmVol_p = smiles_[index_p]->volatility(smiles_[index_p]->atmStrike(dt_c, at_c));
617 for (auto const& d : currentDeltas_) {
618 putVols_p.push_back(
619 smiles_[index_p]->volatility(smiles_[index_p]->strikeFromDelta(Option::Put, d, dt_c)));
620 callVols_p.push_back(
621 smiles_[index_p]->volatility(smiles_[index_p]->strikeFromDelta(Option::Call, d, dt_c)));
622 }
623 } catch (const std::exception& e) {
624 smileHasError_[index_p] = true;
625 smileErrorMessage_[index_p] = e.what();
626 return blackVolImpl(t, strike);
627 }
628 }
629
630 /* find the interpolated vols */
631
632 Real atmVol_i;
633 std::vector<Real> putVols_i, callVols_i;
634
635 if (index_p == Null<Size>()) {
636 // extrapolate beyond last expiry
637 atmVol_i = atmVol_m;
638 putVols_i = putVols_m;
639 callVols_i = callVols_m;
640 QL_REQUIRE(atmVol_i > 0.0, "BlackVolatilitySurfaceBFRR: negative front-extrapolated atm vol " << atmVol_i);
641 for (Size i = 0; i < currentDeltas_.size(); ++i) {
642 QL_REQUIRE(putVols_i[i] > 0.0,
643 "BlackVolatilitySurfaceBFRR: negative front-extrapolated put vol " << putVols_i[i]);
644 QL_REQUIRE(callVols_i[i] > 0.0,
645 "BlackVolatilitySurfaceBFRR: negative front-extrapolated call vol " << callVols_i[i]);
646 }
647 } else if (index_m == Null<Size>()) {
648 // extrapolate before first expiry
649 atmVol_i = atmVol_p;
650 putVols_i = putVols_p;
651 callVols_i = callVols_p;
652 QL_REQUIRE(atmVol_i > 0.0, "BlackVolatilitySurfaceBFRR: negative back-extrapolated atm vol " << atmVol_i);
653 for (Size i = 0; i < currentDeltas_.size(); ++i) {
654 QL_REQUIRE(putVols_i[i] > 0.0,
655 "BlackVolatilitySurfaceBFRR: negative back-extrapolated put vol " << putVols_i[i]);
656 QL_REQUIRE(callVols_i[i] > 0.0,
657 "BlackVolatilitySurfaceBFRR: negative back-extrapolated call vol " << callVols_i[i]);
658 }
659 } else {
660 // interpolate between two expiries
661 Real a = (t - expiryTimes_[index_m]) / (expiryTimes_[index_p] - expiryTimes_[index_m]);
662 atmVol_i = (1.0 - a) * atmVol_m + a * atmVol_p;
663 QL_REQUIRE(atmVol_i > 0.0, "BlackVolatilitySurfaceBFRR: negative atm vol "
664 << atmVol_i << " = " << (1.0 - a) << " * " << atmVol_m << " + " << a << " * "
665 << atmVol_p);
666 for (Size i = 0; i < currentDeltas_.size(); ++i) {
667 putVols_i.push_back((1.0 - a) * putVols_m[i] + a * putVols_p[i]);
668 callVols_i.push_back((1.0 - a) * callVols_m[i] + a * callVols_p[i]);
669 QL_REQUIRE(putVols_i.back() > 0.0, "BlackVolatilitySurfaceBFRR: negative put vol for delta="
670 << currentDeltas_[i] << ", " << putVols_i.back() << " = "
671 << (1.0 - a) << " * " << putVols_m[i] << " + " << a << " * "
672 << putVols_p[i]);
673 QL_REQUIRE(callVols_i.back() > 0.0, "BlackVolatilitySurfaceBFRR: negative call vol for delta="
674 << currentDeltas_[i] << ", " << callVols_i.back() << " = "
675 << (1.0 - a) << " * " << callVols_m[i] << " + " << a << " * "
676 << callVols_p[i]);
677 }
678 }
679
680 /* build a new smile using the interpolated vols and artificial conventions
681 (querying the dom / for TS at t + settlLag_ is not entirely correct, because
682 - of possibly different dcs in the curves and the vol ts and
683 - because settLag_ is the time from today
684 to today's settlement date,
685 but the best we can realistically do in this context, because we don't have a date
686 corresponding to t) */
687
688 boost::shared_ptr<detail::SimpleDeltaInterpolatedSmile> smile;
689
690 try {
691 smile = boost::make_shared<detail::SimpleDeltaInterpolatedSmile>(
692 spot_->value(), domesticTS_->discount(t + settlLag_) / settlDomDisc_,
693 foreignTS_->discount(t + settlLag_) / settlForDisc_, t, currentDeltas_, putVols_i, callVols_i, atmVol_i,
694 dt_c, at_c, smileInterpolation_);
695 } catch (const std::exception& e) {
696 // the interpolated smile failed to build => mark the "m" smile as a failure if available and retry, otherwise
697 // market the "p" smile as a failure and retry
698 Size failureIndex = index_m != Null<Size>() ? index_m : index_p;
699 smileHasError_[failureIndex] = true;
700 smileErrorMessage_[failureIndex] = e.what();
701 return blackVolImpl(t, strike);
702 }
703
704 /* store the new smile in the cache */
705
706 cachedInterpolatedSmiles_[t] = smile;
707
708 /* return the volatility from the smile we just created */
709
710 return smile->volatility(strike);
711}
