182 CmbLeg::operator Leg() const {
183  Leg leg;
184  for (Size i = 0; i < schedule_.size() - 1; i++) {
185  Date paymentDate = paymentCalendar_.adjust(schedule_[i + 1], paymentAdjustment_);
186  boost::shared_ptr<CmbCoupon> coupon
187  = boost::make_shared<CmbCoupon>(paymentDate, notionals_[i], schedule_[i], schedule_[i + 1],
188  fixingDays_[i], bondIndices_[i], gearings_[i], spreads_[i], Date(), Date(),
189  paymentDayCounter_, inArrears_);
190  auto pricer = boost::make_shared<CmbCouponPricer>();
191  coupon->setPricer(pricer);
192  leg.push_back(coupon);
193  }
194  return leg;
195 }
