78 Real AverageOISRateHelper::impliedQuote() const {
79 
80  QL_REQUIRE(termStructure_ != 0, "term structure not set");
81  averageOIS_->deepUpdate();
82 
83  // Calculate the fair fixed rate after accounting for the
84  // spread in the spread quote. Recall, the spread quote was
85  // intentionally not added to instrument averageOIS_.
86  static const Spread basisPoint = 1.0e-4;
87  Real onLegNPV = averageOIS_->overnightLegNPV();
88  Spread onSpread = onSpread_.empty() ? 0.0 : onSpread_->value();
89  Real spreadNPV = averageOIS_->overnightLegBPS() * onSpread / basisPoint;
90  Real onLegNPVwithSpread = onLegNPV + spreadNPV;
91  Real result = -onLegNPVwithSpread / (averageOIS_->fixedLegBPS() / basisPoint);
92  return result;
93 }
