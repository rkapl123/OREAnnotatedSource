21 /*! Commodity APO Engine base class
22  Correlation is parametrized as \f$\rho(s, t) = \exp(-\beta * \abs(s - t))\f$
23  where \f$s\f$ and \f$t\f$ are times to futures expiry.
24  */
25 class CommodityAveragePriceOptionBaseEngine : public CommodityAveragePriceOption::engine {
26 public:
27  CommodityAveragePriceOptionBaseEngine(const QuantLib::Handle<QuantLib::YieldTermStructure>& discountCurve,
28  const QuantLib::Handle<QuantLib::BlackVolTermStructure>& vol,
29  Real beta = 0.0);
30 
31 protected:
32  /*! Calculate accrued portion of the average commodity price if any. There will be an accrued portion of the
33  average price if any of the <em>Pricing Dates</em> are on or before the valuation date.
34 
35  The return value contains the accrued average price and the number of <em>Pricing Dates</em> that were on or
36  before the valution date. If there has been no accrual, the return value is (0.0, 0).
37  */
38  std::pair<QuantLib::Real, QuantLib::Size> calculateAccrued() const;
39 
40  //! Return the correlation between two future expiry dates \p ed_1 and \p ed_2
41  QuantLib::Real rho(const QuantLib::Date& ed_1, const QuantLib::Date& ed_2) const;
42 
43  /*! In certain cases, the APO value is not model dependent. This method returns \c true if the APO value is model
44  dependent. If the APO value is not model dependent, this method returns \c false and populates the results
45  with the model independent value.
46  */
47  bool isModelDependent(const std::pair<QuantLib::Real, QuantLib::Size>& accrued) const;
48 
49  QuantLib::Handle<QuantLib::YieldTermStructure> discountCurve_;
50  QuantLib::Handle<QuantLib::BlackVolTermStructure> volStructure_;
51  QuantLib::Real beta_;
52 };

90  /*! Prepare data for APO calculation. The \p outVolatilities parameter will be populated with separate future
91  contract volatilities taking into account the \p strike level. The number of elements of \p outVolatilities
92  gives the number, N, of future contracts involved in the non-accrued portion of the APO. The matrix
93  \p outSqrtCorr is populated with the square root of the correlation matrix between the future contracts.
94  The \p outPrices vector will be populated with the current future price values. The \p futureIndex is
95  populated with the index of the future to be used on each timestep in the simulation.
96  */
97  void setupFuture(std::vector<QuantLib::Real>& outVolatilities, QuantLib::Matrix& outSqrtCorr,

101  /*! Return the \f$n\f$ timesteps from today, \f$t_0\f$, up to \f$t_n\f$ where \f$n > 0\f$. Note that each
102  \f$t_i\f$ corresponds to a pricing date \f$d_i\f$ that is after today. The method returns the vector of time
103  deltas \f$t_i - t_{i-1}\f$ for \f$i=1,\ldots,n\f$ and populates the vector \p outDates with the dates
104  \f$d_0, d_1,...,d_n\f$. Note that the size of \p outDates is one larger than the size of the return vector.
105  */
