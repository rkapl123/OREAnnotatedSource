31BRLCdiRateHelper::BRLCdiRateHelper(const Period& swapTenor, const Handle<Quote>& fixedRate,
32 const boost::shared_ptr<BRLCdi>& brlCdiIndex,
33 const Handle<YieldTermStructure>& discountingCurve, bool telescopicValueDates)
34 : RelativeDateRateHelper(fixedRate), swapTenor_(swapTenor), brlCdiIndex_(brlCdiIndex),
35 telescopicValueDates_(telescopicValueDates), discountHandle_(discountingCurve) {
36
37 bool onIndexHasCurve = !brlCdiIndex_->forwardingTermStructure().empty();
38 bool haveDiscountCurve = !discountHandle_.empty();
39 QL_REQUIRE(!(onIndexHasCurve && haveDiscountCurve), "Have both curves nothing to solve for.");
40
41 if (!onIndexHasCurve) {
42 boost::shared_ptr<IborIndex> clonedIborIndex(brlCdiIndex_->clone(termStructureHandle_));
43 brlCdiIndex_ = boost::dynamic_pointer_cast<BRLCdi>(clonedIborIndex);
44 brlCdiIndex_->unregisterWith(termStructureHandle_);
45 }
46
47 registerWith(brlCdiIndex_);
48 registerWith(discountHandle_);
49 initializeDates();
50}

52void BRLCdiRateHelper::initializeDates() {
53
54 // Use the overnight index's calendar for all business day adjustments
55 Calendar calendar = brlCdiIndex_->fixingCalendar();
56
57 // Adjust reference date to next good business day if necessary
58 Date referenceDate = Settings::instance().evaluationDate();
59 referenceDate = calendar.adjust(referenceDate);
60
61 // Determine the start date
62 Date startDate = calendar.advance(referenceDate, 2 * Days);
63 startDate = calendar.adjust(startDate, Following);
64
65 // Determine the end date
66 Date endDate = startDate + swapTenor_;
67
68 // Create the BRL CDI swap
69 swap_ = boost::make_shared<BRLCdiSwap>(OvernightIndexedSwap::Payer, 1.0, startDate, endDate, 0.01, brlCdiIndex_,
70 0.0, telescopicValueDates_);
71
72 // Set the pricing engine
73 swap_->setPricingEngine(boost::make_shared<DiscountingSwapEngine>(discountRelinkableHandle_));
74
75 // Update earliest and latest dates
76 earliestDate_ = swap_->startDate();
77 latestDate_ = swap_->maturityDate();
78}

80void BRLCdiRateHelper::setTermStructure(YieldTermStructure* t) {
81
82 bool observer = false;
83 boost::shared_ptr<YieldTermStructure> temp(t, no_deletion);
84 termStructureHandle_.linkTo(temp, observer);
85
86 if (discountHandle_.empty())
87 discountRelinkableHandle_.linkTo(temp, observer);
88 else
89 discountRelinkableHandle_.linkTo(*discountHandle_, observer);
90
91 RelativeDateRateHelper::setTermStructure(t);
92}

94Real BRLCdiRateHelper::impliedQuote() const {
95 QL_REQUIRE(termStructure_ != 0, "BRLCdiRateHelper's term structure not set");
96 swap_->deepUpdate();
97 return swap_->fairRate();
