38void CommoditySpreadOptionAnalyticalEngine::calculate() const {
39 // Populate some additional results that don't change
40 auto& mp = results_.additionalResults;
41 QL_REQUIRE(arguments_.exercise->type() == QuantLib::Exercise::European, "Only European Spread Option supported");
42 QL_REQUIRE(arguments_.longAssetFlow && arguments_.shortAssetFlow, "flows can not be null");
43
44 Date today = Settings::instance().evaluationDate();
45
46 Date exerciseDate = arguments_.exercise->lastDate();
47
48 Date paymentDate = arguments_.paymentDate;
49 if (paymentDate == Date())
50 paymentDate = std::max(arguments_.longAssetFlow->date(), arguments_.shortAssetFlow->date());
51
52 QL_REQUIRE(paymentDate >= exerciseDate, "Payment date needs to be on or after exercise date");
53
54 double df = discountCurve_->discount(paymentDate);
55
56 Time tte = discountCurve_->timeFromReference(exerciseDate);
57
58 auto parameterFlow1 =
59 derivePricingParameterFromFlow(arguments_.longAssetFlow, *volTSLongAsset_, arguments_.longAssetFxIndex);
60
61 auto parameterFlow2 =
62 derivePricingParameterFromFlow(arguments_.shortAssetFlow, *volTSShortAsset_, arguments_.shortAssetFxIndex);
63
64 double F1 = parameterFlow1.atm;
65 double F2 = parameterFlow2.atm;
66 double sigma1 = parameterFlow1.sigma;
67 double sigma2 = parameterFlow2.sigma;
68 double obsTime1 = parameterFlow1.tn;
69 double obsTime2 = parameterFlow2.tn;
70 double accruals1 = parameterFlow1.accruals;
71 double accruals2 = parameterFlow2.accruals;
72
73 double sigma = 0;
74 double stdDev = 0;
75 double Y = 0;
76 double Z = 0;
77 double w1 = arguments_.longAssetFlow->gearing();
78 double w2 = arguments_.shortAssetFlow->gearing();
79 // Adjust strike for past fixings
80 double effectiveStrike = arguments_.effectiveStrike - w1 * accruals1 + w2 * accruals2;
81
82 if (exerciseDate <= today && paymentDate >= today) {
83 results_.value = 0;
84 } else if (exerciseDate <= today && paymentDate < today) {
85 // if observation time is before expiry, continue the process with zero vol and zero drift from pricing date to
86 // expiry
87 double omega = arguments_.type == Option::Call ? 1 : -1;
88
89 results_.value =
90 df * arguments_.quantity * omega * std::max(w1 * F1 - w2 * F2 - arguments_.effectiveStrike, 0.0);
91
92 } else if (effectiveStrike + F2 * w2 < 0) {
93 // Effective strike can be become negative if accrueds large enough
94 if (arguments_.type == Option::Call) {
95 results_.value = df * arguments_.quantity * std::max(w1 * F1 - w2 * F2 - effectiveStrike, 0.0);
96 } else {
97 results_.value = 0.0;
98 }
99
100 } else {
101 sigma1 = sigma1 * std::min(1.0, std::sqrt(obsTime1 / tte));
102 sigma2 = sigma2 * std::min(1.0, std::sqrt(obsTime2 / tte));
103
104 // KirkFormula
105 Y = (F2 * w2 + effectiveStrike);
106 Z = w1 * F1 / Y;
107 double sigmaTilde = sigma2 * F2 * w2 / Y;
108
109 sigma = std::sqrt(std::pow(sigma1, 2.0) + std::pow(sigmaTilde, 2.0) - 2 * sigma1 * sigmaTilde * rho());
110
111 stdDev = sigma * sqrt(tte);
112
113 results_.value = arguments_.quantity * Y * blackFormula(arguments_.type, 1, Z, stdDev, df);
114 }
115
116 // Calendar spread adjustment if observation period is before the exercise date
117 mp["F1"] = F1;
118 mp["accruals1"] = accruals1;
119 mp["sigma1"] = sigma1;
120 mp["obsTime1"] = obsTime1;
121 mp["F2"] = F2;
122 mp["accruals2"] = accruals2;
123 mp["sigma2"] = sigma2;
124 mp["obsTime2"] = obsTime2;
125 mp["tte"] = tte;
126 mp["sigma"] = sigma;
127 mp["stdDev"] = stdDev;
128 mp["Y"] = Y;
129 mp["Z"] = Z;
130 mp["npv"] = results_.value;
131}

134CommoditySpreadOptionAnalyticalEngine::derivePricingParameterFromFlow(const ext::shared_ptr<CommodityCashFlow>& flow,
135 const ext::shared_ptr<BlackVolTermStructure>& vol,
136 const ext::shared_ptr<FxIndex>& fxIndex) const {
137 PricingParameter res;
138 if (auto cf = ext::dynamic_pointer_cast<CommodityIndexedCashFlow>(flow)) {
139 res.accruals = 0.0;
140 res.tn = vol->timeFromReference(cf->pricingDate());
141 double fxSpot = 1.0;
142 if (fxIndex) {
143 fxSpot = fxIndex->fixing(cf->pricingDate());
144 }
145 res.atm = cf->index()->fixing(cf->pricingDate()) * fxSpot;
146 res.sigma = res.tn > 0 && !QuantLib::close_enough(res.tn, 0.0) ? vol->blackVol(res.tn, res.atm, true) : 0.0;
147 } else if (auto avgCf = ext::dynamic_pointer_cast<CommodityIndexedAverageCashFlow>(flow)) {
148 auto parameter = CommodityAveragePriceOptionMomementMatching::matchFirstTwoMomentsTurnbullWakeman(
149 avgCf, vol,
150 std::bind(&CommoditySpreadOptionAnalyticalEngine::intraAssetCorrelation, this, std::placeholders::_1,
151 std::placeholders::_2, vol));
152 res.tn = parameter.tn;
153 res.atm = parameter.forward;
154 res.accruals = parameter.accruals;
155 res.sigma = parameter.sigma;
156 } else {
157 QL_FAIL("SpreadOptionEngine supports only CommodityIndexedCashFlow or CommodityIndexedAverageCashFlow");
158 }
159 return res;
160}

162Real CommoditySpreadOptionAnalyticalEngine::intraAssetCorrelation(
163 const Date& ed_1, const Date& ed_2, const ext::shared_ptr<BlackVolTermStructure>& vol) const {
164 if (beta_ == 0.0 || ed_1 == ed_2) {
165 return 1.0;
166 } else {
167 Time t_1 = vol->timeFromReference(ed_1);
168 Time t_2 = vol->timeFromReference(ed_2);
169 return exp(-beta_ * fabs(t_2 - t_1));
170 }
171}

173Real CommoditySpreadOptionAnalyticalEngine::rho() const {
174 if (arguments_.longAssetFlow->index()->underlyingName() != arguments_.shortAssetFlow->index()->underlyingName()) {
175 return rho_->correlation(arguments_.exercise->lastDate());
176 } else {
177 return intraAssetCorrelation(arguments_.shortAssetLastPricingDate, arguments_.longAssetLastPricingDate,
178 *volTSLongAsset_);
179 }
