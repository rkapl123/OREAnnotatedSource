75 Real FxIndex::fixing(const Date& fixingDate, bool forecastTodaysFixing) const {
76 
77  QL_REQUIRE(isValidFixingDate(fixingDate),
78  "Fixing date " << fixingDate << " is not valid for FxIndex '" << name() << "'");
79 
80  Date today = Settings::instance().evaluationDate();
81 
82  Real result = Null<Decimal>();
83 
84  if (fixingDate > today || (fixingDate == today && forecastTodaysFixing))
85  result = forecastFixing(fixingDate);
86 
87  if (result == Null<Real>()) {
88  if (fixingDate < today || Settings::instance().enforcesTodaysHistoricFixings()) {
89  // must have been fixed
90  // do not catch exceptions
91  result = pastFixing(fixingDate);
92  QL_REQUIRE(result != Null<Real>(), "Missing " << name() << " fixing for " << fixingDate);
93  } else {
94  try {
95  // might have been fixed
96  result = pastFixing(fixingDate);
97  } catch (Error&) {
98  ; // fall through and forecast
99  }
100  if (result == Null<Real>())
101  result = forecastFixing(fixingDate);
102  }
103  }
104 
105  return inverseIndex_ ? 1.0 / result : result;
106 }

108 Real FxIndex::forecastFixing(const Time& fixingTime) const {
109  QL_REQUIRE(!sourceYts_.empty() && !targetYts_.empty(), "null term structure set to this instance of " << name());
110 
111  // we base the forecast always on the exchange rate (and not on today's
112  // fixing)
113  Real rate;
114  if (!useQuote_) {
115  rate = ExchangeRateManager::instance().lookup(sourceCurrency_, targetCurrency_).rate();
116  } else {
117  QL_REQUIRE(!fxQuote_.empty(), "FxIndex::forecastFixing(): fx quote required");
118  rate = fxQuote_->value();
119  }
120 
121  // TODO: Add a time based adjusement for settlement days
122  Real forward = rate * sourceYts_->discount(fixingTime) /
123  targetYts_->discount(fixingTime);
124  return forward;
125 }

127 Real FxIndex::forecastFixing(const Date& fixingDate) const {
128  QL_REQUIRE(!sourceYts_.empty() && !targetYts_.empty(), "null term structure set to this instance of " << name());
129 
130  // we base the forecast always on the exchange rate (and not on today's
131  // fixing)
132  Real rate;
133  if (!useQuote_) {
134  rate = ExchangeRateManager::instance().lookup(sourceCurrency_, targetCurrency_).rate();
135  } else {
136  QL_REQUIRE(!fxQuote_.empty(), "FxIndex::forecastFixing(): fx quote required");
137  rate = fxQuote_->value();
138  }
139 
140  // the exchange rate is interpreted as the spot rate w.r.t. the index's
141  // settlement date
142  Date refValueDate = valueDate(fixingCalendar().adjust(Settings::instance().evaluationDate()));
143 
144  // the fixing is obeying the settlement delay as well
145  Date fixingValueDate = valueDate(fixingDate);
146 
147  // we can assume fixingValueDate >= valueDate
148  QL_REQUIRE(fixingValueDate >= refValueDate, "value date for requested fixing as of "
149  << fixingDate << " (" << fixingValueDate
150  << ") must be greater or equal to today's fixing value date ("
151  << refValueDate << ")");
152 
153  // compute the forecast applying the usual no arbitrage principle
154  Real forward = rate * sourceYts_->discount(fixingValueDate) * targetYts_->discount(refValueDate) /
155  (sourceYts_->discount(refValueDate) * targetYts_->discount(fixingValueDate));
156 
157  return forward;
158 }
