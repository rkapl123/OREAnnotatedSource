29 CarrMadanMarginalProbability::CarrMadanMarginalProbability(const std::vector<Real>& strikes, const Real forward,
30  const std::vector<Real>& callPrices)
31  : strikes_(strikes), forward_(forward), callPrices_(callPrices) {
32 
33  // check input
34 
35  QL_REQUIRE(strikes_.size() == callPrices_.size(), "CarrMadanMarginalProbability: strikes ("
36  << strikes_.size() << ") inconsistent to callPrices ("
37  << callPrices_.size() << ")";)
38 
39  QL_REQUIRE(!strikes_.empty(), "CarrMadanMarginalProbability: input moneyness is empty");
40 
41  // build sort permutation for strikes
42 
43  std::vector<Size> perm(strikes.size());
44  std::iota(perm.begin(), perm.end(), 0);
45  std::sort(perm.begin(), perm.end(), [this](Size a, Size b) { return strikes_[a] < strikes_[b]; });
46 
47  // check strikes are different (and increasing for the found permutation)
48 
49  for (Size i = 1; i < strikes_.size(); ++i) {
50  QL_REQUIRE(strikes_[perm[i]] > strikes_[perm[i - 1]] && !close_enough(strikes_[perm[i]], strikes_[perm[i - 1]]),
51  "CarrMadanMarginalProbability: duplicate strikes at "
52  << perm[i - 1] << ", " << perm[i] << ": " << strikes[perm[i - 1]] << ", " << strikes[perm[i]]);
53  }
54 
55  QL_REQUIRE(strikes_[perm[0]] > 0.0 || close_enough(strikes_[perm[0]], 0.0),
56  "CarrMadanMarginalProbability: all input strikes must be positive or zero, got " << strikes_[perm[0]]);
57 
58  // add strike 0 and corresponding call price (= forward), if not already present
59 
60  bool zeroStrikeAdded = false;
61  if (!close_enough(strikes_[perm[0]], 0.0)) {
62  strikes_.push_back(0.0);
63  callPrices_.push_back(forward_);
64  perm.insert(perm.begin(), strikes_.size() - 1);
65  zeroStrikeAdded = true;
66  } else {
67  QL_REQUIRE(close_enough(callPrices_[perm[0]], forward_),
68  "CarrMadanMarginalProbability: call price for strike 0 ("
69  << callPrices_.front() << ") should match forward (" << forward_ << ")");
70  }
71 
72  // check we have two strikes at least
73 
74  QL_REQUIRE(strikes_.size() >= 2,
75  "CarrMadanMarginalProbability: at least two strikes levels required (after adding 0)");
76 
77  // compute Q
78 
79  std::vector<Real> Q(strikes_.size() - 1);
80  for (Size i = 1; i < strikes_.size(); ++i) {
81  Q[i - 1] = (callPrices_[perm[i - 1]] - callPrices_[perm[i]]) / (strikes_[perm[i]] - strikes_[perm[i - 1]]);
82  }
83 
84  // compute BS
85 
86  std::vector<Real> BS(strikes_.size() - 2);
87  for (Size i = 1; i < strikes_.size() - 1; ++i) {
88  BS[i - 1] = callPrices_[perm[i - 1]] -
89  (strikes_[perm[i + 1]] - strikes_[perm[i - 1]]) / (strikes_[perm[i + 1]] - strikes_[perm[i]]) *
90  callPrices_[perm[i]] +
91  (strikes_[perm[i]] - strikes_[perm[i - 1]]) / (strikes_[perm[i + 1]] - strikes_[perm[i]]) *
92  callPrices_[perm[i + 1]];
93  }
94 
95  // perform the checks 1, 2 from the paper, and populate the set of arbitrage
96 
97  smileIsArbitrageFree_ = true;
98  callSpreadArbitrage_.resize(strikes_.size());
99  butterflyArbitrage_.resize(strikes_.size());
100 
101  // check 1: Q(i,j) in [0,1]
102 
103  for (Size i = 0; i < Q.size(); ++i) {
104  if ((Q[i] < 0.0 && !close_enough(Q[i], 0.0)) || (Q[i] > 1.0 && !close_enough(Q[i], 1.0))) {
105  callSpreadArbitrage_[perm[i]] = true;
106  callSpreadArbitrage_[perm[i + 1]] = true;
107  smileIsArbitrageFree_ = false;
108  }
109  }
110 
111  // check 2: BS(i,j) >= 0
112 
113  for (Size i = 0; i < BS.size(); ++i) {
114  if (BS[i] < -1.0E-10) {
115  butterflyArbitrage_[perm[i]] = true;
116  butterflyArbitrage_[perm[i + 1]] = true;
117  butterflyArbitrage_[perm[i + 2]] = true;
118  smileIsArbitrageFree_ = false;
119  }
120  }
121 
122  // compute the density q
123 
124  q_.resize(strikes_.size());
125  q_[perm[0]] = 1.0 - Q.front();
126  for (Size i = 0; i < Q.size() - 1; ++i) {
127  q_[perm[i + 1]] = Q[i] - Q[i + 1];
128  }
129  q_[perm.back()] = Q.back();
130 
131  // remove zero strike again, if not present from the start
132 
133  if (zeroStrikeAdded) {
134  strikes_.erase(std::next(strikes_.end(), -1));
135  callPrices_.erase(std::next(callPrices_.end(), -1));
136  callSpreadArbitrage_.erase(std::next(callSpreadArbitrage_.begin(), perm.front()));
137  butterflyArbitrage_.erase(std::next(butterflyArbitrage_.begin(), perm.front()));
138  q_.erase(std::next(q_.begin(), perm.front()));
139  }
140 }

165 CarrMadanSurface::CarrMadanSurface(const std::vector<Real>& times, const std::vector<Real>& moneyness, const Real spot,
166  const std::vector<Real>& forwards, const std::vector<std::vector<Real>>& callPrices)
167  : times_(times), moneyness_(moneyness), spot_(spot), forwards_(forwards), callPrices_(callPrices) {
168 
169  // checks
170 
171  QL_REQUIRE(times_.size() == callPrices_.size(),
172  "CarrMadanSurface: times size (" << times_.size() << ") does not match callPrices outer vector size ("
173  << callPrices_.size() << ")");
174  QL_REQUIRE(times.size() == forwards_.size(), "CarrMadanSurface: times size (" << times_.size()
175  << ") does not match forwards size ("
176  << forwards_.size() << ")");
177 
178  QL_REQUIRE(!times_.empty(), "CarrMadanSurface: times are empty");
179 
180  for (Size i = 1; i < times_.size(); ++i) {
181  QL_REQUIRE(times_[i] > times_[i - 1] && !close_enough(times_[i], times_[i - 1]),
182  "CarrMadanSurface: times not increasing at index " << (i - 1) << ", " << i << ": " << times_[i - 1]
183  << ", " << times_[i]);
184  }
185 
186  QL_REQUIRE(times_.front() > 0.0 || close_enough(times_.front(), 0.0),
187  "CarrMadanSurface: all input times must be positive or zero, got " << times_.front());
188 
189  for (Size i = 0; i < times_.size(); ++i) {
190  QL_REQUIRE(callPrices_[i].size() == moneyness_.size(), "CarrMadanSurface: callPrices at time "
191  << times_[i] << "(" << callPrices_[i].size()
192  << ") should match moneyness size ("
193  << moneyness_.size() << ")");
194  }
195 
196  // construct the time slices
197 
198  surfaceIsArbitrageFree_ = true;
199  for (Size i = 0; i < times_.size(); ++i) {
200  std::vector<Real> strikes(moneyness_.size());
201  Real f = forwards_[i];
202  std::transform(moneyness_.begin(), moneyness_.end(), strikes.begin(), [&f](Real m) { return m * f; });
203  timeSlices_.push_back(CarrMadanMarginalProbability(strikes, forwards_[i], callPrices_[i]));
204  surfaceIsArbitrageFree_ = surfaceIsArbitrageFree_ && timeSlices_.back().arbitrageFree();
205  callSpreadArbitrage_.push_back(timeSlices_.back().callSpreadArbitrage());
206  butterflyArbitrage_.push_back(timeSlices_.back().butterflyArbitrage());
207  }
208 
209  // check for calendar arbitrage
210 
211  calendarArbitrage_ = std::vector<std::vector<bool>>(times_.size(), std::vector<bool>(moneyness_.size()));
212  for (Size i = 0; i < moneyness_.size(); ++i) {
213  for (Size j = 0; j < times_.size() - 1; ++j) {
214  Real c2 = callPrices_[j + 1][i] * spot_ / forwards_[j + 1];
215  Real c1 = callPrices_[j][i] * spot_ / forwards_[j];
216  bool af = c2 > c1 || close_enough(c1, c2);
217  if (!af) {
218  calendarArbitrage_[j][i] = true;
219  calendarArbitrage_[j + 1][i] = true;
220  surfaceIsArbitrageFree_ = false;
221  }
222  }
223  }
224 }
