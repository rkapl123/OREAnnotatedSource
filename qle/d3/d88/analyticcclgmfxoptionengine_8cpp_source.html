32 Real AnalyticCcLgmFxOptionEngine::value(const Time t0, const Time t, const boost::shared_ptr<StrikedTypePayoff> payoff,
33  const Real domesticDiscount, const Real fxForward) const {
34  Real H0 = Hz(0).eval(model_.get(), t);
35  Real Hi = Hz(foreignCurrency_ + 1).eval(model_.get(), t);
36 
37  // just a shortcut
38  const Size& i = foreignCurrency_;
39 
40  const CrossAssetModel* x = model_.get();
41 
42  if (cacheDirty_ || !cacheEnabled_ || !(close_enough(cachedT0_, t0) && close_enough(cachedT_, t))) {
43  cachedIntegrals_ =
44  // first term
45  H0 * H0 * (zetaz(0).eval(x, t) - zetaz(0).eval(x, t0)) -
46  2.0 * H0 * integral(x, P(Hz(0), az(0), az(0)), t0, t) + integral(x, P(Hz(0), Hz(0), az(0), az(0)), t0, t) +
47  // second term
48  Hi * Hi * (zetaz(i + 1).eval(x, t) - zetaz(i + 1).eval(x, t0)) -
49  2.0 * Hi * integral(x, P(Hz(i + 1), az(i + 1), az(i + 1)), t0, t) +
50  integral(x, P(Hz(i + 1), Hz(i + 1), az(i + 1), az(i + 1)), t0, t) -
51  // third term
52  2.0 * (H0 * Hi * integral(x, P(az(0), az(i + 1), rzz(0, i + 1)), t0, t) -
53  H0 * integral(x, P(Hz(i + 1), az(i + 1), az(0), rzz(i + 1, 0)), t0, t) -
54  Hi * integral(x, P(Hz(0), az(0), az(i + 1), rzz(0, i + 1)), t0, t) +
55  integral(x, P(Hz(0), Hz(i + 1), az(0), az(i + 1), rzz(0, i + 1)), t0, t));
56  cacheDirty_ = false;
57  cachedT0_ = t0;
58  cachedT_ = t;
59  }
60 
61  Real variance = cachedIntegrals_ +
62  // term two three/fourth
63  (vx(i).eval(x, t) - vx(i).eval(x, t0)) +
64  // forth term
65  2.0 * (H0 * integral(x, P(az(0), sx(i), rzx(0, i)), t0, t) -
66  integral(x, P(Hz(0), az(0), sx(i), rzx(0, i)), t0, t)) -
67  // fifth term
68  2.0 * (Hi * integral(x, P(az(i + 1), sx(i), rzx(i + 1, i)), t0, t) -
69  integral(x, P(Hz(i + 1), az(i + 1), sx(i), rzx(i + 1, i)), t0, t));
70 
71  BlackCalculator black(payoff, fxForward, std::sqrt(variance), domesticDiscount);
72 
73  return black.value();
74 }

76 void AnalyticCcLgmFxOptionEngine::calculate() const {
77 
78  QL_REQUIRE(arguments_.exercise->type() == Exercise::European, "only European options are allowed");
79 
80  boost::shared_ptr<StrikedTypePayoff> payoff = boost::dynamic_pointer_cast<StrikedTypePayoff>(arguments_.payoff);
81  QL_REQUIRE(payoff != NULL, "only striked payoff is allowed");
82 
83  Date expiry = arguments_.exercise->lastDate();
84  Time t = model_->irlgm1f(0)->termStructure()->timeFromReference(expiry);
85 
86  if (t <= 0.0) {
87  // option is expired, we do not value any possibly non settled
88  // flows, i.e. set the npv to zero in this case
89  results_.value = 0.0;
90  return;
91  }
92 
93  Real foreignDiscount = model_->irlgm1f(foreignCurrency_ + 1)->termStructure()->discount(expiry);
94  Real domesticDiscount = model_->irlgm1f(0)->termStructure()->discount(expiry);
95 
96  Real fxForward = model_->fxbs(foreignCurrency_)->fxSpotToday()->value() * foreignDiscount / domesticDiscount;
97 
98  results_.value = value(0.0, t, payoff, domesticDiscount, fxForward);
99 
100 } // calculate()
