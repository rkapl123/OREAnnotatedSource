23 Rate EquityCouponPricer::swapletRate() const {
24  // Start fixing shouldn't include dividends as the assumption of continuous dividends means they will have been paid
25  // as they accrued in the previous period (or at least at the end when performance is measured).
26  Real start = coupon_->initialPrice();
27  Real end = equityCurve_->fixing(coupon_->fixingEndDate(), false, false);
28 
29  // fx rates at start and end, at start we only convert if the initial price is not already in target ccy
30  Real fxStart =
31  fxIndex_ && !coupon_->initialPriceIsInTargetCcy() ? fxIndex_->fixing(coupon_->fixingStartDate()) : 1.0;
32  Real fxEnd = fxIndex_ ? fxIndex_->fixing(coupon_->fixingEndDate()) : 1.0;
33 
34  Real dividends = 0.0;
35 
36  // Dividends that are already fixed dividends + yield accrued over remaining period.
37  // yield accrued = Forward without dividend yield - Forward with dividend yield
38  if (isTotalReturn_) {
39  // projected dividends from today until the fixing end date
40  dividends = equityCurve_->fixing(coupon_->fixingEndDate(), false, true) -
41  equityCurve_->fixing(coupon_->fixingEndDate(), false, false);
42  // subtract projected dividends from today until the fixing start date
43  if (coupon_->fixingStartDate() > Settings::instance().evaluationDate()) {
44  dividends -= (equityCurve_->fixing(coupon_->fixingStartDate(), false, true) -
45  equityCurve_->fixing(coupon_->fixingStartDate(), false, false));
46  }
47  // add historical dividends
48  dividends += equityCurve_->dividendsBetweenDates(coupon_->fixingStartDate(), coupon_->fixingEndDate());
49  }
50 
51  if (start == 0) {
52  return (end + dividends * dividendFactor_) * fxEnd;
53  } else if (absoluteReturn_) {
54  return ((end + dividends * dividendFactor_) * fxEnd - start * fxStart);
55  } else {
56  return ((end + dividends * dividendFactor_) * fxEnd - start * fxStart) / (start * fxStart);
57  }
58 }
