43 Real LinearGaussMarkovModel::bankAccountNumeraire(const Time t, const Real x, const Real y,
44  const Handle<YieldTermStructure> discountCurve) const {
45  QL_REQUIRE(t >= 0.0, "t (" << t << ") >= 0 required in LGM::bankAccountNumeraire");
46  Real Ht = parametrization_->H(t);
47  Real zeta0 = parametrization_->zeta(t);
48  Real zeta2 = parametrization_->zetan(2, t, integrator_);
49  Real Vt = 0.5 * (Ht * Ht * zeta0 + zeta2);
50  return std::exp(Ht * x - y + Vt) /
51  (discountCurve.empty() ? parametrization_->termStructure()->discount(t) : discountCurve->discount(t));
52 }
