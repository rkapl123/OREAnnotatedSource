33 pair<Real, Size> CommodityAveragePriceOptionBaseEngine::calculateAccrued() const {
34 
35  auto result = make_pair(0.0, 0);
36 
37  Date today = Settings::instance().evaluationDate();
38 
39  // If all of the pricing dates are greater than today => no accrued
40  if (today < arguments_.flow->indices().begin()->first) {
41  return result;
42  }
43 
44  // Calculate accrued
45  for (const auto& kv : arguments_.flow->indices()) {
46 
47  // Break on the first pricing date that is greater than today
48  if (today < kv.first) {
49  break;
50  }
51 
52  // Update accrued where pricing date is on or before today
53  result.first += kv.second->fixing(kv.first);
54  result.second++;
55  }
56 
57  // We should have pricing dates in the period but check.
58  auto n = arguments_.flow->indices().size();
59  QL_REQUIRE(n > 0, "APO coupon accrued calculation has a degenerate coupon.");
60  result.first /= n;
61 
62  return result;
63 }

66 
67  if (beta_ == 0.0 || ed_1 == ed_2) {
68  return 1.0;
69  } else {
70  Time t_1 = volStructure_->timeFromReference(ed_1);
71  Time t_2 = volStructure_->timeFromReference(ed_2);
72  return exp(-beta_ * fabs(t_2 - t_1));

76 bool CommodityAveragePriceOptionBaseEngine::isModelDependent(const pair<Real, Size>& accrued) const {
77 
78  // Discount factor to the APO payment date
79  Real discount = discountCurve_->discount(arguments_.flow->date());
80 
81  // Valuation date
82  Date today = Settings::instance().evaluationDate();
83 
84  // If all pricing dates are on or before today. This can happen when the APO payment date is a positive number
85  // of days after the final APO pricing date and today is in between.
86  if (today >= arguments_.flow->indices().rbegin()->first) {
87 
88  // Put call indicator
89  Real omega = arguments_.type == Option::Call ? 1.0 : -1.0;
90 
91  // Populate the result value
92  Real payoff = arguments_.flow->gearing() * max(omega * (accrued.first - arguments_.effectiveStrike), 0.0);
93  results_.value = arguments_.quantity * payoff * discount;
94 
95  return false;
96  }
97 
98  // If a portion of the average price has already accrued, the effective strike of the APO will
99  // have changed by the accrued amount. The strike could be non-positive.
100  Real effectiveStrike = arguments_.effectiveStrike - accrued.first;
101  if (effectiveStrike <= 0.0) {
102 
103  // If the effective strike is <= 0, put payoff is 0.0 and call payoff is [A - K]
104  if (arguments_.type == Option::Call) {
105  results_.value = (arguments_.flow->amount() - arguments_.quantity * arguments_.strikePrice) * discount;
106  } else {
107  results_.value = 0.0;
108  }
109 
110  return false;
111  }
112 
113  // If we get to here, value is model dependent
114  return true;
115 }

117 void CommodityAveragePriceOptionAnalyticalEngine::calculate() const {
118 
119  // Calculate the accrued portion
120  pair<Real, Size> accrued = calculateAccrued();
121 
122  // Populate some additional results that don't change
123  auto& mp = results_.additionalResults;
124  Real discount = discountCurve_->discount(arguments_.flow->date());
125  mp["gearing"] = arguments_.flow->gearing();
126  mp["spread"] = arguments_.flow->spread();
127  mp["strike"] = arguments_.strikePrice;
128  mp["payment_date"] = arguments_.flow->date();
129  mp["accrued"] = accrued.first;
130  mp["discount"] = discount;
131 
132  // If not model dependent, return early.
133  if (!isModelDependent(accrued)) {
134  mp["effective_strike"] = arguments_.effectiveStrike;
135  mp["npv"] = results_.value;
136  return;
137  }
138 
139  // We will read the volatility off the surface at the effective strike
140  // We should only get this far when the effectiveStrike > 0 but will check anyway
141  Real effectiveStrike = arguments_.effectiveStrike - accrued.first;
142  QL_REQUIRE(effectiveStrike > 0.0, "calculateSpot: expected effectiveStrike to be positive");
143 
144  // Valuation date
145  Date today = Settings::instance().evaluationDate();
146 
147  // Expected value of the non-accrued portion of the average commodity prices
148  // In general, m will equal n below if there is no accrued. If accrued, m > n.
149  Real EA = 0.0;
150  vector<Real> forwards;
151  vector<Time> times;
152  vector<Date> futureExpiries;
153  map<Date, Real> futureVols;
154  vector<Real> spotVars;
155  vector<Real> futureVolsVec, spotVolsVec; // additional results only
156  auto m = arguments_.flow->indices().size();
157  for (const auto& p : arguments_.flow->indices()) {
158  if (p.first > today) {
159  forwards.push_back(p.second->fixing(p.first));
160  times.push_back(volStructure_->timeFromReference(p.first));
161  if (arguments_.flow->useFuturePrice()) {
162  Date expiry = p.second->expiryDate();
163  futureExpiries.push_back(expiry);
164  if (futureVols.count(expiry) == 0) {
165  futureVols[expiry] = volStructure_->blackVol(expiry, effectiveStrike);
166  }
167  } else {
168  spotVars.push_back(volStructure_->blackVariance(times.back(), effectiveStrike));
169  spotVolsVec.push_back(std::sqrt(spotVars.back() / times.back()));
170  }
171  EA += forwards.back();
172  }
173  }
174  EA /= m;
175 
176  // Expected value of A^2. Different calculation depending on whether APO references future prices or spot price.
177  Real EA2 = 0.0;
178  Size n = forwards.size();
179  if (arguments_.flow->useFuturePrice()) {
180  // References future prices
181  for (Size i = 0; i < n; ++i) {
182  Date e_i = futureExpiries[i];
183  Volatility v_i = futureVols.at(e_i);
184  EA2 += forwards[i] * forwards[i] * exp(v_i * v_i * times[i]);
185  futureVolsVec.push_back(v_i);
186  for (Size j = 0; j < i; ++j) {
187  Date e_j = futureExpiries[j];
188  Volatility v_j = futureVols.at(e_j);
189  EA2 += 2 * forwards[i] * forwards[j] * exp(rho(e_i, e_j) * v_i * v_j * times[j]);
190  }
191  }
192  mp["futureVols"] = futureVolsVec;
193  } else {
194  // References spot prices
195  for (Size i = 0; i < n; ++i) {
196  EA2 += forwards[i] * forwards[i] * exp(spotVars[i]);
197  for (Size j = 0; j < i; ++j) {
198  EA2 += 2 * forwards[i] * forwards[j] * exp(spotVars[j]);
199  }
200  }
201  mp["spotVols"] = spotVolsVec;
202  }
203  EA2 /= m * m;
204 
205  // Calculate value
206  Real tn = times.back();
207  Real sigma = sqrt(log(EA2 / (EA * EA)) / tn);
208 
209  // Populate results
210  results_.value = arguments_.quantity * arguments_.flow->gearing() *
211  blackFormula(arguments_.type, effectiveStrike, EA, sigma * sqrt(tn), discount);
212 
213  // Add more additional results
214  // Could be part of a strip so we add the value also.
215  mp["effective_strike"] = effectiveStrike;
216  mp["forward"] = EA;
217  mp["exp_A_2"] = EA2;
218  mp["tte"] = tn;
219  mp["sigma"] = sigma;
220  mp["npv"] = results_.value;
221  mp["times"] = times;
222  mp["forwards"] = forwards;
223  mp["beta"] = beta_;
224 }

241 void CommodityAveragePriceOptionMonteCarloEngine::calculateSpot(const pair<Real, Size>& accrued) const {
242 
243  // Discount factor to the APO payment date
244  Real discount = discountCurve_->discount(arguments_.flow->date());
245 
246  // Put call indicator
247  Real omega = arguments_.type == Option::Call ? 1.0 : -1.0;
248 
249  // Variable to hold the payoff
250  Real payoff = 0.0;
251 
252  // Vector of timesteps from today = t_0 out to last pricing date t_n
253  // i.e. {t_1 - t_0, t_2 - t_1,..., t_n - t_{n-1}}
254  vector<Date> dates;
255  vector<Real> dt = timegrid(dates);
256 
257  // On each Monte Carlo sample, we must generate the spot price process path over n time steps.
258  LowDiscrepancy::rsg_type rsg = LowDiscrepancy::make_sequence_generator(dt.size(), seed_);
259 
260  // We will read the volatility off the surface at the effective strike
261  // We will only call this method when the effectiveStrike > 0 but will check anyway
262  Real effectiveStrike = arguments_.effectiveStrike - accrued.first;
263  QL_REQUIRE(effectiveStrike > 0.0, "calculateSpot: expected effectiveStrike to be positive");
264 
265  // Precalculate:
266  // exp(-1/2 forward variance over dt): \exp \left(-\frac{1}{2} \int_{t_{i-1}}^{t_i} \sigma^2(u) du \right)
267  // forward std dev over dt: \exp \left(\sqrt{\int_{t_{i-1}}^{t_i} \sigma^2(u) du} \right)
268  // forward ratio: \exp \left(\int_{t_{i-1}}^{t_i} r(u) - r_f(u) \, du \right) = \frac{F(0, t_i)}{F(0, t_{i-1})}
269  // first period is multiplied by S(0)
270  // factors array is exp(-1/2 forward variance over dt) x forward ratio
271  Array expHalfFwdVar(dt.size(), 0.0);
272  Array fwdStdDev(dt.size(), 0.0);
273  Array fwdRatio(dt.size(), 0.0);
274  Time t = 0.0;
275  for (Size i = 0; i < dt.size(); i++) {
276  t += dt[i];
277  expHalfFwdVar[i] = volStructure_->blackForwardVariance(t - dt[i], t, effectiveStrike);
278  fwdStdDev[i] = sqrt(expHalfFwdVar[i]);
279  expHalfFwdVar[i] = exp(-expHalfFwdVar[i] / 2.0);
280  fwdRatio[i] = arguments_.flow->index()->fixing(dates[i + 1]);
281  if (i > 0) {
282  fwdRatio[i] /= arguments_.flow->index()->fixing(dates[i]);
283  }
284  }
285  Array factors = expHalfFwdVar * fwdRatio;
286 
287  // Loop over each sample
288  auto m = arguments_.flow->indices().size();
289  for (Size k = 0; k < samples_; k++) {
290 
291  // Sequence is n independent standard normal random variables
292  vector<Real> sequence = rsg.nextSequence().value;
293 
294  // Calculate payoff on this sample - price holds the evolved price
295  Real price = 0.0;
296  Real samplePayoff = 0.0;
297  Array path(sequence.begin(), sequence.end());
298  path = Exp(path * fwdStdDev) * factors;
299  for (Size i = 0; i < dt.size(); i++) {
300 
301  // Update price
302  price = i == 0 ? path[i] : price * path[i];
303 
304  // Update sum of the spot prices on the pricing dates after today
305  samplePayoff += price;
306  }
307 
308  // Average price on this sample
309  samplePayoff /= m;
310 
311  // Finally, the payoff on this sample
312  samplePayoff = max(omega * (samplePayoff - effectiveStrike), 0.0);
313 
314  // Update the final average
315  // Note: payoff * k / (k + 1) - left to right precedence => double division.
316  payoff = k == 0 ? samplePayoff : payoff * k / (k + 1) + samplePayoff / (k + 1);
317  }
318 
319  // Populate the result value
320  results_.value = arguments_.quantity * arguments_.flow->gearing() * payoff * discount;
321 }

323 void CommodityAveragePriceOptionMonteCarloEngine::calculateFuture(const pair<Real, Size>& accrued) const {
324 
325  // Discount factor to the APO payment date
326  Real discount = discountCurve_->discount(arguments_.flow->date());
327 
328  // Put call indicator
329  Real omega = arguments_.type == Option::Call ? 1.0 : -1.0;
330 
331  // Variable to hold the payoff
332  Real payoff = 0.0;
333 
334  // We will read the volatility off the surface the effective strike
335  // We will only call this method when the effectiveStrike > 0 but will check anyway
336  Real effectiveStrike = arguments_.effectiveStrike - accrued.first;
337  QL_REQUIRE(effectiveStrike > 0.0, "calculateFuture: expected effectiveStrike to be positive");
338 
339  // Unique future expiry dates i.e. contracts, their volatilities and the correlation matrix between them
340  Matrix sqrtCorr;
341  vector<Real> vols;
342  vector<Real> prices;
343  vector<Size> futureIndex;
344  setupFuture(vols, sqrtCorr, prices, futureIndex, effectiveStrike);
345 
346  // Vector of timesteps from today = t_0 out to last pricing date t_n
347  // i.e. {t_1 - t_0, t_2 - t_1,..., t_n - t_{n-1}}. Don't need the dates here.
348  vector<Date> dates;
349  vector<Real> dt = timegrid(dates);
350 
351  // On each Monte Carlo sample, we must generate the paths for N (size of vols) future contracts where
352  // each path has n time steps. We will represent the paths with an N x n matrix. First step is to fill the
353  // matrix with N x n _independent_ standard normal variables. Then correlate the N variables in each column
354  // using the sqrtCorr matrix and then fill each entry F_{i, j} in the matrix with the value of the i-th
355  // future price process at timestep j. Note, we will possibly simulate contracts past their expiries but not
356  // use the price in the APO rate averaging.
357  LowDiscrepancy::rsg_type rsg = LowDiscrepancy::make_sequence_generator(vols.size() * dt.size(), seed_);
358 
359  // Precalculate exp(-0.5 \sigma_i^2 \delta t_j) and std dev = sqrt(\delta t_j) \sigma_i
360  Matrix factors(vols.size(), dt.size(), 0.0);
361  Matrix stdDev(vols.size(), dt.size(), 0.0);
362  for (Size j = 0; j < factors.columns(); j++) {
363  for (Size i = 0; i < factors.rows(); i++) {
364  factors[i][j] = exp(-vols[i] * vols[i] * dt[j] / 2.0);
365  stdDev[i][j] = vols[i] * sqrt(dt[j]);
366  }
367  }
368 
369  // Loop over each sample
370  auto m = arguments_.flow->indices().size();
371  for (Size k = 0; k < samples_; k++) {
372 
373  // Sequence is N x n independent standard normal random variables
374  vector<Real> sequence = rsg.nextSequence().value;
375 
376  // paths initially holds independent standard normal random variables
377  Matrix paths(vols.size(), dt.size(), sequence.begin(), sequence.end());
378 
379  // Correlate the random variables in each column
380  paths = sqrtCorr * paths;
381 
382  // Fill the paths
383  for (Size i = 0; i < paths.rows(); i++) {
384  for (Size j = 0; j < paths.columns(); j++) {
385  // This is \exp(-\sigma_i^2 \delta t_j / 2) * \exp(\sigma_i \sqrt(\delta t_j) \delta W_{i,j})
386  paths[i][j] = factors[i][j] * exp(stdDev[i][j] * paths[i][j]);
387  if (j == 0) {
388  // If on first timestep, multiply by F_i(0)
389  paths[i][j] *= prices[i];
390  } else {
391  // If on timestep greater than t_0, multiply by F_i(t_{j-1}) i.e. value of contract at
392  // previous time
393  paths[i][j] *= paths[i][j - 1];
394  }
395  }
396  }
397 
398  // Calculate the sum of the commodity future prices on the pricing dates after today
399  Real samplePayoff = 0.0;
400  for (Size j = 0; j < dt.size(); j++) {
401  samplePayoff += paths[futureIndex[j]][j];
402  }
403 
404  // Average price on this sample
405  samplePayoff /= m;
406 
407  // Finally, the payoff on this sample
408  samplePayoff = max(omega * (samplePayoff - effectiveStrike), 0.0);
409 
410  // Update the final average
411  // Note: payoff * k / (k + 1) - left to right precedence => double division.
412  payoff = k == 0 ? samplePayoff : payoff * k / (k + 1) + samplePayoff / (k + 1);
413  }
414 
415  // Populate the result value
416  results_.value = arguments_.quantity * arguments_.flow->gearing() * payoff * discount;
417 }

419 void CommodityAveragePriceOptionMonteCarloEngine::setupFuture(vector<Real>& outVolatilities, Matrix& outSqrtCorr,
420  vector<Real>& prices, vector<Size>& futureIndex,
421  Real strike) const {
422 
423  // Ensure that vectors are empty
424  outVolatilities.clear();
425  prices.clear();
426  futureIndex.clear();
427 
428  // Will hold the result
429  map<Date, Real> result;
430 
431  // Note that here we make the simplifying assumption that the volatility can be read from the volatility
432  // term structure at the future contract's expiry date. In most cases, if the volatility term structure is built
433  // from options on futures, the option contract expiry will be a number of days before the future contract expiry
434  // and we should really read off the term structure at that date. Also populate a temp vector containing the key
435  // dates for use in the loop below where we populate the sqrt correlation matrix.
436 
437  // Initialise result with expiry date keys that are still live in the APO
438  Date today = Settings::instance().evaluationDate();
439  set<Date> expiryDates;
440  for (const auto& p : arguments_.flow->indices()) {
441  if (p.first > today) {
442  Date expiry = p.second->expiryDate();
443  // If expiry has not been encountered yet
444  if (expiryDates.insert(expiry).second) {
445  outVolatilities.push_back(volStructure_->blackVol(expiry, strike));
446  prices.push_back(p.second->fixing(today));
447  }
448  futureIndex.push_back(expiryDates.size() - 1);
449  }
450  }
451 
452  // Populate the outSqrtCorr matrix
453  vector<Date> vExpiryDates(expiryDates.begin(), expiryDates.end());
454  outSqrtCorr = Matrix(vExpiryDates.size(), vExpiryDates.size(), 1.0);
455  for (Size i = 0; i < vExpiryDates.size(); i++) {
456  for (Size j = 0; j < i; j++) {
457  outSqrtCorr[i][j] = outSqrtCorr[j][i] = rho(vExpiryDates[i], vExpiryDates[j]);
458  }
459  }
460  outSqrtCorr = pseudoSqrt(outSqrtCorr, SalvagingAlgorithm::None);
461 }

463 vector<Real> CommodityAveragePriceOptionMonteCarloEngine::timegrid(vector<Date>& outDates) const {
464 
465  // Initialise result with expiry date keys that are still live in the APO
466  // i.e. {t_1 - t_0, t_2 - t_0,..., t_n - t_0} where t_0 corresponds to today
467  vector<Real> times;
468  outDates.clear();
469  Date today = Settings::instance().evaluationDate();
470  outDates.push_back(today);
471  for (const auto& p : arguments_.flow->indices()) {
472  if (p.first > today) {
473  outDates.push_back(p.first);
474  times.push_back(volStructure_->timeFromReference(p.first));
475  }
476  }
477 
478  // Populate time deltas i.e. {t_1 - t_0, t_2 - t_1,..., t_n - t_{n-1}}
479  vector<Real> dt(times.size());
480  adjacent_difference(times.begin(), times.end(), dt.begin());
481 
482  return dt;
483 }
