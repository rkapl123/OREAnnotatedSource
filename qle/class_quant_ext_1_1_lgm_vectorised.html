443 // the times we use for the projection in the LGM model, if t > T1 they are displaced by (t-T1)
444
445 Real T1_lgm = T1, T2_lgm = T2;
446 if (t > T1) {
447 T1_lgm += t - T1;
448 T2_lgm += t - T1;
449 }
450
451 // the discount factors estimated in the lgm model
452
453 RandomVariable disc1 = reducedDiscountBond(t, T1_lgm, x, curve);
454 RandomVariable disc2 = reducedDiscountBond(t, T2_lgm, x, curve);
455
456 // apply a correction to the discount factors
457
458 disc1 *= RandomVariable(x.size(), startDiscount / curve->discount(T1_lgm));
459 disc2 *= RandomVariable(x.size(), endDiscount / curve->discount(T2_lgm));
460
461 // estimate the fixing
462
463 fixing = (disc1 / disc2 - RandomVariable(x.size(), 1.0)) /
464 RandomVariable(x.size(), index->dayCounter().yearFraction(start, end));
465 }
466
467 avgBMA += fixing * RandomVariable(x.size(), (d2 - d1));
468 d1 = d2;
469 }
470
471 avgBMA /= RandomVariable(x.size(), (endDate - startDate));
472
473 return avgBMA;
