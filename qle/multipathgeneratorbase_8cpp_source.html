66const Sample<MultiPath>& MultiPathGeneratorSobolBrownianBridge::next() const {
67 Array asset = process_->initialValues();
68 MultiPath& path = next_.value;
69 for (Size j = 0; j < asset.size(); ++j) {
70 path[j].front() = asset[j];
71 }
72 next_.weight = gen_->nextPath();
73 std::vector<Real> output(process_->factors());
74 for (Size i = 1; i < grid_.size(); ++i) {
75 Real t = grid_[i - 1];
76 Real dt = grid_.dt(i - 1);
77 gen_->nextStep(output);
78 Array tmp(output.begin(), output.end());
79 asset = process_->evolve(t, asset, dt, tmp);
80 for (Size j = 0; j < asset.size(); ++j) {
81 path[j][i] = asset[j];
82 }
83 }
84 return next_;
85}

87boost::shared_ptr<MultiPathGeneratorBase> makeMultiPathGenerator(const SequenceType s,
88 const boost::shared_ptr<StochasticProcess>& process,
89 const TimeGrid& timeGrid, const BigNatural seed,
90 const SobolBrownianGenerator::Ordering ordering,
91 const SobolRsg::DirectionIntegers directionIntegers) {
92 switch (s) {
93 case MersenneTwister:
94 return boost::make_shared<QuantExt::MultiPathGeneratorMersenneTwister>(process, timeGrid, seed, false);
95 case MersenneTwisterAntithetic:
96 return boost::make_shared<QuantExt::MultiPathGeneratorMersenneTwister>(process, timeGrid, seed, true);
97 case Sobol:
98 return boost::make_shared<QuantExt::MultiPathGeneratorSobol>(process, timeGrid, seed, directionIntegers);
99 case SobolBrownianBridge:
100 return boost::make_shared<QuantExt::MultiPathGeneratorSobolBrownianBridge>(process, timeGrid, ordering, seed,
101 directionIntegers);
102 default:
103 QL_FAIL("Unknown sequence type");
104 }
105}
