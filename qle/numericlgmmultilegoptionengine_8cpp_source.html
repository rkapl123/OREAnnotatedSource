41/* For a given cashflow and option exercise date, return the simulation date on which we determine the coupon amout.
42 - If the pay date is <= today, null is returned.
43 - The latestOptionDate is the latest option date for which the cashflow is relevant. This can be Date::minDate(),
44 if the cashflow is not relevant for any option date.
45 The return value will be >= today (or null) always. */
46Date getSimulationDates(const Date& today, const Date& latestOptionDate, const boost::shared_ptr<CashFlow>& c) {
47 if (c->date() <= today)
48 return Date();
49 Date minDate = std::max(today, latestOptionDate);
50 if (auto cpn = boost::dynamic_pointer_cast<Coupon>(c)) {
51 if (auto ibor = boost::dynamic_pointer_cast<IborCoupon>(c)) {
52 return minDate;
53 } else if (auto fix = boost::dynamic_pointer_cast<FixedRateCoupon>(cpn)) {
54 return minDate;
55 } else if (auto on = boost::dynamic_pointer_cast<QuantExt::OvernightIndexedCoupon>(cpn)) {
56 return minDate;
57 } else if (auto av = boost::dynamic_pointer_cast<QuantExt::AverageONIndexedCoupon>(cpn)) {
58 return minDate;
59 } else if (auto av = boost::dynamic_pointer_cast<QuantLib::AverageBMACoupon>(cpn)) {
60 return minDate;
61 } else if (auto cf = boost::dynamic_pointer_cast<QuantLib::CappedFlooredCoupon>(cpn)) {
62 auto und = cf->underlying();
63 if (auto cfibor = boost::dynamic_pointer_cast<QuantLib::IborCoupon>(und)) {
64 return std::max(minDate, und->fixingDate());
65 }
66 } else if (auto cfon = boost::dynamic_pointer_cast<QuantExt::CappedFlooredOvernightIndexedCoupon>(cpn)) {
67 return std::max(minDate, cfon->underlying()->fixingDates().front());
68 } else if (auto cfav = boost::dynamic_pointer_cast<QuantExt::CappedFlooredAverageONIndexedCoupon>(cpn)) {
69 return std::max(minDate, cfav->underlying()->fixingDates().front());
70 } else if (auto sub = boost::dynamic_pointer_cast<QuantExt::SubPeriodsCoupon1>(cpn)) {
71 return std::max(minDate, sub->fixingDates().front());
72 }
73 THROW_ERROR_CPN_NOT_SUPPORTED;
74 } else {
75 return minDate;
76 }
77}

79/* For a given cashflow and exercise date, return true if cashflow is part of the exercise right, otherwise false.
80 If exercise dates <= today or null, false is returned. */
81bool isCashflowRelevantForExercise(const Date& today, const Date& exercise, const boost::shared_ptr<CashFlow>& c) {
82 if (exercise <= today)
83 return false;
84 if (auto cpn = boost::dynamic_pointer_cast<Coupon>(c)) {
85 /* In case we have a coupon, it belongs to the exercise right if and only if exercise <= accrual start,
86 this is a definition. */
87 return exercise <= cpn->accrualStartDate();
88 } else {
89 /* In case we have a cashflow only, it belongs to the exercise right, if and only if exercise <= pay date,
90 again, this is a definition */
91 return exercise <= c->date();
92 }
93}

95/* Get the model-estimate of a cashflow as observed from model time t, state x, deflated by model numeraire */
96RandomVariable getUnderlyingCashflowPv(const LgmVectorised& lgm, const Real t, const RandomVariable& x,
97 const Handle<YieldTermStructure>& discountCurve,
98 const boost::shared_ptr<CashFlow>& c) {
99 Real T = lgm.parametrization()->termStructure()->timeFromReference(c->date());
100 if (auto cpn = boost::dynamic_pointer_cast<Coupon>(c)) {
101 if (auto ibor = boost::dynamic_pointer_cast<IborCoupon>(c)) {
102 return (RandomVariable(x.size(), ibor->gearing()) * lgm.fixing(ibor->index(), ibor->fixingDate(), t, x) +
103 RandomVariable(x.size(), ibor->spread())) *
104 RandomVariable(x.size(), ibor->accrualPeriod() * ibor->nominal()) *
105 lgm.reducedDiscountBond(t, T, x, discountCurve);
106 } else if (auto fix = boost::dynamic_pointer_cast<FixedRateCoupon>(cpn)) {
107 return RandomVariable(x.size(), fix->amount()) * lgm.reducedDiscountBond(t, T, x, discountCurve);
108 } else if (auto on = boost::dynamic_pointer_cast<QuantExt::OvernightIndexedCoupon>(cpn)) {
109 return lgm.compoundedOnRate(boost::dynamic_pointer_cast<OvernightIndex>(on->index()), on->fixingDates(),
110 on->valueDates(), on->dt(), on->rateCutoff(), on->includeSpread(), on->spread(),
111 on->gearing(), on->lookback(), Null<Real>(), Null<Real>(), false, false, t, x) *
112 RandomVariable(x.size(), on->accrualPeriod() * on->nominal()) *
113 lgm.reducedDiscountBond(t, T, x, discountCurve);
114 } else if (auto av = boost::dynamic_pointer_cast<QuantExt::AverageONIndexedCoupon>(cpn)) {
115 return lgm.averagedOnRate(boost::dynamic_pointer_cast<OvernightIndex>(av->index()), av->fixingDates(),
116 av->valueDates(), av->dt(), av->rateCutoff(), false, av->spread(), av->gearing(),
117 av->lookback(), Null<Real>(), Null<Real>(), false, false, t, x) *
118 RandomVariable(x.size(), av->accrualPeriod() * av->nominal()) *
119 lgm.reducedDiscountBond(t, T, x, discountCurve);
120 } else if (auto bma = boost::dynamic_pointer_cast<QuantLib::AverageBMACoupon>(cpn)) {
121 return lgm.averagedBmaRate(boost::dynamic_pointer_cast<BMAIndex>(bma->index()), bma->fixingDates(),
122 bma->accrualStartDate(), bma->accrualEndDate(), false, bma->spread(),
123 bma->gearing(), Null<Real>(), Null<Real>(), false, t, x) *
124 RandomVariable(x.size(), bma->accrualPeriod() * bma->nominal()) *
125 lgm.reducedDiscountBond(t, T, x, discountCurve);
126 } else if (auto cf = boost::dynamic_pointer_cast<QuantLib::CappedFlooredCoupon>(cpn)) {
127 auto und = cf->underlying();
128 RandomVariable cap(x.size(), cf->cap() == Null<Real>() ? QL_MAX_REAL : cf->cap());
129 RandomVariable floor(x.size(), cf->floor() == Null<Real>() ? -QL_MAX_REAL : cf->floor());
130 if (auto undibor = boost::dynamic_pointer_cast<QuantLib::IborCoupon>(und)) {
131 return max(floor, min(cap, (RandomVariable(x.size(), undibor->gearing()) *
132 lgm.fixing(undibor->index(), undibor->fixingDate(), t, x) +
133 RandomVariable(x.size(), undibor->spread())))) *
134 RandomVariable(x.size(), undibor->accrualPeriod() * undibor->nominal()) *
135 lgm.reducedDiscountBond(t, T, x, discountCurve);
136 }
137 } else if (auto cfon = boost::dynamic_pointer_cast<QuantExt::CappedFlooredOvernightIndexedCoupon>(cpn)) {
138 auto und = cfon->underlying();
139 return lgm.compoundedOnRate(boost::dynamic_pointer_cast<OvernightIndex>(und->index()), und->fixingDates(),
140 und->valueDates(), und->dt(), und->rateCutoff(), und->includeSpread(),
141 und->spread(), und->gearing(), und->lookback(), cfon->cap(), cfon->floor(),
142 cfon->localCapFloor(), cfon->nakedOption(), t, x) *
143 RandomVariable(x.size(), cfon->accrualPeriod() * cfon->nominal()) *
144 lgm.reducedDiscountBond(t, T, x, discountCurve);
145 } else if (auto cfav = boost::dynamic_pointer_cast<QuantExt::CappedFlooredAverageONIndexedCoupon>(cpn)) {
146 auto und = cfav->underlying();
147 return lgm.averagedOnRate(boost::dynamic_pointer_cast<OvernightIndex>(und->index()), und->fixingDates(),
148 und->valueDates(), und->dt(), und->rateCutoff(), cfav->includeSpread(),
149 und->spread(), und->gearing(), und->lookback(), cfav->cap(), cfav->floor(),
150 cfav->localCapFloor(), cfav->nakedOption(), t, x) *
151 RandomVariable(x.size(), cfav->accrualPeriod() * cfav->nominal()) *
152 lgm.reducedDiscountBond(t, T, x, discountCurve);
153 } else if (auto cfbma = boost::dynamic_pointer_cast<QuantExt::CappedFlooredAverageBMACoupon>(cpn)) {
154 auto und = cfbma->underlying();
155 return lgm.averagedBmaRate(boost::dynamic_pointer_cast<BMAIndex>(und->index()), und->fixingDates(),
156 und->accrualStartDate(), und->accrualEndDate(), cfbma->includeSpread(),
157 und->spread(), und->gearing(), cfbma->cap(), cfbma->floor(),
158 cfbma->nakedOption(), t, x) *
159 RandomVariable(x.size(), cfbma->accrualPeriod() * cfbma->nominal()) *
160 lgm.reducedDiscountBond(t, T, x, discountCurve);
161 } else if (auto sub = boost::dynamic_pointer_cast<QuantExt::SubPeriodsCoupon1>(cpn)) {
162 return lgm.subPeriodsRate(sub->index(), sub->fixingDates(), t, x) *
163 RandomVariable(x.size(), sub->accrualPeriod() * sub->nominal()) *
164 lgm.reducedDiscountBond(t, T, x, discountCurve);
165 }
166 THROW_ERROR_CPN_NOT_SUPPORTED;
167 } else {
168 return RandomVariable(x.size(), c->amount()) * lgm.reducedDiscountBond(t, T, x, discountCurve);
169 }

172/* Get the rebate amount on an exercise */
173RandomVariable getRebatePv(const LgmVectorised& lgm, const Real t, const RandomVariable& x,
174 const Handle<YieldTermStructure>& discountCurve,
175 const boost::shared_ptr<RebatedExercise>& exercise, const Date& d) {
176 if (exercise == nullptr)
177 return RandomVariable(x.size(), 0.0);
178 auto f = std::find(exercise->dates().begin(), exercise->dates().end(), d);
179 QL_REQUIRE(f != exercise->dates().end(), "NumericLgmMultiLegOptionEngine: internal error: exercise date "
180 << d << " from rebate payment not found amount exercise dates.");
181 Size index = std::distance(exercise->dates().begin(), f);
182 return RandomVariable(x.size(), exercise->rebate(index)) *
183 lgm.reducedDiscountBond(
184 t, lgm.parametrization()->termStructure()->timeFromReference(exercise->rebatePaymentDate(index)), x);
185}

192void NumericLgmMultiLegOptionEngineBase::calculate() const {
193
194 /* TODO ParYieldCurve cash-settled swaptions are priced as if CollateralizedCashPrice, this can be refined. */
195
196 /* The reference date of the valuation. Exercise dates <= refDate are treated as if not present, i.e. it is
197 assumed that these rights were not exercised.
198 TODO we should add exercise data to indicate past exercises and value outstanding cash settlement amounts,
199 or - possibly (?) - the physical underlying in that we exercised into? Or would that be a separate trade? */
200
201 Date refDate = model()->parametrization()->termStructure()->referenceDate();
202
203 /* we might have to handle an exercise with rebate, init a variable for that (null if not applicable) */
204
205 auto rebatedExercise = boost::dynamic_pointer_cast<QuantExt::RebatedExercise>(exercise_);
206
207 /* Check that the engine can handle the instrument, i.e. that there is a unique pay currency and all interest rate
208 indices are in this same currency. */
209
210 for (Size i = 1; i < currency_.size(); ++i) {
211 QL_REQUIRE(currency_[i] == currency_[0],
212 "NumericLgmMultilegOptionEngine: can only handle single currency underlyings, got at least two pay "
213 "currencies: "
214 << currency_[0] << "; " << currency_[i]);
215 }
216
217 for (Size i = 0; i < legs_.size(); ++i) {
218 for (Size j = 0; j < legs_[i].size(); ++j) {
219 if (auto cpn = boost::dynamic_pointer_cast<FloatingRateCoupon>(legs_[i][j])) {
220 QL_REQUIRE(cpn->index()->currency() == currency_[0],
221 "NumericLgmMultilegOptionEngine: can only handle indices ("
222 << cpn->index()->name() << ") with same currency as unqiue pay currency ("
223 << currency_[0]);
224 }
225 }
226 }
227
228 /* Map an option date to the set of relevant cashflows to exercise into, the set will not contain cashflows
229 that are relevant for later option dates. */
230
231 std::map<Date, std::set<std::pair<Size, Size>>> optionDates;
232
233 for (Size i = 0; i < legs_.size(); ++i) {
234 for (Size j = 0; j < legs_[i].size(); ++j) {
235 for (Size k = exercise_->dates().size(); k > 0; --k) {
236 Date exerciseDate = exercise_->dates()[k - 1];
237 if (exerciseDate > refDate) {
238 if (isCashflowRelevantForExercise(refDate, exerciseDate, legs_[i][j])) {
239 optionDates[exerciseDate].insert(std::make_pair(i, j));
240 break;
241 }
242 }
243 }
244 }
245 }
246
247 /* map a actual simulation date => set of (cf leg, cf number) */
248
249 std::map<Date, std::set<std::pair<Size, Size>>> cashflowDates;
250
251 /* Each underlying cashflow has a date on which we can efficiently calculate its amount, depending on
252 the option date for which the cashflow is relevant. We estimate a cashflow amount for the latest
253 option date it is relevant for and then roll it back. */
254
255 for (Size i = 0; i < legs_.size(); ++i) {
256 for (Size j = 0; j < legs_[i].size(); ++j) {
257 Date latestOptionDate = Date::minDate();
258 for (auto const& d : optionDates) {
259 if (d.second.find(std::make_pair(i, j)) != d.second.end())
260 latestOptionDate = std::max(latestOptionDate, d.first);
261 }
262 auto d = getSimulationDates(refDate, latestOptionDate, legs_[i][j]);
263 if (d != Null<Date>())
264 cashflowDates[d].insert(std::make_pair(i, j));
265 }
266 }
267
268 /* Compile the list of simulation dates, include the ref date always */
269
270 std::set<Date> simulationDates;
271
272 for (auto const& d : optionDates)
273 simulationDates.insert(d.first);
274
275 for (auto const& d : cashflowDates)
276 simulationDates.insert(d.first);
277
278 simulationDates.insert(refDate);
279
280 /* Step backwards through simulation dates and compute the option npv */
281
282 LgmVectorised lgm(model()->parametrization());
283
284 // the current option npv that we roll back through the grid
285 RandomVariable optionNpv(gridSize(), 0.0);
286 // cashflow npvs that are part of the latest exercise into option
287 RandomVariable underlyingNpv1(gridSize(), 0.0);
288 // cashflow npvs that are estimated, but not yet part of the latest exercise into option
289 std::map<std::pair<Size, Size>, RandomVariable> underlyingNpv2;
290
291 for (auto it = simulationDates.rbegin(); it != simulationDates.rend(); ++it) {
292
293 Real t_from = model()->parametrization()->termStructure()->timeFromReference(*it);
294 Real t_to = (it != std::next(simulationDates.rend(), -1))
295 ? model()->parametrization()->termStructure()->timeFromReference(*std::next(it, 1))
296 : t_from;
297
298 RandomVariable state = stateGrid(t_from);
299
300 auto option = optionDates.find(*it);
301 auto cashflow = cashflowDates.find(*it);
302
303 // collect cashflow amounts on simulation date
304
305 if (cashflow != cashflowDates.end()) {
306 for (auto const& cf : cashflow->second) {
307 auto tmp = getUnderlyingCashflowPv(lgm, t_from, state, discountCurve_, legs_[cf.first][cf.second]) *
308 RandomVariable(gridSize(), payer_[cf.first]);
309 auto it = underlyingNpv2.find(cf);
310 if (it != underlyingNpv2.end())
311 it->second += tmp;
312 else
313 underlyingNpv2[cf] = tmp;
314 }
315 }
316
317 // handle option date
318
319 if (option != optionDates.end()) {
320
321 // transfer relevant cashflow amounts to exercise into npv
322
323 for (auto const& cf : option->second) {
324 underlyingNpv1 += underlyingNpv2[cf];
325 underlyingNpv2.erase(cf);
326 }
327
328 // update the option value (exercise value including rebate vs. continuation value)
329
330 auto rebateNpv = getRebatePv(lgm, t_from, state, discountCurve_, rebatedExercise, option->first);
331 optionNpv = max(optionNpv, underlyingNpv1 + rebateNpv);
332 }
333
334 // rollback
335
336 if (t_from != t_to) {
337 underlyingNpv1 = rollback(underlyingNpv1, t_from, t_to);
338 for (auto& c : underlyingNpv2) {
339 c.second = rollback(c.second, t_from, t_to);
340 }
341 optionNpv = rollback(optionNpv, t_from, t_to);
342 }
343 }
344
345 /* Set the results */
346
347 npv_ = optionNpv.at(0);
348 underlyingNpv_ = underlyingNpv1.at(0);
349 for (auto const& c : underlyingNpv2) {
350 underlyingNpv_ += c.second.at(0);
351 }
352
353 additionalResults_ = getAdditionalResultsMap(model()->getCalibrationInfo());
354
355 if (rebatedExercise) {
356 for (Size i = 0; i < rebatedExercise->dates().size(); ++i) {
357 std::ostringstream d;
358 d << QuantLib::io::iso_date(rebatedExercise->dates()[i]);
359 additionalResults_["exerciseFee_" + d.str()] = -rebatedExercise->rebate(i);
360 }
361 }
362
363} // NumericLgmMultiLegOptionEngineBase::calculate()
