39 /* For a given cashflow and option exercise date, return the simulation date on which we determine the coupon amout.
40  - If the pay date is <= today, null is returned.
41  - The latestOptionDate is the latest option date for which the cashflow is relevant. This can be Date::minDate(),
42  if the cashflow is not relevant for any option date.
43  The return value will be >= today (or null) always. */
44 Date getSimulationDates(const Date& today, const Date& latestOptionDate, const boost::shared_ptr<CashFlow>& c) {
45  if (c->date() <= today)
46  return Date();
47  Date minDate = std::max(today, latestOptionDate);
48  if (auto cpn = boost::dynamic_pointer_cast<Coupon>(c)) {
49  if (auto ibor = boost::dynamic_pointer_cast<IborCoupon>(c)) {
50  return minDate;
51  } else if (auto fix = boost::dynamic_pointer_cast<FixedRateCoupon>(cpn)) {
52  return minDate;
53  } else if (auto on = boost::dynamic_pointer_cast<QuantExt::OvernightIndexedCoupon>(cpn)) {
54  return minDate;
55  } else if (auto av = boost::dynamic_pointer_cast<QuantExt::AverageONIndexedCoupon>(cpn)) {
56  return minDate;
57  } else if (auto av = boost::dynamic_pointer_cast<QuantLib::AverageBMACoupon>(cpn)) {
58  return minDate;
59  } else if (auto cf = boost::dynamic_pointer_cast<QuantLib::CappedFlooredCoupon>(cpn)) {
60  auto und = cf->underlying();
61  if (auto cfibor = boost::dynamic_pointer_cast<QuantLib::IborCoupon>(und)) {
62  return std::max(minDate, und->fixingDate());
63  }
64  } else if (auto cfon = boost::dynamic_pointer_cast<QuantExt::CappedFlooredOvernightIndexedCoupon>(cpn)) {
65  return std::max(minDate, cfon->underlying()->fixingDates().front());
66  } else if (auto cfav = boost::dynamic_pointer_cast<QuantExt::CappedFlooredAverageONIndexedCoupon>(cpn)) {
67  return std::max(minDate, cfav->underlying()->fixingDates().front());
68  }
69  THROW_ERROR_CPN_NOT_SUPPORTED;
70  } else {
71  return minDate;
72  }
73 }

75 /* For a given cashflow and exercise date, return true if cashflow is part of the exercise right, otherwise false.
76  If exercise dates <= today or null, false is returned. */
77 bool isCashflowRelevantForExercise(const Date& today, const Date& exercise, const boost::shared_ptr<CashFlow>& c) {
78  if (exercise <= today)
79  return false;
80  if (auto cpn = boost::dynamic_pointer_cast<Coupon>(c)) {
81  /* In case we have a coupon, it belongs to the exercise right if and only if exercise <= accrual start,
82  this is a definition. */
83  return exercise <= cpn->accrualStartDate();
84  } else {
85  /* In case we have a cashflow only, it belongs to the exercise right, if and only if exercise <= pay date,
86  again, this is a definition */
87  return exercise <= c->date();
88  }
89 }

91 /* Get the model-estimate of a cashflow as observed from model time t, state x, deflated by model numeraire */
92 RandomVariable getUnderlyingCashflowPv(const LgmVectorised& lgm, const Real t, const RandomVariable& x,
93  const Handle<YieldTermStructure>& discountCurve,
94  const boost::shared_ptr<CashFlow>& c) {
95  Real T = lgm.parametrization()->termStructure()->timeFromReference(c->date());
96  if (auto cpn = boost::dynamic_pointer_cast<Coupon>(c)) {
97  if (auto ibor = boost::dynamic_pointer_cast<IborCoupon>(c)) {
98  return (RandomVariable(x.size(), ibor->gearing()) * lgm.fixing(ibor->index(), ibor->fixingDate(), t, x) +
99  RandomVariable(x.size(), ibor->spread())) *
100  RandomVariable(x.size(), ibor->accrualPeriod() * ibor->nominal()) *
101  lgm.reducedDiscountBond(t, T, x, discountCurve);
102  } else if (auto fix = boost::dynamic_pointer_cast<FixedRateCoupon>(cpn)) {
103  return RandomVariable(x.size(), fix->amount()) * lgm.reducedDiscountBond(t, T, x, discountCurve);
104  } else if (auto on = boost::dynamic_pointer_cast<QuantExt::OvernightIndexedCoupon>(cpn)) {
105  return lgm.compoundedOnRate(boost::dynamic_pointer_cast<OvernightIndex>(on->index()), on->fixingDates(),
106  on->valueDates(), on->dt(), on->rateCutoff(), on->includeSpread(), on->spread(),
107  on->gearing(), on->lookback(), on->dayCounter(),
108  Null<Real>(), Null<Real>(), false, false, t, x) *
109  RandomVariable(x.size(), on->accrualPeriod() * on->nominal()) *
110  lgm.reducedDiscountBond(t, T, x, discountCurve);
111  } else if (auto av = boost::dynamic_pointer_cast<QuantExt::AverageONIndexedCoupon>(cpn)) {
112  return lgm.averagedOnRate(boost::dynamic_pointer_cast<OvernightIndex>(av->index()), av->fixingDates(),
113  av->valueDates(), av->dt(), av->rateCutoff(), false, av->spread(), av->gearing(),
114  av->lookback(), av->dayCounter(), Null<Real>(), Null<Real>(),
115  false, false, t, x) *
116  RandomVariable(x.size(), av->accrualPeriod() * av->nominal()) *
117  lgm.reducedDiscountBond(t, T, x, discountCurve);
118  } else if (auto bma = boost::dynamic_pointer_cast<QuantLib::AverageBMACoupon>(cpn)) {
119  return (RandomVariable(x.size(), bma->gearing()) *
120  lgm.averagedBmaRate(boost::dynamic_pointer_cast<BMAIndex>(bma->index()), bma->fixingDates(),
121  bma->accrualStartDate(), bma->accrualEndDate(), t, x) +
122  RandomVariable(x.size(), bma->spread())) *
123  RandomVariable(x.size(), bma->accrualPeriod() * bma->nominal()) *
124  lgm.reducedDiscountBond(t, T, x, discountCurve);
125  } else if (auto cf = boost::dynamic_pointer_cast<QuantLib::CappedFlooredCoupon>(cpn)) {
126  auto und = cf->underlying();
127  RandomVariable cap(x.size(), cf->cap() == Null<Real>() ? QL_MAX_REAL : cf->cap());
128  RandomVariable floor(x.size(), cf->floor() == Null<Real>() ? -QL_MAX_REAL : cf->floor());
129  if (auto undibor = boost::dynamic_pointer_cast<QuantLib::IborCoupon>(und)) {
130  return max(floor, min(cap, (RandomVariable(x.size(), undibor->gearing()) *
131  lgm.fixing(undibor->index(), undibor->fixingDate(), t, x) +
132  RandomVariable(x.size(), undibor->spread())))) *
133  RandomVariable(x.size(), undibor->accrualPeriod() * undibor->nominal()) *
134  lgm.reducedDiscountBond(t, T, x, discountCurve);
135  }
136  } else if (auto cfon = boost::dynamic_pointer_cast<QuantExt::CappedFlooredOvernightIndexedCoupon>(cpn)) {
137  auto und = cfon->underlying();
138  return lgm.compoundedOnRate(boost::dynamic_pointer_cast<OvernightIndex>(und->index()), und->fixingDates(),
139  und->valueDates(), und->dt(), und->rateCutoff(), und->includeSpread(),
140  und->spread(), und->gearing(), und->lookback(),
141  und->dayCounter(), cfon->cap(), cfon->floor(), cfon->localCapFloor(),
142  cfon->nakedOption(), t, x) *
143  RandomVariable(x.size(), cfon->accrualPeriod() * cfon->nominal()) *
144  lgm.reducedDiscountBond(t, T, x, discountCurve);
145  } else if (auto cfav = boost::dynamic_pointer_cast<QuantExt::CappedFlooredAverageONIndexedCoupon>(cpn)) {
146  auto und = cfav->underlying();
147  return lgm.averagedOnRate(boost::dynamic_pointer_cast<OvernightIndex>(und->index()), und->fixingDates(),
148  und->valueDates(), und->dt(), und->rateCutoff(), cfav->includeSpread(),
149  und->spread(), und->gearing(), und->lookback(),
150  und->dayCounter(), cfav->cap(), cfav->floor(), cfav->localCapFloor(),
151  cfav->nakedOption(), t, x) *
152  RandomVariable(x.size(), cfav->accrualPeriod() * cfav->nominal()) *
153  lgm.reducedDiscountBond(t, T, x, discountCurve);
154  }
155  THROW_ERROR_CPN_NOT_SUPPORTED;
156  } else {
157  return RandomVariable(x.size(), c->amount()) * lgm.reducedDiscountBond(t, T, x, discountCurve);
158  }
159 }

161 /* Get the rebate amount on an exercise */
162 RandomVariable getRebatePv(const LgmVectorised& lgm, const Real t, const RandomVariable& x,
163  const Handle<YieldTermStructure>& discountCurve,
164  const boost::shared_ptr<RebatedExercise>& exercise, const Date& d) {
165  if (exercise == nullptr)
166  return RandomVariable(x.size(), 0.0);
167  auto f = std::find(exercise->dates().begin(), exercise->dates().end(), d);
168  QL_REQUIRE(f != exercise->dates().end(), "NumericLgmMultiLegOptionEngine: internal error: exercise date "
169  << d << " from rebate payment not found amount exercise dates.");
170  Size index = std::distance(exercise->dates().begin(), f);
171  return RandomVariable(x.size(), exercise->rebate(index)) *
172  lgm.reducedDiscountBond(
173  t, lgm.parametrization()->termStructure()->timeFromReference(exercise->rebatePaymentDate(index)), x);
174 }

181 void NumericLgmMultiLegOptionEngineBase::calculate() const {
182 
183  /* TODO ParYieldCurve cash-settled swaptions are priced as if CollateralizedCashPrice, this can be refined. */
184 
185  /* The reference date of the valuation. Exercise dates <= refDate are treated as if not present, i.e. it is
186  assumed that these rights were not exercised.
187  TODO we should add exercise data to indicate past exercises and value outstanding cash settlement amounts,
188  or - possibly (?) - the physical underlying in that we exercised into? Or would that be a separate trade? */
189 
190  Date refDate = model()->parametrization()->termStructure()->referenceDate();
191 
192  /* we might have to handle an exercise with rebate, init a variable for that (null if not applicable) */
193 
194  auto rebatedExercise = boost::dynamic_pointer_cast<QuantExt::RebatedExercise>(exercise_);
195 
196  /* Check that the engine can handle the instrument, i.e. that there is a unique pay currency and all interest rate
197  indices are in this same currency. */
198 
199  for (Size i = 1; i < currency_.size(); ++i) {
200  QL_REQUIRE(currency_[i] == currency_[0],
201  "NumericLgmMultilegOptionEngine: can only handle single currency underlyings, got at least two pay "
202  "currencies: "
203  << currency_[0] << "; " << currency_[i]);
204  }
205 
206  for (Size i = 0; i < legs_.size(); ++i) {
207  for (Size j = 0; j < legs_[i].size(); ++j) {
208  if (auto cpn = boost::dynamic_pointer_cast<FloatingRateCoupon>(legs_[i][j])) {
209  QL_REQUIRE(cpn->index()->currency() == currency_[0],
210  "NumericLgmMultilegOptionEngine: can only handle indices ("
211  << cpn->index()->name() << ") with same currency as unqiue pay currency ("
212  << currency_[0]);
213  }
214  }
215  }
216 
217  /* Map an option date to the set of relevant cashflows to exercise into, the set will not contain cashflows
218  that are relevant for later option dates. */
219 
220  std::map<Date, std::set<std::pair<Size, Size>>> optionDates;
221 
222  for (Size i = 0; i < legs_.size(); ++i) {
223  for (Size j = 0; j < legs_[i].size(); ++j) {
224  for (Size k = exercise_->dates().size(); k > 0; --k) {
225  Date exerciseDate = exercise_->dates()[k - 1];
226  if (isCashflowRelevantForExercise(refDate, exerciseDate, legs_[i][j])) {
227  optionDates[exerciseDate].insert(std::make_pair(i, j));
228  break;
229  }
230  }
231  }
232  }
233 
234  /* map a actual simulation date => set of (cf leg, cf number) */
235 
236  std::map<Date, std::set<std::pair<Size, Size>>> cashflowDates;
237 
238  /* Each underlying cashflow has a date on which we can efficiently calculate its amount, depending on
239  the option date for which the cashflow is relevant. We estimate a cashflow amount for the latest
240  option date it is relevant for and then roll it back. */
241 
242  for (Size i = 0; i < legs_.size(); ++i) {
243  for (Size j = 0; j < legs_[i].size(); ++j) {
244  Date latestOptionDate = Date::minDate();
245  for (auto const& d : optionDates) {
246  if (d.second.find(std::make_pair(i, j)) != d.second.end())
247  latestOptionDate = std::max(latestOptionDate, d.first);
248  }
249  auto d = getSimulationDates(refDate, latestOptionDate, legs_[i][j]);
250  if (d != Null<Date>())
251  cashflowDates[d].insert(std::make_pair(i, j));
252  }
253  }
254 
255  /* Compile the list of simulation dates, include the ref date always */
256 
257  std::set<Date> simulationDates;
258 
259  for (auto const& d : optionDates)
260  simulationDates.insert(d.first);
261 
262  for (auto const& d : cashflowDates)
263  simulationDates.insert(d.first);
264 
265  simulationDates.insert(refDate);
266 
267  /* Step backwards through simulation dates and compute the option npv */
268 
269  LgmVectorised lgm(model()->parametrization());
270 
271  // the current option npv that we roll back through the grid
272  RandomVariable optionNpv(gridSize(), 0.0);
273  // cashflow npvs that are part of the latest exercise into option
274  RandomVariable underlyingNpv1(gridSize(), 0.0);
275  // cashflow npvs that are estimated, but not yet part of the latest exercise into option
276  std::map<std::pair<Size, Size>, RandomVariable> underlyingNpv2;
277 
278  for (auto it = simulationDates.rbegin(); it != simulationDates.rend(); ++it) {
279 
280  Real t_from = model()->parametrization()->termStructure()->timeFromReference(*it);
281  Real t_to = (it != std::next(simulationDates.rend(), -1))
282  ? model()->parametrization()->termStructure()->timeFromReference(*std::next(it, 1))
283  : t_from;
284 
285  RandomVariable state = stateGrid(t_from);
286 
287  auto option = optionDates.find(*it);
288  auto cashflow = cashflowDates.find(*it);
289 
290  // collect cashflow amounts on simulation date
291 
292  if (cashflow != cashflowDates.end()) {
293  for (auto const& cf : cashflow->second) {
294  auto tmp = getUnderlyingCashflowPv(lgm, t_from, state, discountCurve_, legs_[cf.first][cf.second]) *
295  RandomVariable(gridSize(), payer_[cf.first]);
296  auto it = underlyingNpv2.find(cf);
297  if (it != underlyingNpv2.end())
298  it->second += tmp;
299  else
300  underlyingNpv2[cf] = tmp;
301  }
302  }
303 
304  // handle option date
305 
306  if (option != optionDates.end()) {
307 
308  // transfer relevant cashflow amounts to exercise into npv
309 
310  for (auto const& cf : option->second) {
311  underlyingNpv1 += underlyingNpv2[cf];
312  underlyingNpv2.erase(cf);
313  }
314 
315  // update the option value (exercise value including rebate vs. continuation value)
316 
317  auto rebateNpv = getRebatePv(lgm, t_from, state, discountCurve_, rebatedExercise, option->first);
318  optionNpv = max(optionNpv, underlyingNpv1 + rebateNpv);
319  }
320 
321  // rollback
322 
323  if (t_from != t_to) {
324  underlyingNpv1 = rollback(underlyingNpv1, t_from, t_to);
325  for (auto& c : underlyingNpv2) {
326  c.second = rollback(c.second, t_from, t_to);
327  }
328  optionNpv = rollback(optionNpv, t_from, t_to);
329  }
330  }
331 
332  /* Set the results */
333 
334  npv_ = optionNpv.at(0);
335  underlyingNpv_ = underlyingNpv1.at(0);
336  for (auto const& c : underlyingNpv2) {
337  underlyingNpv_ += c.second.at(0);
338  }
339 
340  additionalResults_ = getAdditionalResultsMap(model()->getCalibrationInfo());
341 
342 } // NumericLgmMultiLegOptionEngineBase::calculate()
