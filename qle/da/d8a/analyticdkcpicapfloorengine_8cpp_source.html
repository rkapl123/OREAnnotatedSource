32 void AnalyticDkCpiCapFloorEngine::calculate() const {
33 
34  bool interpolate = arguments_.observationInterpolation == CPI::Linear ||
35  (arguments_.observationInterpolation == CPI::AsIndex && arguments_.infIndex->interpolated());
36  Real t = inflationYearFraction(arguments_.infIndex->frequency(), interpolate,
37  model_->infdk(index_)->termStructure()->dayCounter(),
38  model_->infdk(index_)->termStructure()->baseDate(), arguments_.fixDate);
39 
40  if (t <= 0.0) {
41  // option is expired, we do not value any possibly non settled
42  // flows, i.e. set the npv to zero in this case
43  results_.value = 0.0;
44  return;
45  }
46 
47  // Book, 13.37, 13.38
48 
49  const CrossAssetModel* x = model_.get();
50  Real k = std::pow(1.0 + arguments_.strike, t);
51  Real kTilde = k * arguments_.baseCPI;
52  Real nTilde = arguments_.nominal / arguments_.baseCPI;
53 
54  Real m = baseCPI_ * std::pow(1.0 + model_->infdk(index_)->termStructure()->zeroRate(arguments_.fixDate, Period(0, Days)), t);
55  m = arguments_.infIndex->fixing(arguments_.fixDate);
56  Real Ht = Hy(index_).eval(x, t);
57  Real v = Ht * Ht * zetay(index_).eval(x, t) -
58  2.0 * Ht * integral(x, P(Hy(index_), ay(index_), ay(index_)), 0.0, t) +
59  integral(x, P(Hy(index_), Hy(index_), ay(index_), ay(index_)), 0.0, t);
60 
61  Size irIdx = x->ccyIndex(x->infdk(index_)->currency());
62  Real discount = model_->irlgm1f(irIdx)->termStructure()->discount(arguments_.payDate);
63 
64  results_.value = nTilde * blackFormula(arguments_.type, kTilde, m, std::sqrt(v), discount);
65 
66 } // calculate()
