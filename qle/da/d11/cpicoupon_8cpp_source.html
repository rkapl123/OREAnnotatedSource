34  Rate CPICoupon::rate() const {
35  Rate r = QuantLib::CPICoupon::rate() ;
36  if (subtractInflationNominal_) {
37  Rate adjusted_r = r / fixedRate_ - spread_;
38  r = (adjusted_r - 1) * fixedRate_ + spread_;
39  }
40  return r;

111 Rate CappedFlooredCPICoupon::rate() const {
112  // rate = spread + fixedRate * capped/floored index
113  boost::shared_ptr<BlackCPICouponPricer> blackPricer = boost::dynamic_pointer_cast<BlackCPICouponPricer>(pricer_);
114  QL_REQUIRE(blackPricer, "BlackCPICouponPricer expected");
115  Real capValue = 0.0, floorValue = 0.0;
116  if (isCapped_) {
117  cpiCap_->setPricingEngine(blackPricer->engine());
118  capValue = cpiCap_->NPV();
119  }
120  if (isFloored_) {
121  cpiFloor_->setPricingEngine(blackPricer->engine());
122  floorValue = cpiFloor_->NPV();
123  }
124  Real discount = blackPricer->yieldCurve()->discount(underlying_->date());
125  Real nominal = underlying_->nominal();
126  // normalize, multiplication with nominal, year fraction, discount follows
127  Real capAmount = capValue / (nominal * discount);
128  Real floorAmount = floorValue / (nominal * discount);
129 
130  Rate swapletRate = underlying_->rate();
131  // apply gearing
132  Rate floorletRate = floorAmount * underlying_->fixedRate();
133  Rate capletRate = capAmount * underlying_->fixedRate();
134  // long floor, short cap
135  Rate totalRate = swapletRate + floorletRate - capletRate;
136 
137  return totalRate;
138 }

203 Real CappedFlooredCPICashFlow::amount() const {
204  QL_REQUIRE(pricer_, "pricer not set for capped/floored CPI cashflow");
205  boost::shared_ptr<BlackCPICashFlowPricer> blackPricer =
206  boost::dynamic_pointer_cast<BlackCPICashFlowPricer>(pricer_);
207  Real capValue = 0.0, floorValue = 0.0;
208  if (isCapped_) {
209  cpiCap_->setPricingEngine(blackPricer->engine());
210  capValue = cpiCap_->NPV();
211  }
212  if (isFloored_) {
213  cpiFloor_->setPricingEngine(blackPricer->engine());
214  floorValue = cpiFloor_->NPV();
215  }
216  Real discount = blackPricer->yieldCurve()->discount(underlying_->date());
217  Real capAmount = capValue / discount;
218  Real floorAmount = floorValue / discount;
219  Real underlyingAmount = underlying_->amount();
220  Real totalAmount = underlyingAmount - capAmount + floorAmount;
221  return totalAmount;
222 }

348 CPILeg::operator Leg() const {
349 
350  QL_REQUIRE(!notionals_.empty(), "no notional given");
351  Size n = schedule_.size() - 1;
352  Leg leg;
353  leg.reserve(n + 1); // +1 for notional, we always have some sort ...
354  if (n > 0) {
355  QL_REQUIRE(!fixedRates_.empty() || !spreads_.empty(), "no fixedRates or spreads given");
356 
357  Date refStart, start, refEnd, end;
358 
359  for (Size i = 0; i < n; ++i) {
360  refStart = start = schedule_.date(i);
361  refEnd = end = schedule_.date(i + 1);
362  Date paymentDate = paymentCalendar_.adjust(end, paymentAdjustment_);
363 
364  Date exCouponDate;
365  if (exCouponPeriod_ != Period()) {
366  exCouponDate =
367  exCouponCalendar_.advance(paymentDate, -exCouponPeriod_, exCouponAdjustment_, exCouponEndOfMonth_);
368  }
369 
370  if (i == 0 && schedule_.hasIsRegular() && !schedule_.isRegular(i + 1)) {
371  BusinessDayConvention bdc = schedule_.businessDayConvention();
372  refStart = schedule_.calendar().adjust(end - schedule_.tenor(), bdc);
373  }
374  if (i == n - 1 && schedule_.hasIsRegular() && !schedule_.isRegular(i + 1)) {
375  BusinessDayConvention bdc = schedule_.businessDayConvention();
376  refEnd = schedule_.calendar().adjust(start + schedule_.tenor(), bdc);
377  }
378  if (detail::get(fixedRates_, i, 1.0) == 0.0) { // fixed coupon
379  leg.push_back(ext::shared_ptr<CashFlow>(
380  new FixedRateCoupon(paymentDate, detail::get(notionals_, i, 0.0),
381  detail::effectiveFixedRate(spreads_, caps_, floors_, i), paymentDayCounter_,
382  start, end, refStart, refEnd, exCouponDate)));
383  } else { // zero inflation coupon
384  ext::shared_ptr<CPICoupon> coup;
385 
386  coup = ext::make_shared<CPICoupon>(
387  baseCPI_, // all have same base for ratio
388  paymentDate, detail::get(notionals_, i, 0.0), start, end, detail::get(fixingDays_, i, 0.0), index_,
389  observationLag_, observationInterpolation_, paymentDayCounter_, detail::get(fixedRates_, i, 0.0),
390  detail::get(spreads_, i, 0.0), refStart, refEnd, exCouponDate, subtractInflationNominalAllCoupons_);
391 
392  // set a pricer for the underlying coupon straight away because it only provides computation - not data
393  ext::shared_ptr<CPICouponPricer> pricer(new CPICouponPricer(Handle<YieldTermStructure>(rateCurve_)));
394  coup->setPricer(pricer);
395 
396  if (detail::noOption(caps_, floors_, i)) { // just swaplet
397  leg.push_back(ext::dynamic_pointer_cast<CashFlow>(coup));
398  } else { // cap/floorlet
399  ext::shared_ptr<CappedFlooredCPICoupon> cfCoup = ext::make_shared<CappedFlooredCPICoupon>(
400  coup, startDate_, detail::get(caps_, i, Null<Rate>()), detail::get(floors_, i, Null<Rate>()));
401  // in this case we need to set the "outer" pricer later that handles cap and floor
402  leg.push_back(ext::dynamic_pointer_cast<CashFlow>(cfCoup));
403  }
404  }
405  }
406  }
407 
408  // in CPI legs you always have a notional flow of some sort
409  Date paymentDate = paymentCalendar_.adjust(schedule_.date(n), paymentAdjustment_);
410  Date fixingDate = paymentDate - observationLag_;
411 
412  ext::shared_ptr<CPICashFlow> xnl = ext::make_shared<CPICashFlow>(
413  detail::get(notionals_, n, 0.0), index_, startDate_ - observationLag_, baseCPI_, fixingDate, paymentDate,
414  subtractInflationNominal_, observationInterpolation_, index_->frequency());
415 
416  if (finalFlowCap_ == Null<Real>() && finalFlowFloor_ == Null<Real>()) {
417  leg.push_back(xnl);
418  } else {
419  ext::shared_ptr<CappedFlooredCPICashFlow> cfxnl = ext::make_shared<CappedFlooredCPICashFlow>(
420  xnl, startDate_, observationLag_, finalFlowCap_, finalFlowFloor_);
421  leg.push_back(cfxnl);
422  }
423 
424  return leg;
425 }
