36 class LinkableCalibratedModel::CalibrationFunction : public CostFunction {
37 public:
38  CalibrationFunction(LinkableCalibratedModel* model, const vector<shared_ptr<CalibrationHelper> >& h,
39  const vector<Real>& weights, const Projection& projection)
40  : model_(model, no_deletion), instruments_(h), weights_(weights), projection_(projection) {}
41 
42  virtual ~CalibrationFunction() {}

44  virtual Real value(const Array& params) const {
45  model_->setParams(projection_.include(params));
46  Real value = 0.0;
47  for (Size i = 0; i < instruments_.size(); i++) {
48  Real diff = instruments_[i]->calibrationError();
49  value += diff * diff * weights_[i];
50  }
51  return std::sqrt(value);
52  }
53 
54  virtual Disposable<Array> values(const Array& params) const {
55  model_->setParams(projection_.include(params));
56  Array values(instruments_.size());
57  for (Size i = 0; i < instruments_.size(); i++) {
58  values[i] = instruments_[i]->calibrationError() * std::sqrt(weights_[i]);
59  }
60  return values;
61  }

63  virtual Real finiteDifferenceEpsilon() const { return 1e-6; }

82 void LinkableCalibratedModel::calibrate(const vector<shared_ptr<CalibrationHelper> >& instruments,
83  OptimizationMethod& method, const EndCriteria& endCriteria,
84  const Constraint& additionalConstraint, const vector<Real>& weights,
85  const vector<bool>& fixParameters) {
86 
87  QL_REQUIRE(weights.empty() || weights.size() == instruments.size(), "mismatch between number of instruments ("
88  << instruments.size() << ") and weights("
89  << weights.size() << ")");
90 
91  Constraint c;
92  if (additionalConstraint.empty())
93  c = *constraint_;
94  else
95  c = CompositeConstraint(*constraint_, additionalConstraint);
96  vector<Real> w = weights.empty() ? vector<Real>(instruments.size(), 1.0) : weights;
97 
98  Array prms = params();
99  vector<bool> all(prms.size(), false);
100  Projection proj(prms, fixParameters.size() > 0 ? fixParameters : all);
101  CalibrationFunction f(this, instruments, w, proj);
102  ProjectedConstraint pc(c, proj);
103  Problem prob(f, pc, proj.project(prms));
104  endCriteria_ = method.minimize(prob, endCriteria);
105  Array result(prob.currentValue());
106  setParams(proj.include(result));
107  problemValues_ = prob.values(result);
108 
109  notifyObservers();
110 }
