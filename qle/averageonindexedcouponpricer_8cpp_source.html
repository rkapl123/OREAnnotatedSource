36Rate AverageONIndexedCouponPricer::swapletRate() const {
37
38 std::vector<Date> fixingDates = coupon_->fixingDates();
39 std::vector<Time> accrualFractions = coupon_->dt();
40 Size numPeriods = accrualFractions.size();
41 Real accumulatedRate = 0;
42 QL_REQUIRE(coupon_->rateCutoff() < numPeriods,
43 "rate cutoff (" << coupon_->rateCutoff() << ") must be less than number of fixings in period ("
44 << numPeriods << ")");
45 Size nCutoff = numPeriods - coupon_->rateCutoff();
46
47 if (approximationType_ == Takada) {
48 Size i = 0;
49 Date valuationDate = Settings::instance().evaluationDate();
50 // Deal with past fixings.
51 while (i < numPeriods && fixingDates[std::min(i, nCutoff)] < valuationDate) {
52 Rate pastFixing = overnightIndex_->fixing(fixingDates[std::min(i, nCutoff)]);
53 QL_REQUIRE(pastFixing != Null<Real>(),
54 "Missing " << overnightIndex_->name() << " fixing for " << fixingDates[std::min(i, nCutoff)]);
55 accumulatedRate += pastFixing * accrualFractions[i];
56 ++i;
57 }
58 // Use valuation date's fixing also if available.
59 if (i < numPeriods && fixingDates[std::min(i, nCutoff)] == valuationDate) {
60 Rate valuationDateFixing = IndexManager::instance().getHistory(overnightIndex_->name())[valuationDate];
61 if (valuationDateFixing != Null<Real>()) {
62 accumulatedRate += valuationDateFixing * accrualFractions[i];
63 ++i;
64 }
65 }
66 // Use Takada approximation (2011) for forecasting.
67 if (i < numPeriods) {
68 Handle<YieldTermStructure> projectionCurve = overnightIndex_->forwardingTermStructure();
69 QL_REQUIRE(!projectionCurve.empty(),
70 "Null term structure set to this instance of " << overnightIndex_->name());
71
72 // we know that nCutoff >= i because either
73 // a) i = today, but if nCutoff < today, there is no forward part
74 // b) i = today + 1bd, the we have a historical fixing available on today and if nCutoff = today
75 // again there is no forward part
76 QL_REQUIRE(nCutoff >= i,
77 "internal inconsistency: nCutoff (" << nCutoff << ") >= i (" << i << ") expected.");
78
79 // handle the part until the rate cutoff (might be empty, i.e. startForecast = endForecast)
80 Date startForecast = coupon_->valueDates()[i];
81 Date endForecast = coupon_->valueDates()[nCutoff];
82 DiscountFactor startDiscount = projectionCurve->discount(startForecast);
83 DiscountFactor endDiscount = projectionCurve->discount(endForecast);
84
85 // handle the rate cutoff period (if there is any, i.e. if nCutoff < n)
86 if (nCutoff < numPeriods) {
87 // forward discount factor for one calendar day on the cutoff date
88 DiscountFactor discountCutoffDate = projectionCurve->discount(coupon_->valueDates()[nCutoff] + 1) /
89 projectionCurve->discount(coupon_->valueDates()[nCutoff]);
90 // keep the above forward discount factor constant during the cutoff period
91 endDiscount *=
92 std::pow(discountCutoffDate, coupon_->valueDates()[numPeriods] - coupon_->valueDates()[nCutoff]);
93 }
94
95 accumulatedRate += log(startDiscount / endDiscount);
96 }
97 } else if (approximationType_ == None) {
98 std::vector<Rate> fixings = coupon_->indexFixings();
99 for (Size i = 0; i < numPeriods; ++i) {
100 accumulatedRate += fixings[i] * accrualFractions[i];
101 }
102 } else {
103 QL_FAIL("Invalid Approximation for AverageONIndexedCouponPricer");
104 }
105 // Return factor * rate + spread
106 Rate tau = coupon_->dayCounter().yearFraction(coupon_->valueDates().front(), coupon_->valueDates().back());
107 Rate rate = gearing_ * accumulatedRate / tau + spread_;
108 return rate;
109}
