79 boost::shared_ptr<SmileSection> SpreadedSwaptionVolatility::smileSectionImpl(Time optionTime, Time swapLength) const {
80  calculate();
81  auto baseSection = base_->smileSection(optionTime, swapLength);
82  Real atmLevel = Null<Real>();
83  if (baseSection->atmLevel() == Null<Real>() && strikeSpreads_.size() > 1) {
84  // determine atm level if it is required (more than one strike spread) and the base section does not have it
85  QL_REQUIRE(swapIndexBase_ != nullptr,
86  "SpreadedSwaptionVolatility::smileSectionImpl: require swapIndexBase, since the base vol smile "
87  "section does not provide an atm level and we have more than one strike spread given ("
88  << strikeSpreads_.size() << ")");
89  Date optionDate = optionDateFromTime(optionTime);
90  Rounding rounder(0);
91  Period swapTenor(static_cast<Integer>(rounder(swapLength * 12.0)), Months);
92  optionDate = swapTenor > shortSwapIndexBase_->tenor()
93  ? swapIndexBase_->fixingCalendar().adjust(optionDate, Following)
94  : shortSwapIndexBase_->fixingCalendar().adjust(optionDate, Following);
95  if (swapTenor > shortSwapIndexBase_->tenor()) {
96  atmLevel = swapIndexBase_->clone(swapTenor)->fixing(optionDate);
97  } else {
98  atmLevel = shortSwapIndexBase_->clone(swapTenor)->fixing(optionDate);
99  }
100  }
101  // interpolate vol spreads
102  std::vector<Real> volSpreads(strikeSpreads_.size());
103  for (Size k = 0; k < volSpreads.size(); ++k) {
104  volSpreads[k] = volSpreadInterpolation_[k](swapLength, optionTime);
105  }
106  // create smile section
107  return boost::make_shared<SpreadedSmileSection2>(base_->smileSection(optionTime, swapLength), volSpreads,
108  strikeSpreads_, true, atmLevel);
109 }

111 Volatility SpreadedSwaptionVolatility::volatilityImpl(Time optionTime, Time swapLength, Rate strike) const {
112  if (strike == Null<Real>()) {
113  calculate();
114  // support input strike null, interpret this as atm
115  std::vector<Real> volSpreads(strikeSpreads_.size());
116  for (Size k = 0; k < volSpreads.size(); ++k) {
117  volSpreads[k] = volSpreadInterpolation_[k](swapLength, optionTime);
118  }
119  Real volSpread;
120  if (volSpreads.size() > 1) {
121  auto spreadInterpolation =
122  LinearFlat().interpolate(strikeSpreads_.begin(), strikeSpreads_.end(), volSpreads.begin());
123  volSpread = spreadInterpolation(0.0);
124  } else {
125  volSpread = volSpreads.front();
126  }
127  return base_->volatility(optionTime, swapLength, strike) + volSpread;
128  } else {
129  // if input strike != null, use smile section implementation
130  return smileSectionImpl(optionTime, swapLength)->volatility(strike);
131  }
132 }

134 void SpreadedSwaptionVolatility::performCalculations() const {
135  SwaptionVolatilityDiscrete::performCalculations();
136  for (Size k = 0; k < strikeSpreads_.size(); ++k) {
137  for (Size i = 0; i < optionTenors_.size(); ++i) {
138  for (Size j = 0; j < swapTenors_.size(); ++j) {
139  Size index = i * swapTenors_.size() + j;
140  QL_REQUIRE(!volSpreads_[index][k].empty(), "SpreadedSwaptionVolatility: vol spread quote at index ("
141  << i << "," << j << "," << k << ") is empty");
142  volSpreadValues_[k](i, j) = volSpreads_[index][k]->value();
143  }
144  }
145  volSpreadInterpolation_[k] = FlatExtrapolator2D(boost::make_shared<BilinearInterpolation>(
146  swapLengths_.begin(), swapLengths_.end(), optionTimes_.begin(), optionTimes_.end(), volSpreadValues_[k]));
147  volSpreadInterpolation_[k].enableExtrapolation();
148  }
149 }
