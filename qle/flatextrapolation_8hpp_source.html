37 //! Flat extrapolation given a base interpolation
38 /*! \ingroup math
39  */
40 class FlatExtrapolation : public Interpolation {
41 private:
42  class FlatExtrapolationImpl : public Interpolation::Impl {
43 
44  public:
45  FlatExtrapolationImpl(const boost::shared_ptr<Interpolation>& i) : i_(i) {}
46  void update() { i_->update(); }
47  Real xMin() const { return i_->xMin(); }
48  Real xMax() const { return i_->xMax(); }
49  std::vector<Real> xValues() const { QL_FAIL("not implemented"); }
50  std::vector<Real> yValues() const { QL_FAIL("not implemented"); }
51  bool isInRange(Real x) const { return i_->isInRange(x); }
52  Real value(Real x) const {
53  Real tmp = std::max(std::min(x, i_->xMax()), i_->xMin());
54  return i_->operator()(tmp);
55  }
56  Real primitive(Real x) const {
57  if (x >= i_->xMin() && x <= i_->xMax()) {
58  return i_->primitive(x);
59  }
60  if (x < i_->xMin()) {
61  return i_->primitive(i_->xMin()) - (i_->xMin() - x);
62  } else {
63  return i_->primitive(i_->xMax()) + (x - i_->xMax());
64  }
65  }
66  Real derivative(Real x) const {
67  if (x > i_->xMin() && x < i_->xMax()) {
68  return i_->derivative(x);
69  } else {
70  // that is the left derivative for xmin and
71  // the right derivative for xmax
72  return 0.0;
73  }
74  }
75  Real secondDerivative(Real x) const {
76  if (x > i_->xMin() && x < i_->xMax()) {
77  return i_->secondDerivative(x);
78  } else {
79  // that is the left derivative for xmin and
80  // the right derivative for xmax
81  return 0.0;
82  }
83  }
84 
85  private:
86  const boost::shared_ptr<Interpolation> i_;
87  };
88 
89 public:
90  FlatExtrapolation(const boost::shared_ptr<Interpolation>& i) {
91  impl_ = boost::make_shared<FlatExtrapolationImpl>(i);
92  impl_->update();
93  }
94 };
