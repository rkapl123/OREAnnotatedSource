49 void SubPeriodsSwapHelper::initializeDates() {
50 
51  // build swap
52  Date valuationDate = Settings::instance().evaluationDate();
53  Calendar spotCalendar = iborIndex_->fixingCalendar();
54  Natural spotDays = iborIndex_->fixingDays();
55  // move val date forward in case it is a holiday
56  valuationDate = spotCalendar.adjust(valuationDate);
57  Date effectiveDate = spotCalendar.advance(valuationDate, spotDays * Days);
58 
59  swap_ = boost::shared_ptr<SubPeriodsSwap>(new SubPeriodsSwap(
60  effectiveDate, 1.0, swapTenor_, true, fixedTenor_, 0.0, fixedCalendar_, fixedDayCount_, fixedConvention_,
61  floatPayTenor_, iborIndex_, floatDayCount_, DateGeneration::Backward, type_));
62 
63  boost::shared_ptr<PricingEngine> engine(new DiscountingSwapEngine(discountRelinkableHandle_));
64  swap_->setPricingEngine(engine);
65 
66  // set earliest and latest
67  earliestDate_ = swap_->startDate();
68  latestDate_ = swap_->maturityDate();
69 
70  boost::shared_ptr<FloatingRateCoupon> lastFloating =
71  boost::dynamic_pointer_cast<FloatingRateCoupon>(swap_->floatLeg().back());
72 #ifdef QL_USE_INDEXED_COUPON
73  /* May need to adjust latestDate_ if you are projecting libor based
74  on tenor length rather than from accrual date to accrual date. */
75  Date fixingValueDate = iborIndex_->valueDate(lastFloating->fixingDate());
76  Date endValueDate = iborIndex_->maturityDate(fixingValueDate);
77  latestDate_ = std::max(latestDate_, endValueDate);
78 #else
79  /* Subperiods coupons do not have a par approximation either... */
80  if (boost::dynamic_pointer_cast<QuantExt::SubPeriodsCoupon1>(lastFloating)) {
81  Date fixingValueDate = iborIndex_->valueDate(lastFloating->fixingDate());
82  Date endValueDate = iborIndex_->maturityDate(fixingValueDate);
83  latestDate_ = std::max(latestDate_, endValueDate);
84  }
85 #endif
86 }
