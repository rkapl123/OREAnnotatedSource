78void SwaptionVolatilityConverter::checkInputs() const {
79 QL_REQUIRE(svsIn_->referenceDate() == asof_,
80 "SwaptionVolatilityConverter requires the asof date and reference date to align");
81 QL_REQUIRE(!discount_.empty() && discount_->referenceDate() == asof_,
82 "SwaptionVolatilityConverter requires a valid discount curve with reference date equal to asof date");
83 Handle<YieldTermStructure> forwardCurve = conventions_->floatIndex()->forwardingTermStructure();
84 QL_REQUIRE(!forwardCurve.empty() && forwardCurve->referenceDate() == asof_,
85 "SwaptionVolatilityConverter requires a valid forward curve with reference date equal to asof date");

88boost::shared_ptr<SwaptionVolatilityStructure> SwaptionVolatilityConverter::convert() const {
89
90 boost::shared_ptr<SwaptionVolatilityDiscrete> svDisc;
91
92 // We expect either the wrapper adding ATM to a cube or a swaption vol discrete
93 // instance (like a matrix, a regular cube).
94
95 vector<Real> strikeSpreads(1, 0.0);
96 boost::shared_ptr<SwapIndex> swapIndexBase, shortSwapIndexBase;
97 boost::shared_ptr<SwaptionVolatilityCube> cube;
98 if (boost::dynamic_pointer_cast<SwaptionVolCubeWithATM>(svsIn_)) {
99 cube = boost::static_pointer_cast<SwaptionVolCubeWithATM>(svsIn_)->cube();
100 }
101 if (boost::dynamic_pointer_cast<SwaptionVolatilityCube>(svsIn_)) {
102 cube = boost::static_pointer_cast<SwaptionVolatilityCube>(svsIn_);
103 }
104 if (cube) {
105 svDisc = cube;
106 strikeSpreads = cube->strikeSpreads();
107 swapIndexBase = cube->swapIndexBase();
108 shortSwapIndexBase = cube->shortSwapIndexBase();
109 } else if (boost::dynamic_pointer_cast<SwaptionVolatilityDiscrete>(svsIn_)) {
110 svDisc = boost::static_pointer_cast<SwaptionVolatilityDiscrete>(svsIn_);
111 } else {
112 QL_FAIL("SwaptionVolatilityConverter: unknown input volatility structure");
113 }
114
115 // Some aspects of original volatility structure that we will need
116 DayCounter dayCounter = svDisc->dayCounter();
117 bool extrapolation = svDisc->allowsExtrapolation();
118 Calendar calendar = svDisc->calendar();
119 BusinessDayConvention bdc = svDisc->businessDayConvention();
120
121 const vector<Date>& optionDates = svDisc->optionDates();
122 const vector<Period>& optionTenors = svDisc->optionTenors();
123 const vector<Period>& swapTenors = svDisc->swapTenors();
124 const vector<Time>& optionTimes = svDisc->optionTimes();
125 const vector<Time>& swapLengths = svDisc->swapLengths();
126 Size nOptionTimes = optionTimes.size();
127 Size nSwapLengths = swapLengths.size();
128 Size nStrikeSpreads = strikeSpreads.size();
129
130 Real targetShift = 0.0;
131
132 // If target type is ShiftedLognormal and shifts are provided, check size
133 if (targetType_ == ShiftedLognormal && !targetShifts_.empty()) {
134 QL_REQUIRE(targetShifts_.rows() == nOptionTimes,
135 "SwaptionVolatilityConverter: number of shift rows does not equal the number of option tenors");
136 QL_REQUIRE(targetShifts_.columns() == nSwapLengths,
137 "SwaptionVolatilityConverter: number of shift columns does not equal the number of swap tenors");
138 }
139
140 // Calculate the converted ATM volatilities
141 Matrix volatilities(nOptionTimes, nSwapLengths);
142 for (Size i = 0; i < nOptionTimes; ++i) {
143 for (Size j = 0; j < nSwapLengths; ++j) {
144 if (!targetShifts_.empty())
145 targetShift = targetShifts_[i][j];
146 volatilities[i][j] = convert(optionDates[i], swapTenors[j], 0.0, dayCounter, targetType_, targetShift);
147 }
148 }
149
150 // Build ATM matrix
151 Handle<SwaptionVolatilityStructure> atmStructure;
152 if (calendar.empty() || optionTenors.empty()) {
153 // Original matrix was created with fixed option dates
154 atmStructure = Handle<SwaptionVolatilityStructure>(
155 boost::make_shared<SwaptionVolatilityMatrix>(asof_, calendar, bdc, optionDates, swapTenors, volatilities,
156 Actual365Fixed(), extrapolation, targetType_, targetShifts_));
157 } else {
158 atmStructure = Handle<SwaptionVolatilityStructure>(boost::shared_ptr<SwaptionVolatilityMatrix>(
159 new SwaptionVolatilityMatrix(asof_, calendar, bdc, optionTenors, swapTenors, volatilities, Actual365Fixed(),
160 extrapolation, targetType_, targetShifts_)));
161 }
162
163 // no cube input => we are done
164 if (!cube)
165 return *atmStructure;
166
167 // convert non-ATM volatilities, note that we use the ATM option tenors and swap tenors here!
168
169 std::vector<std::vector<Handle<Quote> > > volSpreads(nOptionTimes * nSwapLengths,
170 std::vector<Handle<Quote> >(nStrikeSpreads));
171 for (Size k = 0; k < nStrikeSpreads; ++k) {
172 for (Size i = 0; i < nOptionTimes; ++i) {
173 for (Size j = 0; j < nSwapLengths; ++j) {
174 if (!targetShifts_.empty())
175 targetShift = targetShifts_[i][j];
176 Real outVol =
177 convert(optionDates[i], swapTenors[j], strikeSpreads[k], dayCounter, targetType_, targetShift);
178 volSpreads[i * nSwapLengths + j][k] = Handle<Quote>(boost::make_shared<SimpleQuote>(outVol));
179 }
180 }
181 }
182
183 // Build and return cube, note that we hardcode flat extrapolation
184 boost::shared_ptr<SwaptionVolatilityCube> cubeOut = boost::shared_ptr<QuantExt::SwaptionVolCube2>(
185 new QuantExt::SwaptionVolCube2(atmStructure, optionTenors, swapTenors, strikeSpreads, volSpreads, swapIndexBase,
186 shortSwapIndexBase, false, true, false));
187 cubeOut->enableExtrapolation(cube->allowsExtrapolation());
188 return boost::make_shared<SwaptionVolCubeWithATM>(cubeOut);
189

197Real SwaptionVolatilityConverter::convert(const Date& expiry, const Period& swapTenor, Real strikeSpread,
198 const DayCounter& volDayCounter, VolatilityType outType,
199 Real outShift) const {
200
201 const boost::shared_ptr<SwapConventions> tmpConv =
202 swapTenor <= shortConventionsTenor_ ? shortConventions_ : conventions_;
203 Handle<YieldTermStructure> tmpDiscount = swapTenor <= shortConventionsTenor_ ? shortDiscount_ : discount_;
204
205 // Create the underlying swap with fixed rate = fair rate
206 // We rely on the fact that MakeVanillaSwap sets the fixed rate to the fair rate if it is left null in the ctor
207 Date effectiveDate = tmpConv->fixedCalendar().advance(expiry, tmpConv->settlementDays(), Days);
208 boost::shared_ptr<PricingEngine> engine = boost::make_shared<DiscountingSwapEngine>(tmpDiscount);
209 boost::shared_ptr<VanillaSwap> swap = MakeVanillaSwap(swapTenor, tmpConv->floatIndex())
210 .withType(strikeSpread < 0.0 ? VanillaSwap::Receiver : VanillaSwap::Payer)
211 .withEffectiveDate(effectiveDate)
212 .withFixedLegCalendar(tmpConv->fixedCalendar())
213 .withFixedLegDayCount(tmpConv->fixedDayCounter())
214 .withFixedLegTenor(tmpConv->fixedTenor())
215 .withFixedLegConvention(tmpConv->fixedConvention())
216 .withFixedLegTerminationDateConvention(tmpConv->fixedConvention())
217 .withFloatingLegSpread(0.0)
218 .withPricingEngine(engine);
219 // we need this also for non-atm swaps
220 Rate atmRate = swap->fairRate(), strike;
221 if (!close_enough(strikeSpread, 0.0)) {
222 strike = atmRate + strikeSpread;
223 swap = MakeVanillaSwap(swapTenor, tmpConv->floatIndex(), strike)
224 .withEffectiveDate(effectiveDate)
225 .withFixedLegTenor(tmpConv->fixedTenor())
226 .withFixedLegDayCount(tmpConv->fixedDayCounter())
227 .withFloatingLegSpread(0.0)
228 .withPricingEngine(engine);
229 } else {
230 strike = atmRate;
231 }
232
233 Real inShift = svsIn_->shift(expiry, swapTenor);
234 VolatilityType inType = svsIn_->volatilityType();
235
236 // if strike is invalid w.r.t. the given input or output vol types, return zero vol
237 // (for a lognormal cube e.g. it is common that some effective strikes are negative)
238 Real inMinStrike = inType == ShiftedLognormal ? -inShift : -QL_MAX_REAL;
239 Real outMinStrike = outType == ShiftedLognormal ? -outShift : -QL_MAX_REAL;
240 if (strike < inMinStrike || strike < outMinStrike)
241 return 0.0;
242
243 Real inVol = svsIn_->volatility(expiry, swapTenor, strike);
244
245 // Create the swaption
246 boost::shared_ptr<Exercise> exercise = boost::make_shared<EuropeanExercise>(expiry);
247 boost::shared_ptr<Swaption> swaption = boost::make_shared<Swaption>(swap, exercise, Settlement::Physical);
248
249 // Price the swaption with the input volatility
250 boost::shared_ptr<PricingEngine> swaptionEngine;
251 if (inType == ShiftedLognormal) {
252 swaptionEngine = boost::make_shared<BlackSwaptionEngine>(discount_, inVol, volDayCounter, inShift);
253 } else {
254 swaptionEngine = boost::make_shared<BachelierSwaptionEngine>(discount_, inVol, volDayCounter);
255 }
256 swaption->setPricingEngine(swaptionEngine);
257
258 // zero might be the actual implied vol (if we e.g. convert from LN to N with a strike near zero),
259 // but the Swaption::impliedVolatility function will not find this, so we check for this case
260 // explicitly here
261 Real vega = swaption->result<Real>("vega");
262 if (vega < minVega_)
263 return 0.0;
264
265 Volatility impliedVol = 0.0;
266 try {
267 Real npv = swaption->NPV();
268
269 // Calculate guess for implied volatility solver
270 Real guess = 0.0;
271 if (outType == ShiftedLognormal) {
272 QL_REQUIRE(atmRate + outShift > 0.0, "SwaptionVolatilityConverter: ATM rate + shift must be > 0.0");
273 if (inType == Normal)
274 guess = inVol / (atmRate + outShift);
275 else
276 guess = inVol * (atmRate + inShift) / (atmRate + outShift);
277 } else {
278 if (inType == Normal)
279 guess = inVol;
280 else
281 guess = inVol * (atmRate + inShift);
282 }
283
284// Note: In implying the volatility the volatility day counter is hardcoded to Actual365Fixed
285#if QL_HEX_VERSION < 0x011000f0
286 impliedVol = swaption->impliedVolatility(npv, discount_, guess, accuracy_, maxEvaluations_, minVol_, maxVol_,
287 outShift, outType);
288#else
289 impliedVol = swaption->impliedVolatility(npv, discount_, guess, accuracy_, maxEvaluations_, minVol_, maxVol_,
290 outType, outShift);
291#endif
292
293 } catch (std::exception& e) {
294 // couldn't find implied volatility
295 QL_FAIL("SwaptionVolatilityConverter: volatility conversion failed while trying to convert volatility"
296 " for expiry "
297 << expiry << " and swap tenor " << swapTenor << ". Error: " << e.what());
298 }
299
300 return impliedVol;
301}
