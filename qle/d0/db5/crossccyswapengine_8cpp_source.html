41 void CrossCcySwapEngine::calculate() const {
42 
43  QL_REQUIRE(!currency1Discountcurve_.empty() && !currency2Discountcurve_.empty(),
44  "Discounting term structure handle is empty.");
45 
46  QL_REQUIRE(!spotFX_.empty(), "FX spot quote handle is empty.");
47 
48  QL_REQUIRE(currency1Discountcurve_->referenceDate() == currency2Discountcurve_->referenceDate(),
49  "Term structures should have the same reference date.");
50  Date referenceDate = currency1Discountcurve_->referenceDate();
51  Date settlementDate = settlementDate_;
52  if (settlementDate_ == Date()) {
53  settlementDate = referenceDate;
54  } else {
55  QL_REQUIRE(settlementDate >= referenceDate, "Settlement date (" << settlementDate
56  << ") cannot be before discount curve "
57  "reference date ("
58  << referenceDate << ")");
59  }
60 
61  Size numLegs = arguments_.legs.size();
62  // - Instrument::Results
63  if (npvDate_ == Date()) {
64  results_.valuationDate = referenceDate;
65  } else {
66  QL_REQUIRE(npvDate_ >= referenceDate, "NPV date (" << npvDate_
67  << ") cannot be before "
68  "discount curve reference date ("
69  << referenceDate << ")");
70  results_.valuationDate = npvDate_;
71  }
72 
73  Date spotFXSettleDate = spotFXSettleDate_;
74  if (spotFXSettleDate_ == Date()) {
75  spotFXSettleDate = referenceDate;
76  } else {
77  QL_REQUIRE(spotFXSettleDate >= referenceDate, "FX settlement date (" << spotFXSettleDate
78  << ") cannot be before discount curve "
79  "reference date ("
80  << referenceDate << ")");
81  }
82 
83  results_.value = 0.0;
84  results_.errorEstimate = Null<Real>();
85  // - Swap::Results
86  results_.legNPV.resize(numLegs);
87  results_.legBPS.resize(numLegs);
88  results_.startDiscounts.resize(numLegs);
89  results_.endDiscounts.resize(numLegs);
90  // - CrossCcySwap::Results
91  results_.inCcyLegNPV.resize(numLegs);
92  results_.inCcyLegBPS.resize(numLegs);
93  results_.npvDateDiscounts.resize(numLegs);
94 
95  bool includeReferenceDateFlows =
96  includeSettlementDateFlows_ ? *includeSettlementDateFlows_ : Settings::instance().includeReferenceDateEvents();
97 
98  for (Size legNo = 0; legNo < numLegs; legNo++) {
99  try {
100  // Choose the correct discount curve for the leg.
101  Handle<YieldTermStructure> legDiscountCurve;
102  if (arguments_.currencies[legNo] == ccy1_) {
103  legDiscountCurve = currency1Discountcurve_;
104  } else {
105  QL_REQUIRE(arguments_.currencies[legNo] == ccy2_, "leg ccy (" << arguments_.currencies[legNo]
106  << ") must be ccy1 (" << ccy1_
107  << ") or ccy2 (" << ccy2_ << ")");
108  legDiscountCurve = currency2Discountcurve_;
109  }
110  results_.npvDateDiscounts[legNo] = legDiscountCurve->discount(results_.valuationDate);
111 
112  // Calculate the NPV and BPS of each leg in its currency.
113  CashFlows::npvbps(arguments_.legs[legNo], **legDiscountCurve, includeReferenceDateFlows, settlementDate,
114  results_.valuationDate, results_.inCcyLegNPV[legNo], results_.inCcyLegBPS[legNo]);
115  results_.inCcyLegNPV[legNo] *= arguments_.payer[legNo];
116  results_.inCcyLegBPS[legNo] *= arguments_.payer[legNo];
117 
118  results_.legNPV[legNo] = results_.inCcyLegNPV[legNo];
119  results_.legBPS[legNo] = results_.inCcyLegBPS[legNo];
120 
121  // Convert to NPV currency if necessary.
122  if (arguments_.currencies[legNo] != ccy1_) {
123  // results_.legNPV[legNo] *= spotFX_->value();
124  // results_.legBPS[legNo] *= spotFX_->value();
125  Real spotFXRate = spotFX_->value();
126  if (spotFXSettleDate != referenceDate) {
127  // Use the parity relation between discount factors and fx rates to compute spotFXRate
128  // Generic formula: fx(T1)/fx(T2) = FwdDF_Quote(T1->T2) / FwdDF_Base(T1->T2),
129  // where fx represents the currency ratio Base/Quote
130  Real ccy1DF = currency1Discountcurve_->discount(spotFXSettleDate);
131  Real ccy2DF = currency2Discountcurve_->discount(spotFXSettleDate);
132  QL_REQUIRE(ccy2DF != 0.0, "Discount Factor associated with currency " << ccy2_
133  << " at maturity " << spotFXSettleDate << " cannot be zero");
134  spotFXRate *= ccy1DF / ccy2DF;
135  }
136  results_.legNPV[legNo] *= spotFXRate;
137  results_.legBPS[legNo] *= spotFXRate;
138  }
139 
140  // Get start date and end date discount for the leg
141  Date startDate = CashFlows::startDate(arguments_.legs[legNo]);
142  if (startDate >= currency1Discountcurve_->referenceDate()) {
143  results_.startDiscounts[legNo] = legDiscountCurve->discount(startDate);
144  } else {
145  results_.startDiscounts[legNo] = Null<DiscountFactor>();
146  }
147 
148  Date maturityDate = CashFlows::maturityDate(arguments_.legs[legNo]);
149  if (maturityDate >= currency1Discountcurve_->referenceDate()) {
150  results_.endDiscounts[legNo] = legDiscountCurve->discount(maturityDate);
151  } else {
152  results_.endDiscounts[legNo] = Null<DiscountFactor>();
153  }
154 
155  } catch (std::exception& e) {
156  QL_FAIL(io::ordinal(legNo + 1) << " leg: " << e.what());
157  }
158 
159  results_.value += results_.legNPV[legNo];
160  }
161 }
