64 void DiscountingRiskyBondEngine::calculate() const {
65  QL_REQUIRE(!discountCurve_.empty(), "discounting term structure handle is empty");
66  results_.valuationDate = (*discountCurve_)->referenceDate();
67 
68  // the npv as of today, excluding cashflows before the settlement date
69 
70  Real npv = calculateNpv(results_.valuationDate, arguments_.settlementDate, arguments_.cashflows,
71  includeSettlementDateFlows_);
72 
73  // the results value is set to the npv as of today including the cashflows before settlement
74 
75  results_.value = npv + cashflowsBeforeSettlementValue_;
76 
77  // the settlement value is excluding cashflows before the settlement date and compounded to the settlement date
78 
79  results_.settlementValue = npv * compoundFactorSettlement_;
80 
81  // set a few more additional results
82 
83  results_.additionalResults["securitySpread"] = securitySpread_.empty() ? 0.0 : securitySpread_->value();
84  Date maturity = CashFlows::maturityDate(arguments_.cashflows);
85  if (maturity > results_.valuationDate) {
86  Real t = discountCurve_->timeFromReference(maturity);
87  results_.additionalResults["maturityTime"] = t;
88  results_.additionalResults["maturityDiscountFactor"] = discountCurve_->discount(t);
89  results_.additionalResults["maturitySurvivalProb"] =
90  defaultCurve_.empty() ? 1.0 : defaultCurve_->survivalProbability(t);
91  results_.additionalResults["recoveryRate"] = recoveryRate_.empty() ? 0.0 : recoveryRate_->value();
92  }
93 }

95 Real DiscountingRiskyBondEngine::calculateNpv(const Date& npvDate, const Date& settlementDate, const Leg& cashflows,
96  boost::optional<bool> includeSettlementDateFlows,
97  const Handle<YieldTermStructure>& incomeCurve,
98  const bool conditionalOnSurvival) const {
99 
100  bool includeRefDateFlows =
101  includeSettlementDateFlows ? *includeSettlementDateFlows_ : Settings::instance().includeReferenceDateEvents();
102 
103  Real npvValue = 0.0;
104  cashflowsBeforeSettlementValue_ = 0.0;
105 
106  // handle case where we wish to price simply with benchmark curve and scalar security spread
107  // i.e. credit curve term structure (and recovery) have not been specified
108  // we set the default probability and recovery rate to zero in this instance (issuer credit worthiness already
109  // captured within security spread)
110  boost::shared_ptr<DefaultProbabilityTermStructure> creditCurvePtr =
111  defaultCurve_.empty()
112  ? boost::make_shared<QuantLib::FlatHazardRate>(results_.valuationDate, 0.0, discountCurve_->dayCounter())
113  : defaultCurve_.currentLink();
114  Rate recoveryVal = recoveryRate_.empty() ? 0.0 : recoveryRate_->value();
115 
116  // compounding factors for npv date
117  Real dfNpv = incomeCurve.empty() ? discountCurve_->discount(npvDate) : incomeCurve->discount(npvDate);
118  Real spNpv = conditionalOnSurvival ? creditCurvePtr->survivalProbability(npvDate) : 1.0;
119 
120  // compound factors for settlement date
121  Real dfSettl =
122  incomeCurve.empty() ? discountCurve_->discount(settlementDate) : incomeCurve->discount(settlementDate);
123  Real spSettl = creditCurvePtr->survivalProbability(settlementDate);
124  if (!conditionalOnSurvival)
125  spSettl /= creditCurvePtr->survivalProbability(npvDate);
126 
127  // effective compound factor to get settlement npv from npv date npv
128  compoundFactorSettlement_ = (dfNpv * spNpv) / (dfSettl * spSettl);
129 
130  Size numCoupons = 0;
131  bool hasLiveCashFlow = false;
132  for (Size i = 0; i < cashflows.size(); i++) {
133  boost::shared_ptr<CashFlow> cf = cashflows[i];
134  if (cf->hasOccurred(npvDate, includeRefDateFlows))
135  continue;
136  hasLiveCashFlow = true;
137 
138  // Coupon value is discounted future payment times the survival probability
139  Probability S = creditCurvePtr->survivalProbability(cf->date()) / spNpv;
140  Real tmp = cf->amount() * S * discountCurve_->discount(cf->date()) / dfNpv;
141  if (!cf->hasOccurred(settlementDate, includeRefDateFlows))
142  npvValue += tmp;
143  else
144  cashflowsBeforeSettlementValue_ += tmp;
145 
146  /* The amount recovered in the case of default is the recoveryrate*Notional*Probability of
147  Default; this is added to the NPV value. For coupon bonds the coupon periods are taken
148  as the timesteps for integrating over the probability of default.
149  */
150  boost::shared_ptr<Coupon> coupon = boost::dynamic_pointer_cast<Coupon>(cf);
151  if (coupon) {
152  numCoupons++;
153  Date startDate = coupon->accrualStartDate();
154  Date endDate = coupon->accrualEndDate();
155  Date effectiveStartDate = (startDate <= npvDate && npvDate <= endDate) ? npvDate : startDate;
156  Date defaultDate = effectiveStartDate + (endDate - effectiveStartDate) / 2;
157  Probability P = creditCurvePtr->defaultProbability(effectiveStartDate, endDate) / spNpv;
158 
159  npvValue += coupon->nominal() * recoveryVal * P * discountCurve_->discount(defaultDate) / dfNpv;
160  }
161  }
162 
163  // the ql instrument might not yet be expired and still have not anything to value if
164  // the npvDate > evaluation date
165  if (!hasLiveCashFlow)
166  return 0.0;
167 
168  if (cashflows.size() > 1 && numCoupons == 0) {
169  QL_FAIL("DiscountingRiskyBondEngine does not support bonds with multiple cashflows but no coupons");
170  }
171 
172  /* If there are no coupon, as in a Zero Bond, we must integrate over the entire period from npv date to
173  maturity. The timestepPeriod specified is used as provide the steps for the integration. This only applies
174  to bonds with 1 cashflow, identified as a final redemption payment.
175  */
176  if (cashflows.size() == 1) {
177  boost::shared_ptr<Redemption> redemption = boost::dynamic_pointer_cast<Redemption>(cashflows[0]);
178  if (redemption) {
179  Date startDate = npvDate;
180  while (startDate < redemption->date()) {
181  Date stepDate = startDate + timestepPeriod_;
182  Date endDate = (stepDate > redemption->date()) ? redemption->date() : stepDate;
183  Date defaultDate = startDate + (endDate - startDate) / 2;
184  Probability P = creditCurvePtr->defaultProbability(startDate, endDate) / spNpv;
185 
186  npvValue += redemption->amount() * recoveryVal * P * discountCurve_->discount(defaultDate) / dfNpv;
187  startDate = stepDate;
188  }
189  }
190  }
191 
192  return npvValue;
193 }
