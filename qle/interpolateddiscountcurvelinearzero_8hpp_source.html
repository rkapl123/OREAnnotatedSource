95  void performCalculations() const {
96  today_ = Settings::instance().evaluationDate();
97  data_[0] = std::log(quotes_[1]->value()) / -times_[1];
98  for (Size i = 1; i < times_.size(); ++i) {

104  DiscountFactor discountImpl(Time t) const {
105  calculate();
106  if (t <= this->times_.back())
107  return std::exp((*interpolation_)(t, true) * -t);
108  // flat fwd extrapolation
109  Time tMax = this->times_.back();
110  Rate zMax = this->data_.back();
111  Rate instFwdMax = zMax + tMax * this->interpolation_->derivative(tMax);
112  Rate r = (zMax * tMax + instFwdMax * (t - tMax)) / t;
113  return std::exp(r * -t);
114  }
