167template <class TimeInterpolator, class SmileInterpolator>
168void OptionletStripperWithAtm<TimeInterpolator, SmileInterpolator>::performCalculations() const {
169
170 // Some localised typedefs and using declarations to make the code more readable
171 using QuantLib::BachelierCapFloorEngine;
172 using QuantLib::BlackCapFloorEngine;
173 using QuantLib::Handle;
174 using QuantLib::MakeCapFloor;
175 using QuantLib::OptionletVolatilityStructure;
176 using QuantLib::Period;
177 using QuantLib::Size;
178 using QuantLib::Time;
179 using QuantLib::Volatility;
180 using QuantLib::YieldTermStructure;
181 using std::vector;
182
183 bool isOis = boost::dynamic_pointer_cast<OvernightIndex>(index_) != nullptr;
184
185 // Possible update based on underlying optionlet stripper data
186 optionletDates_ = osBase_->optionletFixingDates();
187 optionletPaymentDates_ = osBase_->optionletPaymentDates();
188 optionletAccrualPeriods_ = osBase_->optionletAccrualPeriods();
189 optionletTimes_ = osBase_->optionletFixingTimes();
190 atmOptionletRate_ = osBase_->atmOptionletRates();
191 for (Size i = 0; i < optionletTimes_.size(); ++i) {
192 optionletStrikes_[i] = osBase_->optionletStrikes(i);
193 optionletVolatilities_[i] = osBase_->optionletVolatilities(i);
194 }
195
196 // ATM curve tenors
197 const vector<Period>& atmTenors = atmCurve_->optionTenors();
198 vector<Time> atmTimes(atmTenors.size());
199 for (Size i = 0; i < atmTenors.size(); ++i) {
200 Date d = atmCurve_->optionDateFromTenor(atmTenors[i]);
201 atmTimes[i] = atmCurve_->timeFromReference(d);
202 }
203
204 // discount curve
205 const Handle<YieldTermStructure>& discountCurve = discount_.empty() ? index_->forwardingTermStructure() : discount_;
206
207 // Populate ATM strikes and prices
208 for (Size j = 0; j < nAtmExpiries_; ++j) {
209
210 if (isOis) {
211
212 QL_FAIL("OIS cap vol stripping with ATM not yet supported.");
213
214 } else {
215
216 // Create a cap for each pillar point on ATM curve and attach relevant pricing engine
217 Volatility atmVol = atmCurve_->volatility(atmTimes[j], 0.01);
218 boost::shared_ptr<PricingEngine> engine;
219 if (atmVolatilityType_ == ShiftedLognormal) {
220 engine = boost::make_shared<BlackCapFloorEngine>(discountCurve, atmVol, dayCounter_, atmDisplacement_);
221 } else if (atmVolatilityType_ == Normal) {
222 engine = boost::make_shared<BachelierCapFloorEngine>(discountCurve, atmVol, dayCounter_);
223 } else {
224 QL_FAIL("Unknown volatility type: " << atmVolatilityType_);
225 }
226
227 // Using Null<Rate>() as strike => strike will be set to ATM rate. However, to calculate ATM rate, QL
228 // requires a BlackCapFloorEngine to be set (not a BachelierCapFloorEngine)! So, need a temp
229 // BlackCapFloorEngine with a dummy vol to calculate ATM rate. Needs to be fixed in QL.
230 boost::shared_ptr<PricingEngine> tempEngine = boost::make_shared<BlackCapFloorEngine>(discountCurve, 0.01);
231 caps_[j] =
232 MakeCapFloor(CapFloor::Cap, atmTenors[j], index_, Null<Rate>(), 0 * Days).withPricingEngine(tempEngine);
233
234 // Now set correct engine and get the ATM rate and the price
235 caps_[j]->setPricingEngine(engine);
236 atmStrikes_[j] = caps_[j]->atmRate(**discountCurve);
237 atmPrices_[j] = caps_[j]->NPV();
238 }
239 }
240
241 // Create an optionlet volatility structure from the underlying stripped optionlet surface
242 boost::shared_ptr<OptionletVolatilityStructure> ovs =
243 boost::make_shared<QuantExt::StrippedOptionletAdapter<TimeInterpolator, SmileInterpolator> >(
244 atmCurve_->referenceDate(), osBase_);
245 ovs->enableExtrapolation();
246
247 // Imply the volatility spreads that match the ATM prices
248 volSpreads_ = volSpreads(discountCurve, Handle<OptionletVolatilityStructure>(ovs));
249
250 // This stripped optionlet surface is the underlying stripped optionlet surface modified by the implied spreads
251 for (Size j = 0; j < nAtmExpiries_; ++j) {
252 for (Size i = 0; i < optionletTimes_.size(); ++i) {
253 if (i < caps_[j]->floatingLeg().size()) {
254
255 Volatility unadjustedVol = ovs->volatility(optionletTimes_[i], atmStrikes_[j]);
256 Volatility adjustedVol = unadjustedVol + volSpreads_[j];
257
258 // insert ATM strike and adjusted volatility
259 vector<Rate>::const_iterator previous =
260 lower_bound(optionletStrikes_[i].begin(), optionletStrikes_[i].end(), atmStrikes_[j]);
261 Size insertIndex = previous - optionletStrikes_[i].begin();
262 optionletStrikes_[i].insert(optionletStrikes_[i].begin() + insertIndex, atmStrikes_[j]);
263 optionletVolatilities_[i].insert(optionletVolatilities_[i].begin() + insertIndex, adjustedVol);
264 }
265 }
266 }
267}

269template <class TimeInterpolator, class SmileInterpolator>
270inline std::vector<QuantLib::Volatility> OptionletStripperWithAtm<TimeInterpolator, SmileInterpolator>::volSpreads(
271 const QuantLib::Handle<QuantLib::YieldTermStructure>& discount,
272 const QuantLib::Handle<QuantLib::OptionletVolatilityStructure>& ovs) const {
273
274 // Some localised typedefs and using declarations to make the code more readable
275 using QuantLib::Size;
276 using QuantLib::Volatility;
277 using std::vector;
278
279 QuantLib::Brent solver;
280 vector<Volatility> result(nAtmExpiries_);
281 Volatility guess = 0.0001;
282 Volatility minSpread = -0.1;
283 Volatility maxSpread = 0.1;
284
285 for (Size j = 0; j < nAtmExpiries_; ++j) {
286 ObjectiveFunction f(ovs, caps_[j], atmPrices_[j], discount);
287 solver.setMaxEvaluations(maxEvaluations_);
288 Volatility root = solver.solve(f, accuracy_, guess, minSpread, maxSpread);
289 result[j] = root;
290 }
291
292 return result;
293}

295template <class TimeInterpolator, class SmileInterpolator>
296OptionletStripperWithAtm<TimeInterpolator, SmileInterpolator>::ObjectiveFunction::ObjectiveFunction(
297 const QuantLib::Handle<QuantLib::OptionletVolatilityStructure>& ovs,
298 const boost::shared_ptr<QuantLib::CapFloor>& cap, QuantLib::Real targetValue,
299 const QuantLib::Handle<QuantLib::YieldTermStructure>& discount)
300 : cap_(cap), targetValue_(targetValue), discount_(discount) {
301
302 // Some localised typedefs and using declarations to make the code more readable
303 using QuantLib::Handle;
304 using QuantLib::OptionletVolatilityStructure;
305 using QuantLib::Quote;
306 using QuantLib::SimpleQuote;
307
308 // set an implausible value, so that calculation is forced at first operator()(Volatility x) call
309 spreadQuote_ = boost::make_shared<SimpleQuote>(-1.0);
310
311 // Spreaded optionlet volatility structure that will be used to price the ATM cap
312 Handle<OptionletVolatilityStructure> spreadedOvs = Handle<OptionletVolatilityStructure>(
313 boost::make_shared<QuantExt::SpreadedOptionletVolatility>(ovs, Handle<Quote>(spreadQuote_)));
314
315 // Attach the relevant engine to the cap with the spreaded optionlet volatility structure
316 if (ovs->volatilityType() == ShiftedLognormal) {
317 cap_->setPricingEngine(boost::make_shared<BlackCapFloorEngine>(discount_, spreadedOvs, ovs->displacement()));
318 } else if (ovs->volatilityType() == Normal) {
319 cap_->setPricingEngine(boost::make_shared<BachelierCapFloorEngine>(discount_, spreadedOvs));
320 } else {
321 QL_FAIL("Unknown volatility type: " << ovs->volatilityType());
322 }
323}

325template <class TimeInterpolator, class SmileInterpolator>
326QuantLib::Real OptionletStripperWithAtm<TimeInterpolator, SmileInterpolator>::ObjectiveFunction::operator()(
327 QuantLib::Volatility volSpread) const {
328 if (volSpread != spreadQuote_->value()) {
329 spreadQuote_->setValue(volSpread);
330 }
331 return cap_->NPV() - targetValue_;
