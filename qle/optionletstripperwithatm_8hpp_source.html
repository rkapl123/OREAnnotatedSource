167 template <class TimeInterpolator, class SmileInterpolator>
168 void OptionletStripperWithAtm<TimeInterpolator, SmileInterpolator>::performCalculations() const {
169 
170  // Some localised typedefs and using declarations to make the code more readable
171  using QuantLib::BachelierCapFloorEngine;
172  using QuantLib::BlackCapFloorEngine;
173  using QuantLib::Handle;
174  using QuantLib::MakeCapFloor;
175  using QuantLib::OptionletVolatilityStructure;
176  using QuantLib::Period;
177  using QuantLib::Size;
178  using QuantLib::Time;
179  using QuantLib::Volatility;
180  using QuantLib::YieldTermStructure;
181  using std::vector;
182 
183  // Possible update based on underlying optionlet stripper data
184  optionletDates_ = osBase_->optionletFixingDates();
185  optionletPaymentDates_ = osBase_->optionletPaymentDates();
186  optionletAccrualPeriods_ = osBase_->optionletAccrualPeriods();
187  optionletTimes_ = osBase_->optionletFixingTimes();
188  atmOptionletRate_ = osBase_->atmOptionletRates();
189  for (Size i = 0; i < optionletTimes_.size(); ++i) {
190  optionletStrikes_[i] = osBase_->optionletStrikes(i);
191  optionletVolatilities_[i] = osBase_->optionletVolatilities(i);
192  }
193 
194  // ATM curve tenors
195  const vector<Period>& atmTenors = atmCurve_->optionTenors();
196  vector<Time> atmTimes(atmTenors.size());
197  for (Size i = 0; i < atmTenors.size(); ++i) {
198  Date d = atmCurve_->optionDateFromTenor(atmTenors[i]);
199  atmTimes[i] = atmCurve_->timeFromReference(d);
200  }
201 
202  // discount curve
203  const Handle<YieldTermStructure>& discountCurve =
204  discount_.empty() ? iborIndex_->forwardingTermStructure() : discount_;
205 
206  // Populate ATM strikes and prices
207  for (Size j = 0; j < nAtmExpiries_; ++j) {
208 
209  // Create a cap for each pillar point on ATM curve and attach relevant pricing engine
210  Volatility atmVol = atmCurve_->volatility(atmTimes[j], 0.01);
211  boost::shared_ptr<PricingEngine> engine;
212  if (atmVolatilityType_ == ShiftedLognormal) {
213  engine = boost::make_shared<BlackCapFloorEngine>(discountCurve, atmVol, dayCounter_, atmDisplacement_);
214  } else if (atmVolatilityType_ == Normal) {
215  engine = boost::make_shared<BachelierCapFloorEngine>(discountCurve, atmVol, dayCounter_);
216  } else {
217  QL_FAIL("Unknown volatility type: " << atmVolatilityType_);
218  }
219 
220  // Using Null<Rate>() as strike => strike will be set to ATM rate. However, to calculate ATM rate, QL requires
221  // a BlackCapFloorEngine to be set (not a BachelierCapFloorEngine)! So, need a temp BlackCapFloorEngine with a
222  // dummy vol to calculate ATM rate. Needs to be fixed in QL.
223  boost::shared_ptr<PricingEngine> tempEngine = boost::make_shared<BlackCapFloorEngine>(discountCurve, 0.01);
224  caps_[j] =
225  MakeCapFloor(CapFloor::Cap, atmTenors[j], iborIndex_, Null<Rate>(), 0 * Days).withPricingEngine(tempEngine);
226 
227  // Now set correct engine and get the ATM rate and the price
228  caps_[j]->setPricingEngine(engine);
229  atmStrikes_[j] = caps_[j]->atmRate(**discountCurve);
230  atmPrices_[j] = caps_[j]->NPV();
231  }
232 
233  // Create an optionlet volatility structure from the underlying stripped optionlet surface
234  boost::shared_ptr<OptionletVolatilityStructure> ovs =
235  boost::make_shared<QuantExt::StrippedOptionletAdapter<TimeInterpolator, SmileInterpolator> >(
236  atmCurve_->referenceDate(), osBase_);
237  ovs->enableExtrapolation();
238 
239  // Imply the volatility spreads that match the ATM prices
240  volSpreads_ = volSpreads(discountCurve, Handle<OptionletVolatilityStructure>(ovs));
241 
242  // This stripped optionlet surface is the underlying stripped optionlet surface modified by the implied spreads
243  for (Size j = 0; j < nAtmExpiries_; ++j) {
244  for (Size i = 0; i < optionletTimes_.size(); ++i) {
245  if (i < caps_[j]->floatingLeg().size()) {
246 
247  Volatility unadjustedVol = ovs->volatility(optionletTimes_[i], atmStrikes_[j]);
248  Volatility adjustedVol = unadjustedVol + volSpreads_[j];
249 
250  // insert ATM strike and adjusted volatility
251  vector<Rate>::const_iterator previous =
252  lower_bound(optionletStrikes_[i].begin(), optionletStrikes_[i].end(), atmStrikes_[j]);
253  Size insertIndex = previous - optionletStrikes_[i].begin();
254  optionletStrikes_[i].insert(optionletStrikes_[i].begin() + insertIndex, atmStrikes_[j]);
255  optionletVolatilities_[i].insert(optionletVolatilities_[i].begin() + insertIndex, adjustedVol);
256  }
257  }
258  }
259 }

261 template <class TimeInterpolator, class SmileInterpolator>
262 inline std::vector<QuantLib::Volatility> OptionletStripperWithAtm<TimeInterpolator, SmileInterpolator>::volSpreads(
263  const QuantLib::Handle<QuantLib::YieldTermStructure>& discount,
264  const QuantLib::Handle<QuantLib::OptionletVolatilityStructure>& ovs) const {
265 
266  // Some localised typedefs and using declarations to make the code more readable
267  using QuantLib::Size;
268  using QuantLib::Volatility;
269  using std::vector;
270 
271  QuantLib::Brent solver;
272  vector<Volatility> result(nAtmExpiries_);
273  Volatility guess = 0.0001;
274  Volatility minSpread = -0.1;
275  Volatility maxSpread = 0.1;
276 
277  for (Size j = 0; j < nAtmExpiries_; ++j) {
278  ObjectiveFunction f(ovs, caps_[j], atmPrices_[j], discount);
279  solver.setMaxEvaluations(maxEvaluations_);
280  Volatility root = solver.solve(f, accuracy_, guess, minSpread, maxSpread);
281  result[j] = root;
282  }
283 
284  return result;
285 }

287 template <class TimeInterpolator, class SmileInterpolator>
288 OptionletStripperWithAtm<TimeInterpolator, SmileInterpolator>::ObjectiveFunction::ObjectiveFunction(
289  const QuantLib::Handle<QuantLib::OptionletVolatilityStructure>& ovs,
290  const boost::shared_ptr<QuantLib::CapFloor>& cap, QuantLib::Real targetValue,
291  const QuantLib::Handle<QuantLib::YieldTermStructure>& discount)
292  : cap_(cap), targetValue_(targetValue), discount_(discount) {
293 
294  // Some localised typedefs and using declarations to make the code more readable
295  using QuantLib::Handle;
296  using QuantLib::OptionletVolatilityStructure;
297  using QuantLib::Quote;
298  using QuantLib::SimpleQuote;
299 
300  // set an implausible value, so that calculation is forced at first operator()(Volatility x) call
301  spreadQuote_ = boost::make_shared<SimpleQuote>(-1.0);
302 
303  // Spreaded optionlet volatility structure that will be used to price the ATM cap
304  Handle<OptionletVolatilityStructure> spreadedOvs = Handle<OptionletVolatilityStructure>(
305  boost::make_shared<QuantExt::SpreadedOptionletVolatility>(ovs, Handle<Quote>(spreadQuote_)));
306 
307  // Attach the relevant engine to the cap with the spreaded optionlet volatility structure
308  if (ovs->volatilityType() == ShiftedLognormal) {
309  cap_->setPricingEngine(boost::make_shared<BlackCapFloorEngine>(discount_, spreadedOvs, ovs->displacement()));
310  } else if (ovs->volatilityType() == Normal) {
311  cap_->setPricingEngine(boost::make_shared<BachelierCapFloorEngine>(discount_, spreadedOvs));
312  } else {
313  QL_FAIL("Unknown volatility type: " << ovs->volatilityType());
314  }
315 }
