446  void appendToFixedParameterVector(const AssetType t, const AssetType v, const Size param, const Size index,
447  const Size i, std::vector<bool>& res) {
448  for (Size j = 0; j < components(t); ++j) {
449  for (Size k = 0; k < arguments(t, j); ++k) {
450  std::vector<bool> tmp1(p_[idx(t, j)]->parameter(k)->size(), true);
451  if ((param == Null<Size>() || k == param) && t == v && index == j) {
452  for (Size ii = 0; ii < tmp1.size(); ++ii) {
453  if (i == Null<Size>() || i == ii) {
454  tmp1[ii] = false;
455  }
456  }
457  }
458  res.insert(res.end(), tmp1.begin(), tmp1.end());
459  }
460  }
461  }
462 
463  // move parameter param (e.g. vol, reversion, or all if null) of asset type component t / index at step i (or at all
464  // steps if i is null)
465  std::vector<bool> MoveParameter(const AssetType t, const Size param, const Size index, const Size i) {
466  QL_REQUIRE(param == Null<Size>() || param < arguments(t, index),
467  "parameter for " << t << " at " << index << " (" << param << ") out of bounds 0..."
468  << arguments(t, index) - 1);
469  std::vector<bool> res(0);
470  appendToFixedParameterVector(IR, t, param, index, i, res);
471  appendToFixedParameterVector(FX, t, param, index, i, res);
472  appendToFixedParameterVector(INF, t, param, index, i, res);
473  appendToFixedParameterVector(CR, t, param, index, i, res);
474  appendToFixedParameterVector(EQ, t, param, index, i, res);
475  if (measure_ == Measure::BA)
476  appendToFixedParameterVector(AUX, t, param, index, i, res);
477  return res;
478  }
