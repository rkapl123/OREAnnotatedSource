65 /*! Parametrizations must be given in the following order
66 - IR (first parametrization defines the domestic currency)
67 - FX (for all pairs domestic-ccy defined by the IR models)
68 - INF (optionally, ccy must be a subset of the IR ccys)
69 - CR (optionally, ccy must be a subset of the IR ccys)
70 - EQ (for all names equity currency defined in Parametrization)
71 - COM (for all names commodity currency defined in Parametrization)
72 If the correlation matrix is not given, it is initialized
73 as the unit matrix (and can be customized after
74 construction of the model).
75
76 All IR components must be of type HW _or_ LGM1F, i.e. you can't mix the two types.
77 */

448 // cache for infdkI, crlgm1fS method
449 struct cache_key {
450 Size i, ccy;
451 double t, T;
452 bool operator==(const cache_key& o) const { return (i == o.i) && (ccy == o.ccy) && (t == o.t) && (T == o.T); }
453 };
454
455 struct cache_hasher {
456 std::size_t operator()(cache_key const& x) const {
457 std::size_t seed = 0;
458 boost::hash_combine(seed, x.i);
459 boost::hash_combine(seed, x.ccy);
460 boost::hash_combine(seed, x.t);
461 boost::hash_combine(seed, x.T);
462 return seed;
463 }
464 };
