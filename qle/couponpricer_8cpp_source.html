62 void PricerSetter::visit(QuantLib::OvernightIndexedCoupon& c) {
63  // Special pricer for BRL CDI
64  boost::shared_ptr<BRLCdi> brlCdiIndex = boost::dynamic_pointer_cast<BRLCdi>(c.index());
65  if (brlCdiIndex) {
66  const boost::shared_ptr<BRLCdiCouponPricer> brlCdiCouponPricer =
67  boost::dynamic_pointer_cast<BRLCdiCouponPricer>(pricer_);
68  QL_REQUIRE(brlCdiCouponPricer, "Pricer not compatible with BRL CDI coupon");
69  c.setPricer(brlCdiCouponPricer);
70  } else {
71  c.setPricer(pricer_);
72  }

113 void setCouponPricer(const Leg& leg, const boost::shared_ptr<FloatingRateCouponPricer>& pricer) {
114  PricerSetter setter(pricer);

120 void setCouponPricers(const Leg& leg, const std::vector<boost::shared_ptr<FloatingRateCouponPricer> >& pricers) {
121 
122  Size nCashFlows = leg.size();
123  QL_REQUIRE(nCashFlows > 0, "No cashflows");
124 
125  Size nPricers = pricers.size();
126  QL_REQUIRE(nCashFlows >= nPricers,
127  "Mismatch between leg size (" << nCashFlows << ") and number of pricers (" << nPricers << ")");
128 
129  for (Size i = 0; i < nCashFlows; ++i) {
130  PricerSetter setter(i < nPricers ? pricers[i] : pricers[nPricers - 1]);
131  leg[i]->accept(setter);
132  }
133 }
