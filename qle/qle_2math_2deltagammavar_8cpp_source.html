48void moments(const Matrix& omega, const Array& delta, const Matrix& gamma, Real& num, Real& mu, Real& variance) {
49 detail::check(omega, delta, gamma);
50
51 // see Carol Alexander, Market Risk, Vol IV
52 // Formulas IV5.30 and IV5.31 are buggy though:
53 // IV.5.30 should have ... + 3 \delta' \Omega \Gamma \Omega \delta in the numerator
54 // IV.5.31 should have ... + 12 \delta \Omega (\Gamma \Omega)^2 \delta + 3\sigma^4 in the numerator
55
56 num = std::max(QuantExt::detail::absMax(delta), QuantExt::detail::absMax(gamma));
57 if (close_enough(num, 0.0))
58 return;
59
60 Array tmpDelta = 1.0 / num * delta;
61 Matrix tmpGamma = 1.0 / num * gamma;
62
63 Real dOd = DotProduct(tmpDelta, omega * tmpDelta);
64 Matrix go = tmpGamma * omega;
65 Matrix go2 = go * go;
66 Real trGo2 = Trace(go2);
67
68 mu = 0.5 * Trace(go);
69 variance = dOd + 0.5 * trGo2;
70}
71} // namespace

73Real deltaVar(const Matrix& omega, const Array& delta, const Real p, const CovarianceSalvage& sal) {
74 detail::check(p);
75 detail::check(omega, delta);
76 Real num = detail::absMax(delta);
77 if (close_enough(num, 0.0))
78 return 0.0;
79 Array tmpDelta = delta / num;
80 return std::sqrt(DotProduct(tmpDelta, sal.salvage(omega).first * tmpDelta)) *
81 QuantLib::InverseCumulativeNormal()(p) * num;
82} // deltaVar

84Real deltaGammaVarNormal(const Matrix& omega, const Array& delta, const Matrix& gamma, const Real p,
85 const CovarianceSalvage& sal) {
86 detail::check(p);
87 Real s = QuantLib::InverseCumulativeNormal()(p);
88 Real num = 0.0, mu = 0.0, variance = 0.0;
89 moments(sal.salvage(omega).first, delta, gamma, num, mu, variance);
90 if (close_enough(num, 0.0) || close_enough(variance, 0.0))
91 return 0.0;
92 return (std::sqrt(variance) * s + mu) * num;
93
94} // deltaGammaVarNormal
