50 void BaseCorrelationTermStructure::validate() const {
51  Period prevTenor(0, Days);
52  for (size_t i = 0; i < tenors_.size(); ++i) {
53  QL_REQUIRE(tenors_[i] > prevTenor, "Tenors need to be sorted and larger than 0 * Days");
54  }
55 
56  // Ensure detachmentpoints are sorted and between 0.0 and 1.0
57  double prevDetachmentPoint = 0.0;
58  for (size_t i = 0; i < detachmentPoints_.size(); ++i) {
59  QL_REQUIRE(detachmentPoints_[i] > prevDetachmentPoint,
60  "Detachmentpoints need to be sorted and between (0, 1].");
61  QL_REQUIRE(detachmentPoints_[i] <= 1.0, "Detachmentpoints need to be sorted and between (0, 1].");
62  }
63 }

65 void BaseCorrelationTermStructure::initializeDatesAndTimes() const {
66  const Date& refDate = referenceDate();
67  Date start = startDate_ == Date() ? refDate : startDate_;
68  Calendar cldr = calendar();
69 
70  for (Size i = 0; i < tenors_.size(); i++) {
71  Date d;
72  if (rule_) {
73  d = start + tenors_[i];
74  if (*rule_ == DateGeneration::CDS2015 || *rule_ == DateGeneration::CDS ||
75  *rule_ == DateGeneration::OldCDS) {
76  d = cdsMaturity(start, tenors_[i], *rule_);
77  }
78  } else {
79  d = cldr.advance(start, tenors_[i], bdc_);
80  }
81 
82  dates_.push_back(d);
83  times_.push_back(timeFromReference(d));
84 
85  QL_REQUIRE(!dates_.empty(), "no dates left after removing expired dates");
86  }
87 }

89 void BaseCorrelationTermStructure::checkRange(Time t, Real strike, bool extrapolate) const {
90  bool extrapolationNeeded =
91  t < minTime() || t > maxTime() || strike < minDetachmentPoint() || strike > maxDetachmentPoint();
92  QL_REQUIRE(extrapolate || allowsExtrapolation() || !extrapolationNeeded,

94  << t << " to be between (" << minTime() << ", " << maxTime() << ") and detachmentPoint = " << strike
95  << " to be between (" << minDetachmentPoint() << ", " << maxDetachmentPoint() << ").");
