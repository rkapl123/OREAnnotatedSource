53OvernightIndexedCoupon::OvernightIndexedCoupon(const Date& paymentDate, Real nominal, const Date& startDate,
54 const Date& endDate,
55 const ext::shared_ptr<OvernightIndex>& overnightIndex, Real gearing,
56 Spread spread, const Date& refPeriodStart, const Date& refPeriodEnd,
57 const DayCounter& dayCounter, bool telescopicValueDates,
58 bool includeSpread, const Period& lookback, const Natural rateCutoff,
59 const Natural fixingDays, const Date& rateComputationStartDate,
60 const Date& rateComputationEndDate)
61 : FloatingRateCoupon(paymentDate, nominal, startDate, endDate, fixingDays, overnightIndex, gearing, spread,
62 refPeriodStart, refPeriodEnd, dayCounter, false),
63 overnightIndex_(overnightIndex), includeSpread_(includeSpread), lookback_(lookback), rateCutoff_(rateCutoff),
64 rateComputationStartDate_(rateComputationStartDate), rateComputationEndDate_(rateComputationEndDate) {
65
66 Date valueStart = rateComputationStartDate_ == Null<Date>() ? startDate : rateComputationStartDate_;
67 Date valueEnd = rateComputationEndDate_ == Null<Date>() ? endDate : rateComputationEndDate_;
68 if (lookback != 0 * Days) {
69 BusinessDayConvention bdc = lookback.length() > 0 ? Preceding : Following;
70 valueStart = overnightIndex->fixingCalendar().advance(valueStart, -lookback, bdc);
71 valueEnd = overnightIndex->fixingCalendar().advance(valueEnd, -lookback, bdc);
72 }
73
74 // value dates
75 Date tmpEndDate = valueEnd;
76
77 /* For the coupon's valuation only the first and last future valuation
78 dates matter, therefore we can avoid to construct the whole series
79 of valuation dates, a front and back stub will do. However notice
80 that if the global evaluation date moves forward it might run past
81 the front stub of valuation dates we build here (which incorporates
82 a grace period of 7 business after the evaluation date). This will
83 lead to false coupon projections (see the warning the class header). */
84
85 if (telescopicValueDates) {
86 // build optimised value dates schedule: front stub goes
87 // from start date to max(evalDate,valueStart) + 7bd
88 Date evalDate = Settings::instance().evaluationDate();
89 tmpEndDate = overnightIndex->fixingCalendar().advance(std::max(valueStart, evalDate), 7, Days, Following);
90 tmpEndDate = std::min(tmpEndDate, valueEnd);
91 }
92 Schedule sch = MakeSchedule()
93 .from(valueStart)
94 // .to(valueEnd)
95 .to(tmpEndDate)
96 .withTenor(1 * Days)
97 .withCalendar(overnightIndex->fixingCalendar())
98 .withConvention(overnightIndex->businessDayConvention())
99 .backwards();
100 valueDates_ = sch.dates();
101
102 if (telescopicValueDates) {
103 // build optimised value dates schedule: back stub
104 // contains at least two dates and enough periods to cover rate cutoff
105 Date tmp2 = overnightIndex->fixingCalendar().adjust(valueEnd, overnightIndex->businessDayConvention());
106 Date tmp1 = overnightIndex->fixingCalendar().advance(tmp2, -std::max<Size>(rateCutoff_, 1), Days, Preceding);
107 while (tmp1 <= tmp2) {
108 if (tmp1 > valueDates_.back())
109 valueDates_.push_back(tmp1);
110 tmp1 = overnightIndex->fixingCalendar().advance(tmp1, 1, Days, Following);
111 }
112 }
113
114 QL_ENSURE(valueDates_.size() >= 2 + rateCutoff_, "degenerate schedule");
115
116 // the first and last value date should be the unadjusted input value dates
117 if (valueDates_.front() != valueStart)
118 valueDates_.front() = valueStart;
119 if (valueDates_.back() != valueEnd)
120 valueDates_.back() = valueEnd;
121
122 n_ = valueDates_.size() - 1;
123
124 QL_REQUIRE(valueDates_[0] != valueDates_[1],
125 "internal error: first two value dates of on coupon are equal: " << valueDates_[0]);
126 QL_REQUIRE(valueDates_[n_] != valueDates_[n_ - 1],
127 "internal error: last two value dates of on coupon are equal: " << valueDates_[n_]);
128
129 // fixing dates
130 fixingDates_.resize(n_);
131 for (Size i = 0; i < n_; ++i)
132 fixingDates_[i] = overnightIndex->fixingCalendar().advance(
133 valueDates_[i], -static_cast<Integer>(FloatingRateCoupon::fixingDays()), Days, Preceding);
134
135 // accrual (compounding) periods
136 dt_.resize(n_);
137 const DayCounter& dc = overnightIndex->dayCounter();
138 for (Size i = 0; i < n_; ++i)
139 dt_[i] = dc.yearFraction(valueDates_[i], valueDates_[i + 1]);
140
141 setPricer(ext::shared_ptr<FloatingRateCouponPricer>(new OvernightIndexedCouponPricer));
142
143 // check that rate cutoff is < number of fixing dates
144 QL_REQUIRE(rateCutoff_ < n_,
145 "rate cutoff (" << rateCutoff_ << ") must be less than number of fixings in period (" << n_ << ")");
146}

148const vector<Rate>& OvernightIndexedCoupon::indexFixings() const {
149 fixings_.resize(n_);
150 Size i;
151 for (i = 0; i < n_ - rateCutoff_; ++i) {
152 fixings_[i] = index_->fixing(fixingDates_[i]);
153 }
154 Rate cutoffFixing = fixings_[i - 1];
155 while (i < n_) {
156 fixings_[i] = cutoffFixing;
157 i++;
158 }
159 return fixings_;
160}

194void OvernightIndexedCouponPricer::compute() const {
195 ext::shared_ptr<OvernightIndex> index = ext::dynamic_pointer_cast<OvernightIndex>(coupon_->index());
196
197 const vector<Date>& fixingDates = coupon_->fixingDates();
198 const vector<Time>& dt = coupon_->dt();
199
200 Size n = dt.size();
201 Size i = 0;
202 QL_REQUIRE(coupon_->rateCutoff() < n, "rate cutoff (" << coupon_->rateCutoff()
203 << ") must be less than number of fixings in period (" << n
204 << ")");
205 Size nCutoff = n - coupon_->rateCutoff();
206
207 Real compoundFactor = 1.0, compoundFactorWithoutSpread = 1.0;
208
209 // already fixed part
210 Date today = Settings::instance().evaluationDate();
211 while (i < n && fixingDates[std::min(i, nCutoff)] < today) {
212 // rate must have been fixed
213 Rate pastFixing = IndexManager::instance().getHistory(index->name())[fixingDates[std::min(i, nCutoff)]];
214 QL_REQUIRE(pastFixing != Null<Real>(),
215 "Missing " << index->name() << " fixing for " << fixingDates[std::min(i, nCutoff)]);
216 if (coupon_->includeSpread()) {
217 compoundFactorWithoutSpread *= (1.0 + pastFixing * dt[i]);
218 pastFixing += coupon_->spread();
219 }
220 compoundFactor *= (1.0 + pastFixing * dt[i]);
221 ++i;
222 }
223
224 // today is a border case
225 if (i < n && fixingDates[std::min(i, nCutoff)] == today) {
226 // might have been fixed
227 try {
228 Rate pastFixing = IndexManager::instance().getHistory(index->name())[fixingDates[std::min(i, nCutoff)]];
229 if (pastFixing != Null<Real>()) {
230 if (coupon_->includeSpread()) {
231 compoundFactorWithoutSpread *= (1.0 + pastFixing * dt[i]);
232 pastFixing += coupon_->spread();
233 }
234 compoundFactor *= (1.0 + pastFixing * dt[i]);
235 ++i;
236 } else {
237 ; // fall through and forecast
238 }
239 } catch (Error&) {
240 ; // fall through and forecast
241 }
242 }
243
244 // forward part using telescopic property in order
245 // to avoid the evaluation of multiple forward fixings
246 const vector<Date>& dates = coupon_->valueDates();
247 if (i < n) {
248 Handle<YieldTermStructure> curve = index->forwardingTermStructure();
249 QL_REQUIRE(!curve.empty(), "null term structure set to this instance of " << index->name());
250
251 // handle the part until the rate cutoff (might be empty, i.e. startDiscount = endDiscount)
252 DiscountFactor startDiscount = curve->discount(dates[i]);
253 DiscountFactor endDiscount = curve->discount(dates[std::max(nCutoff, i)]);
254
255 // handle the rate cutoff period (if there is any, i.e. if nCutoff < n)
256 if (nCutoff < n) {
257 // forward discount factor for one calendar day on the cutoff date
258 DiscountFactor discountCutoffDate = curve->discount(dates[nCutoff] + 1) / curve->discount(dates[nCutoff]);
259 // keep the above forward discount factor constant during the cutoff period
260 endDiscount *= std::pow(discountCutoffDate, dates[n] - dates[nCutoff]);
261 }
262
263 compoundFactor *= startDiscount / endDiscount;
264
265 if (coupon_->includeSpread()) {
266 compoundFactorWithoutSpread *= startDiscount / endDiscount;
267 // this is an approximation, see "Ester / Daily Spread Curve Setup in ORE":
268 // set tau to an average value
269 Real tau = coupon_->dayCounter().yearFraction(dates[i], dates.back()) / (dates.back() - dates[i]);
270 // now use formula (4) from the paper
271 compoundFactor *= std::pow(1.0 + tau * coupon_->spread(), static_cast<int>(dates.back() - dates[i]));
272 }
273 }
274
275 Rate tau = coupon_->dayCounter().yearFraction(dates.front(), dates.back());
276 Rate rate = (compoundFactor - 1.0) / tau;
277 swapletRate_ = coupon_->gearing() * rate;
278 if (!coupon_->includeSpread()) {
279 swapletRate_ += coupon_->spread();
280 effectiveSpread_ = coupon_->spread();
281 effectiveIndexFixing_ = rate;
282 } else {
283 effectiveSpread_ = rate - (compoundFactorWithoutSpread - 1.0) / tau;
284 effectiveIndexFixing_ = rate - effectiveSpread_;
285 }
286}

307 bool localCapFloor)
308 : FloatingRateCoupon(underlying->date(), underlying->nominal(), underlying->accrualStartDate(),

313
314 QL_REQUIRE(!underlying_->includeSpread() || close_enough(underlying_->gearing(), 1.0),

318 if (!localCapFloor) {

323 cap_ = floor;

325 }
326 } else {
327 cap_ = cap;
328 floor_ = floor;
329 }

331 QL_REQUIRE(cap_ >= floor, "cap level (" << cap_ << ") less than floor level (" << floor_ << ")");

348void CappedFlooredOvernightIndexedCoupon::performCalculations() const {
349 QL_REQUIRE(underlying_->pricer(), "pricer not set");
350 Rate swapletRate = nakedOption_ ? 0.0 : underlying_->rate();
351 if (floor_ != Null<Real>() || cap_ != Null<Real>())
352 pricer()->initialize(*this);
353 Rate floorletRate = 0.;
354 if (floor_ != Null<Real>())
355 floorletRate = pricer()->floorletRate(effectiveFloor());
356 Rate capletRate = 0.;
357 if (cap_ != Null<Real>())
358 capletRate = (nakedOption_ && floor_ == Null<Real>() ? -1.0 : 1.0) * pricer()->capletRate(effectiveCap());
359 rate_ = swapletRate + floorletRate - capletRate;
360}

373Rate CappedFlooredOvernightIndexedCoupon::effectiveCap() const {
374 if (cap_ == Null<Real>())
375 return Null<Real>();
376 /* We have four cases dependent on localCapFloor_ and includeSpread. Notation in the formulas:
377 g gearing,
378 s spread,
379 A coupon amount,
380 f_i daily fixings,
381 \tau_i daily accrual fractions,
382 \tau coupon accrual fraction,
383 C cap rate
384 F floor rate
385 */
386 if (localCapFloor_) {
387 if (underlying_->includeSpread()) {
388 // A = g \cdot \frac{\prod (1 + \tau_i \min ( \max ( f_i + s , F), C)) - 1}{\tau}
389 return cap_ - underlying_->spread();
390 } else {
391 // A = g \cdot \frac{\prod (1 + \tau_i \min ( \max ( f_i , F), C)) - 1}{\tau} + s
392 return cap_;
393 }
394 } else {
395 if (underlying_->includeSpread()) {
396 // A = \min \left( \max \left( g \cdot \frac{\prod (1 + \tau_i(f_i + s)) - 1}{\tau}, F \right), C \right)
397 return (cap_ / gearing() - underlying_->effectiveSpread());
398 } else {
399 // A = \min \left( \max \left( g \cdot \frac{\prod (1 + \tau_i f_i) - 1}{\tau} + s, F \right), C \right)
400 return (cap_ - underlying_->effectiveSpread()) / gearing();
401 }
402 }
403}
404
405Rate CappedFlooredOvernightIndexedCoupon::effectiveFloor() const {
406 if (floor_ == Null<Real>())
407 return Null<Real>();
408 if (localCapFloor_) {
409 if (underlying_->includeSpread()) {
410 return floor_ - underlying_->spread();
411 } else {
412 return floor_;
413 }
414 } else {
415 if (underlying_->includeSpread()) {
416 return (floor_ - underlying_->effectiveSpread());
417 } else {
418 return (floor_ - underlying_->effectiveSpread()) / gearing();
419 }
420 }

570OvernightLeg::operator Leg() const {
571
572 QL_REQUIRE(!notionals_.empty(), "no notional given for compounding overnight leg");
573
574 Leg cashflows;
575
576 Calendar calendar = schedule_.calendar();
577 Calendar paymentCalendar = paymentCalendar_;
578
579 if (calendar.empty())
580 calendar = paymentCalendar;
581 if (calendar.empty())
582 calendar = WeekendsOnly();
583 if (paymentCalendar.empty())
584 paymentCalendar = calendar;
585
586 Date refStart, start, refEnd, end;
587 Date paymentDate;
588
589 Size n = schedule_.size() - 1;
590 for (Size i = 0; i < n; ++i) {
591 refStart = start = schedule_.date(i);
592 refEnd = end = schedule_.date(i + 1);
593 paymentDate = paymentCalendar.advance(end, paymentLag_, Days, paymentAdjustment_);
594
595 // determine refStart and refEnd
596
597 if (i == 0 && schedule_.hasIsRegular() && !schedule_.isRegular(i + 1))
598 refStart = calendar.adjust(end - schedule_.tenor(), paymentAdjustment_);
599 if (i == n - 1 && schedule_.hasIsRegular() && !schedule_.isRegular(i + 1))
600 refEnd = calendar.adjust(start + schedule_.tenor(), paymentAdjustment_);
601
602 // Determine the rate computation start and end date as
603 // - the coupon start and end date, if in arrears, and
604 // - the previous coupon start and end date, if in advance.
605 // In addition, adjust the start date, if a last recent period is given.
606
607 Date rateComputationStartDate, rateComputationEndDate;
608 if (inArrears_) {
609 // in arrears fixing (i.e. the "classic" case)
610 rateComputationStartDate = start;
611 rateComputationEndDate = end;
612 } else {
613 // handle in advance fixing
614 if (i > 0) {
615 // if there is a previous period, we take that
616 rateComputationStartDate = schedule_.date(i - 1);
617 rateComputationEndDate = schedule_.date(i);
618 } else {
619 // otherwise we construct the previous period
620 rateComputationEndDate = start;
621 if (schedule_.hasTenor() && schedule_.tenor() != 0 * Days)
622 rateComputationStartDate = calendar.adjust(start - schedule_.tenor(), Preceding);
623 else
624 rateComputationStartDate = calendar.adjust(start - (end - start), Preceding);
625 }
626 }
627
628 if (lastRecentPeriod_) {
629 rateComputationStartDate = (lastRecentPeriodCalendar_.empty() ? calendar : lastRecentPeriodCalendar_)
630 .advance(rateComputationEndDate, -*lastRecentPeriod_);
631 }
632
633 // build coupon
634
635 if (close_enough(detail::get(gearings_, i, 1.0), 0.0)) {
636 // fixed coupon
637 cashflows.push_back(boost::make_shared<FixedRateCoupon>(
638 paymentDate, detail::get(notionals_, i, 1.0), detail::effectiveFixedRate(spreads_, caps_, floors_, i),
639 paymentDayCounter_, start, end, refStart, refEnd));
640 } else {
641 // floating coupon
