36Date DynamicOptionletVolatilityStructure::maxDate() const {

59Volatility DynamicOptionletVolatilityStructure::volatilityImpl(Time optionTime, Rate strike) const {
60 if (decayMode_ == ConstantVariance) {
61 return source_->volatility(optionTime, strike);
62 }
63
64 // TODO: check validity of ForwardVariance option before using it.
65 QL_REQUIRE(decayMode_ != ForwardForwardVariance,
66 "ForwardVariance not yet supported for DynamicOptionletVolatilityStructure");
67 if (decayMode_ == ForwardForwardVariance) {
68 Real timeToRef = source_->timeFromReference(referenceDate());
69 Real varToRef = source_->blackVariance(timeToRef, strike);
70 Real varToOptTime = source_->blackVariance(timeToRef + optionTime, strike);
71 return std::sqrt((varToOptTime - varToRef) / optionTime);
72 }
73
74 QL_FAIL("Unexpected decay mode (" << decayMode_ << ")");
75}
