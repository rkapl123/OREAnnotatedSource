57 Volatility DynamicOptionletVolatilityStructure::volatilityImpl(Time optionTime, Rate strike) const {
58  if (decayMode_ == ConstantVariance) {
59  return source_->volatility(optionTime, strike);
60  }
61 
62  // TODO: check validity of ForwardVariance option before using it.
63  QL_REQUIRE(decayMode_ != ForwardForwardVariance,
64  "ForwardVariance not yet supported for DynamicOptionletVolatilityStructure");
65  if (decayMode_ == ForwardForwardVariance) {
66  Real timeToRef = source_->timeFromReference(referenceDate());
67  Real varToRef = source_->blackVariance(timeToRef, strike);
68  Real varToOptTime = source_->blackVariance(timeToRef + optionTime, strike);
69  return std::sqrt((varToOptTime - varToRef) / optionTime);
70  }
71 
72  QL_FAIL("Unexpected decay mode (" << decayMode_ << ")");
73 }
