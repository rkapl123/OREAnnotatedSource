72 void MidPointCdsEngineBase::calculate(const Date& refDate, const CreditDefaultSwap::arguments& arguments,
73  CreditDefaultSwap::results& results) const {
74  Date today = Settings::instance().evaluationDate();
75  Date settlementDate = discountCurve_->referenceDate();
76 
77  // Upfront amount.
78  Real upfPVO1 = 0.0;
79  results.upfrontNPV = 0.0;
80  if (arguments.upfrontPayment &&
81  !arguments.upfrontPayment->hasOccurred(settlementDate, includeSettlementDateFlows_)) {
82  upfPVO1 = discountCurve_->discount(arguments.upfrontPayment->date());
83  results.upfrontNPV = upfPVO1 * arguments.upfrontPayment->amount();
84  }
85 
86  // Accrual rebate.
87  results.accrualRebateNPV = 0.;
88  if (arguments.accrualRebate &&
89  !arguments.accrualRebate->hasOccurred(settlementDate, includeSettlementDateFlows_)) {
90  results.accrualRebateNPV = discountCurve_->discount(arguments.accrualRebate->date()) *
91  arguments.accrualRebate->amount();
92  }
93 
94  results.couponLegNPV = 0.0;
95  results.defaultLegNPV = 0.0;
96 
97  std::vector<Date> protectionPaymentDates;
98  std::vector<Real> midpointDiscounts;
99  std::vector<Real> expectedLosses;
100  std::vector<Real> defaultProbabilities;
101 
102  for (Size i = 0; i < arguments.leg.size(); ++i) {
103  if (arguments.leg[i]->hasOccurred(settlementDate, includeSettlementDateFlows_))
104  continue;
105 
106  boost::shared_ptr<FixedRateCoupon> coupon = boost::dynamic_pointer_cast<FixedRateCoupon>(arguments.leg[i]);
107 
108  // In order to avoid a few switches, we calculate the NPV
109  // of both legs as a positive quantity. We'll give them
110  // the right sign at the end.
111 
112  Date paymentDate = coupon->date(), startDate = coupon->accrualStartDate(), endDate = coupon->accrualEndDate();
113  // this is the only point where it might not coincide
114  if (i == 0)
115  startDate = arguments.protectionStart;
116  Date effectiveStartDate = (startDate <= today && today <= endDate) ? today : startDate;
117  Date defaultDate = // mid-point
118  effectiveStartDate + (endDate - effectiveStartDate) / 2;
119 
120  Probability S = survivalProbability(paymentDate);
121  Probability P = defaultProbability(effectiveStartDate, endDate);
122 
123  Date protectionPaymentDate;
124  if (arguments.protectionPaymentTime == CreditDefaultSwap::ProtectionPaymentTime::atDefault) {
125  protectionPaymentDate = defaultDate;
126  } else if (arguments.protectionPaymentTime == CreditDefaultSwap::ProtectionPaymentTime::atPeriodEnd) {
127  protectionPaymentDate = paymentDate;
128  } else if (arguments.protectionPaymentTime == CreditDefaultSwap::ProtectionPaymentTime::atMaturity) {
129  protectionPaymentDate = arguments.maturity;
130  } else {
131  QL_FAIL("protectionPaymentTime not handled");
132  }
133 
134  // on one side, we add the fixed rate payments in case of
135  // survival...
136  results.couponLegNPV += S * coupon->amount() * discountCurve_->discount(paymentDate);
137  // ...possibly including accrual in case of default.
138  if (arguments.settlesAccrual) {
139  results.couponLegNPV +=
140  P * coupon->accruedAmount(defaultDate) * discountCurve_->discount(protectionPaymentDate);
141  }
142 
143  // on the other side, we add the payment in case of default.
144  Real midpointDiscount = discountCurve_->discount(protectionPaymentDate);
145  Real expectLoss = expectedLoss(defaultDate, effectiveStartDate, endDate, coupon->nominal());
146  results.defaultLegNPV += expectLoss *
147  midpointDiscount;
148 
149  protectionPaymentDates.push_back(protectionPaymentDate);
150  midpointDiscounts.push_back(midpointDiscount);
151  expectedLosses.push_back(expectLoss);
152  defaultProbabilities.push_back(P);
153  }
154 
155  results.additionalResults["protectionPaymentDates"] = protectionPaymentDates;
156  results.additionalResults["midpointDiscounts"] = midpointDiscounts;
157  results.additionalResults["expectedLosses"] = expectedLosses;
158  results.additionalResults["defaultProbabilities"] = defaultProbabilities;
159 
160  Real upfrontSign = 1.0;
161  switch (arguments.side) {
162  case Protection::Seller:
163  results.defaultLegNPV *= -1.0;
164  results.accrualRebateNPV *= -1.0;
165  break;
166  case Protection::Buyer:
167  results.couponLegNPV *= -1.0;
168  results.upfrontNPV *= -1.0;
169  upfrontSign = -1.0;
170  break;
171  default:
172  QL_FAIL("unknown protection side");
173  }
174 
175  results.value = results.defaultLegNPV + results.couponLegNPV + results.upfrontNPV + results.accrualRebateNPV;
176  results.errorEstimate = Null<Real>();
177 
178  if (results.couponLegNPV != 0.0) {
179  results.fairSpread =
180  -results.defaultLegNPV * arguments.spread / (results.couponLegNPV + results.accrualRebateNPV);
181  } else {
182  results.fairSpread = Null<Rate>();
183  }
184 
185  Real upfrontSensitivity = upfPVO1 * arguments.notional;
186  if (upfrontSensitivity > 0.0) {
187  results.fairUpfront = -upfrontSign * (results.defaultLegNPV + results.couponLegNPV +
188  results.accrualRebateNPV) / upfrontSensitivity;
189  } else {
190  results.fairUpfront = Null<Rate>();
191  }
192 
193  static const Rate basisPoint = 1.0e-4;
194 
195  if (arguments.spread != 0.0) {
196  results.couponLegBPS = results.couponLegNPV * basisPoint / arguments.spread;
197  } else {
198  results.couponLegBPS = Null<Rate>();
199  }
200 
201  if (arguments.upfront && *arguments.upfront != 0.0) {
202  results.upfrontBPS = results.upfrontNPV * basisPoint / (*arguments.upfront);
203  } else {
204  results.upfrontBPS = Null<Rate>();
205  }
206 
207  results.additionalResults["upfrontPremium"] = arguments.upfrontPayment->amount();
208  results.additionalResults["upfrontPremiumNPV"] = results.upfrontNPV;
209  results.additionalResults["premiumLegNPV"] = results.couponLegNPV;
210  results.additionalResults["accrualRebateNPV"] = results.accrualRebateNPV;
211  results.additionalResults["protectionLegNPV"] = results.defaultLegNPV;
212 
213 } // MidPointCdsEngineBase::calculate()
