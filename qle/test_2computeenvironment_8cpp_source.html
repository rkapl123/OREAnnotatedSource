185 c.declareOutputVariable(r);
186 }
187 }
188 std::vector<std::vector<float>> output(6, std::vector<float>(n));
189 c.finalizeCalculation(output);
190 outputTimings(c);
191 for (auto const& o : output) {
192 boost::accumulators::accumulator_set<
193 float, boost::accumulators::stats<boost::accumulators::tag::mean, boost::accumulators::tag::variance>>
194 acc;
195 for (auto const& v : o) {
196 acc(v);
197 }
198 BOOST_TEST_MESSAGE(" mean = " << boost::accumulators::mean(acc)
199 << ", variance = " << boost::accumulators::variance(acc));
200 BOOST_CHECK_SMALL(boost::accumulators::mean(acc), 0.05f);
201 BOOST_CHECK_CLOSE(boost::accumulators::variance(acc), 1.0f, 1.0f);
202 }
