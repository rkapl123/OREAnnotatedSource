63inline Real RebatedExercise::rebate(Size index) const {
64 QL_REQUIRE(index < rebates_.size(),
65 "rebate with index " << index << " does not exist (0..." << (rebates_.size() - 1) << ")");
66 return rebates_[index];
67}
68
69inline Date RebatedExercise::rebatePaymentDate(Size index) const {
70 QL_REQUIRE(type_ == European || type_ == Bermudan, "for american style exercises the rebate payment date "
71 << "has to be calculted in the client code");
72 Date baseDate = exerciseDates_.empty() ? dates_[index] : exerciseDates_[index];
73 Period settlPeriod = rebateSettlementPeriod_ ? *rebateSettlementPeriod_ : rebateSettlementDays_ * Days;
74 return rebatePaymentCalendar_.advance(baseDate, settlPeriod, rebatePaymentConvention_);
75}
