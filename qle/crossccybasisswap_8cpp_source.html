55void CrossCcyBasisSwap::initialize() {
56 // Pay leg
57 if (auto on = boost::dynamic_pointer_cast<QuantLib::OvernightIndex>(payIndex_)) {
58 // ON leg
59 if (payIsAveraged_ && *payIsAveraged_) {
60 legs_[0] = QuantExt::AverageONLeg(paySchedule_, on)
61 .withNotional(payNominal_)
62 .withSpread(paySpread_)
63 .withGearing(payGearing_)
64 .withPaymentLag(payPaymentLag_)
65 .withLookback(payLookback_ ? *payLookback_ : 0 * Days)
66 .withFixingDays(payFixingDays_ ? *payFixingDays_ : 0)
67 .withRateCutoff(payRateCutoff_ ? *payRateCutoff_ : 0)
68 .withTelescopicValueDates(telescopicValueDates_);
69 } else {
70 legs_[0] = QuantExt::OvernightLeg(paySchedule_, on)
71 .withNotionals(payNominal_)
72 .withSpreads(paySpread_)
73 .withGearings(payGearing_)
74 .withPaymentLag(payPaymentLag_)
75 .includeSpread(payIncludeSpread_ ? *payIncludeSpread_ : false)
76 .withLookback(payLookback_ ? *payLookback_ : 0 * Days)
77 .withFixingDays(payFixingDays_ ? *payFixingDays_ : 0)
78 .withRateCutoff(payRateCutoff_ ? *payRateCutoff_ : 0)
79 .withTelescopicValueDates(telescopicValueDates_);
80 }
81 } else {
82 // Ibor leg
83 legs_[0] = IborLeg(paySchedule_, payIndex_)
84 .withNotionals(payNominal_)
85 .withSpreads(paySpread_)
86 .withGearings(payGearing_)
87 .withPaymentLag(payPaymentLag_);
88 }
89 payer_[0] = -1.0;
90 currencies_[0] = payCurrency_;
91 // Pay leg notional exchange at start.
92 Date initialPayDate = paySchedule_.dates().front();
93 boost::shared_ptr<CashFlow> initialPayCF(new SimpleCashFlow(-payNominal_, initialPayDate));
94 legs_[0].insert(legs_[0].begin(), initialPayCF);
95 // Pay leg notional exchange at end.
96 Date finalPayDate = paySchedule_.dates().back();
97 boost::shared_ptr<CashFlow> finalPayCF(new SimpleCashFlow(payNominal_, finalPayDate));
98 legs_[0].push_back(finalPayCF);
99
100 // Receive leg
101 if (auto on = boost::dynamic_pointer_cast<QuantLib::OvernightIndex>(recIndex_)) {
102 // ON leg
103 if (recIsAveraged_ && *recIsAveraged_) {
104 legs_[1] = QuantExt::AverageONLeg(recSchedule_, on)
105 .withNotional(recNominal_)
106 .withSpread(recSpread_)
107 .withGearing(recGearing_)
108 .withPaymentLag(recPaymentLag_)
109 .withLookback(recLookback_ ? *recLookback_ : 0 * Days)
110 .withFixingDays(recFixingDays_ ? *recFixingDays_ : 0)
111 .withRateCutoff(recRateCutoff_ ? *recRateCutoff_ : 0)
112 .withTelescopicValueDates(telescopicValueDates_);
113 } else {
114 legs_[1] = QuantExt::OvernightLeg(recSchedule_, on)
115 .withNotionals(recNominal_)
116 .withSpreads(recSpread_)
117 .withGearings(recGearing_)
118 .withPaymentLag(recPaymentLag_)
119 .includeSpread(recIncludeSpread_ ? *recIncludeSpread_ : false)
120 .withLookback(recLookback_ ? *recLookback_ : 0 * Days)
121 .withFixingDays(recFixingDays_ ? *recFixingDays_ : 0)
122 .withRateCutoff(recRateCutoff_ ? *recRateCutoff_ : 0)
123 .withTelescopicValueDates(telescopicValueDates_);
124 }
125 } else {
126 // Ibor leg
127 legs_[1] = IborLeg(recSchedule_, recIndex_)
128 .withNotionals(recNominal_)
129 .withSpreads(recSpread_)
130 .withGearings(recGearing_)
131 .withPaymentLag(recPaymentLag_);
132 }
133 payer_[1] = +1.0;
134 currencies_[1] = recCurrency_;
135 // Receive leg notional exchange at start.
136 Date initialRecDate = recSchedule_.dates().front();
137 boost::shared_ptr<CashFlow> initialRecCF(new SimpleCashFlow(-recNominal_, initialRecDate));
138 legs_[1].insert(legs_[1].begin(), initialRecCF);
139 // Receive leg notional exchange at end.
140 Date finalRecDate = recSchedule_.dates().back();
141 boost::shared_ptr<CashFlow> finalRecCF(new SimpleCashFlow(recNominal_, finalRecDate));
142 legs_[1].push_back(finalRecCF);
143
144 // Register the instrument with all cashflows on each leg.
145 for (Size legNo = 0; legNo < 2; legNo++) {
146 Leg::iterator it;
147 for (it = legs_[legNo].begin(); it != legs_[legNo].end(); ++it) {
148 registerWith(*it);
149 }
150 }
151}

154
155 CrossCcySwap::setupArguments(args);

168void CrossCcyBasisSwap::fetchResults(const PricingEngine::results* r) const {
169
170 CrossCcySwap::fetchResults(r);
171
172 const CrossCcyBasisSwap::results* results = dynamic_cast<const CrossCcyBasisSwap::results*>(r);
173 if (results) {
174 /* If PricingEngine::results are of type
175 CrossCcyBasisSwap::results */
176 fairPaySpread_ = results->fairPaySpread;
177 fairRecSpread_ = results->fairRecSpread;
178 } else {
179 /* If not, e.g. if the engine is a CrossCcySwap::engine */
180 fairPaySpread_ = Null<Spread>();
181 fairRecSpread_ = Null<Spread>();
182 }
183
184 /* Calculate the fair pay and receive spreads if they are null */
185 static Spread basisPoint = 1.0e-4;
186 if (fairPaySpread_ == Null<Spread>()) {
187 if (legBPS_[0] != Null<Real>())
188 fairPaySpread_ = paySpread_ - NPV_ / (legBPS_[0] / basisPoint);
189 }
190 if (fairRecSpread_ == Null<Spread>()) {
191 if (legBPS_[1] != Null<Real>())
192 fairRecSpread_ = recSpread_ - NPV_ / (legBPS_[1] / basisPoint);
193 }
194}
