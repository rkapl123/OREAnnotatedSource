42 void CrossCcyBasisSwap::initialize() {
43  // Pay leg
44  legs_[0] =
45  IborLeg(paySchedule_, payIndex_).withNotionals(payNominal_).withSpreads(paySpread_).withGearings(payGearing_);
46  payer_[0] = -1.0;
47  currencies_[0] = payCurrency_;
48  // Pay leg notional exchange at start.
49  Date initialPayDate = paySchedule_.dates().front();
50  boost::shared_ptr<CashFlow> initialPayCF(new SimpleCashFlow(-payNominal_, initialPayDate));
51  legs_[0].insert(legs_[0].begin(), initialPayCF);
52  // Pay leg notional exchange at end.
53  Date finalPayDate = paySchedule_.dates().back();
54  boost::shared_ptr<CashFlow> finalPayCF(new SimpleCashFlow(payNominal_, finalPayDate));
55  legs_[0].push_back(finalPayCF);
56 
57  // Receive leg
58  legs_[1] =
59  IborLeg(recSchedule_, recIndex_).withNotionals(recNominal_).withSpreads(recSpread_).withGearings(recGearing_);
60  payer_[1] = +1.0;
61  currencies_[1] = recCurrency_;
62  // Receive leg notional exchange at start.
63  Date initialRecDate = recSchedule_.dates().front();
64  boost::shared_ptr<CashFlow> initialRecCF(new SimpleCashFlow(-recNominal_, initialRecDate));
65  legs_[1].insert(legs_[1].begin(), initialRecCF);
66  // Receive leg notional exchange at end.
67  Date finalRecDate = recSchedule_.dates().back();
68  boost::shared_ptr<CashFlow> finalRecCF(new SimpleCashFlow(recNominal_, finalRecDate));
69  legs_[1].push_back(finalRecCF);
70 
71  // Register the instrument with all cashflows on each leg.
72  for (Size legNo = 0; legNo < 2; legNo++) {
73  Leg::iterator it;
74  for (it = legs_[legNo].begin(); it != legs_[legNo].end(); ++it) {
75  registerWith(*it);
76  }
77  }
78 }

95 void CrossCcyBasisSwap::fetchResults(const PricingEngine::results* r) const {
96 
97  CrossCcySwap::fetchResults(r);
98 
99  const CrossCcyBasisSwap::results* results = dynamic_cast<const CrossCcyBasisSwap::results*>(r);
100  if (results) {
101  /* If PricingEngine::results are of type
102  CrossCcyBasisSwap::results */
103  fairPaySpread_ = results->fairPaySpread;
104  fairRecSpread_ = results->fairRecSpread;
105  } else {
106  /* If not, e.g. if the engine is a CrossCcySwap::engine */
107  fairPaySpread_ = Null<Spread>();
108  fairRecSpread_ = Null<Spread>();
109  }
110 
111  /* Calculate the fair pay and receive spreads if they are null */
112  static Spread basisPoint = 1.0e-4;
113  if (fairPaySpread_ == Null<Spread>()) {
114  if (legBPS_[0] != Null<Real>())
115  fairPaySpread_ = paySpread_ - NPV_ / (legBPS_[0] / basisPoint);
116  }
117  if (fairRecSpread_ == Null<Spread>()) {
118  if (legBPS_[1] != Null<Real>())
119  fairRecSpread_ = recSpread_ - NPV_ / (legBPS_[1] / basisPoint);
120  }
121 }
