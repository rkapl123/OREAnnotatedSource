29 MakeOISCapFloor::operator Leg() const {
30  Calendar calendar = index_->fixingCalendar();
31 
32  Date startDate;
33  if (effectiveDate_ != Date()) {
34  startDate = effectiveDate_;
35  } else {
36  Date refDate = Settings::instance().evaluationDate();
37  startDate = calendar.advance(calendar.adjust(refDate), settlementDays_ * Days);
38  }
39 
40  Date endDate = calendar.adjust(startDate + tenor_, ModifiedFollowing);
41 
42  Schedule schedule(startDate, endDate, rateComputationPeriod_, calendar, ModifiedFollowing, ModifiedFollowing,
43  DateGeneration::Backward, false);
44 
45  Real cap = Null<Real>(), floor = Null<Real>();
46  if (type_ == CapFloor::Cap)
47  cap = strike_;
48  else if (type_ == CapFloor::Floor)
49  floor = strike_;
50  else {
51  QL_FAIL("MakeOISCapFloor: expected type Cap or Floor");
52  }
53 
54  Leg leg = OvernightLeg(schedule, index_)
55  .withNotionals(nominal_)
56  .withPaymentDayCounter(dayCounter_)
57  .withPaymentAdjustment(convention_)
58  .withCaps(cap)
59  .withFloors(floor)
60  .withNakedOption(true)
61  .withTelescopicValueDates(telescopicValueDates_);
62 
63  if (pricer_) {
64  for (auto c : leg) {
65  auto f = boost::dynamic_pointer_cast<FloatingRateCoupon>(c);
66  if (f) {
67  f->setPricer(pricer_);
68  }
69  }
70  }
71 
72  return leg;
73 }

121 //! get the underlying ON coupons from an OIS cf
122 Leg getOisCapFloorUnderlying(const Leg& oisCapFloor) {
123  Leg underlying;
124  for (auto const& c : oisCapFloor) {
125  auto cfon = boost::dynamic_pointer_cast<CappedFlooredOvernightIndexedCoupon>(c);
126  QL_REQUIRE(cfon, "getOisCapFloorUnderlying(): expected CappedFlooredOvernightIndexedCoupon");
127  underlying.push_back(cfon->underlying());
128  }
129  return underlying;
130 }
