25 SpreadedDiscountCurve::SpreadedDiscountCurve(const Handle<YieldTermStructure>& referenceCurve,
26  const std::vector<Time>& times, const std::vector<Handle<Quote>>& quotes,
27  const Interpolation interpolation, const Extrapolation extrapolation)
28  : YieldTermStructure(referenceCurve->dayCounter()), referenceCurve_(referenceCurve), times_(times), quotes_(quotes),
29  interpolation_(interpolation), extrapolation_(extrapolation), data_(times_.size(), 1.0) {
30  QL_REQUIRE(times_.size() > 1, "SpreadedDiscountCurve: at least two times required");
31  QL_REQUIRE(times_.size() == quotes.size(), "SpreadedDiscountCurve: size of time and quote vectors do not match");
32  QL_REQUIRE(times_[0] == 0.0, "SpreadedDiscountCurve: first time must be 0, got " << times_[0]);
33  for (Size i = 0; i < quotes.size(); ++i) {
34  registerWith(quotes_[i]);
35  }
36  if (interpolation_ == Interpolation::logLinear) {
37  dataInterpolation_ = boost::make_shared<LogLinearInterpolation>(times_.begin(), times_.end(), data_.begin());
38  } else {
39  dataInterpolation_ = boost::make_shared<LinearInterpolation>(times_.begin(), times_.end(), data_.begin());
40  }
41  dataInterpolation_->enableExtrapolation();
42  registerWith(referenceCurve_);
43 }

58 void SpreadedDiscountCurve::performCalculations() const {
59  for (Size i = 0; i < times_.size(); ++i) {
60  QL_REQUIRE(!quotes_[i].empty(), "SpreadedDiscountCurve: quote at index " << i << " is empty");
61  data_[i] = quotes_[i]->value();
62  QL_REQUIRE(data_[i] > 0, "SpreadedDiscountCurve: invalid value " << data_[i] << " at index " << i);
63  if (interpolation_ == Interpolation::linearZero) {
64  data_[i] = -std::log(data_[std::max<Size>(i, 1)]) / times_[std::max<Size>(i, 1)];
65  }
66  }
67  dataInterpolation_->update();
68 }

70 DiscountFactor SpreadedDiscountCurve::discountImpl(Time t) const {
71  calculate();
72  Time tMax = this->times_.back();
73  DiscountFactor dMax =
74  interpolation_ == Interpolation::logLinear ? this->data_.back() : std::exp(-this->data_.back() * tMax);
75  if (t <= this->times_.back()) {
76  Real tmp = (*dataInterpolation_)(t, true);
77  if (interpolation_ == Interpolation::logLinear)
78  return referenceCurve_->discount(t) * tmp;
79  else
80  return referenceCurve_->discount(t) * std::exp(-tmp * t);
81  }
82  if (extrapolation_ == Extrapolation::flatFwd) {
83  Rate instFwdMax = -(*dataInterpolation_).derivative(tMax) / dMax;
84  return referenceCurve_->discount(t) * dMax * std::exp(-instFwdMax * (t - tMax));
85  } else {
86  return referenceCurve_->discount(t) * std::pow(dMax, t / tMax);
87  }
88 }
