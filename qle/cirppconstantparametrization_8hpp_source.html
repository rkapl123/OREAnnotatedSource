46template <class TS>
47CirppConstantParametrization<TS>::CirppConstantParametrization(const Currency& currency,
48 const Handle<TS>& termStructure,
49 const Real kappa, const Real theta,
50 const Real sigma, const Real y0,
51 const bool shifted,
52 const std::string& name)
53 : CirppParametrization<TS>(currency, termStructure, shifted, name),
54 kappa_(boost::make_shared<PseudoParameter>(1)),
55 theta_(boost::make_shared<PseudoParameter>(1)), sigma_(boost::make_shared<PseudoParameter>(1)),
56 y0_(boost::make_shared<PseudoParameter>(1)) {
57 kappa_->setParam(0, inverse(0, kappa));
58 theta_->setParam(0, inverse(1, theta));
59 sigma_->setParam(0, inverse(2, sigma));
60 y0_->setParam(0, inverse(3, y0));
61}

63template <class TS> inline Real CirppConstantParametrization<TS>::direct(const Size i, const Real x) const {
64 constexpr Real eps = 1E-10;
65 return x * x + eps;
66}
67
68template <class TS> inline Real CirppConstantParametrization<TS>::inverse(const Size i, const Real y) const {
69 constexpr Real eps = 1E-10;
70 return std::sqrt(y - eps);
71}
72
73template <class TS> inline Real CirppConstantParametrization<TS>::kappa(const Time) const {
74 return direct(0, kappa_->params()[0]);
75}

77template <class TS> inline Real CirppConstantParametrization<TS>::theta(const Time) const {

81template <class TS> inline Real CirppConstantParametrization<TS>::sigma(const Time) const {

85template <class TS> inline Real CirppConstantParametrization<TS>::y0(const Time) const {

89template <class TS>
90inline const boost::shared_ptr<Parameter> CirppConstantParametrization<TS>::parameter(const Size i) const {
91 QL_REQUIRE(i < 4, "parameter " << i << " does not exist, only have 0..3");
92 if (i == 0)
93 return kappa_;
94 else if (i == 1)
95 return theta_;
96 else if (i == 2)
97 return sigma_;
98 else
99 return y0_;
100}
