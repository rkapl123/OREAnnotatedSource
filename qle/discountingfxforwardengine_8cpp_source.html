38void DiscountingFxForwardEngine::calculate() const {
39 Date npvDate = npvDate_;
40 if (npvDate == Null<Date>()) {
41 npvDate = currency1Discountcurve_->referenceDate();
42 }
43 Date settlementDate = settlementDate_;
44 if (settlementDate == Null<Date>()) {
45 settlementDate = npvDate;
46 }
47
48 Real tmpNominal1, tmpNominal2;
49 bool tmpPayCurrency1;
50 if (ccy1_ == arguments_.currency1) {
51 QL_REQUIRE(ccy2_ == arguments_.currency2, "mismatched currency pairs ("
52 << ccy1_ << "," << ccy2_ << ") in the engine and ("
53 << arguments_.currency1 << "," << arguments_.currency2
54 << ") in the instrument");
55 tmpNominal1 = arguments_.nominal1;
56 tmpNominal2 = arguments_.nominal2;
57 tmpPayCurrency1 = arguments_.payCurrency1;
58 } else {
59 QL_REQUIRE(ccy1_ == arguments_.currency2 && ccy2_ == arguments_.currency1,
60 "mismatched currency pairs (" << ccy1_ << "," << ccy2_ << ") in the engine and ("
61 << arguments_.currency1 << "," << arguments_.currency2
62 << ") in the instrument");
63 tmpNominal1 = arguments_.nominal2;
64 tmpNominal2 = arguments_.nominal1;
65 tmpPayCurrency1 = !arguments_.payCurrency1;
66 }
67
68 QL_REQUIRE(!currency1Discountcurve_.empty() && !currency2Discountcurve_.empty(),
69 "Discounting term structure handle is empty.");
70
71 QL_REQUIRE(currency1Discountcurve_->referenceDate() == currency2Discountcurve_->referenceDate(),
72 "Term structures should have the same reference date.");
73
74 QL_REQUIRE(arguments_.payDate >= currency1Discountcurve_->referenceDate(),
75 "FX forward maturity should exceed or equal the "
76 "discount curve reference date.");
77
78 results_.value = 0.0;
79 results_.fairForwardRate = ExchangeRate(ccy2_, ccy1_, tmpNominal1 / tmpNominal2); // strike rate
80 results_.additionalResults["fairForwardRate"] = tmpNominal1 / tmpNominal2;
81 results_.additionalResults["currency[1]"] = ccy1_.code();
82 results_.additionalResults["currency[2]"] = ccy2_.code();
83
84 if (!detail::simple_event(arguments_.payDate).hasOccurred(settlementDate, includeSettlementDateFlows_)) {
85 Real disc1near = currency1Discountcurve_->discount(npvDate);
86 Real disc1far = currency1Discountcurve_->discount(arguments_.payDate);
87 Real disc2near = currency2Discountcurve_->discount(npvDate);
88 Real disc2far = currency2Discountcurve_->discount(arguments_.payDate);
89 Real fxfwd = disc1near / disc1far * disc2far / disc2near * spotFX_->value();
90
91 // settle ccy is ccy1 if no pay ccy provided
92 Currency settleCcy = arguments_.payCcy.empty() ? ccy1_ : arguments_.payCcy;
93 bool settleCcy1 = ccy1_ == settleCcy;
94
95 Real discNear = settleCcy1 ? disc1near : disc2near;
96 Real discFar = settleCcy1 ? disc1far : disc2far;
97 Real fx1 = settleCcy1 ? 1.0 : fxfwd;
98 Real fx2 = settleCcy1 ? 1 / fxfwd : 1.0;
99
100 QL_REQUIRE(arguments_.isPhysicallySettled || arguments_.payDate <= arguments_.fixingDate ||
101 arguments_.fxIndex != nullptr,
102 "If pay date (" << arguments_.payDate << ") is strictly after fixing date (" << arguments_.fixingDate
103 << "), an FX Index must be given for a cash-settled FX Forward.");
104 if (!arguments_.isPhysicallySettled && arguments_.payDate >= arguments_.fixingDate &&
105 arguments_.fxIndex != nullptr) {
106 fx1 = settleCcy1 ? 1.0 : arguments_.fxIndex->fixing(arguments_.fixingDate);
107 fx2 = settleCcy1 ? arguments_.fxIndex->fixing(arguments_.fixingDate) : 1.0;
108 fxfwd = arguments_.fxIndex->fixing(arguments_.fixingDate);
109 }
110
111 // populate cashflow results
112 std::vector<CashFlowResults> cashFlowResults;
113 CashFlowResults cf1, cf2;
114 cf1.payDate = arguments_.payDate;
115 cf1.type = "Notional";
116 cf2.payDate = arguments_.payDate;
117 cf2.type = "Notional";
118 if (!arguments_.isPhysicallySettled) {
119 if (arguments_.payDate >= arguments_.fixingDate) {
120 cf1.fixingDate = arguments_.fixingDate;
121 cf2.fixingDate = arguments_.fixingDate;
122 }
123 cf1.fixingValue = 1.0 / fx1;
124 cf2.fixingValue = 1.0 / fx2;
125 cf1.amount = (tmpPayCurrency1 ? -1.0 : 1.0) * tmpNominal1 / fx1;
126 cf2.amount = (tmpPayCurrency1 ? -1.0 : 1.0) * (-tmpNominal2 / fx2);
127 cf1.currency = settleCcy.code();
128 cf2.currency = settleCcy.code();
129 } else {
130 cf1.amount = (tmpPayCurrency1 ? -1.0 : 1.0) * tmpNominal1;
131 cf2.amount = (tmpPayCurrency1 ? -1.0 : 1.0) * (-tmpNominal2);
132 cf1.currency = ccy1_.code();
133 cf2.currency = ccy2_.code();
134 }
135 if (ccy1_ == arguments_.currency1) {
136 cf1.legNumber = 0;
137 cf2.legNumber = 1;
138 cashFlowResults.push_back(cf1);
139 cashFlowResults.push_back(cf2);
140 } else {
141 cf1.legNumber = 1;
142 cf2.legNumber = 0;
143 cashFlowResults.push_back(cf2);
144 cashFlowResults.push_back(cf1);
145 }
146 results_.additionalResults["cashFlowResults"] = cashFlowResults;
147
148 results_.value = (tmpPayCurrency1 ? -1.0 : 1.0) * discFar / discNear * (tmpNominal1 / fx1 - tmpNominal2 / fx2);
149
150 results_.npv = Money(settleCcy, results_.value);
151
152 results_.fairForwardRate = ExchangeRate(ccy2_, ccy1_, fxfwd);
153 results_.additionalResults["fairForwardRate"] = fxfwd;
154 results_.additionalResults["fxSpot"] = spotFX_->value();
155 results_.additionalResults["discountFactor[1]"] = disc1far;
156 results_.additionalResults["discountFactor[2]"] = disc2far;
157 results_.additionalResults["legNPV[1]"] = (tmpPayCurrency1 ? -1.0 : 1.0) * discFar / discNear * tmpNominal1 / fx1;
158 results_.additionalResults["legNPV[2]"] = (tmpPayCurrency1 ? -1.0 : 1.0) * discFar / discNear * (-tmpNominal2 / fx2);
159
160 // set notional
161 if (arguments_.isPhysicallySettled) {
162 // Align notional with ISDA AANA/GRID guidance as of November 2020 for deliverable forwards
163 if (tmpNominal1 > tmpNominal2 * fxfwd) {
164 results_.additionalResults["currentNotional"] = tmpNominal1;
165 results_.additionalResults["notionalCurrency"] = ccy1_.code();
166 } else {
167 results_.additionalResults["currentNotional"] = tmpNominal2;
168 results_.additionalResults["notionalCurrency"] = ccy2_.code();
169 }
170 } else {
171 // for cash settled forwards we take the notional from the settlement ccy leg
172 results_.additionalResults["currentNotional"] =
173 arguments_.currency1 == arguments_.payCcy ? arguments_.nominal1 : arguments_.nominal2;
174 results_.additionalResults["notionalCurrency"] = arguments_.payCcy.code();
175 }
176 }
177
178} // calculate
