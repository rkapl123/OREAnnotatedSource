37 void DiscountingFxForwardEngine::calculate() const {
38  Date npvDate = npvDate_;
39  if (npvDate == Null<Date>()) {
40  npvDate = currency1Discountcurve_->referenceDate();
41  }
42  Date settlementDate = settlementDate_;
43  if (settlementDate == Null<Date>()) {
44  settlementDate = npvDate;
45  }
46 
47  Real tmpNominal1, tmpNominal2;
48  bool tmpPayCurrency1;
49  if (ccy1_ == arguments_.currency1) {
50  QL_REQUIRE(ccy2_ == arguments_.currency2, "mismatched currency pairs ("
51  << ccy1_ << "," << ccy2_ << ") in the engine and ("
52  << arguments_.currency1 << "," << arguments_.currency2
53  << ") in the instrument");
54  tmpNominal1 = arguments_.nominal1;
55  tmpNominal2 = arguments_.nominal2;
56  tmpPayCurrency1 = arguments_.payCurrency1;
57  } else {
58  QL_REQUIRE(ccy1_ == arguments_.currency2 && ccy2_ == arguments_.currency1,
59  "mismatched currency pairs (" << ccy1_ << "," << ccy2_ << ") in the engine and ("
60  << arguments_.currency1 << "," << arguments_.currency2
61  << ") in the instrument");
62  tmpNominal1 = arguments_.nominal2;
63  tmpNominal2 = arguments_.nominal1;
64  tmpPayCurrency1 = !arguments_.payCurrency1;
65  }
66 
67  QL_REQUIRE(!currency1Discountcurve_.empty() && !currency2Discountcurve_.empty(),
68  "Discounting term structure handle is empty.");
69 
70  QL_REQUIRE(currency1Discountcurve_->referenceDate() == currency2Discountcurve_->referenceDate(),
71  "Term structures should have the same reference date.");
72 
73  QL_REQUIRE(arguments_.payDate >= currency1Discountcurve_->referenceDate(),
74  "FX forward maturity should exceed or equal the "
75  "discount curve reference date.");
76 
77  results_.value = 0.0;
78  results_.fairForwardRate = ExchangeRate(ccy2_, ccy1_, tmpNominal1 / tmpNominal2); // strike rate
79  results_.additionalResults["fairForwardRate"] = tmpNominal1 / tmpNominal2;
80  results_.additionalResults["currency[1]"] = ccy1_.code();
81  results_.additionalResults["currency[2]"] = ccy2_.code();
82 
83  if (!detail::simple_event(arguments_.payDate).hasOccurred(settlementDate, includeSettlementDateFlows_)) {
84  Real disc1near = currency1Discountcurve_->discount(npvDate);
85  Real disc1far = currency1Discountcurve_->discount(arguments_.payDate);
86  Real disc2near = currency2Discountcurve_->discount(npvDate);
87  Real disc2far = currency2Discountcurve_->discount(arguments_.payDate);
88  Real fxfwd = disc1near / disc1far * disc2far / disc2near * spotFX_->value();
89 
90  // settle ccy is ccy1 if no pay ccy provided
91  Currency settleCcy = arguments_.payCcy.empty() ? ccy1_ : arguments_.payCcy;
92  bool settleCcy1 = ccy1_ == settleCcy;
93 
94  Real discNear = settleCcy1 ? disc1near : disc2near;
95  Real discFar = settleCcy1 ? disc1far : disc2far;
96  Real fx1 = settleCcy1 ? 1.0 : fxfwd;
97  Real fx2 = settleCcy1 ? 1 / fxfwd : 1.0;
98 
99  if (!arguments_.isPhysicallySettled && arguments_.payDate > arguments_.fixingDate) {
100  fx1 = settleCcy1 ? 1.0 : arguments_.fxIndex->fixing(arguments_.fixingDate);
101  fx2 = settleCcy1 ? arguments_.fxIndex->fixing(arguments_.fixingDate) : 1.0;
102  fxfwd = arguments_.fxIndex->fixing(arguments_.fixingDate);
103  }
104 
105  results_.value = (tmpPayCurrency1 ? -1.0 : 1.0) * discFar / discNear * (tmpNominal1 / fx1 - tmpNominal2 / fx2);
106 
107  results_.npv = Money(arguments_.payCcy, results_.value);
108 
109  results_.fairForwardRate = ExchangeRate(ccy2_, ccy1_, fxfwd);
110  results_.additionalResults["fairForwardRate"] = fxfwd;
111  results_.additionalResults["fxSpot"] = spotFX_->value();
112  results_.additionalResults["discountFactor[1]"] = disc1far;
113  results_.additionalResults["discountFactor[2]"] = disc2far;
114  results_.additionalResults["legNPV[1]"] = (tmpPayCurrency1 ? -1.0 : 1.0) * discFar / discNear * tmpNominal1 / fx1;
115  results_.additionalResults["legNPV[2]"] = (tmpPayCurrency1 ? -1.0 : 1.0) * discFar / discNear * (-tmpNominal2 / fx2);
116 
117  // set notional
118  if (arguments_.isPhysicallySettled) {
119  // Align notional with ISDA AANA/GRID guidance as of November 2020 for deliverable forwards
120  if (tmpNominal1 > tmpNominal2 * fxfwd) {
121  results_.additionalResults["currentNotional"] = tmpNominal1;
122  results_.additionalResults["notionalCurrency"] = ccy1_.code();
123  } else {
124  results_.additionalResults["currentNotional"] = tmpNominal2;
125  results_.additionalResults["notionalCurrency"] = ccy2_.code();
126  }
127  } else {
128  // for cash settled forwards we take the notional from the settlement ccy leg
129  results_.additionalResults["currentNotional"] =
130  arguments_.currency1 == arguments_.payCcy ? arguments_.nominal1 : arguments_.nominal2;
131  results_.additionalResults["notionalCurrency"] = arguments_.payCcy.code();
132  }
133  }
134
