55 Real FxRateQuote::value() const {
56  QL_ENSURE(isValid(), "invalid FxRateQuote");
57  if (fixingDays_ == 0)
58  return spotQuote_->value();
59  else {
60  QL_REQUIRE(!sourceYts_.empty() && !targetYts_.empty(),
61  "FxRateQuote: empty curve handles, need curves to discount from spot to today");
62  Date today = sourceYts_->referenceDate();
63  Date refValueDate = fixingCalendar_.advance(today, fixingDays_, Days);
64  return spotQuote_->value() * targetYts_->discount(refValueDate) / sourceYts_->discount(refValueDate);
65  }
66 }

133 const Handle<Quote> FxIndex::fxQuote(bool withSettlementLag) const {
134  Handle<Quote> quote;
135  if (withSettlementLag || fixingDays_ == 0)
136  quote = fxSpot_;
137 
138  if (quote.empty()) {
139  if (fxRate_.empty()) {
140  Handle<Quote> tmpQuote;
141  if (!useQuote_)
142  tmpQuote = Handle<Quote>(boost::make_shared<SimpleQuote>(
143  ExchangeRateManager::instance().lookup(sourceCurrency_, targetCurrency_).rate()));
144  else
145  tmpQuote = fxSpot_;
146 
147  // adjust for spot
148  fxRate_ = Handle<Quote>(
149  boost::make_shared<FxRateQuote>(tmpQuote, sourceYts_, targetYts_, fixingDays_, fixingCalendar_));
150  }
151  quote = fxRate_;
152  }
153  return quote;

156 Real FxIndex::fixing(const Date& fixingDate, bool forecastTodaysFixing) const {
157 
158  Date adjustedFixingDate = fixingCalendar().adjust(fixingDate, Preceding);
159 
160  Date today = Settings::instance().evaluationDate();
161 
162  Real result = Null<Real>();
163 
164  if (adjustedFixingDate > today || (adjustedFixingDate == today && forecastTodaysFixing))
165  result = forecastFixing(adjustedFixingDate);
166 
167  if (result == Null<Real>()) {
168  if (adjustedFixingDate < today || Settings::instance().enforcesTodaysHistoricFixings()) {
169  // must have been fixed
170  // do not catch exceptions
171  result = pastFixing(adjustedFixingDate);
172  QL_REQUIRE(result != Null<Real>(), "Missing " << name() << " fixing for " << adjustedFixingDate);
173  } else {
174  try {
175  // might have been fixed
176  result = pastFixing(adjustedFixingDate);
177  } catch (Error&) {
178  ; // fall through and forecast
179  }
180  if (result == Null<Real>())
181  result = forecastFixing(adjustedFixingDate);
182  }
183  }
184 
185  return result;
186 }

188 Real FxIndex::forecastFixing(const Time& fixingTime) const {
189  QL_REQUIRE(!sourceYts_.empty() && !targetYts_.empty(),
190  "FxIndex::forecastFixing(): null term structure set to this instance of " << name());
191 
192  // we base the forecast always on the exchange rate (and not on today's fixing)
193  Real rate;
194  if (!useQuote_) {
195  rate = ExchangeRateManager::instance().lookup(sourceCurrency_, targetCurrency_).rate();
196  } else {
197  QL_REQUIRE(!fxSpot_.empty(), "FxIndex::forecastFixing(): fx quote required for " << name());
198  rate = fxSpot_->value();
199  }
200 
201  // TODO: do we need to add this a class variable?
202  DayCounter dc = Actual365Fixed();
203 
204  // to make the spot adjustment we get the time to spot, and also add this to fixing time
205  Date refDate = sourceYts_->referenceDate();
206  Date spotValueDate = valueDate(fixingCalendar().adjust(refDate));
207 
208  // time from ref to spot date
209  Real spotTime = dc.yearFraction(refDate, spotValueDate);
210  Real forwardTime = spotTime + fixingTime;
211 
212  QL_REQUIRE(forwardTime > 0.0 || close_enough(forwardTime, 0.0),
213  "FxIndex::forecastFixing(" << fixingTime << "): forwardTime (" << forwardTime << ") is negative for "
214  << name());
215 
216  // compute the forecast applying the usual no arbitrage principle
217  Real forward = rate * sourceYts_->discount(forwardTime) * targetYts_->discount(spotTime) /
218  (targetYts_->discount(forwardTime) * sourceYts_->discount(spotTime));
219  return forward;
220 }

222 Real FxIndex::forecastFixing(const Date& fixingDate) const {
223  QL_REQUIRE(!sourceYts_.empty() && !targetYts_.empty(), "null term structure set to this instance of " << name());
224 
225  // we base the forecast always on the exchange rate (and not on today's
226  // fixing)
227  Real rate;
228  if (!useQuote_) {
229  rate = ExchangeRateManager::instance().lookup(sourceCurrency_, targetCurrency_).rate();
230  } else {
231  QL_REQUIRE(!fxSpot_.empty(), "FxIndex::forecastFixing(): fx quote required for " << name());
232  rate = fxSpot_->value();
233  }
234 
235  // the exchange rate is interpreted as the spot rate w.r.t. the index's
236  // settlement date
237  Date refValueDate = valueDate(fixingCalendar().adjust(sourceYts_->referenceDate()));
238 
239  // the fixing is obeying the settlement delay as well
240  Date fixingValueDate = valueDate(fixingDate);
241 
242  // we can assume fixingValueDate >= valueDate
243  QL_REQUIRE(fixingValueDate >= refValueDate, "value date for requested fixing as of "
244  << fixingDate << " (" << fixingValueDate
245  << ") must be greater or equal to today's fixing value date ("
246  << refValueDate << ") for " << name());
247 
248  // compute the forecast applying the usual no arbitrage principle
249  Real forward = rate * sourceYts_->discount(fixingValueDate) * targetYts_->discount(refValueDate) /
250  (sourceYts_->discount(refValueDate) * targetYts_->discount(fixingValueDate));
251 
252  return forward;
253 }

288 Real FxIndex::pastFixing(const Date& fixingDate) const {
289  QL_REQUIRE(isValidFixingDate(fixingDate), fixingDate << "FxIndex::pastFixing(): is not a valid fixing date for "
290  << name() << " (calendar is " << fixingCalendar().name()
291  << ")");
292 
293  Real fixing = timeSeries()[fixingDate];
294  if (fixing != Null<Real>())
295  return fixing;
296 
297  if (fixingTriangulation_) {
298  // check reverse
299  string revName = familyName_ + " " + targetCurrency_.code() + "/" + sourceCurrency_.code();
300  if (IndexManager::instance().hasHistoricalFixing(revName, fixingDate))
301  return 1.0 / IndexManager::instance().getHistory(revName)[fixingDate];
302 
303  // Now we search for a pair of quotes that we can combine to construct the quote required.
304  // We only search for a pair of quotes a single step apart.
305  //
306  // Suppose we want a USDJPY quote and we have EUR based data, there are 4 combinations to
307  // consider:
308  // EURUSD, EURJPY => we want EURJPY / EURUSD [Triangulation]
309  // EURUSD, JPYEUR => we want 1 / (EURUSD * JPYEUR) [InverseProduct]
310  // USDEUR, EURJPY => we want USDEUR * EURJPY [Product]
311  // USDEUR, JPYEUR => we want USDEUR / JPYEUR [Triangulation (but in the reverse order)]
312  //
313  // Loop over the map, look for domestic then use the map to find the other side of the pair.
314 
315  // Loop over all the possible Indexes
316  vector<string> availIndexes = IndexManager::instance().histories();
317  for (auto i : availIndexes) {
318  if (boost::starts_with(i, familyName_)) {
319  // check for a fixing
320  Real fixing = IndexManager::instance().getHistory(i)[fixingDate];
321  if (fixing != Null<Real>()) {
322 
323  Size l = i.size();
324  string keyDomestic = i.substr(l - 7, 3);
325  string keyForeign = i.substr(l - 3);
326  string domestic = sourceCurrency_.code();
327  string foreign = targetCurrency_.code();
328 
329  if (domestic == keyDomestic) {
330  // we have domestic, now look for foreign/keyForeign
331  // USDEUR, JPYEUR => we want USDEUR / JPYEUR [Triangulation (but in the reverse order)]
332  string forName = familyName_ + " " + foreign + "/" + keyForeign;
333  if (IndexManager::instance().hasHistoricalFixing(forName, fixingDate)) {
334  Real forFixing = IndexManager::instance().getHistory(forName)[fixingDate];
335  return fixing / forFixing;
336  }
337 
338  // USDEUR, EURJPY => we want USDEUR * EURJPY [Product]
339  forName = familyName_ + " " + keyForeign + "/" + foreign;
340  if (IndexManager::instance().hasHistoricalFixing(forName, fixingDate)) {
341  Real forFixing = IndexManager::instance().getHistory(forName)[fixingDate];
342  return fixing * forFixing;
343  }
344  }
345 
346  if (domestic == keyForeign) {
347  // we have fixing, now look for foreign/keyDomestic
348  // EURUSD, JPYEUR => we want 1 / (EURUSD * JPYEUR) [InverseProduct]
349  string forName = familyName_ + " " + foreign + "/" + keyDomestic;
350  if (IndexManager::instance().hasHistoricalFixing(forName, fixingDate)) {
351  Real forFixing = IndexManager::instance().getHistory(forName)[fixingDate];
352  return 1 / (fixing * forFixing);
353  }
354 
355  // EURUSD, EURJPY => we want EURJPY / EURUSD [Triangulation]
356  forName = familyName_ + " " + keyDomestic + "/" + foreign;
357  if (IndexManager::instance().hasHistoricalFixing(forName, fixingDate)) {
358  Real forFixing = IndexManager::instance().getHistory(forName)[fixingDate];
359  return forFixing / fixing;
360  }
361  }
362  }
363  }
364  }
365  }
366  return fixing;
367 }
