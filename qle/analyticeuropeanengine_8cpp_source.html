35 if (flipResults_) {
36 // Invert strike, spot, forward
37 auto resToInvert = vector<string>({"spot", "forward", "strike"});
38 for (const string& res : resToInvert) {
39 auto it = results_.additionalResults.find(res);
40 if (it != results_.additionalResults.end())
41 it->second = 1. / boost::any_cast<Real>(it->second);
42 }
43
44 // Swap riskFreeDiscount/discountFactor and dividendDiscount
45 auto rfDiscountIt = results_.additionalResults.find("riskFreeDiscount");
46 auto divDiscountIt = results_.additionalResults.find("dividendDiscount");
47 if (rfDiscountIt != results_.additionalResults.end() && divDiscountIt != results_.additionalResults.end())
48 std::swap(rfDiscountIt->second, divDiscountIt->second);
49
50 auto discFactorIt = results_.additionalResults.find("discountFactor");
51 if (rfDiscountIt != results_.additionalResults.end() && discFactorIt != results_.additionalResults.end())
52 discFactorIt->second = boost::any_cast<Real>(rfDiscountIt->second);
53 }
54}
