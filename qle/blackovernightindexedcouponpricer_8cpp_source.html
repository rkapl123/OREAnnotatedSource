43 Real BlackOvernightIndexedCouponPricer::optionletRateGlobal(Option::Type optionType, Real effStrike) const {
44  Date lastRelevantFixingDate = coupon_->underlying()->fixingDate();
45  if (lastRelevantFixingDate <= Settings::instance().evaluationDate()) {
46  // the amount is determined
47  Real a, b;
48  if (optionType == Option::Call) {
49  a = effectiveIndexFixing_;
50  b = effStrike;
51  } else {
52  a = effStrike;
53  b = effectiveIndexFixing_;
54  }
55  return gearing_ * std::max(a - b, 0.0);
56  } else {
57  // not yet determined, use Black model
58  // for the standard deviation see Lyashenko, Mercurio, Looking forward to backward looking rates, section 6.3.
59  // the idea is to dampen the average volatility sigma between the fixing start and fixing end date by a
60  // linear function going from (fixing start, 1) to (fixing end, 0)
61  QL_REQUIRE(!capletVolatility().empty(), "BlackOvernightIndexedCouponPricer: missing optionlet volatility");
62  std::vector<Date> fixingDates = coupon_->underlying()->fixingDates();
63  QL_REQUIRE(!fixingDates.empty(), "BlackOvernightIndexedCouponPricer: empty fixing dates");
64  Real fixingStartTime = capletVolatility()->timeFromReference(fixingDates.front());
65  Real fixingEndTime = capletVolatility()->timeFromReference(fixingDates.back());
66  Real sigma = capletVolatility()->volatility(
67  std::max(fixingDates.front(), capletVolatility()->referenceDate() + 1), effStrike);
68  Real T = std::max(fixingStartTime, 0.0);
69  if (!close_enough(fixingEndTime, T))
70  T += std::pow(fixingEndTime - T, 3.0) / std::pow(fixingEndTime - T, 2.0) / 3.0;
71  Real stdDev = sigma * std::sqrt(T);
72  Real shift = capletVolatility()->displacement();
73  bool shiftedLn = capletVolatility()->volatilityType() == ShiftedLognormal;
74  Rate fixing = shiftedLn ? blackFormula(optionType, effStrike, effectiveIndexFixing_, stdDev, 1.0, shift)
75  : bachelierBlackFormula(optionType, effStrike, effectiveIndexFixing_, stdDev, 1.0);
76  return gearing_ * fixing;
77  }
78 }

90 Real BlackOvernightIndexedCouponPricer::optionletRateLocal(Option::Type optionType, Real effStrike) const {
91  // We compute a rate and a rawRate such that
92  // rate * tau * nominal is the amount of the coupon with locally (i.e. daily) capped / floored rates
93  // rawRate * tau * nominal is the amount of the coupon without capping / flooring the rate
94  // We will then return the difference between rate and rawRate (with the correct sign, see below)
95  // as the option component of the coupon.
96 
97  // See CappedFlooredOvernightIndexedCoupon::effectiveCap(), Floor() for what is passed in as effStrike.
98  // From this we back out the absolute strike at which the
99  // - daily rate + spread (spread included) or the
100  // - daily rate (spread excluded)
101  // is capped / floored.
102 
103  Real absStrike = coupon_->underlying()->includeSpread() ? effStrike + coupon_->underlying()->spread() : effStrike;
104 
105  // This following code is inevitably quite similar to the plain ON coupon pricer code, possibly we can refactor
106  // this, but as a first step it seems safer to add the full modified code explicitly here and leave the original
107  // code alone.
108 
109  ext::shared_ptr<OvernightIndex> index = ext::dynamic_pointer_cast<OvernightIndex>(coupon_->index());
110 
111  const std::vector<Date>& fixingDates = coupon_->underlying()->fixingDates();
112  const std::vector<Time>& dt = coupon_->underlying()->dt();
113 
114  Size n = dt.size();
115  Size i = 0;
116  QL_REQUIRE(coupon_->underlying()->rateCutoff() < n,
117  "rate cutoff (" << coupon_->underlying()->rateCutoff()
118  << ") must be less than number of fixings in period (" << n << ")");
119  Size nCutoff = n - coupon_->underlying()->rateCutoff();
120 
121  Real compoundFactor = 1.0, compoundFactorRaw = 1.0;
122 
123  // already fixed part
124  Date today = Settings::instance().evaluationDate();
125  while (i < n && fixingDates[std::min(i, nCutoff)] < today) {
126  // rate must have been fixed
127  Rate pastFixing = IndexManager::instance().getHistory(index->name())[fixingDates[std::min(i, nCutoff)]];
128  QL_REQUIRE(pastFixing != Null<Real>(),
129  "Missing " << index->name() << " fixing for " << fixingDates[std::min(i, nCutoff)]);
130  if (coupon_->underlying()->includeSpread()) {
131  pastFixing += coupon_->spread();
132  }
133  compoundFactor *= 1.0 + cappedFlooredRate(pastFixing, optionType, absStrike) * dt[i];
134  compoundFactorRaw *= 1.0 + pastFixing * dt[i];
135  ++i;
136  }
137 
138  // today is a border case
139  if (i < n && fixingDates[std::min(i, nCutoff)] == today) {
140  // might have been fixed
141  try {
142  Rate pastFixing = IndexManager::instance().getHistory(index->name())[fixingDates[std::min(i, nCutoff)]];
143  if (pastFixing != Null<Real>()) {
144  if (coupon_->underlying()->includeSpread()) {
145  pastFixing += coupon_->spread();
146  }
147  compoundFactor *= 1.0 + cappedFlooredRate(pastFixing, optionType, absStrike) * dt[i];
148  compoundFactorRaw *= 1.0 + pastFixing * dt[i];
149  ++i;
150  } else {
151  ; // fall through and forecast
152  }
153  } catch (Error&) {
154  ; // fall through and forecast
155  }
156  }
157 
158  // forward part, approximation by pricing a cap / floor in the middle of the future period
159  const std::vector<Date>& dates = coupon_->underlying()->valueDates();
160  if (i < n) {
161  Handle<YieldTermStructure> curve = index->forwardingTermStructure();
162  QL_REQUIRE(!curve.empty(), "null term structure set to this instance of " << index->name());
163 
164  DiscountFactor startDiscount = curve->discount(dates[i]);
165  DiscountFactor endDiscount = curve->discount(dates[std::max(nCutoff, i)]);
166 
167  // handle the rate cutoff period (if there is any, i.e. if nCutoff < n)
168  if (nCutoff < n) {
169  // forward discount factor for one calendar day on the cutoff date
170  DiscountFactor discountCutoffDate = curve->discount(dates[nCutoff] + 1) / curve->discount(dates[nCutoff]);
171  // keep the above forward discount factor constant during the cutoff period
172  endDiscount *= std::pow(discountCutoffDate, dates[n] - dates[nCutoff]);
173  }
174 
175  // estimate the average daily rate over the future period (approximate the continuously compounded rate)
176  Real tau = coupon_->dayCounter().yearFraction(dates[i], dates.back());
177  Real averageRate = -std::log(endDiscount / startDiscount) / tau;
178 
179  // compute the value of a cap or floor with fixing in the middle of the future period
180  // (but accounting for the rate cutoff here)
181  Time midPoint =
182  (capletVolatility()->timeFromReference(dates[i]) + capletVolatility()->timeFromReference(dates[nCutoff])) /
183  2.0;
184  Real stdDev = capletVolatility()->volatility(midPoint, effStrike) * std::sqrt(midPoint);
185  Real shift = capletVolatility()->displacement();
186  bool shiftedLn = capletVolatility()->volatilityType() == ShiftedLognormal;
187  Rate cfValue = shiftedLn ? blackFormula(optionType, effStrike, averageRate, stdDev, 1.0, shift)
188  : bachelierBlackFormula(optionType, effStrike, averageRate, stdDev, 1.0);
189 
190  // add spread to average rate
191  if (coupon_->underlying()->includeSpread()) {
192  averageRate += coupon_->underlying()->spread();
193  }
194 
195  // incorporate cap/floor into average rate
196  Real averageRateRaw = averageRate;
197  averageRate += optionType == Option::Call ? (-cfValue) : cfValue;
198 
199  // now assume the averageRate is the effective rate over the future period and update the compoundFactor
200  // this is an approximation, see "Ester / Daily Spread Curve Setup in ORE": set tau to avg value
201  Real dailyTau =
202  coupon_->underlying()->dayCounter().yearFraction(dates[i], dates.back()) / (dates.back() - dates[i]);
203  // now use formula (4) from the paper
204  compoundFactor *= std::pow(1.0 + dailyTau * averageRate, static_cast<int>(dates.back() - dates[i]));
205  compoundFactorRaw *= std::pow(1.0 + dailyTau * averageRateRaw, static_cast<int>(dates.back() - dates[i]));
206  }
207 
208  Rate tau = coupon_->underlying()->lookback() == 0 * Days
209  ? coupon_->accrualPeriod()
210  : coupon_->dayCounter().yearFraction(dates.front(), dates.back());
211  Rate rate = (compoundFactor - 1.0) / tau;
212  Rate rawRate = (compoundFactorRaw - 1.0) / tau;
213 
214  rate *= coupon_->underlying()->gearing();
215  rawRate *= coupon_->underlying()->gearing();
216 
217  if (!coupon_->underlying()->includeSpread()) {
218  rate += coupon_->underlying()->spread();
219  rawRate += coupon_->underlying()->spread();
220  }
221 
222  // return optionletRate := r - rawRate, i.e. the option component only
223  // (see CappedFlooredOvernightIndexedCoupon::rate() for the signs of the capletRate / flooredRate)
224 
225  return (optionType == Option::Call ? -1.0 : 1.0) * (rate - rawRate);
226 }
