33 CrossCcyBasisMtMResetSwapHelper::CrossCcyBasisMtMResetSwapHelper(
34  const Handle<Quote>& spreadQuote, const Handle<Quote>& spotFX, Natural settlementDays,
35  const Calendar& settlementCalendar, const Period& swapTenor, BusinessDayConvention rollConvention,
36  const boost::shared_ptr<QuantLib::IborIndex>& foreignCcyIndex,
37  const boost::shared_ptr<QuantLib::IborIndex>& domesticCcyIndex,
38  const Handle<YieldTermStructure>& foreignCcyDiscountCurve,
39  const Handle<YieldTermStructure>& domesticCcyDiscountCurve,
40  const Handle<YieldTermStructure>& foreignCcyFxFwdRateCurve,
41  const Handle<YieldTermStructure>& domesticCcyFxFwdRateCurve, bool eom, bool spreadOnForeignCcy,
42  boost::optional<Period> foreignTenor, boost::optional<Period> domesticTenor)
43  : RelativeDateRateHelper(spreadQuote), spotFX_(spotFX), settlementDays_(settlementDays),
44  settlementCalendar_(settlementCalendar), swapTenor_(swapTenor), rollConvention_(rollConvention),
45  foreignCcyIndex_(foreignCcyIndex), domesticCcyIndex_(domesticCcyIndex),
46  foreignCcyDiscountCurve_(foreignCcyDiscountCurve), domesticCcyDiscountCurve_(domesticCcyDiscountCurve),
47  foreignCcyFxFwdRateCurve_(foreignCcyFxFwdRateCurve), domesticCcyFxFwdRateCurve_(domesticCcyFxFwdRateCurve),
48  eom_(eom), spreadOnForeignCcy_(spreadOnForeignCcy),
49  foreignTenor_(foreignTenor ? *foreignTenor : foreignCcyIndex_->tenor()),
50  domesticTenor_(domesticTenor ? *domesticTenor : domesticCcyIndex_->tenor()) {
51 
52  foreignCurrency_ = foreignCcyIndex_->currency();
53  domesticCurrency_ = domesticCcyIndex_->currency();
54  QL_REQUIRE(foreignCurrency_ != domesticCurrency_,
55  "matching currencies not allowed on CrossCcyBasisMtMResetSwapHelper");
56 
57  bool foreignIndexHasCurve = !foreignCcyIndex_->forwardingTermStructure().empty();
58  bool domesticIndexHasCurve = !domesticCcyIndex_->forwardingTermStructure().empty();
59  bool haveForeignDiscountCurve = !foreignCcyDiscountCurve_.empty();
60  bool haveDomesticDiscountCurve = !domesticCcyDiscountCurve_.empty();
61 
62  QL_REQUIRE(
63  !(foreignIndexHasCurve && domesticIndexHasCurve && haveForeignDiscountCurve && haveDomesticDiscountCurve),
64  "CrossCcyBasisMtMResetSwapHelper - Have all curves, nothing to solve for.");
65 
66  /* Link the curve being bootstrapped to the index if the index has
67  no projection curve */
68  if (foreignIndexHasCurve && haveForeignDiscountCurve) {
69  if (!domesticIndexHasCurve) {
70  domesticCcyIndex_ = domesticCcyIndex_->clone(termStructureHandle_);
71  domesticCcyIndex_->unregisterWith(termStructureHandle_);
72  }
73  // if we have both index and discounting curve on foreign leg,
74  // check foreignCcyFxFwdRateCurve and link it to foreign discount curve if empty
75  // (we are bootstrapping on domestic leg in this instance, so foreign leg needs to be fully determined
76  if (foreignCcyFxFwdRateCurve_.empty())
77  foreignCcyFxFwdRateCurve_ = foreignCcyDiscountCurve_;
78  } else if (domesticIndexHasCurve && haveDomesticDiscountCurve) {
79  if (!foreignIndexHasCurve) {
80  foreignCcyIndex_ = foreignCcyIndex_->clone(termStructureHandle_);
81  foreignCcyIndex_->unregisterWith(termStructureHandle_);
82  }
83  // if we have both index and discounting curve on domestic leg,
84  // check domesticCcyFxFwdRateCurve and link it to domestic discount curve if empty
85  // (we are bootstrapping on foreign leg in this instance, so domestic leg needs to be fully determined
86  if (domesticCcyFxFwdRateCurve_.empty())
87  domesticCcyFxFwdRateCurve_ = domesticCcyDiscountCurve_;
88  } else {
89  QL_FAIL("Need one leg of the cross currency basis swap to "
90  "have all of its curves.");
91  }
92 
93  registerWith(spotFX_);
94  registerWith(domesticCcyIndex_);
95  registerWith(foreignCcyIndex_);
96  registerWith(foreignCcyDiscountCurve_);
97  registerWith(domesticCcyDiscountCurve_);
98  registerWith(foreignCcyFxFwdRateCurve_);
99  registerWith(domesticCcyFxFwdRateCurve_);
100 
101  initializeDates();
102 }

104 void CrossCcyBasisMtMResetSwapHelper::initializeDates() {
105 
106  Date refDate = evaluationDate_;
107  // if the evaluation date is not a business day
108  // then move to the next business day
109  refDate = settlementCalendar_.adjust(refDate);
110 
111  Date settlementDate = settlementCalendar_.advance(refDate, settlementDays_, Days);
112  Date maturityDate = settlementDate + swapTenor_;
113 
114  Schedule foreignLegSchedule = MakeSchedule()
115  .from(settlementDate)
116  .to(maturityDate)
117  .withTenor(foreignTenor_)
118  .withCalendar(settlementCalendar_)
119  .withConvention(rollConvention_)
120  .endOfMonth(eom_);
121 
122  Schedule domesticLegSchedule = MakeSchedule()
123  .from(settlementDate)
124  .to(maturityDate)
125  .withTenor(domesticTenor_)
126  .withCalendar(settlementCalendar_)
127  .withConvention(rollConvention_)
128  .endOfMonth(eom_);
129 
130  Real foreignNominal = 1.0;
131  // build an FX index for forward rate projection (TODO - review settlement and calendar)
132 
133  boost::shared_ptr<FxIndex> fxIdx =
134  boost::make_shared<FxIndex>("dummy", settlementDays_, foreignCurrency_, domesticCurrency_, settlementCalendar_,
135  spotFX_, foreignCcyFxFwdRateCurveRLH_, domesticCcyFxFwdRateCurveRLH_);
136 
137  swap_ = boost::shared_ptr<CrossCcyBasisMtMResetSwap>(
138  new CrossCcyBasisMtMResetSwap(foreignNominal, foreignCurrency_, foreignLegSchedule, foreignCcyIndex_, 0.0,
139  domesticCurrency_, domesticLegSchedule, domesticCcyIndex_, 0.0, fxIdx));
140 
141  boost::shared_ptr<PricingEngine> engine;
142  engine.reset(new CrossCcySwapEngine(domesticCurrency_, domesticDiscountRLH_, foreignCurrency_, foreignDiscountRLH_,
143  spotFX_));
144  swap_->setPricingEngine(engine);
145 
146  earliestDate_ = swap_->startDate();
147  latestDate_ = swap_->maturityDate();
148 
149 /* May need to adjust latestDate_ if you are projecting libor based
150  on tenor length rather than from accrual date to accrual date. */
151 #ifdef QL_USE_INDEXED_COUPON
152  if (termStructureHandle_ == foreignCcyIndex_->forwardingTermStructure()) {
153  Size numCashflows = swap_->leg(0).size();
154  Date endDate = latestDate_;
155  if (numCashflows > 0) {
156  for (Size i = numCashflows - 1; i >= 0; i--) {
157  boost::shared_ptr<FloatingRateCoupon> lastFloating =
158  boost::dynamic_pointer_cast<FloatingRateCoupon>(swap_->leg(0)[i]);
159  if (!lastFloating)
160  continue;
161  else {
162  Date fixingValueDate = foreignCcyIndex_->valueDate(lastFloating->fixingDate());
163  endDate = domesticCcyIndex_->maturityDate(fixingValueDate);
164  Date endValueDate = foreignCcyIndex_->maturityDate(fixingValueDate);
165  latestDate_ = std::max(latestDate_, endValueDate);
166  break;
167  }
168  }
169  }
170  }
171  if (termStructureHandle_ == domesticCcyIndex_->forwardingTermStructure()) {
172  Size numCashflows = swap_->leg(1).size();
173  Date endDate = latestDate_;
174  if (numCashflows > 0) {
175  for (Size i = numCashflows - 1; i >= 0; i--) {
176  boost::shared_ptr<FloatingRateCoupon> lastFloating =
177  boost::dynamic_pointer_cast<FloatingRateCoupon>(swap_->leg(1)[i]);
178  if (!lastFloating)
179  continue;
180  else {
181  Date fixingValueDate = domesticCcyIndex_->valueDate(lastFloating->fixingDate());
182  endDate = domesticCcyIndex_->maturityDate(fixingValueDate);
183  Date endValueDate = domesticCcyIndex_->maturityDate(fixingValueDate);
184  latestDate_ = std::max(latestDate_, endValueDate);
185  break;
186  }
187  }
188  }
189  }
190 #endif
191 }

193 void CrossCcyBasisMtMResetSwapHelper::setTermStructure(YieldTermStructure* t) {
194 
195  bool observer = false;
196  boost::shared_ptr<YieldTermStructure> temp(t, no_deletion);
197 
198  termStructureHandle_.linkTo(temp, observer);
199 
200  if (foreignCcyDiscountCurve_.empty())
201  foreignDiscountRLH_.linkTo(temp, observer);
202  else
203  foreignDiscountRLH_.linkTo(*foreignCcyDiscountCurve_, observer);
204 
205  if (domesticCcyDiscountCurve_.empty())
206  domesticDiscountRLH_.linkTo(temp, observer);
207  else
208  domesticDiscountRLH_.linkTo(*domesticCcyDiscountCurve_, observer);
209 
210  // the below are the curves used for FX forward rate projection (for the resetting cashflows)
211  if (foreignCcyFxFwdRateCurve_.empty())
212  foreignCcyFxFwdRateCurveRLH_.linkTo(temp, observer);
213  else
214  foreignCcyFxFwdRateCurveRLH_.linkTo(*foreignCcyFxFwdRateCurve_, observer);
215 
216  if (domesticCcyFxFwdRateCurve_.empty())
217  domesticCcyFxFwdRateCurveRLH_.linkTo(temp, observer);
218  else
219  domesticCcyFxFwdRateCurveRLH_.linkTo(*domesticCcyFxFwdRateCurve_, observer);
220 
221  RelativeDateRateHelper::setTermStructure(t);
222 }
