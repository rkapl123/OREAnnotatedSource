46 Handle<DefaultProbabilityTermStructure> CrCirpp::defaultCurve(std::vector<Date> dateGrid) const {
47  if (parametrization_->shifted()) {
48  QL_REQUIRE(!parametrization_->termStructure().empty(), "default curve not set");
49  QL_REQUIRE(dateGrid.size() == 0, "dateGrid without effect for shifted model");
50  return parametrization_->termStructure();
51  } else {
52  // build one on the fly
53  Date today = Settings::instance().evaluationDate();
54  std::vector<Real> survivalProbabilities(1, 1.0);
55  std::vector<Date> dates;
56  DayCounter tsDayCounter = Actual365Fixed();
57  if (dateGrid.size() > 0) {
58  QL_REQUIRE(dateGrid.front() == today, "front date must be today");
59  dates = dateGrid;
60  } else {
61  dates.push_back(today);
62  // up to one year in monthly steps
63  for (Size i = 1; i <= 12; i++)
64  dates.push_back(today + i * Months);
65  // starts in year two annually
66  for (Size i = 2; i <= 10; i++)
67  dates.push_back(today + i * Years);
68  }
69  for (Size i = 1; i < dates.size(); i++) {
70  Real t = tsDayCounter.yearFraction(today, dates[i]);
71  survivalProbabilities.push_back(survivalProbability(0.0, t, parametrization_->y0(t)));
72  }
73  boost::shared_ptr<DefaultProbabilityTermStructure> defaultCurve(
74  new InterpolatedSurvivalProbabilityCurve<LogLinear>(dates, survivalProbabilities, tsDayCounter));
75  defaultCurve->enableExtrapolation();
76  return Handle<DefaultProbabilityTermStructure>(defaultCurve);
77  }

80 Real CrCirpp::A(Real t, Real T) const {
81  Real kappa = parametrization_->kappa(t);
82  Real theta = parametrization_->theta(t);
83  Real sigma = parametrization_->sigma(t);
84  Real sigma2 = sigma * sigma;
85  Real h = sqrt(kappa * kappa + 2.0 * sigma2);
86 
87  Real nominator = 2.0 * h * exp((kappa + h) * (T - t) / 2.0);
88  Real denominator = 2.0 * h + (kappa + h) * (exp((T - t) * h) - 1.0);
89  Real exponent = 2.0 * kappa * theta / sigma2;
90 
91  Real value = std::pow((nominator / denominator), exponent);
92 
93  return value;
94 }

113 Real CrCirpp::survivalProbability(Real t, Real T, Real y) const {
114  Real P_Cir = zeroBond(t, T, y); // A(t,T) * exp(-B(t,T) * y);
115  if (parametrization_->shifted()) {
116  Probability SP_t = parametrization_->termStructure()->survivalProbability(t);
117  // std::cout<<" SP_t " << SP_t << std::endl;
118  Probability SP_T = parametrization_->termStructure()->survivalProbability(T);
119  // std::cout<<" SP_T " << SP_T << std::endl;
120  // std::cout<<" y0 " << parametrization_->y0(t) << std::endl;
121  Real A_bar = (SP_T * A(0, t) * exp(-B(0, t) * parametrization_->y0(t))) /
122  (SP_t * A(0, T) * exp(-B(0, T) * parametrization_->y0(t)));
123  // std::cout<<"A_bar " <<A_bar<< std::endl;
124  return A_bar * P_Cir;
125  } else
126  return P_Cir;
127 }

129 Real CrCirpp::density(Real x, Real t) {
130 
131  Real kappa = parametrization_->kappa(t);
132  Real theta = parametrization_->theta(t);
133  Real sigma = parametrization_->sigma(t);
134  Real y0 = parametrization_->y0(t);
135  Real sigma2 = sigma * sigma;
136 
137  Real c = 4 * kappa / (sigma2 * (1 - exp(-kappa * t)));
138  // degrees of freedom
139  Real df = 4 * kappa * theta / (sigma2);
140  // non-central parameter
141  Real ncp = c * y0 * exp(-kappa * t);
142 
143  return c * nccs(df, ncp, c * x, false);
144 }

163 Real CrCirpp::densityForwardMeasure(Real x, Real t) {
164 
165  Real kappa = parametrization_->kappa(t);
166  Real theta = parametrization_->theta(t);
167  Real sigma = parametrization_->sigma(t);
168  Real y0 = parametrization_->y0(t);
169  Real sigma2 = sigma * sigma;
170 
171  Real h = sqrt(kappa * kappa + 2.0 * sigma2);
172  Real rho = 2 * h / (sigma2 * (exp(h * t) - 1));
173  // q(t,s) in Brigo-Mercuri in (3.28)
174  Real c = 2 * (rho + (kappa + h) / sigma2 + 0.0); // 0.0 stands for the B term which is neglectible
175  // degrees of freedom
176  Real df = 4 * kappa * theta / (sigma2);
177  // non-central parameter
178  // delta(t,s) in Brigo-Mercurio in (3.28)
179  Real ncp = 4 * (rho * rho * y0 * exp(h * t)) / c;
180 
181  return c * nccs(df, ncp, c * x, false);
182 }

207 Real CrCirpp::zeroBondOption(Real eval_t, Real expiry_T, Real maturity_tau, Real strike_k, Real y_t,
208  Real w) {
209 
210  Real value = 0.0;
211  Real kappa = parametrization_->kappa(eval_t);
212  Real theta = parametrization_->theta(eval_t);
213  Real sigma = parametrization_->sigma(eval_t);
214  Real y0 = parametrization_->y0(eval_t);
215  Real sigma2 = sigma * sigma;
216 
217  Real h = sqrt(kappa * kappa + 2.0 * sigma2);
218  Real rho = 2.0 * h / (sigma2 * (exp(h * (expiry_T - eval_t)) - 1.0));
219  Real Psi = (kappa + h) / sigma2;
220 
221  Probability SP_T;
222  Probability SP_tau;
223  if (parametrization_->shifted()) {
224  SP_T = parametrization()->termStructure()->survivalProbability(expiry_T);
225  SP_tau = parametrization()->termStructure()->survivalProbability(maturity_tau);
226  } else {
227  SP_T = survivalProbability(0.0, expiry_T, y0);
228  SP_tau = survivalProbability(0.0, maturity_tau, y0);
229  }
230 
231  Real r_hat = 1.0 / B(expiry_T, maturity_tau) *
232  (log(A(expiry_T, maturity_tau) / strike_k) -
233  log((SP_T * A(0.0, maturity_tau) * exp(-B(0.0, maturity_tau) * y0)) /
234  (SP_tau * A(0.0, expiry_T) * exp(-B(0.0, expiry_T) * y0))));
235 
236  Real denom = rho + Psi + B(expiry_T, maturity_tau);
237  Real df = 4.0 * kappa * theta / sigma2;
238  Real ncp = 2.0 * rho * rho * y_t * exp(h * (expiry_T - eval_t)) / denom;
239  QL_REQUIRE(std::isfinite(df) && df > 0, "CrCirpp::zeroBondOption(): illegal df="
240  << df << ", kappa=" << kappa << ", theta= " << theta
241  << ", sigma=" << sigma);
242 
243  value += nccs(df, ncp, 2 * r_hat * denom, true) * SP_tau;
244 
245  denom = rho + Psi;
246  // df unchanged
247  ncp = 2 * rho * rho * y_t * exp(h * (expiry_T - eval_t)) / denom;
248 
249  value -= nccs(df, ncp, 2 * r_hat * denom, true) * SP_T * strike_k;
250 
251  if (w < 0.0) {
252  // PUT via put-call-paritiy
253  // C - P = S_M(T) - S_M(t)*K
254  // P = C - S_M(T) + S_M(t)*K
255  value -= SP_tau - SP_T * strike_k;
256  }
257 
258  return value;
259 }
