55CrossAssetStateProcess::CrossAssetStateProcess(const CrossAssetModel* const model)
56 : StochasticProcess(), model_(model), cirppCount_(0) {
57
58 if (model_->discretization() == CrossAssetModel::Discretization::Euler) {
59 discretization_ = boost::make_shared<EulerDiscretization>();
60 } else {
61 discretization_ =
62 boost::make_shared<CrossAssetStateProcess::ExactDiscretization>(model_, model_->salvagingAlgorithm());
63 }
64
65 updateSqrtCorrelation();
66
67 // set up CR CIR++ processes, defer the euler discretisation check to evolve()
68 for (Size i = 0; i < model_->components(CrossAssetModel::AssetType::CR); ++i) {
69 if (model_->modelType(CrossAssetModel::AssetType::CR, i) == CrossAssetModel::ModelType::CIRPP) {
70 crCirpp_.push_back(model->crcirppModel(i)->stateProcess());
71 cirppCount_++;
72 } else {
73 crCirpp_.push_back(boost::shared_ptr<CrCirppStateProcess>());
74 }
75 }

96Array CrossAssetStateProcess::initialValues() const {
97 Array res(model_->dimension(), 0.0);
98 /* irlgm1f / irhw processes have initial value 0 */
99 for (Size i = 0; i < model_->components(CrossAssetModel::AssetType::FX); ++i) {
100 /* fxbs processes are in log spot */
101 res[model_->pIdx(CrossAssetModel::AssetType::FX, i, 0)] = std::log(model_->fxbs(i)->fxSpotToday()->value());
102 }
103 for (Size i = 0; i < model_->components(CrossAssetModel::AssetType::EQ); ++i) {
104 /* eqbs processes are in log spot */
105 res[model_->pIdx(CrossAssetModel::AssetType::EQ, i, 0)] = std::log(model_->eqbs(i)->eqSpotToday()->value());
106 }
107 // CR CIR++ components
108 for (Size i = 0; i < model_->components(CrossAssetModel::AssetType::CR); ++i) {
109 if (model_->modelType(CrossAssetModel::AssetType::CR, i) != CrossAssetModel::ModelType::CIRPP)
110 continue;
111 QL_REQUIRE(crCirpp_[i], "crcirpp is null!");
112 Array r = crCirpp_[i]->initialValues();
113 res[model_->pIdx(CrossAssetModel::AssetType::CR, i, 0)] = r[0]; // y0
114 res[model_->pIdx(CrossAssetModel::AssetType::CR, i, 1)] = r[1]; // S(0,0) = 1
115 }
116
117 for (Size i = 0; i < model_->components(CrossAssetModel::AssetType::INF); ++i) {
118 // Second component of JY model is the inflation index process.
119 if (model_->modelType(CrossAssetModel::AssetType::INF, i) == CrossAssetModel::ModelType::JY) {
120 res[model_->pIdx(CrossAssetModel::AssetType::INF, i, 1)] =
121 std::log(model_->infjy(i)->index()->fxSpotToday()->value());
122 }
123 }
124 /* infdk, crlgm1f, commodity, crstate processes have initial value 0 */
125 return res;
126}

128Array CrossAssetStateProcess::drift(Time t, const Array& x) const {
129 Array res(model_->dimension(), 0.0);
130 Size n = model_->components(CrossAssetModel::AssetType::IR);
131 Size n_eq = model_->components(CrossAssetModel::AssetType::EQ);
132 Real H0 = model_->irlgm1f(0)->H(t);
133 Real Hprime0 = model_->irlgm1f(0)->Hprime(t);
134 Real alpha0 = model_->irlgm1f(0)->alpha(t);
135 Real zeta0 = model_->irlgm1f(0)->zeta(t);
136 auto i = cache_m_.find(t);
137 if (i == cache_m_.end()) {
138 /* z0 has drift 0 in the LGM measure but non-zero drift in the bank account measure, so start loop at i = 0 */
139 for (Size i = 0; i < n; ++i) {
140 Real Hi = model_->irlgm1f(i)->H(t);
141 Real alphai = model_->irlgm1f(i)->alpha(t);
142 if (i == 0 && model_->measure() == IrModel::Measure::BA) {
143 // ADD z0 drift in the BA measure
144 res[model_->pIdx(CrossAssetModel::AssetType::IR, i, 0)] = -Hi * alphai * alphai;
145 // the auxiliary state variable is drift-free
146 res[model_->pIdx(CrossAssetModel::AssetType::IR, i, 1)] = 0.0;
147 }
148 if (i > 0) {
149 Real sigmai = model_->fxbs(i - 1)->sigma(t);
150 // ir-ir
151 Real rhozz0i =
152 model_->correlation(CrossAssetModel::AssetType::IR, 0, CrossAssetModel::AssetType::IR, i);
153 // ir-fx
154 Real rhozx0i =
155 model_->correlation(CrossAssetModel::AssetType::IR, 0, CrossAssetModel::AssetType::FX, i - 1);
156 Real rhozxii =
157 model_->correlation(CrossAssetModel::AssetType::IR, i, CrossAssetModel::AssetType::FX, i - 1);
158 // ir drifts
159 res[model_->pIdx(CrossAssetModel::AssetType::IR, i, 0)] =
160 -Hi * alphai * alphai + H0 * alpha0 * alphai * rhozz0i - sigmai * alphai * rhozxii;
161 // log spot fx drifts (z0, zi independent parts)
162 res[model_->pIdx(CrossAssetModel::AssetType::FX, i - 1, 0)] =
163 H0 * alpha0 * sigmai * rhozx0i +
164 model_->irlgm1f(0)->termStructure()->forwardRate(t, t, Continuous) -
165 model_->irlgm1f(i)->termStructure()->forwardRate(t, t, Continuous) - 0.5 * sigmai * sigmai;
166 if (model_->measure() == IrModel::Measure::BA) {
167 // REMOVE the LGM measure drift contributions above
168 res[model_->pIdx(CrossAssetModel::AssetType::IR, i, 0)] -= H0 * alpha0 * alphai * rhozz0i;
169 res[model_->pIdx(CrossAssetModel::AssetType::FX, i - 1, 0)] -= H0 * alpha0 * sigmai * rhozx0i;
170 }
171 }
172 }
173 /* log equity spot drifts (the cache-able parts) */
174 for (Size k = 0; k < n_eq; ++k) {
175 Size i = model_->ccyIndex(model_->eqbs(k)->currency());
176 // ir params (for equity currency)
177 Real eps_ccy = (i == 0) ? 0.0 : 1.0;
178 // Real Hi = model_->irlgm1f(i)->H(t);
179 // Real alphai = model_->irlgm1f(i)->alpha(t);
180 // eq vol
181 Real sigmask = model_->eqbs(k)->sigma(t);
182 // fx vol (eq ccy / base ccy)
183 Real sigmaxi = (i == 0) ? 0.0 : model_->fxbs(i - 1)->sigma(t);
184 // ir-eq corr
185 // Real rhozsik = model_->correlation(EQ, k, CrossAssetModel::AssetType::IR, i); // eq cur
186 Real rhozs0k =
187 model_->correlation(CrossAssetModel::AssetType::EQ, k, CrossAssetModel::AssetType::IR, 0); // base cur
188 // fx-eq corr
189 Real rhoxsik =
190 (i == 0) ? 0.0 : // no fx process for base-ccy
191 model_->correlation(CrossAssetModel::AssetType::FX, i - 1, CrossAssetModel::AssetType::EQ, k);
192 // ir instantaneous forward rate (from curve used for eq forward projection)
193 Real fr_i = model_->eqbs(k)->equityIrCurveToday()->forwardRate(t, t, Continuous);
194 // div yield instantaneous forward rate
195 Real fq_k = model_->eqbs(k)->equityDivYieldCurveToday()->forwardRate(t, t, Continuous);
196 res[model_->pIdx(CrossAssetModel::AssetType::EQ, k, 0)] = fr_i - fq_k + (rhozs0k * H0 * alpha0 * sigmask) -
197 (eps_ccy * rhoxsik * sigmaxi * sigmask) -
198 (0.5 * sigmask * sigmask);
199 }
200
201 // State independent pieces of JY inflation model, if there is a CAM JY component.
202 for (Size j = 0; j < model_->components(CrossAssetModel::AssetType::INF); ++j) {
203
204 if (model_->modelType(CrossAssetModel::AssetType::INF, j) == CrossAssetModel::ModelType::JY) {
205
206 auto p = model_->infjy(j);
207 Size i_j = model_->ccyIndex(p->currency());
208
209 // JY inflation parameter values.
210 Real H_y_j = p->realRate()->H(t);
211 Real Hp_y_j = p->realRate()->Hprime(t);
212 Real zeta_y_j = p->realRate()->zeta(t);
213 Real alpha_y_j = p->realRate()->alpha(t);
214 Real sigma_c_j = p->index()->sigma(t);
215
216 // Inflation nominal currency parameter values
217 Real H_i_j = model_->irlgm1f(i_j)->H(t);
218 Real Hp_i_j = model_->irlgm1f(i_j)->Hprime(t);
219 Real zeta_i_j = model_->irlgm1f(i_j)->zeta(t);
220
221 // Correlations
222 Real rho_zy_0j =
223 model_->correlation(CrossAssetModel::AssetType::IR, 0, CrossAssetModel::AssetType::INF, j, 0, 0);
224 Real rho_yc_ij =
225 model_->correlation(CrossAssetModel::AssetType::INF, j, CrossAssetModel::AssetType::INF, j, 0, 1);
226 Real rho_zc_0j =
227 model_->correlation(CrossAssetModel::AssetType::IR, 0, CrossAssetModel::AssetType::INF, j, 0, 1);
228
229 // JY real rate drift. It is state independent
230 auto rrDrift = -alpha_y_j * alpha_y_j * H_y_j + rho_zy_0j * alpha0 * alpha_y_j * H_y_j -
231 rho_yc_ij * alpha_y_j * sigma_c_j;
232
233 if (i_j > 0) {
234 Real sigma_x_i_j = model_->fxbs(i_j - 1)->sigma(t);
235 Real rho_yx_j_i_j = model_->correlation(CrossAssetModel::AssetType::INF, j,
236 CrossAssetModel::AssetType::FX, i_j - 1, 0, 0);
237 rrDrift -= rho_yx_j_i_j * alpha_y_j * sigma_x_i_j;
238 }
239
240 res[model_->pIdx(CrossAssetModel::AssetType::INF, j, 0)] = rrDrift;
241
242 // JY log inflation index drift (state independent piece).
243 auto indexDrift = rho_zc_0j * alpha0 * sigma_c_j * H0 - 0.5 * sigma_c_j * sigma_c_j +
244 zeta_i_j * Hp_i_j * H_i_j - zeta_y_j * Hp_y_j * H_y_j;
245
246 // Add on the f_n(0, t) - f_r(0, t) piece using the initial zero inflation term structure.
247 // Use the same dt below that is used in yield forward rate calculations.
248 auto ts = p->realRate()->termStructure();
249 Time dt = 0.0001;
250 Time t1 = std::max(t - dt / 2.0, 0.0);
251 Time t2 = t1 + dt;
252 auto z_t = ts->zeroRate(t);
253 auto z_t1 = ts->zeroRate(t1);
254 auto z_t2 = ts->zeroRate(t2);
255 indexDrift += std::log(1 + z_t) + (t / (1 + z_t)) * ((z_t2 - z_t1) / dt);
256
257 if (i_j > 0) {
258 Real sigma_x_i_j = model_->fxbs(i_j - 1)->sigma(t);
259 Real rho_cx_j_i_j = model_->correlation(CrossAssetModel::AssetType::INF, j,
260 CrossAssetModel::AssetType::FX, i_j - 1, 1, 0);
261 indexDrift -= rho_cx_j_i_j * sigma_c_j * sigma_x_i_j;
262 }
263
264 res[model_->pIdx(CrossAssetModel::AssetType::INF, j, 1)] = indexDrift;
265 }
266 }
267
268 cache_m_.insert(std::make_pair(t, res));
269 } else {
270 res = i->second;
271 }
272 // non-cacheable sections of drifts
273 for (Size i = 1; i < n; ++i) {
274 // log spot fx drifts (z0, zi dependent parts)
275 Real Hi = model_->irlgm1f(i)->H(t);
276 Real Hprimei = model_->irlgm1f(i)->Hprime(t);
277 Real zetai = model_->irlgm1f(i)->zeta(t);
278 res[model_->pIdx(CrossAssetModel::AssetType::FX, i - 1, 0)] +=
279 x[model_->pIdx(CrossAssetModel::AssetType::IR, 0, 0)] * Hprime0 + zeta0 * Hprime0 * H0 -
280 x[model_->pIdx(CrossAssetModel::AssetType::IR, i, 0)] * Hprimei - zetai * Hprimei * Hi;
281 }
282 for (Size k = 0; k < n_eq; ++k) {
283 // log equity spot drifts (path-dependent parts)
284 // notice the assumption in below that dividend yield curve is static
285 Size i = model_->ccyIndex(model_->eqbs(k)->currency());
286 // ir params (for equity currency)
287 Real Hi = model_->irlgm1f(i)->H(t);
288 Real Hprimei = model_->irlgm1f(i)->Hprime(t);
289 Real zetai = model_->irlgm1f(i)->zeta(t);
290 res[model_->pIdx(CrossAssetModel::AssetType::EQ, k, 0)] +=
291 (x[model_->pIdx(CrossAssetModel::AssetType::IR, i, 0)] * Hprimei) + (zetai * Hprimei * Hi);
292 }
293
294 // Non-cacheable portion of inflation JY drift, if there is a CAM JY component.
295 for (Size j = 0; j < model_->components(CrossAssetModel::AssetType::INF); ++j) {
296 if (model_->modelType(CrossAssetModel::AssetType::INF, j) == CrossAssetModel::ModelType::JY) {
297
298 auto p = model_->infjy(j);
299 Size i_j = model_->ccyIndex(p->currency());
300
301 // JY inflation parameter values.
302 Real Hp_y_j = p->realRate()->Hprime(t);
303
304 // Inflation nominal currency parameter values
305 Real Hp_i_j = model_->irlgm1f(i_j)->Hprime(t);
306
307 res[model_->pIdx(CrossAssetModel::AssetType::INF, j, 1)] +=
308 x[model_->pIdx(CrossAssetModel::AssetType::IR, i_j, 0)] * Hp_i_j -
309 x[model_->pIdx(CrossAssetModel::AssetType::INF, j, 0)] * Hp_y_j;
310 }
311 }
312
313 // COM drift
314 Size n_com = model_->components(CrossAssetModel::AssetType::COM);
315 for (Size k = 0; k < n_com; ++k) {
316 auto cm = boost::dynamic_pointer_cast<CommoditySchwartzModel>(model_->comModel(k));
317 QL_REQUIRE(cm, "CommoditySchwartzModel not set");
318 if (!cm->parametrization()->driftFreeState()) {
319 // Ornstein-Uhlenbeck drift
320 Real kap = cm->parametrization()->kappaParameter();
321 res[model_->pIdx(CrossAssetModel::AssetType::COM, k, 0)] -=
322 kap * x[model_->pIdx(CrossAssetModel::AssetType::COM, k, 0)];
323 } else {
324 // zero drift
325 }
326 }
327
328 /* no drift for infdk, crlgm1f, crstate components */
329 return res;
330}

347Matrix CrossAssetStateProcess::diffusionOnCorrelatedBrowniansImpl(Time t, const Array&) const {
348 Matrix res(model_->dimension(), model_->brownians(), 0.0);
349 Size n = model_->components(CrossAssetModel::AssetType::IR);
350 Size m = model_->components(CrossAssetModel::AssetType::FX);
351 Size d = model_->components(CrossAssetModel::AssetType::INF);
352 Size c = model_->components(CrossAssetModel::AssetType::CR);
353 Size e = model_->components(CrossAssetModel::AssetType::EQ);
354 Size com = model_->components(CrossAssetModel::AssetType::COM);
355 Size crstates = model_->components(CrossAssetModel::AssetType::CrState);
356 // ir-ir
357 for (Size i = 0; i < n; ++i) {
358 Real alphai = model_->irlgm1f(i)->alpha(t);
359 setValue2(res, alphai, model_, CrossAssetModel::AssetType::IR, i, CrossAssetModel::AssetType::IR, i, 0, 0);
360 }
361 // fx-fx
362 for (Size i = 0; i < m; ++i) {
363 Real sigmai = model_->fxbs(i)->sigma(t);
364 setValue2(res, sigmai, model_, CrossAssetModel::AssetType::FX, i, CrossAssetModel::AssetType::FX, i, 0, 0);
365 }
366 // inf-inf
367 for (Size i = 0; i < d; ++i) {
368 if (model_->modelType(CrossAssetModel::AssetType::INF, i) == CrossAssetModel::ModelType::DK) {
369 Real alphai = model_->infdk(i)->alpha(t);
370 Real Hi = model_->infdk(i)->H(t);
371 // DK z diffusion coefficient
372 setValue2(res, alphai, model_, CrossAssetModel::AssetType::INF, i, CrossAssetModel::AssetType::INF, i, 0,
373 0);
374 // DK y diffusion coefficient
375 setValue2(res, alphai * Hi, model_, CrossAssetModel::AssetType::INF, i, CrossAssetModel::AssetType::INF, i,
376 1, 0);
377 } else {
378 auto p = model_->infjy(i);
379 // JY z diffusion coefficient
380 setValue2(res, p->realRate()->alpha(t), model_, CrossAssetModel::AssetType::INF, i,
381 CrossAssetModel::AssetType::INF, i, 0, 0);
382 // JY I diffusion coefficient
383 setValue2(res, p->index()->sigma(t), model_, CrossAssetModel::AssetType::INF, i,
384 CrossAssetModel::AssetType::INF, i, 1, 1);
385 }
386 }
387 for (Size i = 0; i < c; ++i) {
388 // Skip CR components that are not LGM
389 if (model_->modelType(CrossAssetModel::AssetType::CR, i) != CrossAssetModel::ModelType::LGM1F)
390 continue;
391 Real alphai = model_->crlgm1f(i)->alpha(t);
392 Real Hi = model_->crlgm1f(i)->H(t);
393 // crz-crz
394 setValue2(res, alphai, model_, CrossAssetModel::AssetType::CR, i, CrossAssetModel::AssetType::CR, i, 0, 0);
395 // cry-cry
396 setValue2(res, alphai * Hi, model_, CrossAssetModel::AssetType::CR, i, CrossAssetModel::AssetType::CR, i, 1, 0);
397 }
398 // eq-eq
399 for (Size i = 0; i < e; ++i) {
400 Real sigmai = model_->eqbs(i)->sigma(t);
401 setValue2(res, sigmai, model_, CrossAssetModel::AssetType::EQ, i, CrossAssetModel::AssetType::EQ, i, 0, 0);
402 }
403 // com-com
404 for (Size i = 0; i < com; ++i) {
405 Real sigmai = model_->combs(i)->sigma(t);
406 setValue2(res, sigmai, model_, CrossAssetModel::AssetType::COM, i, CrossAssetModel::AssetType::COM, i, 0, 0);
407 }
408 // creditstate-creditstate
409 for (Size i = 0; i < crstates; ++i) {
410 setValue2(res, 1.0, model_, CrossAssetModel::AssetType::CrState, i, CrossAssetModel::AssetType::CrState, i, 0,
411 0);
412 }
413
414 if (model_->measure() == IrModel::Measure::BA) {
415 // aux-aux
416 Real H0 = model_->irlgm1f(0)->H(t);
417 Real alpha0 = model_->irlgm1f(0)->alpha(t);
418 setValue2(res, alpha0 * H0, model_, CrossAssetModel::AssetType::IR, 0, CrossAssetModel::AssetType::IR, 0, 1, 0);
419 }
420
421 return res;
422}

431void applyFxDriftAdjustment(Array& state, const CrossAssetModel* model, Size i, Time t0, Time dt) {
432
433 // the specifics depend on the ir and fx model types and their discretizations
434
435 if (model->modelType(CrossAssetModel::AssetType::IR, i) == CrossAssetModel::ModelType::HW &&
436 model->modelType(CrossAssetModel::AssetType::FX, i - 1) == CrossAssetModel::ModelType::BS) {
437
438 QL_REQUIRE(model->discretization() == CrossAssetModel::Discretization::Euler,
439 "applyFxDrifAdjustment(): can only handle discretization Euler at the moment.");
440 Matrix corrTmp(model->irModel(i)->m(), 1);
441 for (Size k = 0; k < model->irModel(i)->m(); ++k) {
442 corrTmp(k, 0) =
443 model->correlation(CrossAssetModel::AssetType::IR, i, CrossAssetModel::AssetType::FX, i - 1, k, 0);
444 }
445 Matrix driftAdj = dt * model->fxbs(i - 1)->sigma(t0) * (transpose(model->irhw(i)->sigma_x(t0)) * corrTmp);
446 auto s = std::next(state.begin(), model->pIdx(CrossAssetModel::AssetType::IR, i, 0));
447 for (auto c = driftAdj.column_begin(0); c != driftAdj.column_end(0); ++c, ++s) {
448 *s += *c;
449 }
450
451 } else {
452 QL_FAIL("applyFxDriftAdjustment(): can only handle ir model type HW and fx model type BS currently.");
453 }
454}

457Array CrossAssetStateProcess::evolve(Time t0, const Array& x0, Time dt, const Array& dw) const {
458
459 Array res;
460
461 // handle HW based model
462
463 if (model_->modelType(CrossAssetModel::AssetType::IR, 0) == CrossAssetModel::ModelType::HW) {
464
465 QL_REQUIRE(model_->discretization() == CrossAssetModel::Discretization::Euler,
466 "CrossAssetStateProcess::evolve(): hw-based model only supports Euler discretization.");
467
468 const Array dz = sqrtCorrelation_ * dw;
469
470 res = Array(model_->dimension());
471
472 // eolve ir processes and store current short rates which are needed to evolve the fx components below
473
474 Array shortRates(model_->components(CrossAssetModel::AssetType::IR));
475 for (Size i = 0; i < model_->components(CrossAssetModel::AssetType::IR); ++i) {
476 auto p = model_->irModel(i)->stateProcess();
477 auto r = p->evolve(t0,
478 getProjectedArray(x0, model_->pIdx(CrossAssetModel::AssetType::IR, i, 0),
479 model_->irModel(i)->n() + model_->irModel(i)->n_aux()),
480 dt,
481 getProjectedArray(dz, model_->wIdx(CrossAssetModel::AssetType::IR, i, 0),
482 model_->irModel(i)->m() + model_->irModel(i)->m_aux()));
483 std::copy(r.begin(), r.end(), std::next(res.begin(), model_->pIdx(CrossAssetModel::AssetType::IR, i, 0)));
484 shortRates[i] = model_->irModel(i)->shortRate(
485 t0, getProjectedArray(x0, model_->pIdx(CrossAssetModel::AssetType::IR, i, 0), model_->irModel(i)->n()));
486 }
487
488 // apply drift adjustment to ir processes in non-domestic currency
489
490 for (Size i = 1; i < model_->components(CrossAssetModel::AssetType::IR); ++i) {
491 applyFxDriftAdjustment(res, model_, i, t0, dt);
492 }
493
494 // eolve fx processes
495
496 for (Size i = 0; i < model_->components(CrossAssetModel::AssetType::FX); ++i) {
497 auto r = model_->fxModel(i)->eulerStep(
498 t0, getProjectedArray(x0, model_->pIdx(CrossAssetModel::AssetType::FX, i, 0), model_->fxModel(i)->n()),

505
506 for (Size i = 0; i < model_->components(CrossAssetModel::AssetType::COM); ++i) {
507 auto p = model_->comModel(i)->stateProcess();
508 auto r = p->evolve(
509 t0,
510 getProjectedArray(x0, model_->pIdx(CrossAssetModel::AssetType::COM, i, 0), model_->comModel(i)->n()),

512 getProjectedArray(dz, model_->wIdx(CrossAssetModel::AssetType::COM, i, 0), model_->comModel(i)->m()));

518 model_->components(CrossAssetModel::AssetType::FX) +
519 model_->components(CrossAssetModel::AssetType::COM) ==

521 "CrossAssetStateProcess::evolve(): currently only IR, FX, COM supported.");
522
523 return res;
524 }
525
526 // handle LGM1F based model
527

529 const Array dz = sqrtCorrelation_ * dw;
530 const Matrix df = diffusionOnCorrelatedBrownians(t0, x0);
531 res = apply(expectation(t0, x0, dt), df * dz * std::sqrt(dt));
532
533 // CR CIRPP components
534 if (cirppCount_ > 0) {
535 for (Size i = 0; i < model_->components(CrossAssetModel::AssetType::CR); ++i) {
536 if (!crCirpp_[i])
537 continue; // ignore non-cir cr model
538 Size idx1 = model_->pIdx(CrossAssetModel::AssetType::CR, i, 0);
539 Size idx2 = model_->pIdx(CrossAssetModel::AssetType::CR, i, 1);
540 Size idxw = model_->wIdx(CrossAssetModel::AssetType::CR, i, 0);
541 Array x0Tmp(2), dwTmp(2);
542 x0Tmp[0] = x0[idx1];
543 x0Tmp[1] = x0[idx2];
544 dwTmp[0] = dz[idxw];
545 dwTmp[1] = 0.0; // not used
546 // evolve original process
547 auto r = crCirpp_[i]->evolve(t0, x0Tmp, dt, dwTmp);
548
549 // set result
550 res[idx1] = r[0]; // y
551 res[idx2] = r[1]; // S(0,T)
552 }
553 }
554 } else {
555 QL_REQUIRE(cirppCount_ == 0, "only euler discretization is supported for CIR++");
556 res = StochasticProcess::evolve(t0, x0, dt, dw);
557 }
558
559 return res;

615Array CrossAssetStateProcess::ExactDiscretization::driftImpl1(const StochasticProcess&, Time t0, const Array&,
616 Time dt) const {
617 Size n = model_->components(CrossAssetModel::AssetType::IR);
618 Size m = model_->components(CrossAssetModel::AssetType::FX);
619 Size e = model_->components(CrossAssetModel::AssetType::EQ);
620 Array res(model_->dimension(), 0.0);
621 for (Size i = 0; i < n; ++i) {
622 res[model_->pIdx(CrossAssetModel::AssetType::IR, i, 0)] = ir_expectation_1(model_, i, t0, dt);
623 }
624 for (Size j = 0; j < m; ++j) {
625 res[model_->pIdx(CrossAssetModel::AssetType::FX, j, 0)] = fx_expectation_1(model_, j, t0, dt);
626 }
627 for (Size k = 0; k < e; ++k) {
628 res[model_->pIdx(CrossAssetModel::AssetType::EQ, k, 0)] = eq_expectation_1(model_, k, t0, dt);
629 }
630
631 // If inflation is JY, need to take account of the drift.
632 for (Size i = 0; i < model_->components(CrossAssetModel::AssetType::INF); ++i) {
633 if (model_->modelType(CrossAssetModel::AssetType::INF, i) == CrossAssetModel::ModelType::JY) {
634 std::tie(res[model_->pIdx(CrossAssetModel::AssetType::INF, i, 0)],
635 res[model_->pIdx(CrossAssetModel::AssetType::INF, i, 1)]) =
636 inf_jy_expectation_1(model_, i, t0, dt);
637 }
638 }
639
640 /* no COM driftImpl1 contribution for one-factor non mean-reverting commodity case,
641 no crstate contribution */
642
643 return res;

646Array CrossAssetStateProcess::ExactDiscretization::driftImpl2(const StochasticProcess&, Time t0, const Array& x0,
647 Time dt) const {
648 Size n = model_->components(CrossAssetModel::AssetType::IR);
649 Size m = model_->components(CrossAssetModel::AssetType::FX);
650 Size e = model_->components(CrossAssetModel::AssetType::EQ);
651 Array res(model_->dimension(), 0.0);
652
653 if (model_->measure() == IrModel::Measure::BA) {
654 // zero AUX state drift, i.e. conditional expectation equal to the previous level as for z_i
655 res[model_->pIdx(CrossAssetModel::AssetType::IR, 0, 1)] +=
656 x0[model_->pIdx(CrossAssetModel::AssetType::IR, 0, 1)];
657 }
658
659 for (Size i = 0; i < n; ++i) {
660 res[model_->pIdx(CrossAssetModel::AssetType::IR, i, 0)] +=
661 ir_expectation_2(model_, i, x0[model_->pIdx(CrossAssetModel::AssetType::IR, i, 0)]);
662 }
663 for (Size j = 0; j < m; ++j) {
664 res[model_->pIdx(CrossAssetModel::AssetType::FX, j, 0)] +=
665 fx_expectation_2(model_, j, t0, x0[model_->pIdx(CrossAssetModel::AssetType::FX, j, 0)],
666 x0[model_->pIdx(CrossAssetModel::AssetType::IR, j + 1, 0)],
667 x0[model_->pIdx(CrossAssetModel::AssetType::IR, 0, 0)], dt);
668 }
669 for (Size k = 0; k < e; ++k) {
670 Size eqCcyIdx = model_->ccyIndex(model_->eqbs(k)->currency());
671 res[model_->pIdx(CrossAssetModel::AssetType::EQ, k, 0)] +=
672 eq_expectation_2(model_, k, t0, x0[model_->pIdx(CrossAssetModel::AssetType::EQ, k, 0)],
673 x0[model_->pIdx(CrossAssetModel::AssetType::IR, eqCcyIdx, 0)], dt);
674 }
675
676 // Inflation: JY is state dependent. DK is not. Even for DK, still need to return the conditional expected value.
677 for (Size i = 0; i < model_->components(CrossAssetModel::AssetType::INF); ++i) {
678 if (model_->modelType(CrossAssetModel::AssetType::INF, i) == CrossAssetModel::ModelType::JY) {
679 auto i_i = model_->ccyIndex(model_->infjy(i)->currency());
680 auto zi_i_0 = x0[model_->pIdx(CrossAssetModel::AssetType::IR, i_i, 0)];
681 auto state_0 = std::make_pair(x0[model_->pIdx(CrossAssetModel::AssetType::INF, i, 0)],
682 x0[model_->pIdx(CrossAssetModel::AssetType::INF, i, 1)]);
683 std::tie(res[model_->pIdx(CrossAssetModel::AssetType::INF, i, 0)],
684 res[model_->pIdx(CrossAssetModel::AssetType::INF, i, 1)]) =
685 inf_jy_expectation_2(model_, i, t0, state_0, zi_i_0, dt);
686 } else {
687 res[model_->pIdx(CrossAssetModel::AssetType::INF, i, 0)] =
688 x0[model_->pIdx(CrossAssetModel::AssetType::INF, i, 0)];
689 res[model_->pIdx(CrossAssetModel::AssetType::INF, i, 1)] =
690 x0[model_->pIdx(CrossAssetModel::AssetType::INF, i, 1)];
691 }
692 }
693
694 /*! cr components have integrated drift 0, we have to return the conditional
695 expected value though, since x0 is subtracted later */
696 Size c = model_->components(CrossAssetModel::AssetType::CR);
697 for (Size i = 0; i < c; ++i) {
698 res[model_->pIdx(CrossAssetModel::AssetType::CR, i, 0)] =
699 x0[model_->pIdx(CrossAssetModel::AssetType::CR, i, 0)];
700 res[model_->pIdx(CrossAssetModel::AssetType::CR, i, 1)] =
701 x0[model_->pIdx(CrossAssetModel::AssetType::CR, i, 1)];
702 }
703
704 /* commodity components are drift-free in the one-factor non mean-reverting case */
705 Size com = model_->components(CrossAssetModel::AssetType::COM);
706 for (Size i = 0; i < com; ++i) {
707 // res[model_->pIdx(CrossAssetModel::AssetType::COM, i, 0)] =
708 // x0[model_->pIdx(CrossAssetModel::AssetType::COM, i, 0)];
709 auto cm = boost::dynamic_pointer_cast<CommoditySchwartzModel>(model_->comModel(i));
710 QL_REQUIRE(cm, "CommoditySchwartzModel not set");
711 Real com0 = x0[model_->pIdx(CrossAssetModel::AssetType::COM, i, 0)];
712 if (cm->parametrization()->driftFreeState()) {
713 res[model_->pIdx(CrossAssetModel::AssetType::COM, i, 0)] = com0;
714 } else {
715 Real kap = cm->parametrization()->kappaParameter();
716 res[model_->pIdx(CrossAssetModel::AssetType::COM, i, 0)] = com0 * std::exp(-kap * dt);
717 }
718 }
719
720 for (Size j = 0; j < model_->components(CrossAssetModel::AssetType::CrState); ++j) {
721 Size idx = model_->pIdx(CrossAssetModel::AssetType::CrState, j, 0);
722 res[idx] += x0[idx]; // drift free
723 }
724
725 return res;
726}
