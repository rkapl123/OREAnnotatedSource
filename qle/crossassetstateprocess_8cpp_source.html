93 Array CrossAssetStateProcess::initialValues() const {
94  Array res(model_->dimension(), 0.0);
95  /* irlgm1f processes have initial value 0 */
96  for (Size i = 0; i < model_->components(FX); ++i) {
97  /* fxbs processes are in log spot */
98  res[model_->pIdx(FX, i, 0)] = std::log(model_->fxbs(i)->fxSpotToday()->value());
99  }
100  for (Size i = 0; i < model_->components(EQ); ++i) {
101  /* eqbs processes are in log spot */
102  res[model_->pIdx(EQ, i, 0)] = std::log(model_->eqbs(i)->eqSpotToday()->value());
103  }
104  // CR CIR++ components
105  for (Size i = 0; i < model_->components(CR); ++i) {
106  if (model_->modelType(CR, i) != CIRPP)
107  continue;
108  QL_REQUIRE(crCirpp_[i], "crcirpp is null!");
109  Array r = crCirpp_[i]->initialValues();
110  res[model_->pIdx(CR, i, 0)] = r[0]; // y0
111  res[model_->pIdx(CR, i, 1)] = r[1]; // S(0,0) = 1
112  }
113 
114  for (Size i = 0; i < model_->components(INF); ++i) {
115  // Second component of JY model is the inflation index process.
116  if (model_->modelType(INF, i) == JY) {
117  res[model_->pIdx(INF, i, 1)] = std::log(model_->infjy(i)->index()->fxSpotToday()->value());
118  }
119  }
120  /* infdk, crlgm1f, aux processes have initial value 0 */
121  return res;
122 }

124 Array CrossAssetStateProcess::drift(Time t, const Array& x) const {
125  Array res(model_->dimension(), 0.0);
126  Size n = model_->components(IR);
127  Size n_eq = model_->components(EQ);
128  Real H0 = model_->irlgm1f(0)->H(t);
129  Real Hprime0 = model_->irlgm1f(0)->Hprime(t);
130  Real alpha0 = model_->irlgm1f(0)->alpha(t);
131  Real zeta0 = model_->irlgm1f(0)->zeta(t);
132  auto i = cache_m_.find(t);
133  if (i == cache_m_.end()) {
134  /* z0 has drift 0 in the LGM measure but non-zero drift in the bank account measure, so start loop at i = 0 */
135  for (Size i = 0; i < n; ++i) {
136  Real Hi = model_->irlgm1f(i)->H(t);
137  Real alphai = model_->irlgm1f(i)->alpha(t);
138  if (i == 0 && model_->measure() == Measure::BA) {
139  // ADD z0 drift in the BA measure
140  res[model_->pIdx(IR, i, 0)] = -Hi * alphai * alphai;
141  // the auxiliary state variable is drift-free
142  res[model_->pIdx(AUX, i, 0)] = 0.0;
143  }
144  if (i > 0) {
145  Real sigmai = model_->fxbs(i - 1)->sigma(t);
146  // ir-ir
147  Real rhozz0i = model_->correlation(IR, 0, IR, i);
148  // ir-fx
149  Real rhozx0i = model_->correlation(IR, 0, FX, i - 1);
150  Real rhozxii = model_->correlation(IR, i, FX, i - 1);
151  // ir drifts
152  res[model_->pIdx(IR, i, 0)] =
153  -Hi * alphai * alphai + H0 * alpha0 * alphai * rhozz0i - sigmai * alphai * rhozxii;
154  // log spot fx drifts (z0, zi independent parts)
155  res[model_->pIdx(FX, i - 1, 0)] = H0 * alpha0 * sigmai * rhozx0i +
156  model_->irlgm1f(0)->termStructure()->forwardRate(t, t, Continuous) -
157  model_->irlgm1f(i)->termStructure()->forwardRate(t, t, Continuous) -
158  0.5 * sigmai * sigmai;
159  if (model_->measure() == Measure::BA) {
160  // REMOVE the LGM measure drift contributions above
161  res[model_->pIdx(IR, i, 0)] -= H0 * alpha0 * alphai * rhozz0i;
162  res[model_->pIdx(FX, i - 1, 0)] -= H0 * alpha0 * sigmai * rhozx0i;
163  }
164  }
165  }
166  /* log equity spot drifts (the cache-able parts) */
167  for (Size k = 0; k < n_eq; ++k) {
168  Size i = model_->ccyIndex(model_->eqbs(k)->currency());
169  // ir params (for equity currency)
170  Real eps_ccy = (i == 0) ? 0.0 : 1.0;
171  // Real Hi = model_->irlgm1f(i)->H(t);
172  // Real alphai = model_->irlgm1f(i)->alpha(t);
173  // eq vol
174  Real sigmask = model_->eqbs(k)->sigma(t);
175  // fx vol (eq ccy / base ccy)
176  Real sigmaxi = (i == 0) ? 0.0 : model_->fxbs(i - 1)->sigma(t);
177  // ir-eq corr
178  // Real rhozsik = model_->correlation(EQ, k, IR, i); // eq cur
179  Real rhozs0k = model_->correlation(EQ, k, IR, 0); // base cur
180  // fx-eq corr
181  Real rhoxsik = (i == 0) ? 0.0 : // no fx process for base-ccy
182  model_->correlation(FX, i - 1, EQ, k);
183  // ir instantaneous forward rate (from curve used for eq forward projection)
184  Real fr_i = model_->eqbs(k)->equityIrCurveToday()->forwardRate(t, t, Continuous);
185  // div yield instantaneous forward rate
186  Real fq_k = model_->eqbs(k)->equityDivYieldCurveToday()->forwardRate(t, t, Continuous);
187  res[model_->pIdx(EQ, k, 0)] = fr_i - fq_k + (rhozs0k * H0 * alpha0 * sigmask) -
188  (eps_ccy * rhoxsik * sigmaxi * sigmask) - (0.5 * sigmask * sigmask);
189  }
190 
191  // State independent pieces of JY inflation model, if there is a CAM JY component.
192  for (Size j = 0; j < model_->components(INF); ++j) {
193 
194  if (model_->modelType(INF, j) == JY) {
195 
196  auto p = model_->infjy(j);
197  Size i_j = model_->ccyIndex(p->currency());
198 
199  // JY inflation parameter values.
200  Real H_y_j = p->realRate()->H(t);
201  Real Hp_y_j = p->realRate()->Hprime(t);
202  Real zeta_y_j = p->realRate()->zeta(t);
203  Real alpha_y_j = p->realRate()->alpha(t);
204  Real sigma_c_j = p->index()->sigma(t);
205 
206  // Inflation nominal currency parameter values
207  Real H_i_j = model_->irlgm1f(i_j)->H(t);
208  Real Hp_i_j = model_->irlgm1f(i_j)->Hprime(t);
209  Real zeta_i_j = model_->irlgm1f(i_j)->zeta(t);
210 
211  // Correlations
212  Real rho_zy_0j = model_->correlation(IR, 0, INF, j, 0, 0);
213  Real rho_yc_ij = model_->correlation(INF, j, INF, j, 0, 1);
214  Real rho_zc_0j = model_->correlation(IR, 0, INF, j, 0, 1);
215 
216  // JY real rate drift. It is state independent
217  auto rrDrift = -alpha_y_j * alpha_y_j * H_y_j + rho_zy_0j * alpha0 * alpha_y_j * H_y_j -
218  rho_yc_ij * alpha_y_j * sigma_c_j;
219 
220  if (i_j > 0) {
221  Real sigma_x_i_j = model_->fxbs(i_j - 1)->sigma(t);
222  Real rho_yx_j_i_j = model_->correlation(INF, j, FX, i_j - 1, 0, 0);
223  rrDrift -= rho_yx_j_i_j * alpha_y_j * sigma_x_i_j;
224  }
225 
226  res[model_->pIdx(INF, j, 0)] = rrDrift;
227 
228  // JY log inflation index drift (state independent piece).
229  auto indexDrift = rho_zc_0j * alpha0 * sigma_c_j * H0 - 0.5 * sigma_c_j * sigma_c_j +
230  zeta_i_j * Hp_i_j * H_i_j - zeta_y_j * Hp_y_j * H_y_j;
231 
232  // Add on the f_n(0, t) - f_r(0, t) piece using the initial zero inflation term structure.
233  // Use the same dt below that is used in yield forward rate calculations.
234  auto ts = p->realRate()->termStructure();
235  Time dt = 0.0001;
236  Time t1 = std::max(t - dt / 2.0, 0.0);
237  Time t2 = t1 + dt;
238  auto z_t = ts->zeroRate(t);
239  auto z_t1 = ts->zeroRate(t1);
240  auto z_t2 = ts->zeroRate(t2);
241  indexDrift += std::log(1 + z_t) + (t / (1 + z_t)) * ((z_t2 - z_t1) / dt);
242 
243  if (i_j > 0) {
244  Real sigma_x_i_j = model_->fxbs(i_j - 1)->sigma(t);
245  Real rho_cx_j_i_j = model_->correlation(INF, j, FX, i_j - 1, 1, 0);
246  indexDrift -= rho_cx_j_i_j * sigma_c_j * sigma_x_i_j;
247  }
248 
249  res[model_->pIdx(INF, j, 1)] = indexDrift;
250  }
251  }
252 
253  cache_m_.insert(std::make_pair(t, res));
254  } else {
255  res = i->second;
256  }
257  // non-cacheable sections of drifts
258  for (Size i = 1; i < n; ++i) {
259  // log spot fx drifts (z0, zi dependent parts)
260  Real Hi = model_->irlgm1f(i)->H(t);
261  Real Hprimei = model_->irlgm1f(i)->Hprime(t);
262  Real zetai = model_->irlgm1f(i)->zeta(t);
263  res[model_->pIdx(FX, i - 1, 0)] += x[model_->pIdx(IR, 0, 0)] * Hprime0 + zeta0 * Hprime0 * H0 -
264  x[model_->pIdx(IR, i, 0)] * Hprimei - zetai * Hprimei * Hi;
265  }
266  for (Size k = 0; k < n_eq; ++k) {
267  // log equity spot drifts (path-dependent parts)
268  // notice the assumption in below that dividend yield curve is static
269  Size i = model_->ccyIndex(model_->eqbs(k)->currency());
270  // ir params (for equity currency)
271  Real Hi = model_->irlgm1f(i)->H(t);
272  Real Hprimei = model_->irlgm1f(i)->Hprime(t);
273  Real zetai = model_->irlgm1f(i)->zeta(t);
274  res[model_->pIdx(EQ, k, 0)] += (x[model_->pIdx(IR, i, 0)] * Hprimei) + (zetai * Hprimei * Hi);
275  }
276 
277  // Non-cacheable portion of inflation JY drift, if there is a CAM JY component.
278  for (Size j = 0; j < model_->components(INF); ++j) {
279  if (model_->modelType(INF, j) == JY) {
280 
281  auto p = model_->infjy(j);
282  Size i_j = model_->ccyIndex(p->currency());
283 
284  // JY inflation parameter values.
285  Real Hp_y_j = p->realRate()->Hprime(t);
286 
287  // Inflation nominal currency parameter values
288  Real Hp_i_j = model_->irlgm1f(i_j)->Hprime(t);
289 
290  res[model_->pIdx(INF, j, 1)] += x[model_->pIdx(IR, i_j, 0)] * Hp_i_j - x[model_->pIdx(INF, j, 0)] * Hp_y_j;
291  }
292  }
293 
294  /* no drift for infdk, crlgm1f components */
295  return res;
296 }

313 Matrix CrossAssetStateProcess::diffusionOnCorrelatedBrowniansImpl(Time t, const Array&) const {
314  Matrix res(model_->dimension(), model_->brownians(), 0.0);
315  Size n = model_->components(IR);
316  Size m = model_->components(FX);
317  Size d = model_->components(INF);
318  Size c = model_->components(CR);
319  Size e = model_->components(EQ);
320  // ir-ir
321  for (Size i = 0; i < n; ++i) {
322  Real alphai = model_->irlgm1f(i)->alpha(t);
323  setValue2(res, alphai, model_, IR, i, IR, i, 0, 0);
324  }
325  // fx-fx
326  for (Size i = 0; i < m; ++i) {
327  Real sigmai = model_->fxbs(i)->sigma(t);
328  setValue2(res, sigmai, model_, FX, i, FX, i, 0, 0);
329  }
330  // inf-inf
331  for (Size i = 0; i < d; ++i) {
332  if (model_->modelType(INF, i) == DK) {
333  Real alphai = model_->infdk(i)->alpha(t);
334  Real Hi = model_->infdk(i)->H(t);
335  // DK z diffusion coefficient
336  setValue2(res, alphai, model_, INF, i, INF, i, 0, 0);
337  // DK y diffusion coefficient
338  setValue2(res, alphai * Hi, model_, INF, i, INF, i, 1, 0);
339  } else {
340  auto p = model_->infjy(i);
341  // JY z diffusion coefficient
342  setValue2(res, p->realRate()->alpha(t), model_, INF, i, INF, i, 0, 0);
343  // JY I diffusion coefficient
344  setValue2(res, p->index()->sigma(t), model_, INF, i, INF, i, 1, 1);
345  }
346  }
347  for (Size i = 0; i < c; ++i) {
348  // Skip CR components that are not LGM
349  if (model_->modelType(CR, i) != LGM1F)
350  continue;
351  Real alphai = model_->crlgm1f(i)->alpha(t);
352  Real Hi = model_->crlgm1f(i)->H(t);
353  // crz-crz
354  setValue2(res, alphai, model_, CR, i, CR, i, 0, 0);
355  // cry-cry
356  setValue2(res, alphai * Hi, model_, CR, i, CR, i, 1, 0);
357  }
358  // eq-eq
359  for (Size i = 0; i < e; ++i) {
360  Real sigmai = model_->eqbs(i)->sigma(t);
361  setValue2(res, sigmai, model_, EQ, i, EQ, i, 0, 0);
362  }
363 
364  if (model_->measure() == Measure::BA) {
365  // aux-aux
366  Real H0 = model_->irlgm1f(0)->H(t);
367  Real alpha0 = model_->irlgm1f(0)->alpha(t);
368  setValue2(res, alpha0 * H0, model_, AUX, 0, AUX, 0, 0, 0);
369  }
370 
371  return res;
372 }

374 Array CrossAssetStateProcess::evolve(Time t0, const Array& x0, Time dt, const Array& dw) const {
375 
376  Array res;
377  if (disc_ == euler) {
378  const Array dz = sqrtCorrelation_ * dw;
379  const Matrix df = diffusionOnCorrelatedBrownians(t0, x0);
380  res = apply(expectation(t0, x0, dt), df * dz * std::sqrt(dt));
381 
382  // CR CIRPP components
383  if (cirppCount_ > 0) {
384  for (Size i = 0; i < model_->components(CR); ++i) {
385  if (!crCirpp_[i])
386  continue; // ignore non-cir cr model
387  Size idx1 = model_->pIdx(CR, i, 0);
388  Size idx2 = model_->pIdx(CR, i, 1);
389  Size idxw = model_->cIdx(CR, i, 0);
390  Array x0Tmp(2), dwTmp(2);
391  x0Tmp[0] = x0[idx1];
392  x0Tmp[1] = x0[idx2];
393  dwTmp[0] = dz[idxw];
394  dwTmp[1] = 0.0; // not used
395  // evolve original process
396  auto r = crCirpp_[i]->evolve(t0, x0Tmp, dt, dwTmp);
397 
398  // set result
399  res[idx1] = r[0]; // y
400  res[idx2] = r[1]; // S(0,T)
401  }
402  }
403  } else {
404  QL_REQUIRE(cirppCount_ == 0, "only euler discretization is supported for CIR++");
405  res = StochasticProcess::evolve(t0, x0, dt, dw);
406  }
407 
408  return res;
409 }

416  const Array& x0, Time dt) const {
417  Array res;

420  if (i == cache_m_.end()) {
421  res = driftImpl1(p, t0, x0, dt);

460 Array CrossAssetStateProcess::ExactDiscretization::driftImpl1(const StochasticProcess&, Time t0,
461  const Array&, Time dt) const {
462  Size n = model_->components(IR);
463  Size m = model_->components(FX);
464  Size e = model_->components(EQ);
465  Array res(model_->dimension(), 0.0);
466  for (Size i = 0; i < n; ++i) {
467  res[model_->pIdx(IR, i, 0)] = ir_expectation_1(model_, i, t0, dt);
468  }
469  for (Size j = 0; j < m; ++j) {
470  res[model_->pIdx(FX, j, 0)] = fx_expectation_1(model_, j, t0, dt);
471  }
472  for (Size k = 0; k < e; ++k) {
473  res[model_->pIdx(EQ, k, 0)] = eq_expectation_1(model_, k, t0, dt);
474  }
475 
476  // If inflation is JY, need to take account of the drift.
477  for (Size i = 0; i < model_->components(INF); ++i) {
478  if (model_->modelType(INF, i) == JY) {
479  std::tie(res[model_->pIdx(INF, i, 0)], res[model_->pIdx(INF, i, 1)]) =
480  inf_jy_expectation_1(model_, i, t0, dt);
481  }
482  }
483 
484  return res;
485 }

487 Array CrossAssetStateProcess::ExactDiscretization::driftImpl2(const StochasticProcess&, Time t0,
488  const Array& x0, Time dt) const {
489  Size n = model_->components(IR);
490  Size m = model_->components(FX);
491  Size e = model_->components(EQ);
492  Array res(model_->dimension(), 0.0);
493 
494  if (model_->measure() == Measure::BA) {
495  // zero AUX state drift, i.e. conditional expectation equal to the previous level as for z_i
496  res[model_->pIdx(AUX, 0, 0)] += x0[model_->pIdx(AUX, 0, 0)];
497  }
498 
499  for (Size i = 0; i < n; ++i) {
500  res[model_->pIdx(IR, i, 0)] += ir_expectation_2(model_, i, x0[model_->pIdx(IR, i, 0)]);
501  }
502  for (Size j = 0; j < m; ++j) {
503  res[model_->pIdx(FX, j, 0)] += fx_expectation_2(model_, j, t0, x0[model_->pIdx(FX, j, 0)],
504  x0[model_->pIdx(IR, j + 1, 0)], x0[model_->pIdx(IR, 0, 0)], dt);
505  }
506  for (Size k = 0; k < e; ++k) {
507  Size eqCcyIdx = model_->ccyIndex(model_->eqbs(k)->currency());
508  res[model_->pIdx(EQ, k, 0)] +=
509  eq_expectation_2(model_, k, t0, x0[model_->pIdx(EQ, k, 0)], x0[model_->pIdx(IR, eqCcyIdx, 0)], dt);
510  }
511 
512  // Inflation: JY is state dependent. DK is not. Even for DK, still need to return the conditional expected value.
513  for (Size i = 0; i < model_->components(INF); ++i) {
514  if (model_->modelType(INF, i) == JY) {
515  auto i_i = model_->ccyIndex(model_->infjy(i)->currency());
516  auto zi_i_0 = x0[model_->pIdx(IR, i_i, 0)];
517  auto state_0 = std::make_pair(x0[model_->pIdx(INF, i, 0)], x0[model_->pIdx(INF, i, 1)]);
518  std::tie(res[model_->pIdx(INF, i, 0)], res[model_->pIdx(INF, i, 1)]) =
519  inf_jy_expectation_2(model_, i, t0, state_0, zi_i_0, dt);
520  } else {
521  res[model_->pIdx(INF, i, 0)] = x0[model_->pIdx(INF, i, 0)];
522  res[model_->pIdx(INF, i, 1)] = x0[model_->pIdx(INF, i, 1)];
523  }
524  }
525 
526  /*! cr components have integrated drift 0, we have to return the conditional
527  expected value though, since x0 is subtracted later */
528  Size c = model_->components(CR);
529  for (Size i = 0; i < c; ++i) {
530  res[model_->pIdx(CR, i, 0)] = x0[model_->pIdx(CR, i, 0)];
531  res[model_->pIdx(CR, i, 1)] = x0[model_->pIdx(CR, i, 1)];
532  }
533  return res;
534 }

536 Matrix CrossAssetStateProcess::ExactDiscretization::covarianceImpl(const StochasticProcess&, Time t0,
537  const Array&, Time dt) const {
538  Matrix res(model_->dimension(), model_->dimension());
539  Size n = model_->components(IR);
540  Size m = model_->components(FX);
541  Size d = model_->components(INF);
542  Size c = model_->components(CR);
543  Size e = model_->components(EQ);
544 
545  if (model_->measure() == Measure::BA) {
546  // aux-aux
547  setValue(res, aux_aux_covariance(model_, t0, dt), model_, AUX, 0, AUX, 0, 0, 0);
548  // aux-ir
549  for (Size j = 0; j < n; ++j) {
550  setValue(res, aux_ir_covariance(model_, j, t0, dt), model_, AUX, 0, IR, j, 0, 0);
551  }
552  // aux-fx
553  for (Size j = 0; j < m; ++j) {
554  setValue(res, aux_fx_covariance(model_, j, t0, dt), model_, AUX, 0, FX, j, 0, 0);
555  }
556  }
557  // ir-ir
558  for (Size i = 0; i < n; ++i) {
559  for (Size j = 0; j <= i; ++j) {
560  setValue(res, ir_ir_covariance(model_, i, j, t0, dt), model_, IR, i, IR, j, 0, 0);
561  }
562  }
563  // ir-fx
564  for (Size i = 0; i < n; ++i) {
565  for (Size j = 0; j < m; ++j) {
566  setValue(res, ir_fx_covariance(model_, i, j, t0, dt), model_, IR, i, FX, j, 0, 0);
567  }
568  }
569  // fx-fx
570  for (Size i = 0; i < m; ++i) {
571  for (Size j = 0; j <= i; ++j) {
572  setValue(res, fx_fx_covariance(model_, i, j, t0, dt), model_, FX, i, FX, j);
573  }
574  }
575  // ir,fx,inf - inf
576  for (Size j = 0; j < d; ++j) {
577  for (Size i = 0; i <= j; ++i) {
578  // infz-infz
579  setValue(res, infz_infz_covariance(model_, i, j, t0, dt), model_, INF, i, INF, j, 0, 0);
580  // infz-infy
581  setValue(res, infz_infy_covariance(model_, i, j, t0, dt), model_, INF, i, INF, j, 0, 1);
582  setValue(res, infz_infy_covariance(model_, j, i, t0, dt), model_, INF, i, INF, j, 1, 0);
583  // infy-infy
584  setValue(res, infy_infy_covariance(model_, i, j, t0, dt), model_, INF, i, INF, j, 1, 1);
585  }
586  for (Size i = 0; i < n; ++i) {
587  // ir-inf
588  setValue(res, ir_infz_covariance(model_, i, j, t0, dt), model_, IR, i, INF, j, 0, 0);
589  setValue(res, ir_infy_covariance(model_, i, j, t0, dt), model_, IR, i, INF, j, 0, 1);
590  }
591  for (Size i = 0; i < (n - 1); ++i) {
592  // fx-inf
593  setValue(res, fx_infz_covariance(model_, i, j, t0, dt), model_, FX, i, INF, j, 0, 0);
594  setValue(res, fx_infy_covariance(model_, i, j, t0, dt), model_, FX, i, INF, j, 0, 1);
595  }
596  }
597  // ir,fx,inf,cr - cr
598  for (Size j = 0; j < c; ++j) {
599  // Skip CR components that are not LGM
600  if (model_->modelType(CR, j) != LGM1F)
601  continue;
602  for (Size i = 0; i <= j; ++i) {
603  // Skip CR components that are not LGM
604  if (model_->modelType(CR, i) != LGM1F)
605  continue;
606  // crz-crz
607  setValue(res, crz_crz_covariance(model_, i, j, t0, dt), model_, CR, i, CR, j, 0, 0);
608  // crz-cry
609  setValue(res, crz_cry_covariance(model_, i, j, t0, dt), model_, CR, i, CR, j, 0, 1);
610  setValue(res, crz_cry_covariance(model_, j, i, t0, dt), model_, CR, i, CR, j, 1, 0);
611  // cry-cry
612  setValue(res, cry_cry_covariance(model_, i, j, t0, dt), model_, CR, i, CR, j, 1, 1);
613  }
614  for (Size i = 0; i < n; ++i) {
615  // ir-cr
616  setValue(res, ir_crz_covariance(model_, i, j, t0, dt), model_, IR, i, CR, j, 0, 0);
617  setValue(res, ir_cry_covariance(model_, i, j, t0, dt), model_, IR, i, CR, j, 0, 1);
618  }
619  for (Size i = 0; i < (n - 1); ++i) {
620  // fx-cr
621  setValue(res, fx_crz_covariance(model_, i, j, t0, dt), model_, FX, i, CR, j, 0, 0);
622  setValue(res, fx_cry_covariance(model_, i, j, t0, dt), model_, FX, i, CR, j, 0, 1);
623  }
624  for (Size i = 0; i < d; ++i) {
625  // inf-cr
626  setValue(res, infz_crz_covariance(model_, i, j, t0, dt), model_, INF, i, CR, j, 0, 0);
627  setValue(res, infy_crz_covariance(model_, i, j, t0, dt), model_, INF, i, CR, j, 1, 0);
628  setValue(res, infz_cry_covariance(model_, i, j, t0, dt), model_, INF, i, CR, j, 0, 1);
629  setValue(res, infy_cry_covariance(model_, i, j, t0, dt), model_, INF, i, CR, j, 1, 1);
630  }
631  }
632  // ir,fx,inf,cr,eq - eq
633  for (Size j = 0; j < e; ++j) {
634  for (Size i = 0; i <= j; ++i) {
635  // eq-eq
636  setValue(res, eq_eq_covariance(model_, i, j, t0, dt), model_, EQ, i, EQ, j, 0, 0);
637  }
638  for (Size i = 0; i < n; ++i) {
639  // ir-eq
640  setValue(res, ir_eq_covariance(model_, i, j, t0, dt), model_, IR, i, EQ, j, 0, 0);
641  }
642  for (Size i = 0; i < (n - 1); ++i) {
643  // fx-eq
644  setValue(res, fx_eq_covariance(model_, i, j, t0, dt), model_, FX, i, EQ, j, 0, 0);
645  }
646  for (Size i = 0; i < d; ++i) {
647  // inf-eq
648  setValue(res, infz_eq_covariance(model_, i, j, t0, dt), model_, INF, i, EQ, j, 0, 0);
649  setValue(res, infy_eq_covariance(model_, i, j, t0, dt), model_, INF, i, EQ, j, 1, 0);
650  }
651  for (Size i = 0; i < c; ++i) {
652  // Skip CR components that are not LGM
653  if (model_->modelType(CR, i) != LGM1F)
654  continue;
655  // cr-eq
656  setValue(res, crz_eq_covariance(model_, i, j, t0, dt), model_, CR, i, EQ, j, 0, 0);
657  setValue(res, cry_eq_covariance(model_, i, j, t0, dt), model_, CR, i, EQ, j, 1, 0);
658  }
659  }
660  return res;
661 }
