50  baseShortSwapIndexBase_->fixingCalendar().adjust(optionDate));
51  Real targetAtmLevel =

53  ? targetSwapIndexBase_->clone(swapTenor)->fixing(targetSwapIndexBase_->fixingCalendar().adjust(optionDate))
54  : targetShortSwapIndexBase_->clone(swapTenor)->fixing(
55  targetShortSwapIndexBase_->fixingCalendar().adjust(optionDate));
56  return boost::make_shared<AtmAdjustedSmileSection>(baseVol_->smileSection(optionDate, swapTenor, true),
57  baseAtmLevel, targetAtmLevel);
