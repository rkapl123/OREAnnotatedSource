57 EquityCoupon::EquityCoupon(const Date& paymentDate, Real nominal, const Date& startDate, const Date& endDate,
58  Natural fixingDays, const boost::shared_ptr<EquityIndex>& equityCurve,
59  const DayCounter& dayCounter, EquityReturnType returnType, Real dividendFactor,
60  bool notionalReset, Real initialPrice, Real quantity, const Date& fixingStartDate,
61  const Date& fixingEndDate, const Date& refPeriodStart, const Date& refPeriodEnd,
62  const Date& exCouponDate, const boost::shared_ptr<FxIndex>& fxIndex,
63  const bool initialPriceIsInTargetCcy, Real legInitialNotional, const Date& legFixingDate)
64  : Coupon(paymentDate, nominal, startDate, endDate, refPeriodStart, refPeriodEnd, exCouponDate),
65  fixingDays_(fixingDays), equityCurve_(equityCurve), dayCounter_(dayCounter), returnType_(returnType),
66  dividendFactor_(dividendFactor), notionalReset_(notionalReset), initialPrice_(initialPrice),
67  initialPriceIsInTargetCcy_(initialPriceIsInTargetCcy), quantity_(quantity), fixingStartDate_(fixingStartDate),
68  fixingEndDate_(fixingEndDate), fxIndex_(fxIndex), legInitialNotional_(legInitialNotional),
69  legFixingDate_(legFixingDate) {
70  QL_REQUIRE(dividendFactor_ > 0.0, "Dividend factor should not be negative. It is expected to be between 0 and 1.");
71  QL_REQUIRE(equityCurve_, "Equity underlying an equity swap coupon cannot be empty.");
72 
73  // set up fixing calendar as combined eq / fx calendar
74  Calendar eqCalendar = NullCalendar();
75  Calendar fxCalendar = NullCalendar();
76  if (!equityCurve_->fixingCalendar().empty()) {
77  eqCalendar = equityCurve_->fixingCalendar();
78  }
79  if (fxIndex_ && !fxIndex_->fixingCalendar().empty()) {
80  fxCalendar = fxIndex_->fixingCalendar();
81  }
82  Calendar fixingCalendar = JointCalendar(eqCalendar, fxCalendar);
83 
84  // If a fixing start / end date is provided, use these
85  // else adjust the start/endDate by the FixingDays - defaulted to 0
86  if (fixingStartDate_ == Date())
87  fixingStartDate_ =
88  fixingCalendar.advance(startDate, -static_cast<Integer>(fixingDays_), Days, Preceding);
89 
90  if (fixingEndDate_ == Date())
91  fixingEndDate_ =
92  fixingCalendar.advance(endDate, -static_cast<Integer>(fixingDays_), Days, Preceding);
93 
94  registerWith(equityCurve_);
95  registerWith(fxIndex_);
96  registerWith(Settings::instance().evaluationDate());
97 
98  // QL_REQUIRE(!notionalReset_ || quantity_ != Null<Real>(), "EquityCoupon: quantity required if notional resets");
99  QL_REQUIRE(notionalReset_ || nominal_ != Null<Real>(),
100  "EquityCoupon: notional required if notional does not reset");
101 }

112 Real EquityCoupon::quantity() const {
113  if (notionalReset_ && quantity_ == Null<Real>()) {
114  QL_REQUIRE(legInitialNotional_ != Null<Real>() && legFixingDate_ != Date(),
115  "leg initial notional and fixing date required to compute the missing quantity in case of a resetting equity leg");
116  quantity_ = legInitialNotional_ / equityCurve_->fixing(legFixingDate_, false, false);
117  }
118  return quantity_;
119 }

121 Real EquityCoupon::nominal() const {
122  // use quantity for dividend swaps, this ensures notional resetting is not relevant
123  // swaplet rate returns the absolute dividend to match
124  if (returnType_ == EquityReturnType::Dividend)
125  return quantity();
126  else if(notionalReset_) {
127  Real mult = (initialPrice_ == 0) ? 1 : initialPrice();
128  return mult * (initialPriceIsInTargetCcy_ ? 1.0 : fxRate()) * quantity();
129  } else {
130  return nominal_;
131  }
132 }

148 Real EquityCoupon::accruedAmount(const Date& d) const {
149  if (d <= accrualStartDate_ || d > paymentDate_) {
150  return 0.0;
151  } else {
152  Time fullPeriod = dayCounter().yearFraction(accrualStartDate_, accrualEndDate_, refPeriodStart_, refPeriodEnd_);
153  Time thisPeriod =
154  dayCounter().yearFraction(accrualStartDate_, std::min(d, accrualEndDate_), refPeriodStart_, refPeriodEnd_);
155  return nominal() * rate() * thisPeriod / fullPeriod;
156  }
157 }

253 EquityLeg::operator Leg() const {
254 
255  Leg cashflows;
256  Date startDate;
257  Date endDate;
258  Date paymentDate;
259 
260  Calendar calendar;
261  if (!paymentCalendar_.empty()) {
262  calendar = paymentCalendar_;
263  } else {
264  calendar = schedule_.calendar();
265  }
266 
267  Size numPeriods = schedule_.size() - 1;
268 
269  if (valuationSchedule_.size() > 0) {
270  QL_REQUIRE(valuationSchedule_.size() == schedule_.size(),
271  "mismatch in valuationSchedule (" << valuationSchedule_.size() << ") and scheduleData ("
272  << schedule_.size() << ") sizes");
273  }
274 
275  Date legFixingDate = Date();
276  if (valuationSchedule_.size() > 0)
277  legFixingDate = valuationSchedule_.dates().front();
278  else if (schedule_.size() > 0)
279  legFixingDate = schedule_.dates().front();
280  else
281  QL_FAIL("Cannot build equity leg, neither schedule nor valuation schedule are defined");
282 
283  Real quantity = Null<Real>(), notional = Null<Real>(), legInitialNotional = Null<Real>();
284  if (notionalReset_) {
285  // We need a quantity in each coupon in this case
286  if (quantity_ != Null<Real>()) {
287  quantity = quantity_;
288  QL_REQUIRE(notionals_.empty(), "EquityLeg: notional and quantity are given at the same time");
289  } else {
290  // If we have a notional, but no quantity is given:
291  // a) Compute quantity from notional and initial price
292  // b) If initial price is missing, leave quantity undefined here:
293  // The coupon will compute it from legInitialNotional=notional and legFixingDate
294  QL_REQUIRE(!notionals_.empty(), "EquityLeg: can not compute qunantity, since no notional is given");
295  QL_REQUIRE(fxIndex_ == nullptr || initialPriceIsInTargetCcy_,
296  "EquityLeg: can not compute quantity from nominal when fx conversion is required");
297  notional = notionals_.front();
298  legInitialNotional = notional;
299  if (initialPrice_ != Null<Real>())
300  quantity = (initialPrice_ == 0) ? notional : notional / initialPrice_;
301  }
302  } else {
303  if (!notionals_.empty()) {
304  QL_REQUIRE(quantity_ == Null<Real>(), "EquityLeg: notional and quantity are given at the same time");
305  // notional is determined below in the loop over the periods
306  legInitialNotional = notionals_.front();
307  } else {
308  QL_REQUIRE(initialPrice_ != Null<Real>(), "EquityLeg: can not compute notional, since no intialPrice is given");
309  QL_REQUIRE(quantity_ != Null<Real>(), "EquityLeg: can not compute notional, since no quantity is given");
310  QL_REQUIRE(fxIndex_ == nullptr || initialPriceIsInTargetCcy_,
311  "EquityLeg: can not compute notional from quantity when fx conversion is required");
312  notional = (initialPrice_ == 0) ? quantity_ : quantity_ * initialPrice_;
313  }
314  }
315 
316  for (Size i = 0; i < numPeriods; ++i) {
317  startDate = schedule_.date(i);
318  endDate = schedule_.date(i + 1);
319  paymentDate = calendar.advance(endDate, paymentLag_, Days, paymentAdjustment_);
320 
321  Date fixingStartDate = Date();
322  Date fixingEndDate = Date();
323  if (valuationSchedule_.size() > 0) {
324  fixingStartDate = valuationSchedule_.date(i);
325  fixingEndDate = valuationSchedule_.date(i + 1);
326  }
327 
328  Real initialPrice = (i == 0) ? initialPrice_ : Null<Real>();
329  bool initialPriceIsInTargetCcy = initialPrice != Null<Real>() ? initialPriceIsInTargetCcy_ : false;
330  if (!notionalReset_ && !notionals_.empty()) {
331  notional = detail::get(notionals_, i, 0.0);
332  }
333 
334  boost::shared_ptr<EquityCoupon> cashflow(new EquityCoupon(
335  paymentDate, notional, startDate, endDate, fixingDays_, equityCurve_, paymentDayCounter_, returnType_,
336  dividendFactor_, notionalReset_, initialPrice, quantity, fixingStartDate, fixingEndDate, Date(), Date(),
337  Date(), fxIndex_, initialPriceIsInTargetCcy, legInitialNotional, legFixingDate));
338 
339  boost::shared_ptr<EquityCouponPricer> pricer(new EquityCouponPricer);
340  cashflow->setPricer(pricer);
341 
342  cashflows.push_back(cashflow);
343  }
344  return cashflows;
345 }
