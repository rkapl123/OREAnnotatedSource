29 EquityCoupon::EquityCoupon(const Date& paymentDate, Real nominal, const Date& startDate, const Date& endDate,
30  Natural fixingDays, const boost::shared_ptr<EquityIndex>& equityCurve,
31  const DayCounter& dayCounter, bool isTotalReturn, Real dividendFactor, bool notionalReset,
32  Real initialPrice, Real quantity, const Date& fixingStartDate, const Date& fixingEndDate,
33  const Date& refPeriodStart, const Date& refPeriodEnd, const Date& exCouponDate,
34  const boost::shared_ptr<FxIndex>& fxIndex, const bool initialPriceIsInTargetCcy, const bool absoluteReturn)
35  : Coupon(paymentDate, nominal, startDate, endDate, refPeriodStart, refPeriodEnd, exCouponDate),
36  fixingDays_(fixingDays), equityCurve_(equityCurve), dayCounter_(dayCounter), isTotalReturn_(isTotalReturn),
37  dividendFactor_(dividendFactor), notionalReset_(notionalReset), initialPrice_(initialPrice),
38  initialPriceIsInTargetCcy_(initialPriceIsInTargetCcy), quantity_(quantity), fixingStartDate_(fixingStartDate),
39  fixingEndDate_(fixingEndDate), fxIndex_(fxIndex), isAbsoluteReturn_(absoluteReturn) {
40  QL_REQUIRE(dividendFactor_ > 0.0, "Dividend factor should not be negative. It is expected to be between 0 and 1.");
41  QL_REQUIRE(equityCurve_, "Equity underlying an equity swap coupon cannot be empty.");
42 
43  // set up fixing calendar as combined eq / fx calendar
44  Calendar eqCalendar = NullCalendar();
45  Calendar fxCalendar = NullCalendar();
46  if (!equityCurve_->fixingCalendar().empty()) {
47  eqCalendar = equityCurve_->fixingCalendar();
48  }
49  if (fxIndex_ && !fxIndex_->fixingCalendar().empty()) {
50  fxCalendar = fxIndex_->fixingCalendar();
51  }
52  Calendar fixingCalendar = JointCalendar(eqCalendar, fxCalendar);
53 
54  // If a fixing start / end date is provided, use these
55  // else adjust the start/endDate by the FixingDays - defaulted to 0
56  if (fixingStartDate_ == Date())
57  fixingStartDate_ =
58  fixingCalendar.advance(startDate, -static_cast<Integer>(fixingDays_), Days, Preceding);
59 
60  if (fixingEndDate_ == Date())
61  fixingEndDate_ =
62  fixingCalendar.advance(endDate, -static_cast<Integer>(fixingDays_), Days, Preceding);
63 
64  registerWith(equityCurve_);
65  registerWith(fxIndex_);
66  registerWith(Settings::instance().evaluationDate());
67 
68  QL_REQUIRE(!notionalReset_ || quantity_ != Null<Real>(), "EquityCoupon: quantity required if notional resets");
69  QL_REQUIRE(notionalReset_ || nominal_ != Null<Real>(),
70  "EquityCoupon: notional required if notional does not reset");
71 }

82 Real EquityCoupon::nominal() const {
83  if (notionalReset_) {
84  Real mult = (initialPrice_ == 0) ? 1 : initialPrice();
85  return mult * (initialPriceIsInTargetCcy_ ? 1.0 : fxRate()) * quantity_;
86  } else {
87  return nominal_;
88  }

105 Real EquityCoupon::accruedAmount(const Date& d) const {
106  if (d <= accrualStartDate_ || d > paymentDate_) {
107  return 0.0;
108  } else {
109  Time fullPeriod = dayCounter().yearFraction(accrualStartDate_, accrualEndDate_, refPeriodStart_, refPeriodEnd_);
110  Time thisPeriod =
111  dayCounter().yearFraction(accrualStartDate_, std::min(d, accrualEndDate_), refPeriodStart_, refPeriodEnd_);
112  return nominal() * rate() * thisPeriod / fullPeriod;
113  }
114 }

215 EquityLeg::operator Leg() const {
216 
217  Leg cashflows;
218  Date startDate;
219  Date endDate;
220  Date paymentDate;
221 
222  Calendar calendar;
223  if (!paymentCalendar_.empty()) {
224  calendar = paymentCalendar_;
225  } else {
226  calendar = schedule_.calendar();
227  }
228 
229  Size numPeriods = schedule_.size() - 1;
230 
231  if (valuationSchedule_.size() > 0) {
232  QL_REQUIRE(valuationSchedule_.size() == schedule_.size(),
233  "mismatch in valuationSchedule (" << valuationSchedule_.size() << ") and scheduleData (" << schedule_.size() << ") sizes");
234  }
235 
236  for (Size i = 0; i < numPeriods; ++i) {
237  startDate = schedule_.date(i);
238  endDate = schedule_.date(i + 1);
239  paymentDate = calendar.advance(endDate, paymentLag_, Days, paymentAdjustment_);
240 
241  Date fixingStartDate = Date();
242  Date fixingEndDate = Date();
243  if (valuationSchedule_.size() > 0) {
244  fixingStartDate = valuationSchedule_.date(i);
245  fixingEndDate = valuationSchedule_.date(i + 1);
246  }
247 
248  Real initialPrice = (i == 0) ? initialPrice_ : Null<Real>();
249  bool initialPriceIsInTargetCcy = initialPrice != Null<Real>() ? initialPriceIsInTargetCcy_ : false;
250  Real quantity = Null<Real>(), notional = Null<Real>();
251  if (notionalReset_) {
252  if (quantity_ != Null<Real>()) {
253  quantity = quantity_;
254  QL_REQUIRE(notionals_.empty(), "EquityLeg: notional and quantity are given at the same time");
255  } else {
256  QL_REQUIRE(fxIndex_ == nullptr,
257  "EquityLeg: can not compute quantity from nominal when fx conversion is required");
258  QL_REQUIRE(!notionals_.empty(), "EquityLeg: can not compute qunantity, since no notional is given");
259  QL_REQUIRE(initialPrice_ != Null<Real>(),
260  "EquityLeg: can not compute quantity, since no initialPrice is given");
261  quantity = (initialPrice_ == 0) ? notionals_.front() : notionals_.front() / initialPrice_;
262  }
263  } else {
264  if (!notionals_.empty()) {
265  notional = detail::get(notionals_, i, 0.0);
266  QL_REQUIRE(quantity_ == Null<Real>(), "EquityLeg: notional and quantity are given at the same time");
267  } else {
268  QL_REQUIRE(fxIndex_ == nullptr,
269  "EquityLeg: can not compute notional from quantity when fx conversion is required");
270  QL_REQUIRE(quantity_ != Null<Real>(),
271  "EquityLeg: can not compute notional, since no quantity is given");
272  QL_REQUIRE(initialPrice_ != Null<Real>(),
273  "EquityLeg: can not compute notional, since no intialPrice is given");
274  notional = (initialPrice_ == 0) ? quantity_ : quantity_ * initialPrice_;
275  }
276  }
277 
278  boost::shared_ptr<EquityCoupon> cashflow(
279  new EquityCoupon(paymentDate, notional, startDate, endDate, fixingDays_, equityCurve_, paymentDayCounter_,
280  isTotalReturn_, dividendFactor_, notionalReset_, initialPrice, quantity, fixingStartDate,
281  fixingEndDate, Date(), Date(), Date(), fxIndex_, initialPriceIsInTargetCcy, absoluteReturn_));
282 
283  boost::shared_ptr<EquityCouponPricer> pricer(new EquityCouponPricer);
284  cashflow->setPricer(pricer);
285 
286  cashflows.push_back(cashflow);
287  }
288  return cashflows;
289 }
