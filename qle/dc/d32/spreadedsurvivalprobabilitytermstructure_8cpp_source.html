23 SpreadedSurvivalProbabilityTermStructure::SpreadedSurvivalProbabilityTermStructure(
24  const Handle<DefaultProbabilityTermStructure>& referenceCurve, const std::vector<Time>& times,
25  const std::vector<Handle<Quote>>& spreads)
26  : SurvivalProbabilityStructure(referenceCurve->dayCounter()), referenceCurve_(referenceCurve), times_(times),
27  spreads_(spreads), data_(times.size(), 1.0) {
28  QL_REQUIRE(times_.size() > 1, "at least two times required");
29  QL_REQUIRE(times_.size() == spreads_.size(), "size of time and quote vectors do not match");
30  QL_REQUIRE(times_[0] == 0.0, "First time must be 0, got " << times_[0]);
31  for (Size i = 0; i < spreads_.size(); ++i) {
32  registerWith(spreads_[i]);
33  }
34  interpolation_ = boost::make_shared<LogLinearInterpolation>(times_.begin(), times_.end(), data_.begin());
35  interpolation_->enableExtrapolation();
36  registerWith(Settings::instance().evaluationDate());
37 }

44 void SpreadedSurvivalProbabilityTermStructure::performCalculations() const {
45  for (Size i = 0; i < times_.size(); ++i) {
46  QL_REQUIRE(!spreads_[i].empty(),
47  "SpreadedSurvivalProbabilityTermStructure: quote at index " << i << " is empty");
48  data_[i] = spreads_[i]->value();
49  QL_REQUIRE(data_[i] > 0,
50  "SpreadedSurvivalProbabilityTermStructure: invalid value " << data_[i] << " at index " << i);
51  }
52  interpolation_->update();
53 }

55 Probability SpreadedSurvivalProbabilityTermStructure::survivalProbabilityImpl(Time t) const {
56  calculate();
57  if (t <= this->times_.back())
58  return referenceCurve_->survivalProbability(t) * (*interpolation_)(t, true);
59  // flat fwd extrapolation
60  Real tMax = this->times_.back();
61  Real dMax = this->data_.back();
62  Real instFwdMax = -(*interpolation_).derivative(tMax) / dMax;
63  return referenceCurve_->survivalProbability(t) * dMax * std::exp(-instFwdMax * (t - tMax));
64 }
