59 Real BondIndex::fixing(const Date& fixingDate, bool forecastTodaysFixing) const {
61  QL_REQUIRE(isValidFixingDate(fixingDate), "Fixing date " << fixingDate << " is not valid for '" << name() << "'");
62  Date today = Settings::instance().evaluationDate();
63  if (fixingDate > today || (fixingDate == today && forecastTodaysFixing))
64  return forecastFixing(fixingDate);
65  if (fixingDate < today || Settings::instance().enforcesTodaysHistoricFixings()) {
66  // must have been fixed
67  // do not catch exceptions
68  Rate result = pastFixing(fixingDate);
69  QL_REQUIRE(result != Null<Real>(), "Missing " << name() << " fixing for " << fixingDate);
70  return result;
71  }
72  try {
73  // might have been fixed
74  Rate result = pastFixing(fixingDate);
75  if (result != Null<Real>())
76  return result;
77  else
78  ; // fall through and forecast
79  } catch (Error&) {
80  ; // fall through and forecast
81  }
82  return forecastFixing(fixingDate);
83 }

85 Rate BondIndex::forecastFixing(const Date& fixingDate) const {
86  Date today = Settings::instance().evaluationDate();
87  QL_REQUIRE(fixingDate >= today,
88  "BondIndex::forecastFixing(): fixingDate (" << fixingDate << ") must be >= today (" << today << ")");
89  QL_REQUIRE(bond_, "BondIndex::forecastFixing(): bond required");
90 
91  // on today, try to get the dirty absolute price from the bond itself
92 
93  Real price = Null<Real>();
94  if (fixingDate == today) {
95  try {
96  price = bond_->settlementValue();
97  } catch (...) {
98  }
99  }
100 
101  // for future dates or if the above did not work, assume that the bond can be priced by
102  // simply discounting its cashflows
103 
104  if (price == Null<Real>()) {
105  price = vanillaBondEngine_->calculateNpv(bond_->settlementDate(fixingDate), bond_->settlementDate(fixingDate),
106  bond_->cashflows(), boost::none, incomeCurve_, conditionalOnSurvival_);
107  }
108 
109  if (!dirty_) {
110  price -= bond_->accruedAmount(fixingDate) / 100.0 * bond_->notional(fixingDate);
111  }
112 
113  if (relative_) {
114  if (close_enough(bond_->notional(fixingDate), 0.0))
115  price = 0.0;
116  else
117  price /= bond_->notional(fixingDate);
118  }
119 
120  return price;
121 }

123 Real BondIndex::pastFixing(const Date& fixingDate) const {
124  QL_REQUIRE(isValidFixingDate(fixingDate), fixingDate << " is not a valid fixing date for '" << name() << "'");
125  Real price = timeSeries()[fixingDate];
126  if (price == Null<Real>())
127  return price;
128  if (dirty_) {
129  QL_REQUIRE(bond_, "BondIndex::pastFixing(): bond required for dirty prices");
130  price += bond_->accruedAmount(fixingDate) / 100.0;
131  }
132  if (!relative_) {
133  QL_REQUIRE(bond_, "BondIndex::pastFixing(): bond required for absolute prices");
134  price *= bond_->notional(fixingDate);
135  }
136  return price;
137 }
