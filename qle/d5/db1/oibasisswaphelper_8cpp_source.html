32 OIBSHelper::OIBSHelper(Natural settlementDays,
33  const Period& tenor, // swap maturity
34  const Handle<Quote>& oisSpread, const boost::shared_ptr<OvernightIndex>& overnightIndex,
35  const boost::shared_ptr<IborIndex>& iborIndex, const Handle<YieldTermStructure>& discount)
36  : RelativeDateRateHelper(oisSpread), settlementDays_(settlementDays), tenor_(tenor),
37  overnightIndex_(overnightIndex), iborIndex_(iborIndex), discount_(discount) {
38 
39  /* depending on the given curves we proceed as outlined in the following table
40 
41  x = curve is given
42  . = curve is missing
43 
44  Case | OIS | Ibor | Discount | Action
45  =========================================
46  0 | . | . | . | throw exception
47  1 | . | . | x | throw exception
48  2 | . | x | . | imply OIS = Discount
49  3 | . | x | x | imply OIS
50  4 | x | . | . | imply Ibor, set Discount = OIS
51  5 | x | . | x | imply Ibor
52  6 | x | x | . | imply Discount
53  7 | x | x | x | throw exception
54 
55  */
56 
57  bool oisGiven = !overnightIndex_->forwardingTermStructure().empty();
58  bool iborGiven = !iborIndex_->forwardingTermStructure().empty();
59  bool discountGiven = !discount_.empty();
60 
61  if (!oisGiven && !iborGiven && !discountGiven) {
62  // case 0
63  QL_FAIL("no curve given");
64  } else if (!oisGiven && !iborGiven && discountGiven) {
65  // case 1
66  QL_FAIL("neither OIS nor Ibor curve is given");
67  } else if (!oisGiven && iborGiven && !discountGiven) {
68  // case 2
69  overnightIndex_ = boost::static_pointer_cast<OvernightIndex>(overnightIndex_->clone(termStructureHandle_));
70  overnightIndex_->unregisterWith(termStructureHandle_);
71  discountRelinkableHandle_.linkTo(*termStructureHandle_, false);
72  } else if (!oisGiven && iborGiven && discountGiven) {
73  // case 3
74  overnightIndex_ = boost::static_pointer_cast<OvernightIndex>(overnightIndex_->clone(termStructureHandle_));
75  overnightIndex_->unregisterWith(termStructureHandle_);
76  } else if (oisGiven && !iborGiven && !discountGiven) {
77  // case 4
78  iborIndex_ = iborIndex_->clone(termStructureHandle_);
79  iborIndex_->unregisterWith(termStructureHandle_);
80  discountRelinkableHandle_.linkTo(*overnightIndex_->forwardingTermStructure());
81  } else if (oisGiven && !iborGiven && discountGiven) {
82  // case 5
83  iborIndex_ = iborIndex_->clone(termStructureHandle_);
84  iborIndex_->unregisterWith(termStructureHandle_);
85  } else if (oisGiven && iborGiven && !discountGiven) {
86  // case 6
87  discountRelinkableHandle_.linkTo(*termStructureHandle_, false);
88  } else if (oisGiven && iborGiven && discountGiven) {
89  // case 7
90  QL_FAIL("OIS, Ibor and Discount curves are all given");
91  }
92 
93  registerWith(overnightIndex_);
94  registerWith(iborIndex_);
95  registerWith(discount_);
96  initializeDates();
97 }

99 void OIBSHelper::initializeDates() {
100 
101  Date asof = Settings::instance().evaluationDate();
102  // if the evaluation date is not a business day
103  // then move to the next business day
104  asof = iborIndex_->fixingCalendar().adjust(asof);
105 
106  Date settlementDate = iborIndex_->fixingCalendar().advance(asof, settlementDays_, Days);
107  Schedule oisSchedule = MakeSchedule()
108  .from(settlementDate)
109  .to(settlementDate + tenor_)
110  .withTenor(1 * Years)
111  .withCalendar(overnightIndex_->fixingCalendar())
112  .withConvention(overnightIndex_->businessDayConvention())
113  .forwards();
114  Schedule iborSchedule = MakeSchedule()
115  .from(settlementDate)
116  .to(settlementDate + tenor_)
117  .withTenor(iborIndex_->tenor())
118  .withCalendar(iborIndex_->fixingCalendar())
119  .withConvention(iborIndex_->businessDayConvention())
120  .forwards();
121  swap_ = boost::shared_ptr<OvernightIndexedBasisSwap>(new OvernightIndexedBasisSwap(OvernightIndexedBasisSwap::Payer,
122  10000.0, // arbitrary
123  oisSchedule, overnightIndex_,
124  iborSchedule, iborIndex_));
125  boost::shared_ptr<PricingEngine> engine(
126  new DiscountingSwapEngine(discount_.empty() ? discountRelinkableHandle_ : discount_));
127  swap_->setPricingEngine(engine);
128 
129  earliestDate_ = swap_->startDate();
130  latestDate_ = swap_->maturityDate();
131 }
