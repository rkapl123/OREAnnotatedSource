53void OvernightIndexedBasisSwap::initialize() {
54 legs_[0] = IborLeg(iborSchedule_, iborIndex_).withNotionals(nominals_).withSpreads(iborSpread_);
55
56 legs_[1] = OvernightLeg(oisSchedule_, overnightIndex_)
57 .withNotionals(nominals_)
58 .withSpreads(oisSpread_)
59 .withTelescopicValueDates(telescopicValueDates_);
60
61 for (Size j = 0; j < 2; ++j) {
62 for (Leg::iterator i = legs_[j].begin(); i != legs_[j].end(); ++i)
63 registerWith(*i);
64 }
65
66 switch (type_) {
67 case Payer:
68 payer_[0] = -1.0;
69 payer_[1] = +1.0;
70 break;
71 case Receiver:
72 payer_[0] = +1.0;
73 payer_[1] = -1.0;
74 break;
75 default:
76 QL_FAIL("Unknown overnight-swap type");
77 }
78}
