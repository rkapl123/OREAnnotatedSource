51 void OvernightIndexedBasisSwap::initialize() {
52  legs_[0] = IborLeg(iborSchedule_, iborIndex_).withNotionals(nominals_).withSpreads(iborSpread_);
53 
54  legs_[1] = OvernightLeg(oisSchedule_, overnightIndex_).withNotionals(nominals_).withSpreads(oisSpread_);
55 
56  for (Size j = 0; j < 2; ++j) {
57  for (Leg::iterator i = legs_[j].begin(); i != legs_[j].end(); ++i)
58  registerWith(*i);
59  }
60 
61  switch (type_) {
62  case Payer:
63  payer_[0] = -1.0;
64  payer_[1] = +1.0;
65  break;
66  case Receiver:
67  payer_[0] = +1.0;
68  payer_[1] = -1.0;
69  break;
70  default:
71  QL_FAIL("Unknown overnight-swap type");
72  }
73 }
