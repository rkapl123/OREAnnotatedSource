73void AnalyticLgmSwaptionEngine::calculate() const {
74
75 QL_REQUIRE(arguments_.settlementType == Settlement::Physical, "cash-settled swaptions are not supported ...");
76
77 Date reference = p_->termStructure()->referenceDate();
78
79 Date expiry = arguments_.exercise->dates().back();
80
81 if (expiry <= reference) {
82 // swaption is expired, possibly generated swap is not
83 // valued by this engine, so we set the npv to zero
84 results_.value = 0.0;
85 return;
86 }
87
88 if (!caching_ || S_.empty()) {
89
90 Option::Type type = arguments_.type == VanillaSwap::Payer ? Option::Call : Option::Put;
91
92 QL_REQUIRE(
93 arguments_.swap || arguments_.swapOis,
94 "AnalyticalLgmSwaptionEngine::calculate(): internal error, expected either swap or swapOis to be set.");
95 const Schedule& fixedSchedule =
96 arguments_.swap ? arguments_.swap->fixedSchedule() : arguments_.swapOis->schedule();
97 const Schedule& floatSchedule =
98 arguments_.swap ? arguments_.swap->floatingSchedule() : arguments_.swapOis->schedule();
99
100 j1_ = std::lower_bound(fixedSchedule.dates().begin(), fixedSchedule.dates().end(), expiry) -
101 fixedSchedule.dates().begin();
102 k1_ = std::lower_bound(floatSchedule.dates().begin(), floatSchedule.dates().end(), expiry) -
103 floatSchedule.dates().begin();
104
105 nominal_ = arguments_.swap ? arguments_.swap->nominal() : arguments_.swapOis->nominal();
106
107 fixedLeg_.clear();
108 floatingLeg_.clear();
109 for(auto const& c: (arguments_.swap ? arguments_.swap->fixedLeg() : arguments_.swapOis->fixedLeg())) {
110 fixedLeg_.push_back(boost::dynamic_pointer_cast<FixedRateCoupon>(c));
111 QL_REQUIRE(fixedLeg_.back(),
112 "AnalyticalLgmSwaptionEngine::calculate(): internal error, could not cast to FixedRateCoupon");
113 }
114 for(auto const& c: (arguments_.swap ? arguments_.swap->floatingLeg() : arguments_.swapOis->overnightLeg())) {
115 floatingLeg_.push_back(boost::dynamic_pointer_cast<FloatingRateCoupon>(c));
116 QL_REQUIRE(
117 floatingLeg_.back(),
118 "AnalyticalLgmSwaptionEngine::calculate(): internal error, could not cast to FloatingRateRateCoupon");
119 }
120
121 // compute S_i, i.e. equivalent fixed rate spreads compensating for
122 // a) a possibly non-zero float spread and
123 // b) a spread between the ibor indices forwarding curve and the
124 // discounting curve
125 // here, we do not work with a spread corrections directly, but
126 // with this multiplied by the nominal and accrual basis,
127 // so S_i is really an amount correction.
128
129 S_.resize(fixedLeg_.size() - j1_);
130 for (Size i = 0; i < S_.size(); ++i) {
131 S_[i] = 0.0;
132 }
133 S_m1 = 0.0;
134 Size ratio =
135 static_cast<Size>(static_cast<Real>(floatingLeg_.size()) / static_cast<Real>(fixedLeg_.size()) + 0.5);
136 QL_REQUIRE(ratio >= 1, "floating leg's payment frequency must be equal or "
137 "higher than fixed leg's payment frequency in "
138 "analytic lgm swaption engine");
139
140 Size k = k1_;
141 // The method reduces the problem to a one curve configuration w.r.t. the discount curve and
142 // apply a correction for the discount curve / forwarding curve spread. Furthermore the method
143 // assumes that no historical fixings are present in the floating rate coupons.
144 boost::shared_ptr<IborIndex> index =
145 arguments_.swap ? arguments_.swap->iborIndex() : arguments_.swapOis->overnightIndex();
146 for (Size j = j1_; j < fixedLeg_.size(); ++j) {
147 Real sum1 = 0.0, sum2 = 0.0;
148 for (Size rr = 0; rr < ratio && k < floatingLeg_.size(); ++rr, ++k) {
149 Real amount = Null<Real>();
150 // same strategy as in VanillaSwap::setupArguments()
151 try {
152 amount = floatingLeg_[k]->amount();
153 } catch (...) {
154 }
155 Real lambda1 = 0.0, lambda2 = 1.0;
156 if (floatSpreadMapping_ == proRata) {
157 // we do not use the exact pay dates but the ratio to determine
158 // the distance to the adjacent payment dates
159 lambda2 = static_cast<Real>(rr + 1) / static_cast<Real>(ratio);
160 lambda1 = 1.0 - lambda2;
161 }
162 if (amount != Null<Real>()) {
163 Real flatAmount;
164 if(arguments_.swapOis) {
165 auto on = boost::dynamic_pointer_cast<QuantLib::OvernightIndexedCoupon>(floatingLeg_[k]);
166 QL_REQUIRE(on, "AnalyticalLgmSwaptionEngine::calculate(): internal error, could not cast to "
167 "QuantLib::OvernightIndexedCoupon.");
168 QL_REQUIRE(
169 !on->valueDates().empty(),
170 "AnalyticalLgmSwaptionEngine::calculate(): internal error, no value dates in ois coupon.");
171 Date v1 = std::max(reference, on->valueDates().front());
172 Date v2 = std::max(v1 + 1, on->valueDates().back());
173 Real rate;
174 if (on->averagingMethod() == QuantLib::RateAveraging::Compound)
175 rate =
176 (c_->discount(v1) / c_->discount(v2) - 1.0) / index->dayCounter().yearFraction(v1, v2);
177 else
178 rate = std::log(c_->discount(v1) / c_->discount(v2)) /
179 index->dayCounter().yearFraction(v1, v2);
180 flatAmount = floatingLeg_[k]->accrualPeriod() * nominal_ * rate;
181 } else {
182 if (IborCoupon::Settings::instance().usingAtParCoupons()) {
183 // if par coupons are used, we mimick the fixing estimation in IborCoupon; we make
184 // sure that the estimation period does not start in the past and we do not use
185 // historical fixings
186 Date fixingValueDate = index->fixingCalendar().advance(floatingLeg_[k]->fixingDate(),
187 index->fixingDays(), Days);
188 fixingValueDate = std::max(fixingValueDate, reference);
189 auto cpn = boost::dynamic_pointer_cast<Coupon>(floatingLeg_[k]);
190 QL_REQUIRE(cpn,
191 "AnalyticalLgmSwaptionEngine::calculate(): coupon expected on underlying swap "
192 "floating leg, could not cast");
193 Date nextFixingDate = index->fixingCalendar().advance(
194 cpn->accrualEndDate(), -static_cast<Integer>(index->fixingDays()), Days);
195 Date fixingEndDate =
196 index->fixingCalendar().advance(nextFixingDate, index->fixingDays(), Days);
197 fixingEndDate = std::max(fixingEndDate, fixingValueDate + 1);
198 Real spanningTime = index->dayCounter().yearFraction(fixingValueDate, fixingEndDate);
199 DiscountFactor disc1 = c_->discount(fixingValueDate);
200 DiscountFactor disc2 = c_->discount(fixingEndDate);
201 Real fixing = (disc1 / disc2 - 1.0) / spanningTime;
202 flatAmount = fixing * floatingLeg_[k]->accrualPeriod() * nominal_;
203 } else {
204 // if indexed coupons are used, we use a proper fixing, but make sure that the fixing
205 // date is not in the past and we do not use a historical fixing for "today"
206 auto flatIbor = boost::make_shared<IborIndex>(
207 index->familyName() + " (no fixings)", index->tenor(), index->fixingDays(),
208 index->currency(), index->fixingCalendar(), index->businessDayConvention(),
209 index->endOfMonth(), index->dayCounter(), c_);
210 Date fixingDate =
211 flatIbor->fixingCalendar().adjust(std::max(floatingLeg_[k]->fixingDate(), reference));
212 flatAmount = flatIbor->fixing(fixingDate) * floatingLeg_[k]->accrualPeriod() * nominal_;
213 }
214 }
215 Real correction = (amount - flatAmount) * c_->discount(floatingLeg_[k]->date());
216 sum1 += lambda1 * correction;
217 sum2 += lambda2 * correction;
218 } else {
219 // if no amount is given, we do not need a spread correction
220 // due to different forward / discounting curves since then
221 // no curve is attached to the swap's ibor index and so we
222 // assume a one curve setup;
223 // but we can still have a float spread that has to be converted
224 // into a fixed leg's payment
225 Real correction = nominal_ * floatingLeg_[k]->spread() * floatingLeg_[k]->accrualPeriod() *
226 c_->discount(floatingLeg_[k]->date());
227 sum1 += lambda1 * correction;
228 sum2 += lambda2 * correction;
229 }
230 }
231 if (j > j1_) {
232 S_[j - j1_ - 1] += sum1 / c_->discount(fixedLeg_[j - 1]->date());
233 } else {
234 S_m1 += sum1 / c_->discount(floatingLeg_[k1_]->accrualStartDate());
235 }
236 S_[j - j1_] += sum2 / c_->discount(fixedLeg_[j]->date());
237 }
238
239 w_ = type == Option::Call ? -1.0 : 1.0;
240 D0_ = c_->discount(floatingLeg_[k1_]->accrualStartDate());
241 Dj_.resize(fixedLeg_.size() - j1_);
242 for (Size j = j1_; j < fixedLeg_.size(); ++j) {
243 Dj_[j - j1_] = c_->discount(fixedLeg_[j - j1_]->date());
244 }
245 }
246
247 if (!caching_ || !lgm_H_constant_ || Hj_.empty()) {
248 // it is a requirement that H' does not change its sign,
249 // with u = -1.0 we handle the case H' < 0
250 u_ = p_->Hprime(0.0) > 0.0 ? 1.0 : -1.0;
251
252 H0_ = p_->H(p_->termStructure()->timeFromReference(floatingLeg_[k1_]->accrualStartDate()));
253 Hj_.resize(fixedLeg_.size() - j1_);
254 for (Size j = j1_; j < fixedLeg_.size(); ++j) {
255 Hj_[j - j1_] = p_->H(p_->termStructure()->timeFromReference(fixedLeg_[j]->date()));
256 }
257 }
258
259 if (!caching_ || !lgm_alpha_constant_ || zetaex_ == Null<Real>()) {
260 zetaex_ = p_->zeta(p_->termStructure()->timeFromReference(expiry));
261 }
262
263 Brent b;
264 Real yStar;
265 try {
266 yStar = b.solve(boost::bind(&AnalyticLgmSwaptionEngine::yStarHelper, this, _1), 1.0E-6, 0.0, 0.01);
267 } catch (const std::exception& e) {
268 std::ostringstream os;
269 os << "AnalyticLgmSwaptionEngine: failed to compute yStar (" << e.what() << "), parameter details: [";
270 Real tte = p_->termStructure()->timeFromReference(expiry);
271 os << "tte=" << tte << ", vol=" << std::sqrt(zetaex_ / tte) << ", nominal=" << nominal_
272 << ", d=" << D0_;
273 for (Size j = 0; j < Dj_.size(); ++j)
274 os << ", d" << j << "=" << Dj_[j];
275 os << ", h=" << H0_;
276 for (Size j = 0; j < Hj_.size(); ++j)
277 os << ", h" << j << "=" << Hj_[j];
278 for (Size i = j1_; i < fixedLeg_.size(); ++i) {
279 os << ", cpn" << i << "=(" << QuantLib::io::iso_date(fixedLeg_[i]->accrualStartDate()) << ","
280 << QuantLib::io::iso_date(fixedLeg_[i]->date()) << "," << fixedLeg_[i]->amount() << ")";
281 }
282 os << ", S=" << S_m1;
283 for (Size j = 0; j < S_.size(); ++j) {
284 os << ", S" << j << "=" << S_[j];
285 }
286 os << "]";
287 QL_FAIL(os.str());
288 }
289
290 CumulativeNormalDistribution N;
291 Real sqrt_zetaex = std::sqrt(zetaex_);
292 Real sum = 0.0;
293 for (Size j = j1_; j < fixedLeg_.size(); ++j) {
294 sum += w_ * (fixedLeg_[j]->amount() - S_[j - j1_]) * Dj_[j - j1_] *
295 N(u_ * w_ * (yStar + (Hj_[j - j1_] - H0_) * zetaex_) / sqrt_zetaex);
296 }
297 sum += -w_ * S_m1 * D0_ * N(u_ * w_ * yStar / sqrt_zetaex);
298 sum += w_ * (nominal_ * Dj_.back() * N(u_ * w_ * (yStar + (Hj_.back() - H0_) * zetaex_) / sqrt_zetaex) -
299 nominal_ * D0_ * N(u_ * w_ * yStar / sqrt_zetaex));
300 results_.value = sum;
301
302 results_.additionalResults["fixedAmountCorrectionSettlement"] = S_m1;
303 results_.additionalResults["fixedAmountCorrections"] = S_;
304
305} // calculate

307Real AnalyticLgmSwaptionEngine::yStarHelper(const Real y) const {
308 Real sum = 0.0;
309 for (Size j = j1_; j < fixedLeg_.size(); ++j) {
310 sum += (fixedLeg_[j]->amount() - S_[j - j1_]) * Dj_[j - j1_] *
311 std::exp(-(Hj_[j - j1_] - H0_) * y - 0.5 * (Hj_[j - j1_] - H0_) * (Hj_[j - j1_] - H0_) * zetaex_);
312 }
313 sum += -S_m1 * D0_;
314 sum += Dj_.back() * nominal_ *
315 std::exp(-(Hj_.back() - H0_) * y - 0.5 * (Hj_.back() - H0_) * (Hj_.back() - H0_) * zetaex_);
316 sum -= D0_ * nominal_;
317 return sum;
318}
