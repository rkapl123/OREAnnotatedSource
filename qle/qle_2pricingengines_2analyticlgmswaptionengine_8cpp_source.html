72 void AnalyticLgmSwaptionEngine::calculate() const {
73 
74  QL_REQUIRE(arguments_.settlementType == Settlement::Physical, "cash-settled swaptions are not supported ...");
75 
76  Date reference = p_->termStructure()->referenceDate();
77 
78  Date expiry = arguments_.exercise->dates().back();
79 
80  if (expiry <= reference) {
81  // swaption is expired, possibly generated swap is not
82  // valued by this engine, so we set the npv to zero
83  results_.value = 0.0;
84  return;
85  }
86 
87  if (!caching_ || S_.empty()) {
88 
89  Option::Type type = arguments_.type == VanillaSwap::Payer ? Option::Call : Option::Put;
90  const Schedule& fixedSchedule = arguments_.swap->fixedSchedule();
91  const Schedule& floatSchedule = arguments_.swap->floatingSchedule();
92 
93  j1_ = std::lower_bound(fixedSchedule.dates().begin(), fixedSchedule.dates().end(), expiry) -
94  fixedSchedule.dates().begin();
95  k1_ = std::lower_bound(floatSchedule.dates().begin(), floatSchedule.dates().end(), expiry) -
96  floatSchedule.dates().begin();
97 
98  // compute S_i, i.e. equivalent fixed rate spreads compensating for
99  // a) a possibly non-zero float spread and
100  // b) a spread between the ibor indices forwarding curve and the
101  // discounting curve
102  // here, we do not work with a spread corrections directly, but
103  // with this multiplied by the nominal and accrual basis,
104  // so S_i is really an amount correction.
105 
106  S_.resize(arguments_.fixedCoupons.size() - j1_);
107  for (Size i = 0; i < S_.size(); ++i) {
108  S_[i] = 0.0;
109  }
110  S_m1 = 0.0;
111  Size ratio = static_cast<Size>(static_cast<Real>(arguments_.floatingCoupons.size()) /
112  static_cast<Real>(arguments_.fixedCoupons.size()) +
113  0.5);
114  QL_REQUIRE(ratio >= 1, "floating leg's payment frequency must be equal or "
115  "higher than fixed leg's payment frequency in "
116  "analytic lgm swaption engine");
117 
118  Size k = k1_;
119  // The method reduces the problem to a one curve configuration w.r.t. the discount curve and
120  // apply a correction for the discount curve / forwarding curve spread. Furthermore the method
121  // assumes that no historical fixings are present in the floating rate coupons.
122  auto index = arguments_.swap->iborIndex();
123  for (Size j = j1_; j < arguments_.fixedCoupons.size(); ++j) {
124  Real sum1 = 0.0, sum2 = 0.0;
125  for (Size rr = 0; rr < ratio && k < arguments_.floatingCoupons.size(); ++rr, ++k) {
126  Real amount = arguments_.floatingCoupons[k];
127  Real lambda1 = 0.0, lambda2 = 1.0;
128  if (floatSpreadMapping_ == proRata) {
129  // we do not use the exact pay dates but the ratio to determine
130  // the distance to the adjacent payment dates
131  lambda2 = static_cast<Real>(rr + 1) / static_cast<Real>(ratio);
132  lambda1 = 1.0 - lambda2;
133  }
134  if (amount != Null<Real>()) {
135  Real flatAmount;
136  if (IborCoupon::usingAtParCoupons()) {
137  // if par coupons are used, we mimick the fixing estimation in IborCoupon; we make
138  // sure that the estimation period does not start in the past and we do not use
139  // historical fixings
140  Date fixingValueDate = index->fixingCalendar().advance(arguments_.floatingFixingDates[k],
141  index->fixingDays(), Days);
142  fixingValueDate = std::max(fixingValueDate, reference);
143  auto cpn = boost::dynamic_pointer_cast<Coupon>(arguments_.swap->floatingLeg()[k]);
144  QL_REQUIRE(cpn, "AnalyticalLgmSwaptionEngine::calculate(): coupon expected on underlying swap "
145  "floating leg, could not cast");
146  Date nextFixingDate = index->fixingCalendar().advance(
147  cpn->accrualEndDate(), -static_cast<Integer>(index->fixingDays()), Days);
148  Date fixingEndDate = index->fixingCalendar().advance(nextFixingDate, index->fixingDays(), Days);
149  fixingEndDate = std::max(fixingEndDate, fixingValueDate + 1);
150  Real spanningTime = index->dayCounter().yearFraction(fixingValueDate, fixingEndDate);
151  DiscountFactor disc1 = c_->discount(fixingValueDate);
152  DiscountFactor disc2 = c_->discount(fixingEndDate);
153  Real fixing = (disc1 / disc2 - 1.0) / spanningTime;
154  flatAmount = fixing * arguments_.floatingAccrualTimes[k] * arguments_.nominal;
155  } else {
156  // if indexed coupons are used, we use a proper fixing, but make sure that the fixing date
157  // is not in the past and we do not use a historical fixing for "today"
158  auto flatIbor = boost::make_shared<IborIndex>(
159  index->familyName() + " (no fixings)", index->tenor(), index->fixingDays(),
160  index->currency(), index->fixingCalendar(), index->businessDayConvention(),
161  index->endOfMonth(), index->dayCounter(), c_);
162  Date fixingDate =
163  flatIbor->fixingCalendar().adjust(std::max(arguments_.floatingFixingDates[k], reference));
164  flatAmount =
165  flatIbor->fixing(fixingDate) * arguments_.floatingAccrualTimes[k] * arguments_.nominal;
166  }
167  Real correction = (amount - flatAmount) * c_->discount(arguments_.floatingPayDates[k]);
168  sum1 += lambda1 * correction;
169  sum2 += lambda2 * correction;
170  } else {
171  // if no amount is given, we do not need a spread correction
172  // due to different forward / discounting curves since then
173  // no curve is attached to the swap's ibor index and so we
174  // assume a one curve setup;
175  // but we can still have a float spread that has to be converted
176  // into a fixed leg's payment
177  Real correction = arguments_.nominal * arguments_.floatingSpreads[k] *
178  arguments_.floatingAccrualTimes[k] * c_->discount(arguments_.floatingPayDates[k]);
179  sum1 += lambda1 * correction;
180  sum2 += lambda2 * correction;
181  }
182  }
183  if (j > j1_) {
184  S_[j - j1_ - 1] += sum1 / c_->discount(arguments_.fixedPayDates[j - 1]);
185  } else {
186  S_m1 += sum1 / c_->discount(arguments_.floatingResetDates[k1_]);
187  }
188  S_[j - j1_] += sum2 / c_->discount(arguments_.fixedPayDates[j]);
189  }
190 
191  w_ = type == Option::Call ? -1.0 : 1.0;
192  D0_ = c_->discount(arguments_.floatingResetDates[k1_]);
193  Dj_.resize(arguments_.fixedCoupons.size() - j1_);
194  for (Size j = j1_; j < arguments_.fixedCoupons.size(); ++j) {
195  Dj_[j - j1_] = c_->discount(arguments_.fixedPayDates[j - j1_]);
196  }
197  }
198 
199  if (!caching_ || !lgm_H_constant_ || Hj_.empty()) {
200  // it is a requirement that H' does not change its sign,
201  // with u = -1.0 we handle the case H' < 0
202  u_ = p_->Hprime(0.0) > 0.0 ? 1.0 : -1.0;
203 
204  H0_ = p_->H(p_->termStructure()->timeFromReference(arguments_.floatingResetDates[k1_]));
205  Hj_.resize(arguments_.fixedCoupons.size() - j1_);
206  for (Size j = j1_; j < arguments_.fixedCoupons.size(); ++j) {
207  Hj_[j - j1_] = p_->H(p_->termStructure()->timeFromReference(arguments_.fixedPayDates[j]));
208  }
209  }
210 
211  if (!caching_ || !lgm_alpha_constant_ || zetaex_ == Null<Real>()) {
212  zetaex_ = p_->zeta(p_->termStructure()->timeFromReference(expiry));
213  }
214 
215  Brent b;
216  Real yStar;
217  try {
218  yStar = b.solve(boost::bind(&AnalyticLgmSwaptionEngine::yStarHelper, this, _1), 1.0E-6, 0.0, 0.01);
219  } catch (const std::exception& e) {
220  std::ostringstream os;
221  os << "AnalyticLgmSwaptionEngine: failed to compute yStar (" << e.what() << "), parameter details: [";
222  Real tte = p_->termStructure()->timeFromReference(expiry);
223  os << "tte=" << tte << ", vol=" << std::sqrt(zetaex_ / tte) << ", nominal=" << arguments_.nominal
224  << ", d=" << D0_;
225  for (Size j = 0; j < Dj_.size(); ++j)
226  os << ", d" << j << "=" << Dj_[j];
227  os << ", h=" << H0_;
228  for (Size j = 0; j < Hj_.size(); ++j)
229  os << ", h" << j << "=" << Hj_[j];
230  for (Size i = j1_; i < arguments_.fixedCoupons.size(); ++i) {
231  os << ", cpn" << i << "=(" << QuantLib::io::iso_date(arguments_.fixedResetDates[i]) << ","
232  << QuantLib::io::iso_date(arguments_.fixedPayDates[i]) << "," << arguments_.fixedCoupons[i] << ")";
233  }
234  os << ", S=" << S_m1;
235  for (Size j = 0; j < S_.size(); ++j) {
236  os << ", S" << j << "=" << S_[j];
237  }
238  os << "]";
239  QL_FAIL(os.str());
240  }
241 
242  CumulativeNormalDistribution N;
243  Real sqrt_zetaex = std::sqrt(zetaex_);
244  Real sum = 0.0;
245  for (Size j = j1_; j < arguments_.fixedCoupons.size(); ++j) {
246  sum += w_ * (arguments_.fixedCoupons[j] - S_[j - j1_]) * Dj_[j - j1_] *
247  N(u_ * w_ * (yStar + (Hj_[j - j1_] - H0_) * zetaex_) / sqrt_zetaex);
248  }
249  sum += -w_ * S_m1 * D0_ * N(u_ * w_ * yStar / sqrt_zetaex);
250  sum += w_ * (arguments_.nominal * Dj_.back() * N(u_ * w_ * (yStar + (Hj_.back() - H0_) * zetaex_) / sqrt_zetaex) -
251  arguments_.nominal * D0_ * N(u_ * w_ * yStar / sqrt_zetaex));
252  results_.value = sum;
253 
254  results_.additionalResults["fixedAmountCorrectionSettlement"] = S_m1;
255  results_.additionalResults["fixedAmountCorrections"] = S_;
256 
257 } // calculate

259 Real AnalyticLgmSwaptionEngine::yStarHelper(const Real y) const {
260  Real sum = 0.0;
261  for (Size j = j1_; j < arguments_.fixedCoupons.size(); ++j) {
262  sum += (arguments_.fixedCoupons[j] - S_[j - j1_]) * Dj_[j - j1_] *
263  std::exp(-(Hj_[j - j1_] - H0_) * y - 0.5 * (Hj_[j - j1_] - H0_) * (Hj_[j - j1_] - H0_) * zetaex_);
264  }
265  sum += -S_m1 * D0_;
266  sum += Dj_.back() * arguments_.nominal *
267  std::exp(-(Hj_.back() - H0_) * y - 0.5 * (Hj_.back() - H0_) * (Hj_.back() - H0_) * zetaex_);
268  sum -= D0_ * arguments_.nominal;
269  return sum;
270 }
