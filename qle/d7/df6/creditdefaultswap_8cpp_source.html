114 void CreditDefaultSwap::init(const Schedule& schedule, BusinessDayConvention paymentConvention,
115  const DayCounter& dayCounter, const DayCounter& lastPeriodDayCounter, const Date& upfrontDate) {
116 
117  QL_REQUIRE(!schedule.empty(), "CreditDefaultSwap needs a non-empty schedule.");
118 
119  bool postBigBang = false;
120  if (schedule.hasRule()) {
121  DateGeneration::Rule rule = schedule.rule();
122  postBigBang = rule == DateGeneration::CDS || rule == DateGeneration::CDS2015;
123  }
124 
125  if (!postBigBang) {
126  QL_REQUIRE(protectionStart_ <= schedule[0], "CreditDefaultSwap: protection can not start after accrual");
127  }
128 
129  // If the leg_ has not already been populated via amortised leg ctor, populate it.
130  if (leg_.empty()) {
131  leg_ = FixedRateLeg(schedule)
132  .withNotionals(notional_)
133  .withCouponRates(runningSpread_, dayCounter)
134  .withPaymentAdjustment(paymentConvention)
135  .withLastPeriodDayCounter(lastPeriodDayCounter);
136  }
137 
138  // Deduce the trade date if not given.
139  if (tradeDate_ == Date()) {
140  if (postBigBang) {
141  tradeDate_ = protectionStart_;
142  } else {
143  tradeDate_ = protectionStart_ - 1;
144  }
145  }
146 
147  // Deduce the cash settlement date if not given.
148  Date effectiveUpfrontDate = upfrontDate;
149  if (effectiveUpfrontDate == Date()) {
150  effectiveUpfrontDate = schedule.calendar().advance(tradeDate_,
151  cashSettlementDays_, Days, paymentConvention);
152  }
153  QL_REQUIRE(effectiveUpfrontDate >= protectionStart_, "The cash settlement date must not " <<
154  "be before the protection start date.");
155 
156  // Create the upfront payment. Should always be created as some downstream engines don't expect nullptr.
157  Real upfrontAmount = 0.0;
158  if (upfront_)
159  upfrontAmount = *upfront_ * notional_;
160  upfrontPayment_ = boost::make_shared<SimpleCashFlow>(upfrontAmount, effectiveUpfrontDate);
161 
162  // Set the maturity date.
163  maturity_ = schedule.dates().back();
164 
165  // Deal with the accrual rebate. We use the standard conventions for accrual calculation introduced with the
166  // CDS Big Bang in 2009.
167  if (postBigBang) {
168 
169  Real rebateAmount = 0.0;
170  Date refDate = tradeDate_ + 1;
171 
172  if (tradeDate_ >= schedule.dates().front()) {
173  for (Size i = 0; i < leg_.size(); ++i) {
174  const boost::shared_ptr<CashFlow>& cf = leg_[i];
175  if (refDate < cf->date()) {
176  // Calculate the accrual. The most likely scenario.
177  boost::shared_ptr<FixedRateCoupon> frc = boost::dynamic_pointer_cast<FixedRateCoupon>(cf);
178  rebateAmount = frc->accruedAmount(refDate);
179  break;
180  } else if (refDate == cf->date() && i < leg_.size() - 1) {
181  // If not the last coupon and trade date + 1 is the next coupon payment date,
182  // the accrual is 0 so do nothing.
183  break;
184  } else {
185  // Must have trade date + 1 >= last coupon's payment date. '>' here probably does not make
186  // sense - should possibly have an exception above if trade date >= last coupon's date.
187  boost::shared_ptr<FixedRateCoupon> frc = boost::dynamic_pointer_cast<FixedRateCoupon>(cf);
188  rebateAmount = frc->amount();
189  break;
190  }
191  }
192  }
193 
194  accrualRebate_ = boost::make_shared<SimpleCashFlow>(rebateAmount, effectiveUpfrontDate);
195  }
196 
197  if (!claim_)
198  claim_ = boost::make_shared<FaceValueClaim>();
199  registerWith(claim_);
200 }

320  : target_(target), quote_(quote), engine_(engine), results_(results) {}

342 Rate CreditDefaultSwap::impliedHazardRate(Real targetNPV, const Handle<YieldTermStructure>& discountCurve,
343  const DayCounter& dayCounter, Real recoveryRate, Real accuracy) const {
344 
345  boost::shared_ptr<SimpleQuote> flatRate(new SimpleQuote(0.0));
346 
347  Handle<DefaultProbabilityTermStructure> probability(boost::shared_ptr<DefaultProbabilityTermStructure>(
348  new FlatHazardRate(0, WeekendsOnly(), Handle<Quote>(flatRate), dayCounter)));
349 
350  boost::shared_ptr<PricingEngine> engine = buildPricingEngine(probability, recoveryRate, discountCurve);
351  setupArguments(engine->getArguments());
352  const CreditDefaultSwap::results* results = dynamic_cast<const CreditDefaultSwap::results*>(engine->getResults());
353 
354  ObjectiveFunction f(targetNPV, *flatRate, *engine, results);
355  Rate guess = 0.001;
356  Rate step = guess * 0.1;
357 
358  return Brent().solve(f, accuracy, guess, step);
359 }

361 Rate CreditDefaultSwap::conventionalSpread(Real conventionalRecovery, const Handle<YieldTermStructure>& discountCurve,
362  const DayCounter& dayCounter) const {
363  Rate flatHazardRate = impliedHazardRate(0.0, discountCurve, dayCounter, conventionalRecovery);
364 
365  Handle<DefaultProbabilityTermStructure> probability(boost::shared_ptr<DefaultProbabilityTermStructure>(
366  new FlatHazardRate(0, WeekendsOnly(), flatHazardRate, dayCounter)));
367 
368  boost::shared_ptr<PricingEngine> engine = buildPricingEngine(probability, conventionalRecovery, discountCurve);
369  setupArguments(engine->getArguments());
370  engine->calculate();
371  const CreditDefaultSwap::results* results = dynamic_cast<const CreditDefaultSwap::results*>(engine->getResults());
372  return results->fairSpread;
373 }
