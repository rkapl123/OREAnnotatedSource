46 void CrossCcyBasisMtMResetSwap::initialize() {
47  // Pay (foreign) leg
48  legs_[0] = IborLeg(foreignSchedule_, foreignIndex_).withNotionals(foreignNominal_).withSpreads(foreignSpread_);
49  receiveDomestic_ ? payer_[0] = -1.0 : payer_[0] = +1.0;
50 
51  currencies_[0] = foreignCurrency_;
52  // Pay leg notional exchange at start.
53  Date initialPayDate = foreignSchedule_.dates().front();
54  boost::shared_ptr<CashFlow> initialPayCF(new SimpleCashFlow(-foreignNominal_, initialPayDate));
55  legs_[0].insert(legs_[0].begin(), initialPayCF);
56  // Pay leg notional exchange at end.
57  Date finalPayDate = foreignSchedule_.dates().back();
58  boost::shared_ptr<CashFlow> finalPayCF(new SimpleCashFlow(foreignNominal_, finalPayDate));
59  legs_[0].push_back(finalPayCF);
60 
61  // Receive (domestic/resettable) leg
62  // start by creating a dummy vanilla floating leg
63  legs_[1] = IborLeg(domesticSchedule_, domesticIndex_).withNotionals(0).withSpreads(domesticSpread_);
64  receiveDomestic_ ? payer_[1] = +1.0 : payer_[1] = -1.0;
65  currencies_[1] = domesticCurrency_;
66  // replace the coupons with a FloatingRateFXLinkedNotionalCoupon
67  // (skip the first coupon as it has a fixed notional)
68  for (Size j = 0; j < legs_[1].size(); ++j) {
69  boost::shared_ptr<FloatingRateCoupon> coupon = boost::dynamic_pointer_cast<FloatingRateCoupon>(legs_[1][j]);
70  Date fixingDate = fxIndex_->fixingCalendar().advance(coupon->accrualStartDate(),
71  -static_cast<Integer>(fxIndex_->fixingDays()), Days);
72  boost::shared_ptr<FloatingRateFXLinkedNotionalCoupon> fxLinkedCoupon(
73  new FloatingRateFXLinkedNotionalCoupon(fixingDate, foreignNominal_, fxIndex_, coupon));
74  legs_[1][j] = fxLinkedCoupon;
75  }
76  // now build a separate leg to store the domestic (resetting) notionals
77  receiveDomestic_ ? payer_[2] = +1.0 : payer_[2] = -1.0;
78  currencies_[2] = domesticCurrency_;
79  for (Size j = 0; j < legs_[1].size(); j++) {
80  boost::shared_ptr<Coupon> c = boost::dynamic_pointer_cast<Coupon>(legs_[1][j]);
81  QL_REQUIRE(c, "Resetting XCCY - expected Coupon"); // TODO: fixed fx resetable?
82  // build a pair of notional flows, one at the start and one at the end of
83  // the accrual period. Both with the same FX fixing date
84  Date fixingDate = fxIndex_->fixingCalendar().advance(c->accrualStartDate(),
85  -static_cast<Integer>(fxIndex_->fixingDays()), Days);
86  legs_[2].push_back(boost::shared_ptr<CashFlow>(
87  new FXLinkedCashFlow(c->accrualStartDate(), fixingDate, -foreignNominal_, fxIndex_)));
88  legs_[2].push_back(boost::shared_ptr<CashFlow>(
89  new FXLinkedCashFlow(c->accrualEndDate(), fixingDate, foreignNominal_, fxIndex_)));
90  }
91 
92  // Register the instrument with all cashflows on each leg.
93  for (Size legNo = 0; legNo < legs_.size(); legNo++) {
94  Leg::iterator it;
95  for (it = legs_[legNo].begin(); it != legs_[legNo].end(); ++it) {
96  registerWith(*it);
97  }
98  }
99 }

116 void CrossCcyBasisMtMResetSwap::fetchResults(const PricingEngine::results* r) const {
117 
118  CrossCcySwap::fetchResults(r);
119 
120  const CrossCcyBasisMtMResetSwap::results* results = dynamic_cast<const CrossCcyBasisMtMResetSwap::results*>(r);
121  if (results) {
122  /* If PricingEngine::results are of type
123  CrossCcyBasisSwap::results */
124  fairForeignSpread_ = results->fairForeignSpread;
125  fairDomesticSpread_ = results->fairDomesticSpread;
126  } else {
127  /* If not, e.g. if the engine is a CrossCcySwap::engine */
128  fairForeignSpread_ = Null<Spread>();
129  fairDomesticSpread_ = Null<Spread>();
130  }
131 
132  /* Calculate the fair pay and receive spreads if they are null */
133  static Spread basisPoint = 1.0e-4;
134  if (fairForeignSpread_ == Null<Spread>()) {
135  if (legBPS_[0] != Null<Real>())
136  fairForeignSpread_ = foreignSpread_ - NPV_ / (legBPS_[0] / basisPoint);
137  }
138  if (fairDomesticSpread_ == Null<Spread>()) {
139  if (legBPS_[1] != Null<Real>())
140  fairDomesticSpread_ = domesticSpread_ - NPV_ / (legBPS_[1] / basisPoint);
141  }
142 }
