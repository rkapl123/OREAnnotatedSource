53 template <class TS>
54 CirppConstantWithFellerParametrization<TS>::CirppConstantWithFellerParametrization(
55  const Currency& currency, const Handle<TS>& termStructure, const Real kappa, const Real theta, const Real sigma,
56  const Real y0, const bool shifted, bool relaxed, const Real fellerFactor, const std::string& name)
57  : CirppParametrization<TS>(currency, termStructure, shifted, name),
58  kappa_(boost::make_shared<PseudoParameter>(1)),
59  theta_(boost::make_shared<PseudoParameter>(1)), sigma_(boost::make_shared<PseudoParameter>(1)),
60  y0_(boost::make_shared<PseudoParameter>(1)), relaxed_(relaxed), fellerFactor_(fellerFactor) {
61  QL_REQUIRE((relaxed_ ? 4.0 : 2.0) * kappa * theta > sigma * sigma,
62  "CirppConstantWithFellerParametrization: Feller constraint violated (kappa="
63  << kappa << ", theta=" << theta << ", sigma=" << sigma << " (relaxed=" << std::boolalpha << relaxed_
64  << ")");
65  QL_REQUIRE(fellerFactor_ > 1.0, "CirppConstantWithFellerParametrization: Feller factor ("
66  << fellerFactor_ << ") should be greater than 1.0");
67  kappa_->setParam(0, inverse(0, kappa));
68  theta_->setParam(0, inverse(1, theta));
69  sigma_->setParam(0, inverse(2, sigma));
70  y0_->setParam(0, inverse(3, y0));

74 inline Real CirppConstantWithFellerParametrization<TS>::direct(const Size i, const Real x) const {
75  constexpr Real eps = 1E-10;
76  switch (i) {
77  case 0:
78  case 1:
79  case 3:
80  return x * x + eps;
81  case 2: {
82  const Real fellerBound =
83  std::sqrt((relaxed_ ? 4.0 : 2.0) * direct(0, kappa_->params()[0]) * direct(1, theta_->params()[0]));
84  // return fellerBound * (std::atan(x) + M_PI_2 + eps) / (M_PI - 2.0 * eps); // "free sigma"
85  return fellerBound / std::sqrt(fellerFactor_); // tie sigma to kappa/theta on feller boundary
86  }
87  default:
88  QL_FAIL("Index is not defined!");
89  }
90 }

93 inline Real CirppConstantWithFellerParametrization<TS>::inverse(const Size i, const Real y) const {
94  constexpr Real eps = 1E-10;
95  switch (i) {
96  case 0:
97  case 1:
98  case 3:
99  return sqrt(y - eps);
100  case 2: {
101  const Real fellerBound =
102  std::sqrt((relaxed_ ? 4.0 : 2.0) * direct(0, kappa_->params()[0]) * direct(1, theta_->params()[0]));
103  // return std::tan(y / fellerBound * (M_PI - 2.0 * eps) - M_PI_2 - eps); // see above
104  return fellerBound / std::sqrt(fellerFactor_); // see above
105  }
106  default:
107  QL_FAIL("Index is not defined!");
108  }
