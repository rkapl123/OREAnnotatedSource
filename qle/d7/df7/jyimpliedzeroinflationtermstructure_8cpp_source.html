33 Real JyImpliedZeroInflationTermStructure::zeroRateImpl(Time t) const {
34 
35  QL_REQUIRE(t >= 0.0, "JyImpliedZeroInflationTermStructure::zeroRateImpl: negative time (" << t << ") given");
36 
37  // Zero rate is calculated from the relation: P_n(S, T) (1 + z(S))^t = P_r(S, T)
38  // Here, S in the relation is given by relativeTime_ and T := S + t.
39  // ratio holds \frac{P_r(S, T)}{P_n(S, T)}.
40  auto S = relativeTime_;
41  auto T = relativeTime_ + t;
42  auto ratio = inflationGrowth(model_, index_, S, T, state_[2], state_[0]);
43 
44  // Return the desired z(S) = \left( \frac{P_r(S, T)}{P_n(S, T)} \right)^{\frac{1}{t}} - 1
45  return std::pow(ratio, 1 / t) - 1;
46 }

54 Real inflationGrowth(const boost::shared_ptr<CrossAssetModel>& model, Size index,
55  Time S, Time T, Real irState, Real rrState) {
56 
57  QL_REQUIRE(T >= S, "inflationGrowth: end time (" << T << ") must be >= start time (" << S << ")");
58 
59  // After this step, p_n holds P_n(S, T) * P_n(0, S) / P_n(0, T)
60  // = \exp \left( - \left[ H_n(T) - H_n(S) \right] z_n(S) -
61  // \frac{1}{2} \left[ H^2_n(T) - H^2_n(S) \right] \zeta_n(S) \right)
62  auto irIdx = model->ccyIndex(model->infjy(index)->currency());
63  auto irTs = model->irlgm1f(irIdx)->termStructure();
64  auto p_n = model->discountBond(irIdx, S, T, irState) * irTs->discount(S) / irTs->discount(T);
65 
66  // After this step, p_r holds P_r(S, T) * P_r(0, S) / P_r(0, T)
67  // = \exp \left( - \left[ H_r(T) - H_r(S) \right] z_r(S) -
68  // \frac{1}{2} \left[ H^2_r(T) - H^2_r(S) \right] \zeta_r(S) \right)
69  auto rrParam = model->infjy(index)->realRate();
70  auto H_r_S = rrParam->H(S);
71  auto H_r_T = rrParam->H(T);
72  auto zeta_r_S = rrParam->zeta(S);
73  auto p_r = std::exp(-(H_r_T - H_r_S) * rrState - 0.5 * (H_r_T * H_r_T - H_r_S * H_r_S) * zeta_r_S);
74 
75  // Now, use the original zero inflation term structure to get P_r(0, S) / P_n(0, S) and P_r(0, T) / P_n(0, T) and
76  // return \frac{P_r(S, T)}{P_n(S, T)}
77  const auto& zts = model->infjy(index)->realRate()->termStructure();
78  return inflationGrowth(zts, T) / inflationGrowth(zts, S) * p_r / p_n;
79 
80 }
