54 void BlackCdsOptionEngine::calculate() const {
55 
56  QL_REQUIRE(arguments_.strikeType == CdsOption::Spread, "BlackCdsOptionEngine does not support valuation" <<
57  " of single name options quoted in terms of strike price.");
58 
59  // Reference to the underlying forward starting CDS, from expiry date $t_E$ to maturity $T$.
60  const auto& cds = *arguments_.swap;
61 
62  // Get the additional results of the underlying CDS (need to call NPV first for calculation).
63  cds.NPV();
64  results_.additionalResults = cds.additionalResults();
65 
66  // Add some entries to additional results.
67  Real forward = cds.fairSpread();
68  results_.additionalResults["forwardSpread"] = forward;
69  const Real& strike = arguments_.strike;
70  results_.additionalResults["strikeSpread"] = strike;
71 
72  // Calculate risky PV01, as of the valuation date i.e. time 0, for the period from $t_E$ to underlying
73  // CDS maturity $T$. This risky PV01 does not include the non-risky accrual from the CDS premium leg coupon date
74  // immediately preceding the expiry date up to the expiry date.
75  Real rpv01 = std::abs(cds.couponLegNPV() + cds.accrualRebateNPV()) / (cds.notional() * cds.runningSpread());
76  results_.additionalResults["riskyAnnuity"] = rpv01;
77 
78  // Read the volatility from the volatility surface, assumed to have strike dimension in terms of spread.
79  const Date& exerciseDate = arguments_.exercise->dates().front();
80  Real stdDev = sqrt(volatility_->blackVariance(exerciseDate, strike, true));
81  results_.additionalResults["volatility"] = volatility_->blackVol(exerciseDate, strike, true);
82  results_.additionalResults["standardDeviation"] = stdDev;
83 
84  // Option type
85  Option::Type callPut = cds.side() == Protection::Buyer ? Option::Call : Option::Put;
86  results_.additionalResults["callPut"] = callPut == Option::Call ? string("Call") : string("Put");
87 
88  // NPV, Section 9.3.7 O'Kane 2008.
89  results_.value = rpv01 * cds.notional() * blackFormula(callPut, strike, forward, stdDev, 1.0);
90 
91  // If it is non-knockout and a payer, add the value of the default payout.
92  // Section 2.2 of Richard J.Martin, 2019 or Section 9.3.7 O'Kane 2008.
93  if (!arguments_.knocksOut && cds.side() == Protection::Buyer) {
94 
95  DiscountFactor disc = discount_->discount(exerciseDate);
96  results_.additionalResults["discountToExercise"] = disc;
97 
98  Probability sp = probability_->survivalProbability(exerciseDate);
99  results_.additionalResults["survivalProbabilityToExercise"] = sp;
100 
101  Real nonKoPv = disc * sp * (1 - recovery_);
102  results_.additionalResults["nonKnockoutPv"] = nonKoPv;
103 
104  results_.value += nonKoPv;
105  }
106 }
