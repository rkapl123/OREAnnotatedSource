27 SpreadedSmileSection2::SpreadedSmileSection2(const boost::shared_ptr<SmileSection>& base,
28  const std::vector<Real>& volSpreads, const std::vector<Real>& strikes,
29  const bool strikesRelativeToAtm, const Real atmLevel)
30  : SmileSection(base->exerciseTime(), base->dayCounter(), base->volatilityType(),
31  base->volatilityType() == ShiftedLognormal ? base->shift() : 0.0),
32  base_(base), volSpreads_(volSpreads), strikes_(strikes), strikesRelativeToAtm_(strikesRelativeToAtm),
33  atmLevel_(atmLevel) {
34  registerWith(base_);
35  QL_REQUIRE(!strikes_.empty(), "SpreadedSwaptionSmileSection: strikes empty");
36  QL_REQUIRE(strikes_.size() == volSpreads_.size(), "SpreadedSwaptionSmileSection: strike spreads ("
37  << strikes_.size() << ") inconsistent with vol spreads ("
38  << volSpreads_.size() << ")");
39  QL_REQUIRE(!strikesRelativeToAtm_ || strikes.size() == 1 || atmLevel != Null<Real>(),
40  "SpreadedSmileSection2: if strikes are relative to atm and more than one strike is given, the atm level "
41  "is required");
42  if (volSpreads_.size() > 1) {
43  volSpreadInterpolation_ = LinearFlat().interpolate(strikes_.begin(), strikes_.end(), volSpreads_.begin());
44  volSpreadInterpolation_.enableExtrapolation();
45  }
46 }

52 Volatility SpreadedSmileSection2::volatilityImpl(Rate strike) const {
53  if (volSpreads_.size() == 1)
54  return base_->volatility(strike) + volSpreads_.front();
55  else if (strikesRelativeToAtm_) {
56  Real f = atmLevel();
57  QL_REQUIRE(f != Null<Real>(), "SpreadedSmileSection2: atm level required");
58  return base_->volatility(strike) + volSpreadInterpolation_(strike - f);
59  } else {
60  return base_->volatility(strike) + volSpreadInterpolation_(strike);
61  }
62 }
