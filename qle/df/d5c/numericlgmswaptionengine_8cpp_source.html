33 Real NumericLgmSwaptionEngineBase::rebatePv(const Real x, const Real t, const Size exerciseIndex) const {
34  boost::shared_ptr<QuantExt::RebatedExercise> rebatedExercise =
35  boost::dynamic_pointer_cast<QuantExt::RebatedExercise>(exercise_);
36  if (rebatedExercise) {
37  return rebatedExercise->rebate(exerciseIndex) *
38  model()->discountBond(t,
39  model()->parametrization()->termStructure()->timeFromReference(
40  rebatedExercise->rebatePaymentDate(exerciseIndex)),

48 Real NumericLgmSwaptionEngineBase::calculate() const {
49 
50  iborModelCurve_ = boost::make_shared<LgmImpliedYtsFwdFwdCorrected>(model(), iborIndex_->forwardingTermStructure(),
51  DayCounter(), false, true);
52 
53  iborIndexCorrected_ = iborIndex_->clone(Handle<YieldTermStructure>(iborModelCurve_));
54 
55  Date settlement = model()->parametrization()->termStructure()->referenceDate();
56 
57  if (exercise_->dates().back() <= settlement) { // swaption is expired, possibly generated swap is not
58  // valued
59  return 0.0;
60  }
61 
62  int idx = static_cast<int>(exercise_->dates().size()) - 1;
63  int minIdxAlive =
64  static_cast<int>(std::upper_bound(exercise_->dates().begin(), exercise_->dates().end(), settlement) -
65  exercise_->dates().begin());
66 
67  int options = idx - minIdxAlive + 1;
68 
69  // terminal payoff
70 
71  Real t =
72  model()->parametrization()->termStructure()->timeFromReference(exercise_->dates()[minIdxAlive + options - 1]);
73  std::vector<Real> x = stateGrid(t);
74  std::vector<Real> v(x.size());
75  for (Size k = 0; k < x.size(); ++k) {
76  v[k] = std::max(conditionalSwapValue(x[k], t, exercise_->dates()[minIdxAlive + options - 1]) +
77  rebatePv(x[k], t, minIdxAlive + options - 1),
78  0.0);
79  }
80 
81  // roll back
82 
83  for (int j = options - 1; j > 0; --j) {
84  Real t_to =
85  model()->parametrization()->termStructure()->timeFromReference(exercise_->dates()[minIdxAlive + j - 1]);
86  v = rollback(v, t, t_to);
87  x = stateGrid(t_to);
88  for (Size k = 0; k < x.size(); ++k) {
89  QL_REQUIRE(v[k] > 0 || close_enough(v[k], 0.0), "negative value in rollback: " << v[k]);
90  // choose: continue or exercise
91  v[k] = std::max(v[k], conditionalSwapValue(x[k], t_to, exercise_->dates()[minIdxAlive + j - 1]) +
92  rebatePv(x[k], t_to, minIdxAlive + j - 1));
93  }
94  t = t_to;
95  }
96  v = rollback(v, t, 0.0);
97  return v[0];
98 
99 } // NumericLgmSwaptionEngineBase::calculate

114 Real NumericLgmSwaptionEngine::conditionalSwapValue(Real x, Real t, const Date expiry0) const {
115  Schedule fixedSchedule = arguments_.swap->fixedSchedule();
116  Schedule floatSchedule = arguments_.swap->floatingSchedule();
117  Size j1 = std::upper_bound(fixedSchedule.dates().begin(), fixedSchedule.dates().end(), expiry0 - 1) -
118  fixedSchedule.dates().begin();
119  Size k1 = std::upper_bound(floatSchedule.dates().begin(), floatSchedule.dates().end(), expiry0 - 1) -
120  floatSchedule.dates().begin();
121 
122  iborModelCurve_->move(expiry0, x);
123 
124  Real floatingLegNpv = 0.0;
125  for (Size l = k1; l < arguments_.floatingCoupons.size(); l++) {
126  Real T = model()->parametrization()->termStructure()->timeFromReference(arguments_.floatingPayDates[l]);
127  floatingLegNpv +=
128  arguments_.nominal * arguments_.floatingAccrualTimes[l] *
129  (arguments_.floatingSpreads[l] + iborIndexCorrected_->fixing(arguments_.floatingFixingDates[l])) *
130  model()->reducedDiscountBond(t, T, x, discountCurve_);
131  }
132 
133  Real fixedLegNpv = 0.0;
134  for (Size l = j1; l < arguments_.fixedCoupons.size(); l++) {
135  Real T = model()->parametrization()->termStructure()->timeFromReference(arguments_.fixedPayDates[l]);
136  fixedLegNpv += arguments_.fixedCoupons[l] * model()->reducedDiscountBond(t, T, x, discountCurve_);
137  }
138 
139  Option::Type type = arguments_.type == VanillaSwap::Payer ? Option::Call : Option::Put;
140 
141  Real exerciseValue = (type == Option::Call ? 1.0 : -1.0) * (floatingLegNpv - fixedLegNpv);
142  return exerciseValue;
143 } // NumericLgmSwaptionEngine::conditionalSwapValue

153 Real NumericLgmNonstandardSwaptionEngine::conditionalSwapValue(Real x, Real t, const Date expiry0) const {
154  Schedule fixedSchedule = arguments_.swap->fixedSchedule();
155  Schedule floatSchedule = arguments_.swap->floatingSchedule();
156  Size j1 = std::upper_bound(arguments_.fixedResetDates.begin(), arguments_.fixedResetDates.end(), expiry0 - 1) -
157  arguments_.fixedResetDates.begin();
158  Size k1 =
159  std::upper_bound(arguments_.floatingResetDates.begin(), arguments_.floatingResetDates.end(), expiry0 - 1) -
160  arguments_.floatingResetDates.begin();
161 
162  iborModelCurve_->move(expiry0, x);
163 
164  Real floatingLegNpv = 0.0;
165  for (Size l = k1; l < arguments_.floatingCoupons.size(); l++) {
166  Real T = model()->parametrization()->termStructure()->timeFromReference(arguments_.floatingPayDates[l]);
167  if (arguments_.floatingIsRedemptionFlow[l]) {
168  floatingLegNpv += arguments_.floatingCoupons[l] * model()->reducedDiscountBond(t, T, x, discountCurve_);
169  } else {
170  floatingLegNpv +=
171  arguments_.floatingNominal[l] * arguments_.floatingAccrualTimes[l] *
172  (arguments_.floatingSpreads[l] +
173  arguments_.floatingGearings[l] * iborIndexCorrected_->fixing(arguments_.floatingFixingDates[l])) *
174  model()->reducedDiscountBond(t, T, x, discountCurve_);
175  }
176  }
177 
178  Real fixedLegNpv = 0.0;
179  for (Size l = j1; l < arguments_.fixedCoupons.size(); l++) {
180  Real T = model()->parametrization()->termStructure()->timeFromReference(arguments_.fixedPayDates[l]);
181  fixedLegNpv += arguments_.fixedCoupons[l] * model()->reducedDiscountBond(t, T, x, discountCurve_);
182  }
183 
184  Option::Type type = arguments_.type == VanillaSwap::Payer ? Option::Call : Option::Put;
185 
186  Real exerciseValue = (type == Option::Call ? 1.0 : -1.0) * (floatingLegNpv - fixedLegNpv);
187  return exerciseValue;
188 } // NumericLgmSwaptionEngine::conditionalSwapValue
