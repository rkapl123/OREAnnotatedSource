58 void AverageOIS::initialize(const Schedule& fixedLegSchedule, const Schedule& onLegSchedule) {
59  // Fixed leg.
60  legs_[0] = FixedRateLeg(fixedLegSchedule)
61  .withNotionals(nominals_)
62  .withCouponRates(fixedRates_, fixedDayCounter_)
63  .withPaymentAdjustment(fixedPaymentAdjustment_)
64  .withPaymentCalendar(fixedPaymentCalendar_);
65 
66  // Average ON leg.
67  AverageONLeg tempAverageONLeg = AverageONLeg(onLegSchedule, overnightIndex_)
68  .withNotionals(nominals_)
69  .withPaymentAdjustment(onPaymentAdjustment_)
70  .withPaymentCalendar(onPaymentCalendar_)
71  .withRateCutoff(rateCutoff_)
72  .withSpreads(onSpreads_)
73  .withGearings(onGearings_)
74  .withPaymentDayCounter(onDayCounter_);
75 
76  if (onCouponPricer_) {
77  tempAverageONLeg = tempAverageONLeg.withAverageONIndexedCouponPricer(onCouponPricer_);
78  }
79 
80  legs_[1] = tempAverageONLeg;
81 
82  // Set the fixed and ON leg to pay (receive) and receive (pay) resp.
83  switch (type_) {
84  case Payer:
85  payer_[0] = -1.0;
86  payer_[1] = +1.0;
87  break;
88  case Receiver:
89  payer_[0] = +1.0;
90  payer_[1] = -1.0;
91  break;
92  default:
93  QL_FAIL("Unknown average ON index swap type");
94  }
95 }
