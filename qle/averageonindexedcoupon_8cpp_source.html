34 AverageONIndexedCoupon::AverageONIndexedCoupon(const Date& paymentDate, Real nominal, const Date& startDate,
35  const Date& endDate,
36  const boost::shared_ptr<OvernightIndex>& overnightIndex, Real gearing,
37  Spread spread, Natural rateCutoff, const DayCounter& dayCounter,
38  const Period& lookback, const Size fixingDays,
39  const Date& rateComputationStartDate, const Date& rateComputationEndDate,
40  const bool telescopicValueDates)
41  : FloatingRateCoupon(paymentDate, nominal, startDate, endDate, fixingDays, overnightIndex, gearing, spread, Date(),
42  Date(), dayCounter, false),
43  overnightIndex_(overnightIndex), rateCutoff_(rateCutoff), lookback_(lookback),
44  rateComputationStartDate_(rateComputationStartDate), rateComputationEndDate_(rateComputationEndDate) {
45 
46  Date valueStart = rateComputationStartDate_ == Null<Date>() ? startDate : rateComputationStartDate_;
47  Date valueEnd = rateComputationEndDate_ == Null<Date>() ? endDate : rateComputationEndDate_;
48  if (lookback != 0 * Days) {
49  BusinessDayConvention bdc = lookback.length() > 0 ? Preceding : Following;
50  valueStart = overnightIndex->fixingCalendar().advance(valueStart, -lookback, bdc);
51  valueEnd = overnightIndex->fixingCalendar().advance(valueEnd, -lookback, bdc);
52  }
53 
54  // Populate the value dates.
55  Date tmpEndDate = valueEnd;
56  if(telescopicValueDates) {
57  // same optimization as in OvernightIndexedCoupon
58  Date evalDate = Settings::instance().evaluationDate();
59  tmpEndDate = overnightIndex->fixingCalendar().advance(std::max(valueStart, evalDate), 7, Days, Following);
60  tmpEndDate = std::min(tmpEndDate, valueEnd);
61  }
62 
63  Schedule sch = MakeSchedule()
64  .from(valueStart)
65  .to(tmpEndDate)
66  .withTenor(1 * Days)
67  .withCalendar(overnightIndex->fixingCalendar())
68  .withConvention(overnightIndex->businessDayConvention())
69  .backwards();
70  valueDates_ = sch.dates();
71 
72  if (telescopicValueDates) {
73  // build optimised value dates schedule: back stub
74  // contains at least two dates and enough periods to cover rate cutoff
75  Date tmp2 = overnightIndex->fixingCalendar().adjust(valueEnd, overnightIndex->businessDayConvention());
76  Date tmp1 = overnightIndex->fixingCalendar().advance(tmp2, -std::max<Size>(rateCutoff_, 1), Days, Preceding);
77  while (tmp1 <= tmp2) {
78  if (tmp1 > valueDates_.back())
79  valueDates_.push_back(tmp1);
80  tmp1 = overnightIndex->fixingCalendar().advance(tmp1, 1, Days, Following);
81  }
82  }
83 
84  QL_ENSURE(valueDates_.size() >= 2 + rateCutoff_, "degenerate schedule");
85 
86  // the first and last value date should be the unadjusted input value dates
87  if (valueDates_.front() != valueStart)
88  valueDates_.front() = valueStart;
89  if (valueDates_.back() != valueEnd)
90  valueDates_.back() = valueEnd;
91 
92  numPeriods_ = valueDates_.size() - 1;
93 
94  QL_REQUIRE(valueDates_[0] != valueDates_[1],
95  "internal error: first two value dates of on coupon are equal: " << valueDates_[0]);
96  QL_REQUIRE(valueDates_[numPeriods_] != valueDates_[numPeriods_ - 1],
97  "internal error: last two value dates of on coupon are equal: " << valueDates_[numPeriods_]);
98 
99  // Populate the fixing dates.
100  fixingDates_.resize(numPeriods_);
101  for (Size i = 0; i < numPeriods_; ++i)
102  fixingDates_[i] = overnightIndex->fixingCalendar().advance(
103  valueDates_[i], -static_cast<Integer>(FloatingRateCoupon::fixingDays()), Days, Preceding);
104 
105  // Populate the accrual periods.
106  dt_.resize(numPeriods_);
107  for (Size i = 0; i < numPeriods_; ++i)
108  dt_[i] = dayCounter.yearFraction(valueDates_[i], valueDates_[i + 1]);
109 
110  // check that rate cutoff is < number of fixing dates
111  QL_REQUIRE(rateCutoff_ < numPeriods_, "rate cutoff (" << rateCutoff_
112  << ") must be less than number of fixings in period ("
113  << numPeriods_ << ")");
114 }

116 const std::vector<Rate>& AverageONIndexedCoupon::indexFixings() const {
117 
118  fixings_.resize(numPeriods_);
119  Size i;
120 
121  for (i = 0; i < numPeriods_ - rateCutoff_; ++i) {
122  fixings_[i] = index_->fixing(fixingDates_[i]);
123  }
124 
125  Rate cutoffFixing = fixings_[i - 1];
126  while (i < numPeriods_) {
127  fixings_[i] = cutoffFixing;
128  i++;
129  }
130

157 void CappedFlooredAverageONIndexedCoupon::performCalculations() const {
158  QL_REQUIRE(underlying_->pricer(), "pricer not set");
159  Rate swapletRate = nakedOption_ ? 0.0 : underlying_->rate();
160  if (floor_ != Null<Real>() || cap_ != Null<Real>())
161  pricer()->initialize(*this);
162  Rate floorletRate = 0.;
163  if (floor_ != Null<Real>())
164  floorletRate = pricer()->floorletRate(effectiveFloor());
165  Rate capletRate = 0.;
166  if (cap_ != Null<Real>())
167  capletRate = (nakedOption_ && floor_ == Null<Real>() ? -1.0 : 1.0) * pricer()->capletRate(effectiveCap());
168  rate_ = swapletRate + floorletRate - capletRate;
169 }

182 Rate CappedFlooredAverageONIndexedCoupon::effectiveCap() const {
183  if (cap_ == Null<Real>())
184  return Null<Real>();
185  /* We have four cases dependent on localCapFloor_ and includeSpread. Notation in the formulas:
186  g gearing,
187  s spread,
188  A coupon amount,
189  f_i daily fixings,
190  \tau_i daily accrual fractions,
191  \tau coupon accrual fraction,
192  C cap rate
193  F floor rate
194  */
195  if (localCapFloor_) {
196  if (includeSpread()) {
197  // A = \cdot \frac{\prod (1 + \tau_i \min ( \max ( f_i + s , F), C)) - 1}{\tau}
198  return cap_ - underlying_->spread();
199  } else {
200  // A = g \cdot \frac{\prod (1 + \tau_i \min ( \max ( f_i , F), C)) - 1}{\tau} + s
201  return cap_;
202  }
203  } else {
204  if (includeSpread()) {
205  // A = \min \left( \max \left( \cdot \frac{\prod (1 + \tau_i(f_i + s)) - 1}{\tau}, F \right), C \right)
206  return (cap_ / gearing() - underlying_->spread());
207  } else {
208  // A = \min \left( \max \left( g \cdot \frac{\prod (1 + \tau_i f_i) - 1}{\tau} + s, F \right), C \right)
209  return (cap_ - underlying_->spread()) / gearing();
210  }
211  }
212 }

214 Rate CappedFlooredAverageONIndexedCoupon::effectiveFloor() const {
215  if (floor_ == Null<Real>())
216  return Null<Real>();
217  if (localCapFloor_) {
218  if (includeSpread()) {
219  return floor_ - underlying_->spread();
220  } else {
221  return floor_;
222  }
223  } else {
224  if (includeSpread()) {
225  return (floor_ - underlying_->spread());
226  } else {
227  return (floor_ - underlying_->spread()) / gearing();
228  }
229  }

408 AverageONLeg::operator Leg() const {
409 
410  QL_REQUIRE(!notionals_.empty(), "No notional given for average overnight leg.");
411 
412  Leg cashflows;
413 
414  Calendar calendar = schedule_.calendar();
415  Calendar paymentCalendar = paymentCalendar_;
416 
417  if (calendar.empty())
418  calendar = paymentCalendar;
419  if (calendar.empty())
420  calendar = WeekendsOnly();
421  if (paymentCalendar.empty())
422  paymentCalendar = calendar;
423 
424  Date refStart, start, refEnd, end;
425  Date paymentDate;
426 
427  Size n = schedule_.size() - 1;
428  for (Size i = 0; i < n; ++i) {
429  refStart = start = schedule_.date(i);
430  refEnd = end = schedule_.date(i + 1);
431  paymentDate = paymentCalendar.advance(end, paymentLag_, Days, paymentAdjustment_);
432 
433  // determine refStart and refEnd
434 
435  if (i == 0 && schedule_.hasIsRegular() && !schedule_.isRegular(i + 1))
436  refStart = calendar.adjust(end - schedule_.tenor(), paymentAdjustment_);
437  if (i == n - 1 && schedule_.hasIsRegular() && !schedule_.isRegular(i + 1))
438  refEnd = calendar.adjust(start + schedule_.tenor(), paymentAdjustment_);
439 
440  // Determine the rate computation start and end date as
441  // - the coupon start and end date, if in arrears, and
442  // - the previous coupon start and end date, if in advance.
443  // In addition, adjust the start date, if a last recent period is given.
444 
445  Date rateComputationStartDate, rateComputationEndDate;
446  if (inArrears_) {
447  // in arrears fixing (i.e. the "classic" case)
448  rateComputationStartDate = start;
449  rateComputationEndDate = end;
450  } else {
451  // handle in advance fixing
452  if (i > 0) {
453  // if there is a previous period, we take that
454  rateComputationStartDate = schedule_.date(i - 1);
455  rateComputationEndDate = schedule_.date(i);
456  } else {
457  // otherwise we construct the previous period
458  rateComputationEndDate = start;
459  if (schedule_.hasTenor() && schedule_.tenor() != 0 * Days)
460  rateComputationStartDate = calendar.adjust(start - schedule_.tenor(), Preceding);
461  else
462  rateComputationStartDate = calendar.adjust(start - (end - start), Preceding);
463  }
464  }
465 
466  if (lastRecentPeriod_) {
467  rateComputationStartDate = (lastRecentPeriodCalendar_.empty() ? calendar : lastRecentPeriodCalendar_)
468  .advance(rateComputationEndDate, -*lastRecentPeriod_);
469  }
470 
471  // build coupon
472 
473  if (close_enough(detail::get(gearings_, i, 1.0), 0.0)) {
474  // fixed coupon
475  cashflows.push_back(boost::make_shared<FixedRateCoupon>(paymentDate, detail::get(notionals_, i, 1.0),
476  detail::get(spreads_, i, 0.0), paymentDayCounter_,
477  start, end, refStart, refEnd));
478  } else {
479  // floating coupon
480  auto cpn = boost::make_shared<AverageONIndexedCoupon>(
481  paymentDate, detail::get(notionals_, i, notionals_.back()), start, end, overnightIndex_,
482  detail::get(gearings_, i, 1.0), detail::get(spreads_, i, 0.0), rateCutoff_, paymentDayCounter_,
483  lookback_, fixingDays_, rateComputationStartDate, rateComputationEndDate, telescopicValueDates_);
484  if (couponPricer_) {
485  cpn->setPricer(couponPricer_);
486  }
487  Real cap = detail::get(caps_, i, Null<Real>());
488  Real floor = detail::get(floors_, i, Null<Real>());
489  if (cap == Null<Real>() && floor == Null<Real>()) {
490  cashflows.push_back(cpn);
491  } else {
492  auto cfCpn = ext::make_shared<CappedFlooredAverageONIndexedCoupon>(cpn, cap, floor, nakedOption_,
493  localCapFloor_, includeSpread_);
494  if (capFlooredCouponPricer_) {
495  cfCpn->setPricer(capFlooredCouponPricer_);
496  }
497  cashflows.push_back(cfCpn);
498  }
499  }
500  }
501  return cashflows;
502 }
