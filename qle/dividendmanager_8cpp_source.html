30void applyDividends(const set<Dividend>& dividends) {
31 Size count = 0;
32 map<string, boost::shared_ptr<EquityIndex2>> cache;
33 boost::shared_ptr<EquityIndex2> index;
34 std::string lastIndexName;
35 for (auto& d: dividends) {
36 try {
37 if (lastIndexName != d.name) {
38 index = boost::make_shared<EquityIndex2>(d.name, NullCalendar(), Currency());
39 lastIndexName = d.name;
40 }
41 index->addDividend(d, true);
42 count++;
43 } catch (const std::exception& e) {
44 }
45 }
46}
