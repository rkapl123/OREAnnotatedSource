38 //! Stripped zero inflation volatility structure
39 /*!
40  The surface provides implied CPI Black volatilities for the union of strikes that occur in the underlying
41  cap and floor price surface.
42 
43  The type argument determines which kind of price quotes are used with priority when there is an overlap,
44  i.e. strikes for which we have both cap and floor quotes:
45  If type is Cap: Use cap quotes where available, floor quotes otherwise
46  If type is Floor: Use floor quotes where available, cap quotes otherwise
47  If type is CapFloor: In case of overlap, use floor quotes up to the ATM strike, cap quotes for strikes beyond ATM
48  */

146 template <class Interpolator2D> void StrippedCPIVolatilitySurface<Interpolator2D>::performCalculations() const {
147  strikes_ = priceSurface_->strikes();
148  maturities_ = priceSurface_->maturities();
149  volData_ = QuantLib::Matrix(strikes_.size(), maturities_.size(), QuantLib::Null<QuantLib::Real>());
150  QuantLib::Brent solver;
151  QuantLib::Real guess = (upperVolBound_ + lowerVolBound_) / 2.0;
152  QuantLib::Date startDate = capFloorStartDate();
153  QuantLib::Date underlyingBaseDate =
154  ZeroInflation::fixingDate(startDate, observationLag(), frequency(), indexIsInterpolated());
155  double baseCPI = ZeroInflation::cpiFixing(index_, startDate, observationLag(), indexIsInterpolated());
156  for (QuantLib::Size i = 0; i < strikes_.size(); i++) {
157  for (QuantLib::Size j = 0; j < maturities_.size(); j++) {
158 
159  QuantLib::Date maturityDate = optionMaturityFromTenor(maturities_[j]);
160  QuantLib::Date fixDate =
161  ZeroInflation::fixingDate(maturityDate, observationLag(), frequency(), indexIsInterpolated());
162  Rate I1 = ZeroInflation::cpiFixing(index_, maturityDate, observationLag(), indexIsInterpolated());
163 
164  Time timeToMaturity = dayCounter().yearFraction(underlyingBaseDate, fixDate);
165  QuantLib::Real atmRate = std::pow(I1 / baseCPI, 1 / timeToMaturity) - 1.0;
166 
167  bool useFloor = chooseFloor(strikes_[i], atmRate);
168 
169  // FIXME: Do we need an effective maturity here ?
170  QuantLib::Real priceToMatch = useFloor ? priceSurface_->floorPrice(maturities_[j], strikes_[i])
171  : priceSurface_->capPrice(maturities_[j], strikes_[i]);
172  try {
173  ObjectiveFunction func(priceToMatch, useFloor, strikes_[i], startDate, maturityDate, baseCPI,
174  priceSurface_, index_, engine_);
175  QuantLib::Real found = solver.solve(func, solverTolerance_, guess, lowerVolBound_, upperVolBound_);
176  volData_[i][j] = found;
177  } catch (std::exception& e) {
178  QL_FAIL("failed to find implied vol for "
179  << (useFloor ? "Floor" : "Cap") << " with strike " << strikes_[i] << " and maturity "
180  << maturities_[j] << ", because: " << e.what() << " "
181  << QuantLib::io::iso_date(startDate + maturities_[j]) << " " << maturityDate);
182  }
183  }
184  }
185 
186  maturityTimes_.clear();
187  for (QuantLib::Size i = 0; i < maturities_.size(); i++) {
188  QuantLib::Date d = optionMaturityFromTenor(maturities_[i]);
189  maturityTimes_.push_back(fixingTime(d));
190  }
191 
192  vols_ = interpolator2d_.interpolate(maturityTimes_.begin(), maturityTimes_.end(), strikes_.begin(), strikes_.end(),
193  volData_);
194  vols_.enableExtrapolation();
195 }

197 template <class Interpolator2D>
198 bool StrippedCPIVolatilitySurface<Interpolator2D>::chooseFloor(QuantLib::Real strike, QuantLib::Real atmRate) const {
199  if (preference_ == Floor) {
200  if (strike <= priceSurface_->floorStrikes().back())
201  return true;
202  else
203  return false;
204  }
205 
206  if (preference_ == Cap) {
207  if (strike < priceSurface_->capStrikes().front())
208  return true;
209  else
210  return false;
211  }
212 
213  // else: Use floors where we have floor quotes only, caps where we have cap quotes only,
214  // and decide based on ATM where we have both cap and floor quotes
215 
216  // 1) strike < maxFloorStrike < minCapStrike: Floor!
217  if (strike <= priceSurface_->floorStrikes().back() && strike < priceSurface_->capStrikes().front())
218  return true;
219  // 2) strike > maxFloorStrike and strike >= minCapStrike: Cap!
220  else if (strike > priceSurface_->floorStrikes().back() && strike >= priceSurface_->capStrikes().front())
221  return false;
222  // 3) Overlap, maxFloorStrike > minCapStrike and strike in between: Depends on atmRate which surface we pick
223  else if (strike <= priceSurface_->floorStrikes().back() && strike >= priceSurface_->capStrikes().front()) {
224  // we have overlapping strikes, decide depending on atm level
225  if (strike < atmRate)
226  return true;
227  else
228  return false;
229  // 4) Gap, maxFloorStrike < minCapStrike and strike in the gap: Depends on atmRate which surface we
230  // extrapolate
231  } else if (strike > priceSurface_->floorStrikes().back() && strike < priceSurface_->capStrikes().front()) {
232  // there is a gap between floor end and caps begin, decide again depending on strike level
233  if (strike < atmRate)
234  return true;
235  else
236  return false;
237  } else {
238  QL_FAIL("case not covered in StrippedCPIVolatilitySurface: strike="
239  << strike << " maxFloorStrike=" << priceSurface_->floorStrikes().back()
240  << " minCapStrike=" << priceSurface_->capStrikes().front() << " atm=" << atmRate);
241  return false;
242  }
243 }
