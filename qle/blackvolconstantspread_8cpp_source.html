71 Volatility BlackVolatilityConstantSpread::blackVolImpl(Time t, Rate strike) const {
72  Real s = surface_->blackVol(t, strike, true) - surface_->blackVol(t, Null<Real>(), true);
73  Real v = atm_->blackVol(t, Null<Real>(), true);
74  return v + s;
75 }

77 Real BlackVolatilityConstantSpread::blackVarianceImpl(Time t, Real strike) const {
