86 // cache for exact discretization
87 struct cache_key {
88 double t0, dt;
89 bool operator==(const cache_key& o) const { return (t0 == o.t0) && (dt == o.dt); }
90 };
91
92 struct cache_hasher {
93 std::size_t operator()(cache_key const& x) const {
94 std::size_t seed = 0;
95 boost::hash_combine(seed, x.t0);
96 boost::hash_combine(seed, x.dt);
97 return seed;
98 }
99 };
100 mutable boost::unordered_map<cache_key, Array, cache_hasher> cache_m_;
101 mutable boost::unordered_map<cache_key, Matrix, cache_hasher> cache_v_, cache_d_;
102 }; // ExactDiscretization
103
104 // cache for process drift and diffusion (e.g. used in Euler discretization)

106 std::size_t operator()(double const& x) const {

109 return seed;
110 }
111 };

113 mutable boost::unordered_map<double, Array, cache_hasher> cache_m_;
