49map<Date, Real> JyImpliedYoYInflationTermStructure::yoyRates(const vector<Date>& dts, const Period& obsLag) const {
50
51 // First step is to calculate the YoY swap rate for each maturity date in dts and store in yyiisRates.
52 map<Date, Real> yoySwaplets;
53 map<Date, Real> discounts;
54 map<Date, Real> yyiisRates;
55 auto irIdx = model_->ccyIndex(model_->infjy(index_)->currency());
56
57 // Will need a YoY index below in the helpers.
58 boost::shared_ptr<YoYInflationIndex> index = boost::make_shared<YoYInflationIndexWrapper>(
59 model_->infjy(index_)->inflationIndex(), indexIsInterpolated());
60
61 for (const auto& maturity : dts) {
62
63 // Schedule for the YoY swap with maturity date equal to `maturity`
64 Schedule schedule = MakeSchedule()
65 .from(referenceDate_)
66 .to(maturity)
67 .withTenor(1 * Years)
68 .withConvention(Unadjusted)
69 .withCalendar(calendar())
70 .backwards();
71
72 // Store the value of the model implied YoY leg and the value of the fixed leg annuity.
73 Real yoyLegValue = 0.0;
74 Real fixedLegAnnuity = 0.0;
75
76 for (Size i = 1; i < schedule.dates().size(); ++i) {
77
78 // Start and end of the current YoY swaplet period.
79 const auto& start = schedule.dates()[i - 1];
80 const auto& end = schedule.dates()[i];
81
82 // If we have already calculated the YoY swaplet price for this period, use it.
83 // We should always have a discount factor in this case also.
84 auto it = yoySwaplets.find(end);
85 if (it != yoySwaplets.end()) {
86 yoyLegValue += it->second;
87 fixedLegAnnuity += discounts.at(end);
88 continue;
89 }
90
91 // Need to calculate the YoY swaplet value over the period [start, end]
92 Real swaplet;
93 auto T = relativeTime_ + dayCounter().yearFraction(referenceDate_, end);
94 auto discount = model_->discountBond(irIdx, relativeTime_, T, state_[2]);
95 if (i == 1) {
96 // The first YoY swaplet is a zero coupon swaplet because I_{start} is known.
97 auto growth = inflationGrowth(model_, index_, relativeTime_, T, state_[2], state_[0], indexIsInterpolated_);
98 swaplet = discount * (growth - 1.0);
99 } else {
100 auto S = relativeTime_ + dayCounter().yearFraction(referenceDate_, start);
101 swaplet = yoySwaplet(S, T);
102 }
103
104 // Cache the swaplet value and the discount factor related to this swaplet end date.
105 yoySwaplets[end] = swaplet;
106 discounts[end] = discount;
107
108 // Update the YoY leg value and the fixed leg annuity
109 yoyLegValue += swaplet;
110 fixedLegAnnuity += discount;
111 }
112
113 // The model implied YoY inflation swap rate
114 yyiisRates[maturity] = yoyLegValue / fixedLegAnnuity;
115 }
116
117 QL_REQUIRE(!yyiisRates.empty(), "JyImpliedYoYInflationTermStructure: yoyRates did not create any YoY swap rates.");
118
119 // Will need a discount term structure in the bootstrap below so create it here from the discounts map.
120 vector<Date> dfDates;
121 vector<Real> dfValues;
122
123 if (discounts.count(referenceDate_) == 0) {
124 dfDates.push_back(referenceDate_);
125 dfValues.push_back(1.0);
126 }
127
128 for (const auto& kv : discounts) {
129 dfDates.push_back(kv.first);
130 dfValues.push_back(kv.second);
131 }
132
133 auto irTs = model_->irlgm1f(irIdx)->termStructure();
134 Handle<YieldTermStructure> yts(boost::make_shared<InterpolatedDiscountCurve<LogLinear>>(
135 dfDates, dfValues, irTs->dayCounter(), LogLinear()));
136
137 // Create the YoY swap helpers from the YoY swap rates calculated above.
138 // Using the curve's day counter as the helper's day counter for now.
139 using YoYHelper = BootstrapHelper<YoYInflationTermStructure>;
140 vector<boost::shared_ptr<YoYHelper>> helpers;
141 for (const auto& kv : yyiisRates) {
142 Handle<Quote> yyiisQuote(boost::make_shared<SimpleQuote>(kv.second));
143 helpers.push_back(boost::make_shared<YearOnYearInflationSwapHelper>(yyiisQuote, observationLag(), kv.first,
144 calendar(), Unadjusted, dayCounter(), index, yts));
145 }
146
147 // Create a YoY curve from the helpers
148 // Use Linear here in line with what is in scenariosimmarket and todaysmarket but should probably be more generic.
149 auto lag = obsLag == -1 * Days ? observationLag() : obsLag;
150 auto baseRate = helpers.front()->quote()->value();
151 auto yoyCurve = boost::make_shared<PiecewiseYoYInflationCurve<Linear>>(referenceDate_, calendar(), dayCounter(),
152 lag, frequency(), indexIsInterpolated(), baseRate, helpers, 1e-12);
153
154 // Read the necessary YoY rates from the bootstrapped YoY inflation curve
155 map<Date, Real> result;
156 for (const auto& maturity : dts) {
157 result[maturity] = yoyCurve->yoyRate(maturity);
158 }
159
160 return result;
161}

163Real JyImpliedYoYInflationTermStructure::yoySwaplet(Time S, Time T) const {
164
165 // The JY implied swap rate at time t for the period from S to T is:
166 // N \tau(S, T) \left\{ P_n(t,S) \frac{P_r(t,T)}{P_r(t,S)} e^{C(t,S,T)} - P_n(t,T) \right\}
167 // where N is the nominal, \tau(S, T) is the daycount which we assume equals 1 here.
168 // e^{C(t,S,T)} is the correction term which we deal with below.
169
170 // Get P_n(t,S) and P_n(t,T).
171 auto irIdx = model_->ccyIndex(model_->infjy(index_)->currency());
172 auto irTs = model_->irlgm1f(irIdx)->termStructure();
173 auto p_n_t_S = model_->discountBond(irIdx, relativeTime_, S, state_[2]);
174 auto p_n_t_T = model_->discountBond(irIdx, relativeTime_, T, state_[2]);
175
176 // Get the rrRatio := P_r(t,T)}{P_r(t,S)}.
177 auto rrParam = model_->infjy(index_)->realRate();
178 auto H_r_S = rrParam->H(S);
179 auto H_r_T = rrParam->H(T);
180 auto zeta_r_t = rrParam->zeta(relativeTime_);
181 auto rrRatio = exp(-(H_r_T - H_r_S) * state_[0] - 0.5 * (H_r_T * H_r_T - H_r_S * H_r_S) * zeta_r_t);
182
183 const auto& zts = model_->infjy(index_)->realRate()->termStructure();
184 rrRatio *= (irTs->discount(T) * inflationGrowth(zts, T, indexIsInterpolated_)) / (irTs->discount(S) * inflationGrowth(zts, S, indexIsInterpolated_));
185
186 // Calculate the correction term C(t,S,T)
187 using CrossAssetAnalytics::integral;
188 using CrossAssetAnalytics::LC;
189 using CrossAssetAnalytics::P;
190 using CrossAssetAnalytics::az;
191 using CrossAssetAnalytics::Hz;
192 using CrossAssetAnalytics::ay;
193 using CrossAssetAnalytics::Hy;
194 using CrossAssetAnalytics::sy;
195 using CrossAssetAnalytics::rzy;
196 using CrossAssetAnalytics::ryy;
197
198 auto H_n_S = model_->irlgm1f(irIdx)->H(S);
199 auto zeta_r_S = rrParam->zeta(S);
200
201 auto c = H_r_S * (zeta_r_S - zeta_r_t);
202 c -= H_n_S * integral(model_.get(), P(rzy(irIdx, index_, 0), az(irIdx), ay(index_)), relativeTime_, S);
203 c += integral(model_.get(), LC(0.0, -1.0, P(ay(index_), ay(index_), Hy(index_)),
204 1.0, P(rzy(irIdx, index_, 0), az(irIdx), ay(index_), Hz(irIdx)),
205 -1.0, P(ryy(index_, index_, 0, 1), ay(index_), sy(index_))), relativeTime_, S);
206 c *= (H_r_S - H_r_T);
207
208 return p_n_t_S * rrRatio * exp(c) - p_n_t_T;
209}
