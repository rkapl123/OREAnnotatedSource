96 template <class Interpolator2D>
97 InterpolatedCPIVolatilitySurface<Interpolator2D>::InterpolatedCPIVolatilitySurface(
98  const std::vector<Period>& optionTenors, const std::vector<Real>& strikes,
99  const std::vector<std::vector<Handle<Quote> > > quotes,
100  const boost::shared_ptr<QuantLib::ZeroInflationIndex>& index, const Natural settlementDays, const Calendar& cal, BusinessDayConvention bdc, const DayCounter& dc,
101  const Period& observationLag,
102  const Date& capFloorStartDate,
103  const Interpolator2D& interpolator2d)
104  : CPIVolatilitySurface(settlementDays, cal, bdc, dc, observationLag, index->frequency(), index->interpolated(),
105  capFloorStartDate),
106  optionTenors_(optionTenors), strikes_(strikes), quotes_(quotes), index_(index),
107  interpolator2d_(interpolator2d) {
108  for (Size i = 0; i < optionTenors_.size(); ++i) {
109  QL_REQUIRE(quotes_[i].size() == strikes_.size(), "quotes row " << i << " length does not match strikes size");
110  for (Size j = 0; j < strikes_.size(); ++j)
111  registerWith(quotes_[i][j]);
112  }
113 }

115 template <class Interpolator2D> void InterpolatedCPIVolatilitySurface<Interpolator2D>::performCalculations() const {
116  volData_ = QuantLib::Matrix(strikes_.size(), optionTenors_.size(), QuantLib::Null<QuantLib::Real>());
117  QL_REQUIRE(quotes_.size() == optionTenors_.size(), "quotes rows does not match option tenors size");
118  optionTimes_.clear();
119  for (Size i = 0; i < optionTenors_.size(); ++i) {
120  QuantLib::Date d = optionMaturityFromTenor(optionTenors_[i]);
121  // Save the vols at their fixing times and not maturity
122  optionTimes_.push_back(fixingTime(d));
123  for (QuantLib::Size j = 0; j < strikes_.size(); j++)
124  volData_[j][i] = quotes_[i][j]->value();
125  }
126 
127  vols_ = interpolator2d_.interpolate(optionTimes_.begin(), optionTimes_.end(), strikes_.begin(), strikes_.end(),
128  volData_);
129  // FIXME ?
130  vols_.enableExtrapolation();
131  vols_.update();
132 }

142 template <class Interpolator2D> QuantLib::Date InterpolatedCPIVolatilitySurface<Interpolator2D>::maxDate() const {
