38 CashFlowResults standardCashFlowResults(const boost::shared_ptr<CashFlow>& c, const Real multiplier,
39  const std::string& type, const Size legNo, const Currency& currency,
40  const Handle<YieldTermStructure>& discountCurve) {
41 
42  Date today = Settings::instance().evaluationDate();
43 
44  CashFlowResults cfResults;
45 
46  cfResults.amount = c->amount() * multiplier;
47  cfResults.payDate = c->date();
48 
49  if (!currency.empty())
50  cfResults.currency = currency.code();
51 
52  cfResults.legNumber = legNo;
53  cfResults.type = type;
54 
55  if (auto cpn = boost::dynamic_pointer_cast<Coupon>(c)) {
56  cfResults.rate = cpn->rate();
57  cfResults.accrualStartDate = cpn->accrualStartDate();
58  cfResults.accrualEndDate = cpn->accrualEndDate();
59  cfResults.accruedAmount = cpn->accruedAmount(today);
60  cfResults.notional = cpn->nominal();
61  }
62 
63  if (auto cpn = boost::dynamic_pointer_cast<FloatingRateCoupon>(c)) {
64  cfResults.fixingDate = cpn->fixingDate();
65  }
66 
67  if (!discountCurve.empty()) {
68  cfResults.discountFactor = discountCurve->discount(c->date());
69  cfResults.presentValue = cfResults.amount * cfResults.discountFactor;
70  }
71 
72  return cfResults;
73 }
