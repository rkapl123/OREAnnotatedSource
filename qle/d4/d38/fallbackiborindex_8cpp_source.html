61 boost::shared_ptr<OvernightIndexedCoupon> FallbackIborIndex::onCoupon(const Date& iborFixingDate,
62  const bool telescopicValueDates) const {
63  QL_REQUIRE(iborFixingDate >= switchDate_, "FallbackIborIndex: onCoupon for ibor fixing date "
64  << iborFixingDate << " requested, which is before switch date "
65  << switchDate_ << " for index '" << name() << "'");
66  Date valueDate = originalIndex_->valueDate(iborFixingDate);
67  Date maturityDate = originalIndex_->maturityDate(valueDate);
68  return boost::make_shared<OvernightIndexedCoupon>(maturityDate, 1.0, valueDate, maturityDate, rfrIndex_, 1.0, 0.0,
69  Date(), Date(), DayCounter(), telescopicValueDates, false,
70  2 * Days, 0, Null<Size>());
71 }

73 Real FallbackIborIndex::fixing(const Date& fixingDate, bool forecastTodaysFixing) const {
74  Date today = Settings::instance().evaluationDate();
75  if (today < switchDate_ || fixingDate < switchDate_) {
76  return originalIndex_->fixing(fixingDate, forecastTodaysFixing);
77  }
78  if (fixingDate > today) {
79  return IborIndex::forecastFixing(fixingDate);
80  } else {
81  return onCoupon(fixingDate, true)->rate() + spread_;
82  }
83 }

97 Rate FallbackIborIndex::forecastFixing(const Date& valueDate, const Date& endDate, Time t) const {
98  Date today = Settings::instance().evaluationDate();
99  Handle<YieldTermStructure> curve =
100  today < switchDate_ ? originalIndex_->forwardingTermStructure() : forwardingTermStructure();
101  QL_REQUIRE(!curve.empty(), "FallbackIborIndex: null term structure set for " << name() << ", today=" << today
102  << ", switchDate=" << switchDate_);
103  DiscountFactor disc1 = curve->discount(valueDate);
104  DiscountFactor disc2 = curve->discount(endDate);
105  return (disc1 / disc2 - 1.0) / t;
106 }
