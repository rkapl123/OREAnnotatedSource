32 Rate ZeroInflationIndexWrapper::fixing(const Date& fixingDate, bool /*forecastTodaysFixing*/) const {
33 
34  // duplicated logic from CPICashFlow::amount()
35 
36  // what interpolation do we use? Index / flat / linear
37  if (interpolation_ == CPI::AsIndex) {
38  return source_->fixing(fixingDate);
39  } else {
40  std::pair<Date, Date> dd = inflationPeriod(fixingDate, frequency());
41  Real indexStart = source_->fixing(dd.first);
42  if (interpolation_ == CPI::Linear) {
43  Real indexEnd = source_->fixing(dd.second + Period(1, Days));
44  // linear interpolation
45  return indexStart + (indexEnd - indexStart) * (fixingDate - dd.first) /
46  ((dd.second + Period(1, Days)) -
47  dd.first); // can't get to next period's value within current period
48  } else {
49  // no interpolation, i.e. flat = constant, so use start-of-period value
50  return indexStart;
51  }
52  }
53 }

63 Rate YoYInflationIndexWrapper::fixing(const Date& fixingDate, bool /*forecastTodaysFixing*/) const {
64 
65  // duplicated logic from YoYInflationIndex, this would not be necessary, if forecastFixing
66  // was defined virtual in InflationIndex
67  Date today = Settings::instance().evaluationDate();
68  Date todayMinusLag = today - availabilityLag_;
69  std::pair<Date, Date> lim = inflationPeriod(todayMinusLag, frequency_);
70  Date lastFix = lim.first - 1;
71 
72  Date flatMustForecastOn = lastFix + 1;
73  Date interpMustForecastOn = lastFix + 1 - Period(frequency_);
74 
75  if (interpolated() && fixingDate >= interpMustForecastOn) {
76  return forecastFixing(fixingDate);
77  }
78 
79  if (!interpolated() && fixingDate >= flatMustForecastOn) {
80  return forecastFixing(fixingDate);
81  }
82 
83  // historical fixing
84  return YoYInflationIndex::fixing(fixingDate);
85 }
