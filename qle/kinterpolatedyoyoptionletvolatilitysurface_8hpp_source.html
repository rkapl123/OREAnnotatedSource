155  volatilityImpl(const Date &d, Rate strike) const {
156  updateSlice(d);

158  // extrapolation on interpolator (if enabled in this class)

177  volatilityImpl(Time length, Rate strike) const {
178 
179  Natural years = (Natural)floor(length);
180  Natural days = (Natural)floor((length - years) * 365.0);
181  Date d = referenceDate() + Period(years, Years) + Period(days, Days);
182 
183  return this->volatilityImpl(d, strike);

186  template<class Interpolator1D>
187  void KInterpolatedYoYOptionletVolatilitySurface<Interpolator1D>::
188  updateSlice(const Date &d) const {
189 
190  if (!lastDateisSet_ || d != lastDate_ ) {
191  // patch 2 for QL class:
192  // add observation lag, this is subtracted again in the stripper
193  Date d_eff = d + capFloorPrices_->observationLag();
194  // patch 3 for QL class:
195  // flat extrapolation in date direction, if extrapolation is enabled
196  if (this->allowsExtrapolation())
197  d_eff = std::min(d_eff, maxDate());
198  slice_ = yoyOptionletStripper_->slice(d_eff);
199 
200  tempKinterpolation_ =
201  factory1D_.interpolate( slice_.first.begin(),
202  slice_.first.end(),
203  slice_.second.begin() );
204  lastDateisSet_ = true;
205  lastDate_ = d;
206  }
207  }
