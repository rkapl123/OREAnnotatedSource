50  flatTenor_(flatTenor ? *flatTenor : flatIndex_->tenor()),

52  flatGearing_(flatGearing), spreadGearing_(spreadGearing), flatCalendar_(flatCalendar),
53  spreadCalendar_(spreadCalendar), spotFXSettleDaysVec_(spotFXSettleDaysVec),
54  spotFXSettleCalendarVec_(spotFXSettleCalendarVec), paymentLag_(paymentLag), flatPaymentLag_(flatPaymentLag),
55  includeSpread_(includeSpread), lookback_(lookback), fixingDays_(fixingDays), rateCutoff_(rateCutoff),
56  isAveraged_(isAveraged), flatIncludeSpread_(flatIncludeSpread), flatLookback_(flatLookback),
57  flatFixingDays_(flatFixingDays), flatRateCutoff_(flatRateCutoff), flatIsAveraged_(flatIsAveraged),
58  telescopicValueDates_(telescopicValueDates) {
59 
60  flatLegCurrency_ = flatIndex_->currency();
61  spreadLegCurrency_ = spreadIndex_->currency();
62 
63  bool flatIndexHasCurve = !flatIndex_->forwardingTermStructure().empty();
64  bool spreadIndexHasCurve = !spreadIndex_->forwardingTermStructure().empty();
65  bool haveFlatDiscountCurve = !flatDiscountCurve_.empty();
66  bool haveSpreadDiscountCurve = !spreadDiscountCurve_.empty();
67 
68  QL_REQUIRE(!(flatIndexHasCurve && spreadIndexHasCurve && haveFlatDiscountCurve && haveSpreadDiscountCurve),
69  "Have all curves, "
70  "nothing to solve for.");
71 
72  if (flatCalendar_.empty())
73  flatCalendar_ = settlementCalendar;
74  if (spreadCalendar_.empty())
75  spreadCalendar_ = settlementCalendar;
76 
77  // check spotFXSettleDaysVec_ and spotFXSettleCalendarVec_
78  Size numSpotFXSettleDays = spotFXSettleDaysVec_.size();
79  QL_REQUIRE(numSpotFXSettleDays == spotFXSettleCalendarVec_.size(),
80  "Array size of spot fx settlement days must equal that of spot fx settlement calendars");
81  if (numSpotFXSettleDays == 0) {
82  spotFXSettleDaysVec_.resize(1, 0);
83  spotFXSettleCalendarVec_.resize(1, settlementCalendar);
84  }
85 
86  /* Link the curve being bootstrapped to the index if the index has
87  no projection curve */
88  if (flatIndexHasCurve && haveFlatDiscountCurve) {
89  if (!spreadIndexHasCurve) {
90  spreadIndex_ = spreadIndex_->clone(termStructureHandle_);
91  spreadIndex_->unregisterWith(termStructureHandle_);
92  }
93  } else if (spreadIndexHasCurve && haveSpreadDiscountCurve) {
94  if (!flatIndexHasCurve) {
95  flatIndex_ = flatIndex_->clone(termStructureHandle_);
96  flatIndex_->unregisterWith(termStructureHandle_);
97  }
98  } else {
99  QL_FAIL("Need one leg of the cross currency basis swap to "
100  "have all of its curves.");
101  }
102 
103  registerWith(spotFX_);
104  registerWith(flatIndex_);
105  registerWith(spreadIndex_);
106  registerWith(flatDiscountCurve_);
107  registerWith(spreadDiscountCurve_);
108 
109  initializeDates();
110 }

112 void CrossCcyBasisSwapHelper::initializeDates() {
113 
114  Date refDate = evaluationDate_;
115  // if the evaluation date is not a business day
116  // then move to the next business day
117  refDate = settlementCalendar_.adjust(refDate);
118 
119  Date settlementDate = settlementCalendar_.advance(refDate, settlementDays_, Days);
120  Date maturityDate = settlementDate + swapTenor_;
121 
122  // calc spotFXSettleDate
123  Date spotFXSettleDate = refDate;
124  Size numSpotFXSettleDays = spotFXSettleDaysVec_.size(); // guaranteed to be at least 1
125  for (Size i = 0; i < numSpotFXSettleDays; i++) {
126  // Guaranteed here that spotFXSettleDaysVec_ and spotFXSettleCalendarVec_ have the same size
127  spotFXSettleDate = spotFXSettleCalendarVec_[i].advance(spotFXSettleDate, spotFXSettleDaysVec_[i], Days);
128  }
129 
130  Schedule flatLegSchedule = MakeSchedule()
131  .from(settlementDate)
132  .to(maturityDate)
133  .withTenor(flatTenor_)
134  .withCalendar(flatCalendar_)
135  .withConvention(rollConvention_)
136  .endOfMonth(eom_);
137 
138  Schedule spreadLegSchedule = MakeSchedule()
139  .from(settlementDate)
140  .to(maturityDate)
141  .withTenor(spreadTenor_)
142  .withCalendar(spreadCalendar_)
143  .withConvention(rollConvention_)
144  .endOfMonth(eom_);
145 
146  Real flatLegNominal = 1.0;
147  Real spreadLegNominal = 1.0;
148  if (flatIsDomestic_) {
149  flatLegNominal = spotFX_->value();
150  } else {
151  spreadLegNominal = spotFX_->value();
152  }
153 
154  /* Arbitrarily set the spread leg as the pay leg */
155  swap_ = boost::make_shared<CrossCcyBasisSwap>(
156  spreadLegNominal, spreadLegCurrency_, spreadLegSchedule, spreadIndex_, 0.0, spreadGearing_, flatLegNominal,
157  flatLegCurrency_, flatLegSchedule, flatIndex_, spreadOnFlatLeg_, flatGearing_, paymentLag_, flatPaymentLag_,
158  includeSpread_, lookback_, fixingDays_, rateCutoff_, isAveraged_, flatIncludeSpread_, flatLookback_,
159  flatFixingDays_, flatRateCutoff_, flatIsAveraged_, telescopicValueDates_);
160 
161  boost::shared_ptr<PricingEngine> engine;
162  if (flatIsDomestic_) {
163  engine = boost::make_shared<CrossCcySwapEngine>(flatLegCurrency_, flatDiscountRLH_, spreadLegCurrency_,
164  spreadDiscountRLH_, spotFX_, boost::none, Date(), Date(),
165  spotFXSettleDate);
166  } else {
167  engine = boost::make_shared<CrossCcySwapEngine>(spreadLegCurrency_, spreadDiscountRLH_, flatLegCurrency_,
168  flatDiscountRLH_, spotFX_, boost::none, Date(), Date(),
169  spotFXSettleDate);
170  }
171  swap_->setPricingEngine(engine);
172 
173  earliestDate_ = swap_->startDate();
174  latestDate_ = swap_->maturityDate();
175 
176 /* May need to adjust latestDate_ if you are projecting libor based
177  on tenor length rather than from accrual date to accrual date. */
178 #ifdef QL_USE_INDEXED_COUPON
179  if (termStructureHandle_ == spreadIndex_->forwardingTermStructure()) {
180  Size numCashflows = swap_->leg(0).size();
181  if (numCashflows > 2) {
182  boost::shared_ptr<FloatingRateCoupon> lastFloating =
183  boost::dynamic_pointer_cast<FloatingRateCoupon>(swap_->leg(0)[numCashflows - 2]);
184  Date fixingValueDate = spreadIndex_->valueDate(lastFloating->fixingDate());
185  Date endValueDate = spreadIndex_->maturityDate(fixingValueDate);
186  latestDate_ = std::max(latestDate_, endValueDate);
187  }
188  }
189  if (termStructureHandle_ == flatIndex_->forwardingTermStructure()) {
190  Size numCashflows = swap_->leg(1).size();
191  if (numCashflows > 2) {
192  boost::shared_ptr<FloatingRateCoupon> lastFloating =
193  boost::dynamic_pointer_cast<FloatingRateCoupon>(swap_->leg(1)[numCashflows - 2]);
194  Date fixingValueDate = flatIndex_->valueDate(lastFloating->fixingDate());
195  Date endValueDate = flatIndex_->maturityDate(fixingValueDate);
196  latestDate_ = std::max(latestDate_, endValueDate);
197  }
198  }
199 #endif
200 }

202 void CrossCcyBasisSwapHelper::setTermStructure(YieldTermStructure* t) {
203 
204  bool observer = false;
205  boost::shared_ptr<YieldTermStructure> temp(t, no_deletion);
206 
207  termStructureHandle_.linkTo(temp, observer);
208 
209  if (flatDiscountCurve_.empty())
210  flatDiscountRLH_.linkTo(temp, observer);
211  else
212  flatDiscountRLH_.linkTo(*flatDiscountCurve_, observer);
213 
214  if (spreadDiscountCurve_.empty())
215  spreadDiscountRLH_.linkTo(temp, observer);
216  else
217  spreadDiscountRLH_.linkTo(*spreadDiscountCurve_, observer);
218 
219  RelativeDateRateHelper::setTermStructure(t);
220 }
