27 CompoEquityIndex::CompoEquityIndex(const boost::shared_ptr<EquityIndex>& source,

30  JointCalendar(source->fixingCalendar(), fxIndex->fixingCalendar()), fxIndex->targetCurrency(),

53 void CompoEquityIndex::performCalculations() const {
54  dividendFixings_ = TimeSeries<Real>();
55  auto const& ts = source_->dividendFixings();
56  for (auto const& d : ts) {
57  dividendFixings_[d.first] = d.second * fxIndex_->fixing(fxIndex_->fixingCalendar().adjust(d.first, Preceding));
58  }
59 }
