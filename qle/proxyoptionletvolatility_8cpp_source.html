37Real getOisAtmLevel(const boost::shared_ptr<OvernightIndex>& on, const Date& fixingDate,
38 const Period& rateComputationPeriod) {
39 Date today = Settings::instance().evaluationDate();
40 Date start = on->valueDate(fixingDate);
41 Date end = on->fixingCalendar().advance(start, rateComputationPeriod);
42 Date adjStart = std::max(start, today);
43 Date adjEnd = std::max(adjStart + 1, end);
44 OvernightIndexedCoupon cpn(end, 1.0, adjStart, adjEnd, on);
45 cpn.setPricer(boost::make_shared<OvernightIndexedCouponPricer>());
46 return cpn.rate();
47}

51ProxyOptionletVolatility::ProxyOptionletVolatility(const Handle<OptionletVolatilityStructure>& baseVol,
52 const boost::shared_ptr<IborIndex>& baseIndex,
53 const boost::shared_ptr<IborIndex>& targetIndex,
54 const Period& baseRateComputationPeriod,
55 const Period& targetRateComputationPeriod)
56 : OptionletVolatilityStructure(baseVol->businessDayConvention(), baseVol->dayCounter()), baseVol_(baseVol),
57 baseIndex_(baseIndex), targetIndex_(targetIndex), baseRateComputationPeriod_(baseRateComputationPeriod),
58 targetRateComputationPeriod_(targetRateComputationPeriod) {
59
60 QL_REQUIRE(baseIndex != nullptr, "ProxyOptionletVolatility: no base index given.");
61 QL_REQUIRE(targetIndex != nullptr, "ProxyOptionletVolatility: no target index given.");
62 QL_REQUIRE(boost::dynamic_pointer_cast<QuantLib::OvernightIndex>(targetIndex_) == nullptr ||
63 targetRateComputationPeriod != 0 * Days,
64 "ProxyOptionletVolatility: target index is OIS ("
65 << targetIndex->name() << "), so targetRateComputationPeriod must be given and != 0D.");
66 QL_REQUIRE(boost::dynamic_pointer_cast<QuantLib::OvernightIndex>(baseIndex_) == nullptr ||
67 baseRateComputationPeriod != 0 * Days,
68 "ProxyOptionletVolatility: base index is OIS ("
69 << baseIndex->name() << "), so baseRateComputationPeriod must be given and != 0D.");
70 registerWith(baseVol_);
71 registerWith(baseIndex_);
72 registerWith(targetIndex_);
73 enableExtrapolation(baseVol->allowsExtrapolation());
74}

82boost::shared_ptr<SmileSection> ProxyOptionletVolatility::smileSectionImpl(const Date& fixingDate) const {
83
84 // compute the base and target forward rate levels
85
86 Real baseAtmLevel = isOis(baseIndex_) ? getOisAtmLevel(boost::dynamic_pointer_cast<OvernightIndex>(baseIndex_),
87 baseIndex_->fixingCalendar().adjust(fixingDate), baseRateComputationPeriod_)
88 : baseIndex_->fixing(baseIndex_->fixingCalendar().adjust(fixingDate));
89 Real targetAtmLevel = isOis(targetIndex_)
90 ? getOisAtmLevel(boost::dynamic_pointer_cast<OvernightIndex>(targetIndex_),
91 targetIndex_->fixingCalendar().adjust(fixingDate), targetRateComputationPeriod_)
92 : targetIndex_->fixing(targetIndex_->fixingCalendar().adjust(fixingDate));
93
94 // build the atm-adjusted smile section and return it
95
96 QL_REQUIRE(!baseVol_.empty(), "ProxyOptionletVolatility: no base vol given.");
97 return boost::make_shared<AtmAdjustedSmileSection>(baseVol_->smileSection(fixingDate, true), baseAtmLevel,
98 targetAtmLevel);
99}
