31 class FairShortSpreadHelper {
32 public:
33  FairShortSpreadHelper(const TenorBasisSwap& swap, const Handle<YieldTermStructure>& discountCurve, Real longLegNPV)
34  : longLegNPV_(longLegNPV) {
35 
36  engine_ = boost::shared_ptr<PricingEngine>(new DiscountingSwapEngine(discountCurve));
37  arguments_ = dynamic_cast<Swap::arguments*>(engine_->getArguments());
38  swap.setupArguments(arguments_);
39  shortNo_ = swap.payLongIndex();
40  shortLeg_ = arguments_->legs[shortNo_];
41  results_ = dynamic_cast<const Swap::results*>(engine_->getResults());
42  }
43 
44  Real operator()(Spread shortSpread) const {
45  // Change the spread on the leg and recalculate
46  Leg::const_iterator it;
47  for (it = shortLeg_.begin(); it != shortLeg_.end(); ++it) {
48  boost::shared_ptr<QuantExt::SubPeriodsCoupon1> c = boost::dynamic_pointer_cast<QuantExt::SubPeriodsCoupon1>(*it);
49  c->spread() = shortSpread;
50  }
51  engine_->calculate();
52  return results_->legNPV[shortNo_] + longLegNPV_;
53  }
54 
55 private:
56  boost::shared_ptr<PricingEngine> engine_;
57  Real longLegNPV_;
58  const Swap::results* results_;
59  Swap::arguments* arguments_;
60  Size shortNo_;
61  Leg shortLeg_;
62 };

65 TenorBasisSwap::TenorBasisSwap(const Date& effectiveDate, Real nominal, const Period& swapTenor, bool payLongIndex,
66  const boost::shared_ptr<IborIndex>& longIndex, Spread longSpread,
67  const boost::shared_ptr<IborIndex>& shortIndex, Spread shortSpread,
68  const Period& shortPayTenor, DateGeneration::Rule rule, bool includeSpread,
69  QuantExt::SubPeriodsCoupon1::Type type)
70  : Swap(2), nominal_(nominal), payLongIndex_(payLongIndex), longIndex_(longIndex), longSpread_(longSpread),
71  shortIndex_(shortIndex), shortSpread_(shortSpread), shortPayTenor_(shortPayTenor), includeSpread_(includeSpread),
72  type_(type) {
73 
74  // Checks
75  Period longTenor = longIndex_->tenor();
76  QL_REQUIRE(shortPayTenor_ >= shortIndex_->tenor(), "Expected short payment tenor to exceed/equal shortIndex tenor");
77  QL_REQUIRE(shortPayTenor_ <= longTenor, "Expected short payment tenor to be at most longSchedule tenor");
78 
79  // Create the default long and short schedules
80  Date terminationDate = effectiveDate + swapTenor;
81 
82  boost::shared_ptr<Libor> longIndexAsLibor = boost::dynamic_pointer_cast<Libor>(longIndex_);
83  longIndexCalendar_ = longIndexAsLibor != NULL ? longIndexAsLibor->jointCalendar() : longIndex_->fixingCalendar();
84  boost::shared_ptr<Libor> shortIndexAsLibor = boost::dynamic_pointer_cast<Libor>(shortIndex_);
85  shortIndexCalendar_ =
86  shortIndexAsLibor != NULL ? shortIndexAsLibor->jointCalendar() : shortIndex_->fixingCalendar();
87 
88  longSchedule_ = MakeSchedule()
89  .from(effectiveDate)
90  .to(terminationDate)
91  .withTenor(longTenor)
92  .withCalendar(longIndexCalendar_)
93  .withConvention(longIndex_->businessDayConvention())
94  .withTerminationDateConvention(longIndex_->businessDayConvention())
95  .withRule(rule)
96  .endOfMonth(longIndex_->endOfMonth());
97 
98  shortSchedule_ = MakeSchedule()
99  .from(effectiveDate)
100  .to(terminationDate)
101  .withTenor(shortPayTenor_)
102  .withCalendar(shortIndexCalendar_)
103  .withConvention(shortIndex_->businessDayConvention())
104  .withTerminationDateConvention(shortIndex_->businessDayConvention())
105  .withRule(rule)
106  .endOfMonth(shortIndex_->endOfMonth());
107 
108  // Create legs
109  initializeLegs();
110 }

131 void TenorBasisSwap::initializeLegs() {
132 
133  // Long leg
134  BusinessDayConvention longPmtConvention = longIndex_->businessDayConvention();
135  DayCounter longDayCounter = longIndex_->dayCounter();
136  Leg longLeg = IborLeg(longSchedule_, longIndex_)
137  .withNotionals(nominal_)
138  .withSpreads(longSpread_)
139  .withPaymentAdjustment(longPmtConvention)
140  .withPaymentDayCounter(longDayCounter)
141  .withPaymentCalendar(longIndexCalendar_);
142 
143  // Short leg
144  Leg shortLeg;
145  BusinessDayConvention shortPmtConvention = shortIndex_->businessDayConvention();
146  DayCounter shortDayCounter = shortIndex_->dayCounter();
147  if (shortPayTenor_ == shortIndex_->tenor()) {
148  shortLeg = IborLeg(shortSchedule_, shortIndex_)
149  .withNotionals(nominal_)
150  .withSpreads(shortSpread_)
151  .withPaymentAdjustment(shortPmtConvention)
152  .withPaymentDayCounter(shortDayCounter)
153  .withPaymentCalendar(shortIndexCalendar_);
154  } else {
155  shortLeg = QuantExt::SubPeriodsLeg1(shortSchedule_, shortIndex_)
156  .withNotional(nominal_)
157  .withSpread(shortSpread_)
158  .withPaymentAdjustment(shortPmtConvention)
159  .withPaymentDayCounter(shortDayCounter)
160  .withPaymentCalendar(shortIndexCalendar_)
161  .includeSpread(includeSpread_)
162  .withType(type_);
163  }
164 
165  // Pay (Rec) leg is legs_[0] (legs_[1])
166  payer_[0] = -1.0;
167  payer_[1] = +1.0;
168 
169  longNo_ = !payLongIndex_;
170  shortNo_ = payLongIndex_;
171  legs_[longNo_] = longLeg;
172  legs_[shortNo_] = shortLeg;
173 
174  // Register this instrument with its coupons
175  Leg::const_iterator it;
176  for (it = legs_[0].begin(); it != legs_[0].end(); ++it)
177  registerWith(*it);
178  for (it = legs_[1].begin(); it != legs_[1].end(); ++it)
179  registerWith(*it);
180 }

224 void TenorBasisSwap::fetchResults(const PricingEngine::results* r) const {
225 
226  static const Spread basisPoint = 1.0e-4;
227  Swap::fetchResults(r);
228 
229  const TenorBasisSwap::results* results = dynamic_cast<const TenorBasisSwap::results*>(r);
230 
231  if (results) {
232  fairLongSpread_ = results->fairLongSpread;
233  fairShortSpread_ = results->fairShortSpread;
234  } else {
235  fairLongSpread_ = Null<Spread>();
236  fairShortSpread_ = Null<Spread>();
237  }
238 
239  // Long fair spread should be fine - no averaging or compounding
240  if (fairLongSpread_ == Null<Spread>()) {
241  if (legBPS_[longNo_] != Null<Real>()) {
242  fairLongSpread_ = longSpread_ - NPV_ / (legBPS_[longNo_] / basisPoint);
243  }
244  }
245 
246  /* Short fair spread calculation ok if no averaging/compounding OR
247  if there is averaging/compounding and the spread is added after */
248  if (fairShortSpread_ == Null<Spread>()) {
249  if (shortPayTenor_ == shortIndex_->tenor() || !includeSpread_) {
250  if (legBPS_[shortNo_] != Null<Real>()) {
251  fairShortSpread_ = shortSpread_ - NPV_ / (legBPS_[shortNo_] / basisPoint);
252  }
253  } else {
254  // Need the discount curve
255  Handle<YieldTermStructure> discountCurve;
256  boost::shared_ptr<DiscountingSwapEngine> engine =
257  boost::dynamic_pointer_cast<DiscountingSwapEngine>(engine_);
258  if (engine) {
259  discountCurve = engine->discountCurve();
260  // Calculate a guess
261  Spread guess = 0.0;
262  if (legBPS_[shortNo_] != Null<Real>()) {
263  guess = shortSpread_ - NPV_ / (legBPS_[shortNo_] / basisPoint);
264  }
265  // Attempt to solve for fair spread
266  Spread step = 1e-4;
267  Real accuracy = 1e-8;
268  FairShortSpreadHelper f(*this, discountCurve, legNPV_[longNo_]);
269  Brent solver;
270  fairShortSpread_ = solver.solve(f, accuracy, guess, step);
271  }
272  }
273  }
