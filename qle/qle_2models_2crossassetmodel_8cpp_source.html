182 void CrossAssetModel::correlation(const AssetType s, const Size i, const AssetType t, const Size j, const Real value,
183  const Size iOffset, const Size jOffset) {
184  Size row = cIdx(s, i, iOffset);
185  Size column = cIdx(t, j, jOffset);
186  QL_REQUIRE(row != column || close_enough(value, 1.0), "correlation must be 1 at (" << row << "," << column << ")");
187  QL_REQUIRE(value >= -1.0 && value <= 1.0, "correlation must be in [-1,1] at (" << row << "," << column << ")");
188  // we can not check for non-negative eigenvalues, since we do not
189  // know when the correlation matrix setup is finished, but this
190  // is effectively one in the state process later on anyway and
191  // the user can also use checkCorrelationMatrix() to verify this
192  rho_(row, column) = rho_(column, row) = value;
193  update();
194 }

215 void CrossAssetModel::setIntegrationPolicy(const boost::shared_ptr<Integrator> integrator,
216  const bool usePiecewiseIntegration) const {
217 
218  if (!usePiecewiseIntegration) {
219  integrator_ = integrator;
220  return;
221  }
222 
223  // collect relevant times from parametrizations, we don't have to sort them or make them unique,
224  // this is all done in PiecewiseIntegral for us
225 
226  std::vector<Time> allTimes;
227  for (Size i = 0; i < p_.size(); ++i) {
228  for (Size j = 0; j < getNumberOfParameters(i); ++j)
229  allTimes.insert(allTimes.end(), p_[i]->parameterTimes(j).begin(), p_[i]->parameterTimes(j).end());
230  }
231 
232  // use piecewise integrator avoiding the step points
233  integrator_ = boost::make_shared<PiecewiseIntegral>(integrator, allTimes, true);
234 }

304 void CrossAssetModel::initializeParametrizations() {
305 
306  // count the parametrizations and check their order and their support
307 
308  Size i = 0, j;
309  Size cIdxTmp = 0, pIdxTmp = 0, aIdxTmp = 0;
310  components_.resize(crossAssetModelAssetTypes, 0);
311  idx_.resize(crossAssetModelAssetTypes);
312  cIdx_.resize(crossAssetModelAssetTypes);
313  pIdx_.resize(crossAssetModelAssetTypes);
314  aIdx_.resize(crossAssetModelAssetTypes);
315  brownians_.resize(crossAssetModelAssetTypes);
316  stateVariables_.resize(crossAssetModelAssetTypes);
317  numArguments_.resize(crossAssetModelAssetTypes);
318  modelType_.resize(crossAssetModelAssetTypes);
319 
320  // IR parametrizations
321 
322  bool genericCtor = lgm_.empty();
323  j = 0;
324  while (i < p_.size() && getComponentType(i).first == IR) {
325  // initialize lgm model, if generic constructor was used
326  if (genericCtor) {
327  if (getComponentType(i).second == LGM1F) {
328  lgm_.push_back(boost::make_shared<LinearGaussMarkovModel>(
329  boost::dynamic_pointer_cast<IrLgm1fParametrization>(p_[i])));
330  } else {
331  lgm_.push_back(boost::shared_ptr<LinearGaussMarkovModel>());
332  }
333  }
334  updateIndices(IR, i, cIdxTmp, pIdxTmp, aIdxTmp);
335  cIdxTmp += getNumberOfBrownians(i);
336  pIdxTmp += getNumberOfStateVariables(i);
337  aIdxTmp += getNumberOfParameters(i);
338  ++j;
339  ++i;
340  }
341  components_[IR] = j;
342 
343  // FX parametrizations
344 
345  j = 0;
346  while (i < p_.size() && getComponentType(i).first == FX) {
347  updateIndices(FX, i, cIdxTmp, pIdxTmp, aIdxTmp);
348  cIdxTmp += getNumberOfBrownians(i);
349  pIdxTmp += getNumberOfStateVariables(i);
350  aIdxTmp += getNumberOfParameters(i);
351  ++j;
352  ++i;
353  }
354  components_[FX] = j;
355 
356  QL_REQUIRE(components_[IR] > 0, "at least one ir parametrization must be given");
357 
358  QL_REQUIRE(components_[FX] == components_[IR] - 1, "there must be n-1 fx "
359  "for n ir parametrizations, found "
360  << components_[IR] << " ir and " << components_[FX]
361  << " fx parametrizations");
362 
363  // check currencies
364 
365  // without an order or a hash function on Currency this seems hard
366  // to do in a simpler way ...
367  Size uniqueCurrencies = 0;
368  std::vector<Currency> currencies;
369  for (Size i = 0; i < components_[IR]; ++i) {
370  Size tmp = 1;
371  for (Size j = 0; j < i; ++j) {
372  if (ir(i)->currency() == currencies[j])
373  tmp = 0;
374  }
375  uniqueCurrencies += tmp;
376  currencies.push_back(ir(i)->currency());
377  }
378  QL_REQUIRE(uniqueCurrencies == components_[IR], "there are duplicate currencies "
379  "in the set of ir "
380  "parametrizations");
381  for (Size i = 0; i < components_[FX]; ++i) {
382  QL_REQUIRE(fx(i)->currency() == ir(i + 1)->currency(),
383  "fx parametrization #" << i << " must be for currency of ir parametrization #" << (i + 1)
384  << ", but they are " << fx(i)->currency() << " and "
385  << irlgm1f(i + 1)->currency() << " respectively");
386  }
387 
388  // Inf parametrizations
389 
390  j = 0;
391  while (i < p_.size() && getComponentType(i).first == INF) {
392  updateIndices(INF, i, cIdxTmp, pIdxTmp, aIdxTmp);
393  cIdxTmp += getNumberOfBrownians(i);
394  pIdxTmp += getNumberOfStateVariables(i);
395  aIdxTmp += getNumberOfParameters(i);
396  ++j;
397  ++i;
398  // we do not check the currency, if not present among the model's
399  // currencies, it will throw below
400  }
401  components_[INF] = j;
402 
403  // Cr parametrizations
404 
405  j = 0;
406  while (i < p_.size() && getComponentType(i).first == CR) {
407 
408  if (getComponentType(i).second == CIRPP) {
409  auto tmp = boost::dynamic_pointer_cast<CrCirppParametrization>(p_[i]);
410  QL_REQUIRE(tmp, "CrossAssetModelPlus::initializeParametrizations(): expected CrCirppParametrization");
411  crcirppModel_.push_back(boost::make_shared<CrCirpp>(tmp));
412  } else
413  crcirppModel_.push_back(boost::shared_ptr<CrCirpp>());
414 
415  updateIndices(CR, i, cIdxTmp, pIdxTmp, aIdxTmp);
416  cIdxTmp += getNumberOfBrownians(i);
417  pIdxTmp += getNumberOfStateVariables(i);
418  aIdxTmp += getNumberOfParameters(i);
419  ++j;
420  ++i;
421  // we do not check the currency, if not present among the model's
422  // currencies, it will throw below
423  }
424  components_[CR] = j;
425 
426  // Eq parametrizations
427 
428  j = 0;
429  while (i < p_.size() && getComponentType(i).first == EQ) {
430  updateIndices(EQ, i, cIdxTmp, pIdxTmp, aIdxTmp);
431  cIdxTmp += getNumberOfBrownians(i);
432  pIdxTmp += getNumberOfStateVariables(i);
433  aIdxTmp += getNumberOfParameters(i);
434  ++j;
435  ++i;
436  }
437  components_[EQ] = j;
438 
439  // check the equity currencies to ensure they are covered by CrossAssetModel
440  for (Size i = 0; i < components(EQ); ++i) {
441  Currency eqCcy = eq(i)->currency();
442  try {
443  Size eqCcyIdx = ccyIndex(eqCcy);
444  QL_REQUIRE(eqCcyIdx < components_[IR], "Invalid currency for equity " << eqbs(i)->name());
445  } catch (...) {
446  QL_FAIL("Invalid currency (" << eqCcy.code() << ") for equity " << eqbs(i)->name());
447  }
448  }
449 
450  if (measure_ == Measure::BA) {
451 
452  QL_REQUIRE(components_[INF] == 0, "CAM in BA measure does not support INF components yet");
453  QL_REQUIRE(components_[EQ] == 0, "CAM in BA measure does not support EQ components yet");
454  QL_REQUIRE(components_[CR] == 0, "CAM in BA measure does not support CR components yet");
455 
456  // AUX variable for BA measure simulations
457 
458  components_[AUX] = 1;
459  updateIndices(AUX, i, cIdxTmp, pIdxTmp, aIdxTmp);
460  cIdxTmp += 1;
461  pIdxTmp += 1;
462  }
463 
464  // Summary statistics
465 
466  totalDimension_ = pIdxTmp;
467  totalNumberOfBrownians_ = cIdxTmp;
468 
469 } // initParametrizations

485 void CrossAssetModel::checkCorrelationMatrix() const {
486  Size n = rho_.rows();
487  Size m = rho_.columns();
488  QL_REQUIRE(rho_.columns() == n, "correlation matrix (" << n << " x " << m << " must be square");
489  for (Size i = 0; i < n; ++i) {
490  for (Size j = 0; j < m; ++j) {
491  QL_REQUIRE(close_enough(rho_[i][j], rho_[j][i]), "correlation matrix is not symmetric, for (i,j)=("
492  << i << "," << j << ") rho(i,j)=" << rho_[i][j]
493  << " but rho(j,i)=" << rho_[j][i]);
494  QL_REQUIRE(close_enough(std::abs(rho_[i][j]), 1.0) || (rho_[i][j] > -1.0 && rho_[i][j] < 1.0),
495  "correlation matrix has invalid entry at (i,j)=(" << i << "," << j << ") equal to "
496  << rho_[i][j]);
497  }
498  QL_REQUIRE(close_enough(rho_[i][i], 1.0), "correlation matrix must have unit diagonal elements, "
499  "but rho(i,i)="
500  << rho_[i][i] << " for i=" << i);
501  }
502 
503  SymmetricSchurDecomposition ssd(rho_);
504  for (Size i = 0; i < ssd.eigenvalues().size(); ++i) {
505  QL_REQUIRE(ssd.eigenvalues()[i] >= 0.0,
506  "correlation matrix has negative eigenvalue at " << i << " (" << ssd.eigenvalues()[i] << ")");
507  }
508 }

691  QL_REQUIRE(t < T || close_enough(t, T), "infdkI: t (" << t << ") <= T (" << T << ") required");
692  Real V0, V_tilde;
693  std::pair<Real, Real> Vs = infdkV(i, t, T);
694  V0 = Vs.first;
695  V_tilde = Vs.second;
696  Real Hyt = Hy(i).eval(this, t);
697  Real HyT = Hy(i).eval(this, T);
698 
699  // TODO account for seasonality ...
700  // compute final results depending on z and y
701  const auto& zts = infdk(i)->termStructure();
702  auto dc = irlgm1f(0)->termStructure()->dayCounter();
703  Real growth_t = inflationGrowth(zts, t, dc);
704  Real It = growth_t * std::exp(Hyt * z - y - V0);
705  Real Itilde_t_T = inflationGrowth(zts, T, dc) / growth_t * std::exp((HyT - Hyt) * z + V_tilde);

707  // is not interpolated, we still simulate the index value as of t
708  // (and T), although we should go back to t, T which corresponds to

731 std::pair<Real, Real> CrossAssetModel::crlgm1fS(const Size i, const Size ccy, const Time t, const Time T, const Real z,
732  const Real y) const {
733  QL_REQUIRE(ccy < components(IR), "ccy index (" << ccy << ") must be in 0..." << (components(IR) - 1));
734  QL_REQUIRE(t < T || close_enough(t, T), "crlgm1fS: t (" << t << ") <= T (" << T << ") required");
735  QL_REQUIRE(modelType(CR, i) == LGM1F, "model at " << i << " is not CR-LGM1F");
736  cache_key k = {i, ccy, t, T};
737  boost::unordered_map<cache_key, std::pair<Real, Real>>::const_iterator it = cache_crlgm1fS_.find(k);
738  Real V0, V_tilde;
739  Real Hlt = Hl(i).eval(this, t);
740  Real HlT = Hl(i).eval(this, T);
741 
742  if (it == cache_crlgm1fS_.end()) {
743  // compute V0 and V_tilde
744  if (ccy == 0) {
745  // domestic credit
746  Real Hzt = Hz(0).eval(this, t);
747  Real HzT = Hz(0).eval(this, T);
748  Real zetal0 = zetal(i).eval(this, t);
749  Real zetal1 = integral(this, P(Hl(i), al(i), al(i)), 0.0, t);
750  Real zetal2 = integral(this, P(Hl(i), Hl(i), al(i), al(i)), 0.0, t);
751  Real zetanl0 = integral(this, P(rzl(0, i), az(0), al(i)), 0.0, t);
752  Real zetanl1 = integral(this, P(rzl(0, i), Hl(i), az(0), al(i)), 0.0, t);
753  // opposite signs for last two terms in the book
754  V0 = 0.5 * Hlt * Hlt * zetal0 - Hlt * zetal1 + 0.5 * zetal2 + Hzt * Hlt * zetanl0 - Hzt * zetanl1;
755  V_tilde = -0.5 * (HlT * HlT - Hlt * Hlt) * zetal0 + (HlT - Hlt) * zetal1 -
756  (HzT * HlT - Hzt * Hlt) * zetanl0 + (HzT - Hzt) * zetanl1;
757  } else {
758  // foreign credit
759  V0 = crV(i, ccy, 0, t);
760  V_tilde = crV(i, ccy, t, T) - crV(i, ccy, 0, T) + crV(i, ccy, 0, t);
761  }
762  cache_crlgm1fS_.insert(std::make_pair(k, std::make_pair(V0, V_tilde)));
763  } else {
764  // take V0 and V_tilde from cache
765  V0 = it->second.first;
766  V_tilde = it->second.second;
767  }
768  // compute final results depending on z and y
769  // opposite sign for V0 in the book
770  Real St = crlgm1f(i)->termStructure()->survivalProbability(t) * std::exp(-Hlt * z + y - V0);
771  Real Stilde_t_T = crlgm1f(i)->termStructure()->survivalProbability(T) /
772  crlgm1f(i)->termStructure()->survivalProbability(t) * std::exp(-(HlT - Hlt) * z + V_tilde);
773  return std::make_pair(St, Stilde_t_T);
774 }
