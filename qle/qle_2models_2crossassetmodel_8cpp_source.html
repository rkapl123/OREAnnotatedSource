241void CrossAssetModel::correlation(const AssetType s, const Size i, const AssetType t, const Size j, const Real value,
242 const Size iOffset, const Size jOffset) {
243 Size row = cIdx(s, i, iOffset);
244 Size column = cIdx(t, j, jOffset);
245 QL_REQUIRE(row != column || close_enough(value, 1.0), "correlation must be 1 at (" << row << "," << column << ")");
246 QL_REQUIRE(value >= -1.0 && value <= 1.0, "correlation must be in [-1,1] at (" << row << "," << column << ")");
247 // we can not check for non-negative eigenvalues, since we do not
248 // know when the correlation matrix setup is finished, but this
249 // is effectively one in the state process later on anyway and
250 // the user can also use checkCorrelationMatrix() to verify this
251 rho_(row, column) = rho_(column, row) = value;
252 update();
253}

271void CrossAssetModel::setIntegrationPolicy(const boost::shared_ptr<Integrator> integrator,
272 const bool usePiecewiseIntegration) const {
273
274 if (!usePiecewiseIntegration) {
275 integrator_ = integrator;
276 return;
277 }
278
279 // collect relevant times from parametrizations, we don't have to sort them or make them unique,
280 // this is all done in PiecewiseIntegral for us
281
282 std::vector<Time> allTimes;
283 for (Size i = 0; i < p_.size(); ++i) {
284 for (Size j = 0; j < getNumberOfParameters(i); ++j)
285 allTimes.insert(allTimes.end(), p_[i]->parameterTimes(j).begin(), p_[i]->parameterTimes(j).end());
286 }
287
288 // use piecewise integrator avoiding the step points
289 integrator_ = boost::make_shared<PiecewiseIntegral>(integrator, allTimes, true);
290}

416void CrossAssetModel::initializeParametrizations() {
417
418 // count the parametrizations and check their order and their support
419
420 Size i = 0, j;
421 Size cIdxTmp = 0, wIdxTmp = 0, pIdxTmp = 0, aIdxTmp = 0;
422 components_.resize(numberOfAssetTypes, 0);
423 idx_.resize(numberOfAssetTypes);
424 cIdx_.resize(numberOfAssetTypes);
425 wIdx_.resize(numberOfAssetTypes);
426 pIdx_.resize(numberOfAssetTypes);
427 aIdx_.resize(numberOfAssetTypes);
428 brownians_.resize(numberOfAssetTypes);
429 auxBrownians_.resize(numberOfAssetTypes);
430 stateVariables_.resize(numberOfAssetTypes);
431 numArguments_.resize(numberOfAssetTypes);
432 modelType_.resize(numberOfAssetTypes);
433
434 // IR parametrizations
435
436 bool genericCtor = irModels_.empty();
437 j = 0;
438 while (i < p_.size() && getComponentType(i).first == CrossAssetModel::AssetType::IR) {
439 QL_REQUIRE(j == 0 || getComponentType(i).second == getComponentType(0).second,
440 "All IR models must be of the same type (HW, LGM can not be mixed)");
441 // initialize ir model, if generic constructor was used
442 // evaluate bank account for j = 0 (domestic process
443 if (genericCtor) {
444 if (getComponentType(i).second == ModelType::LGM1F) {
445 irModels_.push_back(boost::make_shared<LinearGaussMarkovModel>(
446 boost::dynamic_pointer_cast<IrLgm1fParametrization>(p_[i]), measure_,
447 getLgm1fDiscretization(discretization_), j == 0));
448 } else if (getComponentType(i).second == ModelType::HW) {
449 irModels_.push_back(boost::make_shared<HwModel>(boost::dynamic_pointer_cast<IrHwParametrization>(p_[i]),
450 measure_, getHwDiscretization(discretization_),
451 j == 0));
452 } else {
453 irModels_.push_back(nullptr);
454 }
455 }
456 updateIndices(CrossAssetModel::AssetType::IR, i, cIdxTmp, wIdxTmp, pIdxTmp, aIdxTmp);
457 cIdxTmp += getNumberOfBrownians(i);
458 wIdxTmp += getNumberOfBrownians(i) + getNumberOfAuxBrownians(i);
459 pIdxTmp += getNumberOfStateVariables(i);
460 aIdxTmp += getNumberOfParameters(i);
461 ++j;
462 ++i;
463 }
464 components_[(Size)CrossAssetModel::AssetType::IR] = j;
465
466 // FX parametrizations
467
468 j = 0;
469 while (i < p_.size() && getComponentType(i).first == CrossAssetModel::AssetType::FX) {
470 fxModels_.push_back(boost::make_shared<FxBsModel>(boost::dynamic_pointer_cast<FxBsParametrization>(p_[i])));
471 updateIndices(CrossAssetModel::AssetType::FX, i, cIdxTmp, wIdxTmp, pIdxTmp, aIdxTmp);
472 cIdxTmp += getNumberOfBrownians(i);
473 wIdxTmp += getNumberOfBrownians(i) + getNumberOfAuxBrownians(i);
474 pIdxTmp += getNumberOfStateVariables(i);
475 aIdxTmp += getNumberOfParameters(i);
476 ++j;
477 ++i;
478 }
479 components_[(Size)CrossAssetModel::AssetType::FX] = j;
480
481 QL_REQUIRE(components_[(Size)CrossAssetModel::AssetType::IR] > 0, "at least one ir parametrization must be given");
482
483 QL_REQUIRE(components_[(Size)CrossAssetModel::AssetType::FX] ==
484 components_[(Size)CrossAssetModel::AssetType::IR] - 1,
485 "there must be n-1 fx "
486 "for n ir parametrizations, found "
487 << components_[(Size)CrossAssetModel::AssetType::IR] << " ir and "
488 << components_[(Size)CrossAssetModel::AssetType::FX] << " fx parametrizations");
489
490 // check currencies
491
492 // without an order or a hash function on Currency this seems hard
493 // to do in a simpler way ...
494 Size uniqueCurrencies = 0;
495 std::vector<Currency> currencies;
496 for (Size i = 0; i < components_[(Size)CrossAssetModel::AssetType::IR]; ++i) {
497 Size tmp = 1;
498 for (Size j = 0; j < i; ++j) {
499 if (ir(i)->currency() == currencies[j])
500 tmp = 0;
501 }
502 uniqueCurrencies += tmp;
503 currencies.push_back(ir(i)->currency());
504 }
505 QL_REQUIRE(uniqueCurrencies == components_[(Size)CrossAssetModel::AssetType::IR], "there are duplicate currencies "
506 "in the set of ir "
507 "parametrizations");
508 for (Size i = 0; i < components_[(Size)CrossAssetModel::AssetType::FX]; ++i) {
509 QL_REQUIRE(fx(i)->currency() == ir(i + 1)->currency(),
510 "fx parametrization #" << i << " must be for currency of ir parametrization #" << (i + 1)
511 << ", but they are " << fx(i)->currency() << " and " << ir(i + 1)->currency()
512 << " respectively");
513 }
514
515 // Inf parametrizations
516
517 j = 0;
518 while (i < p_.size() && getComponentType(i).first == CrossAssetModel::AssetType::INF) {
519 updateIndices(CrossAssetModel::AssetType::INF, i, cIdxTmp, wIdxTmp, pIdxTmp, aIdxTmp);
520 cIdxTmp += getNumberOfBrownians(i);
521 wIdxTmp += getNumberOfBrownians(i) + getNumberOfAuxBrownians(i);
522 pIdxTmp += getNumberOfStateVariables(i);
523 aIdxTmp += getNumberOfParameters(i);
524 ++j;
525 ++i;
526 // we do not check the currency, if not present among the model's
527 // currencies, it will throw below
528 }
529 components_[(Size)CrossAssetModel::AssetType::INF] = j;
530
531 // Cr parametrizations
532
533 j = 0;
534 while (i < p_.size() && getComponentType(i).first == CrossAssetModel::AssetType::CR) {
535
536 if (getComponentType(i).second == CrossAssetModel::ModelType::CIRPP) {
537 auto tmp = boost::dynamic_pointer_cast<CrCirppParametrization>(p_[i]);
538 QL_REQUIRE(tmp, "CrossAssetModelPlus::initializeParametrizations(): expected CrCirppParametrization");
539 crcirppModel_.push_back(boost::make_shared<CrCirpp>(tmp));
540 } else
541 crcirppModel_.push_back(boost::shared_ptr<CrCirpp>());
542
543 updateIndices(CrossAssetModel::AssetType::CR, i, cIdxTmp, wIdxTmp, pIdxTmp, aIdxTmp);
544 cIdxTmp += getNumberOfBrownians(i);
545 wIdxTmp += getNumberOfBrownians(i) + getNumberOfAuxBrownians(i);
546 pIdxTmp += getNumberOfStateVariables(i);
547 aIdxTmp += getNumberOfParameters(i);
548 ++j;
549 ++i;
550 // we do not check the currency, if not present among the model's
551 // currencies, it will throw below
552 }
553 components_[(Size)CrossAssetModel::AssetType::CR] = j;
554
555 // Eq parametrizations
556
557 j = 0;
558 while (i < p_.size() && getComponentType(i).first == CrossAssetModel::AssetType::EQ) {
559 updateIndices(CrossAssetModel::AssetType::EQ, i, cIdxTmp, wIdxTmp, pIdxTmp, aIdxTmp);
560 cIdxTmp += getNumberOfBrownians(i);
561 wIdxTmp += getNumberOfBrownians(i) + getNumberOfAuxBrownians(i);
562 pIdxTmp += getNumberOfStateVariables(i);
563 aIdxTmp += getNumberOfParameters(i);
564 ++j;
565 ++i;
566 }
567 components_[(Size)CrossAssetModel::AssetType::EQ] = j;
568
569 // check the equity currencies to ensure they are covered by CrossAssetModel
570 for (Size i = 0; i < components(CrossAssetModel::AssetType::EQ); ++i) {
571 Currency eqCcy = eq(i)->currency();
572 try {
573 Size eqCcyIdx = ccyIndex(eqCcy);
574 QL_REQUIRE(eqCcyIdx < components_[(Size)CrossAssetModel::AssetType::IR],
575 "Invalid currency for equity " << eqbs(i)->name());
576 } catch (...) {
577 QL_FAIL("Invalid currency (" << eqCcy.code() << ") for equity " << eqbs(i)->name());
578 }
579 }
580
581 // COM parametrizations
582
583 j = 0;
584 while (i < p_.size() && getComponentType(i).first == CrossAssetModel::AssetType::COM) {
585 boost::shared_ptr<CommoditySchwartzParametrization> csp = boost::dynamic_pointer_cast<CommoditySchwartzParametrization>(p_[i]);
586 boost::shared_ptr<CommoditySchwartzModel> csm =
587 csp ? boost::make_shared<CommoditySchwartzModel>(csp, getComSchwartzDiscretization(discretization_))
588 : nullptr;
589 comModels_.push_back(csm);
590 updateIndices(CrossAssetModel::AssetType::COM, i, cIdxTmp, wIdxTmp, pIdxTmp, aIdxTmp);
591 cIdxTmp += getNumberOfBrownians(i);
592 wIdxTmp += getNumberOfBrownians(i) + getNumberOfAuxBrownians(i);
593 pIdxTmp += getNumberOfStateVariables(i);
594 aIdxTmp += getNumberOfParameters(i);
595 ++j;
596 ++i;
597 }
598 components_[(Size)CrossAssetModel::AssetType::COM] = j;
599
600 // check the equity currencies to ensure they are covered by CrossAssetModel
601 for (Size i = 0; i < components(CrossAssetModel::AssetType::COM); ++i) {
602 Currency comCcy = com(i)->currency();
603 try {
604 Size comCcyIdx = ccyIndex(comCcy);
605 QL_REQUIRE(comCcyIdx < components_[(Size)CrossAssetModel::AssetType::IR],
606 "Invalid currency for commodity " << combs(i)->name());
607 } catch (...) {
608 QL_FAIL("Invalid currency (" << comCcy.code() << ") for commodity " << combs(i)->name());
609 }
610 }
611
612 // Summary statistics
613
614 totalDimension_ = pIdxTmp;
615 totalNumberOfBrownians_ = cIdxTmp;
616 totalNumberOfAuxBrownians_ = wIdxTmp - cIdxTmp;
617
618} // initParametrizations

620void CrossAssetModel::initializeCorrelation() {
621 Size n = brownians();
622 if (rho_.empty()) {
623 rho_ = Matrix(n, n, 0.0);
624 for (Size i = 0; i < n; ++i)
625 rho_(i, i) = 1.0;
626 return;
627 }
628 QL_REQUIRE(rho_.rows() == n && rho_.columns() == n, "correlation matrix is " << rho_.rows() << " x "
629 << rho_.columns() << " but should be "
630 << n << " x " << n);
631 checkCorrelationMatrix();
632}
633
634void CrossAssetModel::checkCorrelationMatrix() const {
635 Size n = rho_.rows();
636 Size m = rho_.columns();
637 QL_REQUIRE(rho_.columns() == n, "correlation matrix (" << n << " x " << m << " must be square");
638 for (Size i = 0; i < n; ++i) {
639 for (Size j = 0; j < m; ++j) {
640 QL_REQUIRE(close_enough(rho_[i][j], rho_[j][i]), "correlation matrix is not symmetric, for (i,j)=("
641 << i << "," << j << ") rho(i,j)=" << rho_[i][j]
642 << " but rho(j,i)=" << rho_[j][i]);
643 QL_REQUIRE(close_enough(std::abs(rho_[i][j]), 1.0) || (rho_[i][j] > -1.0 && rho_[i][j] < 1.0),
644 "correlation matrix has invalid entry at (i,j)=(" << i << "," << j << ") equal to "
645 << rho_[i][j]);
646 }
647 QL_REQUIRE(close_enough(rho_[i][i], 1.0), "correlation matrix must have unit diagonal elements, "
648 "but rho(i,i)="
649 << rho_[i][i] << " for i=" << i);
650 }
651
652 // if we salvage the matrix there is no point in checking for negative eigenvalues prior to that
653 if (salvaging_ == SalvagingAlgorithm::None) {
654 SymmetricSchurDecomposition ssd(rho_);
655 for (Size i = 0; i < ssd.eigenvalues().size(); ++i) {
656 QL_REQUIRE(ssd.eigenvalues()[i] >= 0.0,
657 "correlation matrix has negative eigenvalue at " << i << " (" << ssd.eigenvalues()[i] << ")");
658 }
659 }
660}

856std::pair<Real, Real> CrossAssetModel::infdkI(const Size i, const Time t, const Time T, const Real z, const Real y) {
857 QL_REQUIRE(t < T || close_enough(t, T), "infdkI: t (" << t << ") <= T (" << T << ") required");
858 Real V0, V_tilde;
859 std::pair<Real, Real> Vs = infdkV(i, t, T);
860 V0 = Vs.first;
861 V_tilde = Vs.second;
862 Real Hyt = Hy(i).eval(this, t);
863 Real HyT = Hy(i).eval(this, T);
864
865 // TODO account for seasonality ...
866 // compute final results depending on z and y
867 const auto& zts = infdk(i)->termStructure();
868 auto dc = irlgm1f(0)->termStructure()->dayCounter();
869 bool indexIsInterpolated = true; // FIXME, though in line with the comment below
870 Real growth_t = inflationGrowth(zts, t, dc, indexIsInterpolated);
871 Real It = growth_t * std::exp(Hyt * z - y - V0);
872 Real Itilde_t_T = inflationGrowth(zts, T, dc, indexIsInterpolated) / growth_t * std::exp((HyT - Hyt) * z + V_tilde);
873 // concerning interpolation there is an inaccuracy here: if the index
874 // is not interpolated, we still simulate the index value as of t
875 // (and T), although we should go back to t, T which corresponds to
876 // the last actual publication time of the index => is the approximation
877 // here in this sense good enough that we can tolerate this?
878 return std::make_pair(It, Itilde_t_T);
879}

881Real CrossAssetModel::infdkYY(const Size i, const Time t, const Time S, const Time T, const Real z, const Real y,
882 const Real irz) {
883 Size ccy = ccyIndex(infdk(i)->currency());
884
885 // Set Convexity adjustment set to 1.
886 // TODO: Add calculation for DK convexity adjustment
887 Real C_tilde = 1;
888

898std::pair<Real, Real> CrossAssetModel::crlgm1fS(const Size i, const Size ccy, const Time t, const Time T, const Real z,
899 const Real y) const {
900 QL_REQUIRE(ccy < components(CrossAssetModel::AssetType::IR),
901 "ccy index (" << ccy << ") must be in 0..." << (components(CrossAssetModel::AssetType::IR) - 1));
902 QL_REQUIRE(t < T || close_enough(t, T), "crlgm1fS: t (" << t << ") <= T (" << T << ") required");
903 QL_REQUIRE(modelType(CrossAssetModel::AssetType::CR, i) == CrossAssetModel::ModelType::LGM1F,
904 "model at " << i << " is not CR-LGM1F");
905 cache_key k = {i, ccy, t, T};
906 boost::unordered_map<cache_key, std::pair<Real, Real>>::const_iterator it = cache_crlgm1fS_.find(k);
907 Real V0, V_tilde;
908 Real Hlt = Hl(i).eval(this, t);
909 Real HlT = Hl(i).eval(this, T);
910
911 if (it == cache_crlgm1fS_.end()) {
912 // compute V0 and V_tilde
913 if (ccy == 0) {
914 // domestic credit
915 Real Hzt = Hz(0).eval(this, t);
916 Real HzT = Hz(0).eval(this, T);
917 Real zetal0 = zetal(i).eval(this, t);
918 Real zetal1 = integral(this, P(Hl(i), al(i), al(i)), 0.0, t);
919 Real zetal2 = integral(this, P(Hl(i), Hl(i), al(i), al(i)), 0.0, t);
920 Real zetanl0 = integral(this, P(rzl(0, i), az(0), al(i)), 0.0, t);
921 Real zetanl1 = integral(this, P(rzl(0, i), Hl(i), az(0), al(i)), 0.0, t);
922 // opposite signs for last two terms in the book
923 V0 = 0.5 * Hlt * Hlt * zetal0 - Hlt * zetal1 + 0.5 * zetal2 + Hzt * Hlt * zetanl0 - Hzt * zetanl1;
924 V_tilde = -0.5 * (HlT * HlT - Hlt * Hlt) * zetal0 + (HlT - Hlt) * zetal1 -
925 (HzT * HlT - Hzt * Hlt) * zetanl0 + (HzT - Hzt) * zetanl1;
926 } else {
927 // foreign credit
928 V0 = crV(i, ccy, 0, t);
929 V_tilde = crV(i, ccy, t, T) - crV(i, ccy, 0, T) + crV(i, ccy, 0, t);
930 }
931 cache_crlgm1fS_.insert(std::make_pair(k, std::make_pair(V0, V_tilde)));
932 } else {
933 // take V0 and V_tilde from cache
934 V0 = it->second.first;
935 V_tilde = it->second.second;
936 }
937 // compute final results depending on z and y
938 // opposite sign for V0 in the book
939 Real St = crlgm1f(i)->termStructure()->survivalProbability(t) * std::exp(-Hlt * z + y - V0);
940 Real Stilde_t_T = crlgm1f(i)->termStructure()->survivalProbability(T) /
941 crlgm1f(i)->termStructure()->survivalProbability(t) * std::exp(-(HlT - Hlt) * z + V_tilde);
942 return std::make_pair(St, Stilde_t_T);
943}

955Real CrossAssetModel::infV(const Size i, const Size ccy, const Time t, const Time T) const {
956 Real HyT = Hy(i).eval(this, T);
957 Real HdT = irlgm1f(0)->H(T);
958 Real rhody = correlation(CrossAssetModel::AssetType::IR, 0, CrossAssetModel::AssetType::INF, i, 0, 0);
959 Real V;
960 if (ccy == 0) {
961 V = 0.5 * (HyT * HyT * (zetay(i).eval(this, T) - zetay(i).eval(this, t)) -
962 2.0 * HyT * integral(this, P(Hy(i), ay(i), ay(i)), t, T) +
963 integral(this, P(Hy(i), Hy(i), ay(i), ay(i)), t, T)) -
964 rhody * HdT * (HyT * integral(this, P(az(0), ay(i)), t, T) - integral(this, P(az(0), Hy(i), ay(i)), t, T));
965 } else {
966 Real HfT = irlgm1f(ccy)->H(T);
967 Real rhofy = correlation(CrossAssetModel::AssetType::IR, ccy, CrossAssetModel::AssetType::INF, i, 0, 0);
968 Real rhoxy = correlation(CrossAssetModel::AssetType::FX, ccy - 1, CrossAssetModel::AssetType::INF, i, 0, 0);
969 V = 0.5 * (HyT * HyT * (zetay(i).eval(this, T) - zetay(i).eval(this, t)) -
970 2.0 * HyT * integral(this, P(Hy(i), ay(i), ay(i)), t, T) +
971 integral(this, P(Hy(i), Hy(i), ay(i), ay(i)), t, T)) -
972 rhody * (HyT * integral(this, P(Hz(0), az(0), ay(i)), t, T) -
973 integral(this, P(Hz(0), az(0), Hy(i), ay(i)), t, T)) -
974 rhofy * (HfT * HyT * integral(this, P(az(ccy), ay(i)), t, T) -
975 HfT * integral(this, P(az(ccy), Hy(i), ay(i)), t, T) -
976 HyT * integral(this, P(Hz(ccy), az(ccy), ay(i)), t, T) +
977 integral(this, P(Hz(ccy), az(ccy), Hy(i), ay(i)), t, T)) +
978 rhoxy * (HyT * integral(this, P(sx(ccy - 1), ay(i)), t, T) -
979 integral(this, P(sx(ccy - 1), Hy(i), ay(i)), t, T));
980 }
981 return V;
982}

984Real CrossAssetModel::crV(const Size i, const Size ccy, const Time t, const Time T) const {
985 Real HlT = Hl(i).eval(this, T);
986 Real HfT = Hz(ccy).eval(this, T);
987 Real rhodl = correlation(CrossAssetModel::AssetType::IR, 0, CrossAssetModel::AssetType::CR, i, 0, 0);
988 Real rhofl = correlation(CrossAssetModel::AssetType::IR, ccy, CrossAssetModel::AssetType::CR, i, 0, 0);
989 Real rhoxl = correlation(CrossAssetModel::AssetType::FX, ccy - 1, CrossAssetModel::AssetType::CR, i, 0, 0);
990 return 0.5 * (HlT * HlT * (zetal(i).eval(this, T) - zetal(i).eval(this, t)) -
