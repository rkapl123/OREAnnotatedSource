51 void CompoEquityIndex::performCalculations() const {
52  dividendFixings_ = TimeSeries<Real>();
53  auto const& ts = source_->dividendFixings();
54  for (auto const& d : ts) {
55  if (dividendCutoffDate_ == Date() || d.first >= dividendCutoffDate_) {
56  dividendFixings_[d.first] =
57  d.second * fxIndex_->fixing(fxIndex_->fixingCalendar().adjust(d.first, Preceding));
58  }
59  }
60 }
