42 // Shared check of arguments
43 void check(const Date& expiryDate, const Date& paymentDate, bool automaticExercise,
44  const boost::shared_ptr<Index>& underlying, bool exercised, Real priceAtExercise) {
45 
46  using QuantLib::io::iso_date;
47 
48  QL_REQUIRE(paymentDate >= expiryDate, "Cash settled European option payment date ("
49  << iso_date(paymentDate)
50  << ") must be greater than or equal to the expiry date ("
51  << iso_date(expiryDate) << ")");
52 
53  if (automaticExercise) {
54  QL_REQUIRE(underlying, "Cash settled European option has automatic exercise so we need a valid underlying.");
55  }
56 
57  if (exercised) {
58  QL_REQUIRE(priceAtExercise != Null<Real>(), "Cash settled European option was exercised so we need "
59  << "a valid exercise price.");
60  }
61 }

81 CashSettledEuropeanOption::CashSettledEuropeanOption(Option::Type type, Real strike, const Date& expiryDate,
82  Natural paymentLag, const Calendar& paymentCalendar,
83  BusinessDayConvention paymentConvention, bool automaticExercise,
84  const boost::shared_ptr<Index>& underlying, bool exercised,
85  Real priceAtExercise)
86  : VanillaOption(boost::make_shared<PlainVanillaPayoff>(type, strike),
87  boost::make_shared<EuropeanExercise>(expiryDate)),
88  automaticExercise_(automaticExercise), underlying_(underlying), exercised_(false),
89  priceAtExercise_(Null<Real>()) {
90 
91  init(exercised, priceAtExercise);
92 
93  // Derive payment date from exercise date using the lag, calendar and convention.
94  paymentDate_ = paymentCalendar.advance(expiryDate, paymentLag * QuantLib::Days, paymentConvention);
95 
96  check(exercise_->lastDate(), paymentDate_, automaticExercise_, underlying_, exercised_, priceAtExercise_);
97 }

127 void CashSettledEuropeanOption::exercise(Real priceAtExercise) {
128  QL_REQUIRE(priceAtExercise != Null<Real>(), "Cannot exercise with a null price.");
129  QL_REQUIRE(Settings::instance().evaluationDate() >= exercise_->lastDate(),
130  "European option cannot be "
131  << "exercised before expiry date. Valuation date " << iso_date(Settings::instance().evaluationDate())
132  << " is before expiry date " << iso_date(exercise_->lastDate()) << ".");
133  exercised_ = true;
134  priceAtExercise_ = priceAtExercise;
135  update();
136 }
