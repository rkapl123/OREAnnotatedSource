113 IndexedCouponLeg::operator Leg() const {
114  Leg resultLeg;
115  resultLeg.reserve(underlyingLeg_.size());
116 
117  for (Size i = 0; i < underlyingLeg_.size(); ++i) {
118  auto cpn = boost::dynamic_pointer_cast<Coupon>(underlyingLeg_[i]);
119  QL_REQUIRE(cpn, "IndexedCouponLeg: coupon required");
120 
121  bool firstValuationDate = (i == 0);
122  Date fixingDate;
123  if (valuationSchedule_.empty()) {
124  fixingDate = inArrearsFixing_ ? cpn->accrualEndDate() : cpn->accrualStartDate();
125  } else {
126  if (valuationSchedule_.size() == underlyingLeg_.size() + 1) {
127  // valuation schedule corresponds one to one to underlying schedule
128  fixingDate = inArrearsFixing_ ? valuationSchedule_.date(i + 1) : valuationSchedule_.date(i);
129  } else {
130  // look for the latest valuation date less or equal to the underlying accrual start date (if
131  // the indexing is using in advance fixing) resp. accrual end date (for in arrears fixing)
132  auto valDates = valuationSchedule_.dates();
133  Date refDate = inArrearsFixing_ ? cpn->accrualEndDate() : cpn->accrualStartDate();
134  Size index =
135  std::distance(valDates.begin(), std::upper_bound(valDates.begin(), valDates.end(), refDate));
136  QL_REQUIRE(index > 0, "IndexedCouponLeg: First valuation date ("
137  << valDates.front() << ") must be leq accrual "
138  << (inArrearsFixing_ ? "end" : "start") << " date ("
139  << cpn->accrualStartDate() << ") of the " << (i + 1)
140  << "th coupon in the underlying leg");
141  fixingDate = valuationSchedule_.date(index - 1);
142  firstValuationDate = (index == 1);
143  }
144  }
145 
146  fixingDate = fixingCalendar_.advance(fixingDate, -static_cast<int>(fixingDays_), Days, fixingConvention_);
147 
148  if (firstValuationDate && initialFixing_ != Null<Real>()) {
149  resultLeg.push_back(boost::make_shared<IndexedCoupon>(cpn, qty_, initialFixing_));
150  } else {
151  resultLeg.push_back(boost::make_shared<IndexedCoupon>(cpn, qty_, index_, fixingDate));
152  }
153  }
154 
155  return resultLeg;
156 }
