30Sample<std::vector<Array>> MultiPathVariateGeneratorBase::next() const {
31
32 Sample<std::vector<Real>> sequence = nextSequence();
33 Sample<std::vector<Array>> result(std::vector<Array>(timeSteps_, Array(dimension_)), sequence.weight);
34
35 for (Size i = 0; i < timeSteps_; ++i) {
36 Size offset = i * dimension_;
37 std::copy(sequence.value.begin() + offset, sequence.value.begin() + offset + dimension_,
38 result.value[i].begin());
39 }
40
41 return result;
42}

53void MultiPathVariateGeneratorMersenneTwister::reset() {
54 rsg_ = boost::make_shared<
55 InverseCumulativeRsg<RandomSequenceGenerator<MersenneTwisterUniformRng>, InverseCumulativeNormal>>(
56 RandomSequenceGenerator<MersenneTwisterUniformRng>(dimension_ * timeSteps_, MersenneTwisterUniformRng(seed_)),
57 InverseCumulativeNormal());
58 antitheticVariate_ = true;
59}

61Sample<std::vector<Real>> MultiPathVariateGeneratorMersenneTwister::nextSequence() const {
62 if (antitheticSampling_) {
63 antitheticVariate_ = !antitheticVariate_;
64 if (antitheticVariate_) {
65 auto tmp = rsg_->lastSequence();
66 std::transform(tmp.value.begin(), tmp.value.end(), tmp.value.begin(), std::negate<Real>());
67 return tmp;
68 } else {
69 return rsg_->nextSequence();
70 }
71 }
72 return rsg_->nextSequence();
73}

106Sample<std::vector<Array>> MultiPathVariateGeneratorSobolBrownianBridge::next() const {
107 Real weight = gen_->nextPath();
108 std::vector<Array> output(timeSteps_, Array(dimension_));
109 std::vector<Real> tmp(dimension_);
110 for (Size i = 0; i < timeSteps_; ++i) {
111 gen_->nextStep(tmp);
112 std::copy(tmp.begin(), tmp.end(), output[i].begin());
113 }
114 return Sample<std::vector<Array>>(output, weight);
115}

117boost::shared_ptr<MultiPathVariateGeneratorBase>
118makeMultiPathVariateGenerator(const SequenceType s, const Size dimension, const Size timeSteps, const BigNatural seed,
119 const SobolBrownianGenerator::Ordering ordering,
120 const SobolRsg::DirectionIntegers directionIntegers) {
121 switch (s) {
122 case MersenneTwister:
123 return boost::make_shared<QuantExt::MultiPathVariateGeneratorMersenneTwister>(dimension, timeSteps, seed,
124 false);
125 case MersenneTwisterAntithetic:
126 return boost::make_shared<QuantExt::MultiPathVariateGeneratorMersenneTwister>(dimension, timeSteps, seed, true);
127 case Sobol:
128 return boost::make_shared<QuantExt::MultiPathVariateGeneratorSobol>(dimension, timeSteps, seed,
129 directionIntegers);
130 case SobolBrownianBridge:
131 return boost::make_shared<QuantExt::MultiPathVariateGeneratorSobolBrownianBridge>(
132 dimension, timeSteps, ordering, seed, directionIntegers);
133 default:
134 QL_FAIL("Unknown sequence type");
135 }
136}
