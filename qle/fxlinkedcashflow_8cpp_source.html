35Real AverageFXLinked::fxRate() const {
36 Real fx = 0;
37 for (auto const& d: fxFixingDates_)
38 fx += inverted_ ? 1.0 / fxIndex_->fixing(d) : fxIndex_->fixing(d);
39 fx /= fxFixingDates_.size();
40 return inverted_ ? 1.0 / fx : fx;
41}
