115 inline Array Parametrization::parameterValues(const Size i) const {
116  const Array& tmp = parameter(i)->params();
117  Array res(tmp.size());
118  for (Size ii = 0; ii < res.size(); ++ii) {
119  res[ii] = direct(i, tmp[ii]);
120  }
121  return res;
