69McMultiLegBaseEngine::CashflowInfo McMultiLegBaseEngine::createCashflowInfo(boost::shared_ptr<CashFlow> flow,
70 const Currency& payCcy, Real payer,
71 Size legNo, Size cfNo) const {
72
73 constexpr Real tinyTime = 1E-10;
74
75 CashflowInfo info;
76
77 // set some common info: pay time, pay ccy index in the model, payer, exercise into decision time
78
79 info.legNo = legNo;
80 info.cfNo = cfNo;
81 info.payTime = time(flow->date());
82 info.payCcyIndex = model_->ccyIndex(payCcy);
83 info.payer = payer;
84
85 if (auto cpn = boost::dynamic_pointer_cast<Coupon>(flow)) {
86 QL_REQUIRE(cpn->accrualStartDate() < flow->date(),
87 "McMultiLegBaseEngine::createCashflowInfo(): coupon leg "
88 << legNo << " cashflow " << cfNo << " has accrual start date (" << cpn->accrualStartDate()
89 << ") >= pay date (" << flow->date()
90 << "), which breaks an assumption in the engine. This situation is unexpected.");
91 info.exIntoCriterionTime = time(cpn->accrualStartDate()) + tinyTime;
92 } else {
93 info.exIntoCriterionTime = info.payTime;
94 }
95
96 // handle fixed amount cashflows that we immediately recognize as such
97
98 if (boost::dynamic_pointer_cast<FixedRateCoupon>(flow) != nullptr ||
99 boost::dynamic_pointer_cast<SimpleCashFlow>(flow) != nullptr) {
100 info.amountCalculator = [flow](const Size n, const std::vector<std::vector<const RandomVariable*>>& states) {
101 return RandomVariable(n, flow->amount());
102 };
103
104 return info;
105 }
106
107 // handle fx linked fixed cashflow
108
109 if (auto fxl = boost::dynamic_pointer_cast<FXLinkedCashFlow>(flow)) {
110 Date fxLinkedFixingDate = fxl->fxFixingDate();
111 Size fxLinkedSourceCcyIdx = model_->ccyIndex(fxl->fxIndex()->sourceCurrency());
112 Size fxLinkedTargetCcyIdx = model_->ccyIndex(fxl->fxIndex()->targetCurrency());
113 if (fxLinkedFixingDate > today_) {
114 Real fxSimTime = time(fxLinkedFixingDate);
115 info.simulationTimes.push_back(fxSimTime);
116 info.modelIndices.push_back({});
117 if (fxLinkedSourceCcyIdx > 0) {
118 info.modelIndices.front().push_back(
119 model_->pIdx(CrossAssetModel::AssetType::FX, fxLinkedSourceCcyIdx - 1));
120 }
121 if (fxLinkedTargetCcyIdx > 0) {
122 info.modelIndices.front().push_back(
123 model_->pIdx(CrossAssetModel::AssetType::FX, fxLinkedTargetCcyIdx - 1));
124 }
125 }
126 info.amountCalculator = [this, fxLinkedSourceCcyIdx, fxLinkedTargetCcyIdx, fxLinkedFixingDate,
127 fxl](const Size n, const std::vector<std::vector<const RandomVariable*>>& states) {
128 if (fxLinkedFixingDate <= today_)
129 return RandomVariable(n, fxl->amount());
130 RandomVariable fxSource(n, 1.0), fxTarget(n, 1.0);
131 Size fxIdx = 0;
132 if (fxLinkedSourceCcyIdx > 0)
133 fxSource = exp(*states.at(0).at(fxIdx++));
134 if (fxLinkedTargetCcyIdx > 0)
135 fxTarget = exp(*states.at(0).at(fxIdx));
136 return RandomVariable(n, fxl->foreignAmount()) * fxSource / fxTarget;
137 };
138
139 return info;
140 }
141
142 // handle some wrapped coupon types: extract the wrapper info and continue with underlying flow
143
144 bool isCapFloored = false;
145 bool isNakedOption = false;
146 Real effCap = Null<Real>(), effFloor = Null<Real>();
147
148 bool isFxLinked = false;
149 Date fxLinkedFixingDate;
150 Real fxLinkedForeignNominal = Null<Real>();
151 Size fxLinkedSourceCcyIdx = Null<Size>(), fxLinkedTargetCcyIdx = Null<Size>();
152 Real fxLinkedFixedFxRate = Null<Real>(); // if fx fixing date <= today
153 Real fxLinkedSimTime = Null<Real>(); // if fx fixing date > today
154 std::vector<Size> fxLinkedModelIndices; // if fx fixing date > today
155
156 if (auto fxl = boost::dynamic_pointer_cast<FloatingRateFXLinkedNotionalCoupon>(flow)) {
157 isFxLinked = true;
158 fxLinkedFixingDate = fxl->fxFixingDate();
159 fxLinkedSourceCcyIdx = model_->ccyIndex(fxl->fxIndex()->sourceCurrency());
160 fxLinkedTargetCcyIdx = model_->ccyIndex(fxl->fxIndex()->targetCurrency());
161 if (fxLinkedFixingDate <= today_) {
162 fxLinkedFixedFxRate = fxl->fxIndex()->fixing(fxLinkedFixingDate);
163 } else {
164 fxLinkedSimTime = time(fxLinkedFixingDate);
165 if (fxLinkedSourceCcyIdx > 0) {
166 fxLinkedModelIndices.push_back(model_->pIdx(CrossAssetModel::AssetType::FX, fxLinkedSourceCcyIdx - 1));
167 }
168 if (fxLinkedTargetCcyIdx > 0) {
169 fxLinkedModelIndices.push_back(model_->pIdx(CrossAssetModel::AssetType::FX, fxLinkedTargetCcyIdx - 1));
170 }
171 }
172 flow = fxl->underlying();
173 fxLinkedForeignNominal = fxl->foreignAmount();
174 }
175
176 if (auto stripped = boost::dynamic_pointer_cast<StrippedCappedFlooredCoupon>(flow)) {
177 isNakedOption = true;
178 flow = stripped->underlying(); // this is a CappedFlooredCoupon, handled below
179 }
180
181 if (auto cf = boost::dynamic_pointer_cast<CappedFlooredCoupon>(flow)) {
182 isCapFloored = true;
183 effCap = cf->effectiveCap();
184 effFloor = cf->effectiveFloor();
185 flow = cf->underlying();
186 }

188 // handle the coupon types
189
190 if (auto ibor = boost::dynamic_pointer_cast<IborCoupon>(flow)) {
191 Real fixedRate =
192 ibor->fixingDate() <= today_ ? (ibor->rate() - ibor->spread()) / ibor->gearing() : Null<Real>();
193 Size indexCcyIdx = model_->ccyIndex(ibor->index()->currency());
194 Real simTime = time(ibor->fixingDate());
195 if (ibor->fixingDate() > today_) {
196 info.simulationTimes.push_back(simTime);
197 info.modelIndices.push_back({model_->pIdx(CrossAssetModel::AssetType::IR, indexCcyIdx)});
198 }

201 info.simulationTimes.push_back(fxLinkedSimTime);
202 info.modelIndices.push_back(fxLinkedModelIndices);
203 }

205 info.amountCalculator = [this, indexCcyIdx, ibor, simTime, fixedRate, isFxLinked, fxLinkedForeignNominal,
206 fxLinkedSourceCcyIdx, fxLinkedTargetCcyIdx, fxLinkedFixedFxRate, isCapFloored,
207 isNakedOption, effFloor,
208 effCap](const Size n, const std::vector<std::vector<const RandomVariable*>>& states) {
209 RandomVariable fixing = fixedRate != Null<Real>()
210 ? RandomVariable(n, fixedRate)
211 : lgmVectorised_[indexCcyIdx].fixing(ibor->index(), ibor->fixingDate(), simTime,
212 *states.at(0).at(0));
213 RandomVariable fxFixing(n, 1.0);
214 if (isFxLinked) {
215 if (fxLinkedFixedFxRate != Null<Real>()) {
216 fxFixing = RandomVariable(n, fxLinkedFixedFxRate);
217 } else {
218 RandomVariable fxSource(n, 1.0), fxTarget(n, 1.0);
219 Size fxIdx = 0;
220 if (fxLinkedSourceCcyIdx > 0)
221 fxSource = exp(*states.at(1).at(fxIdx++));
222 if (fxLinkedTargetCcyIdx > 0)
223 fxTarget = exp(*states.at(1).at(fxIdx));
224 fxFixing = fxSource / fxTarget;
225 }
226 }
227 RandomVariable effectiveRate;
228 if (isCapFloored) {
229 RandomVariable swapletRate(n, 0.0);
230 RandomVariable floorletRate(n, 0.0);
231 RandomVariable capletRate(n, 0.0);
232 if (!isNakedOption)
233 swapletRate = RandomVariable(n, ibor->gearing()) * fixing + RandomVariable(n, ibor->spread());
234 if (effFloor != Null<Real>())
235 floorletRate = RandomVariable(n, ibor->gearing()) *
236 max(RandomVariable(n, effFloor) - fixing, RandomVariable(n, 0.0));
237 if (effCap != Null<Real>())
238 capletRate = RandomVariable(n, ibor->gearing()) *
239 max(fixing - RandomVariable(n, effCap), RandomVariable(n, 0.0)) *
240 RandomVariable(n, isNakedOption && effFloor == Null<Real>() ? -1.0 : 1.0);
241 effectiveRate = swapletRate + floorletRate - capletRate;
242 } else {
243 effectiveRate = RandomVariable(n, ibor->gearing()) * fixing + RandomVariable(n, ibor->spread());
244 }
245 return RandomVariable(n, (isFxLinked ? fxLinkedForeignNominal : ibor->nominal()) * ibor->accrualPeriod()) *
246 effectiveRate * fxFixing;
247 };
248
249 return info;
250 }

515 Size indexCcyIdx = model_->ccyIndex(cfbma->underlying()->index()->currency());

537 RandomVariable fxSource(n, 1.0), fxTarget(n, 1.0);
538 Size fxIdx = 0;
539 if (fxLinkedSourceCcyIdx > 0)
540 fxSource = exp(*states.at(1).at(fxIdx++));
541 if (fxLinkedTargetCcyIdx > 0)
542 fxTarget = exp(*states.at(1).at(fxIdx));
543 fxFixing = fxSource / fxTarget;

603RandomVariable McMultiLegBaseEngine::cashflowPathValue(const CashflowInfo& cf,
604 const std::vector<std::vector<RandomVariable>>& pathValues,
605 const std::set<Real>& simulationTimes) const {
606
607 Size n = pathValues[0][0].size();
608 auto simTimesPayIdx = timeIndex(cf.payTime, simulationTimes);
609
610 std::vector<RandomVariable> initialValues(model_->stateProcess()->initialValues().size());
611 for (Size i = 0; i < model_->stateProcess()->initialValues().size(); ++i)
612 initialValues[i] = RandomVariable(n, model_->stateProcess()->initialValues()[i]);
613
614 std::vector<std::vector<const RandomVariable*>> states(cf.simulationTimes.size());
615 for (Size i = 0; i < cf.simulationTimes.size(); ++i) {
616 std::vector<const RandomVariable*> tmp(cf.modelIndices[i].size());
617 if (cf.simulationTimes[i] == 0.0) {
618 for (Size j = 0; j < cf.modelIndices[i].size(); ++j) {
619 tmp[j] = &initialValues[cf.modelIndices[i][j]];
620 }
621 } else {
622 auto simTimesIdx = timeIndex(cf.simulationTimes[i], simulationTimes);
623 for (Size j = 0; j < cf.modelIndices[i].size(); ++j) {
624 tmp[j] = &pathValues[simTimesIdx][cf.modelIndices[i][j]];
625 }
626 }
627 states[i] = tmp;
628 }
629
630 auto amount = cf.amountCalculator(n, states) /
631 lgmVectorised_[0].numeraire(
632 cf.payTime, pathValues[simTimesPayIdx][model_->pIdx(CrossAssetModel::AssetType::IR, 0)],
633 discountCurves_[0]);
634
635 if (cf.payCcyIndex > 0) {
636 amount *= exp(pathValues[simTimesPayIdx][model_->pIdx(CrossAssetModel::AssetType::FX, cf.payCcyIndex - 1)]);
637 }
638
639 return amount * RandomVariable(n, cf.payer);
640}

642void McMultiLegBaseEngine::calculate() const {
643
644 McEngineStats::instance().other_timer.resume();
645
646 // check data set by derived engines
647
648 QL_REQUIRE(currency_.size() == leg_.size(), "McMultiLegBaseEngine: number of legs ("
649 << leg_.size() << ") does not match currencies ("
650 << currency_.size() << ")");
651 QL_REQUIRE(payer_.size() == leg_.size(), "McMultiLegBaseEngine: number of legs ("
652 << leg_.size() << ") does not match payer flag (" << payer_.size()
653 << ")");
654
655 // set today's date
656
657 today_ = model_->irlgm1f(0)->termStructure()->referenceDate();
658
659 // set up lgm vectorized instances for each currency
660
661 if (lgmVectorised_.empty()) {
662 for (Size i = 0; i < model_->components(CrossAssetModel::AssetType::IR); ++i) {
663 lgmVectorised_.push_back(LgmVectorised(model_->irlgm1f(i)));
664 }
665 }
666
667 // populate the info to generate the (alive) cashflow amounts
668
669 std::vector<CashflowInfo> cashflowInfo;
670
671 Size legNo = 0;
672 for (auto const& leg : leg_) {
673 Currency currency = currency_[legNo];
674 Real payer = payer_[legNo];
675 Size cashflowNo = 0;
676 for (auto const& cashflow : leg) {
677 // we can skip cashflows that are paid
678 if (cashflow->date() <= today_)
679 continue;
680 // for an alive cashflow, populate the data
681 cashflowInfo.push_back(createCashflowInfo(cashflow, currency, payer, legNo, cashflowNo));
682 // increment counter
683 ++cashflowNo;
684 }
685 ++legNo;
686 }
687
688 /* build exercise times and xva times */
689
690 std::set<Real> exerciseTimes;
691 std::set<Real> xvaTimes;
692
693 if (exercise_ != nullptr) {
694 for (auto const& d : exercise_->dates()) {
695 if (d <= today_)
696 continue;
697 exerciseTimes.insert(time(d));
698 }
699 }
700
701 for (auto const& d : simulationDates_) {
702 xvaTimes.insert(time(d));
703 }
704
705 /* build cashflow generation times */
706
707 std::set<Real> cashflowGenTimes;
708
709 for (auto const& info : cashflowInfo) {
710 cashflowGenTimes.insert(info.simulationTimes.begin(), info.simulationTimes.end());
711 cashflowGenTimes.insert(info.payTime);
712 }
713
714 cashflowGenTimes.erase(0.0); // handled separately, if it is set by a cashflow
715
716 /* build combined time sets */
717
718 std::set<Real> exerciseXvaTimes; // = exercise + xva times
719 std::set<Real> simulationTimes; // = cashflowGen + exercise + xva times
720
721 exerciseXvaTimes.insert(exerciseTimes.begin(), exerciseTimes.end());
722 exerciseXvaTimes.insert(xvaTimes.begin(), xvaTimes.end());
723
724 simulationTimes.insert(cashflowGenTimes.begin(), cashflowGenTimes.end());
725 simulationTimes.insert(exerciseTimes.begin(), exerciseTimes.end());
726 simulationTimes.insert(xvaTimes.begin(), xvaTimes.end());
727
728 McEngineStats::instance().other_timer.stop();
729
730 // simulate the paths for the calibration
731
732 McEngineStats::instance().path_timer.resume();
733
734 QL_REQUIRE(!simulationTimes.empty(),
735 "McMultiLegBaseEngine::calculate(): no simulation times, this is not expected.");
736 std::vector<std::vector<RandomVariable>> pathValues(
737 simulationTimes.size(),
738 std::vector<RandomVariable>(model_->stateProcess()->size(), RandomVariable(calibrationSamples_)));
739
740 for (auto& p : pathValues)
741 for (auto& r : p)
742 r.expand();
743
744 TimeGrid timeGrid(simulationTimes.begin(), simulationTimes.end());
745
746 auto process = model_->stateProcess();
747 if (model_->dimension() == 1) {
748 // use lgm process if possible for better performance
749 auto tmp = boost::make_shared<IrLgm1fStateProcess>(model_->irlgm1f(0));
750 tmp->resetCache(timeGrid.size() - 1);
751 process = tmp;
752 } else if (auto tmp = boost::dynamic_pointer_cast<CrossAssetStateProcess>(process)) {
753 // enable cache
754 tmp->resetCache(timeGrid.size() - 1);
755 }
756
757 auto pathGenerator = makeMultiPathGenerator(calibrationPathGenerator_, process, timeGrid, calibrationSeed_,
758 ordering_, directionIntegers_);
759
760 for (Size i = 0; i < calibrationSamples_; ++i) {
761 const MultiPath& path = pathGenerator->next().value;
762 for (Size j = 0; j < simulationTimes.size(); ++j) {
763 for (Size k = 0; k < model_->stateProcess()->size(); ++k) {
764 pathValues[j][k].data()[i] = path[k][j + 1];
765 }
766 }
767 }
768
769 McEngineStats::instance().path_timer.stop();
770
771 McEngineStats::instance().calc_timer.resume();
772
773 // for each xva and exercise time collect the relevant cashflow amounts and train a model on them
774
775 std::vector<Array> coeffsUndDirty(exerciseXvaTimes.size()); // available on xva times
776 std::vector<Array> coeffsUndExInto(exerciseXvaTimes.size()); // available on xva and ex times
777 std::vector<Array> coeffsContinuationValue(exerciseXvaTimes.size()); // available on ex times
778 std::vector<Array> coeffsOption(exerciseXvaTimes.size()); // available on xva and ex times
779
780 auto basisFns =
781 RandomVariableLsmBasisSystem::multiPathBasisSystem(model_->stateProcess()->size(), polynomOrder_, polynomType_);
782
783 enum class CfStatus { open, cached, done };
784 std::vector<CfStatus> cfStatus(cashflowInfo.size(), CfStatus::open);
785
786 std::vector<const RandomVariable*> regressor(model_->stateProcess()->size());
787
788 RandomVariable pathValueUndDirty(calibrationSamples_);
789 RandomVariable pathValueUndExInto(calibrationSamples_);
790 RandomVariable pathValueOption(calibrationSamples_);
791
792 std::vector<RandomVariable> amountCache(cashflowInfo.size());
793
794 Size counter = exerciseXvaTimes.size() - 1;
795
796 for (auto t = exerciseXvaTimes.rbegin(); t != exerciseXvaTimes.rend(); ++t) {
797
798 bool isExerciseTime = exerciseTimes.find(*t) != exerciseTimes.end();
799 bool isXvaTime = xvaTimes.find(*t) != xvaTimes.end();
800
801 for (Size i = 0; i < cashflowInfo.size(); ++i) {
802
803 /* we assume here that exIntoCriterionTime > t implies payTime > t
804 this must be ensured by the createCashflowInfo method */
805
806 if (cfStatus[i] == CfStatus::open) {
807 if (cashflowInfo[i].exIntoCriterionTime > *t) {
808 auto tmp = cashflowPathValue(cashflowInfo[i], pathValues, simulationTimes);
809 pathValueUndDirty += tmp;
810 pathValueUndExInto += tmp;
811 cfStatus[i] = CfStatus::done;
812 } else if (cashflowInfo[i].payTime > *t) {
813 auto tmp = cashflowPathValue(cashflowInfo[i], pathValues, simulationTimes);
814 pathValueUndDirty += tmp;
815 amountCache[i] = tmp;
816 cfStatus[i] = CfStatus::cached;
817 }
818 } else if (cfStatus[i] == CfStatus::cached) {
819 if (cashflowInfo[i].exIntoCriterionTime > *t) {
820 pathValueUndExInto += amountCache[i];
821 cfStatus[i] = CfStatus::done;
822 amountCache[i].clear();
823 }
824 }
825 }
826
827 /* possible refinement: do the regression on pairs
828 ( std::min(cashflowSimTime, t) , modelIndex state at this time )
829 taken from the cf info sim times, model indices */
830
831 for (Size i = 0; i < model_->stateProcess()->size(); ++i) {
832 regressor[i] = &pathValues[timeIndex(*t, simulationTimes)][i];
833 }
834
835 if (exercise_ != nullptr)
836 coeffsUndExInto[counter] = regressionCoefficients(pathValueUndExInto, regressor, basisFns);
837
838 if (isExerciseTime) {
839 auto exerciseValue = conditionalExpectation(regressor, basisFns, coeffsUndExInto[counter]);
840 coeffsContinuationValue[counter] = regressionCoefficients(
841 pathValueOption, regressor, basisFns, exerciseValue > RandomVariable(calibrationSamples_, 0));
842 auto continuationValue = conditionalExpectation(regressor, basisFns, coeffsContinuationValue[counter]);
843 pathValueOption = conditionalResult(exerciseValue > continuationValue &&
844 exerciseValue > RandomVariable(calibrationSamples_, 0),
845 pathValueUndExInto, pathValueOption);
846 coeffsOption[counter] = regressionCoefficients(pathValueOption, regressor, basisFns);
847 }
848
849 if (isXvaTime) {
850 coeffsUndDirty[counter] = regressionCoefficients(pathValueUndDirty, regressor, basisFns);
851 }
852
853 if (exercise_ != nullptr)
854 coeffsOption[counter] = regressionCoefficients(pathValueOption, regressor, basisFns);
855
856 --counter;
857 }
858
859 // add the remaining live cashflows to get the underlying value
860
861 for (Size i = 0; i < cashflowInfo.size(); ++i) {
862 if (cfStatus[i] == CfStatus::open)
863 pathValueUndDirty += cashflowPathValue(cashflowInfo[i], pathValues, simulationTimes);
864 }
865
866 // set the result value (= underlying value if no exercise is given, otherwise option value)
867
868 resultUnderlyingNpv_ = expectation(pathValueUndDirty).at(0) * model_->numeraire(0, 0.0, 0.0, discountCurves_[0]);
869 resultValue_ = exercise_ == nullptr
870 ? resultUnderlyingNpv_
871 : expectation(pathValueOption).at(0) * model_->numeraire(0, 0.0, 0.0, discountCurves_[0]);
872
873 McEngineStats::instance().calc_timer.stop();
874
875 // construct the amc calculator
876
877 amcCalculator_ = boost::make_shared<MultiLegBaseAmcCalculator>(
878 externalModelIndices_, optionSettlement_, exerciseXvaTimes, exerciseTimes, xvaTimes, coeffsUndDirty,
879 coeffsUndExInto, coeffsContinuationValue, coeffsOption, basisFns, resultValue_,
880 model_->stateProcess()->initialValues(), model_->irlgm1f(0)->currency());
881}

897std::vector<QuantExt::RandomVariable>
898MultiLegBaseAmcCalculator::simulatePath(const std::vector<QuantLib::Real>& pathTimes,
899 std::vector<std::vector<QuantExt::RandomVariable>>& paths,
900 const std::vector<bool>& isRelevantTime, const bool stickyCloseOutRun) {
901
902 // check input path consistency
903
904 QL_REQUIRE(!paths.empty(), "MultiLegBaseAmcCalculator::simulatePath(): no future path times, this is not allowed.");
905 QL_REQUIRE(pathTimes.size() == paths.size(),
906 "MultiLegBaseAmcCalculator::simulatePath(): inconsistent pathTimes size ("
907 << pathTimes.size() << ") and paths size (" << paths.size() << ") - internal error.");
908
909 /* put together the relevant simulation times on the input paths and check for consistency with xva times,
910 also put together the effective paths by filtering on relevant simulation times and model indices */
911
912 std::vector<std::vector<const RandomVariable*>> effPaths(
913 xvaTimes_.size(), std::vector<const RandomVariable*>(externalModelIndices_.size()));
914
915 std::vector<Real> simTimes;
916 Size timeIndex = 0;
917 for (Size i = 0; i < pathTimes.size(); ++i) {
918 if (isRelevantTime[i]) {
919
920 int ind = stickyCloseOutRun ? i - 1 : i;
921 QL_REQUIRE(ind >= 0,
922 "MultiLegBaseAmcCalculator: sticky close out run time index is negative - internal error.");
923 simTimes.push_back(pathTimes[ind]);
924
925 for (Size j = 0; j < externalModelIndices_.size(); ++j) {
926 effPaths[timeIndex][j] = &paths[i][externalModelIndices_[j]];
927 }
928 ++timeIndex;
929 }
930 }
931
932 QL_REQUIRE(simTimes.size() == xvaTimes_.size(),
933 "MultiLegBaseAmcCalculator::simulatePath(): expected input path size "
934 << xvaTimes_.size() << ", but got " << simTimes.size());
935
936 // init result vector
937
938 Size samples = paths.front().front().size();
939 std::vector<RandomVariable> result(simTimes.size() + 1, RandomVariable(paths.front().front().size(), 0.0));
940
941 // simulate the path: result at first time index is simply the reference date npv
942
943 result[0] = RandomVariable(samples, resultValue_);
944
945 // create the initial state as a vector of pointers to rvs for interpolating states below
946
947 std::vector<RandomVariable> initialState;
948 std::vector<const RandomVariable*> initialStatePointer;
949 for (Size j = 0; j < externalModelIndices_.size(); ++j) {
950 initialState.push_back(RandomVariable(samples, initialState_[j]));
951 initialStatePointer.push_back(&initialState.back());
952 }
953
954 // if we don't have an exercise, we return the dirty npv of the underlying at all times
955
956 if (exerciseTimes_.empty()) {
957 Size counter = 0;
958 for (auto t : xvaTimes_) {
959 Size ind = std::distance(exerciseXvaTimes_.begin(), exerciseXvaTimes_.find(t));
960 QL_REQUIRE(ind < exerciseXvaTimes_.size(),
961 "MultiLegBaseAmcCalculator::simulatePath(): internal error, xva time "
962 << t << " not found in exerciseXvaTimes vector.");
963 result[counter + 1] = conditionalExpectation(effPaths[counter], basisFns_, coeffsUndDirty_[ind]);
964 ++counter;
965 }
966 return result;
967 }
968
969 /* if we have an exercise we need to determine the exercise indicators except for a sticky run
970 where we reuse the last saved indicators */
971
972 if (!stickyCloseOutRun) {
973
974 exercised_ = std::vector<Filter>(exerciseTimes_.size() + 1, Filter(samples, false));
975 Size counter = 0;
976
977 for (auto t : exerciseTimes_) {
978
979 // find the time in the exerciseXvaTimes vector
980 Size ind = std::distance(exerciseXvaTimes_.begin(), exerciseXvaTimes_.find(t));
981 QL_REQUIRE(ind != exerciseXvaTimes_.size(),
982 "MultiLegBaseAmcCalculator::simulatePath(): internal error, exercise time "
983 << t << " not found in exerciseXvaTimes vector.");
984
985 // find the sim times and model states before and after the exercise time
986
987 auto t2 = std::lower_bound(xvaTimes_.begin(), xvaTimes_.end(), t);
988
989 // exercise time is after last simulation time => we never exercise on such a path
990
991 if (t2 == xvaTimes_.end())
992 break;
993
994 Real time2 = *t2;
995 std::vector<const RandomVariable*> s2 = effPaths[std::distance(xvaTimes_.begin(), t2)];
996
997 Real time1;
998 std::vector<const RandomVariable*> s1;
999 if (t2 == xvaTimes_.begin()) {
1000 time1 = 0.0;
1001 s1 = initialStatePointer;
1002 } else {
1003 time1 = *std::next(t2, -1);
1004 s1 = effPaths[std::distance(xvaTimes_.begin(), std::next(t2, -1))];
1005 }
1006
1007 // compute the interpolated state (brownian bridge would be better)
1008
1009 std::vector<RandomVariable> s(externalModelIndices_.size());
1010 std::vector<const RandomVariable*> sp(externalModelIndices_.size());
1011 for (Size j = 0; j < externalModelIndices_.size(); ++j) {
1012 RandomVariable alpha1(samples, (time2 - t) / (time2 - time1));
1013 RandomVariable alpha2(samples, (t - time1) / (time2 - time1));
1014 s[j] = alpha1 * *s1[j] + alpha2 * *s2[j];
1015 sp[j] = &s[j];
1016 }
1017
1018 // make the exercise decision
1019
1020 RandomVariable exerciseValue = conditionalExpectation(sp, basisFns_, coeffsUndExInto_[ind]);
1021 RandomVariable continuationValue = conditionalExpectation(sp, basisFns_, coeffsContinuationValue_[ind]);
1022
1023 exercised_[counter + 1] = !exercised_[counter] && exerciseValue > continuationValue &&
1024 exerciseValue > RandomVariable(samples, 0.0);
1025
1026 ++counter;
1027 }
1028 }
1029
1030 // now we can populate the result using the exercise indicators
1031
1032 Size counter = 0;
1033 Size xvaCounter = 0;
1034 Size exerciseCounter = 0;
1035
1036 Filter cashExerciseValueWasAccountedForOnXvaTime(samples, false);
1037 Filter wasExercised(samples, false);
1038
1039 for (auto t : exerciseXvaTimes_) {
1040
1041 if (auto it = exerciseTimes_.find(t); it != exerciseTimes_.end()) {
1042 ++exerciseCounter;
1043 wasExercised = wasExercised || exercised_[exerciseCounter];
1044 }
1045
1046 if (xvaTimes_.find(t) != xvaTimes_.end()) {
1047
1048 RandomVariable optionValue =
1049 conditionalExpectation(effPaths[xvaCounter], basisFns_, coeffsOption_[counter]);
1050
1051 /* Exercise value is "undExInto" if we are in the period between the date on which the exercise happend and
1052 the next exercise date after that, otherwise it is the full dirty npv. This assumes that two exercise
1053 dates d1, d2 are not so close together that a coupon
1054
1055 - pays after d1, d2
1056 - but does not belong to the exercise-into underlying for both d1 and d2
1057
1058 This assumption seems reasonable, since we would never exercise on d1 but wait until d2 since the
1059 underlying which we exercise into is the same in both cases.
1060 We don't introduce a hard check for this, but we rather assume that the exercise dates are set up
1061 appropriately adjusted to the coupon periods. The worst that can happen is that the exercised value
1062 uses the full dirty npv at a too early time. */
1063
1064 RandomVariable exercisedValue =
1065 conditionalResult(exercised_[exerciseCounter],
1066 conditionalExpectation(effPaths[xvaCounter], basisFns_, coeffsUndExInto_[counter]),
1067 conditionalExpectation(effPaths[xvaCounter], basisFns_, coeffsUndDirty_[counter]));
1068
1069 if (settlement_ == Settlement::Type::Cash) {
1070 exercisedValue = applyInverseFilter(exercisedValue, cashExerciseValueWasAccountedForOnXvaTime);
1071 cashExerciseValueWasAccountedForOnXvaTime = cashExerciseValueWasAccountedForOnXvaTime || wasExercised;
1072 }
