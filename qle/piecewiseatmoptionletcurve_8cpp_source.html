146// BackwardFlat does not work well with interpolation on term cap floors so this allows us to exclude it
147class InterpDataset {
148public:
149 enum { arity = 1 };
150
151 InterpDataset() {
152 samples_.push_back(boost::make_tuple(InterpolationType(Linear()), true));
153 samples_.push_back(boost::make_tuple(InterpolationType(QuantExt::LinearFlat()), true));
154 samples_.push_back(boost::make_tuple(InterpolationType(Cubic()), true));
155 samples_.push_back(boost::make_tuple(InterpolationType(QuantExt::CubicFlat()), true));
156 samples_.push_back(boost::make_tuple(InterpolationType(BackwardFlat()), true));
157 samples_.push_back(boost::make_tuple(InterpolationType(Linear()), false));
158 samples_.push_back(boost::make_tuple(InterpolationType(QuantExt::LinearFlat()), false));
159 samples_.push_back(boost::make_tuple(InterpolationType(Cubic()), false));
160 samples_.push_back(boost::make_tuple(InterpolationType(QuantExt::CubicFlat()), false));
161 }
162
163 bdata::size_t size() const { return samples_.size(); }
164
165 typedef vector<boost::tuple<InterpolationType, bool> >::const_iterator iterator;
166 iterator begin() const { return samples_.begin(); }
167
168private:
169 vector<boost::tuple<InterpolationType, bool> > samples_;
170};

178// So that I can reuse below
179string to_string(const InterpolationType& interpolationType) {
180 string result;
181 switch (interpolationType.which()) {
182 case 0:
183 result = "Linear";

185 case 1:
186 result = "BackwardFlat";
187 break;
188 case 2:
189 result = "LinearFlat";

193 break;
194 case 4:

196 break;
197 default:
198 BOOST_FAIL("Unexpected interpolation type");
199 }
200 return result;
201}
