63Real BondIndex::fixing(const Date& fixingDate, bool forecastTodaysFixing) const {
64 //! this logic is the same as in InterestRateIndex
65 QL_REQUIRE(isValidFixingDate(fixingDate), "Fixing date " << fixingDate << " is not valid for '" << name() << "'");
66 Date today = Settings::instance().evaluationDate();
67 if (fixingDate > today || (fixingDate == today && forecastTodaysFixing))
68 return forecastFixing(fixingDate);
69 Real adj = priceQuoteMethod_ == PriceQuoteMethod::CurrencyPerUnit ? 1.0 / priceQuoteBaseValue_ : 1.0;
70 if (fixingDate < today || Settings::instance().enforcesTodaysHistoricFixings()) {
71 // must have been fixed
72 // do not catch exceptions
73 Rate result = pastFixing(fixingDate);
74 QL_REQUIRE(result != Null<Real>(), "Missing " << name() << " fixing for " << fixingDate);
75 return result * adj;
76 }
77 try {
78 // might have been fixed
79 Rate result = pastFixing(fixingDate);
80 if (result != Null<Real>())
81 return result * adj;
82 else
83 ; // fall through and forecast
84 } catch (Error&) {
85 ; // fall through and forecast
86 }
87 return forecastFixing(fixingDate);
88}

90Rate BondIndex::forecastFixing(const Date& fixingDate) const {
91 Date today = Settings::instance().evaluationDate();
92 QL_REQUIRE(fixingDate >= today,
93 "BondIndex::forecastFixing(): fixingDate (" << fixingDate << ") must be >= today (" << today << ")");
94 QL_REQUIRE(bond_, "BondIndex::forecastFixing(): bond required");
95
96 // on today, try to get the dirty absolute price from the bond itself
97
98 Real price = Null<Real>();
99 if (fixingDate == today) {
100 try {
101 price = bond_->settlementValue();
102 } catch (...) {
103 }
104 }
105
106 // for future dates or if the above did not work, assume that the bond can be priced by
107 // simply discounting its cashflows
108
109 if (price == Null<Real>()) {
110 auto res = vanillaBondEngine_->calculateNpv(bond_->settlementDate(fixingDate),
111 bond_->settlementDate(fixingDate), bond_->cashflows(), boost::none,
112 incomeCurve_, conditionalOnSurvival_, false);
113 price = res.npv;
114 }
115
116 price += bidAskAdjustment_ * bond_->notional(fixingDate);
117
118 if (!dirty_) {
119 price -= bond_->accruedAmount(fixingDate) / 100.0 * bond_->notional(fixingDate);
120 }
121
122 if (relative_) {
123 if (close_enough(bond_->notional(fixingDate), 0.0))
124 price = 0.0;
125 else
126 price /= bond_->notional(fixingDate);
127 }
128
129 return price;
130}

132Real BondIndex::pastFixing(const Date& fixingDate) const {
133 QL_REQUIRE(isValidFixingDate(fixingDate), fixingDate << " is not a valid fixing date for '" << name() << "'");
134 Real price = timeSeries()[fixingDate] + bidAskAdjustment_;
135 if (price == Null<Real>())
136 return price;
137 if (dirty_) {
138 QL_REQUIRE(bond_, "BondIndex::pastFixing(): bond required for dirty prices");
139 price += bond_->accruedAmount(fixingDate) / 100.0;
140 }
141
142 if (isInflationLinked_) {
143 price *= QuantExt::inflationLinkedBondQuoteFactor(bond_);
144 }
145
146 if (!relative_) {
147 QL_REQUIRE(bond_, "BondIndex::pastFixing(): bond required for absolute prices");
148 price *= bond_->notional(fixingDate);
149 }
150 return price;
151}

178 Date today = Settings::instance().evaluationDate();
179 QL_REQUIRE(fixingDate >= today, "BondFuturesIndex::forecastFixing(): fixingDate ("
180 << fixingDate << ") must be >= today (" << today << ")");
181 QL_REQUIRE(bond_, "BondFuturesIndex::forecastFixing(): bond required");
182

184 bond()->settlementDate(expiryDate_), bond()->cashflows(),
185 boost::none, incomeCurve(), conditionalOnSurvival(), false);
186 Real price = bondNpvResults.npv;
187
188 if (!dirty()) {
189 price -= bond()->accruedAmount(expiryDate_) / 100.0 * bond()->notional(expiryDate_);
190 }
191
192 if (relative()) {
193 if (close_enough(bond()->notional(expiryDate_), 0.0))
194 price = 0.0;
195 else
196 price /= bond()->notional(expiryDate_);
197 }
198
199 return price;
