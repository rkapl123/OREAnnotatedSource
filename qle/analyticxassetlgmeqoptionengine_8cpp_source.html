32 Real AnalyticXAssetLgmEquityOptionEngine::value(const Time t0, const Time t,
33  const boost::shared_ptr<StrikedTypePayoff> payoff, const Real discount,
34  const Real eqForward) const {
35 
36  const Size& k = eqIdx_;
37  const Size& i = ccyIdx_;
38  const CrossAssetModel* x = model_.get();
39 
40  Real Hi_t = Hz(i).eval(x, t);
41 
42  // calculate the full variance. This is the equity analogy to eqn: 12.18 in Lichters,Stamm,Gallagher
43  Real variance = 0;
44  variance += (vs(k).eval(x, t) - vs(k).eval(x, t0));
45 
46  variance += Hi_t * Hi_t * (zetaz(i).eval(x, t) - zetaz(i).eval(x, t0));
47  variance -= 2.0 * Hi_t * integral(x, P(Hz(i), az(i), az(i)), t0, t);
48  variance += integral(x, P(Hz(i), Hz(i), az(i), az(i)), t0, t);
49 
50  variance += 2.0 * Hi_t * integral(x, P(rzs(i, k), ss(k), az(i)), t0, t);
51  variance -= 2.0 * integral(x, P(Hz(i), rzs(i, k), ss(k), az(i)), t0, t);
52 
53  Real stdev = sqrt(variance);
54  BlackCalculator black(payoff, eqForward, stdev, discount);
55 
56  return black.value();
57 }

59 void AnalyticXAssetLgmEquityOptionEngine::calculate() const {
60 
61  QL_REQUIRE(arguments_.exercise->type() == Exercise::European, "only European options are allowed");
62 
63  boost::shared_ptr<StrikedTypePayoff> payoff = boost::dynamic_pointer_cast<StrikedTypePayoff>(arguments_.payoff);
64  QL_REQUIRE(payoff != NULL, "only striked payoff is allowed");
65 
66  Date expiry = arguments_.exercise->lastDate();
67  Time t = model_->irlgm1f(0)->termStructure()->timeFromReference(expiry);
68 
69  if (t <= 0.0) {
70  // option is expired, we do not value any possibly non settled
71  // flows, i.e. set the npv to zero in this case
72  results_.value = 0.0;
73  return;
74  }
75 
76  Real divDiscount = model_->eqbs(eqIdx_)->equityDivYieldCurveToday()->discount(expiry);
77  Real eqIrDiscount = model_->eqbs(eqIdx_)->equityIrCurveToday()->discount(expiry);
78  Real cashflowsDiscount = model_->irlgm1f(ccyIdx_)->termStructure()->discount(expiry);
79 
80  Real eqForward = model_->eqbs(eqIdx_)->eqSpotToday()->value() * divDiscount / eqIrDiscount;
81 
82  results_.value = value(0.0, t, payoff, cashflowsDiscount, eqForward);
83 
84 } // calculate()
