23class AtmAdjustedSmileSection : public QuantLib::SmileSection {
24public:
25 AtmAdjustedSmileSection(const boost::shared_ptr<QuantLib::SmileSection>& base, const QuantLib::Real baseAtmLevel,
26 const QuantLib::Real targetAtmLevel)
27 : SmileSection(), base_(base), baseAtmLevel_(baseAtmLevel), targetAtmLevel_(targetAtmLevel) {}
28 QuantLib::Real minStrike() const override { return base_->minStrike(); }
29 QuantLib::Real maxStrike() const override { return base_->maxStrike(); }
30 QuantLib::Real atmLevel() const override { return targetAtmLevel_; }
31 const QuantLib::Date& exerciseDate() const override { return base_->exerciseDate(); }
32 QuantLib::VolatilityType volatilityType() const override { return base_->volatilityType(); }
33 QuantLib::Rate shift() const override { return base_->shift(); }
34 const QuantLib::Date& referenceDate() const override { return base_->referenceDate(); }
35 QuantLib::Time exerciseTime() const override { return base_->exerciseTime(); }
36 const QuantLib::DayCounter& dayCounter() const override { return base_->dayCounter(); }
37
38private:
39 QuantLib::Volatility volatilityImpl(QuantLib::Rate strike) const override {
40 if (strike == QuantLib::Null<QuantLib::Real>()) {
41 return base_->volatility(baseAtmLevel_);
42 }
43 // just a moneyness, but no vol adjustment, so this is only suitable for normal vols
44 return base_->volatility(strike + baseAtmLevel_ - targetAtmLevel_);
45 };
46
47 boost::shared_ptr<QuantLib::SmileSection> base_;
48 QuantLib::Real baseAtmLevel_;
49 QuantLib::Real targetAtmLevel_;
50};
