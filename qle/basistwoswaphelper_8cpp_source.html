32 BasisTwoSwapHelper::BasisTwoSwapHelper(const Handle<Quote>& spread, const Period& swapTenor, const Calendar& calendar,
33  // Long tenor swap
34  Frequency longFixedFrequency, BusinessDayConvention longFixedConvention,
35  const DayCounter& longFixedDayCount,
36  const boost::shared_ptr<IborIndex>& longIndex,
37  // Short tenor swap
38  Frequency shortFixedFrequency, BusinessDayConvention shortFixedConvention,
39  const DayCounter& shortFixedDayCount,
40  const boost::shared_ptr<IborIndex>& shortIndex, bool longMinusShort,
41  // Discount curve
42  const Handle<YieldTermStructure>& discountingCurve)
43  : RelativeDateRateHelper(spread), swapTenor_(swapTenor), calendar_(calendar),
44  longFixedFrequency_(longFixedFrequency), longFixedConvention_(longFixedConvention),
45  longFixedDayCount_(longFixedDayCount), longIndex_(longIndex), shortFixedFrequency_(shortFixedFrequency),
46  shortFixedConvention_(shortFixedConvention), shortFixedDayCount_(shortFixedDayCount), shortIndex_(shortIndex),
47  longMinusShort_(longMinusShort), discountHandle_(discountingCurve) {
48 
49  QL_REQUIRE(longIndex_->tenor() >= shortIndex_->tenor(),
50  "Tenor of longIndex should be at least tenor of shortIndex.");
51 
52  bool longIndexHasCurve = !longIndex_->forwardingTermStructure().empty();
53  bool shortIndexHasCurve = !shortIndex_->forwardingTermStructure().empty();
54  bool haveDiscountCurve = !discountHandle_.empty();
55  QL_REQUIRE(!(longIndexHasCurve && shortIndexHasCurve && haveDiscountCurve),
56  "Have all curves nothing to solve for.");
57 
58  if (longIndexHasCurve && !shortIndexHasCurve) {
59  shortIndex_ = shortIndex_->clone(termStructureHandle_);
60  shortIndex_->unregisterWith(termStructureHandle_);
61  } else if (!longIndexHasCurve && shortIndexHasCurve) {
62  longIndex_ = longIndex_->clone(termStructureHandle_);
63  longIndex_->unregisterWith(termStructureHandle_);
64  } else if (!longIndexHasCurve && !shortIndexHasCurve) {
65  QL_FAIL("Need at least one of the indices to have a valid curve.");
66  }
67 
68  registerWith(longIndex_);
69  registerWith(shortIndex_);
70  registerWith(discountHandle_);
71  initializeDates();
72 }

74 void BasisTwoSwapHelper::initializeDates() {
75 
76  /* Important to use a fixed rate of 0.0 here to avoid the calculation
77  of the atm swap rate in MakeVanillaSwap operator ...(). If it is
78  Null, you get an exception because the discountRelinkableHandle_
79  is initially empty. */
80  longSwap_ = MakeVanillaSwap(swapTenor_, longIndex_, 0.0)
81  .withDiscountingTermStructure(discountRelinkableHandle_)
82  .withFixedLegDayCount(longFixedDayCount_)
83  .withFixedLegTenor(Period(longFixedFrequency_))
84  .withFixedLegConvention(longFixedConvention_)
85  .withFixedLegTerminationDateConvention(longFixedConvention_)
86  .withFixedLegCalendar(calendar_)
87  .withFloatingLegCalendar(calendar_);
88 
89  shortSwap_ = MakeVanillaSwap(swapTenor_, shortIndex_, 0.0)
90  .withDiscountingTermStructure(discountRelinkableHandle_)
91  .withFixedLegDayCount(shortFixedDayCount_)
92  .withFixedLegTenor(Period(shortFixedFrequency_))
93  .withFixedLegConvention(shortFixedConvention_)
94  .withFixedLegTerminationDateConvention(shortFixedConvention_)
95  .withFixedLegCalendar(calendar_)
96  .withFloatingLegCalendar(calendar_);
97 
98  earliestDate_ = std::min(longSwap_->startDate(), shortSwap_->startDate());
99  latestDate_ = std::max(longSwap_->maturityDate(), shortSwap_->maturityDate());
100 
101 /* May need to adjust latestDate_ if you are projecting libor based
102  on tenor length rather than from accrual date to accrual date. */
103 #ifdef QL_USE_INDEXED_COUPON
104  if (termStructureHandle_ == shortIndex_->forwardingTermStructure()) {
105  boost::shared_ptr<FloatingRateCoupon> lastFloating =
106  boost::dynamic_pointer_cast<FloatingRateCoupon>(shortSwap_->floatingLeg().back());
107  Date fixingValueDate = shortIndex_->valueDate(lastFloating->fixingDate());
108  Date endValueDate = shortIndex_->maturityDate(fixingValueDate);
109  latestDate_ = std::max(latestDate_, endValueDate);
110  }
111  if (termStructureHandle_ == longIndex_->forwardingTermStructure()) {
112  boost::shared_ptr<FloatingRateCoupon> lastFloating =
113  boost::dynamic_pointer_cast<FloatingRateCoupon>(longSwap_->floatingLeg().back());
114  Date fixingValueDate = longIndex_->valueDate(lastFloating->fixingDate());
115  Date endValueDate = longIndex_->maturityDate(fixingValueDate);
116  latestDate_ = std::max(latestDate_, endValueDate);
117  }
118 #endif
119 }

121 void BasisTwoSwapHelper::setTermStructure(YieldTermStructure* t) {
122 
123  bool observer = false;
124 
125  boost::shared_ptr<YieldTermStructure> temp(t, no_deletion);
126  termStructureHandle_.linkTo(temp, observer);
127 
128  if (discountHandle_.empty())
129  discountRelinkableHandle_.linkTo(temp, observer);
130  else
131  discountRelinkableHandle_.linkTo(*discountHandle_, observer);
132 
133  RelativeDateRateHelper::setTermStructure(t);
134 }

136 Real BasisTwoSwapHelper::impliedQuote() const {
137  QL_REQUIRE(termStructure_ != 0, "Termstructure not set");
138  longSwap_->deepUpdate();
139  shortSwap_->deepUpdate();
140  if (longMinusShort_)
141  return longSwap_->fairRate() - shortSwap_->fairRate();
142  else
143  return shortSwap_->fairRate() - longSwap_->fairRate();
144 }
