38void OvernightIndexedCrossCcyBasisSwapEngine::calculate() const {
39 results_.value = 0.0;
40 results_.errorEstimate = Null<Real>();
41
42 QL_REQUIRE(!fx_.empty(), "fx handle not set");
43 QL_REQUIRE(!ts1_.empty(), "ts1 handle not set");
44 QL_REQUIRE(!ts2_.empty(), "ts2 handle not set");
45
46 QL_REQUIRE(ts1_->referenceDate() == ts2_->referenceDate(), "reference dates do not match");
47
48 Date npvDate = Settings::instance().evaluationDate();
49
50 results_.valuationDate = npvDate;
51
52 results_.legNPV.resize(arguments_.legs.size());
53 results_.legBPS.resize(arguments_.legs.size());
54
55 bool includeRefDateFlows = Settings::instance().includeReferenceDateEvents();
56
57 for (Size i = 0; i < arguments_.legs.size(); ++i) {
58 try {
59 Handle<YieldTermStructure> yts;
60 Real fx;
61 if (arguments_.currency[i] == ccy1_) {
62 yts = ts1_;
63 fx = 1.0;
64 } else {
65 yts = ts2_;
66 fx = fx_->value();
67 }
68 results_.legNPV[i] = fx * arguments_.payer[i] *
69 CashFlows::npv(arguments_.legs[i], **yts, includeRefDateFlows, npvDate, npvDate);
70 results_.legBPS[i] = fx * arguments_.payer[i] *
71 CashFlows::bps(arguments_.legs[i], **yts, includeRefDateFlows, npvDate, npvDate);
72 } catch (std::exception& e) {
73 QL_FAIL(io::ordinal(i + 1) << " leg: " << e.what());
74 }
75 results_.value += results_.legNPV[i];
76 }
77
78 static Spread basisPoint = 1.0e-4;
79 results_.fairPayLegSpread = arguments_.paySpread - results_.value / (results_.legBPS[0] / basisPoint);
80 results_.fairRecLegSpread = arguments_.recSpread - results_.value / (results_.legBPS[1] / basisPoint);
81}
