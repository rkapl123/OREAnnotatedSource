33 // Local class for printing each report type via fprintf
34 class ReportTypePrinter : public boost::static_visitor<> {
35 public:
36  ReportTypePrinter(FILE* fp, int prec, char quoteChar = '\0', const string& nullString = "#N/A")
37  : fp_(fp), rounding_(prec, QuantLib::Rounding::Closest), quoteChar_(quoteChar), null_(nullString) {}
38 
39  void operator()(const Size i) const {
40  if (i == QuantLib::Null<Size>()) {
41  fprintNull();
42  } else {
43  fprintf(fp_, "%zu", i);
44  }
45  }
46  void operator()(const Real d) const {
47  if (d == QuantLib::Null<Real>() || !std::isfinite(d)) {
48  fprintNull();
49  } else {
50  Real r = rounding_(d);
51  fprintf(fp_, "%.*f", rounding_.precision(), QuantLib::close_enough(r, 0.0) ? 0.0 : r);
52  }
53  }
54  void operator()(const string& s) const { fprintString(s); }
55  void operator()(const Date& d) const {
56  if (d == QuantLib::Null<Date>()) {
57  fprintNull();
58  } else {
59  string s = to_string(d);
60  fprintString(s);
61  }
62  }
63  void operator()(const Period& p) const {
64  string s = to_string(p);
65  fprintString(s);
66  }
67 
68 private:
69  void fprintNull() const { fprintf(fp_, "%s", null_.c_str()); }
70 
71  // Shared implementation to include the quote character.
72  void fprintString(const string& s) const {
73  bool quoted = s.size() > 1 && s[0] == quoteChar_ && s[s.size() - 1] == quoteChar_;
74  if (!quoted && quoteChar_ != '\0')
75  fputc(quoteChar_, fp_);
76  fprintf(fp_, "%s", s.c_str());
77  if (!quoted && quoteChar_ != '\0')
78  fputc(quoteChar_, fp_);
79  }
80 
81  FILE* fp_;
82  QuantLib::Rounding rounding_;
83  char quoteChar_;
84  string null_;
85 };

126 Report& CSVFileReport::addColumn(const string& name, const ReportType& rt, Size precision) {
127  checkIsOpen("addColumn(" + name + ")");
128  columnTypes_.push_back(rt);
129  printers_.push_back(ReportTypePrinter(fp_, precision, quoteChar_, nullString_));
130  if (i_ == 0 && commentCharacter_)
131  fprintf(fp_, "#");
132  if (i_ > 0)
133  fprintf(fp_, "%c", sep_);
134  string cpName = name;
135  if (lowerHeader_ && !cpName.empty())
136  cpName[0] = std::tolower(static_cast<unsigned char>(cpName[0]));
137  fprintf(fp_, "%s", cpName.c_str());
138  i_++;
139  return *this;
140 }

142 Report& CSVFileReport::next() {
143  // check the filesize every for every 1000 rows, and roll if necessary
144  if (rolloverSize_ != Null<Size>()) {
145  if (j_ >= 10000) {
146  auto fileSize = boost::filesystem::file_size(filename_);
147  TLOG("CSV size of " << filename_ << " is " << fileSize);
148  if (fileSize > rolloverSize_ * 1024 * 1024)
149  rollover();
150  j_ = 0;
151  } else
152  j_++;
153  }
154 
155  checkIsOpen("next()");
156  QL_REQUIRE(i_ == columnTypes_.size(), "Cannot go to next line, only " << i_ << " entries filled");
157  fprintf(fp_, "\n");
158  i_ = 0;
159  return *this;
160 }

162 Report& CSVFileReport::add(const ReportType& rt) {
163  checkIsOpen("add()");
164  QL_REQUIRE(i_ < columnTypes_.size(), "No column to add [" << rt << "] to.");
165  QL_REQUIRE(rt.which() == columnTypes_[i_].which(), "Cannot add value " << rt << " of type " << rt.which()
166  << " to column " << i_ << " of type "
167  << columnTypes_[i_].which());
168 
169  if (i_ != 0)
170  fprintf(fp_, "%c", sep_);
171  boost::apply_visitor(printers_[i_], rt);
172  i_++;
173  return *this;
