31//! Option Wrapper
32/*!
33 Wrapper Class for Options
34 Prices underlying instrument if option has been exercised
35 Handles Physical and Cash Settlement
36
37 \ingroup tradedata
38*/
39class OptionWrapper : public InstrumentWrapper {
40public:
41 //! Constructor
42 OptionWrapper(const boost::shared_ptr<QuantLib::Instrument>& inst, const bool isLongOption,
43 const std::vector<QuantLib::Date>& exerciseDate, const bool isPhysicalDelivery,
44 const std::vector<boost::shared_ptr<QuantLib::Instrument>>& undInst,
45 // multiplier as seen from the option holder
46 const Real multiplier = 1.0,
47 // undMultiplier w.r.t. underlying as seen from the option holder
48 const Real undMultiplier = 1.0,
49 const std::vector<boost::shared_ptr<QuantLib::Instrument>>& additionalInstruments =
50 std::vector<boost::shared_ptr<QuantLib::Instrument>>(),
51 const std::vector<Real>& additionalMultipliers = std::vector<Real>());
52 //! \name InstrumentWrapper interface
53 //@{
54 void initialise(const std::vector<QuantLib::Date>& dates) override;
55 void reset() override;
56 QuantLib::Real NPV() const override;
57 const std::map<std::string, boost::any>& additionalResults() const override;
58 void updateQlInstruments() override {
59 for (QuantLib::Size i = 0; i < underlyingInstruments_.size(); ++i)
60 underlyingInstruments_[i]->update();
61 InstrumentWrapper::updateQlInstruments();
62 }
63 bool isOption() override { return true; }
64 //@}
65
66 //! return the underlying instruments
67 const std::vector<boost::shared_ptr<QuantLib::Instrument>>& underlyingInstruments() const {
68 return underlyingInstruments_;
69 }
70
71 /*! return the active underlying instrument
72 Pass true if you trigger a calculation on the returned instrument and want to record
73 the timing for that calculation. If in doubt whether a calculation is triggered, pass false. */
74 const boost::shared_ptr<QuantLib::Instrument>& activeUnderlyingInstrument(const bool calculate = false) const {
75 if (calculate && activeUnderlyingInstrument_ != nullptr) {
76 getTimedNPV(activeUnderlyingInstrument_);
77 }
78 return activeUnderlyingInstrument_;
79 }
80
81 //! return true if option is long, false if option is short
82 bool isLong() const { return isLong_; }
83
84 //! return true if option is exercised
85 bool isExercised() const { return exercised_; }
86
87 //! return true for physical delivery, false for cash settlement
88 bool isPhysicalDelivery() const { return isPhysicalDelivery_; }
89
90 //! the underlying multiplier
91 Real underlyingMultiplier() const { return undMultiplier_; }
92
93 //! the (actual) date the option was exercised
94 const QuantLib::Date& exerciseDate() const { return exerciseDate_; }
95
96 //! disable exercise decisions
97 void enableExercise() { exercisable_ = true; }
98
99 //! enable exercise decisions
100 void disableExercise() { exercisable_ = false; }
101
102 virtual bool exercise() const = 0;
103
104protected:
105 bool isLong_;
106 bool isPhysicalDelivery_;
107 std::vector<QuantLib::Date> contractExerciseDates_;
108 std::vector<QuantLib::Date> effectiveExerciseDates_;
109 std::vector<boost::shared_ptr<QuantLib::Instrument>> underlyingInstruments_;
110 mutable boost::shared_ptr<QuantLib::Instrument> activeUnderlyingInstrument_;
111 Real undMultiplier_;
112 mutable bool exercised_;
113 bool exercisable_;
114 mutable QuantLib::Date exerciseDate_;
115};

162//! Bermudan Option Wrapper

169 const std::vector<QuantLib::Date>& exerciseDates, const bool isPhysicalDelivery,
170 const std::vector<boost::shared_ptr<QuantLib::Instrument>>& undInsts,
171 // multiplier as seen from the option holder
172 const Real multiplier = 1.0,
173 // undMultiplier w.r.t. underlying as seen from the option holder
174 const Real undMultiplier = 1.0,
175 const std::vector<boost::shared_ptr<QuantLib::Instrument>>& additionalInstruments =
176 std::vector<boost::shared_ptr<QuantLib::Instrument>>(),
177 const std::vector<Real>& additionalMultipliers = std::vector<Real>())
178 : OptionWrapper(inst, isLongOption, exerciseDates, isPhysicalDelivery, undInsts, multiplier, undMultiplier,
179 additionalInstruments, additionalMultipliers) {
180 QL_REQUIRE(exerciseDates.size() == undInsts.size(),
181 "sizes of exercise date and underlying instrument vectors do not match");

184 bool exercise() const override;

186private:
