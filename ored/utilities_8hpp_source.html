46 template <typename Helper> Real getCalibrationError(const std::vector<boost::shared_ptr<Helper>>& basket) {
47  Real rmse = 0.0;
48  for (auto const& h : basket) {
49  Real tmp = h->calibrationError();
50  rmse += tmp * tmp;
51  }
52  return std::sqrt(rmse / static_cast<Real>(basket.size()));
53 }
