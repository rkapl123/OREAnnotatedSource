39 EqBsBuilder::EqBsBuilder(const boost::shared_ptr<ore::data::Market>& market, const boost::shared_ptr<EqBsData>& data,
40  const QuantLib::Currency& baseCcy, const std::string& configuration,
41  const std::string& referenceCalibrationGrid)
42  : market_(market), configuration_(configuration), data_(data), referenceCalibrationGrid_(referenceCalibrationGrid),
43  baseCcy_(baseCcy) {
44 
45  optionActive_ = std::vector<bool>(data_->optionExpiries().size(), false);
46  marketObserver_ = boost::make_shared<MarketObserver>();
47  QuantLib::Currency ccy = ore::data::parseCurrency(data->currency());
48  string eqName = data->eqName();
49 
50  LOG("Start building EqBs model for " << eqName);
51 
52  // get market data
53  std::string fxCcyPair = ccy.code() + baseCcy_.code();
54  eqSpot_ = market_->equitySpot(eqName, configuration_);
55  fxSpot_ = market_->fxSpot(fxCcyPair, configuration_);
56  // FIXME using the "discount curve" here instead of the equityReferenceRateCurve?
57  ytsRate_ = market_->discountCurve(ccy.code(), configuration_);
58  ytsDiv_ = market_->equityDividendCurve(eqName, configuration_);
59  eqVol_ = market_->equityVol(eqName, configuration_);
60 
61  // register with market observables except vols
62  marketObserver_->registerWith(eqSpot_);
63  marketObserver_->registerWith(fxSpot_);
64  marketObserver_->registerWith(ytsRate_);
65  marketObserver_->registerWith(ytsDiv_);
66 
67  // register the builder with the vol and the market observer
68  registerWith(eqVol_);
69  registerWith(marketObserver_);
70 
71  // notify observers of all market data changes, not only when not calculated
72  alwaysForwardNotifications();
73 
74  // build option basket and derive parametrization from it
75  if (data->calibrateSigma())
76  buildOptionBasket();
77 
78  Array sigmaTimes, sigma;
79  if (data->sigmaParamType() == ParamType::Constant) {
80  QL_REQUIRE(data->sigmaTimes().size() == 0, "empty sigma time grid expected");
81  QL_REQUIRE(data->sigmaValues().size() == 1, "initial sigma grid size 1 expected");
82  sigmaTimes = Array(0);
83  sigma = Array(data_->sigmaValues().begin(), data_->sigmaValues().end());
84  } else {
85  if (data->calibrateSigma()) { // override
86  QL_REQUIRE(optionExpiries_.size() > 0, "optionExpiries is empty");
87  sigmaTimes = Array(optionExpiries_.begin(), optionExpiries_.end() - 1);
88  sigma = Array(sigmaTimes.size() + 1, data->sigmaValues()[0]);
89  } else {
90  // use input time grid and input alpha array otherwise
91  sigmaTimes = Array(data_->sigmaTimes().begin(), data_->sigmaTimes().end());
92  sigma = Array(data_->sigmaValues().begin(), data_->sigmaValues().end());
93  QL_REQUIRE(sigma.size() == sigmaTimes.size() + 1, "sigma grids do not match");
94  }
95  }
96 
97  // Quotation needs to be consistent with FX spot quotation in the FX calibration basket
98  if (data->sigmaParamType() == ParamType::Piecewise)
99  parametrization_ = boost::make_shared<QuantExt::EqBsPiecewiseConstantParametrization>(
100  ccy, eqName, eqSpot_, fxSpot_, sigmaTimes, sigma, ytsRate_, ytsDiv_);
101  else if (data->sigmaParamType() == ParamType::Constant)
102  parametrization_ = boost::make_shared<QuantExt::EqBsConstantParametrization>(ccy, eqName, eqSpot_, fxSpot_,
103  sigma[0], ytsRate_, ytsDiv_);
104  else
105  QL_FAIL("interpolation type not supported for Equity");
106 }

153  std::string expiryString = data_->optionExpiries()[j];

162 bool EqBsBuilder::volSurfaceChanged(const bool updateCache) const {
163  bool hasUpdated = false;
164 
165  // if cache doesn't exist resize vector
166  if (eqVolCache_.size() != optionBasket_.size())
167  eqVolCache_ = vector<Real>(optionBasket_.size(), Null<Real>());
168 
169  Size optionCounter = 0;
170  for (Size j = 0; j < data_->optionExpiries().size(); j++) {
171  if (!optionActive_[j])
172  continue;
173  Real vol = eqVol_->blackVol(optionExpiry(j), optionStrike(j));
174  if (!close_enough(eqVolCache_[optionCounter], vol)) {
175  if (updateCache)
176  eqVolCache_[optionCounter] = vol;
177  hasUpdated = true;
178  }
179  optionCounter++;
180  }
181  return hasUpdated;
182 }

184 void EqBsBuilder::buildOptionBasket() const {
185  QL_REQUIRE(data_->optionExpiries().size() == data_->optionStrikes().size(), "Eq option vector size mismatch");
186 
187  optionActive_ = std::vector<bool>(data_->optionExpiries().size(), false);
188 
189  DLOG("build reference date grid '" << referenceCalibrationGrid_ << "'");
190  Date lastRefCalDate = Date::minDate();
191  std::vector<Date> referenceCalibrationDates;
192  if (!referenceCalibrationGrid_.empty())
193  referenceCalibrationDates = DateGrid(referenceCalibrationGrid_).dates();
194 
195  std::vector<Time> expiryTimes;
196  optionBasket_.clear();
197  for (Size j = 0; j < data_->optionExpiries().size(); j++) {
198  // may wish to calibrate against specific futures expiry dates...
199  Date expiryDate = optionExpiry(j);
200 
201  // check if we want to keep the helper when a reference calibration grid is given
202  auto refCalDate =
203  std::lower_bound(referenceCalibrationDates.begin(), referenceCalibrationDates.end(), expiryDate);
204  if (refCalDate == referenceCalibrationDates.end() || *refCalDate > lastRefCalDate) {
205  optionActive_[j] = true;
206  Real strikeValue = optionStrike(j);
207  Handle<Quote> volQuote(boost::make_shared<SimpleQuote>(eqVol_->blackVol(expiryDate, strikeValue)));
208  boost::shared_ptr<QuantExt::FxEqOptionHelper> helper = boost::make_shared<QuantExt::FxEqOptionHelper>(
209  expiryDate, strikeValue, eqSpot_, volQuote, ytsRate_, ytsDiv_);
210  optionBasket_.push_back(helper);
211  helper->performCalculations();
212  expiryTimes.push_back(ytsRate_->timeFromReference(helper->option()->exercise()->date(0)));
213  DLOG("Added EquityOptionHelper " << data_->eqName() << " " << QuantLib::io::iso_date(expiryDate) << " "
214  << volQuote->value());
215  if (refCalDate != referenceCalibrationDates.end())
216  lastRefCalDate = *refCalDate;
217  } else {
218  optionActive_[j] = false;
219  }
220  }
221 
222  std::sort(expiryTimes.begin(), expiryTimes.end());
223  auto itExpiryTime = unique(expiryTimes.begin(), expiryTimes.end());
224  expiryTimes.resize(distance(expiryTimes.begin(), itExpiryTime));
225 
226  optionExpiries_ = Array(expiryTimes.size());
227  for (Size j = 0; j < expiryTimes.size(); j++)
228  optionExpiries_[j] = expiryTimes[j];
229 }
