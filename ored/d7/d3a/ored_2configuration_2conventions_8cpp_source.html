52 void checkContinuationMappings(const map<Natural, Natural>& mp, const string& name) {
53 
54  Natural previousValue = 0;
55  for (const auto& kv : mp) {
56  QL_REQUIRE(kv.first <= kv.second, "Not allowed a " << name << " continuation mapping where From (" <<
57  kv.first << ") is greater than To (" << kv.second << ").");
58  QL_REQUIRE(kv.second > previousValue, "The To " << name << " continuation mappings should be strictly " <<
59  "increasing but got " << kv.second << " <= " << previousValue);
60  previousValue = kv.second;
61  }
62 }

369 void IborIndexConvention::build() {
370  // just a check really that the id is consistent with the ibor index name rules
371  vector<string> tokens;
372  split(tokens, localId_, boost::is_any_of("-"));
373  QL_REQUIRE(tokens.size() == 2 || tokens.size() == 3,
374  "Two or three tokens required in IborIndexConvention " << localId_ << ": CCY-INDEX or CCY-INDEX-TERM");
375 
376  // set the Id - this converts the local id term from "7D" to "1W", "28D" to "1M" etc, so if can be picked
377  // up by searches
378  id_ = (tokens.size() == 3) ? (tokens[0] + "-" + tokens[1] + "-" + to_string(parsePeriod(tokens[2]))) : localId_;
379 }
