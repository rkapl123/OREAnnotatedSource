32 void CreditDefaultSwap::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
33  DLOG("CreditDefaultSwap::build() called for trade " << id());
34 
35  const boost::shared_ptr<Market> market = engineFactory->market();
36  boost::shared_ptr<EngineBuilder> builder = engineFactory->builder("CreditDefaultSwap");
37 
38  QL_REQUIRE(swap_.leg().legType() == "Fixed", "CreditDefaultSwap requires Fixed leg");
39  Schedule schedule = makeSchedule(swap_.leg().schedule());
40 
41  BusinessDayConvention payConvention = Following;
42  if (!swap_.leg().paymentConvention().empty()) {
43  payConvention = parseBusinessDayConvention(swap_.leg().paymentConvention());
44  }
45 
46  Protection::Side prot = swap_.leg().isPayer() ? Protection::Side::Buyer : Protection::Side::Seller;
47  notional_ = swap_.leg().notionals().front();
48 
49  Leg amortized_leg;
50  if (swap_.leg().notionals().size() > 1) {
51  auto configuration = builder->configuration(MarketContext::pricing);
52  auto legBuilder = engineFactory->legBuilder(swap_.leg().legType());
53  amortized_leg = legBuilder->buildLeg(swap_.leg(), engineFactory, requiredFixings_, configuration);
54  }
55 
56  DayCounter dc = Actual360();
57  if (!swap_.leg().dayCounter().empty()) {
58  dc = parseDayCounter(swap_.leg().dayCounter());
59  }
60 
61  // In general for CDS and CDS index trades, the standard day counter is Actual/360 and the final
62  // period coupon accrual includes the maturity date.
63  DayCounter lastPeriodDayCounter;
64  auto strLpdc = swap_.leg().lastPeriodDayCounter();
65  if (!strLpdc.empty()) {
66  lastPeriodDayCounter = parseDayCounter(strLpdc);
67  } else {
68  Actual360 standardDayCounter;
69  lastPeriodDayCounter = dc == standardDayCounter ? Actual360(true) : dc;
70  }
71 
72  boost::shared_ptr<FixedLegData> fixedData =
73  boost::dynamic_pointer_cast<FixedLegData>(swap_.leg().concreteLegData());
74  QL_REQUIRE(fixedData, "Wrong LegType, expected Fixed, got " << swap_.leg().legType());
75 
76  QL_REQUIRE(fixedData->rates().size() == 1, "CreditDefaultSwap requires single rate");
77 
78  boost::shared_ptr<QuantExt::CreditDefaultSwap> cds;
79 
80  if (swap_.leg().notionals().size() == 1) {
81  if (swap_.upfrontFee() == Null<Real>()) {
82  cds = boost::make_shared<QuantExt::CreditDefaultSwap>(
83  prot, swap_.leg().notionals().front(), fixedData->rates().front(), schedule, payConvention, dc,
84  swap_.settlesAccrual(), swap_.protectionPaymentTime(), swap_.protectionStart(),
85  boost::shared_ptr<Claim>(), lastPeriodDayCounter, swap_.tradeDate(), swap_.cashSettlementDays());
86  } else {
87  cds = boost::make_shared<QuantExt::CreditDefaultSwap>(
88  prot, notional_, swap_.upfrontFee(), fixedData->rates().front(), schedule, payConvention, dc,
89  swap_.settlesAccrual(), swap_.protectionPaymentTime(), swap_.protectionStart(), swap_.upfrontDate(),
90  boost::shared_ptr<Claim>(), lastPeriodDayCounter, swap_.tradeDate(), swap_.cashSettlementDays());
91  }
92  } else {
93  if (swap_.upfrontFee() == Null<Real>()) {
94  cds = boost::make_shared<QuantExt::CreditDefaultSwap>(
95  prot, notional_, amortized_leg, fixedData->rates().front(), schedule, payConvention, dc,
96  swap_.settlesAccrual(), swap_.protectionPaymentTime(), swap_.protectionStart(),
97  boost::shared_ptr<Claim>(), lastPeriodDayCounter, swap_.tradeDate(), swap_.cashSettlementDays());
98  } else {
99  cds = boost::make_shared<QuantExt::CreditDefaultSwap>(
100  prot, notional_, amortized_leg, swap_.upfrontFee(), fixedData->rates().front(), schedule, payConvention,
101  dc, swap_.settlesAccrual(), swap_.protectionPaymentTime(), swap_.protectionStart(), swap_.upfrontDate(),
102  boost::shared_ptr<Claim>(), lastPeriodDayCounter, swap_.tradeDate(), swap_.cashSettlementDays());
103  }
104  }
105 
106  boost::shared_ptr<CreditDefaultSwapEngineBuilder> cdsBuilder =
107  boost::dynamic_pointer_cast<CreditDefaultSwapEngineBuilder>(builder);
108 
109  npvCurrency_ = swap_.leg().currency();
110 
111  QL_REQUIRE(cdsBuilder, "No Builder found for CreditDefaultSwap: " << id());
112  cds->setPricingEngine(cdsBuilder->engine(parseCurrency(npvCurrency_), swap_.creditCurveId(), swap_.recoveryRate()));
113 
114  instrument_.reset(new VanillaInstrument(cds));
115 
116  maturity_ = cds->coupons().back()->date();
117 
118  legs_ = {cds->coupons()};
119  legCurrencies_ = {npvCurrency_};
120  legPayers_ = {swap_.leg().isPayer()};
121  notionalCurrency_ = swap_.leg().currency();
122 }

124 QuantLib::Real CreditDefaultSwap::notional() const {
125  Date asof = Settings::instance().evaluationDate();
126  // get the current notional from CDS premium leg
127  for (Size i = 0; i < legs_[0].size(); ++i) {
128  boost::shared_ptr<Coupon> coupon = boost::dynamic_pointer_cast<Coupon>(legs_[0][i]);
129  if (coupon->date() > asof)
130  return coupon->nominal();
131  }
132 
133  // if no coupons are given, return the initial notional by default
134  return notional_;
135 }
