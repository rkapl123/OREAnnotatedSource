61 // Generate lookback dates
62 set<Date> generateLookbackDates(const Period& lookbackPeriod, const Calendar& calendar) {
63 
64  set<Date> dates;
65 
66  Date today = Settings::instance().evaluationDate();
67  Date lookback = calendar.advance(today, -lookbackPeriod);
68  do {
69  TLOG("Adding date " << io::iso_date(lookback) << " to fixings.");
70  dates.insert(lookback);
71  lookback = calendar.advance(lookback, 1 * Days);
72  } while (lookback <= today);
73 
74  return dates;
75 }

84 // Return the set of dates on which a fixing will be required, if any.
85 set<Date> needsForecast(const Date& fixingDate, const Date& today, const bool interpolated, const Frequency frequency,
86  const Period& availabilityLag) {
87 
88  set<Date> result;
89 
90  Date todayMinusLag = today - availabilityLag;
91  Date historicalFixingKnown = inflationPeriod(todayMinusLag, frequency).first - 1;
92 
93  pair<Date, Date> lim = inflationPeriod(fixingDate, frequency);
94  result.insert(lim.first);
95  Date latestNeededDate = fixingDate;
96  if (interpolated) {
97  if (fixingDate > lim.first) {
98  latestNeededDate += Period(frequency);
99  result.insert(lim.second + 1);
100  }
101  }
102 
103  if (latestNeededDate <= historicalFixingKnown) {
104  // Know that fixings are available
105  return result;
106  } else if (latestNeededDate > today) {
107  // Know that fixings are not available
108  return {};
109  } else {
110  // Grey area here but for now return nothing
111  return {};
112  }
113 }

115 // Common code for zero inflation index based coupons
116 void addZeroInflationDates(set<Date>& dates, const Date& fixingDate, const Date& today, const bool indexInterpolated,
117  const Frequency indexFrequency, const Period& indexAvailabilityLag,
118  const CPI::InterpolationType interpolation, const Frequency f) {
119 
120  set<Date> fixingDates;
121 
122  if (interpolation == CPI::AsIndex) {
123  fixingDates = needsForecast(fixingDate, today, indexInterpolated, indexFrequency, indexAvailabilityLag);
124  } else {
125  pair<Date, Date> lim = inflationPeriod(fixingDate, f);
126  fixingDates = needsForecast(lim.first, today, indexInterpolated, indexFrequency, indexAvailabilityLag);
127  if (interpolation == CPI::Linear) {
128  auto moreDates =
129  needsForecast(lim.second + 1, today, indexInterpolated, indexFrequency, indexAvailabilityLag);
130  fixingDates.insert(moreDates.begin(), moreDates.end());
131  }
132  }
133 
134  dates.insert(fixingDates.begin(), fixingDates.end());
135 }

155  std::set<ZeroInflationFixingEntry> newZeroInflationFixingDates;
156  std::set<InflationFixingEntry> newYoYInflationFixingDates;

171  yoyInflationFixingDates_ = newYoYInflationFixingDates;

174 std::map<std::string, std::set<Date>> RequiredFixings::fixingDatesIndices(const Date& settlementDate) const {
175 
176  // If settlement date is an empty date, update to evaluation date.
177  Date d = settlementDate == Date() ? Settings::instance().evaluationDate() : settlementDate;
178 
179  std::map<std::string, std::set<Date>> result;
180 
181  // handle the general case
182  for (auto const& f : fixingDates_) {
183  // get the data
184  std::string indexName = std::get<0>(f);
185  Date fixingDate = std::get<1>(f);
186  Date payDate = std::get<2>(f);
187  bool alwaysAddIfPaysOnSettlement = std::get<3>(f);
188  // add to result
189  if (fixingDate > d)
190  continue;
191  SimpleCashFlow dummyCf(0.0, payDate);
192  if (!dummyCf.hasOccurred(d) || (alwaysAddIfPaysOnSettlement && dummyCf.date() == d)) {
193  result[indexName].insert(fixingDate);
194  }
195  }
196 
197  // handle zero inflation index based coupons
198  for (auto const& f : zeroInflationFixingDates_) {
199  // get the data
200  InflationFixingEntry inflationFixingEntry = std::get<0>(f);
201  FixingEntry fixingEntry = std::get<0>(inflationFixingEntry);
202  std::string indexName = std::get<0>(fixingEntry);
203  Date fixingDate = std::get<1>(fixingEntry);
204  Date payDate = std::get<2>(fixingEntry);
205  bool alwaysAddIfPaysOnSettlement = std::get<3>(fixingEntry);
206  bool indexInterpolated = std::get<1>(inflationFixingEntry);
207  Frequency indexFrequency = std::get<2>(inflationFixingEntry);
208  Period indexAvailabilityLag = std::get<3>(inflationFixingEntry);
209  CPI::InterpolationType couponInterpolation = std::get<1>(f);
210  Frequency couponFrequency = std::get<2>(f);
211  // add to result
212  SimpleCashFlow dummyCf(0.0, payDate);
213  if (!dummyCf.hasOccurred(d) || (alwaysAddIfPaysOnSettlement && dummyCf.date() == d)) {
214  std::set<Date> tmp;
215  addZeroInflationDates(tmp, fixingDate, d, indexInterpolated, indexFrequency, indexAvailabilityLag,
216  couponInterpolation, couponFrequency);
217  if (!tmp.empty())
218  result[indexName].insert(tmp.begin(), tmp.end());
219  }
220  }
221 
222  // handle yoy inflation index based coupons
223  for (auto const& f : yoyInflationFixingDates_) {
224  // get the data
225  FixingEntry fixingEntry = std::get<0>(f);
226  std::string indexName = std::get<0>(fixingEntry);
227  Date fixingDate = std::get<1>(fixingEntry);
228  Date payDate = std::get<2>(fixingEntry);
229  bool alwaysAddIfPaysOnSettlement = std::get<3>(fixingEntry);
230  bool indexInterpolated = std::get<1>(f);
231  Frequency indexFrequency = std::get<2>(f);
232  Period indexAvailabilityLag = std::get<3>(f);
233  // add to result
234  SimpleCashFlow dummyCf(0.0, payDate);
235  if (!dummyCf.hasOccurred(d) || (alwaysAddIfPaysOnSettlement && dummyCf.date() == d)) {
236  auto fixingDates = needsForecast(fixingDate, d, indexInterpolated, indexFrequency, indexAvailabilityLag);
237  if (!fixingDates.empty())
238  result[indexName].insert(fixingDates.begin(), fixingDates.end());
239  // Add the previous year's date(s) also if any.
240  for (const auto& d : fixingDates) {
241  result[indexName].insert(d - 1 * Years);
242  }
243  }
244  }
245 
246  return result;
247 }

330 void FixingDateGetter::visit(IborCoupon& c) {
331  if (auto bma = boost::dynamic_pointer_cast<QuantExt::BMAIndexWrapper>(c.index())) {
332  // Handle bma indices which we allow in IborCoupon as an approximation to BMA
333  // coupons. For these we allow fixing dates that are invalid as BMA fixing dates
334  // and adjust these dates to the last valid BMA fixing date in the BMAIndexWrapper.
335  // It is this adjusted date that we want to record here.
336  // Enforce fixing to be added even if coupon pays on settlement.
337  requiredFixings_.addFixingDate(bma->adjustedFixingDate(c.fixingDate()), oreIndexName(c.index()->name()),
338  c.date(), true);
339  } else {
340  auto fallback = boost::dynamic_pointer_cast<FallbackIborIndex>(c.index());
341  if (fallback != nullptr && c.fixingDate() >= fallback->switchDate()) {
342  requiredFixings_.addFixingDates(fallback->onCoupon(c.fixingDate())->fixingDates(),
343  oreIndexName(fallback->rfrIndex()->name()), c.date());
344  } else {
345  visit(static_cast<FloatingRateCoupon&>(c));
346  }
347  }
348 }

460 void amendInflationFixingDates(map<string, set<Date>>& fixings, const boost::shared_ptr<Conventions>& conventions) {
461  // Loop over indices and amend any that are of type InflationIndex
462  for (auto& kv : fixings) {
463  auto p = isInflationIndex(kv.first, conventions);
464  if (p.first) {
465  // We have an inflation index
466  set<Date> newDates;
467  Frequency f = p.second->frequency();
468  for (const Date& d : kv.second) {
469  auto period = inflationPeriod(d, f);
470  if (d == period.first) {
471  // If the fixing date is the start of the inflation period, move it to the end.
472  newDates.insert(period.second);
473  } else {
474  // If the fixing date is not the start of the inflation period, leave it as it is.
475  newDates.insert(d);
476  }
477  }
478  // Update the fixings map that was passed in with the new set of dates
479  kv.second = newDates;
480  }
481  }
482 }

484 void addMarketFixingDates(map<string, set<Date>>& fixings, const TodaysMarketParameters& mktParams,
485  const Conventions& conventions, const Period& iborLookback, const Period& oisLookback,
486  const Period& bmaLookback, const Period& inflationLookback, const string& configuration) {
487 
488  if (mktParams.hasConfiguration(configuration)) {
489 
490  LOG("Start adding market fixing dates for configuration '" << configuration << "'");
491 
492  // If there are ibor indices in the market parameters, add the lookback fixings
493  // IF there are SIFMA / BMA indices, add lookback fixings for the Libor basis index
494  if (mktParams.hasMarketObject(MarketObject::IndexCurve)) {
495 
496  QL_REQUIRE(iborLookback >= 0 * Days, "Ibor lookback period must be non-negative");
497 
498  DLOG("Start adding market fixing dates for interest rate indices.");
499 
500  set<Date> iborDates;
501  set<Date> oisDates;
502  set<Date> bmaDates;
503  WeekendsOnly calendar;
504 
505  // For each of the IR indices in market parameters, insert the dates
506  for (const auto& kv : mktParams.mapping(MarketObject::IndexCurve, configuration)) {
507  if (isOvernightIndex(kv.first)) {
508  if (oisDates.empty()) {
509  TLOG("Generating fixing dates for overnight indices.");
510  oisDates = generateLookbackDates(oisLookback, calendar);
511  }
512  TLOG("Adding extra fixing dates for overnight index " << kv.first);
513  fixings[kv.first].insert(oisDates.begin(), oisDates.end());
514  } else if (isBmaIndex(kv.first)) {
515  if (bmaDates.empty()) {
516  TLOG("Generating fixing dates for bma/sifma indices.");
517  bmaDates = generateLookbackDates(bmaLookback, calendar);
518  }
519  fixings[kv.first].insert(bmaDates.begin(), bmaDates.end());
520  if (iborDates.empty()) {
521  TLOG("Generating fixing dates for ibor indices.");
522  iborDates = generateLookbackDates(iborLookback, calendar);
523  }
524  std::set<string> liborNames;
525  for (auto const& c : conventions.get(Convention::Type::BMABasisSwap)) {
526  auto bma = boost::dynamic_pointer_cast<BMABasisSwapConvention>(c);
527  QL_REQUIRE(
528  bma, "internal error, could not cast to BMABasisSwapConvention in addMarketFixingDates()");
529  if (bma->bmaIndexName() == kv.first) {
530  liborNames.insert(bma->liborIndexName());
531  }
532  }
533  for (auto const& l : liborNames) {
534  TLOG("Adding extra fixing dates for libor index " << l << " from bma/sifma index " << kv.first);
535  fixings[l].insert(iborDates.begin(), iborDates.end());
536  }
537  } else {
538  if (iborDates.empty()) {
539  TLOG("Generating fixing dates for ibor indices.");
540  iborDates = generateLookbackDates(iborLookback, calendar);
541  }
542  TLOG("Adding extra fixing dates for ibor index " << kv.first);
543  fixings[kv.first].insert(iborDates.begin(), iborDates.end());
544  }
545  }
546 
547  DLOG("Finished adding market fixing dates for interest rate indices.");
548  }
549 
550  // If there are inflation indices in the market parameters, add the lookback fixings
551  if (mktParams.hasMarketObject(MarketObject::ZeroInflationCurve) ||
552  mktParams.hasMarketObject(MarketObject::YoYInflationCurve)) {
553 
554  QL_REQUIRE(inflationLookback >= 0 * Days, "Inflation lookback period must be non-negative");
555 
556  // Dates that will be used for each of the inflation indices
557  Date today = Settings::instance().evaluationDate();
558  Date lookback = NullCalendar().advance(today, -inflationLookback);
559  lookback = Date(1, lookback.month(), lookback.year());
560  set<Date> dates;
561  do {
562  TLOG("Adding date " << io::iso_date(lookback) << " to fixings for inflation indices");
563  dates.insert(lookback);
564  lookback = NullCalendar().advance(lookback, 1 * Months);
565  } while (lookback <= today);
566 
567  // For each of the inflation indices in market parameters, insert the dates
568  if (mktParams.hasMarketObject(MarketObject::ZeroInflationCurve)) {
569  for (const auto& kv : mktParams.mapping(MarketObject::ZeroInflationCurve, configuration)) {
570  TLOG("Adding extra fixing dates for (zero) inflation index " << kv.first);
571  fixings[kv.first].insert(dates.begin(), dates.end());
572  }
573  }
574 
575  if (mktParams.hasMarketObject(MarketObject::YoYInflationCurve)) {
576  for (const auto& kv : mktParams.mapping(MarketObject::YoYInflationCurve, configuration)) {
577  TLOG("Adding extra fixing dates for (yoy) inflation index " << kv.first);
578  fixings[kv.first].insert(dates.begin(), dates.end());
579  }
580  }
581  }
582 
583  // If there are commodity curves, add "fixings" for this month and two previous months. We add "fixings" for
584  // future contracts with expiry from two months hence to two months prior.
585  if (mktParams.hasMarketObject(MarketObject::CommodityCurve)) {
586 
587  // "Fixing" dates for commodities.
588  Period commodityLookback = 2 * Months;
589  Date today = Settings::instance().evaluationDate();
590  Date lookback = today - commodityLookback;
591  lookback = Date(1, lookback.month(), lookback.year());
592  set<Date> dates;
593  do {
594  TLOG("Adding date " << io::iso_date(lookback) << " to fixings for commodities");
595  dates.insert(lookback++);
596  } while (lookback <= today);
597 
598  // Expiry months and years for which we require future contract fixings. For our purposes here, using the
599  // 1st of the month does not matter. We will just use the date to get the appropriate commodity future
600  // index name below when adding the dates and the "-01" will be removed (for non-daily contracts).
601  Size numberMonths = 2;
602  vector<Date> contractExpiries;
603  Date startContract = today - numberMonths * Months;
604  Date endContract = today + numberMonths * Months;
605  do {
606  Month m = startContract.month();
607  Year y = startContract.year();
608  TLOG("Adding contract month and year (" << m << "," << y << ")");
609  contractExpiries.push_back(Date(1, m, y));
610  startContract += 1 * Months;
611  } while (startContract <= endContract);
612 
613  // For each of the commodity names, create the future contract name with the relevant expiry and insert
614  // the dates. Skip commodity names that do not have future conventions.
615  for (const auto& kv : mktParams.mapping(MarketObject::CommodityCurve, configuration)) {
616 
617  boost::shared_ptr<CommodityFutureConvention> cfc;
618  if (conventions.has(kv.first)) {
619  cfc = boost::dynamic_pointer_cast<CommodityFutureConvention>(conventions.get(kv.first));
620  }
621 
622  auto commIdx = parseCommodityIndex(kv.first, conventions, false);
623  if (cfc) {
624  if (auto oppIdx = boost::dynamic_pointer_cast<OffPeakPowerIndex>(commIdx)) {
625  DLOG("Commodity " << kv.first << " is off-peak power so adding underlying daily contracts.");
626  const auto& opIndex = oppIdx->offPeakIndex();
627  const auto& pIndex = oppIdx->peakIndex();
628  for (const Date& expiry : dates) {
629  auto tmpIdx = oppIdx->clone(expiry);
630  auto opName = opIndex->clone(expiry)->name();
631  TLOG("Adding (date, id) = (" << io::iso_date(expiry) << "," << opName << ")");
632  fixings[opName].insert(expiry);
633  auto pName = pIndex->clone(expiry)->name();
634  TLOG("Adding (date, id) = (" << io::iso_date(expiry) << "," << pName << ")");
635  fixings[pName].insert(expiry);
636  }
637  } else if (cfc->contractFrequency() == Daily) {
638  DLOG("Commodity " << kv.first << " has daily frequency so adding daily contracts.");
639  for (const Date& expiry : dates) {
640  auto indexName = commIdx->clone(expiry)->name();
641  TLOG("Adding (date, id) = (" << io::iso_date(expiry) << "," << indexName << ")");
642  fixings[indexName].insert(expiry);
643  }
644  } else {
645  DLOG("Commodity " << kv.first << " is not daily so adding the monthly contracts.");
646  for (const Date& expiry : contractExpiries) {
647  auto indexName = commIdx->clone(expiry)->name();
648  TLOG("Adding extra fixing dates for commodity future " << indexName);
649  fixings[indexName].insert(dates.begin(), dates.end());
650  }
651  }
652  } else {
653  // Assumption here is that we have a spot index.
654  DLOG("Commodity " << kv.first << " does not have future conventions so adding daily fixings.");
655  auto indexName = commIdx->name();
656  TLOG("Adding extra fixing dates for commodity spot " << indexName);
657  fixings[indexName].insert(dates.begin(), dates.end());
658  }
659  }
660  }
661 
662  LOG("Finished adding market fixing dates for configuration '" << configuration << "'");
663  }
