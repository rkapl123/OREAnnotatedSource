44// helper that converts a context value to a ql additional result (i.e. boost::any)
45
46struct anyGetterCg : public boost::static_visitor<boost::any> {
47 boost::any operator()(const RandomVariable& x) const { QL_FAIL("unexpected call to anyGetter (RandomVariable)"); }
48 boost::any operator()(const EventVec& x) const { return x.value; }
49 boost::any operator()(const IndexVec& x) const { return x.value; }
50 boost::any operator()(const CurrencyVec& x) const { return x.value; }
51 boost::any operator()(const DaycounterVec& x) const { return x.value; }
52 boost::any operator()(const Filter& x) const { QL_FAIL("unexpected call to anyGetter (Filter)"); }
53};
54
55boost::any valueToAnyCg(const ValueType& v) { return boost::apply_visitor(anyGetterCg(), v); }
56
57double externalAverage(const std::vector<double>& v) {
58 boost::accumulators::accumulator_set<double, boost::accumulators::stats<boost::accumulators::tag::mean>> acc;
59 for (auto const& x : v) {
60 acc(x);
61 }
62 return boost::accumulators::mean(acc);
63}

67void ScriptedInstrumentPricingEngineCG::calculate() const {
68
69 // TODOs
70 QL_REQUIRE(!useExternalComputeFramework_ || !generateAdditionalResults_,
71 "ScriptedInstrumentPricingEngineCG: when using external compute framework, generation of additional "
72 "results is not supported yet.");
73 QL_REQUIRE(!useExternalComputeFramework_ || !useCachedSensis_,
74 "ScriptedInstrumentPricingEngineCG: when using external compute framework, usage of cached sensis is "
75 "not supported yet");
76 QL_REQUIRE(model_->trainingPaths() == Null<Size>(), "ScriptedInstrumentPricingEngineCG: separate training phase "
77 "not supported, trainingSamples can not be specified.");
78
79 lastCalculationWasValid_ = false;
80
81 // build cg, if necessary
82
83 auto g = model_->computationGraph();
84
85 if (cgVersion_ != model_->cgVersion()) {
86
87 // clear NPVMem() regression coefficients
88
89 model_->resetNPVMem();
90
91 // set up copy of initial context to run the cg builder against
92
93 workingContext_ = boost::make_shared<Context>(*context_);
94
95 // set TODAY in the context
96
97 checkDuplicateName(workingContext_, "TODAY");
98 Date referenceDate = model_->referenceDate();
99 workingContext_->scalars["TODAY"] = EventVec{model_->size(), referenceDate};
100 workingContext_->constants.insert("TODAY");
101
102 // set variables from initial context
103
104 for (auto const& v : workingContext_->scalars) {
105 if (v.second.which() == ValueTypeWhich::Number) {
106 auto r = boost::get<RandomVariable>(v.second);
107 QL_REQUIRE(r.deterministic(), "ScriptedInstrumentPricingEngineCG::calculate(): expected variable '"
108 << v.first << "' from initial context to be deterministic, got "
109 << r);
110 g->setVariable(v.first + "_0", cg_const(*g, r.at(0)));
111 }
112 }
113
114 for (auto const& a : workingContext_->arrays) {
115 for (Size i = 0; i < a.second.size(); ++i) {
116 if (a.second[i].which() == ValueTypeWhich::Number) {
117 auto r = boost::get<RandomVariable>(a.second[i]);
118 QL_REQUIRE(r.deterministic(), "ScriptedInstrumentPricingEngineCG::calculate(): expected variable '"
119 << a.first << "[" << i
120 << "]' from initial context to be deterministic, got " << r);
121 g->setVariable(a.first + "_" + std::to_string(i), cg_const(*g, r.at(0)));
122 }
123 }
124 }
125
126 // get ops + labels, grads and op node requirements
127
128 opLabels_ = getRandomVariableOpLabels();
129 opNodeRequirements_ = getRandomVariableOpNodeRequirements();
130 if (useExternalComputeFramework_) {
131 opsExternal_ = getExternalRandomVariableOps();
132 gradsExternal_ = getExternalRandomVariableGradients();
133 } else {
134 ops_ = getRandomVariableOps(model_->size());
135 grads_ = getRandomVariableGradients(model_->size());
136 }
137
138 // build graph
139
140 ComputationGraphBuilder cgBuilder(*g, opLabels_, ast_, workingContext_, model_);
141 cgBuilder.run(generateAdditionalResults_, script_, interactive_);
142 cgVersion_ = model_->cgVersion();
143 DLOG("Built computation graph version " << cgVersion_ << " size is " << g->size());
144 TLOGGERSTREAM(ssaForm(*g, opLabels_));
145 keepNodes_ = cgBuilder.keepNodes();
146 payLogEntries_ = cgBuilder.payLogEntries();
147
148 // clear stored base model params
149
150 haveBaseValues_ = false;
151 }
152
153 if (!haveBaseValues_ || !useCachedSensis_) {
154
155 // calculate NPV and Sensis ("base scenario"), store base npv + sensis + base model params
156
157 bool newExternalCalc = false;
158 if (useExternalComputeFramework_) {
159 QL_REQUIRE(ComputeEnvironment::instance().hasContext(),
160 "ScriptedInstrumentPricingEngineCG::calculate(): no compute enviroment context selected.");
161 std::tie(externalCalculationId_, newExternalCalc) =
162 ComputeEnvironment::instance().context().initiateCalculation(model_->size(), externalCalculationId_,
163 cgVersion_);
164 DLOG("initiated external calculation id " << externalCalculationId_ << ", version " << cgVersion_);
165 }
166
167 // populate values
168
169 std::vector<RandomVariable> values(g->size(), RandomVariable(model_->size()));
170
171 std::vector<ExternalRandomVariable> valuesExternal;
172 if (useExternalComputeFramework_)
173 valuesExternal.resize(g->size());
174
175 // set constants
176
177 for (auto const& c : g->constants()) {
178 if (useExternalComputeFramework_) {
179 valuesExternal[c.second] = ExternalRandomVariable(c.first);
180 } else {
181 values[c.second] = RandomVariable(model_->size(), c.first);
182 }
183 }
184
185 // set model parameters
186
187 baseModelParams_ = model_->modelParameters();
188 for (auto const& p : baseModelParams_) {
189 TLOG("setting model parameter at node " << p.first << " to value " << std::setprecision(16) << p.second);
190 if (useExternalComputeFramework_) {
191 valuesExternal[p.first] = ExternalRandomVariable(p.second);
192 } else {
193 values[p.first] = RandomVariable(model_->size(), p.second);
194 }
195 }
196 DLOG("set " << baseModelParams_.size() << " model parameters");
197
198 // set random variates
199
200 auto const& rv = model_->randomVariates();
201 if (!rv.empty()) {
202 if (useExternalComputeFramework_) {
203 auto gen =
204 ComputeEnvironment::instance().context().createInputVariates(rv.size(), rv.front().size(), 42);
205 for (Size k = 0; k < rv.size(); ++k) {
206 for (Size j = 0; j < rv.front().size(); ++j)
207 valuesExternal[rv[k][j]] = ExternalRandomVariable(gen[k][j]);
208 }
209 } else {
210 auto gen =
211 makeMultiPathVariateGenerator(mcParams_.sequenceType, rv.size(), rv.front().size(), mcParams_.seed,
212 mcParams_.sobolOrdering, mcParams_.sobolDirectionIntegers);
213 for (Size path = 0; path < model_->size(); ++path) {
214 auto p = gen->next();
215 for (Size j = 0; j < rv.front().size(); ++j) {
216 for (Size k = 0; k < rv.size(); ++k) {
217 values[rv[k][j]].set(path, p.value[j][k]);
218 }
219 }
220 }
221 }
222 DLOG("generated random variates for dim = " << rv.size() << ", steps = " << rv.front().size());
223 }
224
225 // set flags for nodes we want to keep (model params, npv and additional results)
226
227 std::vector<bool> keepNodes(g->size(), false);
228
229 keepNodes[cg_var(*g, npv_ + "_0")] = true;
230
231 for (auto const& p : baseModelParams_)
232 keepNodes[p.first] = true;
233
234 if (generateAdditionalResults_) {
235 for (auto const& r : additionalResults_) {
236 auto s = workingContext_->scalars.find(r.second);
237 if (s != workingContext_->scalars.end() && s->second.which() == ValueTypeWhich::Number) {
238 keepNodes[cg_var(*g, r.second + "_0")] = true;
239 }
240 auto v = workingContext_->arrays.find(r.second);
241 if (v != workingContext_->arrays.end()) {
242 for (Size i = 0; i < v->second.size(); ++i) {
243 keepNodes[cg_var(*g, r.second + "_" + std::to_string(i))] = true;
244 }
245 }
246 }
247 for (auto const& n : keepNodes_)
248 keepNodes[n] = true;
249 }
250
251 // run the forward evaluation
252
253 std::size_t baseNpvNode = cg_var(*g, npv_ + "_0");
254
255 if (useExternalComputeFramework_) {
256 if (newExternalCalc) {
257 forwardEvaluation(*g, valuesExternal, opsExternal_, ExternalRandomVariable::deleter, useCachedSensis_,
258 opNodeRequirements_, keepNodes);
259 valuesExternal[baseNpvNode].declareAsOutput();
260 externalOutput_ = std::vector<std::vector<double>>(1, std::vector<double>(model_->size()));
261 externalOutputPtr_ = std::vector<double*>(1, &externalOutput_.front()[0]);
262 DLOG("ran forward evaluation");
263 }
264 } else {
265 forwardEvaluation(*g, values, ops_, RandomVariable::deleter, useCachedSensis_, opNodeRequirements_,
266 keepNodes);
267 DLOG("ran forward evaluation");
268 }
269
270 TLOGGERSTREAM(ssaForm(*g, opLabels_, values));
271
272 // extract npv result and set it
273
274 if (useExternalComputeFramework_) {
275 ComputeEnvironment::instance().context().finalizeCalculation(externalOutputPtr_);
276 baseNpv_ = results_.value = externalAverage(externalOutput_[0]);
277 } else {
278 baseNpv_ = results_.value = model_->extractT0Result(values[baseNpvNode]);
279 }
280
281 DLOG("got NPV = " << results_.value << " " << model_->baseCcy());
282
283 // extract additional results
284
285 if (generateAdditionalResults_) {
286
287 instrumentAdditionalResults_.clear();
288
289 for (auto const& r : additionalResults_) {
290
291 auto s = workingContext_->scalars.find(r.second);
292 bool resultSet = false;
293 if (s != workingContext_->scalars.end()) {
294 if (s->second.which() == ValueTypeWhich::Number) {
295 instrumentAdditionalResults_[r.first] =
296 model_->extractT0Result(values[cg_var(*g, r.second + "_0")]);
297 } else {
298 boost::any t = valueToAnyCg(s->second);
299 instrumentAdditionalResults_[r.first] = t;
300 }
301 DLOG("got additional result '" << r.first << "' referencing script variable '" << r.second << "'");
302 resultSet = true;
303 }
304 auto v = workingContext_->arrays.find(r.second);
305 if (v != workingContext_->arrays.end()) {
306 QL_REQUIRE(!resultSet, "result variable '"
307 << r.first << "' referencing script variable '" << r.second
308 << "' appears both as a scalar and an array, this is unexpected");
309 QL_REQUIRE(!v->second.empty(), "result variable '" << v->first << "' is an empty array.");
310 std::vector<double> tmpdouble;
311 std::vector<std::string> tmpstring;
312 std::vector<QuantLib::Date> tmpdate;
313 for (Size i = 0; i < v->second.size(); ++i) {
314 if (v->second[i].which() == ValueTypeWhich::Number) {
315 tmpdouble.push_back(
316 model_->extractT0Result(values[cg_var(*g, r.second + "_" + std::to_string(i))]));
317 } else {
318 boost::any t = valueToAnyCg(v->second[i]);
319 if (t.type() == typeid(std::string))
320 tmpstring.push_back(boost::any_cast<std::string>(t));
321 else if (t.type() == typeid(QuantLib::Date))
322 tmpdate.push_back(boost::any_cast<QuantLib::Date>(t));
323 else {
324 QL_FAIL("unexpected result type '" << t.type().name() << "' for result variable '"
325 << r.first << "' referencing script variable '"
326 << r.second << "'");
327 }
328 }
329 }
330 QL_REQUIRE((int)!tmpdouble.empty() + (int)!tmpstring.empty() + (int)!tmpdate.empty() == 1,
331 "expected exactly one result type in result array '" << v->first << "'");
332 DLOG("got additional result '" << r.first << "' referencing script variable '" << r.second
333 << "' vector of size "
334 << tmpdouble.size() + tmpstring.size() + tmpdate.size());
335 if (!tmpdouble.empty())
336 instrumentAdditionalResults_[r.first] = tmpdouble;
337 else if (!tmpstring.empty())
338 instrumentAdditionalResults_[r.first] = tmpstring;
339 else if (!tmpdate.empty())
340 instrumentAdditionalResults_[r.first] = tmpdate;
341 else {
342 QL_FAIL("got empty result vector for result variable '"
343 << r.first << "' referencing script variable '" << r.second << "', this is unexpected");
344 }
345 resultSet = true;
346 }
347 QL_REQUIRE(resultSet, "could not set additional result '"
348 << r.first << "' referencing script variable '" << r.second << "'");
349 }
350
351 // set contents from paylog as additional results
352
353 auto paylog = boost::make_shared<PayLog>();
354 for (auto const& p : payLogEntries_) {
355 paylog->write(values[p.value],
356 !close_enough(values[p.filter], RandomVariable(values[p.filter].size(), 0.0)), p.obs,
357 p.pay, p.ccy, p.legNo, p.cashflowType, p.slot);
358 }
359
360 paylog->consolidateAndSort();
361 std::vector<CashFlowResults> cashFlowResults(paylog->size());
362 for (Size i = 0; i < paylog->size(); ++i) {
363 // cashflow is written as expectation of deflated base ccy amount at T0, converted to flow ccy
364 // with the T0 FX Spot and compounded back to the pay date on T0 curves
365 Real fx = model_->getDirectFxSpotT0(paylog->currencies().at(i), model_->baseCcy());
366 Real discount = model_->getDirectDiscountT0(paylog->dates().at(i), paylog->currencies().at(i));
367 cashFlowResults[i].amount = model_->extractT0Result(paylog->amounts().at(i)) / fx / discount;
368 cashFlowResults[i].payDate = paylog->dates().at(i);
369 cashFlowResults[i].currency = paylog->currencies().at(i);
370 cashFlowResults[i].legNumber = paylog->legNos().at(i);
371 cashFlowResults[i].type = paylog->cashflowTypes().at(i);
372 DLOG("got cashflow " << QuantLib::io::iso_date(cashFlowResults[i].payDate) << " "
373 << cashFlowResults[i].currency << cashFlowResults[i].amount << " "
374 << cashFlowResults[i].currency << "-" << model_->baseCcy() << " " << fx
375 << "discount(" << cashFlowResults[i].currency << ") " << discount);
376 }
377 instrumentAdditionalResults_["cashFlowResults"] = cashFlowResults;
378
379 // set additional results from the model
380
381 instrumentAdditionalResults_.insert(model_->additionalResults().begin(), model_->additionalResults().end());
382
383 } // if generate additional results
384
385 // if the engine is amc enabled, add an amc calculator to the additional results
386
387 if (amcEnabled_) {
388 QL_FAIL("AMC not supported by scripted instrument pricing engine CG");
389 // DLOG("add amc calculator to results");
390 // results_.additionalResults["amcCalculator"] =
391 // boost::static_pointer_cast<AmcCalculator>(boost::make_shared<ScriptedInstrumentAmcCalculator>(
392 // npv_, model_, ast_, context_, script_, interactive_, amcStickyCloseOutStates_));
393 }
394
395 if (useCachedSensis_) {
396
397 // extract sensis and store them
398
399 std::vector<RandomVariable> derivatives(g->size(), RandomVariable(model_->size(), 0.0));
400 derivatives[cg_var(*g, npv_ + "_0")] = RandomVariable(model_->size(), 1.0);
401 backwardDerivatives(*g, values, derivatives, grads_, RandomVariable::deleter, keepNodes);
402
403 sensis_.resize(baseModelParams_.size());
404 for (Size i = 0; i < baseModelParams_.size(); ++i) {
405 sensis_[i] = model_->extractT0Result(derivatives[baseModelParams_[i].first]);
406 }
407 DLOG("got backward sensitivities");
408
409 // set flag indicating that we can use cached sensis in subsequent calculations
410
411 haveBaseValues_ = true;
412 }
413
414 } else {
415
416 // useCachedSensis => calculate npv from stored base npv, sensis, model params
417
418 auto modelParams = model_->modelParameters();
419
420 double npv = baseNpv_;
421 DLOG("computing npv using baseNpv " << baseNpv_ << " and sensis.");
422
423 for (Size i = 0; i < baseModelParams_.size(); ++i) {
424 QL_REQUIRE(modelParams[i].first == baseModelParams_[i].first, "internal error: modelParams["
425 << i << "] node " << modelParams[i].first
426 << " does not match baseModelParams node "
427 << baseModelParams_[i].first);
428 Real tmp = sensis_[i] * (modelParams[i].second - baseModelParams_[i].second);
429 npv += tmp;
430 DLOG("node " << modelParams[i].first << ": [" << modelParams[i].second << " (current) - "
431 << baseModelParams_[i].second << " (base) ] * " << sensis_[i] << " (delta) => " << tmp);
432 }
433
434 results_.value = npv;
435 }
436
437 if (generateAdditionalResults_) {
438 results_.additionalResults = instrumentAdditionalResults_;
439 }
440
441 lastCalculationWasValid_ = true;
442}
