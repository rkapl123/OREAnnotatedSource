49void Portfolio::fromXML(XMLNode* node) {
50 XMLUtils::checkNode(node, "Portfolio");
51 vector<XMLNode*> nodes = XMLUtils::getChildrenNodes(node, "Trade");
52 for (Size i = 0; i < nodes.size(); i++) {
53 string tradeType = XMLUtils::getChildValue(nodes[i], "TradeType", true);
54
55 // Get the id attribute
56 string id = XMLUtils::getAttribute(nodes[i], "id");
57 QL_REQUIRE(id != "", "No id attribute in Trade Node");
58 DLOG("Parsing trade id:" << id);
59
60 boost::shared_ptr<Trade> trade;
61 bool failedToLoad = true;
62 try {
63 trade = TradeFactory::instance().build(tradeType);
64 trade->fromXML(nodes[i]);
65 trade->id() = id;
66 add(trade);
67 DLOG("Added Trade " << id << " (" << trade->id() << ")"
68 << " type:" << tradeType);
69 failedToLoad = false;
70 } catch (std::exception& ex) {
71 ALOG(StructuredTradeErrorMessage(id, tradeType, "Error parsing Trade XML", ex.what()));
72 }
73
74 // If trade loading failed, then insert a dummy trade with same id and envelope
75 if (failedToLoad && buildFailedTrades_) {
76 try {
77 trade = TradeFactory::instance().build("Failed");
78 // this loads only type, id and envelope, but type will be set to the original trade's type
79 trade->fromXML(nodes[i]);
80 // create a dummy trade of type "Dummy"
81 boost::shared_ptr<FailedTrade> failedTrade = boost::make_shared<FailedTrade>();
82 // copy id and envelope
83 failedTrade->id() = id;
84 failedTrade->setUnderlyingTradeType(tradeType);
85 failedTrade->envelope() = trade->envelope();
86 // and add it to the portfolio
87 add(failedTrade);
88 WLOG("Added trade id " << failedTrade->id() << " type " << failedTrade->tradeType()
89 << " for original trade type " << trade->tradeType());
90 } catch (std::exception& ex) {
91 ALOG(StructuredTradeErrorMessage(id, tradeType, "Error parsing type and envelope", ex.what()));
92 }
93 }
94 }
95 LOG("Finished Parsing XML doc");
96}

110void Portfolio::removeMatured(const Date& asof) {
111 for (auto it = trades_.begin(); it != trades_.end(); /* manual */) {
112 if ((*it).second->maturity() <= asof) {
113 ALOG(StructuredTradeErrorMessage((*it).second, "", "Trade is Matured"));

116 ++it;

121void Portfolio::build(const boost::shared_ptr<EngineFactory>& engineFactory, const std::string& context,
122 const bool emitStructuredError) {
123 LOG("Building Portfolio of size " << trades_.size() << " for context = '" << context << "'");
124 auto trade = trades_.begin();
125 Size initialSize = trades_.size();
126 Size failedTrades = 0;
127 while (trade != trades_.end()) {
128 auto [ft, success] =
129 buildTrade((*trade).second, engineFactory, context, buildFailedTrades(), emitStructuredError);
130 if (success) {
131 ++trade;
132 } else if (ft) {
133 (*trade).second = ft;
134 ++failedTrades;
135 ++trade;
136 } else {
137 trade = trades_.erase(trade);
138 }
139 }
140 LOG("Built Portfolio. Initial size = " << initialSize << ", size now " << trades_.size() << ", built "
141 << failedTrades << " failed trades, context is " + context);
142
143 QL_REQUIRE(trades_.size() > 0, "Portfolio does not contain any built trades, context is '" + context + "'");
144}

146Date Portfolio::maturity() const {
147 QL_REQUIRE(trades_.size() > 0, "Cannot get maturity of an empty portfolio");
148 Date mat = Date::minDate();
149 for (const auto& t : trades_)
150 mat = std::max(mat, t.second->maturity());
151 return mat;
152}

233std::map<AssetClass, std::set<std::string>>
234Portfolio::underlyingIndices(const boost::shared_ptr<ReferenceDataManager>& referenceDataManager) {
235
236 if (!underlyingIndicesCache_.empty())
237 return underlyingIndicesCache_;
238
239 map<AssetClass, std::set<std::string>> result;
240
241 for (const auto& t : trades_) {
242 try {
243 auto underlyings = t.second->underlyingIndices(referenceDataManager);
244 for (const auto& kv : underlyings) {
245 result[kv.first].insert(kv.second.begin(), kv.second.end());
246 }
247 } catch (const std::exception& e) {
248 ALOG(StructuredTradeErrorMessage(t.second->id(), t.second->tradeType(),
249 "Error retrieving underlying indices", e.what()));
250 }
251 }
252 underlyingIndicesCache_ = result;
253 return underlyingIndicesCache_;
254}

268std::pair<boost::shared_ptr<Trade>, bool> buildTrade(boost::shared_ptr<Trade>& trade,
269 const boost::shared_ptr<EngineFactory>& engineFactory,
270 const std::string& context, const bool buildFailedTrades,
271 const bool emitStructuredError) {
272 try {
273 trade->reset();
274 trade->build(engineFactory);
275 TLOG("Required Fixings for trade " << trade->id() << ":");
276 TLOGGERSTREAM(trade->requiredFixings());
277 return std::make_pair(nullptr, true);
278 } catch (std::exception& e) {
279 if (emitStructuredError) {
280 ALOG(StructuredTradeErrorMessage(trade, "Error building trade for context '" + context + "'", e.what()));
281 } else {
282 ALOG("Error building trade '" << trade->id() << "' for context '" + context + "': " + e.what());
283 }
284 if (buildFailedTrades) {
285 boost::shared_ptr<FailedTrade> failed = boost::make_shared<FailedTrade>();
286 failed->id() = trade->id();
287 failed->setUnderlyingTradeType(trade->tradeType());
288 failed->envelope() = trade->envelope();
289 failed->build(engineFactory);
290 failed->resetPricingStats(trade->getNumberOfPricings(), trade->getCumulativePricingTime());
291 LOG("Built failed trade with id " << failed->id());
292 return std::make_pair(failed, false);
293 } else {
294 return std::make_pair(nullptr, false);
295 }
296 }
