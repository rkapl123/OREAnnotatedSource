69void Portfolio::fromXML(XMLNode* node, const boost::shared_ptr<TradeFactory>& factory,
70 const bool checkForDuplicateIds) {
71 XMLUtils::checkNode(node, "Portfolio");
72 vector<XMLNode*> nodes = XMLUtils::getChildrenNodes(node, "Trade");
73 for (Size i = 0; i < nodes.size(); i++) {
74 string tradeType = XMLUtils::getChildValue(nodes[i], "TradeType", true);
75
76 // Get the id attribute
77 string id = XMLUtils::getAttribute(nodes[i], "id");
78 QL_REQUIRE(id != "", "No id attribute in Trade Node");
79 DLOG("Parsing trade id:" << id);
80 boost::shared_ptr<Trade> trade = factory->build(tradeType);
81
82 bool failedToLoad = false;
83 if (trade) {
84 try {
85 trade->fromXML(nodes[i]);
86 trade->id() = id;
87 add(trade, checkForDuplicateIds);
88
89 DLOG("Added Trade " << id << " (" << trade->id() << ")"
90 << " type:" << tradeType);
91 } catch (std::exception& ex) {
92 ALOG(StructuredTradeErrorMessage(id, tradeType, "Error parsing Trade XML", ex.what()));
93 failedToLoad = true;
94 }
95 } else {
96 ALOG(StructuredTradeErrorMessage(id, tradeType, "Error parsing Trade XML"));
97 failedToLoad = true;
98 }
99
100 // If trade loading failed, then insert a dummy trade with same id and envelope
101 if (failedToLoad && buildFailedTrades_) {
102 try {
103 trade = factory->build("Failed");
104 // this loads only type, id and envelope, but type will be set to the original trade's type
105 trade->fromXML(nodes[i]);
106 // create a dummy trade of type "Dummy"
107 boost::shared_ptr<FailedTrade> failedTrade = boost::make_shared<FailedTrade>();
108 // copy id and envelope
109 failedTrade->id() = id;
110 failedTrade->setUnderlyingTradeType(tradeType);
111 failedTrade->envelope() = trade->envelope();
112 // and add it to the portfolio
113 add(failedTrade, checkForDuplicateIds);
114 WLOG("Added trade id " << failedTrade->id() << " type " << failedTrade->tradeType()
115 << " for original trade type " << trade->tradeType());
116 } catch (std::exception& ex) {
117 ALOG(StructuredTradeErrorMessage(id, tradeType, "Error parsing type and envelope", ex.what()));
118 }
119 }
120 }
121 LOG("Finished Parsing XML doc");
122}

146 tradeLookup_.erase(tradeID);
147 for (auto it = trades_.begin(); it != trades_.end(); ++it) {
148 if ((*it)->id() == tradeID) {

151 }
152 }

156void Portfolio::removeMatured(const Date& asof) {
157 for (auto it = trades_.begin(); it != trades_.end(); /* manual */) {
158 if ((*it)->maturity() < asof) {
159 ALOG(StructuredTradeErrorMessage(*it, "Trade is Matured", ""));
160 tradeLookup_.erase((*it)->id());
161 it = trades_.erase(it);
162 } else {
163 ++it;
164 }
165 }

168void Portfolio::build(const boost::shared_ptr<EngineFactory>& engineFactory, const std::string& context,
169 const bool emitStructuredError) {
170 LOG("Building Portfolio of size " << trades_.size() << " for context = '" << context << "'");
171 auto trade = trades_.begin();
172 Size initialSize = trades_.size();
173 Size failedTrades = 0;
174 while (trade != trades_.end()) {
175 auto [ft, success] = buildTrade(*trade, engineFactory, context, buildFailedTrades(), emitStructuredError);
176 if(success) {
177 ++trade;
178 } else if (ft) {
179 *trade = ft;
180 ++failedTrades;
181 ++trade;
182 } else {
183 tradeLookup_.erase((*trade)->id());
184 trade = trades_.erase(trade);
185 }
186 }
187 LOG("Built Portfolio. Initial size = " << initialSize << ", size now " << trades_.size() << ", built "
188 << failedTrades << " failed trades, context is " + context);
189
190 QL_REQUIRE(trades_.size() > 0, "Portfolio does not contain any built trades, context is '" + context + "'");
191}

193Date Portfolio::maturity() const {
194 QL_REQUIRE(trades_.size() > 0, "Cannot get maturity of an empty portfolio");
195 Date mat = trades_.front()->maturity();
196 for (const auto& t : trades_)
197 mat = std::max(mat, t->maturity());
198 return mat;
199}

215std::vector<std::string> Portfolio::counterparties() const {
216 vector<string> counterparties;
217 for (const auto& t : trades_)
218 counterparties.push_back(t->envelope().counterparty());
219 sort(counterparties.begin(), counterparties.end());
220 counterparties.erase(unique(counterparties.begin(), counterparties.end()), counterparties.end());
221 return counterparties;
222}

316std::pair<boost::shared_ptr<Trade>, bool> buildTrade(boost::shared_ptr<Trade>& trade,
317 const boost::shared_ptr<EngineFactory>& engineFactory,
318 const std::string& context, const bool buildFailedTrades,
319 const bool emitStructuredError) {
320 try {
321 trade->reset();
322 trade->build(engineFactory);
323 TLOG("Required Fixings for trade " << trade->id() << ":");
324 TLOGGERSTREAM(trade->requiredFixings());
325 return std::make_pair(nullptr, true);
326 } catch (std::exception& e) {
327 if(emitStructuredError) {
328 ALOG(StructuredTradeErrorMessage(trade, "Error building trade for context '" + context + "'", e.what()));
329 } else {
330 ALOG("Error building trade '" << trade->id() << "' for context '" + context + "': " + e.what());
331 }
332 if (buildFailedTrades) {
333 boost::shared_ptr<FailedTrade> failed = boost::make_shared<FailedTrade>();
334 failed->id() = trade->id();
335 failed->setUnderlyingTradeType(trade->tradeType());
336 failed->envelope() = trade->envelope();
337 failed->build(engineFactory);
338 failed->resetPricingStats(trade->getNumberOfPricings(), trade->getCumulativePricingTime());
339 LOG("Built failed trade with id " << failed->id());
340 return std::make_pair(failed, false);
341 } else {
342 return std::make_pair(nullptr, false);
343 }
344 }
