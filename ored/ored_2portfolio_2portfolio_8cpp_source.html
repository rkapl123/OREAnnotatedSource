69 void Portfolio::fromXML(XMLNode* node, const boost::shared_ptr<TradeFactory>& factory,
70  const bool checkForDuplicateIds) {
71  XMLUtils::checkNode(node, "Portfolio");
72  vector<XMLNode*> nodes = XMLUtils::getChildrenNodes(node, "Trade");
73  for (Size i = 0; i < nodes.size(); i++) {
74  string tradeType = XMLUtils::getChildValue(nodes[i], "TradeType", true);
75 
76  // Get the id attribute
77  string id = XMLUtils::getAttribute(nodes[i], "id");
78  QL_REQUIRE(id != "", "No id attribute in Trade Node");
79  DLOG("Parsing trade id:" << id);
80  boost::shared_ptr<Trade> trade = factory->build(tradeType);
81 
82  bool failedToLoad = false;
83  if (trade) {
84  try {
85  trade->fromXML(nodes[i]);
86  trade->id() = id;
87  add(trade, checkForDuplicateIds);
88 
89  DLOG("Added Trade " << id << " (" << trade->id() << ")"
90  << " type:" << tradeType);
91  } catch (std::exception& ex) {
92  ALOG(StructuredTradeErrorMessage(id, tradeType, "Error parsing Trade XML", ex.what()));
93  failedToLoad = true;
94  }
95  } else {
96  ALOG(StructuredTradeErrorMessage(id, tradeType, "Error parsing Trade XML"));
97  failedToLoad = true;
98  }
99 
100  // If trade loading failed, then insert a dummy trade with same id and envelope
101  if (failedToLoad && buildFailedTrades_) {
102  try {
103  trade = factory->build("Failed");
104  // this loads only type, id and envelope, but type will be set to the original trade's type
105  trade->fromXML(nodes[i]);
106  // create a dummy trade of type "Dummy"
107  boost::shared_ptr<FailedTrade> failedTrade = boost::make_shared<FailedTrade>();
108  // copy id and envelope
109  failedTrade->id() = id;
110  failedTrade->setUnderlyingTradeType(tradeType);
111  failedTrade->envelope() = trade->envelope();
112  // and add it to the portfolio
113  add(failedTrade, checkForDuplicateIds);
114  WLOG("Added trade id " << failedTrade->id() << " type " << failedTrade->tradeType()
115  << " for original trade type " << trade->tradeType());
116  } catch (std::exception& ex) {
117  ALOG(StructuredTradeErrorMessage(id, tradeType, "Error parsing type and envelope", ex.what()));
118  }
119  }
120  }
121  LOG("Finished Parsing XML doc");
122 }

168 void Portfolio::build(const boost::shared_ptr<EngineFactory>& engineFactory, const std::string& context) {
169  LOG("Building Portfolio of size " << trades_.size() << " for context = '" << context << "'");
170  auto trade = trades_.begin();
171  Size initialSize = trades_.size();
172  Size failedTrades = 0;
173  while (trade != trades_.end()) {
174  auto [ft, success] = buildTrade(*trade, engineFactory, context, buildFailedTrades());
175  if(success) {
176  ++trade;
177  } else if (ft) {
178  *trade = ft;
179  ++failedTrades;
180  ++trade;
181  } else {
182  tradeLookup_.erase((*trade)->id());
183  trade = trades_.erase(trade);
184  }
185  }
186  LOG("Built Portfolio. Initial size = " << initialSize << ", size now " << trades_.size() << ", built "
187  << failedTrades << " failed trades, context is " + context);
188 
189  QL_REQUIRE(trades_.size() > 0, "Portfolio does not contain any built trades, context is '" + context + "'");
190 }

315 std::pair<boost::shared_ptr<Trade>, bool> buildTrade(boost::shared_ptr<Trade>& trade,
316  const boost::shared_ptr<EngineFactory>& engineFactory,
317  const std::string& context, const bool buildFailedTrades) {
318  try {
319  trade->reset();
320  trade->build(engineFactory);
321  TLOG("Required Fixings for trade " << trade->id() << ":");
322  TLOGGERSTREAM(trade->requiredFixings());
323  return std::make_pair(nullptr, true);
324  } catch (std::exception& e) {
325  ALOG(StructuredTradeErrorMessage(trade, "Error building trade for context '" + context + "'", e.what()));
326  if (buildFailedTrades) {
327  boost::shared_ptr<FailedTrade> failed = boost::make_shared<FailedTrade>();
328  failed->id() = trade->id();
329  failed->setUnderlyingTradeType(trade->tradeType());
330  failed->envelope() = trade->envelope();
331  failed->build(engineFactory);
332  failed->resetPricingStats(trade->getNumberOfPricings(), trade->getCumulativePricingTime());
333  LOG("Built failed trade with id " << failed->id());
334  return std::make_pair(failed, false);
335  } else {
336  return std::make_pair(nullptr, false);
337  }
338  }
339 }
