33void FxAverageForward::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
34 LOG("FxAverageForward::build() called");
35 QL_REQUIRE(!settlementCurrency_.empty(), "settlement currency must not be blank");
36 QL_REQUIRE(!referenceCurrency_.empty(), "reference currency must not be blank");
37 QL_REQUIRE(!fxIndex_.empty(), "fx index must not be blank");
38 Currency refCcy = parseCurrency(referenceCurrency_);
39 Currency payCcy = parseCurrency(settlementCurrency_);
40 Date payDate = parseDate(paymentDate_);
41 Schedule observationSchedule = makeSchedule(observationDates_);
42 QL_REQUIRE(payDate >= observationSchedule.dates().back(),
43 "payment date >= last observation date required");
44
45 QL_REQUIRE(!fxIndex_.empty(), "FX settlement index must be specified for averaging forwards");
46 boost::shared_ptr<QuantExt::FxIndex> fxIndex = buildFxIndex(fxIndex_, payCcy.code(), refCcy.code(), engineFactory->market(),
47 engineFactory->configuration(MarketContext::pricing));
48 for (const auto& date : observationSchedule.dates()) {
49 if (date <= Settings::instance().evaluationDate())
50 requiredFixings_.addFixingDate(date, fxIndex_, payDate);
51 }
52
53
54 // Set up Legs
55 inverted_ = parseFxIndex(fxIndex_)->targetCurrency() != payCcy;
56 legs_ = {{boost::make_shared<SimpleCashFlow>(settlementNotional_, payDate)},
57 {boost::make_shared<QuantExt::AverageFXLinkedCashFlow>(payDate, observationSchedule.dates(),
58 referenceNotional_, fxIndex, inverted_)}};
59 legCurrencies_ = {settlementCurrency_, settlementCurrency_};
60 legPayers_ = {fixedPayer_, !fixedPayer_};
61
62 // Set up instrument and engine
63 boost::shared_ptr<QuantLib::Swap> swap(new QuantLib::Swap(legs_, legPayers_));
64 boost::shared_ptr<EngineBuilder> builder = engineFactory->builder("Swap");
65 boost::shared_ptr<SwapEngineBuilderBase> swapBuilder = boost::dynamic_pointer_cast<SwapEngineBuilderBase>(builder);
66 QL_REQUIRE(swapBuilder, "No Builder found for Swap " << id());
67 swap->setPricingEngine(swapBuilder->engine(payCcy));
68 instrument_.reset(new VanillaInstrument(swap));
69
70 npvCurrency_ = settlementCurrency_;
71 notional_ = settlementNotional_;
72 notionalCurrency_ = settlementCurrency_;
73 maturity_ = payDate;
74
75 // ISDA taxonomy
76 additionalData_["isdaAssetClass"] = string("Foreign Exchange");
77 additionalData_["isdaBaseProduct"] = string(settlement_ == "Cash" ? "NDF" : "Forward");
78 additionalData_["isdaSubProduct"] = string("");
79 additionalData_["isdaTransaction"] = string("");
80
81 LOG("FxAverageForward::build() done");
82}

84const std::map<std::string, boost::any>& FxAverageForward::additionalData() const {
85 additionalData_.clear();
86 additionalData_["settlementNotional"] = settlementNotional_;
87 additionalData_["settlementCurrency"] = settlementCurrency_;
88 additionalData_["referenceCurrency"] = referenceCurrency_;
89 additionalData_["referenceNotional"] = referenceNotional_;
90 if (legs_.size() == 2 && !legs_[1].empty()) {
91 auto avg = boost::dynamic_pointer_cast<QuantExt::AverageFXLinkedCashFlow>(legs_[1].front());
92 if (avg) {
93 for (auto [date, value] : avg->fixings())
94 additionalData_["fixing_" + ore::data::to_string(date)] = value;
95 }
96 additionalData_["average rate"] = inverted_ ? 1.0 / avg->fxRate() : avg->fxRate();
97 additionalData_["effective rate"] = avg->fxRate();
98 }
99 return additionalData_;
100}

102void FxAverageForward::fromXML(XMLNode* node) {
103 Trade::fromXML(node);
104 XMLNode* fxNode = XMLUtils::getChildNode(node, "FxAverageForwardData");
105 QL_REQUIRE(fxNode, "No FxAverageForwardData Node");
106 paymentDate_ = XMLUtils::getChildValue(fxNode, "PaymentDate", true);
107 XMLNode* tmp = XMLUtils::getChildNode(fxNode, "ObservationDates");
108 QL_REQUIRE(tmp, "No ObservationDates provided in FxAverageForwardData");
109 observationDates_.fromXML(tmp);
110 fixedPayer_ = XMLUtils::getChildValueAsBool(fxNode, "FixedPayer", true);
111 referenceCurrency_ = XMLUtils::getChildValue(fxNode, "ReferenceCurrency", true);
112 referenceNotional_ = XMLUtils::getChildValueAsDouble(fxNode, "ReferenceNotional", true);
113 settlementCurrency_ = XMLUtils::getChildValue(fxNode, "SettlementCurrency", true);
114 settlementNotional_ = XMLUtils::getChildValueAsDouble(fxNode, "SettlementNotional", true);
115 fxIndex_ = XMLUtils::getChildValue(fxNode, "FXIndex", true);
116 settlement_ = XMLUtils::getChildValue(fxNode, "Settlement", false);
117 if (settlement_ == "")
118 settlement_ = "Cash";
119}

121XMLNode* FxAverageForward::toXML(XMLDocument& doc) {
122 XMLNode* node = Trade::toXML(doc);
123 XMLNode* fxNode = doc.allocNode("FxAverageForwardData");
124 XMLUtils::appendNode(node, fxNode);
125
126 XMLNode* schedNode = observationDates_.toXML(doc);
127 XMLUtils::setNodeName(doc, schedNode, "ObservationDates");
128 XMLUtils::appendNode(fxNode, schedNode);
129
130 XMLUtils::addChild(doc, fxNode, "PaymentDate", paymentDate_);
131 XMLUtils::addChild(doc, fxNode, "FixedPayer", std::to_string(fixedPayer_));
132 XMLUtils::addChild(doc, fxNode, "ReferenceCurrency", referenceCurrency_);
133 XMLUtils::addChild(doc, fxNode, "ReferenceNotional", referenceNotional_);
134 XMLUtils::addChild(doc, fxNode, "SettlementCurrency", settlementCurrency_);
135 XMLUtils::addChild(doc, fxNode, "SettlementNotional", settlementNotional_);
136 XMLUtils::addChild(doc, fxNode, "FXIndex", fxIndex_);
137 XMLUtils::addChild(doc, fxNode, "Settlement", settlement_);
138
139 return node;
140}
