33 void FxAverageForward::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
34  LOG("FxAverageForward::build() called");
35  QL_REQUIRE(!settlementCurrency_.empty(), "settlement currency must not be blank");
36  QL_REQUIRE(!referenceCurrency_.empty(), "reference currency must not be blank");
37  QL_REQUIRE(!fxIndex_.empty(), "fx index must not be blank");
38  Currency refCcy = parseCurrency(referenceCurrency_);
39  Currency payCcy = parseCurrency(settlementCurrency_);
40  Date payDate = parseDate(paymentDate_);
41  Schedule observationSchedule = makeSchedule(observationDates_);
42  QL_REQUIRE(payDate >= observationSchedule.dates().back(),
43  "payment date >= last observation date required");
44 
45  QL_REQUIRE(!fxIndex_.empty(), "FX settlement index must be specified for averaging forwards");
46  boost::shared_ptr<QuantExt::FxIndex> fxIndex = buildFxIndex(fxIndex_, payCcy.code(), refCcy.code(), engineFactory->market(),
47  engineFactory->configuration(MarketContext::pricing));
48  for (const auto& date : observationSchedule.dates()) {
49  if (date <= Settings::instance().evaluationDate())
50  requiredFixings_.addFixingDate(date, fxIndex_, payDate);
51  }
52 
53  // Set up Legs
54  legs_ = {{boost::make_shared<SimpleCashFlow>(settlementNotional_, payDate)},
55  {boost::make_shared<QuantExt::AverageFXLinkedCashFlow>(payDate, observationSchedule.dates(), referenceNotional_, fxIndex)}};
56  legCurrencies_ = {settlementCurrency_, settlementCurrency_};
57  legPayers_ = {fixedPayer_, !fixedPayer_};
58 
59  // Set up instrument and engine
60  boost::shared_ptr<QuantLib::Swap> swap(new QuantLib::Swap(legs_, legPayers_));
61  boost::shared_ptr<EngineBuilder> builder = engineFactory->builder("Swap");
62  boost::shared_ptr<SwapEngineBuilderBase> swapBuilder = boost::dynamic_pointer_cast<SwapEngineBuilderBase>(builder);
63  QL_REQUIRE(swapBuilder, "No Builder found for Swap " << id());
64  swap->setPricingEngine(swapBuilder->engine(payCcy));
65  instrument_.reset(new VanillaInstrument(swap));
66 
67  npvCurrency_ = settlementCurrency_;
68  notional_ = settlementNotional_;
69  notionalCurrency_ = settlementCurrency_;
70  maturity_ = payDate;
71 
72  additionalData_["settlementNotional"] = settlementNotional_;
73  additionalData_["settlementCurrency"] = settlementCurrency_;
74  additionalData_["referenceCurrency"] = referenceCurrency_;
75  additionalData_["referenceNotional"] = referenceNotional_;
76  LOG("FxAverageForward::build() done");
77 }

79 void FxAverageForward::fromXML(XMLNode* node) {
80  Trade::fromXML(node);
81  XMLNode* fxNode = XMLUtils::getChildNode(node, "FxAverageForwardData");
82  QL_REQUIRE(fxNode, "No FxAverageForwardData Node");

84  XMLNode* tmp = XMLUtils::getChildNode(fxNode, "ObservationDates");
