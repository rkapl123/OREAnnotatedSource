47void EquityFutureOption::fromXML(XMLNode* node) {
48 Trade::fromXML(node);
49 XMLNode* eqNode = XMLUtils::getChildNode(node, "EquityFutureOptionData");
50 QL_REQUIRE(eqNode, "No EquityFutureOptionData Node");
51 option_.fromXML(XMLUtils::getChildNode(eqNode, "OptionData"));
52 currency_ = XMLUtils::getChildValue(eqNode, "Currency", true);
53 quantity_ = XMLUtils::getChildValueAsDouble(eqNode, "Quantity", true);
54
55 XMLNode* tmp = XMLUtils::getChildNode(eqNode, "Underlying");
56 if (!tmp)
57 tmp = XMLUtils::getChildNode(eqNode, "Name");
58 UnderlyingBuilder underlyingBuilder("Underlying", "Name");
59 underlyingBuilder.fromXML(tmp);
60 underlying_ = underlyingBuilder.underlying();
61
62 strike_.fromXML(eqNode);
63
64 forwardDate_ = parseDate(XMLUtils::getChildValue(eqNode, "FutureExpiryDate", true));
65}

67XMLNode* EquityFutureOption::toXML(XMLDocument& doc) {
68 XMLNode* node = Trade::toXML(doc);
69 XMLNode* eqNode = doc.allocNode("EquityFutureOptionData");
70 XMLUtils::appendNode(node, eqNode);
71 XMLUtils::appendNode(eqNode, option_.toXML(doc));
72
73 XMLUtils::addChild(doc, eqNode, "Currency", currency_);
74 XMLUtils::addChild(doc, eqNode, "Quantity", quantity_);
75
76 XMLUtils::appendNode(eqNode, underlying_->toXML(doc));
77
78 XMLUtils::appendNode(eqNode, strike_.toXML(doc));
79 XMLUtils::addChild(doc, eqNode, "FutureExpiryDate", to_string(forwardDate_));
80
81 return node;
82}
