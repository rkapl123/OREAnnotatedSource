1061 template <typename T>
1062 vector<T> buildScheduledVector(const vector<T>& values, const vector<string>& dates, const Schedule& schedule,
1063  const bool checkAllValuesAppearInResult) {
1064  if (values.size() < 2 || dates.size() == 0)
1065  return values;
1066 
1067  QL_REQUIRE(values.size() == dates.size(), "Value / Date size mismatch in buildScheduledVector."
1068  << "Value:" << values.size() << ", Dates:" << dates.size());
1069 
1070  // Need to use schedule logic
1071  // Length of data will be 1 less than schedule
1072  //
1073  // Notional 100
1074  // Notional {startDate 2015-01-01} 200
1075  // Notional {startDate 2016-01-01} 300
1076  //
1077  // Given schedule June, Dec from 2014 to 2016 (6 dates, 5 coupons)
1078  // we return 100, 100, 200, 200, 300
1079 
1080  // The first node must not have a date.
1081  // If the second one has a date, all the rest must have, and we process
1082  // If the second one does not have a date, none of them must have one
1083  // and we return the vector unaffected.
1084  QL_REQUIRE(dates[0] == "", "Invalid date " << dates[0] << " for first node");
1085  if (dates[1] == "") {
1086  // They must all be empty and then we return values
1087  for (Size i = 2; i < dates.size(); i++) {
1088  QL_REQUIRE(dates[i] == "", "Invalid date " << dates[i] << " for node " << i
1089  << ". Cannot mix dates and non-dates attributes");
1090  }
1091  return values;
1092  }
1093 
1094  // We have nodes with date attributes now
1095  Size len = schedule.size() - 1;
1096  vector<T> data(len);
1097  Size j = 0, max_j = dates.size() - 1; // j is the index of date/value vector. 0 to max_j
1098  Date d = parseDate(dates[j + 1]); // The first start date
1099  for (Size i = 0; i < len; i++) { // loop over data vector and populate it.
1100  // If j == max_j we just fall through and take the final value
1101  while (schedule[i] >= d && j < max_j) {
1102  j++;
1103  if (j < max_j) {
1104  QL_REQUIRE(dates[j + 1] != "", "Cannot have empty date attribute for node " << j + 1);
1105  d = parseDate(dates[j + 1]);
1106  }
1107  }
1108  data[i] = values[j];
1109  }
1110 
1111  if (checkAllValuesAppearInResult) {
1112  auto it = checkAllValuesAppearInScheduledVector<T>(data, values);
1113  QL_REQUIRE(it == values.end(),
1114  "buildScheduledVector: input value '" << *it << " does not appear in the result vector");
1115  }
1116 
1117  return data;
1118 }
