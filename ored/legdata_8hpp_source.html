1045template <typename T>
1046vector<T> buildScheduledVector(const vector<T>& values, const vector<string>& dates, const Schedule& schedule,
1047 const bool checkAllValuesAppearInResult) {
1048 if (values.size() < 2 || dates.size() == 0)
1049 return values;
1050
1051 QL_REQUIRE(values.size() == dates.size(), "Value / Date size mismatch in buildScheduledVector."
1052 << "Value:" << values.size() << ", Dates:" << dates.size());
1053
1054 // Need to use schedule logic
1055 // Length of data will be 1 less than schedule
1056 //
1057 // Notional 100
1058 // Notional {startDate 2015-01-01} 200
1059 // Notional {startDate 2016-01-01} 300
1060 //
1061 // Given schedule June, Dec from 2014 to 2016 (6 dates, 5 coupons)
1062 // we return 100, 100, 200, 200, 300
1063
1064 // The first node must not have a date.
1065 // If the second one has a date, all the rest must have, and we process
1066 // If the second one does not have a date, none of them must have one
1067 // and we return the vector unaffected.
1068 QL_REQUIRE(dates[0] == "", "Invalid date " << dates[0] << " for first node");
1069 if (dates[1] == "") {
1070 // They must all be empty and then we return values
1071 for (Size i = 2; i < dates.size(); i++) {
1072 QL_REQUIRE(dates[i] == "", "Invalid date " << dates[i] << " for node " << i
1073 << ". Cannot mix dates and non-dates attributes");
1074 }
1075 return values;
1076 }
1077
1078 // We have nodes with date attributes now
1079 Size len = schedule.size() - 1;
1080 vector<T> data(len);
1081 Size j = 0, max_j = dates.size() - 1; // j is the index of date/value vector. 0 to max_j
1082 Date d = parseDate(dates[j + 1]); // The first start date
1083 for (Size i = 0; i < len; i++) { // loop over data vector and populate it.
1084 // If j == max_j we just fall through and take the final value
1085 while (schedule[i] >= d && j < max_j) {
1086 j++;
1087 if (j < max_j) {
1088 QL_REQUIRE(dates[j + 1] != "", "Cannot have empty date attribute for node " << j + 1);
1089 d = parseDate(dates[j + 1]);
1090 }
1091 }
1092 data[i] = values[j];
1093 }
1094
1095 if (checkAllValuesAppearInResult) {
1096 auto it = checkAllValuesAppearInScheduledVector<T>(data, values);
1097 QL_REQUIRE(it == values.end(),
1098 "buildScheduledVector: input value '" << *it << " does not appear in the result vector");
1099 }
1100
1101 return data;
1102}
