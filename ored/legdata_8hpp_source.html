1063template <typename T>
1064vector<T> buildScheduledVector(const vector<T>& values, const vector<string>& dates, const Schedule& schedule,
1065 const bool checkAllValuesAppearInResult) {
1066 if (values.size() < 2 || dates.size() == 0)
1067 return values;
1068
1069 QL_REQUIRE(values.size() == dates.size(), "Value / Date size mismatch in buildScheduledVector."
1070 << "Value:" << values.size() << ", Dates:" << dates.size());
1071
1072 // Need to use schedule logic
1073 // Length of data will be 1 less than schedule
1074 //
1075 // Notional 100
1076 // Notional {startDate 2015-01-01} 200
1077 // Notional {startDate 2016-01-01} 300
1078 //
1079 // Given schedule June, Dec from 2014 to 2016 (6 dates, 5 coupons)
1080 // we return 100, 100, 200, 200, 300
1081
1082 // The first node must not have a date.
1083 // If the second one has a date, all the rest must have, and we process
1084 // If the second one does not have a date, none of them must have one
1085 // and we return the vector unaffected.
1086 QL_REQUIRE(dates[0] == "", "Invalid date " << dates[0] << " for first node");
1087 if (dates[1] == "") {
1088 // They must all be empty and then we return values
1089 for (Size i = 2; i < dates.size(); i++) {
1090 QL_REQUIRE(dates[i] == "", "Invalid date " << dates[i] << " for node " << i
1091 << ". Cannot mix dates and non-dates attributes");
1092 }
1093 return values;
1094 }
1095
1096 // We have nodes with date attributes now
1097 Size len = schedule.size() - 1;
1098 vector<T> data(len);
1099 Size j = 0, max_j = dates.size() - 1; // j is the index of date/value vector. 0 to max_j
1100 Date d = parseDate(dates[j + 1]); // The first start date
1101 for (Size i = 0; i < len; i++) { // loop over data vector and populate it.
1102 // If j == max_j we just fall through and take the final value
1103 while (schedule[i] >= d && j < max_j) {
1104 j++;
1105 if (j < max_j) {
1106 QL_REQUIRE(dates[j + 1] != "", "Cannot have empty date attribute for node " << j + 1);
1107 d = parseDate(dates[j + 1]);
1108 }
1109 }
1110 data[i] = values[j];
1111 }
1112
1113 if (checkAllValuesAppearInResult) {
1114 auto it = checkAllValuesAppearInScheduledVector<T>(data, values);
1115 QL_REQUIRE(it == values.end(),
1116 "buildScheduledVector: input value '" << *it << " does not appear in the result vector");
1117 }
1118
1119 return data;
1120}
