114  data[node] = CallPutDatum(optionType, value);
115  TLOG("Added " << optionType << " option data point with value " << fixed << setprecision(9) <<

142  prices.push_back(forCall ? kv.second.call : kv.second.put);
143 
144  TLOG("Using option datum (" << (forCall ? "Call" : "Put") << "," << io::iso_date(expiries.back()) <<
145  "," << fixed << setprecision(9) << strikes.back() << "," << prices.back() << ")");
146  }
147  }
148 
149  // This is not enough but OptionPriceSurface should throw if data is not sufficient.
150  QL_REQUIRE(!prices.empty(), "Need at least one point for " << (forCall ? "Call" : "Put") <<
151  " commodity option price surface.");
152 
153  return boost::make_shared<OptionPriceSurface>(asof, expiries, strikes, prices, dc);

161 CommodityVolCurve::CommodityVolCurve(const Date& asof, const CommodityVolatilityCurveSpec& spec, const Loader& loader,
162  const CurveConfigurations& curveConfigs,
163  const map<string, boost::shared_ptr<YieldCurve>>& yieldCurves,
164  const map<string, boost::shared_ptr<CommodityCurve>>& commodityCurves,
165  const map<string, boost::shared_ptr<CommodityVolCurve>>& commodityVolCurves,
166  const map<string, boost::shared_ptr<FXVolCurve>>& fxVolCurves,
167  const map<string, boost::shared_ptr<CorrelationCurve>>& correlationCurves,
168  const Market* fxIndices) {
169 
170  try {
171  LOG("CommodityVolCurve: start building commodity volatility structure with ID " << spec.curveConfigID());
172 
173  auto config = *curveConfigs.commodityVolatilityConfig(spec.curveConfigID());
174 
175  boost::shared_ptr<Conventions> conventions = InstrumentConventions::instance().conventions();
176 
177  if (!config.futureConventionsId().empty()) {
178  const auto& cId = config.futureConventionsId();
179  QL_REQUIRE(conventions->has(cId),
180  "Conventions, " << cId << " for config " << config.curveID() << " not found.");
181  convention_ = boost::dynamic_pointer_cast<CommodityFutureConvention>(conventions->get(cId));
182  QL_REQUIRE(convention_, "Convention with ID '" << cId << "' should be of type CommodityFutureConvention");
183  expCalc_ = boost::make_shared<ConventionsBasedFutureExpiry>(*convention_);
184  }
185 
186  calendar_ = parseCalendar(config.calendar());
187  dayCounter_ = parseDayCounter(config.dayCounter());
188 
189  // loop over the volatility configs attempting to build in the order provided
190  DLOG("CommodityVolCurve: Attempting to build equity vol curve from volatilityConfig, "
191  << config.volatilityConfig().size() << " volatility configs provided.");
192  for (auto vc : config.volatilityConfig()) {
193  try {
194  // if the volatility config has it's own calendar, we use that.
195  if (!vc->calendar().empty())
196  calendar_ = vc->calendar();
197  // Do different things depending on the type of volatility configured
198  if (auto eqvc = boost::dynamic_pointer_cast<ProxyVolatilityConfig>(vc)) {
199  buildVolatility(asof, spec, curveConfigs, *eqvc, commodityCurves, commodityVolCurves, fxVolCurves,
200  correlationCurves, fxIndices);
201  } else if (auto qvc = boost::dynamic_pointer_cast<QuoteBasedVolatilityConfig>(vc)) {
202 
203  if (auto cvc = boost::dynamic_pointer_cast<ConstantVolatilityConfig>(vc)) {
204  buildVolatility(asof, config, *cvc, loader);
205  } else if (auto vcc = boost::dynamic_pointer_cast<VolatilityCurveConfig>(vc)) {
206  buildVolatility(asof, config, *vcc, loader);
207  } else if (auto vssc = boost::dynamic_pointer_cast<VolatilityStrikeSurfaceConfig>(vc)) {
208  // Try to populate the price and yield term structure. Need them in some cases here.
209  populateCurves(config, yieldCurves, commodityCurves, true, true);
210  buildVolatility(asof, config, *vssc, loader);
211  } else if (auto vdsc = boost::dynamic_pointer_cast<VolatilityDeltaSurfaceConfig>(vc)) {
212  // Need a yield curve and price curve to create a delta surface.
213  populateCurves(config, yieldCurves, commodityCurves, true);
214  buildVolatility(asof, config, *vdsc, loader);
215  } else if (auto vmsc = boost::dynamic_pointer_cast<VolatilityMoneynessSurfaceConfig>(vc)) {
216  // Need a yield curve (if forward moneyness) and price curve to create a moneyness surface.
217  MoneynessStrike::Type moneynessType = parseMoneynessType(vmsc->moneynessType());
218  bool fwdMoneyness = moneynessType == MoneynessStrike::Type::Forward;
219  populateCurves(config, yieldCurves, commodityCurves, fwdMoneyness);
220  buildVolatility(asof, config, *vmsc, loader);
221  } else if (auto vapo = boost::dynamic_pointer_cast<VolatilityApoFutureSurfaceConfig>(vc)) {
222 
223  // Get the base conventions and create the associated expiry calculator.
224  QL_REQUIRE(!vapo->baseConventionsId().empty(),
225  "The APO FutureConventions must be populated to build a future APO surface");
226  QL_REQUIRE(conventions->has(vapo->baseConventionsId()),
227  "Conventions, " << vapo->baseConventionsId() << " for config " << config.curveID()
228  << " not found.");
229  auto convention = boost::dynamic_pointer_cast<CommodityFutureConvention>(
230  conventions->get(vapo->baseConventionsId()));
231  QL_REQUIRE(convention, "Convention with ID '"
232  << config.futureConventionsId()
233  << "' should be of type CommodityFutureConvention");
234  auto baseExpCalc = boost::make_shared<ConventionsBasedFutureExpiry>(*convention);
235 
236  // Need to get the base commodity volatility structure.
237  QL_REQUIRE(!vapo->baseVolatilityId().empty(),
238  "The APO VolatilityId must be populated to build a future APO surface.");
239  auto itVs = commodityVolCurves.find(vapo->baseVolatilityId());
240  QL_REQUIRE(itVs != commodityVolCurves.end(),
241  "Can't find commodity volatility with id " << vapo->baseVolatilityId());
242  auto baseVs = Handle<BlackVolTermStructure>(itVs->second->volatility());
243 
244  // Need to get the base price curve
245  QL_REQUIRE(!vapo->basePriceCurveId().empty(),
246  "The APO PriceCurveId must be populated to build a future APO surface.");
247  auto itPts = commodityCurves.find(vapo->basePriceCurveId());
248  QL_REQUIRE(itPts != commodityCurves.end(),
249  "Can't find price curve with id " << vapo->basePriceCurveId());
250  auto basePts = Handle<PriceTermStructure>(itPts->second->commodityPriceCurve());
251 
252  // Need a yield curve and price curve to create an APO surface.
253  populateCurves(config, yieldCurves, commodityCurves, true);
254 
255  buildVolatility(asof, config, *vapo, baseVs, basePts);
256 
257  } else {
258  QL_FAIL("Unexpected VolatilityConfig in CommodityVolatilityConfig");
259  }
260  } else {
261  QL_FAIL("CommodityVolCurve: VolatilityConfig must be QuoteBased or a Proxy");
262  }
263 
264  } catch (std::exception& e) {
265  DLOG("CommodityVolCurve: equity vol curve building failed :" << e.what());
266  } catch (...) {
267  DLOG("CommodityVolCurve: equity vol curve building failed: unknown error");
268  }
269  }
270  QL_REQUIRE(volatility_ , "CommodityVolCurve: Failed to build commodity volatility structure from "
271  << config.volatilityConfig().size() << " volatility configs provided.");
272 
273  LOG("CommodityVolCurve: finished building commodity volatility structure with ID " << spec.curveConfigID());
274 
275  } catch (exception& e) {
276  QL_FAIL("Commodity volatility curve building failed : " << e.what());
277  } catch (...) {
278  QL_FAIL("Commodity volatility curve building failed: unknown error");
279  }

310 void CommodityVolCurve::buildVolatility(const QuantLib::Date& asof, const CommodityVolatilityConfig& vc,
311  const VolatilityCurveConfig& vcc, const Loader& loader) {
312 
313  LOG("CommodityVolCurve: start building 1-D volatility curve");
314 
315  // Must have at least one quote
316  QL_REQUIRE(vcc.quotes().size() > 0, "No quotes specified in config " << vc.curveID());
317 
318  QL_REQUIRE(vcc.quoteType() == MarketDatum::QuoteType::RATE_LNVOL, "CommodityVolCurve: only quote type" <<
319  " RATE_LNVOL is currently supported for 1-D commodity volatility curves.");
320 
321  // Check if we are using a regular expression to select the quotes for the curve. If we are, the quotes should
322  // contain exactly one element.
323  auto wildcard = getUniqueWildcard(vcc.quotes());
324 
325  // curveData will be populated with the expiry dates and volatility values.
326  map<Date, Real> curveData;
327 
328  // Different approaches depending on whether we are using a regex or searching for a list of explicit quotes.
329  if (wildcard) {
330 
331  DLOG("Have single quote with pattern " << (*wildcard).pattern());
332 
333  // Loop over quotes and process commodity option quotes matching pattern on asof
334  for (const boost::shared_ptr<MarketDatum>& md : loader.loadQuotes(asof)) {
335 
336  // Go to next quote if the market data point's date does not equal our asof
337  if (md->asofDate() != asof)
338  continue;
339 
340  auto q = boost::dynamic_pointer_cast<CommodityOptionQuote>(md);
341  if (q && (*wildcard).matches(q->name())) {
342 
343  TLOG("The quote " << q->name() << " matched the pattern");
344 
345  Date expiryDate = getExpiry(asof, q->expiry(), vc.futureConventionsId(), vc.optionExpiryRollDays());
346  if (expiryDate > asof) {
347  // Add the quote to the curve data
348  auto it = curveData.find(expiryDate);
349  if (it != curveData.end()) {
350  LOG("Already added quote " << fixed << setprecision(9) << it->second << " for the expiry" <<
351  " date " << io::iso_date(expiryDate) << " for commodity curve " << vc.curveID() <<
352  " so not adding " << q->name());
353  } else {
354  curveData[expiryDate] = q->quote()->value();
355  TLOG("Added quote " << q->name() << ": (" << io::iso_date(expiryDate) << "," <<
356  fixed << setprecision(9) << q->quote()->value() << ")");
357  }
358  }
359  }
360  }
361 
362  // Check that we have quotes in the end
363  QL_REQUIRE(curveData.size() > 0, "No quotes found matching regular expression " << vcc.quotes()[0]);
364 
365  } else {
366 
367  DLOG("Have " << vcc.quotes().size() << " explicit quotes");
368 
369  // Loop over quotes and process commodity option quotes that are explicitly specified in the config
370  Size quotesAdded = 0;
371  Size skippedExpiredQuotes = 0;
372  for (const boost::shared_ptr<MarketDatum>& md : loader.loadQuotes(asof)) {
373 
374  // Go to next quote if the market data point's date does not equal our asof
375  if (md->asofDate() != asof)
376  continue;
377 
378  if (auto q = boost::dynamic_pointer_cast<CommodityOptionQuote>(md)) {
379 
380  // Find quote name in configured quotes.
381  auto it = find(vcc.quotes().begin(), vcc.quotes().end(), q->name());
382 
383  if (it != vcc.quotes().end()) {
384  TLOG("Found the configured quote " << q->name());
385 
386  Date expiryDate = getExpiry(asof, q->expiry(), vc.futureConventionsId(), vc.optionExpiryRollDays());
387  if (expiryDate > asof) {
388  QL_REQUIRE(expiryDate > asof, "Commodity volatility quote '"
389  << q->name() << "' has expiry in the past ("
390  << io::iso_date(expiryDate) << ")");
391  QL_REQUIRE(curveData.count(expiryDate) == 0, "Duplicate quote for the date "
392  << io::iso_date(expiryDate)
393  << " provided by commodity volatility config "
394  << vc.curveID());
395  curveData[expiryDate] = q->quote()->value();
396  quotesAdded++;
397  TLOG("Added quote " << q->name() << ": (" << io::iso_date(expiryDate) << "," << fixed
398  << setprecision(9) << q->quote()->value() << ")");
399  } else {
400  skippedExpiredQuotes++;
401  WLOG("Skipped quote " << q->name() << ": (" << io::iso_date(expiryDate) << "," << fixed
402  << setprecision(9) << q->quote()->value() << ")");
403  }
404  }
405  }
406  }
407 
408  // Check that we have found all of the explicitly configured quotes
409  QL_REQUIRE((quotesAdded + skippedExpiredQuotes) == vcc.quotes().size(),
410  "Found " << quotesAdded << " live quotes and " << skippedExpiredQuotes << " expired quotes"
411  << " but " << vcc.quotes().size() << " quotes were given in config.");
412  }
413 
414  // Create the dates and volatility vector
415  vector<Date> dates;
416  vector<Volatility> volatilities;
417  for (const auto& datum : curveData) {
418  dates.push_back(datum.first);
419  volatilities.push_back(datum.second);
420  TLOG("Added data point (" << io::iso_date(dates.back()) << "," << fixed << setprecision(9)
421  << volatilities.back() << ")");
422  }
423 
424  DLOG("Creating BlackVarianceCurve object.");
425  auto tmp = boost::make_shared<BlackVarianceCurve>(asof, dates, volatilities, dayCounter_);
426 
427  // Set the interpolation.
428  if (vcc.interpolation() == "Linear") {
429  DLOG("Interpolation set to Linear.");
430  } else if (vcc.interpolation() == "Cubic") {
431  DLOG("Setting interpolation to Cubic.");
432  tmp->setInterpolation<Cubic>();
433  } else if (vcc.interpolation() == "LogLinear") {
434  DLOG("Setting interpolation to LogLinear.");
435  tmp->setInterpolation<LogLinear>();
436  } else {
437  DLOG("Interpolation " << vcc.interpolation() << " not recognised so leaving it Linear.");
438  }
439 
440  // Set the volatility_ member after we have possibly updated the interpolation.
441  volatility_ = tmp;
442 
443  // Set the extrapolation
444  if (parseExtrapolation(vcc.extrapolation()) == Extrapolation::Flat) {
445  DLOG("Enabling BlackVarianceCurve flat volatility extrapolation.");
446  volatility_->enableExtrapolation();
447  } else if (parseExtrapolation(vcc.extrapolation()) == Extrapolation::None) {
448  DLOG("Disabling BlackVarianceCurve extrapolation.");
449  volatility_->disableExtrapolation();
450  } else if (parseExtrapolation(vcc.extrapolation()) == Extrapolation::UseInterpolator) {
451  DLOG("BlackVarianceCurve does not support using interpolator for extrapolation "
452  << "so default to flat volatility extrapolation.");
453  volatility_->enableExtrapolation();
454  } else {
455  DLOG("Unexpected extrapolation so default to flat volatility extrapolation.");
456  volatility_->enableExtrapolation();
457  }
458 
459  LOG("CommodityVolCurve: finished building 1-D volatility curve");
460 }

462 void CommodityVolCurve::buildVolatility(const Date& asof, CommodityVolatilityConfig& vc,
463  const VolatilityStrikeSurfaceConfig& vssc, const Loader& loader) {
464 
465  LOG("CommodityVolCurve: start building 2-D volatility absolute strike surface");
466 
467  // We are building a commodity volatility surface here of the form expiry vs strike where the strikes are absolute
468  // numbers. The list of expiries may be explicit or contain a single wildcard character '*'. Similarly, the list
469  // of strikes may be explicit or contain a single wildcard character '*'. So, we have four options here which
470  // ultimately lead to two different types of surface being built:
471  // 1. explicit strikes and explicit expiries => BlackVarianceSurface
472  // 2. wildcard strikes and/or wildcard expiries (3 combinations) => BlackVarianceSurfaceSparse
473 
474  bool expWc = false;
475  if (find(vssc.expiries().begin(), vssc.expiries().end(), "*") != vssc.expiries().end()) {
476  expWc = true;
477  QL_REQUIRE(vssc.expiries().size() == 1, "Wild card expiry specified but more expiries also specified.");
478  DLOG("Have expiry wildcard pattern " << vssc.expiries()[0]);
479  }
480 
481  bool strkWc = false;
482  if (find(vssc.strikes().begin(), vssc.strikes().end(), "*") != vssc.strikes().end()) {
483  strkWc = true;
484  QL_REQUIRE(vssc.strikes().size() == 1, "Wild card strike specified but more strikes also specified.");
485  DLOG("Have strike wildcard pattern " << vssc.strikes()[0]);
486  }
487 
488  // If we do not have a strike wild card, we expect a list of absolute strike values
489  vector<Real> configuredStrikes;
490  if (!strkWc) {
491  // Parse the list of absolute strikes
492  configuredStrikes = parseVectorOfValues<Real>(vssc.strikes(), &parseReal);
493  sort(configuredStrikes.begin(), configuredStrikes.end(), [](Real x, Real y) { return !close(x, y) && x < y; });
494  QL_REQUIRE(adjacent_find(configuredStrikes.begin(), configuredStrikes.end(),
495  [](Real x, Real y) { return close(x, y); }) == configuredStrikes.end(),
496  "The configured strikes contain duplicates");
497  DLOG("Parsed " << configuredStrikes.size() << " unique configured absolute strikes");
498  }
499 
500  // If we do not have an expiry wild card, parse the configured expiries.
501  vector<boost::shared_ptr<Expiry>> configuredExpiries;
502  if (!expWc) {
503  // Parse the list of expiry strings.
504  for (const string& strExpiry : vssc.expiries()) {
505  configuredExpiries.push_back(parseExpiry(strExpiry));
506  }
507  DLOG("Parsed " << configuredExpiries.size() << " unique configured expiries");
508  }
509 
510  // If there are no wildcard strikes or wildcard expiries, delegate to buildVolatilityExplicit.
511  if (!expWc && !strkWc) {
512  buildVolatilityExplicit(asof, vc, vssc, loader, configuredStrikes);
513  return;
514  }
515 
516  DLOG("Expiries and or strikes have been configured via wildcards so building a "
517  << "wildcard based absolute strike surface");
518 
519  // Store grid points along with call and/or put data point
520  CallPutData cpData;
521 
522  // If the price term structure is not empty, populate the map with (expiry, forward) pairs
523  // that will be used below when choosing between a call and a put if have volatilities.
524  map<Date, Real> fwdCurve;
525 
526  // Loop over quotes and process any commodity option quote that matches a wildcard
527  for (const boost::shared_ptr<MarketDatum>& md : loader.loadQuotes(asof)) {
528 
529  // Go to next quote if the market data point's date does not equal our asof.
530  if (md->asofDate() != asof)
531  continue;
532 
533  // Go to next quote if not a commodity option quote.
534  auto q = boost::dynamic_pointer_cast<CommodityOptionQuote>(md);
535  if (!q)
536  continue;
537 
538  // Go to next quote if not a commodity name or currency or quote type does not match config.
539  if (vc.curveID() != q->commodityName() || vc.currency() != q->quoteCurrency() ||
540  vssc.quoteType() != q->quoteType())
541  continue;
542 
543  // This surface is for absolute strikes only.
544  auto strike = boost::dynamic_pointer_cast<AbsoluteStrike>(q->strike());
545  if (!strike)
546  continue;
547 
548  // If we have been given a list of explicit expiries, check that the quote matches one of them.
549  // Move to the next quote if it does not.
550  if (!expWc) {
551  auto expiryIt = find_if(configuredExpiries.begin(), configuredExpiries.end(),
552  [&q](boost::shared_ptr<Expiry> e) { return *e == *q->expiry(); });
553  if (expiryIt == configuredExpiries.end())
554  continue;
555  }
556 
557  // If we have been given a list of explicit strikes, check that the quote matches one of them.
558  // Move to the next quote if it does not.
559  if (!strkWc) {
560  auto strikeIt = find_if(configuredStrikes.begin(), configuredStrikes.end(),
561  [&strike](Real s) { return close(s, strike->strike()); });
562  if (strikeIt == configuredStrikes.end())
563  continue;
564  }
565 
566  // Store in the data container.
567  Date expiry = getExpiry(asof, q->expiry(), vc.futureConventionsId(), vc.optionExpiryRollDays());
568  if (expiry > asof) {
569  ExpiryStrike node(expiry, strike->strike());
570  cpData.addDatum(ExpiryStrike(expiry, strike->strike()), q->optionType(), q->quote()->value());
571 
572  // If we have a price term structure, add forward for expiry if not there already.
573  if (vssc.quoteType() == MarketDatum::QuoteType::RATE_LNVOL && !pts_.empty() &&
574  fwdCurve.count(expiry) == 0) {
575  fwdCurve[expiry] = pts_->price(expiry);
576  }
577 
578  TLOG("Added quote " << q->name() << " to intermediate data.");
579  } else {
580  TLOG("Skipped quote " << q->name() << " to intermediate data, already expired at " << QuantLib::io::iso_date(expiry));
581  }
582  }
583 
584  // Set the strike extrapolation which only matters if extrapolation is turned on for the whole surface.
585  bool flatStrikeExtrap = true;
586  bool flatTimeExtrap = true;
587  if (vssc.extrapolation()) {
588 
589  auto strikeExtrapType = parseExtrapolation(vssc.strikeExtrapolation());
590  if (strikeExtrapType == Extrapolation::UseInterpolator) {
591  DLOG("Strike extrapolation switched to using interpolator.");
592  flatStrikeExtrap = false;
593  } else if (strikeExtrapType == Extrapolation::None) {
594  DLOG("Strike extrapolation cannot be turned off on its own so defaulting to flat.");
595  } else if (strikeExtrapType == Extrapolation::Flat) {
596  DLOG("Strike extrapolation has been set to flat.");
597  } else {
598  DLOG("Strike extrapolation " << strikeExtrapType << " not expected so default to flat.");
599  }
600 
601  auto timeExtrapType = parseExtrapolation(vssc.timeExtrapolation());
602  if (timeExtrapType == Extrapolation::UseInterpolator) {
603  DLOG("Time extrapolation switched to using interpolator.");
604  flatTimeExtrap = false;
605  } else if (timeExtrapType == Extrapolation::None) {
606  DLOG("Time extrapolation cannot be turned off on its own so defaulting to flat.");
607  } else if (timeExtrapType == Extrapolation::Flat) {
608  DLOG("Time extrapolation has been set to flat.");
609  } else {
610  DLOG("Time extrapolation " << timeExtrapType << " not expected so default to flat.");
611  }
612 
613  } else {
614  DLOG("Extrapolation is turned off for the whole surface so the time and"
615  << " strike extrapolation settings are ignored");
616  }
617 
618  // Determine which option type to pick at each expiry and strike.
619  bool preferOutOfTheMoney = !vc.preferOutOfTheMoney() ? true : *vc.preferOutOfTheMoney();
620  TLOG("Prefer out of the money set to: " << ore::data::to_string(preferOutOfTheMoney) << ".");
621 
622  // Build the surface depending on the quote type.
623  if (vssc.quoteType() == MarketDatum::QuoteType::RATE_LNVOL) {
624 
625  DLOG("Creating the BlackVarianceSurfaceSparse object");
626 
627  // Now pick the quotes that we want from the data container. If the fwdCurve was populated above, we use
628  // it in conjunction with preferOutOfTheMoney, to pick the volatilities. If fwdCurve is empty, we just pick
629  // a call volatility if there is one else a put volatility (should have at least one).
630  vector<Real> strikes;
631  vector<Date> expiries;
632  vector<Volatility> vols;
633  Size quotesAdded = 0;
634 
635  for (const auto& kv : cpData.data) {
636 
637  const Date& expiry = kv.first.expiry;
638  const Real& stk = kv.first.strike;
639  const CallPutDatum& cpd = kv.second;
640  QL_REQUIRE(cpd.call != Null<Real>() || cpd.put != Null<Real>(),
641  "CommodityVolCurve: expected the call or put value to be populated.");
642 
643  expiries.push_back(expiry);
644  strikes.push_back(stk);
645  bool useCall = cpd.call != Null<Real>();
646 
647  if ((cpd.call != Null<Real>() && cpd.put != Null<Real>()) && !fwdCurve.empty()) {
648  // We have a call and a put and a forward curve so use it to decide.
649  auto it = fwdCurve.find(expiry);
650  QL_REQUIRE(it != fwdCurve.end(), "CommodityVolCurve: expected forwards for all expiries.");
651  const Real& fwd = it->second;
652  useCall = (preferOutOfTheMoney && stk >= fwd) || (!preferOutOfTheMoney && stk <= fwd);
653  TLOG("(Forward,Strike,UseCall) = (" << fixed << setprecision(6) <<
654  fwd << "," << stk << "," << to_string(useCall) << ").");
655  }
656 
657  vols.push_back(useCall ? cpd.call : cpd.put);
658  quotesAdded++;
659 
660  TLOG("Using option datum (" << (useCall ? "Call" : "Put") << "," << io::iso_date(expiries.back()) <<
661  "," << fixed << setprecision(9) << strikes.back() << "," << vols.back() << ")");
662  }
663 
664  LOG("CommodityVolCurve: added " << quotesAdded << " quotes building wildcard based absolute strike surface.");
665  QL_REQUIRE(quotesAdded > 0, "No quotes loaded for " << vc.curveID());
666 
667  volatility_ = boost::make_shared<BlackVarianceSurfaceSparse>(asof, calendar_, expiries, strikes,
668  vols, dayCounter_, flatStrikeExtrap, flatStrikeExtrap, flatTimeExtrap);
669 
670  } else if (vssc.quoteType() == MarketDatum::QuoteType::PRICE) {
671 
672  QL_REQUIRE(!pts_.empty(), "CommodityVolCurve: require non-empty price term structure" <<
673  " to strip volatilities from prices.");
674  QL_REQUIRE(!yts_.empty(), "CommodityVolCurve: require non-empty yield term structure" <<
675  " to strip volatilities from prices.");
676 
677  // Create the 1D solver options used in the price stripping.
678  Solver1DOptions solverOptions = vc.solverConfig();
679 
680  // Need to create the call and put option price surfaces.
681  auto cSurface = optPriceSurface(cpData, asof, dayCounter_, true);
682  auto pSurface = optPriceSurface(cpData, asof, dayCounter_, false);
683 
684  DLOG("CommodityVolCurve: stripping volatility surface from the option premium surfaces.");
685  auto coss = boost::make_shared<CommodityOptionSurfaceStripper>(pts_, yts_, cSurface, pSurface, calendar_,
686  dayCounter_, vssc.exerciseType(), flatStrikeExtrap, flatStrikeExtrap, flatTimeExtrap,
687  preferOutOfTheMoney, solverOptions);
688  volatility_ = coss->volSurface();
689 
690  // If data level logging, output the stripped volatilities.
691  if (Log::instance().filter(ORE_DATA)) {
692  volatility_->enableExtrapolation(vssc.extrapolation());
693  TLOG("CommodityVolCurve: stripped volatilities:");
694  TLOG("expiry,strike,forward_price,call_price,put_price,discount,volatility");
695  for (const auto& kv : cpData.data) {
696  const Date& expiry = kv.first.expiry;
697  const Real& strike = kv.first.strike;
698  Real fwd = pts_->price(expiry);
699  string cPrice;
700  if (kv.second.call != Null<Real>()) {
701  stringstream ss;
702  ss << fixed << setprecision(6) << kv.second.call;
703  cPrice = ss.str();
704  }
705  string pPrice;
706  if (kv.second.put != Null<Real>()) {
707  stringstream ss;
708  ss << fixed << setprecision(6) << kv.second.put;
709  pPrice = ss.str();
710  }
711  Real discount = yts_->discount(expiry);
712  Real vol = volatility_->blackVol(expiry, strike);
713  TLOG(io::iso_date(expiry) << "," << fixed << setprecision(6) << strike << "," <<
714  fwd << "," << cPrice << "," << pPrice << "," << discount << "," << vol);
715  }
716  }
717 
718  } else {
719  QL_FAIL("CommodityVolCurve: invalid quote type " << vssc.quoteType() << " provided.");
720  }
721 
722  DLOG("Setting BlackVarianceSurfaceSparse extrapolation to " << to_string(vssc.extrapolation()));
723  volatility_->enableExtrapolation(vssc.extrapolation());
724 
725  LOG("CommodityVolCurve: finished building 2-D volatility absolute strike surface");
726 }

728 void CommodityVolCurve::buildVolatilityExplicit(const Date& asof, CommodityVolatilityConfig& vc,
729  const VolatilityStrikeSurfaceConfig& vssc, const Loader& loader,
730  const vector<Real>& configuredStrikes) {
731 
732  LOG("CommodityVolCurve: start building 2-D volatility absolute strike surface with explicit strikes and expiries");
733 
734  QL_REQUIRE(vssc.quoteType() == MarketDatum::QuoteType::RATE_LNVOL, "CommodityVolCurve: only quote type" <<
735  " RATE_LNVOL is currently supported for 2-D volatility strike surface with explicit strikes and expiries.");
736 
737  // Map to hold the rows of the commodity volatility matrix. The keys are the expiry dates and the values are the
738  // vectors of volatilities, one for each configured strike.
739  map<Date, vector<Real>> surfaceData;
740 
741  // Count the number of quotes added. We check at the end that we have added all configured quotes.
742  Size quotesAdded = 0;
743  Size skippedExpiredQuotes = 0;
744  // Loop over quotes and process commodity option quotes that have been requested
745  for (const boost::shared_ptr<MarketDatum>& md : loader.loadQuotes(asof)) {
746 
747  // Go to next quote if the market data point's date does not equal our asof.
748  if (md->asofDate() != asof)
749  continue;
750 
751  // Go to next quote if not a commodity option quote.
752  auto q = boost::dynamic_pointer_cast<CommodityOptionQuote>(md);
753  if (!q)
754  continue;
755 
756  // This surface is for absolute strikes only.
757  auto strike = boost::dynamic_pointer_cast<AbsoluteStrike>(q->strike());
758  if (!strike)
759  continue;
760 
761  // If the quote is not in the configured quotes continue
762  auto it = find(vc.quotes().begin(), vc.quotes().end(), q->name());
763  if (it == vc.quotes().end())
764  continue;
765 
766  // Process the quote
767  Date eDate = getExpiry(asof, q->expiry(), vc.futureConventionsId(), vc.optionExpiryRollDays());
768 
769  if (eDate > asof) {
770 
771  // Position of quote in vector of strikes
772  auto strikeIt = find_if(configuredStrikes.begin(), configuredStrikes.end(),
773  [&strike](Real s) { return close(s, strike->strike()); });
774  Size pos = distance(configuredStrikes.begin(), strikeIt);
775  QL_REQUIRE(pos < configuredStrikes.size(),
776  "The quote '"
777  << q->name()
778  << "' is in the list of configured quotes but does not match any of the configured strikes");
779 
780  // Add quote to surface
781  if (surfaceData.count(eDate) == 0)
782  surfaceData[eDate] = vector<Real>(configuredStrikes.size(), Null<Real>());
783 
784  QL_REQUIRE(surfaceData[eDate][pos] == Null<Real>(),
785  "Quote " << q->name() << " provides a duplicate quote for the date " << io::iso_date(eDate)
786  << " and the strike " << configuredStrikes[pos]);
787  surfaceData[eDate][pos] = q->quote()->value();
788  quotesAdded++;
789 
790  TLOG("Added quote " << q->name() << ": (" << io::iso_date(eDate) << "," << fixed << setprecision(9)
791  << configuredStrikes[pos] << "," << q->quote()->value() << ")");
792  } else {
793  skippedExpiredQuotes++;
794  TLOG("Skipped quote " << q->name() << ": (" << io::iso_date(eDate) << "," << fixed << setprecision(9)
795  << q->quote()->value() << ")");
796  }
797  }
798 
799  LOG("CommodityVolCurve: added " << quotesAdded << " quotes in building explicit absolute strike surface.");
800 
801  QL_REQUIRE(vc.quotes().size() == quotesAdded + skippedExpiredQuotes,
802  "Found " << quotesAdded << " live quotes and " << skippedExpiredQuotes << "expired quotes, but " << vc.quotes().size()
803  << " quotes required by config.");
804 
805  // Set up the BlackVarianceSurface. Note that the rows correspond to strikes and that the columns correspond to
806  // the expiry dates in the matrix that is fed to the BlackVarianceSurface ctor.
807  vector<Date> expiryDates;
808  Matrix volatilities = Matrix(configuredStrikes.size(), surfaceData.size());
809  Size expiryIdx = 0;
810  for (const auto& expiryRow : surfaceData) {
811  expiryDates.push_back(expiryRow.first);
812  for (Size i = 0; i < configuredStrikes.size(); i++) {
813  volatilities[i][expiryIdx] = expiryRow.second[i];
814  }
815  expiryIdx++;
816  }
817 
818  // Trace log the surface
819  TLOG("Explicit strike surface grid points:");
820  TLOG("expiry,strike,volatility");
821  for (Size i = 0; i < volatilities.rows(); i++) {
822  for (Size j = 0; j < volatilities.columns(); j++) {
823  TLOG(io::iso_date(expiryDates[j])
824  << "," << fixed << setprecision(9) << configuredStrikes[i] << "," << volatilities[i][j]);
825  }
826  }
827 
828  // Set the strike extrapolation which only matters if extrapolation is turned on for the whole surface.
829  // BlackVarianceSurface time extrapolation is hard-coded to constant in volatility.
830  BlackVarianceSurface::Extrapolation strikeExtrap = BlackVarianceSurface::ConstantExtrapolation;
831  if (vssc.extrapolation()) {
832 
833  auto strikeExtrapType = parseExtrapolation(vssc.strikeExtrapolation());
834  if (strikeExtrapType == Extrapolation::UseInterpolator) {
835  DLOG("Strike extrapolation switched to using interpolator.");
836  strikeExtrap = BlackVarianceSurface::InterpolatorDefaultExtrapolation;
837  } else if (strikeExtrapType == Extrapolation::None) {
838  DLOG("Strike extrapolation cannot be turned off on its own so defaulting to flat.");
839  } else if (strikeExtrapType == Extrapolation::Flat) {
840  DLOG("Strike extrapolation has been set to flat.");
841  } else {
842  DLOG("Strike extrapolation " << strikeExtrapType << " not expected so default to flat.");
843  }
844 
845  auto timeExtrapType = parseExtrapolation(vssc.timeExtrapolation());
846  if (timeExtrapType != Extrapolation::Flat) {
847  DLOG("BlackVarianceSurface only supports flat volatility extrapolation in the time direction");
848  }
849  } else {
850  DLOG("Extrapolation is turned off for the whole surface so the time and"
851  << " strike extrapolation settings are ignored");
852  }
853 
854  DLOG("Creating BlackVarianceSurface object");
855  auto tmp = boost::make_shared<BlackVarianceSurface>(asof, calendar_, expiryDates, configuredStrikes, volatilities,
856  dayCounter_, strikeExtrap, strikeExtrap);
857 
858  // Set the interpolation if configured properly. The default is Bilinear.
859  if (!(vssc.timeInterpolation() == "Linear" && vssc.strikeInterpolation() == "Linear")) {
860  if (vssc.timeInterpolation() != vssc.strikeInterpolation()) {
861  DLOG("Time and strike interpolation must be the same for BlackVarianceSurface but we got strike "
862  << "interpolation " << vssc.strikeInterpolation() << " and time interpolation "
863  << vssc.timeInterpolation());
864  } else if (vssc.timeInterpolation() == "Cubic") {
865  DLOG("Setting interpolation to BiCubic");
866  tmp->setInterpolation<Bicubic>();
867  } else {
868  DLOG("Interpolation type " << vssc.timeInterpolation() << " not supported in 2 dimensions");
869  }
870  }
871 
872  // Set the volatility_ member after we have possibly updated the interpolation.
873  volatility_ = tmp;
874 
875  DLOG("Setting BlackVarianceSurface extrapolation to " << to_string(vssc.extrapolation()));
876  volatility_->enableExtrapolation(vssc.extrapolation());
877 
878  LOG("CommodityVolCurve: finished building 2-D volatility absolute strike "
879  << "surface with explicit strikes and expiries");
880 }

882 void CommodityVolCurve::buildVolatility(const Date& asof, CommodityVolatilityConfig& vc,
883  const VolatilityDeltaSurfaceConfig& vdsc, const Loader& loader) {
884 
885  using boost::adaptors::transformed;
886  using boost::algorithm::join;
887 
888  LOG("CommodityVolCurve: start building 2-D volatility delta strike surface");
889 
890  QL_REQUIRE(vdsc.quoteType() == MarketDatum::QuoteType::RATE_LNVOL, "CommodityVolCurve: only quote type" <<
891  " RATE_LNVOL is currently supported for a 2-D volatility delta strike surface.");
892 
893  // Parse, sort and check the vector of configured put deltas
894  vector<Real> putDeltas = parseVectorOfValues<Real>(vdsc.putDeltas(), &parseReal);
895  sort(putDeltas.begin(), putDeltas.end(), [](Real x, Real y) { return !close(x, y) && x < y; });
896  QL_REQUIRE(adjacent_find(putDeltas.begin(), putDeltas.end(), [](Real x, Real y) { return close(x, y); }) ==
897  putDeltas.end(),
898  "The configured put deltas contain duplicates");
899  DLOG("Parsed " << putDeltas.size() << " unique configured put deltas");
900  DLOG("Put deltas are: " << join(putDeltas | transformed([](Real d) { return ore::data::to_string(d); }), ","));
901 
902  // Parse, sort descending and check the vector of configured call deltas
903  vector<Real> callDeltas = parseVectorOfValues<Real>(vdsc.callDeltas(), &parseReal);
904  sort(callDeltas.begin(), callDeltas.end(), [](Real x, Real y) { return !close(x, y) && x > y; });
905  QL_REQUIRE(adjacent_find(callDeltas.begin(), callDeltas.end(), [](Real x, Real y) { return close(x, y); }) ==
906  callDeltas.end(),
907  "The configured call deltas contain duplicates");
908  DLOG("Parsed " << callDeltas.size() << " unique configured call deltas");
909  DLOG("Call deltas are: " << join(callDeltas | transformed([](Real d) { return ore::data::to_string(d); }), ","));
910 
911  // Expiries may be configured with a wildcard or given explicitly
912  bool expWc = false;
913  if (find(vdsc.expiries().begin(), vdsc.expiries().end(), "*") != vdsc.expiries().end()) {
914  expWc = true;
915  QL_REQUIRE(vdsc.expiries().size() == 1, "Wild card expiry specified but more expiries also specified.");
916  DLOG("Have expiry wildcard pattern " << vdsc.expiries()[0]);
917  }
918 
919  // Map to hold the rows of the commodity volatility matrix. The keys are the expiry dates and the values are the
920  // vectors of volatilities, one for each configured delta.
921  map<Date, vector<Real>> surfaceData;
922 
923  // Number of strikes = number of put deltas + ATM + number of call deltas
924  Size numStrikes = putDeltas.size() + 1 + callDeltas.size();
925 
926  // Count the number of quotes added. We check at the end that we have added all configured quotes.
927  Size quotesAdded = 0;
928  Size skippedExpiredQuotes = 0;
929  // Configured delta and Atm types.
930  DeltaVolQuote::DeltaType deltaType = parseDeltaType(vdsc.deltaType());
931  DeltaVolQuote::AtmType atmType = parseAtmType(vdsc.atmType());
932  boost::optional<DeltaVolQuote::DeltaType> atmDeltaType;
933  if (!vdsc.atmDeltaType().empty()) {
934  atmDeltaType = parseDeltaType(vdsc.atmDeltaType());
935  }
936 
937  // Populate the configured strikes.
938  vector<boost::shared_ptr<BaseStrike>> strikes;
939  for (const auto& pd : putDeltas) {
940  strikes.push_back(boost::make_shared<DeltaStrike>(deltaType, Option::Put, pd));
941  }
942  strikes.push_back(boost::make_shared<AtmStrike>(atmType, atmDeltaType));
943  for (const auto& cd : callDeltas) {
944  strikes.push_back(boost::make_shared<DeltaStrike>(deltaType, Option::Call, cd));
945  }
946 
947  // Read the quotes to fill the expiry dates and vols matrix.
948  for (const boost::shared_ptr<MarketDatum>& md : loader.loadQuotes(asof)) {
949 
950  // Go to next quote if the market data point's date does not equal our asof.
951  if (md->asofDate() != asof)
952  continue;
953 
954  // Go to next quote if not a commodity option quote.
955  auto q = boost::dynamic_pointer_cast<CommodityOptionQuote>(md);
956  if (!q)
957  continue;
958 
959  // Go to next quote if not a commodity name or currency do not match config.
960  if (vc.curveID() != q->commodityName() || vc.currency() != q->quoteCurrency())
961  continue;
962 
963  // Iterator to one of the configured strikes.
964  vector<boost::shared_ptr<BaseStrike>>::iterator strikeIt;
965 
966  if (!expWc) {
967 
968  // If we have explicitly configured expiries and the quote is not in the configured quotes continue.
969  auto it = find(vc.quotes().begin(), vc.quotes().end(), q->name());
970  if (it == vc.quotes().end())
971  continue;
972 
973  // Check if quote's strike is in the configured strikes.
974  // It should be as we have selected from the explicitly configured quotes in the last step.
975  strikeIt = find_if(strikes.begin(), strikes.end(),
976  [&q](boost::shared_ptr<BaseStrike> s) { return *s == *q->strike(); });
977  QL_REQUIRE(strikeIt != strikes.end(),
978  "The quote '"
979  << q->name()
980  << "' is in the list of configured quotes but does not match any of the configured strikes");
981 
982  } else {
983 
984  // Check if quote's strike is in the configured strikes and continue if it is not.
985  strikeIt = find_if(strikes.begin(), strikes.end(),
986  [&q](boost::shared_ptr<BaseStrike> s) { return *s == *q->strike(); });
987  if (strikeIt == strikes.end())
988  continue;
989  }
990 
991  // Position of quote in vector of strikes
992  Size pos = std::distance(strikes.begin(), strikeIt);
993 
994  // Process the quote
995  Date eDate = getExpiry(asof, q->expiry(), vc.futureConventionsId(), vc.optionExpiryRollDays());
996  if (eDate > asof) {
997 
998  // Add quote to surface
999  if (surfaceData.count(eDate) == 0)
1000  surfaceData[eDate] = vector<Real>(numStrikes, Null<Real>());
1001 
1002  QL_REQUIRE(surfaceData[eDate][pos] == Null<Real>(),
1003  "Quote " << q->name() << " provides a duplicate quote for the date " << io::iso_date(eDate)
1004  << " and strike " << *q->strike());
1005  surfaceData[eDate][pos] = q->quote()->value();
1006  quotesAdded++;
1007 
1008  TLOG("Added quote " << q->name() << ": (" << io::iso_date(eDate) << "," << *q->strike() << "," << fixed
1009  << setprecision(9) << "," << q->quote()->value() << ")");
1010  } else {
1011  skippedExpiredQuotes++;
1012  TLOG("Skiped quote, already expired: " << q->name() << ": (" << io::iso_date(eDate) << "," << *q->strike() << "," << fixed
1013  << setprecision(9) << "," << q->quote()->value() << ")");
1014  }
1015  }
1016 
1017  LOG("CommodityVolCurve: added " << quotesAdded << " quotes in building delta strike surface.");
1018 
1019  // Check the data gathered.
1020  if (!expWc) {

1602  Natural rollDays) const {

1605

1607  result = expiryDate->expiryDate();
1608  } else if (auto expiryPeriod = boost::dynamic_pointer_cast<ExpiryPeriod>(expiry)) {

1615  DLOG("Future option continuation expiry is " << *fcExpiry);
1616 
1617  // Firstly, get the next option expiry on or after the asof date
1618  result = expCalc_->nextExpiry(true, asof, 0, true);
1619  TLOG("CommodityVolCurve::getExpiry: next option expiry relative to " << io::iso_date(asof) << " is "
1620  << io::iso_date(result) << ".");
1621 
1622  // Market quotes may be delivered with a given number of roll days.
1623  if (rollDays > 0) {
1624  Date roll;
1625  roll = calendar_.advance(result, -static_cast<Integer>(rollDays), Days);
1626  TLOG("CommodityVolCurve::getExpiry: roll days is " << rollDays << " giving a roll date "
1627  << io::iso_date(roll) << ".");

1629  if (roll < asof) {
1630  result = expCalc_->nextExpiry(true, asof, 1, true);
1631  roll = calendar_.advance(result, -static_cast<Integer>(rollDays), Days);
1632  QL_REQUIRE(roll > asof, "CommodityVolCurve::getExpiry: expected roll to be greater than asof.");

1635  << io::iso_date(result));
1636  }
1637  }
1638

1640  // days.
1641  TLOG("CommodityVolCurve::getExpiry: first option expiry is " << io::iso_date(result) << ".");

1644  Natural fcIndex = fcExpiry->expiryIndex();
