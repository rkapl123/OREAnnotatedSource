47class FXVolatilityCurveConfig : public CurveConfig {
48public:
49 //! supported volatility structure types
50 /*! For ATM we will only load ATM quotes, for Smile we load ATM, RR, BF or Deltas
51 * SmileInterpolation - currently supports which of the 2 Vanna Volga approximations,
52 * as per Castagna& Mercurio(2006), to use. The second approximation is more accurate
53 * but can ask for the square root of a negative number under unusual circumstances.
54 */
55 enum class Dimension { ATM, SmileVannaVolga, SmileDelta, SmileBFRR, SmileAbsolute, ATMTriangulated };
56 enum class SmileInterpolation {
57 VannaVolga1,
58 VannaVolga2,
59 Linear,
60 Cubic
61 }; // Vanna Volga first/second approximation respectively
