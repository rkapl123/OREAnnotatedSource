47//! Trade base class
48/*! Instrument interface to pricing and risk applications
49 Derived classes should
50 - contain additional serializable data classes
51 - implement a build() function that parses data and constructs QuantLib
52 and QuantExt objects
53 \ingroup portfolio
54*/
55class Trade : public XMLSerializable {
56public:
57
58 //! Default constructor
59 Trade() {}
60
61 //! Base class constructor
62 Trade(const string& tradeType, const Envelope& env = Envelope(), const TradeActions& ta = TradeActions())
63 : tradeType_(tradeType), envelope_(env), tradeActions_(ta) {
64 reset();
65 }
66
67 //! Default destructor
68 virtual ~Trade() {}
69
70 /*! Build QuantLib/QuantExt instrument, link pricing engine. If build() is called multiple times, reset() should
71 be called between these calls. */
72 virtual void build(const boost::shared_ptr<EngineFactory>&) = 0;
73
74 /*! Return the fixings that will be requested in order to price this Trade given the \p settlementDate.
75
76
77
78 If the \p settlementDate is not provided, the current evaluation date is taken as the settlement date.
79 If a Trade does not have any fixings, this method will return an empty map.
80 The map key is the ORE name of the index and the map value is the set of fixing dates.
81
82 \warning This method will return an empty map if the Trade has not been built.
83 */
84 virtual std::map<std::string, std::set<QuantLib::Date>>
85 fixings(const QuantLib::Date& settlementDate = QuantLib::Date()) const {
86 return requiredFixings_.fixingDatesIndices(settlementDate);
87 }
88
89 /*! Return the full required fixing information */
90 const RequiredFixings& requiredFixings() const { return requiredFixings_; }
91
92 virtual std::map<AssetClass, std::set<std::string>>
93 underlyingIndices(const boost::shared_ptr<ReferenceDataManager>& referenceDataManager = nullptr) const {
94 return {};
95 }
96
97 //! \name Serialisation
98 //@{
99 virtual void fromXML(XMLNode* node) override;
100 virtual XMLNode* toXML(XMLDocument& doc) override;
101 //@}
102
103 //! Reset trade, clear all base class data. This does not reset accumulated timings for this trade.
104 void reset();
105
106 //! Reset accumulated timings to given values
107 void resetPricingStats(const std::size_t numberOfPricings = 0,
108 const boost::timer::nanosecond_type cumulativePricingTime = 0) {
109 savedNumberOfPricings_ = numberOfPricings;
110 savedCumulativePricingTime_ = cumulativePricingTime;
111 if (instrument_ != nullptr)
112 instrument_->resetPricingStats();
113 }

164 //@}

167 //@{
168 //! Utility to validate that everything that needs to be set in this base class is actually set
169 void validate() const;
170
171 /*! Utility method indicating if the trade has cashflows for the cashflow report. The default implementation
172 returns \c true so that a trade is automatically considered when cashflows are being written. To prevent a
173 trade from being asked for its cashflows, the method can be overridden to return \c false.
174 */
175 virtual bool hasCashflows() const { return true; }
176 //@}
177
178 //! Get cumulative timing spent on pricing
179 boost::timer::nanosecond_type getCumulativePricingTime() const {
180 return savedCumulativePricingTime_ + (instrument_ != nullptr ? instrument_->getCumulativePricingTime() : 0);
181 }
182
183 //! Get number of pricings
184 std::size_t getNumberOfPricings() const {
185 return savedNumberOfPricings_ + (instrument_ != nullptr ? instrument_->getNumberOfPricings() : 0);
186 }

188protected:

194 string npvCurrency_;

198 string issuer_;

200 std::size_t savedNumberOfPricings_ = 0;
201 boost::timer::nanosecond_type savedCumulativePricingTime_ = 0;

203 // Utility to add premiums such that they are taken into account in pricing and cash flow projection.
204 // For example, an option premium flow is not covered by the underlying option instrument in
205 // QuantLib and needs to be represented separately. This is done by inserting it as an additional instrument
206 // into the InstrumentWrapper. This utility creates the additional instrument. The actual insertion into the
207 // instrument wrapper is done in the individual trade builders when they instantiate the InstrumentWrapper.
208 // The returned date is the latest premium payment date added.
209 Date addPremiums(std::vector<boost::shared_ptr<Instrument>>& instruments, std::vector<Real>& multipliers,
210 const Real tradeMultiplier, const PremiumData& premiumData, const Real premiumMultiplier,
211 const Currency& tradeCurrency, const boost::shared_ptr<EngineFactory>& factory,
212 const string& configuration);
213
214 RequiredFixings requiredFixings_;
215 mutable std::map<std::string,boost::any> additionalData_;
216
217 /* sets additional data based on given internal legNo (0, 1, ...), the result leg id is derived from this
218 as "legNo + 1", i.e. starting with 1 (1, 2, ...). The result leg id can be overwriten using the second
219 parameter resultLegId. */
220 void setLegBasedAdditionalData(const Size legNo, Size resultLegId = Null<Size>()) const;
221
222private:
223 string id_;
224 Envelope envelope_;
225 TradeActions tradeActions_;
226};

228template <class T>
229inline T Trade::additionalDatum(const std::string& tag) const {
230 std::map<std::string,boost::any>::const_iterator value =
231 additionalData_.find(tag);
232 QL_REQUIRE(value != additionalData_.end(),
233 tag << " not provided");
234 return boost::any_cast<T>(value->second);
235}
