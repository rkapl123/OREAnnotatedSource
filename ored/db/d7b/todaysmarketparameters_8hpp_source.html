113 private:
114  /* Maps a configuration label to a MarketConfiguration instance. For the command line application, the configuration
115  label is specified under the Markets node in ore.xml. An example is given by:
116 
117  default => (DiscountCurve => xois_eur)
118  collateral_inccy => (DiscountCurve => ois)
119  collateral_eur => (DiscountCurve => xois_eur)
120  ...
121 
122  The RHS maps each market object to a configuration id, which is an intermediate identifier used to group
123  assignments for each market object together, see the description of marketObjects_ below. Missing market objects
124  are mapped to the market default configuration "default". The latter is defined as a constant in
125  Market::defaultConfiguration.
126 
127  A configuration label "default" is always added, which maps all market objects to the market default
128  configuration "default". The entries for the configuration label "default" can be overwritten though. */
129  map<string, MarketConfiguration> configurations_;
130 
131  /* For each market object type, maps the intermediate configuration id to a list of assignments, e.g.
132 
133  DiscountCurve => (xois_eur => (EUR => Yield/EUR/EUR1D,
134  USD => Yield/USD/USD-IN_EUR)
135  ois => (EUR => Yield/EUR/EUR1D,
136  USD => Yield/USD/USD1D))
137  IndexCurve => ...
138 
139  For each pair (market object, intermediate configuration id) defined in the rhs of the configurations_ map, a
140  mapping should be defined in marketObjects_). */
141  map<MarketObject, map<string, map<string, string>>> marketObjects_;
142 
143  void curveSpecs(const map<string, map<string, string>>&, const string&, vector<string>&) const;

167 inline const map<string, string>& TodaysMarketParameters::mapping(const MarketObject o,
168  const string& configuration) const {
169  static map<string, string> empty;
170  QL_REQUIRE(hasConfiguration(configuration), "configuration " << configuration << " not found");
171  auto it = marketObjects_.find(o);
172  if (it != marketObjects_.end()) {
173  auto it2 = it->second.find(marketObjectId(o, configuration));
174  if (it2 != it->second.end()) {
175  return it2->second;
176  }
177  }
178  return empty;
179 }
