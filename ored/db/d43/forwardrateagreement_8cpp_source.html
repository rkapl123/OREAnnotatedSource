31 void ForwardRateAgreement::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
32  const boost::shared_ptr<Market> market = engineFactory->market();
33 
34  Date startDate = parseDate(startDate_);
35  Date endDate = parseDate(endDate_);
36  Position::Type positionType = parsePositionType(longShort_);
37 
38  Handle<YieldTermStructure> discountTS = market->discountCurve(currency_);
39  Handle<QuantLib::IborIndex> index = market->iborIndex(index_);
40 
41  boost::shared_ptr<QuantLib::ForwardRateAgreement> fra(
42  new QuantLib::ForwardRateAgreement(startDate, endDate, positionType, strike_, amount_, *index, discountTS));
43  instrument_.reset(new VanillaInstrument(fra));
44  npvCurrency_ = currency_;
45  maturity_ = endDate;
46  instrument_->qlInstrument()->update();
47  notional_ = amount_;
48  notionalCurrency_ = currency_;
49  // the QL instrument reads the fixing in setupExpired() (bug?), so we don't add a payment date here to be safe
50  requiredFixings_.addFixingDate(fra->fixingDate(), index_);
51  // add required fixings for an Ibor fallback index
52  if (auto fallback = boost::dynamic_pointer_cast<QuantExt::FallbackIborIndex>(*index)) {
53  requiredFixings_.addFixingDates(fallback->onCoupon(fra->fixingDate())->fixingDates(),
54  engineFactory->iborFallbackConfig().fallbackData(index_).rfrIndex);
55  }
56 }
