78 const vector<string>& InflationCapFloorVolatilityCurveConfig::quotes() {
79  if (quotes_.size() == 0) {
80 
81  string type;
82  if (type_ == Type::ZC)
83  type = "ZC";
84  else if (type_ == Type::YY)
85  type = "YY";
86 
87  // Determine the index string to use for the quotes.
88  string index = quoteIndex_.empty() ? index_ : quoteIndex_;
89 
90  std::stringstream ssBase;
91  if (quoteType_ == QuoteType::Price)
92  ssBase << type << "_INFLATIONCAPFLOOR/PRICE/" << index << "/";
93  else
94  ssBase << type << "_INFLATIONCAPFLOOR/" << volatilityType_ << "/" << index << "/";
95  string base = ssBase.str();
96 
97  // TODO: how to tell if atmFlag or relative flag should be true
98  for (auto t : tenors_) {
99  if (quoteType_ == QuoteType::Price) {
100  for (auto s : capStrikes_) {
101  quotes_.push_back(base + t + "/C/" + s);
102  }
103  for (auto s : floorStrikes_) {
104  quotes_.push_back(base + t + "/F/" + s);
105  }
106  } else {
107  for (auto s : strikes_) {
108  quotes_.push_back(base + t + "/F/" + s);
109  }
110  }
111  }
112 
113  if (volatilityType_ == VolatilityType::ShiftedLognormal) {
114  for (auto t : tenors_) {
115  std::stringstream ss;
116  quotes_.push_back(type + "_INFLATIONCAPFLOOR/SHIFT/" + index + "/" + t);
117  }
118  }
119  }
120  return quotes_;
121 }

123 void InflationCapFloorVolatilityCurveConfig::fromXML(XMLNode* node) {
124  XMLUtils::checkNode(node, "InflationCapFloorVolatility");
125 
126  curveID_ = XMLUtils::getChildValue(node, "CurveId", true);
127  curveDescription_ = XMLUtils::getChildValue(node, "CurveDescription", true);
128 
129  string type = XMLUtils::getChildValue(node, "Type", true);
130  if (type == "ZC") {
131  type_ = Type::ZC;
132  } else if (type == "YY") {
133  type_ = Type::YY;
134  } else
135  QL_FAIL("Type " << type << " not recognized");
136 
137  // Get the quote type
138  string quoteType = XMLUtils::getChildValue(node, "QuoteType", true);
139  if (quoteType == "Price") {
140  quoteType_ = QuoteType::Price;
141  } else if (quoteType == "Volatility") {
142  quoteType_ = QuoteType::Volatility;
143  } else {
144  QL_FAIL("Quote type, " << quoteType << ", not recognized");
145  }
146 
147  // Get the volatility type
148  string volType = XMLUtils::getChildValue(node, "VolatilityType", true);
149  if (volType == "Normal") {
150  volatilityType_ = VolatilityType::Normal;
151  } else if (volType == "Lognormal") {
152  volatilityType_ = VolatilityType::Lognormal;
153  } else if (volType == "ShiftedLognormal") {
154  volatilityType_ = VolatilityType::ShiftedLognormal;
155  } else {
156  QL_FAIL("Volatility type, " << volType << ", not recognized");
157  }
158  extrapolate_ = XMLUtils::getChildValueAsBool(node, "Extrapolation", true);
159  tenors_ = XMLUtils::getChildrenValuesAsStrings(node, "Tenors", true);
160 
161  // We are requiring explicit strikes so there should be at least one strike
162  if (quoteType_ == QuoteType::Price) {
163  capStrikes_ = XMLUtils::getChildrenValuesAsStrings(node, "CapStrikes", true);
164  floorStrikes_ = XMLUtils::getChildrenValuesAsStrings(node, "FloorStrikes", true);
165  QL_REQUIRE(!capStrikes_.empty() || !floorStrikes_.empty(),
166  "CapStrikes or FloorStrikes node should not be empty");
167  // Set strikes to the sorted union of cap and floor strikes
168  std::set<Real> strikeSet;
169  for (Size i = 0; i < capStrikes_.size(); ++i)
170  strikeSet.insert(parseReal(capStrikes_[i]));
171  for (Size i = 0; i < floorStrikes_.size(); ++i)
172  strikeSet.insert(parseReal(floorStrikes_[i]));
173  strikes_.clear();
174  for (auto s : strikeSet)
175  strikes_.push_back(to_string(s));
176  for (Size i = 0; i < strikes_.size(); ++i)
177  DLOG("ZC Inflation Cap/Floor Strike " << i << " = " << strikes_[i]);
178  } else {
179  strikes_ = XMLUtils::getChildrenValuesAsStrings(node, "Strikes", true);
180  QL_REQUIRE(!strikes_.empty(), "Strikes node should not be empty");
181  }
182  settleDays_ = 0; // optional
183  if (XMLNode* n = XMLUtils::getChildNode(node, "SettlementDays")) {
184  Integer d = parseInteger(XMLUtils::getNodeValue(n));
185  QL_REQUIRE(d >= 0, "SettlementDays (" << d << ") must be non-negative");
186  settleDays_ = static_cast<Natural>(d);
187  }
188  calendar_ = parseCalendar(XMLUtils::getChildValue(node, "Calendar", true));
189  dayCounter_ = parseDayCounter(XMLUtils::getChildValue(node, "DayCounter", true));
190  businessDayConvention_ = parseBusinessDayConvention(XMLUtils::getChildValue(node, "BusinessDayConvention", true));
191  index_ = XMLUtils::getChildValue(node, "Index", true);
192  indexCurve_ = XMLUtils::getChildValue(node, "IndexCurve", true);
193  yieldTermStructure_ = XMLUtils::getChildValue(node, "YieldTermStructure", true);
194  observationLag_ = parsePeriod(XMLUtils::getChildValue(node, "ObservationLag", true));
195  quoteIndex_ = XMLUtils::getChildValue(node, "QuoteIndex", false);
196  populateRequiredCurveIds();
197 }
