31 void load(InMemoryLoader& loader, const vector<string>& data, bool isMarket, bool implyTodaysFixings) {
32  LOG("MemoryLoader started");
33 
34  Date today = QuantLib::Settings::instance().evaluationDate();
35 
36  for (Size i = 0; i < data.size(); ++i) {
37  string line = data[i];
38  // skip blank and comment lines
39  if (line.size() > 0 && line[0] != '#') {
40  vector<string> tokens;
41  boost::trim(line);
42  boost::split(tokens, line, boost::is_any_of(",;\t "), boost::token_compress_on);
43 
44  // TODO: should we try, catch and log any invalid lines?
45  QL_REQUIRE(tokens.size() == 3, "Invalid MemoryLoader line, 3 tokens expected " << line);
46  Date date = parseDate(tokens[0]);
47  const string& key = tokens[1];
48  Real value = parseReal(tokens[2]);
49 
50  if (isMarket) {
51  // process market
52  // build market datum and add to map
53  try {
54  loader.add(date, key, value);
55  TLOG("Added MarketDatum " << key);
56  } catch (std::exception& e) {
57  WLOG("Failed to parse MarketDatum " << key << ": " << e.what());
58  }
59  } else {
60  // process fixings
61  if (date < today || (date == today && !implyTodaysFixings))
62  loader.addFixing(date, key, value);
63  }
64  }
65  }
66  LOG("MemoryLoader completed");
67 }
