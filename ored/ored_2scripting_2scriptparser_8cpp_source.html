31std::ostream& operator<<(std::ostream& out, const ParserError& error) {
32 if (!error.stoppedParsingAt.initialised) {
33 out << "parsing succeeded";
34 } else {
35 out << "parsing stopped at " << to_string(error.stoppedParsingAt) << "\n";
36 if (error.expectedWhere.initialised) {
37 out << "expected " << error.expectedWhat << " in " << to_string(error.expectedWhere) << ":\n";
38 out << error.scriptCurrentLine << "\n";
39 out << std::string(std::max<Size>(error.expectedWhere.columnStart, 1) - 1, ' ') << "^--- here\n";
40 } else {
41 out << "remaining input is\n<<<<<<<<<<\n" << error.remainingInput << "\n>>>>>>>>>>\n";
42 }
43 }
44 return out;
45}

47ScriptParser::ScriptParser(const std::string& script) {
48 ScriptGrammarIterator first(script.begin()), iter = first, last(script.end());
49 ScriptGrammar grammar(first);
50 success_ = qi::phrase_parse(iter, last, grammar, boost::spirit::qi::space);
51 if (!success_ || iter != last) {
52 success_ = false;
53 parserError_.stoppedParsingAt =
54 LocationInfo(boost::spirit::get_line(iter), boost::spirit::get_column(first, iter),
55 boost::spirit::get_line(iter), boost::spirit::get_column(first, iter));
56 parserError_.remainingInput = std::string(iter, last);
57 if (grammar.hasError) {
58 parserError_.expectedWhat = ore::data::to_string(grammar.errorWhat);
59 parserError_.expectedWhere = LocationInfo(
60 boost::spirit::get_line(grammar.errorPos), boost::spirit::get_column(first, grammar.errorPos),
61 boost::spirit::get_line(grammar.errorPos), boost::spirit::get_column(first, grammar.errorPos));
62 parserError_.scriptContext = std::string(grammar.errorBegin, grammar.errorEnd);
63 parserError_.errorPos = std::distance(grammar.errorBegin, grammar.errorPos);
64 auto currentLine = boost::spirit::get_current_line(grammar.errorBegin, grammar.errorPos, grammar.errorEnd);
65 parserError_.scriptCurrentLine = std::string(currentLine.begin(), currentLine.end());
66 }
67 } else {
68 QL_REQUIRE(grammar.evalStack.size() == 1,
69 "ScriptParser: unexpected eval stack size (" << grammar.evalStack.size() << "), should be 1");
70 ast_ = grammar.evalStack.top();
71 QL_REQUIRE(ast_, "ScriptParser: ast is null");
72 }
73}

75std::string printCodeContext(std::string script, const ASTNode* loc, bool compact) {
76
77 if (!loc)
78 return "(script reference is not available)\n";
79
80 // assume we have either newline or return, but not return alone (old mac csv format?)
81 boost::replace_all(script, "\r", "");
82
83 auto l = loc->locationInfo;
84 if (l.lineEnd < l.lineStart || (l.lineEnd == l.lineStart && l.columnEnd <= l.columnStart))
85 return "(script reference invalid: " + to_string(l) + ")\n";
86
87 std::string res;
88 // if compact is true, we remove leading/trailing symbols used for DLOGGERSTREAM for cleaner error message in log
89 if (compact == false)
90 res += "<<<<<<<<<<\n";
91
92 Size pos = 0, currentLine = 1;
93 for (Size line = l.lineStart; line <= l.lineEnd; ++line) {
94 // look for the current line
95 while (pos < script.size() && currentLine < line) {
96 if (script[pos] == '\n')
97 ++currentLine;
98 ++pos;
99 }
100 if (pos == script.size())
101 break;
102 // look for the line end position in the current line
103 Size posLineEnd = pos;
104 while (posLineEnd < script.size() && script[posLineEnd] != '\n')
105 ++posLineEnd;
106 // add current line to the result
107 std::string curr = std::string(std::next(script.begin(), pos), std::next(script.begin(), posLineEnd));
108 if (compact) {
109 boost::algorithm::trim(curr);
110 res += curr + ' ';
111 } else {
112 res += curr + '\n';
113 }
114 // underline the relevant part of the current line
115 Size columnStart = 1, columnEnd = posLineEnd - pos + 1;
116 if (line == l.lineStart)
117 columnStart = l.columnStart;
118 if (line == l.lineEnd)
119 columnEnd = l.columnEnd;
120 if (columnEnd < columnStart)
121 return "(script reference internal error: columnEnd (" + std::to_string(columnEnd) +
122 ") should be >= columnStart (" + std::to_string(columnStart) + "))";
123 // adjust column start so that trailing spaces are not underlined
124 while (columnStart < columnEnd && pos + columnStart < script.size() &&
125 script[pos + std::max<Size>(columnStart, 1) - 1] == ' ')
126 ++columnStart;
127 if (compact == false)
128 res +=
129 std::string(std::max<Size>(columnStart, 1) - 1, ' ') + std::string(columnEnd - columnStart, '=') + '\n';
130 }
131
132 if (compact == false)
133 res += ">>>>>>>>>>\n";
134 return res;
135}
