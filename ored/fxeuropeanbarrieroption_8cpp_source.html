38 void FxEuropeanBarrierOption::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
39 
40  const boost::shared_ptr<Market> market = engineFactory->market();
41 
42  // Only European Single Barrier supported for now
43  QL_REQUIRE(option_.style() == "European", "Option Style unknown: " << option_.style());
44  QL_REQUIRE(option_.exerciseDates().size() == 1, "Invalid number of excercise dates");
45  QL_REQUIRE(barrier_.levels().size() == 1, "Invalid number of barrier levels");
46  QL_REQUIRE(barrier_.style().empty() || barrier_.style() == "European", "Only european barrier style suppported");
47  QL_REQUIRE(tradeActions().empty(), "TradeActions not supported for FxEuropeanBarrierOption");
48 
49  Currency boughtCcy = parseCurrency(boughtCurrency_);
50  Currency soldCcy = parseCurrency(soldCurrency_);
51  Real level = barrier_.levels()[0].value();
52  Real rebate = barrier_.rebate();
53  QL_REQUIRE(rebate >= 0, "Rebate must be non-negative");
54 
55  // Replicate the payoff of European Barrier Option (with strike K and barrier B) using combinations of options
56 
57  // Call
58  // Up
59  // In
60  // Long Up&Out Digital Option with barrier B payoff rebate
61  // B > K
62  // Long European Call Option with strike B
63  // Long Up&In Digital Option with barrier B payoff B - K
64  // B <= K
65  // Long European Call Option with strike K
66  // Out
67  // Long Up&In Digital Option with barrier B payoff rebate
68  // B > K
69  // Long European Call Option with strike K
70  // Short European Call Option with strike B
71  // Short Up&In Digital Option with barrier B payoff B - K
72  // B <= K
73  // 0
74  // Down
75  // In
76  // Long Down&Out Digital Option with barrier B payoff rebate
77  // B > K
78  // Long European Call Option with strike K
79  // B <= K
80  // 0
81  // Out
82  // Long Down&In Digital Option with barrier B payoff rebate
83  // B > K
84  // Long European Call Option with strike B
85  // Long Down&Out Digital Option with barrier B payoff B - K
86  // B <= K
87  // Long European Call Option with strike K
88 
89  // Put
90  // Up
91  // In
92  // Long Up&Out Digital Option with barrier B payoff rebate
93  // B > K
94  // 0
95  // B <= K
96  // Long European Put Option with strike K
97  // Short European Put Option with strike B
98  // Short Up&Out Digital Option with barrier B payoff K - B
99  // Out
100  // Long Up&In Digital Option with barrier B payoff rebate
101  // B > K
102  // Long European Put Option with strike K
103  // B <= K
104  // Long European Put Option with strike B
105  // Long Up&Out Digital Option with barrier B payoff K - B
106  // Down
107  // In
108  // Long Down&Out Digital Option with barrier B payoff rebate
109  // B > K
110  // Long European Put Option with strike K
111  // B <= K
112  // Long European Put Option with strike B
113  // Long Down&In Digital Option with barrier B payoff K - B
114  // Out
115  // Long Down&In Digital Option with barrier B payoff rebate
116  // B > K
117  // 0
118  // B <= K
119  // Long European Put Option with strike K
120 
121  Real strike = soldAmount_ / boughtAmount_;
122  Option::Type type = parseOptionType(option_.callPut());
123 
124  // Exercise
125  Date expiryDate = parseDate(option_.exerciseDates().front());
126  boost::shared_ptr<Exercise> exercise = boost::make_shared<EuropeanExercise>(expiryDate);
127 
128  Barrier::Type barrierType = parseBarrierType(barrier_.type());
129 
130  // Payoff - European Option with strike K
131  boost::shared_ptr<StrikedTypePayoff> payoffVanillaK(new PlainVanillaPayoff(type, strike));
132  // Payoff - European Option with strike B
133  boost::shared_ptr<StrikedTypePayoff> payoffVanillaB(new PlainVanillaPayoff(type, level));
134  // Payoff - Digital Option with barrier B payoff abs(B - K)
135  boost::shared_ptr<StrikedTypePayoff> payoffDigital(new CashOrNothingPayoff(type, level, fabs(level - strike)));
136 
137  boost::shared_ptr<Instrument> digital = boost::make_shared<VanillaOption>(payoffDigital, exercise);
138  boost::shared_ptr<Instrument> vanillaK = boost::make_shared<VanillaOption>(payoffVanillaK, exercise);
139  boost::shared_ptr<Instrument> vanillaB = boost::make_shared<VanillaOption>(payoffVanillaB, exercise);
140 
141  boost::shared_ptr<StrikedTypePayoff> rebatePayoff;
142  if (barrierType == Barrier::Type::UpIn || barrierType == Barrier::Type::DownOut) {
143  // Payoff - Up&Out / Down&In Digital Option with barrier B payoff rebate
144  rebatePayoff = boost::make_shared<CashOrNothingPayoff>(Option::Put, level, rebate);
145  } else if (barrierType == Barrier::Type::UpOut || barrierType == Barrier::Type::DownIn) {
146  // Payoff - Up&In / Down&Out Digital Option with barrier B payoff rebate
147  rebatePayoff = boost::make_shared<CashOrNothingPayoff>(Option::Call, level, rebate);
148  }
149  boost::shared_ptr<Instrument> rebateInstrument = boost::make_shared<VanillaOption>(rebatePayoff, exercise);
150 
151  // set pricing engines
152  boost::shared_ptr<EngineBuilder> builder = engineFactory->builder("FxOption");
153  QL_REQUIRE(builder, "No builder found for FxOption");
154  boost::shared_ptr<FxEuropeanOptionEngineBuilder> fxOptBuilder =
155  boost::dynamic_pointer_cast<FxEuropeanOptionEngineBuilder>(builder);
156 
157  builder = engineFactory->builder("FxDigitalOption");
158  QL_REQUIRE(builder, "No builder found for FxDigitalOption");
159  boost::shared_ptr<FxDigitalOptionEngineBuilder> fxDigitalOptBuilder =
160  boost::dynamic_pointer_cast<FxDigitalOptionEngineBuilder>(builder);
161 
162  digital->setPricingEngine(fxDigitalOptBuilder->engine(boughtCcy, soldCcy));
163  vanillaK->setPricingEngine(fxOptBuilder->engine(boughtCcy, soldCcy, expiryDate));
164  vanillaB->setPricingEngine(fxOptBuilder->engine(boughtCcy, soldCcy, expiryDate));
165  rebateInstrument->setPricingEngine(fxDigitalOptBuilder->engine(boughtCcy, soldCcy));
166 
167  boost::shared_ptr<CompositeInstrument> qlInstrument = boost::make_shared<CompositeInstrument>();
168  qlInstrument->add(rebateInstrument);
169  if (type == Option::Call) {
170  if (barrierType == Barrier::Type::UpIn || barrierType == Barrier::Type::DownOut) {
171  if (level > strike) {
172  qlInstrument->add(vanillaB);
173  qlInstrument->add(digital);
174  } else {
175  qlInstrument->add(vanillaK);
176  }
177  } else if (barrierType == Barrier::Type::UpOut || barrierType == Barrier::Type::DownIn) {
178  if (level > strike) {
179  qlInstrument->add(vanillaK);
180  qlInstrument->add(vanillaB, -1);
181  qlInstrument->add(digital, -1);
182  } else {
183  // empty
184  }
185  } else {
186  QL_FAIL("Unknown Barrier Type: " << barrierType);
187  }
188  } else if (type == Option::Put) {
189  if (barrierType == Barrier::Type::UpIn || barrierType == Barrier::Type::DownOut) {
190  if (level > strike) {
191  // empty
192  } else {
193  qlInstrument->add(vanillaK);
194  qlInstrument->add(vanillaB, -1);
195  qlInstrument->add(digital, -1);
196  }
197  } else if (barrierType == Barrier::Type::UpOut || barrierType == Barrier::Type::DownIn) {
198  if (level > strike) {
199  qlInstrument->add(vanillaK);
200  } else {
201  qlInstrument->add(vanillaB);
202  qlInstrument->add(digital);
203  }
204  } else {
205  QL_FAIL("Unknown Barrier Type: " << barrierType);
206  }
207  }
208 
209  // Add additional premium payments
210  Position::Type positionType = parsePositionType(option_.longShort());
211  Real bsInd = (positionType == QuantLib::Position::Long ? 1.0 : -1.0);
212  Real mult = boughtAmount_ * bsInd;
213 
214  std::vector<boost::shared_ptr<Instrument>> additionalInstruments;
215  std::vector<Real> additionalMultipliers;
216  Date lastPremiumDate =
217  addPremiums(additionalInstruments, additionalMultipliers, mult, option_.premiumData(), -bsInd, soldCcy,
218  engineFactory, fxOptBuilder->configuration(MarketContext::pricing));
219 
220  instrument_ = boost::shared_ptr<InstrumentWrapper>(
221  new VanillaInstrument(qlInstrument, mult, additionalInstruments, additionalMultipliers));
222 
223  npvCurrency_ = soldCurrency_; // sold is the domestic
224  notional_ = soldAmount_;
225  notionalCurrency_ = soldCurrency_;
226  maturity_ = std::max(lastPremiumDate, expiryDate);
227 
228  additionalData_["boughtCurrency"] = boughtCurrency_;
229  additionalData_["boughtAmount"] = boughtAmount_;
230  additionalData_["soldCurrency"] = soldCurrency_;
231  additionalData_["soldAmount"] = soldAmount_;
232 }

234 void FxEuropeanBarrierOption::fromXML(XMLNode* node) {
235  Trade::fromXML(node);
236  XMLNode* fxNode = XMLUtils::getChildNode(node, "FxEuropeanBarrierOptionData");
237  QL_REQUIRE(fxNode, "No FxEuropeanBarrierOptionData Node");
238  option_.fromXML(XMLUtils::getChildNode(fxNode, "OptionData"));
239  barrier_.fromXML(XMLUtils::getChildNode(fxNode, "BarrierData"));
240  boughtCurrency_ = XMLUtils::getChildValue(fxNode, "BoughtCurrency", true);
241  soldCurrency_ = XMLUtils::getChildValue(fxNode, "SoldCurrency", true);
242  boughtAmount_ = XMLUtils::getChildValueAsDouble(fxNode, "BoughtAmount", true);
243  soldAmount_ = XMLUtils::getChildValueAsDouble(fxNode, "SoldAmount", true);
244 }

246 XMLNode* FxEuropeanBarrierOption::toXML(XMLDocument& doc) {
247  XMLNode* node = Trade::toXML(doc);
248  XMLNode* fxNode = doc.allocNode("FxEuropeanBarrierOptionData");
249  XMLUtils::appendNode(node, fxNode);
250 
251  XMLUtils::appendNode(fxNode, option_.toXML(doc));
252  XMLUtils::appendNode(fxNode, barrier_.toXML(doc));
253  XMLUtils::addChild(doc, fxNode, "BoughtCurrency", boughtCurrency_);
254  XMLUtils::addChild(doc, fxNode, "BoughtAmount", boughtAmount_);
255  XMLUtils::addChild(doc, fxNode, "SoldCurrency", soldCurrency_);
256  XMLUtils::addChild(doc, fxNode, "SoldAmount", soldAmount_);
257 
258  return node;
259 }
