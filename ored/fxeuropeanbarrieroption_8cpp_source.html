41void FxEuropeanBarrierOption::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
42
43 const boost::shared_ptr<Market> market = engineFactory->market();
44
45 // Only European Single Barrier supported for now
46 QL_REQUIRE(option_.style() == "European", "Option Style unknown: " << option_.style());
47 QL_REQUIRE(option_.exerciseDates().size() == 1, "Invalid number of excercise dates");
48 QL_REQUIRE(barrier_.levels().size() == 1, "Invalid number of barrier levels");
49 QL_REQUIRE(barrier_.style().empty() || barrier_.style() == "European", "Only european barrier style suppported");
50 QL_REQUIRE(tradeActions().empty(), "TradeActions not supported for FxEuropeanBarrierOption");
51
52 Currency boughtCcy = parseCurrency(boughtCurrency_);
53 Currency soldCcy = parseCurrency(soldCurrency_);
54 Real level = barrier_.levels()[0].value();
55 Real rebate = barrier_.rebate();
56 QL_REQUIRE(rebate >= 0, "Rebate must be non-negative");
57
58 // Replicate the payoff of European Barrier Option (with strike K and barrier B) using combinations of options
59
60 // Call
61 // Up
62 // In
63 // Long Up&Out Digital Option with barrier B payoff rebate
64 // B > K
65 // Long European Call Option with strike B
66 // Long Up&In Digital Option with barrier B payoff B - K
67 // B <= K
68 // Long European Call Option with strike K
69 // Out
70 // Long Up&In Digital Option with barrier B payoff rebate
71 // B > K
72 // Long European Call Option with strike K
73 // Short European Call Option with strike B
74 // Short Up&In Digital Option with barrier B payoff B - K
75 // B <= K
76 // 0
77 // Down
78 // In
79 // Long Down&Out Digital Option with barrier B payoff rebate
80 // B > K
81 // Long European Call Option with strike K
82 // B <= K
83 // 0
84 // Out
85 // Long Down&In Digital Option with barrier B payoff rebate
86 // B > K
87 // Long European Call Option with strike B
88 // Long Down&Out Digital Option with barrier B payoff B - K
89 // B <= K
90 // Long European Call Option with strike K
91
92 // Put
93 // Up
94 // In
95 // Long Up&Out Digital Option with barrier B payoff rebate
96 // B > K
97 // 0
98 // B <= K
99 // Long European Put Option with strike K
100 // Short European Put Option with strike B
101 // Short Up&Out Digital Option with barrier B payoff K - B
102 // Out
103 // Long Up&In Digital Option with barrier B payoff rebate
104 // B > K
105 // Long European Put Option with strike K
106 // B <= K
107 // Long European Put Option with strike B
108 // Long Up&Out Digital Option with barrier B payoff K - B
109 // Down
110 // In
111 // Long Down&Out Digital Option with barrier B payoff rebate
112 // B > K
113 // Long European Put Option with strike K
114 // B <= K
115 // Long European Put Option with strike B
116 // Long Down&In Digital Option with barrier B payoff K - B
117 // Out
118 // Long Down&In Digital Option with barrier B payoff rebate
119 // B > K
120 // 0
121 // B <= K
122 // Long European Put Option with strike K
123
124 Real strike = soldAmount_ / boughtAmount_;
125 Option::Type type = parseOptionType(option_.callPut());
126
127 // Exercise
128 Date expiryDate = parseDate(option_.exerciseDates().front());
129 Date paymentDate = expiryDate;
130
131 boost::shared_ptr<Exercise> exercise = boost::make_shared<EuropeanExercise>(expiryDate);
132
133 const boost::optional<OptionPaymentData>& opd = option_.paymentData();
134 if (opd) {
135 if (opd->rulesBased()) {
136 const Calendar& cal = opd->calendar();
137 QL_REQUIRE(cal != Calendar(), "Need a non-empty calendar for rules based payment date.");
138 paymentDate = cal.advance(expiryDate, opd->lag(), Days, opd->convention());
139 } else {
140 const vector<Date>& dates = opd->dates();
141 QL_REQUIRE(dates.size() == 1, "Need exactly one payment date for cash settled European option.");
142 paymentDate = dates[0];
143 }
144 QL_REQUIRE(paymentDate >= expiryDate, "Payment date must be greater than or equal to expiry date.");
145 }
146 boost::shared_ptr<Instrument> digital;
147 boost::shared_ptr<Instrument> vanillaK;
148 boost::shared_ptr<Instrument> vanillaB;
149 boost::shared_ptr<Instrument> rebateInstrument;
150
151 bool exercised = false;
152 Real exercisePrice = Null<Real>();
153 Barrier::Type barrierType = parseBarrierType(barrier_.type());
154
155 Option::Type rebateType;
156 if (barrierType == Barrier::Type::UpIn || barrierType == Barrier::Type::DownOut) {
157 // Payoff - Up&Out / Down&In Digital Option with barrier B payoff rebate
158 rebateType = Option::Put;
159 } else {
160 // Payoff - Up&In / Down&Out Digital Option with barrier B payoff rebate
161 rebateType = Option::Call;
162 }
163
164 if (paymentDate > expiryDate) {
165
166 // Has the option been marked as exercised
167 const boost::optional<OptionExerciseData>& oed = option_.exerciseData();
168 if (oed) {
169 QL_REQUIRE(oed->date() == expiryDate, "The supplied exercise date ("
170 << io::iso_date(oed->date())
171 << ") should equal the option's expiry date ("
172 << io::iso_date(expiryDate) << ").");
173 exercised = true;
174 exercisePrice = oed->price();
175 }
176
177 boost::shared_ptr<FxIndex> fxIndex;
178 if (option_.isAutomaticExercise()) {
179 QL_REQUIRE(!fxIndex_.empty(), "FX european barrier option trade with delay payment "
180 << id() << ": the FXIndex node needs to be populated.");
181 fxIndex = buildFxIndex(fxIndex_, soldCcy.code(), boughtCcy.code(), engineFactory->market(),
182 engineFactory->configuration(MarketContext::pricing));
183 requiredFixings_.addFixingDate(expiryDate, fxIndex_, paymentDate);
184 }
185
186 vanillaK = boost::make_shared<CashSettledEuropeanOption>(
187 type, strike, expiryDate, paymentDate, option_.isAutomaticExercise(), fxIndex, exercised, exercisePrice);
188 vanillaB = boost::make_shared<CashSettledEuropeanOption>(
189 type, level, expiryDate, paymentDate, option_.isAutomaticExercise(), fxIndex, exercised, exercisePrice);
190 digital = boost::make_shared<CashSettledEuropeanOption>(type, level, fabs(level - strike), expiryDate,
191 paymentDate, option_.isAutomaticExercise(), fxIndex,
192 exercised, exercisePrice);
193 rebateInstrument = boost::make_shared<CashSettledEuropeanOption>(rebateType, level, rebate, expiryDate,
194 paymentDate, option_.isAutomaticExercise(), fxIndex,
195 exercised, exercisePrice);
196 } else {
197 // Payoff - European Option with strike K
198 boost::shared_ptr<StrikedTypePayoff> payoffVanillaK(new PlainVanillaPayoff(type, strike));
199 // Payoff - European Option with strike B
200 boost::shared_ptr<StrikedTypePayoff> payoffVanillaB(new PlainVanillaPayoff(type, level));
201 // Payoff - Digital Option with barrier B payoff abs(B - K)
202 boost::shared_ptr<StrikedTypePayoff> payoffDigital(new CashOrNothingPayoff(type, level, fabs(level - strike)));
203 boost::shared_ptr<StrikedTypePayoff> rebatePayoff(new CashOrNothingPayoff(rebateType, level, rebate));
204
205 vanillaK = boost::make_shared<VanillaOption>(payoffVanillaK, exercise);
206 vanillaB = boost::make_shared<VanillaOption>(payoffVanillaB, exercise);
207 digital = boost::make_shared<VanillaOption>(payoffDigital, exercise);
208 rebateInstrument = boost::make_shared<VanillaOption>(rebatePayoff, exercise);
209 }
210
211 // This is for when/if a PayoffCurrency is added to the instrument,
212 // which would require flipping the underlying currency pair
213 const bool flipResults = false;
214
215 // set pricing engines
216 boost::shared_ptr<EngineBuilder> builder;
217 boost::shared_ptr<EngineBuilder> digitalBuilder;
218 boost::shared_ptr<VanillaOptionEngineBuilder> fxOptBuilder;
219
220 if (paymentDate > expiryDate) {
221 builder = engineFactory->builder("FxOptionEuropeanCS");
222 QL_REQUIRE(builder, "No builder found for FxOptionEuropeanCS");
223 fxOptBuilder = boost::dynamic_pointer_cast<FxEuropeanCSOptionEngineBuilder>(builder);
224
225 digitalBuilder = engineFactory->builder("FxDigitalOptionEuropeanCS");
226 QL_REQUIRE(digitalBuilder, "No builder found for FxDigitalOptionEuropeanCS");
227 auto fxDigitalOptBuilder = boost::dynamic_pointer_cast<FxDigitalCSOptionEngineBuilder>(digitalBuilder);
228 digital->setPricingEngine(fxDigitalOptBuilder->engine(boughtCcy, soldCcy));
229 rebateInstrument->setPricingEngine(fxDigitalOptBuilder->engine(boughtCcy, soldCcy));
230 } else {
231 builder = engineFactory->builder("FxOption");
232 QL_REQUIRE(builder, "No builder found for FxOption");
233 fxOptBuilder = boost::dynamic_pointer_cast<FxEuropeanOptionEngineBuilder>(builder);
234
235 digitalBuilder = engineFactory->builder("FxDigitalOption");
236 QL_REQUIRE(digitalBuilder, "No builder found for FxDigitalOption");
237 auto fxDigitalOptBuilder = boost::dynamic_pointer_cast<FxDigitalOptionEngineBuilder>(digitalBuilder);
238 digital->setPricingEngine(fxDigitalOptBuilder->engine(boughtCcy, soldCcy, flipResults));
239 rebateInstrument->setPricingEngine(fxDigitalOptBuilder->engine(boughtCcy, soldCcy, flipResults));
240 }
241
242 vanillaK->setPricingEngine(fxOptBuilder->engine(boughtCcy, soldCcy, paymentDate));
243 vanillaB->setPricingEngine(fxOptBuilder->engine(boughtCcy, soldCcy, paymentDate));
244
245 boost::shared_ptr<CompositeInstrument> qlInstrument = boost::make_shared<CompositeInstrument>();
246 qlInstrument->add(rebateInstrument);
247 if (type == Option::Call) {
248 if (barrierType == Barrier::Type::UpIn || barrierType == Barrier::Type::DownOut) {
249 if (level > strike) {
250 qlInstrument->add(vanillaB);
251 qlInstrument->add(digital);
252 } else {
253 qlInstrument->add(vanillaK);
254 }
255 } else if (barrierType == Barrier::Type::UpOut || barrierType == Barrier::Type::DownIn) {
256 if (level > strike) {
257 qlInstrument->add(vanillaK);
258 qlInstrument->add(vanillaB, -1);
259 qlInstrument->add(digital, -1);
260 } else {
261 // empty
262 }
263 } else {
264 QL_FAIL("Unknown Barrier Type: " << barrierType);
265 }
266 } else if (type == Option::Put) {
267 if (barrierType == Barrier::Type::UpIn || barrierType == Barrier::Type::DownOut) {
268 if (level > strike) {
269 // empty
270 } else {
271 qlInstrument->add(vanillaK);
272 qlInstrument->add(vanillaB, -1);
273 qlInstrument->add(digital, -1);
274 }
275 } else if (barrierType == Barrier::Type::UpOut || barrierType == Barrier::Type::DownIn) {
276 if (level > strike) {
277 qlInstrument->add(vanillaK);
278 } else {
279 qlInstrument->add(vanillaB);
280 qlInstrument->add(digital);
281 }
282 } else {
283 QL_FAIL("Unknown Barrier Type: " << barrierType);
284 }
285 }
286
287 // Add additional premium payments
288 Position::Type positionType = parsePositionType(option_.longShort());
289 Real bsInd = (positionType == QuantLib::Position::Long ? 1.0 : -1.0);
290 Real mult = boughtAmount_ * bsInd;
291
292 std::vector<boost::shared_ptr<Instrument>> additionalInstruments;
293 std::vector<Real> additionalMultipliers;
294 Date lastPremiumDate =
295 addPremiums(additionalInstruments, additionalMultipliers, mult, option_.premiumData(), -bsInd, soldCcy,
296 engineFactory, fxOptBuilder->configuration(MarketContext::pricing));
297
298 instrument_ = boost::shared_ptr<InstrumentWrapper>(
299 new VanillaInstrument(qlInstrument, mult, additionalInstruments, additionalMultipliers));
300
301 npvCurrency_ = soldCurrency_; // sold is the domestic
302 notional_ = soldAmount_;
303 notionalCurrency_ = soldCurrency_;
304 maturity_ = std::max({lastPremiumDate, paymentDate}); // delayed pay date is only affecting the maturity
305
306 additionalData_["boughtCurrency"] = boughtCurrency_;
307 additionalData_["boughtAmount"] = boughtAmount_;
308 additionalData_["soldCurrency"] = soldCurrency_;
309 additionalData_["soldAmount"] = soldAmount_;
310 if (!fxIndex_.empty())
311 additionalData_["FXIndex"] = fxIndex_;
312
313 // ISDA taxonomy
314 // ISDA taxonomy
315 additionalData_["isdaAssetClass"] = string("Foreign Exchange");
316 additionalData_["isdaBaseProduct"] = string("Simple Exotic");
317 additionalData_["isdaSubProduct"] = string("Barrier");
318 additionalData_["isdaTransaction"] = string("");
319}

321void FxEuropeanBarrierOption::fromXML(XMLNode* node) {
322 Trade::fromXML(node);
323 XMLNode* fxNode = XMLUtils::getChildNode(node, "FxEuropeanBarrierOptionData");
324 QL_REQUIRE(fxNode, "No FxEuropeanBarrierOptionData Node");
325 option_.fromXML(XMLUtils::getChildNode(fxNode, "OptionData"));
326 barrier_.fromXML(XMLUtils::getChildNode(fxNode, "BarrierData"));
327 boughtCurrency_ = XMLUtils::getChildValue(fxNode, "BoughtCurrency", true);
328 soldCurrency_ = XMLUtils::getChildValue(fxNode, "SoldCurrency", true);
329 boughtAmount_ = XMLUtils::getChildValueAsDouble(fxNode, "BoughtAmount", true);
330 soldAmount_ = XMLUtils::getChildValueAsDouble(fxNode, "SoldAmount", true);
331 fxIndex_ = XMLUtils::getChildValue(fxNode, "FXIndex", false, "");
332}
333
334XMLNode* FxEuropeanBarrierOption::toXML(XMLDocument& doc) {
335 XMLNode* node = Trade::toXML(doc);
336 XMLNode* fxNode = doc.allocNode("FxEuropeanBarrierOptionData");
337 XMLUtils::appendNode(node, fxNode);
338
339 XMLUtils::appendNode(fxNode, option_.toXML(doc));
340 XMLUtils::appendNode(fxNode, barrier_.toXML(doc));
341 XMLUtils::addChild(doc, fxNode, "BoughtCurrency", boughtCurrency_);
342 XMLUtils::addChild(doc, fxNode, "BoughtAmount", boughtAmount_);
343 XMLUtils::addChild(doc, fxNode, "SoldCurrency", soldCurrency_);
344 XMLUtils::addChild(doc, fxNode, "SoldAmount", soldAmount_);
345
346 if (!fxIndex_.empty())
347 XMLUtils::addChild(doc, fxNode, "FXIndex", fxIndex_);
348
349 return node;
350}
