31 CSVFileReader::CSVFileReader(const std::string& fileName, const bool firstLineContainsHeaders,
32  const std::string& delimiters, const char eolMarker)
33  : fileName_(fileName), hasHeaders_(firstLineContainsHeaders), delimiters_(delimiters), eolMarker_(eolMarker),
34  currentLine_(Null<Size>()), numberOfColumns_(Null<Size>()) {
35  file_.open(fileName);
36  QL_REQUIRE(file_.is_open(), "CSVFileReader: error opening file " << fileName);
37  if (firstLineContainsHeaders) {
38  QL_REQUIRE(!file_.eof(), "CSVFileReader: file is empty: " << fileName);
39  std::string line;
40  getline(file_, line, eolMarker);
41  boost::trim(line);
42  boost::split(headers_, line, boost::is_any_of(delimiters_), boost::token_compress_off);
43  numberOfColumns_ = headers_.size();
44  }
45 }

61 bool CSVFileReader::next() {
62  QL_REQUIRE(file_.is_open(), "CSVFileReader: file is not open, can not move to next line");
63  std::string line = "";
64  // skip empty lines
65  while (line.size() == 0 && !file_.eof()) {
66  getline(file_, line, eolMarker_);
67  boost::trim(line);
68  }
69  if (file_.eof() && line.empty()) {
70  close();
71  return false;
72  }
73  if (currentLine_ == Null<Size>())
74  currentLine_ = 0;
75  else
76  ++currentLine_;
77  boost::split(data_, line, boost::is_any_of(delimiters_), boost::token_compress_off);
78  if (numberOfColumns_ == Null<Size>())
79  numberOfColumns_ = data_.size();
80  else
81  QL_REQUIRE(data_.size() == numberOfColumns_, "CSVFileReader: data line #"
82  << currentLine_ << " has " << data_.size()
83  << " fields, expected " << numberOfColumns_);
84  return true;
85 }

92 std::string CSVFileReader::get(const std::string& field) const {
93  QL_REQUIRE(hasHeaders_, "CSVFileReader: can not get data by field, file does not have headers");
94  QL_REQUIRE(currentLine_ != Null<Size>(), "CSVFileReader: can not get data, need call to next() first");
95  Size index = std::find(headers_.begin(), headers_.end(), field) - headers_.begin();
96  QL_REQUIRE(index < headers_.size(), "CSVFileReader: field \"" << field << "\" not found.");
97  QL_REQUIRE(index < data_.size(), "CSVFileReader: unexpected data size (" << data_.size() << "), required at least "
98  << index + 1 << ", while reading field \""
99  << field << "\"");
100  return data_[index];
101 }
