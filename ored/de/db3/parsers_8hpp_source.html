244 template <class T> std::vector<T> parseListOfValues(string s, std::function<T(string)> parser) {
245  boost::trim(s);
246  std::vector<T> vec;
247  boost::char_separator<char> sep(",");
248  boost::tokenizer<boost::char_separator<char>> tokens(s, sep);
249  for (auto r : tokens) {
250  boost::trim(r);
251  vec.push_back(parser(r));
252  }
253  return vec;
254 }

256 template <class T> std::vector<T> parseVectorOfValues(std::vector<std::string> str, std::function<T(string)> parser) {
257  std::vector<T> vec;
258  for (auto s : str) {
259  vec.push_back(parser(s));
260  }
261  return vec;
262 }

329  DLOG("tryParse: attempting to parse " << str);
330  try {
331  obj = parser(str);
332  } catch (...) {
333  TLOG("String " << str << " could not be parsed");
334  return false;
335  }
