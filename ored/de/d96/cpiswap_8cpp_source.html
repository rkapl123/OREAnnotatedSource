181 BOOST_AUTO_TEST_CASE(testCPISwapPrice) {
182 
183  BOOST_TEST_MESSAGE("Testing CPI Swap Price...");
184 
185  // build market
186  Date today(18, July, 2016);
187  Settings::instance().evaluationDate() = today;
188  boost::shared_ptr<TestMarket> market = boost::make_shared<TestMarket>();
189  Date marketDate = market->asofDate();
190  BOOST_CHECK_EQUAL(today, marketDate);
191  Settings::instance().evaluationDate() = marketDate;
192 
193  // Test if GBP discount curve is empty
194  Handle<YieldTermStructure> dts = market->discountCurve("GBP");
195  QL_REQUIRE(!dts.empty(), "GBP discount curve not found");
196  BOOST_CHECK_CLOSE(market->discountCurve("GBP")->discount(today + 1 * Years), 0.9914, 0.0001);
197 
198  // Test if GBP Libor curve is empty
199  Handle<IborIndex> iis = market->iborIndex("GBP-LIBOR-6M");
200  QL_REQUIRE(!iis.empty(), "GBP LIBOR 6M ibor Index not found");
201  BOOST_TEST_MESSAGE(
202  "CPISwap: Projected Libor fixing: " << market->iborIndex("GBP-LIBOR-6M")->forecastFixing(today + 1 * Years));
203 
204  // Test if GBP discount curve is empty
205  Handle<ZeroInflationIndex> infidx = market->zeroInflationIndex("UKRPI");
206  QL_REQUIRE(!infidx.empty(), "UKRPI inflation index not found");
207  BOOST_TEST_MESSAGE("CPISwap: Projected UKRPI rate: " << infidx->fixing(today + 1 * Years));
208 
209  // envelope
210  Envelope env("CP");
211 
212  // Start/End date
213  Date startDate = today;
214  Date endDate = today + 5 * Years;
215 
216  // date 2 string
217  std::ostringstream oss;
218  oss << io::iso_date(startDate);
219  string start(oss.str());
220  oss.str("");
221  oss.clear();
222  oss << io::iso_date(endDate);
223  string end(oss.str());
224 
225  // Schedules
226  string conv = "MF";
227  string rule = "Forward";
228  ScheduleData scheduleLibor(ScheduleRules(start, end, "6M", "UK", conv, conv, rule));
229  ScheduleData scheduleCPI(ScheduleRules(start, end, "1Y", "UK", conv, conv, rule));
230 
231  // Leg variables
232  bool isInArrears = false;
233  string dc = "ACT/ACT";
234  vector<Real> notional(1, 10000000);
235  string paymentConvention = "F";
236 
237  // GBP Libor Leg
238  bool isPayerLibor = true;
239  string indexLibor = "GBP-LIBOR-6M";
240  vector<Real> spread(1, 0);
241  LegData legLibor(boost::make_shared<FloatingLegData>(indexLibor, 0, isInArrears, spread), isPayerLibor, "GBP",
242  scheduleLibor, "A365F", notional, vector<string>(), paymentConvention);
243 
244  // GBP CPI Leg
245  bool isPayerCPI = false;
246  string indexCPI = "UKRPI";
247  Real baseCPI = 210.0;
248  string CPIlag = "2M";
249  std::vector<double> fixedRate(1, 0.02);
250  bool interpolated = false;
251  LegData legCPI(
252  boost::make_shared<CPILegData>(indexCPI, start, baseCPI, CPIlag, (interpolated ? "Linear" : "Flat"), fixedRate),
253  isPayerCPI, "GBP", scheduleCPI, dc, notional, vector<string>(), paymentConvention, false, true);
254 
255  // Build swap trades
256  boost::shared_ptr<Trade> CPIswap(new ore::data::Swap(env, legLibor, legCPI));
257 
258  // engine data and factory
259  boost::shared_ptr<EngineData> engineData = boost::make_shared<EngineData>();
260  engineData->model("Swap") = "DiscountedCashflows";
261  engineData->engine("Swap") = "DiscountingSwapEngine";
262  boost::shared_ptr<EngineFactory> engineFactory = boost::make_shared<EngineFactory>(engineData, market);
263  engineFactory->registerBuilder(boost::make_shared<SwapEngineBuilder>());
264 
265  // build swaps and portfolio
266  boost::shared_ptr<Portfolio> portfolio(new Portfolio());
267  CPIswap->id() = "CPI_Swap";
268 
269  portfolio->add(CPIswap);
270  portfolio->build(engineFactory);
271 
272  // check CPI swap NPV against pure QL pricing
273  Schedule floatSchedule(startDate, endDate, 6 * Months, UnitedKingdom(), ModifiedFollowing, ModifiedFollowing,
274  DateGeneration::Forward, false);
275  Schedule cpiSchedule(startDate, endDate, 1 * Years, UnitedKingdom(), ModifiedFollowing, ModifiedFollowing,
276  DateGeneration::Forward, false);
277  Leg floatLeg = IborLeg(floatSchedule, *market->hGBP).withNotionals(10000000.0);
278  Leg cpiLeg = CPILeg(cpiSchedule, *market->hUKRPI, baseCPI, 2 * Months)
279  .withFixedRates(0.02)
280  .withNotionals(10000000)
281  .withObservationInterpolation(CPI::Flat)
282  .withPaymentDayCounter(ActualActual())
283  .withPaymentAdjustment(Following);
284  auto pricer = boost::make_shared<CPICouponPricer>(market->hGBP->forwardingTermStructure());
285  for (auto const& c : cpiLeg) {
286  if (auto cpn = boost::dynamic_pointer_cast<CPICoupon>(c))
287  cpn->setPricer(pricer);
288  }
289 
290  QuantLib::Swap qlSwap(floatLeg, cpiLeg);
291  auto dscEngine = boost::make_shared<DiscountingSwapEngine>(market->hGBP->forwardingTermStructure());
292  qlSwap.setPricingEngine(dscEngine);
293  BOOST_TEST_MESSAGE("Leg 1 NPV: ORE = "
294  << boost::static_pointer_cast<QuantLib::Swap>(CPIswap->instrument()->qlInstrument())->legNPV(0)
295  << " QL = " << qlSwap.legNPV(0));
296  BOOST_TEST_MESSAGE("Leg 2 NPV: ORE = "
297  << boost::static_pointer_cast<QuantLib::Swap>(CPIswap->instrument()->qlInstrument())->legNPV(1)
298  << " QL = " << qlSwap.legNPV(1));
299  BOOST_CHECK_CLOSE(CPIswap->instrument()->NPV(), qlSwap.NPV(), 1E-8); // this is 1E-10 rel diff
300 }
