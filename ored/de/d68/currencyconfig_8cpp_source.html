30 void CurrencyConfig::fromXML(XMLNode* baseNode) {
31  currencies_.clear();
32  XMLUtils::checkNode(baseNode, "CurrencyConfig");
33 
34  for (auto node : XMLUtils::getChildrenNodes(baseNode, "Currency")) {
35  string name = XMLUtils::getChildValue(node, "Name", false);
36  string isoCode = XMLUtils::getChildValue(node, "ISOCode", false);
37  DLOG("Loading external currency configuration for " << isoCode);
38  Integer numericCode = parseInteger(XMLUtils::getChildValue(node, "NumericCode", false));
39  string symbol = XMLUtils::getChildValue(node, "Symbol", false);
40  string fractionSymbol = XMLUtils::getChildValue(node, "Symbol", false);
41  Integer fractionsPerUnit = parseInteger(XMLUtils::getChildValue(node, "FractionsPerUnit", false));
42  Rounding::Type roundingType = parseRoundingType(XMLUtils::getChildValue(node, "RoundingType", false));
43  Integer precision = parseInteger(XMLUtils::getChildValue(node, "RoundingPrecision", false));
44  // the digit where we switch form roundng down to rounding up, typically 5 and used across all
45  // Integer digit = parseInteger(XMLUtils::getChildValue(node, "Digit", false));
46  string format = XMLUtils::getChildValue(node, "Format", false);
47  Rounding rounding;
48  switch (roundingType) {
49  case Rounding::Type::Up: {
50  rounding = UpRounding(precision);
51  break;
52  }
53  case Rounding::Type::Down: {
54  rounding = DownRounding(precision);
55  break;
56  }
57  case Rounding::Type::Closest: {
58  rounding = ClosestRounding(precision);
59  break;
60  }
61  case Rounding::Type::Floor: {
62  rounding = FloorTruncation(precision);
63  break;
64  }
65  case Rounding::Type::Ceiling: {
66  rounding = CeilingTruncation(precision);
67  break;
68  }
69  default: {
70  ALOG("Rounding type not recognized, falling back on 'Closest'");
71  rounding = ClosestRounding(precision);
72  }
73  }
74 
75  QuantExt::ConfigurableCurrency c(name, isoCode, numericCode, symbol, fractionSymbol, fractionsPerUnit, rounding,
76  format);
77  currencies_.push_back(c);
78 
79  DLOG("loading configuration for currency code " << isoCode);
80 
81  // this pushes any additional currency into the parser's static map
82  parseCurrency(c.code(), c);
83  }
84 }

86 XMLNode* CurrencyConfig::toXML(XMLDocument& doc) {
87  XMLNode* node = doc.allocNode("CurrencyConfig");
88  for (auto ccy : currencies_) {
89  XMLNode* ccyNode = XMLUtils::addChild(doc, node, "Currency");
90  XMLUtils::addChild(doc, ccyNode, "Name", ccy.name());
91  XMLUtils::addChild(doc, ccyNode, "ISOCode", ccy.code());
92  XMLUtils::addChild(doc, ccyNode, "NumericCode", to_string(ccy.numericCode()));
93  XMLUtils::addChild(doc, ccyNode, "Symbol", ccy.symbol());
94  XMLUtils::addChild(doc, ccyNode, "FractionSymbol", ccy.fractionSymbol());
95  XMLUtils::addChild(doc, ccyNode, "RoundingType", to_string(ccy.rounding().type()));
96  XMLUtils::addChild(doc, ccyNode, "RoundingPrecision", to_string(ccy.rounding().precision()));
97  // XMLUtils::addChild(doc, ccyNode, "Digit", to_string(ccy.rounding().roundingDigit()));
98  XMLUtils::addChild(doc, ccyNode, "Format", ccy.format());
99  }
100  return node;
101 }
