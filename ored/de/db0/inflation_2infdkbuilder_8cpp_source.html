41 InfDkBuilder::InfDkBuilder(const boost::shared_ptr<ore::data::Market>& market, const boost::shared_ptr<InfDkData>& data,
42  const std::string& configuration, const std::string& referenceCalibrationGrid,
43  const bool dontCalibrate)
44  : market_(market), configuration_(configuration), data_(data), referenceCalibrationGrid_(referenceCalibrationGrid),
45  dontCalibrate_(dontCalibrate) {
46 
47  LOG("DkBuilder for " << data_->index());
48 
49  if (!data_->calibrationBaskets().empty()) {
50  QL_REQUIRE(data_->calibrationBaskets().size() == 1, "If there is a calibration basket, expect exactly 1.");
51  const CalibrationBasket& cb = data_->calibrationBaskets()[0];
52  QL_REQUIRE(!cb.empty(), "If there is a calibration basket, expect it to be non-empty.");
53  optionActive_ = vector<bool>(cb.instruments().size(), false);
54  }
55 
56  marketObserver_ = boost::make_shared<MarketObserver>();
57 
58  // get market data
59  inflationIndex_ =
60  boost::dynamic_pointer_cast<ZeroInflationIndex>(*market_->zeroInflationIndex(data_->index(), configuration_));
61  QL_REQUIRE(inflationIndex_, "DkBuilder: requires ZeroInflationIndex, got " << data_->index());
62  rateCurve_ = market_->discountCurve(inflationIndex_->currency().code(), configuration_);
63  infVol_ = market_->cpiInflationCapFloorVolatilitySurface(data_->index(), configuration_);
64 
65  // register with market observables except vols
66  marketObserver_->registerWith(inflationIndex_);
67  marketObserver_->registerWith(rateCurve_);
68 
69  // register the builder with the market observer
70  registerWith(marketObserver_);
71  registerWith(infVol_);
72  // notify observers of all market data changes, not only when not calculated
73  alwaysForwardNotifications();
74 
75  // build option basket and derive parametrization from it
76  const ReversionParameter& reversion = data_->reversion();
77  const VolatilityParameter& volatility = data_->volatility();
78  if (volatility.calibrate() || reversion.calibrate())
79  buildCapFloorBasket();
80 
81  Array aTimes(volatility.times().begin(), volatility.times().end());
82  Array hTimes(reversion.times().begin(), reversion.times().end());
83  Array alpha(volatility.values().begin(), volatility.values().end());
84  Array h(reversion.values().begin(), reversion.values().end());
85 
86  if (volatility.type() == ParamType::Constant) {
87  QL_REQUIRE(volatility.times().size() == 0, "empty alpha times expected");
88  QL_REQUIRE(volatility.values().size() == 1, "initial alpha array should have size 1");
89  } else if (volatility.type() == ParamType::Piecewise) {
90  if (volatility.calibrate() && data_->calibrationType() == CalibrationType::Bootstrap) {
91  if (volatility.times().size() > 0) {
92  DLOG("overriding alpha time grid with option expiries");
93  }
94  QL_REQUIRE(optionExpiries_.size() > 0, "empty option expiries");
95  aTimes = Array(optionExpiries_.begin(), optionExpiries_.end() - 1);
96  alpha = Array(aTimes.size() + 1, volatility.values()[0]);
97  } else { // use input time grid and input alpha array otherwise
98  QL_REQUIRE(alpha.size() == aTimes.size() + 1, "alpha grids do not match");
99  }
100  } else {
101  QL_FAIL("Expected DK model data volatility parameter to be Constant or Piecewise.");
102  }
103 
104  if (reversion.type() == ParamType::Constant) {
105  QL_REQUIRE(reversion.values().size() == 1, "reversion grid size 1 expected");
106  QL_REQUIRE(reversion.times().size() == 0, "empty reversion time grid expected");
107  } else if (reversion.type() == ParamType::Piecewise) {
108  if (reversion.calibrate() && data_->calibrationType() == CalibrationType::Bootstrap) {
109  if (reversion.times().size() > 0) {
110  DLOG("overriding H time grid with option expiries");
111  }
112  QL_REQUIRE(optionExpiries_.size() > 0, "empty option expiries");
113  hTimes = Array(optionExpiries_.begin(), optionExpiries_.end() - 1);
114  h = Array(hTimes.size() + 1, reversion.values()[0]);
115  } else { // use input time grid and input alpha array otherwise
116  QL_REQUIRE(h.size() == hTimes.size() + 1, "H grids do not match");
117  }
118  } else {
119  QL_FAIL("Expected DK model data reversion parameter to be Constant or Piecewise.");
120  }
121 
122  DLOG("before calibration: alpha times = " << aTimes << " values = " << alpha);
123  DLOG("before calibration: h times = " << hTimes << " values = " << h)
124 
125  if (reversion.reversionType() == LgmData::ReversionType::HullWhite &&
126  volatility.volatilityType() == LgmData::VolatilityType::HullWhite) {
127  DLOG("INF parametrization: InfDkPiecewiseConstantHullWhiteAdaptor");
128  parametrization_ = boost::make_shared<InfDkPiecewiseConstantHullWhiteAdaptor>(
129  inflationIndex_->currency(), inflationIndex_->zeroInflationTermStructure(), aTimes, alpha, hTimes, h,
130  data_->index());
131  } else if (reversion.reversionType() == LgmData::ReversionType::HullWhite) {
132  DLOG("INF parametrization for " << data_->index() << ": InfDkPiecewiseConstant");
133  parametrization_ = boost::make_shared<InfDkPiecewiseConstantParametrization>(
134  inflationIndex_->currency(), inflationIndex_->zeroInflationTermStructure(), aTimes, alpha, hTimes, h,
135  data_->index());
136  } else {
137  parametrization_ = boost::make_shared<InfDkPiecewiseLinearParametrization>(
138  inflationIndex_->currency(), inflationIndex_->zeroInflationTermStructure(), aTimes, alpha, hTimes, h,
139  data_->index());
140  DLOG("INF parametrization for " << data_->index() << ": InfDkPiecewiseLinear");
141  }
142 
143  DLOG("alpha times size: " << aTimes.size());
144  DLOG("lambda times size: " << hTimes.size());
145 
146  DLOG("Apply shift horizon and scale");
147  Time horizon = data_->reversionTransformation().horizon();
148  Real scaling = data_->reversionTransformation().scaling();
149  QL_REQUIRE(horizon >= 0.0, "shift horizon must be non negative");
150  QL_REQUIRE(scaling > 0.0, "scaling must be positive");
151 
152  if (horizon > 0.0) {
153  DLOG("Apply shift horizon " << horizon << " to the " << data_->index() << " DK model");
154  parametrization_->shift() = horizon;
155  }
156 
157  if (scaling != 1.0) {
158  DLOG("Apply scaling " << scaling << " to the " << data_->index() << " DK model");
159  parametrization_->scaling() = scaling;
160  }
161 }

189 Date InfDkBuilder::optionMaturityDate(const Size j) const {
190  Date today = Settings::instance().evaluationDate();
191  const auto& ci = data_->calibrationBaskets()[0].instruments();
192  QL_REQUIRE(j < ci.size(), "InfDkBuilder::optionMaturityDate(" << j << "): out of bounds, got " << ci.size()
193  << " calibration instruments");
194  auto cf = boost::dynamic_pointer_cast<CpiCapFloor>(ci.at(j));
195  QL_REQUIRE(cf, "InfDkBuilder::optionMaturityDate("
196  << j << "): expected CpiCapFloor calibration instruments, could not cast");
197  Date res = optionMaturity(cf->maturity(), inflationIndex_->fixingCalendar());
198  QL_REQUIRE(res > today, "expired calibration option expiry " << io::iso_date(res));
199  return res;

212 bool InfDkBuilder::volSurfaceChanged(const bool updateCache) const {
213  bool hasUpdated = false;
214 
215  boost::shared_ptr<QuantExt::CPIBlackCapFloorEngine> engine =
216  boost::make_shared<QuantExt::CPIBlackCapFloorEngine>(rateCurve_, infVol_);
217 
218  Calendar fixCalendar = inflationIndex_->fixingCalendar();
219  BusinessDayConvention bdc = infVol_->businessDayConvention();
220  Date baseDate = inflationIndex_->zeroInflationTermStructure()->baseDate();
221  Real baseCPI = inflationIndex_->fixing(baseDate);
222  Period lag = infVol_->observationLag();
223  Handle<ZeroInflationIndex> hIndex(inflationIndex_);
224 
225  // if cache doesn't exist resize vector
226  if (infPriceCache_.size() != optionBasket_.size())
227  infPriceCache_ = vector<Real>(optionBasket_.size(), Null<Real>());
228 
229  // Handle on calibration instruments. No checks this time.
230  const auto& ci = data_->calibrationBaskets()[0].instruments();
231 
232  Real nominal = 1.0;
233  Size optionCounter = 0;
234  Date today = Settings::instance().evaluationDate();
235  for (Size j = 0; j < ci.size(); j++) {
236 
237  if (!optionActive_[j])
238  continue;
239 
240  auto cpiCapFloor = boost::dynamic_pointer_cast<CpiCapFloor>(ci[j]);
241  QL_REQUIRE(cpiCapFloor, "Expected CpiCapFloor calibration instruments in DK inflation model data.");
242 
243  Date expiryDate = optionMaturityDate(j);
244  auto strikeValue = optionStrikeValue(j);
245 
246  Option::Type capfloor = cpiCapFloor->type() == CapFloor::Cap ? Option::Call : Option::Put;
247 
248  boost::shared_ptr<CPICapFloor> h =
249  boost::make_shared<CPICapFloor>(capfloor, nominal, today, baseCPI, expiryDate, fixCalendar, bdc,
250  fixCalendar, bdc, strikeValue, hIndex, lag);
251  h->setPricingEngine(engine);
252  Real price = h->NPV();
253  if (!close_enough(infPriceCache_[optionCounter], price)) {
254  if (updateCache)
255  infPriceCache_[optionCounter] = price;
256  hasUpdated = true;
257  }
258  optionCounter++;
259  }
260  return hasUpdated;
261 }

263 void InfDkBuilder::buildCapFloorBasket() const {
264 
265  // Checks that there is a basket.
266  QL_REQUIRE(!data_->calibrationBaskets().empty(), "No calibration basket provided in inflation DK model data.");
267  const CalibrationBasket& cb = data_->calibrationBaskets()[0];
268  const auto& ci = cb.instruments();
269 
270  QL_REQUIRE(ci.size() == optionActive_.size(), "Expected the option active vector size to equal the "
271  << "number of calibration instruments");
272  fill(optionActive_.begin(), optionActive_.end(), false);
273 
274  DLOG("build reference date grid '" << referenceCalibrationGrid_ << "'");
275  Date lastRefCalDate = Date::minDate();
276  std::vector<Date> referenceCalibrationDates;
277  if (!referenceCalibrationGrid_.empty())
278  referenceCalibrationDates = DateGrid(referenceCalibrationGrid_).dates();
279 
280  boost::shared_ptr<QuantExt::CPIBlackCapFloorEngine> engine =
281  boost::make_shared<QuantExt::CPIBlackCapFloorEngine>(rateCurve_, infVol_);
282 
283  Calendar fixCalendar = inflationIndex_->fixingCalendar();
284  Date baseDate = inflationIndex_->zeroInflationTermStructure()->baseDate();
285  Real baseCPI = dontCalibrate_ ? 100. : inflationIndex_->fixing(baseDate);
286  BusinessDayConvention bdc = infVol_->businessDayConvention();
287  Period lag = infVol_->observationLag();
288  Handle<ZeroInflationIndex> hIndex(inflationIndex_);
289  Date startDate = Settings::instance().evaluationDate();
290 
291  Real nominal = 1.0;
292  vector<Time> expiryTimes;
293  optionBasket_.clear();
294  for (Size j = 0; j < ci.size(); j++) {
295 
296  auto cpiCapFloor = boost::dynamic_pointer_cast<CpiCapFloor>(ci[j]);
297  QL_REQUIRE(cpiCapFloor, "Expected CpiCapFloor calibration instruments in DK inflation model data.");
298 
299  Date expiryDate = optionMaturityDate(j);
300 
301  // check if we want to keep the helper when a reference calibration grid is given
302  auto refCalDate =
303  std::lower_bound(referenceCalibrationDates.begin(), referenceCalibrationDates.end(), expiryDate);
304  if (refCalDate == referenceCalibrationDates.end() || *refCalDate > lastRefCalDate) {
305  Real strikeValue = optionStrikeValue(j);
306  Option::Type capfloor = cpiCapFloor->type() == CapFloor::Cap ? Option::Call : Option::Put;
307  boost::shared_ptr<CPICapFloor> cf =
308  boost::make_shared<CPICapFloor>(capfloor, nominal, startDate, baseCPI, expiryDate, fixCalendar, bdc,
309  fixCalendar, bdc, strikeValue, hIndex, lag);
310  cf->setPricingEngine(engine);
311  Real tte = inflationYearFraction(inflationIndex_->frequency(), inflationIndex_->interpolated(),
312  inflationIndex_->zeroInflationTermStructure()->dayCounter(), baseDate,
313  cf->fixingDate());
314 
315  Real tteFromBase = infVol_->timeFromBase(expiryDate);
316 
317  Real marketPrem = dontCalibrate_ || tte <= 0 || tteFromBase <= 0 ? 0.01 : cf->NPV();
318  boost::shared_ptr<QuantExt::CpiCapFloorHelper> helper =
319  boost::make_shared<QuantExt::CpiCapFloorHelper>(capfloor, baseCPI, expiryDate, fixCalendar, bdc,
320  fixCalendar, bdc, strikeValue, hIndex, lag, marketPrem);
321 
322  // we might produce duplicate expiry times even if the fixing dates are all different
323  if (tte > 0 && tteFromBase >= 0 && std::find_if(expiryTimes.begin(), expiryTimes.end(), [tte](Real x) {
324  return QuantLib::close_enough(x, tte);
325  }) == expiryTimes.end()) {
326  optionBasket_.push_back(helper);
327  helper->performCalculations();
328  expiryTimes.push_back(tte);
329  DLOG("Added InflationOptionHelper index="
330  << data_->index() << ", type=" << (capfloor == Option::Type::Call ? "Cap" : "Floor") << ", expiry="
331  << QuantLib::io::iso_date(expiryDate) << ", baseCPI=" << baseCPI << ", strike=" << strikeValue
332  << ", lag=" << lag << ", marketPremium=" << marketPrem << ", tte=" << tte);
333  optionActive_[j] = true;
334  if (refCalDate != referenceCalibrationDates.end())
335  lastRefCalDate = *refCalDate;
336  } else {
337  if (data_->ignoreDuplicateCalibrationExpiryTimes()) {
338  DLOG("Skipped InflationOptionHelper index="
339  << data_->index() << ", type=" << (capfloor == Option::Type::Call ? "Cap" : "Floor")
340  << ", expiry=" << QuantLib::io::iso_date(expiryDate) << ", baseCPI=" << baseCPI
341  << ", strike=" << strikeValue << ", lag=" << lag << ", marketPremium=" << marketPrem
342  << ", tte=" << tte << " since we already have a helper with the same expiry time.");
343  } else {
344  QL_FAIL("InfDkBuilder: a CPI cap floor calibration instrument with the expiry time, "
345  << tte << ", was already added.");
346  }
347  }
348  }
349  }
350 
351  std::sort(expiryTimes.begin(), expiryTimes.end());
352  auto itExpiryTime = unique(expiryTimes.begin(), expiryTimes.end());
353  expiryTimes.resize(distance(expiryTimes.begin(), itExpiryTime));
354 
355  optionExpiries_ = Array(expiryTimes.size());
356  for (Size j = 0; j < expiryTimes.size(); j++)
357  optionExpiries_[j] = expiryTimes[j];
358 }
