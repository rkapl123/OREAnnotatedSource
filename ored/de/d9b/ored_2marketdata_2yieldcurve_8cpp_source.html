97 template <template <class> class CurveType>
98 boost::shared_ptr<YieldTermStructure> buildYieldCurve(const vector<Date>& dates, const vector<QuantLib::Real>& rates,
99  const DayCounter& dayCounter,
100  YieldCurve::InterpolationMethod interpolationMethod) {
101 
102  boost::shared_ptr<YieldTermStructure> yieldts;
103  switch (interpolationMethod) {
104  case YieldCurve::InterpolationMethod::Linear:
105  yieldts.reset(new CurveType<QuantLib::Linear>(dates, rates, dayCounter, QuantLib::Linear()));
106  break;
107  case YieldCurve::InterpolationMethod::LogLinear:
108  yieldts.reset(new CurveType<QuantLib::LogLinear>(dates, rates, dayCounter, QuantLib::LogLinear()));
109  break;
110  case YieldCurve::InterpolationMethod::NaturalCubic:
111  yieldts.reset(new CurveType<QuantLib::Cubic>(dates, rates, dayCounter,
112  QuantLib::Cubic(CubicInterpolation::Kruger, true)));
113  break;
114  case YieldCurve::InterpolationMethod::FinancialCubic:
115  yieldts.reset(new CurveType<QuantLib::Cubic>(dates, rates, dayCounter,
116  QuantLib::Cubic(CubicInterpolation::Kruger, true,
117  CubicInterpolation::SecondDerivative, 0.0,
118  CubicInterpolation::FirstDerivative)));
119  break;
120  case YieldCurve::InterpolationMethod::ConvexMonotone:
121  yieldts.reset(new CurveType<QuantLib::ConvexMonotone>(dates, rates, dayCounter));
122  break;
123  case YieldCurve::InterpolationMethod::Quadratic:
124  yieldts.reset(new CurveType<QuantExt::Quadratic>(dates, rates, dayCounter, QuantExt::Quadratic(1, 0, 1, 0, 1)));
125  break;
126  case YieldCurve::InterpolationMethod::LogQuadratic:
127  yieldts.reset(
128  new CurveType<QuantExt::LogQuadratic>(dates, rates, dayCounter, QuantExt::LogQuadratic(1, 0, -1, 0, 1)));
129  break;
130  case YieldCurve::InterpolationMethod::Hermite:
131  yieldts.reset(new CurveType<QuantLib::Cubic>(dates, rates, dayCounter, Cubic(CubicInterpolation::Parabolic)));
132  break;
133  case YieldCurve::InterpolationMethod::CubicSpline:
134  yieldts.reset(new CurveType<QuantLib::Cubic>(dates, rates, dayCounter,
135  Cubic(CubicInterpolation::Spline, false,
136  CubicInterpolation::SecondDerivative, 0.0,
137  CubicInterpolation::SecondDerivative, 0.0)));
138  break;
139  default:
140  QL_FAIL("Interpolation method not recognised.");
141  }
142  return yieldts;
143 }

165 YieldCurve::InterpolationMethod parseYieldCurveInterpolationMethod(const string& s) {
166  if (s == "Linear")
167  return YieldCurve::InterpolationMethod::Linear;
168  else if (s == "LogLinear")
169  return YieldCurve::InterpolationMethod::LogLinear;
170  else if (s == "NaturalCubic")
171  return YieldCurve::InterpolationMethod::NaturalCubic;
172  else if (s == "FinancialCubic")
173  return YieldCurve::InterpolationMethod::FinancialCubic;
174  else if (s == "ConvexMonotone")
175  return YieldCurve::InterpolationMethod::ConvexMonotone;
176  else if (s == "ExponentialSplines")
177  return YieldCurve::InterpolationMethod::ExponentialSplines;
178  else if (s == "Quadratic")
179  return YieldCurve::InterpolationMethod::Quadratic;
180  else if (s == "LogQuadratic")
181  return YieldCurve::InterpolationMethod::LogQuadratic;
182  else if (s == "Hermite")
183  return YieldCurve::InterpolationMethod::Hermite;
184  else if (s == "CubicSpline")
185  return YieldCurve::InterpolationMethod::CubicSpline;
186  else if (s == "NelsonSiegel")
187  return YieldCurve::InterpolationMethod::NelsonSiegel;
188  else if (s == "Svensson")
189  return YieldCurve::InterpolationMethod::Svensson;
190  else
191  QL_FAIL("Yield curve interpolation method " << s << " not recognized");
192 };

205 YieldCurve::YieldCurve(Date asof, YieldCurveSpec curveSpec, const CurveConfigurations& curveConfigs,
206  const Loader& loader, const Conventions& conventions,
207  const map<string, boost::shared_ptr<YieldCurve>>& requiredYieldCurves,
208  const map<string, boost::shared_ptr<DefaultCurve>>& requiredDefaultCurves,
209  const FXTriangulation& fxTriangulation,
210  const boost::shared_ptr<ReferenceDataManager>& referenceData,
211  const IborFallbackConfig& iborFallbackConfig, const bool preserveQuoteLinkage,
212  const map<string, boost::shared_ptr<YieldCurve>>& requiredDiscountCurves)
213  : asofDate_(asof), curveSpec_(curveSpec), loader_(loader), conventions_(conventions),
214  requiredYieldCurves_(requiredYieldCurves), requiredDefaultCurves_(requiredDefaultCurves),
215  fxTriangulation_(fxTriangulation), referenceData_(referenceData), iborFallbackConfig_(iborFallbackConfig),
216  preserveQuoteLinkage_(preserveQuoteLinkage), requiredDiscountCurves_(requiredDiscountCurves) {
217 
218  try {
219 
220  curveConfig_ = curveConfigs.yieldCurveConfig(curveSpec_.curveConfigID());
221  QL_REQUIRE(curveConfig_, "No yield curve configuration found "
222  "for config ID "
223  << curveSpec_.curveConfigID());
224  currency_ = parseCurrency(curveConfig_->currency());
225 
226  /* If discount curve is not the curve being built, look for it in the map that is passed in. */
227  string discountCurveID = curveConfig_->discountCurveID();
228  if (discountCurveID != curveConfig_->curveID() && !discountCurveID.empty()) {
229  discountCurveID = yieldCurveKey(currency_, discountCurveID, asofDate_);
230  map<string, boost::shared_ptr<YieldCurve>>::iterator it;
231  it = requiredYieldCurves_.find(discountCurveID);
232  if (it != requiredYieldCurves_.end()) {
233  discountCurve_ = it->second;
234  } else {
235  QL_FAIL("The discount curve, " << discountCurveID
236  << ", required in the building "
237  "of the curve, "
238  << curveSpec_.name() << ", was not found.");
239  }
240  }
241 
242  curveSegments_ = curveConfig_->curveSegments();
243  interpolationMethod_ = parseYieldCurveInterpolationMethod(curveConfig_->interpolationMethod());
244  interpolationVariable_ = parseYieldCurveInterpolationVariable(curveConfig_->interpolationVariable());
245  zeroDayCounter_ = parseDayCounter(curveConfig_->zeroDayCounter());
246  extrapolation_ = curveConfig_->extrapolation();
247 
248  if (curveSegments_[0]->type() == YieldCurveSegment::Type::Discount) {
249  DLOG("Building DiscountCurve " << curveSpec_);
250  buildDiscountCurve();
251  } else if (curveSegments_[0]->type() == YieldCurveSegment::Type::Zero) {
252  DLOG("Building ZeroCurve " << curveSpec_);
253  buildZeroCurve();
254  } else if (curveSegments_[0]->type() == YieldCurveSegment::Type::ZeroSpread) {
255  DLOG("Building ZeroSpreadedCurve " << curveSpec_);
256  buildZeroSpreadedCurve();
257  } else if (curveSegments_[0]->type() == YieldCurveSegment::Type::DiscountRatio) {
258  DLOG("Building discount ratio yield curve " << curveSpec_);
259  buildDiscountRatioCurve();
260  } else if (curveSegments_[0]->type() == YieldCurveSegment::Type::FittedBond) {
261  DLOG("Building FittedBondCurve " << curveSpec_);
262  buildFittedBondCurve();
263  } else if (curveSegments_[0]->type() == YieldCurveSegment::Type::WeightedAverage) {
264  DLOG("Building WeightedAverageCurve " << curveSpec_);
265  buildWeightedAverageCurve();
266  } else if (curveSegments_[0]->type() == YieldCurveSegment::Type::YieldPlusDefault) {
267  DLOG("Building YieldPlusDefaultCurve " << curveSpec_);
268  buildYieldPlusDefaultCurve();
269  } else if (curveSegments_[0]->type() == YieldCurveSegment::Type::IborFallback) {
270  DLOG("Building IborFallbackCurve " << curveSpec_);
271  buildIborFallbackCurve();
272  } else {
273  DLOG("Bootstrapping YieldCurve " << curveSpec_);
274  buildBootstrappedCurve();
275  }
276 
277  h_.linkTo(p_);
278  if (extrapolation_) {
279  h_->enableExtrapolation();
280  }
281 
282  // populate shared calibration info
283 
284  if (calibrationInfo_ == nullptr)
285  calibrationInfo_ = boost::make_shared<YieldCurveCalibrationInfo>();
286  calibrationInfo_->dayCounter = zeroDayCounter_.name();
287  calibrationInfo_->currency = currency_.code();
288  if (calibrationInfo_->pillarDates.empty()) {
289  for (auto const& p : YieldCurveCalibrationInfo::defaultPeriods)
290  calibrationInfo_->pillarDates.push_back(asofDate_ + p);
291  }
292  for (auto const& d : calibrationInfo_->pillarDates) {
293  calibrationInfo_->zeroRates.push_back(p_->zeroRate(d, zeroDayCounter_, Continuous));
294  calibrationInfo_->discountFactors.push_back(p_->discount(d));
295  calibrationInfo_->times.push_back(p_->timeFromReference(d));
296  }
297 
298  } catch (QuantLib::Error& e) {
299  QL_FAIL("yield curve building failed for curve " << curveSpec_.curveConfigID() << " on date "
300  << io::iso_date(asof) << ": " << e.what());
301  } catch (std::exception& e) {
302  QL_FAIL(e.what());
303  } catch (...) {
304  QL_FAIL("unknown error");
305  }
306 
307  // force bootstrap so that errors are thrown during the build, not later
308  h_->discount(QL_EPSILON);
309 
310  LOG("Yield curve " << curveSpec_.name() << " built");
311 }

581  QL_FAIL("Interpolation variable not recognised.");
582  }

586  else {
587  // Build fixed zero/discount curve that matches the boostrapped curve
588  // initially, but does NOT react to quote changes: This is a workaround
589  // for a QuantLib problem, where a fixed reference date piecewise
590  // yield curve reacts to evaluation date changes because the bootstrap
591  // helper recompute their start date (because they are realtive date
592  // helper for deposits, fras, swaps, etc.).
593  vector<Date> dates(instruments.size() + 1, asofDate_);
594  vector<Real> zeros(instruments.size() + 1, 0.0);
595  vector<Real> discounts(instruments.size() + 1, 1.0);
596  vector<Real> forwards(instruments.size() + 1, 0.0);
597 
598  if (extrapolation_) {
599  yieldts->enableExtrapolation();
600  }
601  for (Size i = 0; i < instruments.size(); i++) {
602  dates[i + 1] = instruments[i]->pillarDate();
603  zeros[i + 1] = yieldts->zeroRate(dates[i + 1], zeroDayCounter_, Continuous);
604  discounts[i + 1] = yieldts->discount(dates[i + 1]);
605  forwards[i + 1] = yieldts->forwardRate(dates[i + 1], dates[i + 1], zeroDayCounter_, Continuous);
606  }
607  zeros[0] = zeros[1];
608  forwards[0] = forwards[1];
609  if (interpolationVariable_ == InterpolationVariable::Zero)
610  p_ = zerocurve(dates, zeros, zeroDayCounter_, interpolationMethod_);
611  else if (interpolationVariable_ == InterpolationVariable::Discount)
612  p_ = discountcurve(dates, discounts, zeroDayCounter_, interpolationMethod_);
613  else if (interpolationVariable_ == InterpolationVariable::Forward)
614  p_ = forwardcurve(dates, forwards, zeroDayCounter_, interpolationMethod_);
615  else
616  QL_FAIL("Interpolation variable not recognised.");
617  }
618 
619  // set calibration info
620  calibrationInfo_ = boost::make_shared<PiecewiseYieldCurveCalibrationInfo>();
621  for (Size i = 0; i < instruments.size(); ++i) {
622  calibrationInfo_->pillarDates.push_back(instruments[i]->pillarDate());
623  }
624 
625  return p_;
626 }

628 void YieldCurve::buildZeroCurve() {
629 
630  QL_REQUIRE(curveSegments_.size() <= 1, "More than one zero curve "
631  "segment not supported yet.");
632  QL_REQUIRE(curveSegments_[0]->type() == YieldCurveSegment::Type::Zero, "The curve segment is not of type Zero.");
633 
634  // Fill a vector of zero quotes.
635  vector<boost::shared_ptr<ZeroQuote>> zeroQuotes;
636  boost::shared_ptr<DirectYieldCurveSegment> zeroCurveSegment =
637  boost::dynamic_pointer_cast<DirectYieldCurveSegment>(curveSegments_[0]);
638  auto zeroQuoteIDs = zeroCurveSegment->quotes();
639 
640  for (Size i = 0; i < zeroQuoteIDs.size(); ++i) {
641  boost::shared_ptr<MarketDatum> marketQuote = loader_.get(zeroQuoteIDs[i], asofDate_);
642  if (marketQuote) {
643  QL_REQUIRE(marketQuote->instrumentType() == MarketDatum::InstrumentType::ZERO,
644  "Market quote not of type zero.");
645  boost::shared_ptr<ZeroQuote> zeroQuote = boost::dynamic_pointer_cast<ZeroQuote>(marketQuote);
646  zeroQuotes.push_back(zeroQuote);
647  }
648  }
649 
650  // Create the (date, zero) pairs.
651  map<Date, Rate> data;
652  boost::shared_ptr<Convention> convention = conventions_.get(curveSegments_[0]->conventionsID());
653  QL_REQUIRE(convention, "No conventions found with ID: " << curveSegments_[0]->conventionsID());
654  QL_REQUIRE(convention->type() == Convention::Type::Zero, "Conventions ID does not give zero rate conventions.");
655  boost::shared_ptr<ZeroRateConvention> zeroConvention = boost::dynamic_pointer_cast<ZeroRateConvention>(convention);
656  DayCounter quoteDayCounter = zeroConvention->dayCounter();
657  for (Size i = 0; i < zeroQuotes.size(); ++i) {
658  QL_REQUIRE(quoteDayCounter == zeroQuotes[i]->dayCounter(),
659  "The day counter should be the same between the conventions"
660  "and the quote.");
661  if (!zeroQuotes[i]->tenorBased()) {
662  data[zeroQuotes[i]->date()] = zeroQuotes[i]->quote()->value();
663  } else {
664  QL_REQUIRE(zeroConvention->tenorBased(), "Using tenor based "
665  "zero rates without tenor based zero rate conventions.");
666  Date zeroDate = asofDate_;
667  if (zeroConvention->spotLag() > 0) {
668  zeroDate = zeroConvention->spotCalendar().advance(zeroDate, zeroConvention->spotLag() * Days);
669  }
670  zeroDate = zeroConvention->tenorCalendar().advance(zeroDate, zeroQuotes[i]->tenor(),
671  zeroConvention->rollConvention(), zeroConvention->eom());
672  data[zeroDate] = zeroQuotes[i]->quote()->value();
673  }
674  }
675 
676  QL_REQUIRE(data.size() > 0, "No market data found for curve spec " << curveSpec_.name() << " with as of date "
677  << io::iso_date(asofDate_));
678 
679  // \todo review - more flexible (flat vs. linear extrap)?
680  if (data.begin()->first > asofDate_) {
681  Rate rate = data.begin()->second;
682  data[asofDate_] = rate;
683  LOG("Insert zero curve point at time zero for " << curveSpec_.name() << ": "
684  << "date " << QuantLib::io::iso_date(asofDate_) << ", "
685  << "zero " << fixed << setprecision(4) << data[asofDate_]);
686  }
687 
688  QL_REQUIRE(data.size() > 1, "The single zero rate quote provided "
689  "should be associated with a date greater than as of date.");
690 
691  // First build temporary curves
692  vector<Date> dates;
693  vector<Rate> zeroes, discounts;
694  dates.push_back(data.begin()->first);
695  zeroes.push_back(data.begin()->second);
696  discounts.push_back(1.0);
697 
698  Compounding zeroCompounding = zeroConvention->compounding();
699  Frequency zeroCompoundingFreq = zeroConvention->compoundingFrequency();
700  map<Date, Rate>::iterator it;
701  for (it = ++data.begin(); it != data.end(); ++it) {
702  dates.push_back(it->first);
703  InterestRate tempRate(it->second, quoteDayCounter, zeroCompounding, zeroCompoundingFreq);
704  Time t = quoteDayCounter.yearFraction(asofDate_, it->first);
705  /* Convert zero rate to continuously compounded if necessary */
706  if (zeroCompounding == Continuous) {
707  zeroes.push_back(it->second);
708  } else {
709  zeroes.push_back(tempRate.equivalentRate(Continuous, Annual, t));
710  }
711  discounts.push_back(tempRate.discountFactor(t));
712  LOG("Add zero curve point for " << curveSpec_.name() << ": " << io::iso_date(dates.back()) << " " << fixed
713  << setprecision(4) << zeroes.back() << " / " << discounts.back());
714  }
715 
716  QL_REQUIRE(dates.size() == zeroes.size(), "Date and zero vectors differ in size.");
717  QL_REQUIRE(dates.size() == discounts.size(), "Date and discount vectors differ in size.");
718 
719  // Now build curve with requested conventions
720  if (interpolationVariable_ == YieldCurve::InterpolationVariable::Zero) {
721  boost::shared_ptr<YieldTermStructure> tempCurve =
722  zerocurve(dates, zeroes, quoteDayCounter, interpolationMethod_);
723  zeroes.clear();
724  for (Size i = 0; i < dates.size(); ++i) {
725  Rate zero = tempCurve->zeroRate(dates[i], zeroDayCounter_, Continuous);
726  zeroes.push_back(zero);
727  }
728  p_ = zerocurve(dates, zeroes, zeroDayCounter_, interpolationMethod_);
729  } else if (interpolationVariable_ == YieldCurve::InterpolationVariable::Discount) {
730  boost::shared_ptr<YieldTermStructure> tempCurve =
731  discountcurve(dates, discounts, quoteDayCounter, interpolationMethod_);
732  discounts.clear();
733  for (Size i = 0; i < dates.size(); ++i) {
734  DiscountFactor discount = tempCurve->discount(dates[i]);
735  discounts.push_back(discount);
736  }
737  p_ = discountcurve(dates, discounts, zeroDayCounter_, interpolationMethod_);
738  } else {
739  QL_FAIL("Unknown yield curve interpolation variable.");
740  }
741 }

1149 #else

1151  switch (interpolationMethod_) {

1154  Array());

1160  break;

1163  calInfo->fittingMethod = "Svensson";

1171  Real minError = QL_MAX_REAL;
1172  HaltonRsg halton(method->size(), 42);
1173  // TODO randomised optimisation seeds are only implemented for NelsonSiegel so far
1174  Size trials = 1;
1175  if (interpolationMethod_ == InterpolationMethod::NelsonSiegel) {
1176  trials = curveConfig_->bootstrapConfig().maxAttempts();
1177  } else {
1178  if (curveConfig_->bootstrapConfig().maxAttempts() > 1) {
1179  WLOG("randomised optimisation seeds not implemented for given interpolation method");
1180  }
1181  }
1182  for (Size i = 0; i < trials; ++i) {
1183  Array guess;
1184  // first guess is the default guess (empty array, will be set to a zero vector in
1185  // FittedBondDiscountCurve::calculate())
1186  if (i == 0) {
1187  if (interpolationMethod_ == InterpolationMethod::NelsonSiegel) {
1188  // first guess will be based on the last bond yield and first bond yield
1189  guess = Array(4);
1190  Integer maxMaturity = static_cast<Integer>(

1192  std::max_element(securityMaturityDates.begin(), securityMaturityDates.end())));

1196  guess[0] = marketYields[maxMaturity]; // long term yield
1197  guess[1] = marketYields[minMaturity] - guess[0]; // short term component
1198  guess[2] = 0.0;
1199  guess[3] = 5.0;
1200  DLOG("using smart NelsonSiegel guess for trial #" << (i + 1) << ": " << guess);
1201  }
1202  } else {
1203  auto seq = halton.nextSequence();
1204  guess = Array(seq.value.begin(), seq.value.end());
1205  if (interpolationMethod_ == InterpolationMethod::NelsonSiegel) {
1206  guess[0] = guess[0] * 0.10 - 0.05; // long term yield
1207  guess[1] = guess[1] * 0.10 - 0.05; // short term component
1208  guess[2] = guess[2] * 0.10 - 0.05; // medium term component
1209  guess[3] = guess[3] * 5.0; // decay factor
1210  DLOG("using random NelsonSiegel guess for trial #" << (i + 1) << ": " << guess);
1211  } else {
1212  QL_FAIL("randomised optimisation seed not implemented");
1213  }
1214  }
1215  current = boost::make_shared<FittedBondDiscountCurve>(asofDate_, helpers, zeroDayCounter_, *method, 1.0e-10,
1216  10000, guess);
1217  Real cost = std::sqrt(current->fitResults().minimumCostValue());
1218  if (cost < minError) {
1219  minError = cost;
1220  tmp = current;
1221  }
1222  DLOG("calibration trial #" << (i + 1) << " out of " << trials << ": cost = " << cost
1223  << ", best so far = " << minError);
1224  if (cost < curveConfig_->bootstrapConfig().accuracy()) {
1225  DLOG("reached desired accuracy (" << curveConfig_->bootstrapConfig().accuracy()
1226  << ") - do not attempt more calibrations");
1227  break;
1228  }
1229  }
1230  QL_REQUIRE(tmp, "no best solution found for fitted bond curve - this is unexpected.");
1231 
1232  if (Norm2(tmp->fitResults().solution()) < 1.0e-4) {
1233  WLOG("Fit solution is close to 0. The curve fitting should be reviewed.");
1234  }
1235 
1236  DLOG("Fitted Bond Curve Summary:");
1237  DLOG(" solution: " << tmp->fitResults().solution());
1238  DLOG(" iterations: " << tmp->fitResults().numberOfIterations());
1239  DLOG(" cost value: " << minError);
1240 
1241  std::vector<Real> modelPrices, modelYields;
1242  auto engine = boost::make_shared<DiscountingBondEngine>(Handle<YieldTermStructure>(tmp));
1243  for (Size i = 0; i < bonds.size(); ++i) {
1244  bonds[i]->setPricingEngine(engine);
1245  modelPrices.push_back(bonds[i]->cleanPrice() / 100.0);
1246  modelYields.push_back(bonds[i]->yield(bonds[i]->cleanPrice(), ActualActual(), Continuous, NoFrequency));
1247  DLOG("bond " << securityIDs[i] << ", model clean price = " << modelPrices.back()
1248  << ", yield (cont,actact) = " << modelYields.back() << ", NPV = " << bonds[i]->NPV());
1249  }
1250 
1251  Real tolerance = curveConfig_->bootstrapConfig().globalAccuracy() == Null<Real>()
1252  ? curveConfig_->bootstrapConfig().accuracy()
1253  : curveConfig_->bootstrapConfig().globalAccuracy();
1254  QL_REQUIRE(curveConfig_->bootstrapConfig().dontThrow() || minError < tolerance,
1255  "Fitted Bond Curve cost value (" << minError << ") exceeds tolerance (" << tolerance << ")");
1256 
1257  if (extrapolation_)
1258  tmp->enableExtrapolation();
1259 
1260  p_ = tmp;
1261 
1262  Array solution = tmp->fitResults().solution();
1263  calInfo->modelPrices = modelPrices;
1264  calInfo->modelYields = modelYields;
1265  calInfo->tolerance = tolerance;
1266  calInfo->costValue = minError;
1267  calInfo->solution = std::vector<double>(solution.begin(), solution.end());
1268  calInfo->iterations = static_cast<int>(tmp->fitResults().numberOfIterations());
1269  calibrationInfo_ = calInfo;
