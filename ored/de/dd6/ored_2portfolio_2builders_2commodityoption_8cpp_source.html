34 boost::shared_ptr<QuantLib::PricingEngine> CommodityOptionEngineBuilder::engineImpl(const string& commodityName,
35  const Currency& ccy) {
36 
37  // Create the commodity yield curve for the process
38  Handle<Quote> commoditySpot = market_->commoditySpot(commodityName, configuration(MarketContext::pricing));
39  Handle<PriceTermStructure> priceCurve =
40  market_->commodityPriceCurve(commodityName, configuration(MarketContext::pricing));
41  Handle<YieldTermStructure> discountCurve =
42  market_->discountCurve(ccy.code(), configuration(MarketContext::pricing));
43 
44  Handle<YieldTermStructure> yield = Handle<YieldTermStructure>(
45  boost::make_shared<PriceTermStructureAdapter>(*commoditySpot, *priceCurve, *discountCurve));
46 
47  // Create the option engine
48  boost::shared_ptr<GeneralizedBlackScholesProcess> process = boost::make_shared<GeneralizedBlackScholesProcess>(
49  commoditySpot, yield, discountCurve,
50  market_->commodityVolatility(commodityName, configuration(MarketContext::pricing)));
51 
52  return boost::make_shared<QuantLib::AnalyticEuropeanEngine>(process, discountCurve);
53 }
