79 void CreditDefaultSwapOption::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
80 
81  DLOG("CreditDefaultSwapOption::build() called for trade " << id());
82 
83  // Notionals
84  const auto& legData = swap_.leg();
85  const auto& ntls = legData.notionals();
86  QL_REQUIRE(ntls.size() == 1, "CreditDefaultSwapOption requires a single notional.");

90  // Type of instrument we build depends on whether the reference entity has already defaulted.

181 void CreditDefaultSwapOption::buildNoDefault(const boost::shared_ptr<EngineFactory>& engineFactory) {
182 
183  DLOG("CreditDefaultSwapOption: building CDS option trade " << id() << " given no default.");
184 
185  // Need fixed leg data with one rate. This should be the standard running coupon on the CDS e.g.
186  // generally 100bp for IG CDS and 500bp for HY CDS. For single name CDS options, one can use this field to give
187  // the strike spread. It may matter for the resulting valuation depending on the engine that is used - see
188  // "A CDS Option Miscellany, Richard J. Martin, 2019, Section 2.4".
189  const auto& legData = swap_.leg();
190  QL_REQUIRE(legData.legType() == "Fixed", "CDS option " << id() << " requires fixed leg.");
191  auto fixedLegData = boost::dynamic_pointer_cast<FixedLegData>(legData.concreteLegData());
192  QL_REQUIRE(fixedLegData->rates().size() == 1, "Index CDS option " << id() << " requires single fixed rate.");
193  auto runningCoupon = fixedLegData->rates().front();
194 
195  // Payer (Receiver) swaption if the leg is paying (receiving).
196  auto side = legData.isPayer() ? Protection::Side::Buyer : Protection::Side::Seller;
197 
198  // Day counter. In general for CDS, the standard day counter is Actual/360 and the final
199  // period coupon accrual includes the maturity date.
200  Actual360 standardDayCounter;
201  const std::string& strDc = legData.dayCounter();
202  DayCounter dc = strDc.empty() ? standardDayCounter : parseDayCounter(strDc);
203  DayCounter lastPeriodDayCounter = dc == standardDayCounter ? Actual360(true) : dc;
204 
205  // Schedule
206  Schedule schedule = makeSchedule(legData.schedule());
207  BusinessDayConvention payConvention = legData.paymentConvention().empty() ? Following :
208  parseBusinessDayConvention(legData.paymentConvention());
209 
210  // Don't support upfront fee on the underlying CDS for the moment.
211  QL_REQUIRE(swap_.upfrontFee() == Null<Real>() || close(swap_.upfrontFee(), 0.0),
212  "Upfront fee on the CDS underlying a CDS option is not supported.");
213 
214  // The underlying CDS trade
215  auto cds = boost::make_shared<QuantExt::CreditDefaultSwap>(side, notional_, runningCoupon, schedule,
216  payConvention, dc, swap_.settlesAccrual(), swap_.protectionPaymentTime(), swap_.protectionStart(),
217  boost::shared_ptr<Claim>(), lastPeriodDayCounter, swap_.tradeDate(), swap_.cashSettlementDays());
218 
219  // Set engine on the underlying CDS.
220  auto cdsBuilder = boost::dynamic_pointer_cast<CreditDefaultSwapEngineBuilder>(
221  engineFactory->builder("CreditDefaultSwap"));
222  QL_REQUIRE(cdsBuilder, "CreditDefaultSwapOption expected CDS engine " <<
223  " builder for underlying while building trade " << id() << ".");
224  npvCurrency_ = legData.currency();
225  auto ccy = parseCurrency(npvCurrency_);
226  cds->setPricingEngine(cdsBuilder->engine(ccy, swap_.creditCurveId(), swap_.recoveryRate()));
227 
228  // Check option data
229  QL_REQUIRE(option_.style() == "European", "CreditDefaultSwapOption option style must" <<
230  " be European but got " << option_.style() << ".");
231  QL_REQUIRE(!option_.payoffAtExpiry(), "CreditDefaultSwapOption payoff must be at exercise.");
232  QL_REQUIRE(option_.exerciseFees().empty(), "CreditDefaultSwapOption cannot handle exercise fees.");
233 
234  // Exercise must be European
235  const auto& exerciseDates = option_.exerciseDates();
236  QL_REQUIRE(exerciseDates.size() == 1, "CreditDefaultSwapOption expects one exercise date" <<
237  " but got " << exerciseDates.size() << " exercise dates.");
238  Date exerciseDate = parseDate(exerciseDates.front());
239  boost::shared_ptr<Exercise> exercise = boost::make_shared<EuropeanExercise>(exerciseDate);
240 
241  // Limit strike type to Spread for now.
242  auto strikeType = parseCdsOptionStrikeType(strikeType_);
243  QL_REQUIRE(strikeType == QuantExt::CdsOption::Spread, "CreditDefaultSwapOption strike type must be Spread.");
244 
245  // If the strike_ is Null, the strike is taken as the running coupon.
246  Real strike = strike_ == Null<Real>() ? runningCoupon : strike_;
247 
248  // Build the option instrument
249  auto cdsOption = boost::make_shared<QuantExt::CdsOption>(cds, exercise, knockOut_, strike, strikeType);
250 
251  // Set the option engine
252  auto cdsOptionEngineBuilder = boost::dynamic_pointer_cast<CreditDefaultSwapOptionEngineBuilder>(
253  engineFactory->builder("CreditDefaultSwapOption"));
254  QL_REQUIRE(cdsOptionEngineBuilder, "CreditDefaultSwapOption expected CDS option engine " <<
255  " builder for underlying while building trade " << id() << ".");
256  cdsOption->setPricingEngine(cdsOptionEngineBuilder->engine(ccy, swap_.creditCurveId(), term_));
257 
258  // Copying here what is done for the index CDS option. The comment there is:
259  // Align option product maturities with ISDA AANA/GRID guidance as of November 2020.
260  maturity_ = cds->coupons().back()->date();
261 
262  // Set Trade members.
263  legs_ = { cds->coupons() };
264  legCurrencies_ = { npvCurrency_ };
265  legPayers_ = { legData.isPayer() };
266 
267  // Include premium if enough information is provided
268  vector<boost::shared_ptr<Instrument>> additionalInstruments;
269  vector<Real> additionalMultipliers;
270  string marketConfig = cdsOptionEngineBuilder->configuration(MarketContext::pricing);
271  addPremium(engineFactory, ccy, marketConfig, additionalInstruments, additionalMultipliers);
272 
273  // Instrument wrapper depends on the settlement type.
274  Position::Type positionType = parsePositionType(option_.longShort());
275  Settlement::Type settleType = parseSettlementType(option_.settlement());
276  if (settleType == Settlement::Cash) {
277  Real indicatorLongShort = positionType == Position::Long ? 1.0 : -1.0;
278  instrument_ = boost::make_shared<VanillaInstrument>(cdsOption, indicatorLongShort,
279  additionalInstruments, additionalMultipliers);
280  } else {
281  bool isLong = positionType == Position::Long;
282  bool isPhysical = settleType == Settlement::Physical;
283  instrument_ = boost::make_shared<EuropeanOptionWrapper>(cdsOption, isLong, exerciseDate,
284  isPhysical, cds, 1.0, 1.0, additionalInstruments, additionalMultipliers);
285  }
286 
287 }

289 void CreditDefaultSwapOption::buildDefaulted(const boost::shared_ptr<EngineFactory>& engineFactory) {
290 
291  DLOG("CreditDefaultSwapOption: building CDS option trade " << id() << " given default occured.");
292 
293  // We add a simple payment for CDS options where the reference entity has already defaulted.
294  // If it is a knock-out CDS option, we add a dummy payment of 0.0 with date today instead of throwing.
295  const auto& legData = swap_.leg();
296  Date paymentDate = engineFactory->market()->asofDate();
297  Real amount = 0.0;
298  if (!knockOut_) {
299  paymentDate = asi_->auctionSettlementDate();
300  amount = notional_ * (1 - asi_->auctionFinalPrice());
301  // If it is a receiver option, i.e. selling protection, the FEP is paid out.
302  if (!legData.isPayer())
303  amount *= -1.0;
304  }
305 
306  // Use the add premiums method to add the payment.
307  string marketConfig = Market::defaultConfiguration;
308  auto ccy = parseCurrency(notionalCurrency_);
309  vector<boost::shared_ptr<Instrument>> additionalInstruments;
310  vector<Real> additionalMultipliers;
311  addPremiums(additionalInstruments, additionalMultipliers, 1.0, PremiumData(amount, notionalCurrency_, paymentDate),
312  1.0, ccy, engineFactory, marketConfig);
313  DLOG("FEP payment (date = " << paymentDate << ", amount = " << amount << ") added for CDS option " << id() << ".");
314 
315  // Use the instrument added as the main instrument and clear the vectors
316  auto qlInst = additionalInstruments.back();
317  QL_REQUIRE(qlInst, "Expected a FEP payment to have been added for CDS option " << id() << ".");
318  maturity_ = paymentDate;
319  additionalInstruments.clear();
320  additionalMultipliers.clear();
321 
322  // Include premium if enough information is provided
323  addPremium(engineFactory, ccy, marketConfig, additionalInstruments, additionalMultipliers);
324 
325  // Instrument wrapper.
326  Position::Type positionType = parsePositionType(option_.longShort());
327  Real indicatorLongShort = positionType == Position::Long ? 1.0 : -1.0;
328  instrument_ = boost::make_shared<VanillaInstrument>(qlInst, indicatorLongShort,
329  additionalInstruments, additionalMultipliers);
330 }

332 void CreditDefaultSwapOption::addPremium(const boost::shared_ptr<EngineFactory>& ef,
333  const Currency& tradeCurrency,
334  const string& marketConfig,
335  vector<boost::shared_ptr<Instrument>>& additionalInstruments,
336  vector<Real>& additionalMultipliers) {
337  // The premium amount is always provided as a non-negative amount. Assign the correct sign here i.e.
338  // pay the premium if long the option and recieve the premium if short the option.
339  Position::Type positionType = parsePositionType(option_.longShort());
340  Real indicatorLongShort = positionType == Position::Long ? 1.0 : -1.0;
341  addPremiums(additionalInstruments, additionalMultipliers, 1.0, option_.premiumData(), indicatorLongShort,
342  tradeCurrency, ef, marketConfig);
343 }
