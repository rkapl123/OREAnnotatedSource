35 static CurveSpec::CurveType parseCurveSpecType(const string& s) {
36  static map<string, CurveSpec::CurveType> b = {
37  {"Yield", CurveSpec::CurveType::Yield},
38  {"CapFloorVolatility", CurveSpec::CurveType::CapFloorVolatility},
39  {"SwaptionVolatility", CurveSpec::CurveType::SwaptionVolatility},
40  {"YieldVolatility", CurveSpec::CurveType::YieldVolatility},
41  {"FX", CurveSpec::CurveType::FX},
42  {"FXVolatility", CurveSpec::CurveType::FXVolatility},
43  {"Default", CurveSpec::CurveType::Default},
44  {"CDSVolatility", CurveSpec::CurveType::CDSVolatility},
45  {"BaseCorrelation", CurveSpec::CurveType::BaseCorrelation},
46  {"Inflation", CurveSpec::CurveType::Inflation},
47  {"InflationCapFloorVolatility", CurveSpec::CurveType::InflationCapFloorVolatility},
48  {"Equity", CurveSpec::CurveType::Equity},
49  {"EquityVolatility", CurveSpec::CurveType::EquityVolatility},
50  {"Security", CurveSpec::CurveType::Security},
51  {"Commodity", CurveSpec::CurveType::Commodity},
52  {"Correlation", CurveSpec::CurveType::Correlation},
53  {"CommodityVolatility", CurveSpec::CurveType::CommodityVolatility}};
54 
55  auto it = b.find(s);
56  if (it != b.end()) {
57  return it->second;
58  } else {
59  QL_FAIL("Cannot convert \"" << s << "\" to CurveSpecType");
60  }
61 }

64 boost::shared_ptr<CurveSpec> parseCurveSpec(const string& s) {
65 
66 
67  boost::escaped_list_separator<char> sep('\\', '/', '\"');
68  boost::tokenizer<boost::escaped_list_separator<char> > tokenSplit(s, sep);
69 
70  vector<string> tokens(tokenSplit.begin(), tokenSplit.end());
71 
72  QL_REQUIRE(tokens.size() > 1, "number of tokens too small in curve spec " << s);
73 
74  CurveSpec::CurveType curveType = parseCurveSpecType(tokens[0]);
75 
76  switch (curveType) {
77 
78  case CurveSpec::CurveType::Yield: {
79  // Expected format: Yield/CCY/CurveConfigID
80  // Example: Yield/EUR/eur-6M-curve
81  QL_REQUIRE(tokens.size() == 3, "Unexpected number"
82  " of tokens in yield curve spec "
83  << s);
84  const string& ccy = tokens[1];
85  const string& curveConfigID = tokens[2];
86  return boost::make_shared<YieldCurveSpec>(ccy, curveConfigID);
87  }
88 
89  case CurveSpec::CurveType::Default: {
90  // Default/USD/CurveConfigID
91  QL_REQUIRE(tokens.size() == 3, "Unexpected number"
92  " of tokens in default curve spec "
93  << s);
94  const string& ccy = tokens[1];
95  const string& curveConfigID = tokens[2];
96  return boost::make_shared<DefaultCurveSpec>(ccy, curveConfigID);
97  }
98 
99  case CurveSpec::CurveType::CDSVolatility: {
100  // CDSVolatility/CurveConfigID
101  QL_REQUIRE(tokens.size() == 2, "Unexpected number"
102  " of tokens in cds vol spec "
103  << s);
104  const string& curveConfigID = tokens[1];
105  return boost::make_shared<CDSVolatilityCurveSpec>(curveConfigID);
106  }
107 
108  case CurveSpec::CurveType::BaseCorrelation: {
109  // BaseCorrelation/CurveConfigID
110  QL_REQUIRE(tokens.size() == 2, "Unexpected number"
111  " of tokens in cds vol spec "
112  << s);
113  const string& curveConfigID = tokens[1];
114  return boost::make_shared<BaseCorrelationCurveSpec>(curveConfigID);
115  }
116 
117  case CurveSpec::CurveType::FX: {
118  // FX/USD/CHF
119  QL_REQUIRE(tokens.size() == 3, "Unexpected number"
120  " of tokens in FX curve spec "
121  << s);
122  const string& unitCcy = tokens[1];
123  const string& ccy = tokens[2];
124  return boost::make_shared<FXSpotSpec>(unitCcy, ccy);
125  }
126 
127  case CurveSpec::CurveType::FXVolatility: {
128  // FX/USD/CHF/CurveConfigID
129  QL_REQUIRE(tokens.size() == 4, "Unexpected number"
130  " of tokens in fx vol curve spec "
131  << s);
132  const string& unitCcy = tokens[1];
133  const string& ccy = tokens[2];
134  const string& curveConfigID = tokens[3];
135  return boost::make_shared<FXVolatilityCurveSpec>(unitCcy, ccy, curveConfigID);
136  }
137 
138  case CurveSpec::CurveType::SwaptionVolatility: {
139  // SwaptionVolatility/EUR/CurveConfigID
140  QL_REQUIRE(tokens.size() == 3, "Unexpected number"
141  " of tokens in swaption vol curve spec "
142  << s);
143  const string& ccy = tokens[1];
144  const string& curveConfigID = tokens[2];
145  return boost::make_shared<SwaptionVolatilityCurveSpec>(ccy, curveConfigID);
146  }
147 
148  case CurveSpec::CurveType::YieldVolatility: {
149  // YieldVolatility/CurveConfigID
150  QL_REQUIRE(tokens.size() == 2, "Unexpected number"
151  " of tokens in yield vol curve spec "
152  << s);
153  const string& curveConfigID = tokens[1];
154  return boost::make_shared<YieldVolatilityCurveSpec>(curveConfigID);
155  }
156 
157  case CurveSpec::CurveType::CapFloorVolatility: {
158  // e.g. CapFloorVolatility/EUR/CurveConfigID
159  QL_REQUIRE(tokens.size() == 3, "Unexpected number"
160  " of tokens in CapFloor volatility curve spec "
161  << s);
162  const string& ccy = tokens[1];
163  const string& curveConfigID = tokens[2];
164  return boost::make_shared<CapFloorVolatilityCurveSpec>(ccy, curveConfigID);
165  }
166 
167  case CurveSpec::CurveType::Inflation: {
168  // Inflation/EUHICPXT/CurveConfigID
169  QL_REQUIRE(tokens.size() == 3, "Unexpected number"
170  " of tokens in inflation curve spec "
171  << s);
172  const string& index = tokens[1];
173  const string& curveConfigID = tokens[2];
174  return boost::make_shared<InflationCurveSpec>(index, curveConfigID);
175  }
176 
177  case CurveSpec::CurveType::InflationCapFloorVolatility: {
178  // e.g. InflationCapFloorVolatility/EUHICPXT/CurveConfigID
179  QL_REQUIRE(tokens.size() == 3, "Unexpected number"
180  " of tokens in InflationCapFloor volatility curve spec "
181  << s);
182  const string& index = tokens[1];
183  const string& curveConfigID = tokens[2];
184  return boost::make_shared<InflationCapFloorVolatilityCurveSpec>(index, curveConfigID);
185  }
186 
187  case CurveSpec::CurveType::Equity: {
188  // Equity/USD/CurveConfigID
189  QL_REQUIRE(tokens.size() == 3, "Unexpected number"
190  " of tokens in default curve spec "
191  << s);
192  const string& ccy = tokens[1];
193  const string& curveConfigID = tokens[2];
194  return boost::make_shared<EquityCurveSpec>(ccy, curveConfigID);
195  }
196 
197  case CurveSpec::CurveType::EquityVolatility: {
198  // EquityVolatility/USD/CurveConfigID
199  QL_REQUIRE(tokens.size() == 3, "Unexpected number"
200  " of tokens in default curve spec "
201  << s);
202  const string& ccy = tokens[1];
203  const string& curveConfigID = tokens[2];
204  return boost::make_shared<EquityVolatilityCurveSpec>(ccy, curveConfigID);
205  }
206 
207  case CurveSpec::CurveType::Security: {
208  // Security/ISIN
209  QL_REQUIRE(tokens.size() == 2, "Unexpected number"
210  " of tokens in Security Spread spec "
211  << s);
212  const string& securityID = tokens[1];
213  return boost::make_shared<SecuritySpec>(securityID);
214  }
215 
216  case CurveSpec::CurveType::Commodity: {
217  // Commodity/CCY/CommodityCurveConfigId
218  QL_REQUIRE(tokens.size() == 3, "Unexpected number of tokens in commodity curve spec " << s);
219  return boost::make_shared<CommodityCurveSpec>(tokens[1], tokens[2]);
220  }
221 
222  case CurveSpec::CurveType::CommodityVolatility: {
223  // CommodityVolatility/CCY/CommodityVolatilityConfigId
224  QL_REQUIRE(tokens.size() == 3, "Unexpected number of tokens in commodity volatility spec " << s);
225  return boost::make_shared<CommodityVolatilityCurveSpec>(tokens[1], tokens[2]);
226  }
227 
228  case CurveSpec::CurveType::Correlation: {
229  // Correlation/CorrelationCurveConfigId
230  QL_REQUIRE(tokens.size() == 2, "Unexpected number of tokens in correlatin spec " << s);
231  string id = tokens[1];
232  return boost::make_shared<CorrelationCurveSpec>(id);
233  }
234 
235  // TODO: the rest...
236  }
237 
238  QL_FAIL("Unable to convert \"" << s << "\" into CurveSpec");
239 }
