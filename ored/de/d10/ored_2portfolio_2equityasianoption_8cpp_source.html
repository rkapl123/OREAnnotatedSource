28  void EquityAsianOption::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
29  // Checks
30  QL_REQUIRE(quantity_ > 0, "Equity Asian option requires a positive quatity");
31  QL_REQUIRE(strike_ > 0, "Equity Asian option requires a positive strike");
32 
33  // Set the assetName_ as it may have changed after lookup
34  assetName_ = equityName();
35 
36  // Populate the index_ in case the option is automatic exercise
37  const boost::shared_ptr<Market>& market = engineFactory->market();
38  index_ = *market->equityCurve(assetName_, engineFactory->configuration(MarketContext::pricing));
39 
40  // Build the trade using shared functionality in the base class
41  AsianOptionTrade::build(engineFactory);
42  }

48  void EquityAsianOption::fromXML(XMLNode* node) {
49  AsianOptionTrade::fromXML(node);
50  XMLNode* eqNode = XMLUtils::getChildNode(node, "EquityAsianOptionData");
51  QL_REQUIRE(eqNode, "No EquityAsianOptionData node");
52 
53  option_.fromXML(XMLUtils::getChildNode(eqNode, "OptionData"));
54  QL_REQUIRE(option_.payoffType() == "Asian", "Expected PayoffType Asian for EquityAsianOption.");
55  XMLNode* asianNode = XMLUtils::getChildNode(eqNode, "AsianData");
56  if (asianNode)
57  asianData_.fromXML(asianNode);
58 
59  XMLNode* scheduleDataNode = XMLUtils::getChildNode(eqNode, "ObservationDates");
60  observationDates_.fromXML(scheduleDataNode);
61 
62  XMLNode* tmp = XMLUtils::getChildNode(eqNode, "Underlying");
63  if (!tmp)
64  tmp = XMLUtils::getChildNode(eqNode, "Name");
65  equityUnderlying_.fromXML(tmp);
66 
67  currency_ = XMLUtils::getChildValue(eqNode, "Currency", true);
68  // Require explicit Strike
69  strike_ = XMLUtils::getChildValueAsDouble(eqNode, "Strike", true);
70  quantity_ = XMLUtils::getChildValueAsDouble(eqNode, "Quantity", true);
71  }

88
