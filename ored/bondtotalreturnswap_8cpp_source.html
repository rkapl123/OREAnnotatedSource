49void BondTRS::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
50 DLOG("BondTRS::build() called for trade " << id());
51
52 const boost::shared_ptr<Market> market = engineFactory->market();
53 boost::shared_ptr<EngineBuilder> builder_trs = engineFactory->builder("BondTRS");
54 bondData_ = originalBondData_;
55 bondData_.populateFromBondReferenceData(engineFactory->referenceData());
56
57 Schedule schedule = makeSchedule(scheduleData_);
58 Calendar calendar = parseCalendar(bondData_.calendar());
59
60 auto configuration = builder_trs->configuration(MarketContext::pricing);
61 auto legBuilder = engineFactory->legBuilder(fundingLegData_.legType());
62
63 // check currency restrictions
64
65 QL_REQUIRE(fundingLegData_.currency() != bondData_.currency() || fxIndex_.empty(),
66 "if funding leg ccy (" << fundingLegData_.currency() << ") = bond ccy (" << bondData_.currency()
67 << "), no fx index must be given");
68 QL_REQUIRE(fundingLegData_.currency() == bondData_.currency() || !fxIndex_.empty(),
69 "if funding leg ccy (" << fundingLegData_.currency() << ") != bond ccy (" << bondData_.currency()
70 << "), a fx index must be given");
71
72 // build return leg valuation and payment schedule
73 DLOG("build valuation and payment dates vectors");
74
75 Period observationLag = observationLag_.empty() ? 0 * Days : parsePeriod(observationLag_);
76 Calendar observationCalendar = parseCalendar(observationCalendar_);
77 BusinessDayConvention observationConvention =
78 observationConvention_.empty() ? Unadjusted : parseBusinessDayConvention(observationConvention_);
79
80 PaymentLag paymentLag = parsePaymentLag(paymentLag_);
81 Period plPeriod = boost::apply_visitor(PaymentLagPeriod(), paymentLag);
82 Calendar paymentCalendar = parseCalendar(paymentCalendar_);
83 BusinessDayConvention paymentConvention =
84 paymentConvention_.empty() ? Unadjusted : parseBusinessDayConvention(paymentConvention_);
85
86 std::vector<Date> valuationDates, paymentDates;
87
88 for (auto const& d : schedule.dates()) {
89 valuationDates.push_back(observationCalendar.advance(d, -observationLag, observationConvention));
90 if (d != schedule.dates().front())
91 paymentDates.push_back(paymentCalendar.advance(d, plPeriod, paymentConvention));
92 }
93
94 if (!paymentDates_.empty()) {
95 paymentDates.clear();
96 QL_REQUIRE(paymentDates_.size() + 1 == valuationDates.size(),
97 "paymentDates size (" << paymentDates_.size() << ") does no match valuatioDates size ("
98 << valuationDates.size() << ") minus 1");
99 for (auto const& s : paymentDates_)
100 paymentDates.push_back(parseDate(s));
101 }
102
103 DLOG("valuation schedule:");
104 for (auto const& d : valuationDates)
105 DLOG(ore::data::to_string(d));
106
107 DLOG("payment schedule:");
108 for (auto const& d : paymentDates)
109 DLOG(ore::data::to_string(d));
110
111 // build fx index for composite bond trs
112
113 boost::shared_ptr<FxIndex> fxIndex;
114 if (!fxIndex_.empty()) {
115 fxIndex =
116 buildFxIndex(fxIndex_, fundingLegData_.currency(), bondData_.currency(), engineFactory->market(),
117 engineFactory->configuration(MarketContext::pricing));
118
119 }
120
121 // build bond index (absolute prices, conditional on survival set to false)
122
123 auto bondIndex =
124 buildBondIndex(bondData_, useDirtyPrices_, false, NullCalendar(), false, engineFactory, requiredFixings_);
125
126 // compute initial price taking into account the possible scaling with priceQuoteBaseValue and 100.0
127 Real effectiveInitialPrice = Null<Real>();
128 if (initialPrice_ != Null<Real>()) {
129 Real adj = bondData_.priceQuoteMethod() == BondIndex::PriceQuoteMethod::CurrencyPerUnit
130 ? 1.0 / bondData_.priceQuoteBaseValue()
131 : 1.0;
132 effectiveInitialPrice = initialPrice_ / 100.0 * adj;
133 }
134
135 // add indexing data from the bond trs leg, if this is desired
136
137 if (fundingLegData_.indexingFromAssetLeg()) {
138 DLOG("adding indexing information from trs leg to funding leg");
139
140 std::vector<string> stringValuationDates;
141 for (auto const& d : valuationDates)
142 stringValuationDates.push_back(ore::data::to_string(d));
143 ScheduleData valuationSchedule(ScheduleDates("", "", "", stringValuationDates, ""));
144
145 // add bond indexing
146 Indexing bondIndexing("BOND-" + bondIndex->securityName(), "", bondIndex->dirty(), bondIndex->relative(),
147 bondIndex->conditionalOnSurvival(), bondData_.bondNotional(), effectiveInitialPrice,
148 Null<Real>(), valuationSchedule, 0, "", "U", false);
149 fundingLegData_.indexing().push_back(bondIndexing);
150
151 // add fx indexing, if applicable
152 if (!fxIndex_.empty()) {
153 Indexing fxIndexing(fxIndex_, "", false, false, false, 1.0, Null<Real>(), Null<Real>(), valuationSchedule,
154 0, "", "U", false);
155 fundingLegData_.indexing().push_back(fxIndexing);
156 }
157
158 // set notional node to 1.0
159 fundingLegData_.notionals() = std::vector<Real>(1, 1.0);
160 fundingLegData_.notionalDates() = std::vector<std::string>();
161
162 // reset flag that told us to pull the indexing information from the equity leg
163 fundingLegData_.indexingFromAssetLeg() = false;
164 }
165
166 // build funding leg (consisting of a coupon leg and (possibly) a notional leg
167
168 Leg fundingLeg = legBuilder->buildLeg(fundingLegData_, engineFactory, requiredFixings_, configuration);
169 Leg fundingNotionalLeg;
170 if (fundingLegData_.notionalInitialExchange() || fundingLegData_.notionalFinalExchange() ||
171 fundingLegData_.notionalAmortizingExchange()) {
172 Natural fundingLegPayLag = 0;
173 fundingNotionalLeg =
174 makeNotionalLeg(fundingLeg, fundingLegData_.notionalInitialExchange(),
175 fundingLegData_.notionalFinalExchange(), fundingLegData_.notionalAmortizingExchange(),
176 fundingLegPayLag, parseBusinessDayConvention(fundingLegData_.paymentConvention()),
177 parseCalendar(fundingLegData_.paymentCalendar()));
178 }
179
180 QL_REQUIRE(fundingLegData_.isPayer() != payTotalReturnLeg_,
181 "funding leg and total return lag are both rec or both pay");
182 DLOG("Before bondTRS");
183 auto bondTRS = boost::make_shared<QuantExt::BondTRS>(
184 bondIndex, bondData_.bondNotional(), effectiveInitialPrice,
185 std::vector<QuantLib::Leg>{fundingLeg, fundingNotionalLeg}, payTotalReturnLeg_, valuationDates, paymentDates,
186 fxIndex, payBondCashFlowsImmediately_, parseCurrency(fundingLegData_.currency()),
187 parseCurrency(bondData_.currency()));
188 DLOG("After bondTRS");
189 boost::shared_ptr<BondTRSEngineBuilder> trsBondBuilder =
190 boost::dynamic_pointer_cast<BondTRSEngineBuilder>(builder_trs);
191 QL_REQUIRE(trsBondBuilder, "No Builder found for BondTRS: " << id());
192 bondTRS->setPricingEngine(trsBondBuilder->engine(fundingLegData_.currency()));
193 instrument_.reset(new VanillaInstrument(bondTRS));
194
195 npvCurrency_ = fundingLegData_.currency();
196 // maturity_ = std::max(valuationDates.back(), paymentDates.back());
197 maturity_ = bondIndex->bond()->maturityDate();
198 notional_ = bondIndex->bond()->notional() * bondData_.bondNotional();
199 notionalCurrency_ = bondData_.currency();
200
201 // cashflows will be generated as additional results in the pricing engine
202
203 legs_ = {};
204 legCurrencies_ = {};
205 legPayers_ = {};
206
207 // add required bond and fx fixings for return calculation
208
209 for (auto const& c : bondTRS->returnLeg()) {
210 if (auto bc = boost::dynamic_pointer_cast<BondTRSCashFlow>(c)) {
211 if (bc->initialPrice() == Null<Real>())
212 requiredFixings_.addFixingDate(bc->fixingStartDate(), "BOND-" + bondData_.securityId(), bc->date());
213 requiredFixings_.addFixingDate(bc->fixingEndDate(), "BOND-" + bondData_.securityId(), bc->date());
214 if (!fxIndex_.empty()) {
215 requiredFixings_.addFixingDate(fxIndex->fixingCalendar().adjust(bc->fixingStartDate(), Preceding),
216 fxIndex_, bc->date());
217 requiredFixings_.addFixingDate(fxIndex->fixingCalendar().adjust(bc->fixingEndDate(), Preceding),
218 fxIndex_, bc->date());
219 }
220 }
221 }
222
223 // add required fx fixings for bond cashflow conversion (see the engine for details)
224
225 if (!fxIndex_.empty()) {
226 for (auto const& c : bondIndex->bond()->cashflows()) {
227 requiredFixings_.addFixingDate(fxIndex->fixingCalendar().adjust(c->date(), Preceding), fxIndex_, c->date());
228 }
229 }
230
231 // ISDA taxonomy
232 additionalData_["isdaAssetClass"] = string("Credit");
233 additionalData_["isdaBaseProduct"] = string("Total Return Swap");
234 additionalData_["isdaSubProduct"] = string("");
235 additionalData_["isdaTransaction"] = string("");
236}

238void BondTRS::fromXML(XMLNode* node) {
239 Trade::fromXML(node);
240 XMLNode* bondTRSNode = XMLUtils::getChildNode(node, "BondTRSData");
241 QL_REQUIRE(bondTRSNode, "No BondTRSData Node");
242 originalBondData_.fromXML(XMLUtils::getChildNode(bondTRSNode, "BondData"));
243 bondData_ = originalBondData_;
244
245 XMLNode* bondTRSDataNode = XMLUtils::getChildNode(bondTRSNode, "TotalReturnData");
246 QL_REQUIRE(bondTRSDataNode, "No bondTRSDataNode Node");
247
248 payTotalReturnLeg_ = parseBool(XMLUtils::getChildValue(bondTRSDataNode, "Payer", true));
249
250 scheduleData_.fromXML(XMLUtils::getChildNode(bondTRSDataNode, "ScheduleData"));
251
252 observationLag_ = XMLUtils::getChildValue(bondTRSDataNode, "ObservationLag");
253 observationConvention_ = XMLUtils::getChildValue(bondTRSDataNode, "ObservationConvention");
254 observationCalendar_ = XMLUtils::getChildValue(bondTRSDataNode, "ObservationCalendar");
255
256 paymentLag_ = XMLUtils::getChildValue(bondTRSDataNode, "PaymentLag");
257 paymentConvention_ = XMLUtils::getChildValue(bondTRSDataNode, "PaymentConvention");
258 paymentCalendar_ = XMLUtils::getChildValue(bondTRSDataNode, "PaymentCalendar");
259 paymentDates_ = XMLUtils::getChildrenValues(bondTRSDataNode, "PaymentDates", "PaymentDate");
260
261 initialPrice_ = Null<Real>();
262 if (auto n = XMLUtils::getChildNode(bondTRSDataNode, "InitialPrice"))
263 initialPrice_ = parseReal(XMLUtils::getNodeValue(n));
264 std::string priceType = XMLUtils::getChildValue(bondTRSDataNode, "PriceType", true);
265 if (priceType == "Dirty")
266 useDirtyPrices_ = true;
267 else if (priceType == "Clean")
268 useDirtyPrices_ = false;
269 else {
270 QL_FAIL("PriceType (" << priceType << ") must be Clean or Dirty");
271 }
272
273 XMLNode* fxt = XMLUtils::getChildNode(bondTRSDataNode, "FXTerms");
274 if (fxt)
275 fxIndex_ = XMLUtils::getChildValue(fxt, "FXIndex", true);
276
277 payBondCashFlowsImmediately_ =
278 XMLUtils::getChildValueAsBool(bondTRSDataNode, "PayBondCashFlowsImmediately", false, false);
279
280 XMLNode* bondTRSFundingNode = XMLUtils::getChildNode(bondTRSNode, "FundingData");
281 XMLNode* fLegNode = XMLUtils::getChildNode(bondTRSFundingNode, "LegData");
282 fundingLegData_ = LegData();
283 fundingLegData_.fromXML(fLegNode);
284}

286XMLNode* BondTRS::toXML(XMLDocument& doc) {
287 XMLNode* node = Trade::toXML(doc);
288 XMLNode* bondTRSNode = doc.allocNode("BondTRSData");
289 XMLUtils::appendNode(node, bondTRSNode);
290 XMLUtils::appendNode(bondTRSNode, originalBondData_.toXML(doc));
291
292 XMLNode* trsDataNode = doc.allocNode("TotalReturnData");
293 XMLUtils::appendNode(bondTRSNode, trsDataNode);
294 XMLUtils::addChild(doc, trsDataNode, "Payer", payTotalReturnLeg_);
295
296 if (initialPrice_ != Null<Real>())
297 XMLUtils::addChild(doc, trsDataNode, "InitialPrice", initialPrice_);
298 XMLUtils::addChild(doc, trsDataNode, "PriceType", useDirtyPrices_ ? "Dirty" : "Clean");
299
300 if (!observationLag_.empty())
301 XMLUtils::addChild(doc, trsDataNode, "ObservationLag", observationLag_);
302 if (!observationConvention_.empty())
303 XMLUtils::addChild(doc, trsDataNode, "ObservationConvention", observationConvention_);
304 if (!observationCalendar_.empty())
305 XMLUtils::addChild(doc, trsDataNode, "ObservationCalendar", observationCalendar_);
306
307 if (!paymentLag_.empty())
308 XMLUtils::addChild(doc, trsDataNode, "PaymentLag", paymentLag_);
309 if (!paymentConvention_.empty())
310 XMLUtils::addChild(doc, trsDataNode, "PaymentConvention", paymentConvention_);
311 if (!paymentCalendar_.empty())
312 XMLUtils::addChild(doc, trsDataNode, "PaymentCalendar", paymentCalendar_);
313 if (!paymentDates_.empty())
314 XMLUtils::addChildren(doc, trsDataNode, "PaymentDates", "PaymentDate", paymentDates_);
315
316 if (!fxIndex_.empty()) {
317 XMLNode* fxNode = doc.allocNode("FXTerms");
318 XMLUtils::addChild(doc, fxNode, "FXIndex", fxIndex_);
319 XMLUtils::appendNode(trsDataNode, fxNode);
320 }
321
322 XMLUtils::appendNode(trsDataNode, scheduleData_.toXML(doc));
323
324 XMLUtils::addChild(doc, trsDataNode, "PayBondCashFlowsImmediately", payBondCashFlowsImmediately_);
325
326 XMLNode* fundingDataNode = doc.allocNode("FundingData");
327 XMLUtils::appendNode(bondTRSNode, fundingDataNode);
328 XMLUtils::appendNode(fundingDataNode, fundingLegData_.toXML(doc));
329 return node;
330}
