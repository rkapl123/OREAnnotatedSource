26 void Indexing::fromXML(XMLNode* node) {
27  XMLUtils::checkNode(node, "Indexing");
28  if (auto n = XMLUtils::getChildNode(node, "Quantity")) {
29  quantity_ = parseReal(XMLUtils::getNodeValue(n));
30  } else {
31  quantity_ = 1.0;
32  }
33  index_ = XMLUtils::getChildValue(node, "Index", false);
34  indexFixingCalendar_ = XMLUtils::getChildValue(node, "IndexFixingCalendar", false);
35  if (XMLUtils::getChildNode(node, "IndexFixingDays")) {
36  WLOG("Indexing::fromXML, node IndexFixingDays has been deprecated, fixing days are "
37  "taken from conventions.");
38  }
39  indexIsDirty_ = XMLUtils::getChildValueAsBool(node, "Dirty", false);
40  indexIsRelative_ = XMLUtils::getChildValueAsBool(node, "Relative", false);
41  indexIsConditionalOnSurvival_ = XMLUtils::getChildValueAsBool(node, "ConditionalOnSurvival", false);
42  initialFixing_ = Null<Real>();
43  if (auto n = XMLUtils::getChildNode(node, "InitialFixing"))
44  initialFixing_ = parseReal(XMLUtils::getNodeValue(n));
45  if (auto tmp = XMLUtils::getChildNode(node, "ValuationSchedule"))
46  valuationSchedule_.fromXML(tmp);
47  fixingDays_ = 0;
48  if (auto n = XMLUtils::getChildNode(node, "FixingDays"))
49  fixingDays_ = parseInteger(XMLUtils::getNodeValue(n));
50  fixingCalendar_ = XMLUtils::getChildValue(node, "FixingCalendar");
51  fixingConvention_ = XMLUtils::getChildValue(node, "FixingConvention");
52  inArrearsFixing_ = false;
53  if (auto n = XMLUtils::getChildNode(node, "IsInArrears"))
54  inArrearsFixing_ = parseBool(XMLUtils::getNodeValue(n));
55  hasData_ = true;
56 }

58 XMLNode* Indexing::toXML(XMLDocument& doc) {
59  XMLNode* node = doc.allocNode("Indexing");
60  XMLUtils::addChild(doc, node, "Quantity", quantity_);
61  XMLUtils::addChild(doc, node, "Index", index_);
62  XMLUtils::addChild(doc, node, "IndexFixingCalendar", indexFixingCalendar_);
63  XMLUtils::addChild(doc, node, "Dirty", indexIsDirty_);
64  XMLUtils::addChild(doc, node, "Relative", indexIsRelative_);
65  XMLUtils::addChild(doc, node, "ConditionalOnSurvival", indexIsConditionalOnSurvival_);
66  if (initialFixing_ != Null<Real>())
67  XMLUtils::addChild(doc, node, "InitialFixing", initialFixing_);
68  if (valuationSchedule_.hasData()) {
69  XMLNode* schedNode = valuationSchedule_.toXML(doc);
70  XMLUtils::setNodeName(doc, schedNode, "ValuationSchedule");
71  XMLUtils::appendNode(node, schedNode);
72  }
73  XMLUtils::addChild(doc, node, "FixingDays", static_cast<int>(fixingDays_));
74  XMLUtils::addChild(doc, node, "FixingCalendar", fixingCalendar_);
75  XMLUtils::addChild(doc, node, "FixingConvention", fixingConvention_);
76  XMLUtils::addChild(doc, node, "IsInArrears", inArrearsFixing_);
77  return node;
78 }
