120// calls parseDateOrPeriod and returns a Date (either the supplied date or asof+period)
121Date getDateFromDateOrPeriod(const string& token, Date asof, QuantLib::Calendar cal, QuantLib::BusinessDayConvention bdc) {
122 Period term; // gets populated by parseDateOrPeriod
123 Date expiryDate; // gets populated by parseDateOrPeriod
124 bool tmpIsDate; // gets populated by parseDateOrPeriod
125 parseDateOrPeriod(token, expiryDate, term, tmpIsDate); // checks if the market string contains a date or a period
126 if (!tmpIsDate)
127 expiryDate = cal.adjust(asof + term, bdc);
128 return expiryDate;
129}

147 bool isPeriod = isdigit(s.front());

156struct FxTenorGetter : boost::static_visitor<Period> {
157 FxTenorGetter() {}
158 Period operator()(const Period& p) const { return p; }
159 Period operator()(const FXForwardQuote::FxFwdString& p) const {
160 // These are all overnight rates
161 return Period(1, Days);
162 }
163};
164
165struct FxStartTenorGetter : boost::static_visitor<Period> {
166
167 FxStartTenorGetter(const boost::shared_ptr<FXConvention>& fxConvention) :
168 fxConvention(fxConvention) {}
169
170 Period operator()(const Period& p) const {
171 Integer days = 0;
172 if (fxConvention)
173 days = fxConvention->spotRelative() ? fxConvention->spotDays() : days;
174 return Period(days, Days);
175 }
176
177 Period operator()(const FXForwardQuote::FxFwdString& p) const {
178 Integer days = 0;
179 if (p == FXForwardQuote::FxFwdString::TN)
180 days = 1;
181 else if (p == FXForwardQuote::FxFwdString::SN && fxConvention) {
182 days = fxConvention->spotDays();
183 }
184 return Period(days, Days);
185 }
186
187 boost::shared_ptr<FXConvention> fxConvention;
188};

190struct FxFwdStringCompare : boost::static_visitor<bool> {
191
192 FxFwdStringCompare(FXForwardQuote::FxFwdString fxFwdString) : fxFwdString(fxFwdString) {}
193
194 bool operator()(const Period& p) const { return false; }
195
196 bool operator()(const FXForwardQuote::FxFwdString& p) const {
197 return p == fxFwdString;
198 }
199
200 FXForwardQuote::FxFwdString fxFwdString;
201};

219//! Function to parse a market datum
220boost::shared_ptr<MarketDatum> parseMarketDatum(const Date& asof, const string& datumName, const Real& value) {
221
222 vector<string> tokens;
223 boost::split(tokens, datumName, boost::is_any_of("/"));
224 QL_REQUIRE(tokens.size() > 2, "more than 2 tokens expected in " << datumName);
225
226 MarketDatum::InstrumentType instrumentType = parseInstrumentType(tokens[0]);
227 MarketDatum::QuoteType quoteType = parseQuoteType(tokens[1]);
228
229 switch (instrumentType) {
230
231 case MarketDatum::InstrumentType::ZERO: {
232 // ZERO/RATE/EUR/EUR1D/A365/1Y
233 QL_REQUIRE(quoteType == MarketDatum::QuoteType::RATE || quoteType == MarketDatum::QuoteType::YIELD_SPREAD,
234 "Invalid quote type for " << datumName);
235 QL_REQUIRE(tokens.size() == 6, "6 tokens expected in " << datumName);
236 const string& ccy = tokens[2];
237 DayCounter dc = parseDayCounter(tokens[4]);
238 // token 5 can be a date, or tenor
239 Date date = Date();
240 Period tenor = Period();
241 bool isDate;
242 parseDateOrPeriod(tokens[5], date, tenor, isDate);
243 return boost::make_shared<ZeroQuote>(value, asof, datumName, quoteType, ccy, date, dc, tenor);
244 }
245
246 case MarketDatum::InstrumentType::DISCOUNT: {
247 // DISCOUNT/RATE/EUR/EUR1D/1Y
248 // DISCOUNT/RATE/EUR/EUR1D/2016-12-15
249 QL_REQUIRE(tokens.size() == 5, "5 tokens expected in " << datumName);
250 const string& ccy = tokens[2];
251 // token 4 can be a date, or tenor
252 Date date = Date();
253 Period tenor = Period();
254 bool isDate;
255 parseDateOrPeriod(tokens[4], date, tenor, isDate);
256 return boost::make_shared<DiscountQuote>(value, asof, datumName, quoteType, ccy, date, tenor);
257 }
258
259 case MarketDatum::InstrumentType::MM: {
260 QL_REQUIRE(tokens.size() == 5 || tokens.size() == 6, "5 or 6 tokens expected in " << datumName);
261 const string& ccy = tokens[2];
262 Size offset = 0;
263 string indexName;
264 if (tokens.size() == 6) {
265 indexName = tokens[3];
266 offset = 1;
267 }
268 Period fwdStart = parsePeriod(tokens[3 + offset]);
269 Period term = parsePeriod(tokens[4 + offset]);
270 return boost::make_shared<MoneyMarketQuote>(value, asof, datumName, quoteType, ccy, fwdStart, term, indexName);
271 }
272
273 case MarketDatum::InstrumentType::MM_FUTURE: {
274 QL_REQUIRE(tokens.size() == 6, "6 tokens expected in " << datumName);
275 const string& ccy = tokens[2];
276 const string& expiry = tokens[3];
277 const string& contract = tokens[4];
278 Period term = parsePeriod(tokens[5]);
279 return boost::make_shared<MMFutureQuote>(value, asof, datumName, quoteType, ccy, expiry, contract, term);
280 }
281
282 case MarketDatum::InstrumentType::OI_FUTURE: {
283 QL_REQUIRE(tokens.size() == 6, "6 tokens expected in " << datumName);
284 const string& ccy = tokens[2];
285 const string& expiry = tokens[3];
286 const string& contract = tokens[4];
287 Period term = parsePeriod(tokens[5]);
288 return boost::make_shared<OIFutureQuote>(value, asof, datumName, quoteType, ccy, expiry, contract, term);
289 }
290
291 case MarketDatum::InstrumentType::FRA: {
292 QL_REQUIRE(tokens.size() == 5, "5 tokens expected in " << datumName);
293 const string& ccy = tokens[2];
294 Period fwdStart = parsePeriod(tokens[3]);
295 Period term = parsePeriod(tokens[4]);
296 return boost::make_shared<FRAQuote>(value, asof, datumName, quoteType, ccy, fwdStart, term);
297 }
298
299 case MarketDatum::InstrumentType::IMM_FRA: {
300 QL_REQUIRE(tokens.size() == 5, "5 tokens expected in " << datumName);
301 const string& ccy = tokens[2];
302 string imm1 = tokens[3];
303 string imm2 = tokens[4];
304 unsigned int m1 = parseInteger(imm1);
305 unsigned int m2 = parseInteger(imm2);
306 QL_REQUIRE(m2 > m1, "Second IMM date must be after the first in " << datumName);
307 return boost::make_shared<ImmFraQuote>(value, asof, datumName, quoteType, ccy, m1, m2);
308 }
309
310 case MarketDatum::InstrumentType::IR_SWAP: {
311 QL_REQUIRE(tokens.size() == 6 || tokens.size() == 7, "6 or 7 tokens expected in " << datumName);
312 const string& ccy = tokens[2];
313 Size offset = 0;
314 string indexName;
315 if (tokens.size() == 7) {
316 indexName = tokens[3];
317 offset = 1;
318 }
319 Period fwdStart = parsePeriod(tokens[3 + offset]);
320 Period tenor = parsePeriod(tokens[4 + offset]);
321 Period term = parsePeriod(tokens[5 + offset]);
322 return boost::make_shared<SwapQuote>(value, asof, datumName, quoteType, ccy, fwdStart, term, tenor, indexName);
323 }
324
325 case MarketDatum::InstrumentType::BASIS_SWAP: {
326 // An optional identifier as a penultimate token supports the following two versions:
327 // BASIS_SWAP/BASIS_SPREAD/3M/1D/USD/5Y
328 // BASIS_SWAP/BASIS_SPREAD/3M/1D/USD/foobar/5Y
329 QL_REQUIRE(tokens.size() == 6 || tokens.size() == 7, "Either 6 or 7 tokens expected in " << datumName);
330 Period flatTerm = parsePeriod(tokens[2]);
331 Period term = parsePeriod(tokens[3]);
332 const string& ccy = tokens[4];
333 Period maturity;
334 if (tokens.size() == 7) {
335 maturity = parsePeriod(tokens[6]);
336 } else {
337 maturity = parsePeriod(tokens[5]);
338 }
339 return boost::make_shared<BasisSwapQuote>(value, asof, datumName, quoteType, flatTerm, term, ccy, maturity);
340 }
341
342 case MarketDatum::InstrumentType::BMA_SWAP: {
343 QL_REQUIRE(tokens.size() == 5, "5 tokens expected in " << datumName);
344 const string& ccy = tokens[2];
345 Period term = parsePeriod(tokens[3]);
346 Period maturity = parsePeriod(tokens[4]);
347 return boost::make_shared<BMASwapQuote>(value, asof, datumName, quoteType, term, ccy, maturity);
348 }
349
350 case MarketDatum::InstrumentType::CC_BASIS_SWAP: {
351 QL_REQUIRE(tokens.size() == 7, "7 tokens expected in " << datumName);
352 const string& flatCcy = tokens[2];
353 Period flatTerm = parsePeriod(tokens[3]);
354 const string& ccy = tokens[4];
355 Period term = parsePeriod(tokens[5]);
356 Period maturity = parsePeriod(tokens[6]);
357 return boost::make_shared<CrossCcyBasisSwapQuote>(value, asof, datumName, quoteType, flatCcy, flatTerm, ccy,
358 term, maturity);
359 }
360
361 case MarketDatum::InstrumentType::CC_FIX_FLOAT_SWAP: {
362 // CC_FIX_FLOAT_SWAP/RATE/USD/3M/TRY/1Y/5Y
363 QL_REQUIRE(tokens.size() == 7, "7 tokens expected in " << datumName);
364 Period floatTenor = parsePeriod(tokens[3]);
365 Period fixedTenor = parsePeriod(tokens[5]);
366 Period maturity = parsePeriod(tokens[6]);
367 return boost::make_shared<CrossCcyFixFloatSwapQuote>(value, asof, datumName, quoteType, tokens[2], floatTenor,
368 tokens[4], fixedTenor, maturity);
369 }
370
371 case MarketDatum::InstrumentType::CDS: {
372 // CDS/[CONV_]CREDIT_SPREAD/Name/Seniority/ccy/term
373 // CDS/[CONV_]CREDIT_SPREAD/Name/Seniority/ccy/term/runningSpread
374 // CDS/[CONV_]CREDIT_SPREAD/Name/Seniority/ccy/doc/term
375 // CDS/[CONV_]CREDIT_SPREAD/Name/Seniority/ccy/doc/term/runningSpread
376 // CDS/PRICE/Name/Seniority/ccy/term
377 // CDS/PRICE/Name/Seniority/ccy/term/runningSpread
378 // CDS/PRICE/Name/Seniority/ccy/doc/term
379 // CDS/PRICE/Name/Seniority/ccy/doc/term/runningSpread
380 QL_REQUIRE(tokens.size() == 6 || tokens.size() == 7 || tokens.size() == 8,
381 "6, 7 or 8 tokens expected in " << datumName);
382 const string& underlyingName = tokens[2];
383 const string& seniority = tokens[3];
384 const string& ccy = tokens[4];
385
386 string docClause;
387 Period term;
388 Real runningSpread = Null<Real>();
389 if (tokens.size() == 6) {
390 term = parsePeriod(tokens[5]);
391 } else if (tokens.size() == 8) {
392 docClause = tokens[5];
393 term = parsePeriod(tokens[6]);
394 runningSpread = parseReal(tokens[7]) / 10000;
395 } else {
396 // 7 tokens => [5]/[6] = doc/term or term/runningSpread
397 CdsDocClause cdsDocClause;
398 if (tryParse<CdsDocClause>(tokens[5], cdsDocClause, &parseCdsDocClause)) {
399 docClause = tokens[5];
400 term = parsePeriod(tokens[6]);
401 } else {
402 term = parsePeriod(tokens[5]);
403 runningSpread = parseReal(tokens[6]) / 10000;
404 }
405 }
406
407 return boost::make_shared<CdsQuote>(value, asof, datumName, quoteType, underlyingName,
408 seniority, ccy, term, docClause, runningSpread);
409 }
410
411 case MarketDatum::InstrumentType::HAZARD_RATE: {
412 QL_REQUIRE(tokens.size() == 6 || tokens.size() == 7, "6 or 7 tokens expected in " << datumName);
413 const string& underlyingName = tokens[2];
414 const string& seniority = tokens[3];
415 const string& ccy = tokens[4];
416 string docClause = tokens.size() == 7 ? tokens[5] : "";
417 Period term = parsePeriod(tokens.back());
418 return boost::make_shared<HazardRateQuote>(value, asof, datumName, underlyingName, seniority, ccy, term,
419 docClause);
420 }
421
422 case MarketDatum::InstrumentType::RECOVERY_RATE: {
423 QL_REQUIRE(tokens.size() == 3 || tokens.size() == 5 || tokens.size() == 6,
424 "3, 5 or 6 tokens expected in " << datumName);
425 const string& underlyingName = tokens[2]; // issuer name for CDS, security ID for bond specific RRs
426 string seniority = "";
427 string ccy = "";
428 string docClause = "";
429 if (tokens.size() >= 5) {
430 // CDS
431 seniority = tokens[3];
432 ccy = tokens[4];
433 if (tokens.size() == 6)
434 docClause = tokens[5];
435 }
436 return boost::make_shared<RecoveryRateQuote>(value, asof, datumName, underlyingName, seniority, ccy, docClause);
437 }
438
439 case MarketDatum::InstrumentType::CAPFLOOR: {
440 QL_REQUIRE((tokens.size() >= 8 && tokens.size() <= 10) || tokens.size() == 4 || tokens.size() == 5,
441 "Either 4, 5 or 8, 9, 10 tokens expected in " << datumName);
442 const string& ccy = tokens[2];
443 Size offset = 0;
444 std::string indexName;
445 Size hasCapFloorToken = (tokens.back() == "C" || tokens.back() == "F") ? 1 : 0;
446 QL_REQUIRE(quoteType != MarketDatum::QuoteType::PRICE || hasCapFloorToken == 1,
447 "CAPFLOOR PRICE quotes must specify whether the datum represents a cap or a floor with a \"C\" or"
448 " \"F\" as the final token.");
449
450 if (tokens.size() == 9 + hasCapFloorToken || tokens.size() == 5 + hasCapFloorToken) {
451 // has an index name token... all later tokens are offset by 1
452 offset = 1;
453 indexName = tokens[3];
454 }
455 if (tokens.size() == 8 + hasCapFloorToken || tokens.size() == 9 + hasCapFloorToken) {
456 Period term = parsePeriod(tokens[3 + offset]);
457 Period tenor = parsePeriod(tokens[4 + offset]);
458 bool atm = parseBool(tokens[5 + offset].c_str());
459 bool relative = parseBool(tokens[6 + offset].c_str());
460 Real strike = parseReal(tokens[7 + offset]);
461 bool isCap = !(hasCapFloorToken==1 && tokens.back() == "F"); // assume cap if omitted
462 return boost::make_shared<CapFloorQuote>(value, asof, datumName, quoteType, ccy, term, tenor, atm, relative,
463 strike, indexName, isCap);
464 } else {
465 // not enough tokens, must be a shift quote
466 Period indexTenor = parsePeriod(tokens[3 + offset]);
467 return boost::make_shared<CapFloorShiftQuote>(value, asof, datumName, quoteType, ccy, indexTenor,
468 indexName);
469 }
470 }
471
472 case MarketDatum::InstrumentType::SWAPTION: {
473 QL_REQUIRE(tokens.size() >= 4 && tokens.size() <= 9, "4...9 tokens expected in " << datumName);
474 const string& ccy = tokens[2];
475 Size offset = isOnePeriod(tokens[3]) ? 0 : 1;
476 std::string quoteTag;
477 Size hasPayReceiveToken = (tokens.back() == "P" || tokens.back() == "R") ? 1 : 0;
478 QL_REQUIRE(quoteType != MarketDatum::QuoteType::PRICE || hasPayReceiveToken == 1,
479 "SWAPTION PRICE quotes must specify whether the datum represents a payer or a receiver"
480 " swaption with a \"P\" or \"R\" as the final token.");
481
482 if (offset == 1)
483 quoteTag = tokens[3];
484 if (tokens.size() >= 6 + offset + hasPayReceiveToken) { // volatility
485 Period expiry = parsePeriod(tokens[3 + offset]);
486 Period term = parsePeriod(tokens[4 + offset]);
487 const string& dimension = tokens[5 + offset];
488 Real strike = 0.0;
489 if (dimension == "ATM")
490 QL_REQUIRE(tokens.size() == 6 + offset + hasPayReceiveToken, 6 + offset + hasPayReceiveToken
491 << " tokens expected in ATM quote " << datumName);
492 else if (dimension == "Smile") {
493 QL_REQUIRE(tokens.size() == 7 + offset + hasPayReceiveToken, 7 + offset + hasPayReceiveToken
494 << " tokens expected in Smile quote " << datumName);
495 strike = parseReal(tokens[6 + offset]);
496 } else
497 QL_FAIL("Swaption vol quote dimension " << dimension << " not recognised");
498 bool isPayer = !(hasPayReceiveToken == 1 && tokens.back() == "R"); // assume payer if omitted
499 return boost::make_shared<SwaptionQuote>(value, asof, datumName, quoteType, ccy, expiry, term, dimension,
500 strike, quoteTag, isPayer);
501 } else { // SLN volatility shift
502 return boost::make_shared<SwaptionShiftQuote>(value, asof, datumName, quoteType, ccy,
503 parsePeriod(tokens[3 + offset]), quoteTag);
504 }
505 }
506
507 case MarketDatum::InstrumentType::BOND_OPTION: {
508 QL_REQUIRE(tokens.size() == 4 || tokens.size() == 6, "4 or 6 tokens expected in " << datumName);
509 const string& qualifier = tokens[2];
510 Period expiry = tokens.size() == 6 ? parsePeriod(tokens[3]) : Period(0 * QuantLib::Days);
511 Period term = tokens.size() == 6 ? parsePeriod(tokens[4]) : parsePeriod(tokens[3]);
512 if (tokens.size() == 6) { // volatility
513 QL_REQUIRE(tokens[5] == "ATM", "only ATM allowed for bond option quotes");
514 return boost::make_shared<BondOptionQuote>(value, asof, datumName, quoteType, qualifier, expiry, term);
515 } else { // SLN volatility shift
516 return boost::make_shared<BondOptionShiftQuote>(value, asof, datumName, quoteType, qualifier, term);
517 }
518 }
519
520 case MarketDatum::InstrumentType::FX_SPOT: {
521 QL_REQUIRE(tokens.size() == 4, "4 tokens expected in " << datumName);
522 const string& unitCcy = tokens[2];
523 const string& ccy = tokens[3];
524 return boost::make_shared<FXSpotQuote>(value, asof, datumName, quoteType, unitCcy, ccy);
525 }

609 const string& equityName = tokens[2];

616 case MarketDatum::InstrumentType::EQUITY_OPTION: {
617 QL_REQUIRE(tokens.size() >= 6 || tokens.size() <= 9, "6 - 9 tokens expected in " << datumName);
618 QL_REQUIRE(quoteType == MarketDatum::QuoteType::RATE_LNVOL || quoteType == MarketDatum::QuoteType::PRICE,
619 "Invalid quote type for " << datumName);
620 const string& equityName = tokens[2];
621 const string& ccy = tokens[3];
622 string expiryString = tokens[4];
623 // do we have a call or put flag as the last token?
624 bool hasCallPutToken = tokens.back() == "C" || tokens.back() == "P";
625 // the following tokens represent the strike, except the last one, if we have a call/put token
626 string strikeStr;
627 for(Size i=5; i < tokens.size() - (hasCallPutToken ? 1 : 0); ++i) {
628 strikeStr += (i > 5 ? "/" : "") + tokens[i];
629 }
630 boost::shared_ptr<BaseStrike> strike;
631 // we support ATM, ATMF as aliases for ATM/AtmSpot, ATM/AtmFwd, plus absolute strikes and MNY/[Spot/Fwd]/1.2
632 if(strikeStr == "ATM")
633 strike = boost::make_shared<AtmStrike>(QuantLib::DeltaVolQuote::AtmType::AtmSpot);
634 else if(strikeStr == "ATMF")
635 strike = boost::make_shared<AtmStrike>(QuantLib::DeltaVolQuote::AtmType::AtmFwd);
636 else
637 strike = parseBaseStrike(strikeStr);
638 bool isCall = true;
639 if (hasCallPutToken) {
640 QL_REQUIRE(tokens.back() == "C" || tokens.back() == "P",
641 "excepted C or P for Call or Put at position " << tokens.size() << " in " << datumName);
642 isCall = tokens.back() == "C";
643 }
644
645 // note how we only store the expiry string - to ensure we can support both Periods and Dates being specified in
646 // the vol curve-config.
647 return boost::make_shared<EquityOptionQuote>(value, asof, datumName, quoteType, equityName, ccy, expiryString,
648 strike, isCall);
649 }

651 case MarketDatum::InstrumentType::BOND: {
652 QL_REQUIRE(tokens.size() == 3, "3 tokens expected in " << datumName);
653 const string& securityID = tokens[2];

655 return boost::make_shared<SecuritySpreadQuote>(value, asof, datumName, securityID);
656 else if (quoteType == MarketDatum::QuoteType::PRICE)

662 QL_REQUIRE(quoteType == MarketDatum::QuoteType::BASE_CORRELATION, "Invalid quote type for " << datumName);
663 const string& cdsIndexName = tokens[2];

667 detachmentPoint);

670 case MarketDatum::InstrumentType::INDEX_CDS_OPTION: {
671
672 // Expects the following form. The strike is optional. The index term is optional for backwards compatibility.
673 // INDEX_CDS_OPTION/RATE_LNVOL/<INDEX_NAME>[/<INDEX_TERM>]/<EXPIRY>[/<STRIKE>]
674 QL_REQUIRE(tokens.size() >= 4 || tokens.size() <= 6, "4, 5 or 6 tokens expected in " << datumName);
675 QL_REQUIRE(quoteType == MarketDatum::QuoteType::RATE_LNVOL, "Invalid quote type for " << datumName);
676
677 boost::shared_ptr<Expiry> expiry;
678 boost::shared_ptr<BaseStrike> strike;
679 string indexTerm;
680 if (tokens.size() == 6) {
681 // We have been given an index term, an expiry and a strike.
682 indexTerm = tokens[3];
683 expiry = parseExpiry(tokens[4]);
684 strike = parseBaseStrike(tokens[5]);
685 } else if (tokens.size() == 5) {
686 // We have been given either 1) an index term and an expiry or 2) an expiry and a strike.
687 // If the last token is a number, we have 2) an expiry and a strike.
688 Real tmp;
689 if (tryParseReal(tokens[4], tmp)) {
690 expiry = parseExpiry(tokens[3]);
691 strike = parseBaseStrike(tokens[4]);
692 } else {
693 indexTerm = tokens[3];
694 expiry = parseExpiry(tokens[4]);
695 }
696 } else {
697 // We have just been given the expiry.
698 expiry = parseExpiry(tokens[3]);
699 }
700
701 return boost::make_shared<IndexCDSOptionQuote>(value, asof, datumName, tokens[2], expiry, indexTerm, strike);
702 }
703
704 case MarketDatum::InstrumentType::COMMODITY_SPOT: {
705 QL_REQUIRE(tokens.size() == 4, "4 tokens expected in " << datumName);
706 QL_REQUIRE(quoteType == MarketDatum::QuoteType::PRICE, "Invalid quote type for " << datumName);
707
708 return boost::make_shared<CommoditySpotQuote>(value, asof, datumName, quoteType, tokens[2], tokens[3]);
709 }
710
711 case MarketDatum::InstrumentType::COMMODITY_FWD: {
712 // Expects the following form:
713 // COMMODITY_FWD/PRICE/<COMDTY_NAME>/<CCY>/<DATE/TENOR>
714 QL_REQUIRE(tokens.size() == 5, "5 tokens expected in " << datumName);
715 QL_REQUIRE(quoteType == MarketDatum::QuoteType::PRICE, "Invalid quote type for " << datumName);
716
717 // The last token can be a string defining a special tenor i.e. ON, TN or SN
718 if (tokens[4] == "ON") {
719 return boost::make_shared<CommodityForwardQuote>(value, asof, datumName, quoteType, tokens[2], tokens[3],
720 1 * Days, 0 * Days);
721 } else if (tokens[4] == "TN") {
722 return boost::make_shared<CommodityForwardQuote>(value, asof, datumName, quoteType, tokens[2], tokens[3],
723 1 * Days, 1 * Days);
724 } else if (tokens[4] == "SN") {
725 return boost::make_shared<CommodityForwardQuote>(value, asof, datumName, quoteType, tokens[2], tokens[3],
726 1 * Days);
727 }
728
729 // The last token can be a date or a standard tenor
730 Date date;
731 Period tenor;
732 bool isDate;
733 parseDateOrPeriod(tokens[4], date, tenor, isDate);
734
735 if (isDate) {
736 return boost::make_shared<CommodityForwardQuote>(value, asof, datumName, quoteType, tokens[2], tokens[3],
737 date);
738 } else {
739 return boost::make_shared<CommodityForwardQuote>(value, asof, datumName, quoteType, tokens[2], tokens[3],
740 tenor);
741 }
742 }
743
744 case MarketDatum::InstrumentType::COMMODITY_OPTION: {
745 // Expects one of the following forms:
746 // COMMODITY_OPTION/<QT>/<COMDTY_NAME>/<CCY>/<EXPIRY>/<STRIKE>
747 // COMMODITY_OPTION/<QT>/<COMDTY_NAME>/<CCY>/<EXPIRY>/<STRIKE>/<OT>
748 // where QT = RATE_LNVOL or PRICE and OT = C (for Call) or P (for Put)
749 using QT = MarketDatum::QuoteType;
750 QL_REQUIRE(tokens.size() >= 6, "At least 6 tokens expected in " << datumName);
751 QL_REQUIRE(quoteType == QT::RATE_LNVOL || quoteType == QT::PRICE,
752 "Quote type for " << datumName << " should be 'RATE_LNVOL' or 'PRICE'");
753
754 boost::shared_ptr<Expiry> expiry = parseExpiry(tokens[4]);
755
756 // If the last token is C or P, process it and update bounds of strike portion.
757 auto itStkBeg = tokens.begin() + 5;
758 auto itStkEnd = tokens.end();
759 Option::Type optionType = Option::Call;
760 if (tokens.back() == "C" || tokens.back() == "P") {
761 if (tokens.back() == "P")
762 optionType = Option::Put;
763 itStkEnd = prev(itStkEnd);
764 }
765
766 // Parse the strike
767 QL_REQUIRE(itStkBeg != itStkEnd, "");
768 string strStrike = boost::algorithm::join(boost::make_iterator_range(itStkBeg, itStkEnd), "/");
769 boost::shared_ptr<BaseStrike> strike = parseBaseStrike(strStrike);
770
771 return boost::make_shared<CommodityOptionQuote>(value, asof, datumName, quoteType, tokens[2],
772 tokens[3], expiry, strike, optionType);
773 }
774
775 case MarketDatum::InstrumentType::CORRELATION: {
776 // Expects the following form:
777 // CORRELATION/RATE/<INDEX1>/<INDEX2>/<TENOR>/<STRIKE>
778 QL_REQUIRE(tokens.size() == 6, "6 tokens expected in " << datumName);
779 QL_REQUIRE(quoteType == MarketDatum::QuoteType::RATE || quoteType == MarketDatum::QuoteType::PRICE,
780 "Quote type for " << datumName << " should be 'CORRELATION' or 'PRICE'");
781
782 return boost::make_shared<CorrelationQuote>(value, asof, datumName, quoteType, tokens[2], tokens[3], tokens[4],
783 tokens[5]);
784 }
785
786 case MarketDatum::InstrumentType::CPR: {
787 QL_REQUIRE(tokens.size() == 3, "3 tokens expected in " << datumName);
788 const string& securityID = tokens[2];
789 QL_REQUIRE(quoteType == MarketDatum::QuoteType::RATE, "Invalid quote type for " << datumName);
790 return boost::make_shared<CPRQuote>(value, asof, datumName, securityID);
791 }
792
793 case MarketDatum::InstrumentType::RATING: {
794 QL_REQUIRE(tokens.size() == 5, "5 tokens expected in " << datumName);
795 const string& name = tokens[2];
796 const string& fromRating = tokens[3];
797 const string& toRating = tokens[4];
798 QL_REQUIRE(quoteType == MarketDatum::QuoteType::TRANSITION_PROBABILITY, "Invalid quote type for " << datumName);
799 return boost::make_shared<TransitionProbabilityQuote>(value, asof, datumName, name, fromRating, toRating);
800 }
801
802 default:
803 QL_FAIL("Cannot convert \"" << datumName << "\" to MarketDatum");
804 } // switch instrument type
