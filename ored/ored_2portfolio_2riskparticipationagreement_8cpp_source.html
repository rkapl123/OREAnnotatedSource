37void RiskParticipationAgreement::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
38
39 LOG("RiskParticipationAgreement::build() for id \"" << id() << "\" called.");
40
41 // do some checks
42
43 QL_REQUIRE(!protectionFee_.empty(), "protection fees must not be empty");
44
45 QL_REQUIRE(underlying_.empty() || tlockData_.empty(),
46 "RiskParticipationAgreement::build(): both LegData and TreasuryLockData given in Underlying node.");
47
48 if (!underlying_.empty())
49 buildWithSwapUnderlying(engineFactory);
50 else if (!tlockData_.empty())
51 buildWithTlockUnderlying(engineFactory);
52 else {
53 QL_FAIL("RiskParticipationAgreement::build(): Underlying node must not be empty, LegData or TreasuryLockData "
54 "required as subnode");
55 }
56
57 // set start date
58 additionalData_["startDate"] = to_string(protectionStart_);
59
60 // ISDA taxonomy
61 additionalData_["isdaAssetClass"] = string("Interest Rate");
62 additionalData_["isdaBaseProduct"] = string("Exotic");
63 additionalData_["isdaSubProduct"] = string("");
64 additionalData_["isdaTransaction"] = string("");
65}

67void RiskParticipationAgreement::buildWithSwapUnderlying(const boost::shared_ptr<EngineFactory>& engineFactory) {
68 npvCurrency_ = notionalCurrency_ = underlying_.front().currency();
69
70 bool isXccy = false;
71 for (auto const& u : underlying_) {
72 isXccy = isXccy || u.currency() != npvCurrency_;
73 }
74
75 for (auto const& p : protectionFee_)
76 QL_REQUIRE(p.isPayer() == protectionFee_.front().isPayer(),
77 "the protection fee legs must all be pay or all receive");
78
79 /* determine product variant used to retrieve engine builder
80
81 RiskParticipationAgreement_Vanilla:
82 - exactly one fixed and one floating leg with opposite payer flags
83 - only fixed, ibor, ois (comp, avg) coupons allowed, no cap / floors, no in arrears fixings for ibor
84
85 RiskParticipationAgreement_Structured:
86 - arbitrary number of fixed, floating, cashflow legs
87 - only fixed, ibor coupons, ois (comp, avg), simple cashflows allowed, but possibly capped / floored,
88 as naked option, with in arrears fixing for ibor
89 - with optionData (i.e. callable underlying), as naked option (i.e. swaption)
90
91 RiskParticipationAgreement_Vanilla_XCcy:
92 - two legs in different currencies with arbitrary coupons allowed, no optionData though
93 */
94
95 std::set<std::string> legTypes;
96 std::set<bool> legPayers;
97 bool hasCapFloors = false, hasIborInArrears = false;
98 for (auto const& l : underlying_) {
99 QL_REQUIRE(l.legType() == "Fixed" || l.legType() == "Floating" || l.legType() == "Cashflow",
100 "RiskParticipationAgreement: leg type " << l.legType()
101 << " not supported, expected Fixed, Floating, Cashflow");
102 legTypes.insert(l.legType());
103 legPayers.insert(l.isPayer());
104 if (auto c = boost::dynamic_pointer_cast<FloatingLegData>(l.concreteLegData())) {
105 hasCapFloors = hasCapFloors || !c->caps().empty();
106 hasCapFloors = hasCapFloors || !c->floors().empty();
107 hasIborInArrears = hasIborInArrears || (c->isInArrears() && !isOvernightIndex(c->index()));
108 }
109 }
110
111 std::string productVariant;
112 if (isXccy) {
113 productVariant = "RiskParticipationAgreement_Vanilla_XCcy";
114 QL_REQUIRE(!optionData_, "XCcy Risk Participation Agreement does not allow for OptionData");
115 } else if (underlying_.size() == 2 &&
116 (legTypes == std::set<std::string>{"Fixed", "Floating"} ||
117 legTypes == std::set<std::string>{"Cashflow", "Floating"}) &&
118 legPayers == std::set<bool>{false, true} && !hasCapFloors && !hasIborInArrears && !optionData_) {
119 productVariant = "RiskParticipationAgreement_Vanilla";
120 } else {
121 productVariant = "RiskParticipationAgreement_Structured";
122 }
123
124 // get engine builder
125
126 DLOG("get engine builder for product variant " << productVariant);
127 auto builder = boost::dynamic_pointer_cast<RiskParticipationAgreementEngineBuilderBase>(
128 engineFactory->builder(productVariant));
129 QL_REQUIRE(builder, "wrong builder, expected RiskParticipationAgreementEngineBuilderBase");
130 auto configuration = builder->configuration(MarketContext::pricing);
131
132 // build underlying legs and protection fee legs
133
134 std::vector<Leg> underlyingLegs, protectionFeeLegs;
135 std::vector<bool> underlyingPayer, protectionPayer;
136 std::vector<string> underlyingCcys, protectionCcys;
137 for (auto const& l : underlying_) {
138 auto legBuilder = engineFactory->legBuilder(l.legType());
139 underlyingLegs.push_back(legBuilder->buildLeg(l, engineFactory, requiredFixings_, configuration));
140 underlyingPayer.push_back(l.isPayer());
141 underlyingCcys.push_back(l.currency());
142 auto leg = buildNotionalLeg(l, underlyingLegs.back(), requiredFixings_, engineFactory->market(), configuration);
143 if (!leg.empty()) {
144 underlyingLegs.push_back(leg);
145 underlyingPayer.push_back(l.isPayer());
146 underlyingCcys.push_back(l.currency());
147 }
148 }
149 for (auto const& l : protectionFee_) {
150 auto legBuilder = engineFactory->legBuilder(l.legType());
151 protectionFeeLegs.push_back(legBuilder->buildLeg(l, engineFactory, requiredFixings_, configuration));
152 protectionPayer.push_back(l.isPayer());
153 protectionCcys.push_back(protectionFeeLegs.back().empty() ? npvCurrency_ : l.currency());
154 }
155
156 // build exercise, if option data is present
157
158 boost::shared_ptr<QuantLib::Exercise> exercise;
159 bool exerciseIsLong = true;
160 if (optionData_) {
161 ExerciseBuilder eb(*optionData_, underlyingLegs);
162 exercise = eb.exercise();
163 exerciseIsLong = parsePositionType((*optionData_).longShort()) == QuantLib::Position::Long;
164 }
165
166 // build ql instrument
167
168 auto qleInstr = boost::make_shared<QuantExt::RiskParticipationAgreement>(
169 underlyingLegs, underlyingPayer, underlyingCcys, protectionFeeLegs, protectionPayer.front(), protectionCcys,
170 participationRate_, protectionStart_, protectionEnd_, settlesAccrual_, fixedRecoveryRate_, exercise,
171 exerciseIsLong, nakedOption_);
172
173 // wrap instrument
174
175 instrument_ = boost::make_shared<VanillaInstrument>(qleInstr);
176
177 // set trade members
178
179 notional_ = 0.0;
180 for (Size i = 0; i < underlyingLegs.size(); ++i) {
181 notional_ = std::max(notional_, currentNotional(underlyingLegs[i]) *
182 engineFactory->market()
183 ->fxRate(underlyingCcys[i] + notionalCurrency_,
184 engineFactory->configuration(MarketContext::pricing))
185 ->value());
186 }
187 legs_ = underlyingLegs;
188 legCurrencies_ = underlyingCcys;
189 legPayers_ = underlyingPayer;
190 legs_.insert(legs_.end(), protectionFeeLegs.begin(), protectionFeeLegs.end());
191 legCurrencies_.insert(legCurrencies_.end(), protectionCcys.begin(), protectionCcys.end());
192 legPayers_.insert(legPayers_.end(), protectionPayer.begin(), protectionPayer.end());
193 maturity_ = qleInstr->maturity();
194
195 // set pricing engine
196 qleInstr->setPricingEngine(builder->engine(id(), this));
197}

201 for (auto const& c : l) {

213void RiskParticipationAgreement::buildWithTlockUnderlying(const boost::shared_ptr<EngineFactory>& engineFactory) {
214 std::string productVariant = "RiskParticipationAgreement_TLock";
215
216 // get bond reference data and build bond
217
218 tlockData_.bondData() = tlockData_.originalBondData();
219 tlockData_.bondData().populateFromBondReferenceData(engineFactory->referenceData());
220 ore::data::Bond tmp(Envelope(), tlockData_.bondData());
221 tmp.build(engineFactory);
222 auto bond = boost::dynamic_pointer_cast<QuantLib::Bond>(tmp.instrument()->qlInstrument());
223 QL_REQUIRE(bond != nullptr, "RiskParticipationAgreement: could not build tlock underlying, cast failed (internal "
224 "error that dev needs to look at)");
225
226 // set currency and notional
227
228 npvCurrency_ = notionalCurrency_ = tlockData_.bondData().currency();
229 notional_ = tlockData_.bondData().bondNotional();
230
231 // checks
232
233 for (auto const& p : protectionFee_)
234 QL_REQUIRE(p.isPayer() == protectionFee_.front().isPayer(),
235 "the protection fee legs must all be pay or all receive");
236
237 // get engine builder
238
239 DLOG("get engine builder for product variant " << productVariant);
240 auto builder = boost::dynamic_pointer_cast<RiskParticipationAgreementEngineBuilderBase>(
241 engineFactory->builder(productVariant));
242 QL_REQUIRE(builder, "wrong builder, expected RiskParticipationAgreementEngineBuilderBase");
243 auto configuration = builder->configuration(MarketContext::pricing);
244
245 // build ql instrument and set the pricing engine
246
247 bool payer = tlockData_.payer();
248 Real referenceRate = tlockData_.referenceRate();
249 DayCounter dayCounter =
250 tlockData_.dayCounter().empty() ? getDayCounter(bond->cashflows()) : parseDayCounter(tlockData_.dayCounter());
251 Date terminationDate = parseDate(tlockData_.terminationDate());
252 Size paymentGap = tlockData_.paymentGap();
253 Calendar paymentCalendar = parseCalendar(tlockData_.paymentCalendar());
254
255 std::vector<Leg> protectionFeeLegs;
256 std::vector<bool> protectionPayer;
257 std::vector<string> protectionCcys;
258 for (auto const& l : protectionFee_) {
259 auto legBuilder = engineFactory->legBuilder(l.legType());
260 protectionFeeLegs.push_back(legBuilder->buildLeg(l, engineFactory, requiredFixings_, configuration));
261 protectionPayer.push_back(l.isPayer());
262 protectionCcys.push_back(protectionFeeLegs.back().empty() ? npvCurrency_ : l.currency());
263 }
264
265 Date paymentDate = paymentCalendar.advance(terminationDate, paymentGap * Days);
266 auto qleInstr = boost::make_shared<QuantExt::RiskParticipationAgreementTLock>(
267 bond, notional_, payer, referenceRate, dayCounter, terminationDate, paymentDate, protectionFeeLegs,
268 protectionPayer.front(), protectionCcys, participationRate_, protectionStart_, protectionEnd_, settlesAccrual_,
269 fixedRecoveryRate_);
270
271 // wrap instrument
272
273 instrument_ = boost::make_shared<VanillaInstrument>(qleInstr);
274
275 // set trade members
276
277 legs_ = {bond->cashflows()};
278 legCurrencies_ = {npvCurrency_};
279 legPayers_ = {payer};
280 legs_.insert(legs_.end(), protectionFeeLegs.begin(), protectionFeeLegs.end());
281 legCurrencies_.insert(legCurrencies_.end(), protectionCcys.begin(), protectionCcys.end());
282 legPayers_.insert(legPayers_.end(), protectionPayer.begin(), protectionPayer.end());
283 maturity_ = qleInstr->maturity();
284
285 // set pricing engine
286
287 qleInstr->setPricingEngine(builder->engine(id(), this));
288}

290void RiskParticipationAgreement::fromXML(XMLNode* node) {
291 Trade::fromXML(node);
292 XMLNode* n = XMLUtils::getChildNode(node, "RiskParticipationAgreementData");
293 QL_REQUIRE(n, "RiskParticipationAgreement::fromXML(): RiskParticipationAgreementData not found");
294 participationRate_ = XMLUtils::getChildValueAsDouble(n, "ParticipationRate", true);
295 protectionStart_ = parseDate(XMLUtils::getChildValue(n, "ProtectionStart", true));
296 protectionEnd_ = parseDate(XMLUtils::getChildValue(n, "ProtectionEnd", true));
297 creditCurveId_ = XMLUtils::getChildValue(n, "CreditCurveId", true);
298 issuerId_ = XMLUtils::getChildValue(n, "IssuerId", false); // defaults to empty string
299 settlesAccrual_ = XMLUtils::getChildValueAsBool(n, "SettlesAccrual", false); // defaults to true
300 tryParseReal(XMLUtils::getChildValue(n, "FixedRecoveryRate"), fixedRecoveryRate_); // defaults to null
301
302 underlying_.clear();
303 XMLNode* u = XMLUtils::getChildNode(n, "Underlying");
304 QL_REQUIRE(u, "RiskParticipationAgreement::fromXML(): Underlying not found");
305
306 vector<XMLNode*> nodes = XMLUtils::getChildrenNodes(u, "LegData");
307 for (auto const n : nodes) {
308 LegData ld;
309 ld.fromXML(n);
310 underlying_.push_back(ld);
311 }
312
313 if (auto tmp = XMLUtils::getChildNode(u, "OptionData")) {
314 optionData_ = OptionData();
315 (*optionData_).fromXML(tmp);
316 }
317
318 nakedOption_ = XMLUtils::getChildValueAsBool(u, "NakedOption", false, false);
319
320 if (auto tmp = XMLUtils::getChildNode(u, "TreasuryLockData")) {
321 tlockData_.fromXML(tmp);
322 }
323
324 protectionFee_.clear();
325 XMLNode* p = XMLUtils::getChildNode(n, "ProtectionFee");
326 QL_REQUIRE(p, "RiskParticipationAgreement::fromXML(): ProtectionFee not found");
327 vector<XMLNode*> nodes2 = XMLUtils::getChildrenNodes(p, "LegData");
328 for (auto const n : nodes2) {
329 LegData ld; // we do not allow ORE+ leg types anyway
330 ld.fromXML(n);
331 protectionFee_.push_back(ld);
332 }
333}

335XMLNode* RiskParticipationAgreement::toXML(XMLDocument& doc) {
336 XMLNode* node = Trade::toXML(doc);
337 XMLNode* n = doc.allocNode("RiskParticipationAgreementData");
338 XMLUtils::appendNode(node, n);
339 XMLUtils::addChild(doc, n, "ParticipationRate", participationRate_);
340 XMLUtils::addChild(doc, n, "ProtectionStart", ore::data::to_string(protectionStart_));
341 XMLUtils::addChild(doc, n, "ProtectionEnd", ore::data::to_string(protectionEnd_));
342 XMLUtils::addChild(doc, n, "CreditCurveId", creditCurveId_);
343 XMLUtils::addChild(doc, n, "IssuerId", issuerId_);
344 XMLUtils::addChild(doc, n, "SettlesAccrual", settlesAccrual_);
345 if (fixedRecoveryRate_ != Null<Real>())
346 XMLUtils::addChild(doc, n, "FixedRecoveryRate", fixedRecoveryRate_);
347 XMLNode* p = doc.allocNode("ProtectionFee");
348 XMLNode* u = doc.allocNode("Underlying");
349 XMLUtils::appendNode(n, p);
350 XMLUtils::appendNode(n, u);
351 if (optionData_)
352 XMLUtils::appendNode(u, (*optionData_).toXML(doc));
353 if (nakedOption_)
354 XMLUtils::addChild(doc, n, "NakedOption", nakedOption_);
355 for (auto& l : protectionFee_)
356 XMLUtils::appendNode(p, l.toXML(doc));
357 for (auto& l : underlying_)
358 XMLUtils::appendNode(u, l.toXML(doc));
359 if (!tlockData_.empty()) {
360 XMLUtils::appendNode(u, tlockData_.toXML(doc));
361 }
362 return node;
363}
