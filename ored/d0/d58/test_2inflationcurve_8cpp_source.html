85 // Test AU CPI bootstrap before and after index publication date in Oct 2020 (28 Oct 2020).
86 // Mirrors the conventions from the example in section 2.5 of AFMA Inflation Product Conventions, May 2017.
87 
88 vector<Date> auCpiTestDates{
89  Date(27, Oct, 2020), // Before Q3 CPI release date => swaps start on 15 Sep referencing Q2 CPI
90  Date(28, Oct, 2020), // On Q3 CPI release date => swap start depends on PublicationRoll
91  Date(29, Oct, 2020) // After Q3 CPI release date => swaps start on 15 Dec referencing Q3 CPI
92 };
93 
94 vector<string> publicationRolls{
95  "on", // Roll on the cpi release date
96  "after" // Roll after the cpi release date
97 };

99 BOOST_DATA_TEST_CASE(testAuCpiZcInflationCurve, bdata::make(auCpiTestDates) * bdata::make(publicationRolls),
100  asof, publicationRoll) {
101 
102  BOOST_TEST_MESSAGE("Testing AU CPI zero coupon inflation curve bootstrap on date " << io::iso_date(asof));
103 
104  // Create the market arguments. The fixing file on the release date depends on the publication roll setting.
105  string marketFile = "market_" + to_string(io::iso_date(asof)) + ".txt";
106  string fixingsFile = "fixings_" + to_string(io::iso_date(asof));
107  if (asof == Date(28, Oct, 2020)) {
108  fixingsFile += "_" + publicationRoll;
109  }
110  fixingsFile += ".txt";
111  string conventionsFile = "conventions_" + publicationRoll + ".xml";
112  TodaysMarketArguments tma(asof, "aucpi_zc", marketFile, fixingsFile, conventionsFile);
113 
114  // Check that the market builds without error.
115  boost::shared_ptr<TodaysMarket> market;
116  BOOST_REQUIRE_NO_THROW(market = boost::make_shared<TodaysMarket>(tma.asof, tma.todaysMarketParameters,
117  tma.loader, tma.curveConfigs, tma.conventions, false, true, false));
118 
119  // Portfolio containing 2 AU CPI zero coupon swaps, AUD 10M, that should price at 0, i.e. NPV < AUD 0.01.
120  // Similar to the fixing file, the helpers on the release date depend on the publication roll setting.
121  string portfolioFile = "aucpi_zc/portfolio_" + to_string(io::iso_date(asof));
122  if (asof == Date(28, Oct, 2020)) {
123  portfolioFile += "_" + publicationRoll;
124  }
125  portfolioFile += ".xml";
126 
127  boost::shared_ptr<EngineData> engineData = boost::make_shared<EngineData>();
128  engineData->fromFile(TEST_INPUT_FILE("aucpi_zc/pricingengine.xml"));
129  boost::shared_ptr<EngineFactory> factory = boost::make_shared<EngineFactory>(engineData, market);
130  boost::shared_ptr<Portfolio> portfolio = boost::make_shared<Portfolio>();
131  portfolio->load(TEST_INPUT_FILE(portfolioFile));
132  portfolio->build(factory);
133 
134  BOOST_CHECK_EQUAL(portfolio->size(), 2);
135  for (const auto& trade : portfolio->trades()) {
136  BOOST_CHECK_SMALL(trade->instrument()->NPV(), 0.01);
137  }
138 }
