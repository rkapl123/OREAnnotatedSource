87  protected:
88  virtual boost::shared_ptr<PricingEngine> engineImpl(const string& assetName, const Currency& ccy,
89  const AssetClass& assetClassUnderlying,
90  const Date& expiryDate) override {
91  bool brownianBridge = ore::data::parseBool(engineParameter("BrownianBridge", "", false, "true"));
92  bool antitheticVariate = ore::data::parseBool(engineParameter("AntitheticVariate", "", false, "true"));
93  bool controlVariate = ore::data::parseBool(engineParameter("ControlVariate", "", false, "true"));
94  Size requiredSamples = ore::data::parseInteger(engineParameter("RequiredSamples", "", false, "0"));
95  Real requiredTolerance = ore::data::parseReal(engineParameter("RequiredTolerance", "", false, "0"));
96  Size maxSamples = ore::data::parseInteger(engineParameter("MaxSamples", "", false, "0"));
97  BigNatural seed = ore::data::parseInteger(engineParameter("Seed", "", false, "123456"));
98 
99  // Check if values defaulted to 0, if so replace by Null<T>().
100  if (requiredSamples == 0)
101  requiredSamples = Null<Size>();
102  if (requiredTolerance == 0)
103  requiredTolerance = Null<Real>();
104  if (maxSamples == 0)
105  maxSamples = Null<Size>();
106  QL_REQUIRE(requiredSamples != QuantLib::Null<Size>() || requiredTolerance != QuantLib::Null<Real>(),
107  "RequiredSamples or RequiredTolerance must be set for engine MCDiscreteArithmeticAPEngine.");
108 
109  boost::shared_ptr<GeneralizedBlackScholesProcess> gbsp =
110  getBlackScholesProcess(assetName, ccy, assetClassUnderlying);
111  return boost::make_shared<MCDiscreteArithmeticAPEngine<LowDiscrepancy>>(gbsp, brownianBridge, antitheticVariate,
112  controlVariate, requiredSamples,
113  requiredTolerance, maxSamples, seed);
114  }
115  };
