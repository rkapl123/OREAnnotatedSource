39 CDSVolCurve::CDSVolCurve(Date asof, CDSVolatilityCurveSpec spec, const Loader& loader,
40  const CurveConfigurations& curveConfigs) {
41 
42  try {
43 
44  LOG("CDSVolCurve: start building CDS volatility structure with ID " << spec.curveConfigID());
45 
46  QL_REQUIRE(curveConfigs.hasCdsVolCurveConfig(spec.curveConfigID()), "No curve configuration found for CDS "
47  << "volatility curve spec with ID "
48  << spec.curveConfigID() << ".");
49  auto config = *curveConfigs.cdsVolCurveConfig(spec.curveConfigID());
50 
51  calendar_ = parseCalendar(config.calendar());
52  dayCounter_ = parseDayCounter(config.dayCounter());
53 
54  // Do different things depending on the type of volatility configured
55  boost::shared_ptr<VolatilityConfig> vc = config.volatilityConfig();
56  if (auto cvc = boost::dynamic_pointer_cast<ConstantVolatilityConfig>(vc)) {
57  buildVolatility(asof, config, *cvc, loader);
58  } else if (auto vcc = boost::dynamic_pointer_cast<VolatilityCurveConfig>(vc)) {
59  buildVolatility(asof, config, *vcc, loader);
60  } else if (auto vssc = boost::dynamic_pointer_cast<VolatilityStrikeSurfaceConfig>(vc)) {
61  buildVolatility(asof, config, *vssc, loader);
62  } else if (auto vdsc = boost::dynamic_pointer_cast<VolatilityDeltaSurfaceConfig>(vc)) {
63  QL_FAIL("CDSVolCurve does not support a VolatilityDeltaSurfaceConfig yet.");
64  } else if (auto vmsc = boost::dynamic_pointer_cast<VolatilityMoneynessSurfaceConfig>(vc)) {
65  QL_FAIL("CDSVolCurve does not support a VolatilityMoneynessSurfaceConfig yet.");
66  } else if (auto vapo = boost::dynamic_pointer_cast<VolatilityApoFutureSurfaceConfig>(vc)) {
67  QL_FAIL("VolatilityApoFutureSurfaceConfig does not make sense for CDSVolCurve.");
68  } else {
69  QL_FAIL("Unexpected VolatilityConfig in CDSVolatilityConfig");
70  }
71 
72  LOG("CDSVolCurve: finished building CDS volatility structure with ID " << spec.curveConfigID());
73 
74  } catch (exception& e) {
75  QL_FAIL("CDS volatility curve building for ID " << spec.curveConfigID() << " failed : " << e.what());
76  } catch (...) {
77  QL_FAIL("CDS volatility curve building for ID " << spec.curveConfigID() << " failed: unknown error");
78  }
79 }

81 void CDSVolCurve::buildVolatility(const Date& asof, const CDSVolatilityCurveConfig& vc,
82  const ConstantVolatilityConfig& cvc, const Loader& loader) {
83 
84  LOG("CDSVolCurve: start building constant volatility structure");
85 
86  // Loop over all market datums and find the single quote
87  // Return error if there are duplicates (this is why we do not use loader.get() method)
88  Real quoteValue = Null<Real>();
89  for (const boost::shared_ptr<MarketDatum>& md : loader.loadQuotes(asof)) {
90  if (md->asofDate() == asof && md->instrumentType() == MarketDatum::InstrumentType::INDEX_CDS_OPTION) {
91 
92  boost::shared_ptr<IndexCDSOptionQuote> q = boost::dynamic_pointer_cast<IndexCDSOptionQuote>(md);
93 
94  if (q->name() == cvc.quote()) {
95  TLOG("Found the constant volatility quote " << q->name());
96  QL_REQUIRE(quoteValue == Null<Real>(), "Duplicate quote found for quote with id " << cvc.quote());
97  quoteValue = q->quote()->value();
98  }
99  }
100  }
101  QL_REQUIRE(quoteValue != Null<Real>(), "Quote not found for id " << cvc.quote());
102 
103  DLOG("Creating BlackConstantVol structure");
104  vol_ = boost::make_shared<BlackConstantVol>(asof, calendar_, quoteValue, dayCounter_);
105 
106  LOG("CDSVolCurve: finished building constant volatility structure");
107 }

109 void CDSVolCurve::buildVolatility(const QuantLib::Date& asof, const CDSVolatilityCurveConfig& vc,
110  const VolatilityCurveConfig& vcc, const Loader& loader) {
111 
112  LOG("CDSVolCurve: start building 1-D volatility curve");
113 
114  // Must have at least one quote
115  QL_REQUIRE(vcc.quotes().size() > 0, "No quotes specified in config " << vc.curveID());
116 
117  // Check if we are using a regular expression to select the quotes for the curve. If we are, the quotes should
118  // contain exactly one element.
119  auto wildcard = getUniqueWildcard(vcc.quotes());
120 
121  // curveData will be populated with the expiry dates and volatility values.
122  map<Date, Real> curveData;
123 
124  // Different approaches depending on whether we are using a regex or searching for a list of explicit quotes.
125  if (wildcard) {
126 
127  DLOG("Have single quote with pattern " << (*wildcard).regex());
128 
129  // Loop over quotes and process CDS option quotes matching pattern on asof
130  for (const boost::shared_ptr<MarketDatum>& md : loader.loadQuotes(asof)) {
131 
132  // Go to next quote if the market data point's date does not equal our asof
133  if (md->asofDate() != asof)
134  continue;
135 
136  auto q = boost::dynamic_pointer_cast<IndexCDSOptionQuote>(md);
137  if (q && (*wildcard).matches(q->name())) {
138 
139  TLOG("The quote " << q->name() << " matched the pattern");
140 
141  Date expiryDate = getExpiry(asof, q->expiry());
142  if (expiryDate > asof) {
143  // Add the quote to the curve data
144  QL_REQUIRE(curveData.count(expiryDate) == 0, "Duplicate quote for the expiry date "
145  << io::iso_date(expiryDate)
146  << " provided by CDS volatility config "
147  << vc.curveID());
148  curveData[expiryDate] = q->quote()->value();
149 
150  TLOG("Added quote " << q->name() << ": (" << io::iso_date(expiryDate) << "," << fixed
151  << setprecision(9) << q->quote()->value() << ")");
152  }
153  }
154  }
155 
156  // Check that we have quotes in the end
157  QL_REQUIRE(curveData.size() > 0, "No quotes found matching regular expression " << vcc.quotes()[0]);
158 
159  } else {
160 
161  DLOG("Have " << vcc.quotes().size() << " explicit quotes");
162 
163  // Loop over quotes and process CDS option quotes that are explicitly specified in the config
164  for (const boost::shared_ptr<MarketDatum>& md : loader.loadQuotes(asof)) {
165 
166  // Go to next quote if the market data point's date does not equal our asof
167  if (md->asofDate() != asof)
168  continue;
169 
170  if (auto q = boost::dynamic_pointer_cast<IndexCDSOptionQuote>(md)) {
171 
172  // Find quote name in configured quotes.
173  auto it = find(vcc.quotes().begin(), vcc.quotes().end(), q->name());
174 
175  if (it != vcc.quotes().end()) {
176  TLOG("Found the configured quote " << q->name());
177 
178  Date expiryDate = getExpiry(asof, q->expiry());
179  QL_REQUIRE(expiryDate > asof, "CDS volatility quote '" << q->name() << "' has expiry in the past ("
180  << io::iso_date(expiryDate) << ")");
181  QL_REQUIRE(curveData.count(expiryDate) == 0, "Duplicate quote for the date "
182  << io::iso_date(expiryDate)
183  << " provided by CDS volatility config "
184  << vc.curveID());
185  curveData[expiryDate] = q->quote()->value();
186 
187  TLOG("Added quote " << q->name() << ": (" << io::iso_date(expiryDate) << "," << fixed
188  << setprecision(9) << q->quote()->value() << ")");
189  }
190  }
191  }
192 
193  // Check that we have found all of the explicitly configured quotes
194  QL_REQUIRE(curveData.size() == vcc.quotes().size(), "Found " << curveData.size() << " quotes, but "
195  << vcc.quotes().size()
196  << " quotes were given in config.");
197  }
198 
199  // Create the dates and volatility vector
200  vector<Date> dates;
201  vector<Volatility> volatilities;
202  for (const auto& datum : curveData) {
203  dates.push_back(datum.first);
204  volatilities.push_back(datum.second);
205  TLOG("Added data point (" << io::iso_date(dates.back()) << "," << fixed << setprecision(9)
206  << volatilities.back() << ")");
207  }
208 
209  DLOG("Creating BlackVarianceCurve object.");
210  auto tmp =
211  boost::make_shared<BlackVarianceCurve>(asof, dates, volatilities, dayCounter_, vcc.enforceMontoneVariance());
212 
213  // Set the interpolation.
214  if (vcc.interpolation() == "Linear") {
215  DLOG("Interpolation set to Linear.");
216  } else if (vcc.interpolation() == "Cubic") {
217  DLOG("Setting interpolation to Cubic.");
218  tmp->setInterpolation<Cubic>();
219  } else if (vcc.interpolation() == "LogLinear") {
220  DLOG("Setting interpolation to LogLinear.");
221  tmp->setInterpolation<LogLinear>();
222  } else {
223  DLOG("Interpolation " << vcc.interpolation() << " not recognised so leaving it Linear.");
224  }
225 
226  // Set the volatility_ member after we have possibly updated the interpolation.
227  vol_ = tmp;
228 
229  // Set the extrapolation
230  if (parseExtrapolation(vcc.extrapolation()) == Extrapolation::Flat) {
231  DLOG("Enabling BlackVarianceCurve flat volatility extrapolation.");
232  vol_->enableExtrapolation();
233  } else if (parseExtrapolation(vcc.extrapolation()) == Extrapolation::None) {
234  DLOG("Disabling BlackVarianceCurve extrapolation.");
235  vol_->disableExtrapolation();
236  } else if (parseExtrapolation(vcc.extrapolation()) == Extrapolation::UseInterpolator) {
237  DLOG("BlackVarianceCurve does not support using interpolator for extrapolation "
238  << "so default to flat volatility extrapolation.");
239  vol_->enableExtrapolation();
240  } else {
241  DLOG("Unexpected extrapolation so default to flat volatility extrapolation.");
242  vol_->enableExtrapolation();
243  }
244 
245  LOG("CDSVolCurve: finished building 1-D volatility curve");
246 }

248 void CDSVolCurve::buildVolatility(const Date& asof, CDSVolatilityCurveConfig& vc,
249  const VolatilityStrikeSurfaceConfig& vssc, const Loader& loader) {
250 
251  LOG("CDSVolCurve: start building 2-D volatility absolute strike surface");
252 
253  // We are building a cds volatility surface here of the form expiry vs strike where the strikes are absolute
254  // numbers. The list of expiries may be explicit or contain a single wildcard character '*'. Similarly, the list
255  // of strikes may be explicit or contain a single wildcard character '*'. So, we have four options here which
256  // ultimately lead to two different types of surface being built:
257  // 1. explicit strikes and explicit expiries => BlackVarianceSurface
258  // 2. wildcard strikes and/or wildcard expiries (3 combinations) => BlackVarianceSurfaceSparse
259 
260  bool expWc = false;
261  if (find(vssc.expiries().begin(), vssc.expiries().end(), "*") != vssc.expiries().end()) {
262  expWc = true;
263  QL_REQUIRE(vssc.expiries().size() == 1, "Wild card expiry specified but more expiries also specified.");
264  DLOG("Have expiry wildcard pattern " << vssc.expiries()[0]);
265  }
266 
267  bool strkWc = false;
268  if (find(vssc.strikes().begin(), vssc.strikes().end(), "*") != vssc.strikes().end()) {
269  strkWc = true;
270  QL_REQUIRE(vssc.strikes().size() == 1, "Wild card strike specified but more strikes also specified.");
271  DLOG("Have strike wildcard pattern " << vssc.strikes()[0]);
272  }
273 
274  // If we do not have a strike wild card, we expect a list of absolute strike values
275  vector<Real> configuredStrikes;
276  if (!strkWc) {
277  // Parse the list of absolute strikes
278  configuredStrikes = parseVectorOfValues<Real>(vssc.strikes(), &parseReal);
279  sort(configuredStrikes.begin(), configuredStrikes.end(), [](Real x, Real y) { return !close(x, y) && x < y; });
280  QL_REQUIRE(adjacent_find(configuredStrikes.begin(), configuredStrikes.end(),
281  [](Real x, Real y) { return close(x, y); }) == configuredStrikes.end(),
282  "The configured strikes contain duplicates");
283  DLOG("Parsed " << configuredStrikes.size() << " unique configured absolute strikes");
284  }
285 
286  // If we do not have an expiry wild card, parse the configured expiries.
287  vector<boost::shared_ptr<Expiry>> configuredExpiries;
288  if (!expWc) {
289  // Parse the list of expiry strings.
290  for (const string& strExpiry : vssc.expiries()) {
291  configuredExpiries.push_back(parseExpiry(strExpiry));
292  }
293  DLOG("Parsed " << configuredExpiries.size() << " unique configured expiries");
294  }
295 
296  // If there are no wildcard strikes or wildcard expiries, delegate to buildVolatilityExplicit.
297  if (!expWc && !strkWc) {
298  buildVolatilityExplicit(asof, vc, vssc, loader, configuredStrikes);
299  return;
300  }
301 
302  DLOG("Expiries and or strikes have been configured via wildcards so building a "
303  << "wildcard based absolute strike surface");
304 
305  // Store aligned strikes, expiries and vols found via wildcard lookup
306  vector<Real> strikes;
307  vector<Date> expiries;
308  vector<Volatility> vols;
309  Size quotesAdded = 0;
310 
311  // Loop over quotes and process any CDS option quote that matches a wildcard
312  for (const boost::shared_ptr<MarketDatum>& md : loader.loadQuotes(asof)) {
313 
314  // Go to next quote if the market data point's date does not equal our asof.
315  if (md->asofDate() != asof)
316  continue;
317 
318  // Go to next quote if not a CDS option quote.
319  auto q = boost::dynamic_pointer_cast<IndexCDSOptionQuote>(md);
320  if (!q)
321  continue;
322 
323  // Go to next quote if index name in the quote does not match the cds vol configuration name.
324  if (vc.curveID() != q->indexName() && vc.quoteName() != q->indexName())
325  continue;
326 
327  // This surface is for absolute strikes only.
328  auto strike = boost::dynamic_pointer_cast<AbsoluteStrike>(q->strike());
329  if (!strike)
330  continue;
331 
332  // Make sure that the CDS index option quote term matches the configured term
333  if (q->indexTerm() != vc.nameTerm().second)
334  continue;
335 
336  // If we have been given a list of explicit expiries, check that the quote matches one of them.
337  // Move to the next quote if it does not.
338  if (!expWc) {
339  auto expiryIt = find_if(configuredExpiries.begin(), configuredExpiries.end(),
340  [&q](boost::shared_ptr<Expiry> e) { return *e == *q->expiry(); });
341  if (expiryIt == configuredExpiries.end())
342  continue;
343  }
344 
345  // If we have been given a list of explicit strikes, check that the quote matches one of them.
346  // Move to the next quote if it does not.
347  if (!strkWc) {
348  auto strikeIt = find_if(configuredStrikes.begin(), configuredStrikes.end(),
349  [&strike](Real s) { return close(s, strike->strike()); });
350  if (strikeIt == configuredStrikes.end())
351  continue;
352  }
353 
354  // If we make it here, add the data to the aligned vectors.
355  expiries.push_back(getExpiry(asof, q->expiry()));
356  strikes.push_back(strike->strike() / vc.strikeFactor());
357  vols.push_back(q->quote()->value());
358  quotesAdded++;
359 
360  TLOG("Added quote " << q->name() << ": (" << io::iso_date(expiries.back()) << "," << fixed << setprecision(9)
361  << strikes.back() << "," << vols.back() << ")");
362  }
363 
364  LOG("CDSVolCurve: added " << quotesAdded << " quotes in building wildcard based absolute strike surface.");
365  QL_REQUIRE(quotesAdded > 0, "No quotes loaded for " << vc.curveID());
366 
367  // Set the strike extrapolation which only matters if extrapolation is turned on for the whole surface.
368  bool flatStrikeExtrap = true;
369  bool flatTimeExtrap = true;
370  if (vssc.extrapolation()) {
371 
372  auto strikeExtrapType = parseExtrapolation(vssc.strikeExtrapolation());
373  if (strikeExtrapType == Extrapolation::UseInterpolator) {
374  DLOG("Strike extrapolation switched to using interpolator.");
375  flatStrikeExtrap = false;
376  } else if (strikeExtrapType == Extrapolation::None) {
377  DLOG("Strike extrapolation cannot be turned off on its own so defaulting to flat.");
378  } else if (strikeExtrapType == Extrapolation::Flat) {
379  DLOG("Strike extrapolation has been set to flat.");
380  } else {
381  DLOG("Strike extrapolation " << strikeExtrapType << " not expected so default to flat.");
382  }
383 
384  auto timeExtrapType = parseExtrapolation(vssc.timeExtrapolation());
385  if (timeExtrapType == Extrapolation::UseInterpolator) {
386  DLOG("Time extrapolation switched to using interpolator.");
387  flatTimeExtrap = false;
388  } else if (timeExtrapType == Extrapolation::None) {
389  DLOG("Time extrapolation cannot be turned off on its own so defaulting to flat.");
390  } else if (timeExtrapType == Extrapolation::Flat) {
391  DLOG("Time extrapolation has been set to flat.");
392  } else {
393  DLOG("Time extrapolation " << timeExtrapType << " not expected so default to flat.");
394  }
395 
396  } else {
397  DLOG("Extrapolation is turned off for the whole surface so the time and"
398  << " strike extrapolation settings are ignored");
399  }
400 
401  DLOG("Creating the BlackVarianceSurfaceSparse object");
402  vol_ = boost::make_shared<BlackVarianceSurfaceSparse>(asof, calendar_, expiries, strikes, vols, dayCounter_,
403  flatStrikeExtrap, flatStrikeExtrap, flatTimeExtrap);
404 
405  DLOG("Setting BlackVarianceSurfaceSparse extrapolation to " << to_string(vssc.extrapolation()));
406  vol_->enableExtrapolation(vssc.extrapolation());
407 
408  LOG("CDSVolCurve: finished building 2-D volatility absolute strike surface");
409 }

411 void CDSVolCurve::buildVolatilityExplicit(const Date& asof, CDSVolatilityCurveConfig& vc,
412  const VolatilityStrikeSurfaceConfig& vssc, const Loader& loader,
413  const vector<Real>& configuredStrikes) {
414 
415  LOG("CDSVolCurve: start building 2-D volatility absolute strike surface with explicit strikes and expiries");
416 
417  // Map to hold the rows of the CDS volatility matrix. The keys are the expiry dates and the values are the
418  // vectors of volatilities, one for each configured strike.
419  map<Date, vector<Real>> surfaceData;
420 
421  // Count the number of quotes added. We check at the end that we have added all configured quotes.
422  Size quotesAdded = 0;
423 
424  // Loop over quotes and process CDS option quotes that have been requested
425  for (const boost::shared_ptr<MarketDatum>& md : loader.loadQuotes(asof)) {
426 
427  // Go to next quote if the market data point's date does not equal our asof.
428  if (md->asofDate() != asof)
429  continue;
430 
431  // Go to next quote if not a CDS option quote.
432  auto q = boost::dynamic_pointer_cast<IndexCDSOptionQuote>(md);
433  if (!q)
434  continue;
435 
436  // This surface is for absolute strikes only.
437  auto strike = boost::dynamic_pointer_cast<AbsoluteStrike>(q->strike());
438  if (!strike)
439  continue;
440 
441  // If the quote is not in the configured quotes continue
442  auto it = find(vc.quotes().begin(), vc.quotes().end(), q->name());
443  if (it == vc.quotes().end())
444  continue;
445 
446  // Process the quote
447  Date eDate = getExpiry(asof, q->expiry());
448 
449  // Position of quote in vector of strikes
450  auto strikeIt = find_if(configuredStrikes.begin(), configuredStrikes.end(),
451  [&strike](Real s) { return close(s, strike->strike()); });
452  Size pos = distance(configuredStrikes.begin(), strikeIt);
453  QL_REQUIRE(
454  pos < configuredStrikes.size(),
455  "The quote '" << q->name()
456  << "' is in the list of configured quotes but does not match any of the configured strikes");
457 
458  // Add quote to surface
459  if (surfaceData.count(eDate) == 0)
460  surfaceData[eDate] = vector<Real>(configuredStrikes.size(), Null<Real>());
461 
462  QL_REQUIRE(surfaceData[eDate][pos] == Null<Real>(),
463  "Quote " << q->name() << " provides a duplicate quote for the date " << io::iso_date(eDate)
464  << " and the strike " << configuredStrikes[pos]);
465  surfaceData[eDate][pos] = q->quote()->value();
466  quotesAdded++;
467 
468  TLOG("Added quote " << q->name() << ": (" << io::iso_date(eDate) << "," << fixed << setprecision(9)
469  << configuredStrikes[pos] << "," << q->quote()->value() << ")");
470  }
471 
472  LOG("CDSVolCurve: added " << quotesAdded << " quotes in building explicit absolute strike surface.");
473 
474  QL_REQUIRE(vc.quotes().size() == quotesAdded,
475  "Found " << quotesAdded << " quotes, but " << vc.quotes().size() << " quotes required by config.");
476 
477  // Set up the BlackVarianceSurface. Note that the rows correspond to strikes and that the columns correspond to
478  // the expiry dates in the matrix that is fed to the BlackVarianceSurface ctor.
479  vector<Date> expiryDates;
480  Matrix volatilities = Matrix(configuredStrikes.size(), surfaceData.size());
481  Size expiryIdx = 0;
482  for (const auto& expiryRow : surfaceData) {
483  expiryDates.push_back(expiryRow.first);
484  for (Size i = 0; i < configuredStrikes.size(); i++) {
485  volatilities[i][expiryIdx] = expiryRow.second[i];
486  }
487  expiryIdx++;
488  }
489 
490  // Trace log the surface
491  TLOG("Explicit strike surface grid points:");
492  TLOG("expiry,strike,volatility");
493  for (Size i = 0; i < volatilities.rows(); i++) {
494  for (Size j = 0; j < volatilities.columns(); j++) {
495  TLOG(io::iso_date(expiryDates[j])
496  << "," << fixed << setprecision(9) << configuredStrikes[i] << "," << volatilities[i][j]);
497  }
498  }
499 
500  // Set the strike extrapolation which only matters if extrapolation is turned on for the whole surface.
501  // BlackVarianceSurface time extrapolation is hard-coded to constant in volatility.
502  BlackVarianceSurface::Extrapolation strikeExtrap = BlackVarianceSurface::ConstantExtrapolation;
503  if (vssc.extrapolation()) {
504 
505  auto strikeExtrapType = parseExtrapolation(vssc.strikeExtrapolation());
506  if (strikeExtrapType == Extrapolation::UseInterpolator) {
507  DLOG("Strike extrapolation switched to using interpolator.");
508  strikeExtrap = BlackVarianceSurface::InterpolatorDefaultExtrapolation;
509  } else if (strikeExtrapType == Extrapolation::None) {
510  DLOG("Strike extrapolation cannot be turned off on its own so defaulting to flat.");
511  } else if (strikeExtrapType == Extrapolation::Flat) {
512  DLOG("Strike extrapolation has been set to flat.");
513  } else {
514  DLOG("Strike extrapolation " << strikeExtrapType << " not expected so default to flat.");
515  }
516 
517  auto timeExtrapType = parseExtrapolation(vssc.timeExtrapolation());
518  if (timeExtrapType != Extrapolation::Flat) {
519  DLOG("BlackVarianceSurface only supports flat volatility extrapolation in the time direction");
520  }
521  } else {
522  DLOG("Extrapolation is turned off for the whole surface so the time and"
523  << " strike extrapolation settings are ignored");
524  }
525 
526  // Divide the configured strikes by the strike factor before using in surface
527  vector<Real> strikes = configuredStrikes;
528  for (Real& strike : strikes) {
529  strike /= vc.strikeFactor();
530  }
531 
532  DLOG("Creating BlackVarianceSurface object");
533  auto tmp = boost::make_shared<BlackVarianceSurface>(asof, calendar_, expiryDates, strikes, volatilities,
534  dayCounter_, strikeExtrap, strikeExtrap);
535 
536  // Set the interpolation if configured properly. The default is Bilinear.
537  if (!(vssc.timeInterpolation() == "Linear" && vssc.strikeInterpolation() == "Linear")) {

547  }
548  }
549

551  vol_ = tmp;
552

555

557  << "surface with explicit strikes and expiries");
