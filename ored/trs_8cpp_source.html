48void TRS::ReturnData::fromXML(XMLNode* node) {
49 XMLUtils::checkNode(node, "ReturnData");
50 payer_ = XMLUtils::getChildValueAsBool(node, "Payer", true);
51 currency_ = XMLUtils::getChildValue(node, "Currency", true);
52 scheduleData_.fromXML(XMLUtils::getChildNode(node, "ScheduleData"));
53 observationLag_ = XMLUtils::getChildValue(node, "ObservationLag", false);
54 observationConvention_ = XMLUtils::getChildValue(node, "ObservationConvention", false);
55 observationCalendar_ = XMLUtils::getChildValue(node, "ObservationCalendar", false);
56 paymentLag_ = XMLUtils::getChildValue(node, "PaymentLag", false);
57 paymentConvention_ = XMLUtils::getChildValue(node, "PaymentConvention", false);
58 paymentCalendar_ = XMLUtils::getChildValue(node, "PaymentCalendar", false);
59 paymentDates_ = XMLUtils::getChildrenValues(node, "PaymentDates", "PaymentDate", false);
60 initialPrice_ = Null<Real>();
61 if (auto n = XMLUtils::getChildNode(node, "InitialPrice")) {
62 initialPrice_ = parseReal(XMLUtils::getNodeValue(n));
63 }
64 initialPriceCurrency_ = XMLUtils::getChildValue(node, "InitialPriceCurrency");
65 payUnderlyingCashFlowsImmediately_ = boost::none;
66 if (auto n = XMLUtils::getChildNode(node, "PayUnderlyingCashFlowsImmediately")) {
67 payUnderlyingCashFlowsImmediately_ = parseBool(XMLUtils::getNodeValue(n));
68 }
69 fxTerms_ = XMLUtils::getChildrenValues(node, "FXTerms", "FXIndex", false);
70}

147std::map<AssetClass, std::set<std::string>>
148TRS::underlyingIndices(const boost::shared_ptr<ReferenceDataManager>& referenceDataManager) const {
149 std::map<AssetClass, std::set<std::string>> result;
150 for (Size i = 0; i < underlying_.size(); ++i) {
151 QL_REQUIRE(underlying_[i], "TRS::underlyingIndices(): underlying trade is null");
152 // a builder might update the underlying (e.g. promote it from bond to convertible bond)
153 if (underlyingDerivativeId_[i].empty()) {
154 for (auto const& b : TrsUnderlyingBuilderFactory::instance().getBuilders()) {
155 b.second->updateUnderlying(referenceDataManager, underlying_[i], id());
156 }
157 }
158 for (auto const& tmp : underlying_[i]->underlyingIndices(referenceDataManager))
159 result[tmp.first].insert(tmp.second.begin(), tmp.second.end());
160 }
161 return result;
162}

164void TRS::fromXML(XMLNode* node) {
165 Trade::fromXML(node);
166
167 // set id early since we use it below to set the underlying trade's id
168 this->id() = XMLUtils::getAttribute(node, "id");
169
170 // trs data node
171 XMLNode* dataNode = XMLUtils::getChildNode(node, tradeType_ + "Data");
172 QL_REQUIRE(dataNode, tradeType_ + "Data node required");
173
174 // read underlying data
175 XMLNode* underlyingDataNode = XMLUtils::getChildNode(dataNode, "UnderlyingData");
176 QL_REQUIRE(underlyingDataNode, "UnderlyingData node required");
177 std::vector<XMLNode*> underlyingTradeNodes = XMLUtils::getChildrenNodes(underlyingDataNode, "Trade");
178 std::vector<XMLNode*> underlyingTradeNodes2 = XMLUtils::getChildrenNodes(underlyingDataNode, "Derivative");
179 QL_REQUIRE(!underlyingTradeNodes.empty() || !underlyingTradeNodes2.empty(),
180 "at least one 'Trade' or 'Derivative' node required");
181 Size underlyingCounter = 0;
182 underlying_.clear();
183 underlyingDerivativeId_.clear();
184 for (auto const n : underlyingTradeNodes) {
185 std::string tradeType = XMLUtils::getChildValue(n, "TradeType", true);
186 boost::shared_ptr<Trade> u;
187 try {
188 u = TradeFactory::instance().build(tradeType);
189 } catch (const std::exception& e) {
190 QL_FAIL("Failed for build TRS underlying trade # " << underlyingCounter + 1 << ": " << e.what());
191 }
192 u->id() = this->id() + "_underlying" +
193 (underlyingTradeNodes.size() > 1 ? "_" + std::to_string(underlyingCounter++) : "");
194 u->fromXML(n);
195 underlyingDerivativeId_.push_back(std::string());
196 underlying_.push_back(u);
197 }
198 for (auto const n : underlyingTradeNodes2) {
199 underlyingDerivativeId_.push_back(XMLUtils::getChildValue(n, "Id", true));
200 auto t = XMLUtils::getChildNode(n, "Trade");
201 QL_REQUIRE(t != nullptr, "expected 'Trade' node under 'Derivative' node");
202 std::string tradeType = XMLUtils::getChildValue(t, "TradeType", true);
203 auto u = TradeFactory::instance().build(tradeType);
204 QL_REQUIRE(u, "No trade builder found for TRS derivative trade type '"
205 << tradeType << "' when processing underlying trade #" << (underlyingCounter + 1));
206 u->id() = this->id() + "_underlying" +
207 (underlyingTradeNodes.size() > 1 ? "_" + std::to_string(underlyingCounter++) : "");
208 u->fromXML(t);
209 underlying_.push_back(u);
210 }
211
212 // read return data
213 XMLNode* returnDataNode = XMLUtils::getChildNode(dataNode, "ReturnData");
214 returnData_.fromXML(returnDataNode);
215
216 // read funding data
217 XMLNode* fundingDataNode = XMLUtils::getChildNode(dataNode, "FundingData");
218 if (fundingDataNode)
219 fundingData_.fromXML(fundingDataNode);
220 else
221 fundingData_ = FundingData();
222
223 // read additional cashflow data
224 XMLNode* additionalCashflowDataNode = XMLUtils::getChildNode(dataNode, "AdditionalCashflowData");
225 if (additionalCashflowDataNode)
226 additionalCashflowData_.fromXML(additionalCashflowDataNode);
227 else
228 additionalCashflowData_ = AdditionalCashflowData();
229}

231XMLNode* TRS::toXML(XMLDocument& doc) {
232 XMLNode* node = Trade::toXML(doc);
233 XMLNode* dataNode = doc.allocNode(tradeType_ + "Data");
234 XMLUtils::appendNode(node, dataNode);
235
236 XMLNode* underlyingDataNode = doc.allocNode("UnderlyingData");
237 XMLUtils::appendNode(dataNode, underlyingDataNode);
238
239 for (Size i = 0; i < underlying_.size(); ++i) {
240 if (underlyingDerivativeId_[i].empty()) {
241 XMLUtils::appendNode(underlyingDataNode, underlying_[i]->toXML(doc));
242 } else {
243 auto d = XMLUtils::addChild(doc, underlyingDataNode, "Derivative");
244 XMLUtils::addChild(doc, d, "Id", underlyingDerivativeId_[i]);
245 XMLUtils::appendNode(d, underlying_[i]->toXML(doc));
246 }
247 }
248
249 XMLUtils::appendNode(dataNode, returnData_.toXML(doc));
250 if (!fundingData_.legData().empty())
251 XMLUtils::appendNode(dataNode, fundingData_.toXML(doc));
252 if (additionalCashflowData_.legData().concreteLegData())
253 XMLUtils::appendNode(dataNode, additionalCashflowData_.toXML(doc));
254
255 return node;
256}

258boost::shared_ptr<QuantExt::FxIndex>
259TRS::getFxIndex(const boost::shared_ptr<Market> market, const std::string& configuration, const std::string& domestic,
260 const std::string& foreign, std::map<std::string, boost::shared_ptr<QuantExt::FxIndex>>& fxIndices,
261 std::set<std::string>& missingFxIndexPairs) const {
262 if (domestic == foreign)
263 return nullptr;
264 std::set<std::string> requiredCcys = {domestic, foreign};
265 for (auto const& f : returnData_.fxTerms()) {
266 auto fx = parseFxIndex(f);
267 std::set<std::string> indexCcys = {fx->sourceCurrency().code(), fx->targetCurrency().code()};
268 if (requiredCcys == indexCcys) {
269 auto h = fxIndices.find(f);
270 if (h != fxIndices.end())
271 return h->second;
272 DLOG("setting up fx index for domestic=" << domestic << " foreign=" << foreign);
273 auto fx = buildFxIndex(f, domestic, foreign, market, configuration, false);
274 fxIndices[f] = fx;
275 return fx;
276 }
277 }
278
279 // build a fx index, so that the processing can continue, but add to the error messages,
280 // which - if not empty - will fail the trade build eventually
281
282 std::string f("FX-GENERIC-" + domestic + "-" + foreign);
283 auto fx = buildFxIndex(f, domestic, foreign, market, configuration, false);
284 fxIndices[f] = fx;
285 missingFxIndexPairs.insert(domestic + foreign);
286 return fx;
287}

289void TRS::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
290
291 DLOG("TRS::build() called for id = " << id());
292
293 // clear trade members
294
295 reset();
296
297 creditRiskCurrency_.clear();
298 creditQualifierMapping_.clear();
299
300 // checks
301
302 std::set<bool> fundingLegPayers;
303 std::set<std::string> fundingCurrencies;
304
305 bool fundingLegPayer = !(returnData_.payer());
306 std::string fundingCurrency = returnData_.currency();
307
308 for (auto const& l : fundingData_.legData()) {
309 fundingLegPayer = l.isPayer();
310 fundingCurrency = l.currency();
311 fundingLegPayers.insert(fundingLegPayer);
312 fundingCurrencies.insert(fundingCurrency);
313 }
314
315 QL_REQUIRE(fundingLegPayers.size() <= 1, "funding leg payer flags must match");
316 QL_REQUIRE(fundingCurrencies.size() <= 1, "funding leg currencies must match");
317
318 // a builder might update the underlying (e.g. promote it from bond to convertible bond)
319
320 for (Size i = 0; i < underlying_.size(); ++i) {
321 if (underlyingDerivativeId_[i].empty()) {
322 for (auto const& b : TrsUnderlyingBuilderFactory::instance().getBuilders()) {
323 b.second->updateUnderlying(engineFactory->referenceData(), underlying_[i], id());
324 }
325 }
326 }
327
328 // build underlying trade, add required fixing from there to this trade
329
330 for (Size i = 0; i < underlying_.size(); ++i) {
331 DLOG("build underlying trade #" << (i + 1) << " of type '" << underlying_[i]->tradeType() << "'");
332 underlying_[i]->reset();
333 underlying_[i]->build(engineFactory);
334 requiredFixings_.addData(underlying_[i]->requiredFixings());
335 }
336
337 // we use dirty prices, so we need accrued amounts in the past
338 requiredFixings_.unsetPayDates();
339
340 // build return leg valuation and payment date vectors
341
342 DLOG("build valuation and payment dates vectors");
343
344 std::vector<Date> valuationDates, paymentDates;
345
346 QuantLib::Schedule schedule = makeSchedule(returnData_.scheduleData());
347 QL_REQUIRE(schedule.dates().size() >= 2, "at least two dates required in return schedule");
348
349 Calendar observationCalendar = parseCalendar(returnData_.observationCalendar());
350 BusinessDayConvention observationConvention = returnData_.observationConvention().empty()
351 ? Unadjusted
352 : parseBusinessDayConvention(returnData_.observationConvention());
353 Period observationLag = returnData_.observationLag().empty() ? 0 * Days : parsePeriod(returnData_.observationLag());
354
355 Calendar paymentCalendar = parseCalendar(returnData_.paymentCalendar());
356 BusinessDayConvention paymentConvention = returnData_.paymentConvention().empty()
357 ? Unadjusted
358 : parseBusinessDayConvention(returnData_.paymentConvention());
359 PaymentLag paymentLag = parsePaymentLag(returnData_.paymentLag());
360 Period plPeriod = boost::apply_visitor(PaymentLagPeriod(), paymentLag);
361
362 for (auto const& d : schedule.dates()) {
363 valuationDates.push_back(observationCalendar.advance(d, -observationLag, observationConvention));
364 if (d != schedule.dates().front())
365 paymentDates.push_back(paymentCalendar.advance(d, plPeriod, paymentConvention));
366 }
367
368 if (!returnData_.paymentDates().empty()) {
369 paymentDates.clear();
370 QL_REQUIRE(returnData_.paymentDates().size() + 1 == valuationDates.size(),
371 "paymentDates size (" << returnData_.paymentDates().size() << ") does no match valuatioDates size ("
372 << valuationDates.size() << ") minus 1");
373 for (auto const& s : returnData_.paymentDates())
374 paymentDates.push_back(parseDate(s));
375 }
376
377 DLOG("valuation schedule:");
378 for (auto const& d : valuationDates)
379 DLOG(ore::data::to_string(d));
380
381 DLOG("payment schedule:");
382 for (auto const& d : paymentDates)
383 DLOG(ore::data::to_string(d));
384
385 // build indices corresponding to underlying trades and populate necessary data
386
387 std::map<std::string, double> indexNamesAndQty;
388 std::map<std::string, boost::shared_ptr<QuantExt::FxIndex>> initialFxIndices, fxIndices, fxIndicesDummy;
389
390 // get fx indices for conversion return and add cf ccy to funding ccy
391
392 std::set<std::string> missingFxIndexPairs;
393
394 auto fxIndexReturn = getFxIndex(engineFactory->market(), engineFactory->configuration(MarketContext::pricing),
395 returnData_.currency(), fundingCurrency, initialFxIndices, missingFxIndexPairs);
396 auto fxIndexAdditionalCashflows =
397 additionalCashflowData_.legData().currency().empty()
398 ? fxIndexReturn
399 : getFxIndex(engineFactory->market(), engineFactory->configuration(MarketContext::pricing),
400 additionalCashflowData_.legData().currency(), fundingCurrency, fxIndicesDummy,
401 missingFxIndexPairs);
402
403 Real initialPrice = returnData_.initialPrice();
404
405 std::vector<boost::shared_ptr<QuantLib::Index>> underlyingIndex(underlying_.size(), nullptr);
406 std::vector<Real> underlyingMultiplier(underlying_.size(), Null<Real>());
407 std::vector<std::string> assetCurrency(underlying_.size(), fundingCurrency);
408 std::vector<boost::shared_ptr<QuantExt::FxIndex>> fxIndexAsset(underlying_.size(), nullptr);
409
410 maturity_ = Date::minDate();
411
412 for (Size i = 0; i < underlying_.size(); ++i) {
413
414 DLOG("build underlying index for underlying #" << (i + 1));
415
416 std::string localCreditRiskCurrency;
417 Date localMaturity;
418 std::map<std::string, double> localIndexNamesAndQuantities;
419 std::map<std::string, boost::shared_ptr<QuantExt::FxIndex>> localFxIndices = initialFxIndices;
420 Real dummyInitialPrice = 1.0; // initial price is only updated if we have one underlying
421
422 std::vector<Leg> returnLegs;
423 auto builder = TrsUnderlyingBuilderFactory::instance().getBuilder(
424 underlyingDerivativeId_[i].empty() ? underlying_[i]->tradeType() : "Derivative");
425 builder->build(id(), underlying_[i], valuationDates, paymentDates, fundingCurrency, engineFactory,
426 underlyingIndex[i], underlyingMultiplier[i], localIndexNamesAndQuantities, localFxIndices,
427 underlying_.size() == 1 ? initialPrice : dummyInitialPrice, assetCurrency[i],
428 localCreditRiskCurrency, creditQualifierMapping_, localMaturity,
429 std::bind(&TRS::getFxIndex, this, std::placeholders::_1, std::placeholders::_2,
430 std::placeholders::_3, std::placeholders::_4, std::placeholders::_5,
431 std::ref(missingFxIndexPairs)),
432 underlyingDerivativeId_[i], requiredFixings_, returnLegs);
433
434 addTRSRequiredFixings(requiredFixings_, returnLegs, fxIndexReturn);
435
436 // update global credit risk currency
437
438 if (creditRiskCurrency_.empty()) {
439 creditRiskCurrency_ = localCreditRiskCurrency;
440 } else if (!localCreditRiskCurrency.empty() && creditRiskCurrency_ != localCreditRiskCurrency) {
441 ALOG(ore::data::StructuredTradeErrorMessage(id(), tradeType(), "Ambiguous SIMM CreditQ currencies for TRS",
442 "Will use '" + creditRiskCurrency_ + "', found '" +
443 localCreditRiskCurrency + "' in addition."));
444 }
445
446 // update global maturity date
447
448 maturity_ = std::max(maturity_, localMaturity);
449
450 // get fx indices for conversion of asset to funding ccy
451
452 DLOG("underlying #" << (i + 1) << " has asset ccy " << assetCurrency[i] << ", funding ccy is "
453 << fundingCurrency << ", return ccy is " << returnData_.currency());
454
455 fxIndexAsset[i] = getFxIndex(engineFactory->market(), engineFactory->configuration(MarketContext::pricing),
456 assetCurrency[i], fundingCurrency, localFxIndices, missingFxIndexPairs);
457 DLOG("underlying #" << (i + 1) << " index (" << underlyingIndex[i]->name() << ") built.");
458 DLOG("underlying #" << (i + 1) << " multiplier is " << underlyingMultiplier[i]);
459
460 // update global indexNames and fxIndices
461
462 for (const auto& [indexName, qty] : localIndexNamesAndQuantities) {
463 indexNamesAndQty[indexName] += (returnData_.payer() ? -1.0 : 1.0) * qty;
464 }
465
466 fxIndices.insert(localFxIndices.begin(), localFxIndices.end());
467 }
468
469 // check that we have all fx terms that we needed to build the fx indices
470
471 QL_REQUIRE(missingFxIndexPairs.empty(), "TRS::build(): missing FXTerms for the following pairs: "
472 << boost::algorithm::join(missingFxIndexPairs, ", "));
473
474 // set initial price currency
475
476 QL_REQUIRE(!assetCurrency.empty(), "TRS::build(): no underlying given.");
477
478 std::string initialPriceCurrency =
479 returnData_.initialPriceCurrency().empty() ? assetCurrency.front() : returnData_.initialPriceCurrency();
480
481 if (initialPrice != Null<Real>() && returnData_.initialPriceCurrency().empty()) {
482 for (auto const& ccy : assetCurrency) {
483 QL_REQUIRE(ccy == initialPriceCurrency, "TRS::build(): can not determine unique initial price currency "
484 "from asset currencies for initial price ("
485 << returnData_.initialPrice()
486 << "), please add the initial price currency to the trade xml");
487 }
488 }
489
490 // log some results from the build, convert initial price to major ccy if necessary
491
492 if (initialPrice != Null<Real>()) {
493 DLOG("initial price is given as " << initialPrice << " " << initialPriceCurrency);
494 initialPrice = convertMinorToMajorCurrency(initialPriceCurrency, initialPrice);
495 DLOG("initial price after conversion to major ccy " << initialPrice);
496 } else {
497 DLOG("no initial price is given");
498 }
499
500 DLOG("fundingCurrency is " << fundingCurrency);
501 DLOG("creditRiskCurrency is " << creditRiskCurrency_);
502 for (Size i = 0; i < assetCurrency.size(); ++i)
503 DLOG("assetCurrency #" << i << " is " << assetCurrency[i]);
504
505 // build funding legs, so far the supported types are
506 // Fixed, Floating
507
508 QL_REQUIRE(
509 fundingData_.notionalType().empty() || fundingData_.notionalType().size() == fundingData_.legData().size(),
510 "TRS::build(): got " << fundingData_.notionalType().size() << " NotionalType tags in FundingData, but "
511 << fundingData_.legData().size()
512 << " LegData nodes. These two must match. The NotionalType can also be omitted entirely.");
513
514 std::vector<Leg> fundingLegs;
515 std::vector<TRS::FundingData::NotionalType> fundingNotionalTypes;
516 for (Size i = 0; i < fundingData_.legData().size(); ++i) {
517
518 auto& ld = fundingData_.legData()[i];
519 QL_REQUIRE(ld.legType() == "Fixed" || ld.legType() == "Floating" || ld.legType() == "CMS" ||
520 ld.legType() == "CMB",
521 "TRS::build(): funding leg type: only fixed, floating, CMS, CMB are supported");
522 TRS::FundingData::NotionalType notionalType =
523 fundingData_.notionalType().empty() ? (ld.notionals().empty() ? TRS::FundingData::NotionalType::PeriodReset
524 : TRS::FundingData::NotionalType::Fixed)
525 : fundingData_.notionalType()[i];
526 QL_REQUIRE(ld.notionals().empty() || notionalType == TRS::FundingData::NotionalType::Fixed,
527 "TRS::build(): if notional is given in funding leg data, the notional type must be fixed, got "
528 << notionalType << " for funding leg #" << (i + 1));
529
530 struct TempNotionalSetter {
531 TempNotionalSetter(LegData& ld) : ld_(ld) {
532 if (ld.notionals().empty()) {
533 ld.notionals() = std::vector<Real>(1, 1.0);
534 ld.notionalDates().clear();
535 restoreEmptyLegDataNotionals_ = true;
536 } else {
537 restoreEmptyLegDataNotionals_ = false;
538 }
539 }
540 ~TempNotionalSetter() {
541 if (restoreEmptyLegDataNotionals_)
542 ld_.notionals().clear();
543 }
544 LegData& ld_;
545 bool restoreEmptyLegDataNotionals_;
546 };
547
548 TempNotionalSetter tmpSetter(ld);
549
550 auto legBuilder = engineFactory->legBuilder(ld.legType());
551 fundingLegs.push_back(legBuilder->buildLeg(ld, engineFactory, requiredFixings_,
552 engineFactory->configuration(MarketContext::pricing)));
553 fundingNotionalTypes.push_back(notionalType);
554
555 // update credit risk currency and credit qualifier mapping for CMB leg
556 if (ld.legType() == "CMB") {
557 auto cmbData = boost::dynamic_pointer_cast<ore::data::CMBLegData>(ld.concreteLegData());
558 QL_REQUIRE(cmbData, "TRS::build(): internal error, could to cast to CMBLegData.");
559 if(creditRiskCurrency_.empty())
560 creditRiskCurrency_ = getCmbLegCreditRiskCurrency(*cmbData, engineFactory->referenceData());
561 auto [source, target] =
562 getCmbLegCreditQualifierMapping(*cmbData, engineFactory->referenceData(), id(), tradeType());
563 creditQualifierMapping_[source] = target;
564 creditQualifierMapping_[ore::data::creditCurveNameFromSecuritySpecificCreditCurveName(source)] = target;
565 }
566 }
567
568 // add required fixings for funding legs with daily resets
569
570 DLOG("add required fixings for fundings legs with daily resets (if any)");
571
572 for (Size i = 0; i < fundingLegs.size(); ++i) {
573 if (fundingNotionalTypes[i] != TRS::FundingData::NotionalType::DailyReset)
574 continue;
575 for (auto const& c : fundingLegs[i]) {
576 if (auto cpn = boost::dynamic_pointer_cast<QuantLib::Coupon>(c)) {
577 QL_REQUIRE(boost::dynamic_pointer_cast<QuantLib::FixedRateCoupon>(c),
578 "daily reset funding legs support fixed rate coupons only");
579 for (QuantLib::Date d = cpn->accrualStartDate(); d < cpn->accrualEndDate(); ++d) {
580 for (Size j = 0; j < underlying_.size(); ++j) {
581 Date fixingDate = underlyingIndex[j]->fixingCalendar().adjust(d, Preceding);
582 for (auto const& [n, _] : indexNamesAndQty)
583 requiredFixings_.addFixingDate(fixingDate, n, cpn->date());
584 for (auto const& n : fxIndices) {
585 requiredFixings_.addFixingDate(n.second->fixingCalendar().adjust(fixingDate, Preceding),
586 n.first, cpn->date());
587 }
588 }
589 }
590 }
591 }
592 }
593
594 // set start date
595
596 Date startDate = Date::maxDate();
597 for (auto const& l : fundingLegs) {
598 for (auto const& cf : l) {
599 boost::shared_ptr<Coupon> coupon = boost::dynamic_pointer_cast<Coupon>(cf);
600 if (coupon)
601 startDate = std::min(startDate, coupon->accrualStartDate());
602 }
603 }
604
605 additionalData_["startDate"] = to_string(startDate);
606 for (const auto& [name, qty] : indexNamesAndQty) {
607 additionalData_["underlying_quantity_" + name] = qty;
608 }
609
610 // build additional cashflow leg (if given)
611
612 DLOG("build additional cashflow leg");
613
614 Leg additionalCashflowLeg;
615 bool additionalCashflowLegPayer = false;
616 std::string additionalCashflowLegCurrency = fundingCurrency;
617 if (additionalCashflowData_.legData().concreteLegData()) {
618 QL_REQUIRE(additionalCashflowData_.legData().legType() == "Cashflow",
619 "TRS::build(): additional cashflow data leg must have type 'Cashflow'");
620 additionalCashflowLeg = engineFactory->legBuilder(additionalCashflowData_.legData().legType())
621 ->buildLeg(additionalCashflowData_.legData(), engineFactory, requiredFixings_,
622 engineFactory->configuration(MarketContext::pricing));
623 additionalCashflowLegPayer = additionalCashflowData_.legData().isPayer();
624 additionalCashflowLegCurrency = additionalCashflowData_.legData().currency();
625 }
626
627 // parse asset currencies
628
629 std::vector<QuantLib::Currency> parsedAssetCurrencies;
630 for (auto const& c : assetCurrency) {
631 parsedAssetCurrencies.push_back(parseCurrency(c));
632 }
633
634 // build instrument
635
636 DLOG("build instrument and set trade member");
637
638 bool includeUnderlyingCashflowsInReturn;
639 if (returnData_.payUnderlyingCashFlowsImmediately()) {
640 includeUnderlyingCashflowsInReturn = !(*returnData_.payUnderlyingCashFlowsImmediately());
641 } else {
642 includeUnderlyingCashflowsInReturn = tradeType_ != "ContractForDifference";
643 }
644
645 auto wrapper = boost::make_shared<TRSWrapper>(
646 underlying_, underlyingIndex, underlyingMultiplier, includeUnderlyingCashflowsInReturn, initialPrice,
647 parseCurrencyWithMinors(initialPriceCurrency), parsedAssetCurrencies, parseCurrency(returnData_.currency()),
648 valuationDates, paymentDates, fundingLegs, fundingNotionalTypes, parseCurrency(fundingCurrency),
649 fundingData_.fundingResetGracePeriod(), returnData_.payer(), fundingLegPayer, additionalCashflowLeg,
650 additionalCashflowLegPayer, parseCurrency(additionalCashflowLegCurrency), fxIndexAsset, fxIndexReturn,
651 fxIndexAdditionalCashflows, fxIndices);
652 wrapper->setPricingEngine(boost::make_shared<TRSWrapperAccrualEngine>());
653 instrument_ = boost::make_shared<VanillaInstrument>(wrapper);
654
655 // set trade member variables (leave legs empty for the time being, we just have the funding leg really)
656
657 npvCurrency_ = fundingCurrency;
658
659 notional_ = 0.0; // we have overridden notional() to return this
660 notionalCurrency_ = returnData_.currency();
661
662 // if the maturity date was not set by the trs underlying builder, set it here
663 if (maturity_ == Date::minDate()) {
664 maturity_ = std::max(valuationDates.back(), paymentDates.back());
665 for (auto const& l : fundingLegs) {
666 maturity_ = std::max(maturity_, CashFlows::maturityDate(l));
667 }
668 }
669
670 // propagate additional data from underlyings to trs trade
671 for (Size i = 0; i < underlying_.size(); ++i) {
672 for (auto const& [key, value] : underlying_[i]->additionalData()) {
673 additionalData_["und_ad_" + std::to_string(i + 1) + "_" + key] = value;
674 }
675 }
676
677 // ISDA taxonomy
678 bool assetClassIsUnique = true;
679 std::string assetClass;
680 for (auto u : underlying_) {
681 auto it = u->additionalData().find("isdaAssetClass");
682 if (it != u->additionalData().end()) {
683 std::string ac = boost::any_cast<std::string>(it->second);
684 if (assetClass == "")
685 assetClass = ac;
686 else if (ac != assetClass)
687 assetClassIsUnique = false;
688 }
689 }
690
691 additionalData_["isdaAssetClass"] = string("");
692 additionalData_["isdaBaseProduct"] = string("");
693 additionalData_["isdaSubProduct"] = string("");
694 additionalData_["isdaTransaction"] = string("");
695
696 if (assetClass == "") {
697 ALOG("ISDA asset class not found for TRS " << id() << ", ISDA taxonomy undefined");
698 }
699 else {
700 additionalData_["isdaAssetClass"] = assetClass;
701 if (!assetClassIsUnique) {
702 WLOG("ISDA asset class not unique in TRS " << id() << " using first hit: " << assetClass);
703 }
704 additionalData_["isdaBaseProduct"] = string("Total Return Swap");
705 if (assetClass == "Equity") {
706 if (tradeType_ == "ContractForDifference")
707 additionalData_["isdaBaseProduct"] = string("Contract For Difference");
708 else
709 additionalData_["isdaBaseProduct"] = string("Swap");
710 additionalData_["isdaSubProduct"] = string("Price Return Basic Performance");
711 }
712 else if (assetClass == "Credit") {
713 additionalData_["isdaBaseProduct"] = string("Total Return Swap");
714 additionalData_["isdaSubProduct"] = string("");
715 }
716 else {
717 WLOG("ISDA asset class " << assetClass << " not explicitly covered for TRS trade " << id()
718 << " using default BaseProduct 'Total Retuen Swap' and leaving sub-product blank");
719 }
720 }
721}
