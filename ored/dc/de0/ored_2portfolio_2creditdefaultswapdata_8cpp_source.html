33 CdsTier parseCdsTier(const string& s) {
34  if (s == "SNRFOR") {
35  return CdsTier::SNRFOR;
36  } else if (s == "SUBLT2") {
37  return CdsTier::SUBLT2;
38  } else if (s == "SNRLAC") {

45  return CdsTier::PREFT1;
46  } else {
47  QL_FAIL("Could not parse \"" << s << "\" to CdsTier");
48  }
49 }

146 bool tryParseCdsInformation(const string& strInfo, CdsReferenceInformation& cdsInfo) {
147 
148  DLOG("tryParseCdsInformation: attempting to parse " << strInfo);
149 
150  // As in documentation comment, expect strInfo of form ID|TIER|CCY|DOCCLAUSE
151  vector<string> tokens;
152  boost::split(tokens, strInfo, boost::is_any_of("|"));
153 
154  if (tokens.size() != 4) {
155  TLOG("String " << strInfo << " not of form ID|TIER|CCY|DOCCLAUSE so parsing failed");
156  return false;
157  }
158 
159  CdsTier cdsTier;
160  if (!tryParse<CdsTier>(tokens[1], cdsTier, &parseCdsTier)) {
161  return false;
162  }
163 
164  Currency ccy;
165  if (!tryParseCurrency(tokens[2], ccy)) {
166  return false;
167  }
168 
169  CdsDocClause cdsDocClause;
170  if (!tryParse<CdsDocClause>(tokens[3], cdsDocClause, &parseCdsDocClause)) {
171  return false;
172  }
173 
174  cdsInfo = CdsReferenceInformation(tokens[0], cdsTier, ccy, cdsDocClause);
175 
176  return true;
177 }

205 void CreditDefaultSwapData::fromXML(XMLNode* node) {
206 
207  check(node);
208 
209  issuerId_ = XMLUtils::getChildValue(node, "IssuerId", false);
210 
211  // May get an explicit CreditCurveId node. If so, we use it. Otherwise, we must have ReferenceInformation node.
212  // XMLNode* tmp = XMLUtils::getChildNode(node, "CreditCurveId");
213  if (auto tmp = XMLUtils::getChildNode(node, "CreditCurveId")) {
214  creditCurveId_ = XMLUtils::getNodeValue(tmp);
215  } else {
216  tmp = XMLUtils::getChildNode(node, "ReferenceInformation");
217  QL_REQUIRE(tmp, "Need either a CreditCurveId or ReferenceInformation node in CreditDefaultSwapData");
218  CdsReferenceInformation ref;
219  ref.fromXML(tmp);
220  referenceInformation_ = ref;
221  creditCurveId_ = ref.id();
222  }
223 
224  settlesAccrual_ = XMLUtils::getChildValueAsBool(node, "SettlesAccrual", false);
225  protectionPaymentTime_ = PPT::atDefault;
226 
227  // for backwards compatibility only
228  if (auto c = XMLUtils::getChildNode(node, "PaysAtDefaultTime"))
229  if (!parseBool(XMLUtils::getNodeValue(c)))
230  protectionPaymentTime_ = PPT::atPeriodEnd;
231 
232  // new node overrides deprecated one, if both should be given
233  if (auto c = XMLUtils::getChildNode(node, "ProtectionPaymentTime")) {
234  if (XMLUtils::getNodeValue(c) == "atDefault")
235  protectionPaymentTime_ = PPT::atDefault;
236  else if (XMLUtils::getNodeValue(c) == "atPeriodEnd")
237  protectionPaymentTime_ = PPT::atPeriodEnd;
238  else if (XMLUtils::getNodeValue(c) == "atMaturity")
239  protectionPaymentTime_ = PPT::atMaturity;
240  else {
241  QL_FAIL("protection payment time '" << XMLUtils::getNodeValue(c) <<
242  "' not known, expected atDefault, atPeriodEnd, atMaturity");
243  }
244  }
245 
246  protectionStart_ = Date();
247  if (auto tmp = XMLUtils::getChildNode(node, "ProtectionStart"))
248  protectionStart_ = parseDate(XMLUtils::getNodeValue(tmp));
249 
250  upfrontDate_ = Date();
251  if (auto tmp = XMLUtils::getChildNode(node, "UpfrontDate"))
252  upfrontDate_ = parseDate(XMLUtils::getNodeValue(tmp));
253 
254  // zero if empty or missing
255  upfrontFee_ = Null<Real>();
256  string strUpfrontFee = XMLUtils::getChildValue(node, "UpfrontFee", false);
257  if (!strUpfrontFee.empty()) {
258  upfrontFee_ = parseReal(strUpfrontFee);
259  }
260 
261  if (upfrontDate_ == Date()) {
262  QL_REQUIRE(close_enough(upfrontFee_, 0.0) || upfrontFee_ == Null<Real>(),
263  "fromXML(): UpfronFee (" << upfrontFee_ << ") must be empty or zero if no upfront date is given");
264  upfrontFee_ = Null<Real>();
265  }
266 
267  // Recovery rate is Null<Real>() on a standard CDS i.e. if "FixedRecoveryRate" field is not populated.
268  recoveryRate_ = Null<Real>();
269  string strRecoveryRate = XMLUtils::getChildValue(node, "FixedRecoveryRate", false);
270  if (!strRecoveryRate.empty()) {
271  recoveryRate_ = parseReal(strRecoveryRate);
272  }
273 
274  tradeDate_ = Date();
275  if (auto n = XMLUtils::getChildNode(node, "TradeDate"))
276  tradeDate_ = parseDate(XMLUtils::getNodeValue(n));
277 
278  strCashSettlementDays_ = XMLUtils::getChildValue(node, "CashSettlementDays", false);
279  cashSettlementDays_ = strCashSettlementDays_.empty() ? 3 : parseInteger(strCashSettlementDays_);
280 
281  leg_.fromXML(XMLUtils::getChildNode(node, "LegData"));
282 }

284 XMLNode* CreditDefaultSwapData::toXML(XMLDocument& doc) {
285 
286  XMLNode* node = alloc(doc);
287 
288  XMLUtils::addChild(doc, node, "IssuerId", issuerId_);
289 
290  // We either have reference information or an explicit credit curve ID
291  if (referenceInformation_) {
292  XMLUtils::appendNode(node, referenceInformation_->toXML(doc));
293  } else {
294  XMLUtils::addChild(doc, node, "CreditCurveId", creditCurveId_);
295  }
296 
297  XMLUtils::addChild(doc, node, "SettlesAccrual", settlesAccrual_);
298  if (protectionPaymentTime_ == PPT::atDefault) {
299  XMLUtils::addChild(doc, node, "ProtectionPaymentTime", "atDefault");
300  } else if (protectionPaymentTime_ == PPT::atPeriodEnd) {
301  XMLUtils::addChild(doc, node, "ProtectionPaymentTime", "atPeriodEnd");
302  } else if (protectionPaymentTime_ == PPT::atMaturity) {
303  XMLUtils::addChild(doc, node, "ProtectionPaymentTime", "atMaturity");
304  } else {
305  QL_FAIL("toXML(): unexpected ProtectionPaymentTime");
306  }
307 
308  if (protectionStart_ != Date()) {
309  XMLUtils::addChild(doc, node, "ProtectionStart", to_string(protectionStart_));
310  }
311 
312  if (upfrontDate_ != Date()) {
313  XMLUtils::addChild(doc, node, "UpfrontDate", to_string(upfrontDate_));
314  }
315 
316  if (upfrontFee_ != Null<Real>())
317  XMLUtils::addChild(doc, node, "UpfrontFee", upfrontFee_);
318 
319  if (recoveryRate_ != Null<Real>())
320  XMLUtils::addChild(doc, node, "FixedRecoveryRate", recoveryRate_);
321 
322  if (tradeDate_ != Date()) {
323  XMLUtils::addChild(doc, node, "TradeDate", to_string(tradeDate_));
324  }
325 
326  if (!strCashSettlementDays_.empty()) {
327  XMLUtils::addChild(doc, node, "CashSettlementDays", strCashSettlementDays_);
328  }
329 
330  XMLUtils::appendNode(node, leg_.toXML(doc));
331 
332  return node;
333 }
