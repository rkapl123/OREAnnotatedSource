68 Handle<YieldTermStructure> MarketImpl::yieldCurve(const YieldCurveType& type, const string& key,
69  const string& configuration) const {
70  // we allow for standard (i.e. not convention based) ibor index names as keys and return the index forward curve in
71  // case of a match
72  boost::shared_ptr<IborIndex> notUsed;
73  if (tryParseIborIndex(key, notUsed, conventions_.has(key, Convention::Type::IborIndex)
74  ? conventions_.get(key) : nullptr)) {
75  return iborIndex(key, configuration)->forwardingTermStructure();
76  }
77  // no ibor index found under key => look for a genuine yield curve
78  DLOG("no ibor index found under '" << key << "' - look for a genuine yield curve");
79  if (type == YieldCurveType::Discount)
80  require(MarketObject::DiscountCurve, key, configuration);
81  else if (type == YieldCurveType::Yield)
82  require(MarketObject::YieldCurve, key, configuration);
83  else if (type == YieldCurveType::EquityDividend)
84  require(MarketObject::EquityCurve, key, configuration);
85  else {
86  QL_FAIL("yield curve type not handled");
87  }
88  return lookup<Handle<YieldTermStructure>>(yieldCurves_, key, type, configuration, "yield curve / ibor index");
89 }

145 Handle<BlackVolTermStructure> MarketImpl::fxVol(const string& ccypair, const string& configuration) const {
146  require(MarketObject::FXVol, ccypair, configuration);
147  auto it = fxVols_.find(make_pair(configuration, ccypair));
148  if (it != fxVols_.end())
149  return it->second;
150  else {
151  // check for reverse EURUSD or USDEUR and add to the map
152  QL_REQUIRE(ccypair.length() == 6, "invalid ccy pair length");
153  std::string ccypairInverted = ccypair.substr(3, 3) + ccypair.substr(0, 3);
154  require(MarketObject::FXVol, ccypairInverted, configuration);
155  it = fxVols_.find(make_pair(configuration, ccypairInverted));
156  if (it != fxVols_.end()) {
157  Handle<BlackVolTermStructure> h(boost::make_shared<QuantExt::BlackInvertedVolTermStructure>(it->second));
158  h->enableExtrapolation();
159  // we have found a surface for the inverted pair.
160  // so we can invert the surface and store that under the original pair.
161  fxVols_[make_pair(configuration, ccypair)] = h;
162  return h;
163  } else {
164  if (configuration == Market::defaultConfiguration)
165  QL_FAIL("did not find fx vol object " << ccypair);
166  else
167  return fxVol(ccypair, Market::defaultConfiguration);
168  }
169  }
170 }

277 Handle<QuantExt::CorrelationTermStructure> MarketImpl::correlationCurve(const string& index1, const string& index2,
278  const string& configuration) const {
279  // straight pair
280  require(MarketObject::Correlation, index1 + "&" + index2, configuration);
281  auto it = correlationCurves_.find(make_tuple(configuration, index1, index2));
282  if (it != correlationCurves_.end())
283  return it->second;
284  // inverse pair
285  require(MarketObject::Correlation, index2 + "&" + index1, configuration);
286  it = correlationCurves_.find(make_tuple(configuration, index2, index1));
287  if (it != correlationCurves_.end())
288  return it->second;
289  // inverse fx index1
290  if (isFxIndex(index1)) {
291  require(MarketObject::Correlation, inverseFxIndex(index1) + "&" + index2, configuration);
292  it = correlationCurves_.find(make_tuple(configuration, inverseFxIndex(index1), index2));
293  if (it != correlationCurves_.end())
294  return Handle<QuantExt::CorrelationTermStructure>(
295  boost::make_shared<QuantExt::NegativeCorrelationTermStructure>(it->second));
296  require(MarketObject::Correlation, index2 + "&" + inverseFxIndex(index1), configuration);
297  it = correlationCurves_.find(make_tuple(configuration, index2, inverseFxIndex(index1)));
298  if (it != correlationCurves_.end())
299  return Handle<QuantExt::CorrelationTermStructure>(
300  boost::make_shared<QuantExt::NegativeCorrelationTermStructure>(it->second));
301  }
302  // inverse fx index2
303  if (isFxIndex(index2)) {
304  require(MarketObject::Correlation, index1 + "&" + inverseFxIndex(index2), configuration);
305  it = correlationCurves_.find(make_tuple(configuration, index1, inverseFxIndex(index2)));
306  if (it != correlationCurves_.end())
307  return Handle<QuantExt::CorrelationTermStructure>(
308  boost::make_shared<QuantExt::NegativeCorrelationTermStructure>(it->second));
309  require(MarketObject::Correlation, inverseFxIndex(index2) + "&" + index1, configuration);
310  it = correlationCurves_.find(make_tuple(configuration, inverseFxIndex(index2), index1));
311  if (it != correlationCurves_.end())
312  return Handle<QuantExt::CorrelationTermStructure>(
313  boost::make_shared<QuantExt::NegativeCorrelationTermStructure>(it->second));
314  }
315  // both fx indices inverted
316  if (isFxIndex(index1) && isFxIndex(index2)) {
317  require(MarketObject::Correlation, inverseFxIndex(index1) + "&" + inverseFxIndex(index2), configuration);
318  it = correlationCurves_.find(make_tuple(configuration, inverseFxIndex(index1), inverseFxIndex(index2)));
319  if (it != correlationCurves_.end())
320  return it->second;
321  require(MarketObject::Correlation, inverseFxIndex(index2) + "&" + inverseFxIndex(index1), configuration);
322  it = correlationCurves_.find(make_tuple(configuration, inverseFxIndex(index2), inverseFxIndex(index1)));
323  if (it != correlationCurves_.end())
324  return it->second;
325  }
326  // if not found, fall back to default configuration
327  if (configuration == Market::defaultConfiguration) {
328  QL_FAIL("did not find object " << index1 << "/" << index2 << " in CorrelationCurves");
329  } else {
330  return correlationCurve(index1, index2, Market::defaultConfiguration);
331  }
332 }

339 void MarketImpl::addSwapIndex(const string& swapIndex, const string& discountIndex, const string& configuration) const {
340  try {
341  std::vector<string> tokens;
342  split(tokens, swapIndex, boost::is_any_of("-"));
343  QL_REQUIRE(tokens.size() == 3 || tokens.size() == 4,
344  "three or four tokens required in " << swapIndex << ": CCY-CMS-TENOR or CCY-CMS-TAG-TENOR");
345  QL_REQUIRE(tokens[0].size() == 3, "invalid currency code in " << swapIndex);
346  QL_REQUIRE(tokens[1] == "CMS", "expected CMS as middle token in " << swapIndex);
347 
348  Handle<YieldTermStructure> discounting, forwarding;
349  boost::shared_ptr<IborIndex> dummeyIndex;
350  if (tryParseIborIndex(discountIndex, dummeyIndex))
351  discounting = iborIndex(discountIndex, configuration)->forwardingTermStructure();
352  else
353  discounting = yieldCurve(discountIndex, configuration);
354 
355  auto swapCon = boost::dynamic_pointer_cast<data::SwapIndexConvention>(conventions_.get(swapIndex));
356  QL_REQUIRE(swapCon, "expected SwapIndexConvention for " << swapIndex);
357  auto con = boost::dynamic_pointer_cast<data::IRSwapConvention>(conventions_.get(swapCon->conventions()));
358  QL_REQUIRE(con, "expected IRSwapConvention for " << swapCon->conventions());
359 
360  string fi = con->indexName();
361  if (isGenericIborIndex(fi))
362  forwarding = discounting;
363  else
364  forwarding = iborIndex(con->indexName(), configuration)->forwardingTermStructure();
365 
366  boost::shared_ptr<SwapIndex> si = data::parseSwapIndex(swapIndex, forwarding, discounting, con, swapCon);
367  swapIndices_[make_pair(configuration, swapIndex)] = Handle<SwapIndex>(si);
368  } catch (std::exception& e) {
369  QL_FAIL("Failure in MarketImpl::addSwapIndex() with index " << swapIndex << " : " << e.what());
370  }
371 }

373 void MarketImpl::refresh(const string& configuration) {
374 
375  auto it = refreshTs_.find(configuration);
376  if (it == refreshTs_.end()) {
377  it = refreshTs_.insert(make_pair(configuration, std::set<boost::shared_ptr<TermStructure>>())).first;
378  }
379 
380  if (it->second.empty()) {
381  for (auto& x : yieldCurves_) {
382  if (get<0>(x.first) == configuration || get<0>(x.first) == Market::defaultConfiguration)
383  it->second.insert(*x.second);
384  }
385  for (auto& x : iborIndices_) {
386  if (x.first.first == configuration || x.first.first == Market::defaultConfiguration) {
387  Handle<YieldTermStructure> y = x.second->forwardingTermStructure();
388  if (!y.empty())
389  it->second.insert(*y);
390  }
391  }
392  for (auto& x : swapIndices_) {
393  if (x.first.first == configuration || x.first.first == Market::defaultConfiguration) {
394  Handle<YieldTermStructure> y = x.second->forwardingTermStructure();
395  if (!y.empty())
396  it->second.insert(*y);
397  y = x.second->discountingTermStructure();
398  if (!y.empty())
399  it->second.insert(*y);
400  }
401  }
402  for (auto& x : swaptionCurves_) {
403  if (x.first.first == configuration || x.first.first == Market::defaultConfiguration)
404  it->second.insert(*x.second);
405  }
406  for (auto& x : capFloorCurves_) {
407  if (x.first.first == configuration || x.first.first == Market::defaultConfiguration)
408  it->second.insert(*x.second);
409  }
410  for (auto& x : yoyCapFloorVolSurfaces_) {
411  if (x.first.first == configuration || x.first.first == Market::defaultConfiguration)
412  it->second.insert(*x.second);
413  }
414  for (auto& x : fxVols_) {
415  if (x.first.first == configuration || x.first.first == Market::defaultConfiguration)
416  it->second.insert(*x.second);
417  }
418  for (auto& x : defaultCurves_) {
419  if (x.first.first == configuration || x.first.first == Market::defaultConfiguration)
420  it->second.insert(*x.second);
421  }
422  for (auto& x : cdsVols_) {
423  if (x.first.first == configuration || x.first.first == Market::defaultConfiguration)
424  it->second.insert(*x.second);
425  }
426  for (auto& x : baseCorrelations_) {
427  if (x.first.first == configuration || x.first.first == Market::defaultConfiguration)
428  it->second.insert(*x.second);
429  }
430  for (auto& x : zeroInflationIndices_) {
431  if (x.first.first == configuration || x.first.first == Market::defaultConfiguration) {
432  it->second.insert(*x.second->zeroInflationTermStructure());
433  }
434  }
435  for (auto& x : yoyInflationIndices_) {
436  if (x.first.first == configuration || x.first.first == Market::defaultConfiguration) {
437  it->second.insert(*x.second->yoyInflationTermStructure());
438  }
439  }
440  for (auto& x : cpiInflationCapFloorVolatilitySurfaces_) {
441  if (x.first.first == configuration || x.first.first == Market::defaultConfiguration)
442  it->second.insert(*x.second);
443  }
444  for (auto& x : yoyCapFloorVolSurfaces_) {
445  if (x.first.first == configuration || x.first.first == Market::defaultConfiguration)
446  it->second.insert(*x.second);
447  }
448  for (auto& x : equityVols_) {
449  if (x.first.first == configuration || x.first.first == Market::defaultConfiguration)
450  it->second.insert(*x.second);
451  }
452  for (auto& x : equityCurves_) {
453  if (x.first.first == configuration || x.first.first == Market::defaultConfiguration) {
454  Handle<YieldTermStructure> y = x.second->equityForecastCurve();
455  if (!y.empty())
456  it->second.insert(*y);
457  y = x.second->equityDividendCurve();
458  if (!y.empty())
459  it->second.insert(*y);
460  }
461  }
462  for (auto& x : commodityIndices_) {
463  if (x.first.first == configuration || x.first.first == Market::defaultConfiguration) {
464  const auto& pts = x.second->priceCurve();
465  if (!pts.empty())
466  it->second.insert(*pts);
467  }
468  }
469  for (auto& x : commodityVols_) {
470  if (x.first.first == configuration || x.first.first == Market::defaultConfiguration)
471  it->second.insert(*x.second);
472  }
473 
474  for (auto& x : correlationCurves_) {
475  if (get<0>(x.first) == configuration || get<0>(x.first) == Market::defaultConfiguration)
476  it->second.insert(*x.second);
477  }
478  }
479 
480  // term structures might be wrappers around nested termstructures that need to be updated as well,
481  // therefore we need to call deepUpdate() (=update() if no such nesting is present)
482  for (auto& x : it->second)
483  x->deepUpdate();
484 
485  // update fx spot quotes
486  auto fxSpots = fxSpots_.find(configuration);
487  if (fxSpots != fxSpots_.end()) {
488  for (auto& x : fxSpots->second.quotes()) {
489  auto dq = boost::dynamic_pointer_cast<Observer>(*x.second);
490  if (dq != nullptr)
491  dq->update();
492  }
493  }
