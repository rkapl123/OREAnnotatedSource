125 BOOST_AUTO_TEST_CASE(testCcySwapWithResetsPrice) {
126 
127  BOOST_TEST_MESSAGE("Testing CcySwapWithResets Price...");
128 
129  // Bloomberg CCYswap and CCYswapReset prices (in USD)
130  Real npvCCYswap = -349.69;
131  Real npvCCYswapReset = 0;
132 
133  // build market
134  boost::shared_ptr<Market> market = boost::make_shared<TestMarket>();
135  Settings::instance().evaluationDate() = market->asofDate();
136 
137  // test asof date
138  Date today = market->asofDate();
139  BOOST_CHECK_EQUAL(today, Date(22, Aug, 2016));
140 
141  // Test if EUR discount curve is empty
142  Handle<YieldTermStructure> dts = market->discountCurve("EUR");
143  QL_REQUIRE(!dts.empty(), "EUR discount curve not found");
144 
145  // Test EUR discount(today+3*Years)) against base value
146  BOOST_CHECK_CLOSE(market->discountCurve("EUR")->discount(today + 3 * Years), 1.006005, 0.0001);
147 
148  // Test USD discount(today+3*Years)) against base value
149  BOOST_CHECK_CLOSE(market->discountCurve("USD")->discount(today + 3 * Years), 0.96908, 0.0001);
150 
151  // Test EURUSD fx spot object against base value
152  BOOST_CHECK_EQUAL(market->fxSpot("EURUSD")->value(), 1.1306);
153 
154  // envelope
155  Envelope env("CP");
156 
157  // Start/End date
158  Date startDate = today;
159  Date endDate = today + 5 * Years;
160 
161  // date 2 string
162  std::ostringstream oss;
163  oss << io::iso_date(startDate);
164  string start(oss.str());
165  oss.str("");
166  oss.clear();
167  oss << io::iso_date(endDate);
168  string end(oss.str());
169 
170  // Schedules
171  string conv = "MF";
172  string rule = "Forward";
173  ScheduleData scheduleEUR(ScheduleRules(start, end, "6M", "TARGET", conv, conv, rule));
174  ScheduleData scheduleUSD(ScheduleRules(start, end, "3M", "US", conv, conv, rule));
175 
176  // EUR Leg without and with Notional resets
177  bool isPayerEUR = true;
178  string indexEUR = "EUR-EURIBOR-6M";
179  bool isInArrears = false;
180  int days = 2, fxFixingDays = 0;
181  vector<Real> spreadEUR(1, 0.000261);
182  string dc = "ACT/360";
183  vector<Real> notionalEUR(1, 8833141.95);
184  string paymentConvention = "F";
185  bool notionalInitialXNL = true;
186  bool notionalFinalXNL = true;
187  bool notionalAmortizingXNL = false;

192  LegData legEUR1(legdataEUR, isPayerEUR, "EUR", scheduleEUR, dc, notionalEUR, vector<string>(), paymentConvention,
193  notionalInitialXNL, notionalFinalXNL, notionalAmortizingXNL, notionalFinalXNL, foreignCCY,
194  foreignAmount, fxIndex, fxFixingDays);
195  LegData legEUR2(legdataEUR, isPayerEUR, "EUR", scheduleEUR, dc, notionalEUR, vector<string>(), paymentConvention,
196  notionalInitialXNL, notionalFinalXNL, notionalAmortizingXNL, false, foreignCCY, foreignAmount,
197  fxIndex, fxFixingDays);
198 
199  // USD Leg without notional resets

202  vector<Real> spreadUSD(1, 0);
