92 // Portfolios, designed such that trade NPVs should add up to zero
93 // The first two cases consist of three trades:
94 // 1) CPI Swap receiving a single zero coupon fixed flow and paying a single indexed redemption flow (resp. CPI cpupons
95 // plus indexed redemption)
96 // 2) CPI Swap as above with capped indexed flow and flipped legs: pay zero coupon fixed, receive
97 // capped indexed redemption (resp. capped CPI coupons plus capped indexed redemption), i.e. short embedded cap(s)
98 // 3) standalone long CPI cap with indexed flow(s) above as underlying
99 // The third portfolio has two trades
100 // 1) A CPI Cap as CapFloor instrument
101 // 2) A CPI Cap as Swap with a single CPI leg and "naked" option set to "Y"
102 
103 vector<string> testCases = {"portfolio_singleflow.xml", "portfolio_multiflow.xml", "portfolio_multiflow_naked.xml"};

111 BOOST_DATA_TEST_CASE_F(F, testCapConsistency, bdata::make(testCases), testCase) {
112 
113  BOOST_TEST_MESSAGE("Testing " << testCase);
114 
115  Portfolio p;
116  p.load(TEST_INPUT_FILE(testCase));
117  Size n = p.size();
118 
119  // Build the portfolio
120  p.build(engineFactory);
121  BOOST_CHECK_MESSAGE(p.size() == n, n << " trades expected after build, found " << p.size());
122  for (Size i = 0; i < p.size(); ++i) {
123  BOOST_TEST_MESSAGE("trade " << p.trades()[i]->id());
124  }
125 
126  // Portfolios are designed such that trade NPVs should add up to zero
127  Real sum = 0.0;
128  Real minimumNPV = QL_MAX_REAL;
129  for (Size i = 0; i < p.size(); ++i) {
130  BOOST_CHECK_NO_THROW(p.trades()[i]->instrument()->NPV());
131  BOOST_TEST_MESSAGE("trade " << p.trades()[i]->id() << " npv " << p.trades()[i]->instrument()->NPV());
132  sum += p.trades()[i]->instrument()->NPV();
133  minimumNPV = std::min(minimumNPV, fabs(p.trades()[i]->instrument()->NPV()));
134  }
135  BOOST_TEST_MESSAGE("mininim absolute NPV = " << minimumNPV);
136  Real tolerance = 1.0e-8 * minimumNPV;
137  BOOST_TEST_MESSAGE("tolerance = " << tolerance);
138  BOOST_TEST_MESSAGE("NPV sum = " << sum);
139  BOOST_CHECK_MESSAGE(fabs(sum) < tolerance, "portfolio NPV should be zero, found " << sum);
140 }
