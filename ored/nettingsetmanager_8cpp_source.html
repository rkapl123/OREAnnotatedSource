29 void NettingSetManager::add(const boost::shared_ptr<NettingSetDefinition>& nettingSet) {

32  std::pair<NettingSetDetails, boost::shared_ptr<NettingSetDefinition>> newNetSetDef(k, nettingSet);

92 void NettingSetManager::fromXML(XMLNode* node) {
93  XMLUtils::checkNode(node, "NettingSetDefinitions");
94  vector<XMLNode*> nettingSetNodes = XMLUtils::getChildrenNodes(node, "NettingSet");
95  for (unsigned i = 0; i < nettingSetNodes.size(); i++) {
96  XMLNode* child = nettingSetNodes[i];
97  try {
98  boost::shared_ptr<NettingSetDefinition> nettingSet(new NettingSetDefinition(child));
99  add(nettingSet);
100  } catch (std::exception& ex) {
101  ALOG(StructuredConfigurationWarningMessage("Netting set manager", "",
102  "Netting set definnition failed to parse", ex.what()));
103  }
104  }
105 }
