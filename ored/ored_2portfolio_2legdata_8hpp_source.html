972 template <typename T>
973 vector<T> buildScheduledVector(const vector<T>& values, const vector<string>& dates, const Schedule& schedule,
974  const bool checkAllValuesAppearInResult) {
975  if (values.size() < 2 || dates.size() == 0)
976  return values;
977 
978  QL_REQUIRE(values.size() == dates.size(), "Value / Date size mismatch in buildScheduledVector."
979  << "Value:" << values.size() << ", Dates:" << dates.size());
980 
981  // Need to use schedule logic
982  // Length of data will be 1 less than schedule
983  //
984  // Notional 100
985  // Notional {startDate 2015-01-01} 200
986  // Notional {startDate 2016-01-01} 300
987  //
988  // Given schedule June, Dec from 2014 to 2016 (6 dates, 5 coupons)
989  // we return 100, 100, 200, 200, 300
990 
991  // The first node must not have a date.
992  // If the second one has a date, all the rest must have, and we process
993  // If the second one does not have a date, none of them must have one
994  // and we return the vector uneffected.
995  QL_REQUIRE(dates[0] == "", "Invalid date " << dates[0] << " for first node");
996  if (dates[1] == "") {
997  // They must all be empty and then we return values
998  for (Size i = 2; i < dates.size(); i++) {
999  QL_REQUIRE(dates[i] == "", "Invalid date " << dates[i] << " for node " << i
1000  << ". Cannot mix dates and non-dates attributes");
1001  }
1002  return values;
1003  }
1004 
1005  // We have nodes with date attributes now
1006  Size len = schedule.size() - 1;
1007  vector<T> data(len);
1008  Size j = 0, max_j = dates.size() - 1; // j is the index of date/value vector. 0 to max_j
1009  Date d = parseDate(dates[j + 1]); // The first start date
1010  for (Size i = 0; i < len; i++) { // loop over data vector and populate it.
1011  // If j == max_j we just fall through and take the final value
1012  while (schedule[i] >= d && j < max_j) {
1013  j++;
1014  if (j < max_j) {
1015  QL_REQUIRE(dates[j + 1] != "", "Cannot have empty date attribute for node " << j + 1);
1016  d = parseDate(dates[j + 1]);
1017  }
1018  }
1019  data[i] = values[j];
1020  }
1021 
1022  if (checkAllValuesAppearInResult) {
1023  auto it = checkAllValuesAppearInScheduledVector<T>(data, values);
1024  QL_REQUIRE(it == values.end(),
1025  "buildScheduledVector: input value '" << *it << " does not appear in the result vector");
1026  }
1027 
1028  return data;
1029 }
