98 Date getOpenEndDateReplacement(const std::string& replacementPeriodStr, const Calendar& calendar) {
99  if (replacementPeriodStr.empty())
100  return QuantLib::Null<QuantLib::Date>();
101  Date today = Settings::instance().evaluationDate();
102  Date result = Date::maxDate() - 365;
103  try {
104  // might throw because we are beyond the last allowed date
105  result = (calendar.empty() ? NullCalendar() : calendar).advance(today, parsePeriod(replacementPeriodStr));
106  } catch (...) {
107  }
108  DLOG("Compute open end date replacement as "
109  << QuantLib::io::iso_date(result) << " (today = " << QuantLib::io::iso_date(today)
110  << ", OpenEndDateReplacement from pricing engine config = " << replacementPeriodStr << ")");
111  return result;
112 }
