48void ForwardBond::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
49 DLOG("ForwardBond::build() called for trade " << id());
50
51 const boost::shared_ptr<Market> market = engineFactory->market();
52
53 boost::shared_ptr<EngineBuilder> builder_fwd = engineFactory->builder("ForwardBond");
54 boost::shared_ptr<EngineBuilder> builder_bd = engineFactory->builder("Bond");
55
56 bondData_ = originalBondData_;
57 bondData_.populateFromBondReferenceData(engineFactory->referenceData());
58
59 QL_REQUIRE(!bondData_.referenceCurveId().empty(), "reference curve id required");
60 QL_REQUIRE(!bondData_.settlementDays().empty(), "settlement days required");
61
62 Date issueDate = parseDate(bondData_.issueDate());
63 Calendar calendar = parseCalendar(bondData_.calendar());
64 Natural settlementDays = boost::lexical_cast<Natural>(bondData_.settlementDays());
65
66 Date fwdMaturityDate = parseDate(fwdMaturityDate_);
67 Date fwdSettlementDate = fwdSettlementDate_.empty() ? fwdMaturityDate : parseDate(fwdSettlementDate_);
68 bool isPhysicallySettled;
69 if (settlement_ == "Physical" || settlement_.empty())
70 isPhysicallySettled = true;
71 else if (settlement_ == "Cash")
72 isPhysicallySettled = false;
73 else {
74 QL_FAIL("ForwardBond: invalid settlement '" << settlement_ << "', expected Cash or Physical");
75 }
76 Real amount = amount_.empty() ? Null<Real>() : parseReal(amount_);
77 Real lockRate = lockRate_.empty() ? Null<Real>() : parseReal(lockRate_);
78 Real dv01 = dv01_.empty() ? Null<Real>() : parseReal(dv01_);
79 DayCounter lockRateDayCounter = lockRateDayCounter_.empty() ? Actual360() : parseDayCounter(lockRateDayCounter_);
80 bool settlementDirty = settlementDirty_.empty() ? true : parseBool(settlementDirty_);
81 Real compensationPayment = parseReal(compensationPayment_);
82 Date compensationPaymentDate = parseDate(compensationPaymentDate_);
83 bool longInForward = parseBool(longInForward_);
84
85 QL_REQUIRE((amount == Null<Real>() && lockRate != Null<Real>()) ||
86 (amount != Null<Real>() && lockRate == Null<Real>()),
87 "ForwardBond: exactly one of Amount of LockRate must be given");
88 QL_REQUIRE(dv01 >= 0.0, "negative DV01 given");
89 QL_REQUIRE(compensationPaymentDate <= fwdMaturityDate, "Premium cannot be paid after forward contract maturity");
90
91 if (lockRate != Null<Real>())
92 isPhysicallySettled = false;
93
94 QL_REQUIRE(!bondData_.coupons().empty(), "ForwardBond: No LegData given. If you want to represent a zero bond, "
95 "set it up as a coupon bond with zero fixed rate");
96
97 bool firstLegIsPayer = bondData_.coupons()[0].isPayer();
98 QL_REQUIRE(firstLegIsPayer == false, "ForwardBond: The underlying bond must be entered with a receiver leg. Use "
99 "LongInBond to specify pay direction of forward payoff");
100 QL_REQUIRE(compensationPayment > 0.0 || close_enough(compensationPayment, 0.0),
101 "ForwardBond: Negative compensation payments ("
102 << compensationPayment
103 << ") are not allowed. Notice that we will ensure that a positive compensation amount will be paid "
104 "by the party being long in the forward contract.");
105
106 boost::shared_ptr<Payoff> payoff;
107 if (amount != Null<Real>()) {
108 payoff = longInForward ? boost::make_shared<QuantExt::ForwardBondTypePayoff>(Position::Long, amount)
109 : boost::make_shared<QuantExt::ForwardBondTypePayoff>(Position::Short, amount);
110 }
111 compensationPayment = longInForward ? compensationPayment : -compensationPayment;
112
113 std::vector<Leg> separateLegs;
114 for (Size i = 0; i < bondData_.coupons().size(); ++i) {
115 Leg leg;
116 auto configuration = builder_bd->configuration(MarketContext::pricing);
117 auto legBuilder = engineFactory->legBuilder(bondData_.coupons()[i].legType());
118 leg = legBuilder->buildLeg(bondData_.coupons()[i], engineFactory, requiredFixings_, configuration);
119 separateLegs.push_back(leg);
120 }
121 Leg leg = joinLegs(separateLegs);
122 auto bond = boost::make_shared<QuantLib::Bond>(settlementDays, calendar, issueDate, leg);
123
124 npvCurrency_ = currency_ = bondData_.coupons().front().currency();
125 maturity_ = bond->cashflows().back()->date();
126 notional_ = currentNotional(bond->cashflows()) * bondData_.bondNotional();
127 notionalCurrency_ = currency_;
128
129 // cashflows will be generated as additional results in the pricing engine
130 legs_ = {};
131 legCurrencies_ = {npvCurrency_};
132 legPayers_ = {firstLegIsPayer};
133 Currency currency = parseCurrency(currency_);
134
135 // first ctor is for vanilla fwd bonds, second for tlocks with a lock rate specifying the payoff
136 boost::shared_ptr<QuantLib::Instrument> fwdBond =
137 payoff ? boost::make_shared<QuantExt::ForwardBond>(bond, payoff, fwdMaturityDate, fwdSettlementDate,
138 isPhysicallySettled, settlementDirty, compensationPayment,
139 compensationPaymentDate, bondData_.bondNotional())
140 : boost::make_shared<QuantExt::ForwardBond>(bond, lockRate, lockRateDayCounter, longInForward,
141 fwdMaturityDate, fwdSettlementDate, isPhysicallySettled,
142 settlementDirty, compensationPayment,
143 compensationPaymentDate, bondData_.bondNotional(), dv01);
144
145 boost::shared_ptr<fwdBondEngineBuilder> fwdBondBuilder =
146 boost::dynamic_pointer_cast<fwdBondEngineBuilder>(builder_fwd);
147 QL_REQUIRE(fwdBondBuilder, "ForwardBond::build(): could not cast builder: " << id());
148
149 fwdBond->setPricingEngine(fwdBondBuilder->engine(id(), currency, bondData_.creditCurveId(),
150 bondData_.hasCreditRisk(), bondData_.securityId(),
151 bondData_.referenceCurveId(), bondData_.incomeCurveId()));
152 instrument_.reset(new VanillaInstrument(fwdBond, 1.0));
153
154 additionalData_["currentNotional"] = currentNotional(bond->cashflows()) * bondData_.bondNotional();
155 additionalData_["originalNotional"] = originalNotional(bond->cashflows()) * bondData_.bondNotional();
156 additionalData_["currency"] = currency_;
157
158 // ISDA taxonomy
159 additionalData_["isdaAssetClass"] = string("Interest Rate");
160 additionalData_["isdaBaseProduct"] = string("Forward");
161 additionalData_["isdaSubProduct"] = string("Debt");
162 additionalData_["isdaTransaction"] = string("");
163}

165void ForwardBond::fromXML(XMLNode* node) {
166 Trade::fromXML(node);
167 XMLNode* fwdBondNode = XMLUtils::getChildNode(node, "ForwardBondData");
168 QL_REQUIRE(fwdBondNode, "No ForwardBondData Node");
169 originalBondData_.fromXML(XMLUtils::getChildNode(fwdBondNode, "BondData"));
170 bondData_ = originalBondData_;
171
172 XMLNode* fwdSettlementNode = XMLUtils::getChildNode(fwdBondNode, "SettlementData");
173 QL_REQUIRE(fwdSettlementNode, "No fwdSettlementNode Node");
174
175 fwdMaturityDate_ = XMLUtils::getChildValue(fwdSettlementNode, "ForwardMaturityDate", true);
176 fwdSettlementDate_ = XMLUtils::getChildValue(fwdSettlementNode, "ForwardSettlementDate", false);
177 settlement_ = XMLUtils::getChildValue(fwdSettlementNode, "Settlement", false);
178 amount_ = XMLUtils::getChildValue(fwdSettlementNode, "Amount", false);
179 lockRate_ = XMLUtils::getChildValue(fwdSettlementNode, "LockRate", false);
180 lockRateDayCounter_ = XMLUtils::getChildValue(fwdSettlementNode, "LockRateDayCounter", false);
181 settlementDirty_ = XMLUtils::getChildValue(fwdSettlementNode, "SettlementDirty", false);
182 dv01_ = XMLUtils::getChildValue(fwdSettlementNode, "dv01", false);
183
184 XMLNode* fwdPremiumNode = XMLUtils::getChildNode(fwdBondNode, "PremiumData");
185 if (fwdPremiumNode) {
186 compensationPayment_ = XMLUtils::getChildValue(fwdPremiumNode, "Amount", true);
187 compensationPaymentDate_ = XMLUtils::getChildValue(fwdPremiumNode, "Date", true);
188 } else {
189 compensationPayment_ = "0.0";
190 compensationPaymentDate_ = fwdMaturityDate_;
191 }
192
193 longInForward_ = XMLUtils::getChildValue(fwdBondNode, "LongInForward", true);
194}

196XMLNode* ForwardBond::toXML(XMLDocument& doc) {
197 XMLNode* node = Trade::toXML(doc);
198 XMLNode* fwdBondNode = doc.allocNode("ForwardBondData");
199 XMLUtils::appendNode(node, fwdBondNode);
200 XMLUtils::appendNode(fwdBondNode, originalBondData_.toXML(doc));
201
202 XMLNode* fwdSettlementNode = doc.allocNode("SettlementData");
203 XMLUtils::appendNode(fwdBondNode, fwdSettlementNode);
204 XMLUtils::addChild(doc, fwdSettlementNode, "ForwardMaturityDate", fwdMaturityDate_);
205 if (!fwdSettlementDate_.empty())
206 XMLUtils::addChild(doc, fwdSettlementNode, "ForwardSettlementDate", fwdSettlementDate_);
207 if (!settlement_.empty())
208 XMLUtils::addChild(doc, fwdSettlementNode, "Settlement", settlement_);
209 if (!amount_.empty())
210 XMLUtils::addChild(doc, fwdSettlementNode, "Amount", amount_);
211 if (!lockRate_.empty())
212 XMLUtils::addChild(doc, fwdSettlementNode, "LockRate", lockRate_);
213 if (!dv01_.empty())
214 XMLUtils::addChild(doc, fwdSettlementNode, "dv01", dv01_);
215 if (!lockRateDayCounter_.empty())
216 XMLUtils::addChild(doc, fwdSettlementNode, "LockRateDayCounter", lockRateDayCounter_);
217 if (!settlementDirty_.empty())
218 XMLUtils::addChild(doc, fwdSettlementNode, "SettlementDirty", settlementDirty_);
219
220 XMLNode* fwdPremiumNode = doc.allocNode("PremiumData");
221 XMLUtils::appendNode(fwdBondNode, fwdPremiumNode);
222 XMLUtils::addChild(doc, fwdPremiumNode, "Amount", compensationPayment_);
223 XMLUtils::addChild(doc, fwdPremiumNode, "Date", compensationPaymentDate_);
224
225 XMLUtils::addChild(doc, fwdBondNode, "LongInForward", longInForward_);
226
227 return node;
228}
