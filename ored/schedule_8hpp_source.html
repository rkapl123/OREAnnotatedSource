197  derived_.emplace_back(derived);
198  hasDerived_ = true;
199  }

210  const bool& hasDerived() const { return hasDerived_; }
211  //@}
212 
213  //! \name Modifiers
214  //@{
215  vector<ScheduleDates>& modifyDates() { return dates_; }
216  vector<ScheduleRules>& modifyRules() { return rules_; }
217  vector<ScheduleDerived>& modifyDerived() { return derived_; }
218  //@}
219 
220  //! \name Serialisation

223  virtual XMLNode* toXML(XMLDocument& doc) override;
224  //@}

233 // Container class to support building of derived schedules
234 class ScheduleBuilder {
235 // USAGE:
236 // 1. Initialise a ScheduleBuilder obj
237 // 2. For each Schedule obj that will be built from a given ScheduleData obj:
238 // i. Initialise an empty Schedule obj
239 // ii. Add the Schedule obj in i. and the corresponding ScheduleData obj into the ScheduleBuilder obj in 1.
240 // using the ScheduleBuilder::add() method.
241 // 3. Once all required schedules are added for the instrument/leg, call ScheduleBuilder::makeSchedules()
242 // with the appropriate openEndDateReplacement, if relevant.
243 // If successful, each Schedule obj, derived or otherwise, should have been assigned the correct schedule.
244 // NOTE / WARNING:
245 // * The schedules should be initialised at the same, or higher, scope as the call to makeSchedules() (and to add())
246 
247 public:
248  void add(QuantLib::Schedule& schedule, const ScheduleData& scheduleData);
249  void makeSchedules(const QuantLib::Date& openEndDateReplacement = QuantLib::Null<QuantLib::Date>());
250 
251 private:
252  map<string, pair<ScheduleData, QuantLib::Schedule&>> schedules_;
253 };
