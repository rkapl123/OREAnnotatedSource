37 Leg DurationAdjustedCmsLegBuilder::buildLeg(const LegData& data, const boost::shared_ptr<EngineFactory>& engineFactory,
38  RequiredFixings& requiredFixings, const string& configuration,
39  const QuantLib::Date& openEndDateReplacement) const {
40 
41  auto cmsData = boost::dynamic_pointer_cast<DurationAdjustedCmsLegData>(data.concreteLegData());
42  QL_REQUIRE(cmsData, "Wrong LegType, expected CMS");
43 
44  string indexName = cmsData->swapIndex();
45  auto index = *engineFactory->market()->swapIndex(indexName, configuration);
46  Schedule schedule = makeSchedule(data.schedule(), openEndDateReplacement);
47  DayCounter dc = parseDayCounter(data.dayCounter());
48  BusinessDayConvention bdc = parseBusinessDayConvention(data.paymentConvention());
49 
50  vector<double> spreads =
51  ore::data::buildScheduledVectorNormalised(cmsData->spreads(), cmsData->spreadDates(), schedule, 0.0);
52  vector<double> gearings =
53  ore::data::buildScheduledVectorNormalised(cmsData->gearings(), cmsData->gearingDates(), schedule, 1.0);
54  vector<double> notionals = buildScheduledVector(data.notionals(), data.notionalDates(), schedule);
55  Size fixingDays = cmsData->fixingDays() == Null<Size>() ? index->fixingDays() : cmsData->fixingDays();
56 
57  applyAmortization(notionals, data, schedule, false);
58 
59  PaymentLag paymentLag = parsePaymentLag(data.paymentLag());
60  DurationAdjustedCmsLeg leg = DurationAdjustedCmsLeg(schedule, index, cmsData->duration())
61  .withNotionals(notionals)
62  .withSpreads(spreads)
63  .withGearings(gearings)
64  .withPaymentDayCounter(dc)
65  .withPaymentAdjustment(bdc)
66  .withPaymentLag(boost::apply_visitor(PaymentLagInteger(), paymentLag))
67  .withFixingDays(fixingDays)
68  .inArrears(cmsData->isInArrears());
69 
70  if (!data.paymentCalendar().empty())
71  leg.withPaymentCalendar(parseCalendar(data.paymentCalendar()));
72 
73  if (cmsData->caps().size() > 0)
74  leg.withCaps(buildScheduledVector(cmsData->caps(), cmsData->capDates(), schedule));
75 
76  if (cmsData->floors().size() > 0)
77  leg.withFloors(buildScheduledVector(cmsData->floors(), cmsData->floorDates(), schedule));
78 
79  // Get a coupon pricer for the leg
80  auto builder = boost::dynamic_pointer_cast<DurationAdjustedCmsCouponPricerBuilder>(
81  engineFactory->builder("DurationAdjustedCMS"));
82  QL_REQUIRE(builder != nullptr, "No builder found for DurationAdjustedCmsLeg");
83  auto couponPricer = builder->engine(IndexNameTranslator::instance().oreName(index->iborIndex()->name()));
84 
85  // Loop over the coupons in the leg and set pricer
86  Leg result = leg;
87  for (auto& c : result) {
88  auto f = boost::dynamic_pointer_cast<FloatingRateCoupon>(c);
89  QL_REQUIRE(f != nullptr,
90  "DurationAdjustedCmsLegBuilder::buildLeg(): internal error, expected FloatingRateCoupon");
91  f->setPricer(couponPricer);
92  }
93 
94  // build naked option leg if required
95  if (cmsData->nakedOption()) {
96  result = StrippedCappedFlooredCouponLeg(result);
97  }
98 
99  std::map<std::string, std::string> qlToOREIndexNames;
100  qlToOREIndexNames[index->name()] = indexName;
101  applyIndexing(result, data, engineFactory, qlToOREIndexNames, requiredFixings);
102  addToRequiredFixings(result, boost::make_shared<FixingDateGetter>(requiredFixings, qlToOREIndexNames));
103  return result;
104 }
