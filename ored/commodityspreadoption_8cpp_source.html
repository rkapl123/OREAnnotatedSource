27enum CommoditySpreadOptionPosition{Begin = 0, Long = 0, Short = 1, End = 2};
28void CommoditySpreadOption::build(const boost::shared_ptr<ore::data::EngineFactory>& engineFactory) {
29
30 reset();
31 npvCurrency_ = settlementCcy_;
32 DLOG("CommoditySpreadOption::build() called for trade " << id());
33 // build the relevant fxIndexes;
34 std::vector<boost::shared_ptr<QuantExt::FxIndex>> FxIndex{nullptr, nullptr};
35 commLegData_.resize(2);
36 Currency ccy = parseCurrency(settlementCcy_);
37 auto optionType = parseOptionType(callPut_);
38 boost::shared_ptr<CommodityIndex> longIndex{nullptr}, shortIndex{nullptr};
39 std::vector<bool> isAveraged{false, false};
40
41 // init engine factory builder
42 boost::shared_ptr<EngineBuilder> builder = engineFactory->builder(tradeType_);
43 auto engineBuilder = boost::dynamic_pointer_cast<CommoditySpreadOptionEngineBuilder>(builder);
44 // get config
45 auto config = builder->configuration(MarketContext::pricing);
46
47 // Build the commodity legs
48 for (int i = CommoditySpreadOptionPosition::Begin; i!=CommoditySpreadOptionPosition::End; i++) { // The order is important, the first leg is always the long position, the second is the short
49 if(i == (int)CommoditySpreadOptionPosition::Long) {
50 QL_REQUIRE(legData_[i].isPayer(), "CommoditySpreadOption: first leg is not relevant for long position.");
51 legPayers_.push_back(true);
52 }
53 if(i == (int)CommoditySpreadOptionPosition::Short) {
54 QL_REQUIRE(!legData_[i].isPayer(), "CommoditySpreadOption: second leg is not relevant for short position.");
55 legPayers_.push_back(false);
56 }
57
58 // build legs
59 auto conLegData = legData_[i].concreteLegData();
60 commLegData_[i] = boost::dynamic_pointer_cast<CommodityFloatingLegData>(conLegData);
61 QL_REQUIRE(commLegData_[i], "CommoditySpreadOption leg data should be of type CommodityFloating");
62 isAveraged[i] = commLegData_[i]->isAveraged();
63 auto legBuilder = engineFactory->legBuilder(legData_[i].legType());
64 auto cflb = boost::dynamic_pointer_cast<CommodityFloatingLegBuilder>(legBuilder);
65 QL_REQUIRE(cflb, "Expected a CommodityFloatingLegBuilder for leg type " << legData_[i].legType());
66 Leg leg = cflb->buildLeg(legData_[i], engineFactory, requiredFixings_, config);
67
68 // setup the cf indexes
69 boost::shared_ptr<CommodityIndex> index{nullptr};
70 if(isAveraged[i]) {
71 index = boost::dynamic_pointer_cast<QuantExt::CommodityIndexedAverageCashFlow>(leg[0])->index();
72 } else {
73 index = boost::dynamic_pointer_cast<QuantExt::CommodityIndexedCashFlow>(leg[0])->index();
74 }
75 (i == (int)CommoditySpreadOptionPosition::Long)?longIndex=index:shortIndex=index;
76
77 // check ccy consistency
78 auto underlyingCcy = index->priceCurve()->currency();
79 auto tmpFx = commLegData_[i]->fxIndex();
80 if(tmpFx.empty()) {
81 std::string pos = i == 0 ? "Long" : "Short";
82 QL_REQUIRE(underlyingCcy.code() == settlementCcy_,
83 "CommoditySpreadOption: No FXIndex provided for the " << pos << "position");
84 }else {
85 auto domestic = npvCurrency_;
86 auto foreign = underlyingCcy.code();
87 FxIndex[i] = buildFxIndex(tmpFx, domestic, foreign, engineFactory->market(),
88 engineFactory->configuration(MarketContext::pricing));
89 // update required fixings. This is handled automatically in the leg builder in the averaging case
90 if(!isAveraged[i]) {
91 for (auto cf : leg) {
92 auto fixingDate = cf->date();
93 if (!FxIndex[i]->fixingCalendar().isBusinessDay(
94 fixingDate)) { // If fx index is not available for the cashflow pricing day,
95 // this ensures to require the previous valid one which will be used in pricing
96 // from fxIndex()->fixing(...)
97 Date adjustedFixingDate = FxIndex[i]->fixingCalendar().adjust(fixingDate, Preceding);
98 requiredFixings_.addFixingDate(adjustedFixingDate, tmpFx);
99 } else {
100 requiredFixings_.addFixingDate(fixingDate, tmpFx);
101 }
102 }
103 }
104 }
105 legs_.push_back(leg);
106 legCurrencies_.push_back(npvCurrency_); // all legs and cf are priced with the same ccy
107 }
108
109 vector<boost::shared_ptr<Instrument>> additionalInstruments;
110 //check cfs and build the instruments
111 QL_REQUIRE(legs_[0].size() == legs_[1].size(), "CommoditySpreadOption: the two legs must contain the same number of cashflows.");
112 for (size_t i =0; i<legs_[0].size();++i){ // I already checked the size is the same for both
113
114 Real lq = (isAveraged[CommoditySpreadOptionPosition::Long])?
115 boost::dynamic_pointer_cast<CommodityIndexedAverageCashFlow>(legs_[CommoditySpreadOptionPosition::Long][i])->quantity():
116 boost::dynamic_pointer_cast<CommodityIndexedCashFlow>(legs_[CommoditySpreadOptionPosition::Long][i])->quantity();
117
118 auto end_date_l = legs_[CommoditySpreadOptionPosition::Long][i]->date();
119
120 Real sq = (isAveraged[CommoditySpreadOptionPosition::Short])?
121 boost::dynamic_pointer_cast<CommodityIndexedAverageCashFlow>(legs_[CommoditySpreadOptionPosition::Short][i])->quantity():
122 boost::dynamic_pointer_cast<CommodityIndexedCashFlow>(legs_[CommoditySpreadOptionPosition::Short][i])->quantity();
123
124 auto end_date_s = legs_[CommoditySpreadOptionPosition::Short][i]->date();
125
126 QL_REQUIRE(QuantLib::close_enough(lq - sq, 0.0), "CommoditySpreadOption: all cashflows must refer to the same quantity");
127 QL_REQUIRE(end_date_l==end_date_s, "CommoditySpreadOption: the computation period for the two cashflows must be the same");
128 quantity_ = lq;
129
130 // set exercise date to the calculation period end
131 boost::shared_ptr<Exercise> exercise = boost::make_shared<EuropeanExercise>(end_date_l);
132 Date settlementDate{Date()};
133 if(i==legs_[0].size()-1){ // if the last flow must be settled with delay, set it here
134 settlementDate = settlementDate_.empty()?Date(): parseDate(settlementDate_);
135 }
136 if(settlementDate!=Date())
137 QL_REQUIRE(settlementDate>=exercise->lastDate(), "CommoditySpreadOption: Trade cannot be settled before exercise");
138
139 //maturity gets overwritten every time, and it is ok. If the last option is settled with delay, maturity is set to the settlement date.
140 maturity_ = (settlementDate == Date())?exercise->lastDate() : settlementDate;
141
142 // build the instrument for the i-th cfs
143 auto spreadOption = boost::make_shared<QuantExt::CommoditySpreadOption>(
144 boost::dynamic_pointer_cast<QuantExt::CommodityCashFlow>(legs_[CommoditySpreadOptionPosition::Long][i]),
145 boost::dynamic_pointer_cast<QuantExt::CommodityCashFlow>(legs_[CommoditySpreadOptionPosition::Short][i]),
146 exercise, quantity_, spreadStrike_, optionType,
147 settlementDate, FxIndex[CommoditySpreadOptionPosition::Long], FxIndex[CommoditySpreadOptionPosition::Short]);
148
149 // build and assign the engine
150 boost::shared_ptr<PricingEngine> engine = engineBuilder->engine(ccy, longIndex, shortIndex, id());
151 spreadOption->setPricingEngine(engine);
152 additionalInstruments.emplace_back(spreadOption);
153 }
154
155 // grep the last computation period instrument as main
156 auto qlInstrument = additionalInstruments.back();
157 additionalInstruments.pop_back(); // remove last instrument, it is already set as main one
158 // set the trade instrument
159 // as the first argument is always the long position and the second is always the short, multiplier is set to 1
160 instrument_ = boost::make_shared<VanillaInstrument>(qlInstrument, 1.0, additionalInstruments, std::vector<Real>(legs_[0].size()-1,1.0));
161}

163void CommoditySpreadOption::fromXML(XMLNode* node){
164 Trade::fromXML(node);
165
166 XMLNode* csoNode = XMLUtils::getChildNode(node, "CommoditySpreadOptionData");
167 QL_REQUIRE(csoNode, "No CommoditySpreadOptionData Node");
168 legData_.resize(2); // two legs expected
169 vector<XMLNode*> nodes = XMLUtils::getChildrenNodes(csoNode, "LegData");
170 QL_REQUIRE(nodes.size() == 2, "CommoditySpreadOption: Exactly two LegData nodes expected");
171 for (auto & node : nodes) {
172 auto ld = createLegData();
173 ld->fromXML(node);
174 legData_[!ld->isPayer()] = (*ld); // isPayer true = long position -> 0th element
175 }
176 QL_REQUIRE(legData_[CommoditySpreadOptionPosition::Long].isPayer() !=
177 legData_[CommoditySpreadOptionPosition::Short].isPayer(), "CommoditySpreadOption: both a long and a short Assets are required.");
178 spreadStrike_ = XMLUtils::getChildValueAsDouble(csoNode, "SpreadStrike", true);
179 callPut_ = XMLUtils::getChildValue(csoNode, "OptionType", false, "Call");
180 settlementDate_ = XMLUtils::getChildValue(csoNode, "SettlementDate",false,"");
181 settlementCcy_ = XMLUtils::getChildValue(csoNode, "Currency", true);
182}

184XMLNode* CommoditySpreadOption::toXML(XMLDocument& doc){
185 XMLNode* node = Trade::toXML(doc);
186
187 XMLNode* csoNode = doc.allocNode("CommoditySpreadOptionData");
188 XMLUtils::appendNode(node, csoNode);
189 for (size_t i = 0; i < legData_.size(); ++i) {
190 XMLUtils::appendNode(csoNode, legData_[i].toXML(doc));
191 }
192 XMLUtils::addChild(doc, csoNode, "SpreadStrike", spreadStrike_);
193 XMLUtils::addChild(doc, csoNode, "OptionType", callPut_);
194 XMLUtils::addChild(doc, csoNode, "Currency", settlementCcy_);
195 if(!settlementDate_.empty())
196 XMLUtils::addChild(doc, csoNode, "SettlementDate", settlementDate_);
197 return node;
198}
