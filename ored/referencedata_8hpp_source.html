35 //! Base class for reference data
36 /*! Each reference datum object is a subclass of this base and has it's own accessor functions.
37  * Instances of ReferenceDatum can be gotten from the ReferenceDataManager below, and then cast up as required.
38  * Each instance should be uniquely identified by it's type (which defines it's subclass, e.g. "Bond" for
39  * BondReferenceDatum) and it's id, which is a string. Here it can be any string but in applications there can be
40  * a naming scheme like ISIN for Bonds.
41  */
42 class ReferenceDatum : public XMLSerializable {
43 public:
44  //! Default Constructor
45  ReferenceDatum() {}
46  //! Base class constructor
47  ReferenceDatum(const std::string& type, const std::string& id) : type_(type), id_(id) {}
48 
49  //! setters
50  void setType(const string& type) { type_ = type; }
51  void setId(const string& id) { id_ = id; }
52 
53  //! getters
54  const std::string& type() const { return type_; }
55  const std::string& id() const { return id_; }
56 
57  void fromXML(XMLNode* node) override;
58  XMLNode* toXML(ore::data::XMLDocument& doc) override;
59 
60 private:
61  std::string type_;
62  std::string id_;
63 };

190 //! Interface for Reference Data lookups
191 /*! The ReferenceDataManager is a repository of ReferenceDatum objects.
192  *
193  * This is an interface, there is a concrete impl below that is BasicReferenceDataManager which is filebased.
194  * It is also possible to have a transactional implementation that will fetch reference data from a DB or via
195  * a restful call, possibly with a cache.
196  *
197  * An instance of this class is made available to Trades as they are built, this code be done with a singleton
198  * but it is clearer to pass a pointer to an instance when the Trade is built itself, as there is no access method
199  * for Trade::build() we instead pass a pointer to the Trade constructors, this is then populated when TradeBuilders
200  * are created for ORE+ trades, this also allows us to have custom ORE+ versions of Trades that overload the build()
201  * method to use this data in place of extracting it from XML.
202  *
203  * The actual Trade objects will take a copy of any reference data they need, this way they own all the required data
204  * and a call to "Trade::toXML()" will write out the "full" trade. For example we might load a CDS Index trade using
205  * reference data from which the basket is created, but if we call "toXML()" on that CDS Index the trade the whole
206  * basket will be written out.
207  */
208 class ReferenceDataManager {
209 public:
210  virtual ~ReferenceDataManager() {}
211  virtual bool hasData(const string& type, const string& id) const = 0;
212  virtual boost::shared_ptr<ReferenceDatum> getData(const string& type, const string& id) = 0;
213  virtual void add(const boost::shared_ptr<ReferenceDatum>& referenceDatum) = 0;
214 };
