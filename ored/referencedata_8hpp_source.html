39//! Base class for reference data
40/*! Each reference datum object is a subclass of this base and has it's own accessor functions.
41 * Instances of ReferenceDatum can be gotten from the ReferenceDataManager below, and then cast up as required.
42 * Each instance should be uniquely identified by it's type (which defines it's subclass, e.g. "Bond" for
43 * BondReferenceDatum) and it's id, which is a string. Here it can be any string but in applications there can be
44 * a naming scheme like ISIN for Bonds.
45 */
46class ReferenceDatum : public XMLSerializable {
47public:
48 //! Default Constructor
49 ReferenceDatum() : validFrom_(QuantLib::Date::minDate()) {}
50 //! Base class constructor
51 ReferenceDatum(const std::string& type, const std::string& id)
52 : type_(type), id_(id), validFrom_(QuantLib::Date::minDate()) {}
53 //! Base class constructor
54 ReferenceDatum(const std::string& type, const std::string& id, const QuantLib::Date& validFrom)
55 : type_(type), id_(id), validFrom_(validFrom) {}
56
57 //! setters
58 void setType(const string& type) { type_ = type; }
59 void setId(const string& id) { id_ = id; }
60 void setValidFrom(const QuantLib::Date& validFrom) { validFrom_ = validFrom; }
61
62 //! getters
63 const std::string& type() const { return type_; }
64 const std::string& id() const { return id_; }
65 const QuantLib::Date& validFrom() const { return validFrom_; }
66
67 void fromXML(XMLNode* node) override;
68 XMLNode* toXML(ore::data::XMLDocument& doc) override;
69
70private:
71 std::string type_;
72 std::string id_;
73 QuantLib::Date validFrom_;
74};

451//! Interface for Reference Data lookups
452/*! The ReferenceDataManager is a repository of ReferenceDatum objects.
453 *
454 * This is an interface, there is a concrete impl below that is BasicReferenceDataManager which is filebased.
455 * It is also possible to have a transactional implementation that will fetch reference data from a DB or via
456 * a restful call, possibly with a cache.
457 *
458 * An instance of this class is made available to Trades as they are built, this code be done with a singleton
459 * but it is clearer to pass a pointer to an instance when the Trade is built itself, as there is no access method
460 * for Trade::build() we instead pass a pointer to the Trade constructors, this is then populated when TradeBuilders
461 * are created for ORE+ trades, this also allows us to have custom ORE+ versions of Trades that overload the build()
462 * method to use this data in place of extracting it from XML.
463 *
464 * The actual Trade objects will take a copy of any reference data they need, this way they own all the required data
465 * and a call to "Trade::toXML()" will write out the "full" trade. For example we might load a CDS Index trade using
466 * reference data from which the basket is created, but if we call "toXML()" on that CDS Index the trade the whole
467 * basket will be written out.
468 */
469class ReferenceDataManager {
470public:
471 virtual ~ReferenceDataManager() {}
472 virtual bool hasData(const string& type, const string& id,
473 const QuantLib::Date& asof = QuantLib::Null<QuantLib::Date>()) const = 0;
474 virtual boost::shared_ptr<ReferenceDatum>
475 getData(const string& type, const string& id, const QuantLib::Date& asof = QuantLib::Null<QuantLib::Date>()) = 0;
476 virtual void add(const boost::shared_ptr<ReferenceDatum>& referenceDatum) = 0;
477};
