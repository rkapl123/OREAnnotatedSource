40 EquityDoubleTouchOption::EquityDoubleTouchOption(Envelope& env, OptionData option, BarrierData barrier,
41  const EquityUnderlying& equityUnderlying, string payoffCurrency,
42  double payoffAmount, string startDate, string calendar)
43  : ore::data::Trade("EquityDoubleTouchOption", env),
44  EquitySingleAssetDerivative("", env, equityUnderlying),
45  option_(option),
46  barrier_(barrier), startDate_(startDate), calendar_(calendar), payoffAmount_(payoffAmount),
47  payoffCurrency_(payoffCurrency) {
48  DoubleBarrier::Type barrierType = parseDoubleBarrierType(barrier_.type());
49  switch (barrierType) {
50  case DoubleBarrier::KnockIn:
51  type_ = "KnockIn";
52  break;
53  case DoubleBarrier::KnockOut:
54  type_ = "KnockOut";
55  break;
56  default:
57  QL_FAIL("unsupported barrier type " << barrierType);
58  }
59 }

61 void EquityDoubleTouchOption::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
62 
63  const boost::shared_ptr<Market> market = engineFactory->market();
64  Date start = ore::data::parseDate(startDate_);
65  Calendar cal = ore::data::parseCalendar(calendar_);
66 
67  QL_REQUIRE(tradeActions().empty(), "TradeActions not supported for FxOption");
68  QL_REQUIRE(option_.exerciseDates().size() == 1, "Invalid number of excercise dates");
69  QL_REQUIRE(barrier_.levels().size() == 2, "Invalid number of barrier levels");
70  QL_REQUIRE(barrier_.style().empty() || barrier_.style() == "American", "Only american barrier style suppported");
71 
72  // Parse trade data
73  string assetName = equityName();
74  Currency ccy = parseCurrency(payoffCurrency_);
75  Date expiryDate = parseDate(option_.exerciseDates().front());
76  DoubleBarrier::Type barrierType = parseDoubleBarrierType(barrier_.type());
77  bool payoffAtExpiry = option_.payoffAtExpiry();
78  double rebate = barrier_.rebate();
79  Position::Type positionType = parsePositionType(option_.longShort());
80 
81  QL_REQUIRE(rebate == 0, "Rebates not supported for EquityDoubleTouchOptions");
82  if (payoffAtExpiry == false) {
83  payoffAtExpiry = true;
84  DLOG("Payoff at hit not yet supported for EquityDoubleTouchOptions, setting to payoff at expiry");
85  }
86 
87  Real levelLow = barrier_.levels()[0].value();
88  Real levelHigh = barrier_.levels()[1].value();
89  QL_REQUIRE(levelLow < levelHigh, "barrier levels are not in ascending order");
90 
91 
92  // create payoff and exercise, as well as leg of underlying instrument
93  boost::shared_ptr<StrikedTypePayoff> payoff(new CashOrNothingPayoff(Option::Call, (levelLow + levelHigh) / 2, 1.0));
94  Leg leg;
95 
96  leg.push_back(boost::shared_ptr<CashFlow>(new SimpleCashFlow(1.0, expiryDate)));
97 
98  boost::shared_ptr<Exercise> exercise = boost::make_shared<EuropeanExercise>(expiryDate);
99 
100  boost::shared_ptr<Instrument> doubleTouch =
101  boost::make_shared<DoubleBarrierOption>(barrierType, levelLow, levelHigh, 0.0, payoff, exercise);
102  boost::shared_ptr<Instrument> underlying = boost::make_shared<Swap>(Leg(), leg);
103 
104 
105 
106  boost::shared_ptr<QuantExt::EquityIndex> eqIndex = engineFactory->market()->equityCurve(assetName).currentLink();
107 
108  // set pricing engines
109  boost::shared_ptr<EngineBuilder> builder = engineFactory->builder(tradeType_);
110  QL_REQUIRE(builder, "No builder found for " << tradeType_);
111  boost::shared_ptr<EquityDoubleTouchOptionEngineBuilder> eqDoubleTouchOptBuilder =
112  boost::dynamic_pointer_cast<EquityDoubleTouchOptionEngineBuilder>(builder);
113  doubleTouch->setPricingEngine(eqDoubleTouchOptBuilder->engine(assetName, ccy));
114  if (type_ == "KnockIn") {
115  // if a knock-in option is triggered it becomes a simple forward cashflow
116  // which we price as a swap
117  builder = engineFactory->builder("Swap");
118  QL_REQUIRE(builder, "No builder found for Swap");
119  boost::shared_ptr<SwapEngineBuilderBase> swapBuilder =
120  boost::dynamic_pointer_cast<SwapEngineBuilderBase>(builder);
121  underlying->setPricingEngine(swapBuilder->engine(parseCurrency(payoffCurrency_)));
122  }
123 
124  bool isLong = (positionType == Position::Long) ? true : false;
125 
126  std::vector<boost::shared_ptr<Instrument>> additionalInstruments;
127  std::vector<Real> additionalMultipliers;
128  Date lastPremiumDate =
129  addPremiums(additionalInstruments, additionalMultipliers, payoffAmount_, option_.premiumData(),
130  isLong ? -1.0 : 1.0, ccy, engineFactory, builder->configuration(MarketContext::pricing));
131 
132  Handle<Quote> spot = market->equitySpot(assetName);
133  instrument_ = boost::make_shared<DoubleBarrierOptionWrapper>(
134  doubleTouch, isLong, expiryDate, false, underlying, barrierType, spot, levelLow, levelHigh, 0, ccy, start, eqIndex, cal,
135  payoffAmount_, payoffAmount_, additionalInstruments, additionalMultipliers);
136  npvCurrency_ = payoffCurrency_;
137  notional_ = payoffAmount_;
138  notionalCurrency_ = payoffCurrency_;
139  maturity_ = std::max(lastPremiumDate, expiryDate);
140 
141  if (start != Date()) {
142  for (Date d = start; d <= expiryDate; d = eqIndex->fixingCalendar().advance(d, 1 * Days))
143  requiredFixings_.addFixingDate(d, "EQ-" + assetName, expiryDate);
144  }
145 
146  additionalData_["payoffAmount"] = payoffAmount_;
147  additionalData_["payoffCurrency"] = payoffCurrency_;
148 }

150 bool EquityDoubleTouchOption::checkBarrier(Real spot, Barrier::Type type, Real barrier) {
151  switch (type) {
152  case Barrier::DownIn:
153  case Barrier::DownOut:
154  return spot <= barrier;
155  case Barrier::UpIn:
156  case Barrier::UpOut:
157  return spot >= barrier;
158  default:
159  QL_FAIL("unknown barrier type " << type);
160  }
161 }

163 void EquityDoubleTouchOption::fromXML(XMLNode* node) {
164  Trade::fromXML(node);
165  XMLNode* eqNode = XMLUtils::getChildNode(node, "EquityDoubleTouchOptionData");
166  QL_REQUIRE(eqNode, "No EquityDoubleTouchOptionData Node");
167  option_.fromXML(XMLUtils::getChildNode(eqNode, "OptionData"));
168  barrier_.fromXML(XMLUtils::getChildNode(eqNode, "BarrierData"));
169  DoubleBarrier::Type barrierType = parseDoubleBarrierType(barrier_.type());
170  switch (barrierType) {
171  case DoubleBarrier::KnockIn:
172  type_ = "KnockIn";
173  break;
174  case DoubleBarrier::KnockOut:
175  type_ = "KnockOut";
176  break;
177  default:
178  QL_FAIL("unsupported barrier type " << barrierType);
179  }
180 
181  XMLNode* tmp = XMLUtils::getChildNode(eqNode, "Underlying");
182  if (!tmp)
183  tmp = XMLUtils::getChildNode(eqNode, "Name");
184  equityUnderlying_.fromXML(tmp);
185  payoffCurrency_ = XMLUtils::getChildValue(eqNode, "PayoffCurrency", true);
186  startDate_ = XMLUtils::getChildValue(eqNode, "StartDate", false);
187  calendar_ = XMLUtils::getChildValue(eqNode, "Calendar", false);
188  payoffAmount_ = XMLUtils::getChildValueAsDouble(eqNode, "PayoffAmount", true);
189 }

202  if (calendar_ != "")
203  XMLUtils::addChild(doc, eqNode, "Calendar", calendar_);
204 
205  return node;
