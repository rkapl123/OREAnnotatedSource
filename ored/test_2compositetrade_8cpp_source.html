104 // Synthetic Forward Test.. Per put-call parity a call - a put = a forward
105 BOOST_AUTO_TEST_CASE(testSyntheticForward) {
106  BOOST_TEST_MESSAGE("Testing SyntheticForwardTrade...");
107 
108  SavedSettings backup;
109 
110  InstrumentConventions::instance().setConventions(boost::make_shared<Conventions>());
111 
112  // build market
113  boost::shared_ptr<Market> market = boost::make_shared<TestMarket>();
114  Settings::instance().evaluationDate() = market->asofDate();
115  Date expiry = market->asofDate() + 6 * Months + 1 * Days;
116  ostringstream o;
117  o << QuantLib::io::iso_date(expiry);
118  string exp_str = o.str();
119 
120  // build EquityOption - expiry in 1 Year
121  OptionData callData("Long", "Call", "European", true, vector<string>(1, exp_str));
122  OptionData putData("Short", "Put", "European", true, vector<string>(1, exp_str));
123  Envelope env("CP1");
124  TradeStrike tradeStrike(95.0, "EUR");
125  boost::shared_ptr<Trade> eqCall =
126  boost::make_shared<EquityOption>(env, callData, EquityUnderlying("eurCorp"), "EUR", 1.0, tradeStrike);
127  eqCall->id() = "Long Call";
128  boost::shared_ptr<Trade> eqPut =
129  boost::make_shared<EquityOption>(env, putData, EquityUnderlying("eurCorp"), "EUR", 1.0, tradeStrike);
130  eqPut->id() = "Short Put";
131  CompositeTrade syntheticForward("EUR", {eqCall, eqPut}, "Mean", 0.0, env);
132  syntheticForward.id() = "Synthetic Forward Test";
133  ore::data::EquityForward eqFwd(env, "Long", EquityUnderlying("eurCorp"), "EUR", 1.0, exp_str, 95.0);
134 
135  // Build and price
136  boost::shared_ptr<EngineData> engineData = boost::make_shared<EngineData>();
137  engineData->model("EquityOption") = "BlackScholesMerton";
138  engineData->engine("EquityOption") = "AnalyticEuropeanEngine";
139  engineData->model("EquityForward") = "DiscountedCashflows";
140  engineData->engine("EquityForward") = "DiscountingEquityForwardEngine";
141  boost::shared_ptr<EngineFactory> engineFactory = boost::make_shared<EngineFactory>(engineData, market);
142  engineFactory->registerBuilder(boost::make_shared<EquityEuropeanOptionEngineBuilder>());
143  engineFactory->registerBuilder(boost::make_shared<EquityForwardEngineBuilder>());
144 
145  syntheticForward.build(engineFactory);
146  eqFwd.build(engineFactory);
147 
148  Real npv_composite = syntheticForward.instrument()->NPV();
149  Real npv_fwd = eqFwd.instrument()->NPV();
150 
151  BOOST_CHECK_CLOSE(npv_composite, npv_fwd, 0.01);
152  BOOST_CHECK_CLOSE(syntheticForward.notional(), eqFwd.notional(), 0.01);
153 }

171  string exp_str = o.str();
172 
173  // build EquityOption - expiry in 1 Year
174  OptionData callData("Long", "Call", "European", true, vector<string>(1, exp_str));
175  Envelope env("CP1");
176  TradeStrike tradeStrike_EUR(95.0, "EUR");
177  boost::shared_ptr<Trade> eurCall =
178  boost::make_shared<EquityOption>(env, callData, EquityUnderlying("eurCorp"), "EUR", 1.0, tradeStrike_EUR);
179  eurCall->id() = "EUR Call";
180  TradeStrike tradeStrike_USD(95.0, "USD");

182  boost::make_shared<EquityOption>(env, callData, EquityUnderlying("usdCorp"), "USD", 1.0, tradeStrike_USD);
183  usdCall->id() = "USD Call";

185  CompositeTrade usdComp("USD", {eurCall, usdCall}, "Sum", 0.0, env);
186  eurComp.id() = "EUR Combo Call Test";
187  usdComp.id() = "USD Combo Call Test";
188 
189  // Build and price
190  boost::shared_ptr<EngineData> engineData = boost::make_shared<EngineData>();
191  engineData->model("EquityOption") = "BlackScholesMerton";
192  engineData->engine("EquityOption") = "AnalyticEuropeanEngine";
193  boost::shared_ptr<EngineFactory> engineFactory = boost::make_shared<EngineFactory>(engineData, market);
194  engineFactory->registerBuilder(boost::make_shared<EquityEuropeanOptionEngineBuilder>());
195 
196  eurComp.build(engineFactory);
197  usdComp.build(engineFactory);
198 
199  Real npv_eur_composite = eurComp.instrument()->NPV();
200  Real npv_usd_composite = usdComp.instrument()->NPV();

202  Real npv_usdCall = usdCall->instrument()->NPV();

212  npv_usd_composite = usdComp.instrument()->NPV();
213  BOOST_CHECK_CLOSE(npv_usd_composite, npv_eurCall * 1.25 + npv_usdCall / 1.0, 0.01);
214 }
