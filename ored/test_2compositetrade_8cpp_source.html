46class TestMarket : public MarketImpl {
47public:
48 TestMarket(map<string, Handle<Quote>> fxRates = {{"EURUSD", Handle<Quote>(boost::make_shared<SimpleQuote>(1.2))}})
49 : MarketImpl(false) {
50 asof_ = Date(3, Feb, 2016);
51
52 // build discount
53 yieldCurves_[make_tuple(Market::defaultConfiguration, YieldCurveType::Discount, "EUR")] = flatRateYts(0.075);
54 yieldCurves_[make_tuple(Market::defaultConfiguration, YieldCurveType::Discount, "USD")] = flatRateYts(0.1);
55
56 // add fx rates
57 std::map<std::string, QuantLib::Handle<QuantLib::Quote>> quotes;
58 for (auto& fxRate : fxRates) {
59 quotes[fxRate.first] = fxRate.second;
60 }
61 fx_ = boost::make_shared<FXTriangulation>(quotes);
62
63 // build fx vols
64 fxVols_[make_pair(Market::defaultConfiguration, "EURUSD")] = flatRateFxv(0.10);
65
66 // add equity spots
67 equitySpots_[make_pair(Market::defaultConfiguration, "eurCorp")] =
68 Handle<Quote>(boost::make_shared<SimpleQuote>(100));
69 equitySpots_[make_pair(Market::defaultConfiguration, "usdCorp")] =
70 Handle<Quote>(boost::make_shared<SimpleQuote>(100));
71
72 // add dividend yield
73 yieldCurves_[make_tuple(Market::defaultConfiguration, YieldCurveType::EquityDividend, "eurCorp")] =
74 flatRateYts(0.05);
75 yieldCurves_[make_tuple(Market::defaultConfiguration, YieldCurveType::EquityDividend, "usdCorp")] =
76 flatRateYts(0.05);
77
78 // add equity curve
79 equityCurves_[make_pair(Market::defaultConfiguration, "eurCorp")] =
80 Handle<EquityIndex2>(boost::make_shared<EquityIndex2>(
81 "eurCorp", TARGET(), parseCurrency("EUR"), equitySpot("eurCorp"),
82 yieldCurve(YieldCurveType::Discount, "EUR"), yieldCurve(YieldCurveType::EquityDividend, "eurCorp")));
83 equityCurves_[make_pair(Market::defaultConfiguration, "usdCorp")] =
84 Handle<EquityIndex2>(boost::make_shared<EquityIndex2>(
85 "usdCorp", TARGET(), parseCurrency("USD"), equitySpot("usdCorp"),
86 yieldCurve(YieldCurveType::Discount, "USD"), yieldCurve(YieldCurveType::EquityDividend, "usdCorp")));
87
88 // build equity vols
89 equityVols_[make_pair(Market::defaultConfiguration, "eurCorp")] = flatRateFxv(0.20);
90 equityVols_[make_pair(Market::defaultConfiguration, "usdCorp")] = flatRateFxv(0.20);
91 }

93private:
94 Handle<YieldTermStructure> flatRateYts(Real forward) {
95 boost::shared_ptr<YieldTermStructure> yts(new FlatForward(0, NullCalendar(), forward, ActualActual(ActualActual::ISDA)));
96 return Handle<YieldTermStructure>(yts);
97 }
98 Handle<BlackVolTermStructure> flatRateFxv(Volatility forward) {
99 boost::shared_ptr<BlackVolTermStructure> fxv(new BlackConstantVol(0, NullCalendar(), forward, ActualActual(ActualActual::ISDA)));
100 return Handle<BlackVolTermStructure>(fxv);
101 }

105// Synthetic Forward Test.. Per put-call parity a call - a put = a forward
106BOOST_AUTO_TEST_CASE(testSyntheticForward) {
107 BOOST_TEST_MESSAGE("Testing SyntheticForwardTrade...");
108
109 SavedSettings backup;
110
111 InstrumentConventions::instance().setConventions(boost::make_shared<Conventions>());
112
113 // build market
114 boost::shared_ptr<Market> market = boost::make_shared<TestMarket>();
115 Settings::instance().evaluationDate() = market->asofDate();
116 Date expiry = market->asofDate() + 6 * Months + 1 * Days;
117 ostringstream o;
118 o << QuantLib::io::iso_date(expiry);
119 string exp_str = o.str();
120
121 // build EquityOption - expiry in 1 Year
122 OptionData callData("Long", "Call", "European", true, vector<string>(1, exp_str));
123 OptionData putData("Short", "Put", "European", true, vector<string>(1, exp_str));
124 Envelope env("CP1");
125 TradeStrike tradeStrike(95.0, "EUR");
126 boost::shared_ptr<Trade> eqCall =
127 boost::make_shared<EquityOption>(env, callData, EquityUnderlying("eurCorp"), "EUR", 1.0, tradeStrike);
128 eqCall->id() = "Long Call";
129 boost::shared_ptr<Trade> eqPut =
130 boost::make_shared<EquityOption>(env, putData, EquityUnderlying("eurCorp"), "EUR", 1.0, tradeStrike);
131 eqPut->id() = "Short Put";
132 CompositeTrade syntheticForward("EUR", {eqCall, eqPut}, "Mean", 0.0, env);
133 syntheticForward.id() = "Synthetic Forward Test";
134 ore::data::EquityForward eqFwd(env, "Long", EquityUnderlying("eurCorp"), "EUR", 1.0, exp_str, 95.0);
135
136 // Build and price
137 boost::shared_ptr<EngineData> engineData = boost::make_shared<EngineData>();
138 engineData->model("EquityOption") = "BlackScholesMerton";
139 engineData->engine("EquityOption") = "AnalyticEuropeanEngine";
140 engineData->model("EquityForward") = "DiscountedCashflows";
141 engineData->engine("EquityForward") = "DiscountingEquityForwardEngine";
142 boost::shared_ptr<EngineFactory> engineFactory = boost::make_shared<EngineFactory>(engineData, market);
143
144 syntheticForward.build(engineFactory);
145 eqFwd.build(engineFactory);
146
147 Real npv_composite = syntheticForward.instrument()->NPV();
148 Real npv_fwd = eqFwd.instrument()->NPV();
149
150 BOOST_CHECK_CLOSE(npv_composite, npv_fwd, 0.01);
151 BOOST_CHECK_CLOSE(syntheticForward.notional(), eqFwd.notional(), 0.01);
152}

154// Simple combination of 2 options in different currencies
155BOOST_AUTO_TEST_CASE(testMultiCcyComposite) {
156 BOOST_TEST_MESSAGE("Testing SyntheticForwardTrade...");
157
158 SavedSettings backup;
159
160 InstrumentConventions::instance().setConventions(boost::make_shared<Conventions>());
161
162 // build market
163 boost::shared_ptr<SimpleQuote> eurusdRate(boost::make_shared<SimpleQuote>(1.2));
164 map<string, Handle<Quote>> fxRates = {{"EURUSD", Handle<Quote>(eurusdRate)}};
165 boost::shared_ptr<Market> market = boost::make_shared<TestMarket>(fxRates);
166 Settings::instance().evaluationDate() = market->asofDate();
167 Date expiry = market->asofDate() + 6 * Months + 1 * Days;
168 ostringstream o;
169 o << QuantLib::io::iso_date(expiry);
170 string exp_str = o.str();
171
172 // build EquityOption - expiry in 1 Year
173 OptionData callData("Long", "Call", "European", true, vector<string>(1, exp_str));
174 Envelope env("CP1");
175 TradeStrike tradeStrike_EUR(95.0, "EUR");
176 boost::shared_ptr<Trade> eurCall =
177 boost::make_shared<EquityOption>(env, callData, EquityUnderlying("eurCorp"), "EUR", 1.0, tradeStrike_EUR);
178 eurCall->id() = "EUR Call";
179 TradeStrike tradeStrike_USD(95.0, "USD");
180 boost::shared_ptr<Trade> usdCall =
181 boost::make_shared<EquityOption>(env, callData, EquityUnderlying("usdCorp"), "USD", 1.0, tradeStrike_USD);
182 usdCall->id() = "USD Call";
183 CompositeTrade eurComp("EUR", {eurCall, usdCall}, "Sum", 0.0, env);
184 CompositeTrade usdComp("USD", {eurCall, usdCall}, "Sum", 0.0, env);
185 eurComp.id() = "EUR Combo Call Test";
186 usdComp.id() = "USD Combo Call Test";
187
188 // Build and price
189 boost::shared_ptr<EngineData> engineData = boost::make_shared<EngineData>();
190 engineData->model("EquityOption") = "BlackScholesMerton";
191 engineData->engine("EquityOption") = "AnalyticEuropeanEngine";
192 boost::shared_ptr<EngineFactory> engineFactory = boost::make_shared<EngineFactory>(engineData, market);
193
194 eurComp.build(engineFactory);
195 usdComp.build(engineFactory);
196
197 Real npv_eur_composite = eurComp.instrument()->NPV();
198 Real npv_usd_composite = usdComp.instrument()->NPV();
199 Real npv_eurCall = eurCall->instrument()->NPV();
200 Real npv_usdCall = usdCall->instrument()->NPV();
201
202 BOOST_CHECK_CLOSE(npv_eur_composite, npv_eurCall * 1.0 + npv_usdCall / 1.2, 0.01);
203 BOOST_CHECK_CLOSE(npv_usd_composite, npv_eurCall * 1.2 + npv_usdCall / 1.0, 0.01);
204 // Check that the notional is calulated correctly.
205 BOOST_CHECK_CLOSE(usdComp.notional(), eurCall->notional() * 2.2, 0.01);
206
207 // let's bump the fx to check that observation is working
208 auto testMarket = boost::dynamic_pointer_cast<TestMarket>(market);
209 eurusdRate->setValue(1.25);
210 npv_usd_composite = usdComp.instrument()->NPV();
211 BOOST_CHECK_CLOSE(npv_usd_composite, npv_eurCall * 1.25 + npv_usdCall / 1.0, 0.01);
212}
