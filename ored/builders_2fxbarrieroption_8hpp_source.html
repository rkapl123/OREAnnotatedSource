39 //! Engine Builder for European FX Barrier Options
40 /*! Pricing engines are cached by currency pair
41 
42  \ingroup portfolio
43  */
44 class FxBarrierOptionEngineBuilder
45  : public ore::data::CachingPricingEngineBuilder<string, const Currency&, const Currency&, const Date&> {
46 
47 protected:
48  FxBarrierOptionEngineBuilder(const string& model, const string& engine)
49  : CachingEngineBuilder(model, engine, {"FxBarrierOption"}) {}
50 
51  virtual string keyImpl(const Currency& forCcy, const Currency& domCcy, const Date& expiryDate) override {
52  return forCcy.code() + "/" + domCcy.code() + "/" + ore::data::to_string(expiryDate);
53  }

55  boost::shared_ptr<GeneralizedBlackScholesProcess>
56  getBlackScholesProcess(const Currency& forCcy, const Currency& domCcy, const std::vector<Time>& timePoints = {}) {
57  const string& ccyPairCode = forCcy.code() + domCcy.code();
58  Handle<BlackVolTermStructure> vol =
59  market_->fxVol(ccyPairCode, configuration(ore::data::MarketContext::pricing));
60  if (!timePoints.empty()) {
61  vol = Handle<BlackVolTermStructure>(
62  boost::make_shared<QuantExt::BlackMonotoneVarVolTermStructure>(vol, timePoints));
63  vol->enableExtrapolation();
64  }
65  return boost::make_shared<GeneralizedBlackScholesProcess>(
66  market_->fxRate(ccyPairCode, configuration(ore::data::MarketContext::pricing)),
67  market_->discountCurve(forCcy.code(), configuration(ore::data::MarketContext::pricing)),
68  market_->discountCurve(domCcy.code(), configuration(ore::data::MarketContext::pricing)), vol);
69  }
70 };

84 class FxBarrierOptionFDEngineBuilder : public FxBarrierOptionEngineBuilder {
85 public:
86  FxBarrierOptionFDEngineBuilder() : FxBarrierOptionEngineBuilder("GarmanKohlhagen", "FdBlackScholesBarrierEngine") {}
87 
88 protected:
89  virtual boost::shared_ptr<PricingEngine> engineImpl(const Currency& forCcy, const Currency& domCcy,
90  const Date& expiryDate) override {
91  // We follow the way FdBlackScholesBarrierEngine determines maturity for time grid generation
92  Handle<YieldTermStructure> riskFreeRate =
93  market_->discountCurve(domCcy.code(), configuration(ore::data::MarketContext::pricing));
94  Time expiry = riskFreeRate->dayCounter().yearFraction(riskFreeRate->referenceDate(),
95  std::max(riskFreeRate->referenceDate(), expiryDate));
96 
97  FdmSchemeDesc scheme = ore::data::parseFdmSchemeDesc(engineParameter("Scheme"));
98  Size tGrid = std::max<Size>(1, (Size)(ore::data::parseInteger(engineParameter("TimeGridPerYear")) * expiry));
99  Size xGrid = ore::data::parseInteger(engineParameter("XGrid"));
100  Size dampingSteps = ore::data::parseInteger(engineParameter("DampingSteps"));
101  bool monotoneVar = ore::data::parseBool(engineParameter("EnforceMonotoneVariance", {}, false, "true"));
102 
103  boost::shared_ptr<GeneralizedBlackScholesProcess> gbsp;
104 
105  if (monotoneVar) {
106  // Replicate the construction of time grid in FiniteDifferenceModel::rollbackImpl
107  // This time grid is required to build a BlackMonotoneVarVolTermStructure which
108  // ensures monotonic variance along the time grid
109  std::vector<Time> timePoints(tGrid + 1);
110  Array timePointsArray(tGrid, expiry, -expiry / tGrid);
111  timePoints[0] = 0.0;
112  for (Size i = 0; i < tGrid; i++)
113  timePoints[timePoints.size() - i - 1] = timePointsArray[i];
114  timePoints.insert(std::upper_bound(timePoints.begin(), timePoints.end(), 0.99 / 365), 0.99 / 365);
115  gbsp = getBlackScholesProcess(forCcy, domCcy, timePoints);
116  } else {
117  gbsp = getBlackScholesProcess(forCcy, domCcy);
118  }
119  return boost::make_shared<FdBlackScholesBarrierEngine>(gbsp, tGrid, xGrid, dampingSteps, scheme);
120  }
121 };
