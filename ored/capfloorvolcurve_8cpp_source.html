54 // Currently, only two possibilities for variable InterpolateOn: TermVolatilities and OptionletVolatilities
55 // Here, we convert the value to a bool for use in building the structures. May need to broaden if we add more.
56 bool interpOnOpt(CapFloorVolatilityCurveConfig& config) {
57  QL_REQUIRE(config.interpolateOn() == "TermVolatilities" || config.interpolateOn() == "OptionletVolatilities",
58  "Expected InterpolateOn to be one of TermVolatilities or OptionletVolatilities");
59  return config.interpolateOn() == "OptionletVolatilities";
60 }

62 CapFloorVolCurve::CapFloorVolCurve(const Date& asof, const CapFloorVolatilityCurveSpec& spec, const Loader& loader,
63  const CurveConfigurations& curveConfigs, boost::shared_ptr<IborIndex> iborIndex,
64  Handle<YieldTermStructure> discountCurve)
65  : spec_(spec) {
66 
67  try {
68  // The configuration
69  const boost::shared_ptr<CapFloorVolatilityCurveConfig>& config =
70  curveConfigs.capFloorVolCurveConfig(spec_.curveConfigID());
71 
72  // Read the shift early if the configured volatility type is shifted lognormal
73  Real shift = 0.0;
74  if (config->volatilityType() == CfgVolType::ShiftedLognormal) {
75  shift = shiftQuote(asof, *config, loader);
76  }
77 
78  // There are three possible cap floor configurations
79  if (config->type() == CfgType::Atm) {
80  atmOptCurve(asof, *config, loader, iborIndex, discountCurve, shift);
81  } else if (config->type() == CfgType::Surface || config->type() == CfgType::SurfaceWithAtm) {
82  optSurface(asof, *config, loader, iborIndex, discountCurve, shift);
83  } else {
84  QL_FAIL("Unexpected type (" << static_cast<int>(config->type()) << ") for cap floor config "
85  << config->curveID());
86  }
87 
88  // Turn on or off extrapolation
89  capletVol_->enableExtrapolation(config->extrapolate());
90 
91  } catch (exception& e) {
92  QL_FAIL("cap/floor vol curve building failed :" << e.what());
93  } catch (...) {
94  QL_FAIL("cap/floor vol curve building failed: unknown error");
95  }
96 
97  // force bootstrap so that errors are thrown during the build, not later
98  capletVol_->volatility(QL_EPSILON, capletVol_->minStrike());
99 }

260

262  bool includeAtm = config.includeAtm();

264  if (includeAtm) {

553 boost::shared_ptr<QuantExt::CapFloorTermVolSurface>
554 CapFloorVolCurve::capSurface(const Date& asof, CapFloorVolatilityCurveConfig& config, const Loader& loader) const {
555 
556  // Map to store the quote values that we load with key (period, strike) where strike
557  // needs a custom comparator to avoid == with double
558  auto comp = [](const pair<Period, Rate>& a, const pair<Period, Rate>& b) {
559  return (a.first < b.first) || (!(b.first < a.first) && (a.second < b.second && !close(a.second, b.second)));
560  };
561  map<pair<Period, Rate>, Real, decltype(comp)> volQuotes(comp);
562 
563  bool optionalQuotes = config.optionalQuotes();
564  Size quoteCounter = 0;
565  bool quoteRelevant = false;
566  bool tenorRelevant = false;
567  bool strikeRelevant = false;
568 
569  vector<Real> qtStrikes;
570  vector<Real> qtData;
571  vector<Period> qtTenors;
572  Period tenor = parsePeriod(config.iborTenor());
573  string currency = config.currency();
574  vector<Period> configTenors = parseVectorOfValues<Period>(config.tenors(), &parsePeriod);
575 
576  for (auto& md : loader.loadQuotes(asof)) {
577  if (md->asofDate() == asof && md->instrumentType() == MarketDatum::InstrumentType::CAPFLOOR &&
578  md->quoteType() == config.quoteType()) {
579 
580  boost::shared_ptr<CapFloorQuote> cfq = boost::dynamic_pointer_cast<CapFloorQuote>(md);
581  if (cfq->ccy() == currency && cfq->underlying() == tenor && !cfq->atm()) {
582  auto j = std::find(configTenors.begin(), configTenors.end(), cfq->term());
583  tenorRelevant = j != configTenors.end();
584 
585  Real strike = cfq->strike();
586  auto i = std::find_if(config.strikes().begin(), config.strikes().end(),
587  [&strike](const std::string& x) {
588  return close_enough(parseReal(x), strike);
589  });
590  strikeRelevant = i != config.strikes().end();
591 
592  quoteRelevant = strikeRelevant && tenorRelevant;
593 
594  if (quoteRelevant) {
595  // if we have optional quotes we just make vectors of all quotes and let the sparse surface
596  // handle them
597  quoteCounter++;
598  if (optionalQuotes) {
599  qtStrikes.push_back(cfq->strike());
600  qtTenors.push_back(cfq->term());
601  qtData.push_back(cfq->quote()->value());
602  }
603  auto key = make_pair(cfq->term(), cfq->strike());
604  auto r = volQuotes.insert(make_pair(key, cfq->quote()->value()));
605  QL_REQUIRE(r.second, "Duplicate cap floor quote in config " << config.curveID() << ", with underlying tenor " << tenor <<
606  " and currency " << currency << ", for tenor " << key.first << " and strike " << key.second);
607  }
608  }
609  }
610  }
611 
612  Size totalQuotes = config.tenors().size() * config.strikes().size();
613  if (quoteCounter < totalQuotes) {
614  WLOG("Found only " << quoteCounter << " out of " << totalQuotes << " quotes for CapFloor surface " << config.curveID());
615  }
616 
617  vector<Period> tenors = parseVectorOfValues<Period>(config.tenors(), &parsePeriod);
618  vector<Rate> strikes = parseVectorOfValues<Real>(config.strikes(), &parseReal);
619  Matrix vols(tenors.size(), strikes.size());
620  for (Size i = 0; i < tenors.size(); i++) {
621  for (Size j = 0; j < strikes.size(); j++) {
622  auto key = make_pair(tenors[i], strikes[j]);
623  auto it = volQuotes.find(key);
624  if (!optionalQuotes) {
625  QL_REQUIRE(it != volQuotes.end(), "Quote with tenor " << key.first << " and strike " << key.second
626  << " not loaded for cap floor config "
627  << config.curveID());
628  // Organise the values in to a square matrix
629  vols[i][j] = it->second;
630  } else {
631  if (it == volQuotes.end()) {
632  DLOG("Could not find quote with tenor " << key.first << " and strike " << key.second <<
633  " for cap floor config " << config.curveID());
634  }
635  }
636  }
637  }
638 
639  DLOG("Found " << quoteCounter << " quotes for capfloor surface " << config.curveID());
640  if (optionalQuotes) {
641  QL_REQUIRE(quoteCounter > 0, "No Quotes provided for CapFloor surface " << config.curveID());
642  if (config.interpolationMethod() == CapFloorTermVolSurfaceExact::Bilinear) {
643  return boost::make_shared<QuantExt::CapFloorTermVolSurfaceSparse<Linear, Linear>>(config.settleDays(), config.calendar(),
644  config.businessDayConvention(), config.dayCounter(), qtTenors, qtStrikes, qtData, false, false);
645  } else if (config.interpolationMethod() == CapFloorTermVolSurfaceExact::BicubicSpline) {
646  return boost::make_shared<QuantExt::CapFloorTermVolSurfaceSparse<Cubic, Cubic>>(config.settleDays(), config.calendar(),
647  config.businessDayConvention(), config.dayCounter(), qtTenors, qtStrikes, qtData, false, false);
648  } else {
649  QL_FAIL("Invalid Interpolation method for capfloor surface " << config.curveID() << ", must be either "
650  << CapFloorTermVolSurfaceExact::Bilinear << " or " << CapFloorTermVolSurfaceExact::BicubicSpline << ".");
651  }
652  } else {
653  // Return for the cap floor term volatility surface
654  return boost::make_shared<QuantExt::CapFloorTermVolSurfaceExact>(config.settleDays(), config.calendar(),
655  config.businessDayConvention(), tenors, strikes, vols,
656  config.dayCounter(), config.interpolationMethod());
657  }

661 CapFloorVolCurve::atmCurve(const Date& asof, CapFloorVolatilityCurveConfig& config, const Loader& loader) const {
662 
663  // Map to store the quote values
664  map<Period, Handle<Quote>> volQuotes;
665 
666  bool optionalQuotes = config.optionalQuotes();
667  Period tenor = parsePeriod(config.iborTenor());
668  string currency = config.currency();
669  // Load the relevant quotes
670  for (auto& md : loader.loadQuotes(asof)) {
671 
672  if (md->asofDate() == asof && md->instrumentType() == MarketDatum::InstrumentType::CAPFLOOR &&
673  md->quoteType() == config.quoteType()) {
674 
675  boost::shared_ptr<CapFloorQuote> cfq = boost::dynamic_pointer_cast<CapFloorQuote>(md);
676  if (cfq->ccy() == currency && cfq->underlying() == tenor && cfq->atm()) {
677  auto j = std::find(config.atmTenors().begin(), config.atmTenors().end(), to_string(cfq->term()));
678  if (j != config.atmTenors().end()) {
679  auto r = volQuotes.insert(make_pair(cfq->term(), cfq->quote()));
680  QL_REQUIRE(r.second, "Duplicate ATM cap floor quote in config " << config.curveID() << " for tenor "
681  << cfq->term());
682  }
683  }
684  }
685  }
686 
687  // Check that the loaded quotes cover all of the configured ATM tenors
688  vector<Period> tenors = parseVectorOfValues<Period>(config.atmTenors(), &parsePeriod);
689  vector<Handle<Quote>> vols;
690  vector<Period> quoteTenors;
691  if (!optionalQuotes) {
692  vols.resize(tenors.size());
693  quoteTenors = tenors;
694  }
695  for (Size i = 0; i < tenors.size(); i++) {
696  auto it = volQuotes.find(tenors[i]);
697  if (!optionalQuotes) {
698  QL_REQUIRE(it != volQuotes.end(),
699  "ATM cap floor quote in config " << config.curveID() << " for tenor " << tenors[i] << " not found ");
700  vols[i] = it->second;
701  } else {
702  if (it == volQuotes.end()) {
703  DLOG("Could not find ATM cap floor quote with tenor " << tenors[i] << " for cap floor config " << config.curveID());
704  } else {
705  vols.push_back(it->second);
706  quoteTenors.push_back(it->first);
707  }
708  }
709  }
710 
711  if (optionalQuotes) {
712  QL_REQUIRE(vols.size() > 0, "No ATM cap floor quotes found for cap floor config " << config.curveID());
713  if (vols.size() == 1)
714  WLOG("Only one ATM cap floor quote found for cap floor config " << config.curveID() << ", using constant volatility");
715  }
716 
717  // Return for the cap floor ATM term volatility curve
718  // The interpolation here is also based on the interpolation method parameter in the configuration
719  // Flat first period is true by default (see ctor)
720  if (config.interpolationMethod() == CftvsInterp::BicubicSpline) {
721  if (config.flatExtrapolation()) {
722  return boost::make_shared<InterpolatedCapFloorTermVolCurve<CubicFlat>>(
723  config.settleDays(), config.calendar(), config.businessDayConvention(), quoteTenors, vols,
724  config.dayCounter());
725  } else {
726  return boost::make_shared<InterpolatedCapFloorTermVolCurve<Cubic>>(config.settleDays(), config.calendar(),
727  config.businessDayConvention(), quoteTenors,
728  vols, config.dayCounter());
729  }
730  } else if (config.interpolationMethod() == CftvsInterp::Bilinear) {
731  if (config.flatExtrapolation()) {
732  return boost::make_shared<InterpolatedCapFloorTermVolCurve<LinearFlat>>(
733  config.settleDays(), config.calendar(), config.businessDayConvention(), quoteTenors, vols,
734  config.dayCounter());
735  } else {
736  return boost::make_shared<InterpolatedCapFloorTermVolCurve<Linear>>(config.settleDays(), config.calendar(),
737  config.businessDayConvention(), quoteTenors,
738  vols, config.dayCounter());
739  }
740  } else {
741  QL_FAIL("Cap floor config " << config.curveID() << " has unexpected interpolation method "
742  << static_cast<int>(config.interpolationMethod()));
743  }

746 Real CapFloorVolCurve::shiftQuote(const QuantLib::Date& asof, CapFloorVolatilityCurveConfig& config,
747  const Loader& loader) const {
748 
749  QL_REQUIRE(config.volatilityType() == CfgVolType::ShiftedLognormal,
750  "Method shiftQuote should not be called with a config who's volatility type is not ShiftedLognormal");
751 
752  // Search for the shift quote in the configured quotes
753  for (const string& quoteId : config.quotes()) {
754 
755  boost::shared_ptr<MarketDatum> md = loader.get(quoteId, asof);
756 
757  // If it is a shift quote
758  if (boost::shared_ptr<CapFloorShiftQuote> sq = boost::dynamic_pointer_cast<CapFloorShiftQuote>(md)) {
759  return sq->quote()->value();
760  }
761  }
762 
763  QL_FAIL("Could not find a shift quote for cap floor config " << config.curveID());
764 }
765 
766 // This method is used to pull out the stripped optionlets from the QuantExt::OptionletStripper instance that is
767 // bootstrapped. The reason is that we do not want all of the cap floor helpers and their coupons in scope during
768 // a potential XVA run as this leads to delays when fixings are updated.
769 boost::shared_ptr<StrippedOptionlet> CapFloorVolCurve::transform(const QuantExt::OptionletStripper& os) const {
770 
771  vector<vector<Handle<Quote>>> vols(os.optionletFixingDates().size());
772  for (Size i = 0; i < os.optionletFixingDates().size(); i++) {
773  for (Size j = 0; j < os.optionletVolatilities(i).size(); j++) {
774  vols[i].push_back(Handle<Quote>(boost::make_shared<SimpleQuote>(os.optionletVolatilities(i)[j])));
775  }
776  }
777 
778  boost::shared_ptr<StrippedOptionlet> res = boost::make_shared<StrippedOptionlet>(
779  os.settlementDays(), os.calendar(), os.businessDayConvention(), os.iborIndex(), os.optionletFixingDates(),
780  os.optionletStrikes(0), vols, os.dayCounter(), os.volatilityType(), os.displacement());
781 
782  res->unregisterWithAll();
783 
784  return res;
785 }
