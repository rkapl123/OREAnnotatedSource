58 // Currently, only two possibilities for variable InterpolateOn: TermVolatilities and OptionletVolatilities
59 // Here, we convert the value to a bool for use in building the structures. May need to broaden if we add more.
60 bool interpOnOpt(CapFloorVolatilityCurveConfig& config) {
61  QL_REQUIRE(config.interpolateOn() == "TermVolatilities" || config.interpolateOn() == "OptionletVolatilities",

63  return config.interpolateOn() == "OptionletVolatilities";

66 CapFloorVolCurve::CapFloorVolCurve(
67  const Date& asof, const CapFloorVolatilityCurveSpec& spec, const Loader& loader,
68  const CurveConfigurations& curveConfigs, boost::shared_ptr<IborIndex> iborIndex,
69  Handle<YieldTermStructure> discountCurve, const boost::shared_ptr<IborIndex> sourceIndex,
70  const boost::shared_ptr<IborIndex> targetIndex,
71  const std::map<std::string, boost::shared_ptr<ore::data::CapFloorVolCurve>>& requiredCapFloorVolCurves,
72  const bool buildCalibrationInfo)
73  : spec_(spec) {
74 
75  try {
76  // The configuration
77  const boost::shared_ptr<CapFloorVolatilityCurveConfig>& config =
78  curveConfigs.capFloorVolCurveConfig(spec_.curveConfigID());
79 
80  if (!config->proxySourceCurveId().empty()) {
81  // handle proxy vol surfaces
82  buildProxyCurve(*config, sourceIndex, targetIndex, requiredCapFloorVolCurves);
83  } else {
84  // Read the shift early if the configured volatility type is shifted lognormal
85  Real shift = 0.0;
86  if (config->volatilityType() == CfgVolType::ShiftedLognormal) {
87  shift = shiftQuote(asof, *config, loader);
88  }
89 
90  // There are three possible cap floor configurations
91  if (config->type() == CfgType::Atm) {
92  atmOptCurve(asof, *config, loader, iborIndex, discountCurve, shift);
93  } else if (config->type() == CfgType::Surface || config->type() == CfgType::SurfaceWithAtm) {
94  optSurface(asof, *config, loader, iborIndex, discountCurve, shift);
95  } else {
96  QL_FAIL("Unexpected type (" << static_cast<int>(config->type()) << ") for cap floor config "
97  << config->curveID());
98  }
99  // Turn on or off extrapolation
100  capletVol_->enableExtrapolation(config->extrapolate());
101  }
102 
103  // Build calibration info
104  if (buildCalibrationInfo) {
105  this->buildCalibrationInfo(asof, curveConfigs, config, iborIndex);
106  }
107 
108  } catch (exception& e) {
109  QL_FAIL("cap/floor vol curve building failed :" << e.what());
110  } catch (...) {
111  QL_FAIL("cap/floor vol curve building failed: unknown error");
112  }
113 
114  // force bootstrap so that errors are thrown during the build, not later
115  capletVol_->volatility(QL_EPSILON, capletVol_->minStrike());
116 }

594 boost::shared_ptr<QuantExt::CapFloorTermVolSurface>
595 CapFloorVolCurve::capSurface(const Date& asof, CapFloorVolatilityCurveConfig& config, const Loader& loader) const {
596 
597  // Map to store the quote values that we load with key (period, strike) where strike
598  // needs a custom comparator to avoid == with double
599  auto comp = [](const pair<Period, Rate>& a, const pair<Period, Rate>& b) {
600  return (a.first < b.first) || (!(b.first < a.first) && (a.second < b.second && !close(a.second, b.second)));
601  };
602  map<pair<Period, Rate>, Real, decltype(comp)> volQuotes(comp);
603 
604  bool optionalQuotes = config.optionalQuotes();
605  Size quoteCounter = 0;
606  bool quoteRelevant = false;
607  bool tenorRelevant = false;
608  bool strikeRelevant = false;
609 
610  vector<Real> qtStrikes;
611  vector<Real> qtData;
612  vector<Period> qtTenors;
613  Period tenor = parsePeriod(config.indexTenor()); // 1D, 1M, 3M, 6M, 12M
614  string currency = config.currency();
615  vector<Period> configTenors = parseVectorOfValues<Period>(config.tenors(), &parsePeriod);
616 
617  for (auto& md : loader.loadQuotes(asof)) {
618  if (md->asofDate() == asof && md->instrumentType() == MarketDatum::InstrumentType::CAPFLOOR &&
619  md->quoteType() == config.quoteType()) {
620 
621  boost::shared_ptr<CapFloorQuote> cfq = boost::dynamic_pointer_cast<CapFloorQuote>(md);
622  if (cfq->ccy() == currency && cfq->underlying() == tenor && !cfq->atm()) {
623  auto j = std::find(configTenors.begin(), configTenors.end(), cfq->term());
624  tenorRelevant = j != configTenors.end();
625 
626  Real strike = cfq->strike();
627  auto i = std::find_if(config.strikes().begin(), config.strikes().end(),
628  [&strike](const std::string& x) {
629  return close_enough(parseReal(x), strike);
630  });
631  strikeRelevant = i != config.strikes().end();
632 
633  quoteRelevant = strikeRelevant && tenorRelevant;
634 
635  if (quoteRelevant) {
636  // if we have optional quotes we just make vectors of all quotes and let the sparse surface
637  // handle them
638  quoteCounter++;
639  if (optionalQuotes) {
640  qtStrikes.push_back(cfq->strike());
641  qtTenors.push_back(cfq->term());
642  qtData.push_back(cfq->quote()->value());
643  }
644  auto key = make_pair(cfq->term(), cfq->strike());
645  auto r = volQuotes.insert(make_pair(key, cfq->quote()->value()));
646  QL_REQUIRE(r.second, "Duplicate cap floor quote in config " << config.curveID() << ", with underlying tenor " << tenor <<
647  " and currency " << currency << ", for tenor " << key.first << " and strike " << key.second);
648  }
649  }
650  }
651  }
652 
653  Size totalQuotes = config.tenors().size() * config.strikes().size();
654  if (quoteCounter < totalQuotes) {
655  WLOG("Found only " << quoteCounter << " out of " << totalQuotes << " quotes for CapFloor surface " << config.curveID());
656  }
657 
658  vector<Period> tenors = parseVectorOfValues<Period>(config.tenors(), &parsePeriod);
659  vector<Rate> strikes = parseVectorOfValues<Real>(config.strikes(), &parseReal);
660  Matrix vols(tenors.size(), strikes.size());
661  for (Size i = 0; i < tenors.size(); i++) {
662  for (Size j = 0; j < strikes.size(); j++) {
663  auto key = make_pair(tenors[i], strikes[j]);
664  auto it = volQuotes.find(key);
665  if (!optionalQuotes) {
666  QL_REQUIRE(it != volQuotes.end(), "Quote with tenor " << key.first << " and strike " << key.second
667  << " not loaded for cap floor config "
668  << config.curveID());
669  // Organise the values in to a square matrix
670  vols[i][j] = it->second;
671  } else {
672  if (it == volQuotes.end()) {
673  DLOG("Could not find quote with tenor " << key.first << " and strike " << key.second <<
674  " for cap floor config " << config.curveID());
675  }
676  }
677  }
678  }
679 
680  DLOG("Found " << quoteCounter << " quotes for capfloor surface " << config.curveID());
681  if (optionalQuotes) {
682  QL_REQUIRE(quoteCounter > 0, "No Quotes provided for CapFloor surface " << config.curveID());
683  if (config.interpolationMethod() == CapFloorTermVolSurfaceExact::Bilinear) {
684  return boost::make_shared<QuantExt::CapFloorTermVolSurfaceSparse<Linear, Linear>>(
685  config.settleDays(), config.calendar(), config.businessDayConvention(), config.dayCounter(), qtTenors,
686  qtStrikes, qtData, true, true);
687  } else if (config.interpolationMethod() == CapFloorTermVolSurfaceExact::BicubicSpline) {
688  return boost::make_shared<QuantExt::CapFloorTermVolSurfaceSparse<Cubic, Cubic>>(
689  config.settleDays(), config.calendar(), config.businessDayConvention(), config.dayCounter(), qtTenors,
690  qtStrikes, qtData, true, true);
691  } else {
692  QL_FAIL("Invalid Interpolation method for capfloor surface " << config.curveID() << ", must be either "
693  << CapFloorTermVolSurfaceExact::Bilinear << " or " << CapFloorTermVolSurfaceExact::BicubicSpline << ".");
694  }
695  } else {
696  // Return for the cap floor term volatility surface
697  return boost::make_shared<QuantExt::CapFloorTermVolSurfaceExact>(config.settleDays(), config.calendar(),
698  config.businessDayConvention(), tenors, strikes, vols,
699  config.dayCounter(), config.interpolationMethod());
700  }

708 
709  bool optionalQuotes = config.optionalQuotes();

770  config.businessDayConvention(), quoteTenors,
771  vols, config.dayCounter());

776  config.settleDays(), config.calendar(), config.businessDayConvention(), quoteTenors, vols,
777  config.dayCounter());

780  config.businessDayConvention(), quoteTenors,
781  vols, config.dayCounter());
782  }
783  } else {

785  << static_cast<int>(config.interpolationMethod()));
