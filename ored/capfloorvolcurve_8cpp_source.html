58// Currently, only two possibilities for variable InterpolateOn: TermVolatilities and OptionletVolatilities
59// Here, we convert the value to a bool for use in building the structures. May need to broaden if we add more.
60bool interpOnOpt(CapFloorVolatilityCurveConfig& config) {
61 QL_REQUIRE(config.interpolateOn() == "TermVolatilities" || config.interpolateOn() == "OptionletVolatilities",
62 "Expected InterpolateOn to be one of TermVolatilities or OptionletVolatilities");
63 return config.interpolateOn() == "OptionletVolatilities";
64}

66CapFloorVolCurve::CapFloorVolCurve(
67 const Date& asof, const CapFloorVolatilityCurveSpec& spec, const Loader& loader,
68 const CurveConfigurations& curveConfigs, boost::shared_ptr<IborIndex> iborIndex,
69 Handle<YieldTermStructure> discountCurve, const boost::shared_ptr<IborIndex> sourceIndex,
70 const boost::shared_ptr<IborIndex> targetIndex,
71 const std::map<std::string,
72 std::pair<boost::shared_ptr<ore::data::CapFloorVolCurve>, std::pair<std::string, QuantLib::Period>>>&
73 requiredCapFloorVolCurves,
74 const bool buildCalibrationInfo)
75 : spec_(spec) {
76
77 try {
78 // The configuration
79 const boost::shared_ptr<CapFloorVolatilityCurveConfig>& config =
80 curveConfigs.capFloorVolCurveConfig(spec_.curveConfigID());
81
82 if (!config->proxySourceCurveId().empty()) {
83 // handle proxy vol surfaces
84 buildProxyCurve(*config, sourceIndex, targetIndex, requiredCapFloorVolCurves);
85 } else {
86 // Read the shift early if the configured volatility type is shifted lognormal
87 Real shift = 0.0;
88 if (config->volatilityType() == CfgVolType::ShiftedLognormal) {
89 shift = shiftQuote(asof, *config, loader);
90 }
91
92 // There are three possible cap floor configurations
93 if (config->type() == CfgType::Atm) {
94 atmOptCurve(asof, *config, loader, iborIndex, discountCurve, shift);
95 } else if (config->type() == CfgType::Surface || config->type() == CfgType::SurfaceWithAtm) {
96 optSurface(asof, *config, loader, iborIndex, discountCurve, shift);
97 } else {
98 QL_FAIL("Unexpected type (" << static_cast<int>(config->type()) << ") for cap floor config "
99 << config->curveID());
100 }
101 // Turn on or off extrapolation
102 capletVol_->enableExtrapolation(config->extrapolate());
103 }
104
105 // Build calibration info
106 if (buildCalibrationInfo) {
107 this->buildCalibrationInfo(asof, curveConfigs, config, iborIndex);
108 }
109
110 } catch (exception& e) {
111 QL_FAIL("cap/floor vol curve building failed :" << e.what());
112 } catch (...) {
113 QL_FAIL("cap/floor vol curve building failed: unknown error");
114 }
115
116 // force bootstrap so that errors are thrown during the build, not later
117 capletVol_->volatility(QL_EPSILON, capletVol_->minStrike());
118}

120void CapFloorVolCurve::buildProxyCurve(
121 const CapFloorVolatilityCurveConfig& config, const boost::shared_ptr<IborIndex>& sourceIndex,
122 const boost::shared_ptr<IborIndex>& targetIndex,
123 const std::map<std::string,
124 std::pair<boost::shared_ptr<ore::data::CapFloorVolCurve>, std::pair<std::string, QuantLib::Period>>>&
125 requiredCapFloorVolCurves) {
126
127 auto sourceVol = requiredCapFloorVolCurves.find(config.proxySourceCurveId());
128 QL_REQUIRE(sourceVol != requiredCapFloorVolCurves.end(),
129 "CapFloorVolCurve::buildProxyCurve(): required source cap vol curve '" << config.proxySourceCurveId()
130 << "' not found.");
131
132 capletVol_ = boost::make_shared<ProxyOptionletVolatility>(
133 Handle<OptionletVolatilityStructure>(sourceVol->second.first->capletVolStructure()), sourceIndex, targetIndex,
134 config.proxySourceRateComputationPeriod(), config.proxyTargetRateComputationPeriod());
135}

294 // Get the cap floor term vol surface

598boost::shared_ptr<QuantExt::CapFloorTermVolSurface>
599CapFloorVolCurve::capSurface(const Date& asof, CapFloorVolatilityCurveConfig& config, const Loader& loader) const {
600
601 // Map to store the quote values that we load with key (period, strike) where strike
602 // needs a custom comparator to avoid == with double
603 auto comp = [](const pair<Period, Rate>& a, const pair<Period, Rate>& b) {
604 return (a.first < b.first) || (!(b.first < a.first) && (a.second < b.second && !close(a.second, b.second)));
605 };
606 map<pair<Period, Rate>, Real, decltype(comp)> volQuotes(comp);
607
608 bool optionalQuotes = config.optionalQuotes();
609 Size quoteCounter = 0;
610 bool quoteRelevant = false;
611 bool tenorRelevant = false;
612 bool strikeRelevant = false;
613
614 vector<Real> qtStrikes;
615 vector<Real> qtData;
616 vector<Period> qtTenors;
617 Period tenor = parsePeriod(config.indexTenor()); // 1D, 1M, 3M, 6M, 12M
618 string currency = config.currency();
619 vector<Period> configTenors = parseVectorOfValues<Period>(config.tenors(), &parsePeriod);
620
621 std::ostringstream ss;
622 ss << MarketDatum::InstrumentType::CAPFLOOR << "/" << config.quoteType() << "/" << currency << "/";
623 if (config.quoteIncludesIndexName())
624 ss << config.index() << "/";
625 ss << "*";
626 Wildcard w(ss.str());
627 for (const auto& md : loader.get(w, asof)) {
628 QL_REQUIRE(md->asofDate() == asof, "MarketDatum asofDate '" << md->asofDate() << "' <> asof '" << asof << "'");
629 boost::shared_ptr<CapFloorQuote> cfq = boost::dynamic_pointer_cast<CapFloorQuote>(md);
630 QL_REQUIRE(cfq, "Internal error: could not downcast MarketDatum '" << md->name() << "' to CapFloorQuote");
631 QL_REQUIRE(cfq->ccy() == currency, "CapFloorQuote ccy '" << cfq->ccy() << "' <> config ccy '" << currency << "'");
632 if (cfq->underlying() == tenor && !cfq->atm()) {
633 auto j = std::find(configTenors.begin(), configTenors.end(), cfq->term());
634 tenorRelevant = j != configTenors.end();
635
636 Real strike = cfq->strike();
637 auto i = std::find_if(config.strikes().begin(), config.strikes().end(),
638 [&strike](const std::string& x) {
639 return close_enough(parseReal(x), strike);
640 });
641 strikeRelevant = i != config.strikes().end();
642
643 quoteRelevant = strikeRelevant && tenorRelevant;
644
645 if (quoteRelevant) {
646 // if we have optional quotes we just make vectors of all quotes and let the sparse surface
647 // handle them
648 quoteCounter++;
649 if (optionalQuotes) {
650 qtStrikes.push_back(cfq->strike());
651 qtTenors.push_back(cfq->term());
652 qtData.push_back(cfq->quote()->value());
653 }
654 auto key = make_pair(cfq->term(), cfq->strike());
655 auto r = volQuotes.insert(make_pair(key, cfq->quote()->value()));
656 if (config.quoteIncludesIndexName())
657 QL_REQUIRE(r.second, "Duplicate cap floor quote in config " << config.curveID() << ", with underlying tenor " << tenor <<
658 " ,currency " << currency << " and index " << config.index() << ", for tenor " << key.first << " and strike " << key.second);
659 else
660 QL_REQUIRE(r.second, "Duplicate cap floor quote in config " << config.curveID() << ", with underlying tenor " << tenor <<
661 " and currency " << currency << ", for tenor " << key.first << " and strike " << key.second);
662 }
663 }
664 }
665
666 Size totalQuotes = config.tenors().size() * config.strikes().size();
667 if (quoteCounter < totalQuotes) {
668 WLOG("Found only " << quoteCounter << " out of " << totalQuotes << " quotes for CapFloor surface " << config.curveID());
669 }
670
671 vector<Period> tenors = parseVectorOfValues<Period>(config.tenors(), &parsePeriod);
672 vector<Rate> strikes = parseVectorOfValues<Real>(config.strikes(), &parseReal);
673 Matrix vols(tenors.size(), strikes.size());
674 for (Size i = 0; i < tenors.size(); i++) {
675 for (Size j = 0; j < strikes.size(); j++) {
676 auto key = make_pair(tenors[i], strikes[j]);
677 auto it = volQuotes.find(key);
678 if (!optionalQuotes) {
679 QL_REQUIRE(it != volQuotes.end(), "Quote with tenor " << key.first << " and strike " << key.second
680 << " not loaded for cap floor config "
681 << config.curveID());
682 // Organise the values in to a square matrix
683 vols[i][j] = it->second;
684 } else {
685 if (it == volQuotes.end()) {
686 DLOG("Could not find quote with tenor " << key.first << " and strike " << key.second <<
687 " for cap floor config " << config.curveID());
688 }
689 }
690 }
691 }
692
693 DLOG("Found " << quoteCounter << " quotes for capfloor surface " << config.curveID());
694 if (optionalQuotes) {
695 QL_REQUIRE(quoteCounter > 0, "No Quotes provided for CapFloor surface " << config.curveID());
696 if (config.interpolationMethod() == CapFloorTermVolSurfaceExact::Bilinear) {
697 return boost::make_shared<QuantExt::CapFloorTermVolSurfaceSparse<Linear, Linear>>(
698 config.settleDays(), config.calendar(), config.businessDayConvention(), config.dayCounter(), qtTenors,
699 qtStrikes, qtData, true, true);
700 } else if (config.interpolationMethod() == CapFloorTermVolSurfaceExact::BicubicSpline) {
701 return boost::make_shared<QuantExt::CapFloorTermVolSurfaceSparse<Cubic, Cubic>>(
702 config.settleDays(), config.calendar(), config.businessDayConvention(), config.dayCounter(), qtTenors,
703 qtStrikes, qtData, true, true);
704 } else {
705 QL_FAIL("Invalid Interpolation method for capfloor surface " << config.curveID() << ", must be either "
706 << CapFloorTermVolSurfaceExact::Bilinear << " or " << CapFloorTermVolSurfaceExact::BicubicSpline << ".");
707 }
708 } else {
709 // Return for the cap floor term volatility surface
710 return boost::make_shared<QuantExt::CapFloorTermVolSurfaceExact>(config.settleDays(), config.calendar(),
711 config.businessDayConvention(), tenors, strikes, vols,
712 config.dayCounter(), config.interpolationMethod());
713 }

717CapFloorVolCurve::atmCurve(const Date& asof, CapFloorVolatilityCurveConfig& config, const Loader& loader) const {
718
719 // Map to store the quote values
720 map<Period, Handle<Quote>> volQuotes;
721
722 bool optionalQuotes = config.optionalQuotes();
723 Period tenor = parsePeriod(config.indexTenor()); // 1D, 1M, 3M, 6M, 12M
724 string currency = config.currency();
725 // Load the relevant quotes
726 std::ostringstream ss;
727 ss << MarketDatum::InstrumentType::CAPFLOOR << "/" << config.quoteType() << "/" << currency << "/*";
728 Wildcard w(ss.str());
729 for (const auto& md : loader.get(w, asof)) {
730 QL_REQUIRE(md->asofDate() == asof, "MarketDatum asofDate '" << md->asofDate() << "' <> asof '" << asof << "'");
731 boost::shared_ptr<CapFloorQuote> cfq = boost::dynamic_pointer_cast<CapFloorQuote>(md);
732 QL_REQUIRE(cfq, "Internal error: could not downcast MarketDatum '" << md->name() << "' to CapFloorQuote");
733 QL_REQUIRE(cfq->ccy() == currency, "CapFloorQuote ccy '" << cfq->ccy() << "' <> config ccy '" << currency << "'");
734 if (cfq->underlying() == tenor && cfq->atm()) {
735 auto j = std::find(config.atmTenors().begin(), config.atmTenors().end(), to_string(cfq->term()));
736 if (j != config.atmTenors().end()) {
737 auto r = volQuotes.insert(make_pair(cfq->term(), cfq->quote()));
738 QL_REQUIRE(r.second, "Duplicate ATM cap floor quote in config " << config.curveID() << " for tenor "
739 << cfq->term());
740 }
741 }
742 }
743
744 // Check that the loaded quotes cover all of the configured ATM tenors
745 vector<Period> tenors = parseVectorOfValues<Period>(config.atmTenors(), &parsePeriod);
746 vector<Handle<Quote>> vols;
747 vector<Period> quoteTenors;
748 if (!optionalQuotes) {
749 vols.resize(tenors.size());
750 quoteTenors = tenors;
751 }
752 for (Size i = 0; i < tenors.size(); i++) {
753 auto it = volQuotes.find(tenors[i]);
754 if (!optionalQuotes) {
755 QL_REQUIRE(it != volQuotes.end(),
756 "ATM cap floor quote in config " << config.curveID() << " for tenor " << tenors[i] << " not found ");
757 vols[i] = it->second;
758 } else {
759 if (it == volQuotes.end()) {
760 DLOG("Could not find ATM cap floor quote with tenor " << tenors[i] << " for cap floor config " << config.curveID());
761 } else {
762 vols.push_back(it->second);
763 quoteTenors.push_back(it->first);
764 }
765 }
766 }
767
768 if (optionalQuotes) {
769 QL_REQUIRE(vols.size() > 0, "No ATM cap floor quotes found for cap floor config " << config.curveID());
770 if (vols.size() == 1)
771 WLOG("Only one ATM cap floor quote found for cap floor config " << config.curveID() << ", using constant volatility");
772 }
773
774 // Return for the cap floor ATM term volatility curve
775 // The interpolation here is also based on the interpolation method parameter in the configuration
776 // Flat first period is true by default (see ctor)
777 if (config.interpolationMethod() == CftvsInterp::BicubicSpline) {
778 if (config.flatExtrapolation()) {
779 return boost::make_shared<InterpolatedCapFloorTermVolCurve<CubicFlat>>(
780 config.settleDays(), config.calendar(), config.businessDayConvention(), quoteTenors, vols,
781 config.dayCounter());
782 } else {
783 return boost::make_shared<InterpolatedCapFloorTermVolCurve<Cubic>>(config.settleDays(), config.calendar(),
784 config.businessDayConvention(), quoteTenors,
785 vols, config.dayCounter());
786 }
787 } else if (config.interpolationMethod() == CftvsInterp::Bilinear) {
788 if (config.flatExtrapolation()) {
789 return boost::make_shared<InterpolatedCapFloorTermVolCurve<LinearFlat>>(
790 config.settleDays(), config.calendar(), config.businessDayConvention(), quoteTenors, vols,
791 config.dayCounter());
792 } else {
793 return boost::make_shared<InterpolatedCapFloorTermVolCurve<Linear>>(config.settleDays(), config.calendar(),
794 config.businessDayConvention(), quoteTenors,
795 vols, config.dayCounter());
796 }
797 } else {
798 QL_FAIL("Cap floor config " << config.curveID() << " has unexpected interpolation method "
799 << static_cast<int>(config.interpolationMethod()));
800 }

823// This method is used to pull out the stripped optionlets from the QuantExt::OptionletStripper instance that is
824// bootstrapped. The reason is that we do not want all of the cap floor helpers and their coupons in scope during
825// a potential XVA run as this leads to delays when fixings are updated.
826boost::shared_ptr<StrippedOptionlet> CapFloorVolCurve::transform(const QuantExt::OptionletStripper& os) const {

830 for (Size j = 0; j < os.optionletVolatilities(i).size(); j++) {
831 vols[i].push_back(Handle<Quote>(boost::make_shared<SimpleQuote>(os.optionletVolatilities(i)[j])));
832 }
833 }
834
835 vector<vector<Real>> optionletStrikes;
836 for (Size i = 0; i < os.optionletFixingDates().size(); i++) {
837 optionletStrikes.push_back(os.optionletStrikes(i));
838 }
839
840 boost::shared_ptr<StrippedOptionlet> res = boost::make_shared<StrippedOptionlet>(
841 os.settlementDays(), os.calendar(), os.businessDayConvention(), os.index(), os.optionletFixingDates(),
842 optionletStrikes, vols, os.dayCounter(), os.volatilityType(), os.displacement());
843
844 res->unregisterWithAll();
845
846 return res;

875void CapFloorVolCurve::buildCalibrationInfo(const Date& asof, const CurveConfigurations& curveConfigs,
876 const boost::shared_ptr<CapFloorVolatilityCurveConfig> config,
877 const boost::shared_ptr<IborIndex>& index) {
878 DLOG("Building calibration info for cap floor vols");
879
880 ReportConfig rc = effectiveReportConfig(curveConfigs.reportConfigIrCapFloorVols(), config->reportConfig());
881 bool reportOnStrikeGrid = *rc.reportOnStrikeGrid();
882 bool reportOnStrikeSpreadGrid = *rc.reportOnStrikeSpreadGrid();

884 std::vector<Real> strikeSpreads = *rc.strikeSpreads();
885 std::vector<Period> expiries = *rc.expiries();
886 std::vector<Period> underlyingTenorsReport(1, index->tenor());
887
888 calibrationInfo_ = boost::make_shared<IrVolCalibrationInfo>();
889
890 calibrationInfo_->dayCounter = config->dayCounter().empty() ? "na" : config->dayCounter().name();
891 calibrationInfo_->calendar = config->calendar().empty() ? "na" : config->calendar().name();
892 calibrationInfo_->volatilityType = ore::data::to_string(capletVol_->volatilityType());
893 calibrationInfo_->underlyingTenors = underlyingTenorsReport;
894
895 bool isOis = boost::dynamic_pointer_cast<OvernightIndex>(index) != nullptr;
896
897 Size onSettlementDays = 0;
898 if (isOis) {
899 onSettlementDays = config->onCapSettlementDays();
900 }
901
902 std::vector<Real> times; // fixing times of caplets
903 std::vector<std::vector<Real>> forwards; // fair rates of caplets
904 for (auto const& p : expiries) {
905 Date fixingDate;
906 Real forward;
907 if (isOis) {
908 Leg dummyCap = MakeOISCapFloor(CapFloor::Cap, p, boost::dynamic_pointer_cast<OvernightIndex>(index),
909 config->rateComputationPeriod(), 0.04)
910 .withTelescopicValueDates(true)
911 .withSettlementDays(onSettlementDays);
912 if (dummyCap.empty())
913 continue;
914 auto lastCoupon = boost::dynamic_pointer_cast<CappedFlooredOvernightIndexedCoupon>(dummyCap.back());
915 QL_REQUIRE(lastCoupon, "OptionletStripper::populateDates(): expected CappedFlooredOvernightIndexedCoupon");
916 fixingDate = std::max(asof + 1, lastCoupon->underlying()->fixingDates().front());
917 forward = lastCoupon->underlying()->rate();
918 } else {
919 CapFloor dummyCap = MakeCapFloor(CapFloor::Cap, p, index, 0.04, 0 * Days);
920 if (dummyCap.floatingLeg().empty())
921 continue;
922 boost::shared_ptr<FloatingRateCoupon> lastCoupon = dummyCap.lastFloatingRateCoupon();
923 fixingDate = lastCoupon->fixingDate();
