31CSVFileReader::CSVFileReader(const std::string& fileName, const bool firstLineContainsHeaders,
32 const std::string& delimiters, const std::string& escapeCharacters,
33 const std::string& quoteCharacters, const char eolMarker)
34 : fileName_(fileName), hasHeaders_(firstLineContainsHeaders), eolMarker_(eolMarker), currentLine_(Null<Size>()),
35 numberOfColumns_(Null<Size>()),
36 tokenizer_(std::string(), boost::escaped_list_separator<char>(escapeCharacters, delimiters, quoteCharacters)) {
37 file_.open(fileName);
38 QL_REQUIRE(file_.is_open(), "CSVFileReader: error opening file " << fileName);
39 if (firstLineContainsHeaders) {
40 QL_REQUIRE(!file_.eof(), "CSVFileReader: file is empty: " << fileName);
41 std::string line;
42 getline(file_, line, eolMarker);
43 boost::trim(line);
44 tokenizer_.assign(line);
45 std::copy(tokenizer_.begin(), tokenizer_.end(), std::back_inserter(headers_));
46 numberOfColumns_ = headers_.size();
47 }
48}

64bool CSVFileReader::next() {
65 QL_REQUIRE(file_.is_open(), "CSVFileReader: file is not open, can not move to next line");
66 std::string line = "";
67 // skip empty lines
68 while (line.size() == 0 && !file_.eof()) {
69 getline(file_, line, eolMarker_);
70 boost::trim(line);
71 }
72 if (file_.eof() && line.empty()) {
73 close();
74 return false;
75 }
76 if (currentLine_ == Null<Size>())
77 currentLine_ = 0;
78 else
79 ++currentLine_;
80 tokenizer_.assign(line);
81 data_.clear();
82 std::copy(tokenizer_.begin(), tokenizer_.end(), std::back_inserter(data_));
83 if (numberOfColumns_ == Null<Size>())
84 numberOfColumns_ = data_.size();
85 else
86 QL_REQUIRE(data_.size() == numberOfColumns_, "CSVFileReader: data line #"
87 << currentLine_ << " has " << data_.size()
88 << " fields, expected " << numberOfColumns_);
89 return true;
90}

97std::string CSVFileReader::get(const std::string& field) const {
98 QL_REQUIRE(hasHeaders_, "CSVFileReader: can not get data by field, file does not have headers");
99 QL_REQUIRE(currentLine_ != Null<Size>(), "CSVFileReader: can not get data, need call to next() first");
100 Size index = std::find(headers_.begin(), headers_.end(), field) - headers_.begin();
101 QL_REQUIRE(index < headers_.size(), "CSVFileReader: field \"" << field << "\" not found.");
102 QL_REQUIRE(index < data_.size(), "CSVFileReader: unexpected data size (" << data_.size() << "), required at least "
103 << index + 1 << ", while reading field \""
104 << field << "\"");
105 return data_[index];
106}
