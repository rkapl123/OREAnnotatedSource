29void CrossCurrencySwap::checkCrossCurrencySwap() {
30 // A Cross Currency Swap must have 2 legs (either Fixed or Floating)
31 QL_REQUIRE(legData_.size() >= 2, "A Cross Currency Swap must have at least 2 legs - Trade: " + id());
32 for (Size i = 0; i < legData_.size(); i++)
33 QL_REQUIRE(legData_[i].legType() == "Fixed" || legData_[i].legType() == "Floating",
34 "CrossCurrencySwap leg #" << i << " must be either Fixed or Floating");
35
36 // Check leg currencies
37 Currency legCcy0 = parseCurrencyWithMinors(legData_[0].currency());
38 Currency legIndexCcy0;
39 if (legData_[0].legType() == "Fixed") {
40 legIndexCcy0 = legCcy0;
41 } else if (legData_[0].legType() == "Floating") {
42 auto floatingLeg = boost::dynamic_pointer_cast<FloatingLegData>(legData_[0].concreteLegData());
43 legIndexCcy0 = parseIborIndex(floatingLeg->index())->currency();
44 }
45
46 Currency legCcy, legIndexCcy;
47 for (Size i = 1; i < legData_.size(); i++) {
48 legCcy = parseCurrencyWithMinors(legData_[i].currency());
49 // Require leg currencies to be different. If they are the same, we do a further check of FloatingLeg
50 // underlying currencies, in which case we still call this a cross currency swap.
51 if (legCcy0 == legCcy) {
52 if (legData_[i].legType() == "Fixed") {
53 legIndexCcy = legCcy;
54 } else if (legData_[i].legType() == "Floating") {
55 auto floatingLeg = boost::dynamic_pointer_cast<FloatingLegData>(legData_[i].concreteLegData());
56 legIndexCcy = parseIborIndex(floatingLeg->index()) ->currency();
57 }
58 QL_REQUIRE(legIndexCcy0 != legIndexCcy, "Cross currency swap legs must have different currencies.");
59 }
60 }
61}
