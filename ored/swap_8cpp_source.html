44 void Swap::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
45  DLOG("Swap::build() called for trade " << id());
46 
47  QL_REQUIRE(legData_.size() >= 1, "Swap must have at least 1 leg");
48  const boost::shared_ptr<Market> market = engineFactory->market();
49 
50  // allow minor currencies in case first leg is equity
51  Currency currency = parseCurrencyWithMinors(legData_[0].currency());
52  string ccy_str = currency.code();
53 
54  Size numLegs = legData_.size();
55  legPayers_ = vector<bool>(numLegs);
56  std::vector<QuantLib::Currency> currencies(numLegs);
57  legs_.resize(numLegs);
58 
59  isXCCY_ = false;
60  isResetting_ = false;
61 
62  for (Size i = 0; i < numLegs; ++i) {
63  // allow minor currencies for Equity legs as some exchanges trade in these, e.g LSE in pence - GBX or GBp
64  // minor currencies on other legs will fail here
65  if (legData_[i].legType() == "Equity")
66  currencies[i] = parseCurrencyWithMinors(legData_[i].currency());
67  else
68  currencies[i] = parseCurrency(legData_[i].currency());
69  if (currencies[i] != currency)
70  isXCCY_ = true;
71  isResetting_ = isResetting_ || (!legData_[i].isNotResetXCCY());
72  }
73 
74  boost::shared_ptr<EngineBuilder> builder =
75  isXCCY_ ? engineFactory->builder("CrossCurrencySwap") : engineFactory->builder("Swap");
76  auto configuration = builder->configuration(MarketContext::pricing);
77 
78  for (Size i = 0; i < numLegs; ++i) {
79  legPayers_[i] = legData_[i].isPayer();
80  auto legBuilder = engineFactory->legBuilder(legData_[i].legType());
81  legs_[i] = legBuilder->buildLeg(legData_[i], engineFactory, requiredFixings_, configuration);
82  DLOG("Swap::build(): currency[" << i << "] = " << currencies[i]);
83 
84  // add notional leg, if applicable
85  auto leg = buildNotionalLeg(legData_[i], legs_[i], requiredFixings_, engineFactory->market(), configuration);
86  if (!leg.empty()) {
87  legs_.push_back(leg);
88  legPayers_.push_back(legPayers_[i]);
89  currencies.push_back(currencies[i]);
90  }
91  } // for legs
92 
93  // The npv currency, notional currency and current notional are taken from the first leg that
94  // appears in the XML that has a notional. If no such leg exists the notional currency
95  // and current notional are left empty and the npv currency is set to the first leg's currency
96 
97  notionalTakenFromLeg_ = 0;
98  for (; notionalTakenFromLeg_ < legData_.size(); ++notionalTakenFromLeg_) {
99  const LegData& d = legData_[notionalTakenFromLeg_];
100  if (!d.notionals().empty())
101  break;
102  }
103 
104  if (notionalTakenFromLeg_ == legData_.size()) {
105  ALOG("no suitable leg found to set notional, set to null and notionalCurrency to empty string");
106  notional_ = Null<Real>();
107  notionalCurrency_ = "";
108  // parse for currency in case first leg is Equity, we only want the major currency for NPV
109  npvCurrency_ = parseCurrencyWithMinors(legData_.front().currency()).code();
110  } else {
111  if (legData_[notionalTakenFromLeg_].schedule().hasData()) {
112  Schedule schedule = makeSchedule(legData_[notionalTakenFromLeg_].schedule());
113  auto notional =
114  buildScheduledVectorNormalised(legData_[notionalTakenFromLeg_].notionals(),
115  legData_[notionalTakenFromLeg_].notionalDates(), schedule, 0.0);
116  Date today = Settings::instance().evaluationDate();
117  auto d = std::upper_bound(schedule.dates().begin(), schedule.dates().end(), today);
118  // forward starting => take first notional
119  // on or after last schedule date => zero notional
120  // in between => notional of current period
121  if (d == schedule.dates().begin())
122  notional_ = notional.at(0);
123  else if (d == schedule.dates().end())
124  notional_ = 0.0;
125  else
126  notional_ = notional.at(std::distance(schedule.dates().begin(), d) - 1);
127  } else {
128  notional_ = legData_[notionalTakenFromLeg_].notionals().at(0);
129  }
130  // parse for currency in case leg is Equity, we only want the major currency for NPV and Notional
131  notionalCurrency_ = parseCurrencyWithMinors(legData_[notionalTakenFromLeg_].currency()).code();
132  npvCurrency_ = parseCurrencyWithMinors(legData_[notionalTakenFromLeg_].currency()).code();
133  DLOG("Notional is " << notional_ << " " << notionalCurrency_);
134  }
135 
136  Currency npvCcy = parseCurrency(npvCurrency_);
137  DLOG("npv currency is " << npvCurrency_);
138 
139  if (isXCCY_) {
140  boost::shared_ptr<QuantExt::CurrencySwap> swap(
141  new QuantExt::CurrencySwap(legs_, legPayers_, currencies, settlement_ == "Physical", isResetting_));
142  boost::shared_ptr<CrossCurrencySwapEngineBuilderBase> swapBuilder =
143  boost::dynamic_pointer_cast<CrossCurrencySwapEngineBuilderBase>(builder);
144  QL_REQUIRE(swapBuilder, "No Builder found for CrossCurrencySwap " << id());
145  swap->setPricingEngine(swapBuilder->engine(currencies, npvCcy));
146  // take the first legs currency as the npv currency (arbitrary choice)
147  instrument_.reset(new VanillaInstrument(swap));
148  } else {
149  boost::shared_ptr<QuantLib::Swap> swap(new QuantLib::Swap(legs_, legPayers_));
150  boost::shared_ptr<SwapEngineBuilderBase> swapBuilder =
151  boost::dynamic_pointer_cast<SwapEngineBuilderBase>(builder);
152  QL_REQUIRE(swapBuilder, "No Builder found for Swap " << id());
153  swap->setPricingEngine(swapBuilder->engine(npvCcy));
154  instrument_.reset(new VanillaInstrument(swap));
155  }
156 
157  DLOG("Set instrument wrapper");
158 
159  // set Leg Currencies
160  legCurrencies_ = vector<string>(currencies.size());
161  for (Size i = 0; i < currencies.size(); i++)
162  legCurrencies_[i] = currencies[i].code();
163 
164  // set maturity
165  maturity_ = Date::minDate();
166  Date startDate = Date::maxDate();
167  for (auto const& l : legs_) {
168  if (!l.empty()) {
169  maturity_ = std::max(maturity_, l.back()->date());
170  startDate = std::min(startDate, l.front()->date());
171  boost::shared_ptr<Coupon> coupon = boost::dynamic_pointer_cast<Coupon>(l.front());
172  if (coupon)
173  startDate = std::min(startDate, coupon->accrualStartDate());
174  }
175  }
176 
177  additionalData_["startDate"] = to_string(startDate);
178 }

180 const std::map<std::string,boost::any>& Swap::additionalData() const {
181  Size numLegs = legData_.size();
182  // use the build time as of date to determine current notionals
183  Date asof = Settings::instance().evaluationDate();
184  boost::shared_ptr<QuantLib::Swap> swap = boost::dynamic_pointer_cast<QuantLib::Swap>(instrument_->qlInstrument());
185  boost::shared_ptr<QuantExt::CurrencySwap> cswap = boost::dynamic_pointer_cast<QuantExt::CurrencySwap>(instrument_->qlInstrument());
186  std::map<std::string, Real> legNpv; // by currency
187  for (Size i = 0; i < numLegs; ++i) {
188  string legID = to_string(i+1);
189  additionalData_["legType[" + legID + "]"] = legData_[i].legType();
190  additionalData_["isPayer[" + legID + "]"] = legData_[i].isPayer();
191  additionalData_["notionalCurrency[" + legID + "]"] = legData_[i].currency();
192  if (!isXCCY_) {
193  if (swap)
194  additionalData_["legNPV[" + legID + "]"] = swap->legNPV(i);
195  else
196  ALOG("single currency swap underlying instrument not set, skip leg npv reporting");
197  }
198  else {
199  if (cswap) {
200  // The currency swap has more legs than the swap wrapper (additional notional legs), so aggregate by currency
201  Real legNpv = 0;
202  Real legNpvInCcy = 0;
203  for (Size j = 0; j < cswap->legs().size(); ++j) {
204  if (cswap->legCurrency(j).code() == legData_[i].currency()) {
205  legNpv += cswap->legNPV(j);
206  legNpvInCcy += cswap->inCcyLegNPV(j);
207  }
208  }
209  additionalData_["legNPV[" + legID + "]"] = legNpv;
210  additionalData_["legNPVCCY[" + legID + "]"] = legNpvInCcy;
211  }
212  else
213  ALOG("cross currency swap underlying instrument not set, skip leg npv reporting");
214  }
215  for (Size j = 0; j < legs_[i].size(); ++j) {
216  boost::shared_ptr<CashFlow> flow = legs_[i][j];
217  // pick flow with earliest future payment date on this leg
218  if (flow->date() > asof) {
219  Real flowAmount = 0.0;
220  try { flowAmount = flow->amount(); }
221  catch(std::exception& e) {
222  ALOG("flow amount could not be determined for trade " << id() << ", set to zero: " << e.what());
223  }
224  additionalData_["amount[" + legID + "]"] = flowAmount;
225  additionalData_["paymentDate[" + legID + "]"] = to_string(flow->date());
226  boost::shared_ptr<Coupon> coupon = boost::dynamic_pointer_cast<Coupon>(flow);
227  if (coupon) {
228  Real currentNotional = 0;
229  try { currentNotional = coupon->nominal(); }
230  catch(std::exception& e) {
231  ALOG("current notional could not be determined for trade " << id() << ", set to zero: " << e.what());
232  }
233  additionalData_["currentNotional[" + legID + "]"] = currentNotional;
234 
235  Real rate = 0;
236  try { rate = coupon->rate(); }
237  catch(std::exception& e) {
238  ALOG("coupon rate could not be determined for trade " << id() << ", set to zero: " << e.what());
239  }
240  additionalData_["rate[" + legID + "]"] = rate;
241 
242  boost::shared_ptr<FloatingRateCoupon> frc = boost::dynamic_pointer_cast<FloatingRateCoupon>(flow);
243  if (frc) {
244  additionalData_["index[" + legID + "]"] = frc->index()->name();
245  additionalData_["spread[" + legID + "]"] = frc->spread();
246  }
247  }
248  break;
249  }
250  }
251  if (legs_[i].size() > 0) {
252  boost::shared_ptr<Coupon> coupon = boost::dynamic_pointer_cast<Coupon>(legs_[i][0]);
253  if (coupon) {
254  Real originalNotional = 0.0;
255  try { originalNotional = coupon->nominal(); }
256  catch(std::exception& e) {
257  ALOG("original nominal could not be determined for trade " << id() << ", set to zero: " << e.what());
258  }
259  additionalData_["originalNotional[" + legID + "]"] = originalNotional;
260  }
261  }
262  }
263  return additionalData_;
264 }

266 QuantLib::Real Swap::notional() const {
267  // try to get the notional from the additional results of the instrument
268  try {
269  return instrument_->qlInstrument(true)->result<Real>("currentNotional");
270  } catch (const std::exception& e) {
271  WLOG("swap engine does not provide current notional: " << e.what() << ", using fallback");
272  // Try getting current notional from coupons
273  if (notionalTakenFromLeg_ < legs_.size()) {
274  Real n = currentNotional(legs_[notionalTakenFromLeg_]);
275  if (fabs(n) > QL_EPSILON) {
276  return n;
277  }
278  }
279  // else return the face value
280  WLOG("swap does not provide coupon notionals, using face value");
281  return notional_;
282  }

297 Swap::underlyingIndices(const boost::shared_ptr<ReferenceDataManager>& referenceDataManager) const {
298 
299  map<AssetClass, set<string>> result;
300  for (const auto& ld : legData_) {
301  for (auto ind : ld.indices()) {
302  // only handle equity and commodity for now
303  if (ind.substr(0, 5) != "COMM-" && ind.substr(0, 3) != "EQ-")
304  continue;
305 
306  boost::shared_ptr<Index> index = parseIndex(ind);
307 
308  if (auto ei = boost::dynamic_pointer_cast<EquityIndex>(index)) {
309  result[AssetClass::EQ].insert(ei->name());
310  } else if (auto ci = boost::dynamic_pointer_cast<QuantExt::CommodityIndex>(index)) {
311  result[AssetClass::COM].insert(ci->name());
312  }
313  }
314  }
315 
316  return result;
317 }

319 void Swap::fromXML(XMLNode* node) {
320  Trade::fromXML(node);
321  legData_.clear();
322  XMLNode* swapNode = XMLUtils::getChildNode(node, tradeType() + "Data");
323  // backwards compatibility
324  if (swapNode == nullptr) {
325  swapNode = XMLUtils::getChildNode(node, "SwapData");
326  }
327  QL_REQUIRE(swapNode, "Swap::fromXML(): expected '" << tradeType() << "Data'"
328  << (tradeType() == "Swap" ? "" : " or 'SwapData'"));
329 
330  settlement_ = XMLUtils::getChildValue(swapNode, "Settlement", false);
331  if (settlement_ == "")
332  settlement_ = "Physical";
333 
334  vector<XMLNode*> nodes = XMLUtils::getChildrenNodes(swapNode, "LegData");
335  for (Size i = 0; i < nodes.size(); i++) {
336  auto ld = createLegData();
337  ld->fromXML(nodes[i]);
338  legData_.push_back(*boost::static_pointer_cast<LegData>(ld));
339  }
340 }
