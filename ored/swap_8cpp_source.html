45void Swap::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
46 DLOG("Swap::build() called for trade " << id());
47
48 QL_REQUIRE(legData_.size() >= 1, "Swap must have at least 1 leg");
49 const boost::shared_ptr<Market> market = engineFactory->market();
50
51 // allow minor currencies in case first leg is equity
52 Currency currency = parseCurrencyWithMinors(legData_[0].currency());
53 string ccy_str = currency.code();
54
55 Size numLegs = legData_.size();
56 legPayers_ = vector<bool>(numLegs);
57 std::vector<QuantLib::Currency> currencies(numLegs);
58 legs_.resize(numLegs);
59
60 isXCCY_ = false;
61 isResetting_ = false;
62
63 for (Size i = 0; i < numLegs; ++i) {
64 // allow minor currencies for Equity legs as some exchanges trade in these, e.g LSE in pence - GBX or GBp
65 // minor currencies on other legs will fail here
66 if (legData_[i].legType() == "Equity")
67 currencies[i] = parseCurrencyWithMinors(legData_[i].currency());
68 else
69 currencies[i] = parseCurrency(legData_[i].currency());
70 if (currencies[i] != currency)
71 isXCCY_ = true;
72 isResetting_ = isResetting_ || (!legData_[i].isNotResetXCCY());
73 }
74
75 static std::set<std::string> eligibleForXbs = {"Fixed", "Floating"};
76 bool useXbsCurves = true;
77 for(Size i=0;i<numLegs;++i) {
78 useXbsCurves = useXbsCurves && (eligibleForXbs.find(legData_[i].legType()) != eligibleForXbs.end());
79 }
80
81 boost::shared_ptr<EngineBuilder> builder =
82 isXCCY_ ? engineFactory->builder("CrossCurrencySwap") : engineFactory->builder("Swap");
83 auto configuration = builder->configuration(MarketContext::pricing);
84
85 for (Size i = 0; i < numLegs; ++i) {
86 legPayers_[i] = legData_[i].isPayer();
87 auto legBuilder = engineFactory->legBuilder(legData_[i].legType());
88 legs_[i] =
89 legBuilder->buildLeg(legData_[i], engineFactory, requiredFixings_, configuration, Null<Date>(), useXbsCurves);
90 DLOG("Swap::build(): currency[" << i << "] = " << currencies[i]);
91
92 // add notional leg, if applicable
93 auto leg = buildNotionalLeg(legData_[i], legs_[i], requiredFixings_, engineFactory->market(), configuration);
94 applyIndexing(leg, legData_[i], engineFactory, requiredFixings_, Null<Date>(), useXbsCurves);
95 if (!leg.empty()) {
96 legs_.push_back(leg);
97 legPayers_.push_back(legPayers_[i]);
98 currencies.push_back(currencies[i]);
99 }
100 } // for legs
101
102 // The npv currency, notional currency and current notional are taken from the first leg that
103 // appears in the XML that has a notional. If no such leg exists the notional currency
104 // and current notional are left empty and the npv currency is set to the first leg's currency
105
106 notionalTakenFromLeg_ = 0;
107 for (; notionalTakenFromLeg_ < legData_.size(); ++notionalTakenFromLeg_) {
108 const LegData& d = legData_[notionalTakenFromLeg_];
109 if (!d.notionals().empty())
110 break;
111 }
112
113 if (notionalTakenFromLeg_ == legData_.size()) {
114 ALOG("no suitable leg found to set notional, set to null and notionalCurrency to empty string");
115 notional_ = Null<Real>();
116 notionalCurrency_ = "";
117 // parse for currency in case first leg is Equity, we only want the major currency for NPV
118 npvCurrency_ = parseCurrencyWithMinors(legData_.front().currency()).code();
119 } else {
120 if (legData_[notionalTakenFromLeg_].schedule().hasData()) {
121 Schedule schedule = makeSchedule(legData_[notionalTakenFromLeg_].schedule());
122 auto notional =
123 buildScheduledVectorNormalised(legData_[notionalTakenFromLeg_].notionals(),
124 legData_[notionalTakenFromLeg_].notionalDates(), schedule, 0.0);
125 Date today = Settings::instance().evaluationDate();
126 auto d = std::upper_bound(schedule.dates().begin(), schedule.dates().end(), today);
127 // forward starting => take first notional
128 // on or after last schedule date => zero notional
129 // in between => notional of current period
130 if (d == schedule.dates().begin())
131 notional_ = notional.at(0);
132 else if (d == schedule.dates().end())
133 notional_ = 0.0;
134 else
135 notional_ = notional.at(std::distance(schedule.dates().begin(), d) - 1);
136 } else {
137 notional_ = legData_[notionalTakenFromLeg_].notionals().at(0);
138 }
139 // parse for currency in case leg is Equity, we only want the major currency for NPV and Notional
140 notionalCurrency_ = parseCurrencyWithMinors(legData_[notionalTakenFromLeg_].currency()).code();
141 npvCurrency_ = parseCurrencyWithMinors(legData_[notionalTakenFromLeg_].currency()).code();
142 DLOG("Notional is " << notional_ << " " << notionalCurrency_);
143 }
144
145 Currency npvCcy = parseCurrency(npvCurrency_);
146 DLOG("npv currency is " << npvCurrency_);
147
148 if (isXCCY_) {
149 boost::shared_ptr<QuantExt::CurrencySwap> swap(
150 new QuantExt::CurrencySwap(legs_, legPayers_, currencies, settlement_ == "Physical", isResetting_));
151 boost::shared_ptr<CrossCurrencySwapEngineBuilderBase> swapBuilder =
152 boost::dynamic_pointer_cast<CrossCurrencySwapEngineBuilderBase>(builder);
153 QL_REQUIRE(swapBuilder, "No Builder found for CrossCurrencySwap " << id());
154 swap->setPricingEngine(swapBuilder->engine(currencies, npvCcy));
155 // take the first legs currency as the npv currency (arbitrary choice)
156 instrument_.reset(new VanillaInstrument(swap));
157 } else {
158 boost::shared_ptr<QuantLib::Swap> swap(new QuantLib::Swap(legs_, legPayers_));
159 boost::shared_ptr<SwapEngineBuilderBase> swapBuilder =
160 boost::dynamic_pointer_cast<SwapEngineBuilderBase>(builder);
161 QL_REQUIRE(swapBuilder, "No Builder found for Swap " << id());
162 swap->setPricingEngine(swapBuilder->engine(npvCcy));
163 instrument_.reset(new VanillaInstrument(swap));
164 }
165
166 DLOG("Set instrument wrapper");
167
168 // set Leg Currencies
169 legCurrencies_ = vector<string>(currencies.size());
170 for (Size i = 0; i < currencies.size(); i++)
171 legCurrencies_[i] = currencies[i].code();
172
173 // set maturity
174 maturity_ = Date::minDate();
175 Date startDate = Date::maxDate();
176 for (auto const& l : legs_) {
177 if (!l.empty()) {
178 maturity_ = std::max(maturity_, l.back()->date());
179 startDate = std::min(startDate, l.front()->date());
180 boost::shared_ptr<Coupon> coupon = boost::dynamic_pointer_cast<Coupon>(l.front());
181 if (coupon)
182 startDate = std::min(startDate, coupon->accrualStartDate());
183 }
184 }
185
186 additionalData_["startDate"] = to_string(startDate);
187
188 // ISDA taxonomy
189 additionalData_["isdaAssetClass"] = string("Interest Rate");
190 additionalData_["isdaBaseProduct"] = string(isXCCY_ ? "Cross Currency" : "IR Swap");
191 additionalData_["isdaSubProduct"] = isdaSubProductSwap(id(), legData_);
192 additionalData_["isdaTransaction"] = string("");
193}

195const std::map<std::string,boost::any>& Swap::additionalData() const {
196 Size numLegs = legData_.size();
197 // use the build time as of date to determine current notionals
198 boost::shared_ptr<QuantLib::Swap> swap = boost::dynamic_pointer_cast<QuantLib::Swap>(instrument_->qlInstrument());
199 boost::shared_ptr<QuantExt::CurrencySwap> cswap = boost::dynamic_pointer_cast<QuantExt::CurrencySwap>(instrument_->qlInstrument());
200 std::map<std::string, Real> legNpv; // by currency
201 for (Size i = 0; i < numLegs; ++i) {
202 string legID = to_string(i+1);
203 additionalData_["legType[" + legID + "]"] = legData_[i].legType();
204 additionalData_["isPayer[" + legID + "]"] = legData_[i].isPayer();
205 additionalData_["notionalCurrency[" + legID + "]"] = legData_[i].currency();
206 if (!isXCCY_) {
207 if (swap)
208 additionalData_["legNPV[" + legID + "]"] = swap->legNPV(i);
209 else
210 ALOG("single currency swap underlying instrument not set, skip leg npv reporting");
211 }
212 else {
213 if (cswap) {
214 // The currency swap has more legs than the swap wrapper (additional notional legs), so aggregate by currency
215 Real legNpv = 0;
216 Real legNpvInCcy = 0;
217 for (Size j = 0; j < cswap->legs().size(); ++j) {
218 if (cswap->legCurrency(j).code() == legData_[i].currency()) {
219 legNpv += cswap->legNPV(j);
220 legNpvInCcy += cswap->inCcyLegNPV(j);
221 }
222 }
223 additionalData_["legNPV[" + legID + "]"] = legNpv;
224 additionalData_["legNPVCCY[" + legID + "]"] = legNpvInCcy;
225 }
226 else
227 ALOG("cross currency swap underlying instrument not set, skip leg npv reporting");
228 }
229 setLegBasedAdditionalData(i);
230 }
231 return additionalData_;
232}

234QuantLib::Real Swap::notional() const {
235 // try to get the notional from the additional results of the instrument
236 try {
237 return instrument_->qlInstrument(true)->result<Real>("currentNotional");
238 } catch (const std::exception& e) {
239 WLOG("swap engine does not provide current notional: " << e.what() << ", using fallback");
240 // Try getting current notional from coupons
241 if (notionalTakenFromLeg_ < legs_.size()) {
242 Real n = currentNotional(legs_[notionalTakenFromLeg_]);
243 if (fabs(n) > QL_EPSILON) {
244 return n;
245 }
246 }
247 // else return the face value
248 WLOG("swap does not provide coupon notionals, using face value");
249 return notional_;
250 }

264map<AssetClass, set<string>>
265Swap::underlyingIndices(const boost::shared_ptr<ReferenceDataManager>& referenceDataManager) const {
266
267 map<AssetClass, set<string>> result;
268 for (const auto& ld : legData_) {
269 for (auto ind : ld.indices()) {
270 // only handle equity and commodity for now
271 if (ind.substr(0, 5) != "COMM-" && ind.substr(0, 3) != "EQ-")
272 continue;
273
274 boost::shared_ptr<Index> index = parseIndex(ind);
275
276 if (auto ei = boost::dynamic_pointer_cast<EquityIndex2>(index)) {
277 result[AssetClass::EQ].insert(ei->name());
278 } else if (auto ci = boost::dynamic_pointer_cast<QuantExt::CommodityIndex>(index)) {
279 result[AssetClass::COM].insert(ci->name());
280 }
281 }
282 }
283
284 return result;
285}

287std::string isdaSubProductSwap(const std::string& tradeId, const vector<LegData>& legData) {
288
289 Size nFixed = 0;
290 Size nFloating = 0;
291 for (Size i = 0; i < legData.size(); ++i) {
292 std::string type = legData[i].legType();
293 if (type == "Fixed" ||
294 type == "ZeroCouponFixed" ||
295 type == "Cashflow")
296 nFixed++;
297 else if (type == "Floating" ||
298 type == "CPI" ||
299 type == "YY" ||
300 type == "CMS" ||
301 type == "DigitalCMS" ||
302 type == "CMSSpread" ||
303 type == "DigitalCMSSpread" ||
304 type == "CMB" ||
305 type == "Equity")
306 nFloating++;
307 else {
308 ALOG("leg type " << type << " not mapped for trade " << tradeId);
309 }
310 }
311
312 if (nFixed == 0)
313 return string("Basis");
314 else if (nFloating >= 1)
315 return string("Fixed Float");
316 else
317 return string("Fixed Fixed");
318}

320void Swap::fromXML(XMLNode* node) {
321 Trade::fromXML(node);
322 legData_.clear();
323 XMLNode* swapNode = XMLUtils::getChildNode(node, tradeType() + "Data");
324 // backwards compatibility
325 if (swapNode == nullptr) {
326 swapNode = XMLUtils::getChildNode(node, "SwapData");
327 }
328 QL_REQUIRE(swapNode, "Swap::fromXML(): expected '" << tradeType() << "Data'"
329 << (tradeType() == "Swap" ? "" : " or 'SwapData'"));
330
331 settlement_ = XMLUtils::getChildValue(swapNode, "Settlement", false);
332 if (settlement_ == "")
333 settlement_ = "Physical";
334
335 vector<XMLNode*> nodes = XMLUtils::getChildrenNodes(swapNode, "LegData");
336 for (Size i = 0; i < nodes.size(); i++) {
337 auto ld = createLegData();
338 ld->fromXML(nodes[i]);
339 legData_.push_back(*boost::static_pointer_cast<LegData>(ld));
340 }
341}

345XMLNode* Swap::toXML(XMLDocument& doc) {
346 XMLNode* node = Trade::toXML(doc);
347 XMLNode* swapNode = doc.allocNode(tradeType() + "Data");
348 XMLUtils::appendNode(node, swapNode);
349
350 if (settlement_ == "Cash")
351 XMLUtils::addChild(doc, swapNode, "Settlement", settlement_);
352 for (Size i = 0; i < legData_.size(); i++)
353 XMLUtils::appendNode(swapNode, legData_[i].toXML(doc));
354 return node;
355}
