45void Swap::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
46 DLOG("Swap::build() called for trade " << id());
47
48 QL_REQUIRE(legData_.size() >= 1, "Swap must have at least 1 leg");
49 const boost::shared_ptr<Market> market = engineFactory->market();
50
51 // allow minor currencies in case first leg is equity
52 Currency currency = parseCurrencyWithMinors(legData_[0].currency());
53 string ccy_str = currency.code();
54
55 Size numLegs = legData_.size();
56 legPayers_ = vector<bool>(numLegs);
57 std::vector<QuantLib::Currency> currencies(numLegs);
58 legs_.resize(numLegs);
59
60 isXCCY_ = false;
61 isResetting_ = false;
62
63 for (Size i = 0; i < numLegs; ++i) {
64 // allow minor currencies for Equity legs as some exchanges trade in these, e.g LSE in pence - GBX or GBp
65 // minor currencies on other legs will fail here
66 if (legData_[i].legType() == "Equity")
67 currencies[i] = parseCurrencyWithMinors(legData_[i].currency());
68 else
69 currencies[i] = parseCurrency(legData_[i].currency());
70 if (currencies[i] != currency)
71 isXCCY_ = true;
72 isResetting_ = isResetting_ || (!legData_[i].isNotResetXCCY());
73 }
74
75 boost::shared_ptr<EngineBuilder> builder =
76 isXCCY_ ? engineFactory->builder("CrossCurrencySwap") : engineFactory->builder("Swap");
77 auto configuration = builder->configuration(MarketContext::pricing);
78
79 for (Size i = 0; i < numLegs; ++i) {
80 legPayers_[i] = legData_[i].isPayer();
81 auto legBuilder = engineFactory->legBuilder(legData_[i].legType());
82 legs_[i] = legBuilder->buildLeg(legData_[i], engineFactory, requiredFixings_, configuration);
83 DLOG("Swap::build(): currency[" << i << "] = " << currencies[i]);
84
85 // add notional leg, if applicable
86 auto leg = buildNotionalLeg(legData_[i], legs_[i], requiredFixings_, engineFactory->market(), configuration);
87 if (!leg.empty()) {
88 legs_.push_back(leg);
89 legPayers_.push_back(legPayers_[i]);
90 currencies.push_back(currencies[i]);
91 }
92 } // for legs
93
94 // The npv currency, notional currency and current notional are taken from the first leg that
95 // appears in the XML that has a notional. If no such leg exists the notional currency
96 // and current notional are left empty and the npv currency is set to the first leg's currency
97
98 notionalTakenFromLeg_ = 0;
99 for (; notionalTakenFromLeg_ < legData_.size(); ++notionalTakenFromLeg_) {
100 const LegData& d = legData_[notionalTakenFromLeg_];
101 if (!d.notionals().empty())
102 break;
103 }
104
105 if (notionalTakenFromLeg_ == legData_.size()) {
106 ALOG("no suitable leg found to set notional, set to null and notionalCurrency to empty string");
107 notional_ = Null<Real>();
108 notionalCurrency_ = "";
109 // parse for currency in case first leg is Equity, we only want the major currency for NPV
110 npvCurrency_ = parseCurrencyWithMinors(legData_.front().currency()).code();
111 } else {
112 if (legData_[notionalTakenFromLeg_].schedule().hasData()) {
113 Schedule schedule = makeSchedule(legData_[notionalTakenFromLeg_].schedule());
114 auto notional =
115 buildScheduledVectorNormalised(legData_[notionalTakenFromLeg_].notionals(),
116 legData_[notionalTakenFromLeg_].notionalDates(), schedule, 0.0);
117 Date today = Settings::instance().evaluationDate();
118 auto d = std::upper_bound(schedule.dates().begin(), schedule.dates().end(), today);
119 // forward starting => take first notional
120 // on or after last schedule date => zero notional
121 // in between => notional of current period
122 if (d == schedule.dates().begin())
123 notional_ = notional.at(0);
124 else if (d == schedule.dates().end())
125 notional_ = 0.0;
126 else
127 notional_ = notional.at(std::distance(schedule.dates().begin(), d) - 1);
128 } else {
129 notional_ = legData_[notionalTakenFromLeg_].notionals().at(0);
130 }
131 // parse for currency in case leg is Equity, we only want the major currency for NPV and Notional
132 notionalCurrency_ = parseCurrencyWithMinors(legData_[notionalTakenFromLeg_].currency()).code();
133 npvCurrency_ = parseCurrencyWithMinors(legData_[notionalTakenFromLeg_].currency()).code();
134 DLOG("Notional is " << notional_ << " " << notionalCurrency_);
135 }
136
137 Currency npvCcy = parseCurrency(npvCurrency_);
138 DLOG("npv currency is " << npvCurrency_);
139
140 if (isXCCY_) {
141 boost::shared_ptr<QuantExt::CurrencySwap> swap(
142 new QuantExt::CurrencySwap(legs_, legPayers_, currencies, settlement_ == "Physical", isResetting_));
143 boost::shared_ptr<CrossCurrencySwapEngineBuilderBase> swapBuilder =
144 boost::dynamic_pointer_cast<CrossCurrencySwapEngineBuilderBase>(builder);
145 QL_REQUIRE(swapBuilder, "No Builder found for CrossCurrencySwap " << id());
146 swap->setPricingEngine(swapBuilder->engine(currencies, npvCcy));
147 // take the first legs currency as the npv currency (arbitrary choice)
148 instrument_.reset(new VanillaInstrument(swap));
149 } else {
150 boost::shared_ptr<QuantLib::Swap> swap(new QuantLib::Swap(legs_, legPayers_));
151 boost::shared_ptr<SwapEngineBuilderBase> swapBuilder =
152 boost::dynamic_pointer_cast<SwapEngineBuilderBase>(builder);
153 QL_REQUIRE(swapBuilder, "No Builder found for Swap " << id());
154 swap->setPricingEngine(swapBuilder->engine(npvCcy));
155 instrument_.reset(new VanillaInstrument(swap));
156 }
157
158 DLOG("Set instrument wrapper");
159
160 // set Leg Currencies
161 legCurrencies_ = vector<string>(currencies.size());
162 for (Size i = 0; i < currencies.size(); i++)
163 legCurrencies_[i] = currencies[i].code();
164
165 // set maturity
166 maturity_ = Date::minDate();
167 Date startDate = Date::maxDate();
168 for (auto const& l : legs_) {
169 if (!l.empty()) {
170 maturity_ = std::max(maturity_, l.back()->date());
171 startDate = std::min(startDate, l.front()->date());
172 boost::shared_ptr<Coupon> coupon = boost::dynamic_pointer_cast<Coupon>(l.front());
173 if (coupon)
174 startDate = std::min(startDate, coupon->accrualStartDate());
175 }
176 }
177
178 additionalData_["startDate"] = to_string(startDate);
179}

181const std::map<std::string,boost::any>& Swap::additionalData() const {
182 Size numLegs = legData_.size();
183 // use the build time as of date to determine current notionals
184 Date asof = Settings::instance().evaluationDate();
185 boost::shared_ptr<QuantLib::Swap> swap = boost::dynamic_pointer_cast<QuantLib::Swap>(instrument_->qlInstrument());
186 boost::shared_ptr<QuantExt::CurrencySwap> cswap = boost::dynamic_pointer_cast<QuantExt::CurrencySwap>(instrument_->qlInstrument());
187 std::map<std::string, Real> legNpv; // by currency
188 for (Size i = 0; i < numLegs; ++i) {
189 string legID = to_string(i+1);
190 additionalData_["legType[" + legID + "]"] = legData_[i].legType();
191 additionalData_["isPayer[" + legID + "]"] = legData_[i].isPayer();
192 additionalData_["notionalCurrency[" + legID + "]"] = legData_[i].currency();
193 if (!isXCCY_) {
194 if (swap)
195 additionalData_["legNPV[" + legID + "]"] = swap->legNPV(i);
196 else
197 ALOG("single currency swap underlying instrument not set, skip leg npv reporting");
198 }
199 else {
200 if (cswap) {
201 // The currency swap has more legs than the swap wrapper (additional notional legs), so aggregate by currency
202 Real legNpv = 0;
203 Real legNpvInCcy = 0;
204 for (Size j = 0; j < cswap->legs().size(); ++j) {
205 if (cswap->legCurrency(j).code() == legData_[i].currency()) {
206 legNpv += cswap->legNPV(j);
207 legNpvInCcy += cswap->inCcyLegNPV(j);
208 }
209 }
210 additionalData_["legNPV[" + legID + "]"] = legNpv;
211 additionalData_["legNPVCCY[" + legID + "]"] = legNpvInCcy;
212 }
213 else
214 ALOG("cross currency swap underlying instrument not set, skip leg npv reporting");
215 }
216 for (Size j = 0; j < legs_[i].size(); ++j) {
217 boost::shared_ptr<CashFlow> flow = legs_[i][j];
218 // pick flow with earliest future payment date on this leg
219 if (flow->date() > asof) {
220 Real flowAmount = 0.0;
221 try { flowAmount = flow->amount(); }
222 catch(std::exception& e) {
223 ALOG("flow amount could not be determined for trade " << id() << ", set to zero: " << e.what());
224 }
225 additionalData_["amount[" + legID + "]"] = flowAmount;
226 additionalData_["paymentDate[" + legID + "]"] = to_string(flow->date());
227 boost::shared_ptr<Coupon> coupon = boost::dynamic_pointer_cast<Coupon>(flow);
228 string couponId = to_string(j);
229 if (coupon) {
230 Real currentNotional = 0;
231 try { currentNotional = coupon->nominal(); }
232 catch(std::exception& e) {
233 ALOG("current notional could not be determined for trade " << id() << ", set to zero: " << e.what());
234 }
235 additionalData_["currentNotional[" + legID + "]"] = currentNotional;
236
237 Real rate = 0;
238 try { rate = coupon->rate(); }
239 catch(std::exception& e) {
240 ALOG("coupon rate could not be determined for trade " << id() << ", set to zero: " << e.what());
241 }
242 additionalData_["rate[" + legID + "][" + couponId + "]"] = rate;
243
244 boost::shared_ptr<FloatingRateCoupon> frc = boost::dynamic_pointer_cast<FloatingRateCoupon>(flow);
245 if (frc) {
246 additionalData_["index[" + legID + "]"] = frc->index()->name();
247 additionalData_["spread[" + legID + "]"] = frc->spread();
248 }
249
250 boost::shared_ptr<EquityCoupon> eqc = boost::dynamic_pointer_cast<EquityCoupon>(flow);
251 if (eqc) {
252 EquityCouponPricer::AdditionalResultCache arc = eqc->pricer()->additionalResultCache();
253 additionalData_["initialPrice[" + legID + "][" + couponId + "]"] = arc.initialPrice;
254 additionalData_["endEquityFixing[" + legID + "][" + couponId + "]"] = arc.endFixing;
255 if (arc.startFixing != Null<Real>())
256 additionalData_["startEquityFixing[" + legID + "][" + couponId + "]"] = arc.startFixing;
257 if (arc.startFixingTotal != Null<Real>())
258 additionalData_["startEquityFixingTotal[" + legID + "][" + couponId + "]"] =
259 arc.startFixingTotal;
260 if (arc.endFixingTotal != Null<Real>())
261 additionalData_["endEquityFixingTotal[" + legID + "][" + couponId + "]"] =
262 arc.endFixingTotal;
263 if (arc.startFxFixing != Null<Real>())
264 additionalData_["startFxFixing[" + legID + "][" + couponId + "]"] = arc.startFxFixing;
265 if (arc.endFxFixing != Null<Real>())
266 additionalData_["endFxFixing[" + legID + "][" + couponId + "]"] = arc.endFxFixing;
267 if (arc.pastDividends != Null<Real>())
268 additionalData_["pastDividends[" + legID + "][" + couponId + "]"] = arc.pastDividends;
269 if (arc.forecastDividends != Null<Real>())
270 additionalData_["forecastDividends[" + legID + "][" + couponId + "]"] =
271 arc.forecastDividends;
272 }
273 }
274 break;
275 }
276 }
277 if (legs_[i].size() > 0) {
278 boost::shared_ptr<Coupon> coupon = boost::dynamic_pointer_cast<Coupon>(legs_[i][0]);
279 if (coupon) {
280 Real originalNotional = 0.0;
281 try { originalNotional = coupon->nominal(); }
282 catch(std::exception& e) {
283 ALOG("original nominal could not be determined for trade " << id() << ", set to zero: " << e.what());
284 }
285 additionalData_["originalNotional[" + legID + "]"] = originalNotional;
286 }
287 }
288 }
289 return additionalData_;
290}

292QuantLib::Real Swap::notional() const {
293 // try to get the notional from the additional results of the instrument
294 try {
295 return instrument_->qlInstrument(true)->result<Real>("currentNotional");
296 } catch (const std::exception& e) {
297 WLOG("swap engine does not provide current notional: " << e.what() << ", using fallback");
298 // Try getting current notional from coupons
299 if (notionalTakenFromLeg_ < legs_.size()) {
300 Real n = currentNotional(legs_[notionalTakenFromLeg_]);
301 if (fabs(n) > QL_EPSILON) {
302 return n;
303 }
304 }
305 // else return the face value
306 WLOG("swap does not provide coupon notionals, using face value");
307 return notional_;
308 }

322map<AssetClass, set<string>>
323Swap::underlyingIndices(const boost::shared_ptr<ReferenceDataManager>& referenceDataManager) const {
324
325 map<AssetClass, set<string>> result;
326 for (const auto& ld : legData_) {
327 for (auto ind : ld.indices()) {
328 // only handle equity and commodity for now
329 if (ind.substr(0, 5) != "COMM-" && ind.substr(0, 3) != "EQ-")
330 continue;
331
332 boost::shared_ptr<Index> index = parseIndex(ind);
333
334 if (auto ei = boost::dynamic_pointer_cast<EquityIndex>(index)) {
335 result[AssetClass::EQ].insert(ei->name());
336 } else if (auto ci = boost::dynamic_pointer_cast<QuantExt::CommodityIndex>(index)) {
337 result[AssetClass::COM].insert(ci->name());
338 }
339 }
340 }
341
342 return result;
343}

345void Swap::fromXML(XMLNode* node) {
346 Trade::fromXML(node);
347 legData_.clear();
348 XMLNode* swapNode = XMLUtils::getChildNode(node, tradeType() + "Data");
349 // backwards compatibility
350 if (swapNode == nullptr) {
351 swapNode = XMLUtils::getChildNode(node, "SwapData");
352 }
353 QL_REQUIRE(swapNode, "Swap::fromXML(): expected '" << tradeType() << "Data'"
354 << (tradeType() == "Swap" ? "" : " or 'SwapData'"));
355
356 settlement_ = XMLUtils::getChildValue(swapNode, "Settlement", false);
357 if (settlement_ == "")
358 settlement_ = "Physical";
359
360 vector<XMLNode*> nodes = XMLUtils::getChildrenNodes(swapNode, "LegData");
361 for (Size i = 0; i < nodes.size(); i++) {
362 auto ld = createLegData();
363 ld->fromXML(nodes[i]);
364 legData_.push_back(*boost::static_pointer_cast<LegData>(ld));
365 }
366}

370XMLNode* Swap::toXML(XMLDocument& doc) {
371 XMLNode* node = Trade::toXML(doc);
372 XMLNode* swapNode = doc.allocNode(tradeType() + "Data");
373 XMLUtils::appendNode(node, swapNode);
374
375 if (settlement_ == "Cash")
376 XMLUtils::addChild(doc, swapNode, "Settlement", settlement_);
377 for (Size i = 0; i < legData_.size(); i++)
378 XMLUtils::appendNode(swapNode, legData_[i].toXML(doc));
379 return node;
