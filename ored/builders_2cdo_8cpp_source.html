38boost::shared_ptr<PricingEngine> GaussCopulaBucketingCdoEngineBuilder::engineImpl(
39 const Currency& ccy, bool isIndexCDS, const vector<string>& creditCurves,
40 const boost::shared_ptr<SimpleQuote>& calibrationFactor,
41 const QuantLib::Real fixedRecovery) {
42 if (isIndexCDS) {
43 Handle<YieldTermStructure> yts = market_->discountCurve(ccy.code(), configuration(MarketContext::pricing));
44
45 std::vector<Handle<DefaultProbabilityTermStructure>> dpts;
46 std::vector<Real> recovery;
47 Handle<DefaultProbabilityTermStructure> clientCurve;
48 Handle<DefaultProbabilityTermStructure> baseCurve;
49 vector<Time> baseCurveTimes;
50
51 for (auto& c : creditCurves) {
52 Real recoveryRate = market_->recoveryRate(c, configuration(MarketContext::pricing))->value();
53 Handle<DefaultProbabilityTermStructure> targetCurve;
54
55 auto it = globalParameters_.find("RunType");
56 if (calibrateConstituentCurve() && it != globalParameters_.end() && it->second != "PortfolioAnalyser") {
57 auto orgCurve = market_->defaultCurve(c, configuration(MarketContext::pricing))->curve();
58 targetCurve =
59 SyntheticCDO::buildCalibratedConstiuentCurve(orgCurve, calibrationFactor);
60 } else {
61 targetCurve = market_->defaultCurve(c, configuration(MarketContext::pricing))->curve();
62 }
63
64 if (!optimizedSensitivityCalculation()) {
65 clientCurve = targetCurve;
66 } else {
67 // Disable sensitivites for the all but the first default probability curve
68 // Shift all other curves along with the first curve, reduce the numbers of calculations
69 // For all but the first curve use a spreaded curve, with the first curve as reference and
70 // the inital spread between the curve and the first curve, keep the spread constant
71 // in case of a spreaded curve we need to use the underlying reference curve to retrieve the
72 // correct pillar times, otherwise the interpolation grid is to coarse and we wouldnt match
73 // today's market prices
74 vector<Time> targetCurveTimes = ore::data::SyntheticCDO::extractTimeGridDefaultCurve(targetCurve);
75 // Sort times we need to interpolate an all pillars of the base and target curve
76 std::sort(targetCurveTimes.begin(), targetCurveTimes.end());
77 // Use the first curve as basis curve and update times
78 if (c == creditCurves.front()) {
79 baseCurve = targetCurve;
80 clientCurve = baseCurve;
81 baseCurveTimes = targetCurveTimes;
82 } else {
83 std::vector<Time> times;
84 std::set_union(baseCurveTimes.begin(), baseCurveTimes.end(), targetCurveTimes.begin(),
85 targetCurveTimes.end(), std::back_inserter(times));
86
87 vector<Handle<Quote>> spreads(times.size());
88 std::transform(times.begin(), times.end(), spreads.begin(), [&baseCurve, &targetCurve](Time t) {
89 Probability spread =
90 targetCurve->survivalProbability(t, true) / baseCurve->survivalProbability(t, true);
91 return Handle<Quote>(boost::make_shared<SimpleQuote>(spread));
92 });
93 clientCurve = Handle<DefaultProbabilityTermStructure>(
94 boost::make_shared<SpreadedSurvivalProbabilityTermStructure>(baseCurve, times, spreads));
95 if(baseCurve->allowsExtrapolation()) clientCurve->enableExtrapolation();
96 }
97 }
98 recovery.push_back(fixedRecovery != Null<Real>() ? fixedRecovery : recoveryRate);
99 dpts.push_back(clientCurve);
100 }
101 return boost::make_shared<QuantExt::MidPointIndexCdsEngine>(dpts, recovery, yts);
102 } else {
103 Handle<YieldTermStructure> yts = market_->discountCurve(ccy.code(), configuration(MarketContext::pricing));
104 return boost::make_shared<QuantExt::IndexCdsTrancheEngine>(yts);
105 }
106}
