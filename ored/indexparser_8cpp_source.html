122// Helper base class to build an IborIndex with a specific period and term structure given an instance of the same
123// IborIndex
124class IborIndexParser {
125public:
126 virtual ~IborIndexParser() {}
127 virtual boost::shared_ptr<IborIndex> build(Period p, const Handle<YieldTermStructure>& h) const = 0;
128 virtual string family() const = 0;
129};
130
131// General case
132template <class T> class IborIndexParserWithPeriod : public IborIndexParser {
133public:
134 boost::shared_ptr<IborIndex> build(Period p, const Handle<YieldTermStructure>& h) const override {
135 return boost::make_shared<T>(p, h);
136 }
137 string family() const override { return T(3 * Months).familyName(); }
138};
139
140// MXN TIIE
141// If tenor equates to 28 Days, i.e. tenor is 4W or 28D, ensure that the index is created
142// with a tenor of 4W under the hood. Things work better this way especially cap floor stripping.
143// We do the same with 91D -> 3M (a la KRW CD below)
144template <> class IborIndexParserWithPeriod<MXNTiie> : public IborIndexParser {
145public:
146 boost::shared_ptr<IborIndex> build(Period p, const Handle<YieldTermStructure>& h) const override {
147 if (p.units() == Days && p.length() == 28) {
148 return boost::make_shared<MXNTiie>(4 * Weeks, h);
149 } else if (p.units() == Days && p.length() == 91) {
150 return boost::make_shared<MXNTiie>(3 * Months, h);
151 } else if (p.units() == Days && (p.length() >= 180 || p.length() <= 183)) {
152 return boost::make_shared<MXNTiie>(6 * Months, h);
153 } else {
154 return boost::make_shared<MXNTiie>(p, h);
155 }
156 }
157
158 string family() const override { return MXNTiie(4 * Weeks).familyName(); }
159};
160
161// KRW CD
162// If tenor equates to 91 Days, ensure that the index is created with a tenor of 3M under the hood.
163template <> class IborIndexParserWithPeriod<KRWCd> : public IborIndexParser {
164public:
165 boost::shared_ptr<IborIndex> build(Period p, const Handle<YieldTermStructure>& h) const override {
166 if (p.units() == Days && p.length() == 91) {
167 return boost::make_shared<KRWCd>(3 * Months, h);
168 } else {
169 return boost::make_shared<KRWCd>(p, h);
170 }
171 }
172
173 string family() const override { return KRWCd(3 * Months).familyName(); }
174};
175
176// CNY REPOFIX
177// If tenor equates to 7 Days, i.e. tenor is 1W or 7D, ensure that the index is created
178// with a tenor of 1W under the hood. Similarly for 14 days, i.e. 2W.
179template <> class IborIndexParserWithPeriod<CNYRepoFix> : public IborIndexParser {
180public:
181 boost::shared_ptr<IborIndex> build(Period p, const Handle<YieldTermStructure>& h) const override {
182 if (p.units() == Days && p.length() == 7) {
183 return boost::make_shared<CNYRepoFix>(1 * Weeks, h);
184 } else if (p.units() == Days && p.length() == 14) {
185 return boost::make_shared<CNYRepoFix>(2 * Weeks, h);
186 } else {
187 return boost::make_shared<CNYRepoFix>(p, h);
188 }
189 }
190
191 string family() const override { return CNYRepoFix(1 * Weeks).familyName(); }
192};

194// Helper function to check that index name to index object is a one-to-one mapping
195void checkOneToOne(const map<string, boost::shared_ptr<OvernightIndex>>& onIndices,
196 const map<string, boost::shared_ptr<IborIndexParser>>& iborIndices) {
197
198 // Should not attempt to add the same family name to the set if the provided mappings are one to one
199 set<string> familyNames;
200
201 for (const auto& kv : onIndices) {
202 auto p = familyNames.insert(kv.second->familyName());
203 QL_REQUIRE(p.second, "Duplicate mapping for overnight index family " << *p.first << " not allowed");
204 }
205
206 for (const auto& kv : iborIndices) {
207 auto p = familyNames.insert(kv.second->family());
208 QL_REQUIRE(p.second, "Duplicate mapping for ibor index family " << *p.first << " not allowed");
209 }
210}

212boost::shared_ptr<FxIndex> parseFxIndex(const string& s, const Handle<Quote>& fxSpot,
213 const Handle<YieldTermStructure>& sourceYts,
214 const Handle<YieldTermStructure>& targetYts, const bool useConventions) {
215 std::vector<string> tokens;
216 split(tokens, s, boost::is_any_of("-"));
217 QL_REQUIRE(tokens.size() == 4, "four tokens required in " << s << ": FX-TAG-CCY1-CCY2");
218 QL_REQUIRE(tokens[0] == "FX", "expected first token to be FX");
219 Natural fixingDays = 0;
220 Calendar fixingCalendar = NullCalendar();
221 BusinessDayConvention bdc;
222 if (useConventions)
223 std::tie(fixingDays, fixingCalendar, bdc) = getFxIndexConventions(s);
224 auto index = boost::make_shared<FxIndex>(tokens[1], fixingDays, parseCurrency(tokens[2]), parseCurrency(tokens[3]),
225 fixingCalendar, fxSpot, sourceYts, targetYts);
226
227 IndexNameTranslator::instance().add(index->name(), s);
228 return index;
229}

231boost::shared_ptr<QuantExt::EquityIndex2> parseEquityIndex(const string& s) {
232 std::vector<string> tokens;
233 split(tokens, s, boost::is_any_of("-"));
234 QL_REQUIRE(tokens.size() == 2, "two tokens required in " << s << ": EQ-NAME");
235 QL_REQUIRE(tokens[0] == "EQ", "expected first token to be EQ");
236 auto index = boost::make_shared<QuantExt::EquityIndex2>(tokens[1], NullCalendar(), Currency());
237 IndexNameTranslator::instance().add(index->name(), s);
238 return index;
239}

263boost::shared_ptr<IborIndex> parseIborIndex(const string& s, string& tenor, const Handle<YieldTermStructure>& h) {
264
265 // Check the index string is of the required form before doing anything
266 vector<string> tokens;
267 split(tokens, s, boost::is_any_of("-"));
268 QL_REQUIRE(tokens.size() == 2 || tokens.size() == 3,
269 "Two or three tokens required in " << s << ": CCY-INDEX or CCY-INDEX-TERM");
270
271 // Variables used below
272 string indexStem = tokens[0] + "-" + tokens[1];
273 if (tokens.size() == 3) {
274 tenor = tokens[2];
275 } else {
276 tenor = "";
277 }
278
279 const boost::shared_ptr<Conventions>& conventions = InstrumentConventions::instance().conventions();
280 boost::shared_ptr<Convention> c;
281 if (conventions->has(s, Convention::Type::IborIndex) || conventions->has(s, Convention::Type::OvernightIndex))
282 c = conventions->get(s);
283
284 // if we have a convention given, set up the index using this convention, this overrides the parsing from
285 // hardcoded strings below if there is an overlap
286 if (c) {
287 QL_REQUIRE(c->id() == s, "ibor index convention id ('"
288 << c->id() << "') not matching ibor index string to parse ('" << s << "'");
289 Currency ccy = parseCurrency(tokens[0]);
290 if (auto conv = boost::dynamic_pointer_cast<OvernightIndexConvention>(c)) {
291 QL_REQUIRE(tenor.empty(), "no tenor allowed for convention based overnight index ('" << s << "')");
292 auto res = boost::make_shared<OvernightIndex>(tokens[0] + "-" + tokens[1], conv->settlementDays(), ccy,
293 parseCalendar(conv->fixingCalendar()),
294 parseDayCounter(conv->dayCounter()), h);
295 IndexNameTranslator::instance().add(res->name(), s);
296 return res;
297 } else if (auto conv = boost::dynamic_pointer_cast<IborIndexConvention>(c)) {
298 QL_REQUIRE(!tenor.empty(), "no tenor given for convention based Ibor index ('" << s << "'");
299 auto res = boost::make_shared<IborIndex>(tokens[0] + "-" + tokens[1], parsePeriod(tenor),
300 conv->settlementDays(), ccy, parseCalendar(conv->fixingCalendar()),
301 parseBusinessDayConvention(conv->businessDayConvention()),
302 conv->endOfMonth(), parseDayCounter(conv->dayCounter()), h);
303 IndexNameTranslator::instance().add(res->name(), s);
304 return res;
305 } else {
306 QL_FAIL("invalid convention passed to parseIborIndex(): expected OvernightIndexConvention or "
307 "IborIndexConvention");
308 }
309 }

401 static bool checked = false;
402 if (!checked) {
403 checkOneToOne(onIndices, iborIndices);
404 checked = true;
405 }
406
407 // Simple single case for USD-SIFMA (i.e. BMA)
408 if (indexStem == "USD-SIFMA") {
409 QL_REQUIRE(tenor.empty(), "A tenor is not allowed with USD-SIFMA as it is implied");
410 auto res = boost::make_shared<BMAIndexWrapper>(boost::make_shared<BMAIndex>(h));
411 IndexNameTranslator::instance().add(res->name(), s);
412 return res;
413 }
414
415 // Ibor indices with a tenor, this includes OIS term rates like USD-SOFR-3M
416 auto it = iborIndices.find(indexStem);
417 if (it != iborIndices.end() && !tenor.empty()) {
418 Period p = parsePeriod(tenor);
419 auto res = it->second->build(p, h);
420 IndexNameTranslator::instance().add(res->name(), s);
421 return res;
422 }
423
424 // Overnight indices
425 auto onIt = onIndices.find(indexStem);
426 if (onIt != onIndices.end()) {
427 QL_REQUIRE(tenor.empty(),
428 "A tenor is not allowed with the overnight index " << indexStem << " as it is implied");

430 IndexNameTranslator::instance().add(res->name(), s);
431 return res;
432 }
433

435 if (tokens[1] == "GENERIC") {
436 Period p = parsePeriod(tenor);
437 auto ccy = parseCurrency(tokens[0]);
438 auto res = boost::make_shared<GenericIborIndex>(p, ccy, h);
439 IndexNameTranslator::instance().add(res->name(), s);
440 return res;
441 }
442
443 QL_FAIL("parseIborIndex \"" << s << "\" not recognized");
444}

505boost::shared_ptr<SwapIndex> parseSwapIndex(const string& s, const Handle<YieldTermStructure>& f,
506 const Handle<YieldTermStructure>& d) {
507
508 std::vector<string> tokens;
509 split(tokens, s, boost::is_any_of("-"));
510 QL_REQUIRE(tokens.size() == 3 || tokens.size() == 4,
511 "three or four tokens required in " << s << ": CCY-CMS-TENOR or CCY-CMS-TAG-TENOR");
512 QL_REQUIRE(tokens[0].size() == 3, "invalid currency code in " << s);
513 QL_REQUIRE(tokens[1] == "CMS", "expected CMS as middle token in " << s);
514 Period p = parsePeriod(tokens.back());
515 // use default family name, if none is given
516 string familyName = tokens.size() == 4 ? tokens[0] + "-CMS-" + tokens[2] : tokens[0] + "LiborSwapIsdaFix";
517 Currency ccy = parseCurrency(tokens[0]);
518
519 const boost::shared_ptr<Conventions>& conventions = InstrumentConventions::instance().conventions();
520 boost::shared_ptr<SwapIndexConvention> swapIndexConvention;
521 boost::shared_ptr<IRSwapConvention> irSwapConvention;
522 boost::shared_ptr<OisConvention> oisCompConvention;
523 boost::shared_ptr<AverageOisConvention> oisAvgConvention;
524 if (conventions && conventions->has(s, Convention::Type::SwapIndex)) {
525 swapIndexConvention = boost::dynamic_pointer_cast<SwapIndexConvention>(conventions->get(s));
526 QL_REQUIRE(swapIndexConvention, "internal error: could not cast to SwapIndexConvention");
527 QL_REQUIRE(conventions->has(swapIndexConvention->conventions(), Convention::Type::Swap) ||
528 conventions->has(swapIndexConvention->conventions(), Convention::Type::OIS) ||
529 conventions->has(swapIndexConvention->conventions(), Convention::Type::AverageOIS),
530 "do not have swap or ois conventions for '"
531 << swapIndexConvention->conventions() << "', required from swap index convention '" << s << "'");
532 irSwapConvention =
533 boost::dynamic_pointer_cast<IRSwapConvention>(conventions->get(swapIndexConvention->conventions()));
534 oisCompConvention =
535 boost::dynamic_pointer_cast<OisConvention>(conventions->get(swapIndexConvention->conventions()));
536 oisAvgConvention =
537 boost::dynamic_pointer_cast<AverageOisConvention>(conventions->get(swapIndexConvention->conventions()));
538 QL_REQUIRE(irSwapConvention || oisCompConvention || oisAvgConvention,
539 "internal error: could not cast to IRSwapConvention, OisConvention, AverageOisConvention");
540 } else {
541 // set default conventions using a generic ibor index
542 irSwapConvention = boost::make_shared<IRSwapConvention>("dummy_swap_conv_" + tokens[0], tokens[0], "Annual",
543 "MF", "A365", tokens[0] + "-GENERIC-3M");
544 swapIndexConvention = boost::make_shared<SwapIndexConvention>("dummy_swapindex_conv_" + tokens[0],
545 "dummy_swap_conv_" + tokens[0], "");
546 }
547
548 boost::shared_ptr<SwapIndex> index;
549 if (irSwapConvention) {
550 Calendar fixingCalendar = swapIndexConvention->fixingCalendar().empty()
551 ? irSwapConvention->fixedCalendar()
552 : parseCalendar(swapIndexConvention->fixingCalendar());
553 index = boost::make_shared<SwapIndex>(familyName, p, irSwapConvention->index()->fixingDays(), ccy,
554 fixingCalendar, Period(irSwapConvention->fixedFrequency()),
555 irSwapConvention->fixedConvention(), irSwapConvention->fixedDayCounter(),
556 irSwapConvention->index()->clone(f), d);
557 } else if (oisCompConvention) {
558 Calendar fixingCalendar = swapIndexConvention->fixingCalendar().empty()
559 ? oisCompConvention->index()->fixingCalendar()
560 : parseCalendar(swapIndexConvention->fixingCalendar());
561 index = boost::make_shared<OvernightIndexedSwapIndex>(
562 familyName, p, oisCompConvention->spotLag(), ccy,
563 boost::dynamic_pointer_cast<OvernightIndex>(oisCompConvention->index()->clone(f)), true,
564 RateAveraging::Compound, Period(oisCompConvention->fixedFrequency()), d);
565 } else if (oisAvgConvention) {
566 Calendar fixingCalendar = swapIndexConvention->fixingCalendar().empty()
567 ? oisAvgConvention->index()->fixingCalendar()
568 : parseCalendar(swapIndexConvention->fixingCalendar());
569 auto index = boost::make_shared<OvernightIndexedSwapIndex>(
570 familyName, p, oisAvgConvention->spotLag(), ccy,
571 boost::dynamic_pointer_cast<OvernightIndex>(oisAvgConvention->index()->clone(f)), true,
572 RateAveraging::Simple, Period(oisAvgConvention->fixedFrequency()), d);
573 IndexNameTranslator::instance().add(index->name(), s);
574 return index;
575 } else {
576 QL_FAIL("internal error: expected irSwapConvention, oisConvention, averageOisConvention to be not null");
577 }
578 IndexNameTranslator::instance().add(index->name(), s);
579 return index;
580}

732boost::shared_ptr<BondIndex> parseBondIndex(const string& name) {
733
734 // Make sure the prefix is correct
735 string prefix = name.substr(0, 5);
736 QL_REQUIRE(prefix == "BOND-", "A bond index string must start with 'BOND-' but got " << prefix);
737
738 // Now take the remainder of the string
739 // for spot indices, this should just be the bond name
740 // for future indices, this is of the form NAME-YYYY-MM or NAME-YYYY-MM-DD where NAME is the commodity name
741 // (possibly containing hyphens) and YYYY-MM(-DD) is the expiry date of the futures contract
742 Date expiry;
743 string nameWoPrefix = name.substr(5);
744 string bondName = nameWoPrefix;
745
746 // Check for form NAME-YYYY-MM-DD
747 if (nameWoPrefix.size() > 10) {
748 string test = nameWoPrefix.substr(nameWoPrefix.size() - 10);
749 if (boost::regex_match(test, boost::regex("\\d{4}-\\d{2}-\\d{2}"))) {
750 expiry = parseDate(test);
751 bondName = nameWoPrefix.substr(0, nameWoPrefix.size() - test.size() - 1);
752 }
753 }
754
755 // Check for form NAME-YYYY-MM if NAME-YYYY-MM-DD failed
756 if (expiry == Date() && nameWoPrefix.size() > 7) {
757 string test = nameWoPrefix.substr(nameWoPrefix.size() - 7);
758 if (boost::regex_match(test, boost::regex("\\d{4}-\\d{2}"))) {
759 expiry = parseDate(test + "-01");
760 bondName = nameWoPrefix.substr(0, nameWoPrefix.size() - test.size() - 1);
761 }
762 }
763
764 // Create and return the required future index
765 boost::shared_ptr<BondIndex> index;
766 if (expiry != Date()) {
767 index= boost::make_shared<BondFuturesIndex>(expiry, bondName);
768 } else {
769 index= boost::make_shared<BondIndex>(bondName);
770 }
771 IndexNameTranslator::instance().add(index->name(), name);
772 return index;
773}

775boost::shared_ptr<ConstantMaturityBondIndex> parseConstantMaturityBondIndex(const string& name) {
776 // Expected bondId structure with at least three tokens, separated by "-", of the form CMB-FAMILY-TERM, for example:
777 // CMB-US-CMT-5Y, CMB-US-TIPS-10Y, CMB-UK-GILT-5Y, CMB-DE-BUND-10Y
778 // with two tokens in the middle to define the family
779 std::vector<string> tokens;
780 split(tokens, name, boost::is_any_of("-"));
781 QL_REQUIRE(tokens.size() >= 3, "Generic Bond ID with at least two tokens separated by - expected, found " << name);
782
783 // Make sure the prefix is correct
784 std::string prefix = tokens[0];
785 QL_REQUIRE(prefix == "CMB", "A constant maturity bond yield index string must start with 'CMB' but got " << prefix);
786
787 std::string securityFamily = tokens[1];
788 for (Size i = 2; i < tokens.size() - 1; ++i)
789 securityFamily = securityFamily + "-" + tokens[i];
790 Period underlyingPeriod = parsePeriod(tokens.back());
791 boost::shared_ptr<ConstantMaturityBondIndex> i;
792 try {
793 i = boost::make_shared<ConstantMaturityBondIndex>(prefix + "-" + securityFamily, underlyingPeriod);
794 } catch(std::exception& e) {
795 ALOG("error creating CMB index: " << e.what());
796 }
797 IndexNameTranslator::instance().add(i->name(), name);
798 return i;
799}

801boost::shared_ptr<QuantExt::CommodityIndex> parseCommodityIndex(const string& name, bool hasPrefix,
802 const Handle<PriceTermStructure>& ts,
803 const Calendar& cal, const bool enforceFutureIndex) {
804
805 // Whether we check for "COMM-" prefix depends on hasPrefix.
806 string commName = name;
807 if (hasPrefix) {
808 // Make sure the prefix is correct
809 string prefix = name.substr(0, 5);
810 QL_REQUIRE(prefix == "COMM-", "A commodity index string must start with 'COMM-' but got " << prefix);
811 commName = name.substr(5);
812 }
813
814 // Now take the remainder of the string
815 // for spot indices, this should just be the commodity name (possibly containing hyphens)
816 // for future indices, this is of the form NAME-YYYY-MM or NAME-YYYY-MM-DD where NAME is the commodity name
817 // (possibly containing hyphens) and YYYY-MM(-DD) is the expiry date of the futures contract
818 Date expiry;
819
820 // Check for form NAME-YYYY-MM-DD
821 if (commName.size() > 10) {
822 string test = commName.substr(commName.size() - 10);
823 if (boost::regex_match(test, boost::regex("\\d{4}-\\d{2}-\\d{2}"))) {
824 expiry = parseDate(test);
825 commName = commName.substr(0, commName.size() - test.size() - 1);
826 }
827 }
828
829 // Check for form NAME-YYYY-MM if NAME-YYYY-MM-DD failed
830 if (expiry == Date() && commName.size() > 7) {
831 string test = commName.substr(commName.size() - 7);
832 if (boost::regex_match(test, boost::regex("\\d{4}-\\d{2}"))) {
833 expiry = parseDate(test + "-01");
834 commName = commName.substr(0, commName.size() - test.size() - 1);
835 }
836 }
837
838 // Name to use when creating the index. This may be updated if we have a commodity future convention and IndexName
839 // is provided by the convention.
840 string indexName = commName;
841
842 // Do we have a commodity future convention for the commodity.
843 boost::shared_ptr<Conventions> conventions = InstrumentConventions::instance().conventions();
844 pair<bool, boost::shared_ptr<Convention>> p = conventions->get(commName, Convention::Type::CommodityFuture);
845 boost::shared_ptr<CommodityFutureConvention> convention;
846 if (p.first) {
847
848 convention = boost::dynamic_pointer_cast<CommodityFutureConvention>(p.second);
849
850 if (!convention->indexName().empty())
851 indexName = convention->indexName();
852
853 // If we have provided OffPeakPowerIndexData, we use that to construct the off peak power commodity index.
854 if (convention->offPeakPowerIndexData()) {
855
856 const auto& oppIdxData = *convention->offPeakPowerIndexData();
857
858 if (expiry == Date()) {
859 // If expiry is still not set use any date (off peak index is calendar daily)
860 expiry = Settings::instance().evaluationDate();
861 }
862 string suffix = "-" + to_string(expiry);
863
864 auto offPeakIndex = boost::dynamic_pointer_cast<CommodityFuturesIndex>(parseCommodityIndex(
865 oppIdxData.offPeakIndex() + suffix, false));
866 auto peakIndex = boost::dynamic_pointer_cast<CommodityFuturesIndex>(parseCommodityIndex(
867 oppIdxData.peakIndex() + suffix, false));
868
869 auto index = boost::make_shared<OffPeakPowerIndex>(indexName, expiry, offPeakIndex, peakIndex,
870 oppIdxData.offPeakHours(), oppIdxData.peakCalendar(), ts);
871 IndexNameTranslator::instance().add(index->name(), hasPrefix ? name : "COMM-" + name);
872 DLOG("parseCommodityIndex(" << name << ") -> " << index->name() << " with expiry " << index->expiryDate());
873 return index;
874 }
875 }
876
877 // Create and return the required future index
878 boost::shared_ptr<CommodityIndex> index;
879 if (expiry != Date() || (convention && enforceFutureIndex)) {
880
881 // If expiry is empty, just use any valid expiry.
882 if (expiry == Date()) {
883 ConventionsBasedFutureExpiry feCalc(*convention);
884 expiry = feCalc.nextExpiry();
885 }
886
887 bool keepDays = convention && convention->contractFrequency() == Daily;
888
889 Calendar cdr = cal;
890 if (convention && cdr == NullCalendar()) {
891 cdr = convention->calendar();
892 }
893
894 auto basisCurve = ts.empty() ? nullptr :
895 boost::dynamic_pointer_cast<CommodityBasisPriceTermStructure>(*ts);
896
897 if (basisCurve) {
898 index = boost::make_shared<CommodityBasisFutureIndex>(indexName, expiry, cdr, basisCurve);
899 } else {
900 index = boost::make_shared<CommodityFuturesIndex>(indexName, expiry, cdr, keepDays, ts);
901 }
902
903
904
905 } else {
906 index = boost::make_shared<CommoditySpotIndex>(commName, cal, ts);
907 }
908 IndexNameTranslator::instance().add(index->name(), index->name());
909 DLOG("parseCommodityIndex(" << name << ") -> " << index->name() << " with expiry " << index->expiryDate());
910 return index;
911}

913boost::shared_ptr<Index> parseIndex(const string& s) {
914 boost::shared_ptr<QuantLib::Index> ret_idx;
915 try {
916 ret_idx = parseEquityIndex(s);
917 } catch (...) {
918 }
919 if (!ret_idx) {
920 try {
921 ret_idx = parseBondIndex(s);
922 } catch (...) {
923 }
924 }
925 if (!ret_idx) {
926 try {
927 ret_idx = parseCommodityIndex(s, true, QuantLib::Handle<QuantExt::PriceTermStructure>(), QuantLib::NullCalendar(), false);
928 } catch (...) {
929 }
930 }
931 if (!ret_idx) {
932 try {
933 ret_idx = parseFxIndex(s);
934 } catch (...) {
935 }
936 }
937 if (!ret_idx) {
938 try {
939 ret_idx = parseGenericIndex(s);
940 } catch (...) {
941 }
942 }
943 if (!ret_idx) {
944 try {
945 ret_idx = parseConstantMaturityBondIndex(s);
946 } catch (...) {
947 }
948 }
949 if (!ret_idx) {
950 try {
951 ret_idx = parseIborIndex(s);
952 } catch (...) {
953 }
954 }
955 if (!ret_idx) {
956 try {
957 ret_idx = parseSwapIndex(s, Handle<YieldTermStructure>(), Handle<YieldTermStructure>());
958 } catch (...) {
959 }
960 }
961 if (!ret_idx) {
962 try {
963 ret_idx = parseZeroInflationIndex(s, Handle<ZeroInflationTermStructure>());
964 } catch (...) {
965 }
966 }
967 QL_REQUIRE(ret_idx, "parseIndex \"" << s << "\" not recognized");
968 return ret_idx;
969}

1001 static map<string, string> m = {{"DKK-TNR", "DKK-DKKOIS"}, {"EUR-EURIB", "EUR-EURIBOR"}, {"CAD-BA", "CAD-CDOR"},
1002 {"EUR-ESTR", "EUR-ESTER"}, {"EUR-STR", "EUR-ESTER"}, {"JPY-TONA", "JPY-TONAR"},
1003 {"JPY-TORF", "JPY-TONAR"}};
1004
1005 // Is start of indexName covered by the map? If so, update it.
1006 string tmpName = tokens[0] + "-" + tokens[1];
1007 if (m.count(tmpName) == 1) {
1008 tmpName = m.at(tmpName);
1009 }
1010
1011 // If there were only two tokens, return the possibly updated two tokens.
1012 if (tokens.size() == 2) {
1013 return tmpName;
1014 }
1015
1016 // Check if we have an overnight index
1017 // This covers cases like USD-FedFunds-1D and returns USD-FedFunds
1018 // (no need to check convention based overnight indices, they are always of the form CCY-INDEX)
1019 if (parsePeriod(tokens[2]) == 1 * Days && isOvernightIndex(tmpName)) {
1020 return tmpName;
1021 }
1022
1023 // Allow USD-SIFMA-1W or USD-SIFMA-7D externally. USD-SIFMA is used internally.
1024 if (tmpName == "USD-SIFMA" && (tokens[2] == "1W" || tokens[2] == "7D")) {
1025 return tmpName;
1026 }
1027
1028 return tmpName + "-" + tokens[2];
1029}

1033 split(tokens, indexName, boost::is_any_of("-"));
1034 return tokens.size() == 4 && tokens[0] == "FX";
1035}
