35/*! InMemoryReport just stores report information in local vectors and provides an interface to access
36 * the values. It could be used as a backend to a GUI
37 \ingroup report
38 */
39class InMemoryReport : public Report {
40public:
41 InMemoryReport() : i_(0) {}
42
43 Report& addColumn(const string& name, const ReportType& rt, Size precision = 0) override {
44 headers_.push_back(name);
45 columnTypes_.push_back(rt);
46 columnPrecision_.push_back(precision);
47 data_.push_back(vector<ReportType>()); // Initialise vector for
48 i_++;
49 return *this;
50 }
51
52 Report& next() override {
53 QL_REQUIRE(i_ == headers_.size(), "Cannot go to next line, only " << i_ << " entires filled");
54 i_ = 0;
55 return *this;
56 }
57
58 Report& add(const ReportType& rt) override {
59 // check type is valid
60 QL_REQUIRE(i_ < headers_.size(), "No column to add [" << rt << "] to.");
61 QL_REQUIRE(rt.which() == columnTypes_[i_].which(),
62 "Cannot add value " << rt << " of type " << rt.which() << " to column " << headers_[i_]
63 << " of type " << columnTypes_[i_].which());
64
65 data_[i_].push_back(rt);
66 i_++;
67 return *this;
68 }
69
70 void end() override {
71 QL_REQUIRE(i_ == headers_.size() || i_ == 0,
72 "report is finalized with incomplete row, got data for " << i_ << " columns out of " << columns());
73 }
74
75 // InMemoryInterface
76 Size columns() const { return headers_.size(); }
77 Size rows() const { return data_[0].size(); }
78 const string& header(Size i) const { return headers_[i]; }
79 bool hasHeader(string h) const { return std::find(headers_.begin(), headers_.end(), h) != headers_.end(); }
80 ReportType columnType(Size i) const { return columnTypes_[i]; }
81 Size columnPrecision(Size i) const { return columnPrecision_[i]; }
82 //! Returns the data
83 const vector<ReportType>& data(Size i) const {
84 QL_REQUIRE(data_[i].size() == rows(), "internal error: report column "
85 << i << " (" << header(i) << ") contains " << data_[i].size()
86 << " rows, expected are " << rows() << " rows.");
87 return data_[i];
88 }
89
90 void toFile(const string& filename, const char sep = ',', const bool commentCharacter = true, char quoteChar = '\0',
91 const string& nullString = "#N/A", bool lowerHeader = false) {
92
93 CSVFileReport cReport(filename, sep, commentCharacter, quoteChar, nullString, lowerHeader);
94
95 for (Size i = 0; i < headers_.size(); i++) {
96 cReport.addColumn(headers_[i], columnTypes_[i], columnPrecision_[i]);
97 }
98
99 auto numColumns = columns();
100 if (numColumns > 0) {
101 auto numRows = data_[0].size();
102
103 for (Size i = 0; i < numRows; i++) {
104 cReport.next();
105 for (Size j = 0; j < numColumns; j++) {
106 cReport.add(data_[j][i]);
107 }
108 }
109 }
110
111 cReport.end();
112 }
113
114private:
115 Size i_;
116 vector<string> headers_;
117 vector<ReportType> columnTypes_;
118 vector<Size> columnPrecision_;
119 vector<vector<ReportType>> data_;
120};
121} // namespace data
