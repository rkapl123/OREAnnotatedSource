42//! Composite Instrument Wrapper
43/*!
44 A Composite Instrument Wrapper will return the sum npv for all wrappers passed in.
45 This is to allow for the pricing of trades through replication
46
47 \ingroup tradedata
48*/
49class CompositeInstrumentWrapper : public ore::data::InstrumentWrapper {
50public:
51 CompositeInstrumentWrapper(const std::vector<boost::shared_ptr<InstrumentWrapper>>& wrappers,
52 const std::vector<Handle<Quote>>& fxRates, const Date& valuationDate = Date())
53 : InstrumentWrapper(boost::shared_ptr<QuantLib::Instrument>()), wrappers_(wrappers), fxRates_(fxRates),
54 valuationDate_(valuationDate) {
55 QL_REQUIRE(wrappers.size() > 0, "no instrument wrappers provided");
56 QL_REQUIRE(fxRates_.size() == wrappers_.size(), "unexpected number of fxRates provided");
57 instrument_ = wrappers[0]->qlInstrument();
58 multiplier_ = wrappers[0]->multiplier();
59
60 isOption_ = false;
61
62 for (auto w : wrappers_) {
63 if (w->isOption()) {
64 isOption_ = true;
65 break;
66 }
67 }
68 }
69
70 //! Initialise with the given date grid
71 void initialise(const std::vector<QuantLib::Date>& dates) override{};
72
73 //! reset is called every time a new path is about to be priced.
74 /*! For path dependent Wrappers, this is when internal state should be reset
75 */
76 void reset() override {
77 for (auto w : wrappers_) {
78 w->reset();
79 }
80 }
81
82 //! Return the NPV of this instrument
83 QuantLib::Real NPV() const override {
84 Date today = Settings::instance().evaluationDate();
85 if (valuationDate_ != Date()) {
86 QL_REQUIRE(today == valuationDate_, "today must be the expected valuation date for this trade");
87 }
88 Real npv = 0;
89 for (Size i = 0; i < wrappers_.size(); i++) {
90 npv += wrappers_[i]->NPV() * fxRates_[i]->value();
91 }
92 return npv;
93 }
94
95 const std::map<std::string, boost::any>& additionalResults() const override {
96 additionalResults_.clear();
97 for (auto const& w : wrappers_) {
98 additionalResults_.insert(w->additionalResults().begin(), w->additionalResults().end());
99 }
100 return additionalResults_;
101 }
102
103 //! call update on enclosed instrumentWrapper(s)
104 void updateQlInstruments() override {
105 // the instrument wrappers might contain nested lazy objects which we also want to be updated
106 for (auto w : wrappers_) {
107 w->updateQlInstruments();
108 }
109 }
110
111 //! is it an Option?
112 bool isOption() override { return isOption_; }
113
114protected:
115 bool isOption_;
116 std::vector<boost::shared_ptr<InstrumentWrapper>> wrappers_;
117 std::vector<QuantLib::Handle<Quote>> fxRates_;
118 Date valuationDate_;
119 mutable std::map<std::string, boost::any> additionalResults_;
120};
121} // namespace data
122} // namespace oreplus
