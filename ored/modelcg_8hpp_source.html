89
90 /* eval index at (past or future) obsdate:
91 - if fwddate != null, fwddate > obsdate is required. A check must be implemented that the obsdate allows for
92 the index projection. For non-inflation indices this check is simply obsdate >= refdate. For zero inflation
93 indices the is is obsdate >= basedate where the basedate is the one from the zero inflation ts asociated
94 to the index.
95 - if a historical fixing is required and missing, the behaviour depends on returnMissingFixingAsNull, if this
96 flag is true, an exception is thrown, if false, the node "nan" is returned
97 - for non-inflation indices, if ignoreTodaysFixing is true, always return the market spot for obsdate =
98 referencedate, even if a historical fixing is available; for inflation indices, ignore this flag
99 */
100 virtual std::size_t eval(const std::string& index, const Date& obsdate, const Date& fwddate,
101 const bool returnMissingFixingAsNull = false,
102 const bool ignoreTodaysFixing = false) const = 0;
103
104 // forward looking daily comp/avg, obsdate <= start < end required, result must be as of max(refdate, obsdate)

108 const Real cap, const Real floor, const bool nakedOption,

130 virtual std::vector<std::pair<std::size_t, double>> modelParameters() const = 0;

132 // get fx spot as of today directly, i.e. bypassing the cg

134
135 // get discount as of today directly, i.e. bypassing the cg
136 virtual Real getDirectDiscountT0(const Date& paydate, const std::string& currency) const = 0;

138protected:

142 // the underlying computation graph
143 boost::shared_ptr<QuantExt::ComputationGraph> g_;

145private:
146 void performCalculations() const override {}

148 // size of random variables within model
