37void EquityOption::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
38
39 // Set the assetName_ as it may have changed after lookup
40 assetName_ = equityName();
41
42 // Populate the index_ in case the option is automatic exercise.
43 const boost::shared_ptr<Market>& market = engineFactory->market();
44 index_ = *market->equityCurve(assetName_, engineFactory->configuration(MarketContext::pricing));
45
46 Currency ccy = parseCurrencyWithMinors(currency_);
47
48 // check the equity currency
49 underlyingCurrency_ =
50 market->equityCurve(assetName_, engineFactory->configuration(MarketContext::pricing))->currency();
51 QL_REQUIRE(!underlyingCurrency_.empty(), "No equity currency in equityCurve for equity " << assetName_ << ".");
52
53 // Set the strike currency - if we have a minor currency, convert the strike
54 if (!strikeCurrency_.empty())
55 strike_.setCurrency(strikeCurrency_);
56 else if (strike_.currency().empty()) {
57 // If payoff currency and underlying currency are equivalent (and payoff currency could be a minor currency)
58 if (ccy == underlyingCurrency_) {
59 TLOG("Setting strike currency to payoff currency " << ccy << " for trade " << id() << ".");
60 strike_.setCurrency(ccy.code());
61 } else {
62 // If quanto payoff, then strike currency must be populated to avoid confusion over what the
63 // currency of the strike payoff is: can be either underlying currency or payoff currency
64 QL_FAIL("Strike currency must be specified for a quanto payoff for trade " << id() << ".");
65 }
66 }
67
68 // Quanto payoff condition, i.e. currency_ != underlyingCurrency_, will be checked in VanillaOptionTrade::build()
69 // Build the trade using the shared functionality in the base class.
70 if (strike_.currency() != underlyingCurrency_.code()) {
71
72 // We have a composite EQ Trade
73
74 Currency strikeCcy = parseCurrencyWithMinors(strikeCurrency_);
75 QL_REQUIRE(ccy == strikeCcy, "Equity composite option requires pay ccy ("
76 << ccy.code() << ") to match strike ccy (" << strikeCcy.code()
77 << "), quanto composite options are not supported (underlying currency is "
78 << underlyingCurrency_.code() << ")");
79
80 Option::Type type = parseOptionType(option_.callPut());
81 boost::shared_ptr<StrikedTypePayoff> payoff(new PlainVanillaPayoff(type, strike_.value()));
82 QuantLib::Exercise::Type exerciseType = parseExerciseType(option_.style());
83 QL_REQUIRE(option_.exerciseDates().size() == 1, "Invalid number of excercise dates");
84 expiryDate_ = parseDate(option_.exerciseDates().front());
85 // Set the maturity date equal to the expiry date. It may get updated below if option is cash settled with
86 // payment after expiry.
87 maturity_ = expiryDate_;
88 // Exercise
89 boost::shared_ptr<Exercise> exercise;
90 switch (exerciseType) {
91 case QuantLib::Exercise::Type::European: {
92 exercise = boost::make_shared<EuropeanExercise>(expiryDate_);
93 break;
94 }
95 default:
96 QL_FAIL("Option Style " << option_.style() << " is not supported for an composite equity option");
97 }
98 Settlement::Type settlementType = parseSettlementType(option_.settlement());
99 // Create the instrument and then populate the name for the engine builder.
100 boost::shared_ptr<Instrument> vanilla;
101 if (exerciseType == Exercise::European && settlementType == Settlement::Cash) {
102 // We have a European cash settled option.
103
104 // Get the payment date.
105 const boost::optional<OptionPaymentData>& opd = option_.paymentData();
106 Date paymentDate = expiryDate_;
107 if (opd) {
108 if (opd->rulesBased()) {
109 const Calendar& cal = opd->calendar();
110 QL_REQUIRE(cal != Calendar(), "Need a non-empty calendar for rules based payment date.");
111 paymentDate = cal.advance(expiryDate_, opd->lag(), Days, opd->convention());
112 } else {
113 const vector<Date>& dates = opd->dates();
114 QL_REQUIRE(dates.size() == 1, "Need exactly one payment date for cash settled European option.");
115 paymentDate = dates[0];
116 }
117 QL_REQUIRE(paymentDate >= expiryDate_, "Payment date must be greater than or equal to expiry date.");
118 }
119
120 QL_REQUIRE(paymentDate <= expiryDate_,
121 "Payment date must equal expiry date for a Composite payoff. Trade: " << id() << ".");
122 }
123 QL_REQUIRE(forwardDate_ ==
124 QuantLib::Date(), "Composite payoff is not currently supported for Forward Options: Trade "
125 << id());
126 vanilla = boost::make_shared<QuantLib::VanillaOption>(payoff, exercise);
127
128 string tradeTypeBuilder = "EquityEuropeanCompositeOption";
129
130 boost::shared_ptr<EngineBuilder> builder = engineFactory->builder(tradeTypeBuilder);
131 QL_REQUIRE(builder, "No builder found for " << tradeTypeBuilder);
132
133 // TODO cast and set pricing engine
134
135 auto compositeBuilder = boost::dynamic_pointer_cast<EquityEuropeanCompositeEngineBuilder>(builder);
136 vanilla->setPricingEngine(compositeBuilder->engine(assetName_, underlyingCurrency_,
137 parseCurrency(strike_.currency()), expiryDate_));
138
139 string configuration = Market::defaultConfiguration;
140 Position::Type positionType = parsePositionType(option_.longShort());
141 Real bsInd = (positionType == QuantLib::Position::Long ? 1.0 : -1.0);
142 Real mult = quantity_ * bsInd;
143
144 std::vector<boost::shared_ptr<Instrument>> additionalInstruments;
145 std::vector<Real> additionalMultipliers;
146 maturity_ =
147 std::max(maturity_, addPremiums(additionalInstruments, additionalMultipliers, mult, option_.premiumData(),
148 -bsInd, ccy, engineFactory, configuration));
149
150 instrument_ = boost::shared_ptr<InstrumentWrapper>(
151 new VanillaInstrument(vanilla, mult, additionalInstruments, additionalMultipliers));
152 npvCurrency_ = ccy.code();
153
154 // Notional - we really need todays spot to get the correct notional.
155 // But rather than having it move around we use strike * quantity
156 notional_ = strike_.value() * quantity_;
157 notionalCurrency_ = ccy.code();
158 } else {
159 VanillaOptionTrade::build(engineFactory);
160 }
161
162 additionalData_["quantity"] = quantity_;
163 additionalData_["strike"] = strike_.value();
164 additionalData_["strikeCurrency"] = strike_.currency();
165
166 // ISDA taxonomy
167 additionalData_["isdaAssetClass"] = string("Equity");
168 additionalData_["isdaBaseProduct"] = string("Option");
169 additionalData_["isdaSubProduct"] = string("Price Return Basic Performance");
170 // skip the transaction level mapping for now
171 additionalData_["isdaTransaction"] = string("");
172}

174void EquityOption::fromXML(XMLNode* node) {
175 VanillaOptionTrade::fromXML(node);
176 XMLNode* eqNode = XMLUtils::getChildNode(node, "EquityOptionData");
177 QL_REQUIRE(eqNode, "No EquityOptionData Node");
178 option_.fromXML(XMLUtils::getChildNode(eqNode, "OptionData"));
179 XMLNode* tmp = XMLUtils::getChildNode(eqNode, "Underlying");
180 if (!tmp)
181 tmp = XMLUtils::getChildNode(eqNode, "Name");
182 equityUnderlying_.fromXML(tmp);
183 currency_ = XMLUtils::getChildValue(eqNode, "Currency", true);
184 strike_.fromXML(eqNode);
185
186 strikeCurrency_ = XMLUtils::getChildValue(eqNode, "StrikeCurrency", false);
187 if (!strikeCurrency_.empty())
188 WLOG("EquityOption::fromXML: node StrikeCurrency is deprecated, please use StrikeData node");
189 quantity_ = XMLUtils::getChildValueAsDouble(eqNode, "Quantity", true);
190}

192XMLNode* EquityOption::toXML(XMLDocument& doc) {
193 XMLNode* node = VanillaOptionTrade::toXML(doc);
194 XMLNode* eqNode = doc.allocNode("EquityOptionData");
195 XMLUtils::appendNode(node, eqNode);
196
197 XMLUtils::appendNode(eqNode, option_.toXML(doc));
198 XMLUtils::appendNode(eqNode, equityUnderlying_.toXML(doc));
199 XMLUtils::addChild(doc, eqNode, "Currency", currency_);
200
201 XMLUtils::appendNode(eqNode, strike_.toXML(doc));
202 if (!strikeCurrency_.empty())
203 XMLUtils::addChild(doc, eqNode, "StrikeCurrency", strikeCurrency_);
204
205 XMLUtils::addChild(doc, eqNode, "Quantity", quantity_);
206
207 return node;
208}
