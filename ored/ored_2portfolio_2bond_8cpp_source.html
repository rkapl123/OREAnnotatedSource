65  securityId_ = XMLUtils::getChildValue(node, "SecurityId", true);
66  referenceCurveId_ = XMLUtils::getChildValue(node, "ReferenceCurveId", false);
67  incomeCurveId_ = XMLUtils::getChildValue(node, "IncomeCurveId", false);
68  volatilityCurveId_ = XMLUtils::getChildValue(node, "VolatilityCurveId", false);
69  settlementDays_ = XMLUtils::getChildValue(node, "SettlementDays", false);
70  calendar_ = XMLUtils::getChildValue(node, "Calendar", false);
71  issueDate_ = XMLUtils::getChildValue(node, "IssueDate", false);
72  priceQuoteMethod_ = XMLUtils::getChildValue(node, "PriceQuoteMethod", false);
73  priceQuoteBaseValue_ = XMLUtils::getChildValue(node, "PriceQuoteBaseValue", false);
74  if (auto n = XMLUtils::getChildNode(node, "BondNotional")) {
75  bondNotional_ = parseReal(XMLUtils::getNodeValue(n));
76  } else {
77  bondNotional_ = 1.0;
78  }
79  XMLNode* legNode = XMLUtils::getChildNode(node, "LegData");
80  isInflationLinked_ = false;
81  while (legNode != nullptr) {
82  LegData ld;
83  ld.fromXML(legNode);
84  coupons_.push_back(ld);
85  if (ld.concreteLegData()->legType() == "CPI") {
86  isInflationLinked_ = true;
87  }
88  legNode = XMLUtils::getNextSibling(legNode, "LegData");
89  }
90  hasCreditRisk_ = XMLUtils::getChildValueAsBool(node, "CreditRisk", false, true);
91  initialise();
92 }

127 void BondData::initialise() {
128 
129  isPayer_ = false;
130  isInflationLinked_ = false;
131 
132  if (!zeroBond()) {
133 
134  // fill currency, if not directly given (which is only the case for zero bonds)
135 
136  for (Size i = 0; i < coupons().size(); ++i) {
137  if (i == 0)
138  currency_ = coupons()[i].currency();
139  else {
140  QL_REQUIRE(currency_ == coupons()[i].currency(),
141  "bond leg #" << i << " currency (" << coupons()[i].currency()
142  << ") not equal to leg #0 currency (" << coupons()[0].currency());
143  }
144  }
145 
146  // fill isPayer, FIXME zero bonds are always long
147 
148  for (Size i = 0; i < coupons().size(); ++i) {
149  if (i == 0)
150  isPayer_ = coupons()[i].isPayer();
151  else {
152  QL_REQUIRE(isPayer_ == coupons()[i].isPayer(),
153  "bond leg #" << i << " isPayer (" << std::boolalpha << coupons()[i].isPayer()
154  << ") not equal to leg #0 isPayer (" << coupons()[0].isPayer());
155  }
156  }
157 
158  // fill isInflationLinked
159  for (Size i = 0; i < coupons().size(); ++i) {
160  if (i == 0)
161  isInflationLinked_ = coupons()[i].concreteLegData()->legType() == "CPI";
162  else {
163  bool isIthCouponInflationLinked = coupons()[i].concreteLegData()->legType() == "CPI";
164  QL_REQUIRE(isInflationLinked_ == isIthCouponInflationLinked,
165  "bond leg #" << i << " isInflationLinked (" << std::boolalpha << isIthCouponInflationLinked
166  << ") not equal to leg #0 isInflationLinked (" << isInflationLinked_);
167  }
168  }
169  }
170 }

210 void Bond::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
211  DLOG("Bond::build() called for trade " << id());
212 
213  const boost::shared_ptr<Market> market = engineFactory->market();
214  boost::shared_ptr<EngineBuilder> builder = engineFactory->builder("Bond");
215  QL_REQUIRE(builder, "Bond::build(): internal error, builder is null");
216 
217  bondData_ = originalBondData_;
218  bondData_.populateFromBondReferenceData(engineFactory->referenceData());
219 
220  Date issueDate = parseDate(bondData_.issueDate());
221  Calendar calendar = parseCalendar(bondData_.calendar());
222  QL_REQUIRE(!bondData_.settlementDays().empty(),
223  "no bond settlement days given, if reference data is used, check if securityId '"
224  << bondData_.securityId() << "' is present and of type Bond.");
225  Natural settlementDays = parseInteger(bondData_.settlementDays());
226  boost::shared_ptr<QuantLib::Bond> bond;
227 
228  std::string openEndDateStr = builder->modelParameter("OpenEndDateReplacement", {}, false, "");
229  Date openEndDateReplacement = getOpenEndDateReplacement(openEndDateStr, calendar);
230  Real mult = bondData_.bondNotional() * (bondData_.isPayer() ? -1.0 : 1.0);
231  std::vector<Leg> separateLegs;
232  if (bondData_.zeroBond()) { // Zero coupon bond
233  bond.reset(new QuantLib::ZeroCouponBond(settlementDays, calendar, bondData_.faceAmount(),
234  parseDate(bondData_.maturityDate())));
235  } else { // Coupon bond
236  for (Size i = 0; i < bondData_.coupons().size(); ++i) {
237  Leg leg;
238  auto configuration = builder->configuration(MarketContext::pricing);
239  auto legBuilder = engineFactory->legBuilder(bondData_.coupons()[i].legType());
240  leg = legBuilder->buildLeg(bondData_.coupons()[i], engineFactory, requiredFixings_, configuration,
241  openEndDateReplacement);
242  separateLegs.push_back(leg);
243  } // for coupons_
244  Leg leg = joinLegs(separateLegs);
245  bond.reset(new QuantLib::Bond(settlementDays, calendar, issueDate, leg));
246  }
247 
248  Currency currency = parseCurrency(bondData_.currency());
249  boost::shared_ptr<BondEngineBuilder> bondBuilder = boost::dynamic_pointer_cast<BondEngineBuilder>(builder);
250  QL_REQUIRE(bondBuilder, "No Builder found for Bond: " << id());
251  bond->setPricingEngine(bondBuilder->engine(currency, bondData_.creditCurveId(), bondData_.hasCreditRisk(),
252  bondData_.securityId(), bondData_.referenceCurveId()));
253  instrument_.reset(new VanillaInstrument(bond, mult));
254 
255  npvCurrency_ = bondData_.currency();
256  maturity_ = bond->cashflows().back()->date();
257  notional_ = currentNotional(bond->cashflows());
258  notionalCurrency_ = bondData_.currency();
259 
260  // Add legs (only 1)
261  legs_ = {bond->cashflows()};
262  legCurrencies_ = {npvCurrency_};
263  legPayers_ = {bondData_.isPayer()};
264 
265  DLOG("Bond::build() finished for trade " << id());
266 }

310 BondBuilder::Result VanillaBondBuilder::build(const boost::shared_ptr<EngineFactory>& engineFactory,
311  const boost::shared_ptr<ReferenceDataManager>& referenceData,
312  const std::string& securityId) const {
313  BondData data(securityId, 1.0);
314  data.populateFromBondReferenceData(referenceData);
315  ore::data::Bond bond(Envelope(), data);
316  bond.id() = "VanillaBondBuilder_" + securityId;
317  bond.build(engineFactory);
318 
319  QL_REQUIRE(bond.instrument(), "VanillaBondBuilder: constructed bond is null, this is unexpected");
320  auto qlBond = boost::dynamic_pointer_cast<QuantLib::Bond>(bond.instrument()->qlInstrument());
321 
322  QL_REQUIRE(bond.instrument() && bond.instrument()->qlInstrument(),
323  "VanillaBondBuilder: constructed bond trade does not provide a valid ql instrument, this is unexpected "
324  "(either the instrument wrapper or the ql instrument is null)");
325 
326  Result res;
327  res.bond = qlBond;
328  if (data.isInflationLinked()) {
329  res.inflationFactor = QuantExt::inflationLinkedBondQuoteFactor(qlBond);
330  res.isInflationLinked = true;
331  }
332  res.hasCreditRisk = data.hasCreditRisk() && !data.creditCurveId().empty();
333  res.currency = data.currency();
334  res.creditCurveId = data.creditCurveId();
335  res.securityId = data.securityId();
336  res.creditGroup = data.creditGroup();
337  res.priceQuoteMethod = data.priceQuoteMethod();
338  res.priceQuoteBaseValue = data.priceQuoteBaseValue();
339  return res;
340 }
