42 void BondData::fromXML(XMLNode* node) {
43  XMLUtils::checkNode(node, "BondData");
44  QL_REQUIRE(node, "No BondData Node");
45  issuerId_ = XMLUtils::getChildValue(node, "IssuerId", false);
46  creditCurveId_ = XMLUtils::getChildValue(node, "CreditCurveId", false);
47  securityId_ = XMLUtils::getChildValue(node, "SecurityId", true);
48  referenceCurveId_ = XMLUtils::getChildValue(node, "ReferenceCurveId", false);
49  proxySecurityId_ = XMLUtils::getChildValue(node, "ProxySecurityId", false);
50  incomeCurveId_ = XMLUtils::getChildValue(node, "IncomeCurveId", false);
51  volatilityCurveId_ = XMLUtils::getChildValue(node, "VolatilityCurveId", false);
52  settlementDays_ = XMLUtils::getChildValue(node, "SettlementDays", false);
53  calendar_ = XMLUtils::getChildValue(node, "Calendar", false);
54  issueDate_ = XMLUtils::getChildValue(node, "IssueDate", false);
55  if (auto n = XMLUtils::getChildNode(node, "BondNotional")) {
56  bondNotional_ = parseReal(XMLUtils::getNodeValue(n));
57  } else {
58  bondNotional_ = 1.0;
59  }
60  XMLNode* legNode = XMLUtils::getChildNode(node, "LegData");
61  while (legNode != nullptr) {
62  LegData ld;
63  ld.fromXML(legNode);
64  coupons_.push_back(ld);
65  legNode = XMLUtils::getNextSibling(legNode, "LegData");
66  }
67  hasCreditRisk_ = XMLUtils::getChildValueAsBool(node, "CreditRisk", false, true);
68  initialise();
69 }

71 XMLNode* BondData::toXML(XMLDocument& doc) {
72  XMLNode* bondNode = doc.allocNode("BondData");
73  XMLUtils::addChild(doc, bondNode, "IssuerId", issuerId_);
74  XMLUtils::addChild(doc, bondNode, "CreditCurveId", creditCurveId_);
75  XMLUtils::addChild(doc, bondNode, "SecurityId", securityId_);
76  XMLUtils::addChild(doc, bondNode, "ReferenceCurveId", referenceCurveId_);
77  if (!proxySecurityId_.empty())
78  XMLUtils::addChild(doc, bondNode, "ProxySecurityId", proxySecurityId_);
79  if (!incomeCurveId_.empty())
80  XMLUtils::addChild(doc, bondNode, "IncomeCurveId", incomeCurveId_);
81  if (!volatilityCurveId_.empty())
82  XMLUtils::addChild(doc, bondNode, "VolatilityCurveId", volatilityCurveId_);
83  XMLUtils::addChild(doc, bondNode, "SettlementDays", settlementDays_);
84  XMLUtils::addChild(doc, bondNode, "Calendar", calendar_);
85  XMLUtils::addChild(doc, bondNode, "IssueDate", issueDate_);
86  XMLUtils::addChild(doc, bondNode, "BondNotional", bondNotional_);
87  for (auto& c : coupons_)
88  XMLUtils::appendNode(bondNode, c.toXML(doc));
89  if (!hasCreditRisk_)
90  XMLUtils::addChild(doc, bondNode, "CreditRisk", hasCreditRisk_);
91  return bondNode;
92 }

94 void BondData::initialise() {
95 
96  isPayer_ = false;
97 
98  if (!hasCreditRisk() && !creditCurveId().empty()) {
99  WLOG("BondData: CreditCurveId provided, but CreditRisk set to False, continuing without CreditRisk");
100  creditCurveId_ = "";
101  }
102 
103  if (!zeroBond()) {
104 
105  // fill currency, if not directly given (which is only the case for zero bonds)
106 
107  for (Size i = 0; i < coupons().size(); ++i) {
108  if (i == 0)
109  currency_ = coupons()[i].currency();
110  else {
111  QL_REQUIRE(currency_ == coupons()[i].currency(),
112  "bond leg #" << i << " currency (" << coupons()[i].currency()
113  << ") not equal to leg #0 currency (" << coupons()[0].currency());
114  }
115  }
116 
117  // fill isPayer, FIXME zero bonds are always long
118 
119  for (Size i = 0; i < coupons().size(); ++i) {
120  if (i == 0)
121  isPayer_ = coupons()[i].isPayer();
122  else {
123  QL_REQUIRE(isPayer_ == coupons()[i].isPayer(),
124  "bond leg #" << i << " isPayer (" << std::boolalpha << coupons()[i].isPayer()
125  << ") not equal to leg #0 isPayer (" << coupons()[0].isPayer());
126  }
127  }
128  }
129 }

166 void Bond::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
167  DLOG("Bond::build() called for trade " << id());
168 
169  const boost::shared_ptr<Market> market = engineFactory->market();
170  boost::shared_ptr<EngineBuilder> builder = engineFactory->builder("Bond");
171  QL_REQUIRE(builder, "Bond::build(): internal error, builder is null");
172 
173  bondData_.populateFromBondReferenceData(engineFactory->referenceData());
174 
175  Date issueDate = parseDate(bondData_.issueDate());
176  Calendar calendar = parseCalendar(bondData_.calendar());
177  QL_REQUIRE(!bondData_.settlementDays().empty(),
178  "no bond settlement days given, if reference data is used, check if securityId '"
179  << bondData_.securityId() << "' is present and of type Bond.");
180  Natural settlementDays = parseInteger(bondData_.settlementDays());
181  boost::shared_ptr<QuantLib::Bond> bond;
182 
183  std::string openEndDateStr = builder->modelParameter("OpenEndDateReplacement", "", false, "");
184  Date openEndDateReplacement = getOpenEndDateReplacement(openEndDateStr, calendar);
185  Real mult = bondData_.bondNotional() * (bondData_.isPayer() ? -1.0 : 1.0);
186  std::vector<Leg> separateLegs;
187  if (bondData_.zeroBond()) { // Zero coupon bond
188  bond.reset(new QuantLib::ZeroCouponBond(settlementDays, calendar, bondData_.faceAmount(),
189  parseDate(bondData_.maturityDate())));
190  } else { // Coupon bond
191  for (Size i = 0; i < bondData_.coupons().size(); ++i) {
192  Leg leg;
193  auto configuration = builder->configuration(MarketContext::pricing);
194  auto legBuilder = engineFactory->legBuilder(bondData_.coupons()[i].legType());
195  leg = legBuilder->buildLeg(bondData_.coupons()[i], engineFactory, requiredFixings_, configuration,
196  openEndDateReplacement);
197  separateLegs.push_back(leg);
198  } // for coupons_
199  Leg leg = joinLegs(separateLegs);
200  bond.reset(new QuantLib::Bond(settlementDays, calendar, issueDate, leg));
201  }
202 
203  Currency currency = parseCurrency(bondData_.currency());
204  boost::shared_ptr<BondEngineBuilder> bondBuilder = boost::dynamic_pointer_cast<BondEngineBuilder>(builder);
205  QL_REQUIRE(bondBuilder, "No Builder found for Bond: " << id());
206  bond->setPricingEngine(
207  bondBuilder->engine(currency, bondData_.creditCurveId(), bondData_.securityId(), bondData_.referenceCurveId()));
208  instrument_.reset(new VanillaInstrument(bond, mult));
209 
210  npvCurrency_ = bondData_.currency();
211  maturity_ = bond->cashflows().back()->date();
212  notional_ = currentNotional(bond->cashflows());
213  notionalCurrency_ = bondData_.currency();
214 
215  // Add legs (only 1)
216  legs_ = {bond->cashflows()};
217  legCurrencies_ = {npvCurrency_};
218  legPayers_ = {bondData_.isPayer()};
219 }

259 boost::shared_ptr<QuantLib::Bond>
260 VanillaBondBuilder::build(const boost::shared_ptr<EngineFactory>& engineFactory,
261  const boost::shared_ptr<ReferenceDataManager>& referenceData,
262  const std::string& securityId) const {
263  BondData data(securityId, 1.0);
264  data.populateFromBondReferenceData(referenceData);
265  ore::data::Bond bond(Envelope(), data);
266  bond.id() = "VanillaBondBuilder_" + securityId;
267  bond.build(engineFactory);
268 
269  QL_REQUIRE(bond.instrument(), "VanillaBondBuilder: constructed bond is null, this is unexpected");
270  auto qlBond = boost::dynamic_pointer_cast<QuantLib::Bond>(bond.instrument()->qlInstrument());
271 
272  QL_REQUIRE(bond.instrument() && bond.instrument()->qlInstrument(),
273  "VanillaBondBuilder: constructed bond trade does not provide a valid ql instrument, this is unexpected "
274  "(either the instrument wrapper or the ql instrument is null)");
275 
276  return qlBond;
277 }
