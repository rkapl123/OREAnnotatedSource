42 void FxDigitalBarrierOption::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
43 
44  const boost::shared_ptr<Market> market = engineFactory->market();
45 
46  // Only American supported for now
47  QL_REQUIRE(tradeActions().empty(), "TradeActions not supported for FxOption");
48  QL_REQUIRE(option_.style() == "European", "Option Style unknown: " << option_.style());
49  QL_REQUIRE(option_.exerciseDates().size() == 1, "Invalid number of excercise dates");
50  QL_REQUIRE(strike_ > 0.0 && strike_ != Null<Real>(), "Invalid strike " << strike_);
51 
52  Currency boughtCcy = parseCurrency(foreignCurrency_);
53  Currency soldCcy = parseCurrency(domesticCurrency_);
54  Real level = barrier_.levels()[0].value();
55  Date start = ore::data::parseDate(startDate_);
56  Real rebate = barrier_.rebate();
57  QL_REQUIRE(rebate >= 0, "rebate must be non-negative");
58 
59  QL_REQUIRE(level > 0.0 && level != Null<Real>(), "Invalid level " << level);
60 
61  // Payoff and Barrier Type
62  QL_REQUIRE(barrier_.style().empty() || barrier_.style() == "American", "Only american barrier style suppported");
63  Option::Type type = parseOptionType(option_.callPut());
64  Barrier::Type barrierType = parseBarrierType(barrier_.type());
65 
66  // Handle PayoffCurrency, we might have to flip the trade here
67  Real strike = strike_;
68  if (payoffCurrency_ == "") {
69  DLOG("PayoffCurrency defaulting to " << domesticCurrency_ << " for FxDigitalBarrierOption " << id());
70  } else if (payoffCurrency_ == foreignCurrency_) {
71  // Invert the trade, switch dom and for and flip Put/Call
72  strike = 1.0 / strike;
73  level = 1.0 / level;
74  std::swap(boughtCcy, soldCcy);
75  type = type == Option::Call ? Option::Put : Option::Call;
76  switch (barrierType) {
77  case Barrier::DownIn:
78  barrierType = Barrier::UpIn;
79  break;
80  case Barrier::UpIn:
81  barrierType = Barrier::DownIn;
82  break;
83  case Barrier::DownOut:
84  barrierType = Barrier::UpOut;
85  break;
86  case Barrier::UpOut:
87  barrierType = Barrier::DownOut;
88  break;
89  }
90  } else if (payoffCurrency_ != domesticCurrency_) {
91  QL_FAIL("Invalid Payoff currency (" << payoffCurrency_ << ") for FxDigitalBarrierOption " << foreignCurrency_
92  << domesticCurrency_);
93  }
94  DLOG("Setting up FxDigitalBarrierOption with strike " << strike << " level " << level << " foreign/bought "
95  << boughtCcy << " domestic/sold " << soldCcy);
96  // from this point on it's important not to use domesticCurrency_, foreignCurrency_, strike_, barrier_.level(), etc
97  // rather the local variables (boughtCcy, soldCcy, strike, level, etc) should be used as they may have been flipped.
98 
99  // Create a CashOrNothing payoff for digital options
100  boost::shared_ptr<StrikedTypePayoff> payoff(new CashOrNothingPayoff(type, strike, payoffAmount_));
101 
102  // Exercise
103  // Digital Barrier Options assume an American exercise that pays at expiry
104  Date expiryDate = parseDate(option_.exerciseDates().front());
105  boost::shared_ptr<Exercise> exercise = boost::make_shared<EuropeanExercise>(expiryDate);
106 
107  // QL does not have an FXDigitalBarrierOption, so we add a barrier option here and wrap
108  // it in a composite
109  boost::shared_ptr<Instrument> vanilla = boost::make_shared<VanillaOption>(payoff, exercise);
110  boost::shared_ptr<Instrument> barrier =
111  boost::make_shared<BarrierOption>(barrierType, level, rebate, payoff, exercise);
112 
113  // Check if the barrier has been triggered already
114  Calendar cal = ore::data::parseCalendar(calendar_);
115  boost::shared_ptr<QuantExt::FxIndex> fxIndex;
116  if (!fxIndex_.empty())
117  fxIndex = buildFxIndex(fxIndex_, soldCcy.code(), boughtCcy.code(), engineFactory->market(),
118  engineFactory->configuration(MarketContext::pricing));
119 
120  // set pricing engines
121  // we buy foreign with domestic(=sold ccy).
122  boost::shared_ptr<EngineBuilder> builder = engineFactory->builder(tradeType_);
123  QL_REQUIRE(builder, "No builder found for " << tradeType_);
124  boost::shared_ptr<FxDigitalBarrierOptionEngineBuilder> fxBarrierOptBuilder =
125  boost::dynamic_pointer_cast<FxDigitalBarrierOptionEngineBuilder>(builder);
126  // if an 'in' option is triggered it becomes an FxDigitalOption, so we need an fxDigitalOption pricer
127  builder = engineFactory->builder("FxDigitalOption");
128  QL_REQUIRE(builder, "No builder found for FxDigitalOption");
129  boost::shared_ptr<FxDigitalOptionEngineBuilder> fxOptBuilder =
130  boost::dynamic_pointer_cast<FxDigitalOptionEngineBuilder>(builder);
131 
132  barrier->setPricingEngine(fxBarrierOptBuilder->engine(boughtCcy, soldCcy, expiryDate));
133  vanilla->setPricingEngine(fxOptBuilder->engine(boughtCcy, soldCcy));
134 
135  Position::Type positionType = parsePositionType(option_.longShort());
136  Real bsInd = (positionType == QuantLib::Position::Long ? 1.0 : -1.0);
137 
138  // If premium data is provided
139  // 1) build the fee trade and pass it to the instrument wrapper for pricing
140  // 2) add fee payment as additional trade leg for cash flow reporting
141  std::vector<boost::shared_ptr<Instrument>> additionalInstruments;
142  std::vector<Real> additionalMultipliers;
143  Date lastPremiumDate = addPremiums(additionalInstruments, additionalMultipliers, 1.0, option_.premiumData(), -bsInd,
144  soldCcy, engineFactory, fxOptBuilder->configuration(MarketContext::pricing));
145 
146  Settlement::Type settleType = parseSettlementType(option_.settlement());
147 
148  Handle<Quote> spot = market->fxRate(boughtCcy.code() + soldCcy.code());
149  instrument_ = boost::shared_ptr<InstrumentWrapper>(new SingleBarrierOptionWrapper(
150  barrier, positionType == Position::Long ? true : false, expiryDate,
151  settleType == Settlement::Physical ? true : false, vanilla, barrierType, spot, level, rebate, soldCcy,
152  start, fxIndex, cal, 1, 1, additionalInstruments, additionalMultipliers));
153 
154  npvCurrency_ = soldCcy.code(); // sold is the domestic
155  notional_ = payoffAmount_;
156  notionalCurrency_ = payoffCurrency_ != "" ? payoffCurrency_ : domesticCurrency_; // see logic above
157  maturity_ = std::max(lastPremiumDate, expiryDate);
158 
159  if (start != Date()) {
160  for (Date d = start; d <= expiryDate; d = cal.advance(d, 1 * Days)) {
161  requiredFixings_.addFixingDate(d, fxIndex_, expiryDate);
162  }
163  }
164 
165  additionalData_["payoffAmount"] = payoffAmount_;
166  additionalData_["payoffCurrency"] = payoffCurrency_;
167 }

169 bool checkBarrier(Real spot, Barrier::Type type, Real barrier) {
170  switch (type) {
171  case Barrier::DownIn:
172  case Barrier::DownOut:
173  return spot <= barrier;
174  case Barrier::UpIn:
175  case Barrier::UpOut:
176  return spot >= barrier;
177  default:
178  QL_FAIL("unknown barrier type " << type);
179  }
180 }

182 void FxDigitalBarrierOption::fromXML(XMLNode* node) {
183  Trade::fromXML(node);
184  XMLNode* fxNode = XMLUtils::getChildNode(node, "FxDigitalBarrierOptionData");
185  QL_REQUIRE(fxNode, "No FxDigitalBarrierOptionData Node");
186  option_.fromXML(XMLUtils::getChildNode(fxNode, "OptionData"));
187  barrier_.fromXML(XMLUtils::getChildNode(fxNode, "BarrierData"));
188  startDate_ = XMLUtils::getChildValue(fxNode, "StartDate", false);
189  calendar_ = XMLUtils::getChildValue(fxNode, "Calendar", false);
190  fxIndex_ = XMLUtils::getChildValue(fxNode, "FXIndex", false);
191  strike_ = XMLUtils::getChildValueAsDouble(fxNode, "Strike", true);
192  payoffAmount_ = XMLUtils::getChildValueAsDouble(fxNode, "PayoffAmount", true);
193  payoffCurrency_ = XMLUtils::getChildValue(fxNode, "PayoffCurrency", false); // optional
194  foreignCurrency_ = XMLUtils::getChildValue(fxNode, "ForeignCurrency", true);
195  domesticCurrency_ = XMLUtils::getChildValue(fxNode, "DomesticCurrency", true);
196 }

205  if (startDate_ != "")
206  XMLUtils::addChild(doc, fxNode, "StartDate", startDate_);

208  XMLUtils::addChild(doc, fxNode, "Calendar", calendar_);
209  if (fxIndex_ != "")
210  XMLUtils::addChild(doc, fxNode, "FXIndex", fxIndex_);
211  XMLUtils::addChild(doc, fxNode, "Strike", strike_);
212  XMLUtils::addChild(doc, fxNode, "PayoffAmount", payoffAmount_);
213  if (payoffCurrency_ != "")
214  XMLUtils::addChild(doc, fxNode, "PayoffCurrency", payoffCurrency_);
215  XMLUtils::addChild(doc, fxNode, "ForeignCurrency", foreignCurrency_);
216  XMLUtils::addChild(doc, fxNode, "DomesticCurrency", domesticCurrency_);
217 
218  return node;
219 }
