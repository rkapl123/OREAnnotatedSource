42void FxDigitalBarrierOption::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
43
44 const boost::shared_ptr<Market> market = engineFactory->market();
45
46 // Only American supported for now
47 QL_REQUIRE(tradeActions().empty(), "TradeActions not supported for FxOption");
48 QL_REQUIRE(option_.style() == "European", "Option Style unknown: " << option_.style());
49 QL_REQUIRE(option_.exerciseDates().size() == 1, "Invalid number of excercise dates");
50 QL_REQUIRE(strike_ > 0.0 && strike_ != Null<Real>(), "Invalid strike " << strike_);
51
52 Currency boughtCcy = parseCurrency(foreignCurrency_);
53 Currency soldCcy = parseCurrency(domesticCurrency_);
54 Real level = barrier_.levels()[0].value();
55 Date start = ore::data::parseDate(startDate_);
56 Real rebate = barrier_.rebate();
57 QL_REQUIRE(rebate >= 0, "rebate must be non-negative");
58
59 QL_REQUIRE(level > 0.0 && level != Null<Real>(), "Invalid level " << level);
60
61 // Payoff and Barrier Type
62 QL_REQUIRE(barrier_.style().empty() || barrier_.style() == "American", "Only american barrier style suppported");
63 Option::Type type = parseOptionType(option_.callPut());
64 Barrier::Type barrierType = parseBarrierType(barrier_.type());
65
66 // Handle PayoffCurrency, we might have to flip the trade here
67 Real strike = strike_;
68 bool flipResults = false;
69 if (payoffCurrency_ == "") {
70 DLOG("PayoffCurrency defaulting to " << domesticCurrency_ << " for FxDigitalBarrierOption " << id());
71 } else if (payoffCurrency_ == foreignCurrency_) {
72 // Invert the trade, switch dom and for and flip Put/Call
73 strike = 1.0 / strike;
74 level = 1.0 / level;
75 std::swap(boughtCcy, soldCcy);
76 type = type == Option::Call ? Option::Put : Option::Call;
77 switch (barrierType) {
78 case Barrier::DownIn:
79 barrierType = Barrier::UpIn;
80 break;
81 case Barrier::UpIn:
82 barrierType = Barrier::DownIn;
83 break;
84 case Barrier::DownOut:
85 barrierType = Barrier::UpOut;
86 break;
87 case Barrier::UpOut:
88 barrierType = Barrier::DownOut;
89 break;
90 }
91 flipResults = true;
92 } else if (payoffCurrency_ != domesticCurrency_) {
93 QL_FAIL("Invalid Payoff currency (" << payoffCurrency_ << ") for FxDigitalBarrierOption " << foreignCurrency_
94 << domesticCurrency_);
95 }
96 DLOG("Setting up FxDigitalBarrierOption with strike " << strike << " level " << level << " foreign/bought "
97 << boughtCcy << " domestic/sold " << soldCcy);
98 // from this point on it's important not to use domesticCurrency_, foreignCurrency_, strike_, barrier_.level(), etc
99 // rather the local variables (boughtCcy, soldCcy, strike, level, etc) should be used as they may have been flipped.
100
101 // Create a CashOrNothing payoff for digital options
102 boost::shared_ptr<StrikedTypePayoff> payoff(new CashOrNothingPayoff(type, strike, payoffAmount_));
103
104 // Exercise
105 // Digital Barrier Options assume an American exercise that pays at expiry
106 Date expiryDate = parseDate(option_.exerciseDates().front());
107 boost::shared_ptr<Exercise> exercise = boost::make_shared<EuropeanExercise>(expiryDate);
108
109 // QL does not have an FXDigitalBarrierOption, so we add a barrier option here and wrap
110 // it in a composite
111 boost::shared_ptr<Instrument> vanilla = boost::make_shared<VanillaOption>(payoff, exercise);
112 boost::shared_ptr<Instrument> barrier =
113 boost::make_shared<BarrierOption>(barrierType, level, rebate, payoff, exercise);
114
115 // Check if the barrier has been triggered already
116 Calendar cal = ore::data::parseCalendar(calendar_);
117 boost::shared_ptr<QuantExt::FxIndex> fxIndex;
118 if (!fxIndex_.empty())
119 fxIndex = buildFxIndex(fxIndex_, soldCcy.code(), boughtCcy.code(), engineFactory->market(),
120 engineFactory->configuration(MarketContext::pricing));
121
122 // set pricing engines
123 // we buy foreign with domestic(=sold ccy).
124 boost::shared_ptr<EngineBuilder> builder = engineFactory->builder(tradeType_);
125 QL_REQUIRE(builder, "No builder found for " << tradeType_);
126 boost::shared_ptr<FxDigitalBarrierOptionEngineBuilder> fxBarrierOptBuilder =
127 boost::dynamic_pointer_cast<FxDigitalBarrierOptionEngineBuilder>(builder);
128 // if an 'in' option is triggered it becomes an FxDigitalOption, so we need an fxDigitalOption pricer
129 builder = engineFactory->builder("FxDigitalOption");
130 QL_REQUIRE(builder, "No builder found for FxDigitalOption");
131 boost::shared_ptr<FxDigitalOptionEngineBuilder> fxOptBuilder =
132 boost::dynamic_pointer_cast<FxDigitalOptionEngineBuilder>(builder);
133
134 barrier->setPricingEngine(fxBarrierOptBuilder->engine(boughtCcy, soldCcy, expiryDate));
135 vanilla->setPricingEngine(fxOptBuilder->engine(boughtCcy, soldCcy, flipResults));
136
137 Position::Type positionType = parsePositionType(option_.longShort());
138 Real bsInd = (positionType == QuantLib::Position::Long ? 1.0 : -1.0);
139
140 // If premium data is provided
141 // 1) build the fee trade and pass it to the instrument wrapper for pricing
142 // 2) add fee payment as additional trade leg for cash flow reporting
143 std::vector<boost::shared_ptr<Instrument>> additionalInstruments;
144 std::vector<Real> additionalMultipliers;
145 Date lastPremiumDate = addPremiums(additionalInstruments, additionalMultipliers, 1.0, option_.premiumData(), -bsInd,
146 soldCcy, engineFactory, fxOptBuilder->configuration(MarketContext::pricing));
147
148 Settlement::Type settleType = parseSettlementType(option_.settlement());
149
150 Handle<Quote> spot = market->fxSpot(boughtCcy.code() + soldCcy.code());
151 instrument_ = boost::shared_ptr<InstrumentWrapper>(new SingleBarrierOptionWrapper(
152 barrier, positionType == Position::Long ? true : false, expiryDate,
153 settleType == Settlement::Physical ? true : false, vanilla, barrierType, spot, level, rebate, soldCcy,
154 start, fxIndex, cal, 1, 1, additionalInstruments, additionalMultipliers));
155
156 npvCurrency_ = soldCcy.code(); // sold is the domestic
157 notional_ = payoffAmount_;
158 notionalCurrency_ = payoffCurrency_ != "" ? payoffCurrency_ : domesticCurrency_; // see logic above
159 maturity_ = std::max(lastPremiumDate, expiryDate);
160
161 if (start != Date()) {
162 for (Date d = start; d <= expiryDate; d = cal.advance(d, 1 * Days)) {
163 requiredFixings_.addFixingDate(d, fxIndex_, expiryDate);
164 }
165 }
166
167 additionalData_["payoffAmount"] = payoffAmount_;
168 additionalData_["payoffCurrency"] = payoffCurrency_;
169 additionalData_["effectiveForeignCurrency"] = boughtCcy.code();
170 additionalData_["effectiveDomesticCurrency"] = soldCcy.code();
171
172 // ISDA taxonomy
173 additionalData_["isdaAssetClass"] = string("Foreign Exchange");
174 additionalData_["isdaBaseProduct"] = string("Simple Exotic");
175 additionalData_["isdaSubProduct"] = string("Digital");
176 additionalData_["isdaTransaction"] = string("");
177}

179bool checkBarrier(Real spot, Barrier::Type type, Real barrier) {
180 switch (type) {
181 case Barrier::DownIn:
182 case Barrier::DownOut:
183 return spot <= barrier;
184 case Barrier::UpIn:
185 case Barrier::UpOut:
186 return spot >= barrier;
187 default:
188 QL_FAIL("unknown barrier type " << type);
189 }
190}

192void FxDigitalBarrierOption::fromXML(XMLNode* node) {
193 Trade::fromXML(node);
194 XMLNode* fxNode = XMLUtils::getChildNode(node, "FxDigitalBarrierOptionData");
195 QL_REQUIRE(fxNode, "No FxDigitalBarrierOptionData Node");
196 option_.fromXML(XMLUtils::getChildNode(fxNode, "OptionData"));
197 barrier_.fromXML(XMLUtils::getChildNode(fxNode, "BarrierData"));
198 startDate_ = XMLUtils::getChildValue(fxNode, "StartDate", false);
199 calendar_ = XMLUtils::getChildValue(fxNode, "Calendar", false);
200 fxIndex_ = XMLUtils::getChildValue(fxNode, "FXIndex", false);
201 strike_ = XMLUtils::getChildValueAsDouble(fxNode, "Strike", true);
202 payoffAmount_ = XMLUtils::getChildValueAsDouble(fxNode, "PayoffAmount", true);
203 payoffCurrency_ = XMLUtils::getChildValue(fxNode, "PayoffCurrency", false); // optional
204 foreignCurrency_ = XMLUtils::getChildValue(fxNode, "ForeignCurrency", true);
205 domesticCurrency_ = XMLUtils::getChildValue(fxNode, "DomesticCurrency", true);
206}

208XMLNode* FxDigitalBarrierOption::toXML(XMLDocument& doc) {
209 XMLNode* node = Trade::toXML(doc);
210 XMLNode* fxNode = doc.allocNode("FxDigitalBarrierOptionData");
211 XMLUtils::appendNode(node, fxNode);
212
213 XMLUtils::appendNode(fxNode, option_.toXML(doc));
214 XMLUtils::appendNode(fxNode, barrier_.toXML(doc));
215 if (startDate_ != "")
216 XMLUtils::addChild(doc, fxNode, "StartDate", startDate_);
217 if (calendar_ != "")
218 XMLUtils::addChild(doc, fxNode, "Calendar", calendar_);
219 if (fxIndex_ != "")
220 XMLUtils::addChild(doc, fxNode, "FXIndex", fxIndex_);
221 XMLUtils::addChild(doc, fxNode, "Strike", strike_);
222 XMLUtils::addChild(doc, fxNode, "PayoffAmount", payoffAmount_);
223 if (payoffCurrency_ != "")
224 XMLUtils::addChild(doc, fxNode, "PayoffCurrency", payoffCurrency_);
225 XMLUtils::addChild(doc, fxNode, "ForeignCurrency", foreignCurrency_);
226 XMLUtils::addChild(doc, fxNode, "DomesticCurrency", domesticCurrency_);
227
228 return node;
229}
