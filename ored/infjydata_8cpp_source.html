69 void InfJyData::fromXML(XMLNode* node) {
70 
71  // Check the node is not null and that name is LGM or DodgsonKainth. LGM is for backward compatibility.
72  XMLUtils::checkNode(node, "JarrowYildirim");
73 
74  InflationModelData::fromXML(node);
75 
76  // Get reversion and volatility for the real rate process
77  XMLNode* rrNode = XMLUtils::getChildNode(node, "RealRate");
78  QL_REQUIRE(rrNode, "JarrowYildirim inflation model data should have RealRate node.");
79  realRateReversion_.fromXML(XMLUtils::getChildNode(rrNode, "Reversion"));
80  realRateVolatility_.fromXML(XMLUtils::getChildNode(rrNode, "Volatility"));
81  if (XMLNode* n = XMLUtils::getChildNode(rrNode, "ParameterTransformation")) {
82  reversionTransformation_.fromXML(n);
83  }
84 
85  // Get volatility for the inflation index process
86  XMLNode* idxNode = XMLUtils::getChildNode(node, "Index");
87  QL_REQUIRE(idxNode, "JarrowYildirim inflation model data should have Index node.");
88  indexVolatility_.fromXML(XMLUtils::getChildNode(idxNode, "Volatility"));
89 
90  // Get the calibration configuration
91  XMLNode* ccNode = XMLUtils::getChildNode(node, "CalibrationConfiguration");
92  if (ccNode)
93  calibrationConfiguration_.fromXML(ccNode);
94 }

96 XMLNode* InfJyData::toXML(XMLDocument& doc) {
97 
98  XMLNode* node = doc.allocNode("JarrowYildirim");
99  InflationModelData::append(doc, node);
100 
101  XMLNode* rrNode = doc.allocNode("RealRate");
102  XMLUtils::appendNode(rrNode, realRateReversion_.toXML(doc));
103  XMLUtils::appendNode(rrNode, realRateVolatility_.toXML(doc));
104  XMLUtils::appendNode(rrNode, reversionTransformation_.toXML(doc));
105  XMLUtils::appendNode(node, rrNode);
106 
107  XMLNode* idxNode = doc.allocNode("Index");
108  XMLUtils::appendNode(idxNode, indexVolatility_.toXML(doc));
109  XMLUtils::appendNode(node, idxNode);
110 
111  XMLUtils::appendNode(node, calibrationConfiguration_.toXML(doc));
112 
113  return node;
114 }
