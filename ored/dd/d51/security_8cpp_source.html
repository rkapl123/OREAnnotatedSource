31 Security::Security(const Date& asof, SecuritySpec spec, const Loader& loader, const CurveConfigurations& curveConfigs) {
32 
33  try {
34  const boost::shared_ptr<SecurityConfig>& config = curveConfigs.securityConfig(spec.securityID());
35 
36  // get spread quote, this may be specified but not existent in loader, because it is implied from
37  // a liquid price and added to the loader in a subsequent step
38  string spreadQuote = config->spreadQuote();
39  if (spreadQuote != "" && loader.has(spreadQuote, asof)) {
40  boost::shared_ptr<SecuritySpreadQuote> q =
41  boost::dynamic_pointer_cast<SecuritySpreadQuote>(loader.get(spreadQuote, asof));
42  QL_REQUIRE(q, "Failed to cast " << spreadQuote << " to SecuritySpreadQuote");
43  spread_ = q->quote();
44  }
45 
46  // get recovery quote
47  string recoveryQuote = config->recoveryRatesQuote();
48  if (recoveryQuote != "") {
49  QL_REQUIRE(loader.has(recoveryQuote, asof),
50  "required recovery quote " << recoveryQuote << " not found for " << spec);
51  boost::shared_ptr<RecoveryRateQuote> q =
52  boost::dynamic_pointer_cast<RecoveryRateQuote>(loader.get(recoveryQuote, asof));
53  QL_REQUIRE(q, "Failed to cast " << recoveryQuote << " to RecoveryRateQuote");
54  recoveryRate_ = q->quote();
55  }
56 
57  // get cpr quote
58  string cprQuote = config->cprQuote();
59  if (cprQuote != "") {
60  QL_REQUIRE(loader.has(cprQuote, asof), "required cpr quote " << cprQuote << " not found for " << spec);
61  boost::shared_ptr<CPRQuote> q = boost::dynamic_pointer_cast<CPRQuote>(loader.get(cprQuote, asof));
62  QL_REQUIRE(q, "Failed to cast " << cprQuote << " to CPRQuote");
63  cpr_ = q->quote();
64  }
65 
66  // get price quote
67  string priceQuote = config->priceQuote();
68  if (priceQuote != "") {
69  QL_REQUIRE(loader.has(priceQuote, asof),
70  "required price quote " << priceQuote << " not found for " << spec);
71  boost::shared_ptr<BondPriceQuote> q =
72  boost::dynamic_pointer_cast<BondPriceQuote>(loader.get(priceQuote, asof));
73  QL_REQUIRE(q, "Failed to cast " << priceQuote << " to BondPriceQuote");
74  price_ = q->quote();
75  }
76 
77  if (recoveryRate_.empty())
78  WLOG("No security-specific recovery rate found for " << spec);
79  if (cpr_.empty())
80  WLOG("No security-specific cpr found for " << spec);
81  if (price_.empty())
82  WLOG("No security-specific price found for " << spec);
83  if (spread_.empty())
84  WLOG("No security-specific spread found for " << spec);
85 
86  } catch (std::exception& e) {
87  QL_FAIL("Security building failed for curve " << spec.curveConfigID() << " on date " << io::iso_date(asof)
88  << ": " << e.what());
89  } catch (...) {
90  QL_FAIL("Security building failed: unknown error");
91  }
92 
93  return;
94 }
