51 void CommodityOption::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
52 
53  // Checks
54  QL_REQUIRE(quantity_ > 0, "Commodity option requires a positive quatity");
55  QL_REQUIRE(strike_ > 0, "Commodity option requires a positive strike");
56 
57  // Populate the index_ in case the option is automatic exercise.
58  // Intentionally use null calendar because we will ask for index value on the expiry date without adjustment.
59  const boost::shared_ptr<Market>& market = engineFactory->market();
60  index_ = *market->commodityIndex(assetName_, engineFactory->configuration(MarketContext::pricing));
61  if (!isFuturePrice_ || *isFuturePrice_) {
62 
63  // Assume future price if isFuturePrice_ is not explicitly set or if it is and true.
64 
65  auto index = *market->commodityIndex(assetName_, engineFactory->configuration(MarketContext::pricing));
66 
67  // If we are given an explicit future contract expiry date, use it, otherwise use option's expiry.
68  Date expiryDate;
69  if (futureExpiryDate_ != Date()) {
70  expiryDate = futureExpiryDate_;
71  } else {
72  // Get the expiry date of the option. This is the expiry date of the commodity future index.
73  const vector<string>& expiryDates = option_.exerciseDates();
74  QL_REQUIRE(expiryDates.size() == 1,
75  "Expected exactly one expiry date for CommodityOption but got " << expiryDates.size() << ".");
76  expiryDate = parseDate(expiryDates[0]);
77  }
78 
79  // Clone the index with the relevant expiry date.
80  index_ = index->clone(expiryDate);
81 
82  // Set the VanillaOptionTrade forwardDate_ if the index is a CommodityFuturesIndex - we possibly still have a
83  // CommoditySpotIndex at this point so check. Also, will only work for European exercise.
84  auto et = parseExerciseType(option_.style());
85  if (et == Exercise::European && boost::dynamic_pointer_cast<CommodityFuturesIndex>(index_)) {
86  forwardDate_ = expiryDate;
87  }
88 
89  }
90 
91  VanillaOptionTrade::build(engineFactory);
92 
93  // LOG the volatility if the trade expiry date is in the future.
94  if (expiryDate_ > Settings::instance().evaluationDate()) {
95  DLOG("Implied vol for " << tradeType_ << " on " << assetName_ << " with expiry " << expiryDate_
96  << " and strike " << strike_ << " is "
97  << market->commodityVolatility(assetName_)->blackVol(expiryDate_, strike_));
98  }
99 
100  additionalData_["quantity"] = quantity_;
101  additionalData_["strike"] = strike_;
102  additionalData_["strikeCurrency"] = currency_;
103 }

110 void CommodityOption::fromXML(XMLNode* node) {
111 
112  Trade::fromXML(node);
113 
114  XMLNode* commodityNode = XMLUtils::getChildNode(node, "CommodityOptionData");
115  QL_REQUIRE(commodityNode, "A commodity option needs a 'CommodityOptionData' node");
116 
117  option_.fromXML(XMLUtils::getChildNode(commodityNode, "OptionData"));
118 
119  assetName_ = XMLUtils::getChildValue(commodityNode, "Name", true);
120  currency_ = XMLUtils::getChildValue(commodityNode, "Currency", true);
121  strike_ = XMLUtils::getChildValueAsDouble(commodityNode, "Strike", true);
122  quantity_ = XMLUtils::getChildValueAsDouble(commodityNode, "Quantity", true);
123 
124  isFuturePrice_ = boost::none;
125  if (XMLNode* n = XMLUtils::getChildNode(commodityNode, "IsFuturePrice"))
126  isFuturePrice_ = parseBool(XMLUtils::getNodeValue(n));
127 
128  futureExpiryDate_ = Date();
129  if (XMLNode* n = XMLUtils::getChildNode(commodityNode, "FutureExpiryDate"))
130  futureExpiryDate_ = parseDate(XMLUtils::getNodeValue(n));
131 }

133 XMLNode* CommodityOption::toXML(XMLDocument& doc) {
134 
135  XMLNode* node = Trade::toXML(doc);
136 
137  XMLNode* eqNode = doc.allocNode("CommodityOptionData");
138  XMLUtils::appendNode(node, eqNode);
139 
140  XMLUtils::appendNode(eqNode, option_.toXML(doc));
141 
142  XMLUtils::addChild(doc, eqNode, "Name", assetName_);
143  XMLUtils::addChild(doc, eqNode, "Currency", currency_);
144  XMLUtils::addChild(doc, eqNode, "Strike", strike_);
145  XMLUtils::addChild(doc, eqNode, "Quantity", quantity_);
146 
147  if (isFuturePrice_)
148  XMLUtils::addChild(doc, eqNode, "IsFuturePrice", *isFuturePrice_);
149 
150  if (futureExpiryDate_ != Date())
151  XMLUtils::addChild(doc, eqNode, "FutureExpiryDate", to_string(futureExpiryDate_));
152 
153  return node;
154 }
