127 void TodaysMarketParameters::fromXML(XMLNode* node) {
128 
129  // add default configuration if we do not have one (may be overwritten below)
130  if (!hasConfiguration(Market::defaultConfiguration))
131  addConfiguration(Market::defaultConfiguration, MarketConfiguration());
132 
133  // fill data from XML
134  XMLUtils::checkNode(node, "TodaysMarket");
135  XMLNode* n = XMLUtils::getChildNode(node);
136  while (n) {
137  if (XMLUtils::getNodeName(n) == "Configuration") {
138  MarketConfiguration tmp;
139  for (Size i = 0; i < marketObjectData.size(); ++i) {
140  tmp.setId(marketObjectData[i].obj,
141  XMLUtils::getChildValue(n, marketObjectData[i].xmlName + "Id", false));
142  addConfiguration(XMLUtils::getAttribute(n, "id"), tmp);
143  }
144  } else {
145  Size i = 0;
146  for (; i < marketObjectData.size(); ++i) {
147  if (XMLUtils::getNodeName(n) == marketObjectData[i].xmlName) {
148  string id = XMLUtils::getAttribute(n, "id");
149  if (id == "")
150  id = Market::defaultConfiguration;
151  // The XML schema for swap indices is different ...
152  if (marketObjectData[i].obj == MarketObject::SwapIndexCurve) {
153  vector<XMLNode*> nodes = XMLUtils::getChildrenNodes(n, marketObjectData[i].xmlSingleName.first);
154  map<string, string> swapIndices;
155  for (XMLNode* xn : nodes) {
156  string name = XMLUtils::getAttribute(xn, marketObjectData[i].xmlSingleName.second);
157  QL_REQUIRE(name != "", "no name given for SwapIndex");
158  QL_REQUIRE(swapIndices.find(name) == swapIndices.end(),
159  "Duplicate SwapIndex found for " << name);
160  string disc = XMLUtils::getChildValue(xn, "Discounting", true);
161  swapIndices[name] = disc; //.emplace(name, { ibor, disc }); won't work?
162  }
163  addMarketObject(MarketObject::SwapIndexCurve, id, swapIndices);
164 
165  } else {
166  auto mp =
167  XMLUtils::getChildrenAttributesAndValues(n, marketObjectData[i].xmlSingleName.first,
168  marketObjectData[i].xmlSingleName.second, false);
169  Size nc = XMLUtils::getChildrenNodes(n, "").size();
170  QL_REQUIRE(mp.size() == nc, "could not recognise " << (nc - mp.size()) << " sub nodes under "
171  << marketObjectData[i].xmlName);
172  addMarketObject(marketObjectData[i].obj, id, mp);
173  }
174  break;
175  }
176  }
177  QL_REQUIRE(i < marketObjectData.size(),
178  "TodaysMarketParameters::fromXML(): node not recognized: " << XMLUtils::getNodeName(n));
179  }
180  n = XMLUtils::getNextSibling(n);
181  } // while(n)
182 }

184 XMLNode* TodaysMarketParameters::toXML(XMLDocument& doc) {
185 
186  XMLNode* todaysMarketNode = doc.allocNode("TodaysMarket");
187 
188  // configurations
189  if (configurations_.size() > 0) {
190  for (auto iterator = configurations_.begin(); iterator != configurations_.end(); iterator++) {
191  XMLNode* configurationsNode = XMLUtils::addChild(doc, todaysMarketNode, "Configuration");
192  XMLUtils::addAttribute(doc, configurationsNode, "id", iterator->first.c_str());
193  for (Size i = 0; i < marketObjectData.size(); ++i) {
194  XMLUtils::addChild(doc, configurationsNode, marketObjectData[i].xmlName + "Id",
195  iterator->second(marketObjectData[i].obj)); // Added the "Id" for schema test
196  }
197  }
198  }
199 
200  for (Size i = 0; i < marketObjectData.size(); ++i) {
201  if (marketObjects_.find(marketObjectData[i].obj) != marketObjects_.end()) {
202  auto mapping = marketObjects_.at(marketObjectData[i].obj);
203  for (auto mappingSetIterator = mapping.begin(); mappingSetIterator != mapping.end(); mappingSetIterator++) {
204 
205  XMLNode* node = XMLUtils::addChild(doc, todaysMarketNode, marketObjectData[i].xmlName);
206  XMLUtils::addAttribute(doc, node, "id", mappingSetIterator->first.c_str());
207 
208  for (auto singleMappingIterator = mappingSetIterator->second.begin();
209  singleMappingIterator != mappingSetIterator->second.end(); singleMappingIterator++) {
210  // Again, swap indices are different...
211  if (marketObjectData[i].obj == MarketObject::SwapIndexCurve) {
212  XMLNode* swapIndexNode = XMLUtils::addChild(doc, node, marketObjectData[i].xmlSingleName.first);
213  XMLUtils::addAttribute(doc, swapIndexNode, marketObjectData[i].xmlSingleName.second,
214  singleMappingIterator->first.c_str());
215  XMLUtils::addChild(doc, swapIndexNode, "Discounting",
216  (string)singleMappingIterator->second.c_str());
217  } else {
218  XMLNode* singleMappingNode =
219  doc.allocNode(marketObjectData[i].xmlSingleName.first, singleMappingIterator->second);
220  XMLUtils::appendNode(node, singleMappingNode);
221  XMLUtils::addAttribute(doc, singleMappingNode, marketObjectData[i].xmlSingleName.second,
222  singleMappingIterator->first);
223  }
224  }
225  }
226  }
227  }
228  return todaysMarketNode;
229 }

255 void TodaysMarketParameters::addMarketObject(const MarketObject o, const string& id,
256  const map<string, string>& assignments) {
257 
258  // check that we do not have an inconcsistent mapping within one market object
259  auto mo = marketObjects_.find(o);
260  if (mo != marketObjects_.end()) {
261  auto mp = mo->second.find(id);
262  if (mp != mo->second.end()) {
263  for (auto const& m : mp->second) {
264  auto a = assignments.find(m.first);
265  if (a != assignments.end()) {
266  QL_REQUIRE(m.first != a->first || m.second == a->second,
267  "TodaysMarketParameters, inconsistent mapping is added for market object type "
268  << o << ", id " << id << ": " << a->first << " " << a->second << ", but have "
269  << m.first << " " << m.second << " already.");
270  }
271  }
272  }
273  }
274 
275  // check that we do not have an overlap of names for yield curves and index curves
276  if (o == MarketObject::YieldCurve || o == MarketObject::IndexCurve) {
277  auto mo =
278  marketObjects_.find(o == MarketObject::YieldCurve ? MarketObject::IndexCurve : MarketObject::YieldCurve);
279  if (mo != marketObjects_.end()) {
280  auto mp = mo->second.find(id);
281  if (mp != mo->second.end()) {
282  for (auto const& m : mp->second) {
283  auto a = assignments.find(m.first);
284  QL_REQUIRE(a == assignments.end(),
285  "TodaysMarketParameters, overlap between YieldCurve and IndexCurve names, try to add "
286  "mapping for market object type "
287  << o << ", id " << id << ": " << a->first << " " << a->second << ", but have "
288  << m.first << " " << m.second << " already in other market object's mappnig");
289  }
290  }
291  }
292  }
293 
294  // add the mapping
295  marketObjects_[o][id].insert(assignments.begin(), assignments.end());
296  for (auto s : assignments)
297  DLOG("TodaysMarketParameters, add market objects of type " << o << ": " << id << " " << s.first << " "
298  << s.second);
299 }
