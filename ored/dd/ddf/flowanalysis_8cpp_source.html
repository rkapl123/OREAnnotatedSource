76 void AnalysisGenerator::reset() {
77  flowAnalysis_.clear();
78 
79  vector<string> headings(numberOfColumns_);
80  headings[PAYMENT_DATE] = string("Payment Date");
81  headings[ACCRUAL_START_DATE] = string("Accrual Start Date");

150 vector<vector<string>> flowAnalysis(const QuantLib::Leg& leg) {
151  AnalysisGenerator generator;
152  for (QuantLib::Size i = 0; i < leg.size(); ++i)
153  leg[i]->accept(generator);
154  return generator.analysis();
155 }
