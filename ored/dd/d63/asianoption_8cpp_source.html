23  void AsianOptionTrade::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
24  QuantLib::Date today = engineFactory->market()->asofDate();
25  Currency ccy = parseCurrency(currency_);
26 
27  QL_REQUIRE(tradeActions().empty(), "TradeActions not supported for AsianOption");
28 
29  // Payoff
30  Option::Type type = parseOptionType(option_.callPut());
31  boost::shared_ptr<StrikedTypePayoff> payoff(new PlainVanillaPayoff(type, strike_));
32 
33  QuantLib::Exercise::Type exerciseType = parseExerciseType(option_.style());
34  QL_REQUIRE(option_.exerciseDates().size() == 1, "Invalid number of excercise dates");
35  expiryDate_ = parseDate(option_.exerciseDates().front());
36 
37  // Set the maturity date equal to the expiry date. It may get updated below if option is cash settled with
38  // payment after expiry.
39  maturity_ = expiryDate_;
40 
41  // Exercise
42  boost::shared_ptr<Exercise> exercise;
43  switch (exerciseType) {
44  case QuantLib::Exercise::Type::European: {
45  exercise = boost::make_shared<EuropeanExercise>(expiryDate_);
46  break;
47  }
48  case QuantLib::Exercise::Type::American: {
49  ELOG("The implemented Asian pricing engines do not support American options.");
50  QL_FAIL("The implemented Asian pricing engines do not support American options.");
51  //exercise = boost::make_shared<AmericanExercise>(expiryDate_, option_.payoffAtExpiry());
52  break;
53  }
54  default:
55  QL_FAIL("Option style " << option_.style() << " is not supported");
56  }
57 
58  // Create the instrument and the populate the name for the engine builder.
59  boost::shared_ptr<Instrument> asian;
60  string tradeTypeBuilder = tradeType_;
61  tradeTypeBuilder += (exerciseType == Exercise::European) ? "" : "American";
62  Settlement::Type settlementType = parseSettlementType(option_.settlement());
63 
64  bool isEuropDeferredCS = false;
65  if (exerciseType == Exercise::European && settlementType == Settlement::Cash) {
66  // We have a European cash-settled option
67 
68  // Get payment data
69  const boost::optional<OptionPaymentData>& opd = option_.paymentData();
70  Date paymentDate = expiryDate_;
71  if (opd) {
72  if (opd->rulesBased()) {
73  const Calendar& cal = opd->calendar();
74  QL_REQUIRE(cal != Calendar(), "Need non-empty calendar for rules based payment date.");
75  paymentDate = cal.advance(expiryDate_, opd->lag(), Days, opd->convention());
76  } else {
77  const vector<Date>& dates = opd->dates();
78  QL_REQUIRE(dates.size() == 1, "Need exactly one payment date for cash settled European option.");
79  paymentDate = dates[0];
80  }
81  QL_REQUIRE(paymentDate >= expiryDate_, "Payment date must be greater than or equal to expiry date.");
82  }
83 
84  if (paymentDate > expiryDate_) {
85  isEuropDeferredCS = true;
86  tradeTypeBuilder += "EuropeanCS";
87  // TODO: If support is needed/added, add logic from ored/portfolio/vanillaoption.cpp for updating of maturity date.
88  ELOG("Asian options do not support deferred cash settlement payments, "
89  << "i.e. where payment date > maturity date.");
90  QL_FAIL("Asian options do not support deferred cash-settlement payments, i.e. where payment date > "
91  "maturity date.");
92  }
93  }
94 
95  // Create builder in advance to determine if continuous or discrete asian option.
96  Average::Type averageType = asianData_.averageType();
97  OptionAsianData::AsianType asianType = asianData_.asianType();
98  tradeTypeBuilder += to_string(averageType); // Add Arithmetic/Geometric
99  tradeTypeBuilder += to_string(asianType); // Add Price/Strike
100  boost::shared_ptr<EngineBuilder> builder = engineFactory->builder(tradeTypeBuilder);
101  QL_REQUIRE(builder, "No builder found for " << tradeTypeBuilder);
102 
103  boost::shared_ptr<AsianOptionEngineBuilder> asianOptionBuilder =
104  boost::dynamic_pointer_cast<AsianOptionEngineBuilder>(builder);
105  QL_REQUIRE(asianOptionBuilder, "engine builder is not an AsianOption engine builder" << tradeTypeBuilder);
106 
107  std::string processType = asianOptionBuilder->processType();
108  QL_REQUIRE(processType != "", "ProcessType must be configured, this is unexpected");
109 
110  if (!isEuropDeferredCS) {
111  // If the option is not a European deferred CS contract, i.e. the standard European/American physical/cash-settled ones.
112  if (processType == "Discrete") {
113  Real runningAccumulator = 0;
114  if (averageType == Average::Type::Geometric) {
115  runningAccumulator = 1;
116  }
117  Size pastFixings = 0;
118 
119  Schedule observationSchedule;
120  if (observationDates_.hasData())
121  observationSchedule = makeSchedule(observationDates_);
122  std::vector<QuantLib::Date> observationDates = observationSchedule.dates();
123 
124  // Sort for the engine's sake. Not needed - instrument also sorts...?
125  std::sort(observationDates.begin(), observationDates.end());
126 
127  QL_REQUIRE(index_, "Asian option trade " << id() << " needs a valid index for historical fixings.");
128  // If index name has not been populated, use logic here to populate it from the index object.
129  string indexName = indexName_;
130  if (indexName.empty()) {
131  indexName = index_->name();
132  if (assetClassUnderlying_ == AssetClass::EQ)
133  indexName = "EQ-" + indexName;
134  }
135  for (QuantLib::Date observationDate : observationDates) {
136  // TODO: Verify. Should today be read too? a enforcesTodaysHistoricFixings() be used?
137  if (observationDate < today ||
138  (observationDate == today && Settings::instance().enforcesTodaysHistoricFixings())) {
139  requiredFixings_.addFixingDate(observationDate, indexName);
140  Real fixingValue = index_->fixing(observationDate);
141  if (averageType == Average::Type::Geometric) {
142  runningAccumulator *= fixingValue;
143  } else if (averageType == Average::Type::Arithmetic) {
144  runningAccumulator += fixingValue;
145  }
146  ++pastFixings;
147  }
148  }
149 
150  asian = boost::make_shared<QuantLib::DiscreteAveragingAsianOption>(
151  averageType, runningAccumulator, pastFixings, observationDates, payoff, exercise);
152  } else if (processType == "Continuous") {
153  asian = boost::make_shared<QuantLib::ContinuousAveragingAsianOption>(averageType, payoff, exercise);
154  } else {
155  QL_FAIL("unexpected ProcessType, valid options are Discrete/Continuous");
156  }
157  } else {
158  ELOG("Asian options do not support deferred cash settlement payments, "
159  << "i.e. where payment date > maturity date.");
160  QL_FAIL("Asian options do not support deferred cash-settlement payments, i.e. where payment date > "
161  << "maturity date.");
162  }
163 
164 
165  // Only try to set an engine on the option instrument if it is not expired. This avoids errors in
166  // engine builders that rely on the expiry date being in the future.
167  string configuration = Market::defaultConfiguration;
168  if (!asian->isExpired()) {
169  asian->setPricingEngine(
170  asianOptionBuilder->engine(assetName_, ccy, expiryDate_));
171 
172  configuration = asianOptionBuilder->configuration(MarketContext::pricing);
173  } else {
174  DLOG("No engine attached for option on trade " << id() << " with expiry date " << io::iso_date(expiryDate_)
175  << " because it is expired.");
176  }
177 
178  Position::Type positionType = parsePositionType(option_.longShort());
179  Real bsInd = (positionType == QuantLib::Position::Long ? 1.0 : -1.0);
180  Real mult = quantity_ * bsInd;
181 
182  // If premium data is provided
183  // 1) build the fee trade and pass it to the instrument wrapper for pricing
184  // 2) add fee payment as additional trade leg for cash flow reporting
185  std::vector<boost::shared_ptr<Instrument>> additionalInstruments;
186  std::vector<Real> additionalMultipliers;
187  addPremiums(additionalInstruments, additionalMultipliers, 1.0, option_.premiumData(), 1.0, ccy,
188  engineFactory, configuration);
189 
190  // TODO: Use something like EuropeanOptionWrapper instead? After all, an asian option is not
191  // a vanilla instrument. However, the available European-/AmericanOptionWrapper is not directly
192  // applicable as it calls for both an instrument and an underlying instrument.
193  // Given how the asians are priced, though, we still manage fine using the VanillaInstrument wrapper
194  // for the time being.
195  instrument_ = boost::shared_ptr<InstrumentWrapper>(
196  new VanillaInstrument(asian, mult, additionalInstruments, additionalMultipliers));
197  //instrument_ = boost::shared_ptr<InstrumentWrapper>(
198  // boost::make_shared<EuropeanOptionWrapper>(...));
199 
200  npvCurrency_ = currency_;
201 
202  // Notional - we really need todays spot to get the correct notional.
203  // But rather than having it move around we use strike * quantity
204  notional_ = strike_ * quantity_;
205  notionalCurrency_ = currency_;
206  }
