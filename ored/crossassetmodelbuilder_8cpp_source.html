78CrossAssetModelBuilder::CrossAssetModelBuilder(
79 const boost::shared_ptr<ore::data::Market>& market, const boost::shared_ptr<CrossAssetModelData>& config,
80 const std::string& configurationLgmCalibration, const std::string& configurationFxCalibration,
81 const std::string& configurationEqCalibration, const std::string& configurationInfCalibration,
82 const std::string& configurationCrCalibration, const std::string& configurationFinalModel, const bool dontCalibrate,
83 const bool continueOnError, const std::string& referenceCalibrationGrid, const SalvagingAlgorithm::Type salvaging)
84 : market_(market), config_(config), configurationLgmCalibration_(configurationLgmCalibration),
85 configurationFxCalibration_(configurationFxCalibration), configurationEqCalibration_(configurationEqCalibration),
86 configurationInfCalibration_(configurationInfCalibration),
87 configurationCrCalibration_(configurationCrCalibration),
88 configurationComCalibration_(Market::defaultConfiguration), configurationFinalModel_(configurationFinalModel),
89 dontCalibrate_(dontCalibrate), continueOnError_(continueOnError),
90 referenceCalibrationGrid_(referenceCalibrationGrid), salvaging_(salvaging),
91 optimizationMethod_(boost::shared_ptr<OptimizationMethod>(new LevenbergMarquardt(1E-8, 1E-8, 1E-8))),
92 endCriteria_(EndCriteria(1000, 500, 1E-8, 1E-8, 1E-8)) {
93 buildModel();
94 registerWithSubBuilders();
95 // register market observer with correlations
96 marketObserver_ = boost::make_shared<MarketObserver>();
97 for (auto const& c : config->correlations())
98 marketObserver_->addObservable(c.second);
99 // reset market observer's updated flag
100 marketObserver_->hasUpdated(true);
101}

129void CrossAssetModelBuilder::unregisterWithSubBuilders() {
130 for (auto okv : subBuilders_)
131 for (auto ikv : okv.second)
132 unregisterWith(ikv.second);
133}
134
135void CrossAssetModelBuilder::registerWithSubBuilders() {
136 for (auto okv : subBuilders_)
137 for (auto ikv : okv.second)
138 registerWith(ikv.second);
139}
140
141bool CrossAssetModelBuilder::requiresRecalibration() const {
142 for (auto okv : subBuilders_)
143 for (auto ikv : okv.second)
144 if (ikv.second->requiresRecalibration())
145 return true;
146
147 return marketObserver_->hasUpdated(false);
148}
149
150void CrossAssetModelBuilder::performCalculations() const {
151 // if any of the sub models requires a recalibration, we rebuilt the model
152 // TODO we could do this more selectively
153 if (!dontCalibrate_ && requiresRecalibration()) {
154 // reset market observer update flag
155 marketObserver_->hasUpdated(true);
156 // the cast is a bit ugly, but we pretty much know what we are doing here
157 const_cast<CrossAssetModelBuilder*>(this)->unregisterWithSubBuilders();
158 buildModel();
159 const_cast<CrossAssetModelBuilder*>(this)->registerWithSubBuilders();
160 }
161}

163void CrossAssetModelBuilder::buildModel() const {
164
165 QL_REQUIRE(market_ != NULL, "CrossAssetModelBuilder: no market given");
166 LOG("Start building CrossAssetModel");
167 DLOG("configurations: LgmCalibration "
168 << configurationLgmCalibration_ << ", FxCalibration " << configurationFxCalibration_ << ", EqCalibration "
169 << configurationEqCalibration_ << ", InfCalibration " << configurationInfCalibration_ << ", CrCalibration"
170 << configurationCrCalibration_ << ", ComCalibration" << configurationComCalibration_
171 << ", FinalModel " << configurationFinalModel_);
172 if (dontCalibrate_) {
173 DLOG("Calibration of the model is disabled.");
174 }
175
176 QL_REQUIRE(config_->irConfigs().size() > 0, "missing IR configurations");
177 QL_REQUIRE(config_->irConfigs().size() == config_->fxConfigs().size() + 1,
178 "FX configuration size " << config_->fxConfigs().size() << " inconsistent with IR configuration size "
179 << config_->irConfigs().size());
180
181 swaptionBaskets_.resize(config_->irConfigs().size());
182 optionExpiries_.resize(config_->irConfigs().size());
183 swaptionMaturities_.resize(config_->irConfigs().size());
184 swaptionCalibrationErrors_.resize(config_->irConfigs().size());
185 fxOptionBaskets_.resize(config_->fxConfigs().size());
186 fxOptionExpiries_.resize(config_->fxConfigs().size());
187 fxOptionCalibrationErrors_.resize(config_->fxConfigs().size());
188 eqOptionBaskets_.resize(config_->eqConfigs().size());
189 eqOptionExpiries_.resize(config_->eqConfigs().size());
190 eqOptionCalibrationErrors_.resize(config_->eqConfigs().size());
191 inflationCalibrationErrors_.resize(config_->infConfigs().size());
192 comOptionBaskets_.resize(config_->comConfigs().size());
193 comOptionExpiries_.resize(config_->comConfigs().size());
194 comOptionCalibrationErrors_.resize(config_->comConfigs().size());
195
196 subBuilders_.clear();
197
198 // Store information on the number of factors for each process. This is used when requesting a correlation matrix
199 // from the CorrelationMatrixBuilder below.
200 CorrelationMatrixBuilder::ProcessInfo processInfo;
201
202 // Set the measure
203 IrModel::Measure measure = IrModel::Measure::LGM;
204 if (config_->measure() == "BA") {
205 measure = IrModel::Measure::BA;
206 DLOG("Setting measure to BA");
207 } else if (config_->measure() == "LGM") {
208 measure = IrModel::Measure::LGM;
209 DLOG("Setting measure to LGM");
210 } else if (config_->measure() == "") {
211 DLOG("Defaulting to LGM measure");
212 } else {
213 QL_FAIL("Measure " << config_->measure() << " not recognized");
214 }
215
216 /*******************************************************
217 * Build the IR parametrizations and calibration baskets
218 */
219 std::vector<boost::shared_ptr<QuantExt::Parametrization>> irParametrizations;
220 std::vector<RelinkableHandle<YieldTermStructure>> irDiscountCurves;
221 std::vector<std::string> currencies, regions, crNames, eqNames, infIndices, comNames;
222 std::vector<boost::shared_ptr<LgmBuilder>> lgmBuilder;
223 std::vector<boost::shared_ptr<HwBuilder>> hwBuilder;
224 std::vector<boost::shared_ptr<CommoditySchwartzModelBuilder>> csBuilder;
225
226 for (Size i = 0; i < config_->irConfigs().size(); i++) {
227 auto irConfig = config_->irConfigs()[i];
228 DLOG("IR Parametrization " << i << " qualifier " << irConfig->qualifier());
229
230 if (auto ir = boost::dynamic_pointer_cast<IrLgmData>(irConfig)) {
231
232 auto builder = boost::make_shared<LgmBuilder>(market_, ir, configurationLgmCalibration_,
233 config_->bootstrapTolerance(),
234 continueOnError_, referenceCalibrationGrid_);
235 if (dontCalibrate_)
236 builder->freeze();
237 lgmBuilder.push_back(builder);
238 auto parametrization = builder->parametrization();
239 swaptionBaskets_[i] = builder->swaptionBasket();
240 QL_REQUIRE(std::find(currencies.begin(), currencies.end(), parametrization->currency().code()) ==
241 currencies.end(),
242 "Duplicate IR parameterization for currency "
243 << parametrization->currency().code()
244 << " - are there maybe two indices with the same currency in CrossAssetModelData?");
245 currencies.push_back(parametrization->currency().code());
246 irParametrizations.push_back(parametrization);
247 irDiscountCurves.push_back(builder->discountCurve());
248 subBuilders_[CrossAssetModel::AssetType::IR][i] = builder;
249 processInfo[CrossAssetModel::AssetType::IR].emplace_back(ir->ccy(), 1);
250 }
251 else if(auto ir = boost::dynamic_pointer_cast<HwModelData>(irConfig)) {
252 bool evaluateBankAccount = true; // updated in cross asset model for non-base ccys
253 bool setCalibrationInfo = false;
254 HwModel::Discretization discr = HwModel::Discretization::Euler;
255 auto builder = boost::make_shared<HwBuilder>(
256 market_, ir, measure, discr, evaluateBankAccount, configurationLgmCalibration_,
257 config_->bootstrapTolerance(), continueOnError_, referenceCalibrationGrid_, setCalibrationInfo);
258 if (dontCalibrate_)
259 builder->freeze();
260 hwBuilder.push_back(builder);
261 auto parametrization = builder->parametrization();
262 swaptionBaskets_[i] = builder->swaptionBasket();
263 QL_REQUIRE(std::find(currencies.begin(), currencies.end(), parametrization->currency().code()) ==
264 currencies.end(),
265 "Duplicate IR parameterization for currency "
266 << parametrization->currency().code()
267 << " - are there maybe two indices with the same currency in CrossAssetModelData?");
268 currencies.push_back(parametrization->currency().code());
269 irParametrizations.push_back(parametrization);
270 irDiscountCurves.push_back(builder->discountCurve());
271 subBuilders_[CrossAssetModel::AssetType::IR][i] = builder;
272 processInfo[CrossAssetModel::AssetType::IR].emplace_back(ir->ccy(), parametrization->m());
273 }
274 }
275
276 QL_REQUIRE(irParametrizations.size() > 0, "missing IR parametrizations");
277
278 QuantLib::Currency domesticCcy = irParametrizations[0]->currency();
279
280 /*******************************************************
281 * Build the FX parametrizations and calibration baskets
282 */
283 std::vector<boost::shared_ptr<QuantExt::FxBsParametrization>> fxParametrizations;
284 for (Size i = 0; i < config_->fxConfigs().size(); i++) {
285 DLOG("FX Parametrization " << i);
286 boost::shared_ptr<FxBsData> fx = config_->fxConfigs()[i];
287 QuantLib::Currency ccy = ore::data::parseCurrency(fx->foreignCcy());
288 QuantLib::Currency domCcy = ore::data::parseCurrency(fx->domesticCcy());
289
290 QL_REQUIRE(ccy.code() == irParametrizations[i + 1]->currency().code(),
291 "FX parametrization currency[" << i << "]=" << ccy << " does not match IR currency[" << i + 1
292 << "]=" << irParametrizations[i + 1]->currency().code());
293
294 QL_REQUIRE(domCcy == domesticCcy, "FX parametrization [" << i << "]=" << ccy << "/" << domCcy
295 << " does not match domestic ccy " << domesticCcy);
296
297 boost::shared_ptr<FxBsBuilder> builder =
298 boost::make_shared<FxBsBuilder>(market_, fx, configurationFxCalibration_, referenceCalibrationGrid_);
299 boost::shared_ptr<QuantExt::FxBsParametrization> parametrization = builder->parametrization();
300
301 fxOptionBaskets_[i] = builder->optionBasket();
302 fxParametrizations.push_back(parametrization);
303 subBuilders_[CrossAssetModel::AssetType::FX][i] = builder;
304 processInfo[CrossAssetModel::AssetType::FX].emplace_back(ccy.code() + domCcy.code(), 1);
305 }
306
307 /*******************************************************
308 * Build the EQ parametrizations and calibration baskets
309 */
310 std::vector<boost::shared_ptr<QuantExt::EqBsParametrization>> eqParametrizations;
311 for (Size i = 0; i < config_->eqConfigs().size(); i++) {
312 DLOG("EQ Parametrization " << i);
313 boost::shared_ptr<EqBsData> eq = config_->eqConfigs()[i];
314 string eqName = eq->eqName();
315 QuantLib::Currency eqCcy = ore::data::parseCurrency(eq->currency());
316 QL_REQUIRE(std::find(currencies.begin(), currencies.end(), eqCcy.code()) != currencies.end(),
317 "Currency (" << eqCcy << ") for equity " << eqName << " not covered by CrossAssetModelData");
318 boost::shared_ptr<EqBsBuilder> builder = boost::make_shared<EqBsBuilder>(
319 market_, eq, domesticCcy, configurationEqCalibration_, referenceCalibrationGrid_);
320 boost::shared_ptr<QuantExt::EqBsParametrization> parametrization = builder->parametrization();
321 eqOptionBaskets_[i] = builder->optionBasket();
322 eqParametrizations.push_back(parametrization);
323 eqNames.push_back(eqName);
324 subBuilders_[CrossAssetModel::AssetType::EQ][i] = builder;
325 processInfo[CrossAssetModel::AssetType::EQ].emplace_back(eqName, 1);
326 }
327
328 // Build the INF parametrizations and calibration baskets
329 vector<boost::shared_ptr<Parametrization>> infParameterizations;
330 for (Size i = 0; i < config_->infConfigs().size(); i++) {
331 boost::shared_ptr<InflationModelData> imData = config_->infConfigs()[i];
332 DLOG("Inflation parameterisation (" << i << ") for index " << imData->index());
333 if (auto dkData = boost::dynamic_pointer_cast<InfDkData>(imData)) {
334 boost::shared_ptr<InfDkBuilder> builder = boost::make_shared<InfDkBuilder>(
335 market_, dkData, configurationInfCalibration_, referenceCalibrationGrid_, dontCalibrate_);
336 if (dontCalibrate_)
337 builder->freeze();
338 infParameterizations.push_back(builder->parametrization());
339 subBuilders_[CrossAssetModel::AssetType::INF][i] = builder;
340 processInfo[CrossAssetModel::AssetType::INF].emplace_back(dkData->index(), 1);
341 } else if (auto jyData = boost::dynamic_pointer_cast<InfJyData>(imData)) {
342 boost::shared_ptr<InfJyBuilder> builder = boost::make_shared<InfJyBuilder>(
343 market_, jyData, configurationInfCalibration_, referenceCalibrationGrid_);
344 infParameterizations.push_back(builder->parameterization());
345 subBuilders_[CrossAssetModel::AssetType::INF][i] = builder;
346 processInfo[CrossAssetModel::AssetType::INF].emplace_back(jyData->index(), 2);
347 } else {
348 QL_FAIL("CrossAssetModelBuilder expects either DK or JY inflation model data.");
349 }
350 infIndices.push_back(imData->index());
351 }
352
353 /*******************************************************
354 * Build the CR parametrizations and calibration baskets
355 */
356 // LGM (if any)
357 std::vector<boost::shared_ptr<QuantExt::CrLgm1fParametrization>> crLgmParametrizations;
358 for (Size i = 0; i < config_->crLgmConfigs().size(); ++i) {
359 LOG("CR LGM Parametrization " << i);
360 boost::shared_ptr<CrLgmData> cr = config_->crLgmConfigs()[i];
361 string crName = cr->name();
362 boost::shared_ptr<CrLgmBuilder> builder =
363 boost::make_shared<CrLgmBuilder>(market_, cr, configurationCrCalibration_);
364 boost::shared_ptr<QuantExt::CrLgm1fParametrization> parametrization = builder->parametrization();
365 crLgmParametrizations.push_back(parametrization);
366 crNames.push_back(crName);
367 subBuilders_[CrossAssetModel::AssetType::CR][i] = builder;
368 processInfo[CrossAssetModel::AssetType::CR].emplace_back(crName, 1);
369 }
370
371 // CIR (if any)
372 std::vector<boost::shared_ptr<QuantExt::CrCirppParametrization>> crCirParametrizations;
373 for (Size i = 0; i < config_->crCirConfigs().size(); ++i) {
374 LOG("CR CIR Parametrization " << i);
375 boost::shared_ptr<CrCirData> cr = config_->crCirConfigs()[i];
376 string crName = cr->name();
377 boost::shared_ptr<CrCirBuilder> builder =
378 boost::make_shared<CrCirBuilder>(market_, cr, configurationCrCalibration_);
379 boost::shared_ptr<QuantExt::CrCirppParametrization> parametrization = builder->parametrization();
380 crCirParametrizations.push_back(parametrization);
381 crNames.push_back(crName);
382 subBuilders_[CrossAssetModel::AssetType::CR][i] = builder;
383 processInfo[CrossAssetModel::AssetType::CR].emplace_back(crName, 1);
384 }
385
386 /*******************************************************
387 * Build the COM parametrizations and calibration baskets
388 */
389 std::vector<boost::shared_ptr<QuantExt::CommoditySchwartzParametrization>> comParametrizations;
390 for (Size i = 0; i < config_->comConfigs().size(); i++) {
391 DLOG("COM Parametrization " << i);
392 boost::shared_ptr<CommoditySchwartzData> com = config_->comConfigs()[i];
393 string comName = com->name();
394 QuantLib::Currency comCcy = ore::data::parseCurrency(com->currency());
395 QL_REQUIRE(std::find(currencies.begin(), currencies.end(), comCcy.code()) != currencies.end(),
396 "Currency (" << comCcy << ") for commodity " << comName << " not covered by CrossAssetModelData");
397 boost::shared_ptr<CommoditySchwartzModelBuilder> builder = boost::make_shared<CommoditySchwartzModelBuilder>(
398 market_, com, domesticCcy, configurationComCalibration_, referenceCalibrationGrid_);
399 csBuilder.push_back(builder);
400 boost::shared_ptr<QuantExt::CommoditySchwartzParametrization> parametrization = builder->parametrization();
401 comOptionBaskets_[i] = builder->optionBasket();
402 comParametrizations.push_back(parametrization);
403 comNames.push_back(comName);
404 subBuilders_[CrossAssetModel::AssetType::COM][i] = builder;
405 processInfo[CrossAssetModel::AssetType::COM].emplace_back(comName, 1);
406 }
407
408 /*******************************************************
409 * Build the CrState parametrizations
410 */
411 std::vector<boost::shared_ptr<QuantExt::CrStateParametrization>> crStateParametrizations;
412 for (Size i = 0; i < config_->numberOfCreditStates(); i++) {
413 DLOG("CrState Parametrization " << i);
414 crStateParametrizations.push_back(boost::make_shared<QuantExt::CrStateParametrization>(i));
415 processInfo[CrossAssetModel::AssetType::CrState].emplace_back(std::to_string(i), 1);
416 }
417
418 std::vector<boost::shared_ptr<QuantExt::Parametrization>> parametrizations;
419 for (Size i = 0; i < irParametrizations.size(); i++)
420 parametrizations.push_back(irParametrizations[i]);
421 for (Size i = 0; i < fxParametrizations.size(); i++)
422 parametrizations.push_back(fxParametrizations[i]);
423 for (Size i = 0; i < eqParametrizations.size(); i++)
424 parametrizations.push_back(eqParametrizations[i]);
425 parametrizations.insert(parametrizations.end(), infParameterizations.begin(), infParameterizations.end());
426 for (Size i = 0; i < crLgmParametrizations.size(); i++)
427 parametrizations.push_back(crLgmParametrizations[i]);
428 for (Size i = 0; i < crCirParametrizations.size(); i++)
429 parametrizations.push_back(crCirParametrizations[i]);
430 for (Size i = 0; i < comParametrizations.size(); i++)
431 parametrizations.push_back(comParametrizations[i]);
432 for (Size i = 0; i < crStateParametrizations.size(); i++)
433 parametrizations.push_back(crStateParametrizations[i]);
434
435 QL_REQUIRE(fxParametrizations.size() == irParametrizations.size() - 1, "mismatch in IR/FX parametrization sizes");
436
437 /******************************
438 * Build the correlation matrix
439 */
440 DLOG("CrossAssetModelBuilder: adding correlations.");
441 CorrelationMatrixBuilder cmb;
442
443 for (auto it = config_->correlations().begin(); it != config_->correlations().end(); it++) {
444 cmb.addCorrelation(it->first.first, it->first.second, it->second);
445 }
446
447 Matrix corrMatrix = cmb.correlationMatrix(processInfo);
448
449 TLOG("CAM correlation matrix:");
450 TLOGGERSTREAM(corrMatrix);
451
452 /*****************************
453 * Build the cross asset model
454 */
455
456 model_.linkTo(boost::make_shared<QuantExt::CrossAssetModel>(parametrizations, corrMatrix, salvaging_, measure,
457 config_->discretization()));
458
459 /*************************
460 * Calibrate IR components
461 */
462
463 for (Size i = 0; i < lgmBuilder.size(); i++) {
464 DLOG("IR Calibration " << i);
465 swaptionCalibrationErrors_[i] = lgmBuilder[i]->error();
466 }
467
468 for (Size i = 0; i < hwBuilder.size(); i++) {
469 DLOG("IR Calibration " << i);
470 swaptionCalibrationErrors_[i] = hwBuilder[i]->error();
471 }
472
473 /*************************
474 * Relink LGM discount curves to curves used for FX calibration
475 */
476
477 for (Size i = 0; i < irParametrizations.size(); i++) {
478 auto p = irParametrizations[i];
479 irDiscountCurves[i].linkTo(*market_->discountCurve(p->currency().code(), configurationFxCalibration_));
480 DLOG("Relinked discounting curve for " << p->currency().code() << " for FX calibration");
481 }
482
483 /*************************
484 * Calibrate FX components
485 */
486
487 for (Size i = 0; i < fxParametrizations.size(); i++) {
488 boost::shared_ptr<FxBsData> fx = config_->fxConfigs()[i];
489
490 if (fx->calibrationType() == CalibrationType::None || !fx->calibrateSigma()) {
491 DLOG("FX Calibration " << i << " skipped");
492 continue;
493 }
494
495 DLOG("FX Calibration " << i);
496
497 // attach pricing engines to helpers
498 boost::shared_ptr<QuantExt::AnalyticCcLgmFxOptionEngine> engine =
499 boost::make_shared<QuantExt::AnalyticCcLgmFxOptionEngine>(*model_, i);
500 // enable caching for calibration
501 // TODO: review this
502 engine->cache(true);
503 for (Size j = 0; j < fxOptionBaskets_[i].size(); j++)
504 fxOptionBaskets_[i][j]->setPricingEngine(engine);
505
506 if (!dontCalibrate_) {
507
508 if (fx->calibrationType() == CalibrationType::Bootstrap && fx->sigmaParamType() == ParamType::Piecewise)
509 model_->calibrateBsVolatilitiesIterative(CrossAssetModel::AssetType::FX, i, fxOptionBaskets_[i],
510 *optimizationMethod_, endCriteria_);
511 else
512 model_->calibrateBsVolatilitiesGlobal(CrossAssetModel::AssetType::FX, i, fxOptionBaskets_[i],
513 *optimizationMethod_, endCriteria_);
514
515 DLOG("FX " << fx->foreignCcy() << " calibration errors:");
516 fxOptionCalibrationErrors_[i] = getCalibrationError(fxOptionBaskets_[i]);
517 if (fx->calibrationType() == CalibrationType::Bootstrap) {
518 if (fabs(fxOptionCalibrationErrors_[i]) < config_->bootstrapTolerance()) {
519 TLOGGERSTREAM("Calibration details:");
520 TLOGGERSTREAM(
521 getCalibrationDetails(fxOptionBaskets_[i], fxParametrizations[i], irParametrizations[0]));
522 TLOGGERSTREAM("rmse = " << fxOptionCalibrationErrors_[i]);
523 } else {
524 std::string exceptionMessage = "FX BS " + std::to_string(i) + " calibration error " +
525 std::to_string(fxOptionCalibrationErrors_[i]) +
526 " exceeds tolerance " +
527 std::to_string(config_->bootstrapTolerance());
528 WLOG(StructuredModelErrorMessage("Failed to calibrate FX BS Model", exceptionMessage));
529 WLOGGERSTREAM("Calibration details:");
530 WLOGGERSTREAM(
531 getCalibrationDetails(fxOptionBaskets_[i], fxParametrizations[i], irParametrizations[0]));
532 WLOGGERSTREAM("rmse = " << fxOptionCalibrationErrors_[i]);
533 if (!continueOnError_)
534 QL_FAIL(exceptionMessage);
535 }
536 }
537 }
538 }
539
540 /*************************
541 * Relink LGM discount curves to curves used for EQ calibration
542 */
543
544 for (Size i = 0; i < irParametrizations.size(); i++) {
545 auto p = irParametrizations[i];
546 irDiscountCurves[i].linkTo(*market_->discountCurve(p->currency().code(), configurationEqCalibration_));
547 DLOG("Relinked discounting curve for " << p->currency().code() << " for EQ calibration");
548 }
549
550 /*************************
551 * Calibrate EQ components
552 */
553
554 for (Size i = 0; i < eqParametrizations.size(); i++) {
555 boost::shared_ptr<EqBsData> eq = config_->eqConfigs()[i];
556 if (!eq->calibrateSigma()) {
557 DLOG("EQ Calibration " << i << " skipped");
558 continue;
559 }
560 DLOG("EQ Calibration " << i);
561 // attach pricing engines to helpers
562 Currency eqCcy = eqParametrizations[i]->currency();
563 Size eqCcyIdx = model_->ccyIndex(eqCcy);
564 boost::shared_ptr<QuantExt::AnalyticXAssetLgmEquityOptionEngine> engine =
565 boost::make_shared<QuantExt::AnalyticXAssetLgmEquityOptionEngine>(*model_, i, eqCcyIdx);
566 for (Size j = 0; j < eqOptionBaskets_[i].size(); j++)
567 eqOptionBaskets_[i][j]->setPricingEngine(engine);
568
569 if (!dontCalibrate_) {
570
571 if (eq->calibrationType() == CalibrationType::Bootstrap && eq->sigmaParamType() == ParamType::Piecewise)
572 model_->calibrateBsVolatilitiesIterative(CrossAssetModel::AssetType::EQ, i, eqOptionBaskets_[i],
573 *optimizationMethod_, endCriteria_);
574 else
575 model_->calibrateBsVolatilitiesGlobal(CrossAssetModel::AssetType::EQ, i, eqOptionBaskets_[i],
576 *optimizationMethod_, endCriteria_);
577 DLOG("EQ " << eq->eqName() << " calibration errors:");
578 eqOptionCalibrationErrors_[i] = getCalibrationError(eqOptionBaskets_[i]);
579 if (eq->calibrationType() == CalibrationType::Bootstrap) {
580 if (fabs(eqOptionCalibrationErrors_[i]) < config_->bootstrapTolerance()) {
581 TLOGGERSTREAM("Calibration details:");
582 TLOGGERSTREAM(
583 getCalibrationDetails(eqOptionBaskets_[i], eqParametrizations[i], irParametrizations[0]));
584 TLOGGERSTREAM("rmse = " << eqOptionCalibrationErrors_[i]);
585 } else {
586 std::string exceptionMessage = "EQ BS " + std::to_string(i) + " calibration error " +
587 std::to_string(eqOptionCalibrationErrors_[i]) +
588 " exceeds tolerance " +
589 std::to_string(config_->bootstrapTolerance());
590 WLOG(StructuredModelErrorMessage("Failed to calibrate EQ BS Model", exceptionMessage));
591 WLOGGERSTREAM("Calibration details:");
592 WLOGGERSTREAM(
593 getCalibrationDetails(eqOptionBaskets_[i], eqParametrizations[i], irParametrizations[0]));
594 WLOGGERSTREAM("rmse = " << eqOptionCalibrationErrors_[i]);
595 if (!continueOnError_)
596 QL_FAIL(exceptionMessage);
597 }
598 }
599 }
600 }
601
602 /*************************
603 * Calibrate COM components
604 */
605
606 for (Size i = 0; i < csBuilder.size(); i++) {
607 DLOG("COM Calibration " << i);
608 comOptionCalibrationErrors_[i] = csBuilder[i]->error();
609 }
610
611 /*************************
612 * Relink LGM discount curves to curves used for INF calibration
613 */
614
615 for (Size i = 0; i < irParametrizations.size(); i++) {
616 auto p = irParametrizations[i];
617 irDiscountCurves[i].linkTo(*market_->discountCurve(p->currency().code(), configurationInfCalibration_));
618 DLOG("Relinked discounting curve for " << p->currency().code() << " for INF calibration");
619 }
620
621 // Calibrate INF components
622 for (Size i = 0; i < infParameterizations.size(); i++) {
623 boost::shared_ptr<InflationModelData> imData = config_->infConfigs()[i];
624 if (auto dkData = boost::dynamic_pointer_cast<InfDkData>(imData)) {
625 auto dkParam = boost::dynamic_pointer_cast<InfDkParametrization>(infParameterizations[i]);
626 QL_REQUIRE(dkParam, "Expected DK model data to have given a DK parameterisation.");
627 const auto& builder = subBuilders_.at(CrossAssetModel::AssetType::INF).at(i);
628 const auto& dkBuilder = boost::dynamic_pointer_cast<InfDkBuilder>(builder);
629 calibrateInflation(*dkData, i, dkBuilder->optionBasket(), dkParam);
630 } else if (auto jyData = boost::dynamic_pointer_cast<InfJyData>(imData)) {
631 auto jyParam = boost::dynamic_pointer_cast<InfJyParameterization>(infParameterizations[i]);
632 QL_REQUIRE(jyParam, "Expected JY model data to have given a JY parameterisation.");
633 const auto& builder = subBuilders_.at(CrossAssetModel::AssetType::INF).at(i);
634 const auto& jyBuilder = boost::dynamic_pointer_cast<InfJyBuilder>(builder);
635 calibrateInflation(*jyData, i, jyBuilder, jyParam);
636 } else {
637 QL_FAIL("CrossAssetModelBuilder expects either DK or JY inflation model data.");
638 }
639 }
640
641 /*************************
642 * Relink LGM discount curves to final model curves
643 */
644
645 for (Size i = 0; i < irParametrizations.size(); i++) {
646 auto p = irParametrizations[i];
647 irDiscountCurves[i].linkTo(*market_->discountCurve(p->currency().code(), configurationFinalModel_));
648 DLOG("Relinked discounting curve for " << p->currency().code() << " as final model curves");
649 }
650
651 // play safe (although the cache of the model should be empty at
652 // this point from all what we know...)
653 model_->update();
654
655 DLOG("Building CrossAssetModel done");
656}

664void CrossAssetModelBuilder::calibrateInflation(const InfDkData& data, Size modelIdx,
665 const vector<boost::shared_ptr<BlackCalibrationHelper>>& cb,
666 const boost::shared_ptr<InfDkParametrization>& inflationParam) const {
667
668 LOG("Calibrate DK inflation model for inflation index " << data.index());
669
670 if ((!data.volatility().calibrate() && !data.reversion().calibrate()) ||
671 (data.calibrationType() == CalibrationType::None)) {
672 LOG("Calibration of DK inflation model for inflation index " << data.index() << " not requested.");
673 return;
674 }
675
676 Handle<ZeroInflationIndex> zInfIndex =
677 market_->zeroInflationIndex(model_->infdk(modelIdx)->name(), configurationInfCalibration_);
678 Real baseCPI = dontCalibrate_ ? 100. : zInfIndex->fixing(zInfIndex->zeroInflationTermStructure()->baseDate());
679 auto engine = boost::make_shared<QuantExt::AnalyticDkCpiCapFloorEngine>(*model_, modelIdx, baseCPI);
680 for (Size j = 0; j < cb.size(); j++)
681 cb[j]->setPricingEngine(engine);
682
683 if (dontCalibrate_)
684 return;
685
686 if (data.volatility().calibrate() && !data.reversion().calibrate()) {
687 if (data.calibrationType() == CalibrationType::Bootstrap && data.volatility().type() == ParamType::Piecewise) {
688 model_->calibrateInfDkVolatilitiesIterative(modelIdx, cb, *optimizationMethod_, endCriteria_);
689 } else {
690 model_->calibrateInfDkVolatilitiesGlobal(modelIdx, cb, *optimizationMethod_, endCriteria_);
691 }
692 } else if (!data.volatility().calibrate() && data.reversion().calibrate()) {
693 if (data.calibrationType() == CalibrationType::Bootstrap && data.reversion().type() == ParamType::Piecewise) {
694 model_->calibrateInfDkReversionsIterative(modelIdx, cb, *optimizationMethod_, endCriteria_);
695 } else {
696 model_->calibrateInfDkReversionsGlobal(modelIdx, cb, *optimizationMethod_, endCriteria_);
697 }
698 } else {
699 model_->calibrate(cb, *optimizationMethod_, endCriteria_);
700 }
701
702 DLOG("INF (DK) " << data.index() << " calibration errors:");
703 inflationCalibrationErrors_[modelIdx] = getCalibrationError(cb);
704 if (data.calibrationType() == CalibrationType::Bootstrap) {
705 if (fabs(inflationCalibrationErrors_[modelIdx]) < config_->bootstrapTolerance()) {
706 TLOGGERSTREAM("Calibration details:");
707 TLOGGERSTREAM(getCalibrationDetails(cb, inflationParam, false));
708 TLOGGERSTREAM("rmse = " << inflationCalibrationErrors_[modelIdx]);
709 } else {
710 string exceptionMessage = "INF (DK) " + std::to_string(modelIdx) + " calibration error " +
711 std::to_string(inflationCalibrationErrors_[modelIdx]) + " exceeds tolerance " +
712 std::to_string(config_->bootstrapTolerance());
713 WLOG(StructuredModelErrorMessage("Failed to calibrate INF DK Model", exceptionMessage));
714 WLOGGERSTREAM("Calibration details:");
715 WLOGGERSTREAM(getCalibrationDetails(cb, inflationParam, false));
716 WLOGGERSTREAM("rmse = " << inflationCalibrationErrors_[modelIdx]);
717 if (!continueOnError_)
718 QL_FAIL(exceptionMessage);
719 }
720 }

723void CrossAssetModelBuilder::calibrateInflation(const InfJyData& data, Size modelIdx,
724 const boost::shared_ptr<InfJyBuilder>& jyBuilder,
725 const boost::shared_ptr<InfJyParameterization>& inflationParam) const {
726
727 LOG("Calibrate JY inflation model for inflation index " << data.index());
728
729 const auto& rrVol = data.realRateVolatility();
730 const auto& rrRev = data.realRateReversion();
731 const auto& idxVol = data.indexVolatility();
732
733 // Check if calibration is needed at all.
734 if ((!rrVol.calibrate() && !rrRev.calibrate() && !idxVol.calibrate()) ||
735 (data.calibrationType() == CalibrationType::None)) {
736 LOG("Calibration of JY inflation model for inflation index " << data.index() << " not requested.");
737 return;
738 }
739
740 Handle<ZeroInflationIndex> zInfIndex =
741 market_->zeroInflationIndex(model_->infjy(modelIdx)->name(), configurationInfCalibration_);
742
743 // We will need the 2 baskets of helpers
744 auto rrBasket = jyBuilder->realRateBasket();
745 auto idxBasket = jyBuilder->indexBasket();
746
747 // Attach engines to the helpers.
748 setJyPricingEngine(modelIdx, rrBasket, false);
749 setJyPricingEngine(modelIdx, idxBasket, false);
750
751 if (dontCalibrate_)
752 return;
753
754 // Single basket of helpers is useful in various places below.
755 vector<boost::shared_ptr<CalibrationHelper>> allHelpers = rrBasket;
756 allHelpers.insert(allHelpers.end(), idxBasket.begin(), idxBasket.end());
757
758 // Calibration configuration.
759 const auto& cc = data.calibrationConfiguration();
760
761 if (data.calibrationType() == CalibrationType::BestFit) {
762
763 // If calibration type is BestFit, do a global optimisation on the parameters that need to be calibrated.
764 DLOG("Calibration BestFit of JY inflation model for inflation index " << data.index() << " requested.");
765
766 // Indicate the parameters to calibrate
767 map<Size, bool> toCalibrate;
768 toCalibrate[0] = rrVol.calibrate();
769 toCalibrate[1] = rrRev.calibrate();
770 toCalibrate[2] = idxVol.calibrate();
771
772 // Calibrate the model.
773 model_->calibrateInfJyGlobal(modelIdx, allHelpers, *optimizationMethod_, endCriteria_, toCalibrate);
774
775 } else {
776
777 // Calibration type is now Bootstrap, there are multiple options.
778 QL_REQUIRE(data.calibrationType() == CalibrationType::Bootstrap,
779 "JY inflation calibration expected a "
780 << "calibration type of None, BestFit or Bootstrap.");
781 QL_REQUIRE(!(rrRev.calibrate() && rrVol.calibrate()),
782 "Calibrating both the "
783 << "real rate reversion and real rate volatility using Bootstrap is not supported.");
784
785 if ((!rrVol.calibrate() && !rrRev.calibrate()) && idxVol.calibrate()) {
786
787 // Bootstrap the inflation index volatility only.
788 DLOG("Bootstrap calibration of JY index volatility for index " << data.index() << ".");
789 QL_REQUIRE(idxVol.type() == ParamType::Piecewise, "Index volatility parameter should be Piecewise for "
790 << "a Bootstrap calibration.");
791 model_->calibrateInfJyIterative(modelIdx, 2, idxBasket, *optimizationMethod_, endCriteria_);
792
793 } else if (rrVol.calibrate() && !idxVol.calibrate()) {
794
795 // Bootstrap the real rate volatility only
796 DLOG("Bootstrap calibration of JY real rate volatility for index " << data.index() << ".");
797 QL_REQUIRE(rrVol.type() == ParamType::Piecewise, "Real rate volatility parameter should be "
798 << "Piecewise for a Bootstrap calibration.");
799 model_->calibrateInfJyIterative(modelIdx, 0, rrBasket, *optimizationMethod_, endCriteria_);
800
801 } else if (rrRev.calibrate() && !idxVol.calibrate()) {
802
803 // Bootstrap the real rate reversion only
804 DLOG("Bootstrap calibration of JY real rate reversion for index " << data.index() << ".");
805 QL_REQUIRE(rrRev.type() == ParamType::Piecewise, "Real rate reversion parameter should be "
806 << "Piecewise for a Bootstrap calibration.");
807 model_->calibrateInfJyIterative(modelIdx, 1, rrBasket, *optimizationMethod_, endCriteria_);
808
809 } else if ((rrVol.calibrate() && idxVol.calibrate()) || (rrRev.calibrate() && idxVol.calibrate())) {
810
811 if (rrVol.calibrate()) {
812 DLOG("Bootstrap calibration of JY real rate volatility and index volatility for index " << data.index()
813 << ".");
814 } else {
815 DLOG("Bootstrap calibration of JY real rate reversion and index volatility for index " << data.index()
816 << ".");
817 }
818
819 // Bootstrap the real rate volatility and the index volatility
820 Size rrIdx = rrVol.calibrate() ? 0 : 1;
821 Size numIts = 0;
822 inflationCalibrationErrors_[modelIdx] = getCalibrationError(allHelpers);
823
824 while (inflationCalibrationErrors_[modelIdx] > cc.rmseTolerance() && numIts < cc.maxIterations()) {
825 model_->calibrateInfJyIterative(modelIdx, 2, idxBasket, *optimizationMethod_, endCriteria_);
826 model_->calibrateInfJyIterative(modelIdx, rrIdx, rrBasket, *optimizationMethod_, endCriteria_);
827 numIts++;
828 inflationCalibrationErrors_[modelIdx] = getCalibrationError(allHelpers);
829 }
830
831 DLOG("Bootstrap calibration of JY model stopped with number of iterations "
832 << numIts << " and rmse equal to " << std::scientific << std::setprecision(6)
833 << inflationCalibrationErrors_[modelIdx] << ".");
834
835 } else {
836 QL_FAIL("JY inflation bootstrap calibration does not support the combination of real rate volatility = "
837 << std::boolalpha << rrVol.calibrate() << ", real rate reversion = " << rrRev.calibrate() << " and "
838 << "index volatility = " << idxVol.calibrate() << ".");
839 }
840 }
841
842 // Log the calibration details.
843 DLOG("INF (JY) " << data.index() << " calibration errors:");
844 inflationCalibrationErrors_[modelIdx] = getCalibrationError(allHelpers);
845 if (data.calibrationType() == CalibrationType::Bootstrap) {
846 if (fabs(inflationCalibrationErrors_[modelIdx]) < cc.rmseTolerance()) {
847 TLOGGERSTREAM("Calibration details:");
848 TLOGGERSTREAM(getCalibrationDetails(rrBasket, idxBasket, inflationParam, rrVol.calibrate()));
849 TLOGGERSTREAM("rmse = " << inflationCalibrationErrors_[modelIdx]);
850 } else {
851 std::stringstream ss;
852 ss << "INF (JY) " << modelIdx << " calibration error " << std::scientific
853 << inflationCalibrationErrors_[modelIdx] << " exceeds tolerance " << cc.rmseTolerance();
854 string exceptionMessage = ss.str();
855 WLOG(StructuredModelErrorMessage("Failed to calibrate INF JY Model", exceptionMessage));
856 WLOGGERSTREAM("Calibration details:");
857 WLOGGERSTREAM(getCalibrationDetails(rrBasket, idxBasket, inflationParam, rrVol.calibrate()));
858 WLOGGERSTREAM("rmse = " << inflationCalibrationErrors_[modelIdx]);
859 if (!continueOnError_)
860 QL_FAIL(exceptionMessage);
861 }
862 }
863
864 LOG("Finished calibrating JY inflation model for inflation index " << data.index());
865}

867void CrossAssetModelBuilder::setJyPricingEngine(
868 Size modelIdx, const vector<boost::shared_ptr<CalibrationHelper>>& calibrationBasket, bool indexIsInterpolated) const {
869
870 DLOG("Start setting pricing engines on JY calibration instruments.");
871
872 // JY supports three types of calibration helpers. Generally, all of the calibration instruments in a basket will
873 // be of the same type but we support all three here.
874 boost::shared_ptr<PricingEngine> cpiCapFloorEngine;
875 boost::shared_ptr<PricingEngine> yoyCapFloorEngine;
876 boost::shared_ptr<PricingEngine> yoySwapEngine;
877 boost::shared_ptr<InflationCouponPricer> yoyCouponPricer;
878
879 for (auto& ci : calibrationBasket) {
880
881 if (boost::shared_ptr<CpiCapFloorHelper> h = boost::dynamic_pointer_cast<CpiCapFloorHelper>(ci)) {
882 if (!cpiCapFloorEngine) {
883 cpiCapFloorEngine = boost::make_shared<AnalyticJyCpiCapFloorEngine>(*model_, modelIdx);
884 }
885 h->setPricingEngine(cpiCapFloorEngine);
886 continue;
887 }
888
889 if (boost::shared_ptr<YoYCapFloorHelper> h = boost::dynamic_pointer_cast<YoYCapFloorHelper>(ci)) {
890 if (!yoyCapFloorEngine) {
891 yoyCapFloorEngine = boost::make_shared<AnalyticJyYoYCapFloorEngine>(*model_, modelIdx, indexIsInterpolated);
892 }
893 h->setPricingEngine(yoyCapFloorEngine);
894 continue;
895 }
896
897 if (boost::shared_ptr<YoYSwapHelper> h = boost::dynamic_pointer_cast<YoYSwapHelper>(ci)) {
898 // Here we need to attach the coupon pricer to all the YoY coupons and then the generic discounting swap
899 // engine to the helper.
900 if (!yoyCouponPricer) {
901 yoyCouponPricer = boost::make_shared<JyYoYInflationCouponPricer>(*model_, modelIdx);
902
903 Size irIdx = model_->ccyIndex(model_->infjy(modelIdx)->currency());
904 auto yts = model_->irlgm1f(irIdx)->termStructure();
905 yoySwapEngine = boost::make_shared<DiscountingSwapEngine>(yts);
906 }
907
908 const auto& yoyLeg = h->yoySwap()->yoyLeg();
909 for (const auto& cf : yoyLeg) {
910 if (auto yoyCoupon = boost::dynamic_pointer_cast<YoYInflationCoupon>(cf))
911 yoyCoupon->setPricer(yoyCouponPricer);
912 }
913
914 h->setPricingEngine(yoySwapEngine);
915 continue;
916 }
917
918 QL_FAIL("Only CPI cap floors, YoY cap floors and YoY swaps are supported for JY calibration.");
919 }
920
921 DLOG("Finished setting pricing engines on JY calibration instruments.");
922}
