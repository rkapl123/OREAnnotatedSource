39void IndexCreditDefaultSwap::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
40 DLOG("IndexCreditDefaultSwap::build() called for trade " << id());
41
42 const boost::shared_ptr<Market> market = engineFactory->market();
43 boost::shared_ptr<EngineBuilder> builder = engineFactory->builder("IndexCreditDefaultSwap");
44
45 QL_REQUIRE(swap_.leg().legType() == "Fixed", "IndexCreditDefaultSwap requires Fixed leg");
46 boost::shared_ptr<FixedLegData> fixedLegData =
47 boost::dynamic_pointer_cast<FixedLegData>(swap_.leg().concreteLegData());
48
49 auto configuration = builder->configuration(MarketContext::pricing);
50 FixedLegBuilder flb;
51 legs_.resize(1);
52 legs_[0] = flb.buildLeg(swap_.leg(), engineFactory, requiredFixings_, configuration);
53
54 Schedule schedule = makeSchedule(swap_.leg().schedule());
55 BusinessDayConvention payConvention = parseBusinessDayConvention(swap_.leg().paymentConvention());
56 Protection::Side prot = swap_.leg().isPayer() ? Protection::Side::Buyer : Protection::Side::Seller;
57
58 notional_ = notional();
59 DayCounter dc = parseDayCounter(swap_.leg().dayCounter());
60
61 // In general for CDS and CDS index trades, the standard day counter is Actual/360 and the final
62 // period coupon accrual includes the maturity date.
63 Actual360 standardDayCounter;
64 DayCounter lastPeriodDayCounter = dc == standardDayCounter ? Actual360(true) : dc;
65
66 QL_REQUIRE(fixedLegData->rates().size() == 1, "IndexCreditDefaultSwap requires single rate");
67
68 Real indexFactor = 1.0;
69
70 // From the basket data or reference data, we need a vector of notionals and a vector of creditCurves
71 vector<Real> basketNotionals;
72 vector<string> basketCreditCurves;
73
74 if (!swap_.basket().constituents().empty()) {
75
76 const auto& constituents = swap_.basket().constituents();
77 DLOG("Building constituents from basket data containing " << constituents.size() << " elements.");
78
79 Real totalNtl = 0.0;
80 for (const auto& c : constituents) {
81 Real ntl = Null<Real>();
82
83 const auto& creditCurve = c.creditCurveId();
84
85 if (c.weightInsteadOfNotional()) {
86 ntl = c.weight() * notional_;
87 } else {
88 ntl = c.notional();
89 }
90
91 if (!close(0.0, ntl) && ntl > 0.0) {
92 if (std::find(basketCreditCurves.begin(), basketCreditCurves.end(), creditCurve) ==
93 basketCreditCurves.end()) {
94 DLOG("Adding underlying " << creditCurve << " with notional " << ntl);
95 constituents_[creditCurve] = ntl;
96 basketCreditCurves.push_back(creditCurve);
97 basketNotionals.push_back(ntl);
98 totalNtl += ntl;
99 } else {
100 ALOG(StructuredTradeErrorMessage(id(), "IndexCDS", "Error building trade",
101 ("Invalid Basket: found a duplicate credit curve " + creditCurve +
102 ", skip it. Check the basket data for possible errors.")
103 .c_str()));
104 }
105
106 } else {
107 DLOG("Skipped adding underlying, " << creditCurve << ", because its notional, " << ntl
108 << ", was non-positive.");
109 }
110 }
111 QL_REQUIRE(basketCreditCurves.size() == basketNotionals.size(),
112 "numbers of defaults curves (" << basketCreditCurves.size() << ") and notionals ("
113 << basketNotionals.size() << ") doesnt match");
114 DLOG("All underlyings added, total notional = " << totalNtl);
115
116 if (totalNtl > notional_ * (1.0 + 1.0E-4)) {
117 ALOG(StructuredTradeErrorMessage(id(), "IndexCDS", "Error building trade",
118 ("Sum of basket notionals (" + std::to_string(totalNtl) +
119 ") is greater than trade notional (" + std::to_string(notional_) +
120 "). Check the basket data for possible errors.")
121 .c_str()));
122 }
123
124 indexFactor = totalNtl / notional_;
125
126 DLOG("Finished building constituents using basket data.");
127
128 } else {
129 // get data from ReferenceDatum
130 string id = ore::data::splitCurveIdWithTenor(swap_.creditCurveId()).first;
131 DLOG("Getting CreditIndexReferenceDatum for id " << id);
132 QL_REQUIRE(engineFactory->referenceData(), "No BasketData or ReferenceDataManager");
133 QL_REQUIRE(engineFactory->referenceData()->hasData(CreditIndexReferenceDatum::TYPE, id),
134 "No CreditIndex reference data for " << id);
135 boost::shared_ptr<ReferenceDatum> refData =
136 engineFactory->referenceData()->getData(CreditIndexReferenceDatum::TYPE, id);
137 boost::shared_ptr<CreditIndexReferenceDatum> creditRefData =
138 boost::dynamic_pointer_cast<CreditIndexReferenceDatum>(refData);
139 DLOG("Got CreditIndexReferenceDatum for id " << id);
140
141 Real totalWeight = 0.0;
142 for (const auto& c : creditRefData->constituents()) {
143
144 const auto& name = c.name();
145 auto weight = c.weight();
146
147 if (!close(0.0, weight) && weight > 0.0) {
148 DLOG("Adding underlying " << name << " with weight " << weight);
149 constituents_[name] = notional_ * weight;
150 basketCreditCurves.push_back(name);
151 basketNotionals.push_back(notional_ * weight);
152 totalWeight += weight;
153 } else {
154 DLOG("Skipped adding underlying, " << name << ", because its weight, " << weight
155 << ", was non-positive.");
156 }
157 }
158
159 indexFactor = totalWeight;
160
161 DLOG("All underlyings added, total weight = " << totalWeight);
162
163 if (!close(1.0, totalWeight) && totalWeight > 1.0) {
164 ALOG("Total weight is greater than 1, possible error in CreditIndexReferenceDatum");
165 }
166 }
167
168 boost::shared_ptr<QuantExt::IndexCreditDefaultSwap> cds;
169 if (swap_.upfrontFee() == Null<Real>()) {
170 cds = boost::make_shared<QuantExt::IndexCreditDefaultSwap>(
171 prot, indexFactor * notional_, basketNotionals, fixedLegData->rates().front(), schedule, payConvention, dc,
172 swap_.settlesAccrual(), swap_.protectionPaymentTime(), swap_.protectionStart(), boost::shared_ptr<Claim>(),
173 lastPeriodDayCounter, true, swap_.tradeDate(), swap_.cashSettlementDays());
174 } else {
175 cds = boost::make_shared<QuantExt::IndexCreditDefaultSwap>(
176 prot, indexFactor * notional_, basketNotionals, swap_.upfrontFee(), fixedLegData->rates().front(), schedule,
177 payConvention, dc, swap_.settlesAccrual(), swap_.protectionPaymentTime(), swap_.protectionStart(),
178 swap_.upfrontDate(), boost::shared_ptr<Claim>(), lastPeriodDayCounter, true, swap_.tradeDate(),
179 swap_.cashSettlementDays());
180 }
181
182 boost::shared_ptr<IndexCreditDefaultSwapEngineBuilder> cdsBuilder =
183 boost::dynamic_pointer_cast<IndexCreditDefaultSwapEngineBuilder>(builder);
184
185 npvCurrency_ = swap_.leg().currency();
186 notionalCurrency_ = swap_.leg().currency();
187
188 QL_REQUIRE(cdsBuilder, "No Builder found for IndexCreditDefaultSwap: " << id());
189 std::string curveIdWithTerm = swap_.creditCurveIdWithTerm();
190 // warn if the term can not be implied, except when a custom baskets is defined
191 if (swap_.basket().constituents().empty() && splitCurveIdWithTenor(curveIdWithTerm).second == 0 * Days) {
192 ALOG(StructuredTradeWarningMessage(
193 id(), tradeType(), "Could not imply Index CDS term.",
194 "Index CDS term could not be derived from start, end date, are these dates correct (credit curve id is '" +
195 swap_.creditCurveId() + "')"));
196 }
197
198 cds->setPricingEngine(cdsBuilder->engine(parseCurrency(npvCurrency_), swap_.creditCurveIdWithTerm(),
199 basketCreditCurves, boost::none, swap_.recoveryRate(), false));
200
201 instrument_.reset(new VanillaInstrument(cds));
202
203 maturity_ = cds->coupons().back()->date();
204
205 legs_ = {cds->coupons()};
206 legCurrencies_ = {npvCurrency_};
207 legPayers_ = {swap_.leg().isPayer()};
208
209 if (swap_.protectionStart() != Date())
210 additionalData_["startDate"] = to_string(swap_.protectionStart());
211 else
212 additionalData_["startDate"] = to_string(schedule.dates().front());
213
214 sensitivityDecomposition_ = cdsBuilder->sensitivityDecomposition();
215
216 // ISDA taxonomy
217 additionalData_["isdaAssetClass"] = string("Credit");
218 additionalData_["isdaBaseProduct"] = string("Index");
219 string entity = swap_.creditCurveId();
220 boost::shared_ptr<ReferenceDataManager> refData = engineFactory->referenceData();
221 if (refData && refData->hasData("CreditIndex", entity)) {
222 auto refDatum = refData->getData("CreditIndex", entity);
223 boost::shared_ptr<CreditIndexReferenceDatum> creditIndexRefDatum = boost::dynamic_pointer_cast<CreditIndexReferenceDatum>(refDatum);
224 additionalData_["isdaSubProduct"] = creditIndexRefDatum->indexFamily();
225 if (creditIndexRefDatum->indexFamily() == "") {
226 ALOG("IndexFamily is blank in credit index reference data for entity " << entity);
227 }
228 } else {
229 ALOG("Credit index reference data missing for entity " << entity << ", isdaSubProduct left blank");
230 }
231 // skip the transaction level mapping for now
232 additionalData_["isdaTransaction"] = string("");
233
234}

236const std::map<std::string, boost::any>& IndexCreditDefaultSwap::additionalData() const {
237 setLegBasedAdditionalData(0, 2);
238 additionalData_["legNPV[1]"] = instrument_->qlInstrument()->result<Real>("protectionLegNPV");
239 additionalData_["legNPV[2]"] = instrument_->qlInstrument()->result<Real>("premiumLegNPVDirty") +
240 instrument_->qlInstrument()->result<Real>("upfrontPremiumNPV") +
241 instrument_->qlInstrument()->result<Real>("accrualRebateNPV");
242 additionalData_["isPayer[1]"] = !swap_.leg().isPayer();
243 additionalData_["isPayer[2]"] = swap_.leg().isPayer();
244 additionalData_["legType[2]"] = swap_.leg().legType();
245 additionalData_["legType[1]"] = std::string("Protection");
246 additionalData_["currentNotional[1]"] = additionalData_["currentNotional[2]"];
247 additionalData_["originalNotional[1]"] = additionalData_["originalNotional[2]"];
248 additionalData_["notionalCurrency[1]"] = notionalCurrency_;
249 additionalData_["notionalCurrency[2]"] = notionalCurrency_;
250 return additionalData_;
251}

254QuantLib::Real IndexCreditDefaultSwap::notional() const {
255 Date asof = Settings::instance().evaluationDate();
256 // get the current notional from the premium leg
257 for (Size i = 0; i < legs_[0].size(); ++i) {
258 boost::shared_ptr<Coupon> coupon = boost::dynamic_pointer_cast<Coupon>(legs_[0][i]);
259 if (coupon->date() > asof)
260 return coupon->nominal();
261 }
262
263 // if not provided, return null
264 ALOG("Error retrieving current notional for index credit default swap " << id() << " as of " << io::iso_date(asof));
265 return Null<Real>();
266}
