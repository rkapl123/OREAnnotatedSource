197  case ore::data::CreditEventType::FAILURE_TO_PAY:

212  }
213  break;
214  case ore::data::CreditEventType::GOVERNMENTAL_INTERVENTION:

219  case ore::data::CdsDocClause::CR:

230  break;
231  }
232  case ore::data::CreditEventType::OBLIGATION_ACCELERATION:
233  case ore::data::CreditEventType::OBLIGATION_DEFAULT:

235  return false;
236  break;
237  default:
238  break;
239  }
240  QL_FAIL("Could not recognize CreditEventType "
241  << static_cast<int>(creditEventType) << " or CdsDocClause " << static_cast<int>(contractDocClause)

427 bool tryParseCdsInformation(const string& strInfo, CdsReferenceInformation& cdsInfo) {
428 
429  DLOG("tryParseCdsInformation: attempting to parse " << strInfo);
430 
431  // As in documentation comment, expect strInfo of form ID|TIER|CCY|DOCCLAUSE
432  vector<string> tokens;
433  boost::split(tokens, strInfo, boost::is_any_of("|"));
434 
435  if (tokens.size() != 4) {
436  TLOG("String " << strInfo << " not of form ID|TIER|CCY|DOCCLAUSE so parsing failed");
437  return false;
438  }
439 
440  CdsTier cdsTier;
441  if (!tryParse<CdsTier>(tokens[1], cdsTier, &parseCdsTier)) {
442  return false;
443  }
444 
445  Currency ccy;
446  if (!tryParseCurrency(tokens[2], ccy)) {
447  return false;
448  }
449 
450  CdsDocClause cdsDocClause;
451  if (!tryParse<CdsDocClause>(tokens[3], cdsDocClause, &parseCdsDocClause)) {
452  return false;
453  }
454 
455  cdsInfo = CdsReferenceInformation(tokens[0], cdsTier, ccy, cdsDocClause);
456 
457  return true;
458 }

489 void CreditDefaultSwapData::fromXML(XMLNode* node) {
490 
491  check(node);
492 
493  issuerId_ = XMLUtils::getChildValue(node, "IssuerId", false);
494 
495  // May get an explicit CreditCurveId node. If so, we use it. Otherwise, we must have ReferenceInformation node.
496  // XMLNode* tmp = XMLUtils::getChildNode(node, "CreditCurveId");
497  if (auto tmp = XMLUtils::getChildNode(node, "CreditCurveId")) {
498  creditCurveId_ = XMLUtils::getNodeValue(tmp);
499  } else {
500  tmp = XMLUtils::getChildNode(node, "ReferenceInformation");
501  QL_REQUIRE(tmp, "Need either a CreditCurveId or ReferenceInformation node in CreditDefaultSwapData");
502  CdsReferenceInformation ref;
503  ref.fromXML(tmp);
504  referenceInformation_ = ref;
505  creditCurveId_ = ref.id();
506  }
507 
508  settlesAccrual_ = XMLUtils::getChildValueAsBool(node, "SettlesAccrual", false);
509  rebatesAccrual_ = XMLUtils::getChildValueAsBool(node, "RebatesAccrual", false);
510 
511  protectionPaymentTime_ = PPT::atDefault;
512 
513  // for backwards compatibility only
514  if (auto c = XMLUtils::getChildNode(node, "PaysAtDefaultTime"))
515  if (!parseBool(XMLUtils::getNodeValue(c)))
516  protectionPaymentTime_ = PPT::atPeriodEnd;
517 
518  // new node overrides deprecated one, if both should be given
519  if (auto c = XMLUtils::getChildNode(node, "ProtectionPaymentTime")) {
520  if (XMLUtils::getNodeValue(c) == "atDefault")
521  protectionPaymentTime_ = PPT::atDefault;
522  else if (XMLUtils::getNodeValue(c) == "atPeriodEnd")
523  protectionPaymentTime_ = PPT::atPeriodEnd;
524  else if (XMLUtils::getNodeValue(c) == "atMaturity")
525  protectionPaymentTime_ = PPT::atMaturity;
526  else {
527  QL_FAIL("protection payment time '" << XMLUtils::getNodeValue(c) <<
528  "' not known, expected atDefault, atPeriodEnd, atMaturity");
529  }
530  }
531 
532  protectionStart_ = Date();
533  if (auto tmp = XMLUtils::getChildNode(node, "ProtectionStart"))
534  protectionStart_ = parseDate(XMLUtils::getNodeValue(tmp));
535 
536  upfrontDate_ = Date();
537  if (auto tmp = XMLUtils::getChildNode(node, "UpfrontDate"))
538  upfrontDate_ = parseDate(XMLUtils::getNodeValue(tmp));
539 
540  // zero if empty or missing
541  upfrontFee_ = Null<Real>();
542  string strUpfrontFee = XMLUtils::getChildValue(node, "UpfrontFee", false);
543  if (!strUpfrontFee.empty()) {
544  upfrontFee_ = parseReal(strUpfrontFee);
545  }
546 
547  if (upfrontDate_ == Date()) {
548  QL_REQUIRE(close_enough(upfrontFee_, 0.0) || upfrontFee_ == Null<Real>(),
549  "fromXML(): UpfronFee (" << upfrontFee_ << ") must be empty or zero if no upfront date is given");
550  upfrontFee_ = Null<Real>();
551  }
552 
553  // Recovery rate is Null<Real>() on a standard CDS i.e. if "FixedRecoveryRate" field is not populated.
554  recoveryRate_ = Null<Real>();
555  string strRecoveryRate = XMLUtils::getChildValue(node, "FixedRecoveryRate", false);
556  if (!strRecoveryRate.empty()) {
557  recoveryRate_ = parseReal(strRecoveryRate);
558  }
559 
560  tradeDate_ = Date();
561  if (auto n = XMLUtils::getChildNode(node, "TradeDate"))
562  tradeDate_ = parseDate(XMLUtils::getNodeValue(n));
563 
564  strCashSettlementDays_ = XMLUtils::getChildValue(node, "CashSettlementDays", false);
565  cashSettlementDays_ = strCashSettlementDays_.empty() ? 3 : parseInteger(strCashSettlementDays_);
566 
567  leg_.fromXML(XMLUtils::getChildNode(node, "LegData"));
568 }
