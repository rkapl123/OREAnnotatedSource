56 bool flatExtrapolation, bool inlcudeAtm, const vector<string>& tenors, const vector<string>& strikes,

97void CapFloorVolatilityCurveConfig::fromXML(XMLNode* node) {
98
99 XMLUtils::checkNode(node, "CapFloorVolatility");
100 curveID_ = XMLUtils::getChildValue(node, "CurveId", true);
101 curveDescription_ = XMLUtils::getChildValue(node, "CurveDescription", true);
102
103 if (auto p = XMLUtils::getChildNode(node, "ProxyConfig")) {
104 // read in proxy config
105
106 auto source = XMLUtils::getChildNode(p, "Source");
107 QL_REQUIRE(source != nullptr,
108 "CapFloorVolatilityCurveConfig (" << curveID_ << "): ProxyConfig requires child node 'Source'");
109 proxySourceCurveId_ = XMLUtils::getChildValue(source, "CurveId", true);
110 proxySourceIndex_ = XMLUtils::getChildValue(source, "Index", true);
111 proxySourceRateComputationPeriod_ =
112 parsePeriod(XMLUtils::getChildValue(source, "RateComputationPeriod", false, "0D"));
113
114 auto target = XMLUtils::getChildNode(p, "Target");
115 QL_REQUIRE(target != nullptr,
116 "CapFloorVolatilityCurveConfig (" << curveID_ << "): ProxyConfig requires child node 'Target'");
117 proxyTargetIndex_ = index_ = XMLUtils::getChildValue(target, "Index", true);
118 proxyTargetRateComputationPeriod_ = rateComputationPeriod_ =
119 parsePeriod(XMLUtils::getChildValue(target, "RateComputationPeriod", false, "0D"));
120 onCapSettlementDays_ = parseInteger(XMLUtils::getChildValue(target, "ONCapSettlementDays", false, "0"));
121
122 populateRequiredCurveIds();
123
124 } else {
125 // read in quote-based config
126
127 // Set the volatility type
128 configureVolatilityType(XMLUtils::getChildValue(node, "VolatilityType", true));
129
130 // Set the extrapolation variables
131 extrapolation_ = XMLUtils::getChildValue(node, "Extrapolation", true);
132 configureExtrapolation(extrapolation_);
133
134 // The mandatory variables
135 includeAtm_ = XMLUtils::getChildValueAsBool(node, "IncludeAtm", true);
136 calendar_ = parseCalendar(XMLUtils::getChildValue(node, "Calendar", true));
137 dayCounter_ = parseDayCounter(XMLUtils::getChildValue(node, "DayCounter", true));
138 businessDayConvention_ =
139 parseBusinessDayConvention(XMLUtils::getChildValue(node, "BusinessDayConvention", true));
140 if (auto iborNode = XMLUtils::getChildNode(node, "IborIndex")) {
141 WLOG("CapFloorVolatilityCurveConfig (" << curveID_
142 << "): The IborIndex node is deprecated, use Index instead.");
143 index_ = XMLUtils::getNodeValue(iborNode);
144 } else if (auto indexNode = XMLUtils::getChildNode(node, "Index")) {
145 index_ = XMLUtils::getNodeValue(indexNode);
146 } else {
147 QL_FAIL("CapFloorVOlatilityCurveConfig (" << curveID_
148 << "): Index node (or the deprecated IborIndex node) expected");
149 }
150 discountCurve_ = XMLUtils::getChildValue(node, "DiscountCurve", true);
151
152 // rate computation period, only required for OIS indices (which we don't check here)
153 rateComputationPeriod_ = 0 * Days;
154 if (auto rcpNode = XMLUtils::getChildNode(node, "RateComputationPeriod")) {
155 rateComputationPeriod_ = parsePeriod(XMLUtils::getNodeValue(rcpNode));
156 }
157
158 // on cap settlement days, optional and only relevant for OIS indices
159 onCapSettlementDays_ = 0;
160 if (auto onsNode = XMLUtils::getChildNode(node, "ONCapSettlementDays")) {
161 onCapSettlementDays_ = parseInteger(XMLUtils::getNodeValue(onsNode));
162 }
163
164 // Settlement days (optional)
165 settleDays_ = 0;
166 if (XMLNode* n = XMLUtils::getChildNode(node, "SettlementDays")) {
167 Integer d = parseInteger(XMLUtils::getNodeValue(n));
168 QL_REQUIRE(d >= 0, "SettlementDays (" << d << ") must be non-negative");
169 settleDays_ = static_cast<Natural>(d);
170 }
171
172 // Variable on which to interpolate (optional)
173 interpolateOn_ = "TermVolatilities";
174 if (XMLNode* n = XMLUtils::getChildNode(node, "InterpolateOn")) {
175 interpolateOn_ = XMLUtils::getNodeValue(n);
176 }
177
178 // Interpolation in time direction (optional)
179 timeInterpolation_ = "LinearFlat";
180 if (XMLNode* n = XMLUtils::getChildNode(node, "TimeInterpolation")) {
181 timeInterpolation_ = XMLUtils::getNodeValue(n);
182 }
183
184 // Interpolation in strike direction (optional)
185 strikeInterpolation_ = "LinearFlat";
186 if (XMLNode* n = XMLUtils::getChildNode(node, "StrikeInterpolation")) {
187 strikeInterpolation_ = XMLUtils::getNodeValue(n);
188 }
189
190 quoteIncludesIndexName_ = XMLUtils::getChildValueAsBool(node, "QuoteIncludesIndexName", false, false);
191
192 // Tenors and strikes. Optional as we may have an ATM curve and hence only AtmTenors.
193 tenors_ = XMLUtils::getChildrenValuesAsStrings(node, "Tenors", false);
194 strikes_ = XMLUtils::getChildrenValuesAsStrings(node, "Strikes", false);
195
196 // Optional flag, if set to true some tenor/strike quotes can be omitted
197 optionalQuotes_ = XMLUtils::getChildValueAsBool(node, "OptionalQuotes", false, false);
198
199 // Interpolation for cap floor term volatilities (optional)
200 interpolationMethod_ = "BicubicSpline";
201 if (XMLNode* n = XMLUtils::getChildNode(node, "InterpolationMethod")) {
202 interpolationMethod_ = XMLUtils::getNodeValue(n);
203 }
204
205 // Tenors for ATM volatilities.
206 atmTenors_ = XMLUtils::getChildrenValuesAsStrings(node, "AtmTenors", false);
207 QL_REQUIRE(!tenors_.empty() || !atmTenors_.empty(), "Tenors and AtmTenors cannot both be empty");
208 if (atmTenors_.empty()) {
209 atmTenors_ = tenors_;
210 }
211
212 // Optional bootstrap configuration
213 if (XMLNode* n = XMLUtils::getChildNode(node, "BootstrapConfig")) {
214 bootstrapConfig_.fromXML(n);
215 }
216
217 // Optional Input Type
218 inputType_ = "TermVolatilities";
219 if (XMLNode* n = XMLUtils::getChildNode(node, "InputType")) {
220 inputType_ = XMLUtils::getNodeValue(n);
221 }
222
223 // Set type_
224 configureType();
225
226 // Check that we have a valid configuration
227 validate();
228
229 // Populate quotes
230 populateQuotes();
231
232 // Populate required curve ids
233 populateRequiredCurveIds();
234 }
235
236 // Optional report config
237 if (auto tmp = XMLUtils::getChildNode(node, "Report")) {
238 reportConfig_.fromXML(tmp);
239 }
240}

242XMLNode* CapFloorVolatilityCurveConfig::toXML(XMLDocument& doc) {
243
244 XMLNode* node = doc.allocNode("CapFloorVolatility");
245 XMLUtils::addChild(doc, node, "CurveId", curveID_);
246 XMLUtils::addChild(doc, node, "CurveDescription", curveDescription_);
247
248 if (!proxySourceCurveId_.empty()) {
249 // write out proxy config
250 auto proxy = XMLUtils::addChild(doc, node, "ProxyConfig");
251 auto source = XMLUtils::addChild(doc, proxy, "Source");
252 auto target = XMLUtils::addChild(doc, proxy, "Target");
253 XMLUtils::addChild(doc, source, "CurveId", proxySourceCurveId_);
254 XMLUtils::addChild(doc, source, "Index", proxySourceIndex_);
255 XMLUtils::addChild(doc, target, "Index", proxyTargetIndex_);
256 if (proxySourceRateComputationPeriod_ != 0 * Days)
257 XMLUtils::addChild(doc, source, "RateComputationPeriod", proxySourceRateComputationPeriod_);
258 if (proxyTargetRateComputationPeriod_ != 0 * Days)
259 XMLUtils::addChild(doc, target, "RateComputationPeriod", proxyTargetRateComputationPeriod_);
260 } else {
261 // write out quote based config
262 XMLUtils::addChild(doc, node, "VolatilityType", toString(volatilityType_));
263 XMLUtils::addChild(doc, node, "Extrapolation", extrapolation_);
264 XMLUtils::addChild(doc, node, "InterpolationMethod", interpolationMethod_);
265 XMLUtils::addChild(doc, node, "IncludeAtm", includeAtm_);
266 XMLUtils::addChild(doc, node, "DayCounter", to_string(dayCounter_));
267 XMLUtils::addChild(doc, node, "Calendar", to_string(calendar_));
268 XMLUtils::addChild(doc, node, "BusinessDayConvention", to_string(businessDayConvention_));
269 XMLUtils::addGenericChildAsList(doc, node, "Tenors", tenors_);
270 XMLUtils::addGenericChildAsList(doc, node, "Strikes", strikes_);
271 XMLUtils::addChild(doc, node, "OptionalQuotes", optionalQuotes_);
272 XMLUtils::addChild(doc, node, "Index", index_);
273 if (rateComputationPeriod_ != 0 * Days) {
274 XMLUtils::addChild(doc, node, "RateComputationPeriod", rateComputationPeriod_);
275 }
276 if (onCapSettlementDays_ != 0) {
277 XMLUtils::addChild(doc, node, "ONCapSettlementDays", (int)onCapSettlementDays_);
278 }
279 XMLUtils::addChild(doc, node, "DiscountCurve", discountCurve_);
280 XMLUtils::addGenericChildAsList(doc, node, "AtmTenors", atmTenors_);
281 XMLUtils::addChild(doc, node, "SettlementDays", static_cast<int>(settleDays_));
282 XMLUtils::addChild(doc, node, "InterpolateOn", interpolateOn_);
283 XMLUtils::addChild(doc, node, "TimeInterpolation", timeInterpolation_);
284 XMLUtils::addChild(doc, node, "StrikeInterpolation", strikeInterpolation_);
285 XMLUtils::addChild(doc, node, "QuoteIncludesIndexName", quoteIncludesIndexName_);
286 XMLUtils::appendNode(node, bootstrapConfig_.toXML(doc));
287 XMLUtils::addChild(doc, node, "InputType", inputType_);
288 }
289
290 XMLUtils::appendNode(node, reportConfig_.toXML(doc));
291 return node;
292}

335void CapFloorVolatilityCurveConfig::populateQuotes() {
336
337 // Cap floor quotes are for the form:
338 // CAPFLOOR/(RATE_LNVOL|RATE_NVOL|RATE_SLNVOL)/<CCY>/<TENOR>/<IBOR_TENOR>/<ATM>/<RELATIVE>/<STRIKE> or
339 // CAPFLOOR/(RATE_LNVOL|RATE_NVOL|RATE_SLNVOL)/<CCY>/<IndexName>/<TENOR>/<IBOR_TENOR>/<ATM>/<RELATIVE>/<STRIKE>
340 string ccy = currency();
341 string tenor = indexTenor();
342
343 // Volatility quote stem
344 MarketDatum::QuoteType qType = quoteType();
345 string stem = "CAPFLOOR/" + to_string(qType) + "/" + ccy + "/";
346 if(quoteIncludesIndexName())
347 stem += index() + "/";
348
349 // Cap floor matrix quotes. So, ATM flag is false i.e. 0 and RELATIVE flag is false also as strikes are absolute.
350 for (const string& t : tenors_) {
351 for (const string& s : strikes_) {
352 quotes_.push_back(stem + t + "/" + tenor + "/0/0/" + s);
353 }
354 }
355
356 // ATM quotes. So, ATM flag is true i.e. 1 and RELATIVE flag is true with strike set to 0.
357 if (type_ == Type::TermAtm || type_ == Type::TermSurfaceWithAtm) {
358 for (const string& t : atmTenors_) {
359 quotes_.push_back(stem + t + "/" + tenor + "/1/1/0");
360 }
361 }
362
363 // Cap floor shift quote depends only on the currency and ibor tenor and is of the form:
364 // CAPFLOOR/SHIFT/<CCY>/<IBOR_TERM> or
365 // CAPFLOOR/SHIFT/<CCY>/<IndexName>/<IBOR_TERM> or
366 if (volatilityType_ == VolatilityType::ShiftedLognormal) {
367 quotes_.push_back("CAPFLOOR/SHIFT/" + ccy + "/" + (quoteIncludesIndexName() ? index() + "/" : "") + tenor);
368 }
369}

372 QL_REQUIRE(extrapolation == "Linear" || extrapolation == "Flat" || extrapolation == "None",

383void CapFloorVolatilityCurveConfig::configureType() {
384 if (inputType_ == "TermVolatilities")
385 type_ = strikes_.empty() ? Type::TermAtm : (includeAtm_ ? Type::TermSurfaceWithAtm : Type::TermSurface);
386 else if (inputType_ == "OptionletVolatilities")
387 type_ = strikes_.empty() ? Type::OptionletAtm : (includeAtm_ ? Type::OptionletSurfaceWithAtm : Type::OptionletSurface);
388 else
389 QL_FAIL("InputType " << inputType_ << " not supported");
390}

392void CapFloorVolatilityCurveConfig::validate() const {
393 QL_REQUIRE(interpolateOn_ == "TermVolatilities" || interpolateOn_ == "OptionletVolatilities",
394 "InterpolateOn (" << interpolateOn_ << ") must be TermVolatilities or OptionletVolatilities");
395 QL_REQUIRE(validInterps.count(timeInterpolation_) == 1,
396 "TimeInterpolation, " << timeInterpolation_ << ", not recognised");
397 QL_REQUIRE(validInterps.count(strikeInterpolation_) == 1,
398 "StrikeInterpolation, " << strikeInterpolation_ << ", not recognised");
399 QL_REQUIRE(strikeInterpolation_ != "BackwardFlat", "BackwardFlat StrikeInterpolation is not allowed");
400 if (!strikes_.empty()) {
401 QL_REQUIRE(!tenors_.empty(), "Tenors must be given for a surface (strikes are given)");
402 }
403}

405MarketDatum::QuoteType CapFloorVolatilityCurveConfig::quoteType() const {
406 switch (volatilityType_) {
407 case CapFloorVolatilityCurveConfig::VolatilityType::Lognormal:
408 return MarketDatum::QuoteType::RATE_LNVOL;
409 case CapFloorVolatilityCurveConfig::VolatilityType::ShiftedLognormal:
410 return MarketDatum::QuoteType::RATE_SLNVOL;
411 case CapFloorVolatilityCurveConfig::VolatilityType::Normal:
412 return MarketDatum::QuoteType::RATE_NVOL;
413 default:
414 QL_FAIL("Unknown VolatilityType (" << static_cast<int>(volatilityType_) << ")");
415 }
416}
