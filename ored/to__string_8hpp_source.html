59//! Convert vector to std::string
60/*!
61 Returns a vector into a single string, with elemenst separated by Period as a string as up to QuantLib 1.25, e.g. 13M is written as 1Y1M etc.
62
63 \ingroup utilities
64*/
65template <class T> std::string to_string(const std::vector<T>& vec, const std::string& sep = ",") {
66 std::ostringstream oss;
67 for (Size i = 0; i < vec.size(); ++i) {
68 oss << vec[i];
69 if (i < vec.size() - 1)
70 oss << sep;
71 }
72 return oss.str();
73}
74
75//! Convert set to std::string
76/*!
77 \ingroup utilities
78*/
79template <class T> std::string to_string(const std::set<T>& set, const std::string& sep = ",") {
80 std::ostringstream oss;
81 Size count = 1;
82 for (auto s: set) {
83 oss << s;
84 if (count < set.size())
85 oss << sep;
86 count++;
87 }
88 return oss.str();
89}

91//! Convert type to std::string
92/*!
93 Utility to give to_string() interface to classes and enums that have ostream<< operators defined.
94 \ingroup utilities
95*/
96template <class T> std::string to_string(const T& t) {
97 std::ostringstream oss;
98 oss << t;
99 return oss.str();
100}
