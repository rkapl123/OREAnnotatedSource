15boost::shared_ptr<CurrencyHedgedEquityIndexDecomposition>
16loadCurrencyHedgedIndexDecomposition(const std::string& name, const boost::shared_ptr<ReferenceDataManager>& refDataMgr,
17 const boost::shared_ptr<CurveConfigurations>& curveConfigs) {
18 boost::shared_ptr<CurrencyHedgedEquityIndexReferenceDatum> indexRefData;
19 boost::shared_ptr<EquityIndexReferenceDatum> underlyingRefData;
20 std::map<std::string, std::pair<double, std::string>> currencyWeightsAndFxIndexNames;
21
22 if (refDataMgr) {
23 try {
24 indexRefData = boost::dynamic_pointer_cast<CurrencyHedgedEquityIndexReferenceDatum>(
25 refDataMgr->getData("CurrencyHedgedEquityIndex", name));
26 } catch (...) {
27 // Index not a CurrencyHedgedEquityIndex or referencedata is missing, don't throw here
28 }
29
30 if (indexRefData != nullptr) {
31 try {
32 underlyingRefData = boost::dynamic_pointer_cast<EquityIndexReferenceDatum>(
33 refDataMgr->getData("EquityIndex", indexRefData->underlyingIndexName()));
34 } catch (...) {
35 // referencedata is missing, but don't throw here, return null ptr
36 }
37 }
38 }
39
40 if (indexRefData == nullptr || underlyingRefData == nullptr) {
41 return nullptr;
42 }
43
44 // Load currency Weights
45 std::map<std::string, double> currencyWeights;
46 std::string indexCurrency;
47 std::string underlyingIndexCurrency;
48 std::string fxIndexName;
49
50 // Get currency hedged index currency
51 if (curveConfigs && curveConfigs->hasEquityCurveConfig(indexRefData->id())) {
52 indexCurrency = curveConfigs->equityCurveConfig(indexRefData->id())->currency();
53 } else {
54 WLOG("Can not find curveConfig for " << indexRefData->id() << " and can not determine the index currecy");
55 return nullptr;
56 }
57
58
59 if (curveConfigs && curveConfigs->hasEquityCurveConfig(indexRefData->underlyingIndexName())) {
60
61 // Get Fx Index to convert UnderlyingIndexCCY into HedgedIndexCurrency
62 std::string underlyingIndexName = indexRefData->underlyingIndexName();
63 underlyingIndexCurrency =
64 curveConfigs->equityCurveConfig(indexRefData->underlyingIndexName())->currency();
65
66 auto fxIndexIt = indexRefData->fxIndexes().find(underlyingIndexCurrency);
67 if (fxIndexIt != indexRefData->fxIndexes().end()) {
68 fxIndexName = fxIndexIt->second;
69 } else {
70 fxIndexName = "FX-GENERIC-" + indexCurrency + "-" + underlyingIndexCurrency;
71 }
72
73 // Load the currency weights at referenceDate for hedge notionals
74 QuantLib::Date refDate =
75 CurrencyHedgedEquityIndexDecomposition::referenceDate(indexRefData, Settings::instance().evaluationDate());
76
77 std::vector<std::pair<std::string, double>> underlyingIndexWeightsAtRebalancing;
78
79 if (indexRefData->currencyWeights().empty()) {
80 boost::shared_ptr<ReferenceDatum> undIndexRefDataAtRefDate;
81 try {
82 undIndexRefDataAtRefDate = refDataMgr->getData("EquityIndex", underlyingIndexName, refDate);
83 } catch (...) {
84 // Try to load ref data, but don't throw on error
85 }
86 if (undIndexRefDataAtRefDate) {
87 underlyingIndexWeightsAtRebalancing =
88 boost::dynamic_pointer_cast<EquityIndexReferenceDatum>(undIndexRefDataAtRefDate)->underlyings();
89 }
90 } else {
91 underlyingIndexWeightsAtRebalancing = indexRefData->currencyWeights();
92 }
93
94 if (underlyingIndexWeightsAtRebalancing.empty()) {
95 currencyWeights[underlyingIndexCurrency] = 1.0;
96 } else {
97 for (const auto& [name, weight] : underlyingIndexWeightsAtRebalancing) {
98 // try look up currency in reference data and add if FX delta risk if necessary
99 if (curveConfigs->hasEquityCurveConfig(name)) {
100 auto ecc = curveConfigs->equityCurveConfig(name);
101 auto eqCcy = ecc->currency();
102 currencyWeights[eqCcy] += weight;
103 } else {
104 // Assume Index currency
105 currencyWeights[underlyingIndexCurrency] += weight;
106 }
107 }
108 }
109 }
110
111 for (const auto& [currency, weight] : currencyWeights) {
112 if (currency != indexCurrency) {
113 auto defaultIndexIt = indexRefData->fxIndexes().find(currency);
114 if (defaultIndexIt != indexRefData->fxIndexes().end()) {
115 currencyWeightsAndFxIndexNames[currency] = std::make_pair(weight, defaultIndexIt->second);
116 } else {
117 currencyWeightsAndFxIndexNames[currency] =
118 std::make_pair(weight, "FX-GENERIC-" + indexCurrency + "-" + currency);
119 }
120 }
121 }
122 return boost::make_shared<CurrencyHedgedEquityIndexDecomposition>(name, indexRefData, underlyingRefData,
123 indexCurrency, underlyingIndexCurrency,
124 fxIndexName, currencyWeightsAndFxIndexNames);
125}

127QuantLib::Date CurrencyHedgedEquityIndexDecomposition::referenceDate(
128 const boost::shared_ptr<CurrencyHedgedEquityIndexReferenceDatum>& refData,
129 const QuantLib::Date& asof) {
130 QuantLib::Date hedgingDate = CurrencyHedgedEquityIndexDecomposition::rebalancingDate(refData, asof);
131 if (hedgingDate == QuantLib::Date()) {
132 return QuantLib::Date();
133 } else {
134 return refData->hedgeCalendar().advance(hedgingDate, -refData->referenceDateOffset() * QuantLib::Days,
135 QuantLib::Preceding);
136 }
137}

139QuantLib::Date CurrencyHedgedEquityIndexDecomposition::rebalancingDate(
140 const boost::shared_ptr<CurrencyHedgedEquityIndexReferenceDatum>& refData,
141 const QuantLib::Date& asof) {
142 if (refData->rebalancingStrategy() ==
143 ore::data::CurrencyHedgedEquityIndexReferenceDatum::RebalancingDate::EndOfMonth) {
144 QuantLib::Date lastBusinessDayOfCurrentMonth = QuantLib::Date::endOfMonth(asof);
145 lastBusinessDayOfCurrentMonth =
146 refData->hedgeCalendar().adjust(lastBusinessDayOfCurrentMonth, QuantLib::Preceding);
147 if (asof == lastBusinessDayOfCurrentMonth) {
148 return asof;
149 } else {
150 return refData->hedgeCalendar().advance(QuantLib::Date(1, asof.month(), asof.year()), -1 * QuantLib::Days,
151 QuantLib::Preceding);
152 }
153 }
154 return QuantLib::Date();
155}

165std::map<std::string, double> CurrencyHedgedEquityIndexDecomposition::fxSpotRiskFromForwards(
166 const double quantity, const QuantLib::Date& asof, const boost::shared_ptr<ore::data::Market>& todaysMarket) const {
167
168 std::map<std::string, double> fxRisks;
169 auto indexCurve = todaysMarket->equityCurve(indexName());
170 auto underlyingCurve = todaysMarket->equityCurve(underlyingIndexName());
171 QuantLib::Date refDate = referenceDate(asof);
172 double adjustmentFactor = 1.0;
173 // If adjustement is daily, the fxForward notional will be adjusted by the relative return of the underlying index
174 if (indexRefData_->hedgeAdjustmentRule() == CurrencyHedgedEquityIndexReferenceDatum::HedgeAdjustment::Daily) {
175 adjustmentFactor = underlyingCurve->fixing(asof) / underlyingCurve->fixing(rebalancingDate(asof));
176 }
177 // Compute notionals and fxSpotRisks
178 for (const auto& [ccy, weightAndIndex] : currencyWeightsAndFxIndexNames()) {
179 double weight = weightAndIndex.first;
180 auto fxIndexFamily = parseFxIndex(fxIndexName())->familyName();
181 auto fxIndex = todaysMarket->fxIndex("FX-" + fxIndexFamily + "-" + underlyingIndexCurrency_ + "-" + indexCurrency_);
182 double forwardNotional =
183 quantity * adjustmentFactor * weight * indexCurve->fixing(refDate) / fxIndex->fixing(refDate);
184 fxRisks[ccy] = 0.01 * forwardNotional * fxIndex->fixing(asof);
185 }
186 return fxRisks;
187}

189double
190CurrencyHedgedEquityIndexDecomposition::unhedgedDelta(double hedgedDelta, const double quantity,
191 const QuantLib::Date& asof,
192 const boost::shared_ptr<ore::data::Market>& todaysMarket) const {
193 auto indexCurve = todaysMarket->equityCurve(indexName());
194 auto underlyingCurve = todaysMarket->equityCurve(underlyingIndexName());
195 QuantLib::Date rebalanceDt = rebalancingDate(asof);
196 auto fxIndexFamily = parseFxIndex(fxIndexName())->familyName();
197 auto fxIndex = todaysMarket->fxIndex("FX-" + fxIndexFamily + "-" + underlyingIndexCurrency_ + "-" + indexCurrency_);
198 // In case we have a option unit delta isnt one
199 double scaling = (hedgedDelta * 100 / quantity) / indexCurve->fixing(asof);
200 double fxReturn = fxIndex->fixing(asof) / fxIndex->fixing(rebalanceDt);
201 double underlyingIndexReturn = underlyingCurve->equitySpot()->value() / underlyingCurve->fixing(rebalanceDt);
202 double unhedgedDelta= indexCurve->fixing(rebalanceDt) * underlyingIndexReturn * fxReturn;
203 return scaling * 0.01 * quantity * unhedgedDelta;
204}

206void CurrencyHedgedEquityIndexDecomposition::addAdditionalFixingsForEquityIndexDecomposition(
207 const QuantLib::Date& asof, std::map<std::string, std::set<QuantLib::Date>>& fixings) const {
208 if (isValid()) {
209 QuantLib::Date rebalancingDt = rebalancingDate(asof);
210 QuantLib::Date referenceDt = referenceDate(asof);
211 fixings[IndexNameTranslator::instance().oreName(indexName())].insert(rebalancingDt);
212 fixings[IndexNameTranslator::instance().oreName(indexName())].insert(referenceDt);
213
214 IndexNameTranslator::instance().add(underlyingIndexName(), "EQ-" + underlyingIndexName());
215 fixings[IndexNameTranslator::instance().oreName(underlyingIndexName())].insert(rebalancingDt);
216 fixings[IndexNameTranslator::instance().oreName(underlyingIndexName())].insert(referenceDt);
217
218 fixings[fxIndexName()].insert(referenceDt);
219 fixings[fxIndexName()].insert(rebalancingDt);
220
221 for (const auto& [currency, name] : currencyWeightsAndFxIndexNames()) {
222 fixings[name.second].insert(referenceDt);
223 fixings[name.second].insert(rebalancingDt);
224 }
225 }
