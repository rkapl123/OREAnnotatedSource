51 void CommodityDigitalOption::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
52 
53  // Checks
54  QL_REQUIRE(payoff_ > 0, "Commodity digital option requires a positive quatity");
55  QL_REQUIRE(strike_ > 0, "Commodity digital option requires a positive strike");
56  QL_REQUIRE(optionData_.exerciseDates().size() == 1, "Invalid number of excercise dates");
57 
58  expiryDate_ = parseDate(optionData_.exerciseDates().front());
59 
60  // Populate the index_ in case the option is automatic exercise.
61  // Intentionally use null calendar because we will ask for index value on the expiry date without adjustment.
62  const boost::shared_ptr<Market>& market = engineFactory->market();
63  index_ = *market->commodityIndex(name_, engineFactory->configuration(MarketContext::pricing));
64  if (!isFuturePrice_ || *isFuturePrice_) {
65 
66  // Assume future price if isFuturePrice_ is not explicitly set or if it is and true.
67  auto index = *market->commodityIndex(name_, engineFactory->configuration(MarketContext::pricing));
68 
69  // If we are given an explicit future contract expiry date, use it, otherwise use option's expiry.
70  Date expiryDate;
71  if (futureExpiryDate_ != Date()) {
72  expiryDate = futureExpiryDate_;
73  } else {
74  // Get the expiry date of the option. This is the expiry date of the commodity future index.
75  const vector<string>& expiryDates = optionData_.exerciseDates();
76  QL_REQUIRE(expiryDates.size() == 1,
77  "Expected exactly one expiry date for CommodityDigitalOption but got " << expiryDates.size() << ".");
78  expiryDate = parseDate(expiryDates[0]);
79  }
80 
81  // Clone the index with the relevant expiry date.
82  index_ = index->clone(expiryDate);
83 
84  // Set the VanillaOptionTrade forwardDate_ if the index is a CommodityFuturesIndex - we possibly still have a
85  // CommoditySpotIndex at this point so check. Also, will only work for European exercise.
86  auto et = parseExerciseType(optionData_.style());
87  QL_REQUIRE(et == Exercise::European, "European style expected for CommodityDigitalOption");
88  if (et == Exercise::European && boost::dynamic_pointer_cast<CommodityFuturesIndex>(index_)) {
89  forwardDate_ = expiryDate;
90  }
91  }
92 
93  // Build digital as call or put spread
94  Real strikeSpread = strike_ * 0.01; // FIXME, what is a usual spread here, and where should we put it?
95  Real strike1 = strike_ - strikeSpread/2;
96  Real strike2 = strike_ + strikeSpread/2;
97  CommodityOption opt1(envelope(), optionData_, name_, currency_, 1.0, TradeStrike(strike1, currency_), isFuturePrice_, futureExpiryDate_);
98  CommodityOption opt2(envelope(), optionData_, name_, currency_, 1.0, TradeStrike(strike2, currency_),
99  isFuturePrice_, futureExpiryDate_);
100  opt1.build(engineFactory);
101  opt2.build(engineFactory);
102  boost::shared_ptr<Instrument> inst1 = opt1.instrument()->qlInstrument();
103  boost::shared_ptr<Instrument> inst2 = opt2.instrument()->qlInstrument();
104 
105  boost::shared_ptr<CompositeInstrument> composite = boost::make_shared<CompositeInstrument>();
106  // add and subtract such that the long call spread and long put spread have positive values
107  if (optionData_.callPut() == "Call") {
108  composite->add(inst1);
109  composite->subtract(inst2);
110  }
111  else if (optionData_.callPut() == "Put") {
112  composite->add(inst2);
113  composite->subtract(inst1);
114  }
115  else {
116  QL_FAIL("OptionType Call or Put required in CommodityDigitalOption " << id());
117  }
118 
119  Position::Type positionType = parsePositionType(optionData_.longShort());
120  Real bsIndicator = (positionType == QuantLib::Position::Long ? 1.0 : -1.0);
121  Real multiplier = payoff_ * bsIndicator / strikeSpread;
122  std::vector<boost::shared_ptr<Instrument>> additionalInstruments;
123  std::vector<Real> additionalMultipliers;
124  // FIXME: Do we need to retrieve the engine builder's configuration
125  string configuration = Market::defaultConfiguration;
126  Currency ccy = parseCurrencyWithMinors(currency_);
127  maturity_ = std::max(expiryDate_, addPremiums(additionalInstruments, additionalMultipliers, multiplier,
128  optionData_.premiumData(), -bsIndicator, ccy, engineFactory, configuration));
129 
130  instrument_ = boost::shared_ptr<InstrumentWrapper>(
131  new VanillaInstrument(composite, multiplier, additionalInstruments, additionalMultipliers));
132 
133  npvCurrency_ = currency_;
134  notional_ = payoff_;
135  notionalCurrency_ = currency_;
136 
137  // LOG the volatility if the trade expiry date is in the future.
138  if (expiryDate_ > Settings::instance().evaluationDate()) {
139  DLOG("Implied vol for " << tradeType_ << " on " << name_ << " with expiry " << expiryDate_
140  << " and strike " << strike_ << " is "
141  << market->commodityVolatility(name_)->blackVol(expiryDate_, strike_));
142  }
143 
144  additionalData_["payoff"] = payoff_;
145  additionalData_["strike"] = strike_;
146  additionalData_["optionType"] = optionData_.callPut();
147  additionalData_["strikeCurrency"] = currency_;
148 }

155 void CommodityDigitalOption::fromXML(XMLNode* node) {
156 
157  Trade::fromXML(node);
158 
159  XMLNode* commodityNode = XMLUtils::getChildNode(node, "CommodityDigitalOptionData");
160  QL_REQUIRE(commodityNode, "A commodity option needs a 'CommodityDigitalOptionData' node");
161 
162  optionData_.fromXML(XMLUtils::getChildNode(commodityNode, "OptionData"));
163 
164  name_ = XMLUtils::getChildValue(commodityNode, "Name", true);
165  currency_ = XMLUtils::getChildValue(commodityNode, "Currency", true);
166  strike_ = XMLUtils::getChildValueAsDouble(commodityNode, "Strike", true);
167  payoff_ = XMLUtils::getChildValueAsDouble(commodityNode, "Payoff", true);
168 
169  isFuturePrice_ = boost::none;
170  if (XMLNode* n = XMLUtils::getChildNode(commodityNode, "IsFuturePrice"))
171  isFuturePrice_ = parseBool(XMLUtils::getNodeValue(n));
172 
173  futureExpiryDate_ = Date();
174  if (XMLNode* n = XMLUtils::getChildNode(commodityNode, "FutureExpiryDate"))
175  futureExpiryDate_ = parseDate(XMLUtils::getNodeValue(n));
176 }

178 XMLNode* CommodityDigitalOption::toXML(XMLDocument& doc) {
179 
180  XMLNode* node = Trade::toXML(doc);
181 
182  XMLNode* commodityNode = doc.allocNode("CommodityDigitalOptionData");
183  XMLUtils::appendNode(node, commodityNode);
184 
185  XMLUtils::appendNode(commodityNode, optionData_.toXML(doc));
186 
187  XMLUtils::addChild(doc, commodityNode, "Name", name_);
188  XMLUtils::addChild(doc, commodityNode, "Currency", currency_);
189  XMLUtils::addChild(doc, commodityNode, "Strike", strike_);
190  XMLUtils::addChild(doc, commodityNode, "Payoff", payoff_);
191 
192  if (isFuturePrice_)
193  XMLUtils::addChild(doc, commodityNode, "IsFuturePrice", *isFuturePrice_);
194 
195  if (futureExpiryDate_ != Date())
196  XMLUtils::addChild(doc, commodityNode, "FutureExpiryDate", to_string(futureExpiryDate_));
197 
198  return node;
199 }
