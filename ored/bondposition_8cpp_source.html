47void BondPositionData::populateFromBondBasketReferenceData(const boost::shared_ptr<ReferenceDataManager>& ref) {
48 QL_REQUIRE(!identifier_.empty(), "BondPositionData::populateFromBondBasketReferenceData(): no identifier given");
49 if (!ref || !ref->hasData(BondBasketReferenceDatum::TYPE, identifier_)) {
50 DLOG("could not get BondBasketReferenceDatum for '" << identifier_ << "' leave data in trade unchanged");
51 } else {
52 DLOG("got BondBasketReferenceDatum for '" << identifier_ << "':");
53 auto r = boost::dynamic_pointer_cast<BondBasketReferenceDatum>(
54 ref->getData(BondBasketReferenceDatum::TYPE, identifier_));
55 QL_REQUIRE(r, "BondPositionData::populateFromBondBasketReferenceData(): internal error, could not cast "
56 "reference datum to expected type.");
57 underlyings_ = r->underlyingData();
58 DLOG("updated " << underlyings_.size() << " Underlying nodes.");
59 }

62void BondPosition::build(const boost::shared_ptr<ore::data::EngineFactory>& engineFactory) {
63 DLOG("BondPosition::build() called for " << id());
64
65 bonds_.clear();
66 weights_.clear();
67 bidAskAdjustments_.clear();
68 fxConversion_.clear();
69
70 data_ = originalData_;
71 data_.populateFromBondBasketReferenceData(engineFactory->referenceData());
72
73 QL_REQUIRE(!data_.underlyings().empty(), "BondPosition::build(): no underlyings given");
74
75 maturity_ = Date::minDate();
76 for (auto const& u : data_.underlyings()) {
77 bonds_.push_back(BondFactory::instance().build(engineFactory, engineFactory->referenceData(), u.name()));
78 weights_.push_back(u.weight());
79 bidAskAdjustments_.push_back(u.bidAskAdjustment());
80 maturity_ = std::max(bonds_.back().bond->maturityDate(), maturity_);
81 }
82
83 // get fx quotes
84
85 isSingleCurrency_ = true;
86 npvCurrency_ = bonds_.front().currency;
87 for (auto const& b : bonds_) {
88 fxConversion_.push_back(engineFactory->market()->fxSpot(b.currency + npvCurrency_,
89 engineFactory->configuration(MarketContext::pricing)));
90 if (npvCurrency_ != b.currency)
91 isSingleCurrency_ = false;
92 }
93
94 // set instrument
95 std::vector<boost::shared_ptr<QuantLib::Bond>> qlbonds;
96 std::transform(bonds_.begin(), bonds_.end(), std::back_inserter(qlbonds),
97 [](const BondBuilder::Result& d) { return d.bond; });
98 instrument_ = boost::make_shared<BondPositionInstrumentWrapper>(data_.quantity(), qlbonds, weights_,
99 bidAskAdjustments_, fxConversion_);
100
101 // leave legs empty, leave notional empty for the time being
102 notional_ = Null<Real>();
103 notionalCurrency_ = "";
104
105 // ISDA taxonomy: not a derivative, but define the asset class at least
106 // so that we can determine a TRS asset class that has a Bond position underlying
107 additionalData_["isdaAssetClass"] = string("Credit");
108 additionalData_["isdaBaseProduct"] = string("");
109 additionalData_["isdaSubProduct"] = string("");
110 additionalData_["isdaTransaction"] = string("");
111}

140BondPositionInstrumentWrapper::BondPositionInstrumentWrapper(

142 const std::vector<Real>& bidAskAdjustments, const std::vector<Handle<Quote>>& fxConversion)

160Real BondPositionInstrumentWrapper::NPV() const {
161 Real result = 0.0;
162 for (Size i = 0; i < bonds_.size(); ++i) {
163 // - divide by current notional, because weights are supposed to include any amortization factors
164 // - add bid ask adjustment to relative price in bond ccy
165 Real tmp = quantity_ * (bonds_[i]->NPV() / bonds_[i]->notional() + bidAskAdjustments_[i]);
166 if (!fxConversion_[i].empty()) {
167 tmp *= fxConversion_[i]->value();
168 }
169 result += tmp * weights_[i];
170 }
171 if (!npvCcyConversion_.empty()) {
172 result *= npvCcyConversion_->value();
173 }
174 return result;
175}
