55void EquityVolatilityCurveConfig::populateQuotes() {
56 // add quotes from all the volatility configs
57 for (auto vc : volatilityConfig_) {
58 if (boost::dynamic_pointer_cast<QuoteBasedVolatilityConfig>(vc)) {
59 // The quotes depend on the type of volatility structure that has been configured.
60 if (auto c = boost::dynamic_pointer_cast<ConstantVolatilityConfig>(vc)) {
61 quotes_.push_back(c->quote());
62 } else if (auto c = boost::dynamic_pointer_cast<VolatilityCurveConfig>(vc)) {
63 auto qs = c->quotes();
64 quotes_.insert(quotes_.end(), qs.begin(), qs.end());
65 } else if (auto c = boost::dynamic_pointer_cast<VolatilitySurfaceConfig>(vc)) {
66 // Clear the quotes_ if necessary and populate with surface quotes
67 string quoteStr;
68 string volType = to_string<MarketDatum::QuoteType>(c->quoteType());
69 for (const pair<string, string>& p : c->quotes()) {
70 if (p.first == "*" || p.second == "*") {
71 quoteStr = quoteStem(volType) + "*";
72 } else {
73 quoteStr = quoteStem(volType) + p.first + "/" + p.second;
74 }
75 quotes_.push_back(quoteStr);
76 }
77 }
78 }
79 }
80}

82void EquityVolatilityCurveConfig::populateRequiredCurveIds() {
83 for (auto vc : volatilityConfig_) {
84 if (auto p = boost::dynamic_pointer_cast<ProxyVolatilityConfig>(vc)) {
85 requiredCurveIds_[CurveSpec::CurveType::Equity].insert(p->proxyVolatilityCurve());
86 requiredCurveIds_[CurveSpec::CurveType::EquityVolatility].insert(p->proxyVolatilityCurve());
87 if (!p->fxVolatilityCurve().empty())
88 requiredCurveIds_[CurveSpec::CurveType::FXVolatility].insert(p->fxVolatilityCurve());
89 if (!p->correlationCurve().empty())
90 requiredCurveIds_[CurveSpec::CurveType::Correlation].insert(p->correlationCurve());
91 }
92 }

95void EquityVolatilityCurveConfig::fromXML(XMLNode* node) {
96 XMLUtils::checkNode(node, "EquityVolatility");
97
98 curveID_ = XMLUtils::getChildValue(node, "CurveId", true);
99 curveDescription_ = XMLUtils::getChildValue(node, "CurveDescription", true);
100 ccy_ = XMLUtils::getChildValue(node, "Currency", true);
101
102 calendar_ = XMLUtils::getChildValue(node, "Calendar", false);
103
104 XMLNode* n;
105 dayCounter_ = "A365";
106 if ((n = XMLUtils::getChildNode(node, "DayCounter")))
107 dayCounter_ = XMLUtils::getNodeValue(n);
108
109 solverConfig_ = OneDimSolverConfig();
110 if (XMLNode* n = XMLUtils::getChildNode(node, "OneDimSolverConfig")) {
111 solverConfig_.fromXML(n);
112 }
113
114 preferOutOfTheMoney_ = boost::none;
115 if (XMLNode* n = XMLUtils::getChildNode(node, "PreferOutOfTheMoney")) {
116 preferOutOfTheMoney_ = parseBool(XMLUtils::getNodeValue(n));
117 }
118
119 // In order to remain backward compatible, we first check for a dimension node
120 // If this is present we read the nodes as before but create volatilityConfigs from the inputs
121 // If no dimension nodes then we expect VolatilityConfig nodes - this is the preferred configuration
122 string dim = XMLUtils::getChildValue(node, "Dimension", false);
123 if (dim == "ATM" || dim == "Smile") {
124 vector<string> expiries = XMLUtils::getChildrenValuesAsStrings(node, "Expiries", true);
125 string strikeExtrapolation = "Flat";
126 string timeExtrapolation = "Flat";
127 XMLNode* timeNode = XMLUtils::getChildNode(node, "TimeExtrapolation");
128 if (timeNode) {
129 timeExtrapolation = XMLUtils::getChildValue(node, "TimeExtrapolation", true);
130 }
131 XMLNode* strikeNode = XMLUtils::getChildNode(node, "StrikeExtrapolation");
132 if (strikeNode) {
133 strikeExtrapolation = XMLUtils::getChildValue(node, "StrikeExtrapolation", true);
134 }
135 vector<string> strikes = XMLUtils::getChildrenValuesAsStrings(node, "Strikes", false);
136 if (dim == "ATM") {
137 QL_REQUIRE(strikes.size() == 0,
138 "Dimension ATM, but multiple strikes provided for EquityVolatility " << curveID_);
139 // if ATM create VolatilityCurveConfig which requires quotes to be provided
140 vector<string> quotes(expiries.size());
141 string quoteStem = "EQUITY_OPTION/RATE_LNVOL/" + curveID_ + "/" + ccy_ + "/";
142 if (expiries.size() == 1 && expiries.front() == "*") {
143 quotes[0] = (quoteStem + "*");
144 } else {
145 Size i = 0;
146 for (auto ex : expiries) {
147 quotes[i] = (quoteStem + ex + "/ATMF");
148 i++;
149 }
150 }
151 volatilityConfig_.push_back(
152 boost::make_shared<VolatilityCurveConfig>(quotes, timeExtrapolation, timeExtrapolation));
153 } else {
154 // if Smile create VolatilityStrikeSurfaceConfig
155 volatilityConfig_.push_back(boost::make_shared<VolatilityStrikeSurfaceConfig>(
156 strikes, expiries, "Linear", "Linear", true, timeExtrapolation, strikeExtrapolation));
157 }
158
159 } else if (dim == "") {
160 VolatilityConfigBuilder vcb;
161 vcb.fromXML(node);
162 volatilityConfig_ = vcb.volatilityConfig();
163 } else {
164 QL_FAIL("Only ATM and Smile dimensions, or Volatility Config supported for EquityVolatility " << curveID_);
165 }
166
167 if (auto tmp = XMLUtils::getChildNode(node, "Report")) {
168 reportConfig_.fromXML(tmp);
169 }
170
171 populateQuotes();
172 populateRequiredCurveIds();
173}

175XMLNode* EquityVolatilityCurveConfig::toXML(XMLDocument& doc) {
176
177 XMLNode* node = doc.allocNode("EquityVolatility");
178
179 XMLUtils::addChild(doc, node, "CurveId", curveID_);
180 XMLUtils::addChild(doc, node, "CurveDescription", curveDescription_);
181 XMLUtils::addChild(doc, node, "Currency", ccy_);
182 XMLUtils::addChild(doc, node, "DayCounter", dayCounter_);
183
184 XMLNode* vnode = doc.allocNode("VolatilityConfig");
185 for (auto vc : volatilityConfig_) {
186 XMLNode* n = vc->toXML(doc);
187 XMLUtils::appendNode(vnode, n);
188 }
189 XMLUtils::appendNode(node, vnode);
190
191 if (calendar_ != "NullCalendar")
192 XMLUtils::addChild(doc, node, "Calendar", calendar_);
193
194 if (!solverConfig_.empty())
195 XMLUtils::appendNode(node, solverConfig_.toXML(doc));
196
197 if (preferOutOfTheMoney_)
198 XMLUtils::addChild(doc, node, "PreferOutOfTheMoney", *preferOutOfTheMoney_);
199
200 XMLUtils::appendNode(node, reportConfig_.toXML(doc));
201
202 return node;
203}
