29 Strike parseStrike(const std::string& s) {
30  boost::regex m1("^(ATM|atm)");
31  boost::regex m1b("^(ATMF|atmf)");
32  boost::regex m2("^(ATM|atm)(\\+|\\-)([0-9]+[.]?[0-9]*)");
33  boost::regex m3("^(\\+|\\-)?([0-9]+[.]?[0-9]*)");
34  boost::regex m4("^(\\+|\\-)?([0-9]+[.]?[0-9]*)(d|D)");
35  boost::regex m4b("(d|D)");
36  boost::regex m5("^(\\+|\\-)?([0-9]+[.]?[0-9]*)(c|C)");
37  boost::regex m5b("^(c|C)");
38  boost::regex m6("^(\\+|\\-)?([0-9]+[.]?[0-9]*)(p|P)");
39  boost::regex m6b("^(p|P)");
40  boost::regex m7("^(\\+|\\-)?([0-9]+[.]?[0-9]*)(bf|BF)");
41  boost::regex m7b("^(bf|BF)");
42  boost::regex m8("^(\\+|\\-)?([0-9]+[.]?[0-9]*)(rr|RR)");
43  boost::regex m8b("^(rr|RR)");
44  boost::regex m9("^(\\+|\\-)?([0-9]+[.]?[0-9]*)(ATMF|atmf)");
45  boost::regex m9b("(ATMF|atmf)");
46  boost::regex m10("^(\\+|\\-)?([0-9]+[.]?[0-9]*)(ATM|atm)");
47  boost::regex m10b("(ATM|atm)");
48  Strike result;
49  if (boost::regex_match(s, m1)) {
50  result.type = Strike::Type::ATM;
51  result.value = 0.0;
52  return result;
53  }
54  if (boost::regex_match(s, m1b)) {
55  result.type = Strike::Type::ATMF;
56  result.value = 0.0;
57  return result;
58  }
59  if (boost::regex_match(s, m2)) {
60  result.type = Strike::Type::ATM_Offset;
61  result.value = parseReal(regex_replace(s, m1, std::string("")));
62  return result;
63  }
64  if (boost::regex_match(s, m3)) {
65  result.type = Strike::Type::Absolute;
66  result.value = parseReal(s);
67  return result;
68  }
69  if (boost::regex_match(s, m4)) {
70  result.type = Strike::Type::Delta;
71  result.value = parseReal(regex_replace(s, m4b, std::string("")));
72  return result;
73  }
74  if (boost::regex_match(s, m5)) {
75  result.type = Strike::Type::DeltaCall;
76  result.value = parseReal(regex_replace(s, m5b, std::string("")));
77  return result;
78  }
79  if (boost::regex_match(s, m6)) {
80  result.type = Strike::Type::DeltaPut;
81  result.value = parseReal(regex_replace(s, m6b, std::string("")));
82  return result;
83  }
84  if (boost::regex_match(s, m7)) {
85  result.type = Strike::Type::BF;
86  result.value = parseReal(regex_replace(s, m7b, std::string("")));
87  return result;
88  }
89  if (boost::regex_match(s, m8)) {
90  result.type = Strike::Type::RR;
91  result.value = parseReal(regex_replace(s, m8b, std::string("")));
92  return result;
93  }
94  if (boost::regex_match(s, m9)) {
95  result.type = Strike::Type::ATMF_Moneyness;
96  result.value = parseReal(regex_replace(s, m9b, std::string("")));
97  return result;
98  }
99  if (boost::regex_match(s, m10)) {
100  result.type = Strike::Type::ATM_Moneyness;
101  result.value = parseReal(regex_replace(s, m10b, std::string("")));
102  return result;
103  }
104  QL_FAIL("could not parse strike given by " << s);
105 }

117  break;

122  out << "Delta";
123  break;
124  case Strike::Type::ATMF_Moneyness:
125  out << "ATMF_Moneyness";
126  break;
127  case Strike::Type::ATM_Moneyness:
128  out << "ATM_Moneyness";
129  break;
130  default:
131  out << "UNKNOWN";
132  break;
133  }
134  if (s.type == Strike::Type::ATM_Offset || s.type == Strike::Type::Absolute || s.type == Strike::Type::Delta ||
135  s.type == Strike::Type::ATMF_Moneyness || s.type == Strike::Type::ATM_Moneyness) {
136  if (s.value >= 0.0)
137  out << "+";
138  else
139  out << "-";
140  out << s.value;
141  }
142  return out;
143 }

146 Strike normaliseStrike(const Strike& s) {
147  if (s.type == Strike::Type::ATM_Offset && QuantLib::close_enough(s.value, 0.0))
148  return Strike({Strike::Type::ATM, 0.0});
149  if (s.type == Strike::Type::ATMF_Moneyness && QuantLib::close_enough(s.value, 1.0))
150  return Strike({Strike::Type::ATMF, 0.0});
151  if (s.type == Strike::Type::ATM_Moneyness && QuantLib::close_enough(s.value, 1.0))
152  return Strike({Strike::Type::ATM, 0.0});
153  return s;
154 }

163 QuantLib::Real computeAbsoluteStrike(const Strike& s, const QuantLib::Real atm, const QuantLib::Real atmf) {
164  switch (s.type) {
165  case Strike::Type::ATM:
166  return atm;
167  case Strike::Type::ATMF:
168  return atmf;
169  case Strike::Type::ATM_Offset:
170  return atm + s.value;
171  case Strike::Type::Absolute:
172  return s.value;
173  case Strike::Type::Delta:
174  QL_FAIL("can not compute absolute strike for type delta");
175  case Strike::Type::ATMF_Moneyness:
176  return atmf * s.value;
177  case Strike::Type::ATM_Moneyness:
178  return atm * s.value;
179  default:
180  QL_FAIL("can not compute strike for unknown type " << s);
181  }
182 }

184 DeltaString::DeltaString(const std::string& s) {
185  QL_REQUIRE(!s.empty() && (s.back() == 'P' || s.back() == 'C' || s == "ATM"),
186  "invalid delta quote, expected ATM, 10P, 25C, ...");
187  isAtm_ = s == "ATM";
188  isPut_ = !s.empty() && s.back() == 'P';
189  isCall_ = !s.empty() && s.back() == 'C';
190  if (isPut_ || isCall_) {
191  try {
192  delta_ = ore::data::parseReal(s.substr(0, s.size() - 1)) / 100.0 * (isPut_ ? -1.0 : 1.0);
193  } catch (const std::exception& e) {
194  QL_FAIL("DeltaString: can not convert call / put delta '" << s << "' to numeric value: " << e.what());
195  }
196  }
197 }
