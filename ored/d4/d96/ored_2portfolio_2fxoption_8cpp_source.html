35 void FxOption::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
36 
37  const boost::shared_ptr<Market>& market = engineFactory->market();
38 
39  // If automatic exercise, check that we have a non-empty FX index string, parse it and attach curves from market.
40  if (option_.isAutomaticExercise()) {
41 
42  QL_REQUIRE(!fxIndex_.empty(),
43  "FX option trade " << id() << " has automatic exercise so the FXIndex node needs to be populated.");
44 
45  // The strike is the number of units of sold currency (currency_) per unit of bought currency (assetName_).
46  // So, the convention here is that the sold currency is domestic and the bought currency is foreign.
47  // Note: intentionally use null calendar and 0 day fixing lag here because we will ask the FX index for its
48  // value on the expiry date without adjustment.
49  index_ = buildFxIndex(fxIndex_, currency_, assetName_, market,
50  engineFactory->configuration(MarketContext::pricing), "NullCalendar", 0);
51 
52  // Populate the external index name so that fixings work.
53  indexName_ = fxIndex_;
54  }
55 
56  // Build the trade using the shared functionality in the base class.
57  VanillaOptionTrade::build(engineFactory);
58 
59  // LOG the volatility if the trade expiry date is in the future.
60  if (expiryDate_ > Settings::instance().evaluationDate()) {
61  const string& ccyPairCode = assetName_ + currency_;
62  DLOG("Implied vol for " << tradeType_ << " on " << ccyPairCode << " with expiry " << expiryDate_
63  << " and strike " << std::setprecision(6) << strike_ << " is "
64  << market->fxVol(ccyPairCode)->blackVol(expiryDate_, strike_));
65  }
66 
67  additionalData_["boughtCurrency"] = assetName_;
68  additionalData_["boughtAmount"] = quantity_;
69  additionalData_["soldCurrency"] = currency_;
70  additionalData_["soldAmount"] = quantity_ * strike_;
71 }

73 void FxOption::fromXML(XMLNode* node) {
74  VanillaOptionTrade::fromXML(node);
75  XMLNode* fxNode = XMLUtils::getChildNode(node, "FxOptionData");
76  QL_REQUIRE(fxNode, "No FxOptionData Node");
77  option_.fromXML(XMLUtils::getChildNode(fxNode, "OptionData"));
78  assetName_ = XMLUtils::getChildValue(fxNode, "BoughtCurrency", true);
79  currency_ = XMLUtils::getChildValue(fxNode, "SoldCurrency", true);
80  double boughtAmount = XMLUtils::getChildValueAsDouble(fxNode, "BoughtAmount", true);
81  double soldAmount = XMLUtils::getChildValueAsDouble(fxNode, "SoldAmount", true);
82  strike_ = soldAmount / boughtAmount;
83  quantity_ = boughtAmount;
84  fxIndex_ = XMLUtils::getChildValue(fxNode, "FXIndex", false);
85  QL_REQUIRE(boughtAmount > 0.0, "positive BoughtAmount required");
86  QL_REQUIRE(soldAmount > 0.0, "positive SoldAmount required");
87 }
