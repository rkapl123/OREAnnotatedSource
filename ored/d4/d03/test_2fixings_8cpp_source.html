52 // Give back the expected results for the tests on the various trade types below
53 // The results are read in from the file "test_trade_types_expected.csv"
54 map<tuple<string, string, bool, bool>, map<string, set<Date>>> tradeTypeExpected() {
55 
56  static map<tuple<string, string, bool, bool>, map<string, set<Date>>> exp;
57 
58  string expResultsFile = TEST_INPUT_FILE("test_trade_types_expected.csv");
59  CSVFileReader reader(expResultsFile, true, ",");
60 
61  while (reader.next()) {
62  string tradeType = reader.get("trade_type");
63  string tradeCase = reader.get("trade_case");
64  bool includeSettlementDateFlows = parseBool(reader.get("isdf"));
65  bool enforcesTodaysHistoricFixings = parseBool(reader.get("ethf"));
66  string indexName = reader.get("index_name");
67 
68  string datesList = reader.get("dates");
69  vector<string> strDates;
70  boost::split(strDates, datesList, boost::is_any_of("|"));
71  set<Date> dates;
72  for (const auto& strDate : strDates) {
73  dates.insert(parseDate(strDate));
74  }
75 
76  auto key = make_tuple(tradeType, tradeCase, includeSettlementDateFlows, enforcesTodaysHistoricFixings);
77  if (exp.count(key) == 0) {
78  exp[key] = {};
79  }
80  exp[key][indexName] = dates;
81  }
82 
83  return exp;
84 }

86 // Give back the dummy fixings keyed on [Index Name, Date] pair. We will load chosen elements
87 // from this map at the end of the trade tests below to check that the trade prices.
88 map<tuple<string, Date>, Fixing> dummyFixings() {
89 
90  static map<tuple<string, Date>, Fixing> result;
91 
92  string dummyFixingsFile = TEST_INPUT_FILE("test_trade_types_dummy_fixings.csv");
93  CSVFileReader reader(dummyFixingsFile, true, ",");
94 
95  while (reader.next()) {
96  string name = reader.get("name");
97  Date date = parseDate(reader.get("date"));
98  Real fixing = parseReal(reader.get("value"));
99 
100  result.emplace(make_pair(name, date), Fixing(date, name, fixing));
101  }
102 
103  return result;
104 }

106 // Load the requested fixings
107 void loadFixings(const map<string, set<Date>>& requestedFixings, const boost::shared_ptr<Conventions>& conventions) {
108 
109  // Get the dummy fixings that we have provided in the input directory
110  auto fixingValues = dummyFixings();
111 
112  // Fetch the relevant fixings using the requestedFixings argument
113  vector<Fixing> relevantFixings;
114  for (const auto& kv : requestedFixings) {
115  for (const auto& d : kv.second) {
116  relevantFixings.push_back(fixingValues.at(make_pair(kv.first, d)));
117  }
118  }
119 
120  // Add the fixings in QuantLib index manager
121  applyFixings(relevantFixings, conventions);
122 }

186 BOOST_DATA_TEST_CASE_F(F, testTradeTypes,
187  bdata::make(tradeTypes) * bdata::make(tradeCases) * bdata::make(bools) * bdata::make(bools),
188  tradeType, tradeCase, includeSettlementDateFlows, enforcesTodaysHistoricFixings) {
189 
190  // Set the flag determining what happens if fixings are required today
191  Settings::instance().enforcesTodaysHistoricFixings() = enforcesTodaysHistoricFixings;
192 
193  // Set the flag determining what happens when cashflows happen today
194  Settings::instance().includeTodaysCashFlows() = includeSettlementDateFlows;
195 
196  // Read in the trade
197  Portfolio p;
198  string portfolioFile = "trades/" + tradeType + "/" + tradeCase + ".xml";
199  p.load(TEST_INPUT_FILE(portfolioFile));
200  BOOST_REQUIRE_MESSAGE(p.size() == 1, "Expected portfolio to contain a single trade");
201 
202  // Ask for fixings before trades are built should return empty set
203  auto m = p.fixings(today);
204  BOOST_CHECK_MESSAGE(m.empty(), "Expected fixings to be empty when trades not built");
205 
206  // Build the portfolio and retrieve the fixings
207  p.build(engineFactory);
208  m = p.fixings(today);
209 
210  // Check the retrieved fixings against the expected results
211  auto exp = tradeTypeExpected();
212  auto key = make_tuple(tradeType, tradeCase, includeSettlementDateFlows, enforcesTodaysHistoricFixings);
213  if (exp.count(key) == 0) {
214  // Expected result is no required fixings
215  BOOST_CHECK_MESSAGE(m.empty(), "Expected no required fixings for ["
216  << tradeType << ", " << tradeCase << ", "
217  << ore::data::to_string(includeSettlementDateFlows) << ", "
218  << ore::data::to_string(enforcesTodaysHistoricFixings)
219  << "] but got a map containing " << m.size() << " indices");
220 
221  // Trade should not throw if we ask for NPV
222  BOOST_CHECK_NO_THROW(p.trades()[0]->instrument()->NPV());
223 
224  } else {
225  // Check the retrieved fixings against the expected fixings
226  auto expMap = exp.at(key);
227  BOOST_CHECK_EQUAL(expMap.size(), m.size());
228  for (const auto& kv : expMap) {
229  BOOST_CHECK_MESSAGE(m.count(kv.first), "Could not find index " << kv.first << " in retrieved fixings");
230  BOOST_CHECK_EQUAL_COLLECTIONS(kv.second.begin(), kv.second.end(), m.at(kv.first).begin(),
231  m.at(kv.first).end());
232  }
233 
234  // Trade should throw if we ask for NPV and have not added the fixings
235  // If it is the zciis trade, it won't throw because the fixings were added for the bootstrap
236  if (tradeType != "zciis_with_interp" && tradeType != "cpi_swap_with_interp") {
237  BOOST_CHECK_THROW(p.trades()[0]->instrument()->NPV(), Error);
238  }
239 
240  // Add the fixings
241  loadFixings(m, conventions);
242 
243  // Trade should now not throw when we try to price it
244  BOOST_CHECK_NO_THROW(p.trades()[0]->instrument()->NPV());
245  }
246 }

300  // Add yoy inflation curves
