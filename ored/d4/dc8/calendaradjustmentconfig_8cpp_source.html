96 void CalendarAdjustmentConfig::fromXML(XMLNode* node) {
97  XMLUtils::checkNode(node, "CalendarAdjustments");
98 
99  //we loop once skipping any new calendars
100  for (auto calnode : XMLUtils::getChildrenNodes(node, "Calendar")) {
101  string calname = XMLUtils::getAttribute(calnode, "name");
102  string baseCalendar = XMLUtils::getChildValue(calnode, "BaseCalendar", false);
103  if (baseCalendar != "") {
104  // we check here if the baseCalendar is in the map before the new calendars are added
105  // this is because we don't want any new calendars to be defined in terms of other new calendars
106  parseCalendar(baseCalendar);
107  continue;
108  }
109  Calendar cal = parseCalendar(calname);
110 
111  vector<string> holidayDates = XMLUtils::getChildrenValues(calnode, "AdditionalHolidays", "Date");
112  for (auto holiday : holidayDates) {
113  Date h = parseDate(holiday);
114  addHolidays(calname, h);
115  cal.addHoliday(h);
116  }
117  vector<string> businessDates = XMLUtils::getChildrenValues(calnode, "AdditionalBusinessDays", "Date");
118  for (auto businessDay : businessDates) {
119  Date b = parseDate(businessDay);
120  addBusinessDays(calname, b);
121  cal.removeHoliday(b);
122  }
123  }
124  //then loop again adding the new calendars
125  for (auto calnode : XMLUtils::getChildrenNodes(node, "Calendar")) {
126  string calname = XMLUtils::getAttribute(calnode, "name");
127  string baseCalendar = XMLUtils::getChildValue(calnode, "BaseCalendar", false);
128  if (baseCalendar == "")
129  continue;
130  Calendar cal = parseCalendar(baseCalendar, calname);
131 
132  vector<string> holidayDates = XMLUtils::getChildrenValues(calnode, "AdditionalHolidays", "Date");
133  for (auto holiday : holidayDates) {
134  Date h = parseDate(holiday);
135  addHolidays(calname, h);
136  cal.addHoliday(h);
137  }
138  vector<string> businessDates = XMLUtils::getChildrenValues(calnode, "AdditionalBusinessDays", "Date");
139  for (auto businessDay : businessDates) {
140  Date b = parseDate(businessDay);
141  addBusinessDays(calname, b);
142  cal.removeHoliday(b);
143  }
144 
145  addBaseCalendar(calname, baseCalendar);
146  }
147 
148 }

150 XMLNode* CalendarAdjustmentConfig::toXML(XMLDocument& doc) {
151  XMLNode* node = doc.allocNode("CalendarAdjustments");
152  for (auto cal : getCalendars()) {
153  XMLNode* calendarNode = XMLUtils::addChild(doc, node, "Calendar");
154  XMLUtils::addAttribute(doc, calendarNode, "name", cal);
155  if (getBaseCalendar(cal) != "")
156  XMLUtils::addChild(doc, calendarNode, "BaseCalendar", getBaseCalendar(cal));
157  XMLNode* ahd = XMLUtils::addChild(doc, calendarNode, "AdditionalHolidays");
158  for (auto hol : getHolidays(cal)) {
159  XMLUtils::addChild(doc, ahd, "Date", to_string(hol));
160  }
161  XMLNode* abd = XMLUtils::addChild(doc, calendarNode, "AdditionalBusinessDays");
162  for (auto bd : getBusinessDays(cal)) {
163  XMLUtils::addChild(doc, abd, "Date", to_string(bd));
164  }
165  }
166  return node;
167 }
