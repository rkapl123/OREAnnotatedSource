31 DefaultCurveConfig::DefaultCurveConfig(
32  const string& curveID, const string& curveDescription, const string& currency, const Type& type,
33  const string& discountCurveID, const string& recoveryRateQuote, const DayCounter& dayCounter,
34  const string& conventionID, const std::vector<std::pair<std::string, bool>>& cdsQuotes, bool extrapolation,
35  const string& benchmarkCurveID, const string& sourceCurveID, const std::vector<string>& pillars,
36  const Calendar& calendar, const Size spotLag, const Date& startDate, const BootstrapConfig& bootstrapConfig,
37  QuantLib::Real runningSpread, const boost::optional<bool>& implyDefaultFromMarket, const bool allowNegativeRates)
38  : CurveConfig(curveID, curveDescription), cdsQuotes_(cdsQuotes), currency_(currency), type_(type),
39  discountCurveID_(discountCurveID), recoveryRateQuote_(recoveryRateQuote), dayCounter_(dayCounter),
40  conventionID_(conventionID), extrapolation_(extrapolation), benchmarkCurveID_(benchmarkCurveID),
41  sourceCurveID_(sourceCurveID), pillars_(pillars), calendar_(calendar), spotLag_(spotLag), startDate_(startDate),
42  bootstrapConfig_(bootstrapConfig), runningSpread_(runningSpread), implyDefaultFromMarket_(implyDefaultFromMarket),
43  allowNegativeRates_(allowNegativeRates) {
44 
45  for (const auto& kv : cdsQuotes) {
46  quotes_.push_back(kv.first);
47  }
48  quotes_.insert(quotes_.begin(), recoveryRateQuote_);
49 
50  if (type_ != Type::SpreadCDS && startDate_ != Date()) {
51  WLOG("'StartDate' is only used when type is 'SpreadCDS'");
52  }
53 
54  populateRequiredCurveIds();
55 }

57 void DefaultCurveConfig::populateRequiredCurveIds() {
58  if (!discountCurveID().empty())
59  requiredCurveIds_[CurveSpec::CurveType::Yield].insert(parseCurveSpec(discountCurveID())->curveConfigID());
60  if (!benchmarkCurveID().empty())
61  requiredCurveIds_[CurveSpec::CurveType::Yield].insert(parseCurveSpec(benchmarkCurveID())->curveConfigID());
62  if (!sourceCurveID().empty())
63  requiredCurveIds_[CurveSpec::CurveType::Yield].insert(parseCurveSpec(sourceCurveID())->curveConfigID());
64  for (auto const& s : multiSectionSourceCurveIds_) {
65  if (!s.empty())
66  requiredCurveIds_[CurveSpec::CurveType::Default].insert(parseCurveSpec(s)->curveConfigID());
67  }

70 void DefaultCurveConfig::fromXML(XMLNode* node) {
71  XMLUtils::checkNode(node, "DefaultCurve");
72 
73  // Just in case
74  cdsQuotes_.clear();
75  quotes_.clear();
76 
77  curveID_ = XMLUtils::getChildValue(node, "CurveId", true);
78  curveDescription_ = XMLUtils::getChildValue(node, "CurveDescription", true);
79  currency_ = XMLUtils::getChildValue(node, "Currency", true);
80 
81  string type = XMLUtils::getChildValue(node, "Type", true);
82  if (type == "SpreadCDS") {
83  type_ = Type::SpreadCDS;
84  } else if (type == "HazardRate") {
85  type_ = Type::HazardRate;
86  } else if (type == "Price") {
87  type_ = Type::Price;
88  } else if (type == "Benchmark") {
89  type_ = Type::Benchmark;
90  } else if (type == "MultiSection") {
91  type_ = Type::MultiSection;
92  } else {
93  QL_FAIL("Type " << type << " not recognized");
94  }
95 
96  string dc = XMLUtils::getChildValue(node, "DayCounter", true);
97  dayCounter_ = parseDayCounter(dc);
98  extrapolation_ = XMLUtils::getChildValueAsBool(node, "Extrapolation"); // defaults to true
99 
100  allowNegativeRates_ = false;
101  if (XMLNode* n = XMLUtils::getChildNode(node, "AllowNegativeRates")) {
102  allowNegativeRates_ = parseBool(XMLUtils::getNodeValue(n));
103  }
104 
105  if (type_ == Type::Benchmark) {
106  benchmarkCurveID_ = XMLUtils::getChildValue(node, "BenchmarkCurve", true);
107  sourceCurveID_ = XMLUtils::getChildValue(node, "SourceCurve", true);
108  pillars_ = XMLUtils::getChildrenValuesAsStrings(node, "Pillars", true);
109  spotLag_ = parseInteger(XMLUtils::getChildValue(node, "SpotLag", true));
110  calendar_ = parseCalendar(XMLUtils::getChildValue(node, "Calendar", true));
111  discountCurveID_ = conventionID_ = "";
112  recoveryRateQuote_ = XMLUtils::getChildValue(node, "RecoveryRate", false);
113  // we allow for a numeric recovery rate, in which case we do not add it to the quotes
114  Real dummy;
115  if (!tryParseReal(recoveryRateQuote_, dummy))
116  quotes_.insert(quotes_.end(), recoveryRateQuote_);
117  } else if (type_ == Type::MultiSection) {
118  multiSectionSourceCurveIds_ = XMLUtils::getChildrenValues(node, "SourceCurves", "SourceCurve", true);
119  multiSectionSwitchDates_ = XMLUtils::getChildrenValues(node, "SwitchDates", "SwitchDate", true);
120  discountCurveID_ = conventionID_ = "";
121  recoveryRateQuote_ = XMLUtils::getChildValue(node, "RecoveryRate", false);
122  // we allow for a numeric recovery rate, in which case we do not add it to the quotes
123  Real dummy;
124  if (!tryParseReal(recoveryRateQuote_, dummy))
125  quotes_.insert(quotes_.end(), recoveryRateQuote_);
126  } else {
127  discountCurveID_ = XMLUtils::getChildValue(node, "DiscountCurve", false);
128  conventionID_ = XMLUtils::getChildValue(node, "Conventions", true);
129 
130  XMLNode* quotesNode = XMLUtils::getChildNode(node, "Quotes");
131  if (quotesNode) {
132  for (auto n : XMLUtils::getChildrenNodes(quotesNode, "Quote")) {
133  string attr = XMLUtils::getAttribute(n, "optional");
134  bool opt = (!attr.empty() && parseBool(attr));
135  cdsQuotes_.emplace_back(make_pair(XMLUtils::getNodeValue(n), opt));
136  quotes_.push_back(XMLUtils::getNodeValue(n));
137  }
138  }
139 
140  recoveryRateQuote_ = XMLUtils::getChildValue(node, "RecoveryRate", false);
141  quotes_.insert(quotes_.begin(), recoveryRateQuote_);
142  benchmarkCurveID_ = sourceCurveID_ = "";
143  calendar_ = Calendar();
144  spotLag_ = 0;
145  pillars_.clear();
146 
147  // Read the optional start date
148  string d = XMLUtils::getChildValue(node, "StartDate", false);
149  if (d != "") {
150  if (type_ == Type::SpreadCDS || type_ == Type::Price) {
151  startDate_ = parseDate(d);
152  } else {
153  WLOG("'StartDate' is only used when type is 'SpreadCDS' or 'Price'");
154  }
155  }
156 
157  string s = XMLUtils::getChildValue(node, "RunningSpread", false);
158  if (s.empty() && type_ == Type::Price) {
159  DLOG("'RunningSpread' is empty and type is 'Price' for default curve " << curveID_ <<
160  " so the running spread will need to be provided in the market quote.");
161  }
162  if (s != "") {
163  if (type_ == Type::Price) {
164  runningSpread_ = parseReal(s);
165  } else {
166  WLOG("'RunningSpread' is only used when type is 'Price'");
167  }
168  }
169 
170  implyDefaultFromMarket_ = boost::none;
171  if (XMLNode* n = XMLUtils::getChildNode(node, "ImplyDefaultFromMarket"))
172  implyDefaultFromMarket_ = parseBool(XMLUtils::getNodeValue(n));
173 
174  // Optional bootstrap configuration
175  if (XMLNode* n = XMLUtils::getChildNode(node, "BootstrapConfig")) {
176  bootstrapConfig_.fromXML(n);
177  }
178  }
179 
180  populateRequiredCurveIds();
181 }

183 XMLNode* DefaultCurveConfig::toXML(XMLDocument& doc) {
184  XMLNode* node = doc.allocNode("DefaultCurve");
185 
186  XMLUtils::addChild(doc, node, "CurveId", curveID_);
187  XMLUtils::addChild(doc, node, "CurveDescription", curveDescription_);
188  XMLUtils::addChild(doc, node, "Currency", currency_);
189 
190  if (type_ == Type::SpreadCDS || type_ == Type::HazardRate || type_ == Type::Price) {
191  if (type_ == Type::SpreadCDS) {
192  XMLUtils::addChild(doc, node, "Type", "SpreadCDS");
193  } else if (type_ == Type::HazardRate) {
194  XMLUtils::addChild(doc, node, "Type", "HazardRate");
195  } else {
196  XMLUtils::addChild(doc, node, "Type", "Price");
197  }
198  XMLUtils::addChild(doc, node, "DiscountCurve", discountCurveID_);
199  XMLUtils::addChild(doc, node, "DayCounter", to_string(dayCounter_));
200  XMLUtils::addChild(doc, node, "RecoveryRate", recoveryRateQuote_);
201  XMLNode* quotesNode = XMLUtils::addChild(doc, node, "Quotes");
202  for (auto q : cdsQuotes_) {
203  XMLNode* qNode = doc.allocNode("Quote", q.first);
204  if (q.second)
205  XMLUtils::addAttribute(doc, qNode, "optional", "true");
206  XMLUtils::appendNode(quotesNode, qNode);
207  }
208  } else if (type_ == Type::Benchmark) {
209  XMLUtils::addChild(doc, node, "Type", "Benchmark");
210  XMLUtils::addChild(doc, node, "DayCounter", to_string(dayCounter_));
211  XMLUtils::addChild(doc, node, "RecoveryRate", recoveryRateQuote_);
212  XMLUtils::addChild(doc, node, "BenchmarkCurve", benchmarkCurveID_);
213  XMLUtils::addChild(doc, node, "SourceCurve", sourceCurveID_);
214  XMLUtils::addGenericChildAsList(doc, node, "Pillars", pillars_);
215  XMLUtils::addChild(doc, node, "SpotLag", (int)spotLag_);
216  XMLUtils::addChild(doc, node, "Calendar", calendar_.name());
217  } else if (type_ == Type::MultiSection) {
218  XMLUtils::addChild(doc, node, "RecoveryRate", recoveryRateQuote_);
219  XMLUtils::addChildren(doc, node, "SourceCurves", "SourceCurve", multiSectionSourceCurveIds_);
220  XMLUtils::addChildren(doc, node, "SwitchDates", "SwitchDate", multiSectionSwitchDates_);
221  } else {
222  QL_FAIL("Unkown type in DefaultCurveConfig::toXML()");
223  }
224  XMLUtils::addChild(doc, node, "Conventions", conventionID_);
225  XMLUtils::addChild(doc, node, "Extrapolation", extrapolation_);
226 
227  if (startDate_ != Date())
228  XMLUtils::addChild(doc, node, "StartDate", to_string(startDate_));
229 
230  if (runningSpread_ != QuantLib::Null<Real>())
231  XMLUtils::addChild(doc, node, "RunningSpread", to_string(runningSpread_));
232 
233  if (implyDefaultFromMarket_)
234  XMLUtils::addChild(doc, node, "ImplyDefaultFromMarket", *implyDefaultFromMarket_);
235 
236  XMLUtils::appendNode(node, bootstrapConfig_.toXML(doc));
237 
238  XMLUtils::addChild(doc, node, "AllowNegativeRates", allowNegativeRates_);
239 
240  return node;
241 }
