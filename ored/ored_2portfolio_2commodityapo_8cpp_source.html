63void CommodityAveragePriceOption::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
64
65 reset();
66
67
68 DLOG("CommodityAveragePriceOption::build() called for trade " << id());
69
70
71 QL_REQUIRE(gearing_ > 0.0, "Gearing (" << gearing_ << ") should be positive.");
72 QL_REQUIRE(spread_ < strike_ || QuantLib::close_enough(spread_, strike_),
73 "Spread (" << spread_ << ") should be less than strike (" << strike_ << ").");
74
75 // Allow exercise dates not to be specified for an APO. In this case, the exercise date is deduced below when
76 // building the APO or a standard option.
77 Date exDate;
78 if (!optionData_.exerciseDates().empty()) {
79 QL_REQUIRE(optionData_.exerciseDates().size() == 1, "Commodity average price option must be European");
80 exDate = parseDate(optionData_.exerciseDates().front());
81 }
82
83 // Just to get the configuration string for now
84 boost::shared_ptr<EngineBuilder> builder = engineFactory->builder(
85 barrierData_.initialized() ? "CommodityAveragePriceBarrierOption" : "CommodityAveragePriceOption");
86 string configuration = builder->configuration(MarketContext::pricing);
87
88 // Build the leg. In this method the allAveraging_ flag is set to true if the future itself is averaging and we
89 // can then use a standard commodity option pricer below.
90 Leg leg = buildLeg(engineFactory, configuration);
91
92 // Notional = effective_quantity * effective_strike = (G x Q) x ((K - s) / G) = Q x (K - s)
93 notional_ = quantity_ * (strike_ - spread_);
94 notionalCurrency_ = currency_;
95
96 // Based on allAveraging_ flag, set up a standard or averaging commodity option
97 npvCurrency_ = currency_;
98 if (allAveraging_) {
99 buildStandardOption(engineFactory, leg, exDate);
100 } else {
101 buildApo(engineFactory, leg, exDate, builder);
102 }
103
104 // Add leg to legs_ so that fixings method can work.
105 legs_.push_back(leg);
106 legPayers_.push_back(false);
107 legCurrencies_.push_back(currency_);
108
109 additionalData_["quantity"] = quantity_;
110 additionalData_["strike"] = strike_;
111 additionalData_["strikeCurrency"] = currency_;
112
113 // ISDA taxonomy, assuming Commodity follows the Equity template
114 additionalData_["isdaAssetClass"] = std::string("Commodity");
115 additionalData_["isdaBaseProduct"] = std::string("Option");
116 additionalData_["isdaSubProduct"] = std::string("Price Return Basic Performance");
117 // skip the transaction level mapping for now
118 additionalData_["isdaTransaction"] = std::string();
119}

217Leg CommodityAveragePriceOption::buildLeg(const boost::shared_ptr<EngineFactory>& engineFactory,
218 const string& configuration) {
219
220 // Create the ScheduleData for use in the LegData. Tenor is not needed.
221 ScheduleData scheduleData(ScheduleDates("NullCalendar", "Unadjusted", "", {startDate_, endDate_}));
222
223 // Create the CommodityFloatingLegData. Want to generate a single averaging commodity coupon.
224 vector<Real> quantities{quantity_};
225 vector<string> quantityDates;
226 vector<Real> spreads{spread_};
227 vector<string> spreadDates;
228 vector<Real> gearings{gearing_};
229 vector<string> gearingDates;
230 vector<string> pricingDates;
231 bool isAveraged = true;
232 bool isInArrears = false;
233 boost::shared_ptr<CommodityFloatingLegData> commLegData = boost::make_shared<CommodityFloatingLegData>(
234 name_, priceType_, quantities, quantityDates, commodityQuantityFrequency_, commodityPayRelativeTo_, spreads,
235 spreadDates, gearings, gearingDates, CommodityPricingDateRule::FutureExpiryDate, pricingCalendar_, 0,
236 pricingDates, isAveraged, isInArrears, futureMonthOffset_, deliveryRollDays_, includePeriodEnd_, true,
237 QuantLib::Null<QuantLib::Natural>(), true, "", QuantLib::Null<QuantLib::Natural>(), false, QuantLib::Null<QuantLib::Natural>(),
238 fxIndex_);
239
240 // Create the LegData. All defaults are as in the LegData ctor.
241 vector<string> paymentDates = paymentDate_.empty() ? vector<string>() : vector<string>(1, paymentDate_);
242 LegData legData(commLegData, true, currency_, scheduleData, "", vector<Real>(), vector<string>(),
243 paymentConvention_, false, false, false, true, "", 0, "", vector<AmortizationData>(),
244 paymentLag_, paymentCalendar_, paymentDates);
245
246 // Get the leg builder, set the allAveraging_ flag and build the leg
247 auto legBuilder = engineFactory->legBuilder(legData.legType());
248 boost::shared_ptr<CommodityFloatingLegBuilder> cflb =
249 boost::dynamic_pointer_cast<CommodityFloatingLegBuilder>(legBuilder);
250 QL_REQUIRE(cflb, "Expected a CommodityFloatingLegBuilder for leg type " << legData.legType());
251 Leg leg = cflb->buildLeg(legData, engineFactory, requiredFixings_, configuration);
252 allAveraging_ = cflb->allAveraging();
253
254 return leg;
255}

257void CommodityAveragePriceOption::buildStandardOption(const boost::shared_ptr<EngineFactory>& engineFactory,
258 const Leg& leg, Date exerciseDate) {
259
260 QL_REQUIRE(!barrierData_.initialized(), "Commodity APO: standard option does not support barriers");
261
262 // Expect the leg to hold a single commodity averaging cashflow on which the option is written
263 QL_REQUIRE(leg.size() == 1, "Single flow expected but found " << leg.size());
264 auto flow = boost::dynamic_pointer_cast<CommodityIndexedCashFlow>(leg[0]);
265 QL_REQUIRE(flow, "Expected a cashflow of type CommodityIndexedCashFlow");
266
267 // If exercise date is given use it. If not given, take the cashflow's pricing date.
268 if (exerciseDate != Date()) {
269 QL_REQUIRE(exerciseDate >= flow->pricingDate(),
270 "Exercise date, " << io::iso_date(exerciseDate) << ", should be on or after the pricing date, "
271 << io::iso_date(flow->pricingDate()));
272 DLOG("buildStandardOption: explicit exercise date given for APO " << io::iso_date(exerciseDate) << ".");
273 } else {
274 exerciseDate = flow->pricingDate();
275 optionData_.setExerciseDates({to_string(exerciseDate)});
276 DLOG("buildStandardOption: set exercise date on APO to cashflow's pricing date " << io::iso_date(exerciseDate)
277 << ".");
278 }
279 DLOG("buildStandardOption: pricing date on APO is " << io::iso_date(flow->pricingDate()) << ".");
280
281 // Update the optionData_ if necessary
282 if (!optionData_.automaticExercise()) {
283 optionData_.setAutomaticExercise(true);
284 DLOG("buildStandardOption: setting automatic exercise to true on APO.");
285 }
286
287 if (!optionData_.paymentData()) {
288 QL_REQUIRE(exerciseDate <= flow->date(), "Exercise date, " << io::iso_date(exerciseDate)
289 << ", should be on or before payment date, "
290 << io::iso_date(flow->date()));
291 string strDate = to_string(flow->date());
292 optionData_.setPaymentData(OptionPaymentData({strDate}));
293 DLOG("buildStandardOption: setting payment date to " << strDate << " on APO.");
294 } else {
295 DLOG("buildStandardOption: using explicitly provided payment data on APO.");
296 }
297
298 // Build the commodity option.
299 TradeStrike effectiveStrike((strike_ - spread_) / gearing_, currency_);
300 Real effectiveQuantity = gearing_ * quantity_;
301 CommodityOption commOption(envelope(), optionData_, name_, currency_, effectiveQuantity, effectiveStrike,
302 flow->index()->isFuturesIndex(), flow->pricingDate());
303 commOption.build(engineFactory);
304 instrument_ = commOption.instrument();
305 maturity_ = commOption.maturity();
306}

308void CommodityAveragePriceOption::buildApo(const boost::shared_ptr<EngineFactory>& engineFactory, const Leg& leg,
309 Date exerciseDate, const boost::shared_ptr<EngineBuilder>& builder) {
310
311 // Expect the leg to hold a single commodity averaging cashflow on which the option is written
312 QL_REQUIRE(leg.size() == 1, "Single flow expected but found " << leg.size());
313 auto apoFlow = boost::dynamic_pointer_cast<CommodityIndexedAverageCashFlow>(leg[0]);
314 QL_REQUIRE(apoFlow, "Expected a cashflow of type CommodityIndexedAverageCashFlow");
315 Date lastApoFixingDate = apoFlow->indices().rbegin()->first;
316
317 // If exercise date is given use it. If not given, take the cashflow's last pricing date.
318 if (exerciseDate != Date()) {
319 QL_REQUIRE(exerciseDate >= lastApoFixingDate, "Exercise date, "
320 << io::iso_date(exerciseDate)
321 << ", should be on or after the last APO fixing date, "
322 << io::iso_date(lastApoFixingDate));
323 DLOG("buildApo: explicit exercise date given for APO " << io::iso_date(exerciseDate) << ".");
324 } else {
325 exerciseDate = lastApoFixingDate;
326 string strDate = to_string(lastApoFixingDate);
327 optionData_.setExerciseDates({strDate});
328 DLOG("buildApo: set exercise date on APO to cashflow's last pricing date " << io::iso_date(lastApoFixingDate)
329 << ".");
330 }
331 DLOG("buildApo: pricing date on APO is " << io::iso_date(lastApoFixingDate) << ".");
332
333 QL_REQUIRE(exerciseDate <= apoFlow->date(), "Exercise date, " << io::iso_date(exerciseDate)
334 << ", should be on or before payment date, "
335 << io::iso_date(apoFlow->date()));
336
337 // If the apo is payout and the underlying are quoted in different currencies, handle the fxIndex
338 boost::shared_ptr<FxIndex> fxIndex;
339 if(!fxIndex_.empty()) {
340 auto underlyingCcy =
341 boost::dynamic_pointer_cast<CommodityIndexedAverageCashFlow>(leg[0])->index()->priceCurve()->currency();
342 // check currencies consistency
343 QL_REQUIRE(npvCurrency_ == underlyingCcy.code() || npvCurrency_ == currency_, "Commodity cross-currency APO: inconsistent currencies in trade.");
344
345 // only add an fx index if underlying currency differs from trade currency
346 if (npvCurrency_ != underlyingCcy.code()) {
347 fxIndex = buildFxIndex(fxIndex_, npvCurrency_, underlyingCcy.code(), engineFactory->market(),
348 engineFactory->configuration(MarketContext::pricing));
349 for (auto cf : leg) { // request appropriate fixings
350 auto cacf = boost::dynamic_pointer_cast<CommodityIndexedAverageCashFlow>(cf);
351 for (auto kv : cacf->indices()) {
352 if (!fxIndex->fixingCalendar().isBusinessDay(
353 kv.first)) { // If fx index is not available for the commodity pricing day,
354 // this ensures to require the previous valid one which will be used in pricing
355 // from fxIndex()->fixing(...)
356 Date adjustedFixingDate = fxIndex->fixingCalendar().adjust(kv.first, Preceding);
357 requiredFixings_.addFixingDate(adjustedFixingDate, fxIndex_);
358
359 } else {
360 requiredFixings_.addFixingDate(kv.first, fxIndex_);
361 }
362 }
363 }
364 }
365 }
366
367 // get barrier info
368 Real barrierLevel = Null<Real>();
369 Barrier::Type barrierType = Barrier::DownIn;
370 Exercise::Type barrierStyle = Exercise::American;
371 if (barrierData_.initialized()) {
372 QL_REQUIRE(barrierData_.levels().size() == 1, "Commodity APO: Excepted exactly one barrier level.");
373 barrierLevel = barrierData_.levels().front().value();
374 barrierType = parseBarrierType(barrierData_.type());
375 if (!barrierData_.style().empty()) {
376 barrierStyle = parseExerciseType(barrierData_.style());
377 QL_REQUIRE(barrierStyle == Exercise::European || barrierStyle == Exercise::American,
378 "Commodity APO: Expected 'European' or 'American' as barrier style");
379 }
380 }
381
382 // Create the APO instrument
383 boost::shared_ptr<QuantLib::Exercise> exercise = boost::make_shared<EuropeanExercise>(exerciseDate);
384 auto apo = boost::make_shared<QuantExt::CommodityAveragePriceOption>(
385 apoFlow, exercise, apoFlow->periodQuantity(), strike_, parseOptionType(optionData_.callPut()),
386 Settlement::Physical, Settlement::PhysicalOTC, barrierLevel, barrierType, barrierStyle, fxIndex);
387
388 // Set the pricing engine
389 Currency ccy = parseCurrency(currency_);
390 auto engineBuilder = boost::dynamic_pointer_cast<CommodityApoBaseEngineBuilder>(builder);
391 boost::shared_ptr<PricingEngine> engine = engineBuilder->engine(ccy, name_, id(), apo);
392 apo->setPricingEngine(engine);
393
394 // position type and trade multiplier
395 Position::Type positionType = parsePositionType(optionData_.longShort());
396 Real multiplier = positionType == Position::Long ? 1.0 : -1.0;
397
398 // Take care of fees
399 vector<boost::shared_ptr<Instrument>> additionalInstruments;
400 vector<Real> additionalMultipliers;
401 Date lastPremiumDate = addPremiums(additionalInstruments, additionalMultipliers, multiplier,
402 optionData_.premiumData(), positionType == Position::Long ? -1.0 : 1.0, ccy,
403 engineFactory, engineBuilder->configuration(MarketContext::pricing));
404
405 // Populate instrument wrapper
406 instrument_ = boost::make_shared<VanillaInstrument>(apo, multiplier, additionalInstruments, additionalMultipliers);
407
408 // Populate relevant Trade members
409 maturity_ = std::max(lastPremiumDate, apoFlow->date());
410}
