32 bool operator<(const MarketDatum& a, const MarketDatum& b) {
33  if (a.asofDate() != b.asofDate())
34  return a.asofDate() < b.asofDate();
35  // the name determines the instrument and quote type uniquely by construction
36  return a.name() < b.name();
37 }

138 Natural yearFromExpiryString(const std::string& expiry) {
139  QL_REQUIRE(expiry.length() == 7, "The expiry string must be of "
140  "the form YYYY-MM");
141  string strExpiryYear = expiry.substr(0, 4);
142  Natural expiryYear;
143  try {
144  expiryYear = lexical_cast<Natural>(strExpiryYear);
145  } catch (const bad_lexical_cast&) {
146  QL_FAIL("Could not convert year string, " << strExpiryYear << ", to number.");
147  }
148  return expiryYear;
149 }

151 Month monthFromExpiryString(const std::string& expiry) {

173 QuantLib::Size SeasonalityQuote::applyMonth() const {
174  QL_REQUIRE(month_.length() == 3, "The month string must be of "
175  "the form MMM");
176  std::vector<std::string> allMonths = {"JAN", "FEB", "MAR", "APR", "MAY", "JUN",
177  "JUL", "AUG", "SEP", "OCT", "NOV", "DEC"};
178  QuantLib::Size applyMonth;
179  auto it = std::find(allMonths.begin(), allMonths.end(), month_);
180  if (it != allMonths.end()) {
181  applyMonth = std::distance(allMonths.begin(), it) + 1;
182  } else {
183  QL_FAIL("Unknown month string: " << month_);
184  }
185  return applyMonth;
186 }

201 CorrelationQuote::CorrelationQuote(Real value, const Date& asof, const string& name, QuoteType quoteType,
202  const string& index1, const string& index2, const string& expiry,
203  const string& strike)
204  : MarketDatum(value, asof, name, quoteType, InstrumentType::CORRELATION), index1_(index1), index2_(index2),
205  expiry_(expiry), strike_(strike) {
206 
207  // If strike is not ATM, it must parse to Real
208  if (strike != "ATM") {
209  Real result;
210  QL_REQUIRE(tryParseReal(strike_, result),
211  "Commodity option quote strike (" << strike_ << ") must be either ATM or an actual strike price");
212  }
213 
214  // Call parser to check that the expiry_ resolves to a period or a date
215  Date outDate;
216  Period outPeriod;
217  bool outBool;
218  parseDateOrPeriod(expiry_, outDate, outPeriod, outBool);
219 
220  if (outBool)
221  QL_REQUIRE(asof <= outDate, "CorrelationQuote: Invalid CorrelationQuote, expiry date "
222  << outDate << " must be after asof date " << asof);
223 }

225 template <class Archive> void MarketDatum::serialize(Archive& ar, const unsigned int version) {
226  Real value;
227  // save / load the value of the quote, do not try to serialize the quote as such
228  if (Archive::is_saving::value) {
229  value = quote_->value();
230  ar& value;
231  } else {
232  ar& value;
233  quote_ = Handle<Quote>(boost::make_shared<SimpleQuote>(value));
234  }
235  ar& asofDate_;
236  ar& name_;
237  ar& instrumentType_;
238  ar& quoteType_;
239 }
