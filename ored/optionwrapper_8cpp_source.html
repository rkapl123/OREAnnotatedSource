45 void OptionWrapper::initialise(const vector<Date>& dateGrid) {
46  // set "effective" exercise dates which get used to determine exercise during cube valuation
47  // this is necessary since there is no guarantee that actual exercise dates are included in
48  // the valuation date grid
49  Date today = Settings::instance().evaluationDate();
50  for (Size i = 0; i < contractExerciseDates_.size(); ++i) {
51  effectiveExerciseDates_[i] = Null<Date>();
52  if (contractExerciseDates_[i] > today && contractExerciseDates_[i] <= dateGrid.back()) {
53  // Find the effective exercise date in our grid. We simulate the exercise just after the actual
54  // exercise.This ensures that the QL instrument's NPV is a proper continuation value, i.e.
55  // does not contain the possibility of exercising in to the underlying on the current exercise
56  // date and can therefore it be used as such in the exercise decision made in the exercise()
57  // method of derived classes.
58  auto it = std::lower_bound(dateGrid.begin(), dateGrid.end(), contractExerciseDates_[i]);
59  effectiveExerciseDates_[i] = *it;
60  }
61  }
62 }

69 Real OptionWrapper::NPV() const {
70  Real addNPV = additionalInstrumentsNPV();
71 
72  Date today = Settings::instance().evaluationDate();
73  if (!exercised_) {
74  for (Size i = 0; i < effectiveExerciseDates_.size(); ++i) {
75  if (today == effectiveExerciseDates_[i]) {
76  if (exercise()) {
77  exercised_ = true;
78  exerciseDate_ = today;
79  }
80  }
81  }
82  }
83  if (exercised_) {
84  // if exercised, return underlying npv for physical settlement and also for
85  // cash settlement if we are still on the exercise date (since the cash
86  // settlement takes place strictly after the exercise date usually)
87  // FIXME: we assume that the settlement date lies strictly after the exercise
88  // date, but before or on the next simulation date. Check this explicitly
89  // by introducing the cash settlement date into the option wrapper (note
90  // that we will probably need an effective cash settlement date then to
91  // maintain the relative position to the effective exercise date).
92  Real npv = (isPhysicalDelivery_ || today == exerciseDate_)
93  ? (isLong_ ? 1.0 : -1.0) * getTimedNPV(activeUnderlyingInstrument_) * undMultiplier_
94  : 0.0;
95  return npv + addNPV;
96  } else {
97  // if not exercised we just return the original option's NPV
98  Real npv = (isLong_ ? 1.0 : -1.0) * getTimedNPV(instrument_) * multiplier_;
99  return npv + addNPV;
100  }
101 }

116 bool EuropeanOptionWrapper::exercise() const {
117  if (!exercisable_)
118  return false;
119 
120  // for European Exercise, we only require that underlying has positive PV
121  bool res = getTimedNPV(activeUnderlyingInstrument_) * undMultiplier_ > 0.0;

125 bool AmericanOptionWrapper::exercise() const {
126  if (!exercisable_)
127  return false;
128 
129  if (Settings::instance().evaluationDate() == effectiveExerciseDates_.back()) {
130  bool res = getTimedNPV(activeUnderlyingInstrument_) * undMultiplier_ > 0.0;
131  return res;
132  } else {
133  bool res = getTimedNPV(activeUnderlyingInstrument_) * undMultiplier_ > getTimedNPV(instrument_) * multiplier_;
134  return res;
135  }

138 bool BermudanOptionWrapper::exercise() const {
139  if (!exercisable_)
140  return false;
141 
142  // set active underlying instrument
143  Date today = Settings::instance().evaluationDate();
144  for (Size i = 0; i < effectiveExerciseDates_.size(); ++i) {
145  if (today == effectiveExerciseDates_[i]) {
146  activeUnderlyingInstrument_ = underlyingInstruments_[i];
147  break;
148  }
149  }
150  bool exercise = getTimedNPV(activeUnderlyingInstrument_) * undMultiplier_ > getTimedNPV(instrument_) * multiplier_;
151  return exercise;
