33void CompositeTrade::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
34 DLOG("Building Composite Trade: " << id());
35 npvCurrency_ = currency_;
36 boost::shared_ptr<MultiCcyCompositeInstrument> compositeInstrument =
37 boost::make_shared<MultiCcyCompositeInstrument>();
38 fxRates_.clear();
39 fxRatesNotional_.clear();
40 legs_.clear();
41 for (const boost::shared_ptr<Trade>& trade : trades_) {
42
43 trade->reset();
44 trade->build(engineFactory);
45 trade->validate();
46
47 Handle<Quote> fx = Handle<Quote>(boost::make_shared<SimpleQuote>(1.0));
48 if (trade->npvCurrency() != npvCurrency_)
49 fx = engineFactory->market()->fxRate(trade->npvCurrency() + npvCurrency_);
50 fxRates_.push_back(fx);
51
52 Handle<Quote> fxNotional = Handle<Quote>(boost::make_shared<SimpleQuote>(1.0));
53 if (trade->notionalCurrency().empty()) {
54 // trade is not guaranteed to provide a non-null notional, but if it does we require a notional currency
55 if (trade->notional() != Null<Real>()) {
56 ALOG(StructuredTradeErrorMessage(
57 trade, "Error building composite trade '" + id() + "'",
58 "Component trade '" + trade->id() + "' does not provide notional currency for notional " +
59 std::to_string(trade->notional()) + ". Assuming " + npvCurrency_ + "."));
60 }
61 } else if (trade->notionalCurrency() != npvCurrency_)
62 fxNotional = engineFactory->market()->fxRate(trade->notionalCurrency() + npvCurrency_);
63 fxRatesNotional_.push_back(fxNotional);
64
65 boost::shared_ptr<InstrumentWrapper> instrumentWrapper = trade->instrument();
66 Real effectiveMultiplier = instrumentWrapper->multiplier();
67 if (auto optionWrapper = boost::dynamic_pointer_cast<ore::data::OptionWrapper>(instrumentWrapper)) {
68 effectiveMultiplier *= optionWrapper->isLong() ? 1.0 : -1.0;
69 }
70
71 compositeInstrument->add(instrumentWrapper->qlInstrument(), effectiveMultiplier, fx);
72 for (Size i = 0; i < instrumentWrapper->additionalInstruments().size(); ++i) {
73 compositeInstrument->add(instrumentWrapper->additionalInstruments()[i],
74 instrumentWrapper->additionalMultipliers()[i]);
75 }
76
77 // For cashflows
78 legs_.insert(legs_.end(), trade->legs().begin(), trade->legs().end());
79 legPayers_.insert(legPayers_.end(), trade->legPayers().begin(), trade->legPayers().end());
80 legCurrencies_.insert(legCurrencies_.end(), trade->legCurrencies().begin(), trade->legCurrencies().end());
81
82 maturity_ = std::max(maturity_, trade->maturity());
83 }
84 instrument_ = boost::shared_ptr<InstrumentWrapper>(new VanillaInstrument(compositeInstrument));
85
86 notionalCurrency_ = npvCurrency_;
87
88 // set required fixings
89 for (auto const& t : trades_)
90 requiredFixings_.addData(t->requiredFixings());
91}

93QuantLib::Real CompositeTrade::notional() const {
94 vector<Real> notionals;
95 vector<Handle<Quote>> fxRates;
96 // trade is not guaranteed to provide a non-null notional
97 for (const boost::shared_ptr<Trade>& trade : trades_)
98 notionals.push_back(trade->notional() != Null<Real>() ? trade->notional() : 0.0);
99
100 // need to convert the component notionals to the composite currency.
101 auto notionalConverter = [](const Real ntnl, const Handle<Quote>& fx) { return (ntnl * fx->value()); };
102 std::transform(begin(notionals), end(notionals), begin(fxRates_), begin(notionals), notionalConverter);
103 return calculateNotional(notionals);
104}

106void CompositeTrade::fromXML(XMLNode* node) {
107 QL_REQUIRE("CompositeTrade" == XMLUtils::getChildValue(node, "TradeType", true),
108 "Wrong trade type in composite trade builder.");
109 Trade::fromXML(node);
110 this->id() = XMLUtils::getAttribute(node, "id");
111
112 // We read the data particular to composite trades
113 XMLNode* compNode = XMLUtils::getChildNode(node, "CompositeTradeData");
114 QL_REQUIRE(compNode, "Could not find CompositeTradeData node.");
115 currency_ = XMLUtils::getChildValue(compNode, "Currency", true);
116 // The notional logic is as follows:
117 // If the notional override is specified then it is used, regardless of the "NotionalCalculation" field
118 // Otherwise we calculate the notional as per the calculation specified
119 if (XMLUtils::getChildNode(compNode, "NotionalOverride") != nullptr) {
120 notionalOverride_ = XMLUtils::getChildValueAsDouble(compNode, "NotionalOverride");
121 QL_REQUIRE(notionalOverride_ >= 0, "Non-negative notional expected.");
122 DLOG("Using override notional of " << notionalOverride_);
123 notionalCalculation_ = "Override";
124 } else {
125 // Convert everything to proper case to match xml schema
126 notionalCalculation_ = boost::to_lower_copy(XMLUtils::getChildValue(compNode, "NotionalCalculation"));
127 notionalCalculation_[0] = toupper(notionalCalculation_[0]);
128 QL_REQUIRE(notionalCalculation_ != "Override", "Notional override value has not been provided.");
129 }
130
131 XMLNode* tradesNode = XMLUtils::getChildNode(compNode, "Components");
132 QL_REQUIRE(tradesNode, "Could not find Components node.");
133
134 vector<XMLNode*> nodes = XMLUtils::getChildrenNodes(tradesNode, "Trade");
135 for (Size i = 0; i < nodes.size(); i++) {
136 string tradeType = XMLUtils::getChildValue(nodes[i], "TradeType", true);
137
138 string id = XMLUtils::getAttribute(nodes[i], "id");
139 if (id == "") {
140 WLOG("Empty component trade id being overwritten in composite trade " << this->id() << ".");
141 }
142 id = this->id() + "_" + std::to_string(i);
143 DLOG("Parsing composite trade " << this->id() << " node " << i << " with id: " << id);
144
145 boost::shared_ptr<Trade> trade;
146 try {
147 trade = TradeFactory::instance().build(tradeType);
148 trade->id() = id;
149 // we prefer that the component trades don't have envelopes, but we need one for validation
150 trade->envelope() = this->envelope();
151 trade->fromXML(nodes[i]);
152 trades_.push_back(trade);
153 DLOG("Added Trade " << id << " (" << trade->id() << ")"
154 << " type:" << tradeType << " to composite trade " << this->id() << ".");
155 } catch (std::exception& ex) {
156 ALOG("Failed to build subtrade with id '" << id << "' inside composite trade: " << ex.what());
157 }
158 }
159 LOG("Finished Parsing XML doc");
160}

162XMLNode* CompositeTrade::toXML(XMLDocument& doc) {
163 XMLNode* node = Trade::toXML(doc);
164 XMLNode* compNode = doc.allocNode("CompositeTradeData");
165 XMLUtils::appendNode(node, compNode);
166
167 XMLUtils::addChild(doc, compNode, "Currency", currency_);
168 if (notionalCalculation_ == "Override")
169 XMLUtils::addChild(doc, compNode, "NotionalOverride", notionalOverride_);
170 XMLUtils::addChild(doc, compNode, "NotionalCalculation", notionalCalculation_);
171
172 XMLNode* tradesNode = doc.allocNode("Components");
173 XMLUtils::appendNode(compNode, tradesNode);
174 for (auto trade : trades_) {
175 XMLNode* subTradeNode = trade->toXML(doc);
176 XMLUtils::appendNode(tradesNode, subTradeNode);
177 }
178 return node;
179}

181Real CompositeTrade::calculateNotional(const vector<Real>& notionals) const {
182 if (notionalCalculation_ == "Sum" || notionalCalculation_ == "")
183 return std::accumulate(notionals.begin(), notionals.end(), 0.0);
184 else if (notionalCalculation_ == "Mean" || notionalCalculation_ == "Average")
185 return std::accumulate(notionals.begin(), notionals.end(), 0.0) / notionals.size();
186 else if (notionalCalculation_ == "First")
187 return notionals[0];
188 else if (notionalCalculation_ == "Last")
189 return notionals.back();
190 else if (notionalCalculation_ == "Min")
191 return *std::min_element(notionals.begin(), notionals.end());
192 else if (notionalCalculation_ == "Max")
193 return *std::min_element(notionals.begin(), notionals.end());
194 else if (notionalCalculation_ == "Override")
195 return notionalOverride_;
196 else
197 QL_FAIL("Unsupported notional calculation type.");
198}

200map<string, set<Date>> CompositeTrade::fixings(const Date& settlementDate) const {
201
202 map<string, set<Date>> result;
203 for (const auto& t : trades_) {
204 auto fixings = t->fixings(settlementDate);
205 for (const auto& kv : fixings) {
206 result[kv.first].insert(kv.second.begin(), kv.second.end());
207 }
208 }
209 return result;
