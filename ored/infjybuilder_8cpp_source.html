77InfJyBuilder::InfJyBuilder(
78 const boost::shared_ptr<Market>& market,
79 const boost::shared_ptr<InfJyData>& data,
80 const string& configuration,
81 const string& referenceCalibrationGrid)
82 : market_(market),
83 configuration_(configuration),
84 data_(data),
85 referenceCalibrationGrid_(referenceCalibrationGrid),
86 marketObserver_(boost::make_shared<MarketObserver>()),
87 zeroInflationIndex_(*market_->zeroInflationIndex(data_->index(), configuration_)) {
88
89 LOG("InfJyBuilder: building model for inflation index " << data_->index());
90
91 // Get rate curve
92 rateCurve_ = market_->discountCurve(zeroInflationIndex_->currency().code(), configuration_);
93
94 // Register with market observables except volatilities
95 marketObserver_->registerWith(zeroInflationIndex_);
96 marketObserver_->registerWith(rateCurve_);
97 initialiseMarket();
98
99 // Register the model builder with the market observer
100 registerWith(marketObserver_);
101
102 // Notify observers of all market data changes, not only when not calculated
103 alwaysForwardNotifications();
104
105 // Build the calibration instruments
106 buildCalibrationBaskets();
107
108 // Create the JY parameterisation.
109 parameterization_ = boost::make_shared<QuantExt::InfJyParameterization>(
110 createRealRateParam(), createIndexParam(), zeroInflationIndex_);
111}

132bool InfJyBuilder::requiresRecalibration() const {
133 return (data_->realRateVolatility().calibrate() ||
134 data_->realRateReversion().calibrate() ||
135 data_->indexVolatility().calibrate()) &&
136 (marketObserver_->hasUpdated(false) ||
137 forceCalibration_ ||
138 pricesChanged(false));

155void InfJyBuilder::buildCalibrationBaskets() const {
156
157 // If calibration type is None, don't build any baskets.
158 if (data_->calibrationType() == CalibrationType::None) {
159 DLOG("InfJyBuilder: calibration type is None so no calibration baskets built.");
160 return;
161 }
162
163 const auto& cbs = data_->calibrationBaskets();
164
165 // If calibration type is BestFit, check that we have at least one calibration basket. Build up to a maximum of
166 // two calibration baskets. Log a warning if more than two are given. Arbitrarily assign the built baskets to
167 // the realRateBasket_ and indexBasket_ members. They will be combined again in any case for BestFit calibration.
168 if (data_->calibrationType() == CalibrationType::BestFit) {
169 QL_REQUIRE(cbs.size() > 0, "InfJyBuilder: calibration type is BestFit but no calibration baskets provided.");
170 rrInstActive_ = vector<bool>(cbs[0].instruments().size(), false);
171 realRateBasket_ = buildCalibrationBasket(cbs[0], rrInstActive_, rrInstExpiries_);
172 if (cbs.size() > 1) {
173 indexInstActive_ = vector<bool>(cbs[1].instruments().size(), false);
174 indexBasket_ = buildCalibrationBasket(cbs[1], indexInstActive_, indexInstExpiries_);
175 }
176 if (cbs.size() > 2)
177 WLOG("InfJyBuilder: only 2 calibration baskets can be processed but " << cbs.size() <<
178 " were supplied. The extra baskets are ignored.");
179 return;
180 }
181
182 // Make sure that the calibration type is now Bootstrap.
183 QL_REQUIRE(data_->calibrationType() == CalibrationType::Bootstrap, "InfJyBuilder: expected the calibration " <<
184 "type to be one of None, BestFit or Bootstrap.");
185
186 const VolatilityParameter& idxVolatility = data_->indexVolatility();
187 const ReversionParameter& rrReversion = data_->realRateReversion();
188 const VolatilityParameter& rrVolatility = data_->realRateVolatility();
189
190 // Firstly, look at the inflation index portion i.e. are we calibrating it.
191 if (idxVolatility.calibrate()) {
192
193 DLOG("InfJyBuilder: building calibration basket for JY index bootstrap calibration.");
194
195 // If we are not calibrating the real rate portion, then we expect exactly one calibration basket. Otherwise
196 // we need to find a basket with the 'Index' parameter.
197 if (!rrReversion.calibrate() && !rrVolatility.calibrate()) {
198 QL_REQUIRE(cbs.size() == 1, "InfJyBuilder: calibrating only JY index volatility using Bootstrap so " <<
199 "expected exactly one basket but got " << cbs.size() << ".");
200 const auto& cb = cbs[0];
201 if (!cb.parameter().empty() && cb.parameter() != "Index") {
202 WLOG("InfJyBuilder: calibrating only JY index volatility using Bootstrap so expected the " <<
203 "calibration basket parameter to be 'Index' but got '" << cb.parameter() << "'.");
204 }
205 indexInstActive_ = vector<bool>(cb.instruments().size(), false);
206 indexBasket_ = buildCalibrationBasket(cb, indexInstActive_, indexInstExpiries_);
207 } else {
208 DLOG("InfJyBuilder: need a calibration basket with parameter equal to 'Index'.");
209 const auto& cb = calibrationBasket("Index");
210 indexInstActive_ = vector<bool>(cb.instruments().size(), false);
211 indexBasket_ = buildCalibrationBasket(cb, indexInstActive_, indexInstExpiries_);
212 }
213 }
214
215 // Secondly, look at the real rate portion i.e. are we calibrating it.
216 if (rrReversion.calibrate() || rrVolatility.calibrate()) {
217
218 DLOG("InfJyBuilder: building calibration basket for JY real rate bootstrap calibration.");
219 QL_REQUIRE(!(rrReversion.calibrate() && rrVolatility.calibrate()), "InfJyBuilder: calibrating both the " <<
220 "real rate reversion and real rate volatility using Bootstrap is not supported.");
221
222 // If we are not calibrating the index portion, then we expect exactly one calibration basket. Otherwise
223 // we need to find a basket with the 'RealRate' parameter.
224 if (!idxVolatility.calibrate()) {
225 QL_REQUIRE(cbs.size() == 1, "InfJyBuilder: calibrating only JY real rate using Bootstrap so " <<
226 "expected exactly one basket but got " << cbs.size() << ".");
227 const auto& cb = cbs[0];
228 if (!cb.parameter().empty() && cb.parameter() != "RealRate") {
229 WLOG("InfJyBuilder: calibrating only JY real rate using Bootstrap so expected the " <<
230 "calibration basket parameter to be 'RealRate' but got '" << cb.parameter() << "'.");
231 }
232 rrInstActive_ = vector<bool>(cb.instruments().size(), false);
233 realRateBasket_ = buildCalibrationBasket(cb, rrInstActive_, rrInstExpiries_, rrReversion.calibrate());
234 } else {
235 DLOG("InfJyBuilder: need a calibration basket with parameter equal to 'RealRate'.");
236 const auto& cb = calibrationBasket("RealRate");
237 rrInstActive_ = vector<bool>(cb.instruments().size(), false);
238 realRateBasket_ = buildCalibrationBasket(cb, rrInstActive_, rrInstExpiries_, rrReversion.calibrate());
239 }
240 }

244Helpers InfJyBuilder::buildCalibrationBasket(const CalibrationBasket& cb,
245 vector<bool>& active, Array& expiries, bool forRealRateReversion) const {
246
247 QL_REQUIRE(!cb.empty(), "InfJyBuilder: calibration basket should not be empty.");
248
249 const auto& ci = cb.instruments();
250 QL_REQUIRE(ci.size() == active.size(), "InfJyBuilder: expected the active instruments vector " <<
251 "size to equal the number of calibration instruments");
252 fill(active.begin(), active.end(), false);
253
254 if (cb.instrumentType() == "CpiCapFloor") {
255 return buildCpiCapFloorBasket(cb, active, expiries);
256 } else if (cb.instrumentType() == "YoYCapFloor") {
257 return buildYoYCapFloorBasket(cb, active, expiries);
258 } else if (cb.instrumentType() == "YoYSwap") {
259 return buildYoYSwapBasket(cb, active, expiries, forRealRateReversion);
260 } else {
261 QL_FAIL("InfJyBuilder: expected calibration instrument to be one of CpiCapFloor, YoYCapFloor or YoYSwap");
262 }

265Helpers InfJyBuilder::buildCpiCapFloorBasket(const CalibrationBasket& cb,
266 vector<bool>& active, Array& expiries) const {
267
268 DLOG("InfJyBuilder: start building the CPI cap floor calibration basket.");
269
270 QL_REQUIRE(!cpiVolatility_.empty(), "InfJyBuilder: need a non-empty CPI cap floor volatility structure " <<
271 "to build a CPI cap floor calibration basket.");
272
273 // Procedure is to create a CPI cap floor as described by each instrument in the calibration basket. We then value
274 // each of the CPI cap floor instruments using market data and an engine and pass the NPV as the market premium to
275 // helper that we create.
276
277 Helpers helpers;
278
279 // Create the engine
280 auto zts = zeroInflationIndex_->zeroInflationTermStructure();
281
282 boost::shared_ptr<QuantExt::CPICapFloorEngine> engine;
283 bool isLogNormalVol = QuantExt::ZeroInflation::isCPIVolSurfaceLogNormal(cpiVolatility_.currentLink());
284 if (isLogNormalVol) {
285 engine = boost::make_shared<QuantExt::CPIBlackCapFloorEngine>(rateCurve_, cpiVolatility_);
286 } else {
287 engine = boost::make_shared<QuantExt::CPIBachelierCapFloorEngine>(rateCurve_, cpiVolatility_);
288 }
289 // CPI cap floor calibration instrument details. Assumed to equal those from the index and market structures.
290 // Some of these should possibly come from conventions.
291 // Also some variables used in the loop below.
292 auto calendar = zeroInflationIndex_->fixingCalendar();
293 auto baseDate = zts->baseDate();
294 auto baseCpi = zeroInflationIndex_->fixing(baseDate);
295 auto bdc = cpiVolatility_->businessDayConvention();
296 auto obsLag = cpiVolatility_->observationLag();
297
298 Handle<ZeroInflationIndex> inflationIndex(zeroInflationIndex_);
299 Date today = Settings::instance().evaluationDate();
300 Real nominal = 1.0;
301
302 // Avoid instruments with duplicate expiry times in the loop below
303 set<Time, CloseCmp> expiryTimes;
304
305 // Reference calibration dates if any. If they are given, we only include one calibration instrument from each
306 // period in the grid. Logic copied from other builders.
307 auto rcDates = referenceCalibrationDates();
308 auto prevRcDate = Date::minDate();
309
310 // Add calibration instruments to the helpers vector.
311 const auto& ci = cb.instruments();
312
313 auto observationInterpolation = cpiVolatility_->indexIsInterpolated() ? CPI::Linear : CPI::Flat;
314
315 for (Size i = 0; i < ci.size(); ++i) {
316
317 auto cpiCapFloor = boost::dynamic_pointer_cast<CpiCapFloor>(ci[i]);
318 QL_REQUIRE(cpiCapFloor, "InfJyBuilder: expected CpiCapFloor calibration instrument.");
319 auto maturity = optionMaturity(cpiCapFloor->maturity(), calendar);
320
321 // Deal with reference calibration date grid stuff.
322 auto rcDate = lower_bound(rcDates.begin(), rcDates.end(), maturity);
323 if (!(rcDate == rcDates.end() || *rcDate > prevRcDate)) {
324 active[i] = false;
325 continue;
326 }
327
328 if (rcDate != rcDates.end())
329 prevRcDate = *rcDate;
330
331 // Build the CPI calibration instrument in order to calculate its NPV.
332
333 /* FIXME - the maturity date is not adjusted on eval date changes even if given as a tenor
334 - if the strike is atm, the value will not be updated on eval date changes */
335 Real strikeValue =
336 cpiCapFloorStrikeValue(cpiCapFloor->strike(), *zeroInflationIndex_->zeroInflationTermStructure(), maturity);
337 Option::Type capfloor = cpiCapFloor->type() == CapFloor::Cap ? Option::Call : Option::Put;
338 auto inst = boost::make_shared<CPICapFloor>(capfloor, nominal, today, baseCpi, maturity, calendar, bdc, calendar, bdc,
339 strikeValue, zeroInflationIndex_, obsLag, observationInterpolation);
340 inst->setPricingEngine(engine);
341
342 // Build the helper using the NPV as the premium.
343 auto premium = inst->NPV();
344 auto helper = boost::make_shared<CpiCapFloorHelper>(capfloor, baseCpi, maturity, calendar, bdc, calendar, bdc,
345 strikeValue, inflationIndex, obsLag, premium,
346 observationInterpolation);
347
348
349 // Add the helper's time to expiry.
350 auto fixingDate = helper->instrument()->fixingDate();
351 auto t = inflationTime(fixingDate, *zts, false);
352 auto p = expiryTimes.insert(t);
353 QL_REQUIRE(data_->ignoreDuplicateCalibrationExpiryTimes() || p.second,
354 "InfJyBuilder: a CPI cap floor calibration "
355 << "instrument with the expiry time, " << t << ", was already added.");
356
357 if(p.second)
358 helpers.push_back(helper);
359
360 TLOG("InfJyBuilder: " << (p.second ?
361 "added CPICapFloor helper" :
362 "skipped CPICapFloor helper due to duplicate expiry time (" + std::to_string(t) + ")") <<
363 ": index = " << data_->index() <<
364 ", type = " << cpiCapFloor->type() <<
365 ", expiry = " << io::iso_date(maturity) <<
366 ", base CPI = " << baseCpi <<
367 ", strike = " << strikeValue <<
368 ", obs lag = " << obsLag <<
369 ", market premium = " << premium);
370 }
371
372 // Populate the expiry times array with the unique sorted expiry times.
373 expiries = Array(expiryTimes.begin(), expiryTimes.end());
374
375 DLOG("InfJyBuilder: finished building the CPI cap floor calibration basket.");
376
377 return helpers;
378}

380Helpers InfJyBuilder::buildYoYCapFloorBasket(const CalibrationBasket& cb, vector<bool>& active, Array& expiries) const {
381
382 DLOG("InfJyBuilder: start building the YoY cap floor calibration basket.");
383
384 // Initial checks.
385 QL_REQUIRE(yoyInflationIndex_, "InfJyBuilder: need a valid year on year inflation index "
386 << "to build a year on year cap floor calibration basket.");
387 auto yoyTs = yoyInflationIndex_->yoyInflationTermStructure();
388 QL_REQUIRE(!yoyTs.empty(), "InfJyBuilder: need a valid year on year term structure "
389 << "to build a year on year cap floor calibration basket.");
390 QL_REQUIRE(!yoyVolatility_.empty(), "InfJyBuilder: need a valid year on year volatility "
391 << "structure to build a year on year cap floor calibration basket.");
392
393 // Procedure is to create a YoY cap floor as described by each instrument in the calibration basket. We then value
394 // each of the YoY cap floor instruments using market data and an engine and pass the NPV as the market premium to
395 // helper that we create.
396
397 Helpers helpers;
398
399 // Create the engine which depends on the type of the YoY volatility and the shift.
400 boost::shared_ptr<PricingEngine> engine;
401 auto ovsType = yoyVolatility_->volatilityType();
402 if (ovsType == Normal)
403 engine = boost::make_shared<YoYInflationBachelierCapFloorEngine>(yoyInflationIndex_, yoyVolatility_, rateCurve_);
404 else if (ovsType == ShiftedLognormal && close(yoyVolatility_->displacement(), 0.0))
405 engine = boost::make_shared<YoYInflationBlackCapFloorEngine>(yoyInflationIndex_, yoyVolatility_, rateCurve_);
406 else if (ovsType == ShiftedLognormal)
407 engine =
408 boost::make_shared<YoYInflationUnitDisplacedBlackCapFloorEngine>(yoyInflationIndex_, yoyVolatility_, rateCurve_);
409 else
410 QL_FAIL("InfJyBuilder: can't create engine with yoy volatility type, " << ovsType << ".");
411
412 // YoY cap floor calibration instrument details. Assumed to equal those from the index and market structures.
413 // Some of these should possibly come from conventions. Also some variables used in the loop below.
414 Natural settlementDays = 2;
415 auto calendar = yoyInflationIndex_->fixingCalendar();
416 DayCounter dc = Thirty360(Thirty360::BondBasis);
417 auto bdc = Following;
418 auto obsLag = yoyVolatility_->observationLag();
419
420 // Avoid instruments with duplicate expiry times in the loop below
421 set<Time, CloseCmp> expiryTimes;
422
423 // Reference calibration dates if any. If they are given, we only include one calibration instrument from each
424 // period in the grid. Logic copied from other builders.
425 auto rcDates = referenceCalibrationDates();
426 auto prevRcDate = Date::minDate();
427
428 // Add calibration instruments to the helpers vector.
429 const auto& ci = cb.instruments();
430 for (Size i = 0; i < ci.size(); ++i) {
431
432 auto yoyCapFloor = boost::dynamic_pointer_cast<YoYCapFloor>(ci[i]);
433 QL_REQUIRE(yoyCapFloor, "InfJyBuilder: expected YoYCapFloor calibration instrument.");
434
435 /*! Get the configured strike.
436 FIXME If the strike is atm, the value will not be updated on evaluation date changes */
437 Date today = Settings::instance().evaluationDate();
438 Date maturityDate = calendar.advance(calendar.advance(today, settlementDays * Days), yoyCapFloor->tenor(), bdc);
439 Real strikeValue = yoyCapFloorStrikeValue(yoyCapFloor->strike(), *yoyTs, maturityDate);
440
441 // Build the YoY cap floor helper.
442 auto quote = boost::make_shared<SimpleQuote>(0.01);
443 auto helper = boost::make_shared<YoYCapFloorHelper>(Handle<Quote>(quote), yoyCapFloor->type(), strikeValue,
444 settlementDays, yoyCapFloor->tenor(), yoyInflationIndex_, obsLag, calendar, bdc, dc, calendar, bdc);
445
446 // Deal with reference calibration date grid stuff based on maturity of helper instrument.
447 auto helperInst = helper->yoyCapFloor();
448 auto maturity = helperInst->maturityDate();
449 auto rcDate = lower_bound(rcDates.begin(), rcDates.end(), maturity);
450 if (!(rcDate == rcDates.end() || *rcDate > prevRcDate)) {
451 active[i] = false;
452 continue;
453 }
454
455 if (rcDate != rcDates.end())
456 prevRcDate = *rcDate;
457
458 // Price the underlying helper instrument to get its fair premium.
459 helperInst->setPricingEngine(engine);
460
461 // Update the helper's market quote with the fair rate.
462 quote->setValue(helperInst->NPV());
463
464 // Add the helper's time to expiry.
465 auto fixingDate = helperInst->lastYoYInflationCoupon()->fixingDate();
466 auto t = inflationTime(fixingDate, *yoyTs, yoyInflationIndex_->interpolated());
467 auto p = expiryTimes.insert(t);
468 QL_REQUIRE(data_->ignoreDuplicateCalibrationExpiryTimes() || p.second,
469 "InfJyBuilder: a YoY cap floor calibration "
470 << "instrument with the expiry time, " << t << ", was already added.");
471
472 // Add the helper to the calibration helpers.
473 if(p.second)
474 helpers.push_back(helper);
475
476 TLOG("InfJyBuilder: " << (p.second ?
477 "added YoYCapFloor helper" :
478 "skipped YoYCapFloor helper due to duplicate expiry time (" + std::to_string(t) + ")") <<
479 ": index = " << data_->index() <<
480 ", type = " << yoyCapFloor->type() <<
481 ", expiry = " << io::iso_date(maturity) <<
482 ", strike = " << strikeValue <<
483 ", obs lag = " << obsLag <<
484 ", market premium = " << quote->value());
485 }
486
487 // Populate the expiry times array with the unique sorted expiry times.
488 expiries = Array(expiryTimes.begin(), expiryTimes.end());
489
490 DLOG("InfJyBuilder: finished building the YoY cap floor calibration basket.");
491
492 return helpers;
493}

495Helpers InfJyBuilder::buildYoYSwapBasket(const CalibrationBasket& cb,
496 vector<bool>& active, Array& expiries, bool forRealRateReversion) const {
497
498 DLOG("InfJyBuilder: start building the YoY swap calibration basket.");
499
500 // Initial checks.
501 QL_REQUIRE(yoyInflationIndex_, "InfJyBuilder: need a valid year on year inflation index " <<
502 "to build a year on year swap calibration basket.");
503 auto yoyTs = yoyInflationIndex_->yoyInflationTermStructure();
504 QL_REQUIRE(!yoyTs.empty(), "InfJyBuilder: need a valid year on year term structure " <<
505 "to build a year on year swap calibration basket.");
506
507 // Procedure is to create a YoY cap floor as described by each instrument in the calibration basket. We then value
508 // each of the YoY cap floor instruments using market data and an engine and pass the NPV as the market premium to
509 // helper that we create.
510
511 Helpers helpers;
512
513 // Create the engine
514 auto engine = boost::make_shared<DiscountingSwapEngine>(rateCurve_);
515
516 // YoY swap calibration instrument details. Assumed to equal those from the index and market structures.
517 // Some of these should possibly come from conventions. Hardcoded some common values here.
518 // Also some variables used in the loop below.
519 Natural settlementDays = 2;
520 auto calendar = yoyInflationIndex_->fixingCalendar();
521 auto dc = Thirty360(Thirty360::BondBasis);
522 auto bdc = Following;
523 auto obsLag = yoyTs->observationLag();
524
525 // Avoid instruments with duplicate expiry times in the loop below
526 set<Time, CloseCmp> expiryTimes;
527
528 // Reference calibration dates if any. If they are given, we only include one calibration instrument from each
529 // period in the grid. Logic copied from other builders.
530 auto rcDates = referenceCalibrationDates();
531 auto prevRcDate = Date::minDate();
532
533 // Add calibration instruments to the helpers vector.
534 const auto& ci = cb.instruments();
535 for (Size i = 0; i < ci.size(); ++i) {
536
537 auto yoySwap = boost::dynamic_pointer_cast<YoYSwap>(ci[i]);
538 QL_REQUIRE(yoySwap, "InfJyBuilder: expected YoYSwap calibration instrument.");
539
540 // Build the YoY helper.
541 auto quote = boost::make_shared<SimpleQuote>(0.01);
542 auto helper = boost::make_shared<YoYSwapHelper>(Handle<Quote>(quote), settlementDays, yoySwap->tenor(),
543 yoyInflationIndex_, rateCurve_, obsLag, calendar, bdc, dc,
544 calendar, bdc, dc, calendar, bdc);
545
546 // Deal with reference calibration date grid stuff based on maturity of helper instrument.
547 auto helperInst = helper->yoySwap();
548 auto maturity = helperInst->maturityDate();
549 auto rcDate = lower_bound(rcDates.begin(), rcDates.end(), maturity);
550 if (!(rcDate == rcDates.end() || *rcDate > prevRcDate)) {
551 active[i] = false;
552 continue;
553 }
554
555 if (rcDate != rcDates.end())
556 prevRcDate = *rcDate;
557
558 // Price the underlying helper instrument to get its fair rate.
559 helperInst->setPricingEngine(engine);
560
561 // Update the helper's market quote with the fair rate.
562 quote->setValue(helperInst->fairRate());
563
564 // For JY calibration to YoY swaps, the parameter's time depends on whether you are calibrating the real rate
565 // reversion or the real rate volatility (probably don't want to calibrate the inflation index vol to YoY
566 // swaps as it only shows up via the drift). If you are calibrating to real rate reversion, you want the time
567 // to the numerator index fixing date on the last YoY swaplet on the YoY leg. If you are calibrating to real
568 // rate volatility, you want the time to the denominator index fixing date on the last YoY swaplet on the YoY
569 // leg. We use numerator fixing date - 1 * Years here for this. You can see this from the parameter
570 // dependencies in the YoY swaplet formula in Section 13 of the book (i.e. T vs. S).
571 // If t is not positive, we log a message and skip this helper.
572 Time t = 0.0;
573 QL_REQUIRE(!helperInst->yoyLeg().empty(), "InfJyBuilder: expected YoYSwap to have non-empty YoY leg.");
574 auto finalYoYCoupon = boost::dynamic_pointer_cast<YoYInflationCoupon>(helperInst->yoyLeg().back());
575 Date numFixingDate = finalYoYCoupon->fixingDate();
576 if (forRealRateReversion) {
577 t = inflationTime(numFixingDate, *yoyTs, yoyInflationIndex_->interpolated());
578 } else {
579 auto denFixingDate = numFixingDate - 1 * Years;
580 t = inflationTime(denFixingDate, *yoyTs, yoyInflationIndex_->interpolated());
581 }
582
583 if (t <= 0) {
584 DLOG("The year on year swap with maturity tenor, " << yoySwap->tenor() << ", and date, " << maturity <<
585 ", has a non-positive parameter time, " << t << ", so skipping this as a calibration instrument.");
586 continue;
587 }
588
589 // Add the helper to the calibration helpers.
590 helpers.push_back(helper);
591
592 auto p = expiryTimes.insert(t);
593 QL_REQUIRE(p.second, "InfJyBuilder: a YoY swap calibration instrument with the expiry " <<
594 "time, " << t << ", was already added.");
595
596 TLOG("InfJyBuilder: added year on year swap helper" <<
597 ": index = " << data_->index() <<
598 ", maturity = " << io::iso_date(maturity) <<
599 ", obs lag = " << obsLag <<
600 ", market rate = " << quote->value());
601 }
602
603 // Populate the expiry times array with the unique sorted expiry times.
604 expiries = Array(expiryTimes.begin(), expiryTimes.end());
605
606 DLOG("InfJyBuilder: finished building the YoY swap calibration basket.");
607
608 return helpers;
609}

622boost::shared_ptr<Lgm1fParametrization<ZeroInflationTermStructure>> InfJyBuilder::createRealRateParam() const {
623
624 DLOG("InfJyBuilder: start creating the real rate parameterisation.");
625
626 // Initial parameter setup as provided by the data_.
627 const ReversionParameter& rrReversion = data_->realRateReversion();
628 const VolatilityParameter& rrVolatility = data_->realRateVolatility();
629 Array rrVolatilityTimes(rrVolatility.times().begin(), rrVolatility.times().end());
630 Array rrVolatilityValues(rrVolatility.values().begin(), rrVolatility.values().end());
631 Array rrReversionTimes(rrReversion.times().begin(), rrReversion.times().end());
632 Array rrReversionValues(rrReversion.values().begin(), rrReversion.values().end());
633
634 // Perform checks and in the event of bootstrap calibration, may need to restructure the parameters.
635 setupParams(rrReversion, rrReversionTimes, rrReversionValues, rrInstExpiries_, "RealRate reversion");
636 setupParams(rrVolatility, rrVolatilityTimes, rrVolatilityValues, rrInstExpiries_, "RealRate volatility");
637
638 // Create the JY parameterization.
639 using RT = LgmData::ReversionType;
640 using VT = LgmData::VolatilityType;
641
642 // Real rate parameter constraints
643 const auto& cc = data_->calibrationConfiguration();
644 auto rrVolConstraint = cc.constraint("RealRateVolatility");
645 auto rrRevConstraint = cc.constraint("RealRateReversion");
646
647 // Create the real rate portion of the parameterization
648 using QuantLib::ZeroInflationTermStructure;
649 boost::shared_ptr<QuantExt::Lgm1fParametrization<ZeroInflationTermStructure>> realRateParam;
650 if (rrReversion.reversionType() == RT::HullWhite && rrVolatility.volatilityType() == VT::HullWhite) {
651 using QuantExt::Lgm1fPiecewiseConstantHullWhiteAdaptor;
652 DLOG("InfJyBuilder: real rate parameterization is Lgm1fPiecewiseConstantHullWhiteAdaptor");
653 realRateParam = boost::make_shared<Lgm1fPiecewiseConstantHullWhiteAdaptor<ZeroInflationTermStructure>>(
654 zeroInflationIndex_->currency(), zeroInflationIndex_->zeroInflationTermStructure(), rrVolatilityTimes,
655 rrVolatilityValues, rrReversionTimes, rrReversionValues, data_->index(), rrVolConstraint, rrRevConstraint);
656 } else if (rrReversion.reversionType() == RT::HullWhite && rrVolatility.volatilityType() == VT::Hagan) {
657 using QuantExt::Lgm1fPiecewiseConstantParametrization;
658 DLOG("InfJyBuilder: real rate parameterization is Lgm1fPiecewiseConstantParametrization");
659 realRateParam = boost::make_shared<Lgm1fPiecewiseConstantParametrization<ZeroInflationTermStructure>>(
660 zeroInflationIndex_->currency(), zeroInflationIndex_->zeroInflationTermStructure(), rrVolatilityTimes,
661 rrVolatilityValues, rrReversionTimes, rrReversionValues, data_->index(), rrVolConstraint, rrRevConstraint);
662 } else if (rrReversion.reversionType() == RT::Hagan && rrVolatility.volatilityType() == VT::Hagan) {
663 using QuantExt::Lgm1fPiecewiseLinearParametrization;
664 DLOG("InfJyBuilder: real rate parameterization is Lgm1fPiecewiseLinearParametrization");
665 realRateParam = boost::make_shared<Lgm1fPiecewiseLinearParametrization<ZeroInflationTermStructure>>(
666 zeroInflationIndex_->currency(), zeroInflationIndex_->zeroInflationTermStructure(), rrVolatilityTimes,
667 rrVolatilityValues, rrReversionTimes, rrReversionValues, data_->index(), rrVolConstraint, rrRevConstraint);
668 } else {
669 QL_FAIL("InfJyBuilder: reversion type Hagan and volatility type HullWhite not supported.");
670 }
671
672 Time horizon = data_->reversionTransformation().horizon();
673 if (horizon >= 0.0) {
674 DLOG("InfJyBuilder: apply shift horizon " << horizon << " to the JY real rate parameterisation for index " <<
675 data_->index() << ".");
676 realRateParam->shift() = horizon;
677 } else {
678 WLOG("InfJyBuilder: ignoring negative horizon, " << horizon <<
679 ", passed to the JY real rate parameterisation for index " << data_->index() << ".");
680 }
681
682 Real scaling = data_->reversionTransformation().scaling();
683 if (scaling > 0.0) {
684 DLOG("InfJyBuilder: apply scaling " << scaling << " to the JY real rate parameterisation for index " <<
685 data_->index() << ".");
686 realRateParam->scaling() = scaling;
687 } else {
688 WLOG("Ignoring non-positive scaling, " << scaling <<
689 ", passed to the JY real rate parameterisation for index " << data_->index() << ".");
690 }
691
692 DLOG("InfJyBuilder: finished creating the real rate parameterisation.");
693
694 return realRateParam;
695}

738void InfJyBuilder::setupParams(const ModelParameter& param, Array& times, Array& values,
739 const Array& expiries, const string& paramName) const {
740
741 DLOG("InfJyBuilder: start setting up parameters for " << paramName);
742
743 if (param.type() == ParamType::Constant) {
744 QL_REQUIRE(param.times().size() == 0, "InfJyBuilder: parameter is constant so empty times expected");
745 QL_REQUIRE(param.values().size() == 1, "InfJyBuilder: parameter is constant so initial value array " <<
746 "should have 1 element.");
747 } else if (param.type() == ParamType::Piecewise) {
748
749 if (param.calibrate() && data_->calibrationType() == CalibrationType::Bootstrap) {
750 QL_REQUIRE(!expiries.empty(), "InfJyBuilder: calibration instrument expiries are empty.");
751 QL_REQUIRE(!values.empty(), "InfJyBuilder: expected at least one initial value.");
752 DLOG("InfJyBuilder: overriding initial times " << times << " with option calibration instrument " <<
753 "expiries " << expiries << ".");
754 times = Array(expiries.begin(), expiries.end() - 1);
755 values = Array(times.size() + 1, values[0]);
756 } else {
757 QL_REQUIRE(values.size() == times.size() + 1, "InfJyBuilder: size of values grid, " << values.size() <<
758 ", should be 1 greater than the size of the times grid, " << times.size() << ".");
759 }
760
761 } else {
762 QL_FAIL("Expected " << paramName << " parameter to be Constant or Piecewise.");
763 }
764
765 DLOG("InfJyBuilder: finished setting up parameters for " << paramName);
766}

809bool InfJyBuilder::pricesChanged(bool updateCache) const {
810
811 // Build the calibration instruments again before checking the market price below.
812 // Don't need to do this if updateCache is true, because only called above after buildCalibrationBaskets().
813 if (!updateCache)
814 buildCalibrationBaskets();
815
816 // Resize the cache to match the number of calibration instruments
817 auto numInsts = realRateBasket_.size() + indexBasket_.size();
818 if (priceCache_.size() != numInsts)
819 priceCache_ = vector<Real>(numInsts, Null<Real>());
820
821 // Check if any market prices have changed. Return true if they have and false if they have not.
822 // If asked to update the cached prices, via updateCache being true, update the prices, if necessary.
823 bool result = false;
824 Size ctr = 0;
825 for (const auto& ci : boost::range::join(realRateBasket_, indexBasket_)) {
826 auto mp = marketPrice(ci);
827 if (!close_enough(priceCache_[ctr], mp)) {
828 if (updateCache)
829 priceCache_[ctr] = mp;
830 result = true;
831 }
832 ctr++;
833 }
834
835 return result;
836}
