41 InflationCurve::InflationCurve(Date asof, InflationCurveSpec spec, const Loader& loader,
42  const CurveConfigurations& curveConfigs,
43  map<string, boost::shared_ptr<YieldCurve>>& yieldCurves,
44  const bool buildCalibrationInfo) {
45 
46  try {
47 
48  const boost::shared_ptr<InflationCurveConfig>& config = curveConfigs.inflationCurveConfig(spec.curveConfigID());
49 
50  const boost::shared_ptr<Conventions>& conventions = InstrumentConventions::instance().conventions();
51  boost::shared_ptr<InflationSwapConvention> conv =
52  boost::dynamic_pointer_cast<InflationSwapConvention>(conventions->get(config->conventions()));
53 
54  QL_REQUIRE(conv != nullptr, "convention " << config->conventions() << " could not be found.");
55 
56  Handle<YieldTermStructure> nominalTs;
57  auto it = yieldCurves.find(config->nominalTermStructure());
58  if (it != yieldCurves.end()) {
59  nominalTs = it->second->handle();
60  } else {
61  QL_FAIL("The nominal term structure, " << config->nominalTermStructure()
62  << ", required in the building "
63  "of the curve, "
64  << spec.name() << ", was not found.");
65  }
66 
67  // We loop over all market data, looking for quotes that match the configuration
68 
69  const std::vector<string> strQuotes = config->swapQuotes();
70  std::vector<Handle<Quote>> quotes(strQuotes.size(), Handle<Quote>());
71  std::vector<Period> terms(strQuotes.size());
72  std::vector<bool> isZc(strQuotes.size(), true);
73 
74  for (auto& md : loader.loadQuotes(asof)) {
75 
76  if (md->asofDate() == asof && (md->instrumentType() == MarketDatum::InstrumentType::ZC_INFLATIONSWAP ||
77  (md->instrumentType() == MarketDatum::InstrumentType::YY_INFLATIONSWAP &&
78  config->type() == InflationCurveConfig::Type::YY))) {
79 
80  boost::shared_ptr<ZcInflationSwapQuote> q = boost::dynamic_pointer_cast<ZcInflationSwapQuote>(md);
81  if (q) {
82  auto it = std::find(strQuotes.begin(), strQuotes.end(), q->name());
83  if (it != strQuotes.end()) {
84  QL_REQUIRE(quotes[it - strQuotes.begin()].empty(), "duplicate quote " << q->name());
85  quotes[it - strQuotes.begin()] = q->quote();
86  terms[it - strQuotes.begin()] = q->term();
87  isZc[it - strQuotes.begin()] = true;
88  }
89  }
90 
91  boost::shared_ptr<YoYInflationSwapQuote> q2 = boost::dynamic_pointer_cast<YoYInflationSwapQuote>(md);
92  if (q2) {
93  auto it = std::find(strQuotes.begin(), strQuotes.end(), q2->name());
94  if (it != strQuotes.end()) {
95  QL_REQUIRE(quotes[it - strQuotes.begin()].empty(), "duplicate quote " << q2->name());
96  quotes[it - strQuotes.begin()] = q2->quote();
97  terms[it - strQuotes.begin()] = q2->term();
98  isZc[it - strQuotes.begin()] = false;
99  }
100  }
101  }
102  }
103 
104  // do we have all quotes and do we derive yoy quotes from zc ?
105  for (Size i = 0; i < strQuotes.size(); ++i) {
106  QL_REQUIRE(!quotes[i].empty(), "quote " << strQuotes[i] << " not found in market data.");
107  QL_REQUIRE(isZc[i] == isZc[0], "mixed zc and yoy quotes");
108  }
109  bool derive_yoy_from_zc = (config->type() == InflationCurveConfig::Type::YY && isZc[0]);
110 
111  // construct seasonality
112  boost::shared_ptr<Seasonality> seasonality;
113  if (config->seasonalityBaseDate() != Null<Date>()) {
114  if (!config->overrideSeasonalityFactors().empty()) {
115  // override market data by explicit list
116  seasonality = boost::make_shared<MultiplicativePriceSeasonality>(config->seasonalityBaseDate(),
117  config->seasonalityFrequency(),
118  config->overrideSeasonalityFactors());
119  } else {
120  std::vector<string> strFactorIDs = config->seasonalityFactors();
121  std::vector<double> factors(strFactorIDs.size());
122  for (Size i = 0; i < strFactorIDs.size(); i++) {
123  boost::shared_ptr<MarketDatum> marketQuote = loader.get(strFactorIDs[i], asof);
124  // Check that we have a valid seasonality factor
125  if (marketQuote) {
126  QL_REQUIRE(marketQuote->instrumentType() == MarketDatum::InstrumentType::SEASONALITY,
127  "Market quote (" << marketQuote->name() << ") not of type seasonality.");
128  // Currently only monthly seasonality with 12 multiplicative factors os allowed
129  QL_REQUIRE(config->seasonalityFrequency() == Monthly && strFactorIDs.size() == 12,
130  "Only monthly seasonality with 12 factors is allowed. Provided "
131  << config->seasonalityFrequency() << " with " << strFactorIDs.size()
132  << " factors.");
133  boost::shared_ptr<SeasonalityQuote> sq =
134  boost::dynamic_pointer_cast<SeasonalityQuote>(marketQuote);
135  QL_REQUIRE(sq, "Could not cast to SeasonalityQuote, internal error.");
136  QL_REQUIRE(sq->type() == "MULT",
137  "Market quote (" << sq->name() << ") not of multiplicative type.");
138  Size seasBaseDateMonth = ((Size)config->seasonalityBaseDate().month());
139  int findex = sq->applyMonth() - seasBaseDateMonth;
140  if (findex < 0)
141  findex += 12;
142  QL_REQUIRE(findex >= 0 && findex < 12, "Unexpected seasonality index " << findex);
143  factors[findex] = sq->quote()->value();
144  } else {
145  QL_FAIL("Could not find quote for ID " << strFactorIDs[i] << " with as of date "
146  << io::iso_date(asof) << ".");
147  }
148  }
149  QL_REQUIRE(!factors.empty(), "no seasonality factors found");
150  seasonality = boost::make_shared<MultiplicativePriceSeasonality>(
151  config->seasonalityBaseDate(), config->seasonalityFrequency(), factors);
152  }
153  }
154 
155  // Get helper start date and curve's obs lag.
156  auto p = getStartAndLag(asof, *conv);
157  Date swapStart = p.first;
158  Period curveObsLag = p.second == Period() ? config->lag() : p.second;
159 
160  // construct curve (ZC or YY depending on configuration)
161  std::vector<Date> pillarDates;
162 
163  interpolatedIndex_ = conv->interpolated();
164  CPI::InterpolationType observationInterpolation = interpolatedIndex_ ? CPI::Linear : CPI::Flat;
165  boost::shared_ptr<YoYInflationIndex> zc_to_yoy_conversion_index;
166  if (config->type() == InflationCurveConfig::Type::ZC || derive_yoy_from_zc) {
167  // ZC Curve
168  std::vector<boost::shared_ptr<QuantExt::ZeroInflationTraits::helper>> instruments;
169  boost::shared_ptr<ZeroInflationIndex> index = conv->index();
170  for (Size i = 0; i < strQuotes.size(); ++i) {
171  // QL conventions do not incorporate settlement delay => patch here once QL is patched
172  Date maturity = swapStart + terms[i];
173  boost::shared_ptr<QuantExt::ZeroInflationTraits::helper> instrument =
174  boost::make_shared<ZeroCouponInflationSwapHelper>(
175  quotes[i], conv->observationLag(), maturity, conv->fixCalendar(), conv->fixConvention(),
176  conv->dayCounter(), index, observationInterpolation, nominalTs, swapStart);
177  // The instrument gets registered to update on change of evaluation date. This triggers a
178  // rebootstrapping of the curve. In order to avoid this during simulation we unregister from the
179  // evaluationDate.
180  instrument->unregisterWith(Settings::instance().evaluationDate());
181  instruments.push_back(instrument);
182  }
183  // base zero / yoy rate: if given, take it, otherwise set it to observered zeroRate
184  Real baseRate = quotes[0]->value();
185  if (config->baseRate() != Null<Real>()) {
186  baseRate = config->baseRate();
187  } else if (index) {
188  baseRate = QuantExt::ZeroInflation::guessCurveBaseRate(
189  config->useLastAvailableFixingAsBaseDate(), swapStart, terms[0], conv->dayCounter(),
190  conv->observationLag(), quotes[0]->value(), curveObsLag, config->dayCounter(), index,
191  interpolatedIndex_);
192  }
193 
194  curve_ = boost::shared_ptr<QuantExt::PiecewiseZeroInflationCurve<Linear>>(
195  new QuantExt::PiecewiseZeroInflationCurve<Linear>(
196  asof, config->calendar(), config->dayCounter(), curveObsLag, config->frequency(), baseRate,
197  instruments, config->tolerance(), index, config->useLastAvailableFixingAsBaseDate()));
198 
199  // force bootstrap so that errors are thrown during the build, not later
200  boost::static_pointer_cast<QuantExt::PiecewiseZeroInflationCurve<Linear>>(curve_)->zeroRate(QL_EPSILON);
201  if (derive_yoy_from_zc) {
202  // set up yoy wrapper with empty ts, so that zero index is used to forecast fixings
203  // for this link the appropriate curve to the zero index
204  zc_to_yoy_conversion_index = boost::make_shared<QuantExt::YoYInflationIndexWrapper>(
205  index->clone(Handle<ZeroInflationTermStructure>(
206  boost::dynamic_pointer_cast<ZeroInflationTermStructure>(curve_))),
207  interpolatedIndex_);
208  } else {
209  }
210  }
211  if (config->type() == InflationCurveConfig::Type::YY) {
212  // YOY Curve
213  std::vector<boost::shared_ptr<YoYInflationTraits::helper>> instruments;
214  boost::shared_ptr<ZeroInflationIndex> zcindex = conv->index();
215  boost::shared_ptr<YoYInflationIndex> index =
216  boost::make_shared<QuantExt::YoYInflationIndexWrapper>(zcindex, interpolatedIndex_);
217  boost::shared_ptr<InflationCouponPricer> yoyCpnPricer =
218  boost::make_shared<YoYInflationCouponPricer>(nominalTs);
219  for (Size i = 0; i < strQuotes.size(); ++i) {
220  Date maturity = swapStart + terms[i];
221  Real effectiveQuote = quotes[i]->value();
222  if (derive_yoy_from_zc) {
223  // construct a yoy swap just as it is done in the yoy inflation helper
224  Schedule schedule = MakeSchedule()
225  .from(swapStart)
226  .to(maturity)
227  .withTenor(1 * Years)
228  .withConvention(Unadjusted)
229  .withCalendar(conv->fixCalendar())
230  .backwards();
231  YearOnYearInflationSwap tmp(YearOnYearInflationSwap::Payer, 1000000.0, schedule, 0.02,
232  conv->dayCounter(), schedule, zc_to_yoy_conversion_index,
233  conv->observationLag(), 0.0, conv->dayCounter(), conv->fixCalendar(),
234  conv->fixConvention());
235  for (auto& c : tmp.yoyLeg()) {
236  auto cpn = boost::dynamic_pointer_cast<YoYInflationCoupon>(c);
237  QL_REQUIRE(cpn, "yoy inflation coupon expected, could not cast");
238  cpn->setPricer(yoyCpnPricer);
239  }
240  boost::shared_ptr<PricingEngine> engine =
241  boost::make_shared<QuantLib::DiscountingSwapEngine>(nominalTs);
242  tmp.setPricingEngine(engine);
243  effectiveQuote = tmp.fairRate();
244  DLOG("Derive " << terms[i] << " yoy quote " << effectiveQuote << " from zc quote "
245  << quotes[i]->value());
246  }
247  // QL conventions do not incorporate settlement delay => patch here once QL is patched
248  boost::shared_ptr<YoYInflationTraits::helper> instrument =
249  boost::make_shared<YearOnYearInflationSwapHelper>(
250  Handle<Quote>(boost::make_shared<SimpleQuote>(effectiveQuote)), conv->observationLag(),
251  maturity, conv->fixCalendar(), conv->fixConvention(), conv->dayCounter(), index, nominalTs,
252  swapStart);
253  instrument->unregisterWith(Settings::instance().evaluationDate());
254  instruments.push_back(instrument);
255  pillarDates.push_back(instrument->pillarDate());
256  }
257  // base zero rate: if given, take it, otherwise set it to first quote
258  Real baseRate =
259  config->baseRate() != Null<Real>() ? config->baseRate() : instruments.front()->quote()->value();
260  curve_ = boost::shared_ptr<PiecewiseYoYInflationCurve<Linear>>(new PiecewiseYoYInflationCurve<Linear>(
261  asof, config->calendar(), config->dayCounter(), curveObsLag, config->frequency(), interpolatedIndex_,
262  baseRate, instruments, config->tolerance()));
263  // force bootstrap so that errors are thrown during the build, not later
264  boost::static_pointer_cast<PiecewiseYoYInflationCurve<Linear>>(curve_)->yoyRate(QL_EPSILON);
265  }
266  if (seasonality != nullptr) {
267  curve_->setSeasonality(seasonality);
268  }
269  curve_->enableExtrapolation(config->extrapolate());
270  curve_->unregisterWith(Settings::instance().evaluationDate());
271 
272  // set calibration info
273 
274  if (buildCalibrationInfo) {
275 
276  if (pillarDates.empty()) {
277  // default: fill pillar dates with monthly schedule up to last term given in the curve config (but max
278  // 60y)
279  Date maturity = swapStart + terms.back();
280  for (Size i = 1; i < 60 * 12; ++i) {
281  Date current = inflationPeriod(curve_->baseDate() + i * Months, curve_->frequency()).first;
282  if (current + curve_->observationLag() <= maturity)
283  pillarDates.push_back(current);
284  else
285  break;
286  }
287  }
288 
289  if (config->type() == InflationCurveConfig::Type::YY) {
290  auto yoyCurve = boost::dynamic_pointer_cast<YoYInflationCurve>(curve_);
291  QL_REQUIRE(yoyCurve, "internal error: expected YoYInflationCurve (inflation curve builder)");
292  auto calInfo = boost::make_shared<YoYInflationCurveCalibrationInfo>();
293  calInfo->dayCounter = config->dayCounter().name();
294  calInfo->calendar = config->calendar().empty() ? "null" : config->calendar().name();
295  calInfo->baseDate = curve_->baseDate();
296  for (Size i = 0; i < pillarDates.size(); ++i) {
297  calInfo->pillarDates.push_back(pillarDates[i]);
298  calInfo->yoyRates.push_back(yoyCurve->yoyRate(pillarDates[i], 0 * Days));
299  calInfo->times.push_back(yoyCurve->timeFromReference(pillarDates[i]));
300  }
301  calibrationInfo_ = calInfo;
302  }
303 
304  if (config->type() == InflationCurveConfig::Type::ZC) {
305  auto zcCurve = boost::dynamic_pointer_cast<ZeroInflationTermStructure>(curve_);
306  QL_REQUIRE(zcCurve, "internal error: expected ZeroInflationCurve (inflation curve builder)");
307  auto zcIndex = conv->index()->clone(Handle<ZeroInflationTermStructure>(zcCurve));
308  auto calInfo = boost::make_shared<ZeroInflationCurveCalibrationInfo>();
309  calInfo->dayCounter = config->dayCounter().name();
310  calInfo->calendar = config->calendar().empty() ? "null" : config->calendar().name();
311  calInfo->baseDate = curve_->baseDate();
312  auto lim = inflationPeriod(curve_->baseDate(), curve_->frequency());
313  try {
314  calInfo->baseCpi = conv->index()->fixing(lim.first);
315  } catch (...) {
316  }
317  for (Size i = 0; i < pillarDates.size(); ++i) {
318  calInfo->pillarDates.push_back(pillarDates[i]);
319  calInfo->zeroRates.push_back(zcCurve->zeroRate(pillarDates[i], 0 * Days));
320  calInfo->times.push_back(zcCurve->timeFromReference(pillarDates[i]));
321  Real cpi = 0.0;
322  try {
323  cpi = zcIndex->fixing(pillarDates[i]);
324  } catch (...) {
325  }
326  calInfo->forwardCpis.push_back(cpi);
327  }
328  calibrationInfo_ = calInfo;
329  }
330  }
331 
332  } catch (std::exception& e) {
333  QL_FAIL("inflation curve building failed: " << e.what());
334  } catch (...) {
335  QL_FAIL("inflation curve building failed: unknown error");
336  }
337 }
