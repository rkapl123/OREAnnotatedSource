39LocalVolModelBuilder::LocalVolModelBuilder(
40 const std::vector<Handle<YieldTermStructure>>& curves,
41 const std::vector<boost::shared_ptr<GeneralizedBlackScholesProcess>>& processes,
42 const std::set<Date>& simulationDates, const std::set<Date>& addDates, const Size timeStepsPerYear,
43 const Type lvType, const std::vector<Real>& calibrationMoneyness, const bool dontCalibrate)
44 : BlackScholesModelBuilderBase(curves, processes, simulationDates, addDates, timeStepsPerYear), lvType_(lvType),
45 calibrationMoneyness_(calibrationMoneyness), dontCalibrate_(dontCalibrate) {
46 // we have to observe the whole vol surface for the Dupire implementation unfortunately; we can specify the time
47 // steps that are relevant, but not a set of discrete strikes
48 if (lvType == Type::Dupire) {
49 for (auto const& p : processes_) {
50 marketObserver_->registerWith(p->blackVolatility());
51 }
52 }
53}

55std::vector<boost::shared_ptr<GeneralizedBlackScholesProcess>> LocalVolModelBuilder::getCalibratedProcesses() const {
56
57 QL_REQUIRE(lvType_ != Type::AndreasenHuge || !calibrationMoneyness_.empty(), "no calibration moneyness provided");
58
59 std::vector<boost::shared_ptr<GeneralizedBlackScholesProcess>> processes;
60
61 for (Size l = 0; l < processes_.size(); ++l) {
62
63 Handle<LocalVolTermStructure> localVol;
64 if (dontCalibrate_) {
65 localVol = Handle<LocalVolTermStructure>(
66 boost::make_shared<LocalConstantVol>(0, NullCalendar(), 0.10, ActualActual(ActualActual::ISDA)));
67 } else if (lvType_ == Type::AndreasenHuge) {
68 // for checking arbitrage free input prices, just for logging purposes at this point
69 // notice that we need a uniform strike grid here, so this is not the same as the one below
70 // we choose the strike grid to be the one for the last calibration point
71 std::vector<Real> checkMaturities, checkMoneynesses, atmForwards;
72 std::vector<std::vector<Real>> callPrices;
73
74 // set up Andreasen Huge Volatility interpolation for each underyling
75 // we choose the calibration set to be OTM options on the effective future simulation dates
76 // with strikes given in terms of moneyness K / atmForward
77 AndreasenHugeVolatilityInterpl::CalibrationSet calSet;
78 for (auto const& d : effectiveSimulationDates_) {
79 if (d <= curves_.front()->referenceDate())
80 continue;
81 Real t = processes_.front()->riskFreeRate()->timeFromReference(d);
82 checkMaturities.push_back(t);
83 Real atmLevel =
84 atmForward(processes_[l]->x0(), processes_[l]->riskFreeRate(), processes_[l]->dividendYield(), t);
85 Real atmMarketVol = std::max(1e-4, processes_[l]->blackVolatility()->blackVol(t, atmLevel));
86 callPrices.push_back(std::vector<Real>());
87 atmForwards.push_back(atmLevel);
88 for (Size i = 0; i < calibrationMoneyness_.size(); ++i) {
89 Real strike = atmLevel * std::exp(calibrationMoneyness_[i] * atmMarketVol * std::sqrt(t));
90 Real marketVol = processes_[l]->blackVolatility()->blackVol(t, strike);
91 // skip option with effective moneyness < 0.0001 or > 0.9999 (TODO, hardcoded limits here?)
92 if (std::fabs(calibrationMoneyness_[i]) > 3.72)
93 continue;
94 auto option =
95 boost::make_shared<VanillaOption>(boost::make_shared<PlainVanillaPayoff>(Option::Call, strike),
96 boost::make_shared<EuropeanExercise>(d));
97 calSet.push_back(std::make_pair(option, boost::make_shared<SimpleQuote>(marketVol)));
98 option->setPricingEngine(boost::make_shared<AnalyticEuropeanEngine>(processes_[l]));
99 callPrices.back().push_back(option->NPV());
100 if (d == *effectiveSimulationDates_.rbegin()) {
101 checkMoneynesses.push_back(strike / atmLevel);
102 }
103 }
104 }
105
106 // arbitrage check
107 QuantExt::CarrMadanSurface cmCheck(checkMaturities, checkMoneynesses, processes_[l]->x0(), atmForwards,
108 callPrices);
109 if (!cmCheck.arbitrageFree()) {
110 WLOG("Andreasen-Huge local vol calibration for process #" << l
111 << ":, input vol is not arbitrage free:");
112 DLOG("time,moneyness,callSpread,butterfly,calendar");
113 for (Size i = 0; i < checkMaturities.size(); ++i)
114 for (Size j = 0; j < checkMoneynesses.size(); ++j)
115 DLOG(checkMaturities[i] << "," << checkMoneynesses[i] << "," << std::boolalpha
116 << cmCheck.callSpreadArbitrage()[i][j] << ","
117 << cmCheck.butterflyArbitrage()[i][j] << ","
118 << cmCheck.calendarArbitrage()[i][j]);
119 }
120
121 // TODO using some hardcoded values here, expose to configuration?
122 auto ah = boost::make_shared<AndreasenHugeVolatilityInterpl>(
123 calSet, processes_[l]->stateVariable(), processes_[l]->riskFreeRate(), processes_[l]->dividendYield(),
124 AndreasenHugeVolatilityInterpl::CubicSpline, AndreasenHugeVolatilityInterpl::Call, 500, Null<Real>(),
125 Null<Real>());
126 localVol = Handle<LocalVolTermStructure>(boost::make_shared<AndreasenHugeLocalVolAdapter>(ah));
127 DLOG("Andreasen-Huge local vol calibration for process #"
128 << l
129 << ": "
130 "calibration error min="
131 << std::scientific << std::setprecision(6) << boost::get<0>(ah->calibrationError()) << " max="
132 << boost::get<1>(ah->calibrationError()) << " avg=" << boost::get<2>(ah->calibrationError()));
133 } else if (lvType_ == Type::Dupire) {
134 localVol = Handle<LocalVolTermStructure>(
135 boost::make_shared<LocalVolSurface>(processes_[l]->blackVolatility(), processes_[l]->riskFreeRate(),
136 processes_[l]->dividendYield(), processes_[l]->stateVariable()));
137 } else {
138 QL_FAIL("unexpected local vol type");
139 }
140
141 processes.push_back(boost::make_shared<GeneralizedBlackScholesProcess>(
142 processes_[l]->stateVariable(), processes_[l]->dividendYield(), processes_[l]->riskFreeRate(),
143 processes_[l]->blackVolatility(), localVol));
144 }
145
146 return processes;
147}

149std::vector<std::vector<Real>> LocalVolModelBuilder::getCurveTimes() const {
150 std::vector<Real> timesExt(discretisationTimeGrid_.begin() + 1, discretisationTimeGrid_.end());
151 for (auto const& d : addDates_) {
152 if (d > curves_.front()->referenceDate()) {
153 timesExt.push_back(curves_.front()->timeFromReference(d));
154 }
155 }
156 std::sort(timesExt.begin(), timesExt.end());
157 auto it = std::unique(timesExt.begin(), timesExt.end(),
158 [](const Real x, const Real y) { return QuantLib::close_enough(x, y); });
159 timesExt.resize(std::distance(timesExt.begin(), it));
160 return std::vector<std::vector<Real>>(allCurves_.size(), timesExt);
161}

163std::vector<std::vector<std::pair<Real, Real>>> LocalVolModelBuilder::getVolTimesStrikes() const {
164 std::vector<std::vector<std::pair<Real, Real>>> volTimesStrikes;
165 // for the Dupire implementation we observe the whole vol surface anyhow (see ctor above)
166 if (lvType_ == Type::Dupire)
167 return volTimesStrikes;
168 std::vector<Real> times;
169 if (lvType_ == Type::AndreasenHuge) {
170 for (auto const& d : effectiveSimulationDates_) {
171 if (d > curves_.front()->referenceDate())
172 times.push_back(processes_.front()->riskFreeRate()->timeFromReference(d));
173 }
174 } else {
175 times = std::vector<Real>(discretisationTimeGrid_.begin() + 1, discretisationTimeGrid_.end());
176 }
177 for (auto const& p : processes_) {
178 volTimesStrikes.push_back(std::vector<std::pair<Real, Real>>());
179 for (auto const t : times) {
180 Real atmLevel = atmForward(p->x0(), p->riskFreeRate(), p->dividendYield(), t);
181 Real atmMarketVol = std::max(1e-4, p->blackVolatility()->blackVol(t, atmLevel));
182 for (auto const m : calibrationMoneyness_) {
183 Real strike = atmLevel * std::exp(m * atmMarketVol * std::sqrt(t));
184 volTimesStrikes.back().push_back(std::make_pair(t, strike));
185 }
186 }
187 }
188 return volTimesStrikes;
189}
