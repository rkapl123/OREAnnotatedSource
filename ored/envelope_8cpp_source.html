27void Envelope::fromXML(XMLNode* node) {
28 XMLUtils::checkNode(node, "Envelope");
29 counterparty_ = XMLUtils::getChildValue(node, "CounterParty", true);
30
31 XMLNode* nettingSetDetailsNode = XMLUtils::getChildNode(node, "NettingSetDetails");
32 if (nettingSetDetailsNode) {
33 nettingSetDetails_.fromXML(nettingSetDetailsNode);
34 } else {
35 string nettingSetId = XMLUtils::getChildValue(node, "NettingSetId", false);
36 nettingSetDetails_ = NettingSetDetails(nettingSetId);
37 }
38
39 portfolioIds_.clear();
40 XMLNode* portfolioNode = XMLUtils::getChildNode(node, "PortfolioIds");
41 if (portfolioNode) {
42 for (auto const& c : XMLUtils::getChildrenNodes(portfolioNode, "PortfolioId"))
43 portfolioIds_.insert(XMLUtils::getNodeValue(c));
44 }
45
46 std::function<boost::any(XMLNode*)> getValue;
47 getValue = [&getValue](XMLNode* node) {
48 boost::any value;
49 vector<XMLNode*> children = XMLUtils::getChildrenNodes(node, "");
50 // If node is a single-value node
51 if (children.size() == 1 && XMLUtils::getNodeName(children[0]) == "") {
52 value = XMLUtils::getNodeValue(node);
53 } else {
54 std::multimap<string, boost::any> subAddFields;
55 for (XMLNode* child : children) {
56 const string& name = XMLUtils::getNodeName(child);
57 boost::any childValue = getValue(child);
58 subAddFields.insert({name, childValue});
59 }
60 value = subAddFields;
61 }
62 return value;
63 };
64
65 additionalFields_.clear();
66 XMLNode* additionalNode = XMLUtils::getChildNode(node, "AdditionalFields");
67 if (additionalNode) {
68 for (XMLNode* child = XMLUtils::getChildNode(additionalNode); child; child = XMLUtils::getNextSibling(child)) {
69 additionalFields_[XMLUtils::getNodeName(child)] = getValue(child);
70 }
71 }
72}

74XMLNode* Envelope::toXML(XMLDocument& doc) {
75 XMLNode* node = doc.allocNode("Envelope");
76 XMLUtils::addChild(doc, node, "CounterParty", counterparty_);
77 if (nettingSetDetails_.emptyOptionalFields()) {
78 XMLUtils::addChild(doc, node, "NettingSetId", nettingSetDetails_.nettingSetId());
79 } else {
80 XMLUtils::appendNode(node, nettingSetDetails_.toXML(doc));
81 }
82 XMLNode* portfolioNode = doc.allocNode("PortfolioIds");
83 XMLUtils::appendNode(node, portfolioNode);
84 for (const auto& p : portfolioIds_)
85 XMLUtils::addChild(doc, portfolioNode, "PortfolioId", p);
86 XMLNode* additionalNode = doc.allocNode("AdditionalFields");
87 XMLUtils::appendNode(node, additionalNode);
88
89 std::function<void(XMLNode*, const string&, const boost::any&)> addChild;
90 addChild = [&addChild, &doc](XMLNode* node, const string& name, const boost::any& val) {
91 if (val.type() == typeid(string)) {
92 XMLUtils::addChild(doc, node, name, boost::any_cast<string>(val));
93 } else {
94 QL_REQUIRE(val.type() == typeid(std::multimap<string, boost::any>),
95 "Additional field type must be either string or map<string, boost::any>");
96 XMLNode* childNode = doc.allocNode(name);
97 XMLUtils::appendNode(node, childNode);
98 for (const auto& kv : boost::any_cast<std::multimap<string, boost::any>>(val)) {
99 addChild(childNode, kv.first, kv.second);
100 }
101 }
102 };
103
104 for (const auto& it : additionalFields_)
105 addChild(additionalNode, it.first, it.second);
106 return node;
107}
