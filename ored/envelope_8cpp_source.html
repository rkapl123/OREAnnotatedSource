27 void Envelope::fromXML(XMLNode* node) {
28  XMLUtils::checkNode(node, "Envelope");
29  counterparty_ = XMLUtils::getChildValue(node, "CounterParty", true);
30 
31  XMLNode* nettingSetDetailsNode = XMLUtils::getChildNode(node, "NettingSetDetails");
32  if (nettingSetDetailsNode) {
33  nettingSetDetails_.fromXML(nettingSetDetailsNode);
34  } else {
35  string nettingSetId = XMLUtils::getChildValue(node, "NettingSetId", false);
36  nettingSetDetails_ = NettingSetDetails(nettingSetId);
37  }
38 
39  portfolioIds_.clear();
40  XMLNode* portfolioNode = XMLUtils::getChildNode(node, "PortfolioIds");
41  if (portfolioNode) {
42  for (auto const& c : XMLUtils::getChildrenNodes(portfolioNode, "PortfolioId"))
43  portfolioIds_.insert(XMLUtils::getNodeValue(c));
44  }
45 
46  additionalFields_.clear();
47  XMLNode* additionalNode = XMLUtils::getChildNode(node, "AdditionalFields");
48  if (additionalNode) {
49  for (XMLNode* child = XMLUtils::getChildNode(additionalNode); child; child = XMLUtils::getNextSibling(child)) {
50  additionalFields_[XMLUtils::getNodeName(child)] = XMLUtils::getNodeValue(child);
51  }
52  }
53 }

55 XMLNode* Envelope::toXML(XMLDocument& doc) {
56  XMLNode* node = doc.allocNode("Envelope");
57  XMLUtils::addChild(doc, node, "CounterParty", counterparty_);
58  if (nettingSetDetails_.emptyOptionalFields()) {
59  XMLUtils::addChild(doc, node, "NettingSetId", nettingSetDetails_.nettingSetId());
60  } else {
61  XMLUtils::appendNode(node, nettingSetDetails_.toXML(doc));
62  }
63  XMLNode* portfolioNode = doc.allocNode("PortfolioIds");
64  XMLUtils::appendNode(node, portfolioNode);
65  for (const auto& p : portfolioIds_)
66  XMLUtils::addChild(doc, portfolioNode, "PortfolioId", p);
67  XMLNode* additionalNode = doc.allocNode("AdditionalFields");
68  XMLUtils::appendNode(node, additionalNode);
69  for (const auto& it : additionalFields_)
70  XMLUtils::addChild(doc, additionalNode, it.first, it.second);
71  return node;
72 }
