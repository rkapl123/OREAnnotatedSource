52Date parseDate(const string& s) {
53 // TODO: review
54
55 if (s == "")
56 return Date();
57
58 // guess formats from token number and sizes
59 // check permissible lengths
60 QL_REQUIRE((s.size() >= 3 && s.size() <= 6) || s.size() == 8 || s.size() == 10,
61 "invalid date format of \"" << s << "\", date string length 8 or 10 or between 3 and 6 required");
62
63 vector<string> tokens;
64 boost::split(tokens, s, boost::is_any_of("-/.:"));
65
66 if (tokens.size() == 1) {
67 if (s.size() == 8) {
68 // yyyymmdd
69 int y = parseInteger(s.substr(0, 4));
70 int m = parseInteger(s.substr(4, 2));
71 int d = parseInteger(s.substr(6, 2));
72 return Date(d, Month(m), y);
73 } else if (s.size() >= 3 && s.size() <= 6) {
74 // Excel format
75 // Boundaries will be checked by Date constructor
76 // Boundaries are minDate = 367 i.e. Jan 1st, 1901
77 // and maxDate = 109574 i.e. Dec 31st, 2199
78 BigInteger serial = parseInteger(s);
79 return Date(serial);
80 }
81 } else if (tokens.size() == 3) {
82 if (tokens[0].size() == 4) {
83 // yyyy-mm-dd
84 // yyyy/mm/dd
85 // yyyy.mm.dd
86 int y = parseInteger(tokens[0]);
87 int m = parseInteger(tokens[1]);
88 int d = parseInteger(tokens[2]);
89 return Date(d, Month(m), y);
90 } else if (tokens[0].size() == 2) {
91 // dd-mm-yy
92 // dd/mm/yy
93 // dd.mm.yy
94 // dd-mm-yyyy
95 // dd/mm/yyyy
96 // dd.mm.yyyy
97 int d = parseInteger(tokens[0]);
98 int m = parseInteger(tokens[1]);
99 int y = parseInteger(tokens[2]);
100 if (y < 100) {
101 if (y > 80)
102 y += 1900;
103 else
104 y += 2000;
105 }
106 return Date(d, Month(m), y);
107 }
108 }
109
110 QL_FAIL("Cannot convert \"" << s << "\" to Date.");
111}

156bool isOnePeriod(const string& s) {
157 if (s.empty())
158 return false;
159 char c = std::toupper(s.back());
160 if (!(c == 'D' || c == 'W' || c == 'M' || c == 'Y'))
161 return false;
162 for (auto c = s.cbegin(); c != std::next(s.end(), -1); std::advance(c, 1))
163 if (*c < '0' || *c > '9')
164 return false;
165 return true;
166}

800pair<string, string> parseBoostAny(const boost::any& anyType, Size precision) {
801 string resultType;
802 std::ostringstream oss;
803
804 if (anyType.type() == typeid(int)) {
805 resultType = "int";
806 int r = boost::any_cast<int>(anyType);
807 oss << std::fixed << std::setprecision(precision) << r;
808 } else if (anyType.type() == typeid(Size)) {
809 resultType = "size";
810 int r = boost::any_cast<Size>(anyType);
811 oss << std::fixed << std::setprecision(precision) << r;
812 } else if (anyType.type() == typeid(double)) {
813 resultType = "double";
814 double r = boost::any_cast<double>(anyType);
815 if(r != Null<Real>())
816 oss << std::fixed << std::setprecision(precision) << r;
817 } else if (anyType.type() == typeid(std::string)) {
818 resultType = "string";
819 std::string r = boost::any_cast<std::string>(anyType);
820 oss << std::fixed << std::setprecision(precision) << r;
821 } else if (anyType.type() == typeid(Date)) {
822 resultType = "date";
823 oss << io::iso_date(boost::any_cast<Date>(anyType));
824 } else if (anyType.type() == typeid(bool)) {
825 resultType = "bool";
826 oss << std::boolalpha << boost::any_cast<bool>(anyType);
827 } else if (anyType.type() == typeid(std::vector<bool>)) {
828 resultType = "vector_bool";
829 std::vector<bool> r = boost::any_cast<std::vector<bool>>(anyType);
830 if (r.size() == 0) {
831 oss << "";
832 } else {
833 oss << std::boolalpha << "\"" << boost::any_cast<bool>(anyType);
834 for (Size i = 1; i < r.size(); i++) {
835 oss << ", " << r[i];
836 }
837 oss << "\"";
838 }
839 } else if (anyType.type() == typeid(std::vector<double>)) {
840 resultType = "vector_double";
841 std::vector<double> r = boost::any_cast<std::vector<double>>(anyType);
842 if (r.size() == 0) {
843 oss << "";
844 } else {
845 oss << std::fixed << std::setprecision(precision) << "\"";
846 if (r[0] != Null<Real>())
847 oss << r[0];
848 for (Size i = 1; i < r.size(); i++) {
849 oss << ", ";
850 if (r[i] != Null<Real>())
851 oss << r[i];
852 }
853 oss << "\"";
854 }
855 } else if (anyType.type() == typeid(std::vector<Date>)) {
856 resultType = "vector_date";
857 std::vector<Date> r = boost::any_cast<std::vector<Date>>(anyType);
858 if (r.size() == 0) {
859 oss << "";
860 } else {
861 oss << std::fixed << std::setprecision(precision) << "\"" << to_string(r[0]);
862 for (Size i = 1; i < r.size(); i++) {
863 oss << ", " << to_string(r[i]);
864 }
865 oss << "\"";
866 }
867 } else if (anyType.type() == typeid(std::vector<std::string>)) {
868 resultType = "vector_string";
869 std::vector<std::string> r = boost::any_cast<std::vector<std::string>>(anyType);
870 if (r.size() == 0) {
871 oss << "";
872 } else {
873 oss << std::fixed << std::setprecision(precision) << "\"" << r[0];
874 for (Size i = 1; i < r.size(); i++) {
875 oss << ", " << r[i];
876 }
877 oss << "\"";
878 }
879 } else if (anyType.type() == typeid(std::vector<CashFlowResults>)) {
880 resultType = "vector_cashflows";
881 std::vector<CashFlowResults> r = boost::any_cast<std::vector<CashFlowResults>>(anyType);
882 if (!r.empty()) {
883 oss << std::fixed << std::setprecision(precision) << "\"" << r[0];
884 for (Size i = 1; i < r.size(); ++i) {
885 oss << ", " << r[i];
886 }
887 oss << "\"";
888 }
889 } else if (anyType.type() == typeid(QuantLib::Matrix)) {
890 resultType = "matrix";
891 QuantLib::Matrix r = boost::any_cast<QuantLib::Matrix>(anyType);
892 std::regex pattern("\n");
893 std::ostringstream tmp;
894 tmp << std::setprecision(precision) << r;
895 oss << std::fixed << std::regex_replace(tmp.str(), pattern, std::string(""));
896 } else {
897 ALOG("Unsupported Boost::Any type");
898 resultType = "unsupported_type";
899 }
900 return make_pair(resultType, oss.str());
901}

1274 else if (s2 == "JPY")
1275 return s1 + s2;
1276 else {
1277 if (p1 != dominance.end())
1278 return s1 + s2;
1279 else
1280 return s2 + s1;
1281 }
