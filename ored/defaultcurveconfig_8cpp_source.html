40void DefaultCurveConfig::populateRequiredCurveIds(const std::string& discountCurveID,
41 const std::string& benchmarkCurveID, const std::string& sourceCurveID,
42 const std::vector<std::string>& multiSectionSourceCurveIds) {
43 if (!discountCurveID.empty())
44 requiredCurveIds_[CurveSpec::CurveType::Yield].insert(parseCurveSpec(discountCurveID)->curveConfigID());
45 if (!benchmarkCurveID.empty())
46 requiredCurveIds_[CurveSpec::CurveType::Yield].insert(parseCurveSpec(benchmarkCurveID)->curveConfigID());
47 if (!sourceCurveID.empty())
48 requiredCurveIds_[CurveSpec::CurveType::Yield].insert(parseCurveSpec(sourceCurveID)->curveConfigID());
49 for (auto const& s : multiSectionSourceCurveIds) {
50 if (!s.empty())
51 requiredCurveIds_[CurveSpec::CurveType::Default].insert(parseCurveSpec(s)->curveConfigID());
52 }
53}

62void DefaultCurveConfig::populateQuotes() {
63 quotes_.clear();
64 for (auto const& config : configs_) {
65 for (const auto& kv : config.second.cdsQuotes()) {
66 quotes_.push_back(kv.first);
67 }
68 // recovery rate might be a hardcoded number, in which case we must not add this to the set of quotes
69 Real dummy;
70 if (!config.second.recoveryRateQuote().empty() && !tryParseReal(config.second.recoveryRateQuote(), dummy))
71 quotes_.insert(quotes_.begin(), config.second.recoveryRateQuote());
72 }
73}

75void DefaultCurveConfig::fromXML(XMLNode* node) {
76 XMLUtils::checkNode(node, "DefaultCurve");
77 curveID_ = XMLUtils::getChildValue(node, "CurveId", true);
78 curveDescription_ = XMLUtils::getChildValue(node, "CurveDescription", true);
79 currency_ = XMLUtils::getChildValue(node, "Currency", true);
80 if (auto configs = XMLUtils::getChildNode(node, "Configurations")) {
81 for (auto const& config : XMLUtils::getChildrenNodes(configs, "Configuration")) {
82 Config tmp;
83 tmp.fromXML(config);
84 QL_REQUIRE(configs_.find(tmp.priority()) == configs_.end(),
85 "DefaultCurveConfig::fromXML(): several configurations with same priority '" << tmp.priority()
86 << "' found.");
87 configs_[tmp.priority()] = tmp;
88 }
89 } else {
90 Config tmp;
91 tmp.fromXML(node);
92 configs_[0] = tmp;
93 }
94 populateQuotes();
95 populateRequiredCurveIds();
96}

129void DefaultCurveConfig::Config::fromXML(XMLNode* node) {
130 auto prioStr = XMLUtils::getAttribute(node, "priority");
131 if (!prioStr.empty())
132 priority_ = parseInteger(prioStr);
133 cdsQuotes_.clear();
134 string type = XMLUtils::getChildValue(node, "Type", true);
135 if (type == "SpreadCDS") {
136 type_ = Type::SpreadCDS;
137 } else if (type == "HazardRate") {
138 type_ = Type::HazardRate;
139 } else if (type == "Price") {
140 type_ = Type::Price;
141 } else if (type == "Benchmark") {
142 type_ = Type::Benchmark;
143 } else if (type == "MultiSection") {
144 type_ = Type::MultiSection;
145 } else if (type == "Null") {
146 type_ = Type::Null;
147 } else {
148 QL_FAIL("Type " << type << " not recognized");
149 }
150 string dc = XMLUtils::getChildValue(node, "DayCounter", true);
151 dayCounter_ = parseDayCounter(dc);
152 extrapolation_ = XMLUtils::getChildValueAsBool(node, "Extrapolation"); // defaults to true
153 allowNegativeRates_ = false;
154 if (XMLNode* n = XMLUtils::getChildNode(node, "AllowNegativeRates")) {
155 allowNegativeRates_ = parseBool(XMLUtils::getNodeValue(n));
156 }
157 if (type_ == Type::Benchmark) {
158 benchmarkCurveID_ = XMLUtils::getChildValue(node, "BenchmarkCurve", true);
159 sourceCurveID_ = XMLUtils::getChildValue(node, "SourceCurve", true);
160 pillars_ = XMLUtils::getChildrenValuesAsStrings(node, "Pillars", true);
161 spotLag_ = parseInteger(XMLUtils::getChildValue(node, "SpotLag", true));
162 calendar_ = parseCalendar(XMLUtils::getChildValue(node, "Calendar", true));
163 discountCurveID_ = conventionID_ = "";
164 recoveryRateQuote_ = XMLUtils::getChildValue(node, "RecoveryRate", false);
165 } else if (type_ == Type::MultiSection) {
166 multiSectionSourceCurveIds_ = XMLUtils::getChildrenValues(node, "SourceCurves", "SourceCurve", true);
167 multiSectionSwitchDates_ = XMLUtils::getChildrenValues(node, "SwitchDates", "SwitchDate", true);
168 discountCurveID_ = conventionID_ = "";
169 recoveryRateQuote_ = XMLUtils::getChildValue(node, "RecoveryRate", false);
170 } else {
171 discountCurveID_ = XMLUtils::getChildValue(node, "DiscountCurve", false);
172 conventionID_ = XMLUtils::getChildValue(node, "Conventions", true);
173 XMLNode* quotesNode = XMLUtils::getChildNode(node, "Quotes");
174 if (quotesNode) {
175 for (auto n : XMLUtils::getChildrenNodes(quotesNode, "Quote")) {
176 string attr = XMLUtils::getAttribute(n, "optional");
177 bool opt = (!attr.empty() && parseBool(attr));
178 cdsQuotes_.emplace_back(make_pair(XMLUtils::getNodeValue(n), opt));
179 }
180 }
181 recoveryRateQuote_ = XMLUtils::getChildValue(node, "RecoveryRate", false);
182 benchmarkCurveID_ = sourceCurveID_ = "";
183 calendar_ = Calendar();
184 spotLag_ = 0;
185 pillars_.clear();
186 // Read the optional start date
187 string d = XMLUtils::getChildValue(node, "StartDate", false);
188 if (d != "") {
189 if (type_ == Type::SpreadCDS || type_ == Type::Price) {
190 startDate_ = parseDate(d);
191 } else {
192 WLOG("'StartDate' is only used when type is 'SpreadCDS' or 'Price'");
193 }
194 }
195 string s = XMLUtils::getChildValue(node, "RunningSpread", false);
196 if (s.empty() && type_ == Type::Price) {
197 DLOG("'RunningSpread' is empty and type is 'Price' for default curve "
198 "so the running spread will need to be provided in the market quote.");
199 }
200 if (!s.empty()) {
201 runningSpread_ = parseReal(s);
202 }
203 string t = XMLUtils::getChildValue(node, "IndexTerm", false);
204 indexTerm_ = t.empty() ? 0 * Days : parsePeriod(t);
205 implyDefaultFromMarket_ = boost::none;
206 if (XMLNode* n = XMLUtils::getChildNode(node, "ImplyDefaultFromMarket"))
207 implyDefaultFromMarket_ = parseBool(XMLUtils::getNodeValue(n));
208 // Optional bootstrap configuration
209 if (XMLNode* n = XMLUtils::getChildNode(node, "BootstrapConfig")) {
210 bootstrapConfig_.fromXML(n);
211 }
212 }

215XMLNode* DefaultCurveConfig::Config::toXML(XMLDocument& doc) {
216 XMLNode* node = doc.allocNode("Configuration");
217 XMLUtils::addAttribute(doc, node, "priority", std::to_string(priority_));
218 if (type_ == Type::SpreadCDS || type_ == Type::HazardRate || type_ == Type::Price) {
219 if (type_ == Type::SpreadCDS) {
220 XMLUtils::addChild(doc, node, "Type", "SpreadCDS");
221 } else if (type_ == Type::HazardRate) {
222 XMLUtils::addChild(doc, node, "Type", "HazardRate");
223 } else {
224 XMLUtils::addChild(doc, node, "Type", "Price");
225 }
226 XMLUtils::addChild(doc, node, "DiscountCurve", discountCurveID_);
227 XMLUtils::addChild(doc, node, "DayCounter", to_string(dayCounter_));
228 XMLUtils::addChild(doc, node, "RecoveryRate", recoveryRateQuote_);
229 XMLNode* quotesNode = XMLUtils::addChild(doc, node, "Quotes");
230 for (auto q : cdsQuotes_) {
231 XMLNode* qNode = doc.allocNode("Quote", q.first);
232 if (q.second)
233 XMLUtils::addAttribute(doc, qNode, "optional", "true");
234 XMLUtils::appendNode(quotesNode, qNode);
235 }
236 } else if (type_ == Type::Benchmark) {
237 XMLUtils::addChild(doc, node, "Type", "Benchmark");
238 XMLUtils::addChild(doc, node, "DayCounter", to_string(dayCounter_));
239 XMLUtils::addChild(doc, node, "RecoveryRate", recoveryRateQuote_);
240 XMLUtils::addChild(doc, node, "BenchmarkCurve", benchmarkCurveID_);
241 XMLUtils::addChild(doc, node, "SourceCurve", sourceCurveID_);
242 XMLUtils::addGenericChildAsList(doc, node, "Pillars", pillars_);
243 XMLUtils::addChild(doc, node, "SpotLag", (int)spotLag_);
244 XMLUtils::addChild(doc, node, "Calendar", calendar_.name());
245 } else if (type_ == Type::MultiSection) {
246 XMLUtils::addChild(doc, node, "RecoveryRate", recoveryRateQuote_);
247 XMLUtils::addChildren(doc, node, "SourceCurves", "SourceCurve", multiSectionSourceCurveIds_);
248 XMLUtils::addChildren(doc, node, "SwitchDates", "SwitchDate", multiSectionSwitchDates_);
249 } else if (type_ == Type::Null) {
250 XMLUtils::addChild(doc, node, "Type", "Null");
251 XMLUtils::addChild(doc, node, "DayCounter", to_string(dayCounter_));
252 XMLUtils::addChild(doc, node, "DiscountCurve", discountCurveID_);
253 } else {
254 QL_FAIL("Unknown type in DefaultCurveConfig::toXML()");
255 }
256 XMLUtils::addChild(doc, node, "Conventions", conventionID_);
257 XMLUtils::addChild(doc, node, "Extrapolation", extrapolation_);
258 if (startDate_ != Date())
259 XMLUtils::addChild(doc, node, "StartDate", to_string(startDate_));
260 if (runningSpread_ != QuantLib::Null<Real>())
261 XMLUtils::addChild(doc, node, "RunningSpread", to_string(runningSpread_));
262 if (indexTerm_ != 0 * Days) {
263 XMLUtils::addChild(doc, node, "IndexTerm", indexTerm_);
264 }
265 if (implyDefaultFromMarket_)
266 XMLUtils::addChild(doc, node, "ImplyDefaultFromMarket", *implyDefaultFromMarket_);
267 XMLUtils::appendNode(node, bootstrapConfig_.toXML(doc));
268 XMLUtils::addChild(doc, node, "AllowNegativeRates", allowNegativeRates_);
269 return node;
270}
