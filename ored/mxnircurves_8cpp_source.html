51 BOOST_AUTO_TEST_CASE(testSingleCurrencyYieldCurveBootstrap) {
52

73  engineData->fromFile(TEST_INPUT_FILE("pricingengine_01.xml"));

142  // Portfolio to test market
143  boost::shared_ptr<EngineData> engineData = boost::make_shared<EngineData>();
144  engineData->fromFile(TEST_INPUT_FILE("pricingengine_03.xml"));
145  boost::shared_ptr<EngineFactory> factory = boost::make_shared<EngineFactory>(engineData, market);
146  boost::shared_ptr<Portfolio> portfolio = boost::make_shared<Portfolio>();
147  portfolio->load(TEST_INPUT_FILE("mxn_ir_cap.xml"));
148  portfolio->build(factory);
149 
150  // The single trade in the portfolio is a MXN 10Y cap, i.e. 10 x 13 28D coupons (without first caplet), with
151  // nominal USD 100 million.
152  BOOST_CHECK_EQUAL(portfolio->size(), 1);
153 
154  // Get the npv of the trade using the market i.e. the stripped optionlet surface from TodaysMarket
155  Real npvTodaysMarket = portfolio->trades()[0]->instrument()->NPV();
156  BOOST_TEST_MESSAGE("NPV using TodaysMarket is: " << npvTodaysMarket);
157 
158  // Price the same cap using the constant volatility from the market
159  BOOST_REQUIRE(portfolio->trades()[0]);
160  BOOST_REQUIRE(portfolio->trades()[0]->legs().size() == 1);
161  auto pricer = boost::make_shared<BlackIborCouponPricer>(Handle<OptionletVolatilityStructure>(
162  boost::make_shared<ConstantOptionletVolatility>(0, NullCalendar(), Unadjusted, 0.20320, Actual365Fixed())));
163  Leg leg = portfolio->trades()[0]->legs().front();
164  setCouponPricer(leg, pricer);
165  Real npvMarketVol = CashFlows::npv(leg, **market->discountCurve("MXN"), false);
166  BOOST_TEST_MESSAGE("NPV using the constant market volatility is: " << npvMarketVol);
167 
168  // Check the difference in the NPVs. Should be small if the stripping is working correctly.
169  BOOST_CHECK_SMALL(fabs(npvTodaysMarket - npvMarketVol), 0.01);
170 }
