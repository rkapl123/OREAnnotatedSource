62 void CommodityForward::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
63 
64  // Create the underlying commodity index for the forward
65  const boost::shared_ptr<Market>& market = engineFactory->market();
66  auto index = *market->commodityIndex(commodityName_, engineFactory->configuration(MarketContext::pricing));
67  maturity_ = parseDate(maturityDate_);
68  if (isFuturePrice_ && *isFuturePrice_) {
69 
70  // Get the commodity index from the market.
71  index = *market->commodityIndex(commodityName_, engineFactory->configuration(MarketContext::pricing));
72 
73  // May have been given an explicit future expiry date or an offset and calendar or neither.
74  Date expiryDate = maturity_;
75  if (futureExpiryDate_ != Date()) {
76  expiryDate = futureExpiryDate_;
77  } else if (futureExpiryOffset_ != Period()) {
78  Calendar cal = offsetCalendar_.empty() ? NullCalendar() : offsetCalendar_;
79  expiryDate = cal.advance(maturity_, futureExpiryOffset_);
80  }
81 
82  // Clone the index with the relevant expiry date.
83  index = index->clone(expiryDate);
84 
85  }
86 
87  Date paymentDate = paymentDate_;
88  bool physicallySettled = true;
89  if (physicallySettled_ && !(*physicallySettled_)) {
90  // If cash settled and given a payment date that is not greater than the maturity date, set it equal to the
91  // maturity date and log a warning to continue processing.
92  physicallySettled = false;
93  if (paymentDate_ != Date() && paymentDate_ < maturity_) {
94  WLOG("Commodity forward " << id() << " has payment date (" << io::iso_date(paymentDate_) <<
95  ") before the maturity date (" << io::iso_date(maturity_) << "). Setting payment date" <<
96  " equal to the maturity date.");
97  paymentDate = maturity_;
98  }
99  } else {
100  // If physically settled and given a payment date, log a warning that it is ignored.
101  if (paymentDate_ != Date()) {
102  WLOG("Commodity forward " << id() << " supplies a payment date (" << io::iso_date(paymentDate_) <<
103  ") but is physically settled. The payment date is ignored.");
104  paymentDate = Date();
105  }
106  }
107 
108  // Create the commodity forward instrument
109  Currency currency = parseCurrency(currency_);
110  Position::Type position = parsePositionType(position_);
111  boost::shared_ptr<Instrument> commodityForward = boost::make_shared<QuantExt::CommodityForward>(
112  index, currency, position, quantity_, maturity_, strike_, physicallySettled, paymentDate);
113 
114  // Pricing engine
115  boost::shared_ptr<EngineBuilder> builder = engineFactory->builder(tradeType_);
116  QL_REQUIRE(builder, "No builder found for " << tradeType_);
117  boost::shared_ptr<CommodityForwardEngineBuilder> commodityForwardEngineBuilder =
118  boost::dynamic_pointer_cast<CommodityForwardEngineBuilder>(builder);
119  commodityForward->setPricingEngine(commodityForwardEngineBuilder->engine(currency));
120 
121  // set up other Trade details
122  instrument_ = boost::make_shared<VanillaInstrument>(commodityForward);
123  npvCurrency_ = currency_;
124 
125  // notional_ = strike_ * quantity_;
126  notional_ = Null<Real>(); // is handled by override of notional()
127  notionalCurrency_ = currency_;
128 
129  additionalData_["quantity"] = quantity_;
130  additionalData_["strike"] = strike_;
131  additionalData_["strikeCurrency"] = currency_;
132 }

134 Real CommodityForward::notional() const {
135  // try to get the notional from the additional results of the instrument
136  try {
137  return instrument_->qlInstrument()->result<Real>("currentNotional");
138  } catch (const std::exception& e) {
139  if (strcmp(e.what(), "currentNotional not provided"))
140  ALOG("error when retrieving notional: " << e.what());
141  }
142  // if not provided, return null
143  return Null<Real>();
144 }

151 void CommodityForward::fromXML(XMLNode* node) {
152 
153  Trade::fromXML(node);
154  XMLNode* commodityDataNode = XMLUtils::getChildNode(node, "CommodityForwardData");
155 
156  position_ = XMLUtils::getChildValue(commodityDataNode, "Position", true);
157  commodityName_ = XMLUtils::getChildValue(commodityDataNode, "Name", true);
158  currency_ = XMLUtils::getChildValue(commodityDataNode, "Currency", true);
159  quantity_ = XMLUtils::getChildValueAsDouble(commodityDataNode, "Quantity", true);
160  maturityDate_ = XMLUtils::getChildValue(commodityDataNode, "Maturity", true);
161  strike_ = XMLUtils::getChildValueAsDouble(commodityDataNode, "Strike", true);
162 
163  isFuturePrice_ = boost::none;
164  if (XMLNode* n = XMLUtils::getChildNode(commodityDataNode, "IsFuturePrice"))
165  isFuturePrice_ = parseBool(XMLUtils::getNodeValue(n));
166 
167  futureExpiryDate_ = Date();
168  if (XMLNode* n = XMLUtils::getChildNode(commodityDataNode, "FutureExpiryDate"))
169  futureExpiryDate_ = parseDate(XMLUtils::getNodeValue(n));
170 
171  // If not given an explicit future expiry date, check for offset and calendar.
172  if (futureExpiryDate_ == Date()) {
173  futureExpiryOffset_ = Period();
174  if (XMLNode* n = XMLUtils::getChildNode(commodityDataNode, "FutureExpiryOffset"))
175  futureExpiryOffset_ = parsePeriod(XMLUtils::getNodeValue(n));
176 
177  offsetCalendar_ = Calendar();
178  if (XMLNode* n = XMLUtils::getChildNode(commodityDataNode, "FutureExpiryOffsetCalendar"))
179  offsetCalendar_ = parseCalendar(XMLUtils::getNodeValue(n));
180  }
181 
182  physicallySettled_ = boost::none;
183  if (XMLNode* n = XMLUtils::getChildNode(commodityDataNode, "PhysicallySettled"))
184  physicallySettled_ = parseBool(XMLUtils::getNodeValue(n));
185 
186  paymentDate_ = Date();
187  if (XMLNode* n = XMLUtils::getChildNode(commodityDataNode, "PaymentDate"))
188  paymentDate_ = parseDate(XMLUtils::getNodeValue(n));
189 }

206

208  XMLUtils::addChild(doc, commodityDataNode, "FutureExpiryDate", to_string(futureExpiryDate_));
209 
210  if (futureExpiryOffset_ != Period())
211  XMLUtils::addChild(doc, commodityDataNode, "FutureExpiryOffset", to_string(futureExpiryOffset_));
212 
213  if (offsetCalendar_ != Calendar())
214  XMLUtils::addChild(doc, commodityDataNode, "FutureExpiryOffsetCalendar", to_string(offsetCalendar_));
215 
216  if (physicallySettled_)
217  XMLUtils::addChild(doc, commodityDataNode, "PhysicallySettled", *physicallySettled_);
218 
219  if (paymentDate_ != Date())
220  XMLUtils::addChild(doc, commodityDataNode, "PaymentDate", to_string(paymentDate_));
221 
222  return node;
223 }
