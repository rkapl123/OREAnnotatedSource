67void CommodityForward::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
68
69 // Create the underlying commodity index for the forward
70 const boost::shared_ptr<Market>& market = engineFactory->market();
71 boost::shared_ptr<QuantExt::FxIndex> fxIndex = nullptr;
72 auto index = *market->commodityIndex(commodityName_, engineFactory->configuration(MarketContext::pricing));
73 maturity_ = parseDate(maturityDate_);
74 bool isFutureAccordingToConventions = InstrumentConventions::instance().conventions()->has(commodityName_, Convention::Type::CommodityFuture);
75
76 // adjust the maturity date if not a valid fixing date for the index
77 maturity_ = index->fixingCalendar().adjust(maturity_, Preceding);
78
79 if ((isFuturePrice_ && *isFuturePrice_) || isFutureAccordingToConventions) {
80
81 // Get the commodity index from the market.
82 index = *market->commodityIndex(commodityName_, engineFactory->configuration(MarketContext::pricing));
83
84 // May have been given an explicit future expiry date or an offset and calendar or neither.
85 Date expiryDate = maturity_;
86 if (futureExpiryDate_ != Date()) {
87 expiryDate = futureExpiryDate_;
88 } else if (futureExpiryOffset_ != Period()) {
89 Calendar cal = offsetCalendar_.empty() ? NullCalendar() : offsetCalendar_;
90 expiryDate = cal.advance(maturity_, futureExpiryOffset_);
91 }
92
93 // Clone the index with the relevant expiry date.
94 index = index->clone(expiryDate);
95 }
96
97 Date paymentDate = paymentDate_;
98 bool physicallySettled = true;
99 if (physicallySettled_ && !(*physicallySettled_)) {
100 // If cash settled and given a payment date that is not greater than the maturity date, set it equal to the
101 // maturity date and log a warning to continue processing.
102 physicallySettled = false;
103 if (paymentDate_ != Date() && paymentDate_ < maturity_) {
104 WLOG("Commodity forward " << id() << " has payment date (" << io::iso_date(paymentDate_) <<
105 ") before the maturity date (" << io::iso_date(maturity_) << "). Setting payment date" <<
106 " equal to the maturity date.");
107 paymentDate = maturity_;
108 }
109 } else {
110 // If physically settled and given a payment date, log a warning that it is ignored.
111 if (paymentDate_ != Date()) {
112 WLOG("Commodity forward " << id() << " supplies a payment date (" << io::iso_date(paymentDate_) <<
113 ") but is physically settled. The payment date is ignored.");
114 paymentDate = Date();
115 }
116 }
117
118 // add required commodity fixing
119 DLOG("commodity forward " << id() << " paymentDate is " << paymentDate);
120 requiredFixings_.addFixingDate(maturity_, index->name(),
121 paymentDate == Date() ? maturity_ : paymentDate);
122
123 // Create the commodity forward instrument
124 Currency currency = parseCurrency(currency_);
125 Position::Type position = parsePositionType(position_);
126 auto payCcy = Currency();
127 if(!fxIndex_.empty()){
128 payCcy = parseCurrency(payCcy_);
129 requiredFixings_.addFixingDate( fixingDate_, fxIndex_, paymentDate);
130 fxIndex = buildFxIndex(fxIndex_, currency.code(), payCcy.code(), engineFactory->market(),
131 engineFactory->configuration(MarketContext::pricing));
132 npvCurrency_ = payCcy_;
133 }
134 boost::shared_ptr<Instrument> commodityForward = boost::make_shared<QuantExt::CommodityForward>(
135 index, currency, position, quantity_, maturity_, strike_, physicallySettled, paymentDate,
136 payCcy, fixingDate_, fxIndex);
137
138 // Pricing engine
139 boost::shared_ptr<EngineBuilder> builder = engineFactory->builder(tradeType_);
140 QL_REQUIRE(builder, "No builder found for " << tradeType_);
141 boost::shared_ptr<CommodityForwardEngineBuilder> commodityForwardEngineBuilder =
142 boost::dynamic_pointer_cast<CommodityForwardEngineBuilder>(builder);
143 commodityForward->setPricingEngine(commodityForwardEngineBuilder->engine(currency)); // the engine accounts for NDF if settlement data are present
144
145 // set up other Trade details
146 instrument_ = boost::make_shared<VanillaInstrument>(commodityForward);
147 npvCurrency_ = fixingDate_==Date() ? currency_ : payCcy_;
148
149 // notional_ = strike_ * quantity_;
150 notional_ = Null<Real>(); // is handled by override of notional()
151 notionalCurrency_ = currency_;
152
153 additionalData_["quantity"] = quantity_;
154 additionalData_["strike"] = strike_;
155 additionalData_["strikeCurrency"] = currency_;
156 if(fixingDate_ != Date()) {
157 additionalData_["settlementCurrency"] = payCcy_;
158 additionalData_["fixingDate"] = fixingDate_;
159 additionalData_["fxIndex"] = fxIndex;
160 }
161
162 // ISDA taxonomy
163 additionalData_["isdaAssetClass"] = string("Commodity");
164 additionalData_["isdaBaseProduct"] = string("Forward");
165 additionalData_["isdaSubProduct"] = string("Price Return Basic Performance");
166 // skip the transaction level mapping for now
167 additionalData_["isdaTransaction"] = string("");
168}

170Real CommodityForward::notional() const {
171 // try to get the notional from the additional results of the instrument
172 try {
173 return instrument_->qlInstrument(true)->result<Real>("currentNotional");
174 } catch (const std::exception& e) {
175 if (strcmp(e.what(), "currentNotional not provided"))
176 ALOG("error when retrieving notional: " << e.what());
177 }
178 // if not provided, return null
179 return Null<Real>();
180}

187void CommodityForward::fromXML(XMLNode* node) {
188
189 Trade::fromXML(node);
190 XMLNode* commodityDataNode = XMLUtils::getChildNode(node, "CommodityForwardData");
191
192 position_ = XMLUtils::getChildValue(commodityDataNode, "Position", true);
193 commodityName_ = XMLUtils::getChildValue(commodityDataNode, "Name", true);
194 currency_ = XMLUtils::getChildValue(commodityDataNode, "Currency", true);
195 quantity_ = XMLUtils::getChildValueAsDouble(commodityDataNode, "Quantity", true);
196 maturityDate_ = XMLUtils::getChildValue(commodityDataNode, "Maturity", true);
197 strike_ = XMLUtils::getChildValueAsDouble(commodityDataNode, "Strike", true);
198
199 isFuturePrice_ = boost::none;
200 if (XMLNode* n = XMLUtils::getChildNode(commodityDataNode, "IsFuturePrice"))
201 isFuturePrice_ = parseBool(XMLUtils::getNodeValue(n));
202
203 futureExpiryDate_ = Date();
204 if (XMLNode* n = XMLUtils::getChildNode(commodityDataNode, "FutureExpiryDate"))
205 futureExpiryDate_ = parseDate(XMLUtils::getNodeValue(n));
206
207 // If not given an explicit future expiry date, check for offset and calendar.
208 if (futureExpiryDate_ == Date()) {
209 futureExpiryOffset_ = Period();
210 if (XMLNode* n = XMLUtils::getChildNode(commodityDataNode, "FutureExpiryOffset"))
211 futureExpiryOffset_ = parsePeriod(XMLUtils::getNodeValue(n));
212
213 offsetCalendar_ = Calendar();
214 if (XMLNode* n = XMLUtils::getChildNode(commodityDataNode, "FutureExpiryOffsetCalendar"))
215 offsetCalendar_ = parseCalendar(XMLUtils::getNodeValue(n));
216 }
217
218 physicallySettled_ = boost::none;
219 if (XMLNode* n = XMLUtils::getChildNode(commodityDataNode, "PhysicallySettled"))
220 physicallySettled_ = parseBool(XMLUtils::getNodeValue(n));
221
222 paymentDate_ = Date();
223 if (XMLNode* n = XMLUtils::getChildNode(commodityDataNode, "PaymentDate"))
224 paymentDate_ = parseDate(XMLUtils::getNodeValue(n));
225
226 if (XMLNode* settlementDataNode = XMLUtils::getChildNode(commodityDataNode, "SettlementData")) {
227 // this node is used to provide data for NDF. This includes a fixing date, a settlement currency and the
228 // quote/settlement fx index.
229 payCcy_ = XMLUtils::getChildValue(settlementDataNode, "PayCurrency", true);
230 fxIndex_ = XMLUtils::getChildValue(settlementDataNode, "FXIndex", true);
231 fixingDate_ = parseDate(XMLUtils::getChildValue(settlementDataNode, "FixingDate", true));
232 }
233}

235XMLNode* CommodityForward::toXML(XMLDocument& doc) {
236
237 XMLNode* node = Trade::toXML(doc);
238 XMLNode* commodityDataNode = doc.allocNode("CommodityForwardData");
239 XMLUtils::appendNode(node, commodityDataNode);
240
241 XMLUtils::addChild(doc, commodityDataNode, "Position", position_);
242 XMLUtils::addChild(doc, commodityDataNode, "Maturity", maturityDate_);
243 XMLUtils::addChild(doc, commodityDataNode, "Name", commodityName_);
244 XMLUtils::addChild(doc, commodityDataNode, "Currency", currency_);
245 XMLUtils::addChild(doc, commodityDataNode, "Strike", strike_);
246 XMLUtils::addChild(doc, commodityDataNode, "Quantity", quantity_);
247
248 if (isFuturePrice_)
249 XMLUtils::addChild(doc, commodityDataNode, "IsFuturePrice", *isFuturePrice_);
250
251 if (futureExpiryDate_ != Date())
252 XMLUtils::addChild(doc, commodityDataNode, "FutureExpiryDate", to_string(futureExpiryDate_));
253
254 if (futureExpiryOffset_ != Period())
255 XMLUtils::addChild(doc, commodityDataNode, "FutureExpiryOffset", to_string(futureExpiryOffset_));
256
257 if (offsetCalendar_ != Calendar())
258 XMLUtils::addChild(doc, commodityDataNode, "FutureExpiryOffsetCalendar", to_string(offsetCalendar_));
259
260 if (physicallySettled_)
261 XMLUtils::addChild(doc, commodityDataNode, "PhysicallySettled", *physicallySettled_);
262
263 if (paymentDate_ != Date())
264 XMLUtils::addChild(doc, commodityDataNode, "PaymentDate", to_string(paymentDate_));
265
266 if(fixingDate_!=Date()){ //NDF
267 XMLNode* settlementDataNode = doc.allocNode("SettlementData");
268 XMLUtils::appendNode(commodityDataNode, settlementDataNode);
269 XMLUtils::addChild(doc, settlementDataNode, "PayCurrency", payCcy_);
270 XMLUtils::addChild(doc, settlementDataNode, "FXIndex", fxIndex_);
271 XMLUtils::addChild(doc, settlementDataNode, "FixingDate", to_string(fixingDate_));
272 }
273
274 return node;
275}
