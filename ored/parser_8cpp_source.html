126void checkStrikeParser(const std::string& s, const ore::data::Strike::Type expectedType, const Real expectedValue) {
127 if (ore::data::parseStrike(s).type != expectedType) {
128 BOOST_FAIL("unexpected strike type parsed from input string " << s);
129 }
130 if (!close_enough(ore::data::parseStrike(s).value, expectedValue)) {
131 BOOST_FAIL("unexpected strike value parsed from input string " << s);
132 }
133 return;
134}
