53PseudoCurrencyMarketParameters buildPseudoCurrencyMarketParameters(const std::map<string, string>& pegp) {
54 PseudoCurrencyMarketParameters params;
55
56 // default
57 params.treatAsFX = true;
58
59 auto it = pegp.find("PseudoCurrency.TreatAsFX");
60 if (it != pegp.end()) {
61 DLOG("Building PseudoCurrencyMarketParameters from PricingEngine GlobalParameters");
62 try {
63 // Build the params
64 params.treatAsFX = parseBool(it->second);
65 it = pegp.find("PseudoCurrency.BaseCurrency");
66 QL_REQUIRE(it != pegp.end(), "No BaseCurrency field");
67 params.baseCurrency = it->second;
68 parseCurrency(params.baseCurrency); // to check it is valid
69
70 // Now just search for the 4 precious metals and 7 crypto currencies
71 for (const string& pm : CurrencyParser::instance().pseudoCurrencyCodes()) {
72 it = pegp.find("PseudoCurrency.Curve." + pm);
73 if (it != pegp.end())
74 params.curves[pm] = it->second;
75 }
76 QL_REQUIRE(!params.curves.empty(), "At least one PM Curve required");
77
78 // Look for fxIndexTag
79 it = pegp.find("PseudoCurrency.FXIndexTag");
80 QL_REQUIRE(it != pegp.end(), "No FXIndexTag field");
81 params.fxIndexTag = it->second;
82
83 // Look for Optional default correlation
84 it = pegp.find("PseudoCurrency.DefaultCorrelation");
85 if (it == pegp.end()) {
86 LOG("No Default Correlation present");
87 params.defaultCorrelation = Null<Real>();
88 } else {
89 LOG("Default Correlation is \"" << it->second << "\"");
90 params.defaultCorrelation = parseReal(it->second);
91 QL_REQUIRE(-1.0 <= params.defaultCorrelation && params.defaultCorrelation <= 1.0,
92 "Invalid DefaultCorrelation value " << it->second);
93 }
94 } catch (std::exception& e) {
95 QL_FAIL("Failed to build PseudoCurrencyMarketParameters : " << e.what());
96 }
97 } else {
98 DLOG("Building default PseudoCurrencyMarketParameters");
99 }
100 DLOG(params);
101
102 return params;
103}

150QuantLib::Handle<QuantExt::FxIndex> Market::fxIndex(const string& fxIndex, const string& configuration) const {
151 if (!handlePseudoCurrencies_ || GlobalPseudoCurrencyMarketParameters::instance().get().treatAsFX)
152 return fxIndexImpl(fxIndex, configuration);
153
154 std::string familyName;
155 std::string forCcy;
156 std::string domCcy;
157
158 if (isFxIndex(fxIndex)) {
159 auto ind = parseFxIndex(fxIndex);
160 familyName = ind->familyName();
161 forCcy = ind->sourceCurrency().code();
162 domCcy = ind->targetCurrency().code();
163 } else {
164 familyName = "GENERIC";
165 forCcy = fxIndex.substr(0, 3);
166 domCcy = fxIndex.substr(3);
167 }
168
169 if (hasPseudoCurrencyConfigPair(forCcy + domCcy)) {
170 DLOG("Market::fxIndex() requested for PM pair " << forCcy << domCcy);
171 string index = "FX-" + familyName + "-" + forCcy + "-" + domCcy;
172 Handle<QuantExt::FxIndex> fxInd;
173 auto it = fxIndicesCache_.find(make_pair(configuration, index));
174 // if no index found we build it
175 if (it == fxIndicesCache_.end()) {
176 // Parse the index we have with no term structures
177 boost::shared_ptr<QuantExt::FxIndex> fxIndexBase = parseFxIndex(index);
178
179 string source = fxIndexBase->sourceCurrency().code();
180 string target = fxIndexBase->targetCurrency().code();
181
182 // use todays rate here
183 Handle<Quote> spot = fxRate(source + target, configuration);
184
185 Handle<YieldTermStructure> sorTS = discountCurve(source, configuration);
186 Handle<YieldTermStructure> tarTS = discountCurve(target, configuration);
187
188 // spot is always zero here as we use the fxRates, which give rate today
189 Natural spotDays = 0;
190 Calendar calendar = NullCalendar();
191
192 if (source != target) {
193 const boost::shared_ptr<Conventions>& conventions = InstrumentConventions::instance().conventions();
194 // first check if we have a convention specific to the pair (e.g. XAUUSD), otherwise use
195 try {
196 auto comCon =
197 boost::dynamic_pointer_cast<data::CommodityForwardConvention>(conventions->get(fxIndex));
198 calendar = comCon->advanceCalendar();
199 } catch (...) {
200 WLOG("Market::fxIndex Cannot find commodity conventions for " << fxIndex);
201 }
202 }
203
204 fxInd = Handle<QuantExt::FxIndex>(boost::make_shared<QuantExt::FxIndex>(
205 fxIndexBase->familyName(), spotDays, fxIndexBase->sourceCurrency(), fxIndexBase->targetCurrency(),
206 calendar, spot, sorTS, tarTS));
207 // add it to the cache
208 fxIndicesCache_[make_pair(configuration, index)] = fxInd;
209 } else {
210 fxInd = it->second;
211 }
212 return fxInd;
213 }
214 return fxIndexImpl(fxIndex, configuration);
215}

219Handle<Quote> Market::fxRate(const string& pair, const string& config) const {
220 if (!handlePseudoCurrencies_ || GlobalPseudoCurrencyMarketParameters::instance().get().treatAsFX)
221 return fxRateImpl(pair, config);
222 if (hasPseudoCurrencyConfigPair(pair)) {
223 DLOG("Market::fxSpot() requested for PM pair " << pair);
224 if (fxRateCache_.find(pair) == fxRateCache_.end()) {
225 // Get the FX Spot rate. Rather than deal with all the combinations we just get the FX rate for each vs
226 // the baseCcy and create a ratio quote. This might mean we have combined a USD/USD spot quote below,
227 // but it all works fine and this is cleaner code.
228 //
229 // Note that we could just call market->fxSpot(forCode, domCode) and this would give us the correct
230 // quote from the markets FXTriangulation, however this would create a depedendacy on the pair
231 // and cause the configuration builder to go off building XAU-IN-USD and the like.
232 auto forBaseSpot = getFxBaseQuote(pair.substr(0, 3), config);
233 auto domBaseSpot = getFxBaseQuote(pair.substr(3), config);
234 auto fx = Handle<Quote>(boost::make_shared<CompositeQuote<std::function<Real(Real, Real)>>>(
235 forBaseSpot, domBaseSpot, [](Real a, Real b) { return b > 0.0 ? a / b : 0.0; }));
236 DLOG("Market returning " << fx->value() << " for " << pair << ".");
237 fxRateCache_[pair] = fx;
238 }
239 return fxRateCache_[pair];
240 }
241 return fxRateImpl(pair, config);
242}

290Handle<BlackVolTermStructure> Market::fxVol(const string& pair, const string& config) const {
291 if (!handlePseudoCurrencies_ || GlobalPseudoCurrencyMarketParameters::instance().get().treatAsFX)
292 return fxVolImpl(pair, config);
293
294 if (hasPseudoCurrencyConfigPair(pair)) {
295 DLOG("Market::fxVol() requested for PM pair " << pair);
296 if (volCache_.find(pair) == volCache_.end()) {
297
298 Handle<BlackVolTermStructure> vol;
299
300 auto forCode = pair.substr(0, 3);
301 auto domCode = pair.substr(3);
302
303 // we handle the easy and common case first.
304 if (forCode == GlobalPseudoCurrencyMarketParameters::instance().get().baseCurrency ||
305 domCode == GlobalPseudoCurrencyMarketParameters::instance().get().baseCurrency) {
306 // this is a straight mapping
307 string pm =
308 forCode == GlobalPseudoCurrencyMarketParameters::instance().get().baseCurrency ? domCode : forCode;
309 auto comVol = commodityVolatility(commodityCurveLookup(pm), config);
310 if (domCode == GlobalPseudoCurrencyMarketParameters::instance().get().baseCurrency)
311 vol = comVol;
312 else
313 vol = Handle<BlackVolTermStructure>(
314 boost::make_shared<QuantExt::BlackInvertedVolTermStructure>(comVol));
315 } else {
316
317 // otherwise we must triangulate - we get both surfaces vs baseCcy
318 auto forBaseVol = getVolatility(forCode, config);
319 auto domBaseVol = getVolatility(domCode, config);
320
321 // get the correlation
322 string forIndex = getCorrelationIndexName(forCode);
323 string domIndex = getCorrelationIndexName(domCode);
324 Handle<QuantExt::CorrelationTermStructure> rho;
325 try {
326 rho = correlationCurve(forIndex, domIndex, config);
327 } catch (std::exception& e) {
328 // No correlation, if we have a default we use it
329 WLOG("No correlation found for " << forIndex << "/" << domIndex);
330 if (GlobalPseudoCurrencyMarketParameters::instance().get().defaultCorrelation != Null<Real>()) {
331 WLOG("Using default correlation value "
332 << GlobalPseudoCurrencyMarketParameters::instance().get().defaultCorrelation);
333 rho = Handle<QuantExt::CorrelationTermStructure>(boost::make_shared<QuantExt::FlatCorrelation>(
334 asofDate(), GlobalPseudoCurrencyMarketParameters::instance().get().defaultCorrelation,
335 ActualActual(ActualActual::ISDA)));
336 } else {
337 QL_FAIL("No Correlation which is needed for PseudoCurrency Volatility :" << e.what());
338 }
339 }
340
341 // build and return triangulation
342 vol = Handle<BlackVolTermStructure>(
343 boost::make_shared<QuantExt::BlackTriangulationATMVolTermStructure>(forBaseVol, domBaseVol, rho));
344 }
345
346 DLOG("Market returning vol surface for " << pair << ".");
347 volCache_[pair] = vol;
348 }
349 return volCache_[pair];
350 }
351 return fxVolImpl(pair, config);
352}

354Handle<YieldTermStructure> Market::discountCurve(const string& ccy, const string& config) const {
355 if (!handlePseudoCurrencies_ || GlobalPseudoCurrencyMarketParameters::instance().get().treatAsFX)
356 return discountCurveImpl(ccy, config);
357
358 string baseCcy = GlobalPseudoCurrencyMarketParameters::instance().get().baseCurrency;
359
360 if (hasPseudoCurrencyConfig(ccy)) {
361 DLOG("Market::discount() requested for PM " << ccy);
362 if (discountCurveCache_.find(ccy) == discountCurveCache_.end()) {
363 auto baseDiscount = discountCurveImpl(baseCcy, config);
364 auto priceCurve = commodityPriceCurve(commodityCurveLookup(ccy), config);
365 QL_REQUIRE(!priceCurve.empty(),
366 "Failed to get Commodity Price curve for " << ccy << " using " << commodityCurveLookup(ccy));
367 discountCurveCache_[ccy] =
368 Handle<YieldTermStructure>(boost::make_shared<QuantExt::PriceTermStructureAdapter>(
369 priceCurve.currentLink(), baseDiscount.currentLink(), fxRate(ccy + baseCcy, config)));
370 discountCurveCache_[ccy]->enableExtrapolation();
371 }
372 return discountCurveCache_[ccy];
373 }
374
375 return discountCurveImpl(ccy, config);
376}
