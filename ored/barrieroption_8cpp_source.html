43 void BarrierOption::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
44  QL_REQUIRE(tradeActions().empty(), "TradeActions not supported for FxBarrierOption");
45 
46  checkBarriers();
47 
48  QL_REQUIRE(barrier_.levels().size() > 0 && barrier_.levels().size() <= 2, "BarrierOption must have 1 or 2 levels");
49  QL_REQUIRE(option_.style() == "European", "Option Style unknown: " << option_.style());
50  QL_REQUIRE(option_.exerciseDates().size() == 1, "Invalid number of excercise dates");
51 
52  // get the expiry date
53  Date expiryDate = parseDate(option_.exerciseDates().front());
54 
55  Real rebate = barrier_.rebate() / tradeMultiplier();
56  QL_REQUIRE(rebate >= 0, "rebate must be non-negative");
57 
58  // fx base
59  // Payoff
60  Option::Type type = parseOptionType(option_.callPut());
61  boost::shared_ptr<StrikedTypePayoff> payoff(new PlainVanillaPayoff(type, strike()));
62 
63  boost::shared_ptr<Exercise> exercise = boost::make_shared<EuropeanExercise>(expiryDate);
64  // QL does not have an FXBarrierOption, so we add a barrier option and a vanilla option here and wrap
65  // it in a composite to get the notional in.
66 
67  std::vector<boost::shared_ptr<Instrument>> additionalInstruments;
68  std::vector<Real> additionalMultipliers;
69  Settlement::Type settleType = parseSettlementType(option_.settlement());
70  Position::Type positionType = parsePositionType(option_.longShort());
71 
72  boost::shared_ptr<Instrument> vanilla = boost::make_shared<VanillaOption>(payoff, exercise);
73  boost::shared_ptr<Instrument> barrier;
74  boost::shared_ptr<InstrumentWrapper> instWrapper;
75 
76  boost::variant<Barrier::Type, DoubleBarrier::Type> barrierType;
77  if (barrier_.levels().size() < 2) {
78  barrierType = parseBarrierType(barrier_.type());
79  barrier = boost::make_shared<QuantLib::BarrierOption>(boost::get<Barrier::Type>(barrierType), barrier_.levels()[0].value(),
80  rebate, payoff, exercise);
81  } else {
82  barrierType = parseDoubleBarrierType(barrier_.type());
83  barrier = boost::make_shared<QuantLib::DoubleBarrierOption>(boost::get<DoubleBarrier::Type>(barrierType),
84  barrier_.levels()[0].value(), barrier_.levels()[1].value(), rebate, payoff, exercise);
85  }
86 
87  boost::shared_ptr<QuantLib::PricingEngine> barrierEngine = barrierPricigingEngine(engineFactory, expiryDate);
88  boost::shared_ptr<QuantLib::PricingEngine> vanillaEngine = vanillaPricigingEngine(engineFactory, expiryDate);
89 
90  // set pricing engines
91  barrier->setPricingEngine(barrierEngine);
92  vanilla->setPricingEngine(vanillaEngine);
93 
94  boost::shared_ptr<QuantLib::Index> index = getIndex();
95  const QuantLib::Handle<QuantLib::Quote>& spot = spotQuote();
96  if (barrier_.levels().size() < 2)
97  instWrapper = boost::make_shared<SingleBarrierOptionWrapper>(
98  barrier, positionType == Position::Long ? true : false, expiryDate,
99  settleType == Settlement::Physical ? true : false, vanilla, boost::get<Barrier::Type>(barrierType),
100  spot, barrier_.levels()[0].value(), rebate, tradeCurrency(), startDate_, index, calendar_,
101  tradeMultiplier(), tradeMultiplier(), additionalInstruments, additionalMultipliers);
102  else
103  instWrapper = boost::make_shared<DoubleBarrierOptionWrapper>(
104  barrier, positionType == Position::Long ? true : false, expiryDate,
105  settleType == Settlement::Physical ? true : false, vanilla, boost::get<DoubleBarrier::Type>(barrierType),
106  spot, barrier_.levels()[0].value(), barrier_.levels()[1].value(), rebate, tradeCurrency(), startDate_, index, calendar_,
107  tradeMultiplier(), tradeMultiplier(), additionalInstruments, additionalMultipliers);
108 
109  instrument_ = instWrapper;
110 
111  Calendar fixingCal = index ? index->fixingCalendar() : calendar_;
112  if (startDate_ != Null<Date>() && !indexFixingName().empty()) {
113  for (Date d = fixingCal.adjust(startDate_); d <= expiryDate; d = fixingCal.advance(d, 1 * Days)) {
114  requiredFixings_.addFixingDate(d, indexFixingName(), expiryDate);
115  }
116  }
117 
118  // Add additional premium payments
119  Real bsInd = (positionType == QuantLib::Position::Long ? 1.0 : -1.0);
120  Date lastPremiumDate =
121  addPremiums(additionalInstruments, additionalMultipliers, tradeMultiplier(), option_.premiumData(), -bsInd,
122  tradeCurrency(), engineFactory, engineFactory->configuration(MarketContext::pricing));
123 
124  // set the maturity
125  maturity_ = std::max(lastPremiumDate, expiryDate);
126 }

129 void BarrierOption::fromXML(XMLNode* node) {
130  Trade::fromXML(node);
131  XMLNode* dataNode = XMLUtils::getChildNode(node, tradeType_ + "Data");
132  QL_REQUIRE(dataNode, "No " + tradeType_ + " Node");
133  option_.fromXML(XMLUtils::getChildNode(dataNode, "OptionData"));
134  barrier_.fromXML(XMLUtils::getChildNode(dataNode, "BarrierData"));
135  startDate_ = ore::data::parseDate(XMLUtils::getChildValue(dataNode, "StartDate", false));
136  calendarStr_ = XMLUtils::getChildValue(dataNode, "Calendar", false);
137  calendar_ = parseCalendar(calendarStr_);
138 
139  additionalFromXml(dataNode);
140 }

142 XMLNode* BarrierOption::toXML(XMLDocument& doc) {
143  XMLNode* node = Trade::toXML(doc);
144  XMLNode* barNode = doc.allocNode(tradeType_ + "Data");
145  XMLUtils::appendNode(node, barNode);
146

148  XMLUtils::appendNode(barNode, barrier_.toXML(doc));
149  if (startDate_ != Date())
150  XMLUtils::addChild(doc, barNode, "StartDate", to_string(startDate_));
151  if (!calendarStr_.empty())
152  XMLUtils::addChild(doc, barNode, "Calendar", calendarStr_);
153  additionalToXml(doc, barNode);
154 
155  return node;
156 }

200 void EquityOptionWithBarrier::build(const boost::shared_ptr<ore::data::EngineFactory>& ef) {
201 
202  eqIndex_ = ef->market()->equityCurve(equityName()).currentLink();
203 
204  BarrierOption::build(ef);
205 
206  if (tradeStrike_.currency().empty())
207  tradeStrike_.setCurrency(currencyStr_);
208 
209  npvCurrency_ = currency_.code();
210 
211  // Notional - we really need todays spot to get the correct notional.
212  // But rather than having it move around we use strike * quantity
213  notional_ = tradeStrike_.value() * quantity_;
214  notionalCurrency_ = parseCurrencyWithMinors(tradeStrike_.currency()).code();
215 
216  additionalData_["quantity"] = quantity_;
217  additionalData_["strike"] = tradeStrike_.value();
218  additionalData_["strikeCurrency"] = tradeStrike_.currency();
219 }
