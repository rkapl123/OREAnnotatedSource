143 void FloatingLegData::fromXML(XMLNode* node) {
144  XMLUtils::checkNode(node, legNodeName());

149  &parseReal);

155  if (XMLNode* lrNode = XMLUtils::getChildNode(node, "LastRecentPeriod"))
156  lastRecentPeriod_ = parsePeriod(XMLUtils::getNodeValue(lrNode));
157  lastRecentPeriodCalendar_ = XMLUtils::getChildValue(node, "LastRecentPeriodCalendar", false);
158  if (XMLNode* avgNode = XMLUtils::getChildNode(node, "IsAveraged"))
159  isAveraged_ = parseBool(XMLUtils::getNodeValue(avgNode));
160  if (XMLNode* spNode = XMLUtils::getChildNode(node, "HasSubPeriods"))
161  hasSubPeriods_ = parseBool(XMLUtils::getNodeValue(spNode));
162  if (XMLNode* incSpNode = XMLUtils::getChildNode(node, "IncludeSpread"))
163  includeSpread_ = parseBool(XMLUtils::getNodeValue(incSpNode));
164  if (auto n = XMLUtils::getChildNode(node, "FixingDays"))
165  fixingDays_ = parseInteger(XMLUtils::getNodeValue(n));
166  else
167  fixingDays_ = Null<Size>();
168  if (auto n = XMLUtils::getChildNode(node, "Lookback"))
169  lookback_ = parsePeriod(XMLUtils::getNodeValue(n));
170  else
171  lookback_ = 0 * Days;
172  if (auto n = XMLUtils::getChildNode(node, "RateCutoff"))
173  rateCutoff_ = parseInteger(XMLUtils::getNodeValue(n));
174  else
175  rateCutoff_ = Null<Size>();
176  caps_ = XMLUtils::getChildrenValuesWithAttributes<Real>(node, "Caps", "Cap", "startDate", capDates_, &parseReal);
177  floors_ =
178  XMLUtils::getChildrenValuesWithAttributes<Real>(node, "Floors", "Floor", "startDate", floorDates_, &parseReal);
179  gearings_ = XMLUtils::getChildrenValuesWithAttributes<Real>(node, "Gearings", "Gearing", "startDate", gearingDates_,
180  &parseReal, false);
181  if (XMLUtils::getChildNode(node, "NakedOption"))
182  nakedOption_ = XMLUtils::getChildValueAsBool(node, "NakedOption", false);
183  else
184  nakedOption_ = false;
185 
186  if (XMLUtils::getChildNode(node, "LocalCapFloor"))
187  localCapFloor_ = XMLUtils::getChildValueAsBool(node, "LocalCapFloor", false);
188  else
189  localCapFloor_ = false;
190 }

857 Leg makeZCFixedLeg(const LegData& data, const QuantLib::Date& openEndDateReplacement) {
858  boost::shared_ptr<ZeroCouponFixedLegData> zcFixedLegData =
859  boost::dynamic_pointer_cast<ZeroCouponFixedLegData>(data.concreteLegData());
860  QL_REQUIRE(zcFixedLegData, "Wrong LegType, expected Zero Coupon Fixed, got " << data.legType());
861 
862  Schedule schedule = makeSchedule(data.schedule(), openEndDateReplacement);
863  DayCounter dc = parseDayCounter(data.dayCounter());
864  BusinessDayConvention bdc = parseBusinessDayConvention(data.paymentConvention());
865  // check we have a single notional and two dates in the schedule
866  Size numNotionals = data.notionals().size();
867  Size numRates = zcFixedLegData->rates().size();
868  Size numDates = schedule.size();
869  QL_REQUIRE(numDates >= 2, "Incorrect number of schedule dates entered, expected at least 2, got " << numDates);
870  QL_REQUIRE(numNotionals >= 1, "Incorrect number of notional values entered, expected at least1, got " << numNotionals);
871  QL_REQUIRE(numRates >= 1, "Incorrect number of rate values entered, expected at least 1, got " << numRates);
872 
873 
874  vector<Date> dates = schedule.dates();
875 
876  vector<double> rates = buildScheduledVector(zcFixedLegData->rates(), zcFixedLegData->rateDates(), schedule);
877  vector<double> notionals = buildScheduledVector(data.notionals(), data.notionalDates(), schedule);
878 
879  Compounding comp = parseCompounding(zcFixedLegData->compounding());
880  QL_REQUIRE(comp == QuantLib::Compounded || comp == QuantLib::Simple,
881  "Compounding method " << zcFixedLegData->compounding() << " not supported");
882 
883  // we loop over the dates in the schedule, computing the compound factor.
884  // For the Compounded rule:
885  // (1+r)^dcf_0 * (1+r)^dcf_1 * ... = (1+r)^(dcf_0 + dcf_1 + ...)
886  // So we compute the sum of all DayCountFractions in the loop.
887  // For the Simple rule:
888  // (1 + r * dcf_0) * (1 + r * dcf_1)...
889  double totalDCF = 0;
890  double compoundFactor = 1;
891  Leg leg;
892  for (Size i = 0; i < numDates - 1; i++) {
893  double dcf = dc.yearFraction(dates[i], dates[i + 1]);
894  double fixedAmount = i < notionals.size() ? notionals[i] : notionals.back();
895  double fixedRate = i < rates.size() ? rates[i] : rates.back();
896  if (comp == QuantLib::Simple)
897  compoundFactor *= (1 + fixedRate * dcf);
898  else
899  totalDCF += dcf;
900  if (comp == QuantLib::Compounded)
901  compoundFactor = pow(1.0 + fixedRate, totalDCF);
902  if (zcFixedLegData->subtractNotional())
903  fixedAmount *= (compoundFactor - 1);
904  else
905  fixedAmount *= compoundFactor;
906  Date fixedPayDate = schedule.calendar().adjust(dates[i + 1], bdc);
907  leg.push_back(boost::shared_ptr<CashFlow>(new SimpleCashFlow(fixedAmount, fixedPayDate)));
908  }
909  return leg;
910 }
