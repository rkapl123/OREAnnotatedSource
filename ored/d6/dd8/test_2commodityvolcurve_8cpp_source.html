249  // Check time depending volatilities are as expected

251  Real configuredVolatility;

258 
259  configuredVolatility = 0.10;
260  BOOST_CHECK_CLOSE(volatility->blackVol(asof + 2 * Years, 1000.0), configuredVolatility, testTolerance);
261  BOOST_CHECK_CLOSE(volatility->blackVol(asof + 2 * Years, 1200.0), configuredVolatility, testTolerance);

264  BOOST_CHECK_CLOSE(volatility->blackVol(asof + 5 * Years, 1000.0), configuredVolatility, testTolerance);
265  BOOST_CHECK_CLOSE(volatility->blackVol(asof + 5 * Years, 1200.0), configuredVolatility, testTolerance);
266 
267  // Check briefly the default linear interpolation and extrapolation
268  Time t_s = volatility->dayCounter().yearFraction(asof, asof + 2 * Years);
269  Real v_s = 0.10;
270  Time t_e = volatility->dayCounter().yearFraction(asof, asof + 5 * Years);

273  Time t = volatility->dayCounter().yearFraction(asof, asof + 3 * Years);
274  Real v = sqrt((v_s * v_s * t_s + (v_e * v_e * t_e - v_s * v_s * t_s) * (t - t_s) / (t_e - t_s)) / t);

278  BOOST_CHECK_CLOSE(volatility->blackVol(asof + 6 * Years, 1000.0), v_e, testTolerance);
279  BOOST_CHECK_CLOSE(volatility->blackVol(asof + 6 * Years, 1200.0), v_e, testTolerance);
