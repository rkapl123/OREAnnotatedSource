105 class CrossCurrencySwapEngineBuilder : public CrossCurrencySwapEngineBuilderBase {
106 public:
107  CrossCurrencySwapEngineBuilder()
108  : CrossCurrencySwapEngineBuilderBase("DiscountedCashflows", "DiscountingCrossCurrencySwapEngine") {}
109 
110 protected:
111  virtual boost::shared_ptr<PricingEngine> engineImpl(const std::vector<Currency>& ccys,
112  const Currency& base) override {
113 
114  std::vector<Handle<YieldTermStructure>> discountCurves;
115  std::vector<Handle<Quote>> fxQuotes;
116  std::string config = configuration(MarketContext::pricing);
117  std::string baseCcyCode = base.code();
118  for (Size i = 0; i < ccys.size(); ++i) {
119  string ccyCode = ccys[i].code();
120  discountCurves.push_back(xccyYieldCurve(market_, ccyCode, config));
121  string pair = ccyCode + baseCcyCode;
122  fxQuotes.push_back(market_->fxSpot(pair, config));
123  }
124 
125  return boost::make_shared<QuantExt::DiscountingCurrencySwapEngine>(discountCurves, fxQuotes, ccys, base);
126  }
127 };
