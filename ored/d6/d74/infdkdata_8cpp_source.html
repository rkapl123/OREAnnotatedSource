26 void InfDkData::fromXML(XMLNode* node) {
27  index_ = XMLUtils::getAttribute(node, "index");
28  LOG("Cross-Asset Inflation Index = " << index_);
29 
30  currency_ = XMLUtils::getChildValue(node, "Currency");
31  LOG("Cross-Asset Inflation Index = " << currency_);
32 
33  // Calibration CapFloors
34 
35  XMLNode* optionsNode = XMLUtils::getChildNode(node, "CalibrationCapFloors");
36 
37  capfloor_ = XMLUtils::getChildValue(optionsNode, "CapFloor");
38  optionExpiries() = XMLUtils::getChildrenValuesAsStrings(optionsNode, "Expiries", false);
39  optionStrikes() = XMLUtils::getChildrenValuesAsStrings(optionsNode, "Strikes", false);
40  if (optionStrikes().size() > 0) {
41  QL_REQUIRE(optionStrikes().size() == optionExpiries().size(),
42  "vector size mismatch in cap floor expiries/strikes for inflation index " << index_);
43  } else // Default: ATM
44  optionStrikes().resize(optionExpiries().size(), "ATM");
45 
46  for (Size i = 0; i < optionExpiries().size(); i++) {
47  LOG("LGM calibration capfloors " << optionExpiries()[i] << " " << optionStrikes()[i]);
48  }
49  LgmData::fromXML(node);
50 }

52 XMLNode* InfDkData::toXML(XMLDocument& doc) {
53  XMLNode* node = LgmData::toXML(doc);
54  XMLUtils::addAttribute(doc, node, "index", index_);
55  XMLUtils::addChild(doc, node, "Currency", currency_);
56 
57  // capfloor calibration
58  XMLNode* calibrationCapFloorNode = XMLUtils::addChild(doc, node, "CalibrationCapFloors");
59  XMLUtils::addChild(doc, calibrationCapFloorNode, "CapFloor", capfloor_);
60  XMLUtils::addGenericChildAsList(doc, calibrationCapFloorNode, "Expiries", optionExpiries());
61  XMLUtils::addGenericChildAsList(doc, calibrationCapFloorNode, "Strikes", optionStrikes());
62 
63  return node;
64 }
