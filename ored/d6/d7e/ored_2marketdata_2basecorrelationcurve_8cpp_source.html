48 BaseCorrelationCurve::BaseCorrelationCurve(
49  Date asof,
50  BaseCorrelationCurveSpec spec,
51  const Loader& loader,
52  const CurveConfigurations& curveConfigs,
53  const boost::shared_ptr<ReferenceDataManager>& referenceData)
54  : spec_(spec), referenceData_(referenceData) {
55 
56  DLOG("BaseCorrelationCurve: start building base correlation structure with ID " << spec_.curveConfigID());
57 
58  try {
59 
60  const auto& config = *curveConfigs.baseCorrelationCurveConfig(spec_.curveConfigID());
61 
62  // The base correlation surface is of the form term x detachment point. We need at least two detachment points
63  // and at least one term. The list of terms may be explicit or contain a single wildcard character '*'.
64  // Similarly, the list of detachment points may be explicit or contain a single wildcard character '*'.
65 
66  // Terms
67  const vector<string>& termStrs = config.terms();
68  QL_REQUIRE(!termStrs.empty(), "BaseCorrelationCurve: need at least one term.");
69  set<Period> terms;
70  bool termWc = find(termStrs.begin(), termStrs.end(), "*") != termStrs.end();
71  if (termWc) {
72  QL_REQUIRE(termStrs.size() == 1, "BaseCorrelationCurve: only one wild card term can be specified.");
73  DLOG("Have term wildcard pattern " << termStrs[0]);
74  } else {
75  for (const string& termStr : termStrs) {
76  terms.insert(parsePeriod(termStr));
77  }
78  DLOG("Parsed " << terms.size() << " unique configured term(s).");
79  }
80 
81  // Detachment points
82 
83  // Not advised to use set/map with floating point keys but should be ok here.
84  auto cmp = [](Real dp_1, Real dp_2) { return !close_enough(dp_1, dp_2) && dp_1 < dp_2; };
85  set<Real, decltype(cmp)> dps(cmp);
86 
87  const vector<string>& dpStrs = config.detachmentPoints();
88  bool dpsWc = find(dpStrs.begin(), dpStrs.end(), "*") != dpStrs.end();
89  if (dpsWc) {
90  QL_REQUIRE(dpStrs.size() == 1, "BaseCorrelationCurve: only one wild card " <<
91  "detachment point can be specified.");
92  DLOG("Have detachment point wildcard pattern " << dpStrs[0]);
93  } else {
94  for (const string& dpStr : dpStrs) {
95  dps.insert(parseReal(dpStr));
96  }
97  DLOG("Parsed " << dps.size() << " unique configured detachment points.");
98  QL_REQUIRE(dps.size() > 1, "BaseCorrelationCurve: need at least 2 unique detachment points.");
99  }
100 
101  // Read in quotes relevant for the base correlation surface. The points that will be used are stored in data
102  // where the key is the term, detachment point pair and value is the base correlation quote.
103  auto mpCmp = [](pair<Period, Real> k_1, pair<Period, Real> k_2) {
104  if (k_1.first != k_2.first)
105  return k_1.first < k_2.first;
106  else
107  return !close_enough(k_1.second, k_2.second) && k_1.second < k_2.second;
108  };
109  map<pair<Period, Real>, Handle<Quote>, decltype(mpCmp)> data(mpCmp);
110 
111  for (const boost::shared_ptr<MarketDatum>& md : loader.loadQuotes(asof)) {
112 
113  // Go to next quote if the market data point's date does not equal to asof.
114  if (md->asofDate() != asof)
115  continue;
116 
117  // Go to next quote if not a base correlation quote.
118  auto q = boost::dynamic_pointer_cast<BaseCorrelationQuote>(md);
119  if (!q)
120  continue;
121 
122  // Go to next quote if index name in the quote does not match the cds vol configuration name.
123  if (config.quoteName() != q->cdsIndexName())
124  continue;
125 
126  TLOG("Processing quote " << q->name() << ": (" << q->term() << "," << fixed << setprecision(9)
127  << q->detachmentPoint() << "," << q->quote()->value() << ")");
128 
129  // If we have been given a list of explicit terms, check that the quote matches one of them.
130  // Move to the next quote if it does not.
131  if (!termWc) {
132  auto it = terms.find(q->term());
133  if (it == terms.end())
134  continue;
135  }
136 
137  // If we have been given a list of explicit detachment points, check that the quote matches one of them.
138  // Move to the next quote if it does not.
139  if (!dpsWc) {
140  auto it = dps.find(q->detachmentPoint());
141  if (it == dps.end())
142  continue;
143  }
144 
145  // Skip if we have already added a quote for the given term and detachment point.
146  auto key = make_pair(q->term(), q->detachmentPoint());
147  if (data.find(key) != data.end()) {
148  DLOG("Already added base correlation with term " << q->term() << " and detachment point " <<
149  fixed << setprecision(9) << q->detachmentPoint() << " so skipping quote " << q->name());
150  continue;
151  }
152 
153  // If we have wildcards, update the set so that dps and terms are populated.
154  if (termWc)
155  terms.insert(q->term());
156  if (dpsWc)
157  dps.insert(q->detachmentPoint());
158 
159  // Add to the data that we will use.
160  data[key] = q->quote();
161 
162  TLOG("Added quote " << q->name() << ": (" << q->term() << "," << fixed << setprecision(9)
163  << q->detachmentPoint() << "," << q->quote()->value() << ")");
164  }
165 
166  DLOG("After processing the quotes, we have " << terms.size() << " unique term(s), " <<
167  dps.size() << " unique detachment points and " << data.size() << " quotes.");
168  QL_REQUIRE(dps.size() > 1, "BaseCorrelationCurve: need at least 2 unique detachment points.");
169  QL_REQUIRE(dps.size() * terms.size() == data.size(), "BaseCorrelationCurve: number of quotes (" <<
170  data.size() << ") should equal number of detachment points (" << dps.size() <<
171  ") times the number of terms (" << terms.size() << ").");
172 
173  // Need to now fill _completely_ the (number of dps) x (number of terms) quotes surface.
174  vector<vector<Handle<Quote>>> quotes;
175  for (const auto& dp : dps) {
176  quotes.push_back({});
177  for (const auto& term : terms) {
178  auto key = make_pair(term, dp);
179  auto it = data.find(key);
180  QL_REQUIRE(it != data.end(), "BaseCorrelationCurve: do not have a quote for term " << term <<
181  " and detachment point " << fixed << setprecision(9) << dp << ".");
182  quotes.back().push_back(it->second);
183  }
184  }
185 
186  // Need a vector of terms and detachment points for ctor below
187  vector<Period> tmpTerms(terms.begin(), terms.end());
188  vector<Real> tmpDps(dps.begin(), dps.end());
189 
190  if (config.adjustForLosses()) {
191  DLOG("Adjust for losses is true for base correlation " << spec_.curveConfigID());
192  DLOG("Detachment points before: " << Array(tmpDps.begin(), tmpDps.end()));
193  tmpDps = adjustForLosses(tmpDps);
194  DLOG("Detachment points after: " << Array(tmpDps.begin(), tmpDps.end()));
195  }
196 
197  // The QuantLib interpolator expects at least two terms, so add a second column, copying the first
198  if (tmpTerms.size() == 1) {
199  tmpTerms.push_back(tmpTerms[0] + 1 * tmpTerms[0].units());
200  for (Size i = 0; i < tmpDps.size(); ++i)
201  quotes[i].push_back(quotes[i][0]);
202  }
203 
204  baseCorrelation_ = boost::make_shared<BilinearBaseCorrelationTermStructure>(
205  config.settlementDays(), config.calendar(), config.businessDayConvention(), tmpTerms,
206  tmpDps, quotes, config.dayCounter(), config.startDate(), config.rule());
207 
208  baseCorrelation_->enableExtrapolation(config.extrapolate());
209 
210  } catch (std::exception& e) {
211  QL_FAIL("BaseCorrelationCurve: curve building failed :" << e.what());
212  } catch (...) {
213  QL_FAIL("BaseCorrelationCurve: curve building failed: unknown error");
214  }
215 
216  DLOG("BaseCorrelationCurve: finished building base correlation structure with ID " << spec_.curveConfigID());
217 }

219 vector<Real> BaseCorrelationCurve::adjustForLosses(const vector<Real>& detachPoints) const {
220 
221  const auto& cId = spec_.curveConfigID();
222  DLOG("BaseCorrelationCurve::adjustForLosses: start adjusting for losses for base correlation " << cId);
223 
224  if (!referenceData_) {
225  DLOG("Reference data manager is null so cannot adjust for losses.");
226  return detachPoints;
227  }
228 
229  if (!referenceData_->hasData(CreditIndexReferenceDatum::TYPE, cId)) {
230  DLOG("Reference data manager does not have index credit data for " << cId << " so cannot adjust for losses.");
231  return detachPoints;
232  }
233 
234  auto crd = boost::dynamic_pointer_cast<CreditIndexReferenceDatum>(
235  referenceData_->getData(CreditIndexReferenceDatum::TYPE, cId));
236  if (!crd) {
237  DLOG("Index credit data for " << cId << " is not of correct type so cannot adjust for losses.");
238  return detachPoints;
239  }
240 
241  // Process the credit index reference data
242  Real totalRemainingWeight = 0.0;
243  Real totalPriorWeight = 0.0;
244  Real lost = 0.0;
245  Real recovered = 0.0;
246 
247  for (const auto& c : crd->constituents()) {
248 
249  const auto& name = c.name();
250  auto weight = c.weight();
251  validateWeightRec(weight, name, "weight");
252 
253  if (!close(0.0, weight)) {
254  totalRemainingWeight += weight;
255  } else {
256  auto priorWeight = c.priorWeight();
257  QL_REQUIRE(priorWeight != Null<Real>(), "Expecting a valid prior weight for name " << name << ".");
258  validateWeightRec(priorWeight, name, "prior weight");
259  auto recovery = c.recovery();
260  QL_REQUIRE(recovery != Null<Real>(), "Expecting a valid recovery for name " << name << ".");
261  validateWeightRec(recovery, name, "recovery");
262  lost += (1.0 - recovery) * priorWeight;
263  recovered += recovery * priorWeight;
264  totalPriorWeight += priorWeight;
265  }
266  }
267 
268  Real totalWeight = totalRemainingWeight + totalPriorWeight;
269  TLOG("Total remaining weight = " << totalRemainingWeight);
270  TLOG("Total prior weight = " << totalPriorWeight);
271  TLOG("Total weight = " << totalWeight);
272 
273  if (!close(totalRemainingWeight, 1.0) && totalRemainingWeight > 1.0) {
274  ALOG("Total remaining weight is greater than 1, possible error in CreditIndexReferenceDatum for " << cId);
275  }
276 
277  if (!close(totalWeight, 1.0)) {
278  ALOG("Expected the total weight (" << totalWeight << " = " << totalRemainingWeight << " + " <<
279  totalPriorWeight << ") to equal 1, possible error in CreditIndexReferenceDatum for " << cId);
280  }
281 
282  if (close(totalRemainingWeight, 0.0)) {
283  ALOG("The total remaining weight is 0 so cannot adjust for losses.");
284  return detachPoints;
285  }
286 
287  if (close(totalRemainingWeight, 1.0)) {
288  DLOG("Index factor for " << cId << " is 1 so adjustment for losses not required.");
289  return detachPoints;
290  }
291 
292  // Index factor is less than 1 so need to adjust each of the detachment points.
293  vector<Real> result;
294  for (Size i = 0; i < detachPoints.size(); ++i) {
295 
296  // Amounts below, for tranche and above for original (quoted) attachment and detachment points.
297  Real below = i == 0 ? 0.0 : detachPoints[i-1];
298  Real tranche = detachPoints[i] - below;
299  Real above = 1.0 - detachPoints[i];
300  Real newBelow = max(below - lost, 0.0);
301  Real newTranche = tranche - max(min(recovered - above, tranche), 0.0) - max(min(lost - below, tranche), 0.0);
302  Real newDetach = (newBelow + newTranche) / totalRemainingWeight;
303 
304  TLOG("Quoted detachment point " << detachPoints[i] << " adjusted to " << newDetach << ".");
305 
306  if (i > 0 && (newDetach < result.back() || close(newDetach, result.back()))) {
307  ALOG("The " << io::ordinal(i + 1) << " adjusted detachment point is not greater than the previous " <<
308  "adjusted detachment point so cannot adjust for losses.");
309  return detachPoints;
310  }
311 
312  result.push_back(newDetach);
313  }
314 
315  DLOG("BaseCorrelationCurve::adjustForLosses: finished adjusting for losses for base correlation " << cId);
316 
317  return result;
318 }
