94 BOOST_AUTO_TEST_CASE(testEquityTradePrices) {
95 
96  BOOST_TEST_MESSAGE("Testing EquityOption Price...");
97 
98  Date today = Settings::instance().evaluationDate();
99 
100  // build market
101  boost::shared_ptr<Market> market = boost::make_shared<TestMarket>();
102  Settings::instance().evaluationDate() = market->asofDate();
103  Date expiry = market->asofDate() + 6 * Months + 1 * Days;
104  ostringstream o;
105  o << QuantLib::io::iso_date(expiry);
106  string exp_str = o.str();
107 
108  // build EquityOption - expiry in 1 Year
109  OptionData callData("Long", "Call", "European", true, vector<string>(1, exp_str));
110  OptionData callDataPremium("Long", "Call", "European", true, vector<string>(1, exp_str), "Cash", "",
111  PremiumData{1.0, "EUR", expiry});
112  OptionData putData("Short", "Put", "European", true, vector<string>(1, exp_str));
113  OptionData putDataPremium("Short", "Put", "European", true, vector<string>(1, exp_str), "Cash", "",
114  PremiumData{1.0, "EUR", expiry});
115  Envelope env("CP1");
116  EquityOption eqCall(env, callData, EquityUnderlying("zzzCorp"), "EUR", 95.0, 1.0);
117  EquityOption eqCallPremium(env, callDataPremium, EquityUnderlying("zzzCorp"), "EUR", 95.0, 1.0);
118  EquityOption eqPut(env, putData, EquityUnderlying("zzzCorp"), "EUR", 95.0, 1.0);
119  EquityOption eqPutPremium(env, putDataPremium, EquityUnderlying("zzzCorp"), "EUR", 95.0, 1.0);
120  ore::data::EquityForward eqFwd(env, "Long", EquityUnderlying("zzzCorp"), "EUR", 1.0, exp_str, 95.0);
121 
122  Real expectedNPV_Put = -2.4648; // negative for sold option
123  Real expectedNPV_Put_Premium = -1.513558; // less negative due to received premium of 1 EUR at expiry
124 
125  // Build and price
126  boost::shared_ptr<EngineData> engineData = boost::make_shared<EngineData>();
127  engineData->model("EquityOption") = "BlackScholesMerton";
128  engineData->engine("EquityOption") = "AnalyticEuropeanEngine";
129  engineData->model("EquityForward") = "DiscountedCashflows";
130  engineData->engine("EquityForward") = "DiscountingEquityForwardEngine";
131  boost::shared_ptr<EngineFactory> engineFactory = boost::make_shared<EngineFactory>(engineData, market);
132  engineFactory->registerBuilder(boost::make_shared<EquityEuropeanOptionEngineBuilder>());
133  engineFactory->registerBuilder(boost::make_shared<EquityForwardEngineBuilder>());
134 
135  eqCall.build(engineFactory);
136  eqCallPremium.build(engineFactory);
137  eqPut.build(engineFactory);
138  eqPutPremium.build(engineFactory);
139  eqFwd.build(engineFactory);
140 
141  Real npv_call = eqCall.instrument()->NPV();
142  Real npv_call_premium = eqCallPremium.instrument()->NPV();
143  Real npv_put = eqPut.instrument()->NPV();
144  Real npv_put_premium = eqPutPremium.instrument()->NPV();
145  Real npv_fwd = eqFwd.instrument()->NPV();
146 
147  Real put_call_sum = npv_call + npv_put;
148  Real put_call_premium_sum = npv_call_premium + npv_put_premium;
149 
150  BOOST_CHECK_CLOSE(expectedNPV_Put, npv_put, 0.001);
151  BOOST_CHECK_CLOSE(expectedNPV_Put_Premium, npv_put_premium, 0.001);
152  BOOST_CHECK_CLOSE(npv_fwd, put_call_sum, 0.001); // put-call parity check
153  BOOST_CHECK_CLOSE(npv_fwd, put_call_premium_sum, 0.001); // put-call parity check
154 
155  Settings::instance().evaluationDate() = today; // reset
156 }
