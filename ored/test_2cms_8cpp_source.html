245BOOST_AUTO_TEST_CASE(testCMSAnalyticHagan) {
246 BOOST_TEST_MESSAGE("Testing CMS Analytic Hagan price...");
247
248 // build market
249 boost::shared_ptr<Market> market = boost::make_shared<TestMarket>();
250 Settings::instance().evaluationDate() = market->asofDate();
251 CommonVars vars;
252 boost::shared_ptr<ore::data::Swap> cmsSwap = vars.makeSwap();
253
254 // Build and price
255 boost::shared_ptr<EngineData> engineData = boost::make_shared<EngineData>();
256 engineData->model("CMS") = "Hagan";
257 engineData->engine("CMS") = "Analytic";
258 map<string, string> engineparams;
259 engineparams["YieldCurveModel"] = "Standard";
260 engineparams["MeanReversion"] = "0.0";
261 engineData->engineParameters("CMS") = engineparams;
262
263 engineData->model("Swap") = "DiscountedCashflows";
264 engineData->engine("Swap") = "DiscountingSwapEngineOptimised";
265
266 boost::shared_ptr<EngineFactory> engineFactory = boost::make_shared<EngineFactory>(engineData, market);
267 engineFactory->registerBuilder(boost::make_shared<AnalyticHaganCmsCouponPricerBuilder>());
268
269 cmsSwap->build(engineFactory);
270
271 Real expectedNPV = 3440673.46;
272 Real npv = cmsSwap->instrument()->NPV();
273
274 BOOST_TEST_MESSAGE("Hagan Analytic price is " << npv);
275 outputCoupons(cmsSwap);
276
277 BOOST_CHECK_CLOSE(npv, expectedNPV, 1.0);
278}

358 BOOST_TEST_MESSAGE("Testing CMS CapFloor price...");
359
360 // build market
361 boost::shared_ptr<Market> market = boost::make_shared<TestMarket>();
362 Settings::instance().evaluationDate() = market->asofDate();
363
364 CommonVars vars;
365 boost::shared_ptr<EngineData> engineData = boost::make_shared<EngineData>();
366 engineData->model("CMS") = "Hagan";
367 engineData->engine("CMS") = "Analytic";
368 map<string, string> engineparams;
369 engineparams["YieldCurveModel"] = "Standard";
370 engineparams["MeanReversion"] = "0.0";
371 engineData->engineParameters("CMS") = engineparams;
372
373 engineData->model("Swap") = "DiscountedCashflows";
374 engineData->engine("Swap") = "DiscountingSwapEngineOptimised";
375
376 boost::shared_ptr<EngineFactory> engineFactory = boost::make_shared<EngineFactory>(engineData, market);
377 engineFactory->registerBuilder(boost::make_shared<AnalyticHaganCmsCouponPricerBuilder>());
378
379 BOOST_TEST_MESSAGE(
380 "Comparing CMS Cap price to replication by a Single Legged CMS Swap and a Single Leg Capped CMS Swap...");
381 vector<double> capRate(1, 0.021);
382 boost::shared_ptr<ore::data::Swap> cmsLegSwap = vars.makeCmsLegSwap();
383 boost::shared_ptr<ore::data::Swap> cappedCmsLegSwap = vars.makeCappedCmsLegSwap(capRate);
384 boost::shared_ptr<ore::data::CapFloor> cap = vars.makeCap(capRate);
385
386 cmsLegSwap->build(engineFactory);
387 cappedCmsLegSwap->build(engineFactory);
388 cap->build(engineFactory);
389
390 Real cmsLegNpv = cmsLegSwap->instrument()->NPV();
391 Real cappedCmsLegNpv = cappedCmsLegSwap->instrument()->NPV();
392 Real capNpv = cap->instrument()->NPV();
393
394 Real capBySwaps = cmsLegNpv - cappedCmsLegNpv;
395
396 BOOST_TEST_MESSAGE("CMS Leg swap NPV is " << cmsLegNpv);
397 BOOST_TEST_MESSAGE("CMS Capped Leg swap NPV is " << cappedCmsLegNpv);
398 BOOST_TEST_MESSAGE("CMS Cap NPV is " << capNpv);
399 BOOST_TEST_MESSAGE("CMS Cap NPV from Swap replication is " << capBySwaps);
400 BOOST_CHECK_CLOSE(capNpv, capBySwaps, 1.0);
401
402 BOOST_TEST_MESSAGE("Checking CMS Cap with high Cap is zero...");
403 vector<double> capHigh(1, 1.0);
404
405 cap = vars.makeCap(capHigh);
406 cap->build(engineFactory);
407 capNpv = cap->instrument()->NPV();
408 BOOST_TEST_MESSAGE("CMS Cap (Cap of 100%) NPV is " << capNpv);
409 BOOST_CHECK_SMALL(capNpv, 0.01);
410
411 BOOST_TEST_MESSAGE("Checking CMS Cap with low Cap is equal to single leg swap...");
412 vector<double> capLow(1, -1.0);
413
414 cap = vars.makeCap(capLow);
415 cap->build(engineFactory);
416 capNpv = cap->instrument()->NPV();
417 BOOST_TEST_MESSAGE("CMS Cap (Cap of -100%) NPV is " << capNpv);
418 BOOST_CHECK_CLOSE(capNpv, cmsLegNpv, 1.0);
419
420 BOOST_TEST_MESSAGE("Checking CMS Floor with low Cap is equal to zero...");
421 vector<double> floorLow(1, -1.0);
422
423 boost::shared_ptr<ore::data::CapFloor> floor = vars.makeFloor(floorLow);
424 floor->build(engineFactory);
425 Real floorNpv = floor->instrument()->NPV();
426 BOOST_TEST_MESSAGE("CMS Floor (Floor of -100%) NPV is " << floorNpv);
427 BOOST_CHECK_SMALL(floorNpv, 0.01);
428
429 BOOST_TEST_MESSAGE("Checking CMS Cap + CMS Floor = Swap...");
430 vector<double> floorRate(1, 0.021);
431
432 cap = vars.makeCap(capRate);
433 floor = vars.makeFloor(floorRate);
434 boost::shared_ptr<ore::data::Swap> swap = vars.makeSwap(0.021, "6M");
435 cap->build(engineFactory);
436 floor->build(engineFactory);
437 swap->build(engineFactory);
438 capNpv = cap->instrument()->NPV();
439 floorNpv = floor->instrument()->NPV();
440 Real swapNpv = swap->instrument()->NPV();
441 Real capFloorNpv = capNpv - floorNpv;
442 BOOST_TEST_MESSAGE("CMS Cap NPV is " << capNpv);
443 BOOST_TEST_MESSAGE("CMS Floor NPV is " << floorNpv);
444 BOOST_TEST_MESSAGE("CMS Cap + Floor NPV is " << capFloorNpv);
445 BOOST_TEST_MESSAGE("CMS Swap NPV is " << swapNpv);
446 BOOST_CHECK_CLOSE(capFloorNpv, swapNpv, 1.0);
447}
