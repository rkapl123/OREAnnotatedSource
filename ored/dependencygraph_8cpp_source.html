34 std::vector<std::string> getCorrelationTokens(const std::string& name) {
35  // Look for & first as it avoids collisions with : which can be used in an index name
36  // if it is not there we fall back on the old behaviour
37  string delim;
38  if (name.find('&') != std::string::npos)
39  delim = "&";
40  else
41  delim = "/:";
42  vector<string> tokens;
43  boost::split(tokens, name, boost::is_any_of(delim));
44  QL_REQUIRE(tokens.size() == 2,
45  "invalid correlation name '" << name << "', expected Index2:Index1 or Index2/Index1 or Index2&Index1");
46  return tokens;
47 }

54  Graph& g = dependencies_[configuration];
55  IndexMap index = boost::get(boost::vertex_index, g);
56 
57  // add the vertices
58  std::set<MarketObject> t = getMarketObjectTypes(); // from todaysmarket parameter
59 
60  for (auto const& o : t) {
61  auto mapping = params_->mapping(o, configuration);
62  for (auto const& m : mapping) {
63  Vertex v = boost::add_vertex(g);
64  boost::shared_ptr<CurveSpec> spec;
65  // swap index curves do not have a spec
66  if (o != MarketObject::SwapIndexCurve)
67  spec = parseCurveSpec(m.second);
68  // add the vertex to the dependency graph
69  g[v] = {o, m.first, m.second, spec, false};
70  TLOG("add vertex # " << index[v] << ": " << g[v]);
71  }
72  }
73 
74  // add the dependencies based on the required curve ids stored in the curve configs; notice that no dependencies
75  // to FXSpots are stored in the configs, these are not needed because a complete FXTriangulation object is created
76  // upfront that is passed to all curve builders which require it.
77 
78  VertexIterator v, vend, w, wend;
79 
80  for (std::tie(v, vend) = boost::vertices(g); v != vend; ++v) {
81  if (g[*v].curveSpec) {
82  for (auto const& r :
83  curveConfigs_->requiredCurveIds(g[*v].curveSpec->baseType(), g[*v].curveSpec->curveConfigID())) {
84  for (auto const& cId : r.second) {
85  // avoid self reference
86  if (r.first == g[*v].curveSpec->baseType() && cId == g[*v].curveSpec->curveConfigID())
87  continue;
88  bool found = false;
89  for (std::tie(w, wend) = boost::vertices(g); w != wend; ++w) {
90  if (*w != *v && g[*w].curveSpec && r.first == g[*w].curveSpec->baseType() &&
91  cId == g[*w].curveSpec->curveConfigID()) {
92  g.add_edge(*v, *w);
93  TLOG("add edge from vertex #" << index[*v] << " " << g[*v] << " to #" << index[*w] << " "
94  << g[*w]);
95  found = true;
96  // it is enough to insert one dependency
97  break;
98  }
99  }
100  if (!found)
101  buildErrors[g[*v].mapping] = "did not find required curve id " + cId + " of type " +
102  ore::data::to_string(r.first) + " (required from " +
103  ore::data::to_string(g[*v]) +
104  ") in dependency graph for configuration " + configuration;
105  }
106  }
107  }
