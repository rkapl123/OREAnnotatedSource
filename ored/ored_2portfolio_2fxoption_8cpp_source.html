36void FxOption::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
37
38 const boost::shared_ptr<Market>& market = engineFactory->market();
39
40 // If automatic exercise, check that we have a non-empty FX index string, parse it and attach curves from market.
41 if (option_.isAutomaticExercise()) {
42
43 QL_REQUIRE(!fxIndex_.empty(),
44 "FX option trade " << id() << " has automatic exercise so the FXIndex node needs to be populated.");
45
46 // The strike is the number of units of sold currency (currency_) per unit of bought currency (assetName_).
47 // So, the convention here is that the sold currency is domestic and the bought currency is foreign.
48 // Note: intentionally use null calendar and 0 day fixing lag here because we will ask the FX index for its
49 // value on the expiry date without adjustment.
50 index_ = buildFxIndex(fxIndex_, currency_, assetName_, market,
51 engineFactory->configuration(MarketContext::pricing));
52
53 // Populate the external index name so that fixings work.
54 indexName_ = fxIndex_;
55 }
56
57 // Build the trade using the shared functionality in the base class.
58 VanillaOptionTrade::build(engineFactory);
59
60 additionalData_["boughtCurrency"] = assetName_;
61 additionalData_["boughtAmount"] = quantity_;
62 additionalData_["soldCurrency"] = currency_;
63 additionalData_["soldAmount"] = quantity_ * strike_.value();
64}

66void FxOption::fromXML(XMLNode* node) {
67 VanillaOptionTrade::fromXML(node);
68 XMLNode* fxNode = XMLUtils::getChildNode(node, "FxOptionData");
69 QL_REQUIRE(fxNode, "No FxOptionData Node");
70 option_.fromXML(XMLUtils::getChildNode(fxNode, "OptionData"));
71 assetName_ = XMLUtils::getChildValue(fxNode, "BoughtCurrency", true);
72 currency_ = XMLUtils::getChildValue(fxNode, "SoldCurrency", true);
73 double boughtAmount = XMLUtils::getChildValueAsDouble(fxNode, "BoughtAmount", true);
74 double soldAmount = XMLUtils::getChildValueAsDouble(fxNode, "SoldAmount", true);
75 strike_ = TradeStrike(soldAmount / boughtAmount, currency_);
76 quantity_ = boughtAmount;
77 fxIndex_ = XMLUtils::getChildValue(fxNode, "FXIndex", false);
78 QL_REQUIRE(boughtAmount > 0.0, "positive BoughtAmount required");
79 QL_REQUIRE(soldAmount > 0.0, "positive SoldAmount required");
80}

82XMLNode* FxOption::toXML(XMLDocument& doc) {
83 // TODO: Should call parent class to xml?
84 XMLNode* node = Trade::toXML(doc);
85 XMLNode* fxNode = doc.allocNode("FxOptionData");
86 XMLUtils::appendNode(node, fxNode);
87
88 XMLUtils::appendNode(fxNode, option_.toXML(doc));
89 XMLUtils::addChild(doc, fxNode, "BoughtCurrency", boughtCurrency());
90 XMLUtils::addChild(doc, fxNode, "BoughtAmount", boughtAmount());
91 XMLUtils::addChild(doc, fxNode, "SoldCurrency", soldCurrency());
92 XMLUtils::addChild(doc, fxNode, "SoldAmount", soldAmount());
93
94 if (!fxIndex_.empty())
95 XMLUtils::addChild(doc, fxNode, "FXIndex", fxIndex_);
96
97 return node;
98}
