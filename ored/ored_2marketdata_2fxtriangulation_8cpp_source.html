61 void FXTriangulation::addQuote(const std::string& pair, const Handle<Quote>& spot) {
62  auto it = std::find_if(map_.begin(), map_.end(),
63  [&pair](const std::pair<string, Handle<Quote>>& x) { return x.first == pair; });
64  if (it != map_.end())
65  *it = std::make_pair(pair, spot);
66  else
67  map_.push_back(std::make_pair(pair, spot));
68 }

70 Handle<Quote> FXTriangulation::getQuote(const string& pair) const {
71  // First, look for the pair in the map
72  auto it = std::find_if(map_.begin(), map_.end(),
73  [&pair](const std::pair<string, Handle<Quote>>& x) { return x.first == pair; });
74  if (it != map_.end())
75  return it->second;
76 
77  // Now we have to break the pair up and search for it.
78  QL_REQUIRE(pair.size() == 6, "FXTriangulation: Invalid ccypair " << pair);
79  string domestic = pair.substr(0, 3);
80  string foreign = pair.substr(3);
81 
82  // check reverse
83  string reverse = foreign + domestic;
84  it = std::find_if(map_.begin(), map_.end(),
85  [&reverse](const std::pair<string, Handle<Quote>>& x) { return x.first == reverse; });
86  if (it != map_.end()) {
87  Handle<Quote> invertedQuote(boost::make_shared<DerivedQuote<Inverse>>(it->second, Inverse()));
88  map_.push_back(std::make_pair(pair, invertedQuote));
89  return invertedQuote;
90  }
91 
92  // check EUREUR
93  if (foreign == domestic) {
94  static Handle<Quote> unity(boost::make_shared<SimpleQuote>(1.0));
95  return unity;
96  }
97 
98  // Now we search for a pair of quotes that we can combine to construct the quote required.
99  // We only search for a pair of quotes a single step apart.
100  //
101  // Suppose we want a USDJPY quote and we have EUR based data, there are 4 combinations to
102  // consider:
103  // EURUSD, EURJPY => we want EURJPY / EURUSD [Triangulation]
104  // EURUSD, JPYEUR => we want 1 / (EURUSD * JPYEUR) [InverseProduct]
105  // USDEUR, EURJPY => we want USDEUR * EURJPY [Product]
106  // USDEUR, JPYEUR => we want USDEUR / JPYEUR [Triangulation (but in the reverse order)]
107  //
108  // Loop over the map, look for domestic then use the map to find the other side of the pair.
109  for (const auto& kv : map_) {
110  string keyDomestic = kv.first.substr(0, 3);
111  string keyForeign = kv.first.substr(3);
112  const Handle<Quote>& q1 = kv.second;
113 
114  if (domestic == keyDomestic) {
115  // we have domestic, now look for foreign/keyForeign
116  // USDEUR, JPYEUR => we want USDEUR / JPYEUR [Triangulation (but in the reverse order)]
117  it = std::find_if(map_.begin(), map_.end(),
118  [&foreign, &keyForeign](const std::pair<string, Handle<Quote>>& x) {
119  return x.first == foreign + keyForeign;
120  });
121  if (it != map_.end()) {
122  // Here q1 is USDEUR and it->second is JPYEUR
123  auto tmp =
124  Handle<Quote>(boost::make_shared<CompositeQuote<Triangulation>>(q1, it->second, Triangulation()));
125  map_.push_back(std::make_pair(pair, tmp));
126  return tmp;
127  }
128  // USDEUR, EURJPY => we want USDEUR * EURJPY [Product]
129  it = std::find_if(map_.begin(), map_.end(),
130  [&foreign, &keyForeign](const std::pair<string, Handle<Quote>>& x) {
131  return x.first == keyForeign + foreign;
132  });
133  if (it != map_.end()) {
134  auto tmp = Handle<Quote>(boost::make_shared<CompositeQuote<Product>>(q1, it->second, Product()));
135  map_.push_back(std::make_pair(pair, tmp));
136  return tmp;
137  }
138  }
139 
140  if (domestic == keyForeign) {
141  // EURUSD, JPYEUR => we want 1 / (EURUSD * JPYEUR) [InverseProduct]
142  it = std::find_if(map_.begin(), map_.end(),
143  [&foreign, &keyDomestic](const std::pair<string, Handle<Quote>>& x) {
144  return x.first == foreign + keyDomestic;
145  });
146  if (it != map_.end()) {
147  auto tmp =
148  Handle<Quote>(boost::make_shared<CompositeQuote<InverseProduct>>(q1, it->second, InverseProduct()));
149  map_.push_back(std::make_pair(pair, tmp));
150  return tmp;
151  }
152  // EURUSD, EURJPY => we want EURJPY / EURUSD [Triangulation]
153  it = std::find_if(map_.begin(), map_.end(),
154  [&foreign, &keyDomestic](const std::pair<string, Handle<Quote>>& x) {
155  return x.first == keyDomestic + foreign;
156  });
157  if (it != map_.end()) {
158  // Here q1 is EURUSD and it->second is EURJPY
159  auto tmp =
160  Handle<Quote>(boost::make_shared<CompositeQuote<Triangulation>>(it->second, q1, Triangulation()));
161  map_.push_back(std::make_pair(pair, tmp));
162  return tmp;
163  }
164  }
165  }
166 
167  QL_FAIL("FXTriangulation: Unable to build FXQuote for ccy pair " << pair);
168 }
