60 FXTriangulation::FXTriangulation(std::map<std::string, Handle<Quote>> quotes) : quotes_(std::move(quotes)) {
61 
62  LOG("FXTriangulation: initializing");
63 
64  // collect all currencies from the pairs
65 
66  std::set<std::string> ccys;
67  for (auto const& q : quotes_) {
68  auto [ccy1, ccy2] = splitPair(q.first);
69  ccys.insert(ccy1);
70  ccys.insert(ccy2);
71  TLOG("FXTriangulation: adding quote " << q.first);
72  }
73 
74  /* - populate node to ccy vector
75  - we insert currencies in the order we want to use them for triangulation if there
76  are several shortest paths from CCY1 to CCY2 */
77 
78  static vector<string> ccyOrder = {"USD", "EUR", "GBP", "CHF", "JPY", "AUD", "CAD", "ZAR"};
79 
80  std::set<std::string> remainingCcys(ccys);
81 
82  for (auto const& c : ccyOrder) {
83  if (ccys.find(c) != ccys.end()) {
84  nodeToCcy_.push_back(c);
85  remainingCcys.erase(c);
86  }
87  }
88 
89  for (auto const& c : remainingCcys) {
90  nodeToCcy_.push_back(c);
91  }
92 
93  // populate ccy to node vector
94 
95  for (Size i = 0; i < nodeToCcy_.size(); ++i)
96  ccyToNode_[nodeToCcy_[i]] = i;
97 
98  // populate neighbours container
99 
100  neighbours_.resize(nodeToCcy_.size());
101  for (auto const& q : quotes_) {
102  auto [ccy1, ccy2] = splitPair(q.first);
103  Size n1 = ccyToNode_[ccy1];
104  Size n2 = ccyToNode_[ccy2];
105  neighbours_[n1].insert(n2);
106  neighbours_[n2].insert(n1);
107  }
108 
109  LOG("FXTriangulation: initialized with " << quotes_.size() << " quotes, " << ccys.size() << " currencies.");
110 }

112 Handle<Quote> FXTriangulation::getQuote(const std::string& pair) const {
113 
114  // do we have a cached result?
115 
116  if (auto it = quoteCache_.find(pair); it != quoteCache_.end())
117  return it->second;
118 
119  // we need to construct the quote from the input quotes
120 
121  Handle<Quote> result;
122  auto [ccy1, ccy2] = splitPair(pair);
123 
124  // handle trivial case
125 
126  if (ccy1 == ccy2)
127  return Handle<Quote>(boost::make_shared<SimpleQuote>(1.0));
128 
129  // get the path from ccy1 to ccy2
130 
131  auto path = getPath(ccy1, ccy2);
132 
133  if (path.size() == 2) {
134 
135  // we can use a direct or inverted quote, but do not need a composite
136 
137  result = getQuote(path[0], path[1]);
138 
139  } else {
140 
141  // we need a composite quote
142 
143  std::vector<Handle<Quote>> quotes;
144 
145  // collect the quotes on the path and check consistency of the settlement dates
146 
147  for (Size i = 0; i < path.size() - 1; ++i) {
148  quotes.push_back(getQuote(path[i], path[i + 1]));
149  }
150 
151  // build the composite quote
152 
153  auto f = [](const std::vector<Real>& quotes) {
154  return std::accumulate(quotes.begin(), quotes.end(), 1.0, std::multiplies());
155  };
156  result = Handle<Quote>(boost::make_shared<CompositeVectorQuote<decltype(f)>>(quotes, f));
157  }
158 
159  // add the result to the lookup cache and return it
160 
161  quoteCache_[pair] = result;
162  return result;
163 }

165 Handle<FxIndex> FXTriangulation::getIndex(const std::string& indexOrPair, const Market* market) const {
166 
167  // do we have a cached result?
168 
169  if (auto it = indexCache_.find(indexOrPair); it != indexCache_.end()) {
170  return it->second;
171  }
172 
173  // otherwise we need to construct the index
174 
175  Handle<FxIndex> result;
176 
177  std::string familyName;
178  std::string forCcy;
179  std::string domCcy;
180 
181  if (isFxIndex(indexOrPair)) {
182  auto ind = parseFxIndex(indexOrPair);
183  familyName = ind->familyName();
184  forCcy = ind->sourceCurrency().code();
185  domCcy = ind->targetCurrency().code();
186  } else {
187  familyName = "GENERIC";
188  std::tie(forCcy, domCcy) = splitPair(indexOrPair);
189  }
190 
191  // get the conventions of the result index
192 
193  auto [fixingDays, fixingCalendar] = getFxIndexConventions(indexOrPair);
194 
195  // get the discount curves for the result index
196 
197  auto sourceYts = getMarketDiscountCurve(market, forCcy);
198  auto targetYts = getMarketDiscountCurve(market, domCcy);
199 
200  // get the path from ccy1 to ccy2
201 
202  auto path = getPath(forCcy, domCcy);
203 
204  if (path.size() == 2) {
205 
206  // we can use a direct or inverted quote, but do not need a composite
207 
208  auto fxSpot = getQuote(path[0], path[1]);
209  result = Handle<FxIndex>(boost::make_shared<FxIndex>(familyName, fixingDays, parseCurrency(forCcy),
210  parseCurrency(domCcy), fixingCalendar, fxSpot, sourceYts,
211  targetYts));
212 
213  } else {
214 
215  // we need a composite quote
216 
217  std::vector<Handle<Quote>> quotes;
218 
219  // collect the quotes on the path and store them as FxRate quotes ("as of today" - quotes)
220 
221  for (Size i = 0; i < path.size() - 1; ++i) {
222 
223  auto q = getQuote(path[i], path[i + 1]);
224 
225  // we store a quote "as of today" to account for possible spot lag differences
226 
227  auto [fd, fc] = getFxIndexConventions(path[i] + path[i + 1]);
228  auto s_yts = getMarketDiscountCurve(market, path[i]);
229  auto t_yts = getMarketDiscountCurve(market, path[i + 1]);
230  quotes.push_back(Handle<Quote>(boost::make_shared<FxRateQuote>(q, s_yts, t_yts, fd, fc)));
231  }
232 
233  // build the composite quote "as of today"
234 
235  auto f = [](const std::vector<Real>& quotes) {
236  return std::accumulate(quotes.begin(), quotes.end(), 1.0, std::multiplies());
237  };
238  Handle<Quote> compQuote(boost::make_shared<CompositeVectorQuote<decltype(f)>>(quotes, f));
239 
240  // build the spot quote
241 
242  Handle<Quote> spotQuote(
243  boost::make_shared<FxSpotQuote>(compQuote, sourceYts, targetYts, fixingDays, fixingCalendar));
244 
245  // build the idnex
246 
247  result = Handle<FxIndex>(boost::make_shared<FxIndex>(familyName, fixingDays, parseCurrency(forCcy),
248  parseCurrency(domCcy), fixingCalendar, spotQuote,
249  sourceYts, targetYts));
250  }
251 
252  // add the result to the lookup cache and return it
253 
254  indexCache_[indexOrPair] = result;
255  return result;
256 }

258 std::vector<std::string> FXTriangulation::getPath(const std::string& forCcy, const std::string& domCcy) const {
259 
260  // see https://en.wikipedia.org/wiki/Dijkstra%27s_algorithm
261 
262  Size sourceNode, targetNode;
263 
264  if (auto it = ccyToNode_.find(forCcy); it != ccyToNode_.end()) {
265  sourceNode = it->second;
266  } else {
267  QL_FAIL("FXTriangulation: no conversion from '"
268  << forCcy << "' to '" << domCcy << "' possible, since '" << forCcy
269  << "' is not available as one of the currencies in any of the quotes (" << getAllQuotes() << ")");
270  }
271 
272  if (auto it = ccyToNode_.find(domCcy); it != ccyToNode_.end()) {
273  targetNode = it->second;
274  } else {
275  QL_FAIL("FXTriangulation: no conversion from '"
276  << forCcy << "' to '" << domCcy << "' possible, since '" << domCcy
277  << "' is not available as one of the currencies in any of the quotes (" << getAllQuotes() << ")");
278  }
279 
280  // previous node on the current shortest path
281  std::vector<Size> prev(nodeToCcy_.size(), Null<Size>());
282  // distance per node
283  std::vector<Size> dist(nodeToCcy_.size(), std::numeric_limits<Size>::max());
284  // visited flag
285  std::vector<bool> visited(nodeToCcy_.size(), false);
286 
287  // init source
288  dist[sourceNode] = 0;
289 
290  // main loop
291  Size noVisited = 0;
292  while (noVisited < nodeToCcy_.size()) {
293  Size u = Null<Size>(), min = std::numeric_limits<Size>::max();
294  for (Size i = 0; i < dist.size(); ++i) {
295  if (!visited[i] && dist[i] < min) {
296  u = i;
297  min = dist[i];
298  }
299  }
300  QL_REQUIRE(u != Null<Size>(), "FXTriangulation: internal error, no minimum found in dist array for '"
301  << forCcy << "' to '" << domCcy << "'. Quotes = " << getAllQuotes());
302  if (u == targetNode)
303  break;
304  visited[u] = true;
305  ++noVisited;
306  for (auto const& n : neighbours_[u]) {
307  if (visited[n])
308  continue;
309  Size alt = dist[u] + 1;
310  if (alt < dist[n]) {
311  dist[n] = alt;
312  prev[n] = u;
313  }
314  }
315  }
316 
317  // did we find a path?
318  if (dist[targetNode] < std::numeric_limits<Size>::max()) {
319  std::vector<std::string> result;
320  Size u = targetNode;
321  while (u != sourceNode) {
322  result.insert(result.begin(), nodeToCcy_[u]);
323  u = prev[u];
324  QL_REQUIRE(u != Null<Size>(), "FXTriangulation: internal error u == null for '"
325  << forCcy << "' to '" << domCcy
326  << "'. Contact dev. Quotes = " << getAllQuotes() << ".");
327  }
328  result.insert(result.begin(), nodeToCcy_[sourceNode]);
329  TLOG("FXTriangulation: found path of length "
330  << result.size() - 1 << " from '" << forCcy << "' to '" << domCcy << "': "
331  << std::accumulate(
332  result.begin(), result.end(), std::string(),
333  [](const std::string& s1, const std::string& s2) { return s1.empty() ? s2 : s1 + "-" + s2; }));
334  return result;
335  }
336 
337  QL_FAIL("FXTriangulation: no path from '" << forCcy << "' to '" << domCcy
338  << "' found. Quotes = " << getAllQuotes());
339 }

341 Handle<Quote> FXTriangulation::getQuote(const std::string& forCcy, const std::string& domCcy) const {
342  if (auto it = quotes_.find(forCcy + domCcy); it != quotes_.end()) {
343  return it->second;
344  }
345  if (auto it = quotes_.find(domCcy + forCcy); it != quotes_.end()) {
346  auto f = [](Real x) { return 1.0 / x; };
347  return Handle<Quote>(boost::make_shared<DerivedQuote<decltype(f)>>(it->second, f));
348  }
349  QL_FAIL(
350  "FXTriangulation::getQuote(" << forCcy << domCcy
351  << ") - no such quote available. This is an internal error. Contact dev. Quotes = "
352  << getAllQuotes());
353 }
