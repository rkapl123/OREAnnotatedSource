43std::pair<std::string, std::string> splitPair(const std::string& pair) {
44 QL_REQUIRE(pair.size() == 6, "FXTriangulation: Invalid currency pair '" << pair << "'");
45 return std::make_pair(pair.substr(0, 3), pair.substr(3));
46}

48Handle<YieldTermStructure> getMarketDiscountCurve(const Market* market, const std::string& ccy) {
49 try {
50 return market->discountCurve(ccy);
51 } catch (const std::exception&) {
52 WLOG("FXTriangulation: could not get market discount curve '"
53 << ccy

62FXTriangulation::FXTriangulation(std::map<std::string, Handle<Quote>> quotes) : quotes_(std::move(quotes)) {
63
64 LOG("FXTriangulation: initializing");
65
66 // collect all currencies from the pairs
67
68 std::set<std::string> ccys;
69 for (auto const& q : quotes_) {
70 auto [ccy1, ccy2] = splitPair(q.first);
71 ccys.insert(ccy1);
72 ccys.insert(ccy2);
73 TLOG("FXTriangulation: adding quote " << q.first);
74 }
75
76 /* - populate node to ccy vector
77 - we insert currencies in the order we want to use them for triangulation if there
78 are several shortest paths from CCY1 to CCY2 */
79
80 static vector<string> ccyOrder = {"USD", "EUR", "GBP", "CHF", "JPY", "AUD", "CAD", "ZAR"};
81
82 std::set<std::string> remainingCcys(ccys);
83
84 for (auto const& c : ccyOrder) {
85 if (ccys.find(c) != ccys.end()) {
86 nodeToCcy_.push_back(c);
87 remainingCcys.erase(c);
88 }
89 }
90
91 for (auto const& c : remainingCcys) {
92 nodeToCcy_.push_back(c);
93 }
94
95 // populate ccy to node vector
96
97 for (Size i = 0; i < nodeToCcy_.size(); ++i)
98 ccyToNode_[nodeToCcy_[i]] = i;
99
100 // populate neighbours container
101
102 neighbours_.resize(nodeToCcy_.size());
103 for (auto const& q : quotes_) {
104 auto [ccy1, ccy2] = splitPair(q.first);
105 Size n1 = ccyToNode_[ccy1];
106 Size n2 = ccyToNode_[ccy2];
107 neighbours_[n1].insert(n2);
108 neighbours_[n2].insert(n1);
109 }
110
111 LOG("FXTriangulation: initialized with " << quotes_.size() << " quotes, " << ccys.size() << " currencies.");
112}

114Handle<Quote> FXTriangulation::getQuote(const std::string& pair) const {
115
116 // do we have a cached result?
117
118 if (auto it = quoteCache_.find(pair); it != quoteCache_.end())
119 return it->second;
120
121 // we need to construct the quote from the input quotes
122
123 Handle<Quote> result;
124 auto [ccy1, ccy2] = splitPair(pair);
125
126 // handle trivial case
127
128 if (ccy1 == ccy2)
129 return Handle<Quote>(boost::make_shared<SimpleQuote>(1.0));
130
131 // get the path from ccy1 to ccy2
132
133 auto path = getPath(ccy1, ccy2);
134
135 if (path.size() == 2) {
136
137 // we can use a direct or inverted quote, but do not need a composite
138
139 result = getQuote(path[0], path[1]);
140
141 } else {
142
143 // we need a composite quote
144
145 std::vector<Handle<Quote>> quotes;
146
147 // collect the quotes on the path and check consistency of the settlement dates
148
149 for (Size i = 0; i < path.size() - 1; ++i) {
150 quotes.push_back(getQuote(path[i], path[i + 1]));
151 }
152
153 // build the composite quote
154
155 auto f = [](const std::vector<Real>& quotes) {
156 return std::accumulate(quotes.begin(), quotes.end(), 1.0, std::multiplies());
157 };
158 result = Handle<Quote>(boost::make_shared<CompositeVectorQuote<decltype(f)>>(quotes, f));
159 }
160
161 // add the result to the lookup cache and return it
162
163 quoteCache_[pair] = result;
164 return result;
165}

167Handle<FxIndex> FXTriangulation::getIndex(const std::string& indexOrPair, const Market* market) const {
168
169 // do we have a cached result?
170
171 if (auto it = indexCache_.find(indexOrPair); it != indexCache_.end()) {
172 return it->second;
173 }
174
175 // otherwise we need to construct the index
176
177 Handle<FxIndex> result;
178
179 std::string familyName;
180 std::string forCcy;
181 std::string domCcy;
182
183 if (isFxIndex(indexOrPair)) {
184 auto ind = parseFxIndex(indexOrPair);
185 familyName = ind->familyName();
186 forCcy = ind->sourceCurrency().code();
187 domCcy = ind->targetCurrency().code();
188 } else {
189 familyName = "GENERIC";
190 std::tie(forCcy, domCcy) = splitPair(indexOrPair);
191 }
192
193 // get the conventions of the result index
194
195 auto [fixingDays, fixingCalendar, bdc] = getFxIndexConventions(indexOrPair);
196
197 // get the discount curves for the result index
198
199 auto sourceYts = getMarketDiscountCurve(market, forCcy);
200 auto targetYts = getMarketDiscountCurve(market, domCcy);
201
202 // get the path from ccy1 to ccy2
203
204 auto path = getPath(forCcy, domCcy);
205
206 if (path.size() == 2) {
207
208 // we can use a direct or inverted quote, but do not need a composite
209
210 auto fxSpot = getQuote(path[0], path[1]);
211 result = Handle<FxIndex>(boost::make_shared<FxIndex>(familyName, fixingDays, parseCurrency(forCcy),
212 parseCurrency(domCcy), fixingCalendar, fxSpot, sourceYts,
213 targetYts));
214
215 } else {
216
217 // we need a composite quote
218
219 std::vector<Handle<Quote>> quotes;
220
221 // collect the quotes on the path and store them as FxRate quotes ("as of today" - quotes)
222
223 for (Size i = 0; i < path.size() - 1; ++i) {
224
225 auto q = getQuote(path[i], path[i + 1]);
226
227 // we store a quote "as of today" to account for possible spot lag differences
228
229 auto [fd, fc, bdc] = getFxIndexConventions(path[i] + path[i + 1]);
230 auto s_yts = getMarketDiscountCurve(market, path[i]);
231 auto t_yts = getMarketDiscountCurve(market, path[i + 1]);
232 quotes.push_back(Handle<Quote>(boost::make_shared<FxRateQuote>(q, s_yts, t_yts, fd, fc)));
233 }
234
235 // build the composite quote "as of today"
236
237 auto f = [](const std::vector<Real>& quotes) {
238 return std::accumulate(quotes.begin(), quotes.end(), 1.0, std::multiplies());
239 };
240 Handle<Quote> compQuote(boost::make_shared<CompositeVectorQuote<decltype(f)>>(quotes, f));
241
242 // build the spot quote
243
244 Handle<Quote> spotQuote(
245 boost::make_shared<FxSpotQuote>(compQuote, sourceYts, targetYts, fixingDays, fixingCalendar));
246
247 // build the idnex
248
249 result = Handle<FxIndex>(boost::make_shared<FxIndex>(familyName, fixingDays, parseCurrency(forCcy),
250 parseCurrency(domCcy), fixingCalendar, spotQuote,
251 sourceYts, targetYts));
252 }
253
254 // add the result to the lookup cache and return it
255
256 indexCache_[indexOrPair] = result;
257 return result;
258}

260std::vector<std::string> FXTriangulation::getPath(const std::string& forCcy, const std::string& domCcy) const {
261
262 // see https://en.wikipedia.org/wiki/Dijkstra%27s_algorithm
263
264 Size sourceNode, targetNode;
265
266 if (auto it = ccyToNode_.find(forCcy); it != ccyToNode_.end()) {
267 sourceNode = it->second;
268 } else {
269 QL_FAIL("FXTriangulation: no conversion from '"
270 << forCcy << "' to '" << domCcy << "' possible, since '" << forCcy
271 << "' is not available as one of the currencies in any of the quotes (" << getAllQuotes() << ")");
272 }
273
274 if (auto it = ccyToNode_.find(domCcy); it != ccyToNode_.end()) {
275 targetNode = it->second;
276 } else {
277 QL_FAIL("FXTriangulation: no conversion from '"
278 << forCcy << "' to '" << domCcy << "' possible, since '" << domCcy
279 << "' is not available as one of the currencies in any of the quotes (" << getAllQuotes() << ")");
280 }
281
282 // previous node on the current shortest path
283 std::vector<Size> prev(nodeToCcy_.size(), Null<Size>());
284 // distance per node
285 std::vector<Size> dist(nodeToCcy_.size(), std::numeric_limits<Size>::max());
286 // visited flag
287 std::vector<bool> visited(nodeToCcy_.size(), false);
288
289 // init source
290 dist[sourceNode] = 0;
291
292 // main loop
293 Size noVisited = 0;
294 while (noVisited < nodeToCcy_.size()) {
295 Size u = Null<Size>(), min = std::numeric_limits<Size>::max();
296 for (Size i = 0; i < dist.size(); ++i) {
297 if (!visited[i] && dist[i] < min) {
298 u = i;
299 min = dist[i];
300 }
301 }
302 QL_REQUIRE(u != Null<Size>(), "FXTriangulation: internal error, no minimum found in dist array for '"
303 << forCcy << "' to '" << domCcy << "'. Quotes = " << getAllQuotes());
304 if (u == targetNode)
305 break;
306 visited[u] = true;
307 ++noVisited;
308 for (auto const& n : neighbours_[u]) {
309 if (visited[n])
310 continue;
311 Size alt = dist[u] + 1;
312 if (alt < dist[n]) {
313 dist[n] = alt;
314 prev[n] = u;
315 }
316 }
317 }
318
319 // did we find a path?
320 if (dist[targetNode] < std::numeric_limits<Size>::max()) {
321 std::vector<std::string> result;
322 Size u = targetNode;
323 while (u != sourceNode) {
324 result.insert(result.begin(), nodeToCcy_[u]);
325 u = prev[u];
326 QL_REQUIRE(u != Null<Size>(), "FXTriangulation: internal error u == null for '"
327 << forCcy << "' to '" << domCcy
328 << "'. Contact dev. Quotes = " << getAllQuotes() << ".");
329 }
330 result.insert(result.begin(), nodeToCcy_[sourceNode]);
331 TLOG("FXTriangulation: found path of length "
332 << result.size() - 1 << " from '" << forCcy << "' to '" << domCcy << "': "
333 << std::accumulate(
334 result.begin(), result.end(), std::string(),
335 [](const std::string& s1, const std::string& s2) { return s1.empty() ? s2 : s1 + "-" + s2; }));
336 return result;
337 }
338
339 QL_FAIL("FXTriangulation: no path from '" << forCcy << "' to '" << domCcy
340 << "' found. Quotes = " << getAllQuotes());
341}

343Handle<Quote> FXTriangulation::getQuote(const std::string& forCcy, const std::string& domCcy) const {
344 if (auto it = quotes_.find(forCcy + domCcy); it != quotes_.end()) {
345 return it->second;
346 }
347 if (auto it = quotes_.find(domCcy + forCcy); it != quotes_.end()) {
348 auto f = [](Real x) { return 1.0 / x; };
349 return Handle<Quote>(boost::make_shared<DerivedQuote<decltype(f)>>(it->second, f));
350 }
351 QL_FAIL(
352 "FXTriangulation::getQuote(" << forCcy << domCcy
353 << ") - no such quote available. This is an internal error. Contact dev. Quotes = "
354 << getAllQuotes());
355}

357std::string FXTriangulation::getAllQuotes() const {
358 std::string result;
359 if (quotes_.size() > 0) {
360 for (auto const& d : quotes_) {
361 result += d.first + ",";
362 }
363 result.erase(std::next(result.end(), -1));
364 }
365 return result;
366}
