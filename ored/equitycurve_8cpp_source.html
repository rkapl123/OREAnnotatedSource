43 EquityCurve::EquityCurve(Date asof, EquityCurveSpec spec, const Loader& loader, const CurveConfigurations& curveConfigs,
44  const map<string, boost::shared_ptr<YieldCurve>>& requiredYieldCurves,
45  const bool buildCalibrationInfo) {
46 
47  try {
48 
49  const boost::shared_ptr<EquityCurveConfig>& config = curveConfigs.equityCurveConfig(spec.curveConfigID());
50 
51  dc_ = Actual365Fixed();
52  if (config->dayCountID() == "") {
53  DLOG("No Day Count convention specified for " << spec.curveConfigID() << ", using A365F as default");
54  } else {
55  dc_ = parseDayCounter(config->dayCountID());
56  }
57 
58  // set the calendar to the ccy based calendar, if provided by config we try to use that
59  Calendar calendar;
60  if (!config->calendar().empty()) {
61  try {
62  calendar = parseCalendar(config->calendar());
63  } catch (exception& ex) {
64  WLOG("Failed to get Calendar name for " << config->calendar() << ":" << ex.what());
65  }
66  }
67  if (calendar.empty()) {
68  calendar = parseCalendar(config->currency());
69  }
70 
71  // Set the Curve type - EquityFwd / OptionPrice / DividendYield
72  curveType_ = config->type();
73 
74  // declare spot and yields
75  Handle<Quote> equitySpot;
76  Handle<YieldTermStructure> forecastYieldTermStructure;
77  Handle<YieldTermStructure> dividendYieldTermStructure;
78 
79  // Set the Equity Forecast curve
80  YieldCurveSpec ycspec(config->currency(), config->forecastingCurve());
81 
82  // at this stage we should have built the curve already
83  // (consider building curve on fly if not? Would need to work around fact that requiredYieldCurves is currently
84  // const ref)
85  auto itr = requiredYieldCurves.find(ycspec.name());
86  QL_REQUIRE(itr != requiredYieldCurves.end(),
87  "Yield Curve Spec - " << ycspec.name() << " - not found during equity curve build");
88  boost::shared_ptr<YieldCurve> yieldCurve = itr->second;
89  forecastYieldTermStructure = yieldCurve->handle();
90 
91  // Set the interpolation variables
92  dividendInterpVariable_ = parseYieldCurveInterpolationVariable(config->dividendInterpolationVariable());
93  dividendInterpMethod_ = parseYieldCurveInterpolationMethod(config->dividendInterpolationMethod());
94 
95  // We loop over all market data, looking for quotes that match the configuration
96  // until we found the whole set of quotes or do not have more quotes in the
97  // market data
98 
99  vector<boost::shared_ptr<EquityForwardQuote>> qt; // for sorting quotes_/terms_ pairs
100  vector<boost::shared_ptr<EquityOptionQuote>> oqt; // store any equity vol quotes
101  Size quotesRead = 0;
102  Size quotesExpired = 0;
103  // in case of wild-card in config
104  auto wildcard = getUniqueWildcard(config->fwdQuotes());
105 
106  if ((config->type() == EquityCurveConfig::Type::ForwardPrice ||
107  config->type() == EquityCurveConfig::Type::ForwardDividendPrice ||
108  config->type() == EquityCurveConfig::Type::OptionPremium) &&
109  wildcard) {
110  DLOG("Wild card quote specified for " << config->curveID())
111  } else {
112  if (config->type() == EquityCurveConfig::Type::OptionPremium) {
113  oqt.reserve(config->fwdQuotes().size());
114  } else {
115  quotes_.reserve(config->fwdQuotes().size());
116  terms_.reserve(config->fwdQuotes().size());
117  }
118  }
119 
120  // load the quotes
121 
122  auto q = boost::dynamic_pointer_cast<EquitySpotQuote>(loader.get(config->equitySpotQuoteID(), asof));
123  QL_REQUIRE(q != nullptr, "expected '" << config->equitySpotQuoteID() << "' to be an EquitySpotQuote");
124  Real spot = convertMinorToMajorCurrency(q->ccy(), q->quote()->value());
125  // convert quote from minor to major currency if needed
126  equitySpot = Handle<Quote>(boost::make_shared<SimpleQuote>(spot));
127 
128  if ((config->type() == EquityCurveConfig::Type::ForwardPrice ||
129  config->type() == EquityCurveConfig::Type::ForwardDividendPrice)) {
130  if (wildcard) {
131  for (auto const& md : loader.get(*wildcard, asof)) {
132  auto q = boost::dynamic_pointer_cast<EquityForwardQuote>(md);
133  QL_REQUIRE(q != nullptr, "expected '" << md->name() << "' to be an EquityForwardQuote");
134  QL_REQUIRE(find(qt.begin(), qt.end(), q) == qt.end(),
135  "duplicate market datum found for " << q->name());
136  if (asof < q->expiryDate()) {
137  DLOG("EquityCurve Forward Price found for quote: " << q->name());
138  qt.push_back(q); // terms_ and quotes_
139  quotesRead++;
140  } else {
141  ++quotesExpired;
142  DLOG("Ignore expired ForwardPrice/ForwardDividendPrice quote "
143  << q->name() << ", expired at " << io::iso_date(q->expiryDate()));
144  }
145  }
146  } else {
147  for (auto const& md :
148  loader.get(std::set<std::string>(config->fwdQuotes().begin(), config->fwdQuotes().end()), asof)) {
149  auto q = boost::dynamic_pointer_cast<EquityForwardQuote>(md);
150  QL_REQUIRE(q != nullptr, "expected '" << md->name() << "' to be an EquityForwardQuote");
151  if (asof < q->expiryDate()) {
152  bool isUnique = find(terms_.begin(), terms_.end(), q->expiryDate()) == terms_.end();
153  QL_REQUIRE(isUnique, "duplicate market datum found for " << q->name());
154  terms_.push_back(q->expiryDate());
155  // convert quote from minor to major currency if needed
156  quotes_.push_back(convertMinorToMajorCurrency(q->ccy(), q->quote()->value()));
157  quotesRead++;
158  } else {
159  DLOG("Ignore expired ForwardPrice/ForwardDividendPrice quote "
160  << q->name() << ", expired at " << io::iso_date(q->expiryDate()));
161  ++quotesExpired;
162  }
163  }
164  }
165  }
166 
167  if (config->type() == EquityCurveConfig::Type::OptionPremium) {
168  if (wildcard) {
169  for (auto const& md : loader.get(*wildcard, asof)) {
170  auto q = boost::dynamic_pointer_cast<EquityOptionQuote>(md);
171  QL_REQUIRE(q != nullptr, "expected '" << md->name() << "' to be an EquityOptionQuote");
172  QL_REQUIRE(find(oqt.begin(), oqt.end(), q) == oqt.end(),
173  "duplicate market datum found for " << q->name());
174  if (asof < parseDate(q->expiry())) {
175  DLOG("EquityCurve Volatility Price found for quote: " << q->name());
176  oqt.push_back(q);
177  quotesRead++;
178  } else {
179  ++quotesExpired;
180  DLOG("Ignore expired OptionPremium quote " << q->name() << ", expired at " << q->expiry());
181  }
182  }
183  } else {
184  for (auto const& md :
185  loader.get(std::set<std::string>(config->fwdQuotes().begin(), config->fwdQuotes().end()), asof)) {
186  auto q = boost::dynamic_pointer_cast<EquityOptionQuote>(md);
187  QL_REQUIRE(q != nullptr, "expected '" << md->name() << "' to be an EquityOptionQuote");
188  if (asof < parseDate(q->expiry())) {
189  DLOG("EquityCurve Volatility Price found for quote: " << q->name());
190  oqt.push_back(q);
191  quotesRead++;
192  } else {
193  ++quotesExpired;
194  DLOG("Ignore expired OptionPremium quote " << q->name() << ", expired at " << q->expiry());
195  }
196  }
197  }
198  }
199 
200  if (config->type() == EquityCurveConfig::Type::DividendYield) {
201  for (auto const& md :
202  loader.get(std::set<std::string>(config->fwdQuotes().begin(), config->fwdQuotes().end()), asof)) {
203  boost::shared_ptr<EquityDividendYieldQuote> q =
204  boost::dynamic_pointer_cast<EquityDividendYieldQuote>(md);
205  QL_REQUIRE(q != nullptr, "expected '" << md->name() << "' to be an EquityDividendYieldQuote");
206  if (q->tenorDate() > asof) {
207  bool isUnique = find(terms_.begin(), terms_.end(), q->tenorDate()) == terms_.end();
208  QL_REQUIRE(isUnique, "duplicate market datum found for " << q->name());
209  DLOG("EquityCurve Dividend Yield found for quote: " << q->name());
210  terms_.push_back(q->tenorDate());
211  quotes_.push_back(q->quote()->value());
212  quotesRead++;
213  } else {
214  DLOG("Ignore expired DividendYield quote " << q->name() << ", expired at "
215  << io::iso_date(q->tenorDate()));
216  ++quotesExpired;
217  }
218  }
219  }
220 
221  // some checks on the quotes read
222  DLOG("EquityCurve: read " << quotesRead + quotesExpired << " quotes of type " << config->type());
223  DLOG("EquityCurve: ignored " << quotesExpired << " expired quotes.");
224 
225  if (!wildcard) {
226  QL_REQUIRE(quotesRead + quotesExpired == config->fwdQuotes().size(),
227  "read " << quotesRead << "quotes and " << quotesExpired << " expire quotes, but "
228  << config->fwdQuotes().size() << " required.");
229  }
230 
231  // Sort the quotes and terms by date
232 
233  QL_REQUIRE(terms_.size() == quotes_.size(), "Internal error: terms and quotes mismatch");
234  if (!terms_.empty()) {
235  vector<pair<Date, Real>> tmpSort;
236 
237  std::transform(terms_.begin(), terms_.end(), quotes_.begin(), back_inserter(tmpSort),
238  [](const Date& d, const Real& q) { return make_pair(d, q); });
239 
240  std::sort(tmpSort.begin(), tmpSort.end(), [](const pair<Date, Real>& left, const pair<Date, Real>& right) {
241  return left.first < right.first;
242  });
243 
244  for (Size i = 0; i < tmpSort.size(); ++i) {
245  terms_[i] = tmpSort[i].first;
246  quotes_[i] = tmpSort[i].second;
247  }
248 
249  for (Size i = 0; i < terms_.size(); i++) {
250  QL_REQUIRE(terms_[i] > asof, "Invalid Fwd Expiry " << terms_[i] << " vs. " << asof);
251  if (i > 0) {
252  QL_REQUIRE(terms_[i] > terms_[i - 1], "terms must be increasing in curve config");
253  }
254  }
255  }
256 
257  // the curve type that we will build
258  EquityCurveConfig::Type buildCurveType = curveType_;
259 
260  // for curveType ForwardPrice or OptionPremium populate the terms_ and quotes_ with forward prices
261  if (curveType_ == EquityCurveConfig::Type::ForwardPrice ||
262  curveType_ == EquityCurveConfig::Type::ForwardDividendPrice) {
263 
264  if (qt.size() > 0) {
265  DLOG("Building Equity Dividend Yield curve from Forward/Future prices");
266 
267  // sort quotes and terms in case of wild-card
268  if (wildcard) {
269  QL_REQUIRE(quotesRead > 0, "Wild card quote specified, but no quotes read.");
270 
271  // sort
272  std::sort(qt.begin(), qt.end(),
273  [](const boost::shared_ptr<EquityForwardQuote>& a,
274  const boost::shared_ptr<EquityForwardQuote>& b) -> bool {
275  return a->expiryDate() < b->expiryDate();
276  });
277 
278  // populate individual quote, term vectors
279  for (Size i = 0; i < qt.size(); i++) {
280  terms_.push_back(qt[i]->expiryDate());
281  // convert quote from minor to major currency if needed
282  quotes_.push_back(convertMinorToMajorCurrency(qt[i]->ccy(), qt[i]->quote()->value()));
283  }
284  }
285  }
286  if (quotes_.size() == 0) {
287  DLOG("No Equity Forward quotes provided for " << config->curveID()
288  << ", continuing without dividend curve.");
289  buildCurveType = EquityCurveConfig::Type::NoDividends;
290  }
291  } else if (curveType_ == EquityCurveConfig::Type::OptionPremium) {
292 
293  if (oqt.size() == 0) {
294  DLOG("No Equity Option quotes provided for " << config->curveID()
295  << ", continuing without dividend curve.");
296  buildCurveType = EquityCurveConfig::Type::NoDividends;
297  } else {
298  DLOG("Building Equity Dividend Yield curve from Option Volatilities");
299  vector<boost::shared_ptr<EquityOptionQuote>> calls, puts;
300  vector<Date> callDates, putDates;
301  vector<Real> callStrikes, putStrikes;
302  vector<Real> callPremiums, putPremiums;
303  Calendar cal = NullCalendar();
304 
305  // Split the quotes into call and puts
306  for (auto q : oqt) {
307  if (q->quote()->value() > 0) {
308  if (q->isCall()) {
309  calls.push_back(q);
310  } else {
311  puts.push_back(q);
312  }
313  }
314  }
315 
316  // loop through the quotes and get the expiries, strikes, vols and types
317  // We only want overlapping expiry/strike pairs
318  for (Size i = 0; i < calls.size(); i++) {
319  for (Size j = 0; j < puts.size(); j++) {
320  if (calls[i]->expiry() == puts[j]->expiry()) {
321  auto callAbsoluteStrike = boost::dynamic_pointer_cast<AbsoluteStrike>(calls[i]->strike());
322  auto putAbsoluteStrike = boost::dynamic_pointer_cast<AbsoluteStrike>(puts[j]->strike());
323  QL_REQUIRE(callAbsoluteStrike, "Expected absolute strike for quote " << calls[i]->name());
324  QL_REQUIRE(putAbsoluteStrike, "Expected absolute strike for quote " << puts[j]->name());
325  if (*calls[i]->strike() == *puts[j]->strike()) {
326  TLOG("Adding Call and Put for strike/expiry pair : " << calls[i]->expiry() << "/"
327  << calls[i]->strike()->toString());
328  callDates.push_back(getDateFromDateOrPeriod(calls[i]->expiry(), asof));
329  putDates.push_back(getDateFromDateOrPeriod(puts[j]->expiry(), asof));
330 
331  // convert strike to major currency if quoted in minor
332  Real callStrike =
333  convertMinorToMajorCurrency(calls[i]->ccy(), callAbsoluteStrike->strike());
334  Real putStrike =
335  convertMinorToMajorCurrency(puts[j]->ccy(), putAbsoluteStrike->strike());
336  callStrikes.push_back(callStrike);
337  putStrikes.push_back(putStrike);
338  // convert premium to major currency if quoted in minor
339  callPremiums.push_back(
340  convertMinorToMajorCurrency(calls[i]->ccy(), calls[i]->quote()->value()));
341  putPremiums.push_back(
342  convertMinorToMajorCurrency(puts[j]->ccy(), puts[j]->quote()->value()));
343  }
344  }
345  }
346  }
347 
348  if (callDates.size() > 0 && putDates.size() > 0) {
349  DLOG("Found " << callDates.size() << " Call and Put Option Volatilities");
350 
351  DLOG("Building a Sparse Volatility surface for calls and puts");
352  // Build a Black Variance Sparse matrix
353  boost::shared_ptr<OptionPriceSurface> callSurface =
354  boost::make_shared<OptionPriceSurface>(asof, callDates, callStrikes, callPremiums, dc_);
355  boost::shared_ptr<OptionPriceSurface> putSurface =
356  boost::make_shared<OptionPriceSurface>(asof, putDates, putStrikes, putPremiums, dc_);
357  DLOG("CallSurface contains " << callSurface->expiries().size() << " expiries.");
358 
359  DLOG("Stripping equity forwards from the option premium surfaces");
360  boost::shared_ptr<EquityForwardCurveStripper> efcs = boost::make_shared<EquityForwardCurveStripper>(
361  callSurface, putSurface, forecastYieldTermStructure, equitySpot, config->exerciseStyle());
362 
363  // set terms and quotes from the stripper
364  terms_ = efcs->expiries();
365  quotes_ = efcs->forwards();
366  } else {
367  DLOG("No overlapping call and put quotes for equity " << spec.curveConfigID()
368  << " building NoDividends curve");
369  buildCurveType = EquityCurveConfig::Type::NoDividends;
370  }
371  }
372  }
373 
374  // Build the Dividend Yield curve from the quotes loaded
375  vector<Rate> dividendRates;
376  if (buildCurveType == EquityCurveConfig::Type::ForwardPrice ||
377  buildCurveType == EquityCurveConfig::Type::ForwardDividendPrice ||
378  buildCurveType == EquityCurveConfig::Type::OptionPremium) {
379  // Convert Fwds into dividends.
380  // Fwd = Spot e^{(r-q)T}
381  // => q = 1/T Log(Spot/Fwd) + r
382  for (Size i = 0; i < quotes_.size(); i++) {
383  QL_REQUIRE(quotes_[i] > 0, "Invalid Forward Price " << quotes_[i] << " for " << spec_.name()
384  << ", expiry: " << terms_[i]);
385  Time t = dc_.yearFraction(asof, terms_[i]);
386  Rate ir_rate = forecastYieldTermStructure->zeroRate(t, Continuous);
387  dividendRates.push_back(::log(equitySpot->value() / quotes_[i]) / t + ir_rate);
388  }
389  } else if (buildCurveType == EquityCurveConfig::Type::DividendYield) {
390  DLOG("Building Equity Dividend Yield curve from Dividend Yield rates");
391  dividendRates = quotes_;
392  } else if (buildCurveType == EquityCurveConfig::Type::NoDividends) {
393  DLOG("Building flat Equity Dividend Yield curve as no quotes provided");
394  // Return a flat curve @ 0%
395  dividendYieldTermStructure = Handle<YieldTermStructure>(boost::make_shared<FlatForward>(asof, 0.0, dc_));
396  equityIndex_ =
397  boost::make_shared<EquityIndex>(spec.curveConfigID(), calendar, parseCurrency(config->currency()),
398  equitySpot, forecastYieldTermStructure, dividendYieldTermStructure);
399  return;
400  } else
401  QL_FAIL("Invalid Equity curve configuration type for " << spec_.name());
402 
403  QL_REQUIRE(dividendRates.size() > 0, "No dividend yield rates extracted for " << spec_.name());
404  QL_REQUIRE(dividendRates.size() == terms_.size(), "vector size mismatch - dividend rates ("
405  << dividendRates.size() << ") vs terms (" << terms_.size()
406  << ")");
407 
408  // store "dividend discount factors" - in case we wish to interpolate according to discounts
409  vector<Real> dividendDiscountFactors;
410  for (Size i = 0; i < quotes_.size(); i++) {
411  Time t = dc_.yearFraction(asof, terms_[i]);
412  dividendDiscountFactors.push_back(std::exp(-dividendRates[i] * t));
413  }
414 
415  boost::shared_ptr<YieldTermStructure> divCurve;
416  // Build Dividend Term Structure
417  if (dividendRates.size() == 1) {
418  // We only have 1 quote so we build a flat curve
419  divCurve.reset(new FlatForward(asof, dividendRates[0], dc_));
420  } else {
421  // Build a ZeroCurve
422  Size n = terms_.size();
423  vector<Date> dates(n + 1); // n+1 so we include asof
424  vector<Rate> rates(n + 1);
425  vector<Real> discounts(n + 1);
426  for (Size i = 0; i < n; i++) {
427  dates[i + 1] = terms_[i];
428  rates[i + 1] = dividendRates[i];
429  discounts[i + 1] = dividendDiscountFactors[i];
430  }
431  dates[0] = asof;
432  rates[0] = rates[1];
433  discounts[0] = 1.0;
434  if (forecastYieldTermStructure->maxDate() > dates.back()) {
435  dates.push_back(forecastYieldTermStructure->maxDate());
436  rates.push_back(rates.back());
437  Time maxTime = dc_.yearFraction(asof, forecastYieldTermStructure->maxDate());
438  discounts.push_back(
439  std::exp(-rates.back() * maxTime)); // flat zero extrapolation used to imply dividend DF
440  }
441  if (dividendInterpVariable_ == YieldCurve::InterpolationVariable::Zero) {
442  divCurve = zerocurve(dates, rates, dc_, dividendInterpMethod_);
443  } else if (dividendInterpVariable_ == YieldCurve::InterpolationVariable::Discount) {
444  divCurve = discountcurve(dates, discounts, dc_, dividendInterpMethod_);
445  } else {
446  QL_FAIL("Unsupported interpolation variable for dividend yield curve");
447  }
448  divCurve->enableExtrapolation();
449  }
450  dividendYieldTermStructure = Handle<YieldTermStructure>(divCurve);
451 
452  equityIndex_ =
453  boost::make_shared<EquityIndex>(spec.curveConfigID(), calendar, parseCurrency(config->currency()),
454  equitySpot, forecastYieldTermStructure, dividendYieldTermStructure);
455 
456  if (buildCalibrationInfo) {
457 
458  // set calibration info
459 
460  calibrationInfo_ = boost::make_shared<YieldCurveCalibrationInfo>();
461  calibrationInfo_->dayCounter = dc_.name();
462  calibrationInfo_->currency = config->currency();
463  for (auto const& p : YieldCurveCalibrationInfo::defaultPeriods) {
464  Date d = asof + p;
465  calibrationInfo_->pillarDates.push_back(d);
466  calibrationInfo_->zeroRates.push_back(dividendYieldTermStructure->zeroRate(d, dc_, Continuous));
467  calibrationInfo_->discountFactors.push_back(dividendYieldTermStructure->discount(d));
468  calibrationInfo_->times.push_back(dividendYieldTermStructure->timeFromReference(d));
469  }
470  }
471 
472  } catch (std::exception& e) {
473  QL_FAIL("equity curve building failed: " << e.what());
474  } catch (...) {
475  QL_FAIL("equity curve building failed: unknown error");
476  }
477 };
