65void EquityUnderlying::setEquityName() {
66 if (equityName_.empty()) {
67 string name = name_;
68 if (!identifierType_.empty())
69 name = identifierType_ + ":" + name_;
70 if (!currency_.empty())
71 name = name + ":" + currency_;
72 if (!exchange_.empty()) {
73 if (currency_.empty())
74 name = name + ":";
75 name = name + ":" + exchange_;
76 }
77 equityName_ = name;
78 }
79}

81void EquityUnderlying::fromXML(XMLNode* node) {
82 if (XMLUtils::getNodeName(node) == basicUnderlyingNodeName_) {
83 name_ = XMLUtils::getNodeValue(node);
84 isBasic_ = true;
85 } else if (XMLUtils::getNodeName(node) == nodeName_) {
86 Underlying::fromXML(node);
87 QL_REQUIRE(type_ == "Equity", "Underlying must be of type 'Equity'.");
88 identifierType_ = XMLUtils::getChildValue(node, "IdentifierType", false);
89 // if no identifier is provided, we just use name
90 if (!identifierType_.empty()) {
91 currency_ = XMLUtils::getChildValue(node, "Currency", false);
92 exchange_ = XMLUtils::getChildValue(node, "Exchange", false);
93 }
94 setEquityName();
95 isBasic_ = false;
96 } else {
97 QL_FAIL("Need either a " << basicUnderlyingNodeName_ << " or " << nodeName_ << " for EquityUnderlying.");
98 }
99 setType("Equity");
100}

118void CommodityUnderlying::fromXML(XMLNode* node) {

137 QL_FAIL("Need either a Name or Underlying node for CommodityUnderlying.");

297void UnderlyingBuilder::fromXML(XMLNode* node) {
298 if (XMLUtils::getNodeName(node) == basicUnderlyingNodeName_) {
299 underlying_ = boost::make_shared<BasicUnderlying>();
300 } else if (XMLUtils::getNodeName(node) == nodeName_) {
301 string type = XMLUtils::getChildValue(node, "Type", true);
302 if (type == "Equity")
303 underlying_ = boost::make_shared<EquityUnderlying>();
304 else if (type == "Commodity")
305 underlying_ = boost::make_shared<CommodityUnderlying>();
306 else if (type == "FX")
307 underlying_ = boost::make_shared<FXUnderlying>();
308 else if (type == "InterestRate")
309 underlying_ = boost::make_shared<InterestRateUnderlying>();
310 else if (type == "Inflation")
311 underlying_ = boost::make_shared<InflationUnderlying>();
312 else if (type == "Credit")
313 underlying_ = boost::make_shared<CreditUnderlying>();
314 else if (type == "Bond")
315 underlying_ = boost::make_shared<BondUnderlying>();
316 else {
317 QL_FAIL("Unknown Underlying type " << type);
318 }
319 } else {
320 QL_FAIL("Need either a " << basicUnderlyingNodeName_ << " or " << nodeName_ << " node for Underlying.");
321 }
322 QL_REQUIRE(underlying_ != nullptr, "UnderlyingBuilder: underlying_ is null, this is unexpected");
323 underlying_->setNodeName(nodeName_);
324 underlying_->setBasicUnderlyingNodeName(basicUnderlyingNodeName_);
325 underlying_->fromXML(node);
326}
