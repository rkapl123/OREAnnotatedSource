44BOOST_AUTO_TEST_CASE(testCommodityAPO) {
45
46 BOOST_TEST_MESSAGE("Testing Commodity APO Analytical Approximation vs MC Pricing");
47
48 SavedSettings backup;
49
50 Date today(5, Feb, 2019);
51 Settings::instance().evaluationDate() = today;
52 Calendar cal = UnitedStates(UnitedStates::Settlement);
53
54 // Market - flat price curve
55 std::vector<Date> dates = {today + 1 * Years, today + 5 * Years, today + 10 * Years};
56 std::vector<Real> prices = {100.0, 100.0, 100.0};
57 DayCounter dc = Actual365Fixed();
58 Handle<QuantExt::PriceTermStructure> priceCurve(
59 boost::make_shared<InterpolatedPriceCurve<Linear>>(today, dates, prices, dc, USDCurrency()));
60 priceCurve->enableExtrapolation();
61
62 // Market - flat discount curve
63 Handle<Quote> rateQuote(boost::make_shared<SimpleQuote>(0.01));
64 Handle<YieldTermStructure> discountCurve(boost::make_shared<FlatForward>(today, rateQuote, dc, Compounded, Annual));
65
66 // Market - flat volatility structure
67 Handle<QuantLib::BlackVolTermStructure> vol(boost::make_shared<QuantLib::BlackConstantVol>(today, cal, 0.3, dc));
68
69 // Analytical engine
70 Real beta = 0.0;
71 boost::shared_ptr<PricingEngine> analyticalEngine =
72 boost::make_shared<CommodityAveragePriceOptionAnalyticalEngine>(discountCurve, vol, beta);
73
74 // Monte Carlo engine
75 Size samples = 10000;
76 boost::shared_ptr<PricingEngine> mcEngine =
77 boost::make_shared<CommodityAveragePriceOptionMonteCarloEngine>(discountCurve, vol, samples, beta);
78
79 // Instrument
80 Real quantity = 1.0;
81 std::string name = "CL";
82 Period term = 1 * Months;
83
84 // std::vector<ApoTestCase> cases = {{100.0, Option::Call}, {100.0, Option::Put}};
85 std::vector<ApoTestCase> cases = {{100.0, Option::Call}, {120.0, Option::Call}, {80.0, Option::Call},
86 {100.0, Option::Put}, {120.0, Option::Put}, {80.0, Option::Put}};
87
88 for (Size k = 0; k < cases.size(); ++k) {
89
90 Real strikePrice = cases[k].strikePrice;
91 QuantLib::Option::Type optionType = cases[k].optionType;
92 std::string optionTypeString = (optionType == Option::Call ? "Call" : "Put");
93
94 // Vary APO start dates
95 for (Size i = 1; i <= 10; ++i) {
96
97 // Instrument
98 Period startTerm = i * Years;
99 Date startDate = today + startTerm;
100 Date endDate = startDate + term;
101 Date payDate = endDate;
102 boost::shared_ptr<CommoditySpotIndex> index = boost::make_shared<CommoditySpotIndex>(name, cal, priceCurve);
103 boost::shared_ptr<CommodityIndexedAverageCashFlow> flow =
104 boost::make_shared<CommodityIndexedAverageCashFlow>(quantity, startDate, endDate, payDate, index);
105 boost::shared_ptr<Exercise> exercise = boost::make_shared<EuropeanExercise>(endDate);
106
107 boost::shared_ptr<CommodityAveragePriceOption> apo =
108 boost::make_shared<CommodityAveragePriceOption>(flow, exercise, quantity, strikePrice, optionType);
109
110 boost::timer::cpu_timer tan;
111 apo->setPricingEngine(analyticalEngine);
112 Real anPrice = apo->NPV();
113 tan.stop();
114 Real anTime = tan.elapsed().wall * 1e-6;
115
116 boost::timer::cpu_timer tmc;
117 apo->setPricingEngine(mcEngine);
118 Real mcPrice = apo->NPV();
119 Real mcTime = tmc.elapsed().wall * 1e-6;
120
121 Real error1 = 100.0 * fabs(mcPrice - anPrice) / mcPrice;
122 Real error2 = 100.0 * fabs(mcPrice - anPrice) / anPrice;
123
124 BOOST_TEST_MESSAGE(optionTypeString << " " << std::fixed << std::setprecision(2) << strikePrice << " "
125 << startTerm << " Analytical vs MC price: " << std::fixed
126 << std::setprecision(2) << anPrice << " vs " << mcPrice << " diff "
127 << anPrice - mcPrice << " error " << std::max(error1, error2) << "% ("
128 << anTime << " ms, " << mcTime << " ms)");
129 BOOST_CHECK_CLOSE(anPrice, mcPrice, 1.0); // accept 1% relative error
130 }
131 }
132}
