133void TradeFactory::addExtraBuilders(std::map<string, boost::shared_ptr<AbstractTradeBuilder>> extraBuilders) {
134 if (extraBuilders.size() > 0) {
135 LOG("adding " << extraBuilders.size() << " extra trade builders");
136 for (auto eb : extraBuilders)
137 addBuilder(eb.first, eb.second);
138 }
139}

141boost::shared_ptr<Trade> TradeFactory::build(const string& className) const {
142 auto it = builders_.find(className);
143
144 if (it == builders_.end()) {
145 return boost::shared_ptr<Trade>();}
146 else {
147 return it->second->build();}
148}
