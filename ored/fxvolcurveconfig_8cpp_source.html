52const vector<string>& FXVolatilityCurveConfig::quotes() {
53 if (quotes_.size() == 0) {
54 vector<string> tokens;
55 boost::split(tokens, fxSpotID(), boost::is_any_of("/"));
56 QL_REQUIRE(tokens.size() == 3, "Expected 3 tokens FX/CCY1/CCY2 in fxSpotID (" << fxSpotID() << ")");
57 quotes_.push_back("FX/RATE/" + tokens[1] + "/" + tokens[2]);
58 string base = "FX_OPTION/RATE_LNVOL/" + tokens[1] + "/" + tokens[2] + "/";
59 for (auto e : expiries_) {
60 quotes_.push_back(base + e + "/ATM");
61 if (dimension_ == Dimension::SmileVannaVolga || dimension_ == Dimension::SmileBFRR) {
62 for (auto const& d : smileDelta_) {
63 quotes_.push_back(base + e + "/" + to_string(d) + "RR");
64 quotes_.push_back(base + e + "/" + to_string(d) + "BF");
65 }
66 } else if (dimension_ == Dimension::SmileDelta || dimension_ == Dimension::SmileAbsolute) {
67 for (auto d : deltas_) {
68 quotes_.push_back(base + e + "/" + d);
69 }
70 }
71 }
72 }
73 return quotes_;
74}

76void FXVolatilityCurveConfig::fromXML(XMLNode* node) {
77 XMLUtils::checkNode(node, "FXVolatility");
78
79 curveID_ = XMLUtils::getChildValue(node, "CurveId", true);
80 curveDescription_ = XMLUtils::getChildValue(node, "CurveDescription", true);
81 string dim = XMLUtils::getChildValue(node, "Dimension", true);
82 string cal = XMLUtils::getChildValue(node, "Calendar");
83 string smileInterp = XMLUtils::getChildValue(node, "SmileInterpolation");
84
85 fxSpotID_ = XMLUtils::getChildValue(node, "FXSpotID", true);
86
87 vector<string> tokens;
88 boost::split(tokens, fxSpotID_, boost::is_any_of("/"));
89 QL_REQUIRE(tokens.size() == 3, "Expected 3 tokens FX/CCY1/CCY2 in fxSpotID (" << fxSpotID_ << ")");
90 if (cal == "") {
91 cal = tokens[1] + "," + tokens[2];
92 }
93 calendar_ = parseCalendar(cal);
94
95 string dc = XMLUtils::getChildValue(node, "DayCounter");
96 if (dc == "")
97 dc = "A365";
98 dayCounter_ = parseDayCounter(dc);
99
100 if (dim == "ATMTriangulated") {
101 dimension_ = Dimension::ATMTriangulated;
102 baseVolatility1_ = XMLUtils::getChildValue(node, "BaseVolatility1", true);
103 baseVolatility2_ = XMLUtils::getChildValue(node, "BaseVolatility2", true);
104
105 string fxIndexTag = XMLUtils::getChildValue(node, "FXIndexTag");
106 if (fxIndexTag == "")
107 fxIndexTag = "GENERIC";
108 fxIndexTag_ = fxIndexTag;
109 } else {
110 if (dim == "ATM") {
111 dimension_ = Dimension::ATM;
112 } else if (dim == "Smile") {
113
114 conventionsID_ = XMLUtils::getChildValue(node, "Conventions", false);
115 string smileType = XMLUtils::getChildValue(node, "SmileType");
116 if (smileType == "" || smileType == "VannaVolga") {
117 dimension_ = Dimension::SmileVannaVolga;
118
119 // only read smile interpolation method if dimension is smile.
120 if (smileInterp == "") {
121 smileInterpolation_ = SmileInterpolation::VannaVolga2; // default to VannaVolga 2nd approximation
122 } else if (smileInterp == "VannaVolga1") {
123 smileInterpolation_ = SmileInterpolation::VannaVolga1;
124 } else if (smileInterp == "VannaVolga2") {
125 smileInterpolation_ = SmileInterpolation::VannaVolga2;
126 } else {
127 QL_FAIL("SmileInterpolation " << smileInterp << " not supported");
128 }
129
130 string sDelta = XMLUtils::getChildValue(node, "SmileDelta");
131 if (sDelta == "")
132 smileDelta_ = {25};
133 else
134 smileDelta_ = parseListOfValues<Size>(sDelta, &parseInteger);
135 } else if (smileType == "Delta") {
136 dimension_ = Dimension::SmileDelta;
137 // only read smile interpolation method if dimension is smile.
138 if (smileInterp == "" || smileInterp == "Linear") {
139 smileInterpolation_ = SmileInterpolation::Linear;
140 } else if (smileInterp == "Cubic") {
141 smileInterpolation_ = SmileInterpolation::Cubic;
142 } else {
143 QL_FAIL("SmileInterpolation " << smileInterp << " not supported");
144 }
145 deltas_ = XMLUtils::getChildrenValuesAsStrings(node, "Deltas", true);
146
147 // check that these are valid deltas
148 for (auto d : deltas_) {
149 QL_REQUIRE(d == "ATM" || d.back() == 'P' || d.back() == 'C',
150 "this is not a valid value for delta, " << d);
151 if (d != "ATM") {
152 parseReal(d.substr(0, d.size() - 1));
153 }
154 }
155 } else if (smileType == "BFRR") {
156 dimension_ = Dimension::SmileBFRR;
157 if (smileInterp == "" || smileInterp == "Cubic") {
158 smileInterpolation_ = SmileInterpolation::Cubic;
159 } else if (smileInterp == "Linear") {
160 smileInterpolation_ = SmileInterpolation::Linear;
161 } else {
162 QL_FAIL("SmileInterpolation " << smileInterp << " not supported");
163 }
164 string sDelta = XMLUtils::getChildValue(node, "SmileDelta");
165 if (sDelta == "")
166 smileDelta_ = {10, 25};
167 else
168 smileDelta_ = parseListOfValues<Size>(sDelta, &parseInteger);
169 } else if (smileType == "Absolute") {
170 dimension_ = Dimension::SmileAbsolute;
171 if (smileInterp == "" || smileInterp == "Cubic") {
172 smileInterpolation_ = SmileInterpolation::Cubic;
173 } else if (smileInterp == "Linear") {
174 smileInterpolation_ = SmileInterpolation::Linear;
175 } else {
176 QL_FAIL("SmileInterpolation " << smileInterp << " not supported");
177 }
178 } else {
179 QL_FAIL("SmileType '" << smileType << "' not supported, expected VannaVolga, Delta, BFRR");
180 }
181 } else {
182 QL_FAIL("Dimension " << dim << " not supported yet");
183 }
184 expiries_ = XMLUtils::getChildrenValuesAsStrings(node, "Expiries", true);
185
186 bool curvesRequired = dimension_ == Dimension::SmileVannaVolga || dimension_ == Dimension::SmileDelta ||
187 dimension_ == Dimension::SmileBFRR;
188 fxForeignYieldCurveID_ = XMLUtils::getChildValue(node, "FXForeignCurveID", curvesRequired);
189 fxDomesticYieldCurveID_ = XMLUtils::getChildValue(node, "FXDomesticCurveID", curvesRequired);
190 }
191
192 if (auto tmp = XMLUtils::getChildNode(node, "Report")) {
193 reportConfig_.fromXML(tmp);
194 }
195
196 populateRequiredCurveIds();
197}

199XMLNode* FXVolatilityCurveConfig::toXML(XMLDocument& doc) {
200 XMLNode* node = doc.allocNode("FXVolatility");
201
202 XMLUtils::addChild(doc, node, "CurveId", curveID_);
203 XMLUtils::addChild(doc, node, "CurveDescription", curveDescription_);
204 if (dimension_ == Dimension::ATM) {
205 XMLUtils::addChild(doc, node, "Dimension", "ATM");
206 } else if (dimension_ == Dimension::ATMTriangulated) {
207 XMLUtils::addChild(doc, node, "Dimension", "ATMTriangulated");
208 XMLUtils::addChild(doc, node, "FXSpotID", fxSpotID_);
209 XMLUtils::addChild(doc, node, "FXIndexTag", fxIndexTag_);
210 XMLUtils::addChild(doc, node, "BaseVolatility1", baseVolatility1_);
211 XMLUtils::addChild(doc, node, "BaseVolatility2", baseVolatility2_);
212
213 return node;
214 } else if (dimension_ == Dimension::SmileVannaVolga) {
215 XMLUtils::addChild(doc, node, "Dimension", "Smile");
216 XMLUtils::addChild(doc, node, "SmileType", "VannaVolga");
217 // only write smile interpolation if dimension is smile
218 if (smileInterpolation_ == SmileInterpolation::VannaVolga1) {
219 XMLUtils::addChild(doc, node, "SmileInterpolation", "VannaVolga1");
220 } else if (smileInterpolation_ == SmileInterpolation::VannaVolga2) {
221 XMLUtils::addChild(doc, node, "SmileInterpolation", "VannaVolga2");
222 } else {
223 QL_FAIL("Unknown SmileInterpolation in FXVolatilityCurveConfig::toXML()");
224 }
225 XMLUtils::addGenericChildAsList(doc, node, "SmileDelta", deltas_);
226 XMLUtils::addChild(doc, node, "Conventions", to_string(conventionsID_));
227 } else if (dimension_ == Dimension::SmileDelta) {
228 XMLUtils::addChild(doc, node, "Dimension", "Smile");
229 XMLUtils::addChild(doc, node, "SmileType", "Delta");
230 // only write smile interpolation if dimension is smile
231 if (smileInterpolation_ == SmileInterpolation::Linear) {
232 XMLUtils::addChild(doc, node, "SmileInterpolation", "Linear");
233 } else if (smileInterpolation_ == SmileInterpolation::Cubic) {
234 XMLUtils::addChild(doc, node, "SmileInterpolation", "Cubic");
235 } else {
236 QL_FAIL("Unknown SmileInterpolation in FXVolatilityCurveConfig::toXML()");
237 }
238 XMLUtils::addChild(doc, node, "Conventions", to_string(conventionsID_));
239 XMLUtils::addGenericChildAsList(doc, node, "Deltas", deltas_);
240 } else if (dimension_ == Dimension::SmileBFRR) {
241 XMLUtils::addChild(doc, node, "Dimension", "Smile");
242 XMLUtils::addChild(doc, node, "SmileType", "BFRR");
243 if (smileInterpolation_ == SmileInterpolation::Linear) {
244 XMLUtils::addChild(doc, node, "SmileInterpolation", "Linear");
245 } else if (smileInterpolation_ == SmileInterpolation::Cubic) {
246 XMLUtils::addChild(doc, node, "SmileInterpolation", "Cubic");
247 } else {
248 QL_FAIL("Unknown SmileInterpolation in FXVolatilityCurveConfig::toXML()");
249 }
250 XMLUtils::addGenericChildAsList(doc, node, "SmileDelta", smileDelta_);
251 XMLUtils::addChild(doc, node, "Conventions", to_string(conventionsID_));
252 } else if (dimension_ == Dimension::SmileAbsolute) {
253 XMLUtils::addChild(doc, node, "Dimension", "Smile");
254 XMLUtils::addChild(doc, node, "SmileType", "Absolute");
255 if (smileInterpolation_ == SmileInterpolation::Linear) {
256 XMLUtils::addChild(doc, node, "SmileInterpolation", "Linear");
257 } else if (smileInterpolation_ == SmileInterpolation::Cubic) {
258 XMLUtils::addChild(doc, node, "SmileInterpolation", "Cubic");
259 } else {
260 QL_FAIL("Unknown SmileInterpolation in FXVolatilityCurveConfig::toXML()");
261 }
262 XMLUtils::addChild(doc, node, "Conventions", to_string(conventionsID_));
263 } else {
264 QL_FAIL("Unknown Dimension in FXVolatilityCurveConfig::toXML()");
265 }
266 XMLUtils::addGenericChildAsList(doc, node, "Expiries", expiries_);
267 XMLUtils::addChild(doc, node, "FXSpotID", fxSpotID_);
268 if (!fxForeignYieldCurveID_.empty())
269 XMLUtils::addChild(doc, node, "FXForeignCurveID", fxForeignYieldCurveID_);
270 if (!fxDomesticYieldCurveID_.empty())
271 XMLUtils::addChild(doc, node, "FXDomesticCurveID", fxDomesticYieldCurveID_);
272 XMLUtils::addChild(doc, node, "Calendar", to_string(calendar_));
273 XMLUtils::addChild(doc, node, "DayCounter", to_string(dayCounter_));
274 XMLUtils::appendNode(node, reportConfig_.toXML(doc));
275
276 return node;
277}

279void FXVolatilityCurveConfig::populateRequiredCurveIds() {
280 if (!fxDomesticYieldCurveID_.empty() && !fxForeignYieldCurveID_.empty()) {
281 std::vector<string> domTokens, forTokens;
282 split(domTokens, fxDomesticYieldCurveID_, boost::is_any_of("/"));
283 split(forTokens, fxForeignYieldCurveID_, boost::is_any_of("/"));
284
285 if (domTokens.size() == 3 && domTokens[0] == "Yield") {
286 requiredCurveIds_[CurveSpec::CurveType::Yield].insert(domTokens[2]);
287 } else if (domTokens.size() == 1) {
288 requiredCurveIds_[CurveSpec::CurveType::Yield].insert(fxDomesticYieldCurveID_);
289 } else {
290 QL_FAIL("Cannot determine the required domestic yield curve for fx vol curve " << curveID_);
291 }
292
293 if (forTokens.size() == 3 && forTokens[0] == "Yield") {
294 requiredCurveIds_[CurveSpec::CurveType::Yield].insert(forTokens[2]);
295 } else if (forTokens.size() == 1) {
296 requiredCurveIds_[CurveSpec::CurveType::Yield].insert(fxForeignYieldCurveID_);
297 } else {
298 QL_FAIL("Cannot determine the required foreign yield curve for fx vol curve " << curveID_);
299 }
300 }
301
302 if (dimension_ == Dimension::ATMTriangulated) {
303 requiredCurveIds_[CurveSpec::CurveType::FXVolatility].insert(baseVolatility1_);
304 requiredCurveIds_[CurveSpec::CurveType::FXVolatility].insert(baseVolatility2_);
305
306 vector<string> tokens;
307 boost::split(tokens, fxSpotID_, boost::is_any_of("/"));
308 QL_REQUIRE(tokens.size() == 3, "unexpected fxSpot format: " << fxSpotID_);
309 auto forTarget = tokens[1];
310 auto domTarget = tokens[2];
311
312 // we need to include inverse ccy pairs as well
313 QL_REQUIRE(baseVolatility1_.size() == 6, "invalid ccy pair length");
314 QL_REQUIRE(baseVolatility2_.size() == 6, "invalid ccy pair length");
315 auto forBase1 = baseVolatility1_.substr(0, 3);
316 auto domBase1 = baseVolatility1_.substr(3);
317 auto forBase2 = baseVolatility2_.substr(0, 3);
318 auto domBase2 = baseVolatility2_.substr(3);
319
320 requiredCurveIds_[CurveSpec::CurveType::FXVolatility].insert(domBase1 + forBase1);
321 requiredCurveIds_[CurveSpec::CurveType::FXVolatility].insert(domBase2 + forBase2);
322
323 // we need to establish the common currency between the two pairs to include the correlations
324 std::string baseCcy = "";
325 if (forBase1 == forBase2 || forBase1 == domBase2) {
326 baseCcy = forBase1;
327 } else {
328 QL_REQUIRE(domBase1 == forBase2 || domBase1 == domBase2, "no common currency found for baseVolatilities");
329 baseCcy = domBase1;
330 }
331
332 // straight pair
333 std::string forIndex = "FX-" + fxIndexTag_ + "-" + forTarget + "-" + baseCcy;
334 std::string domIndex = "FX-" + fxIndexTag_ + "-" + domTarget + "-" + baseCcy;
335 requiredCurveIds_[CurveSpec::CurveType::Correlation].insert(forIndex + "&" + domIndex);
336 // inverse pair
337 requiredCurveIds_[CurveSpec::CurveType::Correlation].insert(domIndex + "&" + forIndex);
338 // inverse fx index1
339 std::string forIndexInverse = "FX-" + fxIndexTag_ + "-" + baseCcy + "-" + forTarget;
340 requiredCurveIds_[CurveSpec::CurveType::Correlation].insert(forIndexInverse + "&" + domIndex);
341 requiredCurveIds_[CurveSpec::CurveType::Correlation].insert(domIndex + "&" + forIndexInverse);
342 // inverse fx index2
343 std::string domIndexInverse = "FX-" + fxIndexTag_ + "-" + baseCcy + "-" + domTarget;
344 requiredCurveIds_[CurveSpec::CurveType::Correlation].insert(forIndex + "&" + domIndexInverse);
345 requiredCurveIds_[CurveSpec::CurveType::Correlation].insert(domIndexInverse + "&" + forIndex);
346 // both fx indices inverted
347 requiredCurveIds_[CurveSpec::CurveType::Correlation].insert(forIndexInverse + "&" + domIndexInverse);
348 requiredCurveIds_[CurveSpec::CurveType::Correlation].insert(domIndexInverse + "&" + forIndexInverse);
349 }
