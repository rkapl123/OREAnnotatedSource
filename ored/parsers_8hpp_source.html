292//! Convert comma separated list of values to vector of values
293/*!
294\ingroup utilities
295*/
296template <class T> std::vector<T> parseListOfValues(string s, std::function<T(string)> parser) {
297 boost::trim(s);
298 std::vector<T> vec;
299 boost::char_separator<char> sep(",");
300 boost::tokenizer<boost::char_separator<char>> tokens(s, sep);
301 for (auto r : tokens) {
302 boost::trim(r);
303 vec.push_back(parser(r));
304 }
305 return vec;
306}
307
308template <class T> std::vector<T> parseVectorOfValues(std::vector<std::string> str, std::function<T(string)> parser) {
309 std::vector<T> vec;
310 for (auto s : str) {
311 vec.push_back(parser(s));
312 }
313 return vec;
314}

391/*! Attempt to parse string \p str to \p obj of type \c T using \p parser
392 \param[in] str The string we wish to parse.
393 \param[out] obj The resulting object if the parsing was successful.
394 \param[in] parser The function to use to attempt to parse \p str. This function may throw.
395
396 \return \c true if the parsing was successful and \c false if not.
397
398 \ingroup utilities
399*/
400template <class T> bool tryParse(const std::string& str, T& obj, std::function<T(std::string)> parser) {
401 DLOG("tryParse: attempting to parse " << str);
402 try {
403 obj = parser(str);
404 } catch (...) {
405 TLOG("String " << str << " could not be parsed");
406 return false;
407 }
408 return true;
409}
