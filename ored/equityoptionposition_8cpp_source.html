67void EquityOptionPosition::build(const boost::shared_ptr<ore::data::EngineFactory>& engineFactory) {
68 DLOG("EquityOptionPosition::build() called for " << id());
69 QL_REQUIRE(!data_.underlyings().empty(), "EquityOptionPosition::build(): no underlyings given");
70 options_.clear();
71 indices_.clear();
72 weights_.clear();
73 positions_.clear();
74 currencies_.clear();
75 fxConversion_.clear();
76
77 for (auto const& u : data_.underlyings()) {
78
79 // get equity, populate weight, currency
80
81 auto eq = *engineFactory->market()->equityCurve(u.underlying().name(),
82 engineFactory->configuration(MarketContext::pricing));
83 weights_.push_back(u.underlying().weight());
84 QL_REQUIRE(!eq->currency().empty(),
85 "did not get currency for equity name '" << u.underlying().name() << "', is this set up?");
86 currencies_.push_back(eq->currency().code());
87 QuantLib::Position::Type pos = parsePositionType(u.optionData().longShort());
88 Real posInd = (pos == QuantLib::Position::Long ? 1.0 : -1.0);
89 positions_.push_back(posInd);
90
91 // build vanilla option and attach engine
92
93 Option::Type optionType = parseOptionType(u.optionData().callPut());
94 QuantLib::Exercise::Type exerciseType = parseExerciseType(u.optionData().style());
95 QL_REQUIRE(u.optionData().exerciseDates().size() == 1, "Invalid number of excercise dates");
96 Date optionExpiry = parseDate(u.optionData().exerciseDates().front());
97 boost::shared_ptr<Exercise> exercise;
98 switch (exerciseType) {
99 case QuantLib::Exercise::Type::European: {
100 exercise = boost::make_shared<EuropeanExercise>(optionExpiry);
101 break;
102 }
103 case QuantLib::Exercise::Type::American: {
104 exercise = boost::make_shared<AmericanExercise>(optionExpiry, u.optionData().payoffAtExpiry());
105 break;
106 }
107 default:
108 QL_FAIL("Option Style " << u.optionData().style() << " is not supported");
109 }
110 options_.push_back(boost::make_shared<VanillaOption>(
111 boost::make_shared<PlainVanillaPayoff>(optionType, u.strike()), exercise));
112 if (!options_.back()->isExpired()) {
113 std::string tradeTypeBuilder =
114 (exerciseType == QuantLib::Exercise::Type::European ? "EquityOption" : "EquityOptionAmerican");
115 boost::shared_ptr<VanillaOptionEngineBuilder> builder =
116 boost::dynamic_pointer_cast<VanillaOptionEngineBuilder>(engineFactory->builder(tradeTypeBuilder));
117 QL_REQUIRE(builder, "EquityOptionPosition::build(): no engine builder for '" << tradeTypeBuilder << "'");
118 options_.back()->setPricingEngine(builder->engine(u.underlying().name(), eq->currency(), optionExpiry));
119 }
120
121 // populate index for historical prices
122
123 // ensure the strike appears as 2400.2 (i.e. with decimal places as necessary)
124 std::ostringstream strikeStr;
125 strikeStr << u.strike();
126
127 string underlyingName = u.underlying().name();
128 if (engineFactory->referenceData() && engineFactory->referenceData()->hasData("Equity", underlyingName)) {
129 const auto& underlyingRef = engineFactory->referenceData()->getData("Equity", underlyingName);
130 if (auto equityRef = boost::dynamic_pointer_cast<EquityReferenceDatum>(underlyingRef))
131 underlyingName = equityRef->equityData().equityId;
132 }
133 indices_.push_back(boost::make_shared<QuantExt::GenericIndex>(
134 "GENERIC-MD/EQUITY_OPTION/PRICE/" + underlyingName + "/" + eq->currency().code() + "/" +
135 ore::data::to_string(optionExpiry) + "/" + strikeStr.str() + "/" +
136 (optionType == Option::Call ? "C" : "P")));
137 }
138
139 // get fx quotes
140
141 isSingleCurrency_ = true;
142 npvCurrency_ = currencies_.front();
143 for (auto const& c : currencies_) {
144 // we use fxSpot() as opposed to fxRate() here to ensure consistency between NPV() and the fixing of an
145 // equivalent index representing the same basket
146 fxConversion_.push_back(
147 engineFactory->market()->fxSpot(c + npvCurrency_, engineFactory->configuration(MarketContext::pricing)));
148 if (npvCurrency_ != c)
149 isSingleCurrency_ = false;
150 }
151
152 // set instrument
153 auto qlInstr =
154 boost::make_shared<EquityOptionPositionInstrumentWrapper>(data_.quantity(), options_, weights_, positions_, fxConversion_);
155 qlInstr->setPricingEngine(boost::make_shared<EquityOptionPositionInstrumentWrapperEngine>());
156 instrument_ = boost::make_shared<VanillaInstrument>(qlInstr);
157
158 // no sensible way to set these members
159 maturity_ = Date::maxDate();
160 notional_ = Null<Real>();
161 notionalCurrency_ = "";
162
163 // ISDA taxonomy: not a derivative, but define the asset class at least
164 // so that we can determine a TRS asset class that has an EQ position underlying
165 additionalData_["isdaAssetClass"] = string("Equity");
166 additionalData_["isdaBaseProduct"] = string("");
167 additionalData_["isdaSubProduct"] = string("");
168 additionalData_["isdaTransaction"] = string("");
169
170 // leave legs empty
171}

199EquityOptionPositionInstrumentWrapper::EquityOptionPositionInstrumentWrapper(
200 const Real quantity, const std::vector<boost::shared_ptr<QuantLib::VanillaOption>>& options,
201 const std::vector<Real>& weights, const std::vector<Real>& positions, const std::vector<Handle<Quote>>& fxConversion)
202 : quantity_(quantity), options_(options), weights_(weights), positions_(positions), fxConversion_(fxConversion) {
203 QL_REQUIRE(options_.size() == weights_.size(), "EquityOptionPositionInstrumentWrapper: options size ("
204 << options_.size() << ") must match weights size ("
205 << weights_.size() << ")");
206 QL_REQUIRE(fxConversion_.empty() || fxConversion_.size() == options_.size(),
207 "EquityPositionInstrumentWrapper: fxConversion size ("
208 << fxConversion_.size() << ") must match options size (" << options_.size() << ")");
209 for (auto const& o : options)
210 registerWith(o);
211 for (auto const& fx : fxConversion)
212 registerWith(fx);
213}

245void EquityOptionPositionInstrumentWrapperEngine::calculate() const {
246 Real result = 0.0;
247 for (Size i = 0; i < arguments_.options_.size(); ++i) {
248 Real tmp = arguments_.quantity_ * arguments_.options_[i]->NPV();
249 if (!arguments_.fxConversion_[i].empty()) {
250 tmp *= arguments_.fxConversion_[i]->value();
251 }
252 result += tmp * arguments_.weights_[i] * arguments_.positions_[i];
253 }
254 if (!arguments_.npvCcyConversion_.empty()) {
255 result *= arguments_.npvCcyConversion_->value();
256 }
257 results_.value = result;
258}
