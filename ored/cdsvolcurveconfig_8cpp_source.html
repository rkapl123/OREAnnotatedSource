69 void CDSVolatilityCurveConfig::fromXML(XMLNode* node) {
70 
71  XMLUtils::checkNode(node, "CDSVolatility");
72 
73  curveID_ = XMLUtils::getChildValue(node, "CurveId", true);
74  curveDescription_ = XMLUtils::getChildValue(node, "CurveDescription", true);
75 
76  terms_.clear();
77  termCurves_.clear();
78  if (auto n = XMLUtils::getChildNode(node, "Terms")) {
79  terms_.clear();
80  termCurves_.clear();
81  for (auto c : XMLUtils::getChildrenNodes(n, "Term")) {
82  terms_.push_back(XMLUtils::getChildValueAsPeriod(c, "Label", true));
83  termCurves_.push_back(XMLUtils::getChildValue(c, "Curve", true));
84  }
85  }
86 
87  quoteName_ = "";
88  if (auto n = XMLUtils::getChildNode(node, "QuoteName"))
89  quoteName_ = XMLUtils::getNodeValue(n);
90 
91  if (XMLUtils::getChildNode(node, "Expiries")) {
92 
93  // Giving just an Expiries node is allowed to be backwards compatible but is discouraged.
94  WLOG("Using an Expiries node only in CDSVolatilityCurveConfig is deprecated. "
95  << "A volatility configuration node should be used instead.");
96 
97  // Get the expiries
98  vector<string> expiries = XMLUtils::getChildrenValuesAsStrings(node, "Expiries", true);
99  QL_REQUIRE(expiries.size() > 0, "Need at least one expiry in the Expiries node.");
100 
101  // Build the quotes by appending the expiries and terms to the quote stem.
102  std::vector<std::string> quotes;
103  string stem = quoteStem();
104  for (const string& exp : expiries) {
105  for (auto const& p : terms_) {
106  quotes.push_back(stem + exp + "/" + ore::data::to_string(p));
107  }
108  }
109 
110  // If we have at most 1 term specified, we add quotes without term as well
111  for (const string& exp : expiries) {
112  quotes.push_back(stem + exp);
113  }
114 
115  // Create the relevant volatilityConfig_ object.
116  if (quotes.size() == 1) {
117  volatilityConfig_ = boost::make_shared<ConstantVolatilityConfig>(quotes[0]);
118  } else {
119  volatilityConfig_ = boost::make_shared<VolatilityCurveConfig>(quotes, "Linear", "Flat");
120  }
121 
122  } else {
123  XMLNode* n;
124  if ((n = XMLUtils::getChildNode(node, "Constant"))) {
125  volatilityConfig_ = boost::make_shared<ConstantVolatilityConfig>();
126  } else if ((n = XMLUtils::getChildNode(node, "Curve"))) {
127  volatilityConfig_ = boost::make_shared<VolatilityCurveConfig>();
128  } else if ((n = XMLUtils::getChildNode(node, "StrikeSurface"))) {
129  volatilityConfig_ = boost::make_shared<VolatilityStrikeSurfaceConfig>();
130  } else if ((n = XMLUtils::getChildNode(node, "DeltaSurface"))) {
131  QL_FAIL("CDSVolatilityCurveConfig does not yet support a DeltaSurface.");
132  } else if ((n = XMLUtils::getChildNode(node, "MoneynessSurface"))) {
133  QL_FAIL("CDSVolatilityCurveConfig does not yet support a MoneynessSurface.");
134  } else if ((n = XMLUtils::getChildNode(node, "ProxySurface"))) {
135  volatilityConfig_ = boost::make_shared<CDSProxyVolatilityConfig>();
136  } else {
137  QL_FAIL("CDSVolatility node expects one child node with name in list: Constant,"
138  << " Curve, StrikeSurface, ProxySurface.");
139  }
140  volatilityConfig_->fromXML(n);
141  }
142 
143  dayCounter_ = "A365";
144  if (auto n = XMLUtils::getChildNode(node, "DayCounter"))
145  dayCounter_ = XMLUtils::getNodeValue(n);
146 
147  calendar_ = "NullCalendar";
148  if (auto n = XMLUtils::getChildNode(node, "Calendar"))
149  calendar_ = XMLUtils::getNodeValue(n);
150 
151  strikeType_ = "";
152  if (auto n = XMLUtils::getChildNode(node, "StrikeType"))
153  strikeType_ = XMLUtils::getNodeValue(n);
154 
155  strikeFactor_ = 1.0;
156  if (auto n = XMLUtils::getChildNode(node, "StrikeFactor"))
157  strikeFactor_ = parseReal(XMLUtils::getNodeValue(n));
158 
159  smileDynamics_ = XMLUtils::getChildValue(node, "SmileDynamics", false, "");
160 
161  populateQuotes();
162  populateRequiredCurveIds();
163 }

165 XMLNode* CDSVolatilityCurveConfig::toXML(XMLDocument& doc) {
166 
167  XMLNode* node = doc.allocNode("CDSVolatility");
168 
169  XMLUtils::addChild(doc, node, "CurveId", curveID_);
170  XMLUtils::addChild(doc, node, "CurveDescription", curveDescription_);
171 
172  if (!terms_.empty()) {
173  QL_REQUIRE(terms_.size() == termCurves_.size(),
174  "CDSVolatilityCurveConfig::toXML(): internal error, terms size ("
175  << terms_.size() << ") != termCurves size (" << termCurves_.size() << "), curveId = curveID_");
176  auto termsNode = XMLUtils::addChild(doc, node, "Terms");
177  for (Size i = 0; i < terms_.size(); ++i) {
178  auto tmp = XMLUtils::addChild(doc, termsNode, "Term");
179  XMLUtils::addChild(doc, tmp, "Label", ore::data::to_string(terms_[i]));
180  XMLUtils::addChild(doc, tmp, "Curve", ore::data::to_string(termCurves_[i]));
181  }
182  }
183 
184  XMLNode* n = volatilityConfig_->toXML(doc);
185  XMLUtils::appendNode(node, n);
186 
187  XMLUtils::addChild(doc, node, "DayCounter", dayCounter_);
188  XMLUtils::addChild(doc, node, "Calendar", calendar_);
189  if (!strikeType_.empty())
190  XMLUtils::addChild(doc, node, "StrikeType", strikeType_);
191  if (!quoteName_.empty())
192  XMLUtils::addChild(doc, node, "QuoteName", quoteName_);
193  XMLUtils::addChild(doc, node, "StrikeFactor", strikeFactor_);
194  XMLUtils::addChild(doc, node, "SmileDynamics", smileDynamics_);
195 
196  return node;
197 }

199 void CDSVolatilityCurveConfig::populateQuotes() {
200 
201  // The quotes depend on the type of volatility structure that has been configured.
202  if (auto vc = boost::dynamic_pointer_cast<ConstantVolatilityConfig>(volatilityConfig_)) {
203  quotes_ = {vc->quote()};
204  } else if (auto vc = boost::dynamic_pointer_cast<VolatilityCurveConfig>(volatilityConfig_)) {
205  quotes_ = vc->quotes();
206  } else if (auto vc = boost::dynamic_pointer_cast<VolatilitySurfaceConfig>(volatilityConfig_)) {
207 
208  // Clear the quotes_ if necessary and populate with surface quotes
209  quotes_.clear();
210 
211  string stem = quoteStem();
212  for (const pair<string, string>& p : vc->quotes()) {
213  // build quotes of the form .../TERM/EXPIRY/STRIKE
214  for (auto const& t : terms_) {
215  quotes_.push_back(stem + ore::data::to_string(t) + "/" + p.first + "/" + p.second);
216  }
217  // if only one or even no term is configured, also build quotes of the form .../EXPIRY/STRIKE
218  if (terms_.size() <= 1) {
219  quotes_.push_back(stem + p.first + "/" + p.second);
220  }
221  }
222 
223  } else if (auto vc = boost::dynamic_pointer_cast<CDSProxyVolatilityConfig>(volatilityConfig_)) {
224  // no quotes required in this case
225  } else {
226  QL_FAIL("CDSVolatilityCurveConfig expected a constant, curve or surface");
227  }
228 }

230 void CDSVolatilityCurveConfig::populateRequiredCurveIds() {
231  if (auto vc = boost::dynamic_pointer_cast<CDSProxyVolatilityConfig>(volatilityConfig_)) {
232  requiredCurveIds_[CurveSpec::CurveType::CDSVolatility].insert(vc->cdsVolatilityCurve());
233  }
234  for (auto const& c : termCurves_) {
235  auto spec = parseCurveSpec(c);
236  requiredCurveIds_[CurveSpec::CurveType::Default].insert(spec->curveConfigID());
237  }
238 }

240 string CDSVolatilityCurveConfig::quoteStem() const {
241 
242  string stem{"INDEX_CDS_OPTION/RATE_LNVOL/"};
243 
244  if (!quoteName_.empty()) {
245  // If an explicit quote name has been provided use this.
246  stem += quoteName_;
247  } else {
248  // If not, fallback to just using the curveID_.
249  stem += curveID_;
250  }
251  stem += "/";
252  return stem;
253 }
