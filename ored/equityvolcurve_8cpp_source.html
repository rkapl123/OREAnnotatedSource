57EquityVolCurve::EquityVolCurve(Date asof, EquityVolatilityCurveSpec spec, const Loader& loader,
58 const CurveConfigurations& curveConfigs, const Handle<EquityIndex2>& eqIndex,
59 const map<string, boost::shared_ptr<EquityCurve>>& requiredEquityCurves,
60 const map<string, boost::shared_ptr<EquityVolCurve>>& requiredEquityVolCurves,
61 const map<string, boost::shared_ptr<FXVolCurve>>& requiredFxVolCurves,
62 const map<string, boost::shared_ptr<CorrelationCurve>>& requiredCorrelationCurves,
63 const Market* fxIndices, const bool buildCalibrationInfo) {
64
65 try {
66 LOG("EquityVolCurve: start building equity volatility structure with ID " << spec.curveConfigID());
67
68 auto config = *curveConfigs.equityVolCurveConfig(spec.curveConfigID());
69
70 // if calendar was omitted or left blank, use ccy calendar instead
71 if (config.calendar().empty())
72 calendar_ = parseCalendar(config.ccy());
73 else
74 calendar_ = parseCalendar(config.calendar());
75
76 dayCounter_ = parseDayCounter(config.dayCounter());
77
78 // loop over the volatility configs attempting to build in the order provided
79 DLOG("EquityVolCurve: Attempting to build equity vol curve from volatilityConfig, " << config.volatilityConfig().size() << " volatility configs provided.");
80 for (auto vc : config.volatilityConfig()) {
81 try {
82 // if the volatility config has it's own calendar, we use that.
83 if (!vc->calendar().empty())
84 calendar_ = vc->calendar();
85
86 if (auto eqvc = boost::dynamic_pointer_cast<ProxyVolatilityConfig>(vc)) {
87 buildVolatility(asof, spec, curveConfigs, *eqvc, requiredEquityCurves, requiredEquityVolCurves,
88 requiredFxVolCurves, requiredCorrelationCurves, fxIndices);
89 } else if (auto qvc = boost::dynamic_pointer_cast<QuoteBasedVolatilityConfig>(vc)) {
90 // if the config is quote based (all except proxy surfaces?), do some checks
91 QL_REQUIRE(qvc->quoteType() == MarketDatum::QuoteType::PRICE ||
92 qvc->quoteType() == MarketDatum::QuoteType::RATE_LNVOL,
93 "EquityVolCurve: Only lognormal volatilities and option premiums supported for equity "
94 "volatility surfaces.");
95 if (auto cvc = boost::dynamic_pointer_cast<ConstantVolatilityConfig>(vc)) {
96 buildVolatility(asof, config, *cvc, loader);
97 } else if (auto vcc = boost::dynamic_pointer_cast<VolatilityCurveConfig>(vc)) {
98 buildVolatility(asof, config, *vcc, loader);
99 } else if (auto vssc = boost::dynamic_pointer_cast<VolatilityStrikeSurfaceConfig>(vc)) {
100 buildVolatility(asof, config, *vssc, loader, eqIndex);
101 } else if (auto vmsc = boost::dynamic_pointer_cast<VolatilityMoneynessSurfaceConfig>(vc)) {
102 buildVolatility(asof, config, *vmsc, loader, eqIndex);
103 } else if (auto vdsc = boost::dynamic_pointer_cast<VolatilityDeltaSurfaceConfig>(vc)) {
104 buildVolatility(asof, config, *vdsc, loader, eqIndex);
105 } else if (auto vdsc = boost::dynamic_pointer_cast<VolatilityApoFutureSurfaceConfig>(vc)) {
106 QL_FAIL("EquityVolCurve: VolatilityApoFutureSurfaceConfig surface not supported for Equities");
107 } else {
108 QL_FAIL("EquityVolCurve: Unexpected VolatilityConfig");
109 }
110 } else {
111 QL_FAIL("EquityVolCurve: VolatilityConfig must be QuoteBased or a Proxy");
112 }
113 // if we've successfully built a surface, save the config and exit the loop
114 volatilityConfig_ = vc;
115 break;
116 }
117 catch (std::exception& e) {
118 DLOG("EquityVolCurve: equity vol curve building failed :" << e.what());
119 }
120 catch (...) {
121 DLOG("EquityVolCurve: equity vol curve building failed: unknown error");
122 }
123 }
124 if (!vol_)
125 QL_FAIL("EquityVolCurve: Failed to build equity volatility structure from " << config.volatilityConfig().size() << " volatility configs provided.");
126 LOG("EquityVolCurve: finished building equity volatility structure with ID " << spec.curveConfigID());
127
128 if (buildCalibrationInfo) {
129 this->buildCalibrationInfo(asof, curveConfigs, config, eqIndex);
130 }
131
132 } catch (exception& e) {
133 QL_FAIL("Equity volatility curve building failed : " << e.what());
134 } catch (...) {
135 QL_FAIL("Equity volatility curve building failed: unknown error");
136 }
137}

139void EquityVolCurve::buildVolatility(const Date& asof, const EquityVolatilityCurveConfig& vc,
140 const ConstantVolatilityConfig& cvc, const Loader& loader) {
141 DLOG("EquityVolCurve: start building constant volatility structure");
142
143 QL_REQUIRE(cvc.quoteType() == MarketDatum::QuoteType::RATE_LNVOL ||
144 cvc.quoteType() == MarketDatum::QuoteType::RATE_SLNVOL ||
145 cvc.quoteType() == MarketDatum::QuoteType::RATE_NVOL,
146 "Quote for Equity Constant Volatility Config must be a Volatility");
147
148 const boost::shared_ptr<MarketDatum>& md = loader.get(cvc.quote(), asof);
149 QL_REQUIRE(md->asofDate() == asof, "MarketDatum asofDate '" << md->asofDate() << "' <> asof '" << asof << "'");
150 QL_REQUIRE(md->instrumentType() == MarketDatum::InstrumentType::EQUITY_OPTION,
151 "MarketDatum instrument type '" << md->instrumentType() << "' <> 'MarketDatum::InstrumentType::EQUITY_OPTION'");
152 boost::shared_ptr<EquityOptionQuote> q = boost::dynamic_pointer_cast<EquityOptionQuote>(md);
153 QL_REQUIRE(q, "Internal error: could not downcast MarketDatum '" << md->name() << "' to EquityOptionQuote");
154 QL_REQUIRE(q->name() == cvc.quote(),
155 "EquityOptionQuote name '" << q->name() << "' <> ConstantVolatilityConfig quote '" << cvc.quote() << "'");
156 TLOG("Found the constant volatility quote " << q->name());
157 // convert quote from minor to major currency if needed
158 Real quoteValue = convertMinorToMajorCurrency(q->ccy(), q->quote()->value());
159
160 DLOG("Creating BlackConstantVol structure");
161 vol_ = boost::make_shared<BlackConstantVol>(asof, calendar_, quoteValue, dayCounter_);
162
163 DLOG("EquityVolCurve: finished building constant volatility structure");
164}

166void EquityVolCurve::buildVolatility(const Date& asof, const EquityVolatilityCurveConfig& vc,
167 const VolatilityCurveConfig& vcc, const Loader& loader) {
168
169 DLOG("EquityVolCurve: start building 1-D volatility curve");
170
171 QL_REQUIRE(vcc.quoteType() == MarketDatum::QuoteType::RATE_LNVOL ||
172 vcc.quoteType() == MarketDatum::QuoteType::RATE_SLNVOL ||
173 vcc.quoteType() == MarketDatum::QuoteType::RATE_NVOL,
174 "Quote for Equity Constant Volatility Config must be a Volatility");
175
176 // Must have at least one quote
177 QL_REQUIRE(vcc.quotes().size() > 0, "No quotes specified in config " << vc.curveID());
178
179 // Check if we are using a regular expression to select the quotes for the curve. If we are, the quotes should
180 // contain exactly one element.
181 auto wildcard = getUniqueWildcard(vcc.quotes());
182
183 // curveData will be populated with the expiry dates and volatility values.
184 map<Date, Real> curveData;
185
186 // Different approaches depending on whether we are using a regex or searching for a list of explicit quotes.
187 if (wildcard) {
188 DLOG("Have single quote with pattern " << wildcard->pattern());
189
190 // Loop over quotes and process equity option quotes matching pattern on asof
191 for (const auto& md : loader.get(*wildcard, asof)) {
192
193 QL_REQUIRE(md->asofDate() == asof, "MarketDatum asofDate '" << md->asofDate() << "' <> asof '" << asof << "'");
194
195 auto q = boost::dynamic_pointer_cast<EquityOptionQuote>(md);
196 QL_REQUIRE(q, "Internal error: could not downcast MarketDatum '" << md->name() << "' to EquityOptionQuote");
197 QL_REQUIRE(q->quoteType() == vcc.quoteType(),
198 "EquityOptionQuote type '" << q->quoteType() << "' <> VolatilityCurveConfig quote type '" << vcc.quoteType() << "'");
199
200 TLOG("The quote " << q->name() << " matched the pattern");
201
202 Date expiryDate = getDateFromDateOrPeriod(q->expiry(), asof, calendar_);
203 if (expiryDate > asof) {
204 // Add the quote to the curve data
205 QL_REQUIRE(curveData.count(expiryDate) == 0, "Duplicate quote for the expiry date "
206 << io::iso_date(expiryDate)
207 << " provided by equity volatility config "
208 << vc.curveID());
209 // convert quote from minor to major currency if needed
210 curveData[expiryDate] = convertMinorToMajorCurrency(q->ccy(), q->quote()->value());
211
212 TLOG("Added quote " << q->name() << ": (" << io::iso_date(expiryDate) << "," << fixed
213 << setprecision(9) << q->quote()->value() << ")");
214 }
215 }
216 // Check that we have quotes in the end
217 QL_REQUIRE(curveData.size() > 0, "No quotes found matching regular expression " << vcc.quotes()[0]);
218
219 } else {
220
221 DLOG("Have " << vcc.quotes().size() << " explicit quotes");
222
223 Size excludedAlreadyExpired = 0;
224
225 // Loop over quotes and process equity option quotes that are explicitly specified in the config
226 std::ostringstream ss;
227 ss << MarketDatum::InstrumentType::EQUITY_OPTION << "/" << vcc.quoteType() << "/" << vc.curveID() << "/"
228 << vc.ccy() << "/*";
229 Wildcard w(ss.str());
230 for (const auto& md : loader.get(w, asof)) {
231
232 QL_REQUIRE(md->asofDate() == asof, "MarketDatum asofDate '" << md->asofDate() << "' <> asof '" << asof << "'");
233
234 auto q = boost::dynamic_pointer_cast<EquityOptionQuote>(md);
235 QL_REQUIRE(q, "Internal error: could not downcast MarketDatum '" << md->name() << "' to EquityOptionQuote");
236
237 // Find quote name in configured quotes.
238 auto it = find(vcc.quotes().begin(), vcc.quotes().end(), q->name());
239
240 if (it != vcc.quotes().end()) {
241 TLOG("Found the configured quote " << q->name());
242
243 Date expiryDate = getDateFromDateOrPeriod(q->expiry(), asof, calendar_);
244 if (expiryDate <= asof) {
245 LOG("Warning Stale Marketdata: Equity volatility quote '"
246 << q->name() << "' has expired in the past (" << io::iso_date(expiryDate) << ") and is ignored");
247 ++excludedAlreadyExpired;
248 continue;
249 }
250 QL_REQUIRE(curveData.count(expiryDate) == 0, "Duplicate quote for the date "
251 << io::iso_date(expiryDate)
252 << " provided by equity volatility config "
253 << vc.curveID());
254 // convert quote from minor to major currency if needed
255 curveData[expiryDate] = convertMinorToMajorCurrency(q->ccy(), q->quote()->value());
256
257 TLOG("Added quote " << q->name() << ": (" << io::iso_date(expiryDate) << "," << fixed
258 << setprecision(9) << q->quote()->value() << ")");
259 }
260 }
261 QL_REQUIRE(curveData.size() > 0, "No 'live' quotes found");
262 // Check that we have found all of the explicitly configured quotes
263 QL_REQUIRE((curveData.size() - excludedAlreadyExpired) == vcc.quotes().size(),
264 "Found " << curveData.size() + excludedAlreadyExpired << " quotes, of which "
265 << excludedAlreadyExpired << " has been in the past but " << vcc.quotes().size()
266 << " quotes were given in config.");
267 }
268
269 // Create the dates and volatility vector
270 vector<Date> dates;
271 vector<Volatility> volatilities;
272 for (const auto& datum : curveData) {
273 dates.push_back(datum.first);
274 volatilities.push_back(datum.second);
275 TLOG("Added data point (" << io::iso_date(dates.back()) << "," << fixed << setprecision(9)
276 << volatilities.back() << ")");
277 }
278
279 DLOG("Creating BlackVarianceCurve object.");
280 auto tmp = boost::make_shared<BlackVarianceCurve>(asof, dates, volatilities, dayCounter_);
281
282 // set max expiry date (used in buildCalibrationInfo())
283 if (!dates.empty())
284 maxExpiry_ = dates.back();
285
286 // Set the interpolation.
287 if (vcc.interpolation() == "Linear") {
288 DLOG("Interpolation set to Linear.");
289 } else if (vcc.interpolation() == "Cubic") {
290 DLOG("Setting interpolation to Cubic.");
291 tmp->setInterpolation<Cubic>();
292 } else if (vcc.interpolation() == "LogLinear") {
293 DLOG("Setting interpolation to LogLinear.");
294 tmp->setInterpolation<LogLinear>();
295 } else {
296 DLOG("Interpolation " << vcc.interpolation() << " not recognised so leaving it Linear.");
297 }
298
299 // Set the volatility_ member after we have possibly updated the interpolation.
300 vol_ = tmp;
301
302 // Set the extrapolation
303 if (parseExtrapolation(vcc.extrapolation()) == Extrapolation::Flat) {
304 DLOG("Enabling BlackVarianceCurve flat volatility extrapolation.");
305 vol_->enableExtrapolation();
306 } else if (parseExtrapolation(vcc.extrapolation()) == Extrapolation::None) {
307 DLOG("Disabling BlackVarianceCurve extrapolation.");
308 vol_->disableExtrapolation();
309 } else if (parseExtrapolation(vcc.extrapolation()) == Extrapolation::UseInterpolator) {
310 DLOG("BlackVarianceCurve does not support using interpolator for extrapolation "
311 << "so default to flat volatility extrapolation.");
312 vol_->enableExtrapolation();
313 } else {
314 DLOG("Unexpected extrapolation so default to flat volatility extrapolation.");
315 vol_->enableExtrapolation();
316 }
317
318 DLOG("EquityVolCurve: finished building 1-D volatility curve");
319}

321void EquityVolCurve::buildVolatility(const Date& asof, EquityVolatilityCurveConfig& vc,
322 const VolatilityStrikeSurfaceConfig& vssc, const Loader& loader,
323 const QuantLib::Handle<EquityIndex2>& eqIndex) {
324
325 DLOG("EquityVolCurve: start building 2-D strike volatility surface");
326
327 QL_REQUIRE(vssc.expiries().size() > 0, "No expiries defined");
328 QL_REQUIRE(vssc.strikes().size() > 0, "No strikes defined");
329
330 // check for wild cards
331 bool expiriesWc = find(vssc.expiries().begin(), vssc.expiries().end(), "*") != vssc.expiries().end();
332 bool strikesWc = find(vssc.strikes().begin(), vssc.strikes().end(), "*") != vssc.strikes().end();
333 if (expiriesWc) {
334 QL_REQUIRE(vssc.expiries().size() == 1, "Wild card expiry specified but more expiries also specified.");
335 }
336 if (strikesWc) {
337 QL_REQUIRE(vssc.strikes().size() == 1, "Wild card strike specified but more strikes also specified.");
338 }
339 bool wildcard = strikesWc || expiriesWc;
340
341 vector<Real> callStrikes, putStrikes;
342 vector<Real> callData, putData;
343 vector<Date> callExpiries, putExpiries;
344
345 // In case of wild card we need the following granularity within the mkt data loop
346 bool strikeRelevant = strikesWc;
347 bool expiryRelevant = expiriesWc;
348 bool quoteRelevant = true;
349
350 // We loop over all market data, looking for quotes that match the configuration
351 Size callQuotesAdded = 0;
352 Size putQuotesAdded = 0;
353 Size excludedAlreadyExpired = 0;
354 std::ostringstream ss;
355 ss << MarketDatum::InstrumentType::EQUITY_OPTION << "/" << vssc.quoteType() << "/" << vc.curveID() << "/"
356 << vc.ccy() << "/*";
357 Wildcard w(ss.str());
358 for (const auto& md : loader.get(w, asof)) {
359 QL_REQUIRE(md->asofDate() == asof, "MarketDatum asofDate '" << md->asofDate() << "' <> asof '" << asof << "'");
360 boost::shared_ptr<EquityOptionQuote> q = boost::dynamic_pointer_cast<EquityOptionQuote>(md);
361 QL_REQUIRE(q, "Internal error: could not downcast MarketDatum '" << md->name() << "' to EquityOptionQuote");
362 // todo - for now we will ignore ATM, ATMF quotes both for explicit strikes and in case of strike wild
363 // card. ----
364 auto absoluteStrike = boost::dynamic_pointer_cast<AbsoluteStrike>(q->strike());
365 if (!absoluteStrike)
366 continue;
367 if (!expiriesWc) {
368 auto j = std::find(vssc.expiries().begin(), vssc.expiries().end(), q->expiry());
369 expiryRelevant = j != vssc.expiries().end();
370 }
371 if (!strikesWc) {
372 auto i = std::find_if(vssc.strikes().begin(), vssc.strikes().end(),
373 [&absoluteStrike](const std::string& x) {
374 return close_enough(parseReal(x), absoluteStrike->strike());
375 });
376 strikeRelevant = i != vssc.strikes().end();
377 }
378 quoteRelevant = strikeRelevant && expiryRelevant;
379
380 // add quote to vectors, if relevant
381 // If a quote doesn't include a call/put flag (an Implied Vol for example), it
382 // defaults to a call. For an explicit surface we expect either a call and put
383 // for every point, or just a vol at every point
384 if (quoteRelevant) {
385 Date tmpDate = getDateFromDateOrPeriod(q->expiry(), asof, calendar_);
386 if (tmpDate <= asof) {
387 LOG("expired Equity volatility quote '"
388 << q->name() << "' ignored, expired on (" << io::iso_date(tmpDate)
389 << ")");
390 ++excludedAlreadyExpired;
391 continue;
392 }
393 // get values and strikes, convert from minor to major currency if needed
394 Real quoteValue = q->quote()->value();
395 if (vssc.quoteType() == MarketDatum::QuoteType::PRICE)
396 quoteValue = convertMinorToMajorCurrency(q->ccy(), quoteValue);
397 Real strikeValue = convertMinorToMajorCurrency(q->ccy(), absoluteStrike->strike());
398
399 if (q->isCall()) {
400 callStrikes.push_back(strikeValue);
401 callData.push_back(quoteValue);
402 callExpiries.push_back(tmpDate);
403 callQuotesAdded++;
404 } else {
405 putStrikes.push_back(strikeValue);
406 putData.push_back(quoteValue);
407 putExpiries.push_back(tmpDate);
408 putQuotesAdded++;
409 }
410 }
411 }
412
413 QL_REQUIRE(callQuotesAdded > 0, "No valid equity volatility quotes provided");
414 bool callSurfaceOnly = false;
415 if (callQuotesAdded > 0 && putQuotesAdded == 0) {
416 QL_REQUIRE(vssc.quoteType() != MarketDatum::QuoteType::PRICE,
417 "For Premium quotes, call and put quotes must be supplied.");
418 DLOG("EquityVolCurve: " << vc.curveID() << ": Only one set of quotes, can build surface directly");
419 callSurfaceOnly = true;
420 }
421 // Check loaded quotes
422 if (!wildcard) {
423 Size explicitGridSize = vssc.expiries().size() * vssc.strikes().size();
424
425 QL_REQUIRE(callQuotesAdded + excludedAlreadyExpired == explicitGridSize,
426 "EquityVolCurve: " << vc.curveID() << ": " << callQuotesAdded +excludedAlreadyExpired
427 << " quotes provided, of which " << excludedAlreadyExpired << "have been excluded, but "
428 << explicitGridSize << " expected.");
429 if (!callSurfaceOnly) {
430 QL_REQUIRE(callQuotesAdded == putQuotesAdded,
431 "Call and Put quotes must match for explicitly defined surface, "
432 << callQuotesAdded << " call quotes, and " << putQuotesAdded << " put quotes");
433 DLOG("EquityVolCurve: " << vc.curveID() << ": Complete set of " << callQuotesAdded
434 << ", call and put quotes found.");
435 }
436 }
437
438 QL_REQUIRE(callStrikes.size() == callData.size() && callData.size() == callExpiries.size(),
439 "Quotes loaded don't produce strike,vol,expiry vectors of equal length.");
440 QL_REQUIRE(putStrikes.size() == putData.size() && putData.size() == putExpiries.size(),
441 "Quotes loaded don't produce strike,vol,expiry vectors of equal length.");
442 DLOG("EquityVolCurve: " << vc.curveID() << ": Found " << callQuotesAdded << ", call quotes and "
443 << putQuotesAdded << " put quotes using wildcard.");
444
445 // Set the strike extrapolation which only matters if extrapolation is turned on for the whole surface.
446 bool flatStrikeExtrap = true;
447 bool flatTimeExtrap = true;
448 if (vssc.extrapolation()) {
449
450 auto strikeExtrapType = parseExtrapolation(vssc.strikeExtrapolation());
451 if (strikeExtrapType == Extrapolation::UseInterpolator) {
452 TLOG("EquityVolCurve: Strike extrapolation switched to using interpolator.");
453 flatStrikeExtrap = false;
454 } else if (strikeExtrapType == Extrapolation::None) {
455 TLOG("EquityVolCurve: Strike extrapolation cannot be turned off on its own so defaulting to flat.");
456 } else if (strikeExtrapType == Extrapolation::Flat) {
457 TLOG("EquityVolCurve: Strike extrapolation has been set to flat.");
458 } else {
459 TLOG("EquityVolCurve: Strike extrapolation " << strikeExtrapType << " not expected so default to flat.");
460 }
461
462 auto timeExtrapType = parseExtrapolation(vssc.timeExtrapolation());
463 if (timeExtrapType == Extrapolation::UseInterpolator) {
464 TLOG("EquityVolCurve: Time extrapolation switched to using interpolator.");
465 flatTimeExtrap = false;
466 } else if (timeExtrapType == Extrapolation::None) {
467 TLOG("EquityVolCurve: Time extrapolation cannot be turned off on its own so defaulting to flat.");
468 } else if (timeExtrapType == Extrapolation::Flat) {
469 TLOG("EquityVolCurve: Time extrapolation has been set to flat.");
470 } else {
471 TLOG("EquityVolCurve: Time extrapolation " << timeExtrapType << " not expected so default to flat.");
472 }
473
474 } else {
475 TLOG("EquityVolCurve: Extrapolation is turned off for the whole surface so the time and"
476 << " strike extrapolation settings are ignored");
477 }
478
479 // set max expiry date (used in buildCalibrationInfo())
480 maxExpiry_ = Date::minDate();
481 for (auto const& d : callExpiries)
482 maxExpiry_ = std::max(maxExpiry_, d);
483 for (auto const& d : putExpiries)
484 maxExpiry_ = std::max(maxExpiry_, d);
485 if (maxExpiry_ == Date::minDate())
486 maxExpiry_ = Date();
487
488 bool preferOutOfTheMoney = vc.preferOutOfTheMoney() ? *vc.preferOutOfTheMoney() : true;
489
490 if (vssc.quoteType() == MarketDatum::QuoteType::PRICE) {
491
492 // Create the 1D solver options used in the price stripping.
493 Solver1DOptions solverOptions = vc.solverConfig();
494
495 DLOG("EquityVolCurve: Building a option price surface for calls and puts");
496 boost::shared_ptr<OptionPriceSurface> callSurface =
497 boost::make_shared<OptionPriceSurface>(asof, callExpiries, callStrikes, callData, dayCounter_);
498 boost::shared_ptr<OptionPriceSurface> putSurface =
499 boost::make_shared<OptionPriceSurface>(asof, putExpiries, putStrikes, putData, dayCounter_);
500
501 DLOG("EquityVolCurve: CallSurface contains " << callSurface->expiries().size() << " expiries.");
502
503 DLOG("EquityVolCurve: Stripping equity volatility surface from the option premium surfaces");
504 boost::shared_ptr<EquityOptionSurfaceStripper> eoss = boost::make_shared<EquityOptionSurfaceStripper>(
505 eqIndex, callSurface, putSurface, calendar_, dayCounter_, vssc.exerciseType(), flatStrikeExtrap,
506 flatStrikeExtrap, flatTimeExtrap, preferOutOfTheMoney, solverOptions);
507 vol_ = eoss->volSurface();
508
509 } else if (vssc.quoteType() == MarketDatum::QuoteType::RATE_LNVOL) {
510
511 if (callExpiries.size() == 1 && callStrikes.size() == 1) {
512 DLOG("EquityVolCurve: Building BlackConstantVol");
513 vol_ = boost::shared_ptr<BlackVolTermStructure>(
514 new BlackConstantVol(asof, Calendar(), callData[0], dayCounter_));
515 } else {
516 // create a vol surface from the calls
517 boost::shared_ptr<BlackVarianceSurfaceSparse> callSurface =
518 boost::make_shared<BlackVarianceSurfaceSparse>(asof, calendar_, callExpiries, callStrikes, callData,
519 dayCounter_, flatStrikeExtrap, flatStrikeExtrap,
520 flatTimeExtrap);
521
522 if (callSurfaceOnly) {
523 // if only a call surface provided use that
524 vol_ = callSurface;
525 } else {
526 // otherwise create a vol surface from puts and strip for a final surface
527 boost::shared_ptr<BlackVarianceSurfaceSparse> putSurface =
528 boost::make_shared<BlackVarianceSurfaceSparse>(asof, calendar_, putExpiries, putStrikes,
529 putData, dayCounter_, flatStrikeExtrap,
530 flatStrikeExtrap, flatTimeExtrap);
531
532 boost::shared_ptr<EquityOptionSurfaceStripper> eoss =
533 boost::make_shared<EquityOptionSurfaceStripper>(
534 eqIndex, callSurface, putSurface, calendar_, dayCounter_, Exercise::European,
535 flatStrikeExtrap, flatStrikeExtrap, flatTimeExtrap, preferOutOfTheMoney);
536 vol_ = eoss->volSurface();
537 }
538 }
539 } else {
540 QL_FAIL("EquityVolCurve: Invalid quote type provided.");
541 }
542 DLOG("EquityVolCurve: Setting BlackVarianceSurfaceSparse extrapolation to " << to_string(vssc.extrapolation()));
543 vol_->enableExtrapolation(vssc.extrapolation());
544
545 DLOG("EquityVolCurve: EquityVolCurve: finished building 2-D strike volatility surface");
546}

567void EquityVolCurve::buildVolatility(const Date& asof, EquityVolatilityCurveConfig& vc,
568 const VolatilityMoneynessSurfaceConfig& vmsc, const Loader& loader,
569 const QuantLib::Handle<EquityIndex2>& eqIndex) {
570
571 LOG("EquityVolCurve: start building 2-D volatility moneyness strike surface");
572 using boost::adaptors::transformed;
573 using boost::algorithm::join;
574
575 // Check that the quote type is volatility, we do not support price
576 QL_REQUIRE(vmsc.quoteType() == MarketDatum::QuoteType::RATE_LNVOL,
577 "EquityVolCurve: Equity Moneyness Surface supports lognormal volatility quotes only");
578
579 // Parse, sort and check the vector of configured moneyness levels
580 vector<Real> moneynessLevels = checkMoneyness(vmsc.moneynessLevels());
581
582 // Expiries may be configured with a wildcard or given explicitly
583 bool expWc = false;
584 if (find(vmsc.expiries().begin(), vmsc.expiries().end(), "*") != vmsc.expiries().end()) {
585 expWc = true;
586 QL_REQUIRE(vmsc.expiries().size() == 1, "EquityVolCurve: Wild card expiry specified but more expiries also specified.");
587 DLOG("EquityVolCurve: Have expiry wildcard pattern " << vmsc.expiries()[0]);
588 }
589
590 // Map to hold the rows of the volatility matrix. The keys are the expiry dates and the values are the
591 // vectors of volatilities, one for each configured moneyness.
592 map<Date, vector<Real>> surfaceData;
593
594 // Count the number of quotes added. We check at the end that we have added all configured quotes.
595 Size quotesAdded = 0;
596
597 // Configured moneyness type.
598 MoneynessStrike::Type moneynessType = parseMoneynessType(vmsc.moneynessType());
599
600 // Populate the configured strikes.
601 vector<boost::shared_ptr<BaseStrike>> strikes;
602 for (const auto& moneynessLevel : moneynessLevels) {
603 strikes.push_back(boost::make_shared<MoneynessStrike>(moneynessType, moneynessLevel));
604 }
605
606 // Read the quotes to fill the expiry dates and vols matrix.
607 std::ostringstream ss;
608 ss << MarketDatum::InstrumentType::EQUITY_OPTION << "/" << vmsc.quoteType() << "/" << vc.curveID() << "/"
609 << vc.ccy() << "/*";
610 Wildcard w(ss.str());
611 for (const auto& md : loader.get(w, asof)) {
612
613 QL_REQUIRE(md->asofDate() == asof, "MarketDatum asofDate '" << md->asofDate() << "' <> asof '" << asof << "'");
614
615 auto q = boost::dynamic_pointer_cast<EquityOptionQuote>(md);
616 QL_REQUIRE(q, "Internal error: could not downcast MarketDatum '" << md->name() << "' to EquityOptionQuote");
617 QL_REQUIRE(q->eqName() == vc.curveID(),
618 "EquityOptionQuote eqName '" << q->eqName() << "' <> EquityVolatilityCurveConfig curveID '" << vc.curveID() << "'");
619 QL_REQUIRE(q->ccy() == vc.ccy(),
620 "EquityOptionQuote ccy '" << q->ccy() << "' <> EquityVolatilityCurveConfig ccy '" << vc.ccy() << "'");
621 QL_REQUIRE(q->quoteType() == vmsc.quoteType(),
622 "EquityOptionQuote quoteType '" << q->quoteType() << "' <> VolatilityMoneynessSurfaceConfig quoteType '" << vmsc.quoteType() << "'");
623
624 // Iterator to one of the configured strikes.
625 vector<boost::shared_ptr<BaseStrike>>::iterator strikeIt;
626
627 if (expWc) {
628 // Check if quote's strike is in the configured strikes and continue if it is not.
629 strikeIt = find_if(strikes.begin(), strikes.end(),
630 [&q](boost::shared_ptr<BaseStrike> s) { return *s == *q->strike(); });
631 if (strikeIt == strikes.end())
632 continue;
633 } else {
634 // If we have explicitly configured expiries and the quote is not in the configured quotes continue.
635 auto it = find(vc.quotes().begin(), vc.quotes().end(), q->name());
636 if (it == vc.quotes().end())
637 continue;
638
639 // Check if quote's strike is in the configured strikes and continue if it is not.
640 strikeIt = find_if(strikes.begin(), strikes.end(),
641 [&q](boost::shared_ptr<BaseStrike> s) { return *s == *q->strike(); });
642 if (strikeIt == strikes.end())
643 continue;
644 }
645
646 // Position of quote in vector of strikes
647 Size pos = std::distance(strikes.begin(), strikeIt);
648
649 // Process the quote
650 Date eDate = getDateFromDateOrPeriod(q->expiry(), asof, calendar_);
651
652 // Add quote to surface
653 if (surfaceData.count(eDate) == 0)
654 surfaceData[eDate] = vector<Real>(moneynessLevels.size(), Null<Real>());
655
656 QL_REQUIRE(surfaceData[eDate][pos] == Null<Real>(),
657 "EquityVolCurve: Quote " << q->name() << " provides a duplicate quote for the date " << io::iso_date(eDate)
658 << " and strike " << *q->strike());
659 surfaceData[eDate][pos] = q->quote()->value();
660 quotesAdded++;
661
662 TLOG("EquityVolCurve: Added quote " << q->name() << ": (" << io::iso_date(eDate) << "," << *q->strike() << "," << fixed
663 << setprecision(9) << "," << q->quote()->value() << ")");
664 }
665
666 DLOG("EquityVolCurve: added " << quotesAdded << " quotes in building moneyness strike surface.");
667
668 // Check the data gathered.
669 if (expWc) {
670 // check we have non-empty surface data
671 QL_REQUIRE(!surfaceData.empty(), "EquityVolCurve: Moneyness Surface Data is empty");
672 // If the expiries were configured via a wildcard, check that no surfaceData element has a Null<Real>().
673 for (const auto& kv : surfaceData) {
674 for (Size j = 0; j < moneynessLevels.size(); j++) {
675 QL_REQUIRE(kv.second[j] != Null<Real>(), "EquityVolCurve: Volatility for expiry date "
676 << io::iso_date(kv.first) << " and strike " << *strikes[j]
677 << " not found. Cannot proceed with a sparse matrix.");
678 }
679 }
680 } else {
681 // If expiries were configured explicitly, the number of configured quotes should equal the
682 // number of quotes added.
683 QL_REQUIRE(vc.quotes().size() == quotesAdded,
684 "EquityVolCurve: Found " << quotesAdded << " quotes, but " << vc.quotes().size() << " quotes required by config.");
685 }
686
687 // Populate the volatility quotes and the expiry times.
688 // Rows are moneyness levels and columns are expiry times - this is what the ctor needs below.
689 vector<Date> expiryDates(surfaceData.size());
690 vector<Time> expiryTimes(surfaceData.size());
691 vector<vector<Handle<Quote>>> vols(moneynessLevels.size());
692 for (const auto row : surfaceData | boost::adaptors::indexed(0)) {
693 expiryDates[row.index()] = row.value().first;
694 expiryTimes[row.index()] = dayCounter_.yearFraction(asof, row.value().first);
695 for (Size i = 0; i < row.value().second.size(); i++) {
696 vols[i].push_back(Handle<Quote>(boost::make_shared<SimpleQuote>(row.value().second[i])));
697 }
698 }
699
700 // set max expiry date (used in buildCalibrationInfo())
701 if (!expiryDates.empty())
702 maxExpiry_ = expiryDates.back();
703
704 // Set the strike extrapolation which only matters if extrapolation is turned on for the whole surface.
705 // BlackVarianceSurfaceMoneyness time extrapolation is hard-coded to constant in volatility.
706 bool flatExtrapolation = true;
707 if (vmsc.extrapolation()) {
708
709 auto strikeExtrapType = parseExtrapolation(vmsc.strikeExtrapolation());
710 if (strikeExtrapType == Extrapolation::UseInterpolator) {
711 TLOG("EquityVolCurve: Strike extrapolation switched to using interpolator.");
712 flatExtrapolation = false;
713 } else if (strikeExtrapType == Extrapolation::None) {
714 TLOG("EquityVolCurve: Strike extrapolation cannot be turned off on its own so defaulting to flat.");
715 } else if (strikeExtrapType == Extrapolation::Flat) {
716 TLOG("EquityVolCurve: Strike extrapolation has been set to flat.");
717 } else {
718 TLOG("EquityVolCurve: Strike extrapolation " << strikeExtrapType << " not expected so default to flat.");
719 }
720
721 auto timeExtrapType = parseExtrapolation(vmsc.timeExtrapolation());
722 if (timeExtrapType != Extrapolation::Flat) {
723 TLOG("EquityVolCurve: BlackVarianceSurfaceMoneyness only supports flat volatility extrapolation in the time direction");
724 }
725 } else {
726 TLOG("EquityVolCurve: Extrapolation is turned off for the whole surface so the time and"
727 << " strike extrapolation settings are ignored");
728 }
729
730 // Time interpolation
731 if (vmsc.timeInterpolation() != "Linear") {
732 TLOG("EquityVolCurve: BlackVarianceSurfaceMoneyness only supports linear time interpolation in variance.");
733 }
734
735 // Strike interpolation
736 if (vmsc.strikeInterpolation() != "Linear") {
737 TLOG("EquityVolCurve: BlackVarianceSurfaceMoneyness only supports linear strike interpolation in variance.");
738 }
739
740 // Both moneyness surfaces need a spot quote.
741
742 // The choice of false here is important for forward moneyness. It means that we use the cpts and yts in the
743 // BlackVarianceSurfaceMoneynessForward to get the forward value at all times and in particular at times that
744 // are after the last expiry time. If we set it to true, BlackVarianceSurfaceMoneynessForward uses a linear
745 // interpolated forward curve on the expiry times internally which is poor.
746 bool stickyStrike = false;
747
748 if (moneynessType == MoneynessStrike::Type::Forward) {
749 DLOG("EquityVolCurve: Creating BlackVarianceSurfaceMoneynessForward object");
750 vol_ = boost::make_shared<BlackVarianceSurfaceMoneynessForward>(
751 calendar_, eqIndex->equitySpot(), expiryTimes, moneynessLevels, vols, dayCounter_,
752 eqIndex->equityDividendCurve(), eqIndex->equityForecastCurve(), stickyStrike, flatExtrapolation);
753
754 } else {
755 DLOG("EquityVolCurve: Creating BlackVarianceSurfaceMoneynessSpot object");
756 vol_ = boost::make_shared<BlackVarianceSurfaceMoneynessSpot>(calendar_, eqIndex->equitySpot(), expiryTimes,
757 moneynessLevels, vols, dayCounter_, stickyStrike,
758 flatExtrapolation);
759 }
760
761 DLOG("EquityVolCurve: Setting BlackVarianceSurfaceMoneyness extrapolation to " << to_string(vmsc.extrapolation()));
762 vol_->enableExtrapolation(vmsc.extrapolation());
763
764 DLOG("EquityVolCurve: EquityVolCurve: finished building 2-D volatility moneyness strike surface");
765}

767void EquityVolCurve::buildVolatility(const QuantLib::Date& asof, EquityVolatilityCurveConfig& vc,
768 const VolatilityDeltaSurfaceConfig& vdsc, const Loader& loader,
769 const QuantLib::Handle<QuantExt::EquityIndex2>& eqIndex) {
770
771 using boost::adaptors::transformed;
772 using boost::algorithm::join;
773
774 DLOG("EquityVolCurve: start building 2-D volatility delta strike surface");
775
776 QL_REQUIRE(vdsc.quoteType() == MarketDatum::QuoteType::RATE_LNVOL,
777 "EquityVolCurve: only quote type"
778 << " RATE_LNVOL is currently supported for a 2-D volatility delta strike surface.");
779
780 // Parse, sort and check the vector of configured put deltas
781 vector<Real> putDeltas = parseVectorOfValues<Real>(vdsc.putDeltas(), &parseReal);
782 sort(putDeltas.begin(), putDeltas.end(), [](Real x, Real y) { return !close(x, y) && x < y; });
783 QL_REQUIRE(adjacent_find(putDeltas.begin(), putDeltas.end(), [](Real x, Real y) { return close(x, y); }) ==
784 putDeltas.end(),
785 "EquityVolCurve: The configured put deltas contain duplicates");
786 DLOG("EquityVolCurve: Parsed " << putDeltas.size() << " unique configured put deltas");
787 DLOG("EquityVolCurve: Put deltas are: " << join(putDeltas | transformed([](Real d) { return ore::data::to_string(d); }), ","));
788
789 // Parse, sort descending and check the vector of configured call deltas
790 vector<Real> callDeltas = parseVectorOfValues<Real>(vdsc.callDeltas(), &parseReal);
791 sort(callDeltas.begin(), callDeltas.end(), [](Real x, Real y) { return !close(x, y) && x > y; });
792 QL_REQUIRE(adjacent_find(callDeltas.begin(), callDeltas.end(), [](Real x, Real y) { return close(x, y); }) ==
793 callDeltas.end(),
794 "EquityVolCurve: The configured call deltas contain duplicates");
795 DLOG("EquityVolCurve: Parsed " << callDeltas.size() << " unique configured call deltas");
796 DLOG("EquityVolCurve: Call deltas are: " << join(callDeltas | transformed([](Real d) { return ore::data::to_string(d); }), ","));
797
798 // Expiries may be configured with a wildcard or given explicitly
799 bool expWc = false;
800 if (find(vdsc.expiries().begin(), vdsc.expiries().end(), "*") != vdsc.expiries().end()) {
801 expWc = true;
802 QL_REQUIRE(vdsc.expiries().size() == 1, "Wild card expiry specified but more expiries also specified.");
803 DLOG("EquityVolCurve: Have expiry wildcard pattern " << vdsc.expiries()[0]);
804 }
805
806 // Map to hold the rows of the equity volatility matrix. The keys are the expiry dates and the values are the
807 // vectors of volatilities, one for each configured delta.
808 map<Date, vector<Real>> surfaceData;
809
810 // Number of strikes = number of put deltas + ATM + number of call deltas
811 Size numStrikes = putDeltas.size() + 1 + callDeltas.size();
812
813 // Count the number of quotes added. We check at the end that we have added all configured quotes.
814 Size quotesAdded = 0;
815
816 // Configured delta and Atm types.
817 DeltaVolQuote::DeltaType deltaType = parseDeltaType(vdsc.deltaType());
818 DeltaVolQuote::AtmType atmType = parseAtmType(vdsc.atmType());
819 boost::optional<DeltaVolQuote::DeltaType> atmDeltaType;
820 if (!vdsc.atmDeltaType().empty()) {
821 atmDeltaType = parseDeltaType(vdsc.atmDeltaType());
822 }
823
824 // Populate the configured strikes.
825 vector<boost::shared_ptr<BaseStrike>> strikes;
826 for (const auto& pd : putDeltas) {
827 strikes.push_back(boost::make_shared<DeltaStrike>(deltaType, Option::Put, pd));
828 }
829 strikes.push_back(boost::make_shared<AtmStrike>(atmType, atmDeltaType));
830 for (const auto& cd : callDeltas) {
831 strikes.push_back(boost::make_shared<DeltaStrike>(deltaType, Option::Call, cd));
832 }
833
834 // Read the quotes to fill the expiry dates and vols matrix.
835 std::ostringstream ss;
836 ss << MarketDatum::InstrumentType::EQUITY_OPTION << "/" << vdsc.quoteType() << "/" << vc.curveID() << "/"
837 << vc.ccy() << "/*";
838 Wildcard w(ss.str());
839 for (const auto& md : loader.get(w, asof)) {
840
841 QL_REQUIRE(md->asofDate() == asof, "MarketDatum asofDate '" << md->asofDate() << "' <> asof '" << asof << "'");
842
843 auto q = boost::dynamic_pointer_cast<EquityOptionQuote>(md);
844 QL_REQUIRE(q, "Internal error: could not downcast MarketDatum '" << md->name() << "' to EquityOptionQuote");
845 QL_REQUIRE(q->eqName() == vc.curveID(),
846 "EquityOptionQuote eqName '" << q->eqName() << "' <> EquityVolatilityCurveConfig curveID '" << vc.curveID() << "'");
847 QL_REQUIRE(q->ccy() == vc.ccy(),
848 "EquityOptionQuote ccy '" << q->ccy() << "' <> EquityVolatilityCurveConfig ccy '" << vc.ccy() << "'");
849 QL_REQUIRE(q->quoteType() == vdsc.quoteType(),
850 "EquityOptionQuote quoteType '" << q->quoteType() << "' <> VolatilityMoneynessSurfaceConfig quoteType '" << vdsc.quoteType() << "'");
851
852 // Iterator to one of the configured strikes.
853 vector<boost::shared_ptr<BaseStrike>>::iterator strikeIt;
854
855 if (expWc) {
856 // Check if quote's strike is in the configured strikes and continue if it is not.
857 strikeIt = find_if(strikes.begin(), strikes.end(),
858 [&q](boost::shared_ptr<BaseStrike> s) { return *s == *q->strike(); });
859 if (strikeIt == strikes.end())
860 continue;
861 } else {
862 // If we have explicitly configured expiries and the quote is not in the configured quotes continue.
863 auto it = find(vc.quotes().begin(), vc.quotes().end(), q->name());
864 if (it == vc.quotes().end())
865 continue;
866
867 // Check if quote's strike is in the configured strikes.
868 // It should be as we have selected from the explicitly configured quotes in the last step.
869 strikeIt = find_if(strikes.begin(), strikes.end(),
870 [&q](boost::shared_ptr<BaseStrike> s) { return *s == *q->strike(); });
871 QL_REQUIRE(strikeIt != strikes.end(),
872 "EquityVolCurve: The quote '"
873 << q->name()
874 << "' is in the list of configured quotes but does not match any of the configured strikes");
875 }
876
877 // Position of quote in vector of strikes
878 Size pos = std::distance(strikes.begin(), strikeIt);
879
880 // Process the quote
881 Date eDate;
882 boost::shared_ptr<Expiry> expiry = parseExpiry(q->expiry());
883 if (auto expiryDate = boost::dynamic_pointer_cast<ExpiryDate>(expiry)) {
884 eDate = expiryDate->expiryDate();
885 } else if (auto expiryPeriod = boost::dynamic_pointer_cast<ExpiryPeriod>(expiry)) {
886 // We may need more conventions here eventually.
887 eDate = calendar_.adjust(asof + expiryPeriod->expiryPeriod());
888 }
889
890 // Add quote to surface
891 if (surfaceData.count(eDate) == 0)
892 surfaceData[eDate] = vector<Real>(numStrikes, Null<Real>());
893
894 QL_REQUIRE(surfaceData[eDate][pos] == Null<Real>(),
895 "EquityVolCurve: Quote " << q->name() << " provides a duplicate quote for the date " << io::iso_date(eDate)
896 << " and strike " << *q->strike());
897 surfaceData[eDate][pos] = q->quote()->value();
898 quotesAdded++;
899
900 TLOG("EquityVolCurve: Added quote " << q->name() << ": (" << io::iso_date(eDate) << "," << *q->strike() << "," << fixed
901 << setprecision(9) << "," << q->quote()->value() << ")");
902 }
903
904 DLOG("EquityVolCurve: EquityVolCurve: added " << quotesAdded << " quotes in building delta strike surface.");
905
906 // Check the data gathered.
907 if (expWc) {
908 // If the expiries were configured via a wildcard, check that no surfaceData element has a Null<Real>().
909 for (const auto& kv : surfaceData) {
910 for (Size j = 0; j < numStrikes; j++) {
911 QL_REQUIRE(kv.second[j] != Null<Real>(), "EquityVolCurve: Volatility for expiry date "
912 << io::iso_date(kv.first) << " and strike " << *strikes[j]
913 << " not found. Cannot proceed with a sparse matrix.");
914 }
915 }
916 } else {
917 // If expiries were configured explicitly, the number of configured quotes should equal the
918 // number of quotes added.
919 QL_REQUIRE(vc.quotes().size() == quotesAdded,
920 "EquityVolCurve: Found " << quotesAdded << " quotes, but " << vc.quotes().size() << " quotes required by config.");
921 }
922
923 // Populate the matrix of volatilities and the expiry dates.
924 vector<Date> expiryDates;
925 Matrix vols(surfaceData.size(), numStrikes);
926 for (const auto row : surfaceData | boost::adaptors::indexed(0)) {
927 expiryDates.push_back(row.value().first);
928 copy(row.value().second.begin(), row.value().second.end(), vols.row_begin(row.index()));
929 }
930
931 // Need to multiply each put delta value by -1 before passing it to the BlackVolatilitySurfaceDelta ctor
932 // i.e. a put delta of 0.25 that is passed in to the config must be -0.25 when passed to the ctor.
933 transform(putDeltas.begin(), putDeltas.end(), putDeltas.begin(), [](Real pd) { return -1.0 * pd; });
934 DLOG("EquityVolCurve: Multiply put deltas by -1.0 before creating BlackVolatilitySurfaceDelta object.");
935 DLOG("EquityVolCurve: Put deltas are: " << join(putDeltas | transformed([](Real d) { return ore::data::to_string(d); }), ","));
936
937 // Set the strike extrapolation which only matters if extrapolation is turned on for the whole surface.
938 // BlackVolatilitySurfaceDelta time extrapolation is hard-coded to constant in volatility.
939 bool flatExtrapolation = true;
940 if (vdsc.extrapolation()) {
941
942 auto strikeExtrapType = parseExtrapolation(vdsc.strikeExtrapolation());
943 if (strikeExtrapType == Extrapolation::UseInterpolator) {
944 TLOG("EquityVolCurve: Strike extrapolation switched to using interpolator.");
945 flatExtrapolation = false;
946 } else if (strikeExtrapType == Extrapolation::None) {
947 TLOG("EquityVolCurve: Strike extrapolation cannot be turned off on its own so defaulting to flat.");
948 } else if (strikeExtrapType == Extrapolation::Flat) {
949 TLOG("EquityVolCurve: Strike extrapolation has been set to flat.");
950 } else {
951 TLOG("EquityVolCurve: Strike extrapolation " << strikeExtrapType << " not expected so default to flat.");
952 }
953
954 auto timeExtrapType = parseExtrapolation(vdsc.timeExtrapolation());
955 if (timeExtrapType != Extrapolation::Flat) {
956 TLOG("EquityVolCurve: BlackVolatilitySurfaceDelta only supports flat volatility extrapolation in the time direction");
957 }
958 } else {
959 TLOG("EquityVolCurve: Extrapolation is turned off for the whole surface so the time and"
960 << " strike extrapolation settings are ignored");
961 }
962
963 // Time interpolation
964 if (vdsc.timeInterpolation() != "Linear") {
965 TLOG("EquityVolCurve: BlackVolatilitySurfaceDelta only supports linear time interpolation.");
966 }
967
968 // Strike interpolation
969 InterpolatedSmileSection::InterpolationMethod im;
970 if (vdsc.strikeInterpolation() == "Linear") {
971 im = InterpolatedSmileSection::InterpolationMethod::Linear;
972 } else if (vdsc.strikeInterpolation() == "NaturalCubic") {
973 im = InterpolatedSmileSection::InterpolationMethod::NaturalCubic;
974 } else if (vdsc.strikeInterpolation() == "FinancialCubic") {
975 im = InterpolatedSmileSection::InterpolationMethod::FinancialCubic;
976 } else if (vdsc.strikeInterpolation() == "CubicSpline") {
977 im = InterpolatedSmileSection::InterpolationMethod::CubicSpline;
978 } else {
979 im = InterpolatedSmileSection::InterpolationMethod::Linear;
980 DLOG("EquityVolCurve: BlackVolatilitySurfaceDelta does not support strike interpolation '" << vdsc.strikeInterpolation()
981 << "' so setting it to linear.");
982 }
983
984 // set max expiry date (used in buildCalibrationInfo())
985 if (!expiryDates.empty())
986 maxExpiry_ = expiryDates.back();
987
988 DLOG("EquityVolCurve: Creating BlackVolatilitySurfaceDelta object");
989 bool hasAtm = true;
990 vol_ = boost::make_shared<BlackVolatilitySurfaceDelta>(
991 asof, expiryDates, putDeltas, callDeltas, hasAtm, vols, dayCounter_, calendar_, eqIndex->equitySpot(),
992 eqIndex->equityForecastCurve(), eqIndex->equityDividendCurve(), deltaType, atmType, atmDeltaType, 0 * Days,
993 deltaType, atmType, atmDeltaType, im, flatExtrapolation);
994

1078 const EquityVolatilityCurveConfig& config,
1079 const Handle<EquityIndex2>& eqIndex) {
1080
1081 DLOG("EquityVolCurve: Building calibration info for eq vol surface");
1082
1083 try {
1084
1085 ReportConfig rc = effectiveReportConfig(curveConfigs.reportConfigEqVols(), config.reportConfig());
1086
1087 bool reportOnDeltaGrid = *rc.reportOnDeltaGrid();
1088 bool reportOnMoneynessGrid = *rc.reportOnMoneynessGrid();

1090 std::vector<std::string> deltas = *rc.deltas();
1091 std::vector<Period> expiries = *rc.expiries();
1092
1093 calibrationInfo_ = boost::make_shared<FxEqCommVolCalibrationInfo>();
1094
1095 DeltaVolQuote::AtmType atmType = DeltaVolQuote::AtmType::AtmDeltaNeutral;
1096 DeltaVolQuote::DeltaType deltaType = DeltaVolQuote::DeltaType::Fwd;
1097
1098 if (auto vdsc = boost::dynamic_pointer_cast<VolatilityDeltaSurfaceConfig>(volatilityConfig_)) {
1099 atmType = parseAtmType(vdsc->atmType());
1100 deltaType = parseDeltaType(vdsc->deltaType());
1101 }
1102
1103 calibrationInfo_->dayCounter = config.dayCounter().empty() ? "na" : config.dayCounter();
1104 calibrationInfo_->calendar = config.calendar().empty() ? "na" : config.calendar();
1105 calibrationInfo_->atmType = ore::data::to_string(atmType);
1106 calibrationInfo_->deltaType = ore::data::to_string(deltaType);
1107 calibrationInfo_->longTermAtmType = ore::data::to_string(atmType);
1108 calibrationInfo_->longTermDeltaType = ore::data::to_string(deltaType);
1109 calibrationInfo_->switchTenor = "na";
1110 calibrationInfo_->riskReversalInFavorOf = "na";
1111 calibrationInfo_->butterflyStyle = "na";
1112
1113 std::vector<Real> times, forwards, rfDisc, divDisc;
1114 for (auto const& p : expiries) {
1115 Date d = vol_->optionDateFromTenor(p);
1116 calibrationInfo_->expiryDates.push_back(d);
1117 times.push_back(vol_->dayCounter().empty() ? Actual365Fixed().yearFraction(asof, d)
1118 : vol_->timeFromReference(d));
1119 forwards.push_back(eqIndex->forecastFixing(d));
1120 rfDisc.push_back(eqIndex->equityForecastCurve()->discount(d));
1121 divDisc.push_back(eqIndex->equityDividendCurve()->discount(d));
1122 }
1123
1124 calibrationInfo_->times = times;
1125 calibrationInfo_->forwards = forwards;
1126
1127 std::vector<std::vector<Real>> callPricesDelta(times.size(), std::vector<Real>(deltas.size(), 0.0));
1128 std::vector<std::vector<Real>> callPricesMoneyness(times.size(), std::vector<Real>(moneyness.size(), 0.0));
1129
1130 calibrationInfo_->isArbitrageFree = true;
1131
1132 if (reportOnDeltaGrid) {
1133 calibrationInfo_->deltas = deltas;
1134 calibrationInfo_->deltaGridStrikes =
1135 std::vector<std::vector<Real>>(times.size(), std::vector<Real>(deltas.size(), 0.0));
1136 calibrationInfo_->deltaGridProb =
1137 std::vector<std::vector<Real>>(times.size(), std::vector<Real>(deltas.size(), 0.0));
1138 calibrationInfo_->deltaGridImpliedVolatility =
1139 std::vector<std::vector<Real>>(times.size(), std::vector<Real>(deltas.size(), 0.0));
1140 calibrationInfo_->deltaGridCallSpreadArbitrage =
1141 std::vector<std::vector<bool>>(times.size(), std::vector<bool>(deltas.size(), true));
1142 calibrationInfo_->deltaGridButterflyArbitrage =
1143 std::vector<std::vector<bool>>(times.size(), std::vector<bool>(deltas.size(), true));
1144 TLOG("EquityVolCurve: Delta surface arbitrage analysis result (no calendar spread arbitrage included):");
1145 Real maxTime = QL_MAX_REAL;
1146 if (maxExpiry_ != Date()) {
1147 if (vol_->dayCounter().empty())
1148 maxTime = Actual365Fixed().yearFraction(asof, maxExpiry_);
1149 else
1150 maxTime = vol_->timeFromReference(maxExpiry_);
1151 }
1152 DeltaVolQuote::AtmType at;
1153 DeltaVolQuote::DeltaType dt;
1154 for (Size i = 0; i < times.size(); ++i) {
1155 Real t = times[i];
1156 at = atmType;
1157 dt = deltaType;
1158 // for times after the last quoted expiry we use artificial conventions to avoid problems with strike
1159 // from delta conversions: we use fwd delta always and ATM DNS
1160 if (t > maxTime) {
1161 at = DeltaVolQuote::AtmDeltaNeutral;
1162 dt = DeltaVolQuote::Fwd;
1163 }
1164 bool validSlice = true;
1165 for (Size j = 0; j < deltas.size(); ++j) {
1166 DeltaString d(deltas[j]);
1167 try {
1168 Real strike;
1169 if (d.isAtm()) {
1170 strike = QuantExt::getAtmStrike(dt, at, eqIndex->equitySpot()->value(), rfDisc[i],
1171 divDisc[i], vol_, t);
1172 } else if (d.isCall()) {
1173 strike = QuantExt::getStrikeFromDelta(Option::Call, d.delta(), dt,
1174 eqIndex->equitySpot()->value(), rfDisc[i], divDisc[i],
1175 vol_, t);
1176 } else {
1177 strike =
1178 QuantExt::getStrikeFromDelta(Option::Put, d.delta(), dt, eqIndex->equitySpot()->value(),
1179 rfDisc[i], divDisc[i], vol_, t);
1180 }
1181 Real stddev = std::sqrt(vol_->blackVariance(t, strike));
1182 callPricesDelta[i][j] = blackFormula(Option::Call, strike, forwards[i], stddev);
1183 calibrationInfo_->deltaGridStrikes[i][j] = strike;
1184 calibrationInfo_->deltaGridImpliedVolatility[i][j] = stddev / std::sqrt(t);
1185 } catch (const std::exception& e) {
1186 validSlice = false;
1187 TLOG("EquityVolCurve: error for time " << t << " delta " << deltas[j] << ": " << e.what());
1188 }
1189 }
1190 if (validSlice) {
1191 try {
1192 QuantExt::CarrMadanMarginalProbability cm(calibrationInfo_->deltaGridStrikes[i], forwards[i],
1193 callPricesDelta[i]);
1194 calibrationInfo_->deltaGridCallSpreadArbitrage[i] = cm.callSpreadArbitrage();
1195 calibrationInfo_->deltaGridButterflyArbitrage[i] = cm.butterflyArbitrage();
1196 if (!cm.arbitrageFree())
1197 calibrationInfo_->isArbitrageFree = false;
1198 calibrationInfo_->deltaGridProb[i] = cm.density();
1199 TLOGGERSTREAM(arbitrageAsString(cm));
1200 } catch (const std::exception& e) {
1201 TLOG("error for time " << t << ": " << e.what());
1202 calibrationInfo_->isArbitrageFree = false;
1203 TLOGGERSTREAM("..(invalid slice)..");
1204 }
1205 } else {
1206 calibrationInfo_->isArbitrageFree = false;
1207 TLOGGERSTREAM("..(invalid slice)..");
1208 }
1209 }
1210 TLOG("EquityVolCurve: Delta surface arbitrage analysis completed.");
1211 }
1212
1213 if (reportOnMoneynessGrid) {
1214 calibrationInfo_->moneyness = moneyness;
1215 calibrationInfo_->moneynessGridStrikes =
1216 std::vector<std::vector<Real>>(times.size(), std::vector<Real>(moneyness.size(), 0.0));
1217 calibrationInfo_->moneynessGridProb =
1218 std::vector<std::vector<Real>>(times.size(), std::vector<Real>(moneyness.size(), 0.0));
1219 calibrationInfo_->moneynessGridImpliedVolatility =
1220 std::vector<std::vector<Real>>(times.size(), std::vector<Real>(moneyness.size(), 0.0));
1221 calibrationInfo_->moneynessGridCallSpreadArbitrage =
1222 std::vector<std::vector<bool>>(times.size(), std::vector<bool>(moneyness.size(), true));
1223 calibrationInfo_->moneynessGridButterflyArbitrage =
1224 std::vector<std::vector<bool>>(times.size(), std::vector<bool>(moneyness.size(), true));
1225 calibrationInfo_->moneynessGridCalendarArbitrage =
1226 std::vector<std::vector<bool>>(times.size(), std::vector<bool>(moneyness.size(), true));
1227 for (Size i = 0; i < times.size(); ++i) {
1228 Real t = times[i];
1229 for (Size j = 0; j < moneyness.size(); ++j) {
1230 try {
1231 Real strike = moneyness[j] * forwards[i];
1232 calibrationInfo_->moneynessGridStrikes[i][j] = strike;
1233 Real stddev = std::sqrt(vol_->blackVariance(t, strike));
1234 callPricesMoneyness[i][j] = blackFormula(Option::Call, strike, forwards[i], stddev);
1235 calibrationInfo_->moneynessGridImpliedVolatility[i][j] = stddev / std::sqrt(t);
1236 } catch (const std::exception& e) {
1237 TLOG("EquityVolCurve: error for time " << t << " moneyness " << moneyness[j] << ": " << e.what());
1238 }
1239 }
1240 }
1241 if (!times.empty() && !moneyness.empty()) {
1242 try {
1243 QuantExt::CarrMadanSurface cm(times, moneyness, eqIndex->equitySpot()->value(), forwards,
1244 callPricesMoneyness);
1245 for (Size i = 0; i < times.size(); ++i) {
1246 calibrationInfo_->moneynessGridProb[i] = cm.timeSlices()[i].density();
1247 }
1248 calibrationInfo_->moneynessGridCallSpreadArbitrage = cm.callSpreadArbitrage();
1249 calibrationInfo_->moneynessGridButterflyArbitrage = cm.butterflyArbitrage();
1250 calibrationInfo_->moneynessGridCalendarArbitrage = cm.calendarArbitrage();
1251 if (!cm.arbitrageFree())
1252 calibrationInfo_->isArbitrageFree = false;
1253 TLOG("EquityVolCurve: Moneyness surface Arbitrage analysis result:");
1254 TLOGGERSTREAM(arbitrageAsString(cm));
1255 } catch (const std::exception& e) {
1256 TLOG("EquityVolCurve: error: " << e.what());
1257 calibrationInfo_->isArbitrageFree = false;
1258 }
1259 TLOG("EquityVolCurve: Moneyness surface Arbitrage analysis completed:");
1260 }
1261 }
1262
1263 DLOG("EquityVolCurve: Building calibration info for eq vol surface completed.");
1264
1265 } catch (std::exception& e) {
1266 QL_FAIL("EquityVolCurve: calibration info building failed: " << e.what());
1267 } catch (...) {
1268 QL_FAIL("EquityVolCurve: calibration info building failed: unknown error");
1269 }
