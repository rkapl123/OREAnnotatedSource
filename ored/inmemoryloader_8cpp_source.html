67std::set<boost::shared_ptr<MarketDatum>> InMemoryLoader::get(const Wildcard& wildcard,
68 const QuantLib::Date& asof) const {
69 if (!wildcard.hasWildcard()) {
70 // no wildcard => use get by name function
71 try {
72 return {get(wildcard.pattern(), asof)};
73 } catch (...) {
74 }
75 return {};
76 }
77 auto it = data_.find(asof);
78 if (it == data_.end())
79 return {};
80 std::set<boost::shared_ptr<MarketDatum>> result;
81 std::set<boost::shared_ptr<MarketDatum>>::iterator it1, it2;
82 if (wildcard.wildcardPos() == 0) {
83 // wildcard at first position => we have to search all of the data
84 it1 = it->second.begin();
85 it2 = it->second.end();
86 } else {
87 // search the range matching the substring of the pattern until the wildcard
88 std::string prefix = wildcard.pattern().substr(0, wildcard.wildcardPos());
89 it1 = it->second.lower_bound(makeDummyMarketDatum(asof, prefix));
90 it2 = it->second.upper_bound(makeDummyMarketDatum(asof, prefix + "\xFF"));
91 }
92 for (auto it = it1; it != it2; ++it) {
93 if (wildcard.isPrefix() || wildcard.matches((*it)->name()))
94 result.insert(*it);
95 }
96 return result;
97}

104void InMemoryLoader::add(QuantLib::Date date, const string& name, QuantLib::Real value) {
105 boost::shared_ptr<MarketDatum> md;
106 try {
107 md = parseMarketDatum(date, name, value);
108 } catch (std::exception& e) {
109 WLOG("Failed to parse MarketDatum " << name << ": " << e.what());
110 }
111 if (md != nullptr) {
112 if (data_[date].insert(md).second) {
113 TLOG("Added MarketDatum " << name);
114 } else {
115 WLOG("Skipped MarketDatum " << name << " - this is already present.");
116 }
117 }
118}

139void load(InMemoryLoader& loader, const vector<string>& data, bool isMarket, bool implyTodaysFixings) {
140 LOG("MemoryLoader started");
141
142 Date today = QuantLib::Settings::instance().evaluationDate();
143
144 for (Size i = 0; i < data.size(); ++i) {
145 string line = data[i];
146 // skip blank and comment lines
147 if (line.size() > 0 && line[0] != '#') {
148 vector<string> tokens;
149 boost::trim(line);
150 boost::split(tokens, line, boost::is_any_of(",;\t "), boost::token_compress_on);
151
152 // TODO: should we try, catch and log any invalid lines?
153 QL_REQUIRE(tokens.size() == 3, "Invalid MemoryLoader line, 3 tokens expected " << line);
154 Date date = parseDate(tokens[0]);
155 const string& key = tokens[1];
156 Real value = parseReal(tokens[2]);
157
158 if (isMarket) {
159 // process market
160 // build market datum and add to map
161 try {
162 loader.add(date, key, value);
163 TLOG("Added MarketDatum " << key);
164 } catch (std::exception& e) {
165 WLOG("Failed to parse MarketDatum " << key << ": " << e.what());
166 }
167 } else {
168 // process fixings
169 if (date < today || (date == today && !implyTodaysFixings))
170 loader.addFixing(date, key, value);
171 }
172 }
173 }
174 LOG("MemoryLoader completed");
175}
