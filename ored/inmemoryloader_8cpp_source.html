67 std::set<boost::shared_ptr<MarketDatum>> InMemoryLoader::get(const Wildcard& wildcard,
68  const QuantLib::Date& asof) const {
69  auto it = data_.find(asof);
70  if (it == data_.end())
71  return {};
72  std::set<boost::shared_ptr<MarketDatum>> result;
73  std::set<boost::shared_ptr<MarketDatum>>::iterator it1, it2;
74  if (wildcard.wildcardPos() == std::string::npos || wildcard.wildcardPos() == 0) {
75  // no wildcard => we have to search all of the data
76  it1 = it->second.begin();
77  it2 = it->second.end();
78  } else {
79  // search the range matching the substring of the pattern until the wildcard
80  std::string prefix = wildcard.pattern().substr(0, wildcard.wildcardPos());
81  it1 = it->second.lower_bound(makeDummyMarketDatum(asof, prefix));
82  it2 = it->second.upper_bound(makeDummyMarketDatum(asof, prefix + "\xFF"));
83  }
84  for (auto it = it1; it != it2; ++it) {
85  if (wildcard.isPrefix() || wildcard.matches((*it)->name()))
86  result.insert(*it);
87  }
88  return result;
89 }

130 void load(InMemoryLoader& loader, const vector<string>& data, bool isMarket, bool implyTodaysFixings) {
131  LOG("MemoryLoader started");
132 
133  Date today = QuantLib::Settings::instance().evaluationDate();
134 
135  for (Size i = 0; i < data.size(); ++i) {
136  string line = data[i];
137  // skip blank and comment lines
138  if (line.size() > 0 && line[0] != '#') {
139  vector<string> tokens;
140  boost::trim(line);
141  boost::split(tokens, line, boost::is_any_of(",;\t "), boost::token_compress_on);
142 
143  // TODO: should we try, catch and log any invalid lines?
144  QL_REQUIRE(tokens.size() == 3, "Invalid MemoryLoader line, 3 tokens expected " << line);
145  Date date = parseDate(tokens[0]);
146  const string& key = tokens[1];
147  Real value = parseReal(tokens[2]);
148 
149  if (isMarket) {
150  // process market
151  // build market datum and add to map
152  try {
153  loader.add(date, key, value);
154  TLOG("Added MarketDatum " << key);
155  } catch (std::exception& e) {
156  WLOG("Failed to parse MarketDatum " << key << ": " << e.what());
157  }
158  } else {
159  // process fixings
160  if (date < today || (date == today && !implyTodaysFixings))
161  loader.addFixing(date, key, value);
162  }
163  }
164  }
165  LOG("MemoryLoader completed");
166 }
