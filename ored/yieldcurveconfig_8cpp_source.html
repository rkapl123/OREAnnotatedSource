40YieldCurveSegment::Type parseYieldCurveSegment(const string& s) {
41 if (iequals(s, "Zero"))
42 return YieldCurveSegment::Type::Zero;
43 else if (iequals(s, "Zero Spread"))
44 return YieldCurveSegment::Type::ZeroSpread;
45 else if (iequals(s, "Discount"))
46 return YieldCurveSegment::Type::Discount;
47 else if (iequals(s, "Deposit"))
48 return YieldCurveSegment::Type::Deposit;
49 else if (iequals(s, "FRA"))
50 return YieldCurveSegment::Type::FRA;
51 else if (iequals(s, "Future"))
52 return YieldCurveSegment::Type::Future;
53 else if (iequals(s, "OIS"))
54 return YieldCurveSegment::Type::OIS;
55 else if (iequals(s, "Swap"))
56 return YieldCurveSegment::Type::Swap;
57 else if (iequals(s, "Average OIS"))
58 return YieldCurveSegment::Type::AverageOIS;
59 else if (iequals(s, "Tenor Basis Swap"))
60 return YieldCurveSegment::Type::TenorBasis;
61 else if (iequals(s, "Tenor Basis Two Swaps"))
62 return YieldCurveSegment::Type::TenorBasisTwo;
63 else if (iequals(s, "BMA Basis Swap"))
64 return YieldCurveSegment::Type::BMABasis;
65 else if (iequals(s, "FX Forward"))
66 return YieldCurveSegment::Type::FXForward;
67 else if (iequals(s, "Cross Currency Basis Swap"))
68 return YieldCurveSegment::Type::CrossCcyBasis;
69 else if (iequals(s, "Cross Currency Fix Float Swap"))
70 return YieldCurveSegment::Type::CrossCcyFixFloat;
71 else if (iequals(s, "Discount Ratio"))
72 return YieldCurveSegment::Type::DiscountRatio;
73 else if (iequals(s, "FittedBond"))
74 return YieldCurveSegment::Type::FittedBond;
75 else if (iequals(s, "Yield Plus Default"))
76 return YieldCurveSegment::Type::YieldPlusDefault;
77 else if (iequals(s, "Weighted Average"))
78 return YieldCurveSegment::Type::WeightedAverage;
79 else if (iequals(s, "Ibor Fallback"))
80 return YieldCurveSegment::Type::IborFallback;
81 else if (iequals(s, "Bond Yield Shifted"))
82 return YieldCurveSegment::Type::BondYieldShifted;
83 QL_FAIL("Yield curve segment type " << s << " not recognized");
84}

228const vector<string>& YieldCurveConfig::quotes() {
229 if (quotes_.size() == 0) {
230 bool addedFxSpot = false;
231 for (auto c : curveSegments_) {
232 for (auto segmentQuote : c->quotes())
233 quotes_.push_back(segmentQuote.first);
234
235 // Check if the segment is a CrossCcyYieldCurveSegment and add the FX spot rate to the
236 // set of quotes needed for the YieldCurveConfig if it has not already been added.
237 if (auto xccySegment = boost::dynamic_pointer_cast<CrossCcyYieldCurveSegment>(c)) {
238 if (!addedFxSpot)
239 quotes_.push_back(xccySegment->spotRateID());
240 }
241 }
242 }
243 return quotes_;
244}

246void YieldCurveConfig::fromXML(XMLNode* node) {
247
248 XMLUtils::checkNode(node, "YieldCurve");
249
250 // Read in the mandatory nodes.
251 curveID_ = XMLUtils::getChildValue(node, "CurveId", true);
252 curveDescription_ = XMLUtils::getChildValue(node, "CurveDescription", true);
253 currency_ = XMLUtils::getChildValue(node, "Currency", true);
254 discountCurveID_ = XMLUtils::getChildValue(node, "DiscountCurve", true);
255
256 // Read in the segments.
257 XMLNode* segmentsNode = XMLUtils::getChildNode(node, "Segments");
258 if (segmentsNode) {
259 for (XMLNode* child = XMLUtils::getChildNode(segmentsNode); child; child = XMLUtils::getNextSibling(child)) {
260
261 boost::shared_ptr<YieldCurveSegment> segment;
262 string childName = XMLUtils::getNodeName(child);
263
264 if (childName == "Direct") {
265 segment.reset(new DirectYieldCurveSegment());
266 } else if (childName == "Simple") {
267 segment.reset(new SimpleYieldCurveSegment());
268 } else if (childName == "AverageOIS") {
269 segment.reset(new AverageOISYieldCurveSegment());
270 } else if (childName == "TenorBasis") {
271 segment.reset(new TenorBasisYieldCurveSegment());
272 } else if (childName == "CrossCurrency") {
273 segment.reset(new CrossCcyYieldCurveSegment());
274 } else if (childName == "ZeroSpread") {
275 segment.reset(new ZeroSpreadedYieldCurveSegment());
276 } else if (childName == "DiscountRatio") {
277 segment.reset(new DiscountRatioYieldCurveSegment());
278 } else if (childName == "FittedBond") {
279 segment.reset(new FittedBondYieldCurveSegment());
280 } else if (childName == "BondYieldShifted") {
281 segment.reset(new BondYieldShiftedYieldCurveSegment());
282 } else if (childName == "WeightedAverage") {
283 segment.reset(new WeightedAverageYieldCurveSegment());
284 } else if (childName == "YieldPlusDefault") {
285 segment.reset(new YieldPlusDefaultYieldCurveSegment());
286 } else if(childName == "IborFallback"){
287 segment.reset(new IborFallbackCurveSegment());
288 } else {
289 QL_FAIL("Yield curve segment node name '" << childName << "' not recognized.");
290 }
291
292 if (segment) {
293 try {
294 segment->fromXML(child);
295 } catch (std::exception& ex) {
296 QL_FAIL("Exception parsing yield curve segment XML Node, name = "
297 << childName << " and curveID = " << curveID_ << " : " << ex.what());
298 }
299 } else {
300 QL_FAIL("Unable to build yield curve segment for name = " << childName
301 << " and curveID = " << curveID_);
302 }
303 curveSegments_.push_back(segment);
304 }
305 } else {
306 QL_FAIL("No Segments node in XML doc for yield curve ID = " << curveID_);
307 }
308
309 // Read in the optional nodes.
310
311 // Empty strings if not there (or if there and empty).
312 interpolationVariable_ = XMLUtils::getChildValue(node, "InterpolationVariable", false);
313 interpolationMethod_ = XMLUtils::getChildValue(node, "InterpolationMethod", false);
314 zeroDayCounter_ = XMLUtils::getChildValue(node, "YieldCurveDayCounter", false);
315
316 // Add hardcoded defaults for now.
317 if (interpolationVariable_.empty()) {
318 interpolationVariable_ = "Discount";
319 }
320 if (interpolationMethod_.empty()) {
321 interpolationMethod_ = interpolationVariable_ == "Zero" ? "Linear" : "LogLinear";
322 }
323 if (zeroDayCounter_.empty()) {
324 zeroDayCounter_ = "A365";
325 }
326 XMLNode* nodeToTest = XMLUtils::getChildNode(node, "Extrapolation");
327 if (nodeToTest) {
328 extrapolation_ = XMLUtils::getChildValueAsBool(node, "Extrapolation", false);
329 } else {
330 extrapolation_ = true;
331 }
332
333 // Optional bootstrap configuration
334 if (XMLNode* n = XMLUtils::getChildNode(node, "BootstrapConfig")) {
335 bootstrapConfig_.fromXML(n);
336 }
337
338 // Tolerance is deprecated in favour of Accuracy in BootstrapConfig. However, if it is
339 // still provided, use it as the accuracy and global accuracy in the bootstrap.
340 if (XMLUtils::getChildNode(node, "Tolerance")) {
341 Real accuracy = XMLUtils::getChildValueAsDouble(node, "Tolerance", false);
342 bootstrapConfig_ =
343 BootstrapConfig(accuracy, accuracy, bootstrapConfig_.dontThrow(), bootstrapConfig_.maxAttempts(),
344 bootstrapConfig_.maxFactor(), bootstrapConfig_.minFactor());
345 }
346
347 populateRequiredCurveIds();
348}

350XMLNode* YieldCurveConfig::toXML(XMLDocument& doc) {
351 // Allocate a node.
352 XMLNode* node = doc.allocNode("YieldCurve");
353
354 // Add the mandatory members.
355 XMLUtils::addChild(doc, node, "CurveId", curveID_);
356 XMLUtils::addChild(doc, node, "CurveDescription", curveDescription_);
357 XMLUtils::addChild(doc, node, "Currency", currency_);
358 XMLUtils::addChild(doc, node, "DiscountCurve", discountCurveID_);
359
360 // Add the segments node.
361 XMLNode* segmentsNode = doc.allocNode("Segments");
362 XMLUtils::appendNode(node, segmentsNode);
363 for (Size i = 0; i < curveSegments_.size(); ++i) {
364 XMLUtils::appendNode(segmentsNode, curveSegments_[i]->toXML(doc));
365 }
366
367 // Add the defaultable elements.
368 XMLUtils::addChild(doc, node, "InterpolationVariable", interpolationVariable_);
369 XMLUtils::addChild(doc, node, "InterpolationMethod", interpolationMethod_);
370 XMLUtils::addChild(doc, node, "YieldCurveDayCounter", zeroDayCounter_);
371 XMLUtils::addChild(doc, node, "Tolerance", bootstrapConfig_.accuracy());
372 XMLUtils::addChild(doc, node, "Extrapolation", extrapolation_);
373 XMLUtils::appendNode(node, bootstrapConfig_.toXML(doc));
374
375 return node;
376}

399void YieldCurveSegment::fromXML(XMLNode* node) {
400 typeID_ = XMLUtils::getChildValue(node, "Type", true);
401 string name = XMLUtils::getNodeName(node);
402
403 // Check if curve type is valid for the given segment name
404 std::map<std::string, std::list<std::string>> validSegmentTypes = {
405 {"Direct", {"Zero", "Discount"}},
406 {"Simple", {"Deposit", "FRA", "Future", "OIS", "Swap", "BMA Basis Swap"}},
407 {"AverageOIS", {"Average OIS"}},
408 {"TenorBasis", {"Tenor Basis Swap", "Tenor Basis Two Swaps"}},
409 {"CrossCurrency", {"FX Forward", "Cross Currency Basis Swap", "Cross Currency Fix Float Swap"}},
410 {"ZeroSpread", {"Zero Spread"}},
411 {"FittedBond", {"FittedBond"}},
412 {"YieldPlusDefault", {"Yield Plus Default"}},
413 {"WeightedAverage", {"Weighted Average"}},
414 {"DiscountRatio", {"Discount Ratio"}},
415 {"IborFallback", {"Ibor Fallback"}},
416 {"BondYieldShifted", {"Bond Yield Shifted"}}
417 };
418
419 std::list<std::string> validTypes = validSegmentTypes.at(name);
420 QL_REQUIRE(std::find(validTypes.begin(), validTypes.end(), typeID_) != validTypes.end(),
421 "The curve type " << typeID_ << " is not a valid " << name << " curve segment type");
422
423 quotes_.clear();
424 if (name == "AverageOIS") {
425 XMLNode* quotesNode = XMLUtils::getChildNode(node, "Quotes");
426 if (quotesNode) {
427 for (XMLNode* child = XMLUtils::getChildNode(quotesNode, "CompositeQuote"); child;
428 child = XMLUtils::getNextSibling(child)) {
429 quotes_.push_back(quote(XMLUtils::getChildValue(child, "RateQuote", true)));
430 quotes_.push_back(quote(XMLUtils::getChildValue(child, "SpreadQuote", true)));
431 }
432 } else {
433 QL_FAIL("No Quotes in segment. Remove segment or add quotes.");
434 }
435 } else {
436 XMLNode* quotesNode = XMLUtils::getChildNode(node, "Quotes");
437 if (quotesNode) {
438 for (auto n : XMLUtils::getChildrenNodes(quotesNode, "Quote")) {
439 string attr = XMLUtils::getAttribute(n, "optional"); // return "" if not present
440 bool opt = (!attr.empty() && parseBool(attr));
441 quotes_.emplace_back(quote(XMLUtils::getNodeValue(n), opt));
442 }
443 }
444 }
445 type_ = parseYieldCurveSegment(typeID_);
446 conventionsID_ = XMLUtils::getChildValue(node, "Conventions", false);
447 pillarChoice_ = parsePillarChoice(XMLUtils::getChildValue(node, "PillarChoice", false, "LastRelevantDate"));
448 QL_REQUIRE(pillarChoice_ == QuantLib::Pillar::MaturityDate || pillarChoice_ == QuantLib::Pillar::LastRelevantDate,
449 "PillarChoice " << pillarChoice_ << " not supported, expected MaturityDate, LastRelevantDate");
450}

452XMLNode* YieldCurveSegment::toXML(XMLDocument& doc) {
453 XMLNode* node = doc.allocNode("Segment");
454 XMLUtils::addChild(doc, node, "Type", typeID_);
455 if (!quotes_.empty()) {
456 XMLNode* quotesNode = doc.allocNode("Quotes");
457 // Special case handling for AverageOIS where the quotes are stored as pairs
458 // Spread and Rate.
459 if (type_ == YieldCurveSegment::Type::AverageOIS) {
460 QL_REQUIRE(quotes_.size() % 2 == 0, "Invalid quotes vector should be even");
461 for (Size i = 0; i < quotes_.size(); i = i + 2) {
462 string rateQuote = quotes_[i].first;
463 string spreadQuote = quotes_[i + 1].first;
464
465 XMLNode* compositeQuoteNode = doc.allocNode("CompositeQuote");
466 XMLUtils::addChild(doc, compositeQuoteNode, "SpreadQuote", spreadQuote);
467 XMLUtils::addChild(doc, compositeQuoteNode, "RateQuote", rateQuote);
468 XMLUtils::appendNode(quotesNode, compositeQuoteNode);
469 }
470 } else {
471 for (auto q : quotes_) {
472 XMLNode* qNode = doc.allocNode("Quote", q.first);
473 if (q.second)
474 XMLUtils::addAttribute(doc, qNode, "optional", "true");
475 XMLUtils::appendNode(quotesNode, qNode);
476 }
477 }
478 XMLUtils::appendNode(node, quotesNode);
479 }
480
481 XMLUtils::addChild(doc, node, "Conventions", conventionsID_);
482 XMLUtils::addChild(doc, node, "PillarChoice", ore::data::to_string(pillarChoice_));
483 return node;
484}
