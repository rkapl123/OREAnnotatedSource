72 : MultiLegOptionEngineBuilderBase("CrossAssetModel", "AMC"), cam_(cam), simulationDates_(simulationDates) {}
73
74protected:

78 const std::vector<QuantLib::ext::shared_ptr<QuantLib::InterestRateIndex>>& indexes) override {
