49bool IborFallbackConfig::isIndexReplaced(const string& iborIndex, const Date& asof) const {
50 if (!enableIborFallbacks())
51 return false;
52 auto i = fallbacks_.find(iborIndex);
53 return i != fallbacks_.end() && asof >= i->second.switchDate;
54}

56const IborFallbackConfig::FallbackData& IborFallbackConfig::fallbackData(const string& iborIndex) const {
57 auto i = fallbacks_.find(iborIndex);
58 QL_REQUIRE(
59 i != fallbacks_.end(),
60 "No fallback data found for ibor index '"
61 << iborIndex
62 << "', client code should check whether an index is replaced with IsIndexReplaced() before querying data.");
63 return i->second;
64}

66void IborFallbackConfig::fromXML(XMLNode* node) {
67 clear();
68 XMLUtils::checkNode(node, "IborFallbackConfig");
69 if (auto global = XMLUtils::getChildNode(node, "GlobalSettings")) {
70 enableIborFallbacks_ = XMLUtils::getChildValueAsBool(global, "EnableIborFallbacks", true);
71 useRfrCurveInTodaysMarket_ = XMLUtils::getChildValueAsBool(global, "UseRfrCurveInTodaysMarket", true);
72 useRfrCurveInSimulationMarket_ = XMLUtils::getChildValueAsBool(global, "UseRfrCurveInSimulationMarket", true);
73 }
74 if (auto fallbacks = XMLUtils::getChildNode(node, "Fallbacks")) {
75 for (auto const repl : XMLUtils::getChildrenNodes(fallbacks, "Fallback")) {
76 XMLUtils::checkNode(repl, "Fallback");
77 string ibor = XMLUtils::getChildValue(repl, "IborIndex", true);
78 fallbacks_[ibor].rfrIndex = XMLUtils::getChildValue(repl, "RfrIndex", true);
79 fallbacks_[ibor].spread = parseReal(XMLUtils::getChildValue(repl, "Spread", true));
80 fallbacks_[ibor].switchDate = parseDate(XMLUtils::getChildValue(repl, "SwitchDate", true));
81 }
82 }
83}

85XMLNode* IborFallbackConfig::toXML(XMLDocument& doc) {
86 XMLNode* node = doc.allocNode("IborFallbackConfig");
87 XMLNode* global = XMLUtils::addChild(doc, node, "GlobalSettings");
88 XMLUtils::addChild(doc, global, "EnableIborFallbacks", enableIborFallbacks_);
89 XMLUtils::addChild(doc, global, "UseRfrCurveInTodaysMarket", useRfrCurveInTodaysMarket_);
90 XMLUtils::addChild(doc, global, "UseRfrCurveInSimulationMarket", useRfrCurveInSimulationMarket_);
91 XMLNode* repl = XMLUtils::addChild(doc, node, "Fallbacks");
92 for (auto const& r : fallbacks_) {
93 XMLNode* tmp = XMLUtils::addChild(doc, repl, "Fallback");
94 XMLUtils::addChild(doc, tmp, "IborIndex", r.first);
95 XMLUtils::addChild(doc, tmp, "RfrIndex", r.second.rfrIndex);
96 XMLUtils::addChild(doc, tmp, "Spread", r.second.spread);
97 XMLUtils::addChild(doc, tmp, "SwitchDate", ore::data::to_string(r.second.switchDate));
98 }
99 return node;
100}

190void IborFallbackConfig::updateSwitchDate(QuantLib::Date targetSwitchDate, const std::string& indexName) {
191 for (std::map<std::string, FallbackData>::iterator f = fallbacks_.begin(); f != fallbacks_.end(); f++) {
192 if ((f->first == indexName || indexName == "") && // selected index or all of them if indexName is left blank
193 f->second.switchDate > targetSwitchDate) { // skipping IBORs with switch dates before the target switch date
194 WLOG(StructuredConfigurationWarningMessage("IborFallbackConfig", f->first,
195 "",
196 "Updating switch date from " + to_string(f->second.switchDate) +
197 " to " + to_string(targetSwitchDate)));
198 f->second.switchDate = targetSwitchDate;
199 }
200 }
201}
