65 template <class T>
66 void CurveConfigurations::parseNode(XMLNode* node, const char* parentName, const char* childName,
67  map<string, boost::shared_ptr<T>>& m) {
68 
69  XMLNode* parentNode = XMLUtils::getChildNode(node, parentName);
70  if (parentNode) {
71  for (XMLNode* child = XMLUtils::getChildNode(parentNode, childName); child;
72  child = XMLUtils::getNextSibling(child, childName)) {
73  boost::shared_ptr<T> curveConfig(new T());
74  try {
75  curveConfig->fromXML(child);
76  const string& id = curveConfig->curveID();
77  m[id] = curveConfig;
78  DLOG("Added curve config with ID = " << id);
79  } catch (std::exception& ex) {
80  string id = "(unknown curve id)";
81  try {
82  // try to get the curve id for which an error was thrown
83  id = XMLUtils::getChildValue(child, "CurveId", true);
84  // if we got it, store the error, so that get() can include that information in its error message
85  parseErrors_[std::make_pair(std::type_index(typeid(T)), id)] =
86  std::make_pair(string(parentName), ex.what());
87  } catch (...) {
88  }
89  ALOG("Exception parsing curve config under node '" << parentName << "' for curve id '" << id
90  << "': " << ex.what());
91  }
92  }
93  }

96 template <class T>
97 const boost::shared_ptr<T>& CurveConfigurations::get(const string& id,
98  const map<string, boost::shared_ptr<T>>& m) const {
99  auto it = m.find(id);
100  if (it != m.end())
101  return it->second;
102  auto err = parseErrors_.find(std::make_pair(std::type_index(typeid(T)), id));
103  if (err != parseErrors_.end()) {
104  QL_FAIL("no curve id for '" << id << "' under node '" << err->second.first
105  << "' due to parser error: " << err->second.second);
106  } else {
107  QL_FAIL("no curve id for '" << id << "', is the id present in the curve configuration?");
108  }
109 }

111 boost::shared_ptr<CurveConfigurations>
112 CurveConfigurations::minimalCurveConfig(const boost::shared_ptr<TodaysMarketParameters> todaysMarketParams,
113  const set<string>& configurations) const {
114 
115  boost::shared_ptr<CurveConfigurations> minimum = boost::make_shared<CurveConfigurations>();
116  // If tmparams is not null, organise its specs in to a map [CurveType, set of CurveConfigID]
117  map<CurveSpec::CurveType, set<string>> curveConfigIds;
118  // This set of FXSpotSpec is used below
119  set<boost::shared_ptr<FXSpotSpec>> fxSpotSpecs;
120 
121  for (const auto& config : configurations) {
122  for (const auto& strSpec : todaysMarketParams->curveSpecs(config)) {
123  auto spec = parseCurveSpec(strSpec);
124  if (curveConfigIds.count(spec->baseType())) {
125  curveConfigIds[spec->baseType()].insert(spec->curveConfigID());
126  } else {
127  curveConfigIds[spec->baseType()] = {spec->curveConfigID()};
128  }
129 
130  if (spec->baseType() == CurveSpec::CurveType::FX) {
131  boost::shared_ptr<FXSpotSpec> fxss = boost::dynamic_pointer_cast<FXSpotSpec>(spec);
132  QL_REQUIRE(fxss, "Expected an FXSpotSpec but did not get one");
133  fxSpotSpecs.insert(fxss);
134  }
135  }
136  }
137 
138  // follow order in xsd
139  addMinimalCurves("FXSpots", fxSpotConfigs_, minimum->fxSpotConfigs_, CurveSpec::CurveType::FX, curveConfigIds);
140  addMinimalCurves("FXVolatilities", fxVolCurveConfigs_, minimum->fxVolCurveConfigs_,
141  CurveSpec::CurveType::FXVolatility, curveConfigIds);
142  addMinimalCurves("SwaptionVolatilities", swaptionVolCurveConfigs_, minimum->swaptionVolCurveConfigs_,
143  CurveSpec::CurveType::SwaptionVolatility, curveConfigIds);
144  addMinimalCurves("YieldVolatilities", yieldVolCurveConfigs_, minimum->yieldVolCurveConfigs_,
145  CurveSpec::CurveType::YieldVolatility, curveConfigIds);
146  addMinimalCurves("CapFloorVolatilities", capFloorVolCurveConfigs_, minimum->capFloorVolCurveConfigs_,
147  CurveSpec::CurveType::CapFloorVolatility, curveConfigIds);
148  addMinimalCurves("CDSVolatilities", cdsVolCurveConfigs_, minimum->cdsVolCurveConfigs_,
149  CurveSpec::CurveType::CDSVolatility, curveConfigIds);
150  addMinimalCurves("DefaultCurves", defaultCurveConfigs_, minimum->defaultCurveConfigs_,
151  CurveSpec::CurveType::Default, curveConfigIds);
152  addMinimalCurves("YieldCurves", yieldCurveConfigs_, minimum->yieldCurveConfigs_, CurveSpec::CurveType::Yield,
153  curveConfigIds);
154  addMinimalCurves("InflationCurves", inflationCurveConfigs_, minimum->inflationCurveConfigs_,
155  CurveSpec::CurveType::Inflation, curveConfigIds);
156  addMinimalCurves("InflationCapFloorVolatilities", inflationCapFloorVolCurveConfigs_,
157  minimum->inflationCapFloorVolCurveConfigs_, CurveSpec::CurveType::InflationCapFloorVolatility,
158  curveConfigIds);
159  addMinimalCurves("EquityCurves", equityCurveConfigs_, minimum->equityCurveConfigs_, CurveSpec::CurveType::Equity,
160  curveConfigIds);
161  addMinimalCurves("EquityVolatilities", equityVolCurveConfigs_, minimum->equityVolCurveConfigs_,
162  CurveSpec::CurveType::EquityVolatility, curveConfigIds);
163  addMinimalCurves("Securities", securityConfigs_, minimum->securityConfigs_, CurveSpec::CurveType::Security,
164  curveConfigIds);
165  addMinimalCurves("BaseCorrelations", baseCorrelationCurveConfigs_, minimum->baseCorrelationCurveConfigs_,
166  CurveSpec::CurveType::BaseCorrelation, curveConfigIds);
167  addMinimalCurves("CommodityCurves", commodityCurveConfigs_, minimum->commodityCurveConfigs_,
168  CurveSpec::CurveType::Commodity, curveConfigIds);
169  addMinimalCurves("CommodityVolatilities", commodityVolatilityConfigs_, minimum->commodityVolatilityConfigs_,
170  CurveSpec::CurveType::CommodityVolatility, curveConfigIds);
171  addMinimalCurves("Correlations", correlationCurveConfigs_, minimum->correlationCurveConfigs_,
172  CurveSpec::CurveType::Correlation, curveConfigIds);
173 
174  return minimum;
175 }

177 std::set<string> CurveConfigurations::quotes(const boost::shared_ptr<TodaysMarketParameters> todaysMarketParams,
178  const set<string>& configurations) const {
179 
180  std::set<string> quotes = minimalCurveConfig(todaysMarketParams, configurations)->quotes();
181 
182  // FX spot is special in that we generally do not enter a curve configuration for it. Above, we ran over the
183  // curve configurations asking each for its quotes. We may end up missing FX spot quotes that are specified in a
184  // TodaysMarketParameters but do not have a CurveConfig. If we have a TodaysMarketParameters instance we can add
185  // them here directly using it.
186 
187  for (const auto& config : configurations) {
188  for (const auto& strSpec : todaysMarketParams->curveSpecs(config)) {
189  auto spec = parseCurveSpec(strSpec);
190  if (spec->baseType() == CurveSpec::CurveType::FX) {
191  boost::shared_ptr<FXSpotSpec> fxss = boost::dynamic_pointer_cast<FXSpotSpec>(spec);
192  QL_REQUIRE(fxss, "Expected an FXSpotSpec but did not get one");
193  string strQuote = "FX/RATE/" + fxss->unitCcy() + "/" + fxss->ccy();
194  quotes.insert(strQuote);
195  }
196  }
197  }
198 
199  return quotes;
200 }

298 std::map<CurveSpec::CurveType, std::set<string>>
299 CurveConfigurations::requiredCurveIds(const CurveSpec::CurveType& type, const std::string& curveId) const {
300  std::map<CurveSpec::CurveType, std::set<string>> result;
301  if (type == CurveSpec::CurveType::Yield)
302  addRequiredCurveIds(curveId, yieldCurveConfigs_, result);
303  else if (type == CurveSpec::CurveType::FXVolatility)
304  addRequiredCurveIds(curveId, fxVolCurveConfigs_, result);
305  else if (type == CurveSpec::CurveType::SwaptionVolatility)
306  addRequiredCurveIds(curveId, swaptionVolCurveConfigs_, result);
307  else if (type == CurveSpec::CurveType::YieldVolatility)
308  addRequiredCurveIds(curveId, yieldVolCurveConfigs_, result);
309  else if (type == CurveSpec::CurveType::CapFloorVolatility)
310  addRequiredCurveIds(curveId, capFloorVolCurveConfigs_, result);
311  else if (type == CurveSpec::CurveType::Default)
312  addRequiredCurveIds(curveId, defaultCurveConfigs_, result);
313  else if (type == CurveSpec::CurveType::CDSVolatility)
314  addRequiredCurveIds(curveId, cdsVolCurveConfigs_, result);
315  else if (type == CurveSpec::CurveType::BaseCorrelation)
316  addRequiredCurveIds(curveId, baseCorrelationCurveConfigs_, result);
317  else if (type == CurveSpec::CurveType::Inflation)
318  addRequiredCurveIds(curveId, inflationCurveConfigs_, result);
319  else if (type == CurveSpec::CurveType::InflationCapFloorVolatility)
320  addRequiredCurveIds(curveId, inflationCapFloorVolCurveConfigs_, result);
321  else if (type == CurveSpec::CurveType::Equity)
322  addRequiredCurveIds(curveId, equityCurveConfigs_, result);
323  else if (type == CurveSpec::CurveType::EquityVolatility)
324  addRequiredCurveIds(curveId, equityVolCurveConfigs_, result);
325  else if (type == CurveSpec::CurveType::Security)
326  addRequiredCurveIds(curveId, securityConfigs_, result);
327  else if (type == CurveSpec::CurveType::FX)
328  addRequiredCurveIds(curveId, fxSpotConfigs_, result);
329  else if (type == CurveSpec::CurveType::Commodity)
330  addRequiredCurveIds(curveId, commodityCurveConfigs_, result);
331  else if (type == CurveSpec::CurveType::CommodityVolatility)
332  addRequiredCurveIds(curveId, commodityVolatilityConfigs_, result);
333  else if (type == CurveSpec::CurveType::Correlation)
334  addRequiredCurveIds(curveId, correlationCurveConfigs_, result);
335  else {
336  QL_FAIL("CurveConfigurations::requiredCurveIds(): unhandled curve spec type");
337  }
338  return result;
339 }

477 void CurveConfigurations::fromXML(XMLNode* node) {
478  XMLUtils::checkNode(node, "CurveConfiguration");
479 
480  // Load global settings
481  if (auto tmp = XMLUtils::getChildNode(node, "ReportConfiguration")) {
482  if (auto tmp2 = XMLUtils::getChildNode(tmp, "EquityVolatilities")) {
483  if (auto tmp3 = XMLUtils::getChildNode(tmp2, "Report"))
484  reportConfigEqVols_.fromXML(tmp3);
485  }
486  if (auto tmp2 = XMLUtils::getChildNode(tmp, "FXVolatilities")) {
487  if (auto tmp3 = XMLUtils::getChildNode(tmp2, "Report"))
488  reportConfigFxVols_.fromXML(tmp3);
489  }
490  }
491 
492  // Load YieldCurves, FXVols, etc, etc
493  parseNode(node, "YieldCurves", "YieldCurve", yieldCurveConfigs_);
494  parseNode(node, "FXVolatilities", "FXVolatility", fxVolCurveConfigs_);
495  parseNode(node, "SwaptionVolatilities", "SwaptionVolatility", swaptionVolCurveConfigs_);
496  parseNode(node, "YieldVolatilities", "YieldVolatility", yieldVolCurveConfigs_);
497  parseNode(node, "CapFloorVolatilities", "CapFloorVolatility", capFloorVolCurveConfigs_);
498  parseNode(node, "DefaultCurves", "DefaultCurve", defaultCurveConfigs_);
499  parseNode(node, "CDSVolatilities", "CDSVolatility", cdsVolCurveConfigs_);
500  parseNode(node, "BaseCorrelations", "BaseCorrelation", baseCorrelationCurveConfigs_);
501  parseNode(node, "EquityCurves", "EquityCurve", equityCurveConfigs_);
502  parseNode(node, "EquityVolatilities", "EquityVolatility", equityVolCurveConfigs_);
503  parseNode(node, "InflationCurves", "InflationCurve", inflationCurveConfigs_);
504  parseNode(node, "InflationCapFloorVolatilities", "InflationCapFloorVolatility", inflationCapFloorVolCurveConfigs_);
505  parseNode(node, "Securities", "Security", securityConfigs_);
506  parseNode(node, "FXSpots", "FXSpot", fxSpotConfigs_);
