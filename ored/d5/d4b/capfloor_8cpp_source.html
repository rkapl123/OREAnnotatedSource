42 void CapFloor::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
43 
44  DLOG("CapFloor::build() called for trade " << id() << ", leg type is " << legData_.legType());
45 
46  QL_REQUIRE((legData_.legType() == "Floating") || (legData_.legType() == "CMS") ||
47  (legData_.legType() == "DurationAdjustedCMS") || (legData_.legType() == "CPI") ||
48  (legData_.legType() == "YY"),
49  "CapFloor build error, LegType must be Floating, CMS, DurationAdjustedCMS, CPI or YY");
50 
51  QL_REQUIRE(caps_.size() > 0 || floors_.size() > 0, "CapFloor build error, no cap rates or floor rates provided");
52  QuantLib::CapFloor::Type capFloorType;
53  if (floors_.size() == 0) {
54  capFloorType = QuantLib::CapFloor::Cap;
55  } else if (caps_.size() == 0) {
56  capFloorType = QuantLib::CapFloor::Floor;
57  } else {
58  capFloorType = QuantLib::CapFloor::Collar;
59  }
60 
61  legs_.clear();
62  boost::shared_ptr<EngineBuilder> builder;
63  std::string underlyingIndex, qlIndexName;
64  boost::shared_ptr<QuantLib::Instrument> qlInstrument;
65 
66  // Account for long / short multiplier. In the following we expect the qlInstrument to be set up
67  // as a long cap resp. a long floor resp. as a collar which by definition is a long cap + short floor
68  // (this is opposite to the definition of a leg with naked option = true!)
69  // The isPayer flag in the leg data is ignored.
70  Real multiplier = (parsePositionType(longShort_) == Position::Long ? 1.0 : -1.0);
71 
72  if (legData_.legType() == "Floating") {
73 
74  boost::shared_ptr<FloatingLegData> floatData =
75  boost::dynamic_pointer_cast<FloatingLegData>(legData_.concreteLegData());
76  QL_REQUIRE(floatData, "Wrong LegType, expected Floating, got " << legData_.legType());
77  underlyingIndex = floatData->index();
78  Handle<IborIndex> hIndex =
79  engineFactory->market()->iborIndex(underlyingIndex, engineFactory->configuration(MarketContext::pricing));
80  QL_REQUIRE(!hIndex.empty(), "Could not find ibor index " << underlyingIndex << " in market.");
81  boost::shared_ptr<IborIndex> index = hIndex.currentLink();
82  qlIndexName = index->name();
83 
84  QL_REQUIRE(floatData->caps().empty() && floatData->floors().empty(),
85  "CapFloor build error, Floating leg section must not have caps and floors");
86 
87  bool isBmaIndex = boost::dynamic_pointer_cast<QuantExt::BMAIndexWrapper>(index) != nullptr;
88 
89  if (!isBmaIndex && !floatData->hasSubPeriods()) {
90  // For the cases where we support caps and floors in the regular way, we build a floating leg with
91  // the nakedOption flag set to true, this avoids maintaining all features in legs with associated
92  // coupon pricers and at the same time in the QuaantLib::CapFloor instrument and pricing engine.
93  // Supported cases are
94  // - Ibor coupon without sub periods (hasSubPeriods = false)
95  // - compounded ON coupon, averaged ON coupon
96  // The other cases are handled below.
97  LegData tmpLegData = legData_;
98  boost::shared_ptr<FloatingLegData> tmpFloatData = boost::make_shared<FloatingLegData>(*floatData);
99  tmpFloatData->floors() = floors_;
100  tmpFloatData->caps() = caps_;
101  tmpFloatData->nakedOption() = true;
102  tmpLegData.concreteLegData() = tmpFloatData;
103  legs_.push_back(engineFactory->legBuilder(tmpLegData.legType())
104  ->buildLeg(tmpLegData, engineFactory, requiredFixings_,
105  engineFactory->configuration(MarketContext::pricing)));
106  // if both caps and floors are given, we have to use a payer leg, since in this case
107  // the StrippedCappedFlooredCoupon used to extract the naked options assumes a long floor
108  // and a short cap while we have documented a collar to be a short floor and long cap
109  qlInstrument =
110  boost::make_shared<QuantLib::Swap>(legs_, std::vector<bool>{!floors_.empty() && !caps_.empty()});
111  qlInstrument->setPricingEngine(
112  boost::make_shared<DiscountingSwapEngine>(engineFactory->market()->discountCurve(legData_.currency())));
113  maturity_ = CashFlows::maturityDate(legs_.front());
114  } else {
115  // For the cases where we don't have regular cap / floor support we treat the index approximately as an Ibor
116  // index and build an QuantLib::CapFloor with associated pricing engine. These cases comprise:
117  // - BMA coupons
118  // - Ibor coupons with sub periods (hasSubPeriods = true)
119 
120  ALOG("CapFloor trade " << id()
121  << " on a) BMA or b) sub periods Ibor or c) averaged ON underlying (index = '"
122  << underlyingIndex
123  << "') built, will treat the index approximately as an ibor index");
124  builder = engineFactory->builder(tradeType_);
125  legs_.push_back(makeIborLeg(legData_, index, engineFactory));
126 
127  // If a vector of cap/floor rates are provided, ensure they align with the number of schedule periods
128  if (floors_.size() > 1) {
129  QL_REQUIRE(floors_.size() == legs_[0].size(),
130  "The number of floor rates provided does not match the number of schedule periods");
131  }
132 
133  if (caps_.size() > 1) {
134  QL_REQUIRE(caps_.size() == legs_[0].size(),
135  "The number of cap rates provided does not match the number of schedule periods");
136  }
137 
138  // If one cap/floor rate is given, extend the vector to align with the number of schedule periods
139  if (floors_.size() == 1)
140  floors_.resize(legs_[0].size(), floors_[0]);
141 
142  if (caps_.size() == 1)
143  caps_.resize(legs_[0].size(), caps_[0]);
144 
145  // Create QL CapFloor instrument
146  qlInstrument = boost::make_shared<QuantLib::CapFloor>(capFloorType, legs_[0], caps_, floors_);
147 
148  boost::shared_ptr<CapFloorEngineBuilder> capFloorBuilder =
149  boost::dynamic_pointer_cast<CapFloorEngineBuilder>(builder);
150  qlInstrument->setPricingEngine(capFloorBuilder->engine(parseCurrency(legData_.currency())));
151 
152  maturity_ = boost::dynamic_pointer_cast<QuantLib::CapFloor>(qlInstrument)->maturityDate();
153  }
154 
155  } else if (legData_.legType() == "CMS") {
156  builder = engineFactory->builder("Swap");
157 
158  boost::shared_ptr<CMSLegData> cmsData = boost::dynamic_pointer_cast<CMSLegData>(legData_.concreteLegData());
159  QL_REQUIRE(cmsData, "Wrong LegType, expected CMS");
160 
161  underlyingIndex = cmsData->swapIndex();
162  Handle<SwapIndex> hIndex =
163  engineFactory->market()->swapIndex(underlyingIndex, builder->configuration(MarketContext::pricing));
164  QL_REQUIRE(!hIndex.empty(), "Could not find swap index " << underlyingIndex << " in market.");
165 
166  boost::shared_ptr<SwapIndex> index = hIndex.currentLink();
167  qlIndexName = index->name();
168 
169  vector<bool> legPayers;
170  if (capFloorType == QuantLib::CapFloor::Collar)
171  // long cap, short floor
172  legPayers = {true, false};
173  else if(capFloorType == QuantLib::CapFloor::Cap)
174  // long cap
175  legPayers = {true, false};
176  else if(capFloorType == QuantLib::CapFloor::Floor)
177  // long floor
178  legPayers = {false, true};
179  legs_.push_back(makeCMSLeg(legData_, index, engineFactory, caps_, floors_));
180  legs_.push_back(makeCMSLeg(legData_, index, engineFactory));
181 
182  qlInstrument = boost::make_shared<QuantLib::Swap>(legs_, legPayers);
183  boost::shared_ptr<SwapEngineBuilderBase> cmsCapFloorBuilder =
184  boost::dynamic_pointer_cast<SwapEngineBuilderBase>(builder);
185  qlInstrument->setPricingEngine(cmsCapFloorBuilder->engine(parseCurrency(legData_.currency())));
186 
187  maturity_ = boost::dynamic_pointer_cast<QuantLib::Swap>(qlInstrument)->maturityDate();
188 
189  } else if (legData_.legType() == "DurationAdjustedCMS") {
190  auto cmsData = boost::dynamic_pointer_cast<DurationAdjustedCmsLegData>(legData_.concreteLegData());
191  QL_REQUIRE(cmsData, "Wrong LegType, expected DurationAdjustedCmsLegData");
192  LegData tmpLegData = legData_;
193  auto tmpCmsData = boost::make_shared<DurationAdjustedCmsLegData>(*cmsData);
194  tmpCmsData->floors() = floors_;
195  tmpCmsData->caps() = caps_;
196  tmpCmsData->nakedOption() = true;
197  tmpLegData.concreteLegData() = tmpCmsData;
198  legs_.push_back(engineFactory->legBuilder(tmpLegData.legType())
199  ->buildLeg(tmpLegData, engineFactory, requiredFixings_,
200  engineFactory->configuration(MarketContext::pricing)));
201  // if both caps and floors are given, we have to use a payer leg, since in this case
202  // the StrippedCappedFlooredCoupon used to extract the naked options assumes a long floor
203  // and a short cap while we have documented a collar to be a short floor and long cap
204  qlInstrument = boost::make_shared<QuantLib::Swap>(legs_, std::vector<bool>{!floors_.empty() && !caps_.empty()});
205  qlInstrument->setPricingEngine(
206  boost::make_shared<DiscountingSwapEngine>(engineFactory->market()->discountCurve(legData_.currency())));
207  maturity_ = CashFlows::maturityDate(legs_.front());
208  } else if (legData_.legType() == "CPI") {
209  DLOG("CPI CapFloor Type " << capFloorType << " ID " << id());
210 
211  builder = engineFactory->builder("CpiCapFloor");
212 
213  boost::shared_ptr<CPILegData> cpiData = boost::dynamic_pointer_cast<CPILegData>(legData_.concreteLegData());
214  QL_REQUIRE(cpiData, "Wrong LegType, expected CPI");
215 
216  underlyingIndex = cpiData->index();
217  Handle<ZeroInflationIndex> zeroIndex = engineFactory->market()->zeroInflationIndex(
218  underlyingIndex, builder->configuration(MarketContext::pricing));
219  qlIndexName = zeroIndex->name();
220 
221  // the cpi leg uses the first schedule date as the start date, which only makes sense if there are at least
222  // two dates in the schedule, otherwise the only date in the schedule is the pay date of the cf and a a separate
223  // start date is expected; if both the separate start date and a schedule with more than one date is given
224  Date startDate;
225  Schedule schedule = makeSchedule(legData_.schedule());
226 
227  const string& start = cpiData->startDate();
228  if (schedule.size() < 2) {
229  QL_REQUIRE(!start.empty(), "Only one schedule date, a 'StartDate' must be given.");
230  startDate = parseDate(start);
231  } else if (!start.empty()) {
232  DLOG("Schedule with more than 2 dates was provided. The first schedule date " <<
233  io::iso_date(schedule.dates().front()) << " is used as the start date. The 'StartDate' of " <<
234  start << " is not used.");
235  startDate = schedule.dates().front();
236  }
237 
238  Real baseCPI = cpiData->baseCPI();
239  Period observationLag = parsePeriod(cpiData->observationLag());
240  CPI::InterpolationType interpolation = parseObservationInterpolation(cpiData->interpolation());
241  Calendar cal = zeroIndex->fixingCalendar();
242  BusinessDayConvention conv = Unadjusted; // not used in the CPI CapFloor engine
243 
244  QL_REQUIRE(!zeroIndex.empty(), "Zero Inflation Index is empty");
245 
246  legs_.push_back(makeCPILeg(legData_, zeroIndex.currentLink(), engineFactory));
247 
248  // If a vector of cap/floor rates are provided, ensure they align with the number of schedule periods
249  if (floors_.size() > 1) {
250  QL_REQUIRE(floors_.size() == legs_[0].size(),
251  "The number of floor rates provided does not match the number of schedule periods");
252  }
253 
254  if (caps_.size() > 1) {
255  QL_REQUIRE(caps_.size() == legs_[0].size(),
256  "The number of cap rates provided does not match the number of schedule periods");
257  }
258 
259  // If one cap/floor rate is given, extend the vector to align with the number of schedule periods
260  if (floors_.size() == 1)
261  floors_.resize(legs_[0].size(), floors_[0]);
262 
263  if (caps_.size() == 1)
264  caps_.resize(legs_[0].size(), caps_[0]);
265 
266  boost::shared_ptr<CpiCapFloorEngineBuilder> capFloorBuilder =
267  boost::dynamic_pointer_cast<CpiCapFloorEngineBuilder>(builder);
268 
269  // Create QL CPI CapFloor instruments and add to a composite
270  qlInstrument = boost::make_shared<CompositeInstrument>();
271  maturity_ = Date::minDate();
272  for (Size i = 0; i < legs_[0].size(); ++i) {
273  DLOG("Create composite " << i);
274  Real nominal, gearing;
275  Date paymentDate;
276  boost::shared_ptr<CPICoupon> coupon = boost::dynamic_pointer_cast<CPICoupon>(legs_[0][i]);
277  boost::shared_ptr<CPICashFlow> cashflow = boost::dynamic_pointer_cast<CPICashFlow>(legs_[0][i]);
278  if (coupon) {
279  nominal = coupon->nominal();
280  gearing = coupon->fixedRate() * coupon->accrualPeriod();
281  paymentDate = coupon->date();
282  } else if (cashflow) {
283  nominal = cashflow->notional();
284  gearing = 1.0; // no gearing here
285  paymentDate = cashflow->date();
286  } else {
287  QL_FAIL("Failed to interprete CPI flow");
288  }
289 
290  if (capFloorType == QuantLib::CapFloor::Cap || capFloorType == QuantLib::CapFloor::Collar) {
291  boost::shared_ptr<CPICapFloor> capfloor =
292  boost::make_shared<CPICapFloor>(Option::Call, nominal, startDate, baseCPI, paymentDate, cal, conv,
293  cal, conv, caps_[i], zeroIndex, observationLag, interpolation);
294  capfloor->setPricingEngine(capFloorBuilder->engine(underlyingIndex));
295  boost::dynamic_pointer_cast<QuantLib::CompositeInstrument>(qlInstrument)->add(capfloor, gearing);
296  maturity_ = std::max(maturity_, capfloor->payDate());
297  }
298 
299  if (capFloorType == QuantLib::CapFloor::Floor || capFloorType == QuantLib::CapFloor::Collar) {
300  // for collars we want a long cap, short floor
301  Real sign = capFloorType == QuantLib::CapFloor::Floor ? 1.0 : -1.0;
302  boost::shared_ptr<CPICapFloor> capfloor =
303  boost::make_shared<CPICapFloor>(Option::Put, nominal, startDate, baseCPI, paymentDate, cal, conv,
304  cal, conv, floors_[i], zeroIndex, observationLag, interpolation);
305  capfloor->setPricingEngine(capFloorBuilder->engine(underlyingIndex));
306  boost::dynamic_pointer_cast<QuantLib::CompositeInstrument>(qlInstrument)->add(capfloor, sign * gearing);
307  maturity_ = std::max(maturity_, capfloor->payDate());
308  }
309  }
310 
311  } else if (legData_.legType() == "YY") {
312  builder = engineFactory->builder("YYCapFloor");
313 
314  boost::shared_ptr<YoYLegData> yyData = boost::dynamic_pointer_cast<YoYLegData>(legData_.concreteLegData());
315  QL_REQUIRE(yyData, "Wrong LegType, expected YY");
316 
317  underlyingIndex = yyData->index();
318  Handle<YoYInflationIndex> yoyIndex;
319  // look for yoy inflation index
320  yoyIndex =
321  engineFactory->market()->yoyInflationIndex(underlyingIndex, builder->configuration(MarketContext::pricing));
322  qlIndexName = yoyIndex->name();
323 
324  // we must have either an yoy or a zero inflation index in the market, if no yoy curve, get teh zero
325  // and create a yoy index from it
326  if (yoyIndex.empty()) {
327  Handle<ZeroInflationIndex> zeroIndex = engineFactory->market()->zeroInflationIndex(
328  underlyingIndex, builder->configuration(MarketContext::pricing));
329  QL_REQUIRE(!zeroIndex.empty(), "Could not find inflation index (of type either zero or yoy) "
330  << underlyingIndex << " in market.");
331  yoyIndex = Handle<YoYInflationIndex>(boost::make_shared<QuantExt::YoYInflationIndexWrapper>(
332  zeroIndex.currentLink(), zeroIndex->interpolated()));
333  }
334 
335  legs_.push_back(makeYoYLeg(legData_, yoyIndex.currentLink(), engineFactory));
336 
337  // If a vector of cap/floor rates are provided, ensure they align with the number of schedule periods
338  if (floors_.size() > 1) {
339  QL_REQUIRE(floors_.size() == legs_[0].size(),
340  "The number of floor rates provided does not match the number of schedule periods");
341  }
342 
343  if (caps_.size() > 1) {
344  QL_REQUIRE(caps_.size() == legs_[0].size(),
345  "The number of cap rates provided does not match the number of schedule periods");
346  }
347 
348  // If one cap/floor rate is given, extend the vector to align with the number of schedule periods
349  if (floors_.size() == 1)
350  floors_.resize(legs_[0].size(), floors_[0]);
351 
352  if (caps_.size() == 1)
353  caps_.resize(legs_[0].size(), caps_[0]);
354 
355  // Create QL YoY Inflation CapFloor instrument
356  if (capFloorType == QuantLib::CapFloor::Cap) {
357  qlInstrument = boost::shared_ptr<YoYInflationCapFloor>(new YoYInflationCap(legs_[0], caps_));
358  } else if (capFloorType == QuantLib::CapFloor::Floor) {
359  qlInstrument = boost::shared_ptr<YoYInflationCapFloor>(new YoYInflationFloor(legs_[0], floors_));
360  } else if (capFloorType == QuantLib::CapFloor::Collar) {
361  qlInstrument = boost::shared_ptr<YoYInflationCapFloor>(
362  new YoYInflationCapFloor(QuantLib::YoYInflationCapFloor::Collar, legs_[0], caps_, floors_));
363  } else {
364  QL_FAIL("unknown YoYInflation cap/floor type");
365  }
366 
367  boost::shared_ptr<YoYCapFloorEngineBuilder> capFloorBuilder =
368  boost::dynamic_pointer_cast<YoYCapFloorEngineBuilder>(builder);
369  qlInstrument->setPricingEngine(capFloorBuilder->engine(underlyingIndex));
370 
371  // Wrap the QL instrument in a vanilla instrument
372 
373  maturity_ = boost::dynamic_pointer_cast<QuantLib::YoYInflationCapFloor>(qlInstrument)->maturityDate();
374  } else {
375  QL_FAIL("Invalid legType " << legData_.legType() << " for CapFloor");
376  }
377 
378  std::vector<boost::shared_ptr<Instrument>> additionalInstruments;
379  std::vector<Real> additionalMultipliers;
380  addPremiums(additionalInstruments, additionalMultipliers, 1.0, premiumData_, -multiplier,
381  parseCurrency(legData_.currency()), engineFactory,
382  engineFactory->configuration(MarketContext::pricing));
383 
384  // set instrument
385  instrument_ =
386  boost::make_shared<VanillaInstrument>(qlInstrument, multiplier, additionalInstruments, additionalMultipliers);
387 
388  // add required fixings
389  if (!qlIndexName.empty() && !underlyingIndex.empty()) {
390  std::map<string, string> indexNameMapper = {{qlIndexName, underlyingIndex}};
391  if (engineFactory->iborFallbackConfig().isIndexReplaced(underlyingIndex)) {
392  string rfrIndexName = engineFactory->iborFallbackConfig().fallbackData(underlyingIndex).rfrIndex;
393  // we don't support convention based rfr fallback indices, with ore ticket 1758 this might change
394  indexNameMapper[parseIborIndex(rfrIndexName)->name()] = rfrIndexName;
395  }
396  auto fdg = boost::make_shared<FixingDateGetter>(requiredFixings_, indexNameMapper);
397  for (auto const& l : legs_)
398  addToRequiredFixings(l, fdg);
399  }
400 
401  // Fill in remaining Trade member data
402  legCurrencies_.push_back(legData_.currency());
403  legPayers_.push_back(false); // already accounted for via the instrument multiplier
404  npvCurrency_ = legData_.currency();
405  notionalCurrency_ = legData_.currency();
406  notional_ = currentNotional(legs_[0]);
407 }

409 void CapFloor::fromXML(XMLNode* node) {
410  Trade::fromXML(node);
411  XMLNode* capFloorNode = XMLUtils::getChildNode(node, "CapFloorData");
412  longShort_ = XMLUtils::getChildValue(capFloorNode, "LongShort", true);
413  legData_.fromXML(XMLUtils::getChildNode(capFloorNode, "LegData"));
414  caps_ = XMLUtils::getChildrenValuesAsDoubles(capFloorNode, "Caps", "Cap");
415  floors_ = XMLUtils::getChildrenValuesAsDoubles(capFloorNode, "Floors", "Floor");
416  premiumData_.fromXML(capFloorNode);
417 }

419 XMLNode* CapFloor::toXML(XMLDocument& doc) {
420  XMLNode* node = Trade::toXML(doc);
421  XMLNode* capFloorNode = doc.allocNode("CapFloorData");
422  XMLUtils::appendNode(node, capFloorNode);
423  XMLUtils::addChild(doc, capFloorNode, "LongShort", longShort_);
424  XMLUtils::appendNode(capFloorNode, legData_.toXML(doc));
425  XMLUtils::addChildren(doc, capFloorNode, "Caps", "Cap", caps_);
426  XMLUtils::addChildren(doc, capFloorNode, "Floors", "Floor", floors_);
427  XMLUtils::appendNode(capFloorNode, premiumData_.toXML(doc));
428  return node;
429 }
