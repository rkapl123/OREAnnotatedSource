36 void EquityOption::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
37 
38  // Set the assetName_ as it may have changed after lookup
39  assetName_ = equityName();
40 
41  // Populate the index_ in case the option is automatic exercise.
42  const boost::shared_ptr<Market>& market = engineFactory->market();
43  index_ = *market->equityCurve(assetName_, engineFactory->configuration(MarketContext::pricing));
44 
45  // check the equity currency
46  Currency equityCurrency =
47  market->equityCurve(assetName_, engineFactory->configuration(MarketContext::pricing))->currency();
48  QL_REQUIRE(!equityCurrency.empty(), "No equity currency in equityCurve for equity " << assetName_ << ".");
49 
50  // Set the strike currency - if we have a minor currency, convert the strike
51  if (!strikeCurrency_.empty()) {
52  QL_REQUIRE(parseCurrencyWithMinors(strikeCurrency_) == equityCurrency,
53  "Strike currency " << strikeCurrency_ << " does not match equity currency " << equityCurrency
54  << " for trade " << id() << ".");
55  strike_ = convertMinorToMajorCurrency(strikeCurrency_, localStrike_);
56  } else {
57  // If payoff currency and underlying currency are equivalent (and payoff currency could be a minor currency)
58  if (parseCurrencyWithMinors(localCurrency_) == equityCurrency) {
59  strike_ = convertMinorToMajorCurrency(localCurrency_, localStrike_);
60  TLOG("Setting strike currency to payoff currency " << localCurrency_ << " for trade " << id() << ".");
61  strikeCurrency_ = localCurrency_;
62  } else {
63  // If quanto payoff, then strike currency must be populated to avoid confusion over what the
64  // currency of the strike payoff is: can be either underlying currency or payoff currency
65  QL_FAIL("Strike currency must be specified for a quanto payoff for trade " << id() << ".");
66  }
67  }
68 
69  // Quanto payoff condition, i.e. currency_ != underlyingCurrency_, will be checked in VanillaOptionTrade::build()
70  currency_ = parseCurrencyWithMinors(localCurrency_).code();
71  underlyingCurrency_ = equityCurrency.code();
72 
73  // Build the trade using the shared functionality in the base class.
74  VanillaOptionTrade::build(engineFactory);
75 
76  // LOG the volatility if the trade expiry date is in the future.
77  if (expiryDate_ > Settings::instance().evaluationDate()) {
78  DLOG("Implied vol for " << tradeType_ << " on " << assetName_ << " with expiry " << expiryDate_
79  << " and strike " << strike_ << " is "
80  << market->equityVol(assetName_)->blackVol(expiryDate_, strike_));
81  }
82 
83  additionalData_["quantity"] = quantity_;
84  additionalData_["strike"] = localStrike_;
85  additionalData_["strikeCurrency"] = strikeCurrency_;
86 }

88 void EquityOption::fromXML(XMLNode* node) {
89  VanillaOptionTrade::fromXML(node);
90  XMLNode* eqNode = XMLUtils::getChildNode(node, "EquityOptionData");
91  QL_REQUIRE(eqNode, "No EquityOptionData Node");
92  option_.fromXML(XMLUtils::getChildNode(eqNode, "OptionData"));
93  XMLNode* tmp = XMLUtils::getChildNode(eqNode, "Underlying");
94  if (!tmp)
95  tmp = XMLUtils::getChildNode(eqNode, "Name");
96  equityUnderlying_.fromXML(tmp);
97  localCurrency_ = XMLUtils::getChildValue(eqNode, "Currency", true);
98  localStrike_ = XMLUtils::getChildValueAsDouble(eqNode, "Strike", true);
99  strikeCurrency_ = XMLUtils::getChildValue(eqNode, "StrikeCurrency", false);
100  quantity_ = XMLUtils::getChildValueAsDouble(eqNode, "Quantity", true);
101 }

103 XMLNode* EquityOption::toXML(XMLDocument& doc) {
104  XMLNode* node = VanillaOptionTrade::toXML(doc);
105  XMLNode* eqNode = doc.allocNode("EquityOptionData");
106  XMLUtils::appendNode(node, eqNode);
107 
108  XMLUtils::appendNode(eqNode, option_.toXML(doc));
109  XMLUtils::appendNode(eqNode, equityUnderlying_.toXML(doc));
110  XMLUtils::addChild(doc, eqNode, "Currency", localCurrency_);
111  XMLUtils::addChild(doc, eqNode, "Strike", localStrike_);
112 
113  Currency ccy = parseCurrencyWithMinors(localCurrency_);
114  Currency strikeCcy = parseCurrencyWithMinors(strikeCurrency_);
115  if (!strikeCurrency_.empty() && ccy != strikeCcy)
116  XMLUtils::addChild(doc, eqNode, "StrikeCurrency", strikeCurrency_);
117 
118  XMLUtils::addChild(doc, eqNode, "Quantity", quantity_);
119 
120  return node;
121 }
