80 void addMarketFixing(const string& idxConvId, const Date& expiry, Real value, const Conventions& conventions) {
81  auto p = conventions.get(idxConvId, Convention::Type::CommodityFuture);
82  if (p.first) {
83  auto idx = parseCommodityIndex(idxConvId, conventions, false);
84  idx = idx->clone(expiry);
85  if (idx->isValidFixingDate(expiry)) {
86  try {
87  idx->addFixing(expiry, value);
88  TLOG("Added fixing (" << iso_date(expiry) << "," << idx->name() << "," << value << ").");
89  } catch (const Error& e) {
90  TLOG("Failed to add fixing (" << iso_date(expiry) << "," << idx->name() << "," <<
91  value << "): " << e.what());
92  }
93  } else {
94  TLOG("Failed to add fixing (" << iso_date(expiry) << "," << idx->name() << "," <<
95  value << ") because " << iso_date(expiry) << " is not a valid fixing date.");
96  }
97  } else {
98  TLOG("Failed to add fixing because no commodity future convention for " << idxConvId << ".");
99  }
100 }

110 CommodityCurve::CommodityCurve(const Date& asof, const CommodityCurveSpec& spec, const Loader& loader,
111  const CurveConfigurations& curveConfigs, const Conventions& conventions,
112  const FXTriangulation& fxSpots,
113  const map<string, boost::shared_ptr<YieldCurve>>& yieldCurves,
114  const map<string, boost::shared_ptr<CommodityCurve>>& commodityCurves)
115  : spec_(spec), commoditySpot_(Null<Real>()), onValue_(Null<Real>()), tnValue_(Null<Real>()), regexQuotes_(false) {
116 
117  try {
118 
119  boost::shared_ptr<CommodityCurveConfig> config = curveConfigs.commodityCurveConfig(spec_.curveConfigID());
120 
121  dayCounter_ = config->dayCountId() == "" ? Actual365Fixed() : parseDayCounter(config->dayCountId());
122  interpolationMethod_ = config->interpolationMethod() == "" ? "Linear" : config->interpolationMethod();
123 
124  if (config->type() == CommodityCurveConfig::Type::Direct) {
125 
126  // Populate the raw price curve data
127  map<Date, Handle<Quote>> data;
128  populateData(data, asof, config, loader, conventions);
129 
130  // Create the commodity price curve
131  buildCurve(asof, data, config);
132 
133  } else if (config->type() == CommodityCurveConfig::Type::Basis) {
134 
135  // We have a commodity basis configuration
136 
137  // Look up the required base price curve in the commodityCurves map
138  CommodityCurveSpec ccSpec(config->currency(), config->basePriceCurveId());
139  DLOG("Looking for base price curve with id, " << config->basePriceCurveId() << ", and spec, " << ccSpec
140  << ".");
141  auto itCc = commodityCurves.find(ccSpec.name());
142  QL_REQUIRE(itCc != commodityCurves.end(), "Can't find price curve with id " << config->basePriceCurveId());
143  auto pts = Handle<PriceTermStructure>(itCc->second->commodityPriceCurve());
144 
145  buildBasisPriceCurve(asof, *config, conventions, pts, loader);
146 
147  } else if (config->type() == CommodityCurveConfig::Type::Piecewise) {
148 
149  // We have a piecewise commodity configuration
150  buildPiecewiseCurve(asof, *config, conventions, loader, commodityCurves);
151 
152  } else {
153 
154  // We have a cross currency type commodity curve configuration
155  boost::shared_ptr<CommodityCurveConfig> baseConfig =
156  curveConfigs.commodityCurveConfig(config->basePriceCurveId());
157 
158  buildCrossCurrencyPriceCurve(asof, config, baseConfig, fxSpots, yieldCurves, commodityCurves);
159  }
160 
161  // Apply extrapolation from the curve configuration
162  commodityPriceCurve_->enableExtrapolation(config->extrapolation());
163 
164  // Ask for price now so that errors are thrown during the build, not later.
165  commodityPriceCurve_->price(asof + 1 * Days);
166 
167  } catch (std::exception& e) {
168  QL_FAIL("commodity curve building failed: " << e.what());
169  } catch (...) {
170  QL_FAIL("commodity curve building failed: unknown error");
171  }
172 }

174 void CommodityCurve::populateData(map<Date, Handle<Quote>>& data, const Date& asof,
175  const boost::shared_ptr<CommodityCurveConfig>& config, const Loader& loader,
176  const Conventions& conventions) {
177 
178  // Some default conventions for building the commodity curve
179  Period spotTenor = 2 * Days;
180  Real pointsFactor = 1.0;
181  Calendar cal = parseCalendar(config->currency());
182  bool spotRelative = true;
183  BusinessDayConvention bdc = Following;
184  bool outright = true;
185 
186  // Overwrite the default conventions if the commodity curve config provides explicit conventions
187  if (!config->conventionsId().empty()) {
188  QL_REQUIRE(conventions.has(config->conventionsId()),
189  "Commodity conventions " << config->conventionsId() << " requested by commodity config "
190  << config->curveID() << " not found");
191  auto convention =
192  boost::dynamic_pointer_cast<CommodityForwardConvention>(conventions.get(config->conventionsId()));
193  QL_REQUIRE(convention, "Convention " << config->conventionsId() << " not of expected type CommodityConvention");
194 
195  spotTenor = convention->spotDays() * Days;
196  pointsFactor = convention->pointsFactor();
197  if (convention->advanceCalendar() != NullCalendar())
198  cal = convention->advanceCalendar();
199  spotRelative = convention->spotRelative();
200  bdc = convention->bdc();
201  outright = convention->outright();
202  }
203 
204  // Commodity spot quote if provided by the configuration
205  Date spotDate = cal.advance(asof, spotTenor);
206  if (!config->commoditySpotQuoteId().empty()) {
207  auto spot = loader.get(config->commoditySpotQuoteId(), asof)->quote();
208  commoditySpot_ = spot->value();
209  data[spotDate] = spot;
210  } else {
211  QL_REQUIRE(outright, "If the commodity forward quotes are not outright,"
212  << " a commodity spot quote needs to be configured");
213  }
214 
215  // Add the forward quotes to the curve data
216  for (auto& q : getQuotes(asof, config->curveID(), config->fwdQuotes(), loader)) {
217 
218  // We add ON and TN quotes after this loop if they are given and not outright quotes
219  TLOG("Commodity Forward Price found for quote: " << q->name());
220  Date expiry;
221  Real value = q->quote()->value();
222  if (!q->tenorBased()) {
223  expiry = q->expiryDate();
224  add(asof, expiry, value, data, outright, pointsFactor);
225  } else {
226  if (q->startTenor() == boost::none) {
227  expiry = cal.advance(spotRelative ? spotDate : asof, q->tenor(), bdc);
228  add(asof, expiry, value, data, outright, pointsFactor);
229  } else {
230  if (*q->startTenor() == 0 * Days && q->tenor() == 1 * Days) {
231  onValue_ = q->quote()->value();
232  if (outright)
233  add(asof, asof, value, data, outright);
234  } else if (*q->startTenor() == 1 * Days && q->tenor() == 1 * Days) {
235  tnValue_ = q->quote()->value();
236  if (outright) {
237  expiry = cal.advance(asof, 1 * Days, bdc);
238  add(asof, expiry, value, data, outright);
239  }
240  } else {
241  expiry = cal.advance(cal.advance(asof, *q->startTenor(), bdc), q->tenor(), bdc);
242  add(asof, expiry, value, data, outright, pointsFactor);
243  }
244  }
245  }
246  }
247 
248  // Deal with ON and TN if quotes are not outright quotes
249  if (spotTenor == 2 * Days && tnValue_ != Null<Real>() && !outright) {
250  add(asof, cal.advance(asof, 1 * Days, bdc), -tnValue_, data, outright, pointsFactor);
251  if (onValue_ != Null<Real>()) {
252  add(asof, asof, -onValue_ - tnValue_, data, outright, pointsFactor);
253  }
254  }
255 
256  // Some logging and checks
257  LOG("Read " << data.size() << " quotes for commodity curve " << config->curveID());
258  if (!regexQuotes_) {
259  QL_REQUIRE(data.size() == config->quotes().size(), "Found " << data.size() << " quotes, but "
260  << config->quotes().size()
261  << " quotes given in config " << config->curveID());
262  } else {
263  QL_REQUIRE(data.size() > 0,
264  "Regular expression specified in commodity config " << config->curveID() << " but no quotes read");
265  }
266 }

268 void CommodityCurve::add(const Date& asof, const Date& expiry, Real value, map<Date, Handle<Quote>>& data, bool outright,
269  Real pointsFactor) {
270 
271  if (expiry < asof)
272  return;
273 
274  QL_REQUIRE(data.find(expiry) == data.end(), "Duplicate quote for expiry " << io::iso_date(expiry) << " found.");
275 
276  if (!outright) {
277  QL_REQUIRE(commoditySpot_ != Null<Real>(), "Can't use forward points without a commodity spot value");
278  value = commoditySpot_ + value / pointsFactor;
279  }
280 
281  data[expiry] = Handle<Quote>(boost::make_shared<SimpleQuote>(value));
282 }

284 void CommodityCurve::buildCurve(const Date& asof, const map<Date, Handle<Quote>>& data,
285  const boost::shared_ptr<CommodityCurveConfig>& config) {
286 
287  vector<Date> curveDates;
288  curveDates.reserve(data.size());
289  vector<Handle<Quote>> curvePrices;
290  curvePrices.reserve(data.size());
291  for (auto const& datum : data) {
292  curveDates.push_back(datum.first);
293  curvePrices.push_back(datum.second);
294  }
295 
296  // Build the curve using the data
297  populateCurve<InterpolatedPriceCurve>(asof, curveDates, curvePrices, dayCounter_,
298  parseCurrency(config->currency()));
299 }

301 void CommodityCurve::buildCrossCurrencyPriceCurve(
302  const Date& asof, const boost::shared_ptr<CommodityCurveConfig>& config,
303  const boost::shared_ptr<CommodityCurveConfig>& baseConfig, const FXTriangulation& fxSpots,
304  const map<string, boost::shared_ptr<YieldCurve>>& yieldCurves,
305  const map<string, boost::shared_ptr<CommodityCurve>>& commodityCurves) {
306 
307  // Look up the required base price curve in the commodityCurves map
308  // We pass in the commodity curve ID only in the member basePriceCurveId of config e.g. PM:XAUUSD.
309  // But, the map commodityCurves is keyed on the spec name e.g. Commodity/USD/PM:XAUUSD
310  auto commIt = commodityCurves.find(CommodityCurveSpec(baseConfig->currency(), baseConfig->curveID()).name());
311  QL_REQUIRE(commIt != commodityCurves.end(), "Could not find base commodity curve with id "
312  << baseConfig->curveID()
313  << " required in the building of commodity curve with id "
314  << config->curveID());
315 
316  // Look up the two yield curves in the yieldCurves map
317  auto baseYtsIt = yieldCurves.find(YieldCurveSpec(baseConfig->currency(), config->baseYieldCurveId()).name());
318  QL_REQUIRE(baseYtsIt != yieldCurves.end(),
319  "Could not find base yield curve with id "
320  << config->baseYieldCurveId() << " and currency " << baseConfig->currency()
321  << " required in the building of commodity curve with id " << config->curveID());
322 
323  auto ytsIt = yieldCurves.find(YieldCurveSpec(config->currency(), config->yieldCurveId()).name());
324  QL_REQUIRE(ytsIt != yieldCurves.end(), "Could not find yield curve with id "
325  << config->yieldCurveId() << " and currency " << config->currency()
326  << " required in the building of commodity curve with id "
327  << config->curveID());
328 
329  // Get the FX spot rate, number of units of this currency per unit of base currency
330  Handle<Quote> fxSpot = fxSpots.getQuote(baseConfig->currency() + config->currency());
331 
332  // Populate the commodityPriceCurve_ member
333  commodityPriceCurve_ = boost::make_shared<CrossCurrencyPriceTermStructure>(
334  asof, QuantLib::Handle<PriceTermStructure>(commIt->second->commodityPriceCurve()), fxSpot,
335  baseYtsIt->second->handle(), ytsIt->second->handle(), parseCurrency(config->currency()));
336 }

338 void CommodityCurve::buildBasisPriceCurve(const Date& asof, const CommodityCurveConfig& config,
339  const Conventions& conventions, const Handle<PriceTermStructure>& basePts,
340  const Loader& loader) {
341 
342  LOG("CommodityCurve: start building commodity basis curve.");
343 
344  // We need to have commodity future conventions for both the base curve and the basis curve
345  QL_REQUIRE(conventions.has(config.conventionsId()), "Commodity conventions " << config.conventionsId()
346  << " requested by commodity config "
347  << config.curveID() << " not found");
348  auto basisConvention =
349  boost::dynamic_pointer_cast<CommodityFutureConvention>(conventions.get(config.conventionsId()));
350  QL_REQUIRE(basisConvention,
351  "Convention " << config.conventionsId() << " not of expected type CommodityFutureConvention");
352  auto basisFec = boost::make_shared<ConventionsBasedFutureExpiry>(*basisConvention);
353 
354  QL_REQUIRE(conventions.has(config.baseConventionsId()),
355  "Commodity conventions " << config.baseConventionsId() << " requested by commodity config "
356  << config.curveID() << " not found");
357  auto baseConvention =
358  boost::dynamic_pointer_cast<CommodityFutureConvention>(conventions.get(config.baseConventionsId()));
359  QL_REQUIRE(baseConvention,
360  "Convention " << config.baseConventionsId() << " not of expected type CommodityFutureConvention");
361  auto baseFec = boost::make_shared<ConventionsBasedFutureExpiry>(*baseConvention);
362 
363  // Construct the commodity index.
364  auto index = parseCommodityIndex(baseConvention->id(), conventions, false, basePts);
365 
366  // Sort the configured quotes on expiry dates
367  // Ignore tenor based quotes i.e. we expect an explicit expiry date and log a warning if the expiry date does not
368  // match our own calculated expiry date based on the basis conventions.
369  map<Date, Handle<Quote>> basisData;
370  for (auto& q : getQuotes(asof, config.curveID(), config.fwdQuotes(), loader, true)) {
371 
372  QL_REQUIRE(basisData.find(q->expiryDate()) == basisData.end(), "Found duplicate quote, "
373  << q->name() << ", for expiry date "
374  << io::iso_date(q->expiryDate()) << ".");
375 
376  basisData[q->expiryDate()] = q->quote();
377  TLOG("Using quote " << q->name() << " in commodity basis curve.");
378 
379  // We expect the expiry date in the quotes to match our calculated expiry date. The code will work if it does
380  // not but we log a warning in this case.
381  Date calcExpiry = basisFec->nextExpiry(true, q->expiryDate());
382  if (calcExpiry != q->expiryDate()) {
383  WLOG("Calculated expiry date, " << io::iso_date(calcExpiry) << ", does not equal quote's expiry date "
384  << io::iso_date(q->expiryDate()) << ".");
385  }
386  }
387 
388  if (basisConvention->isAveraging()) {
389  // We are building a curve that will be used to return an average price.
390  if (!baseConvention->isAveraging() && config.averageBase()) {
391  DLOG("Creating a CommodityAverageBasisPriceCurve.");
392  populateCurve<CommodityAverageBasisPriceCurve>(asof, basisData, basisFec, index,
393  basePts, baseFec, config.addBasis());
394  } else {
395  // Either 1) base convention is not averaging and config.averageBase() is false or 2) the base convention
396  // is averaging. Either way, we build a CommodityBasisPriceCurve.
397  DLOG("Creating a CommodityBasisPriceCurve for an average price curve.");
398  populateCurve<CommodityBasisPriceCurve>(asof, basisData, basisFec, index, basePts,
399  baseFec, config.addBasis(), config.monthOffset());
400  }
401  } else {
402  // We are building a curve that will be used to return a price on a single date.
403  QL_REQUIRE(!baseConvention->isAveraging(), "A commodity basis curve with non-averaging" <<
404  " basis and averaging base is not valid.");
405 
406  DLOG("Creating a CommodityBasisPriceCurve.");
407  populateCurve<CommodityBasisPriceCurve>(asof, basisData, basisFec, index, basePts,
408  baseFec, config.addBasis(), config.monthOffset());
409  }
410 
411  LOG("CommodityCurve: finished building commodity basis curve.");
412 }

418 void CommodityCurve::buildPiecewiseCurve(const Date& asof, const CommodityCurveConfig& config,
419  const Conventions& conventions, const Loader& loader,
420  const map<string, boost::shared_ptr<CommodityCurve>>& commodityCurves) {
421 
422  LOG("CommodityCurve: start building commodity piecewise curve.");
423 
424  // We store the instruments in a map. The key is the instrument's pillar date. The segments are ordered in
425  // priority so if we encounter the same pillar date later, we ignore it with a debug log.
426  map<Date, boost::shared_ptr<Helper>> mpInstruments;
427  const auto& priceSegments = config.priceSegments();
428  QL_REQUIRE(!priceSegments.empty(), "CommodityCurve: need at least one price segment to build piecewise curve.");
429  for (const auto& kv : priceSegments) {
430  if (kv.second.type() != PriceSegment::Type::OffPeakPowerDaily) {
431  addInstruments(asof, loader, config.curveID(), config.currency(), kv.second,
432  conventions, commodityCurves, mpInstruments);
433  } else {
434  addOffPeakPowerInstruments(asof, loader, config.curveID(), kv.second, conventions, mpInstruments);
435  }
436  }
437 
438  // Populate the vector of helpers.
439  vector<boost::shared_ptr<Helper>> instruments;
440  instruments.reserve(mpInstruments.size());
441  for (const auto& kv : mpInstruments) {
442  instruments.push_back(kv.second);
443  }
444 
445  // Use bootstrap configuration if provided.
446  BootstrapConfig bc;
447  if (config.bootstrapConfig()) {
448  bc = *config.bootstrapConfig();
449  }
450  Real acc = bc.accuracy();
451  Real globalAcc = bc.globalAccuracy();
452  bool noThrow = bc.dontThrow();
453  Size maxAttempts = bc.maxAttempts();
454  Real maxF = bc.maxFactor();
455  Real minF = bc.minFactor();
456  Size noThrowSteps = bc.dontThrowSteps();
457 
458  // Create curve based on interpolation method provided.
459  Currency ccy = parseCurrency(config.currency());
460  if (interpolationMethod_ == "Linear") {
461  BS<Crv<Linear>> bs(acc, globalAcc, noThrow, maxAttempts, maxF, minF, noThrowSteps);
462  commodityPriceCurve_ = boost::make_shared<Crv<Linear>>(asof, instruments, dayCounter_, ccy, Linear(), bs);
463  } else if (interpolationMethod_ == "LogLinear") {
464  BS<Crv<QuantLib::LogLinear>> bs(acc, globalAcc, noThrow, maxAttempts, maxF, minF, noThrowSteps);
465  commodityPriceCurve_ = boost::make_shared<Crv<QuantLib::LogLinear>>(asof, instruments,
466  dayCounter_, ccy, QuantLib::LogLinear(), bs);
467  } else if (interpolationMethod_ == "Cubic") {
468  BS<Crv<QuantLib::Cubic>> bs(acc, globalAcc, noThrow, maxAttempts, maxF, minF, noThrowSteps);
469  commodityPriceCurve_ = boost::make_shared<Crv<QuantLib::Cubic>>(asof, instruments,
470  dayCounter_, ccy, QuantLib::Cubic(), bs);
471  } else if (interpolationMethod_ == "LinearFlat") {
472  BS<Crv<QuantExt::LinearFlat>> bs(acc, globalAcc, noThrow, maxAttempts, maxF, minF, noThrowSteps);
473  commodityPriceCurve_ = boost::make_shared<Crv<QuantExt::LinearFlat>>(asof, instruments,
474  dayCounter_, ccy, QuantExt::LinearFlat(), bs);
475  } else if (interpolationMethod_ == "LogLinearFlat") {
476  BS<Crv<QuantExt::LogLinearFlat>> bs(acc, globalAcc, noThrow, maxAttempts, maxF, minF, noThrowSteps);
477  commodityPriceCurve_ = boost::make_shared<Crv<QuantExt::LogLinearFlat>>(asof, instruments,
478  dayCounter_, ccy, QuantExt::LogLinearFlat(), bs);
479  } else if (interpolationMethod_ == "CubicFlat") {
480  BS<Crv<QuantExt::CubicFlat>> bs(acc, globalAcc, noThrow, maxAttempts, maxF, minF, noThrowSteps);
481  commodityPriceCurve_ = boost::make_shared<Crv<QuantExt::CubicFlat>>(asof, instruments,
482  dayCounter_, ccy, QuantExt::CubicFlat(), bs);
483  } else if (interpolationMethod_ == "BackwardFlat") {
484  BS<Crv<BackwardFlat>> bs(acc, globalAcc, noThrow, maxAttempts, maxF, minF, noThrowSteps);
485  commodityPriceCurve_ = boost::make_shared<Crv<BackwardFlat>>(asof, instruments,
486  dayCounter_, ccy, BackwardFlat(), bs);
487  } else {
488  QL_FAIL("The interpolation method, " << interpolationMethod_ << ", is not supported.");
489  }
490 
491  LOG("CommodityCurve: finished building commodity piecewise curve.");
492 }

494 vector<boost::shared_ptr<CommodityForwardQuote>>
495 CommodityCurve::getQuotes(const Date& asof, const string& configId, const vector<string>& quotes,
496  const Loader& loader, bool filter) {
497 
498  LOG("CommodityCurve: start getting configured commodity quotes.");
499 
500  // Check if we are using a regular expression to select the quotes for the curve. If we are, the quotes should
501  // contain exactly one element.
502  auto wildcard = getUniqueWildcard(quotes);
503  regexQuotes_ = wildcard != boost::none;
504 
505  // Add the relevant forward quotes to the result vector
506  vector<boost::shared_ptr<CommodityForwardQuote>> result;
507  for (auto& md : loader.loadQuotes(asof)) {
508 
509  // Only looking for quotes on asof date, with quote type PRICE and instrument type commodity forward
510  if (md->asofDate() == asof && md->quoteType() == MarketDatum::QuoteType::PRICE &&
511  md->instrumentType() == MarketDatum::InstrumentType::COMMODITY_FWD) {
512 
513  boost::shared_ptr<CommodityForwardQuote> q = boost::dynamic_pointer_cast<CommodityForwardQuote>(md);
514 
515  // Check if the quote is requested by the config and if it isn't continue to the next quote
516  if (!wildcard) {
517  vector<string>::const_iterator it =
518  find(quotes.begin(), quotes.end(), q->name());
519  if (it == quotes.end())
520  continue;
521  } else {
522  if (!(*wildcard).matches(q->name()))
523  continue;
524  }
525 
526  // If filter is true, remove tenor based quotes and quotes with expiry before asof.
527  if (filter) {
528  if (q->tenorBased()) {
529  TLOG("Skipping tenor based quote, " << q->name() << ".");
530  continue;
531  }
532  if (q->expiryDate() < asof) {
533  TLOG("Skipping quote because its expiry date, " << io::iso_date(q->expiryDate())
534  << ", is before the market date " << io::iso_date(asof));
535  continue;
536  }
537  }
538 
539  // If we make it here, the quote is relevant.
540  result.push_back(q);
541  TLOG("Added quote " << q->name() << ".");
542  }
543  }
544 
545  LOG("CommodityCurve: finished getting configured commodity quotes.");
546 
547  return result;

550 void CommodityCurve::addInstruments(const Date& asof, const Loader& loader, const string& configId,
551  const string& currency, const PriceSegment& priceSegment, const Conventions& conventions,
552  const map<string, boost::shared_ptr<CommodityCurve>>& commodityCurves,
553  map<Date, boost::shared_ptr<Helper>>& instruments) {
554 
555  using PST = PriceSegment::Type;
556  using AD = CommodityFutureConvention::AveragingData;
557  PST type = priceSegment.type();
558 
559  // Pre-populate some variables if averaging segment.
560  boost::shared_ptr<CommodityFutureConvention> convention;
561  AD ad;
562  boost::shared_ptr<CommodityIndex> index;
563  boost::shared_ptr<FutureExpiryCalculator> uFec;
564  if (type == PST::AveragingFuture || type == PST::AveragingSpot || type == PST::AveragingOffPeakPower) {
565 
566  // Get the associated averaging commodity future convention.
567  convention = boost::dynamic_pointer_cast<CommodityFutureConvention>(
568  conventions.get(priceSegment.conventionsId()));
569  QL_REQUIRE(convention, "Convention " << priceSegment.conventionsId() <<
570  " not of expected type CommodityFutureConvention.");
571 
572  ad = convention->averagingData();
573  QL_REQUIRE(!ad.empty(), "CommodityCurve: convention " << convention->id() <<
574  " should have non-empty averaging data for piecewise price curve construction.");
575 
576  // The commodity index for which we are building a price curve.
577  index = parseCommodityIndex(ad.commodityName(), conventions, false);
578 
579  // If referencing a future, we need conventions for the underlying future that is being averaged.
580  if (type == PST::AveragingFuture || type == PST::AveragingOffPeakPower) {
581 
582  auto uConvention = boost::dynamic_pointer_cast<CommodityFutureConvention>(
583  conventions.get(ad.conventionsId()));
584  QL_REQUIRE(uConvention, "Convention " << priceSegment.conventionsId() <<
585  " not of expected type CommodityFutureConvention.");
586  uFec = boost::make_shared<ConventionsBasedFutureExpiry>(*uConvention);
587 
588  if (ad.dailyExpiryOffset() != Null<Natural>() && ad.dailyExpiryOffset() > 0) {
589  QL_REQUIRE(uConvention->contractFrequency() == Daily, "CommodityCurve: the averaging data has" <<
590  " a positive DailyExpiryOffset (" << ad.dailyExpiryOffset() << ") but the underlying future" <<
591  " contract frequency is not daily (" << uConvention->contractFrequency() << ").");
592  }
593  }
594 
595  }
596 
597  // Pre-populate some variables if the price segment is AveragingOffPeakPower.
598  boost::shared_ptr<CommodityIndex> peakIndex;
599  Natural peakHoursPerDay = 16;
600  Calendar peakCalendar;
601  if (type == PST::AveragingOffPeakPower) {
602 
603  // Look up the peak price curve in the commodityCurves map
604  const string& ppId = priceSegment.peakPriceCurveId();
605  QL_REQUIRE(!ppId.empty(), "CommodityCurve: AveragingOffPeakPower segment in " <<
606  " curve configuration " << configId << " does not provide a peak price curve ID.");
607  CommodityCurveSpec ccSpec(currency, ppId);
608  DLOG("Looking for peak price curve with id, " << ppId << ", and spec, " << ccSpec << ".");
609  auto itCc = commodityCurves.find(ccSpec.name());
610  QL_REQUIRE(itCc != commodityCurves.end(), "Can't find peak price curve with id " << ppId);
611  auto peakPts = Handle<PriceTermStructure>(itCc->second->commodityPriceCurve());
612 
613  // Create the daily peak price index linked to the peak price term structure.
614  peakIndex = parseCommodityIndex(ppId, conventions, false, peakPts);
615 
616  // Calendar defining the peak business days.
617  peakCalendar = parseCalendar(priceSegment.peakPriceCalendar());
618 
619  // Look up the conventions for the peak price commodity to determine peak hours per day.
620  if (conventions.has(ppId)) {
621  auto peakConvention = boost::dynamic_pointer_cast<CommodityFutureConvention>(conventions.get(ppId));
622  if (peakConvention && peakConvention->hoursPerDay() != Null<Natural>()) {
623  peakHoursPerDay = peakConvention->hoursPerDay();
624  }
625  }
626  }
627 
628  // Get the relevant quotes
629  auto quotes = getQuotes(asof, configId, priceSegment.quotes(), loader, true);
630 
631  // Add an instrument for each relevant quote.
632  for (const auto& quote : quotes) {
633 
634  const Date& expiry = quote->expiryDate();
635  switch (type) {
636  case PST::Future:
637  if (expiry == asof) {
638  TLOG("Quote " << quote->name() << " has expiry date " << io::iso_date(expiry) << " equal to asof" <<
639  " so not adding to instruments. Attempt to add as fixing instead.");
640  addMarketFixing(priceSegment.conventionsId(), expiry, quote->quote()->value(), conventions);
641  } else if (instruments.count(expiry) == 0) {
642  instruments[expiry] = boost::make_shared<FuturePriceHelper>(quote->quote(), expiry);
643  } else {
644  TLOG("Skipping quote, " << quote->name() << ", because its expiry date, " <<
645  io::iso_date(expiry) << ", is already in the instrument set.");
646  }
647  break;
648 
649  // An averaging future referencing an underlying future or spot. Setup is similar.
650  case PST::AveragingFuture:
651  case PST::AveragingSpot:
652  case PST::AveragingOffPeakPower: {
653 
654  // Determine the calculation period.
655  using ADCP = AD::CalculationPeriod;
656  Date start;
657  Date end;
658  if (ad.period() == ADCP::ExpiryToExpiry) {
659 
660  auto fec = boost::make_shared<ConventionsBasedFutureExpiry>(*convention);
661  end = fec->nextExpiry(true, expiry);
662  if (end != expiry) {
663  WLOG("Calculated expiry date, " << io::iso_date(end) << ", does not equal quote's expiry date "
664  << io::iso_date(expiry) << ". Proceed with quote's expiry.");
665  }
666  start = fec->priorExpiry(false, end) + 1;
667 
668  } else if (ad.period() == ADCP::PreviousMonth) {
669 
670  end = Date::endOfMonth(expiry - 1 * Months);
671  start = Date(1, end.month(), end.year());
672  }
673 
674  boost::shared_ptr<Helper> helper;
675  if (type == PST::AveragingOffPeakPower) {
676  TLOG("Building average off-peak power helper from quote, " << quote->name() << ".");
677  helper = boost::make_shared<AverageOffPeakPowerHelper>(quote->quote(), index, start,
678  end, uFec, peakIndex, peakCalendar, peakHoursPerDay);
679  } else {
680  TLOG("Building average future price helper from quote, " << quote->name() << ".");
681  helper = boost::make_shared<AverageFuturePriceHelper>(quote->quote(), index, start, end, uFec,
682  ad.pricingCalendar(), ad.deliveryRollDays(), ad.futureMonthOffset(), ad.useBusinessDays(),
683  ad.dailyExpiryOffset());
684  }
685 
686  // Only add to instruments if an instrument with the same pillar date is not there already.
687  Date pillar = helper->pillarDate();
688  if (instruments.count(pillar) == 0) {
689  instruments[pillar] = helper;
690  } else {
691  TLOG("Skipping quote, " << quote->name() << ", because an instrument with its pillar date, " <<
692  io::iso_date(pillar) << ", is already in the instrument set.");
693  }
694  break;
695  }
696 
697  default:
698  QL_FAIL("CommodityCurve: unrecognised price segment type.");
699  break;
700  }
701  }

704 void CommodityCurve::addOffPeakPowerInstruments(const Date& asof, const Loader& loader, const string& configId,
705  const PriceSegment& priceSegment, const Conventions& conventions,
706  map<Date, boost::shared_ptr<Helper>>& instruments) {
707 
708  // Check that we have been called with the expected segment type.
709  using PST = PriceSegment::Type;
710  QL_REQUIRE(priceSegment.type() == PST::OffPeakPowerDaily, "Expecting a price segment type of OffPeakPowerDaily.");
711 
712  // Check we have a commodity future convention for the price segment.
713  const string& convId = priceSegment.conventionsId();
714  auto p = conventions.get(convId, Convention::Type::CommodityFuture);
715  QL_REQUIRE(p.first, "Could not get conventions with id " << convId << " for OffPeakPowerDaily price segment" <<
716  " in curve configuration " << configId << ".");
717  auto convention = boost::dynamic_pointer_cast<CommodityFutureConvention>(p.second);
718 
719  // Check that the commodity future convention has off-peak information for the name.
720  const auto& oppIdxData = convention->offPeakPowerIndexData();
721  QL_REQUIRE(oppIdxData, "Conventions with id " << convId << " for OffPeakPowerDaily price segment" <<
722  " should have an OffPeakPowerIndexData section.");
723  Real offPeakHours = oppIdxData->offPeakHours();
724  TLOG("Off-peak hours is " << offPeakHours);
725  const Calendar& peakCalendar = oppIdxData->peakCalendar();
726 
727  // Check that the price segment has off-peak daily section.
728  const auto& opd = priceSegment.offPeakDaily();
729  QL_REQUIRE(opd, "The OffPeakPowerDaily price segment for curve configuration " << configId <<
730  " should have an OffPeakDaily section.");
731 
732  // Get all the peak and off-peak quotes that we have and store them in a map. The map key is the expiry date and
733  // the map value is a pair of values the first being the off-peak value for that expiry and the second being the
734  // peak value for that expiry. We only need the peak portion to form the quote on peakCalendar holidays. We need
735  // the off-peak portion always.
736  map<Date, pair<Real, Real>> quotes;
737 
738  auto opqs = getQuotes(asof, configId, opd->offPeakQuotes(), loader, true);
739  for (const auto& q : opqs) {
740  Real value = q->quote()->value();
741  Date expiry = q->expiryDate();
742  if (quotes.count(expiry) != 0) {
743  TLOG("Already have off-peak quote with expiry " << io::iso_date(expiry) << " so skipping " << q->name());
744  } else {
745  TLOG("Adding off-peak quote " << q->name() << ": " << io::iso_date(expiry) << "," << value);
746  quotes[expiry] = make_pair(value, Null<Real>());
747  }
748  }
749 
750  auto pqs = getQuotes(asof, configId, opd->peakQuotes(), loader, true);
751  for (const auto& q : pqs) {
752  Real value = q->quote()->value();
753  Date expiry = q->expiryDate();
754  auto it = quotes.find(expiry);
755  if (it == quotes.end()) {
756  TLOG("Have no off-peak quote with expiry " << io::iso_date(expiry) << " so skipping " << q->name());
757  } else if (it->second.second != Null<Real>()) {
758  TLOG("Already have a peak quote with expiry " << io::iso_date(expiry) << " so skipping " << q->name());
759  } else {
760  TLOG("Adding peak quote " << q->name() << ": " << io::iso_date(expiry) << "," << value);
761  it->second.second = value;
762  }
763  }
764 
765  // Now, use the quotes to create the future instruments in the curve.
766  for (const auto& kv : quotes) {
767 
768  // If the expiry is already in the instrument set, we skip it.
769  const Date& expiry = kv.first;
770  if (instruments.count(expiry) != 0) {
771  TLOG("Skipping expiry " << io::iso_date(expiry) << " because it is already in the instrument set.");
772  continue;
773  }
774 
775  // If the expiry is equal to the asof, we add fixings.
776  if (expiry == asof) {
777  TLOG("The off-peak power expiry date " << io::iso_date(expiry) << " is equal to asof" <<
778  " so not adding to instruments. Attempt to add fixing(s) instead.");
779  if (peakCalendar.isHoliday(expiry) && kv.second.second == Null<Real>()) {
780  DLOG("The peak portion of the quote on holiday " << io::iso_date(expiry) <<
781  " is missing so can't add fixings.");
782  } else {
783  // Add the off-peak and if necessary peak fixing
784  addMarketFixing(oppIdxData->offPeakIndex(), expiry, kv.second.first, conventions);
785  if (peakCalendar.isHoliday(expiry))
786  addMarketFixing(oppIdxData->peakIndex(), expiry, kv.second.second, conventions);
787  }
788  continue;
789  }
790 
791  // Determine the quote that we will use in the future instrument for this expiry.
792  Real quote = 0.0;
793  if (peakCalendar.isHoliday(expiry)) {
794  Real peakValue = kv.second.second;
795  if (peakValue == Null<Real>()) {
796  DLOG("The peak portion of the quote on holiday " << io::iso_date(expiry) << " is missing so skip.");
797  continue;
798  } else {
799  Real offPeakValue = kv.second.first;
800  quote = (offPeakHours * offPeakValue + (24.0 - offPeakHours) * peakValue) / 24.0;
801  TLOG("The quote on holiday " << io::iso_date(expiry) << " is " << quote << ". (off-peak,peak) is" <<
802  " (" << offPeakValue << "," << peakValue << ").");
803  }
804  } else {
805  quote = kv.second.first;
806  TLOG("The quote on business day " << io::iso_date(expiry) << " is the off-peak value " << quote << ".");
807  }
808 
809  // Add the future helper for this expiry.
810  instruments[expiry] = boost::make_shared<FuturePriceHelper>(quote, expiry);
811 
812  }
813 }
