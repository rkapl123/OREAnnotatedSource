55  TestMarket(Real spot, Real q, Real r, Real vol, bool withFixings = false) {
56  asof_ = Date(3, Feb, 2016);
57 
58  // build discount
59  yieldCurves_[make_tuple(Market::defaultConfiguration, YieldCurveType::Discount, "EUR")] =
60  flatRateYts(r, Actual360());
61  yieldCurves_[make_tuple(Market::defaultConfiguration, YieldCurveType::Discount, "JPY")] =
62  flatRateYts(q, Actual360());
63 
64  // add fx rates
65  fxSpots_[Market::defaultConfiguration].addQuote("JPYEUR", Handle<Quote>(boost::make_shared<SimpleQuote>(spot)));
66 
67  // build fx vols
68  fxVols_[make_pair(Market::defaultConfiguration, "JPYEUR")] = flatRateFxv(vol, Actual360());
69 
70  if (withFixings) {
71  // add fixings
72  TimeSeries<Real> pastFixings;
73  pastFixings[Date(1, Feb, 2016)] = 100;
74  pastFixings[Date(2, Feb, 2016)] = 90;
75  IndexManager::instance().setHistory("FX/Reuters JPY/EUR", pastFixings);
76  }
77  }

79 private:
80  Handle<YieldTermStructure> flatRateYts(Real forward, const DayCounter& dc = ActualActual()) {
81  boost::shared_ptr<YieldTermStructure> yts(new FlatForward(0, NullCalendar(), forward, dc));
82  return Handle<YieldTermStructure>(yts);
83  }
84  Handle<BlackVolTermStructure> flatRateFxv(Volatility forward, const DayCounter& dc = ActualActual()) {
85  boost::shared_ptr<BlackVolTermStructure> fxv(new BlackConstantVol(0, NullCalendar(), forward, dc));
86  return Handle<BlackVolTermStructure>(fxv);
87  }

96 BOOST_AUTO_TEST_CASE(testFXOptionPrice) {
97 
98  BOOST_TEST_MESSAGE("Testing FXOption Price...");
99 
100  Date today = Settings::instance().evaluationDate();
101 
102  // build market
103  boost::shared_ptr<Market> market = boost::make_shared<TestMarket>();
104  Settings::instance().evaluationDate() = market->asofDate();
105 
106  // build FXOption - expiry in 1 Year
107  OptionData optionData("Long", "Call", "European", true, vector<string>(1, "20170203"));
108  OptionData optionDataPremiumUSD("Long", "Call", "European", true, vector<string>(1, "20170203"), "Cash", "",
109  PremiumData(10000.0, "USD", Date(3, Feb, 2017)));
110  OptionData optionDataPremiumEUR("Long", "Call", "European", true, vector<string>(1, "20170203"), "Cash", "",
111  PremiumData(10000.0, "EUR", Date(3, Feb, 2017)));
112  Envelope env("CP1");
113  FxOption fxOption(env, optionData, "EUR", 1000000, // bought
114  "USD", 1250000); // sold
115  FxOption fxOptionPremiumUSD(env, optionDataPremiumUSD, "EUR", 1000000, "USD", 1250000);
116  FxOption fxOptionPremiumEUR(env, optionDataPremiumEUR, "EUR", 1000000, "USD", 1250000);
117 
118  // NPV currency = sold currency = USD
119 
120  Real expectedNPV_USD = 29148.0;
121  Real expectedNPV_EUR = 24290.0;
122  Real expectedNPV_USD_Premium_USD = 19495.6;
123  Real expectedNPV_USD_Premium_EUR = 17496.4;
124 
125  // Build and price
126  boost::shared_ptr<EngineData> engineData = boost::make_shared<EngineData>();
127  engineData->model("FxOption") = "GarmanKohlhagen";
128  engineData->engine("FxOption") = "AnalyticEuropeanEngine";
129  boost::shared_ptr<EngineFactory> engineFactory = boost::make_shared<EngineFactory>(engineData, market);
130  engineFactory->registerBuilder(boost::make_shared<FxEuropeanOptionEngineBuilder>());
131 
132  fxOption.build(engineFactory);
133  fxOptionPremiumUSD.build(engineFactory);
134  fxOptionPremiumEUR.build(engineFactory);
135 
136  Real npv = fxOption.instrument()->NPV();
137  Real npv_prem_usd = fxOptionPremiumUSD.instrument()->NPV();
138  Real npv_prem_eur = fxOptionPremiumEUR.instrument()->NPV();
139 
140  BOOST_TEST_MESSAGE("FX Option, NPV Currency " << fxOption.npvCurrency());
141  BOOST_TEST_MESSAGE("NPV = " << npv);
142  BOOST_TEST_MESSAGE("NPV with premium in USD = " << npv_prem_usd);
143  BOOST_TEST_MESSAGE("NPV with premium in EUR = " << npv_prem_eur);
144 
145  // Check NPV matches expected values. Expected value from Wystup is rounded at each step of
146  // calculation so we get a difference of $50 here.
147  QL_REQUIRE(fxOption.npvCurrency() == "USD", "unexpected NPV currency");
148  if (fxOption.npvCurrency() == "EUR") {
149  BOOST_CHECK_CLOSE(npv, expectedNPV_EUR, 0.2);
150  } else if (fxOption.npvCurrency() == "USD") {
151  BOOST_CHECK_CLOSE(npv, expectedNPV_USD, 0.2);
152  BOOST_CHECK_CLOSE(npv_prem_usd, expectedNPV_USD_Premium_USD, 0.001);
153  BOOST_CHECK_CLOSE(npv_prem_eur, expectedNPV_USD_Premium_EUR, 0.001);
154  } else {
155  BOOST_FAIL("Unexpected FX Option npv currency " << fxOption.npvCurrency());
156  }
157 
158  Settings::instance().evaluationDate() = today; // reset
159 }
