77 void test(string nearDate, string farDate, string nearBoughtCurrency, double nearBoughtAmount, string nearSoldCurrency,
78  double nearSoldAmount, double farBoughtAmount, double farSoldAmount) {
79 
80  // build market
81  boost::shared_ptr<Market> market = boost::make_shared<TestMarket>();
82  Settings::instance().evaluationDate() = market->asofDate();
83  // fxswap's npv should equal that of two separate fxforwards
84  // build first fxforward
85  Envelope env1("FxForward1");
86 
87  boost::shared_ptr<EngineData> engineData = boost::make_shared<EngineData>();
88  engineData->model("FxForward") = "DiscountedCashflows";
89  engineData->engine("FxForward") = "DiscountingFxForwardEngine";
90  boost::shared_ptr<EngineFactory> engineFactory = boost::make_shared<EngineFactory>(engineData, market);
91  // this trade has buyer and seller switched so that it returns it's npv in the same currency as the second forward
92  // fxswap_npv= - fxfor1_npv + fxfor2_npv
93  FxForward fxFor1(env1, nearDate, nearSoldCurrency, nearSoldAmount, nearBoughtCurrency, nearBoughtAmount);
94 
95  fxFor1.build(engineFactory);
96 
97  // build second fxforward
98  Envelope env2("FxForward2");
99  FxForward fxFor2(env2, farDate, nearSoldCurrency, farBoughtAmount, nearBoughtCurrency, farSoldAmount);
100  fxFor2.build(engineFactory);
101 
102  // build fx swap
103 
104  Envelope env3("FxSwap");
105 
106  FxSwap fxswap(env3, nearDate, farDate, nearBoughtCurrency, nearBoughtAmount, nearSoldCurrency, nearSoldAmount,
107  farBoughtAmount, farSoldAmount);
108  fxswap.build(engineFactory);
109  fxswap.instrument()->NPV();
110  Real fx1 = fxFor1.instrument()->NPV();
111  Real fx2 = fxFor2.instrument()->NPV();
112  Real npvForward = fx2 - fx1;
113  if (fxswap.instrument()->NPV() != npvForward)
114  BOOST_FAIL("The FxSwap has NPV: " << fxswap.instrument()->NPV()
115  << ", which does not equal the sum of two Fxforwards: " << npvForward);
116 }
117 } // namespace
