46 Leg FloatingLegBuilder::buildLeg(const LegData& data, const boost::shared_ptr<EngineFactory>& engineFactory,
47  RequiredFixings& requiredFixings, const string& configuration,
48  const QuantLib::Date& openEndDateReplacement) const {
49  auto floatData = boost::dynamic_pointer_cast<FloatingLegData>(data.concreteLegData());
50  QL_REQUIRE(floatData, "Wrong LegType, expected Floating");
51  string indexName = floatData->index();
52  auto index = *engineFactory->market()->iborIndex(indexName, configuration);
53  auto ois = boost::dynamic_pointer_cast<OvernightIndex>(index);
54  Leg result;
55  if (ois != nullptr)
56  result = makeOISLeg(data, ois, engineFactory, true, openEndDateReplacement);
57  else {
58  auto bma = boost::dynamic_pointer_cast<QuantExt::BMAIndexWrapper>(index);
59  if (bma != nullptr)
60  result = makeBMALeg(data, bma, openEndDateReplacement);
61  else
62  result = makeIborLeg(data, index, engineFactory, true, openEndDateReplacement);
63  }
64  std::map<std::string, std::string> qlToOREIndexNames;
65  applyIndexing(result, data, engineFactory, qlToOREIndexNames, requiredFixings, openEndDateReplacement);
66  qlToOREIndexNames[index->name()] = indexName;
67  if (engineFactory->iborFallbackConfig().isIndexReplaced(indexName)) {
68  auto rfrName = engineFactory->iborFallbackConfig().fallbackData(indexName).rfrIndex;
69  // we don't support convention based rfr fallback indices, with ore ticket 1758 this might change
70  qlToOREIndexNames[parseIborIndex(rfrName)->name()] = rfrName;
71  }
72 
73  addToRequiredFixings(result, boost::make_shared<FixingDateGetter>(requiredFixings, qlToOREIndexNames));
74 
75  // handle fx resetting Ibor leg
76 
77  if (data.legType() == "Floating" && !data.isNotResetXCCY()) {
78  QL_REQUIRE(!data.fxIndex().empty(), "FloatingRateLegBuilder: need fx index for fx resetting leg");
79  auto fxIndex = buildFxIndex(data.fxIndex(), data.currency(), data.foreignCurrency(), engineFactory->market(),
80  configuration, data.fixingCalendar(), data.fixingDays(), true);
81 
82  // If the domestic notional value is not specified, i.e. there are no notionals specified in the leg
83  // data, then all coupons including the first will be FX linked. If the first coupon's FX fixing date
84  // is in the past, a FX fixing will be used to determine the first domestic notional. If the first
85  // coupon's FX fixing date is in the future, the first coupon's domestic notional will be determined
86  // by the FX forward rate on that future fixing date.
87 
88  Size j = 0;
89  if (data.notionals().size() == 0) {
90  DLOG("Building FX Resettable with unspecified domestic notional");
91  } else {
92  // First coupon a plain floating rate coupon i.e. it is not FX linked because the initial notional is
93  // known. But, we need to add it to additionalLegs_ so that we don't miss the first coupon's ibor fixing
94  LOG("Building FX Resettable with first domestic notional specified explicitly");
95  j = 1;
96  }
97 
98  // Make the necessary FX linked floating rate coupons
99  for (; j < result.size(); ++j) {
100  boost::shared_ptr<FloatingRateCoupon> coupon = boost::dynamic_pointer_cast<FloatingRateCoupon>(result[j]);
101 
102  Date fixingDate = fxIndex->fixingCalendar().advance(coupon->accrualStartDate(),
103  -static_cast<Integer>(fxIndex->fixingDays()), Days);
104  boost::shared_ptr<FloatingRateFXLinkedNotionalCoupon> fxLinkedCoupon =
105  boost::make_shared<FloatingRateFXLinkedNotionalCoupon>(fixingDate, data.foreignAmount(), fxIndex,
106  coupon);
107  // set the same pricer
108  fxLinkedCoupon->setPricer(coupon->pricer());
109  result[j] = fxLinkedCoupon;
110 
111  // Add the FX fixing to the required fixings
112  requiredFixings.addFixingDate(fixingDate, data.fxIndex(), fxLinkedCoupon->date());
113  }
114  }
115 
116  return result;
117 }

155  addToRequiredFixings(result,
156  boost::make_shared<FixingDateGetter>(
157  requiredFixings, std::map<string, string>{{index->name(), inflationIndexName}}));
158  }
159  return result;

240 Leg EquityLegBuilder::buildLeg(const LegData& data, const boost::shared_ptr<EngineFactory>& engineFactory,
241  RequiredFixings& requiredFixings, const string& configuration,
242  const QuantLib::Date& openEndDateReplacement) const {
243  auto eqData = boost::dynamic_pointer_cast<EquityLegData>(data.concreteLegData());
244  QL_REQUIRE(eqData, "Wrong LegType, expected Equity");
245  string eqName = eqData->eqName();
246  auto eqCurve = *engineFactory->market()->equityCurve(eqName, configuration);
247 
248  Currency dataCurrency = parseCurrencyWithMinors(data.currency());
249  Currency eqCurrency;
250  // Set the equity currency if provided
251  if (!eqData->eqCurrency().empty())
252  eqCurrency = parseCurrencyWithMinors(eqData->eqCurrency());
253 
254  if (eqCurve->currency().empty()) {
255  WLOG("No equity currency set in EquityIndex for equity " << eqCurve->name());
256  } else {
257  // check if it equity currency not set, use from market else check if it matches what is in market
258  if (!eqCurrency.empty())
259  QL_REQUIRE(eqCurve->currency() == eqCurrency,
260  "Equity Currency provided does not match currency of Equity Curve");
261  else
262  eqCurrency = eqCurve->currency();
263  }
264 
265  boost::shared_ptr<QuantExt::FxIndex> fxIndex = nullptr;
266  // if equity currency differs from the leg currency we need an FxIndex
267  if (!eqCurrency.empty() && dataCurrency != eqCurrency) {
268  QL_REQUIRE(eqData->fxIndex() != "",
269  "No FxIndex - if equity currency differs from leg currency an FxIndex must be provided");
270 
271  // An extra check, this ensures that the equity currency provided to use in the FX Index, matches that in
272  // equity curves in the market, this is required as future cashflows will be in the equity curve currency
273  if (!eqCurve->currency().empty() && !eqCurrency.empty()) {
274  QL_REQUIRE(eqCurve->currency() == eqCurrency,
275  "Equity Currency provided does not match currency of Equity Curve");
276  }
277 
278  fxIndex = buildFxIndex(eqData->fxIndex(), data.currency(), eqCurrency.code(), engineFactory->market(),
279  configuration, eqData->fxIndexCalendar(), eqData->fxIndexFixingDays());
280  }
281 
282  Leg result = makeEquityLeg(data, eqCurve, fxIndex, openEndDateReplacement);
283  addToRequiredFixings(
284  result, boost::make_shared<FixingDateGetter>(
285  requiredFixings,
286  std::map<string, string>{{eqCurve->name(), "EQ-" + eqName},
287  {fxIndex != nullptr ? fxIndex->name() : "na", eqData->fxIndex()}}));
288  return result;
289 }
