85//! Engine Builder base class for Cross Currency Swaps
86/*! Pricing engines are cached by currencies (represented as a string list)
87
88 \ingroup builders
89*/
90class CrossCurrencySwapEngineBuilderBase
91 : public CachingPricingEngineBuilder<string, const std::vector<Currency>&, const Currency&> {
92public:
93 CrossCurrencySwapEngineBuilderBase(const std::string& model, const std::string& engine)
94 : CachingEngineBuilder(model, engine, {"CrossCurrencySwap"}) {}
95
96protected:
97 virtual string keyImpl(const std::vector<Currency>& ccys, const Currency& base) override {
98 std::ostringstream ccyskey;
99 ccyskey << base << "/";
100 for (Size i = 0; i < ccys.size(); ++i)
101 ccyskey << ccys[i] << ((i < ccys.size() - 1) ? "-" : "");
102 return ccyskey.str();
103 }
104};

106//! Discounted Cashflows Engine Builder for Cross Currency Swaps
107class CrossCurrencySwapEngineBuilder : public CrossCurrencySwapEngineBuilderBase {
108public:
109 CrossCurrencySwapEngineBuilder()
110 : CrossCurrencySwapEngineBuilderBase("DiscountedCashflows", "DiscountingCrossCurrencySwapEngine") {}
111
112protected:
113 virtual boost::shared_ptr<PricingEngine> engineImpl(const std::vector<Currency>& ccys,
114 const Currency& base) override {
115
116 std::vector<Handle<YieldTermStructure>> discountCurves;
117 std::vector<Handle<Quote>> fxQuotes;
118 std::string config = configuration(MarketContext::pricing);
119 std::string baseCcyCode = base.code();
120 for (Size i = 0; i < ccys.size(); ++i) {
121 string ccyCode = ccys[i].code();
122 discountCurves.push_back(xccyYieldCurve(market_, ccyCode, config));
123 string pair = ccyCode + baseCcyCode;
124 fxQuotes.push_back(market_->fxRate(pair, config));
125 }
126
127 return boost::make_shared<QuantExt::DiscountingCurrencySwapEngine>(discountCurves, fxQuotes, ccys, base);
128 }
129};
