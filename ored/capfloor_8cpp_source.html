45 void CapFloor::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
46 
47  DLOG("CapFloor::build() called for trade " << id() << ", leg type is " << legData_.legType());
48 
49  QL_REQUIRE((legData_.legType() == "Floating") || (legData_.legType() == "CMS") ||
50  (legData_.legType() == "DurationAdjustedCMS") || (legData_.legType() == "CMSSpread") || (legData_.legType() == "CPI") ||
51  (legData_.legType() == "YY"),
52  "CapFloor build error, LegType must be Floating, CMS, DurationAdjustedCMS, CMSSpread, CPI or YY");
53 
54  QL_REQUIRE(caps_.size() > 0 || floors_.size() > 0, "CapFloor build error, no cap rates or floor rates provided");
55  QuantLib::CapFloor::Type capFloorType;
56  if (floors_.size() == 0) {
57  capFloorType = QuantLib::CapFloor::Cap;
58  } else if (caps_.size() == 0) {
59  capFloorType = QuantLib::CapFloor::Floor;
60  } else {
61  capFloorType = QuantLib::CapFloor::Collar;
62  }
63 
64  legs_.clear();
65  boost::shared_ptr<EngineBuilder> builder;
66  std::string underlyingIndex, qlIndexName;
67  boost::shared_ptr<QuantLib::Instrument> qlInstrument;
68 
69  // Account for long / short multiplier. In the following we expect the qlInstrument to be set up
70  // as a long cap resp. a long floor resp. as a collar which by definition is a long cap + short floor
71  // (this is opposite to the definition of a leg with naked option = true!)
72  // The isPayer flag in the leg data is ignored.
73  Real multiplier = (parsePositionType(longShort_) == Position::Long ? 1.0 : -1.0);
74 
75  if (legData_.legType() == "Floating") {
76 
77  boost::shared_ptr<FloatingLegData> floatData =
78  boost::dynamic_pointer_cast<FloatingLegData>(legData_.concreteLegData());
79  QL_REQUIRE(floatData, "Wrong LegType, expected Floating, got " << legData_.legType());
80  underlyingIndex = floatData->index();
81  Handle<IborIndex> hIndex =
82  engineFactory->market()->iborIndex(underlyingIndex, engineFactory->configuration(MarketContext::pricing));
83  QL_REQUIRE(!hIndex.empty(), "Could not find ibor index " << underlyingIndex << " in market.");
84  boost::shared_ptr<IborIndex> index = hIndex.currentLink();
85  qlIndexName = index->name();
86 
87  QL_REQUIRE(floatData->caps().empty() && floatData->floors().empty(),
88  "CapFloor build error, Floating leg section must not have caps and floors");
89 
90  bool isBmaIndex = boost::dynamic_pointer_cast<QuantExt::BMAIndexWrapper>(index) != nullptr;
91 
92  if (!isBmaIndex && !floatData->hasSubPeriods()) {
93  // For the cases where we support caps and floors in the regular way, we build a floating leg with
94  // the nakedOption flag set to true, this avoids maintaining all features in legs with associated
95  // coupon pricers and at the same time in the QuaantLib::CapFloor instrument and pricing engine.
96  // Supported cases are
97  // - Ibor coupon without sub periods (hasSubPeriods = false)
98  // - compounded ON coupon, averaged ON coupon
99  // The other cases are handled below.
100  LegData tmpLegData = legData_;
101  boost::shared_ptr<FloatingLegData> tmpFloatData = boost::make_shared<FloatingLegData>(*floatData);
102  tmpFloatData->floors() = floors_;
103  tmpFloatData->caps() = caps_;
104  tmpFloatData->nakedOption() = true;
105  tmpLegData.concreteLegData() = tmpFloatData;
106  legs_.push_back(engineFactory->legBuilder(tmpLegData.legType())
107  ->buildLeg(tmpLegData, engineFactory, requiredFixings_,
108  engineFactory->configuration(MarketContext::pricing)));
109  // if both caps and floors are given, we have to use a payer leg, since in this case
110  // the StrippedCappedFlooredCoupon used to extract the naked options assumes a long floor
111  // and a short cap while we have documented a collar to be a short floor and long cap
112  qlInstrument =
113  boost::make_shared<QuantLib::Swap>(legs_, std::vector<bool>{!floors_.empty() && !caps_.empty()});
114  qlInstrument->setPricingEngine(
115  boost::make_shared<DiscountingSwapEngine>(engineFactory->market()->discountCurve(legData_.currency())));
116  maturity_ = CashFlows::maturityDate(legs_.front());
117  } else {
118  // For the cases where we don't have regular cap / floor support we treat the index approximately as an Ibor
119  // index and build an QuantLib::CapFloor with associated pricing engine. These cases comprise:
120  // - BMA coupons
121  // - Ibor coupons with sub periods (hasSubPeriods = true)
122 
123  ALOG("CapFloor trade " << id() << " on a) BMA or b) sub periods Ibor (index = '" << underlyingIndex
124  << "') built, will treat the index approximately as an ibor index");
125  builder = engineFactory->builder(tradeType_);
126  legs_.push_back(makeIborLeg(legData_, index, engineFactory));
127 
128  // If a vector of cap/floor rates are provided, ensure they align with the number of schedule periods
129  if (floors_.size() > 1) {
130  QL_REQUIRE(floors_.size() == legs_[0].size(),
131  "The number of floor rates provided does not match the number of schedule periods");
132  }
133 
134  if (caps_.size() > 1) {
135  QL_REQUIRE(caps_.size() == legs_[0].size(),
136  "The number of cap rates provided does not match the number of schedule periods");
137  }
138 
139  // If one cap/floor rate is given, extend the vector to align with the number of schedule periods
140  if (floors_.size() == 1)
141  floors_.resize(legs_[0].size(), floors_[0]);
142 
143  if (caps_.size() == 1)
144  caps_.resize(legs_[0].size(), caps_[0]);
145 
146  // Create QL CapFloor instrument
147  qlInstrument = boost::make_shared<QuantLib::CapFloor>(capFloorType, legs_[0], caps_, floors_);
148 
149  boost::shared_ptr<CapFloorEngineBuilder> capFloorBuilder =
150  boost::dynamic_pointer_cast<CapFloorEngineBuilder>(builder);
151  qlInstrument->setPricingEngine(capFloorBuilder->engine(underlyingIndex));
152 
153  maturity_ = boost::dynamic_pointer_cast<QuantLib::CapFloor>(qlInstrument)->maturityDate();
154  }
155 
156  } else if (legData_.legType() == "CMS") {
157  builder = engineFactory->builder("Swap");

184  boost::make_shared<DiscountingSwapEngine>(engineFactory->market()->discountCurve(legData_.currency())));
185  maturity_ = CashFlows::maturityDate(legs_.front());

187  } else if (legData_.legType() == "DurationAdjustedCMS") {
188  auto cmsData = boost::dynamic_pointer_cast<DurationAdjustedCmsLegData>(legData_.concreteLegData());
189  QL_REQUIRE(cmsData, "Wrong LegType, expected DurationAdjustedCmsLegData");
190  LegData tmpLegData = legData_;
191  auto tmpCmsData = boost::make_shared<DurationAdjustedCmsLegData>(*cmsData);
192  tmpCmsData->floors() = floors_;
193  tmpCmsData->caps() = caps_;
194  tmpCmsData->nakedOption() = true;

201  // and a short cap while we have documented a collar to be a short floor and long cap

203  qlInstrument->setPricingEngine(
204  boost::make_shared<DiscountingSwapEngine>(engineFactory->market()->discountCurve(legData_.currency())));
205  maturity_ = CashFlows::maturityDate(legs_.front());
206  }
207  else if (legData_.legType() == "CMSSpread") {
208  builder = engineFactory->builder("Swap");
209  boost::shared_ptr<CMSSpreadLegData> cmsSpreadData = boost::dynamic_pointer_cast<CMSSpreadLegData>(legData_.concreteLegData());
210  QL_REQUIRE(cmsSpreadData, "Wrong LegType, expected CMSSpread");
211  LegData tmpLegData = legData_;
212  boost::shared_ptr<CMSSpreadLegData> tmpFloatData = boost::make_shared<CMSSpreadLegData>(*cmsSpreadData);
213  tmpFloatData->floors() = floors_;
214  tmpFloatData->caps() = caps_;
215  tmpFloatData->nakedOption() = true;
216  tmpLegData.concreteLegData() = tmpFloatData;
217  legs_.push_back(engineFactory->legBuilder(tmpLegData.legType())
218  ->buildLeg(tmpLegData, engineFactory, requiredFixings_,
219  engineFactory->configuration(MarketContext::pricing)));
220  // if both caps and floors are given, we have to use a payer leg, since in this case
221  // the StrippedCappedFlooredCoupon used to extract the naked options assumes a long floor
222  // and a short cap while we have documented a collar to be a short floor and long cap
223  qlInstrument = boost::make_shared<QuantLib::Swap>(legs_, std::vector<bool>{!floors_.empty() && !caps_.empty()});
224  qlInstrument->setPricingEngine(
225  boost::make_shared<DiscountingSwapEngine>(engineFactory->market()->discountCurve(legData_.currency())));
226  maturity_ = CashFlows::maturityDate(legs_.front());
227  }
228  else if (legData_.legType() == "CPI") {
229  DLOG("CPI CapFloor Type " << capFloorType << " ID " << id());
230 
231  builder = engineFactory->builder("CpiCapFloor");
232 
233  boost::shared_ptr<CPILegData> cpiData = boost::dynamic_pointer_cast<CPILegData>(legData_.concreteLegData());
234  QL_REQUIRE(cpiData, "Wrong LegType, expected CPI");
235 
236  underlyingIndex = cpiData->index();
237  Handle<ZeroInflationIndex> zeroIndex = engineFactory->market()->zeroInflationIndex(
238  underlyingIndex, builder->configuration(MarketContext::pricing));
239  qlIndexName = zeroIndex->name();
240 
241  // the cpi leg uses the first schedule date as the start date, which only makes sense if there are at least
242  // two dates in the schedule, otherwise the only date in the schedule is the pay date of the cf and a a separate
243  // start date is expected; if both the separate start date and a schedule with more than one date is given
244  Date startDate;
245  Schedule schedule = makeSchedule(legData_.schedule());
246 
247  const string& start = cpiData->startDate();
248  if (schedule.size() < 2) {
249  QL_REQUIRE(!start.empty(), "Only one schedule date, a 'StartDate' must be given.");
250  startDate = parseDate(start);
251  } else if (!start.empty()) {
252  DLOG("Schedule with more than 2 dates was provided. The first schedule date "
253  << io::iso_date(schedule.dates().front()) << " is used as the start date. The 'StartDate' of " << start
254  << " is not used.");
255  startDate = schedule.dates().front();
256  }
257 
258  Real baseCPI = cpiData->baseCPI();
259 
260  boost::shared_ptr<InflationSwapConvention> cpiSwapConvention = nullptr;
261 
262  auto inflationConventions =
263  InstrumentConventions::instance().conventions()->get(underlyingIndex + "_INFLATIONSWAP", Convention::Type::InflationSwap);
264 
265  if (inflationConventions.first)
266  cpiSwapConvention = boost::dynamic_pointer_cast<InflationSwapConvention>(inflationConventions.second);
267 
268  Period observationLag;
269  if (cpiData->observationLag().empty()) {
270  QL_REQUIRE(cpiSwapConvention, "observationLag is not specified in legData and couldn't find convention for "
271  << underlyingIndex
272  << ". Please add field to trade xml or add convention");
273  DLOG("Build CPI Leg and use observation lag from standard inflationswap convention");
274  observationLag = cpiSwapConvention->observationLag();
275  } else {
276  observationLag = parsePeriod(cpiData->observationLag());
277  }
278 
279  CPI::InterpolationType interpolationMethod;
280  if (cpiData->interpolation().empty()) {
281  QL_REQUIRE(cpiSwapConvention, "observationLag is not specified in legData and couldn't find convention for "
282  << underlyingIndex
283  << ". Please add field to trade xml or add convention");
284  DLOG("Build CPI Leg and use observation lag from standard inflationswap convention");
285  interpolationMethod = cpiSwapConvention->interpolated() ? CPI::Linear : CPI::Flat;
286  } else {
287  interpolationMethod = parseObservationInterpolation(cpiData->interpolation());
288  }
289 
290  Calendar cal = zeroIndex->fixingCalendar();
291  BusinessDayConvention conv = Unadjusted; // not used in the CPI CapFloor engine
292 
293  QL_REQUIRE(!zeroIndex.empty(), "Zero Inflation Index is empty");
294 
295  legs_.push_back(makeCPILeg(legData_, zeroIndex.currentLink(), engineFactory));
296 
297  // If a vector of cap/floor rates are provided, ensure they align with the number of schedule periods
298  if (floors_.size() > 1) {
299  QL_REQUIRE(floors_.size() == legs_[0].size(),
300  "The number of floor rates provided does not match the number of schedule periods");
301  }
302 
303  if (caps_.size() > 1) {
304  QL_REQUIRE(caps_.size() == legs_[0].size(),
305  "The number of cap rates provided does not match the number of schedule periods");
306  }
307 
308  // If one cap/floor rate is given, extend the vector to align with the number of schedule periods
309  if (floors_.size() == 1)
310  floors_.resize(legs_[0].size(), floors_[0]);
311 
312  if (caps_.size() == 1)
313  caps_.resize(legs_[0].size(), caps_[0]);
314 
315  boost::shared_ptr<CpiCapFloorEngineBuilder> capFloorBuilder =
316  boost::dynamic_pointer_cast<CpiCapFloorEngineBuilder>(builder);
317 
318  // Create QL CPI CapFloor instruments and add to a composite
319  qlInstrument = boost::make_shared<CompositeInstrument>();
320  maturity_ = Date::minDate();
321  for (Size i = 0; i < legs_[0].size(); ++i) {
322  DLOG("Create composite " << i);
323  Real nominal, gearing;
324  Date paymentDate;
325  boost::shared_ptr<CPICoupon> coupon = boost::dynamic_pointer_cast<CPICoupon>(legs_[0][i]);
326  boost::shared_ptr<CPICashFlow> cashflow = boost::dynamic_pointer_cast<CPICashFlow>(legs_[0][i]);

330  paymentDate = coupon->date();
331  } else if (cashflow) {

335  } else {

337  }
338

346  }

351  boost::shared_ptr<CPICapFloor> capfloor =
352  boost::make_shared<CPICapFloor>(Option::Put, nominal, startDate, baseCPI, paymentDate, cal, conv, cal, conv, floors_[i], zeroIndex,
353  observationLag, interpolationMethod);
354  capfloor->setPricingEngine(capFloorBuilder->engine(underlyingIndex));
355  boost::dynamic_pointer_cast<QuantLib::CompositeInstrument>(qlInstrument)->add(capfloor, sign * gearing);
356  maturity_ = std::max(maturity_, capfloor->payDate());
357  }
358  }

407  } else if (capFloorType == QuantLib::CapFloor::Floor) {

409  } else if (capFloorType == QuantLib::CapFloor::Collar) {
410  qlInstrument = boost::shared_ptr<YoYInflationCapFloor>(
411  new YoYInflationCapFloor(QuantLib::YoYInflationCapFloor::Collar, legs_[0], caps_, floors_));
412  } else {
413  QL_FAIL("unknown YoYInflation cap/floor type");
414  }
415 
416  boost::shared_ptr<YoYCapFloorEngineBuilder> capFloorBuilder =
417  boost::dynamic_pointer_cast<YoYCapFloorEngineBuilder>(builder);
418  qlInstrument->setPricingEngine(capFloorBuilder->engine(underlyingIndex));
419 
420  // Wrap the QL instrument in a vanilla instrument
421 
422  maturity_ = boost::dynamic_pointer_cast<QuantLib::YoYInflationCapFloor>(qlInstrument)->maturityDate();
423  } else {
424  QL_FAIL("Invalid legType " << legData_.legType() << " for CapFloor");
425  }
426

428

430 
431  legCurrencies_.push_back(legData_.currency());
432  legPayers_.push_back(false); // already accounted for via the instrument multiplier
433  npvCurrency_ = legData_.currency();
434  notionalCurrency_ = legData_.currency();

436

438

440  std::vector<Real> additionalMultipliers;

442  -multiplier, parseCurrency(legData_.currency()), engineFactory,
443  engineFactory->configuration(MarketContext::pricing)));
444

446  instrument_ =

448 
449  // add required fixings

451  std::map<string, string> indexNameMapper = {{qlIndexName, underlyingIndex}};
452  if (engineFactory->iborFallbackConfig().isIndexReplaced(underlyingIndex)) {
453  string rfrIndexName = engineFactory->iborFallbackConfig().fallbackData(underlyingIndex).rfrIndex;
454  // we don't support convention based rfr fallback indices, with ore ticket 1758 this might change
455  indexNameMapper[parseIborIndex(rfrIndexName)->name()] = rfrIndexName;
456  }

460  }
461

464  if (!l.empty()) {
465  startDate = std::min(startDate, l.front()->date());
466  boost::shared_ptr<Coupon> coupon = boost::dynamic_pointer_cast<Coupon>(l.front());
467  if (coupon)
468  startDate = std::min(startDate, coupon->accrualStartDate());
469  }
470  }
471 
472  additionalData_["startDate"] = to_string(startDate);

475 const std::map<std::string, boost::any>& CapFloor::additionalData() const {
476  // use the build time as of date to determine current notionals
477  Date asof = Settings::instance().evaluationDate();
478 
479  additionalData_["legType"] = legData_.legType();
480  additionalData_["isPayer"] = legData_.isPayer();
481  additionalData_["notionalCurrency"] = legData_.currency();

483  boost::shared_ptr<CashFlow> flow = legs_[0][j];
