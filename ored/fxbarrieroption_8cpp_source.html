46boost::shared_ptr<QuantLib::PricingEngine>
47FxBarrierOption::vanillaPricigingEngine(const boost::shared_ptr<EngineFactory>& ef, const QuantLib::Date& expiryDate,
48 const QuantLib::Date& paymentDate) {
49
50 if (paymentDate > expiryDate) {
51 boost::shared_ptr<EngineBuilder> builder = ef->builder("FxOptionEuropeanCS");
52 QL_REQUIRE(builder, "No builder found for FxOptionEuropeanCS");
53
54 boost::shared_ptr<FxEuropeanCSOptionEngineBuilder> fxOptBuilder =
55 boost::dynamic_pointer_cast<FxEuropeanCSOptionEngineBuilder>(builder);
56 QL_REQUIRE(fxOptBuilder, "No FxEuropeanCSOptionEngineBuilder found");
57
58 return fxOptBuilder->engine(parseCurrency(boughtCurrency_), parseCurrency(soldCurrency_), paymentDate);
59 } else {
60 boost::shared_ptr<EngineBuilder> builder = ef->builder("FxOption");
61 QL_REQUIRE(builder, "No builder found for FxOption");
62
63 boost::shared_ptr<FxEuropeanOptionEngineBuilder> fxOptBuilder =
64 boost::dynamic_pointer_cast<FxEuropeanOptionEngineBuilder>(builder);
65 QL_REQUIRE(fxOptBuilder, "No FxEuropeanOptionEngineBuilder found");
66
67 return fxOptBuilder->engine(parseCurrency(boughtCurrency_), parseCurrency(soldCurrency_), expiryDate);
68 }
69}

71boost::shared_ptr<QuantLib::PricingEngine>
72FxBarrierOption::barrierPricigingEngine(const boost::shared_ptr<EngineFactory>& ef, const QuantLib::Date& expiryDate,
73 const QuantLib::Date& paymentDate) {
74
75 boost::shared_ptr<EngineBuilder> builder = ef->builder(tradeType_);
76 QL_REQUIRE(builder, "No builder found for " << tradeType_);
77
78 boost::shared_ptr<FxBarrierOptionEngineBuilder> fxBarrierOptBuilder =
79 boost::dynamic_pointer_cast<FxBarrierOptionEngineBuilder>(builder);
80 QL_REQUIRE(fxBarrierOptBuilder, "No FxBarrierOptionEngineBuilder found");
81
82 return fxBarrierOptBuilder->engine(parseCurrency(boughtCurrency_), parseCurrency(soldCurrency_), expiryDate, paymentDate);
83}
