59void ModelParameter::fromXML(XMLNode* node) {
60 calibrate_ = XMLUtils::getChildValueAsBool(node, "Calibrate", true);
61 type_ = parseParamType(XMLUtils::getChildValue(node, "ParamType", true));
62 values_ = XMLUtils::getChildrenValuesAsDoublesCompact(node, "InitialValue", true);
63 if (type_ != ParamType::Constant) {
64 times_ = XMLUtils::getChildrenValuesAsDoublesCompact(node, "TimeGrid", true);
65 }
66 check();
67}
68
69void ModelParameter::append(XMLDocument& doc, XMLNode* node) const {
70 XMLUtils::addChild(doc, node, "Calibrate", calibrate_);
71 XMLUtils::addGenericChild(doc, node, "ParamType", type_);
72 XMLUtils::addGenericChildAsList(doc, node, "TimeGrid", times_);
73 XMLUtils::addGenericChildAsList(doc, node, "InitialValue", values_);
74}
75
76void ModelParameter::check() const {
77 if (type_ == ParamType::Constant) {
78 QL_REQUIRE(values_.size() == 1, "Parameter type is Constant so expecting a single InitialValue.");
79 QL_REQUIRE(times_.empty(), "Parameter type is Constant so expecting an empty time vector.");
80 } else if (type_ == ParamType::Piecewise) {
81 QL_REQUIRE(values_.size() == times_.size() + 1, "Parameter type is Piecewise so expecting the size of the " <<
82 "InitialValue vector (" << values_.size() << ") to be one greater than size of time vector (" <<
83 times_.size() << ").");
84 }
85}
