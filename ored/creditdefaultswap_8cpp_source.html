34void CreditDefaultSwap::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
35 DLOG("CreditDefaultSwap::build() called for trade " << id());
36
37 const boost::shared_ptr<Market> market = engineFactory->market();
38 boost::shared_ptr<EngineBuilder> builder = engineFactory->builder("CreditDefaultSwap");
39
40 auto legData = swap_.leg(); // copy
41 const auto& notionals = swap_.leg().notionals();
42
43 QL_REQUIRE(legData.legType() == "Fixed", "CreditDefaultSwap requires Fixed leg");
44 Schedule schedule = makeSchedule(legData.schedule());
45 QL_REQUIRE(schedule.size() > 1, "CreditDefaultSwap requires at least two dates in the schedule");
46
47 BusinessDayConvention payConvention = Following;
48 if (!legData.paymentConvention().empty()) {
49 payConvention = parseBusinessDayConvention(legData.paymentConvention());
50 }
51
52 Protection::Side prot = legData.isPayer() ? Protection::Side::Buyer : Protection::Side::Seller;
53 notional_ = notionals.front();
54
55 DayCounter dc = Actual360();
56 if (!legData.dayCounter().empty()) {
57 dc = parseDayCounter(legData.dayCounter());
58 }
59
60 // In general for CDS and CDS index trades, the standard day counter is Actual/360 and the final
61 // period coupon accrual includes the maturity date.
62 DayCounter lastPeriodDayCounter;
63 auto strLpdc = legData.lastPeriodDayCounter();
64 if (!strLpdc.empty()) {
65 lastPeriodDayCounter = parseDayCounter(strLpdc);
66 } else {
67 Actual360 standardDayCounter;
68 if (dc == standardDayCounter) {
69 lastPeriodDayCounter = dc == standardDayCounter ? Actual360(true) : dc;
70 legData.lastPeriodDayCounter() = "A360 (Incl Last)";
71 } else {
72 lastPeriodDayCounter = dc;
73 }
74 }
75
76 // Build the coupon leg
77 auto legBuilder = engineFactory->legBuilder(legData.legType());
78 auto couponLeg =
79 legBuilder->buildLeg(legData, engineFactory, requiredFixings_, builder->configuration(MarketContext::pricing));
80 // for the accrual rebate calculation we may need historical coupons that are already paid
81 requiredFixings_.unsetPayDates();
82
83 /* If we have an indexed leg we don't allow for an upfront fee, since we would need to index that
84 as well, btu the ql instrument / engine cdoes not support this currently */
85 QL_REQUIRE(legData.indexing().empty() || swap_.upfrontFee() == Null<Real>(),
86 "CreditDefaultSwap with indexed coupon leg does not allow for an upfront fee");
87
88 /* The rate is really only used to compute the fair spread in the add results and we support that only
89 for fixed coupons with a sinlge rate, otherwise we set this rate to zero */
90 Real fixedRate = 0.0;
91 if (auto fixedData = boost::dynamic_pointer_cast<FixedLegData>(legData.concreteLegData())) {
92 if (fixedData->rates().size() == 1)
93 fixedRate = fixedData->rates().front();
94 }
95
96 boost::shared_ptr<QuantLib::CreditDefaultSwap> cds;
97
98 if (swap_.upfrontFee() == Null<Real>()) {
99 cds = boost::make_shared<QuantLib::CreditDefaultSwap>(
100 prot, notional_, couponLeg, fixedRate, schedule, payConvention, dc, swap_.settlesAccrual(),
101 swap_.protectionPaymentTime(), swap_.protectionStart(), boost::shared_ptr<Claim>(), lastPeriodDayCounter,
102 swap_.rebatesAccrual(), swap_.tradeDate(), swap_.cashSettlementDays());
103 } else {
104 cds = boost::make_shared<QuantLib::CreditDefaultSwap>(
105 prot, notional_, couponLeg, swap_.upfrontFee(), fixedRate, schedule, payConvention, dc,
106 swap_.settlesAccrual(), swap_.protectionPaymentTime(), swap_.protectionStart(), swap_.upfrontDate(),
107 boost::shared_ptr<Claim>(), lastPeriodDayCounter, swap_.rebatesAccrual(), swap_.tradeDate(),
108 swap_.cashSettlementDays());
109 }
110
111 boost::shared_ptr<CreditDefaultSwapEngineBuilder> cdsBuilder =
112 boost::dynamic_pointer_cast<CreditDefaultSwapEngineBuilder>(builder);
113
114 npvCurrency_ = legData.currency();
115
116 QL_REQUIRE(cdsBuilder, "No Builder found for CreditDefaultSwap: " << id());
117 cds->setPricingEngine(cdsBuilder->engine(parseCurrency(npvCurrency_), swap_.creditCurveId(), swap_.recoveryRate()));
118
119 instrument_.reset(new VanillaInstrument(cds));
120
121 maturity_ = cds->coupons().back()->date();
122
123 legs_ = {cds->coupons()};
124 legCurrencies_ = {npvCurrency_};
125 legPayers_ = {legData.isPayer()};
126 notionalCurrency_ = legData.currency();
127
128 if (swap_.protectionStart() != Date())
129 additionalData_["startDate"] = to_string(swap_.protectionStart());
130 else
131 additionalData_["startDate"] = to_string(schedule.dates().front());
132
133 issuer_ = swap_.issuerId();
134
135 // ISDA taxonomy
136 additionalData_["isdaAssetClass"] = string("Credit");
137 additionalData_["isdaBaseProduct"] = string("Single Name");
138 // set isdaSubProduct to entityType in credit reference data
139 additionalData_["isdaSubProduct"] = string("");
140 string entity = swap_.referenceInformation() ?
141 swap_.referenceInformation()->id() :
142 swap_.creditCurveId();
143 boost::shared_ptr<ReferenceDataManager> refData = engineFactory->referenceData();
144 if (refData && refData->hasData("Credit", entity)) {
145 auto refDatum = refData->getData("Credit", entity);
146 boost::shared_ptr<CreditReferenceDatum> creditRefDatum = boost::dynamic_pointer_cast<CreditReferenceDatum>(refDatum);
147 additionalData_["isdaSubProduct"] = creditRefDatum->creditData().entityType;
148 if (creditRefDatum->creditData().entityType == "") {
149 ALOG("EntityType is blank in credit reference data for entity " << entity);
150 }
151 } else {
152 ALOG("Credit reference data missing for entity " << entity << ", isdaSubProduct left blank");
153 }
154 // skip the transaction level mapping for now
155 additionalData_["isdaTransaction"] = string("");
156}

158const std::map<std::string, boost::any>& CreditDefaultSwap::additionalData() const {
159 setLegBasedAdditionalData(0, 2);
160 additionalData_["legNPV[1]"] = instrument_->qlInstrument()->result<Real>("protectionLegNPV");
161 additionalData_["legNPV[2]"] = instrument_->qlInstrument()->result<Real>("premiumLegNPVDirty") +
162 instrument_->qlInstrument()->result<Real>("upfrontPremiumNPV") +
163 instrument_->qlInstrument()->result<Real>("accrualRebateNPV");
164 additionalData_["isPayer[1]"] = !swap_.leg().isPayer();
165 additionalData_["isPayer[2]"] = swap_.leg().isPayer();
166 additionalData_["legType[2]"] = swap_.leg().legType();
167 additionalData_["legType[1]"] = std::string("Protection");
168 additionalData_["currentNotional[1]"] = additionalData_["currentNotional[2]"];
169 additionalData_["originalNotional[1]"] = additionalData_["originalNotional[2]"];
170 additionalData_["notionalCurrency[1]"] = notionalCurrency_;
171 additionalData_["notionalCurrency[2]"] = notionalCurrency_;
172 return additionalData_;
173}

175QuantLib::Real CreditDefaultSwap::notional() const {
176 Date asof = Settings::instance().evaluationDate();
177 // get the current notional from CDS premium leg
178 for (Size i = 0; i < legs_[0].size(); ++i) {
179 boost::shared_ptr<Coupon> coupon = boost::dynamic_pointer_cast<Coupon>(legs_[0][i]);
180 if (coupon->date() > asof)
181 return coupon->nominal();
182 }
183
184 // if no coupons are given, return the initial notional by default
185 return notional_;
186}
