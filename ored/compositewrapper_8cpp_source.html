121 BOOST_AUTO_TEST_CASE(testCompositeInstrumentWrapperPrice) {
122  CommonVars vars;
123  boost::shared_ptr<CurveConfigurations> curveConfig = boost::make_shared<CurveConfigurations>();
124  boost::shared_ptr<Conventions> conventions = boost::make_shared<Conventions>();
125  boost::shared_ptr<TodaysMarketParameters> todaysMarketConfig = boost::make_shared<TodaysMarketParameters>();
126  boost::shared_ptr<EngineData> pricingEngineConfig = boost::make_shared<EngineData>();
127  boost::shared_ptr<Portfolio> portfolio = boost::make_shared<Portfolio>();
128 
129  loadFromXMLString(*curveConfig, vars.curveConfig);
130  loadFromXMLString(*conventions, vars.conventions);
131  InstrumentConventions::instance().setConventions(conventions);
132 
133  loadFromXMLString(*todaysMarketConfig, vars.todaysMarketConfig);
134  loadFromXMLString(*pricingEngineConfig, vars.pricingEngineConfig);
135 
136  auto tradeFactory = boost::make_shared<TradeFactory>();
137  tradeFactory->addExtraBuilders(getExtraTradeBuilders());
138 
139  portfolio->loadFromXMLString(vars.portfolio, tradeFactory);
140 
141  boost::shared_ptr<Market> market =
142  boost::make_shared<TodaysMarket>(vars.asof, todaysMarketConfig, vars.loader, curveConfig, true);
143  auto configurations = std::map<MarketContext, string>();
144  boost::shared_ptr<EngineFactory> factory =
145  boost::make_shared<EngineFactory>(pricingEngineConfig, market, configurations, getExtraEngineBuilders());
146 
147  BOOST_TEST_MESSAGE("number trades " << portfolio->size());
148  portfolio->build(factory);
149  BOOST_TEST_MESSAGE("number trades " << portfolio->size());
150 
151  Handle<Quote> fx = Handle<Quote>(boost::make_shared<SimpleQuote>(1.0));
152  vector<boost::shared_ptr<InstrumentWrapper>> iw;
153  std::vector<Handle<Quote>> fxRates;
154 
155  Real totalNPV = 0;
156  for (auto t : portfolio->trades()) {
157  Handle<Quote> fx = Handle<Quote>(boost::make_shared<SimpleQuote>(1.0));
158  if (t->npvCurrency() != "USD")
159  fx = factory->market()->fxRate(t->npvCurrency() + "USD");
160  fxRates.push_back(fx);
161  iw.push_back(t->instrument());
162  BOOST_TEST_MESSAGE("NPV " << iw.back()->NPV());
163  BOOST_TEST_MESSAGE("FX " << fxRates.back()->value());
164 
165  totalNPV += iw.back()->NPV() * fxRates.back()->value();
166  }
167  boost::shared_ptr<InstrumentWrapper> instrument =
168  boost::shared_ptr<InstrumentWrapper>(new ore::data::CompositeInstrumentWrapper(iw, fxRates, vars.asof));
169 
170  BOOST_TEST_MESSAGE(instrument->NPV());
171 
172  BOOST_CHECK_CLOSE(instrument->NPV(), totalNPV, 0.01);
173 }
