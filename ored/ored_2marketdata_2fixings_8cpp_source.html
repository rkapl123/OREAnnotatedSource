40 void applyFixings(const vector<Fixing>& fixings, const boost::shared_ptr<data::Conventions>& conventions) {
41  Size count = 0;
42  map<string, boost::shared_ptr<Index>> cache;
43  cpu_timer timer;
44  boost::shared_ptr<Index> index;
45  for (auto& f : fixings) {
46  try {
47  auto it = cache.find(f.name);
48  if (it == cache.end()) {
49  index = parseIndex(f.name, conventions);
50  cache[f.name] = index;
51  } else {
52  index = it->second;
53  }
54  index->addFixing(f.date, f.fixing, true);
55  TLOG("Added fixing for " << f.name << " (" << io::iso_date(f.date) << ") value:" << f.fixing);
56  count++;
57  } catch (const std::exception& e) {
58  WLOG("Error during adding fixing for " << f.name << ": " << e.what());
59  }
60  }
61  timer.stop();
62  LOG("Added " << count << " of " << fixings.size() << " fixings in " << timer.format(default_places, "%w")
63  << " seconds");
64 }

66 void applyDividends(const vector<Fixing>& dividends) {
67  Size count = 0;
68  map<string, boost::shared_ptr<EquityIndex>> cache;
69  cpu_timer timer;
70  boost::shared_ptr<EquityIndex> index;
71  for (auto& f : dividends) {
72  try {
73  auto it = cache.find(f.name);
74  if (it == cache.end()) {
75  index = boost::make_shared<EquityIndex>(f.name, NullCalendar(), Currency());
76  cache[f.name] = index;
77  } else {
78  index = it->second;
79  }
80  index->addDividend(f.date, f.fixing, true);
81  TLOG("Added dividend for " << f.name << " (" << io::iso_date(f.date) << ") value:" << f.fixing);
82  count++;
83  } catch (const std::exception& e) {
84  WLOG("Error during adding dividend for " << f.name << ": " << e.what());
85  }
86  }
87  timer.stop();
88  DLOG("Added " << count << " of " << dividends.size() << " dividends in " << timer.format(default_places, "%w")
