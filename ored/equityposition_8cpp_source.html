48void EquityPosition::build(const boost::shared_ptr<ore::data::EngineFactory>& engineFactory) {
49 DLOG("EquityPosition::build() called for " << id());
50 QL_REQUIRE(!data_.underlyings().empty(), "EquityPosition::build(): no underlyings given");
51 indices_.clear();
52 weights_.clear();
53 fxConversion_.clear();
54
55 std::vector<std::string> currencies;
56 for (auto const& u : data_.underlyings()) {
57 indices_.push_back(
58 *engineFactory->market()->equityCurve(u.name(), engineFactory->configuration(MarketContext::pricing)));
59 weights_.push_back(u.weight());
60 QL_REQUIRE(!indices_.back()->currency().empty(),
61 "did not get currency for equity name '" << u.name() << "', is this set up?");
62 currencies.push_back(indices_.back()->currency().code());
63 }
64
65 // get fx quotes
66
67 isSingleCurrency_ = true;
68 npvCurrency_ = currencies.front();
69 for (auto const& c : currencies) {
70 // we use fxSpot() as opposed to fxRate() here to ensure consistency between NPV() and the fixing of an
71 // equivalent index representing the same basket
72 fxConversion_.push_back(
73 engineFactory->market()->fxSpot(c + npvCurrency_, engineFactory->configuration(MarketContext::pricing)));
74 if (npvCurrency_ != c)
75 isSingleCurrency_ = false;
76 }
77
78 // set instrument
79 auto qlInstr =
80 boost::make_shared<EquityPositionInstrumentWrapper>(data_.quantity(), indices_, weights_, fxConversion_);
81 qlInstr->setPricingEngine(boost::make_shared<EquityPositionInstrumentWrapperEngine>());
82 instrument_ = boost::make_shared<VanillaInstrument>(qlInstr);
83
84 // no sensible way to set these members
85 maturity_ = Date::maxDate();
86 notional_ = Null<Real>();
87 notionalCurrency_ = "";
88
89 // ISDA taxonomy: not a derivative, but define the asset class at least
90 // so that we can determine a TRS asset class that has an EQ position underlying
91 additionalData_["isdaAssetClass"] = string("Equity");
92 additionalData_["isdaBaseProduct"] = string("");
93 additionalData_["isdaSubProduct"] = string("");
94 additionalData_["isdaTransaction"] = string("");
95
96 // leave legs empty
97}

125EquityPositionInstrumentWrapper::EquityPositionInstrumentWrapper(
126 const Real quantity, const std::vector<boost::shared_ptr<QuantExt::EquityIndex2>>& equities,
127 const std::vector<Real>& weights, const std::vector<Handle<Quote>>& fxConversion)
128 : quantity_(quantity), equities_(equities), weights_(weights), fxConversion_(fxConversion) {
129 QL_REQUIRE(equities_.size() == weights_.size(), "EquityPositionInstrumentWrapper: equities size ("
130 << equities_.size() << ") must match weights size ("
131 << weights_.size() << ")");
132 QL_REQUIRE(fxConversion_.empty() || fxConversion_.size() == equities_.size(),
133 "EquityPositionInstrumentWrapper: fxConversion size ("
134 << fxConversion_.size() << ") must match equities size (" << equities_.size() << ")");
135 for (auto const& i : equities_)
136 registerWith(i);
137 for (auto const& f : fxConversion_)
138 registerWith(f);
139 registerWith(npvCcyConversion_);
140}

164void EquityPositionInstrumentWrapperEngine::calculate() const {
165 Real result = 0.0;
166 for (Size i = 0; i < arguments_.equities_.size(); ++i) {
167 Real tmp = arguments_.quantity_ * arguments_.equities_[i]->equitySpot()->value();
168 if (!arguments_.fxConversion_[i].empty()) {
169 tmp *= arguments_.fxConversion_[i]->value();
170 }
171 result += tmp * arguments_.weights_[i];
172 }
173 if (!arguments_.npvCcyConversion_.empty()) {
174 result *= arguments_.npvCcyConversion_->value();
175 }
176 results_.value = result;
177}
