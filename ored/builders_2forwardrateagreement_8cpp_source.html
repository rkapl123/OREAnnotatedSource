51boost::shared_ptr<PricingEngine> LgmAmcFraEngineBuilder::engineImpl(const Currency& ccy) {
52 DLOG("Building AMC Fra engine for ccy " << ccy << " (from externally given CAM)");
53
54 QL_REQUIRE(cam_ != nullptr, "CamAmcFraEngineBuilder::engineImpl: cam is null");
55 Size currIdx = cam_->ccyIndex(ccy);
56 auto lgm = cam_->lgm(currIdx);
57 std::vector<Size> modelIndex(1, cam_->pIdx(CrossAssetModel::AssetType::IR, currIdx));
58
59 // we assume that the given cam has pricing discount curves attached already
60 Handle<YieldTermStructure> discountCurve;
61 return buildMcEngine(lgm, discountCurve, simulationDates_, modelIndex);
62}
