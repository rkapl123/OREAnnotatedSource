98void TodaysMarket::initialise(const Date& asof) {
99
100 std::map<std::string, boost::timer::nanosecond_type> timings;
101 std::map<std::string, Count> counts;
102 boost::timer::cpu_timer timer;
103
104 asof_ = asof;
105
106 calibrationInfo_ = boost::make_shared<TodaysMarketCalibrationInfo>();
107 calibrationInfo_->asof = asof;
108
109 // Fixings
110
111 if (loadFixings_) {
112 // Apply them now in case a curve builder needs them
113 LOG("Todays Market Loading Fixings");
114 timer.start();
115 applyFixings(loader_->loadFixings());
116 timings["1 load fixings"] = timer.elapsed().wall;
117 LOG("Todays Market Loading Fixing done.");
118 }
119
120 // Dividends - apply them now in case a curve builder needs them
121
122 LOG("Todays Market Loading Dividends");
123 timer.start();
124 applyDividends(loader_->loadDividends());
125 timings["2 load dividends"] = timer.elapsed().wall;
126 LOG("Todays Market Loading Dividends done.");
127
128 // Add all FX quotes from the loader to Triangulation
129 timer.start();
130 if (loader_->hasQuotes(asof_)) {
131 std::map<std::string, Handle<Quote>> fxQuotes;
132 for (auto& md : loader_->get(Wildcard("FX/RATE/*"), asof_)) {
133 boost::shared_ptr<FXSpotQuote> q = boost::dynamic_pointer_cast<FXSpotQuote>(md);
134 QL_REQUIRE(q, "Failed to cast " << md->name() << " to FXSpotQuote");
135 fxQuotes[q->unitCcy() + q->ccy()] = q->quote();
136 }
137 fx_ = boost::make_shared<FXTriangulation>(fxQuotes);
138 } else {
139 WLOG("TodaysMarket::Initialise: no quotes available for date " << asof_);
140 return;
141 }
142 timings["3 add all fx quotes"] = timer.elapsed().wall;
143
144 // build the dependency graph for all configurations and build all FX Spots
145 timer.start();
146 DependencyGraph dg(asof_, params_, curveConfigs_, iborFallbackConfig_);
147 map<string, string> buildErrors;
148
149 for (const auto& configuration : params_->configurations()) {
150 // Build the graph of objects to build for the current configuration
151 dg.buildDependencyGraph(configuration.first, buildErrors);
152 }
153 dependencies_ = dg.dependencies();
154 timings["4 build dep graphs"] = timer.elapsed().wall;
155
156 // if market is not build lazily, sort the dependency graph and build the objects
157
158 if (!lazyBuild_) {
159
160 // We need to build all discount curves first, since some curve builds ask for discount
161 // curves from specific configurations
162 timer.start();
163 for (const auto& configuration : params_->configurations()) {
164 map<string, string> discountCurves;
165 if (params_->hasMarketObject(MarketObject::DiscountCurve)) {
166 discountCurves = params_->mapping(MarketObject::DiscountCurve, configuration.first);
167 }
168 for (const auto& dc : discountCurves)
169 require(MarketObject::DiscountCurve, dc.first, configuration.first, true);
170 }
171 timings["6 build " + ore::data::to_string(MarketObject::DiscountCurve)] += timer.elapsed().wall;
172 counts["6 build " + ore::data::to_string(MarketObject::DiscountCurve)].inc();
173
174 for (const auto& configuration : params_->configurations()) {
175
176 LOG("Build objects in TodaysMarket configuration " << configuration.first);
177
178 // Sort the graph topologically
179
180 timer.start();
181 Graph& g = dependencies_[configuration.first];
182 IndexMap index = boost::get(boost::vertex_index, g);
183 std::vector<Vertex> order;
184 try {
185 boost::topological_sort(g, std::back_inserter(order));
186 } catch (const std::exception& e) {
187 // topological_sort() might have produced partial results, that we have to discard
188 order.clear();
189 // set error (most likely a circle), and output cycles if any
190 buildErrors["CurveDependencyGraph"] = "Topological sort of dependency graph failed for configuration " +
191 configuration.first + " (" + ore::data::to_string(e.what()) +
192 "). Got cycle(s): " + getCycles(g);
193 }
194 timings["5 topological sort dep graphs"] += timer.elapsed().wall;
195
196 TLOG("Can build objects in the following order:");
197 for (auto const& m : order) {
198 TLOG("vertex #" << index[m] << ": " << g[m]);
199 }
200
201 // Build the objects in the graph in topological order
202
203 Size countSuccess = 0, countError = 0;
204 for (auto const& m : order) {
205 timer.start();
206 try {
207 buildNode(configuration.first, g[m]);
208 ++countSuccess;
209 DLOG("built node " << g[m] << " in configuration " << configuration.first);
210 } catch (const std::exception& e) {
211 if (g[m].curveSpec)
212 buildErrors[g[m].curveSpec->name()] = e.what();
213 else
214 buildErrors[g[m].name] = e.what();
215 ++countError;
216 ALOG("error while building node " << g[m] << " in configuration " << configuration.first << ": "
217 << e.what());
218 }
219 timings["6 build " + ore::data::to_string(g[m].obj)] += timer.elapsed().wall;
220 counts["6 build " + ore::data::to_string(g[m].obj)].inc();
221 }
222
223 LOG("Loaded CurvesSpecs: success: " << countSuccess << ", error: " << countError);
224 }
225
226 } else {
227 LOG("Build objects in TodaysMarket lazily, i.e. when requested.");
228 }
229
230 // output stats on initialisation phase
231
232 LOG("TodaysMarket build stats:");
233 boost::timer::nanosecond_type sum = 0;
234 for (auto const& t : timings) {
235 std::size_t c = counts[t.first].count == 0 ? 1 : counts[t.first].count;
236 double timing = static_cast<double>(t.second) / 1.0E6;
237 LOG(std::left << std::setw(34) << t.first << ": " << std::right << std::setprecision(3) << std::setw(15)
238 << timing << " ms" << std::setw(10) << c << std::setw(15) << timing / c << " ms");
239 sum += t.second;
240 }
241 LOG("Total build time : " << std::setw(15) << static_cast<double>(sum) / 1.0E6 << " ms");
242
243 // output errors from initialisation phase
244
245 if (!buildErrors.empty()) {
246 for (auto const& error : buildErrors) {
247 ALOG(StructuredCurveErrorMessage(error.first, "Failed to Build Curve", error.second));
248 }
249 if (!continueOnError_) {
250 string errStr;
251 for (auto const& error : buildErrors) {
252 errStr += "(" + error.first + ": " + error.second + "); ";
253 }
254 QL_FAIL("Cannot build all required curves! Building failed for: " << errStr);
255 }
256 }

258} // TodaysMarket::initialise()

260void TodaysMarket::buildNode(const std::string& configuration, Node& node) const {
261
262 // if the node is already built, there is nothing to do
263
264 if (node.built)
265 return;
266
267 if (node.curveSpec == nullptr) {
268
269 // not spec-based node, this can only be a SwapIndexCurve
270
271 QL_REQUIRE(node.obj == MarketObject::SwapIndexCurve,
272 "market object '" << node.obj << "' (" << node.name << ") without curve spec, this is unexpected.");
273 const string& swapIndexName = node.name;
274 const string& discountIndex = node.mapping;
275 addSwapIndex(swapIndexName, discountIndex, configuration);
276 DLOG("Added SwapIndex " << swapIndexName << " with DiscountingIndex " << discountIndex);
277 requiredSwapIndices_[configuration][swapIndexName] =
278 swapIndices_.at(std::make_pair(configuration, swapIndexName)).currentLink();
279
280 } else {
281
282 // spec-based node
283
284 auto spec = node.curveSpec;
285 switch (spec->baseType()) {
286
287 // Yield
288 case CurveSpec::CurveType::Yield: {
289 boost::shared_ptr<YieldCurveSpec> ycspec = boost::dynamic_pointer_cast<YieldCurveSpec>(spec);
290 QL_REQUIRE(ycspec, "Failed to convert spec " << *spec << " to yield curve spec");
291
292 boost::shared_ptr<Conventions> conventions = InstrumentConventions::instance().conventions();
293
294 auto itr = requiredYieldCurves_.find(ycspec->name());
295 if (itr == requiredYieldCurves_.end()) {
296 DLOG("Building YieldCurve for asof " << asof_);
297 boost::shared_ptr<YieldCurve> yieldCurve = boost::make_shared<YieldCurve>(
298 asof_, *ycspec, *curveConfigs_, *loader_, requiredYieldCurves_, requiredDefaultCurves_, *fx_,
299 referenceData_, iborFallbackConfig_, preserveQuoteLinkage_, buildCalibrationInfo_, this);
300 calibrationInfo_->yieldCurveCalibrationInfo[ycspec->name()] = yieldCurve->calibrationInfo();
301 itr = requiredYieldCurves_.insert(make_pair(ycspec->name(), yieldCurve)).first;
302 DLOG("Added YieldCurve \"" << ycspec->name() << "\" to requiredYieldCurves map");
303 if (itr->second->currency().code() != ycspec->ccy()) {
304 WLOG("Warning: YieldCurve has ccy " << itr->second->currency() << " but spec has ccy "
305 << ycspec->ccy());
306 }
307 }
308
309 if (node.obj == MarketObject::DiscountCurve) {
310 DLOG("Adding DiscountCurve(" << node.name << ") with spec " << *ycspec << " to configuration "
311 << configuration);
312 yieldCurves_[make_tuple(configuration, YieldCurveType::Discount, node.name)] = itr->second->handle();
313
314 } else if (node.obj == MarketObject::YieldCurve) {
315 DLOG("Adding YieldCurve(" << node.name << ") with spec " << *ycspec << " to configuration "
316 << configuration);
317 yieldCurves_[make_tuple(configuration, YieldCurveType::Yield, node.name)] = itr->second->handle();
318
319 } else if (node.obj == MarketObject::IndexCurve) {
320 DLOG("Adding Index(" << node.name << ") with spec " << *ycspec << " to configuration "
321 << configuration);
322 // ibor fallback handling
323 auto tmpIndex = parseIborIndex(node.name, itr->second->handle());
324 if (iborFallbackConfig_.isIndexReplaced(node.name, asof_)) {
325 auto fallbackData = iborFallbackConfig_.fallbackData(node.name);
326 boost::shared_ptr<IborIndex> rfrIndex;
327 auto f = iborIndices_.find(make_pair(configuration, fallbackData.rfrIndex));
328 if (f == iborIndices_.end()) {
329 f = iborIndices_.find(make_pair(Market::defaultConfiguration, fallbackData.rfrIndex));
330 QL_REQUIRE(f != iborIndices_.end(),
331 "Failed to build ibor fallback index '"
332 << node.name << "', did not find rfr index '" << fallbackData.rfrIndex
333 << "' in configuration '" << configuration
334 << "' or default - is the rfr index configuration in todays market parameters?");
335 }
336 auto oi = boost::dynamic_pointer_cast<OvernightIndex>(*f->second);
337 QL_REQUIRE(oi,
338 "Found rfr index '"
339 << fallbackData.rfrIndex << "' as fallback for ibor index '" << node.name
340 << "', but this is not an overnight index. Are the fallback rules correct here?");
341 if (auto original = boost::dynamic_pointer_cast<OvernightIndex>(tmpIndex))
342 tmpIndex = boost::make_shared<QuantExt::FallbackOvernightIndex>(
343 original, oi, fallbackData.spread, fallbackData.switchDate,
344 iborFallbackConfig_.useRfrCurveInTodaysMarket());
345 else
346 tmpIndex = boost::make_shared<QuantExt::FallbackIborIndex>(
347 tmpIndex, oi, fallbackData.spread, fallbackData.switchDate,
348 iborFallbackConfig_.useRfrCurveInTodaysMarket());
349 TLOG("built ibor fall back index for '" << node.name << "' in configuration " << configuration
350 << " using rfr index '" << fallbackData.rfrIndex
351 << "', spread " << fallbackData.spread
352 << ", will use rfr curve in t0 market: " << std::boolalpha
353 << iborFallbackConfig_.useRfrCurveInTodaysMarket());
354 }
355 iborIndices_[make_pair(configuration, node.name)] = Handle<IborIndex>(tmpIndex);
356 } else {
357 QL_FAIL("unexpected market object type '"
358 << node.obj << "' for yield curve, should be DiscountCurve, YieldCurve, IndexCurve");
359 }
360 break;
361 }
362
363 // FX Spot
364 case CurveSpec::CurveType::FX: {
365 DLOG("Building FXSpot (" << node.name << ") does not require any action.");
366 break;
367 }
368
369 // FX Vol
370 case CurveSpec::CurveType::FXVolatility: {
371 boost::shared_ptr<FXVolatilityCurveSpec> fxvolspec =
372 boost::dynamic_pointer_cast<FXVolatilityCurveSpec>(spec);
373 QL_REQUIRE(fxvolspec, "Failed to convert spec " << *spec);
374
375 // have we built the curve already ?
376 auto itr = requiredFxVolCurves_.find(fxvolspec->name());
377 if (itr == requiredFxVolCurves_.end()) {
378 DLOG("Building FXVolatility for asof " << asof_);
379 boost::shared_ptr<FXVolCurve> fxVolCurve = boost::make_shared<FXVolCurve>(
380 asof_, *fxvolspec, *loader_, *curveConfigs_, *fx_, requiredYieldCurves_, requiredFxVolCurves_,
381 requiredCorrelationCurves_, buildCalibrationInfo_);
382 calibrationInfo_->fxVolCalibrationInfo[fxvolspec->name()] = fxVolCurve->calibrationInfo();
383 itr = requiredFxVolCurves_.insert(make_pair(fxvolspec->name(), fxVolCurve)).first;
384 }
385
386 DLOG("Adding FXVol (" << node.name << ") with spec " << *fxvolspec << " to configuration "
387 << configuration);
388 fxVols_[make_pair(configuration, node.name)] =
389 Handle<BlackVolTermStructure>(itr->second->volTermStructure());
390 break;
391 }
392
393 // Swaption Vol
394 case CurveSpec::CurveType::SwaptionVolatility: {
395 boost::shared_ptr<SwaptionVolatilityCurveSpec> swvolspec =
396 boost::dynamic_pointer_cast<SwaptionVolatilityCurveSpec>(spec);
397 QL_REQUIRE(swvolspec, "Failed to convert spec " << *spec);
398
399 auto itr = requiredGenericYieldVolCurves_.find(swvolspec->name());
400 if (itr == requiredGenericYieldVolCurves_.end()) {
401 DLOG("Building Swaption Volatility (" << node.name << ") for asof " << asof_);
402 boost::shared_ptr<SwaptionVolCurve> swaptionVolCurve = boost::make_shared<SwaptionVolCurve>(
403 asof_, *swvolspec, *loader_, *curveConfigs_, requiredSwapIndices_[configuration],
404 requiredGenericYieldVolCurves_, buildCalibrationInfo_);
405 calibrationInfo_->irVolCalibrationInfo[swvolspec->name()] = swaptionVolCurve->calibrationInfo();
406 itr = requiredGenericYieldVolCurves_.insert(make_pair(swvolspec->name(), swaptionVolCurve)).first;
407 }
408
409 boost::shared_ptr<SwaptionVolatilityCurveConfig> cfg =
410 curveConfigs_->swaptionVolCurveConfig(swvolspec->curveConfigID());
411
412 DLOG("Adding SwaptionVol (" << node.name << ") with spec " << *swvolspec << " to configuration "
413 << configuration);
414 swaptionCurves_[make_pair(configuration, node.name)] =
415 Handle<SwaptionVolatilityStructure>(itr->second->volTermStructure());
416 swaptionIndexBases_[make_pair(configuration, node.name)] =
417 cfg->proxySourceCurveId().empty()
418 ? make_pair(cfg->shortSwapIndexBase(), cfg->swapIndexBase())
419 : make_pair(cfg->proxyTargetShortSwapIndexBase(), cfg->proxyTargetSwapIndexBase());
420 break;
421 }
422
423 // Yield Vol
424 case CurveSpec::CurveType::YieldVolatility: {
425 boost::shared_ptr<YieldVolatilityCurveSpec> ydvolspec =
426 boost::dynamic_pointer_cast<YieldVolatilityCurveSpec>(spec);
427 QL_REQUIRE(ydvolspec, "Failed to convert spec " << *spec);
428 auto itr = requiredGenericYieldVolCurves_.find(ydvolspec->name());
429 if (itr == requiredGenericYieldVolCurves_.end()) {
430 DLOG("Building Yield Volatility for asof " << asof_);
431 boost::shared_ptr<YieldVolCurve> yieldVolCurve = boost::make_shared<YieldVolCurve>(
432 asof_, *ydvolspec, *loader_, *curveConfigs_, buildCalibrationInfo_);
433 calibrationInfo_->irVolCalibrationInfo[ydvolspec->name()] = yieldVolCurve->calibrationInfo();
434 itr = requiredGenericYieldVolCurves_.insert(make_pair(ydvolspec->name(), yieldVolCurve)).first;
435 }
436 DLOG("Adding YieldVol (" << node.name << ") with spec " << *ydvolspec << " to configuration "
437 << configuration);
438 yieldVolCurves_[make_pair(configuration, node.name)] =
439 Handle<SwaptionVolatilityStructure>(itr->second->volTermStructure());
440 break;
441 }
442
443 // Cap Floor Vol
444 case CurveSpec::CurveType::CapFloorVolatility: {
445 boost::shared_ptr<CapFloorVolatilityCurveSpec> cfVolSpec =
446 boost::dynamic_pointer_cast<CapFloorVolatilityCurveSpec>(spec);
447 QL_REQUIRE(cfVolSpec, "Failed to convert spec " << *spec);
448 boost::shared_ptr<CapFloorVolatilityCurveConfig> cfg =
449 curveConfigs_->capFloorVolCurveConfig(cfVolSpec->curveConfigID());
450
451 auto itr = requiredCapFloorVolCurves_.find(cfVolSpec->name());
452 if (itr == requiredCapFloorVolCurves_.end()) {
453 DLOG("Building cap/floor volatility for asof " << asof_);
454
455 // Firstly, need to retrieve ibor index and discount curve
456 // Ibor index
457 std::string iborIndexName = cfg->index();
458 QuantLib::Period rateComputationPeriod = cfg->rateComputationPeriod();
459 Handle<IborIndex> iborIndex = MarketImpl::iborIndex(iborIndexName, configuration);
460 Handle<YieldTermStructure> discountCurve;
461 // Discount curve
462 if (!cfg->discountCurve().empty()) {
463 auto it = requiredYieldCurves_.find(cfg->discountCurve());
464 QL_REQUIRE(it != requiredYieldCurves_.end(), "Discount curve with spec, "
465 << cfg->discountCurve()
466 << ", not found in loaded yield curves");
467 discountCurve = it->second->handle();
468 }
469
470 // for proxy curves we need the source and target indices
471 boost::shared_ptr<IborIndex> sourceIndex, targetIndex;
472 if (!cfg->proxySourceCurveId().empty()) {
473 if (!cfg->proxySourceIndex().empty())
474 sourceIndex = *MarketImpl::iborIndex(cfg->proxySourceIndex(), configuration);
475 if (!cfg->proxyTargetIndex().empty()) {
476 targetIndex = *MarketImpl::iborIndex(cfg->proxyTargetIndex(), configuration);
477 iborIndexName = cfg->proxyTargetIndex();
478 rateComputationPeriod = cfg->proxyTargetRateComputationPeriod();
479 }
480 }
481
482 // Now create cap/floor vol curve
483 boost::shared_ptr<CapFloorVolCurve> capFloorVolCurve = boost::make_shared<CapFloorVolCurve>(
484 asof_, *cfVolSpec, *loader_, *curveConfigs_, iborIndex.currentLink(), discountCurve, sourceIndex,
485 targetIndex, requiredCapFloorVolCurves_, buildCalibrationInfo_);
486 calibrationInfo_->irVolCalibrationInfo[cfVolSpec->name()] = capFloorVolCurve->calibrationInfo();
487 itr = requiredCapFloorVolCurves_
488 .insert(make_pair(
489 cfVolSpec->name(),
490 std::make_pair(capFloorVolCurve, std::make_pair(iborIndexName, rateComputationPeriod))))
491 .first;
492 }
493
494 DLOG("Adding CapFloorVol (" << node.name << ") with spec " << *cfVolSpec << " to configuration "
495 << configuration);
496 capFloorCurves_[make_pair(configuration, node.name)] =
497 Handle<OptionletVolatilityStructure>(itr->second.first->capletVolStructure());
498 capFloorIndexBase_[make_pair(configuration, node.name)] = itr->second.second;
499 break;
500 }
501
502 // Default Curve
503 case CurveSpec::CurveType::Default: {
504 boost::shared_ptr<DefaultCurveSpec> defaultspec = boost::dynamic_pointer_cast<DefaultCurveSpec>(spec);
505 QL_REQUIRE(defaultspec, "Failed to convert spec " << *spec);
506 auto itr = requiredDefaultCurves_.find(defaultspec->name());
507 if (itr == requiredDefaultCurves_.end()) {
508 // build the curve
509 DLOG("Building DefaultCurve for asof " << asof_);
510 boost::shared_ptr<DefaultCurve> defaultCurve = boost::make_shared<DefaultCurve>(
511 asof_, *defaultspec, *loader_, *curveConfigs_, requiredYieldCurves_, requiredDefaultCurves_);
512 itr = requiredDefaultCurves_.insert(make_pair(defaultspec->name(), defaultCurve)).first;
513 }
514 DLOG("Adding DefaultCurve (" << node.name << ") with spec " << *defaultspec << " to configuration "
515 << configuration);
516 defaultCurves_[make_pair(configuration, node.name)] =
517 Handle<QuantExt::CreditCurve>(itr->second->creditCurve());
518 recoveryRates_[make_pair(configuration, node.name)] =
519 Handle<Quote>(boost::make_shared<SimpleQuote>(itr->second->recoveryRate()));
520 break;
521 }
522
523 // CDS Vol
524 case CurveSpec::CurveType::CDSVolatility: {
525 boost::shared_ptr<CDSVolatilityCurveSpec> cdsvolspec =
526 boost::dynamic_pointer_cast<CDSVolatilityCurveSpec>(spec);
527 QL_REQUIRE(cdsvolspec, "Failed to convert spec " << *spec);
528 auto itr = requiredCDSVolCurves_.find(cdsvolspec->name());
529 if (itr == requiredCDSVolCurves_.end()) {
530 DLOG("Building CDSVol for asof " << asof_);
531 boost::shared_ptr<CDSVolCurve> cdsVolCurve = boost::make_shared<CDSVolCurve>(
532 asof_, *cdsvolspec, *loader_, *curveConfigs_, requiredCDSVolCurves_, requiredDefaultCurves_);
533 itr = requiredCDSVolCurves_.insert(make_pair(cdsvolspec->name(), cdsVolCurve)).first;
534 }
535 DLOG("Adding CDSVol (" << node.name << ") with spec " << *cdsvolspec << " to configuration "
536 << configuration);
537 cdsVols_[make_pair(configuration, node.name)] =
538 Handle<QuantExt::CreditVolCurve>(itr->second->volTermStructure());
539 break;
540 }

813void TodaysMarket::require(const MarketObject o, const string& name, const string& configuration,
814 const bool forceBuild) const {
815
816 // if the market is not lazily built, do nothing
817
818 if (!lazyBuild_ && !forceBuild)
819 return;
820
821 // search the node (o, name) in the dependency graph
822
823 DLOG("market object " << o << "(" << name << ") required for configuration '" << configuration << "'");
824
825 auto tmp = dependencies_.find(configuration);
826 if (tmp == dependencies_.end()) {
827 if (configuration != Market::defaultConfiguration) {
828 ALOG(StructuredCurveErrorMessage(ore::data::to_string(o) + "(" + name + ")", "Failed to Build Curve",
829 "Configuration '" + configuration +
830 "' not known, retry with default configuration."));
831 require(o, name, Market::defaultConfiguration);
832 return;
833 } else {
834 ALOG(StructuredCurveErrorMessage(ore::data::to_string(o) + "(" + name + ")", "Failed to Build Curve",
835 "Configuration 'default' not known, this is unexpected. Do nothing."));
836 return;
837 }
838 }
839
840 Vertex node = nullptr;
841
842 Graph& g = tmp->second;
843 IndexMap index = boost::get(boost::vertex_index, g);
844
845 VertexIterator v, vend;
846 bool found = false;
847 for (std::tie(v, vend) = boost::vertices(g); v != vend; ++v) {
848 if (g[*v].obj == o) {
849 if (o == MarketObject::Correlation) {
850 // split the required name and the node name and compare the tokens
851 found = getCorrelationTokens(name) == getCorrelationTokens(g[*v].name);
852 } else {
853 found = (g[*v].name == name);
854 }
855 if (found) {
856 node = *v;
857 break;
858 }
859 }
860 }
861
862 // if we did not find a node, we retry with the default configuration, as required by the interface
863
864 if (!found && configuration != Market::defaultConfiguration) {
865 DLOG("not found, retry with default configuration");
866 require(o, name, Market::defaultConfiguration);
867 return;
868 }
869
870 // if we still have no node, we do nothing, the error handling is done in MarketImpl
871
872 if (!found) {
873 DLOG("not found, do nothing");
874 return;
875 }
876
877 // if the node is already built, we are done
878
879 if (g[node].built) {
880 DLOG("node already built, do nothing.");
881 return;
882 }
883
884 // run a DFS from the found node to identify the required nodes to be built and get a possible order to do this
885
886 map<string, string> buildErrors;
887 std::vector<Vertex> order;
888 bool foundCycle = false;
889
890 DfsVisitor<Vertex> dfs(order, foundCycle);
891 auto colorMap = boost::make_vector_property_map<boost::default_color_type>(index);
892 boost::depth_first_visit(g, node, dfs, colorMap);
893
894 if (foundCycle) {
895 order.clear();
896 buildErrors[g[node].curveSpec ? g[node].curveSpec->name() : g[node].name] = "found cycle";
897 }
898
899 // build the nodes
900
901 TLOG("Can build objects in the following order:");
902 for (auto const& m : order) {
903 TLOG("vertex #" << index[m] << ": " << g[m] << (g[m].built ? " (already built)" : " (not yet built)"));
904 }
905
906 Size countSuccess = 0, countError = 0;
907 for (auto const& m : order) {
908 if (g[m].built)
909 continue;
910 try {
911 buildNode(configuration, g[m]);
912 ++countSuccess;
913 DLOG("built node " << g[m] << " in configuration " << configuration);
914 } catch (const std::exception& e) {
915 if (g[m].curveSpec)
916 buildErrors[g[m].curveSpec->name()] = e.what();
917 else
918 buildErrors[g[m].name] = e.what();
919 ++countError;
920 ALOG("error while building node " << g[m] << " in configuration " << configuration << ": " << e.what());
921 }
922 }
923
924 if (countSuccess + countError > 0) {
925 DLOG("Loaded CurvesSpecs: success: " << countSuccess << ", error: " << countError);
926 }
927
928 // output errors
929
930 if (!buildErrors.empty()) {
931 for (auto const& error : buildErrors) {
932 ALOG(StructuredCurveErrorMessage(error.first, "Failed to Build Curve", error.second));
933 }
934 if (!continueOnError_) {
935 string errStr;
936 for (auto const& error : buildErrors) {
937 errStr += "(" + error.first + ": " + error.second + "); ";
938 }
939 QL_FAIL("Cannot build all required curves! Building failed for: " << errStr);
940 }
941 }
942} // TodaysMarket::require()
