40CommoditySchwartzModelBuilder::CommoditySchwartzModelBuilder(
41 const boost::shared_ptr<ore::data::Market>& market, const boost::shared_ptr<CommoditySchwartzData>& data,
42 const QuantLib::Currency& baseCcy, const std::string& configuration,
43 const std::string& referenceCalibrationGrid)
44 : market_(market), configuration_(configuration), data_(data), referenceCalibrationGrid_(referenceCalibrationGrid),
45 baseCcy_(baseCcy) {
46
47 optionActive_ = std::vector<bool>(data_->optionExpiries().size(), false);
48 marketObserver_ = boost::make_shared<MarketObserver>();
49 QuantLib::Currency ccy = ore::data::parseCurrency(data->currency());
50 string name = data->name();
51
52 LOG("Start building CommoditySchwartz model for " << name);
53
54 // get market data
55 std::string fxCcyPair = ccy.code() + baseCcy_.code();
56 fxSpot_ = market_->fxRate(fxCcyPair, configuration_);
57 curve_ = market_->commodityPriceCurve(name, configuration_);
58 vol_ = market_->commodityVolatility(name, configuration_);
59
60 // register with market observables except vols
61 marketObserver_->registerWith(fxSpot_);
62 marketObserver_->registerWith(curve_);
63
64 // register the builder with the vol and the market observer
65 registerWith(vol_);
66 registerWith(marketObserver_);
67
68 // notify observers of all market data changes, not only when not calculated
69 alwaysForwardNotifications();
70
71 // build option basket and derive parametrization from it
72 if (data->calibrateSigma() || data->calibrateKappa())
73 buildOptionBasket();
74
75 parametrization_ = boost::make_shared<QuantExt::CommoditySchwartzParametrization>(ccy, name, curve_, fxSpot_,
76 data->sigmaValue(), data->kappaValue(),
77 data->driftFreeState());
78 model_ = boost::make_shared<QuantExt::CommoditySchwartzModel>(parametrization_);
79}

105void CommoditySchwartzModelBuilder::performCalculations() const {
106 if (requiresRecalibration()) {
107 DLOG("COM model requires recalibration");
108
109 // reset market observer updated flag
110 marketObserver_->hasUpdated(true);
111 // build option basket
112 buildOptionBasket();
113 // update vol cache
114 volSurfaceChanged(true);
115
116 // attach pricing engine to helpers
117 boost::shared_ptr<QuantExt::CommoditySchwartzFutureOptionEngine> engine =
118 boost::make_shared<QuantExt::CommoditySchwartzFutureOptionEngine>(model_);
119 for (Size j = 0; j < optionBasket_.size(); j++)
120 optionBasket_[j]->setPricingEngine(engine);
121
122 QL_REQUIRE(data_->calibrationType() != CalibrationType::Bootstrap, "Bootstrap COM calibration not supported yet");
123
124 if (data_->calibrationType() == CalibrationType::None ||
125 (data_->calibrateSigma() == false && data_->calibrateKappa() == false)) {
126 LOG("COM calibration is deactivated in the CommoditySchwartzModelData for name " << data_->name());
127 return;
128 }
129
130 // check which parameters are kept fixed
131 std::vector<bool> fix(model_->parametrization()->numberOfParameters(), true);
132 std::vector<Real> weights;
133 Size freeParams = 0;
134 if (data_->calibrateSigma()) {
135 fix[0] = false;
136 freeParams++;
137 LOG("CommoditySchwartzModel: calibrate sigma for name " << data_->name());
138 }
139 if (data_->calibrateKappa()) {
140 fix[1] = false;
141 freeParams++;
142 LOG("CommoditySchwartzModel: calibrate kappa for name " << data_->name());
143 }
144 if (freeParams == 0) {
145 WLOG("CommoditySchwartzModel: skip calibration for name " << data_->name() << ", no free parameters");
146 error_ = 0.0;
147 return;
148 }
149
150 LOG("CommoditySchwartzModel for name " << data_->name() << " before calibration:"
151 << " sigma=" << parametrization_->sigmaParameter()
152 << " kappa=" << parametrization_->kappaParameter());
153
154 model_->calibrate(optionBasket_, *data_->optimizationMethod(), data_->endCriteria(), data_->constraint(), weights, fix);
155
156 LOG("CommoditySchwartzModel for name " << data_->name() << " after calibration:"
157 << " sigma=" << parametrization_->sigmaParameter()
158 << " kappa=" << parametrization_->kappaParameter());
159
160 error_ = getCalibrationError(optionBasket_);
161 LOG("CommoditySchwartzModel calibration rmse error " << error_ << " for name " << data_->name());
162 try {
163 auto d = getCalibrationDetails(optionBasket_, parametrization_);
164 DLOG(d);
165 } catch (const std::exception& e) {
166 WLOG("An error occurred: " << e.what());
167 }
168 }

171Real CommoditySchwartzModelBuilder::optionStrike(const Size j) const {
172 ore::data::Strike strike = ore::data::parseStrike(data_->optionStrikes()[j]);
173 Real strikeValue;
174 // TODO: Extend strike type coverage
175 if (strike.type == ore::data::Strike::Type::ATMF)
176 strikeValue = Null<Real>();
177 else if (strike.type == ore::data::Strike::Type::Absolute)
178 strikeValue = strike.value;
179 else
180 QL_FAIL("strike type ATMF or Absolute expected");
181 return strikeValue;
182}

184Date CommoditySchwartzModelBuilder::optionExpiry(const Size j) const {
185 Date today = Settings::instance().evaluationDate();
186 std::string expiryString = data_->optionExpiries()[j];
187 bool expiryDateBased;
188 Period expiryPb;
189 Date expiryDb;
190 parseDateOrPeriod(expiryString, expiryDb, expiryPb, expiryDateBased);
191 Date expiryDate = expiryDateBased ? expiryDb : today + expiryPb;
192 return expiryDate;

195bool CommoditySchwartzModelBuilder::volSurfaceChanged(const bool updateCache) const {
196 bool hasUpdated = false;
197
198 // if cache doesn't exist resize vector
199 if (volCache_.size() != optionBasket_.size())
200 volCache_ = vector<Real>(optionBasket_.size(), Null<Real>());
201
202 Size optionCounter = 0;
203 for (Size j = 0; j < data_->optionExpiries().size(); j++) {
204 if (!optionActive_[j])
205 continue;
206 Real vol = vol_->blackVol(optionExpiry(j), optionStrike(j));
207 if (!close_enough(volCache_[optionCounter], vol)) {
208 if (updateCache)
209 volCache_[optionCounter] = vol;
210 hasUpdated = true;
211 }
212 optionCounter++;
213 }
214 return hasUpdated;
215}

217void CommoditySchwartzModelBuilder::buildOptionBasket() const {
218
219 QL_REQUIRE(data_->optionExpiries().size() == data_->optionStrikes().size(), "Commodity option vector size mismatch");
220
221 optionActive_ = std::vector<bool>(data_->optionExpiries().size(), false);
222
223 DLOG("build reference date grid '" << referenceCalibrationGrid_ << "'");
224 Date lastRefCalDate = Date::minDate();
225 std::vector<Date> referenceCalibrationDates;
226 if (!referenceCalibrationGrid_.empty())
227 referenceCalibrationDates = DateGrid(referenceCalibrationGrid_).dates();
228
229 std::vector<Time> expiryTimes;
230 optionBasket_.clear();
231 for (Size j = 0; j < data_->optionExpiries().size(); j++) {
232 // may wish to calibrate against specific futures expiry dates...
233 Date expiryDate = optionExpiry(j);
234
235 // check if we want to keep the helper when a reference calibration grid is given
236 auto refCalDate =
237 std::lower_bound(referenceCalibrationDates.begin(), referenceCalibrationDates.end(), expiryDate);
238 if (refCalDate == referenceCalibrationDates.end() || *refCalDate > lastRefCalDate) {
239 optionActive_[j] = true;
240 Real strikeValue = optionStrike(j);
241 Handle<Quote> volQuote(boost::make_shared<SimpleQuote>(vol_->blackVol(expiryDate, strikeValue)));
242 boost::shared_ptr<QuantExt::FutureOptionHelper> helper = boost::make_shared<QuantExt::FutureOptionHelper>(
243 expiryDate, strikeValue, curve_, volQuote, data_->calibrationErrorType());
244 optionBasket_.push_back(helper);
245 helper->performCalculations();
246 expiryTimes.push_back(curve_->timeFromReference(helper->option()->exercise()->date(0)));
247 DLOG("Added FutureOptionHelper " << data_->name() << " " << QuantLib::io::iso_date(expiryDate) << " "
248 << volQuote->value());
249 if (refCalDate != referenceCalibrationDates.end())
250 lastRefCalDate = *refCalDate;
251 } else {
252 optionActive_[j] = false;
253 }
254 }
255
256 std::sort(expiryTimes.begin(), expiryTimes.end());
257 auto itExpiryTime = unique(expiryTimes.begin(), expiryTimes.end());
258 expiryTimes.resize(distance(expiryTimes.begin(), itExpiryTime));
259
260 optionExpiries_ = Array(expiryTimes.size());
261 for (Size j = 0; j < expiryTimes.size(); j++)
262 optionExpiries_[j] = expiryTimes[j];
263}
