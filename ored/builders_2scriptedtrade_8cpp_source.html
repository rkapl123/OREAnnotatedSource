59ScriptedTradeEngineBuilder::correlationCurve(const std::string& index1, const std::string& index2) {
60 if (index1 == index2) {

67 return market_->correlationCurve(index1, index2, configuration(MarketContext::pricing));

71boost::shared_ptr<ScriptedInstrument::engine>
72ScriptedTradeEngineBuilder::engine(const std::string& id, const ScriptedTrade& scriptedTrade,
73 const boost::shared_ptr<ReferenceDataManager>& referenceData,
74 const IborFallbackConfig& iborFallbackConfig) {
75
76 const std::vector<ScriptedTradeEventData>& events = scriptedTrade.events();
77 const std::vector<ScriptedTradeValueTypeData>& numbers = scriptedTrade.numbers();
78 const std::vector<ScriptedTradeValueTypeData>& indices = scriptedTrade.indices();
79 const std::vector<ScriptedTradeValueTypeData>& currencies = scriptedTrade.currencies();
80 const std::vector<ScriptedTradeValueTypeData>& daycounters = scriptedTrade.daycounters();
81
82 LOG("Building engine for scripted trade " << id);
83
84 // 0 clear members
85
86 clear();
87
88 // 1 set the SIMM product class, simple EQ > COM > FX approach for Hybrids, that's all we can do automatically?
89 // also set the assetClassReplacement string which is used to replace {AssetClass} in product tags
90
91 deriveProductClass(indices);
92
93 // 1b get product tag from scripted trade or library and build resolved product tag
94
95 std::string productTag = getScript(scriptedTrade, ScriptLibraryStorage::instance().get(), "", false).first;
96 resolvedProductTag_ = boost::replace_all_copy(productTag, "{AssetClass}", assetClassReplacement_);
97 DLOG("got product tag '" << productTag << "', resolved product tag is '" << resolvedProductTag_);
98
99 // 2 populate model and engine parameters
100
101 populateModelParameters();
102
103 // 3 define purpose, get suitable script and build ast (i.e. parse it or retrieve it from cache)
104
105 std::string purpose = "";
106 if (amcCam_)
107 purpose = "AMC";
108 else if (engineParam_ == "FD")
109 purpose = "FD";
110
111 ScriptedTradeScriptData script =
112 getScript(scriptedTrade, ScriptLibraryStorage::instance().get(), purpose, true).second;
113 auto f = astCache_.find(script.code());
114 if (f != astCache_.end()) {
115 ast_ = f->second;
116 DLOG("retrieved ast from cache");
117 } else {
118 ast_ = parseScript(script.code());
119 astCache_[script.code()] = ast_;
120 DLOGGERSTREAM("built ast:\n" << to_string(ast_));
121 }
122
123 // 4 set up context
124
125 auto context = makeContext(modelSize_, gridCoarsening_, script.schedulesEligibleForCoarsening(), referenceData,
126 events, numbers, indices, currencies, daycounters);
127 addAmcGridToContext(context);
128 addNewSchedulesToContext(context, script.newSchedules());
129
130 DLOG("Built initial context:");
131 DLOGGERSTREAM(*context);
132
133 // 4b set up calibration strike information
134
135 setupCalibrationStrikes(script, context);
136
137 // 5 run static analyser
138
139 DLOG("Run static analyser on script");
140 staticAnalyser_ = boost::make_shared<StaticAnalyser>(ast_, context);
141 staticAnalyser_->run(script.code());
142
143 // 6 extract eq, fx, ir indices from script
144
145 extractIndices(referenceData);
146
147 // 7 populate fixings map
148
149 populateFixingsMap(iborFallbackConfig);
150
151 // 8 init result variable for NPV
152
153 checkDuplicateName(context, script.npv());
154 context->scalars[script.npv()] = RandomVariable(modelSize_, 0.0);
155
156 // 9 extract pay currencies
157
158 extractPayCcys();
159
160 // 10 determine base ccy, this might be overwritten in 11 when an AMC engine is built, see the implementation of 11
161
162 determineBaseCcy();
163
164 // 11 compile the model currency list (depends on the model actually)
165
166 compileModelCcyList();
167
168 // 12 get the t0 curves for each model ccy
169
170 for (auto const& c : modelCcys_) {
171 modelCurves_.push_back(market_->discountCurve(c, configuration(MarketContext::pricing)));
172 DLOG("curve for " << c << " added.");
173 }
174
175 // 13 get the fx spots for each model ccy vs. the base ccy
176
177 for (Size i = 1; i < modelCcys_.size(); ++i) {
178 modelFxSpots_.push_back(market_->fxRate(modelCcys_[i] + baseCcy_, configuration(MarketContext::pricing)));
179 DLOG("fx spot " << modelCcys_[i] + baseCcy_ << " added.");
180 }
181
182 // 14 compile the model index (eq, fx, comm) and ir index lists
183
184 compileModelIndexLists();
185
186 // 15 determine last relevant date as max over index eval, regression, pay obs/pay dates
187
188 setLastRelevantDate();
189
190 // 16 set up correlations between model indices (ir, eq, fx, comm)
191
192 setupCorrelations();
193
194 // 17 compile the processes (needed for BlackScholes, LocalVol only)
195
196 if (modelParam_ == "BlackScholes" || modelParam_ == "LocalVolDupire" || modelParam_ == "LocalVolAndreasenHuge")
197 setupBlackScholesProcesses();
198
199 // 18 setup IR reversion values (needed for Gaussian CAM only)
200
201 if (modelParam_ == "GaussianCam") {
202 setupIrReversions();
203 }
204
205 // 19 compile the sim and add dates required by the model ctors below
206
207 compileSimulationAndAddDates();
208
209 // 20 build the model adapter
210
211 QL_REQUIRE(amcCam_ == nullptr || modelParam_ == "GaussianCam",
212 "model/engine = GaussianCam/MC required to build an amc model, got " << modelParam_ << "/"
213 << engineParam_);
214
215 if(staticAnalyser_->regressionDates().empty())
216 mcParams_.trainingSamples = Null<Size>();
217
218 if (modelParam_ == "BlackScholes" && engineParam_ == "MC") {
219 buildBlackScholes(id, iborFallbackConfig);
220 } else if (modelParam_ == "BlackScholes" && engineParam_ == "FD") {
221 buildFdBlackScholes(id, iborFallbackConfig);
222 } else if ((modelParam_ == "LocalVolDupire" || modelParam_ == "LocalVolAndreasenHuge") && engineParam_ == "MC") {
223 buildLocalVol(id, iborFallbackConfig);
224 } else if (modelParam_ == "GaussianCam" && engineParam_ == "MC") {
225 if (amcCam_ == nullptr)
226 buildGaussianCam(id, iborFallbackConfig, script.conditionalExpectationModelStates());
227 else
228 buildGaussianCamAMC(id, iborFallbackConfig, script.conditionalExpectationModelStates());
229 } else {
230 QL_FAIL("model '" << modelParam_ << "' / engine '" << engineParam_
231 << "' not recognised, expected BlackScholes/[MC|FD], LocalVolDupire/MC, "
232 "LocalVolAndreasenHuge/MC, GaussianCam/MC");
233 }
234
235 QL_REQUIRE(model_ != nullptr || modelCG_ != nullptr, "internal error: both model_ and modelCG_ are null");
236
237 // 21 log some summary information
238
239 DLOG("built model : " << modelParam_ << " / " << engineParam_);
240 DLOG("useCg = " << std::boolalpha << useCg_);
241 DLOG("useAd = " << std::boolalpha << useAd_);
242 DLOG("useExternalDevice = " << std::boolalpha << useExternalComputeDevice_);
243 DLOG("externalDevice = " << (useExternalComputeDevice_ ? externalComputeDevice_ : "na"));
244 DLOG("calibration = " << calibration_);
245 DLOG("base ccy = " << baseCcy_);
246 DLOG("model base (npv) ccy = " << (model_ != nullptr ? model_->baseCcy() : modelCG_->baseCcy()));
247 DLOG("ccys = " << modelCcys_.size());
248 DLOG("eq,fx,com indices = " << modelIndices_.size());
249 DLOG("ir indices = " << irIndices_.size());
250 DLOG("inf indices = " << infIndices_.size());
251 DLOG("sim dates = " << simulationDates_.size());
252 DLOG("add dates = " << addDates_.size());
253 DLOG("timeStepsPerYear = " << timeStepsPerYear_);
254 DLOG("fullDynamicFx = " << std::boolalpha << fullDynamicFx_);
255 if (engineParam_ == "MC") {
256 DLOG("seed = " << mcParams_.seed);
257 DLOG("paths = " << modelSize_);
258 DLOG("regressionOrder = " << mcParams_.regressionOrder);
259 DLOG("sequence type = " << mcParams_.sequenceType);
260 DLOG("polynom type = " << mcParams_.polynomType);
261 if (mcParams_.trainingSamples != Null<Size>()) {
262 DLOG("training seed = " << mcParams_.trainingSeed);
263 DLOG("training paths = " << mcParams_.trainingSamples);
264 DLOG("training seq. type = " << mcParams_.trainingSequenceType);
265 }
266 DLOG("sobol bb ordering = " << mcParams_.sobolOrdering);
267 DLOG("sobol direction int. = " << mcParams_.sobolDirectionIntegers);
268 } else if (engineParam_ == "FD") {
269 DLOG("stateGridPoints = " << modelSize_);
270 DLOG("mesherEpsilon = " << mesherEpsilon_);
271 DLOG("mesherScaling = " << mesherScaling_);
272 DLOG("mesherConcentration = " << mesherConcentration_);
273 DLOG("mesherMaxConcentrPts = " << mesherMaxConcentratingPoints_);
274 DLOG("mesherIsStatic = " << std::boolalpha << mesherIsStatic_);
275 }
276 if (modelParam_ == "GaussianCam") {
277 DLOG("fullDynamicIr = " << std::boolalpha << fullDynamicIr_);
278 DLOG("ref calibration grid = " << referenceCalibrationGrid_);
279 DLOG("bootstrap tolerance = " << bootstrapTolerance_);
280 DLOG("infModelType = " << infModelType_);
281 DLOG("condExpMdlStates = " << boost::algorithm::join(script.conditionalExpectationModelStates(), ","));
282 } else if (modelParam_ == "LocalVolAndreasenHuge") {
283 DLOG("moneyness points = " << calibrationMoneyness_.size());
284 }
285
286 // 22 build the pricing engine and return it
287
288 bool generateAdditionalResults = false;
289 auto p = globalParameters_.find("GenerateAdditionalResults");
290 if (p != globalParameters_.end()) {
291 generateAdditionalResults = parseBool(p->second);
292 }
293
294 boost::shared_ptr<ScriptedInstrument::engine> engine;
295 if (model_) {
296 engine = boost::make_shared<ScriptedInstrumentPricingEngine>(
297 script.npv(), script.results(), model_, ast_, context, script.code(), interactive_, amcCam_ != nullptr,
298 std::set<std::string>(script.stickyCloseOutStates().begin(), script.stickyCloseOutStates().end()),
299 generateAdditionalResults);
300 } else if (modelCG_) {
301 auto rt = globalParameters_.find("RunType");
302 bool useCachedSensis = useAd_ && (rt != globalParameters_.end() && rt->second == "SensitivityDelta");
303 bool useExternalDev = useExternalComputeDevice_ && !generateAdditionalResults && !useCachedSensis;
304 engine = boost::make_shared<ScriptedInstrumentPricingEngineCG>(
305 script.npv(), script.results(), modelCG_, ast_, context, mcParams_, script.code(), interactive_,
306 amcCam_ != nullptr,
307 std::set<std::string>(script.stickyCloseOutStates().begin(), script.stickyCloseOutStates().end()),
308 generateAdditionalResults, useCachedSensis, useExternalDev);
309 if (useExternalDev) {
310 ComputeEnvironment::instance().selectContext(externalComputeDevice_);
311 }
312 }
313
314 LOG("engine built for model " << modelParam_ << " / " << engineParam_ << ", modelSize = " << modelSize_
315 << ", interactive = " << interactive_ << ", amcEnabled = " << (amcCam_ != nullptr)
316 << ", generateAdditionalResults = " << generateAdditionalResults);
317 return engine;
318}

343void ScriptedTradeEngineBuilder::extractIndices(
344 const boost::shared_ptr<ore::data::ReferenceDataManager>& referenceData) {
345 DLOG("Extract indices from script:");
346 for (auto const& i : staticAnalyser_->indexEvalDates()) {
347 IndexInfo ind(i.first);
348 if (ind.isEq()) {
349 eqIndices_.insert(ind);
350 } else if (ind.isIr()) {
351 irIndices_.insert(ind);
352 } else if (ind.isInf()) {
353 infIndices_.insert(ind);
354 } else if (ind.isFx()) {
355 // ignore trivial fx indices
356 if (ind.fx()->sourceCurrency() != ind.fx()->targetCurrency())
357 fxIndices_.insert(ind);
358 } else if (ind.isComm()) {
359 commIndices_.insert(ind);
360 } else if (ind.isGeneric()) {
361 // ignore generic indices, only historical fixings can be retrieved from them
362 } else {
363 QL_FAIL("unexpected index type for '" << ind.name() << "'");
364 }
365 DLOG("got " << ind);
366 }
367 for (auto const& i : staticAnalyser_->fwdCompAvgFixingDates()) {
368 IndexInfo ind(i.first);
369 QL_REQUIRE(ind.isIr(), "expected IR (ON) index for " << ind.name());
370 irIndices_.insert(ind);
371 DLOG("got " << ind);
372 }

375void ScriptedTradeEngineBuilder::deriveProductClass(const std::vector<ScriptedTradeValueTypeData>& indices) {
376 std::set<std::string> names;
377 std::set<IndexInfo> commIndices, eqIndices, fxIndices, irIndices, infIndices;
378
379 for (auto const& i : indices) {
380 if (i.isArray()) {
381 names.insert(i.values().begin(), i.values().end());
382 } else {
383 names.insert(i.value());
384 }
385 }
386
387 for (auto const& n : names) {
388 IndexInfo ind(n);
389 if (ind.isFx())
390 fxIndices.insert(ind);
391 else if (ind.isEq())
392 eqIndices.insert(ind);
393 else if (ind.isComm())
394 commIndices.insert(ind);
395 else if (ind.isIr())
396 irIndices.insert(ind);
397 else if (ind.isInf())
398 infIndices.insert(ind);
399 }
400
401 assetClassReplacement_ = "";
402 if (!commIndices.empty()) {
403 simmProductClass_ = "Commodity";
404 scheduleProductClass_ = "Commodity";
405 assetClassReplacement_ = "COMM";
406 } else if (!eqIndices.empty()) {
407 simmProductClass_ = "Equity";
408 scheduleProductClass_ = "Equity";
409 assetClassReplacement_ = "EQ";
410 } else if (!fxIndices.empty()) {
411 simmProductClass_ = "RatesFX";
412 scheduleProductClass_ = "FX";
413 assetClassReplacement_ = "FX";
414 for (auto const& i : fxIndices) {
415 std::string f = i.fx()->sourceCurrency().code();
416 std::string d = i.fx()->targetCurrency().code();
417 if (isPseudoCurrency(f) || isPseudoCurrency(d)) {
418 simmProductClass_ = "Commodity";
419 scheduleProductClass_ = "Commodity";
420 // in terms of the asset class replacement we stick with FX for precious metals
421 // assetClassReplacement_ = "COMM";
422 }
423 }
424 } else if (!irIndices.empty() || !infIndices.empty()) {
425 simmProductClass_ = "RatesFX";
426 scheduleProductClass_ = "Rates";
427 } else {
428 simmProductClass_ = "RatesFx"; // fallback if we do not have any indices (an edge case really...)
429 scheduleProductClass_ = "Rates"; // fallback if we do not have any indices (an edge case really...)
430 }
431
432 if ((int)!eqIndices.empty() + (int)!fxIndices.empty() + (int)!commIndices.empty() > 1) {
433 WLOG("SIMM product class for hybrid trade is set to " << simmProductClass_);
434 WLOG("IM Schedule product class for hybrid trade is set to " << scheduleProductClass_);
435 assetClassReplacement_ = "HYBRID";
436 } else {
437 LOG("SIMM product class is set to " << simmProductClass_);
438 LOG("IM Schedule product class is set to " << scheduleProductClass_);
439 }

538void ScriptedTradeEngineBuilder::populateFixingsMap(const IborFallbackConfig& iborFallbackConfig) {
539 DLOG("Populate fixing map");
540
541 // this might be a superset of the actually required fixings, since index evaluations with fwd date are also
542 // returned, in which case only future estimations are allowed
543
544 std::map<std::string, std::set<std::pair<Date, bool>>> indexFixings;
545
546 for (auto const& [name, fixings] : staticAnalyser_->probFixingDates()) {
547 for (auto const& d : fixings) {
548 indexFixings[name].insert(std::make_pair(d, true));
549 }
550 }
551
552 for (auto const& [name, fixings] : staticAnalyser_->indexEvalDates()) {
553 for (auto const& d : fixings) {
554 indexFixings[name].insert(std::make_pair(d, false));
555 }
556 }
557
558 for (auto const& [name, fixings] : indexFixings) {
559 IndexInfo i(name);
560 if (i.isComm()) {
561 // COMM indices require a special treatment, since they might need resolution
562 std::map<std::string, Size> stats;
563 for (auto const& [d, _] : fixings) {
564 auto idx = i.comm(d);
565 std::string name = idx->name();
566 fixings_[name].insert(idx->fixingCalendar().adjust(d, Preceding));
567 stats[name]++;
568 }
569 for (auto const& s : stats) {
570 DLOG("added " << s.second << " fixings for '" << s.first << "' (from eval op, prob fcts)");
571 }
572 } else {
573 // all other indices can be handled generically, notice for inf we include the scripting specific
574 // suffixes #L, #F in the index name, this is handled in the scripted trade builder when populating the
575 // required fixings
576
577 if (i.irIborFallback(iborFallbackConfig)) {
578 // well, except ibor fallback indices that we handle here...
579 Size nIbor = 0, nRfr = 0;
580 for (auto [d, _] : fixings) {
581 d = i.index()->fixingCalendar().adjust(d, Preceding);
582 if (d >= i.irIborFallback(iborFallbackConfig)->switchDate()) {
583 auto fd = i.irIborFallback(iborFallbackConfig)->onCoupon(d)->fixingDates();
584 fixings_[iborFallbackConfig.fallbackData(name).rfrIndex].insert(fd.begin(), fd.end());
585 nRfr += fd.size();
586 } else {
587 fixings_[i.name()].insert(d);
588 nIbor++;
589 }
590 }
591 DLOG("added " << nIbor << " Ibor and " << nRfr << " Rfr fixings for ibor fallback '" << i.name()
592 << "' (from eval op, prob fcts)");
593 } else if (i.irOvernightFallback(iborFallbackConfig)) {
594 Size nOis = 0, nRfr = 0;
595 for (auto [d, _] : fixings) {
596 d = i.index()->fixingCalendar().adjust(d, Preceding);
597 if (d >= i.irOvernightFallback(iborFallbackConfig)->switchDate()) {
598 fixings_[iborFallbackConfig.fallbackData(name).rfrIndex].insert(d);
599 nRfr++;
600 } else {
601 fixings_[i.name()].insert(d);
602 nOis++;
603 }
604 }
605 DLOG("added " << nOis << " OIS and " << nRfr << " Rfr fallback fixings for OIS fallback '" << i.name()
606 << "' (from eval op, prob fcts)");
607 } else {
608 // ... and all the others here:
609 IndexInfo imkt(name, market_);
610 for (const auto& [d, prob] : fixings) {
611 fixings_[i.name()].insert((prob ? imkt : i).index()->fixingCalendar().adjust(d, Preceding));
612 }
613 DLOG("added " << fixings.size() << " fixings for '" << i.name() << "' (from eval op, prob fcts)");
614 }
615 }
616 }
617
618 // add fixings from FWDCOMP(), FWDAVG()
619 for (auto const& f : staticAnalyser_->fwdCompAvgFixingDates()) {
620 QL_REQUIRE(IndexInfo(f.first).isIr(),
621 "FWD[COMP|AVG]() only supports IR ON indices, got '" << f.first << "' during fixing map population");
622 fixings_[f.first].insert(f.second.begin(), f.second.end());
623 DLOG("added " << f.second.size() << " fixings for '" << f.first << "' (from FWD[COMP|AVG]())");
624 }

684void ScriptedTradeEngineBuilder::compileModelCcyList() {
685 std::set<std::string> tmpCcys;
686 tmpCcys.insert(baseCcy_);
687
688 DLOG("Compile the model currencies list");
689
690 for (auto const& c : payCcys_) {
691 tmpCcys.insert(c);
692 }
693
694 for (auto const& i : fxIndices_) {
695 std::string f = i.fx()->sourceCurrency().code();
696 std::string d = i.fx()->targetCurrency().code();
697 tmpCcys.insert(f);
698 tmpCcys.insert(d);
699 }
700
701 // ir index currencies are only added for the cam model, for bs or local vol they are not needed
702 // inf index currencies are not needed for the dk in the cam model, but for jy they are
703 if (modelParam_ == "GaussianCam") {
704 for (auto const& i : irIndices_)
705 tmpCcys.insert(i.ir()->currency().code());
706 if (infModelType_ == "DK") {
707 for (auto const& i : infIndices_)
708 tmpCcys.insert(i.inf()->currency().code());
709 }
710 }
711
712 // we only add eqCurrencies / comCurrencies to the modelCcys if we
713 // - build the GaussianCam model which requires all relevant currencies to be present
714 // - or require a dynamic FX process for each currency
715 // In case we build a GaussianCam and fullDynamicIr_ = false, there will be a zero vol process set up
716 // for the IR component though, if there is no requirement for the currency from an IR index.
717 // If fullDynamicFx_ = false the FX components for that currency will be zero vol, too, if there is no
718 // FX index requiring the ccy. See buildGaussianCam().
719 if (fullDynamicFx_ || modelParam_ == "GaussianCam") {
720 for (auto const& e : eqIndices_) {
721 tmpCcys.insert(getEqCcy(e));
722 }
723 for (auto const& c : commIndices_) {
724 tmpCcys.insert(getCommCcy(c));
725 }
726 }
727
728 // if we build an AMC builder, we set the base ccy to the amc model base ccy, otherwise we won't have the
729 // required FX spot processes in the projected model we use for the scripted trade; the only exception is
730 // if we have only one ccy in the final scripted trade model anyway (i.e. only one IR process), in which
731 // case we can go with that one currency and don't need a more complicated model
732 if (amcCam_ != nullptr) {
733 std::string newBaseCcy_ = amcCam_->ir(0)->currency().code();
734 if (newBaseCcy_ == baseCcy_) {
735 DLOG("base ccy and AMC model base ccy are identical (" << baseCcy_ << ")");
736 } else {
737 if (tmpCcys.size() > 1) {
738 DLOG("base ccy " << baseCcy_ << " is overwritten with AMC model base ccy " << newBaseCcy_
739 << ", since more than one ccy is needed in the final model.");
740 baseCcy_ = newBaseCcy_;
741 } else {
742 DLOG("base ccy " << baseCcy_ << " is kept although AMC model base ccy is different (" << newBaseCcy_
743 << "), because it is a single currency model");
744 }
745 }
746 }
747
748 // build currency vector with the base ccy at the front
749 modelCcys_ = std::vector<std::string>(1, baseCcy_);
750 for (auto const& c : tmpCcys)
751 if (c != baseCcy_)
752 modelCcys_.push_back(c);
753
754 // log ccys
755 for (auto const& c : modelCcys_)
756 DLOG("model ccy " << c << " added");
757}

784 }

786 std::find(coveredCcys.begin(), coveredCcys.end(), targetCcy) == coveredCcys.end()) {

788 modelIndicesCurrencies_.push_back(targetCcy);
789 coveredCcys.insert(targetCcy);
790 DLOG("added model index " << modelIndices_.back());
791 }
792 }
793

795 if (fullDynamicFx_) {
796 for (Size i = 1; i < modelCcys_.size(); ++i) {
797 if (std::find(coveredCcys.begin(), coveredCcys.end(), modelCcys_[i]) == coveredCcys.end()) {
798 modelIndices_.push_back("FX-GENERIC-" + modelCcys_[i] + "-" + baseCcy_);

819 DLOG("added model inf index " << i.name());

823void ScriptedTradeEngineBuilder::setupCorrelations() {
824
825 if (zeroVolatility_) {
826 DLOG("skipping correlation setup because we are using zero volatility");
827 return;
828 }
829
830 // collect pairs of model index names and correlation curve lookup names
831 std::set<std::pair<std::string, std::string>> tmp;
832
833 // EQ, FX, COMM indices
834 for (auto const& m : modelIndices_) {
835 IndexInfo ind(m);
836 if (ind.isComm()) {
837 // for COMM indices we expect the correlation on the COMM name level (not on single futures)
838 // notice we might have different COMM indices on the same name (via spot, future, dynamic future
839 // reference) - for those the correlationCurve() call below will return a constant 1.0 correlation
840 tmp.insert(std::make_pair(m, "COMM-" + ind.commName()));
841 } else {
842 // for EQ, FX the lookup name is the same as the model index name
843 tmp.insert(std::make_pair(m, m));
844 }
845 }
846
847 // need the ir, inf indices only for GaussianCam
848 if (modelParam_ == "GaussianCam") {
849 for (auto const& irIdx : modelIrIndices_) {
850 // for IR the lookup name is the same as the model index name
851 tmp.insert(std::make_pair(irIdx.first, irIdx.first));
852 }
853 for (auto const& infIdx : modelInfIndices_) {
854 // for INF the lookup name is without the #L, #F suffix
855 IndexInfo ind(infIdx.first);
856 tmp.insert(std::make_pair(infIdx.first, ind.infName()));
857 }
858 }
859
860 DLOG("adding correlations for indices:");
861 for (auto const& n : tmp)
862 DLOG("model index '" << n.first << "' lookup name '" << n.second << "'");
863
864 std::vector<std::pair<std::string, std::string>> corrModelIndices(tmp.begin(), tmp.end());
865
866 for (Size i = 0; i < corrModelIndices.size(); ++i) {
867 for (Size j = 0; j < i; ++j) {
868 try {
869 correlations_[std::make_pair(corrModelIndices[i].first, corrModelIndices[j].first)] =
870 correlationCurve(corrModelIndices[i].second, corrModelIndices[j].second);
871 DLOG("added correlation for " << corrModelIndices[j].second << " ~ " << corrModelIndices[i].second);
872 } catch (const std::exception& e) {
873 WLOG("no correlation provided for " << corrModelIndices[j].second << " ~ " << corrModelIndices[i].second
874 << "(" << e.what() << ")");
875 }
876 }
877 }

903 Handle<BlackVolTermStructure> vol;
904 if (zeroVolatility_) {
905 vol = Handle<BlackVolTermStructure>(
906 boost::make_shared<BlackConstantVol>(0, NullCalendar(), 0.0, ActualActual(ActualActual::ISDA)));
907 DLOG("using zero volatility processes");

921 std::string name = ind.commName();

923 market_->commodityPriceCurve(name, configuration(MarketContext::pricing))));
924 auto priceCurve = market_->commodityPriceCurve(name, configuration(MarketContext::pricing));
925 auto fc = market_->discountCurve(modelIndicesCurrencies_[i], configuration(MarketContext::pricing));
926 auto div = Handle<YieldTermStructure>(boost::make_shared<PriceTermStructureAdapter>(*priceCurve, *fc));
927 div->enableExtrapolation();
928 if (!zeroVolatility_)
929 vol = market_->commodityVolatility(name, configuration(MarketContext::pricing));
930 processes_.push_back(boost::make_shared<GeneralizedBlackScholesProcess>(spot, div, fc, vol));
931 DLOG("added process for commodity " << name);
932 } else if (ind.isFx()) {
933 std::string targetCcy = ind.fx()->targetCurrency().code();
934 std::string sourceCcy = ind.fx()->sourceCurrency().code();
935 auto spot = market_->fxSpot(sourceCcy + targetCcy, configuration(MarketContext::pricing));
936 auto div = market_->discountCurve(sourceCcy, configuration(MarketContext::pricing));
937 auto fc = market_->discountCurve(targetCcy, configuration(MarketContext::pricing));
938 if (!zeroVolatility_)
939 vol = market_->fxVol(sourceCcy + targetCcy, configuration(MarketContext::pricing));
940 processes_.push_back(boost::make_shared<GeneralizedBlackScholesProcess>(spot, div, fc, vol));
941 DLOG("added process for fx " << sourceCcy << "-" << targetCcy);
942 } else {
943 QL_FAIL("unexpected model index " << ind);
944 }
945 }

987 DLOG("compile simulation and additional dates...");
988
989 for (auto const& s : staticAnalyser_->indexEvalDates()) {
990 IndexInfo info(s.first);
991 // skip generic indices, for them we do not to add the obs date to sim or add dates
992 if (info.isGeneric())
993 continue;
994 // need ir / inf index observation dates only for GaussianCam as simulation dates, for LocalVol, BS
995 // we don't need to add them at all
996 if ((!info.isIr() && !info.isInf()) || modelParam_ == "GaussianCam") {
997 if (info.isInf()) {
998 // inf needs special considerations
999 boost::shared_ptr<ZeroInflationIndex> marketIndex = getInfMarketIndex(info.name(), modelInfIndices_);
1000 Size lag = getInflationSimulationLag(marketIndex);
1001 for (auto const& d : s.second) {
1002 auto lim = inflationPeriod(d, info.inf()->frequency());
1003 simulationDates_.insert(lim.first + lag);
1004 QL_DEPRECATED_DISABLE_WARNING
1005 // This will be removed in a later release and all inf indices are then flat
1006 if (info.inf()->interpolated())
1007 simulationDates_.insert(d + lag);
1008 QL_DEPRECATED_ENABLE_WARNING
1009 }
1010 } else {
1011 // for all other indices we just take the original dates
1012 simulationDates_.insert(s.second.begin(), s.second.end());
1013 }
1014 DLOG("added " << s.second.size() << " simulation dates for '" << s.first << "' (from eval op obs dates)");
1015 }
1016 }
1017
1018 for (auto const& s : staticAnalyser_->indexFwdDates()) {
1019 IndexInfo info(s.first);
1020 // do not need ir / inf index fwd dates (not for LocalVol, BS, but also not for GaussianCam)
1021 if (!info.isIr() && !info.isInf()) {
1022 addDates_.insert(s.second.begin(), s.second.end());
1023 DLOG("added " << s.second.size() << " additional dates for '" << s.first << "' (from eval op fwd dates)");
1024 }
1025 }

1035 IndexInfo ind(s);

1041 addDates_.insert(s.second.begin(), s.second.end());
1042 DLOG("added " << s.second.size() << " additional dates for '" << payCcy << "' (from pay() obs dates)");
1043 }
1044 }
1045
1046 simulationDates_.insert(staticAnalyser_->regressionDates().begin(), staticAnalyser_->regressionDates().end());
1047 DLOG("added " << staticAnalyser_->regressionDates().size() << " simulation dates (from npv() regression dates)");
1048
1049 for (auto const& s : staticAnalyser_->payPayDates()) {
1050 addDates_.insert(s.second.begin(), s.second.end());
1051 DLOG("added " << s.second.size() << " additional dates for '" << s.first << "' (from pay() pay dates)");
1052 }
1053
1054 for (auto const& s : staticAnalyser_->discountObsDates()) {
1055 // need disocunt obs dates only for GaussianCam as simulation dates, for Local Vol, BS
1056 // add them as addDates
1057 if (modelParam_ == "GaussianCam") {
1058 simulationDates_.insert(s.second.begin(), s.second.end());
1059 DLOG("added " << s.second.size() << " simulation dates for '" << s.first

1063 DLOG("added " << s.second.size() << " additional dates for '" << s.first

1079 DLOG("added " << s.second.size() << " simulation dates for '" << s.first

1099 const std::map<std::string, std::vector<Real>>& strikes, const std::vector<std::string>& modelIndices,

1101 QL_REQUIRE(modelIndices.size() == processes.size(), "filterBlackScholesCalibrationStrikes: processes size ("
1102 << processes.size() << ") must match modelIndices size ("
1103 << modelIndices.size() << ")");
1104 std::map<std::string, std::vector<Real>> result;
1105 if (T < 0.0 || close_enough(T, 0.0)) {
1106 DLOG("excluding all calibration strikes, because last relevant time is not positive (" << T << ")");
1107 return result;
1108 }
1109 const Real normInvEps = 2.0 * InverseCumulativeNormal()(1 - 1E-6); // hardcoded epsilon
1110 for (auto const& ks : strikes) {
1111 auto m = std::find(modelIndices.begin(), modelIndices.end(), ks.first);
1112 if (m != modelIndices.end()) {
1113 Size index = std::distance(modelIndices.begin(), m);
1114 Real atmf = processes[index]->x0() / processes[index]->riskFreeRate()->discount(T) *
1115 processes[index]->dividendYield()->discount(T);
1116 Real sigmaSqrtT =
1117 std::max(processes[index]->blackVolatility()->blackVol(T, atmf), 0.1) * std::sqrt(std::max(T, 1.0));
1118 Real xmin = std::exp(std::log(atmf) - normInvEps * sigmaSqrtT);
1119 Real xmax = std::exp(std::log(atmf) + normInvEps * sigmaSqrtT);
1120 for (auto const k : ks.second) {
1121 if (k < xmin || k > xmax) {
1122 DLOG("excluding calibration strike (" << k << ") for index '" << modelIndices[index]
1123 << "', bounds = [" << xmin << "," << xmax << "]");
1124 } else {
1125 result[ks.first].push_back(k);
1126 }
1127 }
1128 } else {
1129 result[ks.first] = ks.second;
1130 }
1131 }
1132 return result;
1133}

1221 // compile cam correlation matrix
1222 // - we want to use the maximum tenor of an ir index in a correlation pair if several are given (to have
1223 // a well defined rule how to derive the LGM IR correlations); to get there we store the correlations
1224 // together with the index tenors (or null if not applicatble), so that we can decide if we overwrite
1225 // an existing correlation with another candidate or not
1226 // - correlations are for index pair names and must be constant; if not given for a pair, we assume zero
1227 // correlation;
1228 // - correlations for IR processes are taken from IR index correlations, if several indices exist
1229 // for one ccy, the index with the longest tenor T is selected; we do not apply an LGM adjustment for this
1230 // value due to time dependent volatility, since this is usually negligible
1231 // - for inf we do not apply an adjustment either => TODO is that suitable for both DK and JY approximately?
1232 // - for inf JY we have two driving factors (f1,f2) where f1 drives the real rate process and f2 drives the
1233 // inflation index ("fx") process; we assume the correlation between f1 and any other factor (including f2)
1234 // to be zero and set the correlation between f2 and any other factor to the correlation read from the
1235 // market data for the inflation index, TODO is that assumption reasonable?
1236 std::map<std::pair<std::string, std::string>,
1237 std::tuple<Handle<QuantExt::CorrelationTermStructure>, Period, Period>>
1238 tmpCorrelations;
1239 for (auto const& c : correlations_) {
1240 std::pair<std::string, Period> firstEntry = convertIndexToCamCorrelationEntry(c.first.first);
1241 std::pair<std::string, Period> secondEntry = convertIndexToCamCorrelationEntry(c.first.second);
1242 // if we have identical CAM entries (e.g. IR:EUR, IR:EUR) we skip this pair, since we can't specify a
1243 // correlation in this case
1244 if (firstEntry.first == secondEntry.first)
1245 continue;
1246 auto e = tmpCorrelations.find(std::make_pair(firstEntry.first, secondEntry.first));
1247 if (e == tmpCorrelations.end() ||
1248 (firstEntry.second > std::get<1>(e->second) && secondEntry.second > std::get<2>(e->second))) {
1249 tmpCorrelations[std::make_pair(firstEntry.first, secondEntry.first)] =
1250 std::make_tuple(c.second, firstEntry.second, secondEntry.second);
1251 }
1252 }

1265 }
1266 auto q = Handle<Quote>(boost::make_shared<CorrelationValue>(std::get<0>(c.second), 0.0));

1269 }
1270
1271 // set up the cam and calibrate it using the cam builder
1272 // if for a non-base currency no fx index is given, we set up a zero vol FX process for this
1273 // if fullDynamicIr is false, we set up a zero vol IR process for currencies that are not irIndex currencies
1274
1275 std::vector<boost::shared_ptr<IrModelData>> irConfigs;
1276 std::vector<boost::shared_ptr<InflationModelData>> infConfigs;
1277 std::vector<boost::shared_ptr<FxBsData>> fxConfigs;

1280 // TODO: populate comConfigs

1282 // calibration expiries and terms for IR, INF, FX, EQ parametrisations (this will only work for a fixed reference
1283 // date, due to the way the cam builder and nested builders work, see ticket #940)
1284 Date referenceDate = modelCurves_.front()->referenceDate();
