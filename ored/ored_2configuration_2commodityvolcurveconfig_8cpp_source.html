52 void CommodityVolatilityConfig::populateRequiredCurveIds() {
53  if (!priceCurveId().empty())
54  requiredCurveIds_[CurveSpec::CurveType::Commodity].insert(parseCurveSpec(priceCurveId())->curveConfigID());
55  if (!yieldCurveId().empty())
56  requiredCurveIds_[CurveSpec::CurveType::Yield].insert(parseCurveSpec(yieldCurveId())->curveConfigID());
57  if (auto vapo = boost::dynamic_pointer_cast<VolatilityApoFutureSurfaceConfig>(volatilityConfig())) {
58  requiredCurveIds_[CurveSpec::CurveType::CommodityVolatility].insert(
59  parseCurveSpec(vapo->baseVolatilityId())->curveConfigID());
60  }

100 void CommodityVolatilityConfig::fromXML(XMLNode* node) {
101 
102  XMLUtils::checkNode(node, "CommodityVolatility");
103 
104  curveID_ = XMLUtils::getChildValue(node, "CurveId", true);
105  curveDescription_ = XMLUtils::getChildValue(node, "CurveDescription", true);
106  currency_ = XMLUtils::getChildValue(node, "Currency", true);
107 
108  XMLNode* n;
109  if ((n = XMLUtils::getChildNode(node, "Constant"))) {
110  volatilityConfig_ = boost::make_shared<ConstantVolatilityConfig>();
111  } else if ((n = XMLUtils::getChildNode(node, "Curve"))) {
112  volatilityConfig_ = boost::make_shared<VolatilityCurveConfig>();
113  } else if ((n = XMLUtils::getChildNode(node, "StrikeSurface"))) {
114  volatilityConfig_ = boost::make_shared<VolatilityStrikeSurfaceConfig>();
115  } else if ((n = XMLUtils::getChildNode(node, "DeltaSurface"))) {
116  volatilityConfig_ = boost::make_shared<VolatilityDeltaSurfaceConfig>();
117  } else if ((n = XMLUtils::getChildNode(node, "MoneynessSurface"))) {
118  volatilityConfig_ = boost::make_shared<VolatilityMoneynessSurfaceConfig>();
119  } else if ((n = XMLUtils::getChildNode(node, "ApoFutureSurface"))) {
120  volatilityConfig_ = boost::make_shared<VolatilityApoFutureSurfaceConfig>();
121  } else {
122  QL_FAIL("CommodityVolatility node expects one child node with name in list: Constant,"
123  << " Curve, StrikeSurface, DeltaSurface, MoneynessSurface, ApoFutureSurface.");
124  }
125  volatilityConfig_->fromXML(n);
126 
127  dayCounter_ = "A365";
128  if ((n = XMLUtils::getChildNode(node, "DayCounter")))
129  dayCounter_ = XMLUtils::getNodeValue(n);
130 
131  calendar_ = "NullCalendar";
132  if ((n = XMLUtils::getChildNode(node, "Calendar")))
133  calendar_ = XMLUtils::getNodeValue(n);
134 
135  futureConventionsId_ = XMLUtils::getChildValue(node, "FutureConventions", false);
136 
137  optionExpiryRollDays_ = 0;
138  if ((n = XMLUtils::getChildNode(node, "OptionExpiryRollDays")))
139  optionExpiryRollDays_ = parseInteger(XMLUtils::getNodeValue(n));
140 
141  priceCurveId_ = XMLUtils::getChildValue(node, "PriceCurveId", false);
142  yieldCurveId_ = XMLUtils::getChildValue(node, "YieldCurveId", false);
143 
144  quoteSuffix_ = XMLUtils::getChildValue(node, "QuoteSuffix", false);
145 
146  solverConfig_ = OneDimSolverConfig();
147  if (XMLNode* n = XMLUtils::getChildNode(node, "OneDimSolverConfig")) {
148  solverConfig_.fromXML(n);
149  }
150 
151  preferOutOfTheMoney_ = boost::none;
152  if (XMLNode* n = XMLUtils::getChildNode(node, "PreferOutOfTheMoney")) {
153  preferOutOfTheMoney_ = parseBool(XMLUtils::getNodeValue(n));
154  }
155 
156  populateQuotes();
157  populateRequiredCurveIds();
158 }

160 XMLNode* CommodityVolatilityConfig::toXML(XMLDocument& doc) {
161 
162  XMLNode* node = doc.allocNode("CommodityVolatility");
163 
164  XMLUtils::addChild(doc, node, "CurveId", curveID_);
165  XMLUtils::addChild(doc, node, "CurveDescription", curveDescription_);
166  XMLUtils::addChild(doc, node, "Currency", currency_);
167 
168  XMLNode* n = volatilityConfig_->toXML(doc);
169  XMLUtils::appendNode(node, n);
170 
171  XMLUtils::addChild(doc, node, "DayCounter", dayCounter_);
172  XMLUtils::addChild(doc, node, "Calendar", calendar_);
173  if (!futureConventionsId_.empty())
174  XMLUtils::addChild(doc, node, "FutureConventions", futureConventionsId_);
175  XMLUtils::addChild(doc, node, "OptionExpiryRollDays", static_cast<int>(optionExpiryRollDays_));
176  if (!priceCurveId_.empty())
177  XMLUtils::addChild(doc, node, "PriceCurveId", priceCurveId_);
178  if (!yieldCurveId_.empty())
179  XMLUtils::addChild(doc, node, "YieldCurveId", yieldCurveId_);
180  if (!quoteSuffix_.empty())
181  XMLUtils::addChild(doc, node, "QuoteSuffix", quoteSuffix_);
182  if (!solverConfig_.empty())
183  XMLUtils::appendNode(node, solverConfig_.toXML(doc));
184  if (preferOutOfTheMoney_)
185  XMLUtils::addChild(doc, node, "PreferOutOfTheMoney", *preferOutOfTheMoney_);
186 
187  return node;
188 }

190 void CommodityVolatilityConfig::populateQuotes() {
191 
192  // The quotes depend on the type of volatility structure that has been configured.
193  if (auto vc = boost::dynamic_pointer_cast<ConstantVolatilityConfig>(volatilityConfig_)) {
194  quotes_ = {vc->quote()};
195  } else if (auto vc = boost::dynamic_pointer_cast<VolatilityCurveConfig>(volatilityConfig_)) {
196  quotes_ = vc->quotes();
197  } else if (auto vc = boost::dynamic_pointer_cast<VolatilitySurfaceConfig>(volatilityConfig_)) {
198  // Clear the quotes_ if necessary and populate with surface quotes
199  quotes_.clear();
200  string quoteType = to_string(volatilityConfig()->quoteType());
201  string stem = "COMMODITY_OPTION/" + quoteType + "/" + curveID_ + "/" + currency_ + "/";
202  for (const pair<string, string>& p : vc->quotes()) {
203  string q = stem + p.first + "/" + p.second;
204  if (!quoteSuffix_.empty())
205  q += "/" + quoteSuffix_;
206  quotes_.push_back(q);
207  }
208  } else {
209  QL_FAIL("CommodityVolatilityConfig expected a constant, curve or surface");
210  }
211 }
