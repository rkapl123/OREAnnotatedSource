56 z = x / y;

59 z = min(x, y);
60 BOOST_REQUIRE_EQUAL(z.which(), 0);

143 constexpr Size n = 1;
144 RandomVariable a(n, 10);
145 EventVec c{n, Date(2, Jan, 2017)};
146 CurrencyVec d{n, "EUR"};
147 IndexVec e{n, "EUR-EURIBOR-6M"};
148 ValueType a2(RandomVariable(n, 0.0));
149 ValueType c2(EventVec{n, Date()});
150 ValueType d2(CurrencyVec{n, ""});
151 ValueType e2(IndexVec{n, ""});
152 // type = type
153 BOOST_REQUIRE_NO_THROW(typeSafeAssign(a2, a););
154 BOOST_REQUIRE_EQUAL(a2, ValueType(a));
155 try {
156 typeSafeAssign(c2, c);
157 } catch (const std::exception& e) {
158 BOOST_FAIL(e.what());
159 }
160 BOOST_REQUIRE_NO_THROW(typeSafeAssign(c2, c););
161 BOOST_REQUIRE_EQUAL(c2, ValueType(c));
162 BOOST_REQUIRE_NO_THROW(typeSafeAssign(d2, d););
163 BOOST_REQUIRE_EQUAL(d2, ValueType(d));
164 BOOST_REQUIRE_NO_THROW(typeSafeAssign(e2, e););
165 BOOST_REQUIRE_EQUAL(e2, ValueType(e));
166}
