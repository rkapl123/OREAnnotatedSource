47Handle<YieldTermStructure> xccyYieldCurve(const boost::shared_ptr<Market>& market, const string& ccyCode,
48 bool& outXccyExists, const string& configuration) {
49
50 Handle<YieldTermStructure> curve;
51 string xccyCurve = xccyCurveName(ccyCode);
52 outXccyExists = true;
53 try {
54 curve = market->yieldCurve(xccyCurve, configuration);
55 } catch (const Error&) {
56 DLOG("Could not link " << ccyCode << " termstructure to cross currency yield curve " << xccyCurve
57 << " so just using " << ccyCode << " discount curve.");
58 curve = market->discountCurve(ccyCode, configuration);
59 outXccyExists = false;
60 }
61
62 return curve;
63}

70std::string creditCurveNameFromSecuritySpecificCreditCurveName(const std::string& name) {
71 if (boost::starts_with(name, "__SECCRCRV_")) {
72 std::size_t pos = name.find("_&_", 11);
73 if (pos != std::string::npos) {
74 std::size_t pos2 = name.find("_&_", pos + 3);
75 if (pos2 != std::string::npos) {
76 auto tmp = name.substr(pos + 3, pos2 - (pos + 3));
77 return tmp;
78 }
79 }
80 }
81 return name;
82}

100std::string prettyPrintInternalCurveName(std::string name) {
101 std::size_t pos = 0;
102 bool found;
103 do {
104 found = false;
105 std::size_t pos2 = name.find("__SECCRCRV_", pos);
106 if (pos2 != std::string::npos) {
107 std::size_t pos3 = name.find("_&_", pos2);
108 if (pos3 != std::string::npos) {
109 std::size_t pos4 = name.find("_&_", pos3 + 3);
110 if (pos4 != std::string::npos) {
111 name.replace(pos2, pos4 + 3 - pos2,
112 name.substr(pos2 + 11, pos3 - (pos2 + 11)) + "(" +
113 name.substr(pos3 + 3, pos4 - (pos3 + 3)) + ")");
114 pos = pos + (pos4 - pos2 - 12);
115 found = true;
116 }
117 }
118 }
119 } while (found);
120 return name;
121}

123boost::shared_ptr<QuantExt::FxIndex> buildFxIndex(const string& fxIndex, const string& domestic, const string& foreign,
124 const boost::shared_ptr<Market>& market, const string& configuration,
125 bool useXbsCurves) {
126
127 auto fxInd = parseFxIndex(fxIndex);
128
129 string source = fxInd->sourceCurrency().code();
130 string target = fxInd->targetCurrency().code();
131 string family = fxInd->familyName();
132
133 fxInd = *market->fxIndex("FX-" + family + "-" + foreign + "-" + domestic);
134
135 QL_REQUIRE((domestic == target && foreign == source) || (domestic == source && foreign == target),
136 "buildFxIndex(): index '" << fxIndex << "' does not match given currencies " << domestic << ", "
137 << foreign);
138
139 if (!useXbsCurves)
140 return fxInd;
141
142 return fxInd->clone(Handle<Quote>(), xccyYieldCurve(market, foreign, configuration),
143 xccyYieldCurve(market, domestic, configuration));
144}

146std::tuple<Natural, Calendar, BusinessDayConvention> getFxIndexConventions(const string& index) {
147 // can take an fx index or ccy pair e.g. EURUSD
148 string ccy1, ccy2;
149 string fixingSource;
150 if (isFxIndex(index)) {
151 auto ind = parseFxIndex(index);
152 ccy1 = ind->sourceCurrency().code();
153 ccy2 = ind->targetCurrency().code();
154 fixingSource = ind->familyName();
155 } else {
156 QL_REQUIRE(index.size() == 6, "getFxIndexConventions: index must be an FXIndex of form FX-ECB-EUR-USD, "
157 << "or a currency pair e.g. EURUSD, got '" + index + "'");
158 ccy1 = index.substr(0, 3);
159 ccy2 = index.substr(3);
160 fixingSource = "GENERIC";
161 }
162
163 if (ccy1 == ccy2)
164 return std::make_tuple(0, NullCalendar(), Unadjusted);
165
166 const boost::shared_ptr<Conventions>& conventions = InstrumentConventions::instance().conventions();
167 boost::shared_ptr<Convention> con;
168 // first look for the index and inverse index directly
169 try {
170 con = conventions->get("FX-" + fixingSource + "-" + ccy1 + "-" + ccy2);
171 } catch (...) {
172 }
173 if (con == nullptr) {
174 try {
175 con = conventions->get("FX-" + fixingSource + "-" + ccy2 + "-" + ccy1);
176 } catch (...) {
177 }
178 }
179 // then by currency pair and inverse currency pair (getFxConvention() handles both)
180 if (con == nullptr) {
181 try {
182 con = conventions->getFxConvention(ccy1, ccy2);
183 } catch (...) {
184 }
185 }
186 if (auto fxCon = boost::dynamic_pointer_cast<FXConvention>(con)) {
187 TLOG("getFxIndexConvention(" << index << "): " << fxCon->spotDays() << " / " << fxCon->advanceCalendar().name()
188 << " from convention.");
189 return std::make_tuple(fxCon->spotDays(), fxCon->advanceCalendar(), fxCon->convention());
190 }
191
192 // default calendar for pseudo currencies is USD
193 if (isPseudoCurrency(ccy1))
194 ccy1 = "USD";
195 if (isPseudoCurrency(ccy2))
196 ccy2 = "USD";
197
198 try {
199 Calendar cal = parseCalendar(ccy1 + "," + ccy2);
200 TLOG("getFxIndexConvention(" << index << "): 2 (default) / " << cal.name()
201 << " (from ccys), no convention found.");
202 return std::make_tuple(2, cal, Following);
203 } catch (const std::exception& e) {
204 ALOG("could not get fx index convention for '" << index << "': " << e.what() << ", continue with 'USD'");
205 }
206 TLOG("getFxIndexConvention(" << index
207 << "): 2 (default) / USD (default), no convention found, could not parse calendar '"
208 << (ccy1 + "," + ccy2) << "'");
209 return std::make_tuple(2, parseCalendar("USD"), Following);
210}

212std::pair<std::string, QuantLib::Period> splitCurveIdWithTenor(const std::string& creditCurveId) {
213 Size pos = creditCurveId.rfind("_");
214 if (pos != std::string::npos) {
215 Period term;
216 string termString = creditCurveId.substr(pos + 1, creditCurveId.length());
217 if (tryParse<Period>(termString, term, parsePeriod)) {
218 return make_pair(creditCurveId.substr(0, pos), term);
219 }
220 }
221 return make_pair(creditCurveId, 0 * Days);
222}
