60//! Base PricingEngine Builder class for a specific model and engine
61/*!
62 * The EngineBuilder is responsible for building pricing engines for a specific
63 * Model and Engine.
64 *
65 * Each builder should implement a method with a signature
66 * @code
67 * boost::shared_ptr<PricingEngine> engine (...);
68 * @endcode
69 * The exact parameters of each method can vary depending on the type of engine.
70 *
71 * An EngineBuilder can cache engines and return the same PricingEngine multiple
72 * times, alternatively the Builder can build a unique PricingEngine each time it
73 * is called.
74 *
75 * For example a swap engine builder can have the interface
76 * @code
77 * boost::shared_ptr<PricingEngine> engine (const Currency&);
78 * @endcode
79 * and so returns the same (cached) engine every time it is asked for a particular
80 * currency.
81 *
82 * The interface of each type of engine builder can be different, then there can
83 * be further sub-classes for different models and engines.
84 *
85 * EngineBuilders are registered in an EngineFactory, multiple engine builders for
86 * the same trade type can be registered with the EngineFactory and it will select
87 * the appropriate one based on configuration.
88 *
89 * Each EngineBuilder must return it's Model and Engine.
90
91 \ingroup tradedata
92 */
93class EngineBuilder {
94public:
95 /*! Constructor that takes a model and engine name
96 * @param model the model name
97 * @param engine the engine name
98 * @param tradeTypes a set of trade types
99 */
100 EngineBuilder(const string& model, const string& engine, const set<string>& tradeTypes)
101 : model_(model), engine_(engine), tradeTypes_(tradeTypes) {}
102
103 //! Virtual destructor
104 virtual ~EngineBuilder() {}
105
106 //! Return the model name
107 const string& model() const { return model_; }
108 //! Return the engine name
109 const string& engine() const { return engine_; }
110 //! Return the possible trade types
111 const set<string>& tradeTypes() const { return tradeTypes_; }
112
113 //! Return a configuration (or the default one if key not found)
114 const string& configuration(const MarketContext& key) {
115 if (configurations_.count(key) > 0) {
116 return configurations_.at(key);
117 } else {
118 return Market::defaultConfiguration;
119 }
120 }
121
122 //! reset the builder (e.g. clear cache)
123 virtual void reset() {}
124
125 //! Initialise this Builder with the market and parameters to use
126 /*! This method should not be called directly, it is called by the EngineFactory
127 * before it is returned.
128 */
129 void init(const boost::shared_ptr<Market> market, const map<MarketContext, string>& configurations,
130 const map<string, string>& modelParameters, const map<string, string>& engineParameters,
131 const std::map<std::string, std::string>& globalParameters = {}) {
132 market_ = market;
133 configurations_ = configurations;
134 modelParameters_ = modelParameters;
135 engineParameters_ = engineParameters;
136 globalParameters_ = globalParameters;
137 }
138
139 //! return model builders
140 const set<std::pair<string, boost::shared_ptr<ModelBuilder>>>& modelBuilders() const { return modelBuilders_; }
141
142 /*! retrieve engine parameter p, first look for p_qualifier, if this does not exist fall back to p */
143 std::string engineParameter(const std::string& p, const std::vector<std::string>& qualifiers = {},
144 const bool mandatory = true, const std::string& defaultValue = "") const;
145 /*! retrieve model parameter p, first look for p_qualifier, if this does not exist fall back to p */
146 std::string modelParameter(const std::string& p, const std::vector<std::string>& qualifiers = {},
147 const bool mandatory = true, const std::string& defaultValue = "") const;
148
149protected:
150 string model_;
151 string engine_;
152 set<string> tradeTypes_;
153 boost::shared_ptr<Market> market_;
154 map<MarketContext, string> configurations_;
155 map<string, string> modelParameters_;
156 map<string, string> engineParameters_;
157 std::map<std::string, std::string> globalParameters_;
158 set<std::pair<string, boost::shared_ptr<ModelBuilder>>> modelBuilders_;
159};

171//! Pricing Engine Factory class
172/*! A Pricing Engine Factory is used when building a portfolio, it provides
173 * QuantLib::PricingEngines to each of the Trade objects.
174 *
175 * An Engine Factory is configured on two levels, both from EngineData.
176 * The first level is the type of builder (Model and Engine) to use for each
177 * trade type, so given a trade type one asks the factory for a builder and
178 * then the builder for a PricingEngine. Each builder must be registered with
179 * the factory and then the configuration defines which builder to use.
180 *
181 * Secondly, the factory maintains builder specific parameters for each Model
182 * and Engine.
183
184 \ingroup tradedata
185 */
186class EngineFactory {
187public:
188 //! Create an engine factory
189 EngineFactory( //! Configuration data
190 const boost::shared_ptr<EngineData>& data,
191 //! The market that is passed to each builder
192 const boost::shared_ptr<Market>& market,
193 //! The market configurations that are passed to each builder
194 const map<MarketContext, string>& configurations = std::map<MarketContext, string>(),
195 //! additional engine builders
196 const std::vector<boost::shared_ptr<EngineBuilder>> extraEngineBuilders = {},
197 //! additional leg builders
198 const std::vector<boost::shared_ptr<LegBuilder>> extraLegBuilders = {},
199 //! optional pointer to reference data
200 const boost::shared_ptr<ReferenceDataManager>& referenceData = nullptr,
201 //! ibor fallback config
202 const IborFallbackConfig& iborFallbackConfig = IborFallbackConfig::defaultConfig());
203
204 //! Return the market used by this EngineFactory
205 const boost::shared_ptr<Market>& market() const { return market_; };
206 //! Return the market configurations used by this EngineFactory
207 const map<MarketContext, string>& configurations() const { return configurations_; };
208 //! Return a configuration (or the default one if key not found)
209 const string& configuration(const MarketContext& key) {
210 if (configurations_.count(key) > 0) {
211 return configurations_.at(key);
212 } else {
213 return Market::defaultConfiguration;
214 }
215 }
216 //! Return the EngineData parameters
217 const boost::shared_ptr<EngineData> engineData() const { return engineData_; };
218 //! Register a builder with the factory
219 void registerBuilder(const boost::shared_ptr<EngineBuilder>& builder);
220 //! Return the reference data used by this EngineFactory
221 const boost::shared_ptr<ReferenceDataManager>& referenceData() const { return referenceData_; };
222 //! Return the ibor fallback config
223 const IborFallbackConfig& iborFallbackConfig() const { return iborFallbackConfig_; }
224
225 //! Get a builder by trade type
226 /*! This will look up configured model/engine for that trade type
227 the returned builder can be cast to the type required for the tradeType.
228
229 The factory will call EngineBuilder::init() before returning it.
230 */
231 boost::shared_ptr<EngineBuilder> builder(const string& tradeType);
232
233 //! Register a leg builder with the factory
234 void registerLegBuilder(const boost::shared_ptr<LegBuilder>& legBuilder);
235
236 //! Get a leg builder by leg type
237 boost::shared_ptr<LegBuilder> legBuilder(const string& legType);
238
239 //! Add a set of default engine and leg builders
240 void addDefaultBuilders();
241 //! Add a set of default engine and leg builders
242 void addExtraBuilders(const std::vector<boost::shared_ptr<EngineBuilder>> extraEngineBuilders,
243 const std::vector<boost::shared_ptr<LegBuilder>> extraLegBuilders);
244
245 //! Clear all builders
246 void clear() {
247 builders_.clear();
248 legBuilders_.clear();
249 }
250
251 //! return model builders
252 set<std::pair<string, boost::shared_ptr<ModelBuilder>>> modelBuilders() const;
253
254private:
255 boost::shared_ptr<Market> market_;
256 boost::shared_ptr<EngineData> engineData_;
257 map<MarketContext, string> configurations_;
258 map<tuple<string, string, set<string>>, boost::shared_ptr<EngineBuilder>> builders_;
259 map<string, boost::shared_ptr<LegBuilder>> legBuilders_;
260 boost::shared_ptr<ReferenceDataManager> referenceData_;
261 IborFallbackConfig iborFallbackConfig_;
262};
