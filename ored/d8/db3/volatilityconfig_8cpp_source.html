31 void VolatilityConfig::fromBaseNode(XMLNode* node) {
32  string qType = XMLUtils::getChildValue(node, "QuoteType", false);
33  if (qType == "ImpliedVolatility" || qType == "") {
34  string volType = XMLUtils::getChildValue(node, "VolatilityType", false);
35  if (volType == "Lognormal" || qType == "") {
36  quoteType_ = MarketDatum::QuoteType::RATE_LNVOL;
37  } else if (volType == "ShiftedLognormal") {
38  quoteType_ = MarketDatum::QuoteType::RATE_SLNVOL;
39  } else if (volType == "Normal") {
40  quoteType_ = MarketDatum::QuoteType::RATE_NVOL;
41  } else {
42  QL_FAIL("Volatiltiy type " << volType << " is not supported;");
43  }
44  } else if (qType == "Premium") {
45  quoteType_ = MarketDatum::QuoteType::PRICE;
46  // If we have premiums the exercise type is required
47  exerciseType_ = parseExerciseType(XMLUtils::getChildValue(node, "ExerciseType", true));
48  } else {
49  QL_FAIL("Invalid quote type for volitility curve , quote type must be ImpliedVolatility or Premium");
50  }
51 }

53 void VolatilityConfig::addBaseNode(XMLDocument& doc, XMLNode* node) {
54 
55  // Check first for premium
56  if (quoteType_ == MarketDatum::QuoteType::PRICE) {
57  XMLUtils::addChild(doc, node, "QuoteType", "Premium");
58  XMLUtils::addChild(doc, node, "ExerciseType", to_string(exerciseType_));
59  return;
60  }
61 
62  // Must be a volatility (or possibly fail)
63  XMLUtils::addChild(doc, node, "QuoteType", "ImpliedVolatility");
64  if (quoteType_ == MarketDatum::QuoteType::RATE_LNVOL) {
65  XMLUtils::addChild(doc, node, "VolatilityType", "Lognormal");
66  } else if (quoteType_ == MarketDatum::QuoteType::RATE_SLNVOL) {
67  XMLUtils::addChild(doc, node, "VolatilityType", "ShiftedLognormal");
68  } else if (quoteType_ == MarketDatum::QuoteType::RATE_NVOL) {
69  XMLUtils::addChild(doc, node, "VolatilityType", "Normal");
70  } else {
71  QL_FAIL("Invalid quote type");
72  }
73 }

251 vector<pair<string, string>> VolatilityDeltaSurfaceConfig::quotes() const {
252 
253  vector<pair<string, string>> result;
254 
255  // ATM strike string
256  string atmString = "ATM/" + atmType_;
257  if (!atmDeltaType_.empty())
258  atmString += "/DEL/" + atmDeltaType_;
259 
260  // Delta stem
261  string stem = "DEL/" + deltaType_ + "/";
262 
263  for (const string& e : expiries()) {
264  result.push_back(make_pair(e, atmString));
265  for (const string& d : putDeltas_) {
266  result.push_back(make_pair(e, stem + "Put/" + d));
267  }
268  for (const string& d : callDeltas_) {
269  result.push_back(make_pair(e, stem + "Call/" + d));
270  }
271  }
272 
273  return result;
274 }
