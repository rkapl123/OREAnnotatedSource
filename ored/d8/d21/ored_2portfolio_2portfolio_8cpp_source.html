63 void Portfolio::fromXML(XMLNode* node, const boost::shared_ptr<TradeFactory>& factory,
64  const bool checkForDuplicateIds) {
65  XMLUtils::checkNode(node, "Portfolio");
66  vector<XMLNode*> nodes = XMLUtils::getChildrenNodes(node, "Trade");
67  for (Size i = 0; i < nodes.size(); i++) {
68  string tradeType = XMLUtils::getChildValue(nodes[i], "TradeType", true);
69 
70  // Get the id attribute
71  string id = XMLUtils::getAttribute(nodes[i], "id");
72  QL_REQUIRE(id != "", "No id attribute in Trade Node");
73  DLOG("Parsing trade id:" << id);
74  boost::shared_ptr<Trade> trade = factory->build(tradeType);
75 
76  if (trade) {
77  try {
78  trade->fromXML(nodes[i]);
79  trade->id() = id;
80  add(trade, checkForDuplicateIds);
81 
82  DLOG("Added Trade " << id << " (" << trade->id() << ")"
83  << " type:" << tradeType);
84  } catch (std::exception& ex) {
85  ALOG(StructuredTradeErrorMessage(id, tradeType, "Error parsing Trade XML", ex.what()));
86  }
87  } else {
88  WLOG("Unable to build Trade for tradeType=" << tradeType);
89  }
90  }
91  LOG("Finished Parsing XML doc");
92 }

127 void Portfolio::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
128  LOG("Building Portfolio of size " << trades_.size());
129  auto trade = trades_.begin();
130  while (trade != trades_.end()) {
131  try {
132  (*trade)->reset();
133  (*trade)->build(engineFactory);
134  TLOG("Required Fixings for trade " << (*trade)->id() << ":");
135  TLOGGERSTREAM << (*trade)->requiredFixings();
136  ++trade;
137  } catch (std::exception& e) {
138  ALOG(StructuredTradeErrorMessage(*trade, "Error building trade", e.what()));
139  trade = trades_.erase(trade);
140  }
141  }
142  LOG("Built Portfolio. Size now " << trades_.size());
143 
144  QL_REQUIRE(trades_.size() > 0, "Portfolio does not contain any built trades");
145 }

147 Date Portfolio::maturity() const {
148  QL_REQUIRE(trades_.size() > 0, "Cannot get maturity of an empty portfolio");
149  Date mat = trades_.front()->maturity();
150  for (const auto& t : trades_)
151  mat = std::max(mat, t->maturity());
152  return mat;
