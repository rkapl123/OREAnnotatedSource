32std::string getParameter(const std::map<std::string, std::string>& m, const std::string& p,
33 const std::vector<std::string>& qs, const bool mandatory, const std::string& defaultValue) {
34 // first look for p_q if one or several qualifiers are given
35 for (auto const& q : qs) {
36 if (!q.empty()) {
37 auto r = m.find(p + "_" + q);
38 if (r != m.end())
39 return r->second;
40 }
41 }
42 // no qualifier given, or fall back on p because p_q was not found
43 auto r = m.find(p);
44 if (r != m.end()) {
45 return r->second;
46 }
47 // if parameter is mandatory throw, otherwise return the default value
48 if (mandatory) {
49 QL_FAIL("parameter " << p << " not found (qualifier list was \"" << boost::algorithm::join(qs, ", ") << "\")");
50 }
51 return defaultValue;
52}

65void EngineBuilderFactory::addEngineBuilder(const std::function<boost::shared_ptr<EngineBuilder>()>& builder,
66 const bool allowOverwrite) {
67 boost::unique_lock<boost::shared_mutex> lock(mutex_);
68 auto tmp = builder();
69 auto key = make_tuple(tmp->model(), tmp->engine(), tmp->tradeTypes());
70 auto it = std::remove_if(engineBuilderBuilders_.begin(), engineBuilderBuilders_.end(),
71 [&key](std::function<boost::shared_ptr<EngineBuilder>()>& b) {
72 auto tmp = b();
73 return key == std::make_tuple(tmp->model(), tmp->engine(), tmp->tradeTypes());
74 });
75 QL_REQUIRE(it == engineBuilderBuilders_.end() || allowOverwrite,
76 "EngineBuilderFactory::addEngineBuilder(" << tmp->model() << "/" << tmp->engine() << "/"
77 << boost::algorithm::join(tmp->tradeTypes(), ",")
78 << "): builder for given key already exists.");
79 engineBuilderBuilders_.erase(it, engineBuilderBuilders_.end());
80 engineBuilderBuilders_.push_back(builder);
81}

158void EngineFactory::registerBuilder(const boost::shared_ptr<EngineBuilder>& builder, const bool allowOverwrite) {
159 const string& modelName = builder->model();
160 const string& engineName = builder->engine();
161 auto key = make_tuple(modelName, engineName, builder->tradeTypes());
162 if (allowOverwrite)
163 builders_.erase(key);
164 QL_REQUIRE(builders_.insert(make_pair(key, builder)).second,
165 "EngineFactory: duplicate engine builder for (" << modelName << "/" << engineName << "/"
166 << boost::algorithm::join(builder->tradeTypes(), ",")
167 << ") - this is an internal error.");
168}

170boost::shared_ptr<EngineBuilder> EngineFactory::builder(const string& tradeType) {
171 // Check that we have a model/engine for tradetype
172 QL_REQUIRE(engineData_->hasProduct(tradeType),
173 "No Pricing Engine configuration was provided for trade type " << tradeType);
174
175 // Find a builder for the model/engine/tradeType
176 const string& model = engineData_->model(tradeType);
177 const string& engine = engineData_->engine(tradeType);
178 typedef pair<tuple<string, string, set<string>>, boost::shared_ptr<EngineBuilder>> map_type;
179 auto it =
180 std::find_if(builders_.begin(), builders_.end(), [&model, &engine, &tradeType](const map_type& v) -> bool {
181 const set<string>& types = std::get<2>(v.first);
182 return std::get<0>(v.first) == model && std::get<1>(v.first) == engine &&
183 std::find(types.begin(), types.end(), tradeType) != types.end();
184 });
185 QL_REQUIRE(it != builders_.end(), "No EngineBuilder for " << model << "/" << engine << "/" << tradeType);
186
187 boost::shared_ptr<EngineBuilder> builder = it->second;
188 string effectiveTradeType = tradeType;
189 if(auto db = boost::dynamic_pointer_cast<DelegatingEngineBuilder>(builder))
190 effectiveTradeType = db->effectiveTradeType();
191
192 builder->init(market_, configurations_, engineData_->modelParameters(effectiveTradeType),
193 engineData_->engineParameters(effectiveTradeType), engineData_->globalParameters());
194
195 return builder;
196}

228 const std::vector<boost::shared_ptr<LegBuilder>> extraLegBuilders,
229 const bool allowOverwrite) {
230
231 if (extraEngineBuilders.size() > 0) {

237 DLOG("adding " << extraLegBuilders.size() << " extra leg builders");
238 for (auto elb : extraLegBuilders)

240 }
