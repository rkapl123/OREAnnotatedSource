82std::string getParameter(const std::map<std::string, std::string>& m, const std::string& p,
83 const std::vector<std::string>& qs, const bool mandatory, const std::string& defaultValue) {
84 // first look for p_q if one or several qualifiers are given
85 for (auto const& q : qs) {
86 if (!q.empty()) {
87 auto r = m.find(p + "_" + q);
88 if (r != m.end())
89 return r->second;
90 }
91 }
92 // no qualifier given, or fall back on p because p_q was not found
93 auto r = m.find(p);
94 if (r != m.end()) {
95 return r->second;
96 }
97 // if parameter is mandatory throw, otherwise return the default value
98 if (mandatory) {
99 QL_FAIL("parameter " << p << " not found (qualifier list was \"" << boost::algorithm::join(qs, ", ") << "\")");
100 }
101 return defaultValue;
102}

135boost::shared_ptr<EngineBuilder> EngineFactory::builder(const string& tradeType) {
136 // Check that we have a model/engine for tradetype
137 QL_REQUIRE(engineData_->hasProduct(tradeType),
138 "No Pricing Engine configuration was provided for trade type " << tradeType);
139
140 // Find a builder for the model/engine/tradeType
141 const string& model = engineData_->model(tradeType);
142 const string& engine = engineData_->engine(tradeType);
143 typedef pair<tuple<string, string, set<string>>, boost::shared_ptr<EngineBuilder>> map_type;
144 auto it =
145 std::find_if(builders_.begin(), builders_.end(), [&model, &engine, &tradeType](const map_type& v) -> bool {
146 const set<string>& types = std::get<2>(v.first);
147 return std::get<0>(v.first) == model && std::get<1>(v.first) == engine &&
148 std::find(types.begin(), types.end(), tradeType) != types.end();
149 });
150 QL_REQUIRE(it != builders_.end(), "No EngineBuilder for " << model << "/" << engine << "/" << tradeType);
151
152 boost::shared_ptr<EngineBuilder> builder = it->second;
153 string effectiveTradeType = tradeType;
154 if(auto db = boost::dynamic_pointer_cast<DelegatingEngineBuilder>(builder))
155 effectiveTradeType = db->effectiveTradeType();
156
157 builder->init(market_, configurations_, engineData_->modelParameters(effectiveTradeType),
158 engineData_->engineParameters(effectiveTradeType), engineData_->globalParameters());
159
160 return builder;
161}

295void EngineFactory::addExtraBuilders(const std::vector<boost::shared_ptr<EngineBuilder>> extraEngineBuilders,
296 const std::vector<boost::shared_ptr<LegBuilder>> extraLegBuilders) {
297
298 if (extraEngineBuilders.size() > 0) {
299 DLOG("adding " << extraEngineBuilders.size() << " extra engine builders");
300 for (auto eb : extraEngineBuilders)
301 registerBuilder(eb);
302 }
303 if (extraLegBuilders.size() > 0) {
304 DLOG("adding " << extraLegBuilders.size() << " extra leg builders");
305 for (auto elb : extraLegBuilders)
306 registerLegBuilder(elb);
307 }
308}
