55void EquityMarginLegData::fromXML(XMLNode* node) {
56 XMLUtils::checkNode(node, legNodeName());
57 rates_ = XMLUtils::getChildrenValuesWithAttributes<Real>(node, "Rates", "Rate", "startDate", rateDates_, parseReal,
58 true);
59 initialMarginFactor_ = XMLUtils::getChildValueAsDouble(node, "InitialMarginFactor", true);
60 multiplier_ = 1;
61 if (XMLUtils::getChildNode(node, "Multiplier"))
62 multiplier_ = XMLUtils::getChildValueAsDouble(node, "Multiplier");
63 XMLNode* equityNode = XMLUtils::getChildNode(node, "EquityLegData");
64 QL_REQUIRE(equityNode, "no equityLegData provided");
65
66 boost::shared_ptr<ore::data::EquityLegData> ld = boost::make_shared<EquityLegData>();
67 ld->fromXML(equityNode);
68 equityLegData_ = ld;
69
70}

72XMLNode* EquityMarginLegData::toXML(XMLDocument& doc) {
73 XMLNode* node = doc.allocNode(legNodeName());
74 XMLUtils::addChildrenWithOptionalAttributes(doc, node, "Rates", "Rate", rates_, "startDate", rateDates_);
75 XMLUtils::addChild(doc, node, "InitialMarginFactor", initialMarginFactor_);
76 XMLUtils::addChild(doc, node, "Multiplier", multiplier_);
77
78 XMLUtils::appendNode(node, equityLegData_->toXML(doc));
79 return node;
80}

82QuantExt::Leg makeEquityMarginLeg(const LegData& data, const boost::shared_ptr<EquityIndex>& equityCurve,
83 const boost::shared_ptr<QuantExt::FxIndex>& fxIndex,
84 const QuantLib::Date& openEndDateReplacement) {
85 boost::shared_ptr<EquityMarginLegData> eqMarginLegData = boost::dynamic_pointer_cast<EquityMarginLegData>(data.concreteLegData());
86 QL_REQUIRE(eqMarginLegData, "Wrong LegType, expected EquityMargin, got " << data.legType());
87 boost::shared_ptr<ore::data::EquityLegData> eqLegData = eqMarginLegData->equityLegData();
88 QL_REQUIRE(eqLegData, "expected equityLegData");
89 QuantExt::Schedule schedule = makeSchedule(data.schedule(), openEndDateReplacement);
90 DayCounter dc = parseDayCounter(data.dayCounter());
91 BusinessDayConvention bdc = ore::data::parseBusinessDayConvention(data.paymentConvention());
92 bool isTotalReturn = eqLegData->returnType() == EquityReturnType::Total;
93 Real dividendFactor = eqLegData->dividendFactor();
94 Real initialPrice = eqLegData->initialPrice();
95 bool initialPriceIsInTargetCcy = false;
96
97 if (!eqLegData->initialPriceCurrency().empty()) {
98 // parse currencies to handle minor currencies
99 Currency initialPriceCurrency = parseCurrencyWithMinors(eqLegData->initialPriceCurrency());
100 Currency dataCurrency = parseCurrencyWithMinors(data.currency());
101 Currency eqCurrency;
102 // set equity currency
103 if (!equityCurve->currency().empty())
104 eqCurrency = equityCurve->currency();
105 else
106 TLOG("Cannot find currency for equity " << equityCurve->name());
107
108 // initial price currency must match leg or equity currency
109 QL_REQUIRE(initialPriceCurrency == dataCurrency ||
110 initialPriceCurrency == eqCurrency || eqCurrency.empty(),
111 "initial price ccy (" << initialPriceCurrency << ") must match either leg ccy ("
112 << dataCurrency << ") or equity ccy (if given, got '"
113 << eqCurrency << "')");
114 initialPriceIsInTargetCcy = initialPriceCurrency == dataCurrency;
115 // adjust for minor currency
116 initialPrice = convertMinorToMajorCurrency(eqLegData->initialPriceCurrency(), initialPrice);
117 }
118 bool notionalReset = eqLegData->notionalReset();
119 Natural fixingDays = eqLegData->fixingDays();
120 PaymentLag paymentLag = parsePaymentLag(data.paymentLag());
121 ScheduleData valuationData = eqLegData->valuationSchedule();
122 Schedule valuationSchedule;
123 if (valuationData.hasData())
124 valuationSchedule = makeSchedule(valuationData, openEndDateReplacement);
125 vector<double> notionals = buildScheduledVector(data.notionals(), data.notionalDates(), schedule);
126 vector<double> rates = buildScheduledVector(eqMarginLegData->rates(), eqMarginLegData->rateDates(), schedule);
127
128 applyAmortization(notionals, data, schedule, false);
129 Leg leg = EquityMarginLeg(schedule, equityCurve, fxIndex)
130 .withCouponRates(rates, dc)
131 .withInitialMarginFactor(eqMarginLegData->initialMarginFactor())
132 .withNotionals(notionals)
133 .withQuantity(eqLegData->quantity())
134 .withPaymentDayCounter(dc)
135 .withPaymentAdjustment(bdc)
136 .withPaymentLag(boost::apply_visitor(PaymentLagInteger(), paymentLag))
137 .withTotalReturn(isTotalReturn)
138 .withDividendFactor(dividendFactor)
139 .withInitialPrice(initialPrice)
140 .withInitialPriceIsInTargetCcy(initialPriceIsInTargetCcy)
141 .withNotionalReset(notionalReset)
142 .withFixingDays(fixingDays)
143 .withValuationSchedule(valuationSchedule)
144 .withMultiplier(eqMarginLegData->multiplier());
145
146 QL_REQUIRE(leg.size() > 0, "Empty Equity Margin Leg");
147
148 return leg;
149}
