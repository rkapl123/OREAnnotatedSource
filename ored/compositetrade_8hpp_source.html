34//! Composite Trade class
35/*! CompositeTrades are single currency strategies consisting of independent financial instruments but regarded as a
36 single position n the portfolio. Examples include straddles, butterflies, iron condors. The class can also be used to
37 create representations of single contracts which can be replicated by linear combinations of other positions. E.g. Bond
38 + Bond Option = Callable Bond. \ingroup portfolio
39*/
40class CompositeTrade : public Trade {
41public:
42 /// This enum decalres how the notional of the CompositeTrade should be calculated
43 enum class NotionalCalculation {
44 Sum, ///< The notional is calculated as the sum of notionals of subtrades
45 Mean, ///< The notional is calculated as the average of notionals of subtrades
46 First, ///< The notional is taken as the first subtrade notional
47 Last, ///< The notional is taken as the last subtrade notional
48 Min, ///< The notional is taken as the minimum subtrade notional
49 Max, ///< The notional is taken as the maximum subtrade notional
50 Override ///< The notional is explicitly overridden
51 };
52
53 //! Constructor requires a trade factory so that subtrades can be built
54 CompositeTrade(const Envelope& env = Envelope(), const TradeActions& ta = TradeActions())
55 : Trade("CompositeTrade", env, ta) {
56 reset();
57 }
58
59 //! Fully-specified Constructor
60 CompositeTrade(const string currency, const vector<boost::shared_ptr<Trade>>& trades,
61 const string notionalCalculation = "", const Real notionalOverride = 0.0,
62 const Envelope& env = Envelope(), const TradeActions& ta = TradeActions())
63 : Trade("CompositeTrade", env, ta), currency_(currency), notionalOverride_(notionalOverride),
64 notionalCalculation_(notionalCalculation), trades_(trades) {}
