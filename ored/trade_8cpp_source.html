33void Trade::fromXML(XMLNode* node) {
34 XMLUtils::checkNode(node, "Trade");
35 tradeType_ = XMLUtils::getChildValue(node, "TradeType", true);
36 if (XMLNode* envNode = XMLUtils::getChildNode(node, "Envelope")) {
37 envelope_.fromXML(envNode);
38 }
39 tradeActions_.clear();
40 XMLNode* taNode = XMLUtils::getChildNode(node, "TradeActions");
41 if (taNode)
42 tradeActions_.fromXML(taNode);
43}

45XMLNode* Trade::toXML(XMLDocument& doc) {
46 // Crete Trade Node with Id attribute.
47 XMLNode* node = doc.allocNode("Trade");
48 QL_REQUIRE(node, "Failed to create trade node");
49 XMLUtils::addAttribute(doc, node, "id", id_);
50 XMLUtils::addChild(doc, node, "TradeType", tradeType_);
51 XMLUtils::appendNode(node, envelope_.toXML(doc));
52 if (!tradeActions_.empty())
53 XMLUtils::appendNode(node, tradeActions_.toXML(doc));
54 return node;
55}

57Date Trade::addPremiums(std::vector<boost::shared_ptr<Instrument>>& addInstruments, std::vector<Real>& addMultipliers,
58 const Real tradeMultiplier, const PremiumData& premiumData, const Real premiumMultiplier,
59 const Currency& tradeCurrency, const boost::shared_ptr<EngineFactory>& factory,
60 const string& configuration) {
61
62 Date latestPremiumPayDate = Date::minDate();
63
64 for (auto const& d : premiumData.premiumData()) {
65 QL_REQUIRE(d.amount != Null<Real>(), "Trade contains invalid premium data.");
66
67 Currency premiumCurrency = parseCurrencyWithMinors(d.ccy);
68 Real premiumAmount = convertMinorToMajorCurrency(d.ccy, d.amount);
69 auto fee = boost::make_shared<QuantExt::Payment>(premiumAmount, premiumCurrency, d.payDate);
70
71 addMultipliers.push_back(premiumMultiplier);
72
73 Handle<YieldTermStructure> yts = factory->market()->discountCurve(d.ccy, configuration);
74 Handle<Quote> fx;
75 if (tradeCurrency.code() != d.ccy) {
76 fx = factory->market()->fxRate(d.ccy + tradeCurrency.code(), configuration);
77 }
78 boost::shared_ptr<PricingEngine> discountingEngine(new QuantExt::PaymentDiscountingEngine(yts, fx));
79 fee->setPricingEngine(discountingEngine);
80
81 // 1) Add to additional instruments for pricing
82 addInstruments.push_back(fee);
83
84 // 2) Add a trade leg for cash flow reporting, divide the amount by the multiplier, because the leg entries
85 // are multiplied with the trade multiplier in the cashflow report (and if used elsewhere)
86 legs_.push_back(
87 Leg(1, boost::make_shared<SimpleCashFlow>(fee->cashFlow()->amount() * premiumMultiplier / tradeMultiplier,
88 fee->cashFlow()->date())));
89 legCurrencies_.push_back(fee->currency().code());
90
91 // premium * premiumMultiplier reflects the correct pay direction, set payer to false therefore
92 legPayers_.push_back(false);
93
94 // update latest premium pay date
95 latestPremiumPayDate = std::max(latestPremiumPayDate, d.payDate);
96
97 DLOG("added fee " << d.amount << " " << d.ccy << " payable on " << d.payDate << " to trade");
98 }
99
100 return latestPremiumPayDate;
101}

103void Trade::validate() const {
104 QL_REQUIRE(id_ != "", "Trade id has not been set.");
105 QL_REQUIRE(tradeType_ != "", "Trade id has not been set.");
106 QL_REQUIRE(instrument_ || legs_.size() > 0,
107 "Trade " << id_ << " requires either QuantLib instruments or legs to be created.");
108 QL_REQUIRE(npvCurrency_ != "", "NPV currency has not been set for trade " << id_ << ".");
109 // QL_REQUIRE(notional_ != Null<Real>(), "Notional has not been set for trade " << id_ << ".");
110 // QL_REQUIRE(notionalCurrency_ != "", "Notional currency has not been set for trade " << id_ << ".");
111 QL_REQUIRE(maturity_ != Null<Date>(), "Maturity not set for trade " << id_ << ".");
112 QL_REQUIRE(!envelope_.empty(), "Envelope not set for trade " << id_ << ".");
113 if (legs_.size() > 0) {
114 QL_REQUIRE(legs_.size() == legPayers_.size(),
115 "Inconsistent number of pay/receive indicators for legs in trade " << id_ << ".");
116 QL_REQUIRE(legs_.size() == legCurrencies_.size(),
117 "Inconsistent number of leg currencies for legs in trade " << id_ << ".");
118 }
119}

121void Trade::reset() {
122 // save accumulated timings from wrapper to trade before resetting
123 if (instrument_ != nullptr) {
124 savedNumberOfPricings_ += instrument_->getNumberOfPricings();
125 savedCumulativePricingTime_ += instrument_->getCumulativePricingTime();
126 }
127 // reset members
128 instrument_ = boost::shared_ptr<InstrumentWrapper>();
129 legs_.clear();
130 legCurrencies_.clear();
131 legPayers_.clear();
132 npvCurrency_ = "";
133 notional_ = Null<Real>();
134 notionalCurrency_ = "";
135 maturity_ = Date();
136 issuer_ = "";
137 requiredFixings_.clear();
138}

142void Trade::setLegBasedAdditionalData(const Size i, Size resultLegId) const {
143 if (legs_.size() < i + 1)
144 return;
145 Date asof = Settings::instance().evaluationDate();
146 string legID = std::to_string(resultLegId == Null<Size>() ? i + 1 : resultLegId);
147 for (Size j = 0; j < legs_[i].size(); ++j) {
148 boost::shared_ptr<CashFlow> flow = legs_[i][j];
149 // pick flow with earliest future payment date on this leg
150 if (flow->date() > asof) {
151 Real flowAmount = 0.0;
152 try {
153 flowAmount = flow->amount();
154 } catch (std::exception& e) {
155 ALOG("flow amount could not be determined for trade " << id() << ", set to zero: " << e.what());
156 }
157 additionalData_["amount[" + legID + "]"] = flowAmount;
158 additionalData_["paymentDate[" + legID + "]"] = ore::data::to_string(flow->date());
159 boost::shared_ptr<Coupon> coupon = boost::dynamic_pointer_cast<Coupon>(flow);
160 if (coupon) {
161 Real currentNotional = 0;
162 try {
163 currentNotional = coupon->nominal();
164 } catch (std::exception& e) {
165 ALOG("current notional could not be determined for trade " << id()
166 << ", set to zero: " << e.what());
167 }
168 additionalData_["currentNotional[" + legID + "]"] = currentNotional;
169
170 Real rate = 0;
171 try {
172 rate = coupon->rate();
173 } catch (std::exception& e) {
174 ALOG("coupon rate could not be determined for trade " << id() << ", set to zero: " << e.what());
175 }
176 additionalData_["rate[" + legID + "]"] = rate;
177
178 if (auto frc = boost::dynamic_pointer_cast<FloatingRateCoupon>(flow)) {
179 additionalData_["index[" + legID + "]"] = frc->index()->name();
180 additionalData_["spread[" + legID + "]"] = frc->spread();
181 }
182
183 if (auto eqc = boost::dynamic_pointer_cast<QuantExt::EquityCoupon>(flow)) {
184 auto arc = eqc->pricer()->additionalResultCache();
185 additionalData_["initialPrice[" + legID + "]"] = arc.initialPrice;
186 additionalData_["endEquityFixing[" + legID + "]"] = arc.endFixing;
187 if (arc.startFixing != Null<Real>())
188 additionalData_["startEquityFixing[" + legID + "]"] = arc.startFixing;
189 if (arc.startFixingTotal != Null<Real>())
190 additionalData_["startEquityFixingTotal[" + legID + "]"] =
191 arc.startFixingTotal;
192 if (arc.endFixingTotal != Null<Real>())
193 additionalData_["endEquityFixingTotal[" + legID + "]"] = arc.endFixingTotal;
194 if (arc.startFxFixing != Null<Real>())
195 additionalData_["startFxFixing[" + legID + "]"] = arc.startFxFixing;
196 if (arc.endFxFixing != Null<Real>())
197 additionalData_["endFxFixing[" + legID + "]"] = arc.endFxFixing;
198 if (arc.pastDividends != Null<Real>())
199 additionalData_["pastDividends[" + legID + "]"] = arc.pastDividends;
200 if (arc.forecastDividends != Null<Real>())
201 additionalData_["forecastDividends[" + legID + "]"] = arc.forecastDividends;
202 }
203
204 if (auto cpic = boost::dynamic_pointer_cast<QuantExt::CPICoupon>(flow)) {
205 Real baseCPI;
206 baseCPI = cpic->baseCPI();
207 if (baseCPI == Null<Real>()) {
208 try {
209 baseCPI =
210 QuantLib::CPI::laggedFixing(cpic->cpiIndex(), cpic->baseDate() + cpic->observationLag(),
211 cpic->observationLag(), cpic->observationInterpolation());
212 } catch (std::exception& e) {
213 ALOG("CPICoupon baseCPI could not be interpolated for additional results for trade " << id()
214 << ".")
215 }
216 }
217
218 additionalData_["baseCPI[" + legID + "]"] = baseCPI;
219 } else if (auto cpicf = boost::dynamic_pointer_cast<QuantLib::CPICashFlow>(flow)) {
220 Real baseCPI;
221 baseCPI = cpicf->baseFixing();
222 if (baseCPI == Null<Real>()) {
223 try {
224 baseCPI = QuantLib::CPI::laggedFixing(cpicf->cpiIndex(),
225 cpicf->baseDate() + cpicf->observationLag(),
226 cpicf->observationLag(), cpicf->interpolation());
227 } catch (std::exception& e) {
228 ALOG("CPICashFlow baseCPI could not be interpolated for additional results for trade " << id()
229 << ".")
230 }
231 }
232
233 additionalData_["baseCPI[" + legID + "]"] = baseCPI;
234 }
235 }
236 break;
237 }
238 }
239 if (legs_[i].size() > 0) {
240 boost::shared_ptr<Coupon> coupon = boost::dynamic_pointer_cast<Coupon>(legs_[i][0]);
241 if (coupon) {
242 Real originalNotional = 0.0;
243 try {
244 originalNotional = coupon->nominal();
245 } catch (std::exception& e) {
246 ALOG("original nominal could not be determined for trade " << id() << ", set to zero: " << e.what());
247 }
248 additionalData_["originalNotional[" + legID + "]"] = originalNotional;
249 }
250 }
251 for (Size j = 0; j < legs_[i].size(); ++j) {
252 boost::shared_ptr<CashFlow> flow = legs_[i][j];
253 if (flow->date() > asof) {
254 Size k = 0;
255 for (auto const& [fixingDate, index, multiplier] :
256 QuantExt::getIndexedCouponOrCashFlowFixingDetails(flow)) {
257 auto label = "[" + legID + "][" + std::to_string(j) + "][" + std::to_string(k) + "]";
258 additionalData_["indexingFixingDate" + label] = fixingDate;
259 additionalData_["indexingIndex" + label] =
260 index == nullptr ? "na" : IndexNameTranslator::instance().oreName(index->name());
261 additionalData_["indexingMultiplier" + label] = multiplier;
262 }
263 }
264 }
265}
