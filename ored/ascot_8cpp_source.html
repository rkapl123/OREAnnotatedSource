38void Ascot::build(const boost::shared_ptr<ore::data::EngineFactory>& engineFactory) {
39 DLOG("Ascot::build() called for trade " << id());
40
41 // build underlying convertible bond
42 bond_.reset();
43 // we need to do set the id manually because it otherwise remains blank
44 bond_.id() = id() + "_Bond";
45 bond_.build(engineFactory);
46 requiredFixings_.addData(bond_.requiredFixings());
47 boost::shared_ptr<QuantExt::ConvertibleBond2> cb =
48 boost::dynamic_pointer_cast<QuantExt::ConvertibleBond2>(bond_.instrument()->qlInstrument());
49
50 // check option fields
51 Exercise::Type exerciseType = parseExerciseType(optionData_.style());
52 QL_REQUIRE(exerciseType == Exercise::American, "expected American exercise type");
53 QL_REQUIRE(optionData_.exerciseDates().size() == 1,
54 "Ascot::build(): exactly one option date required, found " << optionData_.exerciseDates().size());
55 Date exerciseDate = parseDate(optionData_.exerciseDates().back());
56 boost::shared_ptr<Exercise> exercise = boost::make_shared<AmericanExercise>(exerciseDate);
57 Option::Type type = parseOptionType(optionData_.callPut());
58
59 // build funding leg
60 // Payer should be false,
61 // i.e. the swap is entered from the viewpoint of the asset swap buyer
62 QL_REQUIRE(fundingLegData_.isPayer() == false, "expected isPayer == false for funding leg");
63 Leg fundingLeg;
64
65 auto builder = boost::dynamic_pointer_cast<AscotEngineBuilder>(engineFactory->builder("Ascot"));
66 auto configuration = builder->configuration(MarketContext::pricing);
67 auto legBuilder = engineFactory->legBuilder(fundingLegData_.legType());
68 fundingLeg = legBuilder->buildLeg(fundingLegData_, engineFactory, requiredFixings_, configuration);
69
70 QL_REQUIRE(builder, "Ascot::build(): could not cast to AscotBuilder, this is unexpected");
71
72 auto qlAscot =
73 boost::make_shared<QuantExt::Ascot>(type, exercise, bond_.data().bondData().bondNotional(), cb, fundingLeg);
74 qlAscot->setPricingEngine(builder->engine(id(), bond_.data().bondData().currency()));
75
76 Real multiplier = (parsePositionType(optionData_.longShort()) == Position::Long ? 1.0 : -1.0);
77 instrument_ =
78 boost::shared_ptr<ore::data::InstrumentWrapper>(new ore::data::VanillaInstrument(qlAscot, multiplier));
79
80 npvCurrency_ = notionalCurrency_ = bond_.notionalCurrency();
81 legs_ = {cb->cashflows()};
82 legCurrencies_ = {npvCurrency_};
83 legPayers_ = {parsePositionType(optionData_.longShort()) == Position::Long};
84
85 notional_ = bond_.data().bondData().bondNotional();
86 maturity_ = bond_.maturity();
87
88 // ISDA taxonomy
89 additionalData_["isdaAssetClass"] = string("Credit");
90 additionalData_["isdaBaseProduct"] = string("Exotic");
91 additionalData_["isdaSubProduct"] = string("Other");
92 additionalData_["isdaTransaction"] = string("");
93}

95void Ascot::fromXML(XMLNode* node) {
96 Trade::fromXML(node);
97 XMLNode* dataNode = XMLUtils::getChildNode(node, "AscotData");
98 QL_REQUIRE(dataNode, "AscotData node not found");
99
100 ConvertibleBondData bondData;
101 bondData.fromXML(XMLUtils::getChildNode(dataNode, "ConvertibleBondData"));
102 bond_ = ConvertibleBond(envelope(), bondData);
103
104 optionData_.fromXML(XMLUtils::getChildNode(dataNode, "OptionData"));
105
106 XMLNode* tmpNode = XMLUtils::getChildNode(dataNode, "ReferenceSwapData");
107 QL_REQUIRE(tmpNode, "ReferenceSwapData node not found");
108 XMLNode* legNode = XMLUtils::getChildNode(tmpNode, "LegData");
109 QL_REQUIRE(legNode, "LegData node not found");
110 fundingLegData_.fromXML(legNode);
111}

113XMLNode* Ascot::toXML(XMLDocument& doc) {
114 XMLNode* node = Trade::toXML(doc);
115 XMLNode* dataNode = doc.allocNode("AscotData");
116 XMLUtils::appendNode(node, dataNode);
117
118 ConvertibleBondData d = bond_.data();
119 XMLUtils::appendNode(dataNode, d.toXML(doc));
120 XMLUtils::appendNode(dataNode, optionData_.toXML(doc));
121
122 XMLNode* fundingDataNode = doc.allocNode("ReferenceSwapData");
123 XMLUtils::appendNode(dataNode, fundingDataNode);
124 XMLUtils::appendNode(fundingDataNode, fundingLegData_.toXML(doc));
125
126 return node;
127}
