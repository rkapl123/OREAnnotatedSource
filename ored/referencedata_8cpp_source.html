295 <IndexWeightsAtLastRebalancingDate>
296 <Underlying>
297 <Name>Apple</Name>

302 </CurrencyHedgedEquityIndexReferenceDatum>
303</ReferenceDatum>

306void CurrencyHedgedEquityIndexReferenceDatum::fromXML(XMLNode* node) {
307 ReferenceDatum::fromXML(node);
308 XMLNode* innerNode = XMLUtils::getChildNode(node, type() + "ReferenceData");
309 QL_REQUIRE(innerNode, "No " + type() + "ReferenceData node");
310
311 underlyingIndexName_ = XMLUtils::getChildValue(innerNode, "UnderlyingIndex", true);
312
313 auto rebalancingStr = XMLUtils::getChildValue(innerNode, "RebalancingStrategy", false, "EndOfMonth");
314 if (rebalancingStr == "EndOfMonth") {
315 rebalancingStrategy_ = CurrencyHedgedEquityIndexReferenceDatum::RebalancingDate::EndOfMonth;
316 } else {
317 QL_FAIL("unexpected rebalancing strategy " << rebalancingStr);
318 }
319
320 std::string hedgeCalendarStr = XMLUtils::getChildValue(innerNode, "HedgeCalendar", true);
321 hedgeCalendar_ = parseCalendar(hedgeCalendarStr);
322
323 XMLNode* fxIndexesNode = XMLUtils::getChildNode(innerNode, "FxIndexes");
324 if (fxIndexesNode) {
325 for (XMLNode* child = XMLUtils::getChildNode(fxIndexesNode, "FxIndex"); child;
326 child = XMLUtils::getNextSibling(child, "FxIndex")) {
327 string currency = XMLUtils::getChildValue(child, "Currency", true);
328 string indexFamily = XMLUtils::getChildValue(child, "IndexName", true);
329 fxIndexes_[currency] = indexFamily;
330 }
331 }
332 // Optional Fields
333 referenceDateOffset_ = XMLUtils::getChildValueAsInt(innerNode, "ReferenceDateOffset", false, 0);
334 auto hedgeAdjStr = XMLUtils::getChildValue(innerNode, "HedgeAdjustment", false, "None");
335
336 if (hedgeAdjStr == "None") {
337 hedgeAdjustmentRule_ = HedgeAdjustment::None;
338 } else if (hedgeAdjStr == "Daily") {
339 hedgeAdjustmentRule_ = HedgeAdjustment::Daily;
340 } else {
341 QL_FAIL("unexpected rebalancing strategy " << hedgeAdjStr);
342 }
343 // clear map
344 data_.clear();
345
346 // and populate it...
347 XMLNode* currencyWeightNode = XMLUtils::getChildNode(innerNode, "IndexWeightsAtLastRebalancingDate");
348 if (currencyWeightNode) {
349 double totalWeight = 0.0;
350 for (XMLNode* child = XMLUtils::getChildNode(currencyWeightNode, "Underlying"); child;
351 child = XMLUtils::getNextSibling(child, "Underlying")) {
352 string name = XMLUtils::getChildValue(child, "Name", true);
353 double weight = XMLUtils::getChildValueAsDouble(child, "Weight", true);
354 QL_REQUIRE(weight > 0 || QuantLib::close_enough(weight, 0.0),
355 "Try to add negtive weight for Underlying" << name);
356 data_.push_back(make_pair(name, weight));
357 totalWeight += weight;
358 }
359 QL_REQUIRE(data_.empty() || QuantLib::close_enough(totalWeight, 1.0),
360 "Sum of underlying weights at last rebalancing date (" << totalWeight << ") is not 1.0");
361 }
362}

507boost::shared_ptr<ReferenceDatum> BasicReferenceDataManager::addFromXMLNode(XMLNode* node, const std::string& inputId,
508 const QuantLib::Date& inputValidFrom) {
509 string refDataType = XMLUtils::getChildValue(node, "Type", false);
510 boost::shared_ptr<ReferenceDatum> refData;
511
512 if (refDataType.empty()) {
513 ALOG("Found referenceDatum without Type - skipping");
514 return refData;
515 }
516
517 string id = inputId.empty() ? XMLUtils::getAttribute(node, "id") : inputId;
518
519 string validFromStr = XMLUtils::getAttribute(node, "validFrom");
520 QuantLib::Date validFrom;
521 if (validFromStr.empty()) {
522 validFrom = QuantLib::Date::minDate();
523 } else {
524 validFrom = ore::data::parseDate(validFromStr);
525 }
526
527 validFrom = inputValidFrom == QuantLib::Null<QuantLib::Date>() ? validFrom : inputValidFrom;
528
529 if (id.empty()) {
530 ALOG("Found referenceDatum without id - skipping");
531 return refData;
532 }
533
534 if (auto it = data_.find(make_pair(refDataType, id)); it != data_.end()) {
535 if (it->second.count(validFrom) > 0) {
536 duplicates_.insert(make_tuple(refDataType, id, validFrom));
537 ALOG("Found duplicate referenceDatum for type='" << refDataType << "', id='" << id << "', validFrom='"
538 << validFrom << "'");
539 return refData;
540 }
541 }
542
543 try {
544 refData = buildReferenceDatum(refDataType);
545 refData->fromXML(node);
546 // set the type and id at top level (is this needed?)
547 refData->setType(refDataType);
548 refData->setId(id);
549 refData->setValidFrom(validFrom);
550 data_[make_pair(refDataType, id)][validFrom] = refData;
551 TLOG("added referenceDatum for type='" << refDataType << "', id='" << id << "', validFrom='" << validFrom
552 << "'");
553 } catch (const std::exception& e) {
554 buildErrors_[make_pair(refDataType, id)][validFrom] = e.what();
555 ALOG("Error building referenceDatum for type='" << refDataType << "', id='" << id << "', validFrom='" << validFrom
556 << "': " << e.what());
557 }
558
559 return refData;
560}

562boost::shared_ptr<ReferenceDatum> BasicReferenceDataManager::buildReferenceDatum(const string& refDataType) {
563 auto refData = ReferenceDatumFactory::instance().build(refDataType);
564 QL_REQUIRE(refData,
565 "Reference data type " << refDataType << " has not been registered with the reference data factory.");

579std::tuple<QuantLib::Date, boost::shared_ptr<ReferenceDatum>> BasicReferenceDataManager::latestValidFrom(const string& type, const string& id,
580 const QuantLib::Date& asof) const {
581 auto it = data_.find(make_pair(type, id));
582 if (it != data_.end() && !it->second.empty()){
583 auto uB = it->second.upper_bound(asof);
584 if (uB != it->second.begin()) {

586 }
587 }
588 return {QuantLib::Date(), nullptr};

591void BasicReferenceDataManager::check(const string& type, const string& id, const QuantLib::Date& validFrom) const {
592 auto key = make_tuple(type, id, validFrom);
593 if (duplicates_.find(key) != duplicates_.end())
594 ALOG("BasicReferenceDataManager: duplicate entries for type='" << type << "', id='" << id << "', validFrom='"
595 << validFrom << "'");
596 auto err = buildErrors_.find(make_pair(type, id));
597 if (err != buildErrors_.end()) {
598 for (const auto& [validFrom, error] : err->second) {
599 ALOG("BasicReferenceDataManager: Build error for type='" << type << "', id='" << id << "', validFrom='"
600 << validFrom << "': " << error);
601 }
602 }

609 asofDate = Settings::instance().evaluationDate();
