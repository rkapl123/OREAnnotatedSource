253boost::shared_ptr<ReferenceDatum> BasicReferenceDataManager::addFromXMLNode(XMLNode* node, const std::string& inputId) {
254 string refDataType = XMLUtils::getChildValue(node, "Type", false);
255 boost::shared_ptr<ReferenceDatum> refData;
256
257 if (refDataType.empty()) {
258 ALOG("Found referenceDatum without Type - skipping");
259 return refData;
260 }
261
262 string id = inputId.empty() ? XMLUtils::getAttribute(node, "id") : inputId;
263
264 if (id.empty()) {
265 ALOG("Found referenceDatum without id - skipping");
266 return refData;
267 }
268
269 if (data_.find(make_pair(refDataType, id)) != data_.end()) {
270 duplicates_.insert(make_pair(refDataType, id));
271 ALOG("Found duplicate referenceDatum for type='" << refDataType << "', id='" << id << "'");
272 return refData;
273 }
274
275 try {
276 refData = buildReferenceDatum(refDataType);
277 refData->fromXML(node);
278 // set the type and id at top level (is this needed?)
279 refData->setType(refDataType);
280 refData->setId(id);
281 data_[make_pair(refDataType, id)] = refData;
282 TLOG("added referenceDatum for type='" << refDataType << "', id='" << id << "'");
283 } catch (const std::exception& e) {
284 buildErrors_[make_pair(refDataType, id)] = e.what();
285 ALOG("Error building referenceDatum for type='" << refDataType << "', id='" << id << "': " << e.what());
286 }
287
288 return refData;
289}

306void BasicReferenceDataManager::check(const string& type, const string& id) const {
307 auto key = make_pair(type, id);
308 if (duplicates_.find(key) != duplicates_.end())
309 ALOG("BasicReferenceDataManager: duplicate entries for type='" << type << "', id='" << id << "'");
310 auto err = buildErrors_.find(key);
311 if (err != buildErrors_.end())
312 ALOG("BasicReferenceDataManager: Build error for type='" << type << "', id='" << id << "': " << err->second);
313}

315bool BasicReferenceDataManager::hasData(const string& type, const string& id) const {

320boost::shared_ptr<ReferenceDatum> BasicReferenceDataManager::getData(const string& type, const string& id) {
