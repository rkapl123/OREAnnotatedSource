46 void FxKIKOBarrierOption::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
47 
48  Date today = Settings::instance().evaluationDate();
49  const boost::shared_ptr<Market> market = engineFactory->market();
50  Date start = ore::data::parseDate(startDate_);
51  Calendar cal = ore::data::parseCalendar(calendar_);
52 
53  // Only European Barrier supported for now
54  QL_REQUIRE(option_.style() == "European", "Option Style unknown: " << option_.style());
55  QL_REQUIRE(option_.exerciseDates().size() == 1, "Invalid number of excercise dates");
56  QL_REQUIRE(barriers_.size() == 2, "Invalid number of barriers");
57  QL_REQUIRE(barriers_[0].levels().size() == 1, "Invalid number of barrier levels");
58  QL_REQUIRE(barriers_[1].levels().size() == 1, "Invalid number of barrier levels");
59  QL_REQUIRE(barriers_[0].rebate() == 0, "rebates are not supported for KIKO options");
60  QL_REQUIRE(barriers_[1].rebate() == 0, "rebates are not supported for KIKO options");
61  QL_REQUIRE(barriers_[0].style().empty() || barriers_[0].style() == "American",
62  "only american barrier style supported");
63  QL_REQUIRE(barriers_[1].style().empty() || barriers_[1].style() == "American",
64  "only american barrier style supported");
65  QL_REQUIRE(tradeActions().empty(), "TradeActions not supported for FxBarrierOption");
66 
67  Currency boughtCcy = parseCurrency(boughtCurrency_);
68  Currency soldCcy = parseCurrency(soldCurrency_);
69 
70  // Payoff
71  Real strike = soldAmount_ / boughtAmount_;
72  Option::Type type = parseOptionType(option_.callPut());
73  boost::shared_ptr<StrikedTypePayoff> payoff(new PlainVanillaPayoff(type, strike));
74 
75  // Exercise
76  Date expiryDate = parseDate(option_.exerciseDates().front());
77  boost::shared_ptr<Exercise> exercise = boost::make_shared<EuropeanExercise>(expiryDate);
78 
79  // build underlying vanilla option
80  boost::shared_ptr<Instrument> vanilla = boost::make_shared<VanillaOption>(payoff, exercise);
81 
82  // we extract our KnockIn/KnockOut barrier data
83  Size knockInIndex;
84  Size knockOutIndex;
85 
86  Barrier::Type knockInType;
87  Barrier::Type knockOutType;
88 
89  Barrier::Type tmpBarrier = parseBarrierType(barriers_[0].type());
90  switch (tmpBarrier) {
91  case Barrier::Type::UpIn:
92  case Barrier::Type::DownIn:
93  knockInIndex = 0;
94  knockOutIndex = 1;
95  knockInType = tmpBarrier;
96  knockOutType = parseBarrierType(barriers_[1].type());
97  break;
98  case Barrier::Type::UpOut:
99  case Barrier::Type::DownOut:
100  knockInIndex = 1;
101  knockOutIndex = 0;
102  knockOutType = tmpBarrier;
103  knockInType = parseBarrierType(barriers_[1].type());
104  break;
105  default:
106  QL_FAIL("unsupported barrier type provided");
107  }
108  QL_REQUIRE(knockOutType == Barrier::Type::UpOut || knockOutType == Barrier::Type::DownOut,
109  "KIKO barrier requires one KnockOut barrier");
110  QL_REQUIRE(knockInType == Barrier::Type::UpIn || knockInType == Barrier::Type::DownIn,
111  "KIKO barrier requires one KnockIn barrier");
112 
113  Real knockInLevel = barriers_[knockInIndex].levels()[0].value();
114  Real knockOutLevel = barriers_[knockOutIndex].levels()[0].value();
115 
116  QL_REQUIRE(knockInLevel != knockOutLevel, "different levels must be provided");
117  // Check if the barrier has been triggered already
118  bool knockedIn = false;
119  bool knockedOut = false;
120  Handle<Quote> spot = market->fxRate(boughtCurrency_ + soldCurrency_);
121 
122  boost::shared_ptr<QuantExt::FxIndex> fxIndex;
123  if (!fxIndex_.empty())
124  fxIndex = parseFxIndex(fxIndex_);
125 
126  // checking fixings
127  if (startDate_ != "" && parseDate(startDate()) < today) {
128 
129  QL_REQUIRE(fxIndex_ != "", "no fxIndex provided");
130  QL_REQUIRE(calendar_ != "", "no calendar provided");
131  bool inverted = false;
132  if (fxIndex->sourceCurrency() == soldCcy && fxIndex->targetCurrency() == boughtCcy) {
133  inverted = true;
134  } else {
135  QL_REQUIRE(fxIndex->sourceCurrency() == boughtCcy && fxIndex->targetCurrency() == soldCcy,
136  "Invalid FX Index " << fxIndex_ << "for bought " << boughtCcy << "and sold " << soldCcy);
137  }
138 
139  Date d = start;
140  while (d < today && !knockedIn && !knockedOut) {
141  Real fixing = fxIndex->pastFixing(d);
142 
143  if (fixing == 0.0 || fixing == Null<Real>()) {
144  ALOG("Got invalid FX fixing for index " << fxIndex_ << " on " << d
145  << "Skipping this date, assuming no trigger");
146  } else {
147  if (inverted)
148  fixing = 1.0 / fixing;
149  ALOG("Checking FX fixing for index " << fxIndex_ << " on " << d << ", value " << fixing);
150 
151  if (!knockedIn)
152  knockedIn = checkBarrier(fixing, knockInType, knockInLevel);
153  if (!knockedOut)
154  knockedOut = checkBarrier(fixing, knockOutType, knockOutLevel);
155  }
156  d = cal.advance(d, 1, Days);
157  }
158  }
159 
160  // All possible instruments require an underlying vanilla option so we set this up
161  boost::shared_ptr<EngineBuilder> builder = engineFactory->builder("FxOption");
162  QL_REQUIRE(builder, "No FxOption builder found");
163  boost::shared_ptr<FxEuropeanOptionEngineBuilder> fxOptBuilder =
164  boost::dynamic_pointer_cast<FxEuropeanOptionEngineBuilder>(builder);
165  vanilla->setPricingEngine(fxOptBuilder->engine(boughtCcy, soldCcy, expiryDate));
166 
167  // Add additional premium payments
168  Position::Type positionType = parsePositionType(option_.longShort());
169  Real bsInd = (positionType == QuantLib::Position::Long ? 1.0 : -1.0);
170 
171  std::vector<boost::shared_ptr<Instrument>> additionalInstruments;
172  std::vector<Real> additionalMultipliers;
173  Date lastPremiumDate =
174  addPremiums(additionalInstruments, additionalMultipliers, boughtAmount_, option_.premiumData(), -bsInd, soldCcy,
175  engineFactory, fxOptBuilder->configuration(MarketContext::pricing));
176 
177  // we build a knock out option
178  boost::shared_ptr<Instrument> barrier =
179  boost::make_shared<QuantLib::BarrierOption>(knockOutType, knockOutLevel, 0, payoff, exercise);
180 
181  builder = engineFactory->builder("FxBarrierOption");
182  QL_REQUIRE(builder, "No FxBarrierOption builder found");
183  boost::shared_ptr<FxBarrierOptionEngineBuilder> fxBarrierOptBuilder =
184  boost::dynamic_pointer_cast<FxBarrierOptionEngineBuilder>(builder);
185  barrier->setPricingEngine(fxBarrierOptBuilder->engine(boughtCcy, soldCcy, expiryDate));
186  Settlement::Type settleType = parseSettlementType(option_.settlement());
187 
188  boost::shared_ptr<InstrumentWrapper> koInstrument =
189  boost::shared_ptr<InstrumentWrapper>(new SingleBarrierOptionWrapper(
190  barrier, positionType == Position::Long ? true : false, expiryDate,
191  settleType == Settlement::Physical ? true : false, vanilla, knockOutType, spot, knockOutLevel, 0, soldCcy,
192  start, fxIndex, cal, boughtAmount_, boughtAmount_, additionalInstruments, additionalMultipliers));
193 
194  // if the trade is knocked in this is all we price
195  if (knockedIn || knockedOut) {
196  DLOG("This trade has been knocked-in, building a knock out option");
197  instrument_ = koInstrument;
198  // Otherwise we build a composite instrument
199  } else {
200 
201  Handle<Quote> fx = Handle<Quote>(boost::make_shared<SimpleQuote>(1.0));
202  vector<boost::shared_ptr<InstrumentWrapper>> iw;
203  std::vector<Handle<Quote>> fxRates;
204 
205  DLOG("adding a knock out option to our composite trade");
206  iw.push_back(koInstrument);
207  fxRates.push_back(fx);
208 
209  // the second trade's additional instruments should have the opposite multipliers
210  std::vector<Real> flippedAdditionalMultipliers;
211  for (auto a : additionalMultipliers)
212  flippedAdditionalMultipliers.push_back(-1 * a);
213 
214  // For an UpIn/UpOut or a DownIn/DownOut the kiko option is replicated as follows:
215  // V_kiko(L,U) = V_knockout(L) - V_knockout(U)
216  // where L is the knockout level, U is the knockIn level
217  // If the option touches U the trade is knocked in, if it then touches L it is knocked out.
218  if ((knockOutType == Barrier::Type::UpOut && knockInType == Barrier::Type::UpIn) ||
219  (knockOutType == Barrier::Type::DownOut && knockInType == Barrier::Type::DownIn)) {
220  DLOG("Barrier Types are UpIn/UpOut or DownIn/DownOut, we add a single Barrier Knock Out Option to our composite trade");
221 
222  boost::shared_ptr<Instrument> barrier2 =
223  boost::make_shared<QuantLib::BarrierOption>(knockOutType, knockInLevel, 0, payoff, exercise);
224  barrier2->setPricingEngine(fxBarrierOptBuilder->engine(boughtCcy, soldCcy, expiryDate));
225  boost::shared_ptr<InstrumentWrapper> koInstrument2 =
226  boost::shared_ptr<InstrumentWrapper>(new SingleBarrierOptionWrapper(
227  barrier2, positionType == Position::Long ? false : true, expiryDate,
228  settleType == Settlement::Physical ? true : false, vanilla, knockOutType, spot, knockInLevel, 0,
229  soldCcy, start, fxIndex, cal, boughtAmount_, boughtAmount_, additionalInstruments, flippedAdditionalMultipliers));
230 
231  iw.push_back(koInstrument2);
232  fxRates.push_back(fx);
233  // For all other cases the KIKO is replicated as follows:
234  // V_kiko(L,U) = V_knockout(L) - V_doubleknockout(L,U)
235  // where L is the knockout level, U is the knockIn level
236  // the Option is only exercised if L is never touched, and U has been touched
237  } else {
238  DLOG("We add a Double Barrier Knock Out Option to our composite trade");
239  builder = engineFactory->builder("FxDoubleBarrierOption");
240  QL_REQUIRE(builder, "No FxDoubleBarrierOption builder found");
241  boost::shared_ptr<FxDoubleBarrierOptionEngineBuilder> fxDoubleBarrierOptBuilder =
242  boost::dynamic_pointer_cast<FxDoubleBarrierOptionEngineBuilder>(builder);
243 
244  boost::shared_ptr<Instrument> doubleBarrier = boost::make_shared<DoubleBarrierOption>(
245  DoubleBarrier::Type::KnockOut, std::min(knockInLevel, knockOutLevel),
246  std::max(knockInLevel, knockOutLevel), 0, payoff, exercise);
247  doubleBarrier->setPricingEngine(fxDoubleBarrierOptBuilder->engine(boughtCcy, soldCcy));
248 
249  boost::shared_ptr<InstrumentWrapper> dkoInstrument =
250  boost::shared_ptr<InstrumentWrapper>(new DoubleBarrierOptionWrapper(
251  doubleBarrier, positionType == Position::Long ? false : true, expiryDate,
252  settleType == Settlement::Physical ? true : false, vanilla, DoubleBarrier::Type::KnockOut, spot,
253  std::min(knockInLevel, knockOutLevel), std::max(knockInLevel, knockOutLevel), 0, soldCcy,
254  start, fxIndex, cal, boughtAmount_, boughtAmount_, additionalInstruments, flippedAdditionalMultipliers));
255 
256  iw.push_back(dkoInstrument);
257  fxRates.push_back(fx);
258  }
259  instrument_ = boost::shared_ptr<InstrumentWrapper>(new CompositeInstrumentWrapper(iw, fxRates, today));
260  }
261 
262  npvCurrency_ = soldCurrency_; // sold is the domestic
263  notional_ = soldAmount_;
264  notionalCurrency_ = soldCurrency_;
265  maturity_ = std::max(expiryDate, lastPremiumDate);
266 
267  if (start != Date()) {
268  for (Date d = start; d <= expiryDate; d = cal.advance(d, 1 * Days))
269  requiredFixings_.addFixingDate(d, fxIndex_, expiryDate);
270  }
271 
272  additionalData_["boughAmount"] = boughtAmount_;
273  additionalData_["boughtCurrency"] = boughtCurrency_;
274  additionalData_["soldAmount"] = soldAmount_;
275  additionalData_["soldCurrency"] = soldCurrency_;
276 }

278 bool FxKIKOBarrierOption::checkBarrier(Real spot, Barrier::Type type, Real barrier) {
279  switch (type) {
280  case Barrier::DownIn:
281  case Barrier::DownOut:
282  return spot <= barrier;
283  case Barrier::UpIn:
284  case Barrier::UpOut:
285  return spot >= barrier;
286  default:
287  QL_FAIL("unknown barrier type " << type);
288  }
289 }

291 void FxKIKOBarrierOption::fromXML(XMLNode* node) {
292  Trade::fromXML(node);
293  XMLNode* fxNode = XMLUtils::getChildNode(node, "FxKIKOBarrierOptionData");
294  QL_REQUIRE(fxNode, "No FxKIKOBarrierOptionData Node");
295  option_.fromXML(XMLUtils::getChildNode(fxNode, "OptionData"));
296  XMLNode* barrierNode = XMLUtils::getChildNode(fxNode, "Barriers");
297  QL_REQUIRE(barrierNode, "No Barriers node");
298  auto barriers = XMLUtils::getChildrenNodes(barrierNode, "BarrierData");
299  for (auto const& b : barriers) {
300  barriers_.push_back(BarrierData());
301  barriers_.back().fromXML(b);
302  }
303  QL_REQUIRE(barriers_.size() == 2, "A KIKO barrier requires two BarrierData nodes");
304 
305  startDate_ = XMLUtils::getChildValue(fxNode, "StartDate", false);
306  calendar_ = XMLUtils::getChildValue(fxNode, "Calendar", false);
307  fxIndex_ = XMLUtils::getChildValue(fxNode, "FXIndex", false);
308  boughtCurrency_ = XMLUtils::getChildValue(fxNode, "BoughtCurrency", true);
309  soldCurrency_ = XMLUtils::getChildValue(fxNode, "SoldCurrency", true);
310  boughtAmount_ = XMLUtils::getChildValueAsDouble(fxNode, "BoughtAmount", true);
311  soldAmount_ = XMLUtils::getChildValueAsDouble(fxNode, "SoldAmount", true);
312 }

314 XMLNode* FxKIKOBarrierOption::toXML(XMLDocument& doc) {
315  XMLNode* node = Trade::toXML(doc);
316  XMLNode* fxNode = doc.allocNode("FxKIKOBarrierOptionData");
317  XMLUtils::appendNode(node, fxNode);
318 
319  XMLUtils::appendNode(fxNode, option_.toXML(doc));
320  XMLNode* barrierNode = doc.allocNode("Barriers");
321  for (auto& b : barriers_) {
322  XMLUtils::appendNode(barrierNode, b.toXML(doc));
323  }
324  XMLUtils::appendNode(fxNode, barrierNode);
325 
326  XMLUtils::addChild(doc, fxNode, "StartDate", startDate_);
327  XMLUtils::addChild(doc, fxNode, "Calendar", calendar_);
328  XMLUtils::addChild(doc, fxNode, "FXIndex", fxIndex_);
329  XMLUtils::addChild(doc, fxNode, "BoughtCurrency", boughtCurrency_);
330  XMLUtils::addChild(doc, fxNode, "BoughtAmount", boughtAmount_);
331  XMLUtils::addChild(doc, fxNode, "SoldCurrency", soldCurrency_);
332  XMLUtils::addChild(doc, fxNode, "SoldAmount", soldAmount_);
333 
334  return node;
335 }
