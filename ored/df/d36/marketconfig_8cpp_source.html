39 void MarketConfiguration::fromXML(XMLNode* node) {
40 
41  clear();
42 
43  XMLUtils::checkNode(node, "MarketConfiguration");
44 
45  baseCurrency_ = XMLUtils::getChildValue(node, "BaseCurrency", true);
46  fxSpec_ = XMLUtils::getChildrenValues(node, "FxRates", "FxRate", "CurrencyPair", "Spec", true);
47  fxVolSpec_ = XMLUtils::getChildrenValues(node, "FxVolatilities", "FxVolatility", "CurrencyPair", "Spec", true);
48  capVolSpec_ = XMLUtils::getChildrenValues(node, "CapVolatilities", "CapVolatility", "Currency", "Spec", true);
49  swaptionVolSpec_ =
50  XMLUtils::getChildrenValues(node, "SwaptionVolatilities", "SwaptionVolatility", "Currency", "Spec", true);
51  iborIndexSpecs_ = XMLUtils::getChildrenValues(node, "IborIndices", "IborIndex", "Name", "Spec", true);
52 
53  // Get discount curve specs and attributes
54  XMLNode* discountNodes = XMLUtils::getChildNode(node, "DiscountCurves");
55  QL_REQUIRE(discountNodes, "Must provide a DiscountCurves node in market configuration");
56  for (XMLNode* child = XMLUtils::getChildNode(discountNodes); child; child = XMLUtils::getNextSibling(child)) {
57  // Get the child node information
58  string currency = XMLUtils::getChildValue(child, "Currency", true);
59  string spec = XMLUtils::getChildValue(child, "Spec", true);
60  discountCurveSpecs_.insert(make_pair(currency, spec));
61  }
62 
63  // Get yield curve specs and attributes
64  XMLNode* yieldNodes = XMLUtils::getChildNode(node, "YieldCurves");
65  for (XMLNode* child = XMLUtils::getChildNode(yieldNodes); child; child = XMLUtils::getNextSibling(child)) {
66  // Get the child node information
67  string name = XMLUtils::getChildValue(child, "Name", true);
68  string spec = XMLUtils::getChildValue(child, "Spec", true);
69  yieldCurveSpecs_.insert(make_pair(name, spec));
70  }
71 }

83 
84  // Add the DiscountCurves element
85  XMLNode* discountNodes = XMLUtils::addChild(doc, node, "DiscountCurves");
86  map<string, string>::const_iterator it;
87  for (it = discountCurveSpecs_.begin(); it != discountCurveSpecs_.end(); ++it) {
88  // Add the main node data
89  XMLNode* discountNode = XMLUtils::addChild(doc, discountNodes, "DiscountCurve");
90  XMLUtils::addChild(doc, discountNode, "Currency", it->first);

93
