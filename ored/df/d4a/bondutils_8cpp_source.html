62 Date getOpenEndDateReplacement(const std::string& replacementPeriodStr, const Calendar& calendar) {
63  if (replacementPeriodStr.empty())
64  return QuantLib::Null<QuantLib::Date>();
65  Date today = Settings::instance().evaluationDate();
66  Date result = Date::maxDate() - 365;
67  try {
68  // might throw because we are beyond the last allowed date
69  result = (calendar.empty() ? NullCalendar() : calendar).advance(today, parsePeriod(replacementPeriodStr));
70  } catch (...) {
71  }
72  DLOG("Compute open end date replacement as "
73  << QuantLib::io::iso_date(result) << " (today = " << QuantLib::io::iso_date(today)
74  << ", OpenEndDateReplacement from pricing engine config = " << replacementPeriodStr << ")");
75  return result;
76 }
