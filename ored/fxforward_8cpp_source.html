33void FxForward::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
34 // If you Buy EURUSD forward, then you buy EUR and sell USD.
35 // EUR = foreign, USD = Domestic.
36 // You pay in USD, so the Domestic / Sold ccy is the "payer" currency
37 Currency boughtCcy = parseCurrency(boughtCurrency_);
38 Currency soldCcy = parseCurrency(soldCurrency_);
39 Date maturityDate = parseDate(maturityDate_);
40
41 // Derive settlement date from payment data parameters
42 Date payDate;
43 if (payDate_.empty()) {
44 Natural conventionalLag = 0;
45 Calendar conventionalCalendar = NullCalendar();
46 BusinessDayConvention conventionalBdc = Unadjusted;
47 if (!fxIndex_.empty() && settlement_ == "Cash") {
48 std::tie(conventionalLag, conventionalCalendar, conventionalBdc) =
49 getFxIndexConventions(fxIndex_.empty() ? boughtCurrency_ + soldCurrency_ : fxIndex_);
50 }
51 PaymentLag paymentLag;
52 if (payLag_.empty())
53 paymentLag = conventionalLag;
54 else
55 paymentLag = parsePaymentLag(payLag_);
56 Period payLag = boost::apply_visitor(PaymentLagPeriod(), paymentLag);
57 Calendar payCalendar = payCalendar_.empty() ? conventionalCalendar : parseCalendar(payCalendar_);
58 BusinessDayConvention payConvention =
59 payConvention_.empty() ? conventionalBdc : parseBusinessDayConvention(payConvention_);
60 payDate = payCalendar.advance(maturityDate, payLag, payConvention);
61 } else {
62 payDate = parseDate(payDate_);
63 QL_REQUIRE(payDate >= maturityDate, "FX Forward settlement date should equal or exceed the maturity date.");
64 }
65
66 boost::shared_ptr<QuantExt::FxIndex> fxIndex;
67 Currency payCcy;
68
69 if (payCurrency_.empty()) {
70 // If settlement currency is not set, set it to the domestic currency.
71 //LOG("Settlement currency was not specified, defaulting to " << soldCcy.code());
72 payCcy = soldCcy;
73 } else {
74 payCcy = parseCurrency(payCurrency_);
75 QL_REQUIRE(payCcy == boughtCcy || payCcy == soldCcy,
76 "Settlement currency must be either " << boughtCcy.code() << " or " << soldCcy.code() << ".");
77 }
78
79 Date fixingDate;
80 bool fixingRequired = (settlement_ == "Cash") && (payDate > maturityDate);
81 if (fixingRequired) {
82 QL_REQUIRE(!fxIndex_.empty(), "FX settlement index must be specified for non-deliverable forwards");
83 Currency nonPayCcy = payCcy == boughtCcy ? soldCcy : boughtCcy;
84 fxIndex = buildFxIndex(fxIndex_, nonPayCcy.code(), payCcy.code(), engineFactory->market(),
85 engineFactory->configuration(MarketContext::pricing));
86 fixingDate = fxIndex->fixingCalendar().adjust(maturityDate);
87 requiredFixings_.addFixingDate(fixingDate, fxIndex_, payDate);
88 }
89
90 QL_REQUIRE(tradeActions().empty(), "TradeActions not supported for FxForward");
91
92 DLOG("Build FxForward with maturity date " << QuantLib::io::iso_date(maturityDate) << " and pay date "
93 << QuantLib::io::iso_date(payDate));
94
95 boost::shared_ptr<QuantLib::Instrument> instrument =
96 boost::make_shared<QuantExt::FxForward>(boughtAmount_, boughtCcy, soldAmount_, soldCcy, maturityDate, false,
97 settlement_ == "Physical", payDate, payCcy, fixingDate, fxIndex);
98 instrument_.reset(new VanillaInstrument(instrument));
99
100 npvCurrency_ = payCcy.code();
101 notional_ = Null<Real>(); // soldAmount_;
102 notionalCurrency_ = ""; // soldCurrency_;
103 maturity_ = std::max(payDate, maturityDate);
104
105 // Set up Legs
106 legs_ = {{boost::make_shared<SimpleCashFlow>(boughtAmount_, payDate)},
107 {boost::make_shared<SimpleCashFlow>(soldAmount_, payDate)}};
108 legCurrencies_ = {boughtCurrency_, soldCurrency_};
109 legPayers_ = {false, true};
110
111 // set Pricing engine
112 boost::shared_ptr<EngineBuilder> builder = engineFactory->builder(tradeType_);
113 QL_REQUIRE(builder, "No builder found for " << tradeType_);
114 boost::shared_ptr<FxForwardEngineBuilderBase> fxBuilder =
115 boost::dynamic_pointer_cast<FxForwardEngineBuilderBase>(builder);
116
117 instrument_->qlInstrument()->setPricingEngine(fxBuilder->engine(boughtCcy, soldCcy));
118
119 DLOG("FxForward leg 0: " << legs_[0][0]->date() << " " << legs_[0][0]->amount());
120 DLOG("FxForward leg 1: " << legs_[1][0]->date() << " " << legs_[1][0]->amount());
121
122 additionalData_["soldCurrency"] = soldCurrency_;
123 additionalData_["boughtCurrency"] = boughtCurrency_;
124 additionalData_["soldAmount"] = soldAmount_;
125 additionalData_["boughtAmount"] = boughtAmount_;
126}

128QuantLib::Real FxForward::notional() const {
129 // try to get the notional from the additional results of the instrument
130 try {
131 return instrument_->qlInstrument(true)->result<Real>("currentNotional");
132 } catch (const std::exception& e) {
133 if (strcmp(e.what(), "currentNotional not provided"))
134 ALOG("error when retrieving notional: " << e.what());
135 }
136 // if not provided, return null
137 return Null<Real>();
138}

152void FxForward::fromXML(XMLNode* node) {
153 Trade::fromXML(node);
154 XMLNode* fxNode = XMLUtils::getChildNode(node, "FxForwardData");
155 QL_REQUIRE(fxNode, "No FxForwardData Node");
156 maturityDate_ = XMLUtils::getChildValue(fxNode, "ValueDate", true);
157 boughtCurrency_ = XMLUtils::getChildValue(fxNode, "BoughtCurrency", true);
158 soldCurrency_ = XMLUtils::getChildValue(fxNode, "SoldCurrency", true);
159 boughtAmount_ = XMLUtils::getChildValueAsDouble(fxNode, "BoughtAmount", true);
160 soldAmount_ = XMLUtils::getChildValueAsDouble(fxNode, "SoldAmount", true);
161 settlement_ = XMLUtils::getChildValue(fxNode, "Settlement", false);
162 if (settlement_ == "")
163 settlement_ = "Physical";
164
165 if (XMLNode* settlementDataNode = XMLUtils::getChildNode(fxNode, "SettlementData")) {
166 payCurrency_ = XMLUtils::getChildValue(settlementDataNode, "Currency", false);
167 fxIndex_ = XMLUtils::getChildValue(settlementDataNode, "FXIndex", false);
168 payDate_ = XMLUtils::getChildValue(settlementDataNode, "Date", false);
169
170 if (payDate_.empty()) {
171 if (XMLNode* rulesNode = XMLUtils::getChildNode(settlementDataNode, "Rules")) {
172 payLag_ = XMLUtils::getChildValue(rulesNode, "PaymentLag", false);
173 payCalendar_ = XMLUtils::getChildValue(rulesNode, "PaymentCalendar", false);
174 payConvention_ = XMLUtils::getChildValue(rulesNode, "PaymentConvention", false);
175 }
176 }
177 }
178}

180XMLNode* FxForward::toXML(XMLDocument& doc) {
181 XMLNode* node = Trade::toXML(doc);
182 XMLNode* fxNode = doc.allocNode("FxForwardData");
183 XMLUtils::appendNode(node, fxNode);
184 XMLUtils::addChild(doc, fxNode, "ValueDate", maturityDate_);
185 XMLUtils::addChild(doc, fxNode, "BoughtCurrency", boughtCurrency_);
186 XMLUtils::addChild(doc, fxNode, "BoughtAmount", boughtAmount_);
187 XMLUtils::addChild(doc, fxNode, "SoldCurrency", soldCurrency_);
188 XMLUtils::addChild(doc, fxNode, "SoldAmount", soldAmount_);
189 XMLUtils::addChild(doc, fxNode, "Settlement", settlement_);
190
191 XMLNode* settlementDataNode = doc.allocNode("SettlementData");
192 XMLUtils::appendNode(fxNode, settlementDataNode);
193
194 if (!payCurrency_.empty())
195 XMLUtils::addChild(doc, settlementDataNode, "Currency", payCurrency_);
196 if (!fxIndex_.empty())
197 XMLUtils::addChild(doc, settlementDataNode, "FXIndex", fxIndex_);
198 if (!payDate_.empty()) {
199 XMLUtils::addChild(doc, settlementDataNode, "Date", payDate_);
200 } else {
201 XMLNode* rulesNode = doc.allocNode("Rules");
202 XMLUtils::appendNode(settlementDataNode, rulesNode);
203 if (!payLag_.empty())
204 XMLUtils::addChild(doc, rulesNode, "PaymentLag", payLag_);
205 if (!payCalendar_.empty())
206 XMLUtils::addChild(doc, rulesNode, "PaymentCalendar", payCalendar_);
207 if (!payConvention_.empty())
208 XMLUtils::addChild(doc, rulesNode, "PaymentConvention", payConvention_);
209 }
210
211 return node;
212}
