33void FxForward::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
34 // If you Buy EURUSD forward, then you buy EUR and sell USD.
35 // EUR = foreign, USD = Domestic.
36 // You pay in USD, so the Domestic / Sold ccy is the "payer" currency
37 Currency boughtCcy = parseCurrency(boughtCurrency_);
38 Currency soldCcy = parseCurrency(soldCurrency_);
39 Date maturityDate = parseDate(maturityDate_);
40
41 // Derive settlement date from payment data parameters
42 Date payDate;
43 if (payDate_.empty()) {
44 Natural conventionalLag = 0;
45 Calendar conventionalCalendar = NullCalendar();
46 BusinessDayConvention conventionalBdc = Unadjusted;
47 if (!fxIndex_.empty() && settlement_ == "Cash") {
48 std::tie(conventionalLag, conventionalCalendar, conventionalBdc) =
49 getFxIndexConventions(fxIndex_.empty() ? boughtCurrency_ + soldCurrency_ : fxIndex_);
50 }
51 PaymentLag paymentLag;
52 if (payLag_.empty())
53 paymentLag = conventionalLag;
54 else
55 paymentLag = parsePaymentLag(payLag_);
56 Period payLag = boost::apply_visitor(PaymentLagPeriod(), paymentLag);
57 Calendar payCalendar = payCalendar_.empty() ? conventionalCalendar : parseCalendar(payCalendar_);
58 BusinessDayConvention payConvention =
59 payConvention_.empty() ? conventionalBdc : parseBusinessDayConvention(payConvention_);
60 payDate = payCalendar.advance(maturityDate, payLag, payConvention);
61 } else {
62 payDate = parseDate(payDate_);
63 QL_REQUIRE(payDate >= maturityDate, "FX Forward settlement date should equal or exceed the maturity date.");
64 }
65
66 boost::shared_ptr<QuantExt::FxIndex> fxIndex;
67 Currency payCcy;
68
69 if (payCurrency_.empty()) {
70 // If settlement currency is not set, set it to the domestic currency.
71 payCcy = soldCcy;
72 } else {
73 payCcy = parseCurrency(payCurrency_);
74 QL_REQUIRE(payCcy == boughtCcy || payCcy == soldCcy,
75 "Settlement currency must be either " << boughtCcy.code() << " or " << soldCcy.code() << ".");
76 }
77
78 Date fixingDate;
79 if (settlement_ == "Cash") {
80 // We allow for an empty fxIndex if maturiytDate == payDate in order not to break trades that were previously
81 // pricing - we should really require fxIndex whenever cash settlement is specified. If the fxIndex is not
82 // given in this case, we assume that the current FX Spot rate is used to determine the settlement amount.
83 if (maturityDate <= payDate && !fxIndex_.empty()) {
84 Currency nonPayCcy = payCcy == boughtCcy ? soldCcy : boughtCcy;
85 fxIndex = buildFxIndex(fxIndex_, nonPayCcy.code(), payCcy.code(), engineFactory->market(),
86 engineFactory->configuration(MarketContext::pricing));
87 // We also allow for an effective fixing date > payDate in order not to break trades that were preivously
88 // pricing - this should be an error as well. If this is the case we assume that the current FX Spot
89 // rate is used to determine the settlement amount as above.
90 fixingDate = fxIndex->fixingCalendar().adjust(maturityDate);
91 if (fixingDate <= payDate) {
92 requiredFixings_.addFixingDate(fixingDate, fxIndex_, payDate);
93 }
94 } else {
95 QL_REQUIRE(maturityDate >= payDate,
96 "FX settlement index must be specified for non-deliverable forward if value date ("
97 << maturityDate << ") < payDate (" << payDate << ")");
98 }
99 }
100
101 QL_REQUIRE(tradeActions().empty(), "TradeActions not supported for FxForward");
102
103 DLOG("Build FxForward with maturity date " << QuantLib::io::iso_date(maturityDate) << " and pay date "
104 << QuantLib::io::iso_date(payDate));
105
106 boost::shared_ptr<QuantLib::Instrument> instrument =
107 boost::make_shared<QuantExt::FxForward>(boughtAmount_, boughtCcy, soldAmount_, soldCcy, maturityDate, false,
108 settlement_ == "Physical", payDate, payCcy, fixingDate, fxIndex);
109 instrument_.reset(new VanillaInstrument(instrument));
110
111 npvCurrency_ = payCcy.code();
112 notional_ = Null<Real>(); // soldAmount_;
113 notionalCurrency_ = ""; // soldCurrency_;
114 maturity_ = std::max(payDate, maturityDate);
115
116 // Set up Legs
117 legs_ = {{boost::make_shared<SimpleCashFlow>(boughtAmount_, payDate)},
118 {boost::make_shared<SimpleCashFlow>(soldAmount_, payDate)}};
119 legCurrencies_ = {boughtCurrency_, soldCurrency_};
120 legPayers_ = {false, true};
121
122 // set Pricing engine
123 boost::shared_ptr<EngineBuilder> builder = engineFactory->builder(tradeType_);
124 QL_REQUIRE(builder, "No builder found for " << tradeType_);
125 boost::shared_ptr<FxForwardEngineBuilderBase> fxBuilder =
126 boost::dynamic_pointer_cast<FxForwardEngineBuilderBase>(builder);
127
128 instrument_->qlInstrument()->setPricingEngine(fxBuilder->engine(boughtCcy, soldCcy));
129
130 additionalData_["soldCurrency"] = soldCurrency_;
131 additionalData_["boughtCurrency"] = boughtCurrency_;
132 additionalData_["soldAmount"] = soldAmount_;
133 additionalData_["boughtAmount"] = boughtAmount_;
134 additionalData_["valueDate"] = maturityDate_;
135 if (fixingDate != Date()) {
136 if (fixingDate <= payDate)
137 additionalData_["fixingDate"] = fixingDate;
138 else
139 additionalData_["adjustedValueDate"] = fixingDate;
140 }
141 additionalData_["payDate"] = payDate;
142 additionalData_["settlement"] = settlement_;
143
144 // ISDA taxonomy
145 additionalData_["isdaAssetClass"] = string("Foreign Exchange");
146 additionalData_["isdaBaseProduct"] = string(settlement_ == "Cash" ? "NDF" : "Forward");
147 additionalData_["isdaSubProduct"] = string("");
148 additionalData_["isdaTransaction"] = string("");
149}

151QuantLib::Real FxForward::notional() const {
152 // try to get the notional from the additional results of the instrument
153 try {
154 return instrument_->qlInstrument(true)->result<Real>("currentNotional");
155 } catch (const std::exception& e) {
156 if (strcmp(e.what(), "currentNotional not provided"))
157 ALOG("error when retrieving notional: " << e.what());
158 }
159 // if not provided, return null
160 return Null<Real>();
161}

163std::string FxForward::notionalCurrency() const {
164 // try to get the notional ccy from the additional results of the instrument
165 try {
166 return instrument_->qlInstrument(true)->result<std::string>("notionalCurrency");
167 } catch (const std::exception& e) {
168 if (strcmp(e.what(), "notionalCurrency not provided"))
169 ALOG("error when retrieving notional ccy: " << e.what());
170 }
171 // if not provided, return an empty string
172 return "";
173}

175void FxForward::fromXML(XMLNode* node) {
176 Trade::fromXML(node);
177 XMLNode* fxNode = XMLUtils::getChildNode(node, "FxForwardData");
178 QL_REQUIRE(fxNode, "No FxForwardData Node");
179 maturityDate_ = XMLUtils::getChildValue(fxNode, "ValueDate", true);
180 boughtCurrency_ = XMLUtils::getChildValue(fxNode, "BoughtCurrency", true);
181 soldCurrency_ = XMLUtils::getChildValue(fxNode, "SoldCurrency", true);
182 boughtAmount_ = XMLUtils::getChildValueAsDouble(fxNode, "BoughtAmount", true);
183 soldAmount_ = XMLUtils::getChildValueAsDouble(fxNode, "SoldAmount", true);
184 settlement_ = XMLUtils::getChildValue(fxNode, "Settlement", false);
185 if (settlement_ == "")
186 settlement_ = "Physical";
187
188 if (XMLNode* settlementDataNode = XMLUtils::getChildNode(fxNode, "SettlementData")) {
189 payCurrency_ = XMLUtils::getChildValue(settlementDataNode, "Currency", false);
190 fxIndex_ = XMLUtils::getChildValue(settlementDataNode, "FXIndex", false);
191 payDate_ = XMLUtils::getChildValue(settlementDataNode, "Date", false);
192
193 if (payDate_.empty()) {
194 if (XMLNode* rulesNode = XMLUtils::getChildNode(settlementDataNode, "Rules")) {
195 payLag_ = XMLUtils::getChildValue(rulesNode, "PaymentLag", false);
196 payCalendar_ = XMLUtils::getChildValue(rulesNode, "PaymentCalendar", false);
197 payConvention_ = XMLUtils::getChildValue(rulesNode, "PaymentConvention", false);
198 }
199 }
200 }
201}

203XMLNode* FxForward::toXML(XMLDocument& doc) {
204 XMLNode* node = Trade::toXML(doc);
205 XMLNode* fxNode = doc.allocNode("FxForwardData");
206 XMLUtils::appendNode(node, fxNode);
207 XMLUtils::addChild(doc, fxNode, "ValueDate", maturityDate_);
208 XMLUtils::addChild(doc, fxNode, "BoughtCurrency", boughtCurrency_);
209 XMLUtils::addChild(doc, fxNode, "BoughtAmount", boughtAmount_);
210 XMLUtils::addChild(doc, fxNode, "SoldCurrency", soldCurrency_);
211 XMLUtils::addChild(doc, fxNode, "SoldAmount", soldAmount_);
212 XMLUtils::addChild(doc, fxNode, "Settlement", settlement_);
213
214 XMLNode* settlementDataNode = doc.allocNode("SettlementData");
215 XMLUtils::appendNode(fxNode, settlementDataNode);
216
217 if (!payCurrency_.empty())
218 XMLUtils::addChild(doc, settlementDataNode, "Currency", payCurrency_);
219 if (!fxIndex_.empty())
220 XMLUtils::addChild(doc, settlementDataNode, "FXIndex", fxIndex_);
221 if (!payDate_.empty()) {
222 XMLUtils::addChild(doc, settlementDataNode, "Date", payDate_);
223 } else {
224 XMLNode* rulesNode = doc.allocNode("Rules");
225 XMLUtils::appendNode(settlementDataNode, rulesNode);
226 if (!payLag_.empty())
227 XMLUtils::addChild(doc, rulesNode, "PaymentLag", payLag_);
228 if (!payCalendar_.empty())
229 XMLUtils::addChild(doc, rulesNode, "PaymentCalendar", payCalendar_);
230 if (!payConvention_.empty())
231 XMLUtils::addChild(doc, rulesNode, "PaymentConvention", payConvention_);
232 }
233
234 return node;
235}
