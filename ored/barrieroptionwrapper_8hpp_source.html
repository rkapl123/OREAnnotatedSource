126  QL_REQUIRE(barrierType == DoubleBarrier::Type::KnockOut || barrierType == DoubleBarrier::Type::KnockIn,
127  "Invalid barrier type " << barrierType << ". Only KnockOut and KnockIn are supported.");
128  QL_REQUIRE(barrierLow < barrierHigh, "barrierLow has to be less than barrierHigh");
129  }
