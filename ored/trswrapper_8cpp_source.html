36TRSWrapper::TRSWrapper(
37 const std::vector<boost::shared_ptr<ore::data::Trade>>& underlying,
38 const std::vector<boost::shared_ptr<Index>>& underlyingIndex, const std::vector<Real> underlyingMultiplier,
39 const bool includeUnderlyingCashflowsInReturn, const Real initialPrice, const Currency& initialPriceCurrency,
40 const std::vector<Currency>& assetCurrency, const Currency& returnCurrency,
41 const std::vector<Date>& valuationSchedule, const std::vector<Date>& paymentSchedule,
42 const std::vector<Leg>& fundingLegs, const std::vector<TRS::FundingData::NotionalType>& fundingNotionalTypes,
43 const Currency& fundingCurrency, const Size fundingResetGracePeriod, const bool paysAsset, const bool paysFunding,
44 const Leg& additionalCashflowLeg, const bool additionalCashflowLegPayer, const Currency& additionalCashflowCurrency,
45 const std::vector<boost::shared_ptr<FxIndex>>& fxIndexAsset, const boost::shared_ptr<FxIndex>& fxIndexReturn,
46 const boost::shared_ptr<FxIndex>& fxIndexAdditionalCashflows,
47 const std::map<std::string, boost::shared_ptr<QuantExt::FxIndex>>& addFxIndices)
48 : underlying_(underlying), underlyingIndex_(underlyingIndex), underlyingMultiplier_(underlyingMultiplier),
49 includeUnderlyingCashflowsInReturn_(includeUnderlyingCashflowsInReturn), initialPrice_(initialPrice),
50 initialPriceCurrency_(initialPriceCurrency), assetCurrency_(assetCurrency), returnCurrency_(returnCurrency),
51 valuationSchedule_(valuationSchedule), paymentSchedule_(paymentSchedule), fundingLegs_(fundingLegs),
52 fundingNotionalTypes_(fundingNotionalTypes), fundingCurrency_(fundingCurrency),
53 fundingResetGracePeriod_(fundingResetGracePeriod), paysAsset_(paysAsset), paysFunding_(paysFunding),
54 additionalCashflowLeg_(additionalCashflowLeg), additionalCashflowLegPayer_(additionalCashflowLegPayer),
55 additionalCashflowCurrency_(additionalCashflowCurrency), fxIndexAsset_(fxIndexAsset),
56 fxIndexReturn_(fxIndexReturn), fxIndexAdditionalCashflows_(fxIndexAdditionalCashflows),
57 addFxIndices_(addFxIndices) {
58
59 QL_REQUIRE(!paymentSchedule_.empty(), "TRSWrapper::TRSWrapper(): payment schedule must not be empty()");
60
61 QL_REQUIRE(valuationSchedule_.size() == paymentSchedule_.size() + 1,
62 "TRSWrapper::TRSWrapper(): valuation schedule size ("
63 << valuationSchedule_.size() << ") must be payment schedule size (" << paymentSchedule_.size()
64 << ") plus one");
65
66 for (Size i = 0; i < valuationSchedule_.size() - 1; ++i) {
67 QL_REQUIRE(valuationSchedule_[i] < valuationSchedule_[i + 1],
68 "TRSWrapper::TRSWrapper(): valuation schedule dates must be monotonic, at "
69 << i << ": " << valuationSchedule_[i] << ", " << valuationSchedule_[i + 1]);
70 }
71
72 for (Size i = 0; i < paymentSchedule_.size() - 1; ++i) {
73 QL_REQUIRE(paymentSchedule_[i] < paymentSchedule_[i + 1],
74 "TRSWrapper::TRSWrapper(): payment schedule dates must be monotonic, at "
75 << i << ": " << paymentSchedule_[i] << ", " << paymentSchedule_[i + 1]);
76 }
77
78 for (Size i = 0; i < paymentSchedule_.size(); ++i) {
79 QL_REQUIRE(paymentSchedule_[i] >= valuationSchedule_[i + 1], "TRSWrapper::TRSWrapper(): payment date at "
80 << i << " (" << paymentSchedule_[i]
81 << ") must be >= valuation date ("
82 << valuationSchedule_[i + 1]);
83 }
84
85 QL_REQUIRE(fundingLegs_.size() == fundingNotionalTypes_.size(), "TRSWrapper::TRSWrapper(): number of funding legs ("
86 << fundingLegs_.size()
87 << ") must match funding notitional types ("
88 << fundingNotionalTypes_.size() << ")");
89
90 QL_REQUIRE(!underlying_.empty(), "TRSWrapper::TRSWrapper(): no underlying given, at least one is required");
91 QL_REQUIRE(underlying.size() == underlyingIndex.size(),
92 "TRSWrapper::TRSWrapper(): number of underlyings ("
93 << underlying.size() << ") does not match underlying index size (" << underlyingIndex.size() << ")");
94 QL_REQUIRE(underlying.size() == underlyingMultiplier.size(), "TRSWrapper::TRSWrapper(): number of underlyings ("
95 << underlying.size()
96 << ") does not match underlying index size ("
97 << underlyingMultiplier.size() << ")");
98 QL_REQUIRE(underlying.size() == assetCurrency.size(),
99 "TRSWrapper::TRSWrapper(): number of underlyings ("
100 << underlying.size() << ") does not match asset currency size (" << assetCurrency.size() << ")");
101 QL_REQUIRE(underlying.size() == fxIndexAsset.size(),
102 "TRSWrapper::TRSWrapper(): number of underlyings ("
103 << underlying.size() << ") does not match fx index asset size (" << fxIndexAsset.size() << ")");
104
105 for (Size i = 0; i < underlying_.size(); ++i) {
106 registerWith(underlying_[i]->instrument()->qlInstrument());
107 registerWith(underlyingIndex_[i]);
108 }
109
110 for (Size i = 0; i < fundingLegs_.size(); ++i) {
111 for (Size j = 0; j < fundingLegs_[i].size(); ++j) {
112 registerWith(fundingLegs_[i][j]);
113 }
114 }
115
116 for (auto const& f : fxIndexAsset)
117 registerWith(f);
118 registerWith(fxIndexReturn);
119 registerWith(fxIndexAdditionalCashflows);
120
121 // compute last payment date, after this date the TRS is considered expired
122
123 lastDate_ = Date::minDate();
124 for (auto const& d : paymentSchedule_)
125 lastDate_ = std::max(lastDate_, d);
126 for (auto const& l : fundingLegs_)
127 for (auto const& c : l)
128 lastDate_ = std::max(lastDate_, c->date());
129 for (auto const& c : additionalCashflowLeg_)
130 lastDate_ = std::max(lastDate_, c->date());
131}

173bool TRSWrapperAccrualEngine::computeStartValue(std::vector<Real>& underlyingStartValue,
174 std::vector<Real>& fxConversionFactor, QuantLib::Date& startDate,
175 QuantLib::Date& endDate, bool& usingInitialPrice,
176 const Size nth) const {
177 Date today = Settings::instance().evaluationDate();
178 Size payIdx =
179 std::distance(arguments_.paymentSchedule_.begin(),
180 std::upper_bound(arguments_.paymentSchedule_.begin(), arguments_.paymentSchedule_.end(), today)) +
181 nth;
182 Date v0 = payIdx < arguments_.valuationSchedule_.size() ? arguments_.valuationSchedule_[payIdx] : Date::maxDate();
183 Date v1 =
184 payIdx < arguments_.valuationSchedule_.size() - 1 ? arguments_.valuationSchedule_[payIdx + 1] : Date::maxDate();
185
186 /* Check whether there is a "nth" current valuation period, nth > 0. */
187 if (nth > 0 && (payIdx >= arguments_.paymentSchedule_.size() || v0 > today))
188 return false;
189
190 std::fill(underlyingStartValue.begin(), underlyingStartValue.end(), 0.0);
191 std::fill(fxConversionFactor.begin(), fxConversionFactor.end(), 1.0);
192 startDate = Null<Date>();
193 endDate = Null<Date>();
194 usingInitialPrice = false;
195
196 for (Size i = 0; i < arguments_.underlying_.size(); ++i) {
197 if (payIdx < arguments_.paymentSchedule_.size()) {
198 if (v0 > today) {
199 // The start valuation date is > today: we return null, except an initial price is given, in which case
200 // we return this price (possibly converted with todays FX rate to return ccy). This allows for a
201 // reasonable asset leg npv estimation, which would otherwise be zero and jump to its actual value on v0
202 // + 1. Internal consistency check: make sure that v0 is the initial date of the valuation schedule
203 QL_REQUIRE(payIdx == 0, "TRSWrapper: internal error, expected valuation date "
204 << v0 << " for pay date = " << arguments_.paymentSchedule_[payIdx]
205 << " to be the first valuation date, since it is > today (" << today
206 << ")");
207 if (nth == 0 && arguments_.initialPrice_ != Null<Real>()) {
208 if (i == 0) {
209 Real s0 =
210 arguments_.initialPrice_ *
211 (arguments_.underlyingMultiplier_.size() == 1 ? arguments_.underlyingMultiplier_[i] : 1.0);
212 Real fx0 = getFxConversionRate(today, arguments_.initialPriceCurrency_,
213 arguments_.returnCurrency_, false);
214 DLOG("start value (underlying "
215 << std::to_string(i + 1) << "): s0=" << s0 << " (from fixed initial price), fx0=" << fx0
216 << " => " << fx0 * s0 << " on today (valuation start date is " << v0 << ")");
217 underlyingStartValue[i] = s0;
218 fxConversionFactor[i] = fx0;
219 startDate = v0;
220 if(v1 <= today)
221 endDate = v1;
222 usingInitialPrice = true;
223 } else {
224 underlyingStartValue[i] = 0.0;
225 fxConversionFactor[i] = 1.0;
226 }
227 } else {
228 DLOG("start value (underlying " << std::to_string(i + 1) << ") is null, because eval date ("
229 << today << ") is <= start valuation date (" << v0
230 << ") for nth current period " << nth
231 << " and no intiial price is given");
232 underlyingStartValue[i] = Null<Real>();
233 fxConversionFactor[i] = Null<Real>();
234 startDate = Null<Date>();
235 }
236 } else {
237 // The start valuation date is <= today, we determine the start value from the initial price or a
238 // historical fixing
239 Real s0 = 0.0, fx0 = 1.0;
240 if (nth == 0 && arguments_.initialPrice_ != Null<Real>() &&
241 v0 == arguments_.valuationSchedule_.front()) {
242 if (i == 0) {
243 DLOG("initial price is given as " << arguments_.initialPrice_ << " "
244 << arguments_.initialPriceCurrency_);
245 s0 = arguments_.initialPrice_ *
246 (arguments_.underlying_.size() == 1 ? arguments_.underlyingMultiplier_[i] : 1.0);
247 fx0 = getFxConversionRate(v0, arguments_.initialPriceCurrency_, arguments_.returnCurrency_,
248 false);
249 usingInitialPrice = true;
250 }
251 } else {
252 s0 = getUnderlyingFixing(i, v0, false) * arguments_.underlyingMultiplier_[i];
253 fx0 = getFxConversionRate(v0, arguments_.assetCurrency_[i], arguments_.returnCurrency_, false);
254 }
255 DLOG("start value (underlying " << std::to_string(i + 1) << "): s0=" << s0 << " fx0=" << fx0 << " => "
256 << fx0 * s0 << " on " << v0 << " in nth current period " << nth);
257 underlyingStartValue[i] = s0;
258 fxConversionFactor[i] = fx0;
259 startDate = v0;
260 if(v1 <= today)
261 endDate = v1;
262 }
263 } else {
264 // we are beyond the last date in the payment schedule => return null
265 DLOG("start value (underlying " << std::to_string(i + 1) << ") is null, because eval date (" << today
266 << ") is >= last date in payment schedule ("
267 << arguments_.paymentSchedule_.back() << ") in nth current period " << nth);
268 underlyingStartValue[i] = Null<Real>();
269 fxConversionFactor[i] = Null<Real>();
270 startDate = Null<Date>();
271 }
272 } // loop over underlyings
273
274 return true;
275}

278Real getFxIndexFixing(const boost::shared_ptr<FxIndex>& fx, const Currency& source, const Date& d,
279 const bool enforceProjection) {
280 bool invert = fx->targetCurrency() == source;
281 Real res;
282 if (enforceProjection) {
283 res = fx->forecastFixing(0.0);
284 } else {
285 Date adjustedDate = fx->fixingCalendar().adjust(d, Preceding);
286 res = fx->fixing(adjustedDate, false);
287 }
288 return invert ? 1.0 / res : res;
289}

292Real TRSWrapperAccrualEngine::getFxConversionRate(const Date& date, const Currency& source, const Currency& target,
293 const bool enforceProjection) const {
294
295 if (source == target)
296 return 1.0;
297
298 Real result1 = 1.0;
299 if (source != arguments_.fundingCurrency_) {
300 bool found = false;
301 for (Size i = 0; i < arguments_.fxIndexAsset_.size(); ++i) {
302 if (arguments_.fxIndexAsset_[i] == nullptr)
303 continue;
304 if (source == arguments_.fxIndexAsset_[i]->sourceCurrency() ||
305 source == arguments_.fxIndexAsset_[i]->targetCurrency()) {
306 result1 = getFxIndexFixing(arguments_.fxIndexAsset_[i], source, date, enforceProjection);
307 found = true;
308 }
309 }
310 if (!found) {
311 if (arguments_.fxIndexReturn_ != nullptr && (source == arguments_.fxIndexReturn_->sourceCurrency() ||
312 source == arguments_.fxIndexReturn_->targetCurrency())) {
313 result1 = getFxIndexFixing(arguments_.fxIndexReturn_, source, date, enforceProjection);
314 } else if (arguments_.fxIndexAdditionalCashflows_ != nullptr &&
315 (source == arguments_.fxIndexAdditionalCashflows_->sourceCurrency() ||
316 source == arguments_.fxIndexAdditionalCashflows_->targetCurrency())) {
317 result1 = getFxIndexFixing(arguments_.fxIndexAdditionalCashflows_, source, date, enforceProjection);
318 } else {
319 QL_FAIL("TRSWrapperAccrualEngine: could not convert " << source.code() << " to funding currency "
320 << arguments_.fundingCurrency_
321 << ", are all required FXTerms set up?");
322 }
323 }
324 }
325
326 Real result2 = 1.0;
327 if (target != arguments_.fundingCurrency_) {
328 bool found = false;
329 for (Size i = 0; i < arguments_.fxIndexAsset_.size(); ++i) {
330 if (arguments_.fxIndexAsset_[i] == nullptr)
331 continue;
332 if (target == arguments_.fxIndexAsset_[i]->sourceCurrency() ||
333 target == arguments_.fxIndexAsset_[i]->targetCurrency()) {
334 result2 = getFxIndexFixing(arguments_.fxIndexAsset_[i], target, date, enforceProjection);
335 found = true;
336 }
337 }
338 if (!found) {
339 if (arguments_.fxIndexReturn_ != nullptr && (target == arguments_.fxIndexReturn_->sourceCurrency() ||
340 target == arguments_.fxIndexReturn_->targetCurrency())) {
341 result2 = getFxIndexFixing(arguments_.fxIndexReturn_, target, date, enforceProjection);
342 } else if (arguments_.fxIndexAdditionalCashflows_ != nullptr &&
343 (target == arguments_.fxIndexAdditionalCashflows_->sourceCurrency() ||
344 target == arguments_.fxIndexAdditionalCashflows_->targetCurrency())) {
345 result2 = getFxIndexFixing(arguments_.fxIndexAdditionalCashflows_, target, date, enforceProjection);
346 } else {
347 QL_FAIL("TRSWrapperAccrualEngine: could not convert " << source.code() << " to funding currency "
348 << arguments_.fundingCurrency_
349 << ", are all required FXTerms set up?");
350 }
351 }
352 }
353
354 return result1 / result2;
355}

357Real TRSWrapperAccrualEngine::getUnderlyingFixing(const Size i, const Date& date, const bool enforceProjection) const {
358 Date today = Settings::instance().evaluationDate();
359 QL_REQUIRE(date <= today, "TRSWrapperAccrualEngine: internal error, getUnderlyingFixing("
360 << date << ") for future date requested (today=" << today << ")");
361 if (enforceProjection) {
362 return arguments_.underlying_[i]->instrument()->NPV() / arguments_.underlyingMultiplier_[i];
363 }
364 Date adjustedDate = arguments_.underlyingIndex_[i]->fixingCalendar().adjust(date, Preceding);
365 try {
366 auto tmp = arguments_.underlyingIndex_[i]->fixing(adjustedDate);
367 return tmp;
368 } catch (const std::exception&) {
369 if (adjustedDate == today)
370 return arguments_.underlying_[i]->instrument()->NPV() / arguments_.underlyingMultiplier_[i];
371 else
372 throw;
373 }

376void TRSWrapperAccrualEngine::calculate() const {
377
378 Date today = Settings::instance().evaluationDate();
379
380 DLOG("TRSWrapperAccrualEngine: today = " << today << ", paysAsset = " << std::boolalpha << arguments_.paysAsset_
381 << ", paysFunding = " << std::boolalpha << arguments_.paysFunding_);
382
383 Real assetMultiplier = (arguments_.paysAsset_ ? -1.0 : 1.0);
384 Real fundingMultiplier = (arguments_.paysFunding_ ? -1.0 : 1.0);
385
386 results_.additionalResults["returnCurrency"] = arguments_.returnCurrency_.code();
387 results_.additionalResults["fundingCurrency"] = arguments_.fundingCurrency_.code();
388 results_.additionalResults["returnLegInitialPrice"] = arguments_.initialPrice_;
389 results_.additionalResults["returnLegInitialPriceCurrency"] = arguments_.initialPriceCurrency_.code();
390
391 // asset leg valuation (accrual method)
392
393 Real assetLegNpv = 0.0;
394 Size nthCurrentPeriod = 0;
395 std::vector<CashFlowResults> cfResults;
396
397 std::vector<Real> underlyingStartValue(arguments_.underlying_.size(), 0.0);
398 std::vector<Real> fxConversionFactor(arguments_.underlying_.size(), 1.0);
399 Date startDate = Null<Date>();
400 Date endDate = Null<Date>();
401 bool usingInitialPrice;
402
403 while (computeStartValue(underlyingStartValue, fxConversionFactor, startDate, endDate, usingInitialPrice,
404 nthCurrentPeriod)) {
405
406 // vector holding cashflow results, we store these as an additional result
407
408 for (Size i = 0; i < arguments_.underlying_.size(); ++i) {
409
410 std::string resultSuffix = arguments_.underlying_.size() > 1 ? "_" + std::to_string(i + 1) : "";
411 if (nthCurrentPeriod > 0)
412 resultSuffix += "_nth(" + std::to_string(nthCurrentPeriod) + ")";
413
414 results_.additionalResults["underlyingCurrency" + resultSuffix] = arguments_.assetCurrency_[i].code();
415
416 if (underlyingStartValue[i] != Null<Real>()) {
417 Real s1, fx1;
418 if (endDate == Null<Date>()) {
419 s1 = arguments_.underlying_[i]->instrument()->NPV();
420 fx1 = getFxConversionRate(today, arguments_.assetCurrency_[i], arguments_.returnCurrency_, true);
421 } else {
422 s1 = getUnderlyingFixing(i, endDate, false) * arguments_.underlyingMultiplier_[i];
423 fx1 = getFxConversionRate(endDate, arguments_.assetCurrency_[i], arguments_.returnCurrency_, false);
424 }
425 assetLegNpv += fx1 * s1 - underlyingStartValue[i] * fxConversionFactor[i];
426 DLOG("end value (underlying " << std::to_string(i + 1) << "): s1=" << s1 << " fx1=" << fx1 << " => "
427 << fx1 * s1 << " on "
428 << io::iso_date(endDate == Null<Date>() ? today : endDate));
429
430 // add details return leg valuation to additional results
431 results_.additionalResults["s0" + resultSuffix] = underlyingStartValue[i];
432 results_.additionalResults["fx0" + resultSuffix] = fxConversionFactor[i];
433 results_.additionalResults["s1" + resultSuffix] = s1;
434 results_.additionalResults["fx1" + resultSuffix] = fx1;
435 results_.additionalResults["underlyingMultiplier" + resultSuffix] = arguments_.underlyingMultiplier_[i];
436
437 // add return cashflow to additional results
438 cfResults.emplace_back();
439 cfResults.back().amount = fx1 * s1;
440 if (arguments_.underlying_.size() == 1 || !usingInitialPrice) {
441 cfResults.back().amount -= underlyingStartValue[i] * fxConversionFactor[i];
442 }
443 cfResults.back().amount *= assetMultiplier;
444 cfResults.back().payDate = today;
445 cfResults.back().currency = arguments_.returnCurrency_.code();
446 cfResults.back().legNumber = 0;
447 cfResults.back().type = "AccruedReturn" + resultSuffix;
448 cfResults.back().accrualStartDate = startDate;
449 cfResults.back().accrualEndDate = endDate == Null<Date>() ? today : endDate;
450 cfResults.back().fixingValue = s1 / arguments_.underlyingMultiplier_[i];
451 cfResults.back().notional = underlyingStartValue[i] * fxConversionFactor[i];
452
453 // if initial price is used and there is more than one underlying, add cf for initialPrice
454 if (arguments_.underlying_.size() > 1 && usingInitialPrice && i == 0) {
455 cfResults.emplace_back();
456 cfResults.back().amount = -underlyingStartValue[i] * fxConversionFactor[i];
457 cfResults.back().payDate = today;
458 cfResults.back().currency = arguments_.returnCurrency_.code();
459 cfResults.back().legNumber = 0;
460 cfResults.back().type = "AccruedReturn" + resultSuffix;
461 cfResults.back().accrualStartDate = startDate;
462 cfResults.back().accrualEndDate = endDate == Null<Date>() ? today : endDate;
463 cfResults.back().notional = underlyingStartValue[i] * fxConversionFactor[i];
464 }
465
466 // startDate might be >= today, if an initial price is given, see the comment in startValue() above
467 if (arguments_.includeUnderlyingCashflowsInReturn_ && startDate != Null<Date>() && startDate < today) {
468 // add cashflows in return period
469 Real cf = 0.0;
470 for (auto const& l : arguments_.underlying_[i]->legs()) {
471 for (auto const& c : l) {
472 if (!c->hasOccurred(startDate) && c->hasOccurred(today)) {
473 Real tmp = c->amount() * arguments_.underlyingMultiplier_[i];
474 cf += tmp;
475 // add intermediate cashflows to additional results
476 cfResults.emplace_back();
477 cfResults.back().amount = assetMultiplier * (tmp * fx1);
478 cfResults.back().payDate = c->date();
479 cfResults.back().currency = arguments_.returnCurrency_.code();
480 cfResults.back().legNumber = 1;
481 cfResults.back().type = "UnderlyingCashFlow" + resultSuffix;
482 cfResults.back().notional = underlyingStartValue[i] * fxConversionFactor[i];
483 }
484 }
485 }
486 // account for dividends
487 Real dividends = 0.0;
488 if (auto e = boost::dynamic_pointer_cast<EquityIndex2>(arguments_.underlyingIndex_[i])) {
489 dividends +=
490 e->dividendsBetweenDates(startDate + 1, today) * arguments_.underlyingMultiplier_[i];
491 } else if (auto e = boost::dynamic_pointer_cast<CompositeIndex>(arguments_.underlyingIndex_[i])) {
492 dividends +=
493 e->dividendsBetweenDates(startDate + 1, today) * arguments_.underlyingMultiplier_[i];
494 }
495 cf += dividends;
496 if (!close_enough(dividends, 0.0)) {
497 // add dividends as one cashflow to additional results
498 cfResults.emplace_back();
499 cfResults.back().amount = assetMultiplier * (dividends * fx1);
500 cfResults.back().payDate = today;
501 cfResults.back().currency = arguments_.returnCurrency_.code();
502 cfResults.back().legNumber = 2;
503 cfResults.back().type = "UnderlyingDividends" + resultSuffix;
504 cfResults.back().notional = underlyingStartValue[i] * fxConversionFactor[i];
505 }
506
507 DLOG("add cashflows in return period (" << io::iso_date(startDate) << ", " << io::iso_date(today)
508 << "]: amount in asset ccy = " << cf << ", fx conversion "
509 << fx1 << " => " << cf * fx1);
510
511 results_.additionalResults["underlyingCashflows" + resultSuffix] = cf;
512
513 assetLegNpv += cf * fx1;
514 }
515 }
516 } // loop over underlyings
517
518 ++nthCurrentPeriod;
519 } // loop over nth current period
520
521 results_.additionalResults["assetLegNpv"] = assetMultiplier * assetLegNpv;
522 results_.additionalResults["assetLegNpvCurency"] = arguments_.returnCurrency_.code();
523 DLOG("asset leg npv = " << assetMultiplier * assetLegNpv << " " << arguments_.returnCurrency_.code());
524
525 // funding leg valuation (accrual method)
526
527 Real fundingLegNpv = 0.0;
528
529 for (Size i = 0; i < arguments_.fundingLegs_.size(); ++i) {
530
531 Size nthCpn = 0;
532 for (Size cpnNo = 0; cpnNo < arguments_.fundingLegs_[i].size(); ++cpnNo) {
533
534 Real localFundingLegNpv = 0.0; // local per funding coupon
535
536 auto cpn = boost::dynamic_pointer_cast<QuantLib::Coupon>(arguments_.fundingLegs_[i][cpnNo]);
537 if (cpn == nullptr || cpn->date() <= today || cpn->accrualStartDate() >= today)
538 continue;
539
540 // look up latest valuation date <= funding start date, fall back to first valuation date, if no such
541 // date exists
542 Date fundingStartDate = cpn->accrualStartDate();
543 Real fundingCouponNotional = cpn->nominal();
544 Size currentIdx = std::distance(arguments_.valuationSchedule_.begin(),
545 std::upper_bound(arguments_.valuationSchedule_.begin(),
546 arguments_.valuationSchedule_.end(),
547 fundingStartDate + arguments_.fundingResetGracePeriod_));
548 if (currentIdx > 0)
549 --currentIdx;
550
551 if (arguments_.valuationSchedule_[currentIdx] > today) {
552 DLOG("fundingLegNpv = 0 for funding leg #" << (i + 1) << ", because last relevant valuation date ("
553 << arguments_.valuationSchedule_[currentIdx]
554 << ") is >= eval date (" << today << ")");
555 continue;
556 }
557
558 localFundingLegNpv = cpn->accruedAmount(today);
559
560 Real fundingLegNotionalFactor = 0.0;
561 std::string resultSuffix = arguments_.fundingLegs_.size() > 1 ? "_" + std::to_string(i + 1) : "";
562
563 try {
564 results_.additionalResults["fundingCouponRate" + resultSuffix] = cpn->rate();
565 } catch (...) {
566 }
567
568 for (Size j = 0; j < arguments_.underlying_.size(); ++j) {
569
570 std::string resultSuffix2 = arguments_.underlying_.size() > 1 ? "_" + std::to_string(j + 1) : "";
571
572 if (nthCpn > 0)
573 resultSuffix2 += "_nth(" + std::to_string(nthCpn) + ")";
574
575 if (arguments_.fundingNotionalTypes_[i] == TRS::FundingData::NotionalType::Fixed) {
576
577 fundingLegNotionalFactor = 1.0;
578
579 } else if (arguments_.fundingNotionalTypes_[i] == TRS::FundingData::NotionalType::PeriodReset) {
580
581 Real localNotionalFactor = 0.0, localFxFactor = 1.0; // local per underlying
582 if (currentIdx == 0 && arguments_.initialPrice_ != Null<Real>()) {
583 if (j == 0) {
584 localNotionalFactor =
585 arguments_.initialPrice_ *
586 (arguments_.underlying_.size() == 1 ? arguments_.underlyingMultiplier_[j] : 1.0);
587 localFxFactor = getFxConversionRate(arguments_.valuationSchedule_[currentIdx],
588 arguments_.initialPriceCurrency_,
589 arguments_.fundingCurrency_, false);
590 }
591 } else {
592 localNotionalFactor = arguments_.underlyingMultiplier_[j] *
593 getUnderlyingFixing(j, arguments_.valuationSchedule_[currentIdx], false);
594 localFxFactor =
595 getFxConversionRate(arguments_.valuationSchedule_[currentIdx], arguments_.assetCurrency_[j],
596 arguments_.fundingCurrency_, false);
597 }
598
599 fundingLegNotionalFactor += localNotionalFactor * localFxFactor;
600
601 results_.additionalResults["fundingLegNotional" + resultSuffix + resultSuffix2] =
602 localNotionalFactor;
603 results_.additionalResults["fundingLegFxRate" + resultSuffix + resultSuffix2] = localFxFactor;
604
605 } else if (arguments_.fundingNotionalTypes_[i] == TRS::FundingData::NotionalType::DailyReset) {
606
607 auto fixedCpn = boost::dynamic_pointer_cast<QuantLib::FixedRateCoupon>(cpn);
608 QL_REQUIRE(fixedCpn, "NotionalType DailyReset is only supported for fixed rate funding legs");
609
610 Real dcfTotal = fixedCpn->dayCounter().yearFraction(fixedCpn->accrualStartDate(),
611 std::min(fixedCpn->accrualEndDate(), today));
612 for (QuantLib::Date d = fixedCpn->accrualStartDate();
613 d < std::min(fixedCpn->accrualEndDate(), today); ++d) {
614 Real dcfLocal = fixedCpn->dayCounter().yearFraction(d, d + 1);
615 Date fixingDate = arguments_.underlyingIndex_[j]->fixingCalendar().adjust(d, Preceding);
616 Real localNotionalFactor = getUnderlyingFixing(j, fixingDate, false) *
617 arguments_.underlyingMultiplier_[j] * dcfLocal / dcfTotal;
618 Real localFxFactor = getFxConversionRate(fixingDate, arguments_.assetCurrency_[j],
619 arguments_.fundingCurrency_, false);
620 fundingLegNotionalFactor += localNotionalFactor * localFxFactor;
621
622 results_.additionalResults["fundingLegNotional" + resultSuffix + resultSuffix2 + "_" +
623 ore::data::to_string(d)] = localNotionalFactor;
624 results_.additionalResults["fundingLegFxRate" + resultSuffix + resultSuffix2 + "_" +
625 ore::data::to_string(d)] = localFxFactor;
626 }
627
628 } else {
629 QL_FAIL("internal error: unknown notional type, contact dev");
630 }
631 } // loop over underlyings
632
633 DLOG("fundingLegNpv for funding leg #"
634 << (i + 1) << " is " << fundingMultiplier * localFundingLegNpv << " * " << fundingLegNotionalFactor
635 << " = " << fundingMultiplier * localFundingLegNpv * fundingLegNotionalFactor << " "
636 << arguments_.fundingCurrency_ << " (notional type of leg is '" << arguments_.fundingNotionalTypes_[i]
637 << "')");
638
639 localFundingLegNpv *= fundingLegNotionalFactor;
640
641 results_.additionalResults["fundingLegNpv" + resultSuffix] = fundingMultiplier * localFundingLegNpv;
642
643 // add funding leg cashflow to addtional results
644 cfResults.emplace_back();
645 cfResults.back().amount = fundingMultiplier * localFundingLegNpv;
646 cfResults.back().payDate = today;
647 cfResults.back().currency = arguments_.fundingCurrency_.code();
648 cfResults.back().legNumber = 3 + i;
649 cfResults.back().type = "AccruedFunding" + (nthCpn > 0 ? "_nth(" + std::to_string(nthCpn) + ")" : "");
650 cfResults.back().accrualStartDate = std::min(fundingStartDate, today);
651 cfResults.back().accrualEndDate = today;
652 cfResults.back().notional = arguments_.fundingNotionalTypes_[i] == TRS::FundingData::NotionalType::Fixed
653 ? fundingCouponNotional
654 : fundingLegNotionalFactor;
655
656 results_.additionalResults["fundingLegNotional" + resultSuffix] = cfResults.back().notional;
657
658 fundingLegNpv += localFundingLegNpv;
659 ++nthCpn;
660 } // loop over funding leg coupons (indexed by cpnNo)
661 } // loop over funding legs (indexed by i)
662
663 DLOG("total funding leg(s) npv is " << fundingMultiplier * fundingLegNpv);
664
665 results_.additionalResults["fundingLegNpv"] = fundingMultiplier * fundingLegNpv;
666 results_.additionalResults["fundingLegNpvCurrency"] = arguments_.fundingCurrency_.code();
667
668 // additional cashflow leg valuation (take the plain amount of future cashflows as if paid today)
669
670 Real additionalCashflowLegNpv = 0.0;
671 for (auto const& cf : arguments_.additionalCashflowLeg_) {
672 if (cf->date() > today) {
673 Real tmp = cf->amount() * (arguments_.additionalCashflowLegPayer_ ? -1.0 : 1.0);
674 additionalCashflowLegNpv += tmp;
675 // add additional cashflows to additional results
676 cfResults.emplace_back();
677 cfResults.back().amount = tmp;
678 cfResults.back().payDate = cf->date();
679 cfResults.back().currency = arguments_.additionalCashflowCurrency_.code();
680 cfResults.back().legNumber = 0;
681 cfResults.back().type = "AdditionalCashFlow";
682 }
683 }
684 DLOG("additionalCashflowLegNpv = " << additionalCashflowLegNpv << " " << arguments_.additionalCashflowCurrency_);
685 results_.additionalResults["additionalCashflowLegNpv"] = additionalCashflowLegNpv;
686 results_.additionalResults["additionalCashflowLegNpvCurrency"] = arguments_.additionalCashflowCurrency_.code();
687
688 // set npv and current notional, set additional results
689
690 Real fxAssetToPnlCcy = getFxConversionRate(today, arguments_.returnCurrency_, arguments_.fundingCurrency_, true);
691 Real fxAdditionalCashflowLegToPnlCcy =
692 getFxConversionRate(today, arguments_.additionalCashflowCurrency_, arguments_.fundingCurrency_, true);
693
694 results_.additionalResults["fxConversionAssetLegNpvToPnlCurrency"] = fxAssetToPnlCcy;
695 results_.additionalResults["fxConversionAdditionalCashflowLegNpvToPnlCurrency"] = fxAdditionalCashflowLegToPnlCcy;
696 results_.additionalResults["pnlCurrency"] = arguments_.fundingCurrency_.code();
697
698 results_.value = assetMultiplier * assetLegNpv * fxAssetToPnlCcy + fundingMultiplier * fundingLegNpv +
699 additionalCashflowLegNpv * fxAdditionalCashflowLegToPnlCcy;
700
701 Real currentNotional = 0.0;
702 for (Size j = 0; j < arguments_.underlying_.size(); ++j) {
703 // this is using the underlyingStartValue and fxConversionFactor that was populated during the
704 // valuation of the asset leg above in the last "nth current period" which contributed to this npv
705 if (underlyingStartValue[j] == Null<Real>()) {
706 currentNotional +=
707 arguments_.underlyingMultiplier_[j] * getUnderlyingFixing(j, today, true) *
708 getFxConversionRate(today, arguments_.initialPriceCurrency_, arguments_.returnCurrency_, true);
709 } else {
710 currentNotional += underlyingStartValue[j] * fxConversionFactor[j];
711 }
712 }
713
714 for (Size j = 0; j < arguments_.underlying_.size(); ++j) {
715 // the start fixing will refer to the last of the nth current return periods
716 std::string resultSuffix = arguments_.underlying_.size() == 1 ? "" : "_" + std::to_string(j);
717 Real startFixing = Null<Real>(), todaysFixing = Null<Real>();
718 try {
719 startFixing = getUnderlyingFixing(j, startDate, false);
720 } catch (...) {
721 }
722 try {
723 todaysFixing = getUnderlyingFixing(j, today, true);
724 } catch (...) {
725 }
726 results_.additionalResults["startFixing" + resultSuffix] = startFixing;
727 results_.additionalResults["todaysFixing" + resultSuffix] = todaysFixing;
728 }
729
730 for (auto const& d : arguments_.addFxIndices_) {
731 Real startFixing = Null<Real>(), todaysFixing = Null<Real>();
732 try {
733 startFixing = d.second->fixing(d.second->fixingCalendar().adjust(startDate, Preceding));
734 } catch (...) {
735 }
736 try {
737 todaysFixing = d.second->fixing(d.second->fixingCalendar().adjust(today, Preceding), true);
738 } catch (...) {
739 }
740 results_.additionalResults["startFxFixing(" + d.first + ")"] = startFixing;
741 results_.additionalResults["todaysFxFixing(" + d.first + ")"] = todaysFixing;
742 }
743
744 results_.additionalResults["currentNotional"] = currentNotional * fxAssetToPnlCcy;
745 results_.additionalResults["cashFlowResults"] = cfResults;
746
747 // propagate underlying additional results to trswrapper
748
749 for (Size i = 0; i < arguments_.underlying_.size(); ++i) {
750 for (auto const& [key, value] : arguments_.underlying_[i]->instrument()->additionalResults()) {
751 results_.additionalResults["und_ar_" + std::to_string(i + 1) + "_" + key] = value;
752 }
753 }
754
755 DLOG("TRSWrapperAccrualEngine: all done, total npv = " << results_.value << " "
756 << arguments_.fundingCurrency_.code());
757}
