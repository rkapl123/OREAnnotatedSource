46// Return swaption data
47SwaptionData swaptionData(const boost::shared_ptr<Swaption> swaption, const Handle<YieldTermStructure>& yts,
48 const Handle<SwaptionVolatilityStructure>& svts) {
49
50 boost::shared_ptr<PricingEngine> engine;
51 switch (svts->volatilityType()) {
52 case ShiftedLognormal:
53 engine = boost::make_shared<BlackSwaptionEngine>(yts, svts);
54 break;
55 case Normal:
56 engine = boost::make_shared<BachelierSwaptionEngine>(yts, svts);
57 break;
58 default:
59 QL_FAIL("Could not construct swaption engine for volatility type: " << svts->volatilityType());
60 break;
61 }
62
63 swaption->setPricingEngine(engine);
64
65 SwaptionData sd;
66 sd.timeToExpiry = yts->timeFromReference(swaption->exercise()->dates().back());
67 sd.swapLength = swaption->result<Real>("swapLength");
68 sd.strike = swaption->result<Real>("strike");
69 sd.atmForward = swaption->result<Real>("atmForward");
70 sd.annuity = swaption->result<Real>("annuity");
71 sd.vega = swaption->result<Real>("vega");
72 sd.stdDev = swaption->result<Real>("stdDev");
73
74 return sd;
75}

77// Utility function to create swaption helper. Returns helper and (possibly updated) strike
78template <typename E, typename T>
79std::pair<boost::shared_ptr<SwaptionHelper>, double>
80createSwaptionHelper(const E& expiry, const T& term, const Handle<SwaptionVolatilityStructure>& svts,
81 const Handle<Quote>& vol, const boost::shared_ptr<IborIndex>& iborIndex,
82 const Period& fixedLegTenor, const DayCounter& fixedDayCounter, const DayCounter& floatDayCounter,
83 const Handle<YieldTermStructure>& yts, BlackCalibrationHelper::CalibrationErrorType errorType,
84 Real strike, Real shift, const Size settlementDays, const RateAveraging::Type averagingMethod) {
85
86 // hardcoded parameters to ensure a robust cailbration:
87
88 // 1 If the helper's strike is too far away from the ATM level in terms of the relevant std dev, we move the
89 // calibration strike closer to the ATM level
90 static constexpr Real maxAtmStdDev = 3.0;
91
92 // 2 If the helper value is lower than mmv, replace it with a "more reasonable" helper. Here, we replace
93 // the helper with a helper that has the ATM strike. There are other options here.
94 static constexpr Real mmv = 1.0E-20;
95
96 // 3 Switch to PriceError if helper's market value is below smv
97 static constexpr Real smv = 1.0E-8;
98
99 // Notice: the vol that is passed in to this method is in general a dummy value, which is good enough though to
100 // check 2 and 3 above. To check 1, the vol is not needed at all.
101
102 auto vt = svts->volatilityType();
103 auto helper = boost::make_shared<SwaptionHelper>(expiry, term, vol, iborIndex, fixedLegTenor, fixedDayCounter,
104 floatDayCounter, yts, errorType, strike, 1.0, vt, shift,
105 settlementDays, averagingMethod);
106 auto sd = swaptionData(helper->swaption(), yts, svts);
107
108 // ensure point 1 from above
109
110 Real atmStdDev = svts->volatility(sd.timeToExpiry, sd.swapLength, sd.atmForward) * std::sqrt(sd.timeToExpiry);
111 if (vt == ShiftedLognormal) {
112 atmStdDev *= sd.atmForward + shift;
113 }
114 if (strike != Null<Real>() && std::abs(strike - sd.atmForward) > maxAtmStdDev * atmStdDev) {
115 DLOG("Helper with expiry " << expiry << " and term " << term << " has a strike (" << strike
116 << ") that is too far out of the money (atm = " << sd.atmForward << ", atmStdDev = "
117 << atmStdDev << "). Adjusting the strike using maxAtmStdDev " << maxAtmStdDev);
118 if (strike > sd.atmForward)
119 strike = sd.atmForward + maxAtmStdDev * atmStdDev;
120 else
121 strike = sd.atmForward - maxAtmStdDev * atmStdDev;
122 helper = boost::make_shared<SwaptionHelper>(expiry, term, vol, iborIndex, fixedLegTenor, fixedDayCounter,
123 floatDayCounter, yts, errorType, strike, 1.0, vt, shift,
124 settlementDays, averagingMethod);
125 }
126
127 // ensure point 2 from above
128
129 auto mv = std::abs(helper->marketValue());
130 if (mv < mmv) {
131 DLOG("Helper with expiry " << expiry << " and term " << term << " has an absolute market value of "
132 << std::scientific << mv << " which is lower than minimum market value " << mmv
133 << " so switching to helper with atm rate " << sd.atmForward);
134 strike = sd.atmForward;
135 helper = boost::make_shared<SwaptionHelper>(expiry, term, vol, iborIndex, fixedLegTenor, fixedDayCounter,
136 floatDayCounter, yts, errorType, strike, 1.0, vt, shift,
137 settlementDays, averagingMethod);
138 }
139
140 // ensure point 3 from above
141
142 mv = std::abs(helper->marketValue());
143 if (errorType != BlackCalibrationHelper::PriceError && mv < smv) {
144 errorType = BlackCalibrationHelper::PriceError;
145 TLOG("Helper with expiry " << expiry << " and term " << term << " has an absolute market value of "
146 << std::scientific << mv << " which is lower than " << smv
147 << " so switching to a price error helper.");
148 helper = boost::make_shared<SwaptionHelper>(expiry, term, vol, iborIndex, fixedLegTenor, fixedDayCounter,
149 floatDayCounter, yts, errorType, strike, 1.0, vt, shift,
150 settlementDays, averagingMethod);
151 }
152
153 DLOG("Created swaption helper with expiry " << expiry << " and term " << term << ": vol=" << vol->value()
154 << ", index=" << iborIndex->name() << ", strike=" << strike
155 << ", shift=" << shift);
156
157 return std::make_pair(helper, strike);
158}

165LgmBuilder::LgmBuilder(const boost::shared_ptr<ore::data::Market>& market, const boost::shared_ptr<IrLgmData>& data,
166 const std::string& configuration, const Real bootstrapTolerance, const bool continueOnError,
167 const std::string& referenceCalibrationGrid, const bool setCalibrationInfo)
168 : market_(market), configuration_(configuration), data_(data), bootstrapTolerance_(bootstrapTolerance),
169 continueOnError_(continueOnError), referenceCalibrationGrid_(referenceCalibrationGrid),
170 setCalibrationInfo_(setCalibrationInfo),
171 optimizationMethod_(boost::shared_ptr<OptimizationMethod>(new LevenbergMarquardt(1E-8, 1E-8, 1E-8))),
172 endCriteria_(EndCriteria(1000, 500, 1E-8, 1E-8, 1E-8)),
173 calibrationErrorType_(BlackCalibrationHelper::RelativePriceError) {
174
175 marketObserver_ = boost::make_shared<MarketObserver>();
176 string qualifier = data_->qualifier();
177 currency_ = qualifier;
178 boost::shared_ptr<IborIndex> index;
179 if(tryParseIborIndex(qualifier,index)) {
180 currency_ = index->currency().code();
181 }
182 LOG("LgmCalibration for qualifier " << qualifier << " (ccy=" << currency_ << "), configuration is " << configuration_);
183 Currency ccy = parseCurrency(currency_);
184
185 requiresCalibration_ =
186 (data_->calibrateA() || data_->calibrateH()) && data_->calibrationType() != CalibrationType::None;
187
188 // the discount curve underlying the model might be relinked to a different curve outside this builder
189 // the calibration curve should always stay the same though, therefore we create a different handle for this
190 modelDiscountCurve_ = RelinkableHandle<YieldTermStructure>(*market_->discountCurve(currency_, configuration_));
191 calibrationDiscountCurve_ = Handle<YieldTermStructure>(*modelDiscountCurve_);
192
193 if (requiresCalibration_) {
194 svts_ = market_->swaptionVol(data_->qualifier(), configuration_);
195 swapIndex_ = market_->swapIndex(market_->swapIndexBase(data_->qualifier(), configuration_), configuration_);
196 shortSwapIndex_ = market_->swapIndex(market_->shortSwapIndexBase(data_->qualifier(), configuration_), configuration_);
197 registerWith(svts_);
198 marketObserver_->addObservable(swapIndex_->forwardingTermStructure());
199 marketObserver_->addObservable(swapIndex_->discountingTermStructure());
200 marketObserver_->addObservable(shortSwapIndex_->forwardingTermStructure());
201 marketObserver_->addObservable(shortSwapIndex_->discountingTermStructure());
202 }
203 marketObserver_->addObservable(calibrationDiscountCurve_);
204 registerWith(marketObserver_);
205 // notify observers of all market data changes, not only when not calculated
206 alwaysForwardNotifications();
207
208 swaptionActive_ = std::vector<bool>(data_->optionExpiries().size(), false);
209
210 if (requiresCalibration_) {
211 buildSwaptionBasket();
212 }
213
214 Array aTimes(data_->aTimes().begin(), data_->aTimes().end());
215 Array hTimes(data_->hTimes().begin(), data_->hTimes().end());
216 Array alpha(data_->aValues().begin(), data_->aValues().end());
217 Array h(data_->hValues().begin(), data_->hValues().end());
218
219 if (data_->aParamType() == ParamType::Constant) {
220 QL_REQUIRE(data_->aTimes().size() == 0,
221 "LgmBuilder: empty volatility time grid expected for constant parameter type");
222 QL_REQUIRE(data_->aValues().size() == 1,
223 "LgmBuilder: initial volatility values should have size 1 for constant parameter type");
224 } else if (data_->aParamType() == ParamType::Piecewise) {
225 if (data_->calibrateA() && data_->calibrationType() == CalibrationType::Bootstrap) {
226 if (data_->aTimes().size() > 0) {
227 DLOG("overriding alpha time grid with swaption expiries, set all initial values to first given value");
228 }
229 QL_REQUIRE(swaptionExpiries_.size() > 0, "empty swaptionExpiries");
230 aTimes = Array(swaptionExpiries_.begin(), swaptionExpiries_.end() - 1);
231 alpha = Array(aTimes.size() + 1, data_->aValues()[0]);
232 } else {
233 QL_REQUIRE(alpha.size() == aTimes.size() + 1,
234 "LgmBuilder: LGM volatility time and initial value array sizes do not match");
235 }
236 } else
237 QL_FAIL("LgmBuilder: volatility parameter type not covered");
238
239 if (data_->hParamType() == ParamType::Constant) {
240 QL_REQUIRE(data_->hTimes().size() == 0,
241 "LgmBuilder: empty reversion time grid expected for constant parameter type");
242 QL_REQUIRE(data_->hValues().size() == 1,
243 "LgmBuidler: initial reversion values should have size 1 for constant parameter type");
244 } else if (data_->hParamType() == ParamType::Piecewise) {
245 if (data_->calibrateH() && data_->calibrationType() == CalibrationType::Bootstrap) {
246 if (data_->hTimes().size() > 0) {
247 DLOG("overriding h time grid with swaption underlying maturities, set all initial values to first "
248 "given value");
249 }
250 hTimes = swaptionMaturities_;
251 h = Array(hTimes.size() + 1, data_->hValues()[0]);
252 } else { // use input time grid and input h array otherwise
253 QL_REQUIRE(h.size() == hTimes.size() + 1, "H grids do not match");
254 }
255 } else
256 QL_FAIL("LgmBuilder: reversion parameter type case not covered");
257
258 DLOG("before calibration: alpha times = " << aTimes << " values = " << alpha);
259 DLOG("before calibration: h times = " << hTimes << " values = " << h);
260
261 if (data_->reversionType() == LgmData::ReversionType::HullWhite &&
262 data_->volatilityType() == LgmData::VolatilityType::HullWhite) {
263 DLOG("IR parametrization for " << qualifier << ": IrLgm1fPiecewiseConstantHullWhiteAdaptor");
264 parametrization_ = boost::make_shared<QuantExt::IrLgm1fPiecewiseConstantHullWhiteAdaptor>(
265 ccy, modelDiscountCurve_, aTimes, alpha, hTimes, h);
266 } else if (data_->reversionType() == LgmData::ReversionType::HullWhite &&
267 data_->volatilityType() == LgmData::VolatilityType::Hagan) {
268 DLOG("IR parametrization for " << qualifier << ": IrLgm1fPiecewiseConstant");
269 parametrization_ = boost::make_shared<QuantExt::IrLgm1fPiecewiseConstantParametrization>(
270 ccy, modelDiscountCurve_, aTimes, alpha, hTimes, h);
271 } else if (data_->reversionType() == LgmData::ReversionType::Hagan &&
272 data_->volatilityType() == LgmData::VolatilityType::Hagan) {
273 parametrization_ = boost::make_shared<QuantExt::IrLgm1fPiecewiseLinearParametrization>(
274 ccy, modelDiscountCurve_, aTimes, alpha, hTimes, h);
275 DLOG("IR parametrization for " << qualifier << ": IrLgm1fPiecewiseLinear");
276 } else {
277 QL_FAIL("LgmBuilder: Reversion type Hagan and volatility type HullWhite not covered");
278 }
279 DLOG("alpha times size: " << aTimes.size());
280 DLOG("lambda times size: " << hTimes.size());
281
282 model_ = boost::make_shared<QuantExt::LGM>(parametrization_);
283 params_ = model_->params();
284}

311void LgmBuilder::performCalculations() const {
312
313 DLOG("Recalibrate LGM model for qualifier " << data_->qualifier() << " currency " << currency_);
314
315 if (!requiresRecalibration()) {
316 DLOG("Skipping calibration as nothing has changed");
317 return;
318 }
319
320 // reset lgm observer's updated flag
321 marketObserver_->hasUpdated(true);
322
323 if (swaptionBasketRefDate_ != calibrationDiscountCurve_->referenceDate()) {
324 // build swaption basket if required, i.e. if reference date has changed since last build
325 buildSwaptionBasket();
326 volSurfaceChanged(true);
327 updateSwaptionBasketVols();
328 } else {
329 // otherwise just update vols
330 volSurfaceChanged(true);
331 updateSwaptionBasketVols();
332 }
333
334 for (Size j = 0; j < swaptionBasket_.size(); j++) {
335 auto engine = boost::make_shared<QuantExt::AnalyticLgmSwaptionEngine>(model_, calibrationDiscountCurve_);
336 engine->enableCache(!data_->calibrateH(), !data_->calibrateA());
337 swaptionBasket_[j]->setPricingEngine(engine);
338 // necessary if notifications are disabled (observation mode = Disable)
339 swaptionBasket_[j]->update();
340 }
341
342 // reset model parameters to ensure identical results on identical market data input
343 model_->setParams(params_);
344 parametrization_->shift() = 0.0;
345 parametrization_->scaling() = 1.0;
346
347 LgmCalibrationInfo calibrationInfo;
348 error_ = QL_MAX_REAL;
349 try {
350 if (data_->calibrateA() && !data_->calibrateH() && data_->calibrationType() == CalibrationType::Bootstrap) {
351 DLOG("call calibrateVolatilitiesIterative for volatility calibration (bootstrap)");
352 model_->calibrateVolatilitiesIterative(swaptionBasket_, *optimizationMethod_, endCriteria_);
353 } else if (data_->calibrateH() && !data_->calibrateA() &&
354 data_->calibrationType() == CalibrationType::Bootstrap) {
355 DLOG("call calibrateReversionsIterative for reversion calibration (bootstrap)");
356 model_->calibrateVolatilitiesIterative(swaptionBasket_, *optimizationMethod_, endCriteria_);
357 } else {
358 QL_REQUIRE(data_->calibrationType() != CalibrationType::Bootstrap,
359 "LgmBuidler: Calibration type Bootstrap can be used with volatilities and reversions calibrated "
360 "simultaneously. Either choose BestFit oder fix one of these parameters.");
361 if (data_->calibrateA() && !data_->calibrateH()) {
362 DLOG("call calibrateVolatilities for (global) volatility calibration")
363 model_->calibrateVolatilities(swaptionBasket_, *optimizationMethod_, endCriteria_);
364 } else if (data_->calibrateH() && !data_->calibrateA()) {
365 DLOG("call calibrateReversions for (global) reversion calibration")
366 model_->calibrateReversions(swaptionBasket_, *optimizationMethod_, endCriteria_);
367 } else {
368 DLOG("call calibrate for global volatility and reversion calibration");
369 model_->calibrate(swaptionBasket_, *optimizationMethod_, endCriteria_);
370 }
371 }
372 TLOG("LGM " << data_->qualifier() << " calibration errors:");
373 error_ = getCalibrationError(swaptionBasket_);
374 } catch (const std::exception& e) {
375 // just log a warning, we check below if we meet the bootstrap tolerance and handle the result there
376 WLOG(StructuredModelErrorMessage("Error during LGM calibration: ", e.what()));
377 }
378 calibrationInfo.rmse = error_;
379 if (fabs(error_) < bootstrapTolerance_ ||
380 (data_->calibrationType() == CalibrationType::BestFit && error_ != QL_MAX_REAL)) {
381 // we check the log level here to avoid unnecessary computations
382 if (Log::instance().filter(ORE_DATA) || setCalibrationInfo_) {
383 TLOGGERSTREAM("Basket details:");
384 try {
385 auto d = getBasketDetails(calibrationInfo);
386 TLOGGERSTREAM(d);
387 } catch (const std::exception& e) {
388 WLOG("An error occurred: " << e.what());
389 }
390 TLOGGERSTREAM("Calibration details (with time grid = calibration swaption expiries):");
391 try {
392 auto d = getCalibrationDetails(calibrationInfo, swaptionBasket_, parametrization_);
393 TLOGGERSTREAM(d);
394 } catch (const std::exception& e) {
395 WLOG("An error occurred: " << e.what());
396 }
397 TLOGGERSTREAM("Parameter details (with parameter time grid)");
398 TLOGGERSTREAM(getCalibrationDetails(calibrationInfo, swaptionBasket_, parametrization_))
399 TLOGGERSTREAM("rmse = " << error_);
400 calibrationInfo.valid = true;
401 }
402 } else {
403 std::string exceptionMessage = "LGM (" + data_->qualifier() + ") calibration error " + std::to_string(error_) +
404 " exceeds tolerance " + std::to_string(bootstrapTolerance_);
405 WLOG(StructuredModelErrorMessage("Failed to calibrate LGM Model", exceptionMessage));
406 WLOGGERSTREAM("Basket details:");
407 try {
408 auto d = getBasketDetails(calibrationInfo);
409 WLOGGERSTREAM(d);
410 } catch (const std::exception& e) {
411 WLOG("An error occurred: " << e.what());
412 }
413 WLOGGERSTREAM("Calibration details (with time grid = calibration swaption expiries):");
414 try {
415 auto d = getCalibrationDetails(calibrationInfo, swaptionBasket_, parametrization_);
416 WLOGGERSTREAM(d);
417 } catch (const std::exception& e) {
418 WLOG("An error occurred: " << e.what());
419 }
420 WLOGGERSTREAM("Parameter details (with parameter time grid)");
421 WLOGGERSTREAM(getCalibrationDetails(parametrization_));
422 WLOGGERSTREAM("rmse = " << error_);
423 calibrationInfo.valid = true;
424 if (!continueOnError_) {
425 QL_FAIL(exceptionMessage);
426 }
427 }
428 model_->setCalibrationInfo(calibrationInfo);
429
430 DLOG("Apply shift horizon and scale (if not 0.0 and 1.0 respectively)");
431
432 QL_REQUIRE(data_->shiftHorizon() >= 0.0, "shift horizon must be non negative");
433 QL_REQUIRE(data_->scaling() > 0.0, "scaling must be positive");
434
435 if (data_->shiftHorizon() > 0.0) {
436 Real value = -parametrization_->H(data_->shiftHorizon());
437 DLOG("Apply shift horizon " << data_->shiftHorizon() << " (C=" << value << ") to the " << data_->qualifier()
438 << " LGM model");
439 parametrization_->shift() = value;
440 }
441
442 if (data_->scaling() != 1.0) {
443 DLOG("Apply scaling " << data_->scaling() << " to the " << data_->qualifier() << " LGM model");
444 parametrization_->scaling() = data_->scaling();
445 }
446} // performCalculations()

448void LgmBuilder::getExpiryAndTerm(const Size j, Period& expiryPb, Period& termPb, Date& expiryDb, Date& termDb,
449 Real& termT, bool& expiryDateBased, bool& termDateBased) const {
450 std::string expiryString = data_->optionExpiries()[j];
451 std::string termString = data_->optionTerms()[j];
452 parseDateOrPeriod(expiryString, expiryDb, expiryPb, expiryDateBased);
453 parseDateOrPeriod(termString, termDb, termPb, termDateBased);
454 if (termDateBased) {
455 Date tmpExpiry = expiryDateBased ? expiryDb : svts_->optionDateFromTenor(expiryPb);
456 Date tmpStart = swapIndex_->iborIndex()->valueDate(swapIndex_->iborIndex()->fixingCalendar().adjust(tmpExpiry));
457 // ensure that we have a term >= 1 Month, otherwise QL might throw "non-positive swap length (0) given" from
458 // the black swaption engine during calibration helper pricing; also notice that we use the swap length
459 // calculated in the svts (i.e. a length rounded to whole months) to read the volatility from the cube, which is
460 // consistent with what is done in BlackSwaptionEngine (although one might ask whether an interpolated
461 // volatility would be more appropriate)
462 termDb = std::max(termDb, tmpStart + 1 * Months);
463 termT = svts_->swapLength(tmpStart, termDb);
464 } else {
465 termT = svts_->swapLength(termPb);
466 // same as above, make sure the underlying term is at least >= 1 Month, but since Period::operator<
467 // throws in certain circumstances, we do the comparison based on termT here:
468 if (termT < 1.0 / 12.0) {
469 termT = 1.0 / 12.0;
470 termPb = 1 * Months;
471 }
472 }
473}

475Real LgmBuilder::getStrike(const Size j) const {
476 Strike strike = parseStrike(data_->optionStrikes()[j]);
477 Real strikeValue;
478 // TODO: Extend strike type coverage
479 if (strike.type == Strike::Type::ATM)
480 strikeValue = Null<Real>();
481 else if (strike.type == Strike::Type::Absolute)
482 strikeValue = strike.value;
483 else
484 QL_FAIL("strike type ATM or Absolute expected");
485 return strikeValue;
486}

488bool LgmBuilder::volSurfaceChanged(const bool updateCache) const {
489 bool hasUpdated = false;
490
491 // create cache if not equal to required size
492 if (swaptionVolCache_.size() != swaptionBasket_.size())
493 swaptionVolCache_ = vector<Real>(swaptionBasket_.size(), Null<Real>());
494
495 Size swaptionCounter = 0;
496 for (Size j = 0; j < data_->optionExpiries().size(); j++) {
497 if (!swaptionActive_[j])
498 continue;
499
500 Real volCache = swaptionVolCache_.at(swaptionCounter);
501
502 bool expiryDateBased, termDateBased;
503 Period expiryPb, termPb;
504 Date expiryDb, termDb;
505 Real termT;
506
507 getExpiryAndTerm(j, expiryPb, termPb, expiryDb, termDb, termT, expiryDateBased, termDateBased);
508 Real strikeValue = swaptionStrike_.at(swaptionCounter);
509
510 Real vol;
511 if (expiryDateBased && termDateBased) {
512 vol = svts_->volatility(expiryDb, termT, strikeValue);
513 } else if (expiryDateBased && !termDateBased) {
514 vol = svts_->volatility(expiryDb, termPb, strikeValue);
515 } else if (!expiryDateBased && termDateBased) {
516 vol = svts_->volatility(expiryPb, termT, strikeValue);
517 } else {
518 // !expiryDateBased && !termDateBased
519 vol = svts_->volatility(expiryPb, termPb, strikeValue);
520 }
521
522 if (!close_enough(volCache, vol)) {
523 if (updateCache)
524 swaptionVolCache_[swaptionCounter] = vol;
525 hasUpdated = true;
526 }
527 swaptionCounter++;
528 }
529 return hasUpdated;
530}

537void LgmBuilder::buildSwaptionBasket() const {
538
539 DLOG("build swaption basket");
540
541 QL_REQUIRE(data_->optionExpiries().size() == data_->optionTerms().size(), "swaption vector size mismatch");
542 QL_REQUIRE(data_->optionExpiries().size() == data_->optionStrikes().size(), "swaption vector size mismatch");
543
544 std::ostringstream log;
545
546 Handle<YieldTermStructure> yts = market_->discountCurve(currency_, configuration_);
547
548 std::vector<Time> expiryTimes;
549 std::vector<Time> maturityTimes;
550 swaptionBasket_.clear();
551 swaptionBasketVols_.clear();
552 swaptionVolCache_.clear();
553 swaptionStrike_.clear();
554
555 DLOG("build reference date grid '" << referenceCalibrationGrid_ << "'");
556 Date lastRefCalDate = Date::minDate();
557 std::vector<Date> referenceCalibrationDates;
558 if (!referenceCalibrationGrid_.empty())
559 referenceCalibrationDates = DateGrid(referenceCalibrationGrid_).dates();
560
561 for (Size j = 0; j < data_->optionExpiries().size(); j++) {
562 bool expiryDateBased, termDateBased;
563 Period expiryPb, termPb;
564 Date expiryDb, termDb;
565 Real termT = Null<Real>();
566
567 getExpiryAndTerm(j, expiryPb, termPb, expiryDb, termDb, termT, expiryDateBased, termDateBased);
568 Real strikeValue = getStrike(j);
569
570 // rounded to whole years, only used to distinguish between short and long
571 // swap tenors, which in practice always are multiples of whole years
572 Period termTmp = static_cast<Size>(termT + 0.5) * Years;
573 auto iborIndex = termTmp > shortSwapIndex_->tenor() ? swapIndex_->iborIndex() : shortSwapIndex_->iborIndex();
574 auto fixedLegTenor =
575 termTmp > shortSwapIndex_->tenor() ? swapIndex_->fixedLegTenor() : shortSwapIndex_->fixedLegTenor();
576 auto fixedDayCounter =
577 termTmp > shortSwapIndex_->tenor() ? swapIndex_->dayCounter() : shortSwapIndex_->dayCounter();
578 auto floatDayCounter = termTmp > shortSwapIndex_->tenor() ? swapIndex_->iborIndex()->dayCounter()
579 : shortSwapIndex_->iborIndex()->dayCounter();
580 Size settlementDays = Null<Size>();
581 RateAveraging::Type averagingMethod = RateAveraging::Compound;
582 if (auto on = dynamic_pointer_cast<OvernightIndexedSwapIndex>(*swapIndex_)) {
583 settlementDays = on->fixingDays();
584 averagingMethod = on->averagingMethod();
585 }
586
587 Real dummyQuote = svts_->volatilityType() == Normal ? 0.0020 : 0.10;
588 auto volQuote = boost::make_shared<SimpleQuote>(dummyQuote);
589 Handle<Quote> vol = Handle<Quote>(volQuote);
590 boost::shared_ptr<SwaptionHelper> helper;
591 Real updatedStrike;
592
593 if (expiryDateBased && termDateBased) {
594 Real shift = svts_->volatilityType() == ShiftedLognormal ? svts_->shift(expiryDb, termT) : 0.0;
595 std::tie(helper, updatedStrike) = createSwaptionHelper(
596 expiryDb, termDb, svts_, vol, iborIndex, fixedLegTenor, fixedDayCounter, floatDayCounter, yts,
597 calibrationErrorType_, strikeValue, shift, settlementDays, averagingMethod);
598 }
599 if (expiryDateBased && !termDateBased) {
600 Real shift = svts_->volatilityType() == ShiftedLognormal ? svts_->shift(expiryDb, termPb) : 0.0;
601 std::tie(helper, updatedStrike) = createSwaptionHelper(
602 expiryDb, termPb, svts_, vol, iborIndex, fixedLegTenor, fixedDayCounter, floatDayCounter, yts,
603 calibrationErrorType_, strikeValue, shift, settlementDays, averagingMethod);
604 }
605 if (!expiryDateBased && termDateBased) {
606 Date expiry = svts_->optionDateFromTenor(expiryPb);
607 Real shift = svts_->volatilityType() == ShiftedLognormal ? svts_->shift(expiryPb, termT) : 0.0;
608 std::tie(helper, updatedStrike) = createSwaptionHelper(
609 expiry, termDb, svts_, vol, iborIndex, fixedLegTenor, fixedDayCounter, floatDayCounter, yts,
610 calibrationErrorType_, strikeValue, shift, settlementDays, averagingMethod);
611 }
612 if (!expiryDateBased && !termDateBased) {
613 Real shift = svts_->volatilityType() == ShiftedLognormal ? svts_->shift(expiryPb, termPb) : 0.0;
614 std::tie(helper, updatedStrike) = createSwaptionHelper(
615 expiryPb, termPb, svts_, vol, iborIndex, fixedLegTenor, fixedDayCounter, floatDayCounter, yts,
616 calibrationErrorType_, strikeValue, shift, settlementDays, averagingMethod);
617 }
618
619 // check if we want to keep the helper when a reference calibration grid is given
620 Date expiryDate = helper->swaption()->exercise()->date(0);
621 auto refCalDate =
622 std::lower_bound(referenceCalibrationDates.begin(), referenceCalibrationDates.end(), expiryDate);
623 if (refCalDate == referenceCalibrationDates.end() || *refCalDate > lastRefCalDate) {
624 swaptionActive_[j] = true;
625 swaptionBasketVols_.push_back(volQuote);
626 swaptionBasket_.push_back(helper);
627 swaptionStrike_.push_back(updatedStrike);
628 expiryTimes.push_back(yts->timeFromReference(expiryDate));
629 Date matDate = helper->underlyingSwap() ? helper->underlyingSwap()->maturityDate()
630 : helper->underlyingOvernightIndexedSwap()->maturityDate();
631 maturityTimes.push_back(yts->timeFromReference(matDate));
632 if (refCalDate != referenceCalibrationDates.end())
633 lastRefCalDate = *refCalDate;
634 }
635 }
636
637 std::sort(expiryTimes.begin(), expiryTimes.end());
638 auto itExpiryTime = unique(expiryTimes.begin(), expiryTimes.end());
639 expiryTimes.resize(distance(expiryTimes.begin(), itExpiryTime));
640
641 swaptionExpiries_ = Array(expiryTimes.size());
642 for (Size j = 0; j < expiryTimes.size(); j++)
643 swaptionExpiries_[j] = expiryTimes[j];
644
645 std::sort(maturityTimes.begin(), maturityTimes.end());
646 auto itMaturityTime = unique(maturityTimes.begin(), maturityTimes.end());
647 maturityTimes.resize(distance(maturityTimes.begin(), itMaturityTime));
648
649 swaptionMaturities_ = Array(maturityTimes.size());
650 for (Size j = 0; j < maturityTimes.size(); j++)
651 swaptionMaturities_[j] = maturityTimes[j];
652
653 swaptionBasketRefDate_ = calibrationDiscountCurve_->referenceDate();
654}

656std::string LgmBuilder::getBasketDetails(LgmCalibrationInfo& info) const {
657 std::ostringstream log;
658 Handle<YieldTermStructure> yts = market_->discountCurve(currency_, configuration_);
659 log << std::right << std::setw(3) << "#" << std::setw(16) << "expiry" << std::setw(16) << "swapLength"
660 << std::setw(16) << "strike" << std::setw(16) << "atmForward" << std::setw(16) << "annuity" << std::setw(16)
661 << "vega" << std::setw(16) << "vol\n";
662 info.swaptionData.clear();
663 for (Size j = 0; j < swaptionBasket_.size(); ++j) {
664 auto swp = boost::static_pointer_cast<SwaptionHelper>(swaptionBasket_[j])->swaption();
665 auto sd = swaptionData(swp, yts, svts_);
666 log << std::right << std::setw(3) << j << std::setw(16) << sd.timeToExpiry << std::setw(16) << sd.swapLength
667 << std::setw(16) << sd.strike << std::setw(16) << sd.atmForward << std::setw(16) << sd.annuity
668 << std::setw(16) << sd.vega << std::setw(16) << std::setw(16) << sd.stdDev / std::sqrt(sd.timeToExpiry)
669 << "\n";
670 info.swaptionData.push_back(sd);
671 }
672 return log.str();
673}
