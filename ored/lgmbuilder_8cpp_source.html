45 // Return swaption data
46 SwaptionData swaptionData(const boost::shared_ptr<Swaption> swaption, const Handle<YieldTermStructure>& yts,
47  const Handle<SwaptionVolatilityStructure>& svts) {
48 
49  boost::shared_ptr<PricingEngine> engine;
50  switch (svts->volatilityType()) {
51  case ShiftedLognormal:
52  engine = boost::make_shared<BlackSwaptionEngine>(yts, svts);
53  break;
54  case Normal:
55  engine = boost::make_shared<BachelierSwaptionEngine>(yts, svts);
56  break;
57  default:
58  QL_FAIL("Could not construct swaption engine for volatility type: " << svts->volatilityType());
59  break;
60  }
61 
62  swaption->setPricingEngine(engine);
63 
64  SwaptionData sd;
65  sd.timeToExpiry = svts->timeFromReference(swaption->exercise()->dates().back());
66  sd.swapLength = swaption->result<Real>("swapLength");
67  sd.strike = swaption->result<Real>("strike");
68  sd.atmForward = swaption->result<Real>("atmForward");
69  sd.annuity = swaption->result<Real>("annuity");
70  sd.vega = swaption->result<Real>("vega");
71  sd.stdDev = swaption->result<Real>("stdDev");
72 
73  return sd;
74 }

76 // Utility function to create swaption helper.
77 template <typename E, typename T>
78 boost::shared_ptr<SwaptionHelper>
79 createSwaptionHelper(const E& expiry, const T& term, const Handle<SwaptionVolatilityStructure>& svts,
80  const Handle<Quote>& vol, const boost::shared_ptr<IborIndex>& iborIndex,
81  const Period& fixedLegTenor, const DayCounter& fixedDayCounter, const DayCounter& floatDayCounter,
82  const Handle<YieldTermStructure>& yts, BlackCalibrationHelper::CalibrationErrorType errorType,
83  Real strike, Real shift) {
84 
85  auto vt = svts->volatilityType();
86  auto helper = boost::make_shared<SwaptionHelper>(expiry, term, vol, iborIndex, fixedLegTenor, fixedDayCounter,
87  floatDayCounter, yts, errorType, strike, 1.0, vt, shift);
88 
89  // If the helper value is lower than mmv, replace it with a "more reasonable" helper. Here, we replace
90  // the helper with a helper that has the ATM strike. There are other options here.
91  static constexpr Real mmv = 1.0E-20;
92  auto mv = std::abs(helper->marketValue());
93  if (mv < mmv) {
94  auto sd = swaptionData(helper->swaption(), yts, svts);
95  strike = sd.atmForward;
96  helper = boost::make_shared<SwaptionHelper>(expiry, term, vol, iborIndex, fixedLegTenor, fixedDayCounter,
97  floatDayCounter, yts, errorType, strike, 1.0, vt, shift);
98  DLOG("Helper with expiry " << expiry << " and term " << term << " has an absolute market value of " << mv
99  << " which is lower than minimum market value " << mmv
100  << " so switching to helper with atm rate " << strike);
101  }
102 
103  // Switch to PriceError if helper's market value is below 1e-8
104  static constexpr Real smv = 1.0E-8;
105  mv = std::abs(helper->marketValue());
106  if (errorType != BlackCalibrationHelper::PriceError && mv < smv) {
107  helper = boost::make_shared<SwaptionHelper>(expiry, term, vol, iborIndex, fixedLegTenor, fixedDayCounter,
108  floatDayCounter, yts, BlackCalibrationHelper::PriceError, strike,
109  1.0, vt, shift);
110  TLOG("Helper with expiry " << expiry << " and term " << term << " has an absolute market value of " << mv
111  << " which is lower than " << smv << " so switching to a price error helper.");
112  }
113 
114  return helper;
115 }

122 LgmBuilder::LgmBuilder(const boost::shared_ptr<ore::data::Market>& market, const boost::shared_ptr<IrLgmData>& data,
123  const std::string& configuration, const Real bootstrapTolerance, const bool continueOnError,
124  const std::string& referenceCalibrationGrid, const bool setCalibrationInfo)
125  : market_(market), configuration_(configuration), data_(data), bootstrapTolerance_(bootstrapTolerance),
126  continueOnError_(continueOnError), referenceCalibrationGrid_(referenceCalibrationGrid),
127  setCalibrationInfo_(setCalibrationInfo),
128  optimizationMethod_(boost::shared_ptr<OptimizationMethod>(new LevenbergMarquardt(1E-8, 1E-8, 1E-8))),
129  endCriteria_(EndCriteria(1000, 500, 1E-8, 1E-8, 1E-8)),
130  calibrationErrorType_(BlackCalibrationHelper::RelativePriceError) {
131 
132  marketObserver_ = boost::make_shared<MarketObserver>();
133  QuantLib::Currency ccy = parseCurrency(data_->ccy());
134  LOG("LgmCalibration for ccy " << ccy << ", configuration is " << configuration_);
135 
136  requiresCalibration_ =
137  (data_->calibrateA() || data_->calibrateH()) && data_->calibrationType() != CalibrationType::None;
138 
139  // the discount curve underlying the model might be relinked to a different curve outside this builder
140  // the calibration curve should always stay the same though, therefore we create a different handle for this
141  modelDiscountCurve_ = RelinkableHandle<YieldTermStructure>(*market_->discountCurve(ccy.code(), configuration_));
142  calibrationDiscountCurve_ = Handle<YieldTermStructure>(*modelDiscountCurve_);
143 
144  if (requiresCalibration_) {
145  svts_ = market_->swaptionVol(data_->ccy(), configuration_);
146  swapIndex_ = market_->swapIndex(market_->swapIndexBase(data_->ccy(), configuration_), configuration_);
147  shortSwapIndex_ = market_->swapIndex(market_->shortSwapIndexBase(data_->ccy(), configuration_), configuration_);
148  registerWith(svts_);
149  marketObserver_->addObservable(swapIndex_->forwardingTermStructure());
150  marketObserver_->addObservable(swapIndex_->discountingTermStructure());
151  marketObserver_->addObservable(shortSwapIndex_->forwardingTermStructure());
152  marketObserver_->addObservable(shortSwapIndex_->discountingTermStructure());
153  }
154  marketObserver_->addObservable(calibrationDiscountCurve_);
155  registerWith(marketObserver_);
156  // notify observers of all market data changes, not only when not calculated
157  alwaysForwardNotifications();
158 
159  swaptionActive_ = std::vector<bool>(data_->optionExpiries().size(), false);
160 
161  if (requiresCalibration_) {
162  buildSwaptionBasket();
163  }
164 
165  Array aTimes(data_->aTimes().begin(), data_->aTimes().end());
166  Array hTimes(data_->hTimes().begin(), data_->hTimes().end());
167  Array alpha(data_->aValues().begin(), data_->aValues().end());
168  Array h(data_->hValues().begin(), data_->hValues().end());
169 
170  if (data_->aParamType() == ParamType::Constant) {
171  QL_REQUIRE(data_->aTimes().size() == 0,
172  "LgmBuilder: empty volatility time grid expected for constant parameter type");
173  QL_REQUIRE(data_->aValues().size() == 1,
174  "LgmBuilder: initial volatility values should have size 1 for constant parameter type");
175  } else if (data_->aParamType() == ParamType::Piecewise) {
176  if (data_->calibrateA() && data_->calibrationType() == CalibrationType::Bootstrap) {
177  if (data_->aTimes().size() > 0) {
178  DLOG("overriding alpha time grid with swaption expiries, set all initial values to first given value");
179  }
180  QL_REQUIRE(swaptionExpiries_.size() > 0, "empty swaptionExpiries");
181  aTimes = Array(swaptionExpiries_.begin(), swaptionExpiries_.end() - 1);
182  alpha = Array(aTimes.size() + 1, data_->aValues()[0]);
183  } else {
184  QL_REQUIRE(alpha.size() == aTimes.size() + 1,
185  "LgmBuilder: LGM volatility time and initial value array sizes do not match");
186  }
187  } else
188  QL_FAIL("LgmBuilder: volatility parameter type not covered");
189 
190  if (data_->hParamType() == ParamType::Constant) {
191  QL_REQUIRE(data_->hTimes().size() == 0,
192  "LgmBuilder: empty reversion time grid expected for constant parameter type");
193  QL_REQUIRE(data_->hValues().size() == 1,
194  "LgmBuidler: initial reversion values shouuld have size 1 for constant parameter type");
195  } else if (data_->hParamType() == ParamType::Piecewise) {
196  if (data_->calibrateH() && data_->calibrationType() == CalibrationType::Bootstrap) {
197  if (data_->hTimes().size() > 0) {
198  DLOG("overriding h time grid with swaption underlying maturities, set all initial values to first "
199  "given value");
200  }
201  hTimes = swaptionMaturities_;
202  h = Array(hTimes.size() + 1, data_->hValues()[0]);
203  } else { // use input time grid and input h array otherwise
204  QL_REQUIRE(h.size() == hTimes.size() + 1, "H grids do not match");
205  }
206  } else
207  QL_FAIL("LgmBuilder: reversion parameter type case not covered");
208 
209  DLOG("before calibration: alpha times = " << aTimes << " values = " << alpha);
210  DLOG("before calibration: h times = " << hTimes << " values = " << h);
211 
212  if (data_->reversionType() == LgmData::ReversionType::HullWhite &&
213  data_->volatilityType() == LgmData::VolatilityType::HullWhite) {
214  DLOG("IR parametrization for " << ccy << ": IrLgm1fPiecewiseConstantHullWhiteAdaptor");
215  parametrization_ = boost::make_shared<QuantExt::IrLgm1fPiecewiseConstantHullWhiteAdaptor>(
216  ccy, modelDiscountCurve_, aTimes, alpha, hTimes, h);
217  } else if (data_->reversionType() == LgmData::ReversionType::HullWhite &&
218  data_->volatilityType() == LgmData::VolatilityType::Hagan) {
219  DLOG("IR parametrization for " << ccy << ": IrLgm1fPiecewiseConstant");
220  parametrization_ = boost::make_shared<QuantExt::IrLgm1fPiecewiseConstantParametrization>(
221  ccy, modelDiscountCurve_, aTimes, alpha, hTimes, h);
222  } else if (data_->reversionType() == LgmData::ReversionType::Hagan &&
223  data_->volatilityType() == LgmData::VolatilityType::Hagan) {
224  parametrization_ = boost::make_shared<QuantExt::IrLgm1fPiecewiseLinearParametrization>(
225  ccy, modelDiscountCurve_, aTimes, alpha, hTimes, h);
226  DLOG("IR parametrization for " << ccy << ": IrLgm1fPiecewiseLinear");
227  } else {
228  QL_FAIL("LgmBuilder: Reversion type Hagan and volatility type HullWhite not covered");
229  }
230  DLOG("alpha times size: " << aTimes.size());
231  DLOG("lambda times size: " << hTimes.size());
232 
233  model_ = boost::make_shared<QuantExt::LGM>(parametrization_);
234  params_ = model_->params();
235 }

262 void LgmBuilder::performCalculations() const {
263 
264  DLOG("Recalibrate LGM model for currency " << data_->ccy());
265 
266  if (!requiresRecalibration()) {
267  DLOG("Skipping calibration as nothing has changed");
268  return;
269  }
270 
271  // reset lgm observer's updated flag
272  marketObserver_->hasUpdated(true);
273 
274  if (swaptionBasketRefDate_ != calibrationDiscountCurve_->referenceDate()) {
275  // build swaption basket if required, i.e. if reference date has changed since last build
276  buildSwaptionBasket();
277  volSurfaceChanged(true);
278  updateSwaptionBasketVols();
279  } else {
280  // otherwise just update vols
281  volSurfaceChanged(true);
282  updateSwaptionBasketVols();
283  }
284 
285  for (Size j = 0; j < swaptionBasket_.size(); j++) {
286  auto engine = boost::make_shared<QuantExt::AnalyticLgmSwaptionEngine>(model_, calibrationDiscountCurve_);
287  engine->enableCache(!data_->calibrateH(), !data_->calibrateA());
288  swaptionBasket_[j]->setPricingEngine(engine);
289  // necessary if notifications are disabled (observation mode = Disable)
290  swaptionBasket_[j]->update();
291  }
292 
293  // reset model parameters to ensure identical results on identical market data input
294  model_->setParams(params_);
295  parametrization_->shift() = 0.0;
296  parametrization_->scaling() = 1.0;
297 
298  LgmCalibrationInfo calibrationInfo;
299  error_ = QL_MAX_REAL;
300  try {
301  if (data_->calibrateA() && !data_->calibrateH() && data_->calibrationType() == CalibrationType::Bootstrap) {
302  DLOG("call calibrateVolatilitiesIterative for volatility calibration (bootstrap)");
303  model_->calibrateVolatilitiesIterative(swaptionBasket_, *optimizationMethod_, endCriteria_);
304  } else if (data_->calibrateH() && !data_->calibrateA() &&
305  data_->calibrationType() == CalibrationType::Bootstrap) {
306  DLOG("call calibrateReversionsIterative for reversion calibration (bootstrap)");
307  model_->calibrateVolatilitiesIterative(swaptionBasket_, *optimizationMethod_, endCriteria_);
308  } else {
309  QL_REQUIRE(data_->calibrationType() != CalibrationType::Bootstrap,
310  "LgmBuidler: Calibration type Bootstrap can be used with volatilities and reversions calibrated "
311  "simultaneously. Either choose BestFit oder fix one of these parameters.");
312  if (data_->calibrateA() && !data_->calibrateH()) {
313  DLOG("call calibrateVolatilities for (global) volatility calibration")
314  model_->calibrateVolatilities(swaptionBasket_, *optimizationMethod_, endCriteria_);
315  } else if (data_->calibrateH() && !data_->calibrateA()) {
316  DLOG("call calibrateReversions for (global) reversion calibration")
317  model_->calibrateReversions(swaptionBasket_, *optimizationMethod_, endCriteria_);
318  } else {
319  DLOG("call calibrate for global volatility and reversion calibration");
320  model_->calibrate(swaptionBasket_, *optimizationMethod_, endCriteria_);
321  }
322  }
323  TLOG("LGM " << data_->ccy() << " calibration errors:");
324  error_ = getCalibrationError(swaptionBasket_);
325  } catch (const std::exception& e) {
326  // just log a warning, we check below if we meet the bootstrap tolerance and handle the result there
327  WLOG(StructuredModelErrorMessage("Error during LGM calibration: ", e.what()));
328  }
329  calibrationInfo.rmse = error_;
330  if (fabs(error_) < bootstrapTolerance_ ||
331  (data_->calibrationType() == CalibrationType::BestFit && error_ != QL_MAX_REAL)) {
332  // we check the log level here to avoid unncessary computations
333  if (Log::instance().filter(ORE_DATA) || setCalibrationInfo_) {
334  TLOGGERSTREAM << "Basket details:";
335  try {
336  TLOGGERSTREAM << getBasketDetails(calibrationInfo);
337  } catch (const std::exception& e) {
338  WLOG("An error occured: " << e.what());
339  }
340  TLOGGERSTREAM << "Calibration details (with time grid = calibration swaption expiries):";
341  try {
342  TLOGGERSTREAM << getCalibrationDetails(calibrationInfo, swaptionBasket_, parametrization_);
343  } catch (const std::exception& e) {
344  WLOG("An error occured: " << e.what());
345  }
346  TLOGGERSTREAM << "Parameter details (with parameter time grid)";
347  TLOGGERSTREAM << getCalibrationDetails(parametrization_);
348  TLOGGERSTREAM << "rmse = " << error_;
349  calibrationInfo.valid = true;
350  }
351  } else {
352  std::string exceptionMessage = "LGM (" + data_->ccy() + ") calibration error " + std::to_string(error_) +
353  " exceeds tolerance " + std::to_string(bootstrapTolerance_);
354  WLOG(StructuredModelErrorMessage("Failed to calibrate LGM Model", exceptionMessage));
355  WLOGGERSTREAM << "Basket details:";
356  try {
357  WLOGGERSTREAM << getBasketDetails(calibrationInfo);
358  } catch (const std::exception& e) {
359  WLOG("An error occured: " << e.what());
360  }
361  WLOGGERSTREAM << "Calibration details (with time grid = calibration swaption expiries):";
362  try {
363  WLOGGERSTREAM << getCalibrationDetails(calibrationInfo, swaptionBasket_, parametrization_);
364  } catch (const std::exception& e) {
365  WLOG("An error occured: " << e.what());
366  }
367  WLOGGERSTREAM << "Parameter details (with parameter time grid)";
368  WLOGGERSTREAM << getCalibrationDetails(parametrization_);
369  WLOGGERSTREAM << "rmse = " << error_;
370  calibrationInfo.valid = true;
371  if (!continueOnError_) {
372  QL_FAIL(exceptionMessage);
373  }
374  }
375  model_->setCalibrationInfo(calibrationInfo);
376 
377  DLOG("Apply shift horizon and scale (if not 0.0 and 1.0 respectively)");
378 
379  QL_REQUIRE(data_->shiftHorizon() >= 0.0, "shift horizon must be non negative");
380  QL_REQUIRE(data_->scaling() > 0.0, "scaling must be positive");
381 
382  if (data_->shiftHorizon() > 0.0) {
383  Real value = -parametrization_->H(data_->shiftHorizon());
384  DLOG("Apply shift horizon " << data_->shiftHorizon() << " (C=" << value << ") to the " << data_->ccy()
385  << " LGM model");
386  parametrization_->shift() = value;
387  }
388 
389  if (data_->scaling() != 1.0) {
390  DLOG("Apply scaling " << data_->scaling() << " to the " << data_->ccy() << " LGM model");
391  parametrization_->scaling() = data_->scaling();
392  }
393 } // performCalculations()

395 void LgmBuilder::getExpiryAndTerm(const Size j, Period& expiryPb, Period& termPb, Date& expiryDb, Date& termDb,
396  Real& termT, bool& expiryDateBased, bool& termDateBased) const {
397  std::string expiryString = data_->optionExpiries()[j];
398  std::string termString = data_->optionTerms()[j];
399  parseDateOrPeriod(expiryString, expiryDb, expiryPb, expiryDateBased);
400  parseDateOrPeriod(termString, termDb, termPb, termDateBased);
401  if (termDateBased) {
402  Date tmpExpiry = expiryDateBased ? expiryDb : svts_->optionDateFromTenor(expiryPb);
403  Date tmpStart = swapIndex_->iborIndex()->valueDate(swapIndex_->iborIndex()->fixingCalendar().adjust(tmpExpiry));
404  // ensure that we have a term >= 1 Month, otherwise QL might throw "non-positive swap length (0) given" from
405  // the black swaption engine during calibration helper pricing; also notice that we use the swap legnth
406  // calculated in the svts (i.e. a length rounded to whole months) to read the volatility from the cube, which is
407  // consistent with what is done in BlackSwaptionEngine (although one might ask whether an interpolated
408  // volatility would be more appropriate)
409  termDb = std::max(termDb, tmpStart + 1 * Months);
410  termT = svts_->swapLength(tmpStart, termDb);
411  } else {
412  termT = svts_->swapLength(termPb);
413  // same as above, make sure the underlying term is at least >= 1 Month, but since Period::operator<
414  // throws in certain circumstances, we do the comparison based on termT here:
415  if (termT < 1.0 / 12.0) {
416  termT = 1.0 / 12.0;
417  termPb = 1 * Months;
418  }
419  }
420 }

435 bool LgmBuilder::volSurfaceChanged(const bool updateCache) const {
436  bool hasUpdated = false;
437 
438  // create cache if not equal to required size
439  if (swaptionVolCache_.size() != swaptionBasket_.size())
440  swaptionVolCache_ = vector<Real>(swaptionBasket_.size(), Null<Real>());
441 
442  Size swaptionCounter = 0;
443  for (Size j = 0; j < data_->optionExpiries().size(); j++) {
444  if (!swaptionActive_[j])
445  continue;
446 
447  Real volCache = swaptionVolCache_.at(swaptionCounter);
448 
449  bool expiryDateBased, termDateBased;
450  Period expiryPb, termPb;
451  Date expiryDb, termDb;
452  Real termT;
453 
454  getExpiryAndTerm(j, expiryPb, termPb, expiryDb, termDb, termT, expiryDateBased, termDateBased);
455  Real strikeValue = getStrike(j);
456 
457  Real vol;
458  if (expiryDateBased && termDateBased) {
459  vol = svts_->volatility(expiryDb, termT, strikeValue);
460  } else if (expiryDateBased && !termDateBased) {
461  vol = svts_->volatility(expiryDb, termPb, strikeValue);
462  } else if (!expiryDateBased && termDateBased) {
463  vol = svts_->volatility(expiryPb, termT, strikeValue);
464  } else {
465  // !expiryDateBased && !termDateBased
466  vol = svts_->volatility(expiryPb, termPb, strikeValue);
467  }
468 
469  if (!close_enough(volCache, vol)) {
470  if (updateCache)
471  swaptionVolCache_[swaptionCounter] = vol;
472  hasUpdated = true;
473  }
474  swaptionCounter++;
475  }
476  return hasUpdated;
477 }

484 void LgmBuilder::buildSwaptionBasket() const {
485 
486  DLOG("build swaption basket");
487 
488  QL_REQUIRE(data_->optionExpiries().size() == data_->optionTerms().size(), "swaption vector size mismatch");
489  QL_REQUIRE(data_->optionExpiries().size() == data_->optionStrikes().size(), "swaption vector size mismatch");
490 
491  std::ostringstream log;
492 
493  Handle<YieldTermStructure> yts = market_->discountCurve(data_->ccy(), configuration_);
494 
495  std::vector<Time> expiryTimes;
496  std::vector<Time> maturityTimes;
497  swaptionBasket_.clear();
498  swaptionBasketVols_.clear();
499  swaptionVolCache_.clear();
500 
501  DLOG("build reference date grid '" << referenceCalibrationGrid_ << "'");
502  Date lastRefCalDate = Date::minDate();
503  std::vector<Date> referenceCalibrationDates;
504  if (!referenceCalibrationGrid_.empty())
505  referenceCalibrationDates = DateGrid(referenceCalibrationGrid_).dates();
506 
507  for (Size j = 0; j < data_->optionExpiries().size(); j++) {
508  bool expiryDateBased, termDateBased;
509  Period expiryPb, termPb;
510  Date expiryDb, termDb;
511  Real termT = Null<Real>();
512 
513  getExpiryAndTerm(j, expiryPb, termPb, expiryDb, termDb, termT, expiryDateBased, termDateBased);
514  Real strikeValue = getStrike(j);
515 
516  // rounded to whole years, only used to distinguish between short and long
517  // swap tenors, which in practice always are multiples of whole years
518  Period termTmp = static_cast<Size>(termT + 0.5) * Years;
519  auto iborIndex = termTmp > shortSwapIndex_->tenor() ? swapIndex_->iborIndex() : shortSwapIndex_->iborIndex();
520  auto fixedLegTenor =
521  termTmp > shortSwapIndex_->tenor() ? swapIndex_->fixedLegTenor() : shortSwapIndex_->fixedLegTenor();
522  auto fixedDayCounter =
523  termTmp > shortSwapIndex_->tenor() ? swapIndex_->dayCounter() : shortSwapIndex_->dayCounter();
524  auto floatDayCounter = termTmp > shortSwapIndex_->tenor() ? swapIndex_->iborIndex()->dayCounter()
525  : shortSwapIndex_->iborIndex()->dayCounter();
526 
527  Real dummyQuote = svts_->volatilityType() == Normal ? 0.0020 : 0.10;
528  auto volQuote = boost::make_shared<SimpleQuote>(dummyQuote);
529  Handle<Quote> vol = Handle<Quote>(volQuote);
530  boost::shared_ptr<SwaptionHelper> helper;
531 
532  if (expiryDateBased && termDateBased) {
533  Real shift = svts_->volatilityType() == ShiftedLognormal ? svts_->shift(expiryDb, termT) : 0.0;
534  helper = createSwaptionHelper(expiryDb, termDb, svts_, vol, iborIndex, fixedLegTenor, fixedDayCounter,
535  floatDayCounter, yts, calibrationErrorType_, strikeValue, shift);
536  }
537  if (expiryDateBased && !termDateBased) {
538  Real shift = svts_->volatilityType() == ShiftedLognormal ? svts_->shift(expiryDb, termPb) : 0.0;
539  helper = createSwaptionHelper(expiryDb, termPb, svts_, vol, iborIndex, fixedLegTenor, fixedDayCounter,
540  floatDayCounter, yts, calibrationErrorType_, strikeValue, shift);
541  }
542  if (!expiryDateBased && termDateBased) {
543  Date expiry = svts_->optionDateFromTenor(expiryPb);
544  Real shift = svts_->volatilityType() == ShiftedLognormal ? svts_->shift(expiryPb, termT) : 0.0;
545  helper = createSwaptionHelper(expiry, termDb, svts_, vol, iborIndex, fixedLegTenor, fixedDayCounter,
546  floatDayCounter, yts, calibrationErrorType_, strikeValue, shift);
547  }
548  if (!expiryDateBased && !termDateBased) {
549  Real shift = svts_->volatilityType() == ShiftedLognormal ? svts_->shift(expiryPb, termPb) : 0.0;
550  helper = createSwaptionHelper(expiryPb, termPb, svts_, vol, iborIndex, fixedLegTenor, fixedDayCounter,
551  floatDayCounter, yts, calibrationErrorType_, strikeValue, shift);
552  }
553 
554  // check if we want to keep the helper when a reference calibration grid is given
555  Date expiryDate = helper->swaption()->exercise()->date(0);
556  auto refCalDate =
557  std::lower_bound(referenceCalibrationDates.begin(), referenceCalibrationDates.end(), expiryDate);
558  if (refCalDate == referenceCalibrationDates.end() || *refCalDate > lastRefCalDate) {
559  swaptionActive_[j] = true;
560  swaptionBasketVols_.push_back(volQuote);
561  swaptionBasket_.push_back(helper);
562  expiryTimes.push_back(yts->timeFromReference(expiryDate));
563  maturityTimes.push_back(yts->timeFromReference(helper->underlyingSwap()->maturityDate()));
564  if (refCalDate != referenceCalibrationDates.end())
565  lastRefCalDate = *refCalDate;
566  }
567  }
568 
569  std::sort(expiryTimes.begin(), expiryTimes.end());
570  auto itExpiryTime = unique(expiryTimes.begin(), expiryTimes.end());
571  expiryTimes.resize(distance(expiryTimes.begin(), itExpiryTime));
572 
573  swaptionExpiries_ = Array(expiryTimes.size());
574  for (Size j = 0; j < expiryTimes.size(); j++)
575  swaptionExpiries_[j] = expiryTimes[j];
576 
577  std::sort(maturityTimes.begin(), maturityTimes.end());
578  auto itMaturityTime = unique(maturityTimes.begin(), maturityTimes.end());
579  maturityTimes.resize(distance(maturityTimes.begin(), itMaturityTime));
580 
581  swaptionMaturities_ = Array(maturityTimes.size());
582  for (Size j = 0; j < maturityTimes.size(); j++)
583  swaptionMaturities_[j] = maturityTimes[j];
584 
585  swaptionBasketRefDate_ = calibrationDiscountCurve_->referenceDate();
586 }
