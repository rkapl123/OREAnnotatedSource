82//! Outputs stringized representation of the severity level to the stream
83template <typename CharT, typename TraitsT>
84inline std::basic_ostream<CharT, TraitsT>& operator<<(std::basic_ostream<CharT, TraitsT>& strm, oreSeverity lvl) {
85 switch (lvl) {
86 case oreSeverity::alert:
87 strm << "ALERT";
88 break;
89 case oreSeverity::critical:
90 strm << "CRITICAL";
91 break;
92 case oreSeverity::error:
93 strm << "ERROR";
94 break;
95 case oreSeverity::warning:
96 strm << "WARNING";
97 break;
98 case oreSeverity::notice:
99 strm << "NOTICE";
100 break;
101 case oreSeverity::debug:
102 strm << "DEBUG";
103 break;
104 case oreSeverity::data:
105 strm << "DATA";
106 break;
107 case oreSeverity::memory:
108 strm << "MEMORY";
109 break;
110 default:
111 strm << "UNKNOWN";
112 }
113
114 return strm;
115}

325//! Global static Log class
326/*!
327 The Global Log class gets registered with individual loggers and receives application log messages.
328 Once a message is received, it is immediately dispatched to each of the registered loggers, the order in which
329 the loggers are called is not guaranteed.
330
331 Logging is done by the calling thread and the LOG call blocks until all the loggers have returned.
332
333 At start up, the Log class has no loggers and so will ignore any LOG() messages until it is configured.
334
335 To configure the Log class to log to a file "/tmp/my_log.txt"
336 <pre>
337 Log::instance().removeAllLoggers();
338 Log::instance().registerLogger(boost::shared_ptr<Logger>(new FileLogger("/tmp/my_log.txt")));
339 </pre>
340
341 To change the Log class to only use a BufferLogger the user must call
342 <pre>
343 Log::instance().removeAllLoggers();
344 Log::instance().registerLogger(boost::shared_ptr<Logger>(new BufferLogger));
345 </pre>
346 and then to retrieve log messages from the buffer and print them to stdout the user must call:
347 <pre>
348 std::cout << "Begin Log Messages:" << std::endl;
349
350 boost::shared_ptr<BufferLogger> bl = boost::dynamic_pointer_cast<BufferLogger>
351 (Log::instance().logger(BufferLogger::name));
352
353 while (bl.hasNext())
354 std::cout << bl.next() << std::endl;
355 std::cout << "End Log Messages." << std::endl;
356 </pre>

526//! LoggerStream class that is a std::ostream replacement that will log each line
527/*! LoggerStream is a simple wrapper around a std::string stream, it has an explicit
528 cast std::ostream& method so it can be used in place of any std::ostream, this
529 can be used with QuantExt methods that take a std::ostream& for logging purposes.
530
531 Once the stream falls out of focus, it's destructor will take the buffered log
532 messages and pass them the main ore::data::Log::instance().
533
534 Note the following
535 - The timestamps for each log message will correspond to when the LoggerStream
536 destructor has been called, this may not correspond to the actual time the event
537 occurred.
538 - The filename and linenumber in the ore::data::Log() have to be explicitly passed to
539 the LoggerStream, as such the log messages will not correspond to any references
540 in QuantExt (or any other library).
541
542 The LoggerStream relies on falling out of scope to trigger the logging, as such they
543 should not be created using the new operator.
544
545 The proper usage is to use the macro LOGGERSTREAM and DLOGGERSTREAM, if a function takes
546 a std::ostream& as a parameter, use the macro instead.
547
548 \code{.cpp}
549 void function(int x, int y, std::ostream& out);
550
551 void main () {
552
553 // call function
554 function (3, 4, LOGGERSTREAM);
555 // All logging will be completed before this line
556 }
557 \endcode
558 */
559class LoggerStream {
560public:
561 LoggerStream(unsigned mask, const char* filename, unsigned lineNo);
562
563 //! destructor - this is when the logging takes place.
564 ~LoggerStream();
565
566 //! cast this LoggerStream as a ostream&
567 operator std::ostream &() { return ss_; }
568
569private:
570 unsigned mask_;
571 const char* filename_;
572 unsigned lineNo_;
573 std::stringstream ss_;
574};
