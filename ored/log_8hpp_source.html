154//! BufferLogger
155/*!
156 This logger stores each log message in an internal buffer, it can then be probed for log messages at a later point.
157 Log messages are always returned in a FIFO order.
158
159 Typical usage to display log messages would be
160 <pre>
161 while (bLogger.hasNext()) {
162 MsgBox("Log Message", bLogger.next());
163 }
164 </pre>
165 \ingroup utilities
166 \see Log
167 */
168class BufferLogger : public Logger {
169public:
170 //! the name "BufferLogger"
171 static const string name;
172 //! Constructor
173 BufferLogger(unsigned minLevel = ORE_DATA) : Logger(name), minLevel_(minLevel) {}
174 //! Destructor
175 virtual ~BufferLogger() {}
176 //! The log callback
177 virtual void log(unsigned, const string&) override;
178
179 //! Checks if Logger has new messages
180 /*!
181 \return True if this BufferLogger has any new log messages
182 */
183 bool hasNext();
184 //! Retrieve new messages
185 /*!
186 Retrieve the next new message from the buffer, this will throw if the buffer is empty.
187 Messages are returned in a FIFO order. Messages are deleted from the buffer once returned.
188 \return The next message
189 */
190 string next();
191
192private:
193 std::queue<string> buffer_;
194 unsigned minLevel_;
195};

197//! Global static Log class
198/*!
199 The Global Log class gets registered with individual loggers and receives application log messages.
200 Once a message is received, it is immediately dispatched to each of the registered loggers, the order in which
201 the loggers are called is not guaranteed.
202
203 Logging is done by the calling thread and the LOG call blocks until all the loggers have returned.
204
205 At start up, the Log class has no loggers and so will ignore any LOG() messages until it is configured.
206
207 To configure the Log class to log to a file "/tmp/my_log.txt"
208 <pre>
209 Log::instance().removeAllLoggers();
210 Log::instance().registerLogger(boost::shared_ptr<Logger>(new FileLogger("/tmp/my_log.txt")));
211 </pre>
212
213 To change the Log class to only use a BufferLogger the user must call
214 <pre>
215 Log::instance().removeAllLoggers();
216 Log::instance().registerLogger(boost::shared_ptr<Logger>(new BufferLogger));
217 </pre>
218 and then to retrieve log messages from the buffer and print them to stdout the user must call:
219 <pre>
220 std::cout << "Begin Log Messages:" << std::endl;
221
222 boost::shared_ptr<BufferLogger> bl = boost::dynamic_pointer_cast<BufferLogger>
223 (Log::instance().logger(BufferLogger::name));
224
225 while (bl.hasNext())
226 std::cout << bl.next() << std::endl;
227 std::cout << "End Log Messages." << std::endl;
228 </pre>
229 \ingroup utilities
230 */

366//! LoggerStream class that is a std::ostream replacement that will log each line
367/*! LoggerStream is a simple wrapper around a string stream, it has an explicit
368 cast std::ostream& method so it can be used in place of any std::ostream, this
369 can be used with QuantExt methods that take a std::ostream& for logging purposes.
370
371 Once the stream falls out of focus, it's destructor will take the buffered log
372 messages and pass them the main ore::data::Log::instance().
373
374 Note the following
375 - The timestamps for each log message will correspond to when the LoggerStream
376 destructor has been called, this may not correspond to the actual time the event
377 occurred.
378 - The filename and linenumber in the ore::data::Log() have to be explicitly passed to
379 the LoggerStream, as such the log messages will not correspond to any references
380 in QuantExt (or any other library).
381
382 The LoggerStream relies on falling out of scope to trigger the logging, as such they
383 should not be created using the new operator.
384
385 The proper usage is to use the macro LOGGERSTREAM and DLOGGERSTREAM, if a function takes
386 a std::ostream& as a parameter, use the macro instead.
387
388 \code{.cpp}
389 void function(int x, int y, std::ostream& out);
390
391 void main () {
392
393 // call function
394 function (3, 4, LOGGERSTREAM);
395 // All logging will be completed before this line
396 }
397 \endcode
398 */
399class LoggerStream {
400public:
401 LoggerStream(unsigned mask, const char* filename, unsigned lineNo);
402
403 //! destructor - this is when the logging takes place.
404 ~LoggerStream();
405
406 //! cast this LoggerStream as a ostream&
407 operator std::ostream &() { return ss_; }
408
409private:
410 unsigned mask_;
411 const char* filename_;
412 unsigned lineNo_;
413 std::stringstream ss_;
414};
