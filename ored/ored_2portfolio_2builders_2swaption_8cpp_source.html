62boost::shared_ptr<PricingEngine> EuropeanSwaptionEngineBuilder::engineImpl(const string& key) {
63 boost::shared_ptr<IborIndex> index;
64 string ccyCode = tryParseIborIndex(key, index) ? index->currency().code() : key;
65 Handle<YieldTermStructure> yts = market_->discountCurve(ccyCode, configuration(MarketContext::pricing));
66 Handle<SwaptionVolatilityStructure> svts = market_->swaptionVol(key, configuration(MarketContext::pricing));
67 switch (svts->volatilityType()) {
68 case ShiftedLognormal:
69 LOG("Build BlackSwaptionEngine for currency " << ccyCode);
70 return boost::make_shared<BlackSwaptionEngine>(yts, svts);
71 case Normal:
72 LOG("Build BachelierSwaptionEngine for currency " << ccyCode);
73 return boost::make_shared<BachelierSwaptionEngine>(yts, svts);
74 default:
75 QL_FAIL("Swaption volatility type " << svts->volatilityType() << "not covered in EngineFactory");
76 break;
77 }
78}

80boost::shared_ptr<QuantExt::LGM> LGMBermudanSwaptionEngineBuilder::model(const string& id, const string& key,
81 const std::vector<Date>& expiries,
82 const Date& maturity,
83 const std::vector<Real>& strikes) {
84 boost::shared_ptr<IborIndex> index;
85 std::string ccy = tryParseIborIndex(key, index) ? index->currency().code() : key;
86
87 DLOG("Get model data");
88 auto calibration = parseCalibrationType(modelParameter("Calibration"));
89 auto calibrationStrategy = parseCalibrationStrategy(modelParameter("CalibrationStrategy"));
90 std::string referenceCalibrationGrid = modelParameter("ReferenceCalibrationGrid", {}, false, "");
91 Real lambda = parseReal(modelParameter("Reversion", {key, ccy}));
92 vector<Real> sigma = parseListOfValues<Real>(modelParameter("Volatility"), &parseReal);
93 vector<Real> sigmaTimes = parseListOfValues<Real>(modelParameter("VolatilityTimes", {}, false), &parseReal);
94 QL_REQUIRE(sigma.size() == sigmaTimes.size() + 1, "there must be n+1 volatilities (" << sigma.size()
95 << ") for n volatility times ("
96 << sigmaTimes.size() << ")");
97 Real tolerance = parseReal(modelParameter("Tolerance"));
98 auto reversionType = parseReversionType(modelParameter("ReversionType"));
99 auto volatilityType = parseVolatilityType(modelParameter("VolatilityType"));
100 bool continueOnCalibrationError = globalParameters_.count("ContinueOnCalibrationError") > 0 &&
101 parseBool(globalParameters_.at("ContinueOnCalibrationError"));
102
103 auto data = boost::make_shared<IrLgmData>();
104
105 // check for allowed calibration / bermudan strategy settings
106 std::vector<std::pair<CalibrationType, CalibrationStrategy>> validCalPairs = {
107 {CalibrationType::None, CalibrationStrategy::None},
108 {CalibrationType::Bootstrap, CalibrationStrategy::CoterminalATM},
109 {CalibrationType::Bootstrap, CalibrationStrategy::CoterminalDealStrike},
110 {CalibrationType::BestFit, CalibrationStrategy::CoterminalATM},
111 {CalibrationType::BestFit, CalibrationStrategy::CoterminalDealStrike}};
112
113 QL_REQUIRE(std::find(validCalPairs.begin(), validCalPairs.end(),
114 std::make_pair(calibration, calibrationStrategy)) != validCalPairs.end(),
115 "Calibration (" << calibration << ") and CalibrationStrategy (" << calibrationStrategy
116 << ") are not allowed in this combination");
117
118 // compute horizon shift
119 Real shiftHorizon = parseReal(modelParameter("ShiftHorizon", {}, false, "0.5"));
120 Date today = Settings::instance().evaluationDate();
121 shiftHorizon = ActualActual(ActualActual::ISDA).yearFraction(today, maturity) * shiftHorizon;
122
123 // Default: no calibration, constant lambda and sigma from engine configuration
124 data->reset();
125 data->qualifier() = key;
126 data->calibrateH() = false;
127 data->hParamType() = ParamType::Constant;
128 data->hValues() = {lambda};
129 data->reversionType() = reversionType;
130 data->calibrateA() = false;
131 data->aParamType() = ParamType::Piecewise;
132 data->aValues() = sigma;
133 data->aTimes() = sigmaTimes;
134 data->volatilityType() = volatilityType;
135 data->calibrationType() = calibration;
136 data->shiftHorizon() = shiftHorizon;
137
138 if (calibrationStrategy == CalibrationStrategy::CoterminalATM ||
139 calibrationStrategy == CalibrationStrategy::CoterminalDealStrike) {
140 DLOG("Build LgmData for co-terminal specification");
141 vector<string> expiryDates, termDates;
142 for (Size i = 0; i < expiries.size(); ++i) {
143 expiryDates.push_back(to_string(expiries[i]));
144 termDates.push_back(to_string(maturity));
145 }
146 data->optionExpiries() = expiryDates;
147 data->optionTerms() = termDates;
148 data->optionStrikes().resize(expiryDates.size(), "ATM");
149 if (calibrationStrategy == CalibrationStrategy::CoterminalDealStrike) {
150 for (Size i = 0; i < expiries.size(); ++i) {
151 if (strikes[i] != Null<Real>())
152 data->optionStrikes()[i] = std::to_string(strikes[i]);
153 }
154 }
155 if (calibration == CalibrationType::Bootstrap) {
156 DLOG("Calibrate piecewise alpha");
157 data->calibrationType() = CalibrationType::Bootstrap;
158 data->calibrateH() = false;
159 data->hParamType() = ParamType::Constant;
160 data->hValues() = {lambda};
161 data->calibrateA() = true;
162 data->aParamType() = ParamType::Piecewise;
163 data->aValues() = {sigma};
164 } else if (calibration == CalibrationType::BestFit) {
165 DLOG("Calibrate constant sigma");
166 data->calibrationType() = CalibrationType::BestFit;
167 data->calibrateH() = false;
168 data->hParamType() = ParamType::Constant;
169 data->hValues() = {lambda};
170 data->calibrateA() = true;
171 data->aParamType() = ParamType::Constant;
172 data->aValues() = {sigma};
173 } else
174 QL_FAIL("choice of calibration type invalid");
175 }
176
177 bool generateAdditionalResults = false;
178 auto p = globalParameters_.find("GenerateAdditionalResults");
179 if (p != globalParameters_.end()) {
180 generateAdditionalResults = parseBool(p->second);
181 }
182
183 // Build and calibrate model
184 DLOG("Build LGM model");
185 boost::shared_ptr<LgmBuilder> calib =
186 boost::make_shared<LgmBuilder>(market_, data, configuration(MarketContext::irCalibration), tolerance,
187 continueOnCalibrationError, referenceCalibrationGrid, generateAdditionalResults);
188
189 // In some cases, we do not want to calibrate the model
190 boost::shared_ptr<QuantExt::LGM> model;
191 if (globalParameters_.count("Calibrate") == 0 || parseBool(globalParameters_.at("Calibrate"))) {
192 DLOG("Calibrate model (configuration " << configuration(MarketContext::irCalibration) << ")");
193 model = calib->model();
194 } else {
195 DLOG("Skip calibration of model based on global parameters");
196 calib->freeze();
197 model = calib->model();
198 calib->unfreeze();
199 }
200 modelBuilders_.insert(std::make_pair(id, calib));
201
202 return model;
203}

205boost::shared_ptr<PricingEngine> LGMGridBermudanSwaptionEngineBuilder::engineImpl(const string& id, const string& key,
206 const std::vector<Date>& expiries,
207 const Date& maturity,
208 const std::vector<Real>& strikes) {
209 DLOG("Building Bermudan Swaption engine for trade " << id);
210
211 boost::shared_ptr<QuantExt::LGM> lgm = model(id, key, expiries, maturity, strikes);
212
213 DLOG("Get engine data");
214 Real sy = parseReal(engineParameter("sy"));
215 Size ny = parseInteger(engineParameter("ny"));
216 Real sx = parseReal(engineParameter("sx"));
217 Size nx = parseInteger(engineParameter("nx"));
218
219 // Build engine
220 DLOG("Build engine (configuration " << configuration(MarketContext::pricing) << ")");
221 boost::shared_ptr<IborIndex> index;
222 std::string ccy = tryParseIborIndex(key, index) ? index->currency().code() : key;
223 return boost::make_shared<QuantExt::NumericLgmMultiLegOptionEngine>(
224 lgm, sy, ny, sx, nx, market_->discountCurve(ccy, configuration(MarketContext::pricing)));
225}

227boost::shared_ptr<PricingEngine> LgmMcBermudanSwaptionEngineBuilder::engineImpl(const string& id, const string& key,
228 const std::vector<Date>& expiries,
229 const Date& maturity,
230 const std::vector<Real>& strikes) {
231 DLOG("Building MC Bermudan Swaption engine for trade " << id);

234
235 // Build engine

244boost::shared_ptr<PricingEngine> LgmAmcBermudanSwaptionEngineBuilder::engineImpl(const string& id, const string& key,
245 const std::vector<Date>& expiries,
246 const Date& maturity,
247 const std::vector<Real>& strikes) {
248 boost::shared_ptr<IborIndex> index;
249 std::string ccy = tryParseIborIndex(key, index) ? index->currency().code() : key;
250 Currency curr = parseCurrency(ccy);
251 DLOG("Building AMC Bermudan Swaption engine for key " << key << ", ccy " << ccy << " (from externally given CAM)");
252
253 QL_REQUIRE(cam_ != nullptr, "LgmCamBermudanSwaptionEngineBuilder::engineImpl: cam is null");
254 Size currIdx = cam_->ccyIndex(curr);
255 auto lgm = cam_->lgm(currIdx);
256 std::vector<Size> modelIndex(1, cam_->pIdx(CrossAssetModel::AssetType::IR, currIdx));
257
258 // Build engine
259 DLOG("Build engine (configuration " << configuration(MarketContext::pricing) << ")");
260 // we assume that the given cam has pricing discount curves attached already
261 Handle<YieldTermStructure> discountCurve;
262 return buildMcEngine([this](const std::string& p) { return this->engineParameter(p); }, lgm, discountCurve,
263 simulationDates_, modelIndex);
264} // LgmCam engineImpl
