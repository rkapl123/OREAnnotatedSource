29 boost::shared_ptr<PricingEngine> EuropeanSwaptionEngineBuilder::engineImpl(const Currency& ccy) {
30  const string& ccyCode = ccy.code();
31  Handle<YieldTermStructure> yts = market_->discountCurve(ccyCode, configuration(MarketContext::pricing));
32  QL_REQUIRE(!yts.empty(), "engineFactory error: yield term structure not found for currency " << ccyCode);
33  Handle<SwaptionVolatilityStructure> svts = market_->swaptionVol(ccyCode, configuration(MarketContext::pricing));
34  QL_REQUIRE(!svts.empty(), "engineFactory error: swaption vol structure not found for currency " << ccyCode);
35 
36  switch (svts->volatilityType()) {
37  case ShiftedLognormal:
38  LOG("Build BlackSwaptionEngine for currency " << ccyCode);
39  return boost::make_shared<BlackSwaptionEngine>(yts, svts);
40  case Normal:
41  LOG("Build BachelierSwaptionEngine for currency " << ccyCode);
42  return boost::make_shared<BachelierSwaptionEngine>(yts, svts);
43  default:
44  QL_FAIL("Swaption volatility type " << svts->volatilityType() << "not covered in EngineFactory");
45  break;
46  }
47 }

49 boost::shared_ptr<QuantExt::LGM> LGMBermudanSwaptionEngineBuilder::model(const string& id, bool isNonStandard,
50  const string& ccy,
51  const std::vector<Date>& expiries,
52  const Date& maturity,
53  const std::vector<Real>& strikes) {
54 
55  DLOG("Get model data");
56  auto calibration = parseCalibrationType(modelParameter("Calibration"));
57  auto calibrationStrategy = parseCalibrationStrategy(modelParameter("CalibrationStrategy"));
58  std::string referenceCalibrationGrid = modelParameter("ReferenceCalibrationGrid", "", false, "");
59  Real lambda = parseReal(modelParameter("Reversion", ccy));
60  vector<Real> sigma = parseListOfValues<Real>(modelParameter("Volatility"), &parseReal);
61  vector<Real> sigmaTimes = parseListOfValues<Real>(modelParameter("VolatilityTimes", "", false), &parseReal);
62  QL_REQUIRE(sigma.size() == sigmaTimes.size() + 1, "there must be n+1 volatilities (" << sigma.size()
63  << ") for n volatility times ("
64  << sigmaTimes.size() << ")");
65  Real tolerance = parseReal(modelParameter("Tolerance"));
66  auto reversionType = parseReversionType(modelParameter("ReversionType"));
67  auto volatilityType = parseVolatilityType(modelParameter("VolatilityType"));
68  bool continueOnCalibrationError = globalParameters_.count("ContinueOnCalibrationError") > 0 &&
69  parseBool(globalParameters_.at("ContinueOnCalibrationError"));
70 
71  auto data = boost::make_shared<IrLgmData>();
72 
73  // check for allowed calibration / bermudan strategy settings
74  std::vector<std::pair<CalibrationType, CalibrationStrategy>> validCalPairs = {
75  {CalibrationType::None, CalibrationStrategy::None},
76  {CalibrationType::Bootstrap, CalibrationStrategy::CoterminalATM},
77  {CalibrationType::Bootstrap, CalibrationStrategy::CoterminalDealStrike},
78  {CalibrationType::BestFit, CalibrationStrategy::CoterminalATM},
79  {CalibrationType::BestFit, CalibrationStrategy::CoterminalDealStrike}};
80 
81  QL_REQUIRE(std::find(validCalPairs.begin(), validCalPairs.end(),
82  std::make_pair(calibration, calibrationStrategy)) != validCalPairs.end(),
83  "Calibration (" << calibration << ") and CalibrationStrategy (" << calibrationStrategy
84  << ") are not allowed in this combination");
85 
86  // compute horizon shift
87  Real shiftHorizon = parseReal(modelParameter("ShiftHorizon", "", false, "0.5"));
88  Date today = Settings::instance().evaluationDate();
89  shiftHorizon = ActualActual().yearFraction(today, maturity) * shiftHorizon;
90 
91  // Default: no calibration, constant lambda and sigma from engine configuration
92  data->reset();
93  data->ccy() = ccy;
94  data->calibrateH() = false;
95  data->hParamType() = ParamType::Constant;
96  data->hValues() = {lambda};
97  data->reversionType() = reversionType;
98  data->calibrateA() = false;
99  data->aParamType() = ParamType::Piecewise;
100  data->aValues() = sigma;
101  data->aTimes() = sigmaTimes;
102  data->volatilityType() = volatilityType;
103  data->calibrationType() = calibration;
104  data->shiftHorizon() = shiftHorizon;
105 
106  if (calibrationStrategy == CalibrationStrategy::CoterminalATM ||
107  calibrationStrategy == CalibrationStrategy::CoterminalDealStrike) {
108  DLOG("Build LgmData for co-terminal specification");
109  vector<string> expiryDates, termDates;
110  for (Size i = 0; i < expiries.size(); ++i) {
111  expiryDates.push_back(to_string(expiries[i]));
112  termDates.push_back(to_string(maturity));
113  }
114  data->optionExpiries() = expiryDates;
115  data->optionTerms() = termDates;
116  data->optionStrikes().resize(expiryDates.size(), "ATM");
117  if (calibrationStrategy == CalibrationStrategy::CoterminalDealStrike) {
118  for (Size i = 0; i < expiries.size(); ++i) {
119  if (strikes[i] != Null<Real>())
120  data->optionStrikes()[i] = std::to_string(strikes[i]);
121  }
122  }
123  if (calibration == CalibrationType::Bootstrap) {
124  DLOG("Calibrate piecewise alpha");
125  data->calibrationType() = CalibrationType::Bootstrap;
126  data->calibrateH() = false;
127  data->hParamType() = ParamType::Constant;
128  data->hValues() = {lambda};
129  data->calibrateA() = true;
130  data->aParamType() = ParamType::Piecewise;
131  data->aValues() = {sigma};
132  } else if (calibration == CalibrationType::BestFit) {
133  DLOG("Calibrate constant sigma");
134  data->calibrationType() = CalibrationType::BestFit;
135  data->calibrateH() = false;
136  data->hParamType() = ParamType::Constant;
137  data->hValues() = {lambda};
138  data->calibrateA() = true;
139  data->aParamType() = ParamType::Constant;
140  data->aValues() = {sigma};
141  } else
142  QL_FAIL("choice of calibration type invalid");
143  }
144 
145  bool generateAdditionalResults = false;
146  auto p = globalParameters_.find("GenerateAdditionalResults");
147  if (p != globalParameters_.end()) {
148  generateAdditionalResults = parseBool(p->second);
149  }
150 
151  // Build and calibrate model
152  DLOG("Build LGM model");
153  boost::shared_ptr<LgmBuilder> calib =
154  boost::make_shared<LgmBuilder>(market_, data, configuration(MarketContext::irCalibration), tolerance,
155  continueOnCalibrationError, referenceCalibrationGrid, generateAdditionalResults);
156 
157  // In some cases, we do not want to calibrate the model
158  boost::shared_ptr<QuantExt::LGM> model;
159  if (globalParameters_.count("Calibrate") == 0 || parseBool(globalParameters_.at("Calibrate"))) {
160  DLOG("Calibrate model (configuration " << configuration(MarketContext::irCalibration) << ")");
161  model = calib->model();
162  } else {
163  DLOG("Skip calibration of model based on global parameters");
164  calib->freeze();
165  model = calib->model();
166  calib->unfreeze();
167  }
168  modelBuilders_.insert(std::make_pair(id, calib));
169 
170  return model;
171 }

173 boost::shared_ptr<PricingEngine> LGMGridBermudanSwaptionEngineBuilder::engineImpl(const string& id, bool isNonStandard,
174  const string& ccy,
175  const std::vector<Date>& expiries,
176  const Date& maturity,
177  const std::vector<Real>& strikes) {
178  DLOG("Building Bermudan Swaption engine for trade " << id);
179 
180  boost::shared_ptr<QuantExt::LGM> lgm = model(id, isNonStandard, ccy, expiries, maturity, strikes);
181 
182  DLOG("Get engine data");
183  Real sy = parseReal(engineParameter("sy"));
184  Size ny = parseInteger(engineParameter("ny"));
185  Real sx = parseReal(engineParameter("sx"));
186  Size nx = parseInteger(engineParameter("nx"));
187 
188  // Build engine
189  DLOG("Build engine (configuration " << configuration(MarketContext::pricing) << ")");
190  Handle<YieldTermStructure> dscCurve = market_->discountCurve(ccy, configuration(MarketContext::pricing));
191  boost::shared_ptr<PricingEngine> p;
192  if (isNonStandard)
193  return boost::make_shared<QuantExt::NumericLgmNonstandardSwaptionEngine>(lgm, sy, ny, sx, nx, dscCurve);
194  else
195  return boost::make_shared<QuantExt::NumericLgmSwaptionEngine>(lgm, sy, ny, sx, nx, dscCurve);
196 }
