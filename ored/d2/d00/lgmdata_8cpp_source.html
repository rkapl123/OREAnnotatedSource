173 void LgmData::fromFile(const std::string& fileName, const std::string& qualifier) {
174  LOG("load model configuration from " << fileName);
175  clear();
176  XMLDocument doc(fileName);
177  XMLNode* root = doc.getFirstNode("Models");
178  Size count = 0;
179  for (XMLNode* child = XMLUtils::getChildNode(root, "LGM"); child; child = XMLUtils::getNextSibling(child, "LGM")) {
180  std::string childCcy = XMLUtils::getAttribute(child, "qualifier");
181  if (qualifier == childCcy) {
182  fromXML(child);
183  count++;
184  break;
185  }
186  }
187  QL_REQUIRE(count == 1, "LGM configuration not found for qualifier '" << qualifier << "'");
188  LOG("load model configuration from " << fileName << " done.");
189 }

217 void LgmData::fromXML(XMLNode* node) {
218  // XMLUtils::checkNode(node, "Models");
219  // XMLNode* modelNode = XMLUtils::getChildNode(node, "LGM");
220 
221  std::string calibTypeString = XMLUtils::getChildValue(node, "CalibrationType", true);
222  calibrationType_ = parseCalibrationType(calibTypeString);
223  LOG("LGM calibration type = " << calibTypeString);
224 
225  // Volatility config
226 
227  XMLNode* volNode = XMLUtils::getChildNode(node, "Volatility");
228 
229  calibrateA_ = XMLUtils::getChildValueAsBool(volNode, "Calibrate", true);
230  LOG("LGM Volatility calibrate = " << calibrateA_);
231 
232  std::string volTypeString = XMLUtils::getChildValue(volNode, "VolatilityType", true);
233  volType_ = parseVolatilityType(volTypeString);
234  LOG("LGM Volatility type = " << volTypeString);
235 
236  std::string alphaTypeString = XMLUtils::getChildValue(volNode, "ParamType", true);
237  aType_ = parseParamType(alphaTypeString);
238  LOG("LGM Volatility param type = " << alphaTypeString);
239 
240  aTimes_ = XMLUtils::getChildrenValuesAsDoublesCompact(volNode, "TimeGrid", true);
241  LOG("LGM Volatility time grid size = " << aTimes_.size());
242 
243  aValues_ = XMLUtils::getChildrenValuesAsDoublesCompact(volNode, "InitialValue", true);
244  LOG("LGM Volatility initial values size = " << aValues_.size());
245 
246  // Reversion config
247 
248  XMLNode* revNode = XMLUtils::getChildNode(node, "Reversion");
249 
250  calibrateH_ = XMLUtils::getChildValueAsBool(revNode, "Calibrate", true);
251  LOG("LGM Reversion calibrate = " << calibrateH_);
252 
253  std::string revTypeString = XMLUtils::getChildValue(revNode, "ReversionType", true);
254  revType_ = parseReversionType(revTypeString);
255  LOG("LGM Reversion type = " << revTypeString);
256 
257  std::string hTypeString = XMLUtils::getChildValue(revNode, "ParamType", true);
258  hType_ = parseParamType(hTypeString);
259  LOG("LGM Reversion parameter type = " << hTypeString);
260 
261  hTimes_ = XMLUtils::getChildrenValuesAsDoublesCompact(revNode, "TimeGrid", true);
262  LOG("LGM Reversion time grid size = " << hTimes_.size());
263 
264  hValues_ = XMLUtils::getChildrenValuesAsDoublesCompact(revNode, "InitialValue", true);
265  LOG("LGM Reversion initial values size = " << hValues_.size());
266 
267  // Parameter transformation config
268 
269  XMLNode* tranformNode = XMLUtils::getChildNode(node, "ParameterTransformation");
270  shiftHorizon_ = XMLUtils::getChildValueAsDouble(tranformNode, "ShiftHorizon", true);
271  LOG("LGM shift horizon = " << shiftHorizon_);
272 
273  scaling_ = XMLUtils::getChildValueAsDouble(tranformNode, "Scaling", true);
274  LOG("LGM scaling = " << scaling_);
275 
276  LOG("LgmData done");
277 }

279 XMLNode* LgmData::toXML(XMLDocument& doc) {
280 
281  XMLNode* lgmNode = doc.allocNode("LGM");
282 
283  XMLUtils::addGenericChild(doc, lgmNode, "CalibrationType", calibrationType_);
284 
285  // volatility
286  XMLNode* volatilityNode = XMLUtils::addChild(doc, lgmNode, "Volatility");
287  XMLUtils::addChild(doc, volatilityNode, "Calibrate", calibrateA_);
288 
289  XMLNode* volatilityTypeNode = doc.allocNode("VolatilityType", to_string(volType_));
290  XMLUtils::appendNode(volatilityNode, volatilityTypeNode);
291 
292  XMLUtils::addGenericChild(doc, volatilityNode, "ParamType", aType_);
293  XMLUtils::addGenericChildAsList(doc, volatilityNode, "TimeGrid", aTimes_);
294  XMLUtils::addGenericChildAsList(doc, volatilityNode, "InitialValue", aValues_);
295 
296  // reversion
297  XMLNode* reversionNode = XMLUtils::addChild(doc, lgmNode, "Reversion");
298  XMLUtils::addChild(doc, reversionNode, "Calibrate", calibrateH_);
299 
300  XMLNode* reversionTypeNode = doc.allocNode("ReversionType", to_string(revType_));
301  XMLUtils::appendNode(reversionNode, reversionTypeNode);
302 
303  XMLUtils::addGenericChild(doc, reversionNode, "ParamType", hType_);
304  XMLUtils::addGenericChildAsList(doc, reversionNode, "TimeGrid", hTimes_);
305  XMLUtils::addGenericChildAsList(doc, reversionNode, "InitialValue", hValues_);
306 
307  // parameter transformation
308  XMLNode* parameterTransformationNode = XMLUtils::addChild(doc, lgmNode, "ParameterTransformation");
309  XMLUtils::addChild(doc, parameterTransformationNode, "ShiftHorizon", shiftHorizon_);
310  XMLUtils::addChild(doc, parameterTransformationNode, "Scaling", scaling_);
311 
312  return lgmNode;
313 }
