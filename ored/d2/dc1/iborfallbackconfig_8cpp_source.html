48 bool IborFallbackConfig::isIndexReplaced(const string& iborIndex, const Date& asof) const {
49  if (!enableIborFallbacks())
50  return false;
51  auto i = fallbacks_.find(iborIndex);
52  return i != fallbacks_.end() && asof >= i->second.switchDate;
53 }

55 const IborFallbackConfig::FallbackData& IborFallbackConfig::fallbackData(const string& iborIndex) const {
56  auto i = fallbacks_.find(iborIndex);
57  QL_REQUIRE(
58  i != fallbacks_.end(),
59  "No fallback data found for ibor index '"
60  << iborIndex
61  << "', client code should check whether an index is replaced with IsIndexReplaced() before querying data.");
62  return i->second;
63 }

65 void IborFallbackConfig::fromXML(XMLNode* node) {
66  clear();
67  XMLUtils::checkNode(node, "IborFallbackConfig");
68  if (auto global = XMLUtils::getChildNode(node, "GlobalSettings")) {
69  enableIborFallbacks_ = XMLUtils::getChildValueAsBool(global, "EnableIborFallbacks", true);
70  useRfrCurveInTodaysMarket_ = XMLUtils::getChildValueAsBool(global, "UseRfrCurveInTodaysMarket", true);
71  useRfrCurveInSimulationMarket_ = XMLUtils::getChildValueAsBool(global, "UseRfrCurveInSimulationMarket", true);
72  }
73  if (auto fallbacks = XMLUtils::getChildNode(node, "Fallbacks")) {
74  for (auto const repl : XMLUtils::getChildrenNodes(fallbacks, "Fallback")) {
75  XMLUtils::checkNode(repl, "Fallback");
76  string ibor = XMLUtils::getChildValue(repl, "IborIndex", true);
77  fallbacks_[ibor].rfrIndex = XMLUtils::getChildValue(repl, "RfrIndex", true);
78  fallbacks_[ibor].spread = parseReal(XMLUtils::getChildValue(repl, "Spread", true));
79  fallbacks_[ibor].switchDate = parseDate(XMLUtils::getChildValue(repl, "SwitchDate", true));
80  }
81  }
82 }

84 XMLNode* IborFallbackConfig::toXML(XMLDocument& doc) {
85  XMLNode* node = doc.allocNode("IborFallbackConfig");
86  XMLNode* global = XMLUtils::addChild(doc, node, "GlobalSettings");
87  XMLUtils::addChild(doc, global, "EnableIborFallbacks", enableIborFallbacks_);
88  XMLUtils::addChild(doc, global, "UseRfrCurveInTodaysMarket", useRfrCurveInTodaysMarket_);
89  XMLUtils::addChild(doc, global, "UseRfrCurveInSimulationMarket", useRfrCurveInSimulationMarket_);
90  XMLNode* repl = XMLUtils::addChild(doc, node, "Fallbacks");
91  for (auto const& r : fallbacks_) {
92  XMLNode* tmp = XMLUtils::addChild(doc, repl, "Fallback");
93  XMLUtils::addChild(doc, tmp, "IborIndex", r.first);
94  XMLUtils::addChild(doc, tmp, "RfrIndex", r.second.rfrIndex);
95  XMLUtils::addChild(doc, tmp, "Spread", r.second.spread);
96  XMLUtils::addChild(doc, tmp, "SwitchDate", ore::data::to_string(r.second.switchDate));
97  }
98  return node;
99 }
