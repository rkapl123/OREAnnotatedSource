65 void CDSVolatilityCurveConfig::fromXML(XMLNode* node) {
66 
67  XMLUtils::checkNode(node, "CDSVolatility");
68 
69  curveID_ = XMLUtils::getChildValue(node, "CurveId", true);
70  curveDescription_ = XMLUtils::getChildValue(node, "CurveDescription", true);
71 
72  XMLNode* n;
73  if ((n = XMLUtils::getChildNode(node, "ParseTerm")))
74  parseTerm_ = parseBool(XMLUtils::getNodeValue(n));
75 
76  populateNameTerm();
77 
78  quoteName_ = "";
79  if ((n = XMLUtils::getChildNode(node, "QuoteName")))
80  quoteName_ = XMLUtils::getNodeValue(n);
81 
82  if (XMLUtils::getChildNode(node, "Expiries")) {
83 
84  // Giving just an Expiries node is allowed to be backwards compatible but is discouraged.
85  WLOG("Using an Expiries node only in CDSVolatilityCurveConfig is deprecated. "
86  << "A volatility configuration node should be used instead.");
87 
88  // Get the expiries
89  vector<string> quotes = XMLUtils::getChildrenValuesAsStrings(node, "Expiries", true);
90  QL_REQUIRE(quotes.size() > 0, "Need at least one expiry in the Expiries node.");
91 
92  // Build the quotes by appending the expiries to the quote stem.
93  string stem = quoteStem();
94  for (string& q : quotes) {
95  q = stem + q;
96  }
97 
98  // Create the relevant volatilityConfig_ object.
99  if (quotes.size() == 1) {
100  volatilityConfig_ = boost::make_shared<ConstantVolatilityConfig>(quotes[0]);
101  } else {
102  volatilityConfig_ = boost::make_shared<VolatilityCurveConfig>(quotes, "Linear", "Flat");
103  }
104 
105  } else {
106  XMLNode* n;
107  if ((n = XMLUtils::getChildNode(node, "Constant"))) {
108  volatilityConfig_ = boost::make_shared<ConstantVolatilityConfig>();
109  } else if ((n = XMLUtils::getChildNode(node, "Curve"))) {
110  volatilityConfig_ = boost::make_shared<VolatilityCurveConfig>();
111  } else if ((n = XMLUtils::getChildNode(node, "StrikeSurface"))) {
112  volatilityConfig_ = boost::make_shared<VolatilityStrikeSurfaceConfig>();
113  } else if ((n = XMLUtils::getChildNode(node, "DeltaSurface"))) {
114  QL_FAIL("CDSVolatilityCurveConfig does not yet support a DeltaSurface.");
115  } else if ((n = XMLUtils::getChildNode(node, "MoneynessSurface"))) {
116  QL_FAIL("CDSVolatilityCurveConfig does not yet support a MoneynessSurface.");
117  } else {
118  QL_FAIL("CDSVolatility node expects one child node with name in list: Constant,"
119  << " Curve, StrikeSurface.");
120  }
121  volatilityConfig_->fromXML(n);
122  }
123 
124  dayCounter_ = "A365";
125  if ((n = XMLUtils::getChildNode(node, "DayCounter")))
126  dayCounter_ = XMLUtils::getNodeValue(n);
127 
128  calendar_ = "NullCalendar";
129  if ((n = XMLUtils::getChildNode(node, "Calendar")))
130  calendar_ = XMLUtils::getNodeValue(n);
131 
132  strikeType_ = "";
133  if ((n = XMLUtils::getChildNode(node, "StrikeType")))
134  strikeType_ = XMLUtils::getNodeValue(n);
135 
136  strikeFactor_ = 1.0;
137  if ((n = XMLUtils::getChildNode(node, "StrikeFactor")))
138  strikeFactor_ = parseReal(XMLUtils::getNodeValue(n));
139 
140  populateQuotes();
141 }

143 XMLNode* CDSVolatilityCurveConfig::toXML(XMLDocument& doc) {
144 
145  XMLNode* node = doc.allocNode("CDSVolatility");
146 
147  XMLUtils::addChild(doc, node, "CurveId", curveID_);
148  XMLUtils::addChild(doc, node, "CurveDescription", curveDescription_);

154  XMLUtils::addChild(doc, node, "Calendar", calendar_);
155  if (!strikeType_.empty())

157  if (!quoteName_.empty())
158  XMLUtils::addChild(doc, node, "QuoteName", quoteName_);
159  XMLUtils::addChild(doc, node, "StrikeFactor", strikeFactor_);
160  XMLUtils::addChild(doc, node, "ParseTerm", parseTerm_);
161 
162  return node;

165 void CDSVolatilityCurveConfig::populateNameTerm() {
166 
167  // If parsing of term from ID has been turned off, just return early.
168  if (!parseTerm_)
169  return;
170 
171  // Attempt to parse "<name>-<tenor>" from curve configuration ID.
172  // If not of this form, nameTerm_ is left as ("", "").
173  nameTerm_ = extractTermFromId(curveID_);
174 }

176 void CDSVolatilityCurveConfig::populateQuotes() {
177 
178  // The quotes depend on the type of volatility structure that has been configured.
179  if (auto vc = boost::dynamic_pointer_cast<ConstantVolatilityConfig>(volatilityConfig_)) {
180  quotes_ = {vc->quote()};
181  } else if (auto vc = boost::dynamic_pointer_cast<VolatilityCurveConfig>(volatilityConfig_)) {
182  quotes_ = vc->quotes();
183  } else if (auto vc = boost::dynamic_pointer_cast<VolatilitySurfaceConfig>(volatilityConfig_)) {
184 
185  // Clear the quotes_ if necessary and populate with surface quotes
186  quotes_.clear();
187 
188  // Build the quotes by appending the expiries and strikes to the quote stem.
189  string stem = quoteStem();
190  for (const pair<string, string>& p : vc->quotes()) {
191  quotes_.push_back(stem + p.first + "/" + p.second);
192  }
193 
194  } else {
195  QL_FAIL("CDSVolatilityCurveConfig expected a constant, curve or surface");
196  }
197 }

199 string CDSVolatilityCurveConfig::quoteStem() const {
200 
201  string stem{"INDEX_CDS_OPTION/RATE_LNVOL/"};
202 
203  if (!quoteName_.empty()) {
204  // If an explicit quote name has been provided use this.
205  stem += quoteName_;
206  } else if (!nameTerm_.first.empty()) {
207  // If we have parsed the curveID_ from <name>-<term> to a (name, term) pair, use name.
208  stem += nameTerm_.first;
209  } else {
210  // Fallback to just using the curveID_.
211  stem += curveID_;
212  }
213  stem += "/";
214 
215  // If we have parsed the curveID_ from <name>-<term> to a (name, term) pair, append term.
216  if (!nameTerm_.second.empty())
217  stem += nameTerm_.second + "/";
218 
219  return stem;
220 }

222 pair<string, string> extractTermFromId(const string& id) {
223 
224  pair<string, string> result;
225 
226  // Attempt to parse suffix "-<tenor_string>" from id.
227  auto pos = id.find_last_of('-');
228  if (pos != string::npos && pos != id.size() - 1 && pos != 0) {
229  Period tmp;
230  string term = id.substr(pos + 1);
231  if (tryParse<Period>(id.substr(pos + 1), tmp, parsePeriod)) {
232  DLOG("Parsed term suffix, " << term << ", from id " << id);
233  result.first = id.substr(0, pos);
234  result.second = term;
235  } else {
236  DLOG("Could not convert suffix, " << term << ", from id " << id << " to term.");
237  }
238  } else {
239  DLOG("Could not parse a term suffix from id " << id);
240  }
241 
242  return result;
243 }
