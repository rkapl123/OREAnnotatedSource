84 void CapFloorVolatilityCurveConfig::fromXML(XMLNode* node) {
85 
86  XMLUtils::checkNode(node, "CapFloorVolatility");
87  curveID_ = XMLUtils::getChildValue(node, "CurveId", true);
88  curveDescription_ = XMLUtils::getChildValue(node, "CurveDescription", true);
89 
90  // Set the volatility type
91  configureVolatilityType(XMLUtils::getChildValue(node, "VolatilityType", true));
92 
93  // Set the extrapolation variables
94  extrapolation_ = XMLUtils::getChildValue(node, "Extrapolation", true);
95  configureExtrapolation(extrapolation_);
96 
97  // The mandatory variables
98  includeAtm_ = XMLUtils::getChildValueAsBool(node, "IncludeAtm", true);
99  calendar_ = parseCalendar(XMLUtils::getChildValue(node, "Calendar", true));
100  dayCounter_ = parseDayCounter(XMLUtils::getChildValue(node, "DayCounter", true));
101  businessDayConvention_ = parseBusinessDayConvention(XMLUtils::getChildValue(node, "BusinessDayConvention", true));
102  iborIndex_ = XMLUtils::getChildValue(node, "IborIndex", true);
103  discountCurve_ = XMLUtils::getChildValue(node, "DiscountCurve", true);
104 
105  // Settlement days (optional)
106  settleDays_ = 0;
107  if (XMLNode* n = XMLUtils::getChildNode(node, "SettlementDays")) {
108  Integer d = parseInteger(XMLUtils::getNodeValue(n));
109  QL_REQUIRE(d >= 0, "SettlementDays (" << d << ") must be non-negative");
110  settleDays_ = static_cast<Natural>(d);
111  }
112 
113  // Variable on which to interpolate (optional)
114  interpolateOn_ = "TermVolatilities";
115  if (XMLNode* n = XMLUtils::getChildNode(node, "InterpolateOn")) {
116  interpolateOn_ = XMLUtils::getNodeValue(n);
117  }
118 
119  // Interpolation in time direction (optional)
120  timeInterpolation_ = "LinearFlat";
121  if (XMLNode* n = XMLUtils::getChildNode(node, "TimeInterpolation")) {
122  timeInterpolation_ = XMLUtils::getNodeValue(n);
123  }
124 
125  // Interpolation in strike direction (optional)
126  strikeInterpolation_ = "LinearFlat";
127  if (XMLNode* n = XMLUtils::getChildNode(node, "StrikeInterpolation")) {
128  strikeInterpolation_ = XMLUtils::getNodeValue(n);
129  }
130 
131  // Tenors and strikes. Optional as we may have an ATM curve and hence only AtmTenors.
132  tenors_ = XMLUtils::getChildrenValuesAsStrings(node, "Tenors", false);
133  strikes_ = XMLUtils::getChildrenValuesAsStrings(node, "Strikes", false);
134 
135  // Optional flag, if set to true some tenor/strike quotes can be omitted
136  optionalQuotes_ = XMLUtils::getChildValueAsBool(node, "OptionalQuotes", false, false);
137 
138  // Interpolation for cap floor term volatilities (optional)
139  interpolationMethod_ = "BicubicSpline";
140  if (XMLNode* n = XMLUtils::getChildNode(node, "InterpolationMethod")) {
141  interpolationMethod_ = XMLUtils::getNodeValue(n);
142  }
143 
144  // Tenors for ATM volatilities.
145  atmTenors_ = XMLUtils::getChildrenValuesAsStrings(node, "AtmTenors", false);
146  QL_REQUIRE(!tenors_.empty() || !atmTenors_.empty(), "Tenors and AtmTenors cannot both be empty");
147  if (includeAtm_ && atmTenors_.empty()) {
148  // swap so tenors_ is empty
149  atmTenors_.swap(tenors_);
150  }
151 
152  // Optional bootstrap configuration
153  if (XMLNode* n = XMLUtils::getChildNode(node, "BootstrapConfig")) {
154  bootstrapConfig_.fromXML(n);
155  }
156 
157  // Set type_
158  configureType();
159 
160  // Check that we have a valid configuration
161  validate();
162 
163  // Populate required curve ids
164  populateRequiredCurveIds();
165 
166  // Populate quotes
167  populateQuotes();
168 }

234 void CapFloorVolatilityCurveConfig::populateQuotes() {
235 
236  // Cap floor quotes are for the form:
237  // CAPFLOOR/(RATE_LNVOL|RATE_NVOL|RATE_SLNVOL)/<CCY>/<TENOR>/<IBOR_TENOR>/<ATM>/<RELATIVE>/<STRIKE>
238  string ccy = currency();
239  string tenor = iborTenor();
240 
241  // Volatility quote stem
242  MarketDatum::QuoteType qType = quoteType();
243  string stem = "CAPFLOOR/" + to_string(qType) + "/" + ccy + "/";
244 
245  // Cap floor matrix quotes. So, ATM flag is false i.e. 0 and RELATIVE flag is false also as strikes are absolute.
246  for (const string& t : tenors_) {
247  for (const string& s : strikes_) {
248  quotes_.push_back(stem + t + "/" + tenor + "/0/0/" + s);
249  }
250  }
251 
252  // ATM quotes. So, ATM flag is true i.e. 1 and RELATIVE flag is true with strike set to 0.
253  if (type_ == Type::Atm || type_ == Type::SurfaceWithAtm) {
254  for (const string& t : atmTenors_) {
255  quotes_.push_back(stem + t + "/" + tenor + "/1/1/0");
256  }
257  }
258 
259  // Cap floor shift quote depends only on the currency and ibor tenor and is of the form:
260  // CAPFLOOR/SHIFT/<CCY>/<IBOR_TERM>
261  if (volatilityType_ == VolatilityType::ShiftedLognormal) {
262  quotes_.push_back("CAPFLOOR/SHIFT/" + ccy + "/" + tenor);
263  }
264 }

283 void CapFloorVolatilityCurveConfig::validate() const {
284  QL_REQUIRE(interpolateOn_ == "TermVolatilities" || interpolateOn_ == "OptionletVolatilities",
285  "InterpolateOn (" << interpolateOn_ << ") must be TermVolatilities or OptionletVolatilities");
286  QL_REQUIRE(validInterps.count(timeInterpolation_) == 1,
287  "TimeInterpolation, " << timeInterpolation_ << ", not recognised");
288  QL_REQUIRE(validInterps.count(strikeInterpolation_) == 1,
289  "StrikeInterpolation, " << strikeInterpolation_ << ", not recognised");
290  QL_REQUIRE(strikeInterpolation_ != "BackwardFlat", "BackwardFlat StrikeInterpolation is not allowed");
291  if (!tenors_.empty()) {
292  QL_REQUIRE(!strikes_.empty(), "Strikes cannot be empty when configuring a surface");
293  }
294 }

296 MarketDatum::QuoteType CapFloorVolatilityCurveConfig::quoteType() const {
297  switch (volatilityType_) {
298  case CapFloorVolatilityCurveConfig::VolatilityType::Lognormal:
299  return MarketDatum::QuoteType::RATE_LNVOL;
300  case CapFloorVolatilityCurveConfig::VolatilityType::ShiftedLognormal:
301  return MarketDatum::QuoteType::RATE_SLNVOL;
302  case CapFloorVolatilityCurveConfig::VolatilityType::Normal:
303  return MarketDatum::QuoteType::RATE_NVOL;
304  default:
305  QL_FAIL("Unknown VolatilityType (" << static_cast<int>(volatilityType_) << ")");
306  }
307 }
