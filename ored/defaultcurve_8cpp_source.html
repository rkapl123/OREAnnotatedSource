58struct QuoteData {
59 QuoteData() : value(Null<Real>()), runningSpread(Null<Real>()) {}
60
61 QuoteData(const Period& t, Real v, string s, string c, string d, Real rs = Null<Real>())
62 : term(t), value(v), seniority(s), ccy(c), docClause(d), runningSpread(rs) {}
63
64 Period term;
65 Real value;
66 string seniority;
67 string ccy;
68 string docClause;
69 Real runningSpread;
70};

72bool operator<(const QuoteData& lhs, const QuoteData& rhs) { return lhs.term < rhs.term; }

74void addQuote(set<QuoteData>& quotes, const string& configId, const string& name, const Period& tenor, Real value,
75 string seniority, string ccy, string docClause, Real runningSpread = Null<Real>()) {
76
77 // Add to quotes, with a check that we have no duplicate tenors
78 auto r = quotes.insert(QuoteData(tenor, value, seniority, ccy, docClause, runningSpread));
79 QL_REQUIRE(r.second, "duplicate term in quotes found (" << tenor << ") while loading default curve " << configId);
80 TLOG("Loaded quote " << name << " for default curve " << configId);
81}

83set<QuoteData> getRegexQuotes(const Wildcard& wc, const string& configId, DefaultCurveConfig::Config::Type type,
84 const Date& asof, const Loader& loader) {
85
86 using DCCT = DefaultCurveConfig::Config::Type;
87 using MDQT = MarketDatum::QuoteType;
88 using MDIT = MarketDatum::InstrumentType;
89 LOG("Loading regex quotes for default curve " << configId);
90
91 // Loop over the available market data and pick out quotes that match the expression
92 set<QuoteData> result;
93
94 std::ostringstream ss1;
95 ss1 << MDIT::CDS << "/*";
96 Wildcard w1(ss1.str());
97 auto data1 = loader.get(w1, asof);
98
99 std::ostringstream ss2;
100 ss2 << MDIT::HAZARD_RATE << "/*";
101 Wildcard w2(ss2.str());
102 auto data2 = loader.get(w2, asof);
103
104 data1.merge(data2);
105
106 for (const auto& md : data1) {
107
108 QL_REQUIRE(md->asofDate() == asof, "MarketDatum asofDate '" << md->asofDate() << "' <> asof '" << asof << "'");
109
110 auto mdit = md->instrumentType();
111 auto mdqt = md->quoteType();
112
113 // If we have a CDS spread or hazard rate quote, check it and populate tenor and value if it matches
114 if (type == DCCT::SpreadCDS && mdit == MDIT::CDS &&
115 (mdqt == MDQT::CREDIT_SPREAD || mdqt == MDQT::CONV_CREDIT_SPREAD)) {
116
117 auto q = boost::dynamic_pointer_cast<CdsQuote>(md);
118 QL_REQUIRE(q, "Internal error: could not downcast MarketDatum '" << md->name() << "' to CdsQuote");
119 if (wc.matches(q->name())) {
120 addQuote(result, configId, q->name(), q->term(), q->quote()->value(), q->seniority(), q->ccy(),
121 q->docClause(), q->runningSpread());
122 }
123
124 } else if (type == DCCT::Price && (mdit == MDIT::CDS && mdqt == MDQT::PRICE)) {
125
126 auto q = boost::dynamic_pointer_cast<CdsQuote>(md);
127 QL_REQUIRE(q, "Internal error: could not downcast MarketDatum '" << md->name() << "' to CdsQuote");
128 if (wc.matches(q->name())) {
129 addQuote(result, configId, q->name(), q->term(), q->quote()->value(), q->seniority(), q->ccy(),
130 q->docClause(), q->runningSpread());
131 }
132
133 } else if (type == DCCT::HazardRate && (mdit == MDIT::HAZARD_RATE && mdqt == MDQT::RATE)) {
134
135 auto q = boost::dynamic_pointer_cast<HazardRateQuote>(md);
136 QL_REQUIRE(q, "Internal error: could not downcast MarketDatum '" << md->name() << "' to HazardRateQuote");
137 if (wc.matches(q->name())) {
138 addQuote(result, configId, q->name(), q->term(), q->quote()->value(), q->seniority(), q->ccy(),
139 q->docClause());
140 }
141 }
142 }
143
144 // check whether the set of quotes contains more than one seniority, ccy or doc clause,
145 // which probably means that the wildcard has pulled in too many quotes
146
147 std::set<string> seniorities, ccys, docClauses;
148 for (auto const& q : result) {
149 if (!q.seniority.empty())
150 seniorities.insert(q.seniority);
151 if (!q.ccy.empty())
152 ccys.insert(q.ccy);
153 if (!q.docClause.empty())
154 docClauses.insert(q.docClause);
155 }
156
157 QL_REQUIRE(seniorities.size() < 2, "More than one seniority found in wildcard quotes.");
158 QL_REQUIRE(ccys.size() < 2, "More than one seniority found in wildcard quotes.");
159 QL_REQUIRE(docClauses.size() < 2, "More than one seniority found in wildcard quotes.");
160
161 // We don't check for an empty set of CDS quotes here. We check it later because under some circumstances,
162 // it may be allowable to have no quotes.
163 if (type != DCCT::SpreadCDS && type != DCCT::Price) {
164 QL_REQUIRE(!result.empty(), "No market points found for curve config " << configId);
165 }
166
167 LOG("DefaultCurve " << configId << " loaded and using " << result.size() << " quotes.");
168
169 return result;
170}

172set<QuoteData> getExplicitQuotes(const vector<pair<string, bool>>& quotes, const string& configId,
173 DefaultCurveConfig::Config::Type type, const Date& asof, const Loader& loader) {
174
175 using DCCT = DefaultCurveConfig::Config::Type;
176 LOG("Loading explicit quotes for default curve " << configId);
177
178 set<QuoteData> result;
179 for (const auto& p : quotes) {
180 if (boost::shared_ptr<MarketDatum> md = loader.get(p, asof)) {
181 if (type == DCCT::SpreadCDS || type == DCCT::Price) {
182 auto q = boost::dynamic_pointer_cast<CdsQuote>(md);
183 QL_REQUIRE(q, "Quote " << p.first << " for config " << configId << " should be a CdsQuote");
184 addQuote(result, configId, q->name(), q->term(), q->quote()->value(), q->seniority(), q->ccy(),
185 q->docClause(), q->runningSpread());
186 } else {
187 auto q = boost::dynamic_pointer_cast<HazardRateQuote>(md);
188 QL_REQUIRE(q, "Quote " << p.first << " for config " << configId << " should be a HazardRateQuote");
189 addQuote(result, configId, q->name(), q->term(), q->quote()->value(), q->seniority(), q->ccy(),
190 q->docClause());
191 }
192 }
193 }
194
195 // We don't check for an empty set of CDS quotes here. We check it later because under some circumstances,
196 // it may be allowable to have no quotes.
197 if (type != DCCT::SpreadCDS && type != DCCT::Price) {
198 QL_REQUIRE(!result.empty(), "No market points found for curve config " << configId);
199 }
200
201 LOG("DefaultCurve " << configId << " using " << result.size() << " default quotes of " << quotes.size()
202 << " requested quotes.");
203
204 return result;
205}

207set<QuoteData> getConfiguredQuotes(const std::string& curveID, const DefaultCurveConfig::Config& config,
208 const Date& asof, const Loader& loader) {
209
210 using DCCT = DefaultCurveConfig::Config::Type;
211 auto type = config.type();
212 QL_REQUIRE(type == DCCT::SpreadCDS || type == DCCT::Price || type == DCCT::HazardRate,
213 "getConfiguredQuotes expects a curve type of SpreadCDS, Price or HazardRate.");
214 QL_REQUIRE(!config.cdsQuotes().empty(), "No quotes configured for curve " << curveID);
215
216 // We may have a _single_ regex quote or a list of explicit quotes. Check if we have single regex quote.
217 std::vector<std::string> tmp;
218 std::transform(config.cdsQuotes().begin(), config.cdsQuotes().end(), std::back_inserter(tmp),
219 [](const std::pair<std::string, bool>& p) { return p.first; });
220 auto wildcard = getUniqueWildcard(tmp);
221
222 if (wildcard) {
223 return getRegexQuotes(*wildcard, curveID, config.type(), asof, loader);
224 } else {
225 return getExplicitQuotes(config.cdsQuotes(), curveID, config.type(), asof, loader);
226 }
227}

234DefaultCurve::DefaultCurve(Date asof, DefaultCurveSpec spec, const Loader& loader,
235 const CurveConfigurations& curveConfigs,
236 map<string, boost::shared_ptr<YieldCurve>>& yieldCurves,
237 map<string, boost::shared_ptr<DefaultCurve>>& defaultCurves) {
238 const boost::shared_ptr<DefaultCurveConfig>& configs = curveConfigs.defaultCurveConfig(spec.curveConfigID());
239 bool built = false;
240 std::string errors;
241 for (auto const& config : configs->configs()) {
242 try {
243 recoveryRate_ = Null<Real>();
244 if (!config.second.recoveryRateQuote().empty()) {
245 // handle case where the recovery rate is hardcoded in the curve config
246 if (!tryParseReal(config.second.recoveryRateQuote(), recoveryRate_)) {
247 Wildcard wc(config.second.recoveryRateQuote());
248 if (wc.hasWildcard()) {
249 for (auto const& q : loader.get(wc, asof)) {
250 if (wc.matches(q->name())) {
251 QL_REQUIRE(recoveryRate_ == Null<Real>(),
252 "There is more than one recovery rate matching the pattern '" << wc.pattern()
253 << "'.");
254 recoveryRate_ = q->quote()->value();
255 }
256 }
257 } else {
258 QL_REQUIRE(loader.has(config.second.recoveryRateQuote(), asof),
259 "There is no market data for the requested recovery rate "
260 << config.second.recoveryRateQuote());
261 recoveryRate_ = loader.get(config.second.recoveryRateQuote(), asof)->quote()->value();
262 }
263 }
264 }
265 // Build the default curve of the requested type
266 switch (config.second.type()) {
267 case DefaultCurveConfig::Config::Type::SpreadCDS:
268 case DefaultCurveConfig::Config::Type::Price:
269 buildCdsCurve(configs->curveID(), config.second, asof, spec, loader, yieldCurves);
270 break;
271 case DefaultCurveConfig::Config::Type::HazardRate:
272 buildHazardRateCurve(configs->curveID(), config.second, asof, spec, loader);
273 break;
274 case DefaultCurveConfig::Config::Type::Benchmark:
275 buildBenchmarkCurve(configs->curveID(), config.second, asof, spec, loader, yieldCurves);
276 break;
277 case DefaultCurveConfig::Config::Type::MultiSection:
278 buildMultiSectionCurve(configs->curveID(), config.second, asof, spec, loader, defaultCurves);
279 break;
280 case DefaultCurveConfig::Config::Type::TransitionMatrix:
281 buildTransitionMatrixCurve(configs->curveID(), config.second, asof, spec, loader, defaultCurves);
282 break;
283 case DefaultCurveConfig::Config::Type::Null:
284 buildNullCurve(configs->curveID(), config.second, asof, spec);
285 break;
286 default:
287 QL_FAIL("The DefaultCurveConfig type " << static_cast<int>(config.second.type())
288 << " was not recognised");
289 }
290 built = true;
291 break;
292 } catch (exception& e) {
293 std::ostringstream message;
294 message << "build attempt failed for " << configs->curveID() << " using config with priority "
295 << config.first << ": " << e.what();
296 DLOG(message.str());
297 if (!errors.empty())
298 errors += ", ";
299 errors += message.str();
300 }
301 }
302 QL_REQUIRE(built, "default curve building failed for " << spec.curveConfigID() << ": " << errors);
303}

305void DefaultCurve::buildCdsCurve(const std::string& curveID, const DefaultCurveConfig::Config& config, const Date& asof,
306 const DefaultCurveSpec& spec, const Loader& loader,
307 map<string, boost::shared_ptr<YieldCurve>>& yieldCurves) {
308
309 LOG("Start building default curve of type SpreadCDS for curve " << curveID);
310
311 QL_REQUIRE(config.type() == DefaultCurveConfig::Config::Type::SpreadCDS ||
312 config.type() == DefaultCurveConfig::Config::Type::Price,
313 "DefaultCurve::buildCdsCurve expected a default curve configuration with type SpreadCDS/Price");
314 QL_REQUIRE(recoveryRate_ != Null<Real>(), "DefaultCurve: recovery rate needed to build SpreadCDS curve");
315
316 // Get the CDS curve conventions
317 boost::shared_ptr<Conventions> conventions = InstrumentConventions::instance().conventions();
318 QL_REQUIRE(conventions->has(config.conventionID()), "No conventions found with id " << config.conventionID());
319 boost::shared_ptr<CdsConvention> cdsConv =
320 boost::dynamic_pointer_cast<CdsConvention>(conventions->get(config.conventionID()));
321 QL_REQUIRE(cdsConv, "SpreadCDS curves require CDS convention");
322
323 // Get the discount curve for use in the CDS spread curve bootstrap
324 auto it = yieldCurves.find(config.discountCurveID());
325 QL_REQUIRE(it != yieldCurves.end(), "The discount curve, " << config.discountCurveID()
326 << ", required in the building of the curve, "
327 << spec.name() << ", was not found.");
328 Handle<YieldTermStructure> discountCurve = it->second->handle();
329
330 // Get the CDS spread / price curve quotes
331 set<QuoteData> quotes = getConfiguredQuotes(curveID, config, asof, loader);
332
333 // Set up ref data for the curve, except runningSpread which is set below
334 QuantExt::CreditCurve::RefData refData;
335 refData.indexTerm = config.indexTerm();
336 refData.startDate = config.startDate();
337 refData.tenor = Period(cdsConv->frequency());
338 refData.calendar = cdsConv->calendar();
339 refData.convention = cdsConv->paymentConvention();
340 refData.termConvention = cdsConv->paymentConvention();
341 refData.rule = cdsConv->rule();
342 refData.payConvention = cdsConv->paymentConvention();
343 refData.dayCounter = cdsConv->dayCounter();
344 refData.lastPeriodDayCounter = cdsConv->lastPeriodDayCounter();
345 refData.cashSettlementDays = cdsConv->upfrontSettlementDays();
346
347 // If the configuration instructs us to imply a default from the market data, we do it here.
348 if (config.implyDefaultFromMarket() && *config.implyDefaultFromMarket()) {
349 if (recoveryRate_ != Null<Real>() && quotes.empty()) {
350 // Assume entity is in default, between event determination date and auction date. Build a survival
351 // probability curve with value 0.0 tomorrow to approximate this and allow dependent instruments to price.
352 // Need to use small but positive numbers to avoid downstream issues with log linear survivals e.g. below
353 // and in places like ScenarioSimMarket.
354 vector<Date> dates{asof, asof + 1 * Years, asof + 10 * Years};
355 vector<Real> survivalProbs{1.0, 1e-16, 1e-18};
356 curve_ = boost::make_shared<QuantExt::CreditCurve>(
357 Handle<DefaultProbabilityTermStructure>(
358 boost::make_shared<QuantExt::InterpolatedSurvivalProbabilityCurve<LogLinear>>(
359 dates, survivalProbs, config.dayCounter(), Calendar(), std::vector<Handle<Quote>>(),
360 std::vector<Date>(), LogLinear())),
361 discountCurve, Handle<Quote>(boost::make_shared<SimpleQuote>(recoveryRate_)), refData);
362 curve_->curve()->enableExtrapolation();
363 WLOG("DefaultCurve: recovery rate found but no CDS quotes for "
364 << curveID << " and "
365 << "ImplyDefaultFromMarket is true. Curve built that gives default immediately.");
366 return;
367 }
368 } else {
369 QL_REQUIRE(!quotes.empty(), "No market points found for CDS curve config " << curveID);
370 }
371
372 // Create the CDS instrument helpers, only keep alive helpers
373 vector<boost::shared_ptr<QuantExt::DefaultProbabilityHelper>> helpers;
374 std::map<QuantLib::Date, QuantLib::Period> helperQuoteTerms;
375 Real runningSpread = Null<Real>();
376 QuantExt::CreditDefaultSwap::ProtectionPaymentTime ppt = cdsConv->paysAtDefaultTime()
377 ? QuantExt::CreditDefaultSwap::atDefault
378 : QuantExt::CreditDefaultSwap::atPeriodEnd;
379
380 if (config.type() == DefaultCurveConfig::Config::Type::SpreadCDS) {
381 for (auto quote : quotes) {
382 boost::shared_ptr<SpreadCdsHelper> tmp;
383 try {
384 tmp = boost::make_shared<SpreadCdsHelper>(
385 quote.value, quote.term, cdsConv->settlementDays(), cdsConv->calendar(), cdsConv->frequency(),
386 cdsConv->paymentConvention(), cdsConv->rule(), cdsConv->dayCounter(), recoveryRate_, discountCurve,
387 cdsConv->settlesAccrual(), ppt, config.startDate(), cdsConv->lastPeriodDayCounter());
388
389 } catch (exception& e) {
390 if (quote.term == Period(0, Months)) {
391 WLOG("DefaultCurve:: Cannot add quote of term 0M to CDS curve " << curveID << " for asof date "
392 << asof);
393 } else {
394 QL_FAIL("DefaultCurve:: Failed to add quote of term " << quote.term << " to CDS curve " << curveID
395 << " for asof date " << asof
396 << ", with error: " << e.what());
397 }
398 }
399 if (tmp) {
400 if (tmp->latestDate() > asof) {
401 helpers.push_back(tmp);
402 runningSpread = config.runningSpread();
403 }
404 helperQuoteTerms[tmp->latestDate()] = quote.term;
405 }
406 }
407 } else {
408 for (auto quote : quotes) {
409 // If there is no running spread encoded in the quote, the config must have one.
410 runningSpread = quote.runningSpread;
411 if (runningSpread == Null<Real>()) {
412 QL_REQUIRE(config.runningSpread() != Null<Real>(),
413 "A running spread was not provided in the quote "
414 << "string so it must be provided in the config for CDS upfront curve " << curveID);
415 runningSpread = config.runningSpread();
416 }
417 auto tmp = boost::make_shared<UpfrontCdsHelper>(
418 quote.value, runningSpread, quote.term, cdsConv->settlementDays(), cdsConv->calendar(),
419 cdsConv->frequency(), cdsConv->paymentConvention(), cdsConv->rule(), cdsConv->dayCounter(),
420 recoveryRate_, discountCurve, cdsConv->upfrontSettlementDays(), cdsConv->settlesAccrual(), ppt,
421 config.startDate(), cdsConv->lastPeriodDayCounter());
422 if (tmp->latestDate() > asof) {
423 helpers.push_back(tmp);
424 }
425 helperQuoteTerms[tmp->latestDate()] = quote.term;
426 }
427 }
428
429 QL_REQUIRE(!helpers.empty(), "DefaultCurve: no alive quotes found.");
430
431 refData.runningSpread = runningSpread;
432
433 // Ensure that the helpers are sorted. This is done in IterativeBootstrap, but we need
434 // a sorted instruments vector in the code here as well.
435 std::sort(helpers.begin(), helpers.end(), QuantLib::detail::BootstrapHelperSorter());
436
437 // Get configuration values for bootstrap
438 Real accuracy = config.bootstrapConfig().accuracy();
439 Real globalAccuracy = config.bootstrapConfig().globalAccuracy();
440 bool dontThrow = config.bootstrapConfig().dontThrow();
441 Size maxAttempts = config.bootstrapConfig().maxAttempts();
442 Real maxFactor = config.allowNegativeRates() ? config.bootstrapConfig().maxFactor() : 1.0;
443 Real minFactor = config.bootstrapConfig().minFactor();
444 Size dontThrowSteps = config.bootstrapConfig().dontThrowSteps();
445
446 typedef PiecewiseDefaultCurve<QuantExt::SurvivalProbability, LogLinear, QuantExt::IterativeBootstrap> SpCurve;
447 ATTR_UNUSED typedef SpCurve::traits_type dummy;
448 QuantExt::IterativeBootstrap<SpCurve> btconfig(accuracy, globalAccuracy, dontThrow, maxAttempts, maxFactor,
449 minFactor, dontThrowSteps);
450 boost::shared_ptr<DefaultProbabilityTermStructure> qlCurve;
451
452 if (config.indexTerm() != 0 * Days) {
453
454 // build flat index curve picking a quote with identical term as configured if possible
455 // otherwise build interpolated curve using the existing quotes
456
457 std::vector<Real> helperTermTimes;
458 for (auto const& h : helpers) {
459 helperTermTimes.push_back(QuantExt::periodToTime(helperQuoteTerms[h->latestDate()]));
460 }
461
462 Real t = QuantExt::periodToTime(config.indexTerm());
463 Size helperIndex_m, helperIndex_p;
464 Real alpha;
465 std::tie(helperIndex_m, helperIndex_p, alpha) = QuantExt::interpolationIndices(helperTermTimes, t);
466
467 auto tmp1 = boost::make_shared<SpCurve>(
468 asof, std::vector<boost::shared_ptr<QuantExt::DefaultProbabilityHelper>>{helpers[helperIndex_m]},
469 config.dayCounter(), LogLinear(), btconfig);
470 Date d1 = helpers[helperIndex_m]->pillarDate();
471 Real p1 = tmp1->survivalProbability(d1);
472 auto tmp1i = boost::make_shared<QuantExt::InterpolatedSurvivalProbabilityCurve<LogLinear>>(
473 std::vector<Date>{asof, d1}, std::vector<Real>{1.0, p1}, config.dayCounter(), Calendar(),
474 std::vector<Handle<Quote>>(), std::vector<Date>(), LogLinear(), config.allowNegativeRates());
475
476 if (close_enough(alpha, 1.0)) {
477 qlCurve = tmp1i;
478 } else {
479 auto tmp2 = boost::make_shared<SpCurve>(
480 asof, std::vector<boost::shared_ptr<QuantExt::DefaultProbabilityHelper>>{helpers[helperIndex_p]},
481 config.dayCounter(), LogLinear(), btconfig);
482 Date d2 = helpers[helperIndex_p]->pillarDate();
483 Real p2 = tmp2->survivalProbability(d2);
484 auto tmp2i = boost::make_shared<QuantExt::InterpolatedSurvivalProbabilityCurve<LogLinear>>(
485 std::vector<Date>{asof, d2}, std::vector<Real>{1.0, p2}, config.dayCounter(), Calendar(),
486 std::vector<Handle<Quote>>(), std::vector<Date>(), LogLinear(), config.allowNegativeRates());
487 tmp1i->enableExtrapolation();
488 tmp2i->enableExtrapolation();
489 qlCurve = boost::make_shared<QuantExt::TermInterpolatedDefaultCurve>(
490 Handle<DefaultProbabilityTermStructure>(tmp1i), Handle<DefaultProbabilityTermStructure>(tmp2i), alpha);
491 }
492
493 } else {
494
495 // build single name curve
496
497 boost::shared_ptr<DefaultProbabilityTermStructure> tmp = boost::make_shared<SpCurve>(
498 asof, helpers, config.dayCounter(), LogLinear(),
499 QuantExt::IterativeBootstrap<SpCurve>(accuracy, globalAccuracy, dontThrow, maxAttempts, maxFactor,
500 minFactor, dontThrowSteps));
501
502 // As for yield curves we need to copy the piecewise curve because on eval date changes the relative date
503 // helpers with trigger a bootstrap.
504 vector<Date> dates;
505 vector<Real> survivalProbs;
506 dates.push_back(asof);
507 survivalProbs.push_back(1.0);
508
509 for (Size i = 0; i < helpers.size(); ++i) {
510 if (helpers[i]->latestDate() > asof) {
511 Date pillarDate = helpers[i]->pillarDate();
512 Probability sp = tmp->survivalProbability(pillarDate);
513
514 // In some cases the bootstrapped survival probability at one tenor will be `close` to that at a
515 // previous tenor. Here we don't add that survival probability and date to avoid issues when creating
516 // the InterpolatedSurvivalProbabilityCurve below.
517 if (!survivalProbs.empty() && close(survivalProbs.back(), sp)) {
518 DLOG("Survival probability for curve " << spec.name() << " at date " << io::iso_date(pillarDate)
519 << " is the same as that at previous date "
520 << io::iso_date(dates.back()) << " so skipping it.");
521 continue;
522 }
523
524 dates.push_back(pillarDate);
525 survivalProbs.push_back(sp);
526 TLOG(io::iso_date(pillarDate) << "," << fixed << setprecision(9) << sp);
527 }
528 }
529 if (dates.size() == 1) {
530 // We might have removed points above. To make the interpolation work, we need at least two points though.
531 dates.push_back(dates.back() + 1);
532 survivalProbs.push_back(survivalProbs.back());
533 }
534 qlCurve = boost::make_shared<QuantExt::InterpolatedSurvivalProbabilityCurve<LogLinear>>(
535 dates, survivalProbs, config.dayCounter(), Calendar(), std::vector<Handle<Quote>>(), std::vector<Date>(),
536 LogLinear(), config.allowNegativeRates());
537 }
538
539 if (config.extrapolation()) {
540 qlCurve->enableExtrapolation();
541 DLOG("DefaultCurve: Enabled Extrapolation");
542 }
543
544 curve_ = boost::make_shared<QuantExt::CreditCurve>(Handle<DefaultProbabilityTermStructure>(qlCurve), discountCurve,
545 Handle<Quote>(boost::make_shared<SimpleQuote>(recoveryRate_)),
546 refData);
547
548 LOG("Finished building default curve of type SpreadCDS for curve " << curveID);

551void DefaultCurve::buildHazardRateCurve(const std::string& curveID, const DefaultCurveConfig::Config& config,
552 const Date& asof, const DefaultCurveSpec& spec, const Loader& loader) {
553
554 LOG("Start building default curve of type HazardRate for curve " << curveID);
555
556 QL_REQUIRE(config.type() == DefaultCurveConfig::Config::Type::HazardRate,
557 "DefaultCurve::buildHazardRateCurve expected a default curve configuration with type HazardRate");
558
559 // Get the hazard rate curve conventions
560 boost::shared_ptr<Conventions> conventions = InstrumentConventions::instance().conventions();
561 QL_REQUIRE(conventions->has(config.conventionID()), "No conventions found with id " << config.conventionID());
562 boost::shared_ptr<CdsConvention> cdsConv =
563 boost::dynamic_pointer_cast<CdsConvention>(conventions->get(config.conventionID()));
564 QL_REQUIRE(cdsConv, "HazardRate curves require CDS convention");
565
566 // Get the hazard rate quotes
567 set<QuoteData> quotes = getConfiguredQuotes(curveID, config, asof, loader);
568
569 // Build the hazard rate curve
570 Calendar cal = cdsConv->calendar();
571 vector<Date> dates;
572 vector<Real> quoteValues;
573
574 // If first term is not zero, add asof point
575 if (quotes.begin()->term != 0 * Days) {
576 LOG("DefaultCurve: add asof (" << asof << "), hazard rate " << quotes.begin()->value << ", as not given");
577 dates.push_back(asof);
578 quoteValues.push_back(quotes.begin()->value);
579 }
580
581 for (auto quote : quotes) {
582 dates.push_back(cal.advance(asof, quote.term, Following, false));
583 quoteValues.push_back(quote.value);
584 }
585
586 LOG("DefaultCurve: set up interpolated hazard rate curve");
587 curve_ = boost::make_shared<QuantExt::CreditCurve>(
588 Handle<DefaultProbabilityTermStructure>(boost::make_shared<QuantExt::InterpolatedHazardRateCurve<BackwardFlat>>(
589 dates, quoteValues, config.dayCounter(), BackwardFlat(), config.allowNegativeRates())));
590
591 if (config.extrapolation()) {
592 curve_->curve()->enableExtrapolation();
593 DLOG("DefaultCurve: Enabled Extrapolation");
594 }
595
596 if (recoveryRate_ == Null<Real>()) {
597 LOG("DefaultCurve: setting recovery rate to 0.0 for hazard rate curve, because none is given.");
598 recoveryRate_ = 0.0;
599 }
600
601 // Force bootstrap so that errors are thrown during the build, not later
602 curve_->curve()->survivalProbability(QL_EPSILON);
603
604 LOG("Finished building default curve of type HazardRate for curve " << curveID);
605}

607void DefaultCurve::buildBenchmarkCurve(const std::string& curveID, const DefaultCurveConfig::Config& config,
608 const Date& asof, const DefaultCurveSpec& spec, const Loader& loader,
609 map<string, boost::shared_ptr<YieldCurve>>& yieldCurves) {
610
611 LOG("Start building default curve of type Benchmark for curve " << curveID);
612
613 QL_REQUIRE(config.type() == DefaultCurveConfig::Config::Type::Benchmark,
614 "DefaultCurve::buildBenchmarkCurve expected a default curve configuration with type Benchmark");
615
616 if (recoveryRate_ == Null<Real>())
617 recoveryRate_ = 0.0;
618
619 // Populate benchmark yield curve
620 auto it = yieldCurves.find(config.benchmarkCurveID());
621 QL_REQUIRE(it != yieldCurves.end(), "The benchmark curve, " << config.benchmarkCurveID()
622 << ", required in the building of the curve, "
623 << spec.name() << ", was not found.");
624 boost::shared_ptr<YieldCurve> benchmarkCurve = it->second;
625
626 // Populate source yield curve
627 it = yieldCurves.find(config.sourceCurveID());
628 QL_REQUIRE(it != yieldCurves.end(), "The source curve, " << config.sourceCurveID()
629 << ", required in the building of the curve, "
630 << spec.name() << ", was not found.");
631 boost::shared_ptr<YieldCurve> sourceCurve = it->second;
632
633 // Parameters from the configuration
634 vector<Period> pillars = parseVectorOfValues<Period>(config.pillars(), &parsePeriod);
635 Calendar cal = config.calendar();
636 Size spotLag = config.spotLag();
637
638 // Create the implied survival curve
639 vector<Date> dates;
640 vector<Real> impliedSurvProb;
641 Date spot = cal.advance(asof, spotLag * Days);
642 for (Size i = 0; i < pillars.size(); ++i) {
643 dates.push_back(cal.advance(spot, pillars[i]));
644 Real tmp = dates[i] == asof
645 ? 1.0
646 : sourceCurve->handle()->discount(dates[i]) / benchmarkCurve->handle()->discount(dates[i]);
647 // if a non-zero recovery rate is given, we adjust the implied surv probability according to a market value
648 // recovery model (see the documentation of the benchmark curve in the user guide for more details)
649 impliedSurvProb.push_back(std::pow(tmp, 1.0 / (1.0 - recoveryRate_)));
650 }
651 QL_REQUIRE(dates.size() > 0, "DefaultCurve (Benchmark): no dates given");
652
653 // Insert SP = 1.0 at asof if asof date is not in the pillars
654 if (dates[0] != asof) {
655 dates.insert(dates.begin(), asof);
656 impliedSurvProb.insert(impliedSurvProb.begin(), 1.0);
657 }
658
659 LOG("DefaultCurve: set up interpolated surv prob curve as yield over benchmark");
660 curve_ = boost::make_shared<QuantExt::CreditCurve>(Handle<DefaultProbabilityTermStructure>(
661 boost::make_shared<QuantExt::InterpolatedSurvivalProbabilityCurve<LogLinear>>(
662 dates, impliedSurvProb, config.dayCounter(), Calendar(), std::vector<Handle<Quote>>(), std::vector<Date>(),
663 LogLinear(), config.allowNegativeRates())));
664
665 if (config.extrapolation()) {
666 curve_->curve()->enableExtrapolation();
667 DLOG("DefaultCurve: Enabled Extrapolation");
668 }
669
670 // Force bootstrap so that errors are thrown during the build, not later
671 curve_->curve()->survivalProbability(QL_EPSILON);
672
673 LOG("Finished building default curve of type Benchmark for curve " << curveID);
674}

706void DefaultCurve::buildTransitionMatrixCurve(const std::string& curveID, const DefaultCurveConfig::Config& config,
707 const Date& asof, const DefaultCurveSpec& spec, const Loader& loader,
708 map<string, boost::shared_ptr<DefaultCurve>>& defaultCurves) {
709 LOG("Start building default curve of type TransitionMatrix for curve " << curveID);
710 Size dim = config.states().size();
711 QL_REQUIRE(dim >= 2,
712 "DefaultCurve::buildTransitionMatrixCurve(): transition matrix dimension >= 2 required, found " << dim);
713 Matrix transitionMatrix(dim, dim, Null<Real>());
714 map<string, Size> stateIndex;
715 for (Size i = 0; i < config.states().size(); ++i)
716 stateIndex[config.states()[i]] = i;
717 QL_REQUIRE(!config.cdsQuotes().empty(), "DefaultCurve::buildTransitionMatrixCurve(): not quotes given.");
718 std::vector<std::string> tmp;
719 std::transform(config.cdsQuotes().begin(), config.cdsQuotes().end(), std::back_inserter(tmp),
720 [](const std::pair<std::string, bool>& p) { return p.first; });
721 auto wildcard = getUniqueWildcard(tmp);
722 std::set<boost::shared_ptr<MarketDatum>> mdData;
723 if (wildcard) {
724 mdData = loader.get(*wildcard, asof);
725 } else {
726 for (auto const& q : config.cdsQuotes()) {
727 if (auto m = loader.get(q, asof))
728 mdData.insert(m);
729 }
730 }
731 for (const auto& md : mdData) {
732 QL_REQUIRE(md->instrumentType() == MarketDatum::InstrumentType::RATING,
733 "DefaultCurve::buildTransitionMatrixCurve(): quote instrument type must be RATING.");
734 boost::shared_ptr<TransitionProbabilityQuote> q = boost::dynamic_pointer_cast<TransitionProbabilityQuote>(md);
735 Size i = stateIndex[q->fromRating()];
736 Size j = stateIndex[q->toRating()];
737 transitionMatrix[i][j] = q->quote()->value();
738 }
739 for (Size i = 0; i < dim; ++i) {
740 for (Size j = 0; j < dim; ++j) {
741 QL_REQUIRE(transitionMatrix[i][j] != Null<Real>(),
742 "DefaultCurve::buildTransitionMatrixCurve():matrix element "
743 << config.states()[i] << " -> " << config.states()[j] << " missing in market data");
744 }
745 }
746 Size initialStateIndex = stateIndex[config.initialState()];
747 curve_ = boost::make_shared<QuantExt::CreditCurve>(
748 Handle<DefaultProbabilityTermStructure>(boost::make_shared<QuantExt::GeneratorDefaultProbabilityTermStructure>(
749 QuantExt::GeneratorDefaultProbabilityTermStructure::MatrixType::Transition, transitionMatrix,
750 initialStateIndex, asof)));
751 if (recoveryRate_ == Null<Real>())
752 recoveryRate_ = 0.0;
753 LOG("Finished building default curve of type TransitionMatrix for curve " << curveID);
754}
