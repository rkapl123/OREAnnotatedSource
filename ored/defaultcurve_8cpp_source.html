74  string seniority, string ccy, string docClause, Real runningSpread = Null<Real>()) {

82 set<QuoteData> getRegexQuotes(const Wildcard& wc, const string& configId, DefaultCurveConfig::Config::Type type,
83  const Date& asof, const Loader& loader) {
84 
85  using DCCT = DefaultCurveConfig::Config::Type;
86  using MDQT = MarketDatum::QuoteType;
87  using MDIT = MarketDatum::InstrumentType;
88  LOG("Loading regex quotes for default curve " << configId);
89 
90  // Loop over the available market data and pick out quotes that match the expression
91  set<QuoteData> result;
92  for (const auto& md : loader.loadQuotes(asof)) {
93 
94  // Go to next quote if the market data point's date does not equal our asof
95  if (md->asofDate() != asof)
96  continue;
97 
98  auto mdit = md->instrumentType();
99  auto mdqt = md->quoteType();
100 
101  // If we have a CDS spread or hazard rate quote, check it and populate tenor and value if it matches
102  if (type == DCCT::SpreadCDS && mdit == MDIT::CDS &&
103  (mdqt == MDQT::CREDIT_SPREAD || mdqt == MDQT::CONV_CREDIT_SPREAD)) {
104 
105  auto q = boost::dynamic_pointer_cast<CdsQuote>(md);
106  if (wc.matches(q->name())) {
107  addQuote(result, configId, q->name(), q->term(), q->quote()->value(), q->seniority(), q->ccy(),
108  q->docClause(), q->runningSpread());
109  }
110 
111  } else if (type == DCCT::Price && (mdit == MDIT::CDS && mdqt == MDQT::PRICE)) {
112 
113  auto q = boost::dynamic_pointer_cast<CdsQuote>(md);
114  if (wc.matches(q->name())) {
115  addQuote(result, configId, q->name(), q->term(), q->quote()->value(), q->seniority(), q->ccy(),
116  q->docClause(), q->runningSpread());
117  }
118 
119  } else if (type == DCCT::HazardRate && (mdit == MDIT::HAZARD_RATE && mdqt == MDQT::RATE)) {
120 
121  auto q = boost::dynamic_pointer_cast<HazardRateQuote>(md);
122  if (wc.matches(q->name())) {
123  addQuote(result, configId, q->name(), q->term(), q->quote()->value(), q->seniority(), q->ccy(),
124  q->docClause());
125  }
126  }
127  }
128 
129  // check whether the set of quotes contains more than one seniority, ccy or doc clause,
130  // which probably means that the wildcard has pulled in too many quotes
131 
132  std::set<string> seniorities, ccys, docClauses;
133  for (auto const& q : result) {
134  if (!q.seniority.empty())
135  seniorities.insert(q.seniority);
136  if (!q.ccy.empty())
137  ccys.insert(q.ccy);
138  if (!q.docClause.empty())
139  docClauses.insert(q.docClause);
140  }
141 
142  QL_REQUIRE(seniorities.size() < 2, "More than one seniority found in wildcard quotes.");
143  QL_REQUIRE(ccys.size() < 2, "More than one seniority found in wildcard quotes.");
144  QL_REQUIRE(docClauses.size() < 2, "More than one seniority found in wildcard quotes.");
145 
146  // We don't check for an empty set of CDS quotes here. We check it later because under some circumstances,
147  // it may be allowable to have no quotes.
148  if (type != DCCT::SpreadCDS && type != DCCT::Price) {
149  QL_REQUIRE(!result.empty(), "No market points found for curve config " << configId);
150  }
151 
152  LOG("DefaultCurve " << configId << " loaded and using " << result.size() << " quotes.");
153 
154  return result;
155 }

157 set<QuoteData> getExplicitQuotes(const vector<pair<string, bool>>& quotes, const string& configId,
158  DefaultCurveConfig::Config::Type type, const Date& asof, const Loader& loader) {
159 
160  using DCCT = DefaultCurveConfig::Config::Type;
161  LOG("Loading explicit quotes for default curve " << configId);
162 
163  set<QuoteData> result;
164  for (const auto& p : quotes) {
165  if (boost::shared_ptr<MarketDatum> md = loader.get(p, asof)) {
166  if (type == DCCT::SpreadCDS || type == DCCT::Price) {
167  auto q = boost::dynamic_pointer_cast<CdsQuote>(md);
168  QL_REQUIRE(q, "Quote " << p.first << " for config " << configId << " should be a CdsQuote");
169  addQuote(result, configId, q->name(), q->term(), q->quote()->value(), q->seniority(), q->ccy(),
170  q->docClause(), q->runningSpread());
171  } else {
172  auto q = boost::dynamic_pointer_cast<HazardRateQuote>(md);
173  QL_REQUIRE(q, "Quote " << p.first << " for config " << configId << " should be a HazardRateQuote");
174  addQuote(result, configId, q->name(), q->term(), q->quote()->value(), q->seniority(), q->ccy(),
175  q->docClause());
176  }
177  }
178  }
179 
180  // We don't check for an empty set of CDS quotes here. We check it later because under some circumstances,
181  // it may be allowable to have no quotes.
182  if (type != DCCT::SpreadCDS && type != DCCT::Price) {
183  QL_REQUIRE(!result.empty(), "No market points found for curve config " << configId);
184  }
185 
186  LOG("DefaultCurve " << configId << " using " << result.size() << " default quotes of " << quotes.size()
187  << " requested quotes.");
188 
189  return result;
190 }

192 set<QuoteData> getConfiguredQuotes(const std::string& curveID, const DefaultCurveConfig::Config& config,
193  const Date& asof, const Loader& loader) {
194 
195  using DCCT = DefaultCurveConfig::Config::Type;
196  auto type = config.type();
197  QL_REQUIRE(type == DCCT::SpreadCDS || type == DCCT::Price || type == DCCT::HazardRate,
198  "getConfiguredQuotes expects a curve type of SpreadCDS, Price or HazardRate.");
199  QL_REQUIRE(!config.cdsQuotes().empty(), "No quotes configured for curve " << curveID);
200 
201  // We may have a _single_ regex quote or a list of explicit quotes. Check if we have single regex quote.
202  std::vector<std::string> tmp;
203  std::transform(config.cdsQuotes().begin(), config.cdsQuotes().end(), std::back_inserter(tmp),
204  [](const std::pair<std::string, bool>& p) { return p.first; });
205  auto wildcard = getUniqueWildcard(tmp);
206 
207  if (wildcard) {
208  return getRegexQuotes(*wildcard, curveID, config.type(), asof, loader);
209  } else {
210  return getExplicitQuotes(config.cdsQuotes(), curveID, config.type(), asof, loader);
211  }
212 }

219 DefaultCurve::DefaultCurve(Date asof, DefaultCurveSpec spec, const Loader& loader,
220  const CurveConfigurations& curveConfigs,
221  map<string, boost::shared_ptr<YieldCurve>>& yieldCurves,
222  map<string, boost::shared_ptr<DefaultCurve>>& defaultCurves) {
223  const boost::shared_ptr<DefaultCurveConfig>& configs = curveConfigs.defaultCurveConfig(spec.curveConfigID());
224  bool built = false;
225  std::string errors;
226  for (auto const& config : configs->configs()) {
227  try {
228  recoveryRate_ = Null<Real>();
229  if (!config.second.recoveryRateQuote().empty()) {
230  // handle case where the recovery rate is hardcoded in the curve config
231  if (!tryParseReal(config.second.recoveryRateQuote(), recoveryRate_)) {
232  Wildcard wc(config.second.recoveryRateQuote());
233  if (wc.hasWildcard()) {
234  for (auto const& q : loader.loadQuotes(asof)) {
235  if (wc.matches(q->name())) {
236  QL_REQUIRE(recoveryRate_ == Null<Real>(),
237  "There is more than one recovery rate matching the pattern '" << wc.pattern()
238  << "'.");
239  recoveryRate_ = q->quote()->value();
240  }
241  }
242  } else {
243  QL_REQUIRE(loader.has(config.second.recoveryRateQuote(), asof),
244  "There is no market data for the requested recovery rate "
245  << config.second.recoveryRateQuote());
246  recoveryRate_ = loader.get(config.second.recoveryRateQuote(), asof)->quote()->value();
247  }
248  }
249  }
250  // Build the default curve of the requested type
251  switch (config.second.type()) {
252  case DefaultCurveConfig::Config::Type::SpreadCDS:
253  case DefaultCurveConfig::Config::Type::Price:
254  buildCdsCurve(configs->curveID(), config.second, asof, spec, loader, yieldCurves);
255  break;
256  case DefaultCurveConfig::Config::Type::HazardRate:
257  buildHazardRateCurve(configs->curveID(), config.second, asof, spec, loader);
258  break;
259  case DefaultCurveConfig::Config::Type::Benchmark:
260  buildBenchmarkCurve(configs->curveID(), config.second, asof, spec, loader, yieldCurves);
261  break;
262  case DefaultCurveConfig::Config::Type::MultiSection:
263  buildMultiSectionCurve(configs->curveID(), config.second, asof, spec, loader, defaultCurves);
264  break;
265  case DefaultCurveConfig::Config::Type::Null:
266  buildNullCurve(configs->curveID(), config.second, asof, spec);
267  break;
268  default:
269  QL_FAIL("The DefaultCurveConfig type " << static_cast<int>(config.second.type())
270  << " was not recognised");
271  }
272  built = true;
273  } catch (exception& e) {
274  std::ostringstream message;
275  message << "build attempt failed for " << configs->curveID() << " using config with priority "
276  << config.first << ": " << e.what();
277  DLOG(message.str());
278  if (!errors.empty())
279  errors += ", ";
280  errors += message.str();
281  }
282  }
283  QL_REQUIRE(built, "default curve building failed for " << spec.curveConfigID() << ": " << errors);
284 }

286 void DefaultCurve::buildCdsCurve(const std::string& curveID, const DefaultCurveConfig::Config& config, const Date& asof,
287  const DefaultCurveSpec& spec, const Loader& loader,
288  map<string, boost::shared_ptr<YieldCurve>>& yieldCurves) {
289 
290  LOG("Start building default curve of type SpreadCDS for curve " << curveID);
291 
292  QL_REQUIRE(config.type() == DefaultCurveConfig::Config::Type::SpreadCDS ||
293  config.type() == DefaultCurveConfig::Config::Type::Price,
294  "DefaultCurve::buildCdsCurve expected a default curve configuration with type SpreadCDS/Price");
295  QL_REQUIRE(recoveryRate_ != Null<Real>(), "DefaultCurve: recovery rate needed to build SpreadCDS curve");
296 
297  // Get the CDS curve conventions
298  boost::shared_ptr<Conventions> conventions = InstrumentConventions::instance().conventions();
299  QL_REQUIRE(conventions->has(config.conventionID()), "No conventions found with id " << config.conventionID());
300  boost::shared_ptr<CdsConvention> cdsConv =
301  boost::dynamic_pointer_cast<CdsConvention>(conventions->get(config.conventionID()));
302  QL_REQUIRE(cdsConv, "SpreadCDS curves require CDS convention");
303 
304  // Get the discount curve for use in the CDS spread curve bootstrap
305  auto it = yieldCurves.find(config.discountCurveID());
306  QL_REQUIRE(it != yieldCurves.end(), "The discount curve, " << config.discountCurveID()
307  << ", required in the building of the curve, "
308  << spec.name() << ", was not found.");
309  Handle<YieldTermStructure> discountCurve = it->second->handle();
310 
311  // Get the CDS spread / price curve quotes
312  set<QuoteData> quotes = getConfiguredQuotes(curveID, config, asof, loader);
313 
314  // Set up ref data for the curve, except runningSpread which is set below
315  QuantExt::CreditCurve::RefData refData;
316  refData.indexTerm = config.indexTerm();
317  refData.startDate = config.startDate();
318  refData.tenor = Period(cdsConv->frequency());
319  refData.calendar = cdsConv->calendar();
320  refData.convention = cdsConv->paymentConvention();
321  refData.termConvention = cdsConv->paymentConvention();
322  refData.rule = cdsConv->rule();
323  refData.payConvention = cdsConv->paymentConvention();
324  refData.dayCounter = cdsConv->dayCounter();
325  refData.lastPeriodDayCounter = cdsConv->lastPeriodDayCounter();
326  refData.cashSettlementDays = cdsConv->upfrontSettlementDays();
327 
328  // If the configuration instructs us to imply a default from the market data, we do it here.
329  if (config.implyDefaultFromMarket() && *config.implyDefaultFromMarket()) {
330  if (recoveryRate_ != Null<Real>() && quotes.empty()) {
331  // Assume entity is in default, between event determination date and auction date. Build a survival
332  // probability curve with value 0.0 tomorrow to approximate this and allow dependent instruments to price.
333  // Need to use small but positive numbers to avoid downstream issues with log linear survivals e.g. below
334  // and in places like ScenarioSimMarket.
335  vector<Date> dates{asof, asof + 1 * Years, asof + 10 * Years};
336  vector<Real> survivalProbs{1.0, 1e-16, 1e-18};
337  curve_ = boost::make_shared<QuantExt::CreditCurve>(
338  Handle<DefaultProbabilityTermStructure>(
339  boost::make_shared<QuantExt::InterpolatedSurvivalProbabilityCurve<LogLinear>>(
340  dates, survivalProbs, config.dayCounter(), Calendar(), std::vector<Handle<Quote>>(),
341  std::vector<Date>(), LogLinear())),
342  discountCurve, Handle<Quote>(boost::make_shared<SimpleQuote>(recoveryRate_)), refData);
343  curve_->curve()->enableExtrapolation();
344  WLOG("DefaultCurve: recovery rate found but no CDS quotes for "
345  << curveID << " and "
346  << "ImplyDefaultFromMarket is true. Curve built that gives default immediately.");
347  return;
348  }
349  } else {
350  QL_REQUIRE(!quotes.empty(), "No market points found for CDS curve config " << curveID);
351  }
352 
353  // Create the CDS instrument helpers, only keep alive helpers
354  vector<boost::shared_ptr<QuantExt::DefaultProbabilityHelper>> helpers;
355  std::map<QuantLib::Date, QuantLib::Period> helperQuoteTerms;
356  Real runningSpread = Null<Real>();
357  QuantExt::CreditDefaultSwap::ProtectionPaymentTime ppt = cdsConv->paysAtDefaultTime()
358  ? QuantExt::CreditDefaultSwap::atDefault
359  : QuantExt::CreditDefaultSwap::atPeriodEnd;
360 
361  if (config.type() == DefaultCurveConfig::Config::Type::SpreadCDS) {
362  for (auto quote : quotes) {
363  boost::shared_ptr<SpreadCdsHelper> tmp;
364  try {
365  tmp = boost::make_shared<SpreadCdsHelper>(
366  quote.value, quote.term, cdsConv->settlementDays(), cdsConv->calendar(), cdsConv->frequency(),
367  cdsConv->paymentConvention(), cdsConv->rule(), cdsConv->dayCounter(), recoveryRate_, discountCurve,
368  cdsConv->settlesAccrual(), ppt, config.startDate(), cdsConv->lastPeriodDayCounter());
369 
370  } catch (exception& e) {
371  if (quote.term == Period(0, Months)) {
372  WLOG("DefaultCurve:: Cannot add quote of term 0M to CDS curve " << curveID <<
373  " for asof date " << asof);
374  } else {
375  QL_FAIL("DefaultCurve:: Failed to add quote of term " << quote.term << " to CDS curve "
376  << curveID << " for asof date " << asof << ", with error: " << e.what());
377  }
378  }
379  if (tmp) {
380  if (tmp->latestDate() > asof) {
381  helpers.push_back(tmp);
382  runningSpread = config.runningSpread();
383  }
384  helperQuoteTerms[tmp->latestDate()] = quote.term;
385  }
386  }
387  } else {
388  for (auto quote : quotes) {
389  // If there is no running spread encoded in the quote, the config must have one.
390  runningSpread = quote.runningSpread;
391  if (runningSpread == Null<Real>()) {
392  QL_REQUIRE(config.runningSpread() != Null<Real>(),
393  "A running spread was not provided in the quote "
394  << "string so it must be provided in the config for CDS upfront curve " << curveID);
395  runningSpread = config.runningSpread();
396  }
397  auto tmp = boost::make_shared<UpfrontCdsHelper>(
398  quote.value, runningSpread, quote.term, cdsConv->settlementDays(), cdsConv->calendar(),
399  cdsConv->frequency(), cdsConv->paymentConvention(), cdsConv->rule(), cdsConv->dayCounter(),
400  recoveryRate_, discountCurve, cdsConv->upfrontSettlementDays(),
401  cdsConv->settlesAccrual(), ppt, config.startDate(), cdsConv->lastPeriodDayCounter());
402  if (tmp->latestDate() > asof) {
403  helpers.push_back(tmp);
404  }
405  helperQuoteTerms[tmp->latestDate()] = quote.term;
406  }
407  }
408 
409  QL_REQUIRE(!helpers.empty(), "DefaultCurve: no alive quotes found.");
410 
411  refData.runningSpread = runningSpread;
412 
413  // Ensure that the helpers are sorted. This is done in IterativeBootstrap, but we need
414  // a sorted instruments vector in the code here as well.
415  std::sort(helpers.begin(), helpers.end(), QuantLib::detail::BootstrapHelperSorter());
416 
417  // Get configuration values for bootstrap
418  Real accuracy = config.bootstrapConfig().accuracy();
419  Real globalAccuracy = config.bootstrapConfig().globalAccuracy();
420  bool dontThrow = config.bootstrapConfig().dontThrow();
421  Size maxAttempts = config.bootstrapConfig().maxAttempts();
422  Real maxFactor = config.allowNegativeRates() ? config.bootstrapConfig().maxFactor() : 1.0;
423  Real minFactor = config.bootstrapConfig().minFactor();
424  Size dontThrowSteps = config.bootstrapConfig().dontThrowSteps();
425 
426  typedef PiecewiseDefaultCurve<QuantExt::SurvivalProbability, LogLinear, QuantExt::IterativeBootstrap> SpCurve;
427  ATTR_UNUSED typedef SpCurve::traits_type dummy;
428  QuantExt::IterativeBootstrap<SpCurve> btconfig(accuracy, globalAccuracy, dontThrow, maxAttempts, maxFactor,
429  minFactor, dontThrowSteps);
430  boost::shared_ptr<DefaultProbabilityTermStructure> qlCurve;
431 
432  if (config.indexTerm() != 0 * Days) {
433 
434  // build flat index curve picking a quote with identical term as configured if possible
435  // otherwise build interpolated curve using the existing quotes
436 
437  std::vector<Real> helperTermTimes;
438  for (auto const& h : helpers) {
439  helperTermTimes.push_back(QuantExt::periodToTime(helperQuoteTerms[h->latestDate()]));
440  }
441 
442  Real t = QuantExt::periodToTime(config.indexTerm());
443  Size helperIndex_m, helperIndex_p;
444  Real alpha;
445  std::tie(helperIndex_m, helperIndex_p, alpha) = QuantExt::interpolationIndices(helperTermTimes, t);
446 
447  auto tmp1 = boost::make_shared<SpCurve>(
448  asof, std::vector<boost::shared_ptr<QuantExt::DefaultProbabilityHelper>>{helpers[helperIndex_m]},
449  config.dayCounter(), LogLinear(), btconfig);
450  Date d1 = helpers[helperIndex_m]->pillarDate();
451  Real p1 = tmp1->survivalProbability(d1);
452  auto tmp1i = boost::make_shared<QuantExt::InterpolatedSurvivalProbabilityCurve<LogLinear>>(
453  std::vector<Date>{asof, d1}, std::vector<Real>{1.0, p1}, config.dayCounter(), Calendar(),
454  std::vector<Handle<Quote>>(), std::vector<Date>(), LogLinear(), config.allowNegativeRates());
455 
456  if (close_enough(alpha, 1.0)) {
457  qlCurve = tmp1i;
458  } else {
459  auto tmp2 = boost::make_shared<SpCurve>(
460  asof, std::vector<boost::shared_ptr<QuantExt::DefaultProbabilityHelper>>{helpers[helperIndex_p]},
461  config.dayCounter(), LogLinear(), btconfig);
462  Date d2 = helpers[helperIndex_p]->pillarDate();
463  Real p2 = tmp2->survivalProbability(d2);
464  auto tmp2i = boost::make_shared<QuantExt::InterpolatedSurvivalProbabilityCurve<LogLinear>>(
465  std::vector<Date>{asof, d2}, std::vector<Real>{1.0, p2}, config.dayCounter(), Calendar(),
466  std::vector<Handle<Quote>>(), std::vector<Date>(), LogLinear(), config.allowNegativeRates());
467  tmp1i->enableExtrapolation();
468  tmp2i->enableExtrapolation();
469  qlCurve = boost::make_shared<QuantExt::TermInterpolatedDefaultCurve>(
470  Handle<DefaultProbabilityTermStructure>(tmp1i), Handle<DefaultProbabilityTermStructure>(tmp2i), alpha);
471  }
472 
473  } else {
474 
475  // build single name curve
476 
477  boost::shared_ptr<DefaultProbabilityTermStructure> tmp = boost::make_shared<SpCurve>(
478  asof, helpers, config.dayCounter(), LogLinear(),
479  QuantExt::IterativeBootstrap<SpCurve>(accuracy, globalAccuracy, dontThrow, maxAttempts, maxFactor,
480  minFactor, dontThrowSteps));
481 
482  // As for yield curves we need to copy the piecewise curve because on eval date changes the relative date
483  // helpers with trigger a bootstrap.
484  vector<Date> dates;
485  vector<Real> survivalProbs;
486  dates.push_back(asof);
487  survivalProbs.push_back(1.0);
488 
489  for (Size i = 0; i < helpers.size(); ++i) {
490  if (helpers[i]->latestDate() > asof) {
491  Date pillarDate = helpers[i]->pillarDate();
492  Probability sp = tmp->survivalProbability(pillarDate);
493 
494  // In some cases the bootstrapped survival probability at one tenor will be `close` to that at a
495  // previous tenor. Here we don't add that survival probability and date to avoid issues when creating
496  // the InterpolatedSurvivalProbabilityCurve below.
497  if (!survivalProbs.empty() && close(survivalProbs.back(), sp)) {
498  DLOG("Survival probability for curve " << spec.name() << " at date " << io::iso_date(pillarDate)
499  << " is the same as that at previous date "
500  << io::iso_date(dates.back()) << " so skipping it.");
501  continue;
502  }
503 
504  dates.push_back(pillarDate);
505  survivalProbs.push_back(sp);
506  TLOG(io::iso_date(pillarDate) << "," << fixed << setprecision(9) << sp);
507  }
508  }
509  if (dates.size() == 1) {
510  // We might have removed points above. To make the interpolation work, we need at least two points though.
511  dates.push_back(dates.back() + 1);
512  survivalProbs.push_back(survivalProbs.back());
513  }
514  qlCurve = boost::make_shared<QuantExt::InterpolatedSurvivalProbabilityCurve<LogLinear>>(
515  dates, survivalProbs, config.dayCounter(), Calendar(), std::vector<Handle<Quote>>(), std::vector<Date>(),
516  LogLinear(), config.allowNegativeRates());
517  }
518 
519  if (config.extrapolation()) {
520  qlCurve->enableExtrapolation();
521  DLOG("DefaultCurve: Enabled Extrapolation");
522  }
523 
524  curve_ = boost::make_shared<QuantExt::CreditCurve>(Handle<DefaultProbabilityTermStructure>(qlCurve), discountCurve,
525  Handle<Quote>(boost::make_shared<SimpleQuote>(recoveryRate_)),
526  refData);
527

531 void DefaultCurve::buildHazardRateCurve(const std::string& curveID, const DefaultCurveConfig::Config& config,
532  const Date& asof, const DefaultCurveSpec& spec, const Loader& loader) {
533 
534  LOG("Start building default curve of type HazardRate for curve " << curveID);
535 
536  QL_REQUIRE(config.type() == DefaultCurveConfig::Config::Type::HazardRate,
537  "DefaultCurve::buildHazardRateCurve expected a default curve configuration with type HazardRate");
538 
539  // Get the hazard rate curve conventions
540  boost::shared_ptr<Conventions> conventions = InstrumentConventions::instance().conventions();
541  QL_REQUIRE(conventions->has(config.conventionID()), "No conventions found with id " << config.conventionID());
542  boost::shared_ptr<CdsConvention> cdsConv =
543  boost::dynamic_pointer_cast<CdsConvention>(conventions->get(config.conventionID()));
544  QL_REQUIRE(cdsConv, "HazardRate curves require CDS convention");
545 
546  // Get the hazard rate quotes
547  set<QuoteData> quotes = getConfiguredQuotes(curveID, config, asof, loader);
548 
549  // Build the hazard rate curve
550  Calendar cal = cdsConv->calendar();
551  vector<Date> dates;
552  vector<Real> quoteValues;
553 
554  // If first term is not zero, add asof point
555  if (quotes.begin()->term != 0 * Days) {
556  LOG("DefaultCurve: add asof (" << asof << "), hazard rate " << quotes.begin()->value << ", as not given");
557  dates.push_back(asof);
558  quoteValues.push_back(quotes.begin()->value);
559  }
560 
561  for (auto quote : quotes) {
562  dates.push_back(cal.advance(asof, quote.term, Following, false));
563  quoteValues.push_back(quote.value);
564  }
565 
566  LOG("DefaultCurve: set up interpolated hazard rate curve");
567  curve_ = boost::make_shared<QuantExt::CreditCurve>(
568  Handle<DefaultProbabilityTermStructure>(boost::make_shared<QuantExt::InterpolatedHazardRateCurve<BackwardFlat>>(
569  dates, quoteValues, config.dayCounter(), BackwardFlat(), config.allowNegativeRates())));
570 
571  if (config.extrapolation()) {
572  curve_->curve()->enableExtrapolation();
573  DLOG("DefaultCurve: Enabled Extrapolation");
574  }
575 
576  if (recoveryRate_ == Null<Real>()) {
577  LOG("DefaultCurve: setting recovery rate to 0.0 for hazard rate curve, because none is given.");
578  recoveryRate_ = 0.0;
579  }
580 
581  // Force bootstrap so that errors are thrown during the build, not later
582  curve_->curve()->survivalProbability(QL_EPSILON);
583 
584  LOG("Finished building default curve of type HazardRate for curve " << curveID);
585 }

587 void DefaultCurve::buildBenchmarkCurve(const std::string& curveID, const DefaultCurveConfig::Config& config,
588  const Date& asof, const DefaultCurveSpec& spec, const Loader& loader,
589  map<string, boost::shared_ptr<YieldCurve>>& yieldCurves) {
590 
591  LOG("Start building default curve of type Benchmark for curve " << curveID);
592 
593  QL_REQUIRE(config.type() == DefaultCurveConfig::Config::Type::Benchmark,
594  "DefaultCurve::buildBenchmarkCurve expected a default curve configuration with type Benchmark");
595 
596  if (recoveryRate_ == Null<Real>())
597  recoveryRate_ = 0.0;
598 
599  // Populate benchmark yield curve
600  auto it = yieldCurves.find(config.benchmarkCurveID());
601  QL_REQUIRE(it != yieldCurves.end(), "The benchmark curve, " << config.benchmarkCurveID()
602  << ", required in the building of the curve, "
603  << spec.name() << ", was not found.");
604  boost::shared_ptr<YieldCurve> benchmarkCurve = it->second;
605 
606  // Populate source yield curve
607  it = yieldCurves.find(config.sourceCurveID());
608  QL_REQUIRE(it != yieldCurves.end(), "The source curve, " << config.sourceCurveID()
609  << ", required in the building of the curve, "
610  << spec.name() << ", was not found.");
611  boost::shared_ptr<YieldCurve> sourceCurve = it->second;
612 
613  // Parameters from the configuration
614  vector<Period> pillars = parseVectorOfValues<Period>(config.pillars(), &parsePeriod);
615  Calendar cal = config.calendar();
616  Size spotLag = config.spotLag();
617 
618  // Create the implied survival curve
619  vector<Date> dates;
620  vector<Real> impliedSurvProb;
621  Date spot = cal.advance(asof, spotLag * Days);
622  for (Size i = 0; i < pillars.size(); ++i) {
623  dates.push_back(cal.advance(spot, pillars[i]));
624  Real tmp = dates[i] == asof
625  ? 1.0
626  : sourceCurve->handle()->discount(dates[i]) / benchmarkCurve->handle()->discount(dates[i]);
627  // if a non-zero recovery rate is given, we adjust the implied surv probability according to a market value
628  // recovery model (see the documentation of the benchmark curve in the user guide for more details)
629  impliedSurvProb.push_back(std::pow(tmp, 1.0 / (1.0 - recoveryRate_)));
630  }
631  QL_REQUIRE(dates.size() > 0, "DefaultCurve (Benchmark): no dates given");
632 
633  // Insert SP = 1.0 at asof if asof date is not in the pillars
634  if (dates[0] != asof) {
635  dates.insert(dates.begin(), asof);
636  impliedSurvProb.insert(impliedSurvProb.begin(), 1.0);
637  }
638 
639  LOG("DefaultCurve: set up interpolated surv prob curve as yield over benchmark");
640  curve_ = boost::make_shared<QuantExt::CreditCurve>(Handle<DefaultProbabilityTermStructure>(
641  boost::make_shared<QuantExt::InterpolatedSurvivalProbabilityCurve<LogLinear>>(
642  dates, impliedSurvProb, config.dayCounter(), Calendar(), std::vector<Handle<Quote>>(), std::vector<Date>(),
643  LogLinear(), config.allowNegativeRates())));
644 
645  if (config.extrapolation()) {
646  curve_->curve()->enableExtrapolation();
647  DLOG("DefaultCurve: Enabled Extrapolation");
648  }
649 
650  // Force bootstrap so that errors are thrown during the build, not later
651  curve_->curve()->survivalProbability(QL_EPSILON);
652 
653  LOG("Finished building default curve of type Benchmark for curve " << curveID);
654 }
