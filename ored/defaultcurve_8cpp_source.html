78 set<QuoteData> getRegexQuotes(const Wildcard& wc, const string& configId, DefaultCurveConfig::Type type,
79  const Date& asof, const Loader& loader) {
80 
81  using DCCT = DefaultCurveConfig::Type;
82  using MDQT = MarketDatum::QuoteType;
83  using MDIT = MarketDatum::InstrumentType;
84  LOG("Loading regex quotes for default curve " << configId);
85 
86  // Loop over the available market data and pick out quotes that match the expression
87  set<QuoteData> result;
88  for (const auto& md : loader.loadQuotes(asof)) {
89 
90  // Go to next quote if the market data point's date does not equal our asof
91  if (md->asofDate() != asof)
92  continue;
93 
94  auto mdit = md->instrumentType();
95  auto mdqt = md->quoteType();
96 
97  // If we have a CDS spread or hazard rate quote, check it and populate tenor and value if it matches
98  if (type == DCCT::SpreadCDS && mdit == MDIT::CDS &&
99  (mdqt == MDQT::CREDIT_SPREAD || mdqt == MDQT::CONV_CREDIT_SPREAD)) {
100 
101  auto q = boost::dynamic_pointer_cast<CdsQuote>(md);
102  if (wc.matches(q->name())) {
103  addQuote(result, configId, q->name(), q->term(), q->quote()->value(), q->runningSpread());
104  }
105 
106  } else if (type == DCCT::Price && (mdit == MDIT::CDS && mdqt == MDQT::PRICE)) {
107 
108  auto q = boost::dynamic_pointer_cast<CdsQuote>(md);
109  if (wc.matches(q->name())) {
110  addQuote(result, configId, q->name(), q->term(), q->quote()->value(), q->runningSpread());
111  }
112 
113  } else if (type == DCCT::HazardRate && (mdit == MDIT::HAZARD_RATE && mdqt == MDQT::RATE)) {
114 
115  auto q = boost::dynamic_pointer_cast<HazardRateQuote>(md);
116  if (wc.matches(q->name())) {
117  addQuote(result, configId, q->name(), q->term(), q->quote()->value());
118  }
119  }
120  }
121 
122  // We don't check for an empty set of CDS quotes here. We check it later because under some circumstances,
123  // it may be allowable to have no quotes.
124  if (type != DCCT::SpreadCDS && type != DCCT::Price) {
125  QL_REQUIRE(!result.empty(), "No market points found for curve config " << configId);
126  }
127 
128  LOG("DefaultCurve " << configId << " loaded and using " << result.size() << " quotes.");
129 
130  return result;
131 }

133 set<QuoteData> getExplicitQuotes(const vector<pair<string, bool>>& quotes, const string& configId,
134  DefaultCurveConfig::Type type, const Date& asof, const Loader& loader) {
135 
136  using DCCT = DefaultCurveConfig::Type;
137  LOG("Loading explicit quotes for default curve " << configId);
138 
139  set<QuoteData> result;
140  for (const auto& p : quotes) {
141  if (boost::shared_ptr<MarketDatum> md = loader.get(p, asof)) {
142  if (type == DCCT::SpreadCDS || type == DCCT::Price) {
143  auto q = boost::dynamic_pointer_cast<CdsQuote>(md);
144  QL_REQUIRE(q, "Quote " << p.first << " for config " << configId << " should be a CdsQuote");
145  addQuote(result, configId, q->name(), q->term(), q->quote()->value(), q->runningSpread());
146  } else {
147  auto q = boost::dynamic_pointer_cast<HazardRateQuote>(md);
148  QL_REQUIRE(q, "Quote " << p.first << " for config " << configId << " should be a HazardRateQuote");
149  addQuote(result, configId, q->name(), q->term(), q->quote()->value());
150  }
151  }
152  }
153 
154  // We don't check for an empty set of CDS quotes here. We check it later because under some circumstances,
155  // it may be allowable to have no quotes.
156  if (type != DCCT::SpreadCDS && type != DCCT::Price) {
157  QL_REQUIRE(!result.empty(), "No market points found for curve config " << configId);
158  }
159 
160  LOG("DefaultCurve " << configId << " using " << result.size() << " default quotes of " << quotes.size()
161  << " requested quotes.");
162 
163  return result;
164 }

166 set<QuoteData> getConfiguredQuotes(DefaultCurveConfig& config, const Date& asof, const Loader& loader) {
167 
168  using DCCT = DefaultCurveConfig::Type;
169  auto type = config.type();
170  QL_REQUIRE(type == DCCT::SpreadCDS || type == DCCT::Price || type == DCCT::HazardRate,
171  "getConfiguredQuotes expects a curve type of SpreadCDS, Price or HazardRate.");
172  QL_REQUIRE(!config.cdsQuotes().empty(), "No quotes configured for curve " << config.curveID());
173 
174  // We may have a _single_ regex quote or a list of explicit quotes. Check if we have single regex quote.
175  std::vector<std::string> tmp;
176  std::transform(config.cdsQuotes().begin(), config.cdsQuotes().end(), std::back_inserter(tmp),
177  [](const std::pair<std::string, bool>& p) { return p.first; });
178  auto wildcard = getUniqueWildcard(tmp);
179 
180  if (wildcard) {
181  return getRegexQuotes(*wildcard, config.curveID(), config.type(), asof, loader);
182  } else {
183  return getExplicitQuotes(config.cdsQuotes(), config.curveID(), config.type(), asof, loader);
184  }
185 }

192 DefaultCurve::DefaultCurve(Date asof, DefaultCurveSpec spec, const Loader& loader,
193  const CurveConfigurations& curveConfigs, const Conventions& conventions,
194  map<string, boost::shared_ptr<YieldCurve>>& yieldCurves,
195  map<string, boost::shared_ptr<DefaultCurve>>& defaultCurves) {
196 
197  try {
198 
199  const boost::shared_ptr<DefaultCurveConfig>& config = curveConfigs.defaultCurveConfig(spec.curveConfigID());
200 
201  // Set the recovery rate if necessary
202  recoveryRate_ = Null<Real>();
203  if (!config->recoveryRateQuote().empty()) {
204  // for some curve types (Benchmark, MutliSection) a numeric recovery rate is allowed, which we handle here
205  if (!tryParseReal(config->recoveryRateQuote(), recoveryRate_)) {
206  QL_REQUIRE(loader.has(config->recoveryRateQuote(), asof),
207  "There is no market data for the requested recovery rate " << config->recoveryRateQuote());
208  recoveryRate_ = loader.get(config->recoveryRateQuote(), asof)->quote()->value();
209  }
210  }
211 
212  // Build the default curve of the requested type
213  switch (config->type()) {
214  case DefaultCurveConfig::Type::SpreadCDS:
215  case DefaultCurveConfig::Type::Price:
216  buildCdsCurve(*config, asof, spec, loader, conventions, yieldCurves);
217  break;
218  case DefaultCurveConfig::Type::HazardRate:
219  buildHazardRateCurve(*config, asof, spec, loader, conventions);
220  break;
221  case DefaultCurveConfig::Type::Benchmark:
222  buildBenchmarkCurve(*config, asof, spec, loader, conventions, yieldCurves);
223  break;
224  case DefaultCurveConfig::Type::MultiSection:
225  buildMultiSectionCurve(*config, asof, spec, loader, conventions, defaultCurves);
226  break;
227  default:
228  QL_FAIL("The DefaultCurveConfig type " << static_cast<int>(config->type()) << " was not recognised");
229  }
230 
231  } catch (exception& e) {
232  QL_FAIL("default curve building failed for " << spec.curveConfigID() << ": " << e.what());
233  } catch (...) {
234  QL_FAIL("default curve building failed for " << spec.curveConfigID() << ": unknown error");
235  }

238 void DefaultCurve::buildCdsCurve(DefaultCurveConfig& config, const Date& asof, const DefaultCurveSpec& spec,
239  const Loader& loader, const Conventions& conventions,
240  map<string, boost::shared_ptr<YieldCurve>>& yieldCurves) {
241 
242  LOG("Start building default curve of type SpreadCDS for curve " << config.curveID());
243 
244  QL_REQUIRE(config.type() == DefaultCurveConfig::Type::SpreadCDS || config.type() == DefaultCurveConfig::Type::Price,
245  "DefaultCurve::buildCdsCurve expected a default curve configuration with type SpreadCDS/Price");
246  QL_REQUIRE(recoveryRate_ != Null<Real>(), "DefaultCurve: recovery rate needed to build SpreadCDS curve");
247 
248  // Get the CDS curve conventions
249  QL_REQUIRE(conventions.has(config.conventionID()), "No conventions found with id " << config.conventionID());
250  boost::shared_ptr<CdsConvention> cdsConv =
251  boost::dynamic_pointer_cast<CdsConvention>(conventions.get(config.conventionID()));
252  QL_REQUIRE(cdsConv, "SpreadCDS curves require CDS convention");
253 
254  // Get the discount curve for use in the CDS spread curve bootstrap
255  auto it = yieldCurves.find(config.discountCurveID());
256  QL_REQUIRE(it != yieldCurves.end(), "The discount curve, " << config.discountCurveID()
257  << ", required in the building of the curve, "
258  << spec.name() << ", was not found.");
259  Handle<YieldTermStructure> discountCurve = it->second->handle();
260 
261  // Get the CDS spread / price curve quotes
262  set<QuoteData> quotes = getConfiguredQuotes(config, asof, loader);
263 
264  // If the configuration instructs us to imply a default from the market data, we do it here.
265  if (config.implyDefaultFromMarket() && *config.implyDefaultFromMarket()) {
266  if (recoveryRate_ != Null<Real>() && quotes.empty()) {
267  // Assume entity is in default, between event determination date and auction date. Build a survivial
268  // probability curve with value 0.0 tomorrow to approximate this and allow dependent instruments to price.
269  // Need to use small but positive numbers to avoid downstream issues with log linear survivals e.g. below
270  // and in places like ScenarioSimMarket.
271  vector<Date> dates{asof, asof + 1 * Years, asof + 10 * Years};
272  vector<Real> survivalProbs{1.0, 1e-16, 1e-18};
273  curve_ = boost::make_shared<QuantExt::InterpolatedSurvivalProbabilityCurve<LogLinear>>(
274  dates, survivalProbs, config.dayCounter(), Calendar(), std::vector<Handle<Quote>>(),
275  std::vector<Date>(), LogLinear());
276  curve_->enableExtrapolation();
277  WLOG("DefaultCurve: recovery rate found but no CDS quotes for "
278  << config.curveID() << " and "
279  << "ImplyDefaultFromMarket is true. Curve built that gives default immediately.");
280  return;
281  }
282  } else {
283  QL_REQUIRE(!quotes.empty(), "No market points found for CDS curve config " << config.curveID());
284  }
285 
286  // Create the CDS instrument helpers
287  vector<boost::shared_ptr<QuantExt::DefaultProbabilityHelper>> helpers;
288  QuantExt::CreditDefaultSwap::ProtectionPaymentTime ppt = cdsConv->paysAtDefaultTime() ?
289  QuantExt::CreditDefaultSwap::atDefault : QuantExt::CreditDefaultSwap::atPeriodEnd;
290 
291  if (config.type() == DefaultCurveConfig::Type::SpreadCDS) {
292  for (auto quote : quotes) {
293  helpers.push_back(boost::make_shared<QuantExt::SpreadCdsHelper>(
294  quote.value, quote.term, cdsConv->settlementDays(), cdsConv->calendar(), cdsConv->frequency(),
295  cdsConv->paymentConvention(), cdsConv->rule(), cdsConv->dayCounter(), recoveryRate_, discountCurve,
296  config.startDate(), cdsConv->settlesAccrual(), ppt, cdsConv->lastPeriodDayCounter()));
297  }
298  } else {
299  for (auto quote : quotes) {
300 
301  // If there is no running spread encoded in the quote, the config must have one.
302  Real runningSpread = quote.runningSpread;
303  if (runningSpread == Null<Real>()) {
304  QL_REQUIRE(config.runningSpread() != Null<Real>(), "A running spread was not provided in the quote " <<
305  "string so it must be provided in the config for CDS upfront curve " << config.curveID());
306  runningSpread = config.runningSpread();
307  }
308 
309  helpers.push_back(boost::make_shared<QuantExt::UpfrontCdsHelper>(
310  quote.value, runningSpread, quote.term, cdsConv->settlementDays(), cdsConv->calendar(),
311  cdsConv->frequency(), cdsConv->paymentConvention(), cdsConv->rule(), cdsConv->dayCounter(),
312  recoveryRate_, discountCurve, config.startDate(), cdsConv->upfrontSettlementDays(),
313  cdsConv->settlesAccrual(), ppt, cdsConv->lastPeriodDayCounter()));
314  }
315  }
316 
317  // Ensure that the helpers are sorted. This is done in IterativeBootstrap, but we need
318  // a sorted instruments vector in the code here as well.
319  std::sort(helpers.begin(), helpers.end(), QuantLib::detail::BootstrapHelperSorter());
320 
321  // Get configuration values for bootstrap
322  Real accuracy = config.bootstrapConfig().accuracy();
323  Real globalAccuracy = config.bootstrapConfig().globalAccuracy();
324  bool dontThrow = config.bootstrapConfig().dontThrow();
325  Size maxAttempts = config.bootstrapConfig().maxAttempts();
326  Real maxFactor = config.allowNegativeRates() ? config.bootstrapConfig().maxFactor() : 1.0;
327  Real minFactor = config.bootstrapConfig().minFactor();
328  Size dontThrowSteps = config.bootstrapConfig().dontThrowSteps();
329 
330  // Create the default probability term structure
331  typedef PiecewiseDefaultCurve<QuantExt::SurvivalProbability, LogLinear, QuantExt::IterativeBootstrap> SpCurve;
332  ATTR_UNUSED typedef SpCurve::traits_type dummy;
333  boost::shared_ptr<DefaultProbabilityTermStructure> tmp = boost::make_shared<SpCurve>(
334  asof, helpers, config.dayCounter(), LogLinear(),
335  QuantExt::IterativeBootstrap<SpCurve>(accuracy, globalAccuracy, dontThrow, maxAttempts, maxFactor, minFactor,
336  dontThrowSteps));
337 
338  // As for yield curves we need to copy the piecewise curve because on eval date changes the relative date
339  // helpers with trigger a bootstrap.
340  vector<Date> dates;
341  vector<Real> survivalProbs;
342  dates.push_back(asof);
343  survivalProbs.push_back(1.0);
344  TLOG("Survival probabilities for CDS curve " << config.curveID() << ":");
345  TLOG(io::iso_date(dates.back()) << "," << fixed << setprecision(9) << survivalProbs.back());
346 
347  bool gotAliveHelper = false;
348 
349  for (Size i = 0; i < helpers.size(); ++i) {
350  if (helpers[i]->latestDate() > asof) {
351  gotAliveHelper = true;
352 
353  Date pillarDate = helpers[i]->pillarDate();
354  Probability sp = tmp->survivalProbability(pillarDate);
355 
356  // In some cases the bootstrapped survival probability at one tenor will be `close` to that at a previous
357  // tenor. Here we don't add that survival probability and date to avoid issues when creating the
358  // InterpolatedSurvivalProbabilityCurve below.
359  if (!survivalProbs.empty() && close(survivalProbs.back(), sp)) {
360  DLOG("Survival probability for curve " << spec.name() << " at date " << io::iso_date(pillarDate)
361  << " is the same as that at previous date "
362  << io::iso_date(dates.back()) << " so skipping it.");
363  continue;
364  }
365 
366  dates.push_back(pillarDate);
367  survivalProbs.push_back(sp);
368  TLOG(io::iso_date(pillarDate) << "," << fixed << setprecision(9) << sp);
369  }
370  }
371  QL_REQUIRE(gotAliveHelper, "Need at least one alive helper to build the default curve");
372  if (dates.size() == 1) {
373  // We might have removed points above. To make the interpolation work, we need at least two points though.
374  dates.push_back(dates.back() + 1);
375  survivalProbs.push_back(survivalProbs.back());
376  }
377 
378  LOG("DefaultCurve: copy piecewise curve to interpolated survival probability curve");
379  curve_ = boost::make_shared<QuantExt::InterpolatedSurvivalProbabilityCurve<LogLinear>>(
380  dates, survivalProbs, config.dayCounter(), Calendar(), std::vector<Handle<Quote>>(), std::vector<Date>(),
381  LogLinear(), config.allowNegativeRates());
382  if (config.extrapolation()) {
383  curve_->enableExtrapolation();
384  DLOG("DefaultCurve: Enabled Extrapolation");
385  }
386 
387  // Force bootstrap so that errors are thrown during the build, not later
388  curve_->survivalProbability(QL_EPSILON);
389 
390  LOG("Finished building default curve of type SpreadCDS for curve " << config.curveID());
391 }

393 void DefaultCurve::buildHazardRateCurve(DefaultCurveConfig& config, const Date& asof, const DefaultCurveSpec& spec,
394  const Loader& loader, const Conventions& conventions) {
395 
396  LOG("Start building default curve of type HazardRate for curve " << config.curveID());
397 
398  QL_REQUIRE(config.type() == DefaultCurveConfig::Type::HazardRate,
399  "DefaultCurve::buildHazardRateCurve expected a default curve configuration with type HazardRate");
400 
401  // Get the hazard rate curve conventions
402  QL_REQUIRE(conventions.has(config.conventionID()), "No conventions found with id " << config.conventionID());
403  boost::shared_ptr<CdsConvention> cdsConv =
404  boost::dynamic_pointer_cast<CdsConvention>(conventions.get(config.conventionID()));
405  QL_REQUIRE(cdsConv, "HazardRate curves require CDS convention");
406 
407  // Get the hazard rate quotes
408  set<QuoteData> quotes = getConfiguredQuotes(config, asof, loader);
409 
410  // Build the hazard rate curve
411  Calendar cal = cdsConv->calendar();
412  vector<Date> dates;
413  vector<Real> quoteValues;
414 
415  // If first term is not zero, add asof point
416  if (quotes.begin()->term != 0 * Days) {
417  LOG("DefaultCurve: add asof (" << asof << "), hazard rate " << quotes.begin()->value << ", as not given");
418  dates.push_back(asof);
419  quoteValues.push_back(quotes.begin()->value);
420  }
421 
422  for (auto quote : quotes) {
423  dates.push_back(cal.advance(asof, quote.term, Following, false));
424  quoteValues.push_back(quote.value);
425  }
426 
427  LOG("DefaultCurve: set up interpolated hazard rate curve");
428  curve_ = boost::make_shared<QuantExt::InterpolatedHazardRateCurve<BackwardFlat>>(
429  dates, quoteValues, config.dayCounter(), BackwardFlat(), config.allowNegativeRates());
430 
431  if (config.extrapolation()) {
432  curve_->enableExtrapolation();
433  DLOG("DefaultCurve: Enabled Extrapolation");
434  }
435 
436  if (recoveryRate_ == Null<Real>()) {
437  LOG("DefaultCurve: setting recovery rate to 0.0 for hazard rate curve, because none is given.");
438  recoveryRate_ = 0.0;
439  }
440 
441  // Force bootstrap so that errors are thrown during the build, not later
442  curve_->survivalProbability(QL_EPSILON);
443 
444  LOG("Finished building default curve of type HazardRate for curve " << config.curveID());
445 }

447 void DefaultCurve::buildBenchmarkCurve(DefaultCurveConfig& config, const Date& asof, const DefaultCurveSpec& spec,
448  const Loader& loader, const Conventions& conventions,
449  map<string, boost::shared_ptr<YieldCurve>>& yieldCurves) {
450 
451  LOG("Start building default curve of type Benchmark for curve " << config.curveID());
452 
453  QL_REQUIRE(config.type() == DefaultCurveConfig::Type::Benchmark,
454  "DefaultCurve::buildBenchmarkCurve expected a default curve configuration with type Benchmark");
455 
456  if (recoveryRate_ == Null<Real>())
457  recoveryRate_ = 0.0;
458 
459  // Populate benchmark yield curve
460  auto it = yieldCurves.find(config.benchmarkCurveID());
461  QL_REQUIRE(it != yieldCurves.end(), "The benchmark curve, " << config.benchmarkCurveID()
462  << ", required in the building of the curve, "
463  << spec.name() << ", was not found.");
464  boost::shared_ptr<YieldCurve> benchmarkCurve = it->second;
465 
466  // Populate source yield curve
467  it = yieldCurves.find(config.sourceCurveID());
468  QL_REQUIRE(it != yieldCurves.end(), "The source curve, " << config.sourceCurveID()
469  << ", required in the building of the curve, "
470  << spec.name() << ", was not found.");
471  boost::shared_ptr<YieldCurve> sourceCurve = it->second;
472 
473  // Parameters from the configuration
474  vector<Period> pillars = parseVectorOfValues<Period>(config.pillars(), &parsePeriod);
475  Calendar cal = config.calendar();
476  Size spotLag = config.spotLag();
477 
478  // Create the implied survival curve
479  vector<Date> dates;
480  vector<Real> impliedSurvProb;
481  Date spot = cal.advance(asof, spotLag * Days);
482  for (Size i = 0; i < pillars.size(); ++i) {
483  dates.push_back(cal.advance(spot, pillars[i]));
484  Real tmp = dates[i] == asof
485  ? 1.0
486  : sourceCurve->handle()->discount(dates[i]) / benchmarkCurve->handle()->discount(dates[i]);
487  // if a non-zero recovery rate is given, we adjust the implied surv probability according to a market value
488  // recovery model (see the documentation of the benchmark curve in the user guide for more details)
489  impliedSurvProb.push_back(std::pow(tmp, 1.0 / (1.0 - recoveryRate_)));
490  }
491  QL_REQUIRE(dates.size() > 0, "DefaultCurve (Benchmark): no dates given");
492 
493  // Insert SP = 1.0 at asof if asof date is not in the pillars
494  if (dates[0] != asof) {
495  dates.insert(dates.begin(), asof);
496  impliedSurvProb.insert(impliedSurvProb.begin(), 1.0);
497  }
498 
499  LOG("DefaultCurve: set up interpolated surv prob curve as yield over benchmark");
500  curve_ = boost::make_shared<QuantExt::InterpolatedSurvivalProbabilityCurve<LogLinear>>(
501  dates, impliedSurvProb, config.dayCounter(), Calendar(), std::vector<Handle<Quote>>(), std::vector<Date>(),
502  LogLinear(), config.allowNegativeRates());
503 
504  if (config.extrapolation()) {
505  curve_->enableExtrapolation();
506  DLOG("DefaultCurve: Enabled Extrapolation");
507  }
508 
509  // Force bootstrap so that errors are thrown during the build, not later
510  curve_->survivalProbability(QL_EPSILON);
511 
512  LOG("Finished building default curve of type Benchmark for curve " << config.curveID());
513 }

515 void DefaultCurve::buildMultiSectionCurve(DefaultCurveConfig& config, const Date& asof, const DefaultCurveSpec& spec,
516  const Loader& loader, const Conventions& conventions,
517  map<string, boost::shared_ptr<DefaultCurve>>& defaultCurves) {
518  LOG("Start building default curve of type MultiSection for curve " << config.curveID());
519 
520  std::vector<Handle<DefaultProbabilityTermStructure>> curves;
521  std::vector<Handle<Quote>> recoveryRates;
522  std::vector<Date> switchDates;
523 
524  for (auto const& s : config.multiSectionSourceCurveIds()) {
