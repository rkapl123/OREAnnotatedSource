26 void VanillaOptionTrade::build(const boost::shared_ptr<ore::data::EngineFactory>& engineFactory) {
27  Currency ccy = parseCurrencyWithMinors(currency_);
28  QL_REQUIRE(tradeActions().empty(), "TradeActions not supported for VanillaOption");
29 
30  // If underlying currency is empty, then set to payment currency by default.
31  // If non-empty, then check if the currencies are different for a Quanto payoff
32  Currency underlyingCurrency = underlyingCurrency_.empty() ? ccy : parseCurrencyWithMinors(underlyingCurrency_);
33  bool sameCcy = underlyingCurrency == ccy;
34 
35  if (strike_.currency().empty())
36  strike_.setCurrency(ccy.code());
37 
38  // Payoff
39  Option::Type type = parseOptionType(option_.callPut());
40  boost::shared_ptr<StrikedTypePayoff> payoff(new PlainVanillaPayoff(type, strike_.value()));
41  QuantLib::Exercise::Type exerciseType = parseExerciseType(option_.style());
42  QL_REQUIRE(option_.exerciseDates().size() == 1, "Invalid number of exercise dates");
43  expiryDate_ = parseDate(option_.exerciseDates().front());
44  // Set the maturity date equal to the expiry date. It may get updated below if option is cash settled with
45  // payment after expiry.
46  maturity_ = expiryDate_;
47  // Exercise
48  boost::shared_ptr<Exercise> exercise;
49  switch (exerciseType) {
50  case QuantLib::Exercise::Type::European: {
51  exercise = boost::make_shared<EuropeanExercise>(expiryDate_);
52  break;
53  }
54  case QuantLib::Exercise::Type::American: {
55  exercise = boost::make_shared<AmericanExercise>(expiryDate_, option_.payoffAtExpiry());
56  break;
57  }
58  default:
59  QL_FAIL("Option Style " << option_.style() << " is not supported");
60  }
61  // Create the instrument and then populate the name for the engine builder.
62  boost::shared_ptr<Instrument> vanilla;
63  string tradeTypeBuilder = tradeType_;
64  Settlement::Type settlementType = parseSettlementType(option_.settlement());
65 
66  // For Quanto, check for European and Cash, except for an FX underlying
67  if (!sameCcy) {
68  QL_REQUIRE(exerciseType == Exercise::Type::European, "Option exercise must be European for a Quanto payoff.");
69  if (settlementType == Settlement::Type::Physical) {
70  QL_REQUIRE(assetClassUnderlying_ == AssetClass::FX,
71  "Physically settled Quanto options are allowed only for an FX underlying.");
72  }
73  }
74 
75  if (exerciseType == Exercise::European && settlementType == Settlement::Cash) {
76  // We have a European cash settled option.
77 
78  // Get the payment date.
79  const boost::optional<OptionPaymentData>& opd = option_.paymentData();
80  Date paymentDate = expiryDate_;
81  if (opd) {
82  if (opd->rulesBased()) {
83  const Calendar& cal = opd->calendar();
84  QL_REQUIRE(cal != Calendar(), "Need a non-empty calendar for rules based payment date.");
85  paymentDate = cal.advance(expiryDate_, opd->lag(), Days, opd->convention());
86  } else {
87  const vector<Date>& dates = opd->dates();
88  QL_REQUIRE(dates.size() == 1, "Need exactly one payment date for cash settled European option.");
89  paymentDate = dates[0];
90  }
91  QL_REQUIRE(paymentDate >= expiryDate_, "Payment date must be greater than or equal to expiry date.");
92  }
93 
94  if (paymentDate > expiryDate_) {
95  QL_REQUIRE(sameCcy, "Payment date must equal expiry date for a Quanto payoff. Trade: " << id() << ".");
96 
97  // Build a QuantExt::CashSettledEuropeanOption if payment date is strictly greater than expiry.
98 
99  // Has the option been marked as exercised
100  const boost::optional<OptionExerciseData>& oed = option_.exerciseData();
101  bool exercised = false;
102  Real exercisePrice = Null<Real>();
103  if (oed) {
104  QL_REQUIRE(oed->date() == expiryDate_, "The supplied exercise date ("
105  << io::iso_date(oed->date())
106  << ") should equal the option's expiry date ("
107  << io::iso_date(expiryDate_) << ").");
108  exercised = true;
109  exercisePrice = oed->price();
110  }
111 
112  // If automatic exercise, we will need an index fixing on the expiry date.
113  if (option_.isAutomaticExercise()) {
114  QL_REQUIRE(index_, "Option trade " << id() << " has automatic exercise so we need a valid index.");
115  // If index name has not been populated, use logic here to populate it from the index object.
116  string indexName = indexName_;
117  if (indexName.empty()) {
118  indexName = index_->name();
119  if (assetClassUnderlying_ == AssetClass::EQ)
120  indexName = "EQ-" + indexName;
121  }
122  requiredFixings_.addFixingDate(expiryDate_, indexName, paymentDate);
123  }
124 
125  // Build the instrument
126  LOG("Build CashSettledEuropeanOption for trade " << id());
127  vanilla = boost::make_shared<CashSettledEuropeanOption>(
128  type, strike_.value(), expiryDate_, paymentDate, option_.isAutomaticExercise(), index_, exercised, exercisePrice);
129 
130  // Allow for a separate pricing engine that takes care of payment on a date after expiry. Do this by
131  // appending 'EuropeanCS' to the trade type.
132  tradeTypeBuilder = tradeType_ + "EuropeanCS";
133 
134  // Update the maturity date.
135  maturity_ = paymentDate;
136 
137  } else {
138  if (forwardDate_ == QuantLib::Date()) {
139  // If payment date is not greater than expiry, build QuantLib::VanillaOption.
140  if (sameCcy) {
141  LOG("Build VanillaOption for trade " << id());
142  vanilla = boost::make_shared<QuantLib::VanillaOption>(payoff, exercise);
143  }
144  else {
145  LOG("Build QuantoVanillaOption for trade " << id());
146  vanilla = boost::make_shared<QuantLib::QuantoVanillaOption>(payoff, exercise);
147  if (assetClassUnderlying_ == AssetClass::EQ)
148  tradeTypeBuilder = "QuantoEquityOption";
149  else if (assetClassUnderlying_ == AssetClass::COM)
150  tradeTypeBuilder = "QuantoCommodityOption";
151  else
152  QL_FAIL("Option Quanto payoff not supported for " << assetClassUnderlying_ << " class.");
153  }
154  } else {
155  LOG("Build VanillaForwardOption for trade " << id());
156  QL_REQUIRE(sameCcy, "Quanto payoff is not currently supported for Forward Options: Trade " << id());
157  vanilla = boost::make_shared<QuantExt::VanillaForwardOption>(payoff, exercise, forwardDate_);
158  if (assetClassUnderlying_ == AssetClass::COM)
159  tradeTypeBuilder = tradeType_ + "Forward";
160  }
161  }
162 
163  } else {
164  if (forwardDate_ == QuantLib::Date()) {
165  // If not European or not cash settled, build QuantLib::VanillaOption.
166  if (sameCcy) {
167  LOG("Build VanillaOption for trade " << id());
168  vanilla = boost::make_shared<QuantLib::VanillaOption>(payoff, exercise);
169  } else {
170  LOG("Build QuantoVanillaOption for trade " << id());
171  vanilla = boost::make_shared<QuantLib::QuantoVanillaOption>(payoff, exercise);
172  }
173  } else {
174  QL_REQUIRE(exerciseType == QuantLib::Exercise::Type::European, "Only European Forward Options currently supported");
175  LOG("Built VanillaForwardOption for trade " << id());
176  vanilla = boost::make_shared<QuantExt::VanillaForwardOption>(payoff, exercise, forwardDate_);
177  if (assetClassUnderlying_ == AssetClass::COM)
178  tradeTypeBuilder = tradeType_ + "Forward";
179  }
180 
181  // If the tradeTypeBuilder has not been modified yet..
182  if (tradeTypeBuilder == tradeType_) {
183  if (sameCcy)
184  tradeTypeBuilder = tradeType_ + (exerciseType == QuantLib::Exercise::Type::European ? "" : "American");
185  else
186  tradeTypeBuilder = "QuantoFxOption";
187  }
188  }
189  LOG("tradeTypeBuilder set to " << tradeTypeBuilder << " for trade " << id());
190 
191  // Generally we need to set the pricing engine here even if the option is expired at build time, since the valuation date
192  // might change after build, and we get errors for the edge case valuation date = expiry date for European options.
193  string configuration = Market::defaultConfiguration;
194  boost::shared_ptr<EngineBuilder> builder = engineFactory->builder(tradeTypeBuilder);
195  QL_REQUIRE(builder, "No builder found for " << tradeTypeBuilder);
196 
197  if (sameCcy) {
198  boost::shared_ptr<VanillaOptionEngineBuilder> vanillaOptionBuilder =
199  boost::dynamic_pointer_cast<VanillaOptionEngineBuilder>(builder);
200  QL_REQUIRE(vanillaOptionBuilder != nullptr, "No engine builder found for trade type " << tradeTypeBuilder);
201 
202  vanilla->setPricingEngine(vanillaOptionBuilder->engine(assetName_, ccy, expiryDate_));
203 
204  configuration = vanillaOptionBuilder->configuration(MarketContext::pricing);
205  } else {
206  boost::shared_ptr<QuantoVanillaOptionEngineBuilder> quantoVanillaOptionBuilder =
207  boost::dynamic_pointer_cast<QuantoVanillaOptionEngineBuilder>(builder);
208  QL_REQUIRE(quantoVanillaOptionBuilder != nullptr, "No (Quanto) engine builder found for trade type "
209  << tradeTypeBuilder);
210 
211  vanilla->setPricingEngine(quantoVanillaOptionBuilder->engine(assetName_, underlyingCurrency, ccy, expiryDate_));
212 
213  configuration = quantoVanillaOptionBuilder->configuration(MarketContext::pricing);
214  }
215 
216  Position::Type positionType = parsePositionType(option_.longShort());
217  Real bsInd = (positionType == QuantLib::Position::Long ? 1.0 : -1.0);
218  Real mult = quantity_ * bsInd;
219 
220  std::vector<boost::shared_ptr<Instrument>> additionalInstruments;
221  std::vector<Real> additionalMultipliers;
222  maturity_ = std::max(maturity_, addPremiums(additionalInstruments, additionalMultipliers, mult,
223  option_.premiumData(), -bsInd, ccy, engineFactory, configuration));
224 
225  instrument_ = boost::shared_ptr<InstrumentWrapper>(
226  new VanillaInstrument(vanilla, mult, additionalInstruments, additionalMultipliers));
227  npvCurrency_ = ccy.code();
228 
229  // Notional - we really need todays spot to get the correct notional.
230  // But rather than having it move around we use strike * quantity
231  notional_ = strike_.value() * quantity_;
232  // the following is correct for vanilla (sameCcy = true) and quanto (sameCcy = false)
233  notionalCurrency_ = ccy.code();
234 }
