44InfDkBuilder::InfDkBuilder(const boost::shared_ptr<ore::data::Market>& market, const boost::shared_ptr<InfDkData>& data,
45 const std::string& configuration, const std::string& referenceCalibrationGrid,
46 const bool dontCalibrate)
47 : market_(market), configuration_(configuration), data_(data), referenceCalibrationGrid_(referenceCalibrationGrid),
48 dontCalibrate_(dontCalibrate) {
49
50 LOG("DkBuilder for " << data_->index());
51
52 if (!data_->calibrationBaskets().empty()) {
53 QL_REQUIRE(data_->calibrationBaskets().size() == 1, "If there is a calibration basket, expect exactly 1.");
54 const CalibrationBasket& cb = data_->calibrationBaskets()[0];
55 QL_REQUIRE(!cb.empty(), "If there is a calibration basket, expect it to be non-empty.");
56 optionActive_ = vector<bool>(cb.instruments().size(), false);
57 }
58
59 marketObserver_ = boost::make_shared<MarketObserver>();
60
61 // get market data
62 inflationIndex_ =
63 boost::dynamic_pointer_cast<ZeroInflationIndex>(*market_->zeroInflationIndex(data_->index(), configuration_));
64 QL_REQUIRE(inflationIndex_, "DkBuilder: requires ZeroInflationIndex, got " << data_->index());
65 rateCurve_ = market_->discountCurve(inflationIndex_->currency().code(), configuration_);
66 infVol_ = market_->cpiInflationCapFloorVolatilitySurface(data_->index(), configuration_);
67
68 // register with market observables except vols
69 marketObserver_->registerWith(inflationIndex_);
70 marketObserver_->registerWith(rateCurve_);
71
72 // register the builder with the market observer
73 registerWith(marketObserver_);
74 registerWith(infVol_);
75 // notify observers of all market data changes, not only when not calculated
76 alwaysForwardNotifications();
77
78 // build option basket and derive parametrization from it
79 const ReversionParameter& reversion = data_->reversion();
80 const VolatilityParameter& volatility = data_->volatility();
81 if (volatility.calibrate() || reversion.calibrate())
82 buildCapFloorBasket();
83
84 Array aTimes(volatility.times().begin(), volatility.times().end());
85 Array hTimes(reversion.times().begin(), reversion.times().end());
86 Array alpha(volatility.values().begin(), volatility.values().end());
87 Array h(reversion.values().begin(), reversion.values().end());
88
89 if (volatility.type() == ParamType::Constant) {
90 QL_REQUIRE(volatility.times().size() == 0, "empty alpha times expected");
91 QL_REQUIRE(volatility.values().size() == 1, "initial alpha array should have size 1");
92 } else if (volatility.type() == ParamType::Piecewise) {
93 if (volatility.calibrate() && data_->calibrationType() == CalibrationType::Bootstrap) {
94 if (volatility.times().size() > 0) {
95 DLOG("overriding alpha time grid with option expiries");
96 }
97 QL_REQUIRE(optionExpiries_.size() > 0, "empty option expiries");
98 aTimes = Array(optionExpiries_.begin(), optionExpiries_.end() - 1);
99 alpha = Array(aTimes.size() + 1, volatility.values()[0]);
100 } else { // use input time grid and input alpha array otherwise
101 QL_REQUIRE(alpha.size() == aTimes.size() + 1, "alpha grids do not match");
102 }
103 } else {
104 QL_FAIL("Expected DK model data volatility parameter to be Constant or Piecewise.");
105 }
106
107 if (reversion.type() == ParamType::Constant) {
108 QL_REQUIRE(reversion.values().size() == 1, "reversion grid size 1 expected");
109 QL_REQUIRE(reversion.times().size() == 0, "empty reversion time grid expected");
110 } else if (reversion.type() == ParamType::Piecewise) {
111 if (reversion.calibrate() && data_->calibrationType() == CalibrationType::Bootstrap) {
112 if (reversion.times().size() > 0) {
113 DLOG("overriding H time grid with option expiries");
114 }
115 QL_REQUIRE(optionExpiries_.size() > 0, "empty option expiries");
116 hTimes = Array(optionExpiries_.begin(), optionExpiries_.end() - 1);
117 h = Array(hTimes.size() + 1, reversion.values()[0]);
118 } else { // use input time grid and input alpha array otherwise
119 QL_REQUIRE(h.size() == hTimes.size() + 1, "H grids do not match");
120 }
121 } else {
122 QL_FAIL("Expected DK model data reversion parameter to be Constant or Piecewise.");
123 }
124
125 DLOG("before calibration: alpha times = " << aTimes << " values = " << alpha);
126 DLOG("before calibration: h times = " << hTimes << " values = " << h)
127
128 if (reversion.reversionType() == LgmData::ReversionType::HullWhite &&
129 volatility.volatilityType() == LgmData::VolatilityType::HullWhite) {
130 DLOG("INF parametrization: InfDkPiecewiseConstantHullWhiteAdaptor");
131 parametrization_ = boost::make_shared<InfDkPiecewiseConstantHullWhiteAdaptor>(
132 inflationIndex_->currency(), inflationIndex_->zeroInflationTermStructure(), aTimes, alpha, hTimes, h,
133 data_->index());
134 } else if (reversion.reversionType() == LgmData::ReversionType::HullWhite) {
135 DLOG("INF parametrization for " << data_->index() << ": InfDkPiecewiseConstant");
136 parametrization_ = boost::make_shared<InfDkPiecewiseConstantParametrization>(
137 inflationIndex_->currency(), inflationIndex_->zeroInflationTermStructure(), aTimes, alpha, hTimes, h,
138 data_->index());
139 } else {
140 parametrization_ = boost::make_shared<InfDkPiecewiseLinearParametrization>(
141 inflationIndex_->currency(), inflationIndex_->zeroInflationTermStructure(), aTimes, alpha, hTimes, h,
142 data_->index());
143 DLOG("INF parametrization for " << data_->index() << ": InfDkPiecewiseLinear");
144 }
145
146 DLOG("alpha times size: " << aTimes.size());
147 DLOG("lambda times size: " << hTimes.size());
148
149 DLOG("Apply shift horizon and scale");
150 Time horizon = data_->reversionTransformation().horizon();
151 Real scaling = data_->reversionTransformation().scaling();
152 QL_REQUIRE(horizon >= 0.0, "shift horizon must be non negative");
153 QL_REQUIRE(scaling > 0.0, "scaling must be positive");
154
155 if (horizon > 0.0) {
156 DLOG("Apply shift horizon " << horizon << " to the " << data_->index() << " DK model");
157 parametrization_->shift() = horizon;
158 }
159
160 if (scaling != 1.0) {
161 DLOG("Apply scaling " << scaling << " to the " << data_->index() << " DK model");
162 parametrization_->scaling() = scaling;
163 }
164}

176bool InfDkBuilder::requiresRecalibration() const {
177 return (data_->volatility().calibrate() || data_->reversion().calibrate()) &&
178 (volSurfaceChanged(false) || marketObserver_->hasUpdated(false) || forceCalibration_);

181void InfDkBuilder::performCalculations() const {
182 if (requiresRecalibration()) {
183 // reset market observer updated flag
184 marketObserver_->hasUpdated(true);
185 // build option basket
186 buildCapFloorBasket();
187 // update vol cache
188 volSurfaceChanged(true);
189 }

192Date InfDkBuilder::optionMaturityDate(const Size j) const {
193 Date today = Settings::instance().evaluationDate();
194 const auto& ci = data_->calibrationBaskets()[0].instruments();
195 QL_REQUIRE(j < ci.size(), "InfDkBuilder::optionMaturityDate(" << j << "): out of bounds, got " << ci.size()
196 << " calibration instruments");
197 auto cf = boost::dynamic_pointer_cast<CpiCapFloor>(ci.at(j));
198 QL_REQUIRE(cf, "InfDkBuilder::optionMaturityDate("
199 << j << "): expected CpiCapFloor calibration instruments, could not cast");
200 Date res = optionMaturity(cf->maturity(), inflationIndex_->fixingCalendar());
201 QL_REQUIRE(res > today, "expired calibration option expiry " << io::iso_date(res));
202 return res;
203}

215bool InfDkBuilder::volSurfaceChanged(const bool updateCache) const {
216 bool hasUpdated = false;
217
218 boost::shared_ptr<QuantExt::CPICapFloorEngine> engine;
219
220 bool isLogNormalVol = QuantExt::ZeroInflation::isCPIVolSurfaceLogNormal(infVol_.currentLink());
221 if (isLogNormalVol) {
222 engine = boost::make_shared<QuantExt::CPIBlackCapFloorEngine>(rateCurve_, infVol_);
223 } else {
224 engine = boost::make_shared<QuantExt::CPIBachelierCapFloorEngine>(rateCurve_, infVol_);
225 }
226
227 Calendar fixCalendar = inflationIndex_->fixingCalendar();
228 BusinessDayConvention bdc = infVol_->businessDayConvention();
229 Date baseDate = inflationIndex_->zeroInflationTermStructure()->baseDate();
230 Real baseCPI = inflationIndex_->fixing(baseDate);
231 Period lag = infVol_->observationLag();
232
233 // if cache doesn't exist resize vector
234 if (infPriceCache_.size() != optionBasket_.size())
235 infPriceCache_ = vector<Real>(optionBasket_.size(), Null<Real>());
236
237 // Handle on calibration instruments. No checks this time.
238 const auto& ci = data_->calibrationBaskets()[0].instruments();
239
240 Real nominal = 1.0;
241 Size optionCounter = 0;
242 Date today = Settings::instance().evaluationDate();
243 for (Size j = 0; j < ci.size(); j++) {
244
245 if (!optionActive_[j])
246 continue;
247
248 auto cpiCapFloor = boost::dynamic_pointer_cast<CpiCapFloor>(ci[j]);
249 QL_REQUIRE(cpiCapFloor, "Expected CpiCapFloor calibration instruments in DK inflation model data.");
250
251 Date expiryDate = optionMaturityDate(j);
252 auto strikeValue = optionStrikeValue(j);
253
254 Option::Type capfloor = cpiCapFloor->type() == CapFloor::Cap ? Option::Call : Option::Put;
255
256 boost::shared_ptr<CPICapFloor> h =
257 boost::make_shared<CPICapFloor>(capfloor, nominal, today, baseCPI, expiryDate, fixCalendar, bdc,
258 fixCalendar, bdc, strikeValue, inflationIndex_, lag);
259 h->setPricingEngine(engine);
260 Real price = h->NPV();
261 if (!close_enough(infPriceCache_[optionCounter], price)) {
262 if (updateCache)
263 infPriceCache_[optionCounter] = price;
264 hasUpdated = true;
265 }
266 optionCounter++;
267 }
268 return hasUpdated;
269}

271void InfDkBuilder::buildCapFloorBasket() const {
272
273 // Checks that there is a basket.
274 QL_REQUIRE(!data_->calibrationBaskets().empty(), "No calibration basket provided in inflation DK model data.");
275 const CalibrationBasket& cb = data_->calibrationBaskets()[0];
276 const auto& ci = cb.instruments();
277
278 QL_REQUIRE(ci.size() == optionActive_.size(), "Expected the option active vector size to equal the "
279 << "number of calibration instruments");
280 fill(optionActive_.begin(), optionActive_.end(), false);
281
282 DLOG("build reference date grid '" << referenceCalibrationGrid_ << "'");
283 Date lastRefCalDate = Date::minDate();
284 std::vector<Date> referenceCalibrationDates;
285 if (!referenceCalibrationGrid_.empty())
286 referenceCalibrationDates = DateGrid(referenceCalibrationGrid_).dates();
287
288 boost::shared_ptr<QuantExt::CPICapFloorEngine> engine;
289
290 bool isLogNormalVol = QuantExt::ZeroInflation::isCPIVolSurfaceLogNormal(infVol_.currentLink());
291 if (isLogNormalVol) {
292 engine = boost::make_shared<QuantExt::CPIBlackCapFloorEngine>(rateCurve_, infVol_);
293 } else {
294 engine = boost::make_shared<QuantExt::CPIBachelierCapFloorEngine>(rateCurve_, infVol_);
295 }
296
297 Calendar fixCalendar = inflationIndex_->fixingCalendar();
298 Date baseDate = inflationIndex_->zeroInflationTermStructure()->baseDate();
299 Real baseCPI = dontCalibrate_ ? 100. : inflationIndex_->fixing(baseDate);
300 BusinessDayConvention bdc = infVol_->businessDayConvention();
301 Period lag = infVol_->observationLag();
302 Handle<ZeroInflationIndex> hIndex(inflationIndex_);
303 Date startDate = Settings::instance().evaluationDate();
304 bool useInterpolatedCPIFixings = infVol_->indexIsInterpolated();
305 Real nominal = 1.0;
306 vector<Time> expiryTimes;
307 optionBasket_.clear();
308 for (Size j = 0; j < ci.size(); j++) {
309
310 auto cpiCapFloor = boost::dynamic_pointer_cast<CpiCapFloor>(ci[j]);
311 QL_REQUIRE(cpiCapFloor, "Expected CpiCapFloor calibration instruments in DK inflation model data.");
312
313 Date expiryDate = optionMaturityDate(j);
314
315 // check if we want to keep the helper when a reference calibration grid is given
316 auto refCalDate =
317 std::lower_bound(referenceCalibrationDates.begin(), referenceCalibrationDates.end(), expiryDate);
318 if (refCalDate == referenceCalibrationDates.end() || *refCalDate > lastRefCalDate) {
319 Real strikeValue = optionStrikeValue(j);
320 Option::Type capfloor = cpiCapFloor->type() == CapFloor::Cap ? Option::Call : Option::Put;
321 boost::shared_ptr<CPICapFloor> cf =
322 boost::make_shared<CPICapFloor>(capfloor, nominal, startDate, baseCPI, expiryDate, fixCalendar, bdc,
323 fixCalendar, bdc, strikeValue, inflationIndex_, lag);
324 cf->setPricingEngine(engine);
325 Real tte = inflationYearFraction(inflationIndex_->frequency(), useInterpolatedCPIFixings,
326 inflationIndex_->zeroInflationTermStructure()->dayCounter(), baseDate,
327 cf->fixingDate());
328
329 Real tteFromBase = infVol_->timeFromBase(expiryDate);
330
331 Real marketPrem = dontCalibrate_ || tte <= 0 || tteFromBase <= 0 ? 0.01 : cf->NPV();
332 boost::shared_ptr<QuantExt::CpiCapFloorHelper> helper =
333 boost::make_shared<QuantExt::CpiCapFloorHelper>(capfloor, baseCPI, expiryDate, fixCalendar, bdc,
334 fixCalendar, bdc, strikeValue, hIndex, lag, marketPrem);
335
336 // we might produce duplicate expiry times even if the fixing dates are all different
337 if (tte > 0 && tteFromBase >= 0 && std::find_if(expiryTimes.begin(), expiryTimes.end(), [tte](Real x) {
338 return QuantLib::close_enough(x, tte);
339 }) == expiryTimes.end()) {
340 optionBasket_.push_back(helper);
341 helper->performCalculations();
342 expiryTimes.push_back(tte);
343 DLOG("Added InflationOptionHelper index="
344 << data_->index() << ", type=" << (capfloor == Option::Type::Call ? "Cap" : "Floor") << ", expiry="
345 << QuantLib::io::iso_date(expiryDate) << ", baseCPI=" << baseCPI << ", strike=" << strikeValue
346 << ", lag=" << lag << ", marketPremium=" << marketPrem << ", tte=" << tte);
347 optionActive_[j] = true;
348 if (refCalDate != referenceCalibrationDates.end())
349 lastRefCalDate = *refCalDate;
350 } else {
351 if (data_->ignoreDuplicateCalibrationExpiryTimes()) {
352 DLOG("Skipped InflationOptionHelper index="
353 << data_->index() << ", type=" << (capfloor == Option::Type::Call ? "Cap" : "Floor")
354 << ", expiry=" << QuantLib::io::iso_date(expiryDate) << ", baseCPI=" << baseCPI
355 << ", strike=" << strikeValue << ", lag=" << lag << ", marketPremium=" << marketPrem
356 << ", tte=" << tte << " since we already have a helper with the same expiry time.");
357 } else {
358 QL_FAIL("InfDkBuilder: a CPI cap floor calibration instrument with the expiry time, "
359 << tte << ", was already added.");
360 }
361 }
362 }
363 }
364
365 std::sort(expiryTimes.begin(), expiryTimes.end());
366 auto itExpiryTime = unique(expiryTimes.begin(), expiryTimes.end());
367 expiryTimes.resize(distance(expiryTimes.begin(), itExpiryTime));
368
369 optionExpiries_ = Array(expiryTimes.size());
370 for (Size j = 0; j < expiryTimes.size(); j++)
371 optionExpiries_[j] = expiryTimes[j];
372}
