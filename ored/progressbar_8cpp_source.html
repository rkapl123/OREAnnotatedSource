53 void SimpleProgressBar::updateProgress(const unsigned long progress, const unsigned long total) {
54  if (finalized_)
55  return;
56  if (progress >= total) {
57  std::cout << "\r" << std::setw(messageWidth_) << std::left << message_;
58  for (unsigned int i = 0; i < barWidth_; ++i)
59  std::cout << " ";
60  std::cout << " \r";
61  std::cout << std::setw(messageWidth_) << std::left << message_;
62  std::cout.flush();
63  finalized_ = true;
64  return;
65  }
66  if (updateCounter_ > 0 && progress * numberOfScreenUpdates_ < updateCounter_ * total) {
67  return;
68  }
69  std::cout << "\r" << std::setw(messageWidth_) << std::left << message_ << "[";
70  double ratio = static_cast<double>(progress) / static_cast<double>(total);
71  unsigned int pos = static_cast<unsigned int>(static_cast<double>(barWidth_) * ratio);
72  for (unsigned int i = 0; i < barWidth_; ++i) {
73  if (i < pos)
74  std::cout << "=";
75  else if (i == pos && pos != 0)
76  std::cout << ">";
77  else
78  std::cout << " ";
79  }
80  std::cout << "] " << static_cast<int>(ratio * 100.0) << " %\r";
81  std::cout.flush();
82  updateCounter_++;
83 }
