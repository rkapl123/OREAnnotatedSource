36 GenericYieldVolCurve::GenericYieldVolCurve(
37  const Date& asof, const Loader& loader, const boost::shared_ptr<GenericYieldVolatilityCurveConfig>& config,
38  const map<string, boost::shared_ptr<SwapIndex>>& requiredSwapIndices,
39  const std::function<bool(const boost::shared_ptr<MarketDatum>& md, Period& expiry, Period& term)>& matchAtmQuote,
40  const std::function<bool(const boost::shared_ptr<MarketDatum>& md, Period& expiry, Period& term, Real& strike)>&
41  matchSmileQuote,
42  const std::function<bool(const boost::shared_ptr<MarketDatum>& md, Period& term)>& matchShiftQuote) {
43 
44  try {
45 
46  // We loop over all market data, looking for quotes that match the configuration
47  // until we found the whole matrix or do not have more quotes in the market data
48 
49  MarketDatum::QuoteType volatilityType;
50  switch (config->volatilityType()) {
51  case GenericYieldVolatilityCurveConfig::VolatilityType::Lognormal:
52  volatilityType = MarketDatum::QuoteType::RATE_LNVOL;
53  break;
54  case GenericYieldVolatilityCurveConfig::VolatilityType::Normal:
55  volatilityType = MarketDatum::QuoteType::RATE_NVOL;
56  break;
57  case GenericYieldVolatilityCurveConfig::VolatilityType::ShiftedLognormal:
58  volatilityType = MarketDatum::QuoteType::RATE_SLNVOL;
59  break;
60  default:
61  QL_FAIL("unexpected volatility type");
62  }
63  bool isSln = volatilityType == MarketDatum::QuoteType::RATE_SLNVOL;
64  Matrix vols(config->optionTenors().size(), config->underlyingTenors().size(), Null<Real>());
65  Matrix shifts(isSln ? vols.rows() : 0, isSln ? vols.columns() : 0, Null<Real>());
66  Size quotesRead = 0, shiftQuotesRead = 0;
67  vector<Period> optionTenors = parseVectorOfValues<Period>(config->optionTenors(), &parsePeriod);
68  vector<Period> underlyingTenors = parseVectorOfValues<Period>(config->underlyingTenors(), &parsePeriod);
69 
70  for (auto& p : config->quotes()) {
71  // optional, because we do not require all spread quotes; we check below that we have all atm quotes
72  boost::shared_ptr<MarketDatum> md = loader.get(std::make_pair(p, true), asof);
73  if (md == nullptr)
74  continue;
75  Period expiry, term;
76  if (md->quoteType() == volatilityType && matchAtmQuote(md, expiry, term)) {
77  quotesRead++;
78  Size i = std::find(optionTenors.begin(), optionTenors.end(), expiry) - optionTenors.begin();
79  Size j = std::find(underlyingTenors.begin(), underlyingTenors.end(), term) - underlyingTenors.begin();
80  QL_REQUIRE(i < config->optionTenors().size(),
81  "expiry " << expiry << " not in configuration, this is unexpected");
82  QL_REQUIRE(j < config->underlyingTenors().size(),
83  "term " << term << " not in configuration, this is unexpected");
84  vols[i][j] = md->quote()->value();
85  }
86  if (isSln && md->quoteType() == MarketDatum::QuoteType::SHIFT && matchShiftQuote(md, term)) {
87  shiftQuotesRead++;
88  Size j = std::find(underlyingTenors.begin(), underlyingTenors.end(), term) - underlyingTenors.begin();
89  QL_REQUIRE(j < config->underlyingTenors().size(),
90  "term " << term << " not in configuration, this is unexpected");
91  for (Size i = 0; i < shifts.rows(); ++i)
92  shifts[i][j] = md->quote()->value();
93  }
94  }
95 
96  LOG("GenericYieldVolCurve: read " << quotesRead << " vols");
97 
98  // check we have found all requires values
99  bool haveAllAtmValues = true;
100  for (Size i = 0; i < config->optionTenors().size(); ++i) {
101  for (Size j = 0; j < config->underlyingTenors().size(); ++j) {
102  if (vols[i][j] == Null<Real>()) {
103  ALOG("missing ATM vol for " << config->optionTenors()[i] << " / " << config->underlyingTenors()[j]);
104  haveAllAtmValues = false;
105  }
106  if (isSln && shifts[i][j] == Null<Real>()) {
107  ALOG("missing shift for " << config->optionTenors()[i] << " / " << config->underlyingTenors()[j]);
108  haveAllAtmValues = false;
109  }
110  }
111  }
112  QL_REQUIRE(haveAllAtmValues, "Did not find all required quotes to build ATM surface");
113 
114  boost::shared_ptr<SwaptionVolatilityStructure> atm;
115 
116  QL_REQUIRE(quotesRead > 0,
117  "GenericYieldVolCurve: did not read any quotes, are option and swap tenors defined?");
118  if (quotesRead > 1) {
119  atm = boost::shared_ptr<SwaptionVolatilityStructure>(new SwaptionVolatilityMatrix(
120  asof, config->calendar(), config->businessDayConvention(), optionTenors, underlyingTenors, vols,
121  config->dayCounter(), config->flatExtrapolation(),
122  config->volatilityType() == GenericYieldVolatilityCurveConfig::VolatilityType::Normal
123  ? QuantLib::Normal
124  : QuantLib::ShiftedLognormal,
125  isSln ? shifts : Matrix(vols.rows(), vols.columns(), 0.0)));
126 
127  atm->enableExtrapolation(config->extrapolate());
128  TLOG("built atm surface with vols:");
129  TLOGGERSTREAM << vols;
130  if(isSln) {
131  TLOG("built atm surface with shifts:");
132  TLOGGERSTREAM << shifts;
133  }
134  } else {
135  // Constant volatility
136  atm = boost::shared_ptr<SwaptionVolatilityStructure>(new ConstantSwaptionVolatility(
137  asof, config->calendar(), config->businessDayConvention(), vols[0][0], config->dayCounter(),
138  config->volatilityType() == GenericYieldVolatilityCurveConfig::VolatilityType::Normal
139  ? QuantLib::Normal
140  : QuantLib::ShiftedLognormal,
141  !shifts.empty() ? shifts[0][0] : 0.0));
142  }
143 
144  if (config->dimension() == GenericYieldVolatilityCurveConfig::Dimension::ATM) {
145  // Nothing more to do
146  LOG("Returning ATM surface for config " << config->curveID());
147  vol_ = atm;
148  } else {
149  LOG("Building Cube for config " << config->curveID());
150  vector<Period> smileOptionTenors = parseVectorOfValues<Period>(config->smileOptionTenors(), &parsePeriod);
151  vector<Period> smileUnderlyingTenors =
152  parseVectorOfValues<Period>(config->smileUnderlyingTenors(), &parsePeriod);
153  vector<Spread> spreads = parseVectorOfValues<Real>(config->smileSpreads(), &parseReal);
154 
155  // add smile spread 0, if not already existent and sort the spreads
156  if (std::find_if(spreads.begin(), spreads.end(), [](const Real x) { return close_enough(x, 0.0); }) ==
157  spreads.end())
158  spreads.push_back(0.0);
159  std::sort(spreads.begin(), spreads.end());
160 
161  vector<vector<bool>> zero(smileOptionTenors.size() * smileUnderlyingTenors.size(),
162  std::vector<bool>(spreads.size(), true));
163 
164  if (smileOptionTenors.size() == 0)
165  smileOptionTenors = parseVectorOfValues<Period>(config->optionTenors(), &parsePeriod);
166  if (smileUnderlyingTenors.size() == 0)
167  smileUnderlyingTenors = parseVectorOfValues<Period>(config->underlyingTenors(), &parsePeriod);
168  QL_REQUIRE(spreads.size() > 0, "Need at least 1 strike spread for a SwaptionVolCube");
169 
170  Size n = smileOptionTenors.size() * smileUnderlyingTenors.size();
171  vector<vector<Handle<Quote>>> volSpreadHandles(n, vector<Handle<Quote>>(spreads.size()));
172  for (auto& i : volSpreadHandles)
173  for (auto& j : i)
174  j = Handle<Quote>(boost::make_shared<SimpleQuote>(0.0));
175 
176  LOG("vol cube smile option tenors " << smileOptionTenors.size());
177  LOG("vol cube smile swap tenors " << smileUnderlyingTenors.size());
178  LOG("vol cube strike spreads " << spreads.size());
179 
180  Size spreadQuotesRead = 0;
181  for (auto& p : config->quotes()) {
182  // optional because we do not require all spreads
183  // we default them to zero instead and post process them below
184  boost::shared_ptr<MarketDatum> md = loader.get(std::make_pair(p, true), asof);
185  if (md == nullptr)
186  continue;
187  Period expiry, term;
188  Real strike;
189  if (md->quoteType() == volatilityType && matchSmileQuote(md, expiry, term, strike)) {
190 
191  Size i = std::find(smileOptionTenors.begin(), smileOptionTenors.end(), expiry) -
192  smileOptionTenors.begin();
193  Size j = std::find(smileUnderlyingTenors.begin(), smileUnderlyingTenors.end(), term) -
194  smileUnderlyingTenors.begin();
195  // In the MarketDatum we call it a strike, but it's really a spread
196  Size k = std::find(spreads.begin(), spreads.end(), strike) - spreads.begin();
197  QL_REQUIRE(i < smileOptionTenors.size(),
198  "expiry " << expiry << " not in configuration, this is unexpected");
199  QL_REQUIRE(j < smileUnderlyingTenors.size(),
200  "term " << term << " not in configuration, this is unexpected");
201  QL_REQUIRE(k < spreads.size(), "strike " << strike << " not in configuration, this is unexpected");
202 
203  spreadQuotesRead++;
204  // Assume quotes are absolute vols by strike so construct the vol spreads here
205  Volatility atmVol = atm->volatility(smileOptionTenors[i], smileUnderlyingTenors[j], 0.0);
206  volSpreadHandles[i * smileUnderlyingTenors.size() + j][k] =
207  Handle<Quote>(boost::make_shared<SimpleQuote>(md->quote()->value() - atmVol));
208  zero[i * smileUnderlyingTenors.size() + j][k] = close_enough(md->quote()->value(), 0.0);
209  }
210  }
211  LOG("Read " << spreadQuotesRead << " quotes for VolCube.");
212 
213  // post processing: extrapolate leftmost non-zero value flat to the left and overwrite
214  // zero values
215  for (Size i = 0; i < smileOptionTenors.size(); ++i) {
216  for (Size j = 0; j < smileUnderlyingTenors.size(); ++j) {
217  Real lastNonZeroValue = 0.0;
218  for (Size k = 0; k < spreads.size(); ++k) {
219  boost::shared_ptr<SimpleQuote> q = boost::dynamic_pointer_cast<SimpleQuote>(
220  *volSpreadHandles[i * smileUnderlyingTenors.size() + j][spreads.size() - 1 - k]);
221  QL_REQUIRE(q, "internal error: expected simple quote");
222  // do not overwrite vol spread for zero strike spread (ATM point)
223  if (zero[i * smileUnderlyingTenors.size() + j][spreads.size() - 1 - k] &&
224  !close_enough(spreads[spreads.size() - 1 - k], 0.0)) {
225  q->setValue(lastNonZeroValue);
226  WLOG("Overwrite vol spread for "
227  << config->curveID() << "/" << smileOptionTenors[i] << "/" << smileUnderlyingTenors[j]
228  << "/" << spreads[spreads.size() - 1 - k] << " with " << lastNonZeroValue
229  << " since market quote is zero");
230  }
231  // update last non-zero value
232  if (!zero[i * smileUnderlyingTenors.size() + j][spreads.size() - 1 - k]) {
233  lastNonZeroValue = q->value();
234  }
235  }
236  }
237  }
238 
239  // log vols
240  for (Size i = 0; i < smileOptionTenors.size(); ++i) {
241  for (Size j = 0; j < smileUnderlyingTenors.size(); ++j) {
242  ostringstream o;
243  for (Size k = 0; k < spreads.size(); ++k) {
244  o << volSpreadHandles[i * smileUnderlyingTenors.size() + j][k]->value() +
245  atm->volatility(smileOptionTenors[i], smileUnderlyingTenors[j], 0.0)
246  << " ";
247  }
248  DLOG("Vols for " << smileOptionTenors[i] << "/" << smileUnderlyingTenors[j] << ": " << o.str());
249  }
250  }
251 
252  // build swap indices
253  auto it = requiredSwapIndices.find(config->swapIndexBase());
254  QL_REQUIRE(it != requiredSwapIndices.end(), "Unable to find SwapIndex " << config->swapIndexBase());
255  boost::shared_ptr<SwapIndex> swapIndexBase = it->second;
256 
257  it = requiredSwapIndices.find(config->shortSwapIndexBase());
258  QL_REQUIRE(it != requiredSwapIndices.end(), "Unable to find SwapIndex " << config->shortSwapIndexBase());
259  boost::shared_ptr<SwapIndex> shortSwapIndexBase = it->second;
260 
261  bool vegaWeighedSmileFit = false; // TODO
262 
263  Handle<SwaptionVolatilityStructure> hATM(atm);
264  boost::shared_ptr<QuantExt::SwaptionVolCube2> cube = boost::make_shared<QuantExt::SwaptionVolCube2>(
265  hATM, smileOptionTenors, smileUnderlyingTenors, spreads, volSpreadHandles, swapIndexBase,
266  shortSwapIndexBase, vegaWeighedSmileFit, config->flatExtrapolation());
267  cube->enableExtrapolation();
268 
269  // Wrap it in a SwaptionVolCubeWithATM
270  vol_ = boost::make_shared<QuantExt::SwaptionVolCubeWithATM>(cube);
271  }
272 
273  } catch (std::exception& e) {
274  QL_FAIL("generic yield volatility curve building failed for curve " << config->curveID() << " on date "
275  << io::iso_date(asof) << ": " << e.what());
276  } catch (...) {
277  QL_FAIL("generic yield vol curve building failed: unknown error");
278  }
279 }
