49 string convertToString(const Real value) {
50  // We want to write out a double that conforms to xs:double, this means no
51  // scientific notation, so we check for really small numbers here and explicitly set
52  // to 16 decimal places
53  string result;
54  if (std::abs(value) < 1.0e-6) {
55  std::ostringstream obj1;
56  obj1.precision(16);
57  obj1 << std::fixed << value;
58  result = obj1.str();
59  } else {
60  // And here we just use boost::lexical_cast which is better
61  // at precision than std::to_string()
62  result = boost::lexical_cast<std::string>(value);
63  }
64  return result;
65 }

78 XMLDocument::XMLDocument(const string& fileName) : _doc(new rapidxml::xml_document<char>()), _buffer(NULL) {
79  // Need to load the entire file into memory to pass to doc.parse().
80  ifstream t(fileName.c_str());
81  QL_REQUIRE(t.is_open(), "Failed to open file " << fileName);
82  t.seekg(0, std::ios::end); // go to the end
83  Size length = static_cast<Size>(t.tellg()); // report location (this is the length)
84  QL_REQUIRE(length > 0, "File " << fileName << " is empty.");
85  t.seekg(0, std::ios::beg); // go back to the beginning
86  _buffer = new char[length + 1]; // allocate memory for a buffer of appropriate dimension
87  t.read(_buffer, length); // read the whole file into the buffer
88  _buffer[static_cast<int>(t.gcount())] = '\0';
89  t.close(); // close file handle
90  try {
91  _doc->parse<0>(_buffer);
92  } catch (rapidxml::parse_error& pe) {
93  string where(pe.where<char>(), 30); // limit to first 30 chars.
94  QL_FAIL("RapidXML Parse Error : " << pe.what() << ". where=" << where);
95  }
96 }

482  const vector<T>& values, const vector<string>& attrNames,
483  const vector<vector<string>>& attrs) {
484  QL_REQUIRE(attrNames.size() == attrs.size(),
485  "attrNames size (" << attrNames.size() << ") must match attrs size (" << attrs.size() << ")");

496  node->insert_node(0, c);
497  for (Size j = 0; j < attrs.size(); ++j) {
498  if (attrs[j][i] != "")

559  vector<T> vec;
560  rapidxml::xml_node<>* node = parent->first_node(names.c_str());
561  if (mandatory) {
562  QL_REQUIRE(node, "Error: No XML Node " << names << " found.");
563  }
564  if (node) {
565  for (rapidxml::xml_node<>* child = node->first_node(name.c_str()); child;
566  child = child->next_sibling(name.c_str())) {
567 
568  std::string vstr = getNodeValue(child);
569  vec.push_back(parser(vstr));
570 
571  for (Size i = 0; i < attrNames.size(); ++i) {
572  xml_attribute<>* attr = child->first_attribute(attrNames[i].c_str());
573  if (attr && attr->value())
574  ((vector<string>&)attrs[i]).push_back(attr->value());
575  else
576  ((vector<string>&)attrs[i]).push_back("");
577  }
578  }
579  }
580  return vec;
581 }
