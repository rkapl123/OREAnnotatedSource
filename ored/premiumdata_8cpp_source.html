28void PremiumData::fromXML(XMLNode* node) {
29
30 // support deprecated variant, where data is given in single fields under the root node
31
32 auto depr_amount_node = XMLUtils::getChildNode(node, "PremiumAmount");
33 auto depr_ccy_node = XMLUtils::getChildNode(node, "PremiumCurrency");
34 auto depr_date_node = XMLUtils::getChildNode(node, "PremiumPayDate");
35
36 if (depr_amount_node != nullptr) {
37 std::string amountStr = XMLUtils::getNodeValue(depr_amount_node);
38 if (!amountStr.empty()) {
39 double amount = parseReal(amountStr);
40 if (!close_enough(amount, 0.0)) {
41 QL_REQUIRE(depr_ccy_node, "PremiumAmount (" << amount << ") given, but no PremiumCurrency");
42 QL_REQUIRE(depr_date_node, "PremiumAmount (" << amount << ") given, but no PremiumPayDate");
43 std::string ccyStr = XMLUtils::getNodeValue(depr_ccy_node);
44 std::string dateStr = XMLUtils::getNodeValue(depr_date_node);
45 QL_REQUIRE(!ccyStr.empty(), "PremiumAmount (" << amount << ") given, but no PremiumCurrency");
46 QL_REQUIRE(!dateStr.empty(), "PremiumAmount (" << amount << ") given, but no PremiumPayDate");
47 premiumData_.emplace_back(amount, ccyStr, parseDate(dateStr));
48 }
49 }
50 }
51
52 // standard variant, data is given in Premium nodes under Premiums root node
53
54 if (auto p = XMLUtils::getChildNode(node, "Premiums")) {
55 QL_REQUIRE(premiumData_.empty(), "Single PremiumAmount and Premiums node are not allowed simultaneously. Move "
56 "the single premium to the Premiums node instead.");
57 for (auto n : XMLUtils::getChildrenNodes(p, "Premium")) {
58 PremiumDatum d;
59 d.amount = XMLUtils::getChildValueAsDouble(n, "Amount", true);
60 d.ccy = XMLUtils::getChildValue(n, "Currency", true);
61 d.payDate = parseDate(XMLUtils::getChildValue(n, "PayDate", true));
62 premiumData_.push_back(d);
63 }
64 }
65}

67XMLNode* PremiumData::toXML(XMLDocument& doc) {
68 XMLNode* node = doc.allocNode("Premiums");
69 for (auto const& d : premiumData_) {
70 XMLNode* p = XMLUtils::addChild(doc, node, "Premium");
71 XMLUtils::addChild(doc, p, "Amount", d.amount);
72 XMLUtils::addChild(doc, p, "Currency", d.ccy);
73 XMLUtils::addChild(doc, p, "PayDate", ore::data::to_string(d.payDate));
74 }
75 return node;
76}
