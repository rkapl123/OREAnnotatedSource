37//! Instrument Wrapper
38/*!
39 Wrap Instrument base class
40 Derived classes should
41 - store instrument "state" for each scenario
42 - adjust the instrument pricing formula to account for state
43
44 \ingroup tradedata
45*/
46class InstrumentWrapper {
47public:
48 InstrumentWrapper() : multiplier_(1.0) {}
49 InstrumentWrapper(const boost::shared_ptr<QuantLib::Instrument>& inst, const Real multiplier = 1.0,
50 const std::vector<boost::shared_ptr<QuantLib::Instrument>>& additionalInstruments =
51 std::vector<boost::shared_ptr<QuantLib::Instrument>>(),
52 const std::vector<Real>& additionalMultipliers = std::vector<Real>())
53 : instrument_(inst), multiplier_(multiplier), additionalInstruments_(additionalInstruments),
54 additionalMultipliers_(additionalMultipliers) {
55 QL_REQUIRE(additionalInstruments_.size() == additionalMultipliers_.size(),
56 "vector size mismatch, instruments (" << additionalInstruments_.size() << ") vs multipliers ("
57 << additionalMultipliers_.size() << ")");
58 }
59
60 virtual ~InstrumentWrapper() {}
61
62 //! Initialise with the given date grid
63 virtual void initialise(const std::vector<QuantLib::Date>& dates) = 0;
64
65 //! reset is called every time a new path is about to be priced.
66 /*! For path dependent Wrappers, this is when internal state should be reset
67 */
68 virtual void reset() = 0;
69
70 //! Return the NPV of this instrument
71 virtual QuantLib::Real NPV() const = 0;
72
73 //! Return the additional results of this instrument
74 virtual const std::map<std::string, boost::any>& additionalResults() const = 0;
75
76 QuantLib::Real additionalInstrumentsNPV() const {
77 Real npv = 0.0;
78 for (QuantLib::Size i = 0; i < additionalInstruments_.size(); ++i)
79 npv += additionalInstruments_[i]->NPV() * additionalMultipliers_[i];
80 return npv;
81 }
82
83 //! call update on enclosed instrument(s)
84 virtual void updateQlInstruments() {
85 // the instrument might contain nested lazy objects which we also want to be updated
86 instrument_->deepUpdate();
87 for (QuantLib::Size i = 0; i < additionalInstruments_.size(); ++i)
88 additionalInstruments_[i]->deepUpdate();
89 }
90
91 //! is it an Option?
92 virtual bool isOption() { return false; }
93
94 //! Inspectors
95 //@{
96 /*! The "QuantLib" instrument
97 Pass true if you trigger a calculation on the returned instrument and want to record
98 the timing for that calculation. If in doubt whether a calculation is triggered, pass false. */
99 boost::shared_ptr<QuantLib::Instrument> qlInstrument(const bool calculate = false) const {
100 if (calculate && instrument_ != nullptr) {
101 getTimedNPV(instrument_);
102 }
103 return instrument_;
104 }
105 /*! The multiplier */
106 const Real& multiplier() const { return multiplier_; }
107 /*! multiplier to be applied on top of multiplier(), e.g. -1 for short options */
108 virtual Real multiplier2() const { return 1.0; }
109 /*! additional instruments */
110 const std::vector<boost::shared_ptr<QuantLib::Instrument>>& additionalInstruments() const {
111 return additionalInstruments_;
112 }
113 /*! multipliers for additional instruments */
114 const std::vector<Real>& additionalMultipliers() const { return additionalMultipliers_; }
115 //@}
116
117 //! Get cumulative timing spent on pricing
118 boost::timer::nanosecond_type getCumulativePricingTime() const { return cumulativePricingTime_; }
119
120 //! Get number of pricings
121 std::size_t getNumberOfPricings() const { return numberOfPricings_; }
122
123 //! Reset pricing statistics
124 void resetPricingStats() const {
125 numberOfPricings_ = 0;
126 cumulativePricingTime_ = 0;
127 }
128
129protected:
130 boost::shared_ptr<QuantLib::Instrument> instrument_;
131 Real multiplier_;
132 std::vector<boost::shared_ptr<QuantLib::Instrument>> additionalInstruments_;
133 std::vector<Real> additionalMultipliers_;
134
135 // all NPV calls to be logged in the timings should go through this method
136 Real getTimedNPV(const boost::shared_ptr<QuantLib::Instrument>& instr) const {
137 if (instr == nullptr)
138 return 0.0;
139 if (instr->isCalculated() || instr->isExpired())
140 return instr->NPV();
141 boost::timer::cpu_timer timer_;
142 Real tmp = instr->NPV();
143 cumulativePricingTime_ += timer_.elapsed().wall;
144 ++numberOfPricings_;
145 return tmp;
146 }
147
148 mutable std::size_t numberOfPricings_ = 0;
149 mutable boost::timer::nanosecond_type cumulativePricingTime_ = 0;
150};

152//! Vanilla Instrument Wrapper
153/*!
154 Derived from InstrumentWrapper
155 Used for any non path-dependent trades
156
157 \ingroup tradedata
158*/
159class VanillaInstrument : public InstrumentWrapper {
160public:
161 VanillaInstrument(const boost::shared_ptr<QuantLib::Instrument>& inst, const Real multiplier = 1.0,
162 const std::vector<boost::shared_ptr<QuantLib::Instrument>>& additionalInstruments =
163 std::vector<boost::shared_ptr<QuantLib::Instrument>>(),
164 const std::vector<Real>& additionalMultipliers = std::vector<Real>())
165 : InstrumentWrapper(inst, multiplier, additionalInstruments, additionalMultipliers) {}
166
167 void initialise(const std::vector<QuantLib::Date>&) override{};
168 void reset() override {}
169
170 QuantLib::Real NPV() const override { return getTimedNPV(instrument_) * multiplier_ + additionalInstrumentsNPV(); }
171 const std::map<std::string, boost::any>& additionalResults() const override {
172 static std::map<std::string, boost::any> empty;
173 if (instrument_ == nullptr)
174 return empty;
175 getTimedNPV(instrument_);
176 return instrument_->additionalResults();
177 }
178};
