37 //! Instrument Wrapper
38 /*!
39  Wrap Instrument base class
40  Derived classes should
41  - store instrument "state" for each scenario
42  - adjust the instrument pricing formula to account for state
43 
44  \ingroup tradedata
45 */
46 class InstrumentWrapper {
47 public:
48  InstrumentWrapper() : multiplier_(1.0) {}
49  InstrumentWrapper(const boost::shared_ptr<QuantLib::Instrument>& inst, const Real multiplier = 1.0,
50  const std::vector<boost::shared_ptr<QuantLib::Instrument>>& additionalInstruments =
51  std::vector<boost::shared_ptr<QuantLib::Instrument>>(),
52  const std::vector<Real>& additionalMultipliers = std::vector<Real>())
53  : instrument_(inst), multiplier_(multiplier), additionalInstruments_(additionalInstruments),
54  additionalMultipliers_(additionalMultipliers) {
55  QL_REQUIRE(additionalInstruments_.size() == additionalMultipliers_.size(),
56  "vector size mismatch, instruments (" << additionalInstruments_.size() << ") vs multipliers ("
57  << additionalMultipliers_.size() << ")");
58  }

60  virtual ~InstrumentWrapper() {}

63  virtual void initialise(const std::vector<QuantLib::Date>& dates) = 0;

65  //! reset is called every time a new path is about to be priced.
66  /*! For path dependent Wrappers, this is when internal state should be reset
67  */
68  virtual void reset() = 0;

73  //! Return the additional results of this instrument
74  virtual const std::map<std::string, boost::any>& additionalResults() const = 0;

76  QuantLib::Real additionalInstrumentsNPV() const {
77  Real npv = 0.0;
78  for (QuantLib::Size i = 0; i < additionalInstruments_.size(); ++i)
79  npv += additionalInstruments_[i]->NPV() * additionalMultipliers_[i];
80  return npv;
81  }
82 
83  //! call update on enclosed instrument(s)
84  virtual void updateQlInstruments() {
85  // the instrument might contain nested lazy objects which we also want to be updated
86  instrument_->deepUpdate();
87  for (QuantLib::Size i = 0; i < additionalInstruments_.size(); ++i)
88  additionalInstruments_[i]->deepUpdate();
89  }
90 
91  //! is it an Option?
92  virtual bool isOption() { return false; }

94  //! Inspectors
95  //@{
96  /*! The "QuantLib" instrument
97  Pass true if you trigger a calculation on the returned instrument and want to record
98  the timing for that calculation. If in doubt whether a calculation is triggered, pass false. */
99  boost::shared_ptr<QuantLib::Instrument> qlInstrument(const bool calculate = false) const {
100  if (calculate && instrument_ != nullptr) {
101  getTimedNPV(instrument_);
102  }
103  return instrument_;
104  }
105  /*! The multiplier */
106  const Real& multiplier() const { return multiplier_; }
107  /*! additional instruments */
108  const std::vector<boost::shared_ptr<QuantLib::Instrument>>& additionalInstruments() const {
109  return additionalInstruments_;
110  }
111  /*! multipliers for additional instruments */
112  const std::vector<Real>& additionalMultipliers() const { return additionalMultipliers_; }
113  //@}
114 
115  //! Get cumulative timing spent on pricing
116  boost::timer::nanosecond_type getCumulativePricingTime() const { return cumulativePricingTime_; }
117 
118  //! Get number of pricings
119  std::size_t getNumberOfPricings() const { return numberOfPricings_; }
120 
121  //! Reset pricing statistics
122  void resetPricingStats() const {
123  numberOfPricings_ = 0;
124  cumulativePricingTime_ = 0;
125  }
126 
127 protected:
128  boost::shared_ptr<QuantLib::Instrument> instrument_;
129  Real multiplier_;
130  std::vector<boost::shared_ptr<QuantLib::Instrument>> additionalInstruments_;
131  std::vector<Real> additionalMultipliers_;
132 
133  // all NPV calls to be logged in the timings should go through this method
134  Real getTimedNPV(const boost::shared_ptr<QuantLib::Instrument>& instr) const {
135  if(instr == nullptr)
136  return 0.0;
137  if (instr->isCalculated() || instr->isExpired())
138  return instr->NPV();
139  boost::timer::cpu_timer timer_;
140  Real tmp = instr->NPV();
141  cumulativePricingTime_ += timer_.elapsed().wall;
142  ++numberOfPricings_;
143  return tmp;
144  }
145 
146  mutable std::size_t numberOfPricings_ = 0;
147  mutable boost::timer::nanosecond_type cumulativePricingTime_ = 0;
148 };

157 class VanillaInstrument : public InstrumentWrapper {
158 public:
159  VanillaInstrument(const boost::shared_ptr<QuantLib::Instrument>& inst, const Real multiplier = 1.0,
160  const std::vector<boost::shared_ptr<QuantLib::Instrument>>& additionalInstruments =
161  std::vector<boost::shared_ptr<QuantLib::Instrument>>(),
162  const std::vector<Real>& additionalMultipliers = std::vector<Real>())
163  : InstrumentWrapper(inst, multiplier, additionalInstruments, additionalMultipliers) {}
164 
165  void initialise(const std::vector<QuantLib::Date>&) override{};
166  void reset() override {}
167 
168  QuantLib::Real NPV() const override { return getTimedNPV(instrument_) * multiplier_ + additionalInstrumentsNPV(); }
169  const std::map<std::string, boost::any>& additionalResults() const override {
170  static std::map<std::string, boost::any> empty;
171  if(instrument_ == nullptr)
172  return empty;
173  getTimedNPV(instrument_);
174  return instrument_->additionalResults();
175  }
