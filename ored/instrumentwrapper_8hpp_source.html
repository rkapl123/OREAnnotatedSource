34 //! Instrument Wrapper
35 /*!
36  Wrap Instrument base class
37  Derived classes should
38  - store instrument "state" for each scenario
39  - adjust the instrument pricing formula to account for state
40 
41  \ingroup tradedata
42 */
43 class InstrumentWrapper {
44 public:
45  InstrumentWrapper() : multiplier_(1.0) {}
46  InstrumentWrapper(const boost::shared_ptr<QuantLib::Instrument>& inst, const Real multiplier = 1.0,
47  const std::vector<boost::shared_ptr<QuantLib::Instrument>>& additionalInstruments =
48  std::vector<boost::shared_ptr<QuantLib::Instrument>>(),
49  const std::vector<Real>& additionalMultipliers = std::vector<Real>())
50  : instrument_(inst), multiplier_(multiplier), additionalInstruments_(additionalInstruments),
51  additionalMultipliers_(additionalMultipliers) {
52  QL_REQUIRE(additionalInstruments_.size() == additionalMultipliers_.size(),
53  "vector size mismatch, instruments (" << additionalInstruments_.size() << ") vs multipliers ("
54  << additionalMultipliers_.size() << ")");
55  }
56 
57  virtual ~InstrumentWrapper() {}

59  //! Initialise with the given date grid
60  virtual void initialise(const std::vector<QuantLib::Date>& dates) = 0;
61 
62  //! reset is called every time a new path is about to be priced.
63  /*! For path dependent Wrappers, this is when internal state should be reset
64  */
65  virtual void reset() = 0;
66 
67  //! Return the NPV of this instrument
68  virtual QuantLib::Real NPV() const = 0;
69 
70  QuantLib::Real additionalInstrumentsNPV() const {
71  Real npv = 0.0;
72  for (QuantLib::Size i = 0; i < additionalInstruments_.size(); ++i)
73  npv += additionalInstruments_[i]->NPV() * additionalMultipliers_[i];
74  return npv;
75  }
76 
77  //! call update on enclosed instrument(s)
78  virtual void updateQlInstruments() {
79  // the instrument might contain nested lazy objects which we also want to be updated
80  instrument_->deepUpdate();
81  for (QuantLib::Size i = 0; i < additionalInstruments_.size(); ++i)
82  additionalInstruments_[i]->deepUpdate();
83  }
84 
85  //! is it an Option?
86  virtual bool isOption() { return false; }
87 
88  //! Inspectors
89  //@{
90  /*! The "QuantLib" instrument */
91  boost::shared_ptr<QuantLib::Instrument> qlInstrument() const { return instrument_; }
92  /*! The multiplier */
93  const Real& multiplier() const { return multiplier_; }
94  /*! additional instruments */
95  const std::vector<boost::shared_ptr<QuantLib::Instrument>>& additionalInstruments() const {
96  return additionalInstruments_;
97  }
98  /*! mutlipliers for additional instruments */
99  const std::vector<Real>& additionalMultipliers() const { return additionalMultipliers_; }
100  //@}
101 
102 protected:
103  boost::shared_ptr<QuantLib::Instrument> instrument_;
104  Real multiplier_;
105  std::vector<boost::shared_ptr<QuantLib::Instrument>> additionalInstruments_;
106  std::vector<Real> additionalMultipliers_;
107 };
