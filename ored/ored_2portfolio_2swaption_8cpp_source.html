61 void Swaption::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
62 
63  // build underlying swap and copy its required fixings
64 
65  DLOG("Swaption::build() for " << id() << ": build underlying swap");
66 
67  underlying_ = boost::make_shared<ore::data::Swap>(Envelope(), legData_);
68  underlying_->build(engineFactory);
69  requiredFixings_.addData(underlying_->requiredFixings());
70 
71  // build the exercise and parse some fields
72 
73  DLOG("Swaption::build() for " << id() << ": build exercise");
74 
75  exerciseBuilder_ = boost::make_shared<ExerciseBuilder>(optionData_, underlying_->legs());
76 
77  exerciseType_ = parseExerciseType(optionData_.style());
78  settlementType_ = parseSettlementType(optionData_.settlement());
79  settlementMethod_ = optionData_.settlementMethod() == "" ? defaultSettlementMethod(settlementType_)
80  : parseSettlementMethod(optionData_.settlementMethod());
81  positionType_ = parsePositionType(optionData_.longShort());
82 
83  // determine the type: isCrossCcy, isOis, isBma, isStandard
84 
85  std::set<std::string> legTypes;
86  for (auto const& l : legData_) {
87  legTypes.insert(l.legType());
88  }
89 
90  bool isCrossCcy = false;
91  for (Size i = 1; i < legData_.size(); ++i) {
92  isCrossCcy = isCrossCcy || legData_[0].currency() != legData_[1].currency();
93  }
94 
95  bool isOis = false, isBma = false;
96  for (auto const& l : legData_) {
97  if (auto fld = boost::dynamic_pointer_cast<FloatingLegData>(l.concreteLegData())) {
98  auto ind =
99  *engineFactory->market()->iborIndex(fld->index(), engineFactory->configuration(MarketContext::pricing));
100  isOis = isOis || (boost::dynamic_pointer_cast<OvernightIndex>(ind) != nullptr);
101  isBma = isBma || (boost::dynamic_pointer_cast<QuantExt::BMAIndexWrapper>(ind) != nullptr);
102  }
103  }
104 
105  bool isStandard = true;
106  isStandard = isStandard && (legData_.size() == 2 && legTypes == std::set<std::string>{"Fixed", "Floating"} &&
107  !isOis && !isBma && !isCrossCcy && legData_[0].isPayer() != legData_[1].isPayer());
108  Real notional = Null<Real>(), fixedRate = Null<Real>(), gearing = Null<Real>(), spread = Null<Real>();
109  for (auto const& l : underlying_->legs()) {
110  for (auto const& c : l) {
111  if (auto cpn = boost::dynamic_pointer_cast<Coupon>(c)) {
112  // only constant notional allowed
113  if (notional == Null<Real>())
114  notional = cpn->nominal();
115  else
116  isStandard = isStandard && close_enough(notional, cpn->nominal());
117  if (auto f = boost::dynamic_pointer_cast<FixedRateCoupon>(c)) {
118  // only constant fixed rate allowed
119  if (fixedRate == Null<Real>())
120  fixedRate = f->rate();
121  else
122  isStandard = isStandard && close_enough(fixedRate, f->rate());
123  } else if (auto f = boost::dynamic_pointer_cast<IborCoupon>(c)) {
124  // in arrears not allowed
125  isStandard = isStandard && !f->isInArrears();
126  // only constant spread allowed
127  if (spread == Null<Real>())
128  spread = f->spread();
129  else
130  isStandard = isStandard && close_enough(spread, f->spread());
131  // only constant gearing allowed
132  if (gearing == Null<Real>())
133  gearing = f->gearing();
134  else
135  isStandard = isStandard && close_enough(gearing, f->gearing());
136  } else {
137  // no other coupon types allowed
138  isStandard = false;
139  }
140  } else {
141  // no cashflows != coupon allowed
142  isStandard = false;
143  }
144  }
145  }
146 
147  // check for zero notional, this is not handled well in the European engine, so we switch to Bermudan in this case
148 
149  if (isStandard && close_enough(notional, 0.0)) {
150  DLOG("Swaption::build() found zero notional, set isStandard := false to ensure valid pricing");
151  isStandard = false;
152  }
153 
154  DLOG("Swaption::build() for " << id() << ": type: isCrossCcy = " << std::boolalpha << isCrossCcy
155  << ", isOis = " << isOis << ", isBma = " << isBma << ", isStandard = " << isStandard);
156 
157  // we do not support xccy swaptions currently
158 
159  QL_REQUIRE(!isCrossCcy, "Cross Currency Swaptions are not supported at the moment.");
160 
161  // fill currencies and set notional to null (will be retrieved via notional())
162 
163  npvCurrency_ = notionalCurrency_ = "USD"; // only if no legs are given, not relevant in this case
164 
165  if (!legData_.empty()) {
166  npvCurrency_ = notionalCurrency_ = legData_[0].currency();
167  }
168 
169  notional_ = Null<Real>();
170 
171  // if we do not have an active exercise as of today, or no underlying legs we build an expired dummy instrument
172  // FIXME this should be handled in the engines, it's not critical though, since we always simulate into the future
173 
174  Date today = Settings::instance().evaluationDate();
175 
176  if (exerciseBuilder_->exercise() == nullptr || exerciseBuilder_->exercise()->dates().empty() ||
177  exerciseBuilder_->exercise()->dates().back() <= today || legData_.empty()) {
178  legs_ = {{boost::make_shared<QuantLib::SimpleCashFlow>(0.0, today)}};
179  instrument_ = boost::make_shared<VanillaInstrument>(
180  boost::make_shared<QuantLib::Swap>(legs_, std::vector<bool>(1, false)), 1.0,
181  std::vector<boost::shared_ptr<QuantLib::Instrument>>{}, std::vector<Real>{});
182  legCurrencies_ = {npvCurrency_};
183  legPayers_ = {false};
184  notional_ = Null<Real>();
185  maturity_ = today;
186  return;
187  }
188 
189  // fill legs, only include coupons after first exercise
190 
191  legCurrencies_ = underlying_->legCurrencies();
192  legPayers_ = underlying_->legPayers();
193  legs_.clear();
194  Date firstExerciseDate = exerciseBuilder_->exercise()->dates().front();
195  for (auto const& l : underlying_->legs()) {
196  legs_.push_back(Leg());
197  for (auto const& c : l) {
198  if (auto cpn = boost::dynamic_pointer_cast<Coupon>(c)) {
199  if (firstExerciseDate <= cpn->accrualStartDate()) {
200  legs_.back().push_back(c);
201  }
202  } else if (firstExerciseDate <= c->date()) {
203  legs_.back().push_back(c);
204  }
205  }
206  }
207 
208  // build a Euroepan or Bermudan swaption
209 
210  if (exerciseType_ == Exercise::European && isStandard)
211  buildEuropean(engineFactory);
212  else
213  buildBermudan(engineFactory);
214 }

216 void Swaption::buildEuropean(const boost::shared_ptr<EngineFactory>& engineFactory) {
217 
218  DLOG("Building European Swaption " << id());
219 
220  boost::shared_ptr<VanillaSwap> swap = buildVanillaSwap(engineFactory);
221 
222  boost::shared_ptr<EngineBuilder> builder = engineFactory->builder("EuropeanSwaption");
223  boost::shared_ptr<EuropeanSwaptionEngineBuilder> swaptionBuilder =
224  boost::dynamic_pointer_cast<EuropeanSwaptionEngineBuilder>(builder);
225  QL_REQUIRE(swaptionBuilder != nullptr, "internal error: could not cast to EuropeanSwaptionEngineBuilder");
226 
227  boost::shared_ptr<QuantLib::Swaption> swaption =
228  boost::make_shared<QuantLib::Swaption>(swap, exerciseBuilder_->exercise(), settlementType_, settlementMethod_);
229 
230  Currency ccy = parseCurrency(npvCurrency_);
231 
232  std::vector<boost::shared_ptr<Instrument>> additionalInstruments;
233  std::vector<Real> additionalMultipliers;
234  Date lastPremiumDate = addPremiums(additionalInstruments, additionalMultipliers, 1.0, optionData_.premiumData(),
235  positionType_ == Position::Long ? -1.0 : 1.0, ccy, engineFactory,
236  swaptionBuilder->configuration(MarketContext::pricing));
237 
238  swaption->setPricingEngine(
239  swaptionBuilder->engine(IndexNameTranslator::instance().oreName(swap->iborIndex()->name())));
240 
241  if (settlementType_ == Settlement::Physical) {
242  instrument_ = boost::make_shared<EuropeanOptionWrapper>(
243  swaption, positionType_ == Position::Long ? true : false, exerciseBuilder_->exercise()->dates().back(),
244  settlementType_ == Settlement::Physical ? true : false, swap, 1.0, 1.0, additionalInstruments,
245  additionalMultipliers);
246  maturity_ = std::max(swap->fixedSchedule().dates().back(), swap->floatingSchedule().dates().back());
247  } else {
248  instrument_ = boost::make_shared<VanillaInstrument>(swaption, positionType_ == Position::Long ? 1.0 : -1.0,
249  additionalInstruments, additionalMultipliers);
250  // Align with ISDA AANA/GRID guidance as of November 2020
251  // maturity_ = exDate;
252  maturity_ = std::max(swap->fixedSchedule().dates().back(), swap->floatingSchedule().dates().back());
253  }
254 
255  maturity_ = std::max(maturity_, lastPremiumDate);
256 
257  DLOG("Building European Swaption done");
258 }

260 void Swaption::buildBermudan(const boost::shared_ptr<EngineFactory>& engineFactory) {
261 
262  if (settlementType_ == Settlement::Physical)
263  maturity_ = underlying_->maturity();
264  else
265  maturity_ = exerciseBuilder_->noticeDates().back();
266 
267  if (settlementType_ == Settlement::Cash && settlementMethod_ == Settlement::ParYieldCurve)
268  WLOG("Cash-settled Bermudan Swaption (id = "
269  << id() << ") with ParYieldCurve settlement method not supported by Lgm engine. "
270  << "Approximate pricing using CollateralizedCashPrice pricing methodology");
271 
272  std::vector<QuantLib::Currency> ccys;
273  for (auto const& c : underlying_->legCurrencies())
274  ccys.push_back(parseCurrency(c));
275  auto swaption =
276  boost::make_shared<QuantExt::MultiLegOption>(underlying_->legs(), underlying_->legPayers(), ccys,
277  exerciseBuilder_->exercise(), settlementType_, settlementMethod_);
278 
279  // determine strikes for calibration basket (simple approach, a la summit)
280  // also determine the ibor index (if several, chose the first) to get the engine
281  std::vector<Real> strikes(exerciseBuilder_->noticeDates().size(), Null<Real>());
282  boost::shared_ptr<IborIndex> index;
283  for (Size i = 0; i < exerciseBuilder_->noticeDates().size(); ++i) {
284  Real firstFixedRate = Null<Real>();
285  Real firstFloatSpread = Null<Real>();
286  for (auto const& l : underlying_->legs()) {
287  for (auto const& c : l) {
288  if (auto cpn = boost::dynamic_pointer_cast<FixedRateCoupon>(c)) {
289  if (cpn->accrualStartDate() >= exerciseBuilder_->noticeDates()[i] && firstFixedRate == Null<Real>())
290  firstFixedRate = cpn->rate();
291  } else if (auto cpn = boost::dynamic_pointer_cast<FloatingRateCoupon>(c)) {
292  firstFloatSpread = cpn->spread();
293  if (index == nullptr) {
294  if (auto tmp = boost::dynamic_pointer_cast<IborIndex>(cpn->index())) {
295  DLOG("found ibor / ois index '" << tmp->name() << "'");
296  index = tmp;
297  } else if (auto tmp = boost::dynamic_pointer_cast<SwapIndex>(cpn->index())) {
298  DLOG("found cms index " << tmp->name() << ", use key '" << tmp->iborIndex()->name()
299  << "' to look up vol");
300  index = tmp->iborIndex();
301  }
302  }
303  }
304  }
305  }
306  if (firstFixedRate != Null<Real>()) {
307  strikes[i] = firstFixedRate;
308  if (firstFloatSpread != Null<Real>()) {
309  strikes[i] -= firstFloatSpread;
310  }
311  }
312  DLOG("calibration strike for ex date "
313  << QuantLib::io::iso_date(exerciseBuilder_->noticeDates()[i]) << " is "
314  << (strikes[i] == Null<Real>() ? "ATMF" : std::to_string(strikes[i])) << " (fixed rate "
315  << (firstFixedRate == Null<Real>() ? "NA" : std::to_string(firstFixedRate)) << ", spread "
316  << (firstFloatSpread == Null<Real>() ? "NA" : std::to_string(firstFloatSpread)) << ")");
317  }
318 
319  if (index == nullptr) {
320  DLOG("no ibor, ois, cms index found, use ccy key to look up vol");
321  }
322 
323  auto builder = engineFactory->builder("BermudanSwaption");
324  auto swaptionBuilder = boost::dynamic_pointer_cast<BermudanSwaptionEngineBuilder>(builder);
325  QL_REQUIRE(swaptionBuilder, "internal error: could not cast to BermudanSwaptionEngineBuilder");
326 
327  auto tmp = engineFactory->builder("Swap");
328  auto swapBuilder = boost::dynamic_pointer_cast<SwapEngineBuilderBase>(tmp);
329  QL_REQUIRE(swapBuilder, "internal error: could not cast to SwapEngineBuilder");
330 
331  cpu_timer timer;
332  // use ibor / ois index as key, if possible, otherwise the npv currency
333  auto swaptionEngine = swaptionBuilder->engine(
334  id(), index == nullptr ? npvCurrency_ : IndexNameTranslator::instance().oreName(index->name()),
335  exerciseBuilder_->noticeDates(), underlying_->maturity(), strikes);
336  timer.stop();
337  DLOG("Swaption model calibration time: " << timer.format(default_places, "%w") << " s");
338  swaption->setPricingEngine(swaptionEngine);
339 
340  auto swapEngine = swapBuilder->engine(parseCurrency(npvCurrency_));
341 
342  std::vector<boost::shared_ptr<Instrument>> underlyingSwaps =
343  buildUnderlyingSwaps(swapEngine, exerciseBuilder_->noticeDates());
344 
345  // If premium data is provided
346  // 1) build the fee trade and pass it to the instrument wrapper for pricing
347  // 2) add fee payment as additional trade leg for cash flow reporting
348  std::vector<boost::shared_ptr<Instrument>> additionalInstruments;
349  std::vector<Real> additionalMultipliers;
350  Real multiplier = positionType_ == Position::Long ? 1.0 : -1.0;
351  Date lastPremiumDate =
352  addPremiums(additionalInstruments, additionalMultipliers, 1.0, optionData_.premiumData(), -multiplier,
353  parseCurrency(npvCurrency_), engineFactory, swaptionBuilder->configuration(MarketContext::pricing));
354 
355  // instrument_ = boost::shared_ptr<InstrumentWrapper> (new VanillaInstrument (swaption, multiplier));
356  instrument_ = boost::make_shared<BermudanOptionWrapper>(
357  swaption, positionType_ == Position::Long ? true : false, exerciseBuilder_->noticeDates(),
358  settlementType_ == Settlement::Physical ? true : false, underlyingSwaps, 1.0, 1.0, additionalInstruments,
359  additionalMultipliers);
360 
361  maturity_ = std::max(maturity_, lastPremiumDate);
362 
363  DLOG("Building Bermudan Swaption done");
364 }

366 boost::shared_ptr<VanillaSwap> Swaption::buildVanillaSwap(const boost::shared_ptr<EngineFactory>& engineFactory) {
367 
368  Date exerciseDate = exerciseBuilder_->exercise()->dates().front();
369 
370  Size fixedLegIndex, floatingLegIndex;
371  if (legData_[0].legType() == "Floating" && legData_[1].legType() == "Fixed") {
372  floatingLegIndex = 0;
373  fixedLegIndex = 1;
374  } else if (legData_[1].legType() == "Floating" && legData_[0].legType() == "Fixed") {
375  floatingLegIndex = 1;
376  fixedLegIndex = 0;
377  } else {
378  QL_FAIL("Invalid leg types " << legData_[0].legType() << " + " << legData_[1].legType());
379  }
380  boost::shared_ptr<FixedLegData> fixedLegData =
381  boost::dynamic_pointer_cast<FixedLegData>(legData_[fixedLegIndex].concreteLegData());
382  boost::shared_ptr<FloatingLegData> floatingLegData =
383  boost::dynamic_pointer_cast<FloatingLegData>(legData_[floatingLegIndex].concreteLegData());
384 
385  boost::shared_ptr<EngineBuilder> tmp = engineFactory->builder("Swap");
386  boost::shared_ptr<SwapEngineBuilderBase> swapBuilder = boost::dynamic_pointer_cast<SwapEngineBuilderBase>(tmp);
387  QL_REQUIRE(swapBuilder, "No Swap Builder found for Swaption " << id());
388 
389  // Get Trade details
390  // We known that the notional, fixed rate, float spread, gearing are all constant in the underlying leg. To make
391  // sure we get the correct value from the leg data we take the _last_ value in the leg data vectors, because
392  // the first might contain a value that is different, but lies before the schedule start date.
393  string ccy = legData_[0].currency();
394  Currency currency = parseCurrency(ccy);
395  Real nominal = legData_[0].notionals().back();
396 
397  Real rate = fixedLegData->rates().back();
398  Real spread = floatingLegData->spreads().empty() ? 0.0 : floatingLegData->spreads().back();
399  string indexName = floatingLegData->index();
400 
401  Schedule fixedSchedule = makeSchedule(legData_[fixedLegIndex].schedule());
402  DayCounter fixedDayCounter = parseDayCounter(legData_[fixedLegIndex].dayCounter());
403 
404  Schedule floatingSchedule = makeSchedule(legData_[floatingLegIndex].schedule());
405  Handle<IborIndex> index =
406  engineFactory->market()->iborIndex(indexName, swapBuilder->configuration(MarketContext::pricing));
407  DayCounter floatingDayCounter = parseDayCounter(legData_[floatingLegIndex].dayCounter());
408 
409  BusinessDayConvention paymentConvention =
410  parseBusinessDayConvention(legData_[floatingLegIndex].paymentConvention());
411 
412  VanillaSwap::Type type = legData_[fixedLegIndex].isPayer() ? VanillaSwap::Payer : VanillaSwap::Receiver;
413 
414  // only take into account accrual periods with start date on or after first exercise date (if given)
415  std::vector<Date> fixDates = fixedSchedule.dates();
416  auto it1 = std::lower_bound(fixDates.begin(), fixDates.end(), exerciseDate);
417  fixDates.erase(fixDates.begin(), it1);
418  // check we have at least 1 to stop set fault on vector(fixDates.size() - 1, true) but maybe check should be 2
419  QL_REQUIRE(fixDates.size() >= 2, "Not enough schedule dates are left in Swaption fixed leg (check exercise dates)");
420  fixedSchedule = Schedule(fixDates, fixedSchedule.calendar(), Unadjusted, boost::none, boost::none, boost::none,
421  boost::none, std::vector<bool>(fixDates.size() - 1, true));
422  std::vector<Date> floatingDates = floatingSchedule.dates();
423  auto it2 = std::lower_bound(floatingDates.begin(), floatingDates.end(), exerciseDate);
424  floatingDates.erase(floatingDates.begin(), it2);
425  QL_REQUIRE(floatingDates.size() >= 2,
426  "Not enough schedule dates are left in Swaption floating leg (check exercise dates)");
427  floatingSchedule = Schedule(floatingDates, floatingSchedule.calendar(), Unadjusted, boost::none, boost::none,
428  boost::none, boost::none, std::vector<bool>(floatingDates.size() - 1, true));
429 
430  // Build a vanilla (bullet) swap underlying
431  boost::shared_ptr<VanillaSwap> swap =
432  boost::make_shared<VanillaSwap>(type, nominal, fixedSchedule, rate, fixedDayCounter, floatingSchedule, *index,
433  spread, floatingDayCounter, paymentConvention);
434 
435  swap->setPricingEngine(swapBuilder->engine(currency));
436  return swap;
437 }

439 std::vector<boost::shared_ptr<Instrument>>
440 Swaption::buildUnderlyingSwaps(const boost::shared_ptr<PricingEngine>& swapEngine,
441  const std::vector<Date>& exerciseDates) {
442  std::vector<boost::shared_ptr<Instrument>> swaps;
443  for (Size i = 0; i < exerciseDates.size(); ++i) {
444  std::vector<Leg> legs = underlying_->legs();
445  std::vector<bool> payer = underlying_->legPayers();
446  for (Size j = 0; j < legs.size(); ++j) {
447  Date ed = exerciseDates[i];
448  auto it = std::lower_bound(legs[j].begin(), legs[j].end(), exerciseDates[i],
449  [&ed](const boost::shared_ptr<CashFlow>& c, const Date& d) {
450  if (auto cpn = boost::dynamic_pointer_cast<Coupon>(c)) {
451  return cpn->accrualStartDate() < ed;
452  } else {
453  return c->date() < ed;
454  }
455  });
456  if (it != legs[j].begin())
457  --it;
458  legs[j].erase(legs[j].begin(), it);
459  }
460  auto newSwap = boost::make_shared<QuantLib::Swap>(legs, payer);
461  if (swapEngine != nullptr) {
462  newSwap->setPricingEngine(swapEngine);
463  }
464  swaps.push_back(newSwap);
465  for (auto const& l : legs) {
466  if (l.empty()) {
467  WLOG("Added empty leg to underlying swap for exercise " << QuantLib::io::iso_date(exerciseDates[i])
468  << "!");
469  } else {
470  Date d;
471  if (auto cpn = boost::dynamic_pointer_cast<Coupon>(l.front())) {
472  d = cpn->accrualStartDate();
473  } else {
474  d = l.front()->date();
475  }
476  DLOG("Added leg with start date " << QuantLib::io::iso_date(d) << " for exercise "
477  << QuantLib::io::iso_date(exerciseDates[i]));
478  }
479  }
480  }
481  return swaps;
482 }

492 const std::map<std::string, boost::any>& Swaption::additionalData() const {
493  // use the build time as of date to determine current notionals
494  Date asof = Settings::instance().evaluationDate();
495  for (Size i = 0; i < std::min(legData_.size(), legs_.size()); ++i) {
496  string legID = to_string(i + 1);
497  additionalData_["legType[" + legID + "]"] = legData_[i].legType();
498  additionalData_["isPayer[" + legID + "]"] = legData_[i].isPayer();
499  additionalData_["notionalCurrency[" + legID + "]"] = legData_[i].currency();
500  for (Size j = 0; j < legs_[i].size(); ++j) {
501  boost::shared_ptr<CashFlow> flow = legs_[i][j];
502  // pick flow with earliest future payment date on this leg
503  if (flow->date() > asof) {
504  additionalData_["amount[" + legID + "]"] = flow->amount();
505  additionalData_["paymentDate[" + legID + "]"] = to_string(flow->date());
506  boost::shared_ptr<Coupon> coupon = boost::dynamic_pointer_cast<Coupon>(flow);
507  if (coupon) {
508  additionalData_["currentNotional[" + legID + "]"] = coupon->nominal();
509  additionalData_["rate[" + legID + "]"] = coupon->rate();
510  boost::shared_ptr<FloatingRateCoupon> frc = boost::dynamic_pointer_cast<FloatingRateCoupon>(flow);
511  if (frc) {
512  additionalData_["index[" + legID + "]"] = frc->index()->name();
513  additionalData_["spread[" + legID + "]"] = frc->spread();
514  }
515  }
516  break;
517  }
518  }
519  if (legs_[i].size() > 0) {
520  boost::shared_ptr<Coupon> coupon = boost::dynamic_pointer_cast<Coupon>(legs_[i][0]);
521  if (coupon)
522  additionalData_["originalNotional[" + legID + "]"] = coupon->nominal();
523  }
524  }
525  return additionalData_;
