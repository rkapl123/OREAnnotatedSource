61void Swaption::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
62
63 // build underlying swap and copy its required fixings
64
65 DLOG("Swaption::build() for " << id() << ": build underlying swap");
66
67 underlying_ = boost::make_shared<ore::data::Swap>(Envelope(), legData_);
68 underlying_->build(engineFactory);
69 requiredFixings_.addData(underlying_->requiredFixings());
70
71 // build the exercise and parse some fields
72
73 DLOG("Swaption::build() for " << id() << ": build exercise");
74
75 exerciseBuilder_ = boost::make_shared<ExerciseBuilder>(optionData_, underlying_->legs());
76
77 exerciseType_ = parseExerciseType(optionData_.style());
78 settlementType_ = parseSettlementType(optionData_.settlement());
79 settlementMethod_ = optionData_.settlementMethod() == "" ? defaultSettlementMethod(settlementType_)
80 : parseSettlementMethod(optionData_.settlementMethod());
81 positionType_ = parsePositionType(optionData_.longShort());
82
83 // determine the type: isCrossCcy, isOis, isBma, isStandard
84
85 std::set<std::string> legTypes;
86 for (auto const& l : legData_) {
87 legTypes.insert(l.legType());
88 }
89
90 bool isCrossCcy = false;
91 for (Size i = 1; i < legData_.size(); ++i) {
92 isCrossCcy = isCrossCcy || legData_[0].currency() != legData_[1].currency();
93 }
94
95 bool isOis = false, isBma = false;
96 for (auto const& l : legData_) {
97 if (auto fld = boost::dynamic_pointer_cast<FloatingLegData>(l.concreteLegData())) {
98 auto ind =
99 *engineFactory->market()->iborIndex(fld->index(), engineFactory->configuration(MarketContext::pricing));
100 isOis = isOis || (boost::dynamic_pointer_cast<OvernightIndex>(ind) != nullptr);
101 isBma = isBma || (boost::dynamic_pointer_cast<QuantExt::BMAIndexWrapper>(ind) != nullptr);
102 }
103 }
104
105 bool isStandard = true;
106 isStandard = isStandard &&
107 (legData_.size() == 2 && legTypes == std::set<std::string>{"Fixed", "Floating"} && !isOis && !isBma &&
108 !isCrossCcy && legData_[0].isPayer() != legData_[1].isPayer() && optionData_.exerciseFees().empty());
109 Real notional = Null<Real>(), fixedRate = Null<Real>(), gearing = Null<Real>(), spread = Null<Real>();
110 for (auto const& l : underlying_->legs()) {
111 for (auto const& c : l) {
112 if (auto cpn = boost::dynamic_pointer_cast<Coupon>(c)) {
113 // only constant notional allowed
114 if (notional == Null<Real>())
115 notional = cpn->nominal();
116 else
117 isStandard = isStandard && close_enough(notional, cpn->nominal());
118 if (auto f = boost::dynamic_pointer_cast<FixedRateCoupon>(c)) {
119 // only constant fixed rate allowed
120 if (fixedRate == Null<Real>())
121 fixedRate = f->rate();
122 else
123 isStandard = isStandard && close_enough(fixedRate, f->rate());
124 } else if (auto f = boost::dynamic_pointer_cast<IborCoupon>(c)) {
125 // in arrears not allowed
126 isStandard = isStandard && !f->isInArrears();
127 // only constant spread allowed
128 if (spread == Null<Real>())
129 spread = f->spread();
130 else
131 isStandard = isStandard && close_enough(spread, f->spread());
132 // only constant gearing allowed
133 if (gearing == Null<Real>())
134 gearing = f->gearing();
135 else
136 isStandard = isStandard && close_enough(gearing, f->gearing());
137 } else {
138 // no other coupon types allowed
139 isStandard = false;
140 }
141 } else {
142 // no cashflows != coupon allowed
143 isStandard = false;
144 }
145 }
146 }
147
148 // check for zero notional, this is not handled well in the European engine, so we switch to Bermudan in this case
149
150 if (isStandard && close_enough(notional, 0.0)) {
151 DLOG("Swaption::build() found zero notional, set isStandard := false to ensure valid pricing");
152 isStandard = false;
153 }
154
155 DLOG("Swaption::build() for " << id() << ": type: isCrossCcy = " << std::boolalpha << isCrossCcy
156 << ", isOis = " << isOis << ", isBma = " << isBma << ", isStandard = " << isStandard);
157
158 // we do not support xccy swaptions currently
159
160 QL_REQUIRE(!isCrossCcy, "Cross Currency Swaptions are not supported at the moment.");
161
162 // fill currencies and set notional to null (will be retrieved via notional())
163
164 npvCurrency_ = notionalCurrency_ = "USD"; // only if no legs are given, not relevant in this case
165
166 if (!legData_.empty()) {
167 npvCurrency_ = notionalCurrency_ = legData_[0].currency();
168 }
169
170 notional_ = Null<Real>();
171
172 // if we do not have an active exercise as of today, or no underlying legs we build an expired dummy instrument
173 // FIXME this should be handled in the engines, it's not critical though, since we always simulate into the future
174
175 Date today = Settings::instance().evaluationDate();
176
177 if (exerciseBuilder_->exercise() == nullptr || exerciseBuilder_->exercise()->dates().empty() ||
178 exerciseBuilder_->exercise()->dates().back() <= today || legData_.empty()) {
179 legs_ = {{boost::make_shared<QuantLib::SimpleCashFlow>(0.0, today)}};
180 instrument_ = boost::make_shared<VanillaInstrument>(
181 boost::make_shared<QuantLib::Swap>(legs_, std::vector<bool>(1, false)), 1.0,
182 std::vector<boost::shared_ptr<QuantLib::Instrument>>{}, std::vector<Real>{});
183 legCurrencies_ = {npvCurrency_};
184 legPayers_ = {false};
185 notional_ = Null<Real>();
186 maturity_ = today;
187 return;
188 }
189
190 // fill legs, only include coupons after first exercise
191
192 legCurrencies_ = underlying_->legCurrencies();
193 legPayers_ = underlying_->legPayers();
194 legs_.clear();
195 Date firstExerciseDate = exerciseBuilder_->exercise()->dates().front();
196 for (auto const& l : underlying_->legs()) {
197 legs_.push_back(Leg());
198 for (auto const& c : l) {
199 if (auto cpn = boost::dynamic_pointer_cast<Coupon>(c)) {
200 if (firstExerciseDate <= cpn->accrualStartDate()) {
201 legs_.back().push_back(c);
202 }
203 } else if (firstExerciseDate <= c->date()) {
204 legs_.back().push_back(c);
205 }
206 }
207 }
208
209 // build a Euroepan or Bermudan swaption
210
211 if (exerciseType_ == Exercise::European && isStandard)
212 buildEuropean(engineFactory);
213 else
214 buildBermudan(engineFactory);
215
216 // ISDA taxonomy
217 additionalData_["isdaAssetClass"] = string("Interest Rate");
218 additionalData_["isdaBaseProduct"] = string("Option");
219 additionalData_["isdaSubProduct"] = string("Swaption");
220 additionalData_["isdaTransaction"] = string("");
221}

223void Swaption::buildEuropean(const boost::shared_ptr<EngineFactory>& engineFactory) {
224
225 DLOG("Building European Swaption " << id());
226
227 boost::shared_ptr<VanillaSwap> swap = buildVanillaSwap(engineFactory);
228
229 boost::shared_ptr<EngineBuilder> builder = engineFactory->builder("EuropeanSwaption");
230 boost::shared_ptr<EuropeanSwaptionEngineBuilder> swaptionBuilder =
231 boost::dynamic_pointer_cast<EuropeanSwaptionEngineBuilder>(builder);
232 QL_REQUIRE(swaptionBuilder != nullptr, "internal error: could not cast to EuropeanSwaptionEngineBuilder");
233
234 boost::shared_ptr<QuantLib::Swaption> swaption =
235 boost::make_shared<QuantLib::Swaption>(swap, exerciseBuilder_->exercise(), settlementType_, settlementMethod_);
236
237 Currency ccy = parseCurrency(npvCurrency_);
238
239 std::vector<boost::shared_ptr<Instrument>> additionalInstruments;
240 std::vector<Real> additionalMultipliers;
241 Date lastPremiumDate = addPremiums(additionalInstruments, additionalMultipliers, 1.0, optionData_.premiumData(),
242 positionType_ == Position::Long ? -1.0 : 1.0, ccy, engineFactory,
243 swaptionBuilder->configuration(MarketContext::pricing));
244
245 swaption->setPricingEngine(
246 swaptionBuilder->engine(IndexNameTranslator::instance().oreName(swap->iborIndex()->name())));
247
248 if (settlementType_ == Settlement::Physical) {
249 instrument_ = boost::make_shared<EuropeanOptionWrapper>(
250 swaption, positionType_ == Position::Long ? true : false, exerciseBuilder_->exercise()->dates().back(),
251 settlementType_ == Settlement::Physical ? true : false, swap, 1.0, 1.0, additionalInstruments,
252 additionalMultipliers);
253 maturity_ = std::max(swap->fixedSchedule().dates().back(), swap->floatingSchedule().dates().back());
254 } else {
255 instrument_ = boost::make_shared<VanillaInstrument>(swaption, positionType_ == Position::Long ? 1.0 : -1.0,
256 additionalInstruments, additionalMultipliers);
257 // Align with ISDA AANA/GRID guidance as of November 2020
258 // maturity_ = exDate;
259 maturity_ = std::max(swap->fixedSchedule().dates().back(), swap->floatingSchedule().dates().back());
260 }
261
262 maturity_ = std::max(maturity_, lastPremiumDate);
263
264 DLOG("Building European Swaption done");
265}

267void Swaption::buildBermudan(const boost::shared_ptr<EngineFactory>& engineFactory) {
268
269 if (settlementType_ == Settlement::Physical)
270 maturity_ = underlying_->maturity();
271 else
272 maturity_ = exerciseBuilder_->noticeDates().back();
273
274 if (settlementType_ == Settlement::Cash && settlementMethod_ == Settlement::ParYieldCurve)
275 WLOG("Cash-settled Bermudan Swaption (id = "
276 << id() << ") with ParYieldCurve settlement method not supported by Lgm engine. "
277 << "Approximate pricing using CollateralizedCashPrice pricing methodology");
278
279 std::vector<QuantLib::Currency> ccys;
280 for (auto const& c : underlying_->legCurrencies())
281 ccys.push_back(parseCurrency(c));
282 auto swaption =
283 boost::make_shared<QuantExt::MultiLegOption>(underlying_->legs(), underlying_->legPayers(), ccys,
284 exerciseBuilder_->exercise(), settlementType_, settlementMethod_);
285
286 // determine strikes for calibration basket (simple approach, a la summit)
287 // also determine the ibor index (if several, chose the first) to get the engine
288 std::vector<Real> strikes(exerciseBuilder_->noticeDates().size(), Null<Real>());
289 boost::shared_ptr<IborIndex> index;
290 for (Size i = 0; i < exerciseBuilder_->noticeDates().size(); ++i) {
291 Real firstFixedRate = Null<Real>();
292 Real firstFloatSpread = Null<Real>();
293 for (auto const& l : underlying_->legs()) {
294 for (auto const& c : l) {
295 if (auto cpn = boost::dynamic_pointer_cast<FixedRateCoupon>(c)) {
296 if (cpn->accrualStartDate() >= exerciseBuilder_->noticeDates()[i] && firstFixedRate == Null<Real>())
297 firstFixedRate = cpn->rate();
298 } else if (auto cpn = boost::dynamic_pointer_cast<FloatingRateCoupon>(c)) {
299 firstFloatSpread = cpn->spread();
300 if (index == nullptr) {
301 if (auto tmp = boost::dynamic_pointer_cast<IborIndex>(cpn->index())) {
302 DLOG("found ibor / ois index '" << tmp->name() << "'");
303 index = tmp;
304 } else if (auto tmp = boost::dynamic_pointer_cast<SwapIndex>(cpn->index())) {
305 DLOG("found cms index " << tmp->name() << ", use key '" << tmp->iborIndex()->name()
306 << "' to look up vol");
307 index = tmp->iborIndex();
308 }
309 }
310 }
311 }
312 }
313 if (firstFixedRate != Null<Real>()) {
314 strikes[i] = firstFixedRate;
315 if (firstFloatSpread != Null<Real>()) {
316 strikes[i] -= firstFloatSpread;
317 }
318 }
319 DLOG("calibration strike for ex date "
320 << QuantLib::io::iso_date(exerciseBuilder_->noticeDates()[i]) << " is "
321 << (strikes[i] == Null<Real>() ? "ATMF" : std::to_string(strikes[i])) << " (fixed rate "
322 << (firstFixedRate == Null<Real>() ? "NA" : std::to_string(firstFixedRate)) << ", spread "
323 << (firstFloatSpread == Null<Real>() ? "NA" : std::to_string(firstFloatSpread)) << ")");
324 }
325
326 if (index == nullptr) {
327 DLOG("no ibor, ois, cms index found, use ccy key to look up vol");
328 }
329
330 auto builder = engineFactory->builder("BermudanSwaption");
331 auto swaptionBuilder = boost::dynamic_pointer_cast<BermudanSwaptionEngineBuilder>(builder);
332 QL_REQUIRE(swaptionBuilder, "internal error: could not cast to BermudanSwaptionEngineBuilder");
333
334 auto tmp = engineFactory->builder("Swap");
335 auto swapBuilder = boost::dynamic_pointer_cast<SwapEngineBuilderBase>(tmp);
336 QL_REQUIRE(swapBuilder, "internal error: could not cast to SwapEngineBuilder");
337
338 cpu_timer timer;
339 // use ibor / ois index as key, if possible, otherwise the npv currency
340 auto swaptionEngine = swaptionBuilder->engine(
341 id(), index == nullptr ? npvCurrency_ : IndexNameTranslator::instance().oreName(index->name()),
342 exerciseBuilder_->noticeDates(), underlying_->maturity(), strikes);
343 timer.stop();
344 DLOG("Swaption model calibration time: " << timer.format(default_places, "%w") << " s");
345 swaption->setPricingEngine(swaptionEngine);
346
347 auto swapEngine = swapBuilder->engine(parseCurrency(npvCurrency_));
348
349 std::vector<boost::shared_ptr<Instrument>> underlyingSwaps =
350 buildUnderlyingSwaps(swapEngine, exerciseBuilder_->noticeDates());
351
352 // If premium data is provided
353 // 1) build the fee trade and pass it to the instrument wrapper for pricing
354 // 2) add fee payment as additional trade leg for cash flow reporting
355 std::vector<boost::shared_ptr<Instrument>> additionalInstruments;
356 std::vector<Real> additionalMultipliers;
357 Real multiplier = positionType_ == Position::Long ? 1.0 : -1.0;
358 Date lastPremiumDate =
359 addPremiums(additionalInstruments, additionalMultipliers, 1.0, optionData_.premiumData(), -multiplier,
360 parseCurrency(npvCurrency_), engineFactory, swaptionBuilder->configuration(MarketContext::pricing));
361
362 // instrument_ = boost::shared_ptr<InstrumentWrapper> (new VanillaInstrument (swaption, multiplier));
363 instrument_ = boost::make_shared<BermudanOptionWrapper>(
364 swaption, positionType_ == Position::Long ? true : false, exerciseBuilder_->noticeDates(),
365 settlementType_ == Settlement::Physical ? true : false, underlyingSwaps, 1.0, 1.0, additionalInstruments,
366 additionalMultipliers);
367
368 maturity_ = std::max(maturity_, lastPremiumDate);
369
370 DLOG("Building Bermudan Swaption done");
371}

373boost::shared_ptr<VanillaSwap> Swaption::buildVanillaSwap(const boost::shared_ptr<EngineFactory>& engineFactory) {
374
375 Date exerciseDate = exerciseBuilder_->exercise()->dates().front();
376
377 Size fixedLegIndex, floatingLegIndex;
378 if (legData_[0].legType() == "Floating" && legData_[1].legType() == "Fixed") {
379 floatingLegIndex = 0;
380 fixedLegIndex = 1;
381 } else if (legData_[1].legType() == "Floating" && legData_[0].legType() == "Fixed") {
382 floatingLegIndex = 1;
383 fixedLegIndex = 0;
384 } else {
385 QL_FAIL("Invalid leg types " << legData_[0].legType() << " + " << legData_[1].legType());
386 }
387 boost::shared_ptr<FixedLegData> fixedLegData =
388 boost::dynamic_pointer_cast<FixedLegData>(legData_[fixedLegIndex].concreteLegData());
389 boost::shared_ptr<FloatingLegData> floatingLegData =
390 boost::dynamic_pointer_cast<FloatingLegData>(legData_[floatingLegIndex].concreteLegData());
391
392 boost::shared_ptr<EngineBuilder> tmp = engineFactory->builder("Swap");
393 boost::shared_ptr<SwapEngineBuilderBase> swapBuilder = boost::dynamic_pointer_cast<SwapEngineBuilderBase>(tmp);
394 QL_REQUIRE(swapBuilder, "No Swap Builder found for Swaption " << id());
395
396 // Get Trade details
397 // We known that the notional, fixed rate, float spread, gearing are all constant in the underlying leg. To make
398 // sure we get the correct value from the leg data we take the _last_ value in the leg data vectors, because
399 // the first might contain a value that is different, but lies before the schedule start date.
400 string ccy = legData_[0].currency();
401 Currency currency = parseCurrency(ccy);
402 Real nominal = legData_[0].notionals().back();
403
404 Real rate = fixedLegData->rates().back();
405 Real spread = floatingLegData->spreads().empty() ? 0.0 : floatingLegData->spreads().back();
406 string indexName = floatingLegData->index();
407
408 Schedule fixedSchedule = makeSchedule(legData_[fixedLegIndex].schedule());
409 DayCounter fixedDayCounter = parseDayCounter(legData_[fixedLegIndex].dayCounter());
410
411 Schedule floatingSchedule = makeSchedule(legData_[floatingLegIndex].schedule());
412 Handle<IborIndex> index =
413 engineFactory->market()->iborIndex(indexName, swapBuilder->configuration(MarketContext::pricing));
414 DayCounter floatingDayCounter = parseDayCounter(legData_[floatingLegIndex].dayCounter());
415
416 BusinessDayConvention paymentConvention =
417 parseBusinessDayConvention(legData_[floatingLegIndex].paymentConvention());
418
419 VanillaSwap::Type type = legData_[fixedLegIndex].isPayer() ? VanillaSwap::Payer : VanillaSwap::Receiver;
420
421 // only take into account accrual periods with start date on or after first exercise date (if given)
422 std::vector<Date> fixDates = fixedSchedule.dates();
423 auto it1 = std::lower_bound(fixDates.begin(), fixDates.end(), exerciseDate);
424 fixDates.erase(fixDates.begin(), it1);
425 // check we have at least 1 to stop set fault on vector(fixDates.size() - 1, true) but maybe check should be 2
426 QL_REQUIRE(fixDates.size() >= 2, "Not enough schedule dates are left in Swaption fixed leg (check exercise dates)");
427 fixedSchedule = Schedule(fixDates, fixedSchedule.calendar(), Unadjusted, boost::none, boost::none, boost::none,
428 boost::none, std::vector<bool>(fixDates.size() - 1, true));
429 std::vector<Date> floatingDates = floatingSchedule.dates();
430 auto it2 = std::lower_bound(floatingDates.begin(), floatingDates.end(), exerciseDate);
431 floatingDates.erase(floatingDates.begin(), it2);
432 QL_REQUIRE(floatingDates.size() >= 2,
433 "Not enough schedule dates are left in Swaption floating leg (check exercise dates)");
434 floatingSchedule = Schedule(floatingDates, floatingSchedule.calendar(), Unadjusted, boost::none, boost::none,
435 boost::none, boost::none, std::vector<bool>(floatingDates.size() - 1, true));
436
437 // Build a vanilla (bullet) swap underlying
438 boost::shared_ptr<VanillaSwap> swap =
439 boost::make_shared<VanillaSwap>(type, nominal, fixedSchedule, rate, fixedDayCounter, floatingSchedule, *index,
440 spread, floatingDayCounter, paymentConvention);
441
442 swap->setPricingEngine(swapBuilder->engine(currency));
443 return swap;
444}

446std::vector<boost::shared_ptr<Instrument>>
447Swaption::buildUnderlyingSwaps(const boost::shared_ptr<PricingEngine>& swapEngine,
448 const std::vector<Date>& exerciseDates) {
449 std::vector<boost::shared_ptr<Instrument>> swaps;
450 for (Size i = 0; i < exerciseDates.size(); ++i) {
451 std::vector<Leg> legs = underlying_->legs();
452 std::vector<bool> payer = underlying_->legPayers();
453 for (Size j = 0; j < legs.size(); ++j) {
454 Date ed = exerciseDates[i];
455 auto it = std::lower_bound(legs[j].begin(), legs[j].end(), exerciseDates[i],
456 [&ed](const boost::shared_ptr<CashFlow>& c, const Date& d) {
457 if (auto cpn = boost::dynamic_pointer_cast<Coupon>(c)) {
458 return cpn->accrualStartDate() < ed;
459 } else {
460 return c->date() < ed;
461 }
462 });
463 if (it != legs[j].begin())
464 --it;
465 legs[j].erase(legs[j].begin(), it);
466 }
467 auto newSwap = boost::make_shared<QuantLib::Swap>(legs, payer);
468 if (swapEngine != nullptr) {
469 newSwap->setPricingEngine(swapEngine);
470 }
471 swaps.push_back(newSwap);
472 for (auto const& l : legs) {
473 if (l.empty()) {
474 WLOG("Added empty leg to underlying swap for exercise " << QuantLib::io::iso_date(exerciseDates[i])
475 << "!");
476 } else {
477 Date d;
478 if (auto cpn = boost::dynamic_pointer_cast<Coupon>(l.front())) {
479 d = cpn->accrualStartDate();
480 } else {
481 d = l.front()->date();
482 }
483 DLOG("Added leg with start date " << QuantLib::io::iso_date(d) << " for exercise "
484 << QuantLib::io::iso_date(exerciseDates[i]));
485 }
486 }
487 }
488 return swaps;
489}

491QuantLib::Real Swaption::notional() const {
492 Real tmp = 0.0;
493 for (auto const& l : underlying_->legs()) {
494 tmp = std::max(tmp, currentNotional(l));
495 }
496 return tmp;

499const std::map<std::string, boost::any>& Swaption::additionalData() const {
500 // use the build time as of date to determine current notionals
501 Date asof = Settings::instance().evaluationDate();
502 for (Size i = 0; i < std::min(legData_.size(), legs_.size()); ++i) {
503 string legID = to_string(i + 1);
504 additionalData_["legType[" + legID + "]"] = legData_[i].legType();
505 additionalData_["isPayer[" + legID + "]"] = legData_[i].isPayer();
506 additionalData_["notionalCurrency[" + legID + "]"] = legData_[i].currency();
507 for (Size j = 0; j < legs_[i].size(); ++j) {
508 boost::shared_ptr<CashFlow> flow = legs_[i][j];
509 // pick flow with earliest future payment date on this leg
510 if (flow->date() > asof) {
511 additionalData_["amount[" + legID + "]"] = flow->amount();
512 additionalData_["paymentDate[" + legID + "]"] = to_string(flow->date());
513 boost::shared_ptr<Coupon> coupon = boost::dynamic_pointer_cast<Coupon>(flow);
514 if (coupon) {
515 additionalData_["currentNotional[" + legID + "]"] = coupon->nominal();
516 additionalData_["rate[" + legID + "]"] = coupon->rate();
517 boost::shared_ptr<FloatingRateCoupon> frc = boost::dynamic_pointer_cast<FloatingRateCoupon>(flow);
518 if (frc) {
519 additionalData_["index[" + legID + "]"] = frc->index()->name();
520 additionalData_["spread[" + legID + "]"] = frc->spread();
521 }
522 }
523 break;
524 }
525 }
526 if (legs_[i].size() > 0) {
527 boost::shared_ptr<Coupon> coupon = boost::dynamic_pointer_cast<Coupon>(legs_[i][0]);
528 if (coupon)
529 additionalData_["originalNotional[" + legID + "]"] = coupon->nominal();
530 }
531 }
532 return additionalData_;
533}

535void Swaption::fromXML(XMLNode* node) {
536 Trade::fromXML(node);
537 XMLNode* swapNode = XMLUtils::getChildNode(node, "SwaptionData");
538 optionData_.fromXML(XMLUtils::getChildNode(swapNode, "OptionData"));
539 legData_.clear();
540 vector<XMLNode*> nodes = XMLUtils::getChildrenNodes(swapNode, "LegData");
541 for (Size i = 0; i < nodes.size(); i++) {
542 LegData ld;
543 ld.fromXML(nodes[i]);
544 legData_.push_back(ld);
545 }
546}

548XMLNode* Swaption::toXML(XMLDocument& doc) {
549 XMLNode* node = Trade::toXML(doc);
550 XMLNode* swaptionNode = doc.allocNode("SwaptionData");
551 XMLUtils::appendNode(node, swaptionNode);
552
553 XMLUtils::appendNode(swaptionNode, optionData_.toXML(doc));
554 for (Size i = 0; i < legData_.size(); i++)
555 XMLUtils::appendNode(swaptionNode, legData_[i].toXML(doc));
556
557 return node;
558}
