101 const vector<string>& GenericYieldVolatilityCurveConfig::quotes() {
102 
103  if (quotes_.size() == 0 && proxySourceCurveId_.empty()) {
104  std::stringstream ssBase;
105  ssBase << marketDatumInstrumentLabel_ << "/" << volatilityType_ << "/" << qualifier_ << "/";
106  if (!quoteTag_.empty())
107  ssBase << quoteTag_ << "/";
108  string base = ssBase.str();
109 
110  // add atm vols (always required)
111  for (auto o : optionTenors_) {
112  for (auto s : underlyingTenors_) {
113  std::stringstream ss;
114  ss << base << o << "/" << s << "/ATM";
115  quotes_.push_back(ss.str());
116  }
117  }
118 
119  // add Smile Spreads, if dimension is Smile
120  if (dimension_ == Dimension::Smile) {
121  for (auto o : smileOptionTenors_) {
122  for (auto s : smileUnderlyingTenors_) {
123  for (auto sp : smileSpreads_) {
124  std::stringstream ss;
125  ss << base << o << "/" << s << "/Smile/" << sp;
126  quotes_.push_back(ss.str());
127  }
128  }
129  }
130  }
131  // add SHIFT quotes, if vol type is SLN
132  for (auto s : underlyingTenors_) {
133  if (volatilityType_ == VolatilityType::ShiftedLognormal) {
134  std::stringstream ss;
135  ss << marketDatumInstrumentLabel_ << "/SHIFT/" << qualifier_ << "/"
136  << (quoteTag_.empty() ? "" : quoteTag_ + "/") << s;
137  quotes_.push_back(ss.str());
138  }
139  }
140  }
141  return quotes_;
142 }

144 string GenericYieldVolatilityCurveConfig::ccyFromSwapIndexBase(const std::string& swapIndexBase) {
145  std::vector<string> tokens;
146  split(tokens, swapIndexBase, boost::is_any_of("-"));
147  QL_REQUIRE(!tokens.empty() && tokens[0] != "",
148  "GenericYieldVolatilityCurveConfig::fromXML(): can not derive qualifier from SwapIndexBase ("
149  << swapIndexBase << ")");
150  return tokens[0];
151 }

153 void GenericYieldVolatilityCurveConfig::fromXML(XMLNode* node) {
154  XMLUtils::checkNode(node, rootNodeLabel_);
155 
156  curveID_ = XMLUtils::getChildValue(node, "CurveId", true);
157  curveDescription_ = XMLUtils::getChildValue(node, "CurveDescription", true);
158 
159  if (auto p = XMLUtils::getChildNode(node, "ProxyConfig")) {
160  // read in proxy config
161 
162  auto source = XMLUtils::getChildNode(p, "Source");
163  QL_REQUIRE(source != nullptr,
164  "GenericYieldVolatilityCurveConfig::fromXML(): ProxyConfig requires child node 'Source'");
165  proxySourceCurveId_ = XMLUtils::getChildValue(source, "CurveId");
166  proxySourceShortSwapIndexBase_ = XMLUtils::getChildValue(source, "ShortSwapIndexBase");
167  proxySourceSwapIndexBase_ = XMLUtils::getChildValue(source, "SwapIndexBase");
168 
169  auto target = XMLUtils::getChildNode(p, "Target");
170  QL_REQUIRE(target != nullptr,
171  "GenericYieldVolatilityCurveConfig::fromXML(): ProxyConfig requires child node 'Target'");
172  proxyTargetShortSwapIndexBase_ = XMLUtils::getChildValue(target, "ShortSwapIndexBase");
173  proxyTargetSwapIndexBase_ = XMLUtils::getChildValue(target, "SwapIndexBase");
174 
175  populateRequiredCurveIds();
176 
177  } else {
178  // read in quote-based config
179  if (allowSmile_) {
180  string dim = XMLUtils::getChildValue(node, "Dimension", true);
181  if (dim == "ATM") {
182  dimension_ = Dimension::ATM;
183  } else if (dim == "Smile") {
184  dimension_ = Dimension::Smile;
185  } else {
186  QL_FAIL("Dimension " << dim << " not recognized");
187  }
188  } else {
189  dimension_ = Dimension::ATM;
190  }
191 
192  string volType = XMLUtils::getChildValue(node, "VolatilityType", true);
193  if (volType == "Normal") {
194  volatilityType_ = VolatilityType::Normal;
195  } else if (volType == "Lognormal") {
196  volatilityType_ = VolatilityType::Lognormal;
197  } else if (volType == "ShiftedLognormal") {
198  volatilityType_ = VolatilityType::ShiftedLognormal;
199  } else {
200  QL_FAIL("Volatility type " << volType << " not recognized");
201  }
202 
203  string extr = XMLUtils::getChildValue(node, "Extrapolation", true);
204  extrapolate_ = true;
205  flatExtrapolation_ = true;
206  if (extr == "Linear") {
207  flatExtrapolation_ = false;
208  } else if (extr == "Flat") {
209  flatExtrapolation_ = true;
210  } else if (extr == "None") {
211  extrapolate_ = false;
212  } else {
213  QL_FAIL("Extrapolation " << extr << " not recognized");
214  }
215 
216  optionTenors_ = XMLUtils::getChildrenValuesAsStrings(node, "OptionTenors", true);
217 
218  underlyingTenors_ = XMLUtils::getChildrenValuesAsStrings(node, underlyingLabel_ + "Tenors", false);
219 
220  string cal = XMLUtils::getChildValue(node, "Calendar", true);
221  calendar_ = parseCalendar(cal);
222 
223  string dc = XMLUtils::getChildValue(node, "DayCounter", true);
224  dayCounter_ = parseDayCounter(dc);
225 
226  string bdc = XMLUtils::getChildValue(node, "BusinessDayConvention", true);
227  businessDayConvention_ = parseBusinessDayConvention(bdc);
228 
229  if (requireSwapIndexBases_ || dimension_ == Dimension::Smile) {
230  shortSwapIndexBase_ = XMLUtils::getChildValue(node, "ShortSwapIndexBase", true);
231  swapIndexBase_ = XMLUtils::getChildValue(node, "SwapIndexBase", true);
232  }
233 
234  // smile data
235  if (dimension_ == Dimension::Smile) {
236  smileOptionTenors_ = XMLUtils::getChildrenValuesAsStrings(node, "SmileOptionTenors", true);
237  smileUnderlyingTenors_ =
238  XMLUtils::getChildrenValuesAsStrings(node, "Smile" + underlyingLabel_ + "Tenors", true);
239  smileSpreads_ = XMLUtils::getChildrenValuesAsStrings(node, "SmileSpreads", true);
240  }
241 
242  // read qualifier from explicit field
243  if (qualifierLabel_ != "")
244  qualifier_ = XMLUtils::getChildValue(node, qualifierLabel_, true);
245 
246  // derive qualifier (=ccy) from swapIndexBase if not given explicitly
247  if (qualifier_ == "")
248  qualifier_ = ccyFromSwapIndexBase(swapIndexBase_);
249 
250  // optional quote tag to include
251  quoteTag_ = XMLUtils::getChildValue(node, "QuoteTag", false);
252 
253  }
254 
255  smileDynamics_ = XMLUtils::getChildValue(node, "SmileDynamics", false, "");
256 
257  if (auto tmp = XMLUtils::getChildNode(node, "Report")) {
258  reportConfig_.fromXML(tmp);
259  }
260 }

262 XMLNode* GenericYieldVolatilityCurveConfig::toXML(XMLDocument& doc) {
263  XMLNode* node = doc.allocNode(rootNodeLabel_);
264  XMLUtils::addChild(doc, node, "CurveId", curveID_);
265  XMLUtils::addChild(doc, node, "CurveDescription", curveDescription_);
266 
267  if (!proxySourceCurveId_.empty()) {
268  // write out proxy config
269  auto proxy = XMLUtils::addChild(doc, node, "ProxyConfig");
270  auto source = XMLUtils::addChild(doc, proxy, "Source");
271  auto target = XMLUtils::addChild(doc, proxy, "Target");
272  XMLUtils::addChild(doc, source, "CurveId", proxySourceCurveId_);
273  XMLUtils::addChild(doc, source, "ShortSwapIndexBase", proxySourceShortSwapIndexBase_);
274  XMLUtils::addChild(doc, source, "SwapIndexBase", proxySourceSwapIndexBase_);
275  XMLUtils::addChild(doc, target, "ShortSwapIndexBase", proxyTargetShortSwapIndexBase_);
276  XMLUtils::addChild(doc, target, "SwapIndexBase", proxyTargetSwapIndexBase_);
277  } else {
278  // write out quote based config
279  if (qualifierLabel_ != "")
280  XMLUtils::addChild(doc, node, qualifierLabel_, qualifier_);
281 
282  if (dimension_ == Dimension::ATM) {
283  XMLUtils::addChild(doc, node, "Dimension", "ATM");
284  } else if (dimension_ == Dimension::Smile) {
285  XMLUtils::addChild(doc, node, "Dimension", "Smile");
286  } else {
287  QL_FAIL("Unknown Dimension in GenericYieldVolatilityCurveConfig::toXML()");
288  }
289 
290  if (volatilityType_ == VolatilityType::Normal) {
291  XMLUtils::addChild(doc, node, "VolatilityType", "Normal");
292  } else if (volatilityType_ == VolatilityType::Lognormal) {
293  XMLUtils::addChild(doc, node, "VolatilityType", "Lognormal");
294  } else if (volatilityType_ == VolatilityType::ShiftedLognormal) {
295  XMLUtils::addChild(doc, node, "VolatilityType", "ShiftedLognormal");
296  } else {
297  QL_FAIL("Unknown VolatilityType in GenericYieldVolatilityCurveConfig::toXML()");
298  }
299 
300  string extr_str = flatExtrapolation_ ? "Flat" : "Linear";
301  if (!extrapolate_)
302  extr_str = "None";
303  XMLUtils::addChild(doc, node, "Extrapolation", extr_str);
304 
305  XMLUtils::addChild(doc, node, "DayCounter", to_string(dayCounter_));
306  XMLUtils::addChild(doc, node, "Calendar", to_string(calendar_));
307  XMLUtils::addChild(doc, node, "BusinessDayConvention", to_string(businessDayConvention_));
308  XMLUtils::addGenericChildAsList(doc, node, "OptionTenors", optionTenors_);
309  XMLUtils::addGenericChildAsList(doc, node, underlyingLabel_ + "Tenors", underlyingTenors_);
310 
311  if (requireSwapIndexBases_ || dimension_ == Dimension::Smile) {
312  XMLUtils::addChild(doc, node, "ShortSwapIndexBase", shortSwapIndexBase_);
313  XMLUtils::addChild(doc, node, "SwapIndexBase", swapIndexBase_);
314  }
315 
316  if (dimension_ == Dimension::Smile) {
317  XMLUtils::addGenericChildAsList(doc, node, "SmileOptionTenors", smileOptionTenors_);
318  XMLUtils::addGenericChildAsList(doc, node, "Smile" + underlyingLabel_ + "Tenors", smileUnderlyingTenors_);
319  XMLUtils::addGenericChildAsList(doc, node, "SmileSpreads", smileSpreads_);
320  }
321 
322  if (!quoteTag_.empty()) {
323  XMLUtils::addChild(doc, node, "QuoteTag", quoteTag_);
324  }
325  }
326 
327  XMLUtils::addChild(doc, node, "SmileDynamics", smileDynamics_);
328  XMLUtils::appendNode(node, reportConfig_.toXML(doc));
329 
330  return node;
331 }
