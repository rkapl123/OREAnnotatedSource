72 const vector<string>& GenericYieldVolatilityCurveConfig::quotes() {
73 
74  if (quotes_.size() == 0) {
75  std::stringstream ssBase;
76  ssBase << marketDatumInstrumentLabel_ << "/" << volatilityType_ << "/" << qualifier_ << "/";
77  string base = ssBase.str();
78 
79  // add atm vols (always required)
80  for (auto o : optionTenors_) {
81  for (auto s : underlyingTenors_) {
82  std::stringstream ss;
83  ss << base << o << "/" << s << "/ATM";
84  quotes_.push_back(ss.str());
85  }
86  }
87 
88  // add Smile Spreads, if dimension is Smile
89  if (dimension_ == Dimension::Smile) {
90  for (auto o : smileOptionTenors_) {
91  for (auto s : smileUnderlyingTenors_) {
92  for (auto sp : smileSpreads_) {
93  std::stringstream ss;
94  ss << base << o << "/" << s << "/Smile/" << sp;
95  quotes_.push_back(ss.str());
96  }
97  }
98  }
99  }
100  // add SHIFT quotes, if vol type is SLN
101  for (auto s : underlyingTenors_) {
102  if (volatilityType_ == VolatilityType::ShiftedLognormal) {
103  std::stringstream ss;
104  ss << marketDatumInstrumentLabel_ << "/SHIFT/" << qualifier_ << "/" << s;
105  quotes_.push_back(ss.str());
106  }
107  }
108  }
109  return quotes_;
110 }

112 string GenericYieldVolatilityCurveConfig::ccyFromSwapIndexBase() {
113  std::vector<string> tokens;
114  split(tokens, swapIndexBase_, boost::is_any_of("-"));
115  QL_REQUIRE(!tokens.empty() && tokens[0] != "",
116  "GenericYieldVolatilityCurveConfig::fromXML(): can not derive qualifier from SwapIndexBase ("
117  << swapIndexBase_ << ")");
118  return tokens[0];

121 void GenericYieldVolatilityCurveConfig::fromXML(XMLNode* node) {
122  XMLUtils::checkNode(node, rootNodeLabel_);
123 
124  curveID_ = XMLUtils::getChildValue(node, "CurveId", true);
125  curveDescription_ = XMLUtils::getChildValue(node, "CurveDescription", true);
126 
127  if (allowSmile_) {
128  string dim = XMLUtils::getChildValue(node, "Dimension", true);
129  if (dim == "ATM") {
130  dimension_ = Dimension::ATM;
131  } else if (dim == "Smile") {
132  dimension_ = Dimension::Smile;
133  } else {
134  QL_FAIL("Dimension " << dim << " not recognized");
135  }
136  } else {
137  dimension_ = Dimension::ATM;
138  }
139 
140  string volType = XMLUtils::getChildValue(node, "VolatilityType", true);
141  if (volType == "Normal") {
142  volatilityType_ = VolatilityType::Normal;
143  } else if (volType == "Lognormal") {
144  volatilityType_ = VolatilityType::Lognormal;
145  } else if (volType == "ShiftedLognormal") {
146  volatilityType_ = VolatilityType::ShiftedLognormal;
147  } else {
148  QL_FAIL("Volatility type " << volType << " not recognized");
149  }
150 
151  string extr = XMLUtils::getChildValue(node, "Extrapolation", true);
152  extrapolate_ = true;
153  flatExtrapolation_ = true;
154  if (extr == "Linear") {
155  flatExtrapolation_ = false;
156  } else if (extr == "Flat") {
157  flatExtrapolation_ = true;
158  } else if (extr == "None") {
159  extrapolate_ = false;
160  } else {
161  QL_FAIL("Extrapolation " << extr << " not recognized");
162  }
163 
164  optionTenors_ = XMLUtils::getChildrenValuesAsStrings(node, "OptionTenors", true);
165 
166  underlyingTenors_ = XMLUtils::getChildrenValuesAsStrings(node, underlyingLabel_ + "Tenors", false);
167 
168  string cal = XMLUtils::getChildValue(node, "Calendar", true);
169  calendar_ = parseCalendar(cal);
170 
171  string dc = XMLUtils::getChildValue(node, "DayCounter", true);
172  dayCounter_ = parseDayCounter(dc);
173 
174  string bdc = XMLUtils::getChildValue(node, "BusinessDayConvention", true);
175  businessDayConvention_ = parseBusinessDayConvention(bdc);
176 
177  if (requireSwapIndexBases_ || dimension_ == Dimension::Smile) {
178  shortSwapIndexBase_ = XMLUtils::getChildValue(node, "ShortSwapIndexBase", true);
179  swapIndexBase_ = XMLUtils::getChildValue(node, "SwapIndexBase", true);
180  }
181 
182  // smile data
183  if (dimension_ == Dimension::Smile) {
184  smileOptionTenors_ = XMLUtils::getChildrenValuesAsStrings(node, "SmileOptionTenors", true);
185  smileUnderlyingTenors_ =
186  XMLUtils::getChildrenValuesAsStrings(node, "Smile" + underlyingLabel_ + "Tenors", true);
187  smileSpreads_ = XMLUtils::getChildrenValuesAsStrings(node, "SmileSpreads", true);
188  }
189 
190  // read qualifier from explicit field
191  if (qualifierLabel_ != "")
192  qualifier_ = XMLUtils::getChildValue(node, qualifierLabel_, true);
193 
194  // derive qualifier (=ccy) from swapIndexBase if not given explicitly
195  if (qualifier_ == "")
196  qualifier_ = ccyFromSwapIndexBase();
197 }

199 XMLNode* GenericYieldVolatilityCurveConfig::toXML(XMLDocument& doc) {
200  XMLNode* node = doc.allocNode(rootNodeLabel_);
201  XMLUtils::addChild(doc, node, "CurveId", curveID_);
202  XMLUtils::addChild(doc, node, "CurveDescription", curveDescription_);
203  if (qualifierLabel_ != "")
204  XMLUtils::addChild(doc, node, qualifierLabel_, qualifier_);
205 
206  if (dimension_ == Dimension::ATM) {
207  XMLUtils::addChild(doc, node, "Dimension", "ATM");
208  } else if (dimension_ == Dimension::Smile) {
209  XMLUtils::addChild(doc, node, "Dimension", "Smile");
210  } else {
211  QL_FAIL("Unkown Dimension in GenericYieldVolatilityCurveConfig::toXML()");
212  }
213 
214  if (volatilityType_ == VolatilityType::Normal) {
215  XMLUtils::addChild(doc, node, "VolatilityType", "Normal");
216  } else if (volatilityType_ == VolatilityType::Lognormal) {
217  XMLUtils::addChild(doc, node, "VolatilityType", "Lognormal");
218  } else if (volatilityType_ == VolatilityType::ShiftedLognormal) {
219  XMLUtils::addChild(doc, node, "VolatilityType", "ShiftedLognormal");
220  } else {
221  QL_FAIL("Unknown VolatilityType in GenericYieldVolatilityCurveConfig::toXML()");
222  }
223 
224  string extr_str = flatExtrapolation_ ? "Flat" : "Linear";
225  if (!extrapolate_)
226  extr_str = "None";
227  XMLUtils::addChild(doc, node, "Extrapolation", extr_str);
228 
229  XMLUtils::addChild(doc, node, "DayCounter", to_string(dayCounter_));
230  XMLUtils::addChild(doc, node, "Calendar", to_string(calendar_));
231  XMLUtils::addChild(doc, node, "BusinessDayConvention", to_string(businessDayConvention_));
232  XMLUtils::addGenericChildAsList(doc, node, "OptionTenors", optionTenors_);
233  XMLUtils::addGenericChildAsList(doc, node, underlyingLabel_ + "Tenors", underlyingTenors_);
234 
235  if (requireSwapIndexBases_ || dimension_ == Dimension::Smile) {
236  XMLUtils::addChild(doc, node, "ShortSwapIndexBase", shortSwapIndexBase_);
237  XMLUtils::addChild(doc, node, "SwapIndexBase", swapIndexBase_);
238  }
239 
240  if (dimension_ == Dimension::Smile) {
241  XMLUtils::addGenericChildAsList(doc, node, "SmileOptionTenors", smileOptionTenors_);
242  XMLUtils::addGenericChildAsList(doc, node, "Smile" + underlyingLabel_ + "Tenors", smileUnderlyingTenors_);
243  XMLUtils::addGenericChildAsList(doc, node, "SmileSpreads", smileSpreads_);
244  }
245 
246  return node;
247 }
