39void EquityForward::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
40 Currency ccy = parseCurrencyWithMinors(currency_);
41
42 // get the equity currency from the market
43 Currency equityCcy = engineFactory->market()->equityCurve(eqName())->currency();
44
45 // ensure forward currency matches the equity currency
46 QL_REQUIRE(!equityCcy.empty(), "No equity currency in equityCurve for equity " << eqName());
47 QL_REQUIRE(ccy == equityCcy, "EquityForward currency " << ccy << " does not match equity currency " << equityCcy << " for trade " << id());
48
49 // convert strike to the major currency if needed
50 Real strike;
51 if (!strikeCurrency_.empty()) {
52 Currency strikeCcy = parseCurrencyWithMinors(strikeCurrency_);
53 strike = convertMinorToMajorCurrency(strikeCurrency_, strike_);
54 // ensure strike currency matches equity currency
55 QL_REQUIRE(strikeCcy == equityCcy, "Strike currency " << ccy << " does not match equity currency " << equityCcy << " for trade " << id());
56 } else {
57 WLOG("No Strike Currency provide for trade " << id() << ", assuming trade currency " << ccy);
58 strike = convertMinorToMajorCurrency(currency_, strike_);
59 }
60
61 QuantLib::Position::Type longShort = parsePositionType(longShort_);
62 Date maturity = parseDate(maturityDate_);
63
64 string name = eqName();
65
66 boost::shared_ptr<Instrument> inst =
67 boost::make_shared<QuantExt::EquityForward>(name, ccy, longShort, quantity_, maturity, strike);
68
69 // Pricing engine
70 boost::shared_ptr<EngineBuilder> builder = engineFactory->builder(tradeType_);
71 QL_REQUIRE(builder, "No builder found for " << tradeType_);
72 boost::shared_ptr<EquityForwardEngineBuilder> eqFwdBuilder =
73 boost::dynamic_pointer_cast<EquityForwardEngineBuilder>(builder);
74 inst->setPricingEngine(eqFwdBuilder->engine(name, ccy));
75
76 // set up other Trade details
77 instrument_ = boost::shared_ptr<InstrumentWrapper>(new VanillaInstrument(inst));
78 npvCurrency_ = ccy.code();
79 maturity_ = maturity;
80 // Notional - we really need todays spot to get the correct notional.
81 // But rather than having it move around we use strike * quantity
82 notional_ = strike * quantity_;
83 notionalCurrency_ = ccy.code();
84
85 additionalData_["underlyingSecurityId"] = name;
86 additionalData_["strike"] = strike;
87 additionalData_["strikeCurrency"] = strikeCurrency_;
88 additionalData_["quantity"] = quantity_;
89
90 // ISDA taxonomy
91 additionalData_["isdaAssetClass"] = string("Equity");
92 additionalData_["isdaBaseProduct"] = string("Forward");
93 additionalData_["isdaSubProduct"] = string("Price Return Basic Performance");
94 // skip the transaction level mapping for now
95 additionalData_["isdaTransaction"] = string("");
96}

98void EquityForward::fromXML(XMLNode* node) {
99 Trade::fromXML(node);
100 XMLNode* eNode = XMLUtils::getChildNode(node, "EquityForwardData");
101
102 longShort_ = XMLUtils::getChildValue(eNode, "LongShort", true);
103 maturityDate_ = XMLUtils::getChildValue(eNode, "Maturity", true);
104 XMLNode* tmp = XMLUtils::getChildNode(eNode, "Underlying");
105 if (!tmp)
106 tmp = XMLUtils::getChildNode(eNode, "Name");
107 equityUnderlying_.fromXML(tmp);
108 currency_ = XMLUtils::getChildValue(eNode, "Currency", true);
109 strike_ = XMLUtils::getChildValueAsDouble(eNode, "Strike", true);
110 strikeCurrency_ = XMLUtils::getChildValue(eNode, "StrikeCurrency", false);
111 quantity_ = XMLUtils::getChildValueAsDouble(eNode, "Quantity", true);
112}

114XMLNode* EquityForward::toXML(XMLDocument& doc) {
115 XMLNode* node = Trade::toXML(doc);
116 XMLNode* eNode = doc.allocNode("EquityForwardData");
117 XMLUtils::appendNode(node, eNode);
118
119 XMLUtils::addChild(doc, eNode, "LongShort", longShort_);
120 XMLUtils::addChild(doc, eNode, "Maturity", maturityDate_);
121 XMLUtils::appendNode(eNode, equityUnderlying_.toXML(doc));
122 XMLUtils::addChild(doc, eNode, "Currency", currency_);
123 XMLUtils::addChild(doc, eNode, "Strike", strike_);
124 if (!strikeCurrency_.empty())
125 XMLUtils::addChild(doc, eNode, "StrikeCurrency", strikeCurrency_);
126 XMLUtils::addChild(doc, eNode, "Quantity", quantity_);
127 return node;
128}
