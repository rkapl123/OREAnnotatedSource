34 void EquityDigitalOption::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
35 
36  // Only European Vanilla supported for now
37  QL_REQUIRE(option_.style() == "European", "Option Style unknown: " << option_.style());
38  QL_REQUIRE(option_.exerciseDates().size() == 1, "Invalid number of excercise dates");
39  QL_REQUIRE(option_.payoffAtExpiry() == true, "PayoffAtExpiry must be True for EquityDigitalOption");
40  QL_REQUIRE(tradeActions().empty(), "TradeActions not supported for EquityDigitalOption");
41  QL_REQUIRE(strike_ > 0.0 && strike_ != Null<Real>(), "Invalid strike " << strike_);
42  QL_REQUIRE(payoffAmount_ > 0.0 && payoffAmount_ != Null<Real>(), "Invalid payoff amount " << payoffAmount_);
43  QL_REQUIRE(payoffCurrency_ != "", "PayoffCurrency is missing");
44 
45  // Currency
46  Currency ccy = parseCurrency(payoffCurrency_);
47 
48  // Asset Name
49  string assetName = equityName();
50 
51  // Payoff Type
52  Option::Type type = parseOptionType(option_.callPut());
53 
54  // Set up the CashOrNothing
55  Real strike = strike_;
56  boost::shared_ptr<StrikedTypePayoff> payoff(new CashOrNothingPayoff(type, strike, payoffAmount_));
57 
58  // Exercise
59  Date expiryDate = parseDate(option_.exerciseDates().front());
60  boost::shared_ptr<Exercise> exercise = boost::make_shared<EuropeanExercise>(expiryDate);
61 
62  // QL does not have an EquityDigitalOption, so we add a vanilla one here and wrap
63  // it in a composite.
64  boost::shared_ptr<Instrument> vanilla = boost::make_shared<VanillaOption>(payoff, exercise);
65 
66  // set pricing engines
67  boost::shared_ptr<EngineBuilder> builder = engineFactory->builder(tradeType_);
68  QL_REQUIRE(builder, "No builder found for " << tradeType_);
69  boost::shared_ptr<EquityDigitalOptionEngineBuilder> eqOptBuilder =
70  boost::dynamic_pointer_cast<EquityDigitalOptionEngineBuilder>(builder);
71  vanilla->setPricingEngine(eqOptBuilder->engine(assetName, ccy));
72 
73  Position::Type positionType = parsePositionType(option_.longShort());
74  Real bsInd = (positionType == QuantLib::Position::Long ? 1.0 : -1.0);
75  Real mult = bsInd;
76 
77  std::vector<boost::shared_ptr<Instrument>> additionalInstruments;
78  std::vector<Real> additionalMultipliers;
79  Date lastPremiumDate =
80  addPremiums(additionalInstruments, additionalMultipliers, mult * quantity_, option_.premiumData(), -bsInd, ccy,
81  engineFactory, eqOptBuilder->configuration(MarketContext::pricing));
82 
83  instrument_ = boost::shared_ptr<InstrumentWrapper>(
84  new VanillaInstrument(vanilla, mult*quantity_, additionalInstruments, additionalMultipliers));
85 
86  notional_ = payoffAmount_;
87  notionalCurrency_ = payoffCurrency_;
88  npvCurrency_ = payoffCurrency_;
89  maturity_ = std::max(lastPremiumDate, expiryDate);
90 
91  additionalData_["payoffAmount"] = payoffAmount_;
92  additionalData_["payoffCurrency"] = payoffCurrency_;
93 }

95 void EquityDigitalOption::fromXML(XMLNode* node) {
96  Trade::fromXML(node);
97  XMLNode* eqNode = XMLUtils::getChildNode(node, "EquityDigitalOptionData");
98  QL_REQUIRE(eqNode, "No EquityDigitalOptionData Node");
99  option_.fromXML(XMLUtils::getChildNode(eqNode, "OptionData"));
100  strike_ = XMLUtils::getChildValueAsDouble(eqNode, "Strike", true);
101  payoffCurrency_ = XMLUtils::getChildValue(eqNode, "PayoffCurrency", true);
102  payoffAmount_ = XMLUtils::getChildValueAsDouble(eqNode, "PayoffAmount", true);
103  XMLNode* tmp = XMLUtils::getChildNode(eqNode, "Underlying");
104  if (!tmp)
105  tmp = XMLUtils::getChildNode(eqNode, "Name");
106  equityUnderlying_.fromXML(tmp);
107  quantity_ = XMLUtils::getChildValueAsDouble(eqNode, "Quantity", true);
108 }

110 XMLNode* EquityDigitalOption::toXML(XMLDocument& doc) {
111  XMLNode* node = Trade::toXML(doc);
112  XMLNode* eqNode = doc.allocNode("EquityDigitalOptionData");
113  XMLUtils::appendNode(node, eqNode);
114 
115  XMLUtils::appendNode(eqNode, option_.toXML(doc));
116  XMLUtils::addChild(doc, eqNode, "Strike", strike_);
117  XMLUtils::addChild(doc, eqNode, "PayoffCurrency", payoffCurrency_);
118  XMLUtils::addChild(doc, eqNode, "PayoffAmount", payoffAmount_);
119  XMLUtils::appendNode(eqNode, equityUnderlying_.toXML(doc));
120  XMLUtils::addChild(doc, eqNode, "Quantity", quantity_);
121 
122  return node;
123 }
