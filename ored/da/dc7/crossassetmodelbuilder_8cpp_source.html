75 CrossAssetModelBuilder::CrossAssetModelBuilder(
76  const boost::shared_ptr<ore::data::Market>& market, const boost::shared_ptr<CrossAssetModelData>& config,
77  const std::string& configurationLgmCalibration, const std::string& configurationFxCalibration,
78  const std::string& configurationEqCalibration, const std::string& configurationInfCalibration,
79  const std::string& configurationCrCalibration, const std::string& configurationFinalModel,
80  const DayCounter& dayCounter, const bool dontCalibrate, const bool continueOnError,
81  const std::string& referenceCalibrationGrid)
82  : market_(market), config_(config), configurationLgmCalibration_(configurationLgmCalibration),
83  configurationFxCalibration_(configurationFxCalibration), configurationEqCalibration_(configurationEqCalibration),
84  configurationInfCalibration_(configurationInfCalibration),
85  configurationCrCalibration_(configurationCrCalibration), configurationFinalModel_(configurationFinalModel),
86  dayCounter_(dayCounter), dontCalibrate_(dontCalibrate), continueOnError_(continueOnError),
87  referenceCalibrationGrid_(referenceCalibrationGrid),
88  optimizationMethod_(boost::shared_ptr<OptimizationMethod>(new LevenbergMarquardt(1E-8, 1E-8, 1E-8))),
89  endCriteria_(EndCriteria(1000, 500, 1E-8, 1E-8, 1E-8)) {
90  buildModel();
91  registerWithSubBuilders();
92  // register market observer with correlations
93  marketObserver_ = boost::make_shared<MarketObserver>();
94  for (auto const& c : config->correlations())
95  marketObserver_->addObservable(c.second);
96  // reset market observer's updated flag
97  marketObserver_->hasUpdated(true);
98 }

143 void CrossAssetModelBuilder::performCalculations() const {
144  // if any of the sub models requires a recalibration, we rebuilt the model
145  // TODO we could do this more selectively
146  if (!dontCalibrate_ && requiresRecalibration()) {
147  // reset market observer update flag
148  marketObserver_->hasUpdated(true);
149  // the cast is a bit ugly, but we pretty much know what we are doing here
150  const_cast<CrossAssetModelBuilder*>(this)->unregisterWithSubBuilders();
151  buildModel();
152  const_cast<CrossAssetModelBuilder*>(this)->registerWithSubBuilders();
153  }

156 void CrossAssetModelBuilder::buildModel() const {
157 
158  QL_REQUIRE(market_ != NULL, "CrossAssetModelBuilder: no market given");
159  LOG("Start building CrossAssetModel");
160  DLOG("configurations: LgmCalibration "
161  << configurationLgmCalibration_ << ", FxCalibration " << configurationFxCalibration_ << ", EqCalibration "
162  << configurationEqCalibration_ << ", InfCalibration " << configurationInfCalibration_ << ", CrCalibration"
163  << configurationCrCalibration_ << ", FinalModel " << configurationFinalModel_);
164  if (dontCalibrate_) {
165  DLOG("Calibration of the model is disabled.");
166  }
167 
168  QL_REQUIRE(config_->irConfigs().size() > 0, "missing IR configurations");
169  QL_REQUIRE(config_->irConfigs().size() == config_->fxConfigs().size() + 1,
170  "FX configuration size " << config_->fxConfigs().size() << " inconsisitent with IR configuration size "
171  << config_->irConfigs().size());
172 
173  swaptionBaskets_.resize(config_->irConfigs().size());
174  optionExpiries_.resize(config_->irConfigs().size());
175  swaptionMaturities_.resize(config_->irConfigs().size());
176  swaptionCalibrationErrors_.resize(config_->irConfigs().size());
177  fxOptionBaskets_.resize(config_->fxConfigs().size());
178  fxOptionExpiries_.resize(config_->fxConfigs().size());
179  fxOptionCalibrationErrors_.resize(config_->fxConfigs().size());
180  eqOptionBaskets_.resize(config_->eqConfigs().size());
181  eqOptionExpiries_.resize(config_->eqConfigs().size());
182  eqOptionCalibrationErrors_.resize(config_->eqConfigs().size());
183  inflationCalibrationErrors_.resize(config_->infConfigs().size());
184 
185  subBuilders_.clear();
186 
187  // Store information on the number of factors for each process. This is used when requesting a correlation matrix
188  // from the CorrelationMatrixBuilder below.
189  using ProcessInfo = CorrelationMatrixBuilder::ProcessInfo;
190  namespace CT = QuantExt::CrossAssetModelTypes;
191  ProcessInfo processInfo;
192 
193  /*******************************************************
194  * Build the IR parametrizations and calibration baskets
195  */
196  std::vector<boost::shared_ptr<QuantExt::IrLgm1fParametrization>> irParametrizations;
197  std::vector<RelinkableHandle<YieldTermStructure>> irDiscountCurves;
198  std::vector<std::string> currencies, regions, crNames, eqNames, infIndices;
199  std::vector<boost::shared_ptr<LgmBuilder>> irBuilder;
200 
201  for (Size i = 0; i < config_->irConfigs().size(); i++) {
202  boost::shared_ptr<IrLgmData> ir = config_->irConfigs()[i];
203  DLOG("IR Parametrization " << i << " ccy " << ir->ccy());
204  boost::shared_ptr<LgmBuilder> builder =
205  boost::make_shared<LgmBuilder>(market_, ir, configurationLgmCalibration_, config_->bootstrapTolerance(),
206  continueOnError_, referenceCalibrationGrid_);
207  if (dontCalibrate_)
208  builder->freeze();
209  irBuilder.push_back(builder);
210  boost::shared_ptr<QuantExt::IrLgm1fParametrization> parametrization = builder->parametrization();
211  swaptionBaskets_[i] = builder->swaptionBasket();
212  currencies.push_back(ir->ccy());
213  irParametrizations.push_back(parametrization);
214  irDiscountCurves.push_back(builder->discountCurve());
215  subBuilders_[CT::IR][i] = builder;
216  processInfo[CT::IR].emplace_back(ir->ccy(), 1);
217  }
218 
219  QL_REQUIRE(irParametrizations.size() > 0, "missing IR parametrizations");
220 
221  QuantLib::Currency domesticCcy = irParametrizations[0]->currency();
222 
223  /*******************************************************
224  * Build the FX parametrizations and calibration baskets
225  */
226  std::vector<boost::shared_ptr<QuantExt::FxBsParametrization>> fxParametrizations;
227  for (Size i = 0; i < config_->fxConfigs().size(); i++) {
228  DLOG("FX Parametrization " << i);
229  boost::shared_ptr<FxBsData> fx = config_->fxConfigs()[i];
230  QuantLib::Currency ccy = ore::data::parseCurrency(fx->foreignCcy());
231  QuantLib::Currency domCcy = ore::data::parseCurrency(fx->domesticCcy());
232 
233  QL_REQUIRE(ccy.code() == irParametrizations[i + 1]->currency().code(),
234  "FX parametrization currency[" << i << "]=" << ccy << " does not match IR currrency[" << i + 1
235  << "]=" << irParametrizations[i + 1]->currency().code());
236 
237  QL_REQUIRE(domCcy == domesticCcy, "FX parametrization [" << i << "]=" << ccy << "/" << domCcy
238  << " does not match domestic ccy " << domesticCcy);
239 
240  boost::shared_ptr<FxBsBuilder> builder =
241  boost::make_shared<FxBsBuilder>(market_, fx, configurationFxCalibration_, referenceCalibrationGrid_);
242  boost::shared_ptr<QuantExt::FxBsParametrization> parametrization = builder->parametrization();
243 
244  fxOptionBaskets_[i] = builder->optionBasket();
245  fxParametrizations.push_back(parametrization);
246  subBuilders_[CT::FX][i] = builder;
247  processInfo[CT::FX].emplace_back(ccy.code() + domCcy.code(), 1);
248  }
249 
250  /*******************************************************
251  * Build the EQ parametrizations and calibration baskets
252  */
253  std::vector<boost::shared_ptr<QuantExt::EqBsParametrization>> eqParametrizations;
254  for (Size i = 0; i < config_->eqConfigs().size(); i++) {
255  DLOG("EQ Parametrization " << i);
256  boost::shared_ptr<EqBsData> eq = config_->eqConfigs()[i];
257  string eqName = eq->eqName();
258  QuantLib::Currency eqCcy = ore::data::parseCurrency(eq->currency());
259  QL_REQUIRE(std::find(currencies.begin(), currencies.end(), eqCcy.code()) != currencies.end(),
260  "Currency (" << eqCcy << ") for equity " << eqName << " not covered by CrossAssetModelData");
261  boost::shared_ptr<EqBsBuilder> builder = boost::make_shared<EqBsBuilder>(
262  market_, eq, domesticCcy, configurationEqCalibration_, referenceCalibrationGrid_);
263  boost::shared_ptr<QuantExt::EqBsParametrization> parametrization = builder->parametrization();
264  eqOptionBaskets_[i] = builder->optionBasket();
265  eqParametrizations.push_back(parametrization);
266  eqNames.push_back(eqName);
267  subBuilders_[CT::EQ][i] = builder;
268  processInfo[CT::EQ].emplace_back(eqName, 1);
269  }
270 
271  // Build the INF parametrizations and calibration baskets
272  vector<boost::shared_ptr<Parametrization>> infParameterizations;
273  for (Size i = 0; i < config_->infConfigs().size(); i++) {
274  boost::shared_ptr<InflationModelData> imData = config_->infConfigs()[i];
275  DLOG("Inflation parameterisation (" << i << ") for index " << imData->index());
276  if (auto dkData = boost::dynamic_pointer_cast<InfDkData>(imData)) {
277  boost::shared_ptr<InfDkBuilder> builder = boost::make_shared<InfDkBuilder>(
278  market_, dkData, configurationInfCalibration_, referenceCalibrationGrid_, dontCalibrate_);
279  if (dontCalibrate_)
280  builder->freeze();
281  infParameterizations.push_back(builder->parametrization());
282  subBuilders_[CT::INF][i] = builder;
283  processInfo[CT::INF].emplace_back(dkData->index(), 1);
284  } else if (auto jyData = boost::dynamic_pointer_cast<InfJyData>(imData)) {
285  boost::shared_ptr<InfJyBuilder> builder = boost::make_shared<InfJyBuilder>(
286  market_, jyData, configurationInfCalibration_, referenceCalibrationGrid_);
287  infParameterizations.push_back(builder->parameterization());
288  subBuilders_[CT::INF][i] = builder;
289  processInfo[CT::INF].emplace_back(jyData->index(), 2);
290  } else {
291  QL_FAIL("CrossAssetModelBuilder expects either DK or JY inflation model data.");
292  }
293  infIndices.push_back(imData->index());
294  }
295 
296  /*******************************************************
297  * Build the CR parametrizations and calibration baskets
298  */
299  // LGM (if any)
300  std::vector<boost::shared_ptr<QuantExt::CrLgm1fParametrization>> crLgmParametrizations;
301  for (Size i = 0; i < config_->crLgmConfigs().size(); ++i) {
302  LOG("CR LGM Parametrization " << i);
303  boost::shared_ptr<CrLgmData> cr = config_->crLgmConfigs()[i];
304  string crName = cr->name();
305  boost::shared_ptr<CrLgmBuilder> builder =
306  boost::make_shared<CrLgmBuilder>(market_, cr, configurationCrCalibration_);
307  boost::shared_ptr<QuantExt::CrLgm1fParametrization> parametrization = builder->parametrization();
308  crLgmParametrizations.push_back(parametrization);
309  crNames.push_back(crName);
310  subBuilders_[CT::CR][i] = builder;
311  processInfo[CT::CR].emplace_back(crName, 1);
312  }
313 
314  // CIR (if any)
315  std::vector<boost::shared_ptr<QuantExt::CrCirppParametrization>> crCirParametrizations;
316  for (Size i = 0; i < config_->crCirConfigs().size(); ++i) {
317  LOG("CR CIR Parametrization " << i);
318  boost::shared_ptr<CrCirData> cr = config_->crCirConfigs()[i];
319  string crName = cr->name();
320  boost::shared_ptr<CrCirBuilder> builder =
321  boost::make_shared<CrCirBuilder>(market_, cr, configurationCrCalibration_);
322  boost::shared_ptr<QuantExt::CrCirppParametrization> parametrization = builder->parametrization();
323  crCirParametrizations.push_back(parametrization);
324  crNames.push_back(crName);
325  subBuilders_[CT::CR][i] = builder;
326  processInfo[CT::CR].emplace_back(crName, 1);
327  }
328 
329  std::vector<boost::shared_ptr<QuantExt::Parametrization>> parametrizations;
330  for (Size i = 0; i < irParametrizations.size(); i++)
331  parametrizations.push_back(irParametrizations[i]);
332  for (Size i = 0; i < fxParametrizations.size(); i++)
333  parametrizations.push_back(fxParametrizations[i]);
334  for (Size i = 0; i < eqParametrizations.size(); i++)
335  parametrizations.push_back(eqParametrizations[i]);
336  parametrizations.insert(parametrizations.end(), infParameterizations.begin(), infParameterizations.end());
337  for (Size i = 0; i < crLgmParametrizations.size(); i++)
338  parametrizations.push_back(crLgmParametrizations[i]);
339  for (Size i = 0; i < crCirParametrizations.size(); i++)
340  parametrizations.push_back(crCirParametrizations[i]);
341 
342  QL_REQUIRE(fxParametrizations.size() == irParametrizations.size() - 1, "mismatch in IR/FX parametrization sizes");
343 
344  Measure::Type measure = Measure::LGM;
345  if (config_->measure() == "BA") {
346  measure = Measure::BA;
347  DLOG("Setting measure to BA");
348  } else if (config_->measure() == "LGM") {
349  measure = Measure::LGM;
350  DLOG("Setting measure to BA");
351  } else if (config_->measure() == "") {
352  DLOG("Defaulting to LGM measure");
353  } else {
354  QL_FAIL("Measure " << config_->measure() << " not recognized");
355  }
356 
357  // Tag on the parametrization and process info for the auxiliary state variable in the bank account measure
358  if (measure == Measure::BA) {
359  parametrizations.push_back(
360  irParametrizations[0]); // FIXME: Is index 0 safe to reference the domestic IR parameters?
361  processInfo[CT::AUX].emplace_back(config_->domesticCurrency(), 1);
362  }
363 
364  /******************************
365  * Build the correlation matrix
366  */
367  DLOG("CrossAssetModelBuilder: adding correlations.");
368  CorrelationMatrixBuilder cmb;
369 
370  // Perfect instantaneous correlation of auxiliary variable and domestic LGM state variable in the bank account
371  // measure
372  CorrelationFactor domesticFactorIR = {CrossAssetModelTypes::IR, config_->domesticCurrency(), 0};
373  CorrelationFactor domesticFactorAUX = {CrossAssetModelTypes::AUX, config_->domesticCurrency(), 0};
374  if (measure == Measure::BA)
375  cmb.addCorrelation(domesticFactorAUX, domesticFactorIR, 1.0);
376 
377  for (auto it = config_->correlations().begin(); it != config_->correlations().end(); it++) {
378  cmb.addCorrelation(it->first.first, it->first.second, it->second);
379  if (measure == Measure::BA) {
380  // Copy correlation(domesticIR, other) to correlation(domesticAUX, other)
381  if (it->first.first == domesticFactorIR)
382  cmb.addCorrelation(domesticFactorAUX, it->first.second, it->second);
383  if (it->first.second == domesticFactorIR)
384  cmb.addCorrelation(it->first.first, domesticFactorAUX, it->second);
385  }
386  }
387 
388  Matrix corrMatrix = cmb.correlationMatrix(processInfo);
389 
390  TLOG("CAM correlation matrix:");
391  TLOGGERSTREAM << corrMatrix;
392 
393  /*****************************
394  * Build the cross asset model
395  */
396 
397  SalvagingAlgorithm::Type salvaging = SalvagingAlgorithm::None;
398 
399  model_.linkTo(boost::make_shared<QuantExt::CrossAssetModel>(parametrizations, corrMatrix, salvaging, measure));
400 
401  /*************************
402  * Calibrate IR components
403  */
404 
405  for (Size i = 0; i < irBuilder.size(); i++) {
406  DLOG("IR Calibration " << i);
407  swaptionCalibrationErrors_[i] = irBuilder[i]->error();
408  }
409 
410  /*************************
411  * Relink LGM discount curves to curves used for FX calibration
412  */
413 
414  for (Size i = 0; i < irParametrizations.size(); i++) {
415  auto p = irParametrizations[i];
416  irDiscountCurves[i].linkTo(*market_->discountCurve(p->currency().code(), configurationFxCalibration_));
417  DLOG("Relinked discounting curve for " << p->currency().code() << " for FX calibration");
418  }
419 
420  /*************************
421  * Calibrate FX components
422  */
423 
424  for (Size i = 0; i < fxParametrizations.size(); i++) {
425  boost::shared_ptr<FxBsData> fx = config_->fxConfigs()[i];
426 
427  if (fx->calibrationType() == CalibrationType::None || !fx->calibrateSigma()) {
428  DLOG("FX Calibration " << i << " skipped");
429  continue;
430  }
431 
432  DLOG("FX Calibration " << i);
433 
434  // attach pricing engines to helpers
435  boost::shared_ptr<QuantExt::AnalyticCcLgmFxOptionEngine> engine =
436  boost::make_shared<QuantExt::AnalyticCcLgmFxOptionEngine>(*model_, i);
437  // enable caching for calibration
438  // TODO: review this
439  engine->cache(true);
440  for (Size j = 0; j < fxOptionBaskets_[i].size(); j++)
441  fxOptionBaskets_[i][j]->setPricingEngine(engine);
442 
443  if (!dontCalibrate_) {
444 
445  if (fx->calibrationType() == CalibrationType::Bootstrap && fx->sigmaParamType() == ParamType::Piecewise)
446  model_->calibrateBsVolatilitiesIterative(CrossAssetModelTypes::FX, i, fxOptionBaskets_[i],
447  *optimizationMethod_, endCriteria_);
448  else
449  model_->calibrateBsVolatilitiesGlobal(CrossAssetModelTypes::FX, i, fxOptionBaskets_[i],
450  *optimizationMethod_, endCriteria_);
451 
452  DLOG("FX " << fx->foreignCcy() << " calibration errors:");
453  fxOptionCalibrationErrors_[i] = getCalibrationError(fxOptionBaskets_[i]);
454  if (fx->calibrationType() == CalibrationType::Bootstrap) {
455  if (fabs(fxOptionCalibrationErrors_[i]) < config_->bootstrapTolerance()) {
456  // we check the log level here to avoid unncessary computations
457  if (Log::instance().filter(ORE_DATA)) {
458  TLOGGERSTREAM << "Calibration details:";
459  TLOGGERSTREAM << getCalibrationDetails(fxOptionBaskets_[i], fxParametrizations[i],
460  irParametrizations[0]);
461  TLOGGERSTREAM << "rmse = " << fxOptionCalibrationErrors_[i];
462  }
463  } else {
464  std::string exceptionMessage = "FX BS " + std::to_string(i) + " calibration error " +
465  std::to_string(fxOptionCalibrationErrors_[i]) +
466  " exceeds tolerance " +
467  std::to_string(config_->bootstrapTolerance());
468  WLOG(StructuredModelErrorMessage("Failed to calibrate FX BS Model", exceptionMessage));
469  WLOGGERSTREAM << "Calibration details:";
470  WLOGGERSTREAM << getCalibrationDetails(fxOptionBaskets_[i], fxParametrizations[i],
471  irParametrizations[0]);
472  WLOGGERSTREAM << "rmse = " << fxOptionCalibrationErrors_[i];
473  if (!continueOnError_)
474  QL_FAIL(exceptionMessage);
475  }
476  }
477  }
478  }
479 
480  /*************************
481  * Relink LGM discount curves to curves used for EQ calibration
482  */
483 
484  for (Size i = 0; i < irParametrizations.size(); i++) {
485  auto p = irParametrizations[i];
486  irDiscountCurves[i].linkTo(*market_->discountCurve(p->currency().code(), configurationEqCalibration_));
487  DLOG("Relinked discounting curve for " << p->currency().code() << " for EQ calibration");
488  }
489 
490  /*************************
491  * Calibrate EQ components
492  */
493 
494  for (Size i = 0; i < eqParametrizations.size(); i++) {
495  boost::shared_ptr<EqBsData> eq = config_->eqConfigs()[i];
496  if (!eq->calibrateSigma()) {
497  DLOG("EQ Calibration " << i << " skipped");
498  continue;
499  }
500  DLOG("EQ Calibration " << i);
501  // attach pricing engines to helpers
502  Currency eqCcy = eqParametrizations[i]->currency();
503  Size eqCcyIdx = model_->ccyIndex(eqCcy);
504  boost::shared_ptr<QuantExt::AnalyticXAssetLgmEquityOptionEngine> engine =
505  boost::make_shared<QuantExt::AnalyticXAssetLgmEquityOptionEngine>(*model_, i, eqCcyIdx);
506  for (Size j = 0; j < eqOptionBaskets_[i].size(); j++)
507  eqOptionBaskets_[i][j]->setPricingEngine(engine);
508 
509  if (!dontCalibrate_) {
510 
511  if (eq->calibrationType() == CalibrationType::Bootstrap && eq->sigmaParamType() == ParamType::Piecewise)
512  model_->calibrateBsVolatilitiesIterative(CrossAssetModelTypes::EQ, i, eqOptionBaskets_[i],
513  *optimizationMethod_, endCriteria_);
514  else
515  model_->calibrateBsVolatilitiesGlobal(CrossAssetModelTypes::EQ, i, eqOptionBaskets_[i],
516  *optimizationMethod_, endCriteria_);
517  DLOG("EQ " << eq->eqName() << " calibration errors:");
518  eqOptionCalibrationErrors_[i] = getCalibrationError(eqOptionBaskets_[i]);
519  if (eq->calibrationType() == CalibrationType::Bootstrap) {
520  if (fabs(eqOptionCalibrationErrors_[i]) < config_->bootstrapTolerance()) {
521  // we check the log level here to avoid unncessary computations
522  if (Log::instance().filter(ORE_DATA)) {
523  TLOGGERSTREAM << "Calibration details:";
524  TLOGGERSTREAM << getCalibrationDetails(eqOptionBaskets_[i], eqParametrizations[i],
525  irParametrizations[0]);
526  TLOGGERSTREAM << "rmse = " << eqOptionCalibrationErrors_[i];
527  }
528  } else {
529  std::string exceptionMessage = "EQ BS " + std::to_string(i) + " calibration error " +
530  std::to_string(eqOptionCalibrationErrors_[i]) +
531  " exceeds tolerance " +
532  std::to_string(config_->bootstrapTolerance());
533  WLOG(StructuredModelErrorMessage("Failed to calibrate EQ BS Model", exceptionMessage));
534  WLOGGERSTREAM << "Calibration details:";
535  WLOGGERSTREAM << getCalibrationDetails(eqOptionBaskets_[i], eqParametrizations[i],
536  irParametrizations[0]);
537  WLOGGERSTREAM << "rmse = " << eqOptionCalibrationErrors_[i];
538  if (!continueOnError_)
539  QL_FAIL(exceptionMessage);
540  }
541  }
542  }
543  }
544 
545  /*************************
546  * Relink LGM discount curves to curves used for INF calibration
547  */
548 
549  for (Size i = 0; i < irParametrizations.size(); i++) {
550  auto p = irParametrizations[i];
551  irDiscountCurves[i].linkTo(*market_->discountCurve(p->currency().code(), configurationInfCalibration_));
552  DLOG("Relinked discounting curve for " << p->currency().code() << " for INF calibration");
553  }
554 
555  // Calibrate INF components
556  for (Size i = 0; i < infParameterizations.size(); i++) {
557  boost::shared_ptr<InflationModelData> imData = config_->infConfigs()[i];
558  if (auto dkData = boost::dynamic_pointer_cast<InfDkData>(imData)) {
559  auto dkParam = boost::dynamic_pointer_cast<InfDkParametrization>(infParameterizations[i]);
560  QL_REQUIRE(dkParam, "Expected DK model data to have given a DK parameterisation.");
561  const auto& builder = subBuilders_.at(CT::INF).at(i);
562  const auto& dkBuilder = boost::dynamic_pointer_cast<InfDkBuilder>(builder);
563  calibrateInflation(*dkData, i, dkBuilder->optionBasket(), dkParam);
564  } else if (auto jyData = boost::dynamic_pointer_cast<InfJyData>(imData)) {
565  auto jyParam = boost::dynamic_pointer_cast<InfJyParameterization>(infParameterizations[i]);
566  QL_REQUIRE(jyParam, "Expected JY model data to have given a JY parameterisation.");
567  const auto& builder = subBuilders_.at(CT::INF).at(i);
568  const auto& jyBuilder = boost::dynamic_pointer_cast<InfJyBuilder>(builder);
569  calibrateInflation(*jyData, i, jyBuilder, jyParam);
570  } else {
571  QL_FAIL("CrossAssetModelBuilder expects either DK or JY inflation model data.");
572  }
573  }
574 
575  /*************************
576  * Relink LGM discount curves to final model curves
577  */
578 
579  for (Size i = 0; i < irParametrizations.size(); i++) {
580  auto p = irParametrizations[i];
581  irDiscountCurves[i].linkTo(*market_->discountCurve(p->currency().code(), configurationFinalModel_));
582  DLOG("Relinked discounting curve for " << p->currency().code() << " as final model curves");
583  }

586  // this point from all what we know...)
587  model_->update();
588 
589  DLOG("Building CrossAssetModel done");

598 void CrossAssetModelBuilder::calibrateInflation(const InfDkData& data, Size modelIdx,
599  const vector<boost::shared_ptr<BlackCalibrationHelper>>& cb,
600  const boost::shared_ptr<InfDkParametrization>& inflationParam) const {
601 
602  LOG("Calibrate DK inflation model for inflation index " << data.index());
603 
604  if ((!data.volatility().calibrate() && !data.reversion().calibrate()) ||
605  (data.calibrationType() == CalibrationType::None)) {
606  LOG("Calibration of DK inflation model for inflation index " << data.index() << " not requested.");
607  return;
608  }
609 
610  Handle<ZeroInflationIndex> zInfIndex =
611  market_->zeroInflationIndex(model_->infdk(modelIdx)->name(), configurationInfCalibration_);
612  Real baseCPI = dontCalibrate_ ? 100. : zInfIndex->fixing(zInfIndex->zeroInflationTermStructure()->baseDate());
613  auto engine = boost::make_shared<QuantExt::AnalyticDkCpiCapFloorEngine>(*model_, modelIdx, baseCPI);
614  for (Size j = 0; j < cb.size(); j++)
615  cb[j]->setPricingEngine(engine);
616 
617  if (dontCalibrate_)
618  return;
619 
620  if (data.volatility().calibrate() && !data.reversion().calibrate()) {
621  if (data.calibrationType() == CalibrationType::Bootstrap && data.volatility().type() == ParamType::Piecewise) {
622  model_->calibrateInfDkVolatilitiesIterative(modelIdx, cb, *optimizationMethod_, endCriteria_);
623  } else {
624  model_->calibrateInfDkVolatilitiesGlobal(modelIdx, cb, *optimizationMethod_, endCriteria_);
625  }
626  } else if (!data.volatility().calibrate() && data.reversion().calibrate()) {
627  if (data.calibrationType() == CalibrationType::Bootstrap && data.reversion().type() == ParamType::Piecewise) {
628  model_->calibrateInfDkReversionsIterative(modelIdx, cb, *optimizationMethod_, endCriteria_);
629  } else {
630  model_->calibrateInfDkReversionsGlobal(modelIdx, cb, *optimizationMethod_, endCriteria_);
631  }
632  } else {
633  model_->calibrate(cb, *optimizationMethod_, endCriteria_);
634  }
635 
636  DLOG("INF (DK) " << data.index() << " calibration errors:");
637  inflationCalibrationErrors_[modelIdx] = getCalibrationError(cb);
638  if (data.calibrationType() == CalibrationType::Bootstrap) {
639  if (fabs(inflationCalibrationErrors_[modelIdx]) < config_->bootstrapTolerance()) {
640  // we check the log level here to avoid unncessary computations
641  if (Log::instance().filter(ORE_DATA)) {
642  TLOGGERSTREAM << "Calibration details:";
643  TLOGGERSTREAM << getCalibrationDetails(cb, inflationParam);
644  TLOGGERSTREAM << "rmse = " << inflationCalibrationErrors_[modelIdx];
645  }
646  } else {
647  string exceptionMessage = "INF (DK) " + std::to_string(modelIdx) + " calibration error " +
648  std::to_string(inflationCalibrationErrors_[modelIdx]) + " exceeds tolerance " +
649  std::to_string(config_->bootstrapTolerance());
650  WLOG(StructuredModelErrorMessage("Failed to calibrate INF DK Model", exceptionMessage));
651  WLOGGERSTREAM << "Calibration details:";
652  WLOGGERSTREAM << getCalibrationDetails(cb, inflationParam);
653  WLOGGERSTREAM << "rmse = " << inflationCalibrationErrors_[modelIdx];
654  if (!continueOnError_)
655  QL_FAIL(exceptionMessage);
656  }
657  }

660 void CrossAssetModelBuilder::calibrateInflation(const InfJyData& data, Size modelIdx,
661  const boost::shared_ptr<InfJyBuilder>& jyBuilder,
662  const boost::shared_ptr<InfJyParameterization>& inflationParam) const {
663 
664  LOG("Calibrate JY inflation model for inflation index " << data.index());
665 
666  const auto& rrVol = data.realRateVolatility();
667  const auto& rrRev = data.realRateReversion();
668  const auto& idxVol = data.indexVolatility();
669 
670  // Check if calibration is needed at all.
671  if ((!rrVol.calibrate() && !rrRev.calibrate() && !idxVol.calibrate()) ||
672  (data.calibrationType() == CalibrationType::None)) {
673  LOG("Calibration of JY inflation model for inflation index " << data.index() << " not requested.");
674  return;
675  }
676 
677  // We will need the 2 baskets of helpers
678  auto rrBasket = jyBuilder->realRateBasket();
679  auto idxBasket = jyBuilder->indexBasket();
680 
681  // Attach engines to the helpers.
682  setJyPricingEngine(modelIdx, rrBasket);
683  setJyPricingEngine(modelIdx, idxBasket);
684 
685  if (dontCalibrate_)
686  return;
687 
688  // Single basket of helpers is useful in various places below.
689  vector<boost::shared_ptr<CalibrationHelper>> allHelpers = rrBasket;
690  allHelpers.insert(allHelpers.end(), idxBasket.begin(), idxBasket.end());
691 
692  // Calibration configuration.
693  const auto& cc = data.calibrationConfiguration();
694 
695  if (data.calibrationType() == CalibrationType::BestFit) {
696 
697  // If calibration type is BestFit, do a global optimisation on the parameters that need to be calibrated.
698  DLOG("Calibration BestFit of JY inflation model for inflation index " << data.index() << " requested.");
699 
700  // Indicate the parameters to calibrate
701  map<Size, bool> toCalibrate;
702  toCalibrate[0] = rrVol.calibrate();
703  toCalibrate[1] = rrRev.calibrate();
704  toCalibrate[2] = idxVol.calibrate();
705 
706  // Calibrate the model.
707  model_->calibrateInfJyGlobal(modelIdx, allHelpers, *optimizationMethod_, endCriteria_, toCalibrate);
708 
709  } else {
710 
711  // Calibration type is now Bootstrap, there are multiple options.
712  QL_REQUIRE(data.calibrationType() == CalibrationType::Bootstrap,
713  "JY inflation calibration expected a "
714  << "calibration type of None, BestFit or Bootstrap.");
715  QL_REQUIRE(!(rrRev.calibrate() && rrVol.calibrate()),
716  "Calibrating both the "
717  << "real rate reversion and real rate volatility using Bootstrap is not supported.");
718 
719  if ((!rrVol.calibrate() && !rrRev.calibrate()) && idxVol.calibrate()) {
720 
721  // Bootstrap the inflation index volatility only.
722  DLOG("Bootstrap calibration of JY index volatility for index " << data.index() << ".");
723  QL_REQUIRE(idxVol.type() == ParamType::Piecewise, "Index volatility parameter should be Piecewise for "
724  << "a Bootstrap calibration.");
725  model_->calibrateInfJyIterative(modelIdx, 2, idxBasket, *optimizationMethod_, endCriteria_);
726 
727  } else if (rrVol.calibrate() && !idxVol.calibrate()) {
728 
729  // Bootstrap the real rate volatility only
730  DLOG("Bootstrap calibration of JY real rate volatility for index " << data.index() << ".");
731  QL_REQUIRE(rrVol.type() == ParamType::Piecewise, "Real rate volatility parameter should be "
732  << "Piecewise for a Bootstrap calibration.");
733  model_->calibrateInfJyIterative(modelIdx, 0, rrBasket, *optimizationMethod_, endCriteria_);
734 
735  } else if (rrRev.calibrate() && !idxVol.calibrate()) {
736 
737  // Bootstrap the real rate reversion only
738  DLOG("Bootstrap calibration of JY real rate reversion for index " << data.index() << ".");
739  QL_REQUIRE(rrRev.type() == ParamType::Piecewise, "Real rate reversion parameter should be "
740  << "Piecewise for a Bootstrap calibration.");
741  model_->calibrateInfJyIterative(modelIdx, 1, rrBasket, *optimizationMethod_, endCriteria_);
742 
743  } else if ((rrVol.calibrate() && idxVol.calibrate()) || (rrRev.calibrate() && idxVol.calibrate())) {
744 
745  if (rrVol.calibrate()) {
746  DLOG("Bootstrap calibration of JY real rate volatility and index volatility for index " << data.index()
747  << ".");
748  } else {
749  DLOG("Bootstrap calibration of JY real rate reversion and index volatility for index " << data.index()
750  << ".");
751  }
752 
753  // Bootstrap the real rate volatility and the index volatility
754  Size rrIdx = rrVol.calibrate() ? 0 : 1;
755  Size numIts = 0;
756  inflationCalibrationErrors_[modelIdx] = getCalibrationError(allHelpers);
757 
758  while (inflationCalibrationErrors_[modelIdx] > cc.rmseTolerance() && numIts < cc.maxIterations()) {
759  model_->calibrateInfJyIterative(modelIdx, 2, idxBasket, *optimizationMethod_, endCriteria_);
760  model_->calibrateInfJyIterative(modelIdx, rrIdx, rrBasket, *optimizationMethod_, endCriteria_);
761  numIts++;
762  inflationCalibrationErrors_[modelIdx] = getCalibrationError(allHelpers);
763  }
764 
765  DLOG("Bootstrap calibration of JY model stopped with number of iterations "
766  << numIts << " and rmse equal to " << std::scientific << std::setprecision(6)
767  << inflationCalibrationErrors_[modelIdx] << ".");
768 
769  } else {
770  QL_FAIL("JY inflation bootstrap calibration does not support the combination of real rate volatility = "
771  << std::boolalpha << rrVol.calibrate() << ", real rate reversion = " << rrRev.calibrate() << " and "
772  << "index volatility = " << idxVol.calibrate() << ".");
773  }
774  }
775 
776  // Log the calibration details.
777  DLOG("INF (JY) " << data.index() << " calibration errors:");
778  inflationCalibrationErrors_[modelIdx] = getCalibrationError(allHelpers);
779  if (data.calibrationType() == CalibrationType::Bootstrap) {
780  if (fabs(inflationCalibrationErrors_[modelIdx]) < cc.rmseTolerance()) {
781  // we check the log level here to avoid unncessary computations
782  if (Log::instance().filter(ORE_DATA)) {
783  TLOGGERSTREAM << "Calibration details:";
784  TLOGGERSTREAM << getCalibrationDetails(rrBasket, idxBasket, inflationParam, rrVol.calibrate());
785  TLOGGERSTREAM << "rmse = " << inflationCalibrationErrors_[modelIdx];
786  }
787  } else {
788  std::stringstream ss;
789  ss << "INF (JY) " << modelIdx << " calibration error " << std::scientific
790  << inflationCalibrationErrors_[modelIdx] << " exceeds tolerance " << cc.rmseTolerance();
791  string exceptionMessage = ss.str();
792  WLOG(StructuredModelErrorMessage("Failed to calibrate INF JY Model", exceptionMessage));
793  WLOGGERSTREAM << "Calibration details:";
794  WLOGGERSTREAM << getCalibrationDetails(rrBasket, idxBasket, inflationParam, rrVol.calibrate());
795  WLOGGERSTREAM << "rmse = " << inflationCalibrationErrors_[modelIdx];
796  if (!continueOnError_)
797  QL_FAIL(exceptionMessage);
798  }
799  }
800 
801  LOG("Finished calibrating JY inflation model for inflation index " << data.index());
802 }

804 void CrossAssetModelBuilder::setJyPricingEngine(
805  Size modelIdx, const vector<boost::shared_ptr<CalibrationHelper>>& calibrationBasket) const {
806 
807  DLOG("Start setting pricing engines on JY calibration instruments.");
808 
809  // JY supports three types of calibration helpers. Generally, all of the calibration instruments in a basket will
810  // be of the same type but we support all three here.
811  boost::shared_ptr<PricingEngine> cpiCapFloorEngine;
812  boost::shared_ptr<PricingEngine> yoyCapFloorEngine;
813  boost::shared_ptr<PricingEngine> yoySwapEngine;
814  boost::shared_ptr<InflationCouponPricer> yoyCouponPricer;
815 
816  for (auto& ci : calibrationBasket) {
817 
818  if (boost::shared_ptr<CpiCapFloorHelper> h = boost::dynamic_pointer_cast<CpiCapFloorHelper>(ci)) {
819  if (!cpiCapFloorEngine) {
820  cpiCapFloorEngine = boost::make_shared<AnalyticJyCpiCapFloorEngine>(*model_, modelIdx);
821  }
822  h->setPricingEngine(cpiCapFloorEngine);
823  continue;
824  }
825 
826  if (boost::shared_ptr<YoYCapFloorHelper> h = boost::dynamic_pointer_cast<YoYCapFloorHelper>(ci)) {
827  if (!yoyCapFloorEngine) {
828  yoyCapFloorEngine = boost::make_shared<AnalyticJyYoYCapFloorEngine>(*model_, modelIdx);
829  }
830  h->setPricingEngine(yoyCapFloorEngine);
831  continue;
832  }
833 
834  if (boost::shared_ptr<YoYSwapHelper> h = boost::dynamic_pointer_cast<YoYSwapHelper>(ci)) {
835  // Here we need to attach the coupon pricer to all the YoY coupons and then the generic discounting swap
836  // engine to the helper.
837  if (!yoyCouponPricer) {
838  yoyCouponPricer = boost::make_shared<JyYoYInflationCouponPricer>(*model_, modelIdx);
839 
840  Size irIdx = model_->ccyIndex(model_->infjy(modelIdx)->currency());
841  auto yts = model_->irlgm1f(irIdx)->termStructure();
842  yoySwapEngine = boost::make_shared<DiscountingSwapEngine>(yts);
843  }
844 
845  const auto& yoyLeg = h->yoySwap()->yoyLeg();
846  for (const auto& cf : yoyLeg) {
847  if (auto yoyCoupon = boost::dynamic_pointer_cast<YoYInflationCoupon>(cf))
848  yoyCoupon->setPricer(yoyCouponPricer);
849  }
850 
851  h->setPricingEngine(yoySwapEngine);
852  continue;
853  }
854 
855  QL_FAIL("Only CPI cap floors, YoY cap floors and YoY swaps are supported for JY calibration.");
856  }
857 
858  DLOG("Finished setting pricing engines on JY calibration instruments.");
859 }
