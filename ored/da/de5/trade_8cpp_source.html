53 void Trade::addPremiums(std::vector<boost::shared_ptr<Instrument>>& addInstruments, std::vector<Real>& addMultipliers,
54  const Real tradeMultiplier, const PremiumData& premiumData, const Real premiumMultiplier,
55  const Currency& tradeCurrency, const boost::shared_ptr<EngineFactory>& factory,
56  const string& configuration) {
57 
58  for (auto const& d : premiumData.premiumData()) {
59  QL_REQUIRE(d.amount != Null<Real>(), "Trade contains invalid premium data.");
60 
61  Currency premiumCurrency = parseCurrencyWithMinors(d.ccy);
62  Real premiumAmount = convertMinorToMajorCurrency(d.ccy, d.amount);
63  auto fee = boost::make_shared<QuantExt::Payment>(premiumAmount, premiumCurrency, d.payDate);
64 
65  addMultipliers.push_back(premiumMultiplier);
66 
67  Handle<YieldTermStructure> yts = factory->market()->discountCurve(d.ccy, configuration);
68  Handle<Quote> fx;
69  if (tradeCurrency.code() != d.ccy) {
70  fx = factory->market()->fxSpot(d.ccy + tradeCurrency.code(), configuration);
71  }
72  boost::shared_ptr<PricingEngine> discountingEngine(new QuantExt::PaymentDiscountingEngine(yts, fx));
73  fee->setPricingEngine(discountingEngine);
74 
75  // 1) Add to additional instruments for pricing
76  addInstruments.push_back(fee);
77 
78  // 2) Add a trade leg for cash flow reporting, divide the amount by the multiplier, because the leg entries
79  // are multiplied with the trade multiplier in the cashflow report (and if used elsewhere)
80  legs_.push_back(
81  Leg(1, boost::make_shared<SimpleCashFlow>(fee->cashFlow()->amount() * premiumMultiplier / tradeMultiplier,
82  fee->cashFlow()->date())));
83  legCurrencies_.push_back(fee->currency().code());
84 
85  // premium * premiumMultiplier reflects the correct pay direction, set payer to false therefore
86  legPayers_.push_back(false);
87 
88  DLOG("added fee " << d.amount << " " << d.ccy << " payable on " << d.payDate << " to trade");
89  }
90 }

92 void Trade::validate() const {
93  QL_REQUIRE(id_ != "", "Trade id has not been set.");
94  QL_REQUIRE(tradeType_ != "", "Trade id has not been set.");
95  QL_REQUIRE(instrument_ || legs_.size() > 0,
96  "Trade " << id_ << " requires either QuantLib instruments or legs to be created.");
97  QL_REQUIRE(npvCurrency_ != "", "NPV currency has not been set for trade " << id_ << ".");
98  // QL_REQUIRE(notional_ != Null<Real>(), "Notional has not been set for trade " << id_ << ".");
99  // QL_REQUIRE(notionalCurrency_ != "", "Notional currency has not been set for trade " << id_ << ".");
100  QL_REQUIRE(maturity_ != Null<Date>(), "Maturity not set for trade " << id_ << ".");
101  QL_REQUIRE(!envelope_.empty(), "Envelope not set for trade " << id_ << ".");
102  if (legs_.size() > 0) {
103  QL_REQUIRE(legs_.size() == legPayers_.size(),
104  "Inconsistent number of pay/receive indicators for legs in trade " << id_ << ".");
105  QL_REQUIRE(legs_.size() == legCurrencies_.size(),
106  "Inconsistent number of leg currencies for legs in trade " << id_ << ".");
107  }
108 }
