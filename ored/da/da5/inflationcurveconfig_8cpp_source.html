52 void InflationCurveConfig::fromXML(XMLNode* node) {
53  XMLUtils::checkNode(node, "InflationCurve");
54 
55  curveID_ = XMLUtils::getChildValue(node, "CurveId", true);
56  curveDescription_ = XMLUtils::getChildValue(node, "CurveDescription", true);
57  nominalTermStructure_ = XMLUtils::getChildValue(node, "NominalTermStructure", true);
58 
59  string type = XMLUtils::getChildValue(node, "Type", true);
60  if (type == "ZC") {
61  type_ = Type::ZC;
62  } else if (type == "YY") {
63  type_ = Type::YY;
64  } else
65  QL_FAIL("Type " << type << " not recognized");
66 
67  swapQuotes_ = XMLUtils::getChildrenValues(node, "Quotes", "Quote", true);
68 
69  conventions_ = XMLUtils::getChildValue(node, "Conventions", true);
70  extrapolate_ = XMLUtils::getChildValueAsBool(node, "Extrapolation", false);
71 
72  string cal = XMLUtils::getChildValue(node, "Calendar", true);
73  calendar_ = parseCalendar(cal);
74 
75  string dc = XMLUtils::getChildValue(node, "DayCounter", true);
76  dayCounter_ = parseDayCounter(dc);
77 
78  string lag = XMLUtils::getChildValue(node, "Lag", true);
79  lag_ = parsePeriod(lag);
80 
81  string freq = XMLUtils::getChildValue(node, "Frequency", true);
82  frequency_ = parseFrequency(freq);
83 
84  baseRate_ = QuantLib::Null<Real>();
85  string baseZr = XMLUtils::getChildValue(node, "BaseRate", false);
86  if (baseZr != "")
87  baseRate_ = parseReal(baseZr);
88 
89  string tol = XMLUtils::getChildValue(node, "Tolerance", true);
90  tolerance_ = parseReal(tol);
91 
92  XMLNode* seasonalityNode = XMLUtils::getChildNode(node, "Seasonality");
93  seasonalityBaseDate_ = QuantLib::Null<Date>();
94  seasonalityFrequency_ = QuantLib::NoFrequency;
95  seasonalityFactors_.clear();
96  quotes_ = swapQuotes_;
97  if (seasonalityNode != nullptr) {
98  seasonalityBaseDate_ = parseDate(XMLUtils::getChildValue(seasonalityNode, "BaseDate", true));
99  seasonalityFrequency_ = parseFrequency(XMLUtils::getChildValue(seasonalityNode, "Frequency", true));
100  seasonalityFactors_ = XMLUtils::getChildrenValues(seasonalityNode, "Factors", "Factor", false);
101  quotes_.insert(quotes_.end(), seasonalityFactors_.begin(), seasonalityFactors_.end());
102  std::string overrideFctStr = XMLUtils::getChildValue(seasonalityNode, "OverrideFactors", false);
103  overrideSeasonalityFactors_ = parseListOfValues<Real>(overrideFctStr, &parseReal);
104  }
105  populateRequiredCurveIds();
106 }

108 XMLNode* InflationCurveConfig::toXML(XMLDocument& doc) {
109  XMLNode* node = doc.allocNode("InflationCurve");
110 
111  XMLUtils::addChild(doc, node, "CurveId", curveID_);
112  XMLUtils::addChild(doc, node, "CurveDescription", curveDescription_);
113  XMLUtils::addChild(doc, node, "NominalTermStructure", nominalTermStructure_);
114 
115  if (type_ == Type::ZC) {
116  XMLUtils::addChild(doc, node, "Type", "ZC");
117  } else if (type_ == Type::YY) {
118  XMLUtils::addChild(doc, node, "Type", "YY");
119  } else
120  QL_FAIL("Unknown Type in InflationCurveConfig::toXML()");
121 
122  XMLUtils::addChildren(doc, node, "Quotes", "Quote", swapQuotes_);
123  XMLUtils::addChild(doc, node, "Conventions", conventions_);
124  string extrap = (extrapolate_ ? "true" : "false");
125  XMLUtils::addChild(doc, node, "Extrapolation", extrap);
126 
127  string baseRateStr;
128  if (baseRate_ != QuantLib::Null<Real>())
129  baseRateStr = to_string(baseRate_);
130  else
131  baseRateStr = "";
132 
133  XMLUtils::addChild(doc, node, "Calendar", to_string(calendar_));
134  XMLUtils::addChild(doc, node, "DayCounter", to_string(dayCounter_));
135  XMLUtils::addChild(doc, node, "Lag", to_string(lag_));
136  XMLUtils::addChild(doc, node, "Frequency", to_string(frequency_));
137  XMLUtils::addChild(doc, node, "BaseRate", baseRateStr);
138  XMLUtils::addChild(doc, node, "Tolerance", tolerance_);
139 
140  if (seasonalityBaseDate_ != QuantLib::Null<Date>()) {
141  XMLNode* seasonalityNode = XMLUtils::addChild(doc, node, "Seasonality");
142  std::ostringstream dateStr, sFreq;
143  dateStr << QuantLib::io::iso_date(seasonalityBaseDate_);
144  sFreq << seasonalityFrequency_;
145  XMLUtils::addChild(doc, seasonalityNode, "BaseDate", dateStr.str());
146  XMLUtils::addChild(doc, seasonalityNode, "Frequency", sFreq.str());
147  if (!seasonalityFactors_.empty())
148  XMLUtils::addChildren(doc, seasonalityNode, "Factors", "Factor", seasonalityFactors_);
149  if (!overrideSeasonalityFactors_.empty())
150  XMLUtils::addChild(doc, seasonalityNode, "OverrideFactors", overrideSeasonalityFactors_);
151  }
152 
153  return node;
154 }
