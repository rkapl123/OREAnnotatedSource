36 Leg DurationAdjustedCmsLegBuilder::buildLeg(const LegData& data, const boost::shared_ptr<EngineFactory>& engineFactory,
37  RequiredFixings& requiredFixings, const string& configuration,
38  const QuantLib::Date& openEndDateReplacement) const {
39 
40  auto cmsData = boost::dynamic_pointer_cast<DurationAdjustedCmsLegData>(data.concreteLegData());
41  QL_REQUIRE(cmsData, "Wrong LegType, expected CMS");
42 
43  string indexName = cmsData->swapIndex();
44  auto index = *engineFactory->market()->swapIndex(indexName, configuration);
45  Schedule schedule = makeSchedule(data.schedule(), openEndDateReplacement);
46  DayCounter dc = parseDayCounter(data.dayCounter());
47  BusinessDayConvention bdc = parseBusinessDayConvention(data.paymentConvention());
48 
49  vector<double> spreads =
50  ore::data::buildScheduledVectorNormalised(cmsData->spreads(), cmsData->spreadDates(), schedule, 0.0);
51  vector<double> gearings =
52  ore::data::buildScheduledVectorNormalised(cmsData->gearings(), cmsData->gearingDates(), schedule, 1.0);
53  vector<double> notionals = buildScheduledVector(data.notionals(), data.notionalDates(), schedule);
54  Size fixingDays = cmsData->fixingDays() == Null<Size>() ? index->fixingDays() : cmsData->fixingDays();
55 
56  applyAmortization(notionals, data, schedule, false);
57 
58  DurationAdjustedCmsLeg leg = DurationAdjustedCmsLeg(schedule, index, cmsData->duration())
59  .withNotionals(notionals)
60  .withSpreads(spreads)
61  .withGearings(gearings)
62  .withPaymentDayCounter(dc)
63  .withPaymentAdjustment(bdc)
64  .withPaymentLag(data.paymentLag())
65  .withFixingDays(fixingDays)
66  .inArrears(cmsData->isInArrears());
67 
68  if (!data.paymentCalendar().empty())
69  leg.withPaymentCalendar(parseCalendar(data.paymentCalendar()));
70 
71  if (cmsData->caps().size() > 0)
72  leg.withCaps(buildScheduledVector(cmsData->caps(), cmsData->capDates(), schedule));
73 
74  if (cmsData->floors().size() > 0)
75  leg.withFloors(buildScheduledVector(cmsData->floors(), cmsData->floorDates(), schedule));
76 
77  // Get a coupon pricer for the leg
78  auto builder = boost::dynamic_pointer_cast<DurationAdjustedCmsCouponPricerBuilder>(
79  engineFactory->builder("DurationAdjustedCMS"));
80  QL_REQUIRE(builder != nullptr, "No builder found for DurationAdjustedCmsLeg");
81  auto couponPricer = builder->engine(index->currency());
82 
83  // Loop over the coupons in the leg and set pricer
84  Leg result = leg;
85  for (auto& c : result) {
86  auto f = boost::dynamic_pointer_cast<FloatingRateCoupon>(c);
87  QL_REQUIRE(f != nullptr,
88  "DurationAdjustedCmsLegBuilder::buildLeg(): internal error, expected FloatingRateCoupon");
89  f->setPricer(couponPricer);
90  }
91 
92  // build naked option leg if required
93  if (cmsData->nakedOption()) {
94  result = StrippedCappedFlooredCouponLeg(result);
95  }
96 
97  std::map<std::string, std::string> qlToOREIndexNames;
98  qlToOREIndexNames[index->name()] = indexName;
99  applyIndexing(result, data, engineFactory, qlToOREIndexNames, requiredFixings);
100  addToRequiredFixings(result, boost::make_shared<FixingDateGetter>(requiredFixings, qlToOREIndexNames));
101  return result;
102 }
