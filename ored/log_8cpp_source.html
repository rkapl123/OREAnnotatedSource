117void ProgressLogger::setCoutLog(const bool flag) {
118 auto core = logging::core::get();
119 if (flag && !coutSink_) {
120 // Create backend and initialize it with a stream
121 boost::shared_ptr<lsinks::text_ostream_backend> backend = boost::make_shared<lsinks::text_ostream_backend>();
122 backend->add_stream(boost::shared_ptr<std::ostream>(&std::clog, boost::null_deleter()));
123
124 // Wrap it into the frontend and register in the core

128 // Store the sink for removal/retrieval later

178void Log::removeLogger(const string& name) {
179 boost::unique_lock<boost::shared_mutex> lock(mutex_);
180 map<string, boost::shared_ptr<Logger>>::iterator it = loggers_.find(name);
181 map<string, boost::shared_ptr<IndependentLogger>>::iterator iit = independentLoggers_.find(name);
182 if (it != loggers_.end()) {
183 loggers_.erase(it);
184 } else if (iit != independentLoggers_.end()) {
185 iit->second->removeSinks();
186 independentLoggers_.erase(iit);
187 } else {
188 QL_FAIL("No logger found with name " << name);
189 }
190}

199string Log::source(const char* filename, int lineNo) {
200 string filepath;
201 if (rootPath_.empty()) {
202 filepath = filename;
203 } else {
204 filepath = relative(path(filename), rootPath_).string();
205 }
206 int lineNoLen = (int)log10((double)lineNo) + 1; // Length of the int as a string
207 int len = 2 + filepath.length() + 1 + lineNoLen + 1; // " (" + file + ':' + line + ')'
208
209 if (maxLen_ == 0) {
210 return "(" + filepath + ':' + to_string(lineNo) + ')';
211 } else {
212 if (len <= maxLen_) {
213 // pad out spaces
214 return string(maxLen_ - len, ' ') + "(" + filepath + ':' + to_string(lineNo) + ')';
215 } else {
216 // need to trim the filename to fit into maxLen chars
217 // need to remove (len - maxLen_) chars + 3 for the "..."
218 return "(..." + filepath.substr(3 + len - maxLen_) + ':' + to_string(lineNo) + ')';
219 }
220 }

236}

238void Log::header(unsigned m, const char* filename, int lineNo) {
239 // 1. Reset stringstream
240 ls_.str(string());
241 ls_.clear();
242
243 // Write the header to the stream
244 // TYPE [Time Stamp] (file:line)
245 switch (m) {
246 case ORE_ALERT:
247 ls_ << "ALERT ";
248 break;
249 case ORE_CRITICAL:
250 ls_ << "CRITICAL ";
251 break;
252 case ORE_ERROR:
253 ls_ << "ERROR ";
254 break;
255 case ORE_WARNING:
256 ls_ << "WARNING ";
257 break;
258 case ORE_NOTICE:
259 ls_ << "NOTICE ";
260 break;
261 case ORE_DEBUG:
262 ls_ << "DEBUG ";
263 break;
264 case ORE_DATA:
265 ls_ << "DATA ";
266 break;
267 case ORE_MEMORY:
268 ls_ << "MEMORY ";
269 break;
270 }
271
272 // Timestamp
273 // Use boost::posix_time microsecond clock to get better precision (when available).
274 // format is "2014-Apr-04 11:10:16.179347"
275 ls_ << '[' << to_simple_string(microsec_clock::local_time()) << ']';
276
277 // Filename & line no
278 // format is " (file:line)"
279 ls_ << " " << source(filename, lineNo) << " : ";
280
281 // log pid if given
282 if (pid_ > 0)
283 ls_ << " [" << pid_ << "] ";
284
285 // update statistics
286 if (lastLineNo_ == lineNo && lastFileName_ == filename) {
287 ++sameSourceLocationSince_;
288 } else {
289 lastFileName_ = filename;
290 lastLineNo_ = lineNo;
291 sameSourceLocationSince_ = 0;
292 writeSuppressedMessagesHint_ = true;
293 }
294}

296void Log::log(unsigned m) {
297 string msg = ls_.str();
298 map<string, boost::shared_ptr<Logger>>::iterator it;
299 if (sameSourceLocationSince_ <= sameSourceLocationCutoff_) {
300 for (auto& l : loggers_) {
301 l.second->log(m, msg);
302 }
303 } else if (writeSuppressedMessagesHint_) {
304 std::string suffix;
305 if (msg.find(StructuredMessage::name) == string::npos) {
306 suffix = " ... suppressing more messages from same source code location (cutoff = " +
307 std::to_string(sameSourceLocationCutoff_) + " lines)";
308 }
309 for (auto& l : loggers_) {
310 l.second->log(m, msg + suffix);
311 }
312 writeSuppressedMessagesHint_ = false;
313 }
314}

325LoggerStream::~LoggerStream() {
326 string text;
327 while (getline(ss_, text)) {
328 // we expand the MLOG macro here so we can overwrite __FILE__ and __LINE__
329 if (ore::data::Log::instance().enabled() && ore::data::Log::instance().filter(mask_)) {
330 boost::unique_lock<boost::shared_mutex> lock(ore::data::Log::instance().mutex());
331 ore::data::Log::instance().header(mask_, filename_, lineNo_);
332 ore::data::Log::instance().logStream() << text;
333 ore::data::Log::instance().log(mask_);
334 }

338string JSONMessage::jsonify(const boost::any& obj) {
339 if (obj.type() == typeid(map<string, boost::any>)) {
340 string jsonStr = "{ ";
341 Size i = 0;
342 for (const auto& kv : boost::any_cast<map<string, boost::any>>(obj)) {
343 if (i > 0)
344 jsonStr += ", ";
345 jsonStr += '\"' + kv.first + "\": " + jsonify(kv.second);
346 i++;
347 }
348 jsonStr += " }";
349 return jsonStr;
350 } else if (obj.type() == typeid(vector<boost::any>)) {
351 string arrayStr = "[ ";
352 Size i = 0;
353 for (const auto& v : boost::any_cast<vector<boost::any>>(obj)) {
354 if (i > 0)
355 arrayStr += ", ";
356 arrayStr += jsonify(v);
357 i++;
358 }
359 arrayStr += " ]";
360 return arrayStr;
361 } else if (obj.type() == typeid(string)) {
362 string str = boost::any_cast<string>(obj);
363 boost::replace_all(str, "\\", "\\\\"); // do this before the below otherwise we get \\"
364 boost::replace_all(str, "\"", "\\\"");
365 boost::replace_all(str, "\r", "\\r");
366 boost::replace_all(str, "\n", "\\n");
367 return '\"' + str + '\"';
368 } else if (obj.type() == typeid(StructuredMessage::Category)) {
369 return to_string(boost::any_cast<StructuredMessage::Category>(obj));
370 } else if (obj.type() == typeid(StructuredMessage::Group)) {
371 return to_string(boost::any_cast<StructuredMessage::Group>(obj));
372 } else if (obj.type() == typeid(int)) {
373 return to_string(boost::any_cast<int>(obj));
374 } else if (obj.type() == typeid(bool)) {
375 return to_string(boost::any_cast<bool>(obj));
376 } else if (obj.type() == typeid(QuantLib::Size)) {
377 return to_string(boost::any_cast<QuantLib::Size>(obj));
378 } else if (obj.type() == typeid(QuantLib::Real)) {
379 return to_string(boost::any_cast<QuantLib::Real>(obj));
380 } else if (obj.type() == typeid(unsigned int)) {
381 return to_string(boost::any_cast<unsigned int>(obj));
382 } else if (obj.type() == typeid(unsigned short)) {
383 return to_string(boost::any_cast<unsigned short>(obj));
384 } else if (obj.type() == typeid(float)) {
385 return to_string(boost::any_cast<float>(obj));
386 } else {
387 WLOG(StructuredLoggingErrorMessage("JSON Message Logging", "JSONMessage::jsonify() : Unrecognised value type"));
388 return string();
389 }

392StructuredMessage::StructuredMessage(const Category& category, const Group& group, const string& message,
393 const map<string, string>& subFields) {
394 data_["category"] = to_string(category);
395 data_["group"] = to_string(group);
396 data_["message"] = message;
397
398 if (!subFields.empty()) {
399 vector<boost::any> subFieldsVector;
400 bool addedSubField = false;
401 for (const auto& sf : subFields) {
402 if (!sf.second.empty()) {
403 map<string, boost::any> subField({{"name", sf.first}, {"value", sf.second}});
404 subFieldsVector.push_back(subField);
405 addedSubField = true;
406 }
407 }
408 if (addedSubField)
409 data_["sub_fields"] = subFieldsVector;
410 }
411}

419void StructuredMessage::addSubFields(const map<string, string>& subFields) {
420 if (!subFields.empty()) {
421
422 // First check that there is at least one non-empty subfield
423 bool hasNonEmptySubField = false;
424 for (const auto& sf : subFields) {
425 if (!sf.second.empty()) {
426 hasNonEmptySubField = true;
427 break;
428 }
429 }
430 if (!hasNonEmptySubField)
431 return;
432
433 if (data_.find("sub_fields") == data_.end()) {
434 data_["sub_fields"] = vector<boost::any>();
435 }
436
437 for (const auto& sf : subFields) {
438 if (!sf.second.empty()) {
439 map<string, boost::any> subField({{"name", sf.first}, {"value", sf.second}});
440 boost::any_cast<vector<boost::any>&>(data_.at("sub_fields")).push_back(subField);
441 }
442 }
443
444 }
