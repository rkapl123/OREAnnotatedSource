109void Log::header(unsigned m, const char* filename, int lineNo) {
110 // 1. Reset stringstream
111 ls_.str(string());
112 ls_.clear();
113
114 // Write the header to the stream
115 // TYPE [Time Stamp] (file:line)
116 switch (m) {
117 case ORE_ALERT:
118 ls_ << "ALERT ";
119 break;
120 case ORE_CRITICAL:
121 ls_ << "CRITICAL ";
122 break;
123 case ORE_ERROR:
124 ls_ << "ERROR ";
125 break;
126 case ORE_WARNING:
127 ls_ << "WARNING ";
128 break;
129 case ORE_NOTICE:
130 ls_ << "NOTICE ";
131 break;
132 case ORE_DEBUG:
133 ls_ << "DEBUG ";
134 break;
135 case ORE_DATA:
136 ls_ << "DATA ";
137 break;
138 case ORE_MEMORY:
139 ls_ << "MEMORY ";
140 break;
141 }
142
143 // Timestamp
144 // Use boost::posix_time microsecond clock to get better precision (when available).
145 // format is "2014-Apr-04 11:10:16.179347"
146 ls_ << '[' << to_simple_string(microsec_clock::local_time()) << ']';
147
148 // Filename & line no
149 // format is " (file:line)"
150 string filepath;
151 if (rootPath_.empty()) {
152 filepath = filename;
153 } else {
154 filepath = relative(path(filename), rootPath_).string();
155 }
156 int lineNoLen = (int)log10((double)lineNo) + 1; // Length of the int as a string
157 int len = 2 + filepath.length() + 1 + lineNoLen + 1; // " (" + file + ':' + line + ')'
158
159 if (maxLen_ == 0) {
160 ls_ << " (" << filepath << ':' << lineNo << ')';
161 } else {
162 if (len <= maxLen_) {
163 // pad out spaces
164 ls_ << string(maxLen_ - len, ' ') << " (" << filepath << ':' << lineNo << ')';
165 } else {
166 // need to trim the filename to fit into maxLen chars
167 // need to remove (len - maxLen_) chars + 3 for the "..."
168 ls_ << " (..." << filepath.substr(3 + len - maxLen_) << ':' << lineNo << ')';
169 }
170 }
171
172 ls_ << " : ";
173
174 // log pid if given
175 if (pid_ > 0)
176 ls_ << " [" << pid_ << "] ";
177
178 // update statistics
179 if (lastLineNo_ == lineNo && lastFileName_ == filepath) {
180 ++sameSourceLocationSince_;
181 } else {
182 lastFileName_ = filepath;
183 lastLineNo_ = lineNo;
184 sameSourceLocationSince_ = 0;
185 writeSuppressedMessagesHint_ = true;
186 }
187
188}

190void Log::log(unsigned m) {
191 string msg = ls_.str();
192 map<string, boost::shared_ptr<Logger>>::iterator it;
193 if (sameSourceLocationSince_ <= sameSourceLocationCutoff_) {
194 for (auto& l : loggers_) {
195 l.second->log(m, msg);
196 }
197 } else if (writeSuppressedMessagesHint_) {
198 std::string suffix;
199 if (msg.find(StructuredMessage::name) == string::npos) {
200 suffix = " ... suppressing more messages from same source code location (cutoff = " +
201 std::to_string(sameSourceLocationCutoff_) + " lines)";
202 }
203 for (auto& l : loggers_) {
204 l.second->log(m, msg + suffix);
205 }
206 writeSuppressedMessagesHint_ = false;
207 }
208}

212LoggerStream::LoggerStream(unsigned mask, const char* filename, unsigned lineNo)
213 : mask_(mask), filename_(filename), lineNo_(lineNo), ss_() {
214 QL_REQUIRE(mask == ORE_ALERT || mask == ORE_CRITICAL || mask == ORE_ERROR || mask == ORE_WARNING ||

216 "Invalid log mask " << mask);
217}
218
219LoggerStream::~LoggerStream() {
220 string text;
221 while (getline(ss_, text)) {
222 // we expand the MLOG macro here so we can overwrite __FILE__ and __LINE__
223 if (ore::data::Log::instance().enabled() && ore::data::Log::instance().filter(mask_)) {
224 boost::unique_lock<boost::shared_mutex> lock(ore::data::Log::instance().mutex());
225 ore::data::Log::instance().header(mask_, filename_, lineNo_);
226 ore::data::Log::instance().logStream() << text;
227 ore::data::Log::instance().log(mask_);
228 }
229 }

232string StructuredMessage::json() const {
233 string msg = "{ \"category\":\"" + ore::data::to_string(category_) + "\", \"group\":\"" + ore::data::to_string(group_) + "\"," + " \"message\":\"" +
234 jsonify(message_) + "\"";
235
236 if (!subFields_.empty()) {
237 msg += ", \"sub_fields\": [ ";
238 QuantLib::Size i = 0;
239 for (const auto& p : subFields_) {
240 // Only include subFields that are non-empty.
241 if (!p.second.empty()) {
242 if (i > 0)
243 msg += ", ";
244 msg += "{ \"name\": \"" + p.first + "\", \"value\": \"" + jsonify(p.second) + "\" }";
245 i++;
246 }
247 }
248 msg += " ]";
249 }
250 msg += " }";
251
252 return msg;
253}

264string EventMessage::json() const {
265 string msg = "{ \"exception_message\":\"" + jsonify(message_) + "\"";
266
267 if (!data_.empty()) {
268 msg += ", ";
269 QuantLib::Size i = 0;
270 for (const auto& p : data_) {
271 string value;
272 if (p.second.type() == typeid(string)) {
273 value = "\"" + boost::any_cast<string>(p.second) + "\"";
274 } else if (p.second.type() == typeid(boost::posix_time::ptime)) {
275 auto time = boost::any_cast<boost::posix_time::ptime>(p.second);
276 value = boost::posix_time::to_iso_extended_string(time);
277 } else if (p.second.type() == typeid(int)) {
278 value = to_string(boost::any_cast<int>(p.second));
279 } else if (p.second.type() == typeid(unsigned int)) {
280 value = to_string(boost::any_cast<unsigned int>(p.second));
281 } else if (p.second.type() == typeid(unsigned short)) {
282 value = to_string(boost::any_cast<unsigned short>(p.second));
283 } else if (p.second.type() == typeid(float)) {
284 value = to_string(boost::any_cast<float>(p.second));
285 } else if (p.second.type() == typeid(QuantLib::Size)) {
286 value = to_string(boost::any_cast<QuantLib::Size>(p.second));
287 } else if (p.second.type() == typeid(QuantLib::Real)) {
288 value = to_string(boost::any_cast<QuantLib::Real>(p.second));
289 } else if (p.second.type() == typeid(bool)) {
290 value = to_string(boost::any_cast<bool>(p.second));
291 } else {
292 WLOG(StructuredLoggingErrorMessage("Event Message Logging",
293 "Unrecognised value type for key '" + p.first + "'"));
294 }
295
296 if (i > 0)
297 msg += ", ";
298 msg += "\"" + p.first + "\": " + value;
299 i++;
300 }
301 }
302 msg += " }";
303
304 return msg;
305}
