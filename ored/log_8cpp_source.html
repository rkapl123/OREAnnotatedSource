106void Log::header(unsigned m, const char* filename, int lineNo) {
107 // 1. Reset stringstream
108 ls_.str(string());
109 ls_.clear();
110
111 // Write the header to the stream
112 // TYPE [Time Stamp] (file:line)
113 switch (m) {
114 case ORE_ALERT:
115 ls_ << "ALERT ";
116 break;
117 case ORE_CRITICAL:
118 ls_ << "CRITICAL ";
119 break;
120 case ORE_ERROR:
121 ls_ << "ERROR ";
122 break;
123 case ORE_WARNING:
124 ls_ << "WARNING ";
125 break;
126 case ORE_NOTICE:
127 ls_ << "NOTICE ";
128 break;
129 case ORE_DEBUG:
130 ls_ << "DEBUG ";
131 break;
132 case ORE_DATA:
133 ls_ << "DATA ";
134 break;
135 case ORE_MEMORY:
136 ls_ << "MEMORY ";
137 break;
138 }
139
140 // Timestamp
141 // Use boost::posix_time microsecond clock to get better precision (when available).
142 // format is "2014-Apr-04 11:10:16.179347"
143 ls_ << '[' << to_simple_string(microsec_clock::local_time()) << ']';
144
145 // Filename & line no
146 // format is " (file:line)"
147 int lineNoLen = (int)log10((double)lineNo) + 1; // Length of the int as a string
148 int len = 2 + strlen(filename) + 1 + lineNoLen + 1; // " (" + file + ':' + line + ')'
149
150 int maxLen = 30; // gives about 23 chars for the filename
151 if (len <= maxLen) {
152 ls_ << " (" << filename << ':' << lineNo << ')';
153 // pad out spaces
154 ls_ << string(maxLen - len, ' ');
155 } else {
156 // need to trim the filename to fit into maxLen chars
157 // need to remove (len - maxLen) chars + 3 for the "..."
158 ls_ << " (..." << string(filename).substr(3 + len - maxLen) << ':' << lineNo << ')';
159 }
160
161 ls_ << " : ";
162
163 // log pid if given
164 if (pid_ > 0)
165 ls_ << " [" << pid_ << "] ";
166
167 // update statistics
168 if(lastLineNo_ == lineNo && lastFileName_ == filename) {
169 ++sameSourceLocationSince_;
170 } else {
171 lastFileName_ = filename;
172 lastLineNo_ = lineNo;
173 sameSourceLocationSince_ = 0;
174 writeSuppressedMessagesHint_ = true;
175 }
176
177}

179void Log::log(unsigned m) {
180 string msg = ls_.str();
181 map<string, boost::shared_ptr<Logger>>::iterator it;
182 if (sameSourceLocationSince_ <= sameSourceLocationCutoff_) {
183 for (auto& l : loggers_) {
184 l.second->log(m, msg);
185 }
186 } else if (writeSuppressedMessagesHint_) {
187 std::string suffix;
188 if (msg.find(StructuredMessage::name) == string::npos) {
189 suffix = " ... suppressing more messages from same source code location (cutoff = " +
190 std::to_string(sameSourceLocationCutoff_) + " lines)";
191 }
192 for (auto& l : loggers_) {
193 l.second->log(m, msg + suffix);
194 }
195 writeSuppressedMessagesHint_ = false;
196 }
197}

201LoggerStream::LoggerStream(unsigned mask, const char* filename, unsigned lineNo)

208LoggerStream::~LoggerStream() {
209 string text;
210 while (getline(ss_, text)) {
211 // we expand the MLOG macro here so we can overwrite __FILE__ and __LINE__
212 if (ore::data::Log::instance().enabled() && ore::data::Log::instance().filter(mask_)) {
213 boost::unique_lock<boost::shared_mutex> lock(ore::data::Log::instance().mutex());
214 ore::data::Log::instance().header(mask_, filename_, lineNo_);
215 ore::data::Log::instance().logStream() << text;
216 ore::data::Log::instance().log(mask_);
217 }
218 }

221string StructuredMessage::json() const {
222 string msg =
223 "{ \"category\":\"" + category_ + "\", \"group\":\"" + group_ + "\"," + " \"message\":\"" + jsonify(message_);
224
225 if (!subFields_.empty()) {
226 msg += "\", \"sub_fields\": [ ";
227 QuantLib::Size i = 0;
228 for (const auto& p : subFields_) {
229 // Only include subFields that are non-empty.
230 if (!p.second.empty()) {
231 if (i > 0)
232 msg += ", ";
233 msg += "{ \"name\": \"" + p.first + "\", \"value\": \"" + jsonify(p.second) + "\" }";
234 i++;
235 }
236 }
237 msg += " ]";
238 }
239 msg += " }";
240
241 return msg;
242}

244string StructuredMessage::jsonify(const string& s) const {
245 string str = s;
246 boost::replace_all(str, "\\", "\\\\"); // do this before the below otherwise we get \\"
247 boost::replace_all(str, "\"", "\\\"");
248 boost::replace_all(str, "\r", "\\r");
249 boost::replace_all(str, "\n", "\\n");
250 return str;
251}
