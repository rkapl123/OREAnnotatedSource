31boost::shared_ptr<FloatingRateCouponPricer>
32CapFlooredAverageONIndexedCouponLegEngineBuilder::engineImpl(const std::string& index,
33 const QuantLib::Period& rateComputationPeriod) {
34 std::string ccyCode = parseIborIndex(index)->currency().code();
35 Handle<YieldTermStructure> yts = market_->discountCurve(ccyCode, configuration(MarketContext::pricing));
36 Handle<OptionletVolatilityStructure> ovs = market_->capFloorVol(index, configuration(MarketContext::pricing));
37
38 /* if we are pricing an overnight index coupon with rate computation period different from the one on which the
39 market vol surface is based, we apply a moneyness adjustment accounting for this difference */
40 auto [volIndex, volRateComputationPeriod] =
41 market_->capFloorVolIndexBase(index, configuration(MarketContext::pricing));
42 if (volIndex == index && volRateComputationPeriod != rateComputationPeriod &&
43 volRateComputationPeriod != 0 * Days && rateComputationPeriod != 0 * Days) {
44 ovs = Handle<OptionletVolatilityStructure>(boost::make_shared<QuantExt::ProxyOptionletVolatility>(
45 ovs, *market_->iborIndex(volIndex, configuration(MarketContext::pricing)),
46 *market_->iborIndex(index, configuration(MarketContext::pricing)), volRateComputationPeriod,
47 rateComputationPeriod));
48 }
49
50 return boost::make_shared<QuantExt::BlackAverageONIndexedCouponPricer>(ovs);
51}
