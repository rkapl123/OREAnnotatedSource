61 XMLNode* YoYCapFloor::toXML(XMLDocument& doc) {
62  XMLNode* node = doc.allocNode(instrumentType_);
63  if (type_ == YoYInflationCapFloor::Cap)
64  XMLUtils::addChild(doc, node, "Type", "Cap");
65  else if (type_ == YoYInflationCapFloor::Floor)
66  XMLUtils::addChild(doc, node, "Type", "Floor");
67  else
68  QL_FAIL("Failure in YoYCapFloor::toXML, unsupported YoY cap floor type.");
69  XMLUtils::addChild(doc, node, "Tenor", to_string(tenor_));
