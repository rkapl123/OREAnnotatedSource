66// Generate lookback dates
67set<Date> generateLookbackDates(const Date& asof, const Period& lookbackPeriod, const Calendar& calendar) {
68
69 set<Date> dates;
70 Date lookback = calendar.advance(asof, -lookbackPeriod);
71 do {
72 TLOG("Adding date " << io::iso_date(lookback) << " to fixings.");
73 dates.insert(lookback);
74 lookback = calendar.advance(lookback, 1 * Days);
75 } while (lookback <= asof);
76
77 return dates;
78}

87// Return the set of dates on which a fixing will be required, if any.
88set<Date> needsForecast(const Date& fixingDate, const Date& today, const bool interpolated, const Frequency frequency,
89 const Period& availabilityLag) {
90
91 set<Date> result;
92
93 Date todayMinusLag = today - availabilityLag;
94 Date historicalFixingKnown = inflationPeriod(todayMinusLag, frequency).first - 1;
95
96 pair<Date, Date> lim = inflationPeriod(fixingDate, frequency);
97 result.insert(lim.first);
98 Date latestNeededDate = fixingDate;
99 if (interpolated) {
100 if (fixingDate > lim.first) {
101 latestNeededDate += Period(frequency);
102 result.insert(lim.second + 1);
103 }
104 }
105
106 if (latestNeededDate <= historicalFixingKnown) {
107 // Know that fixings are available
108 return result;
109 } else if (latestNeededDate > today) {
110 // Know that fixings are not available
111 return {};
112 } else {
113 // Grey area here but for now return nothing
114 return {};
115 }
116}

118// Common code for zero inflation index based coupons
119void addZeroInflationDates(set<Date>& dates, const Date& fixingDate, const Date& today, const bool indexInterpolated,
120 const Frequency indexFrequency, const Period& indexAvailabilityLag,
121 const CPI::InterpolationType interpolation, const Frequency f) {
122
123 set<Date> fixingDates;
124
125 if (interpolation == CPI::AsIndex) {
126 fixingDates = needsForecast(fixingDate, today, indexInterpolated, indexFrequency, indexAvailabilityLag);
127 } else {
128 pair<Date, Date> lim = inflationPeriod(fixingDate, f);
129 fixingDates = needsForecast(lim.first, today, indexInterpolated, indexFrequency, indexAvailabilityLag);
130 if (interpolation == CPI::Linear) {
131 auto moreDates =
132 needsForecast(lim.second + 1, today, indexInterpolated, indexFrequency, indexAvailabilityLag);
133 fixingDates.insert(moreDates.begin(), moreDates.end());
134 }
135 }
136
137 dates.insert(fixingDates.begin(), fixingDates.end());
138}

155void RequiredFixings::unsetPayDates() {
156 // we can't modify the elements of a set directly, need to make a copy and reassign
157 std::set<FixingEntry> newFixingDates;
158 std::set<ZeroInflationFixingEntry> newZeroInflationFixingDates;
159 std::set<InflationFixingEntry> newYoYInflationFixingDates;
160 for (auto f : fixingDates_) {
161 std::get<2>(f) = Date::maxDate();
162 std::get<3>(f) = true;
163 newFixingDates.insert(f);
164 }
165 for (auto f : zeroInflationFixingDates_) {
166 std::get<2>(std::get<0>(std::get<0>(f))) = Date::maxDate();
167 std::get<3>(std::get<0>(std::get<0>(f))) = true;
168 newZeroInflationFixingDates.insert(f);
169 }
170 for (auto f : yoyInflationFixingDates_) {
171 std::get<2>(std::get<0>(f)) = Date::maxDate();
172 std::get<3>(std::get<0>(f)) = true;
173 newYoYInflationFixingDates.insert(f);
174 }
175 fixingDates_ = newFixingDates;
176 zeroInflationFixingDates_ = newZeroInflationFixingDates;
177 yoyInflationFixingDates_ = newYoYInflationFixingDates;
178}

180std::map<std::string, std::set<Date>> RequiredFixings::fixingDatesIndices(const Date& settlementDate) const {
181
182 // If settlement date is an empty date, update to evaluation date.
183 Date d = settlementDate == Date() ? Settings::instance().evaluationDate() : settlementDate;
184
185 std::map<std::string, std::set<Date>> result;
186
187 // handle the general case
188 for (auto const& f : fixingDates_) {
189 // get the data
190 std::string indexName = std::get<0>(f);
191 Date fixingDate = std::get<1>(f);
192 Date payDate = std::get<2>(f);
193 bool alwaysAddIfPaysOnSettlement = std::get<3>(f);
194 // add to result
195 if (fixingDate > d)
196 continue;
197 SimpleCashFlow dummyCf(0.0, payDate);
198 if (!dummyCf.hasOccurred(d) || (alwaysAddIfPaysOnSettlement && dummyCf.date() == d)) {
199 result[indexName].insert(fixingDate);
200 }
201 }
202
203 // handle zero inflation index based coupons
204 for (auto const& f : zeroInflationFixingDates_) {
205 // get the data
206 InflationFixingEntry inflationFixingEntry = std::get<0>(f);
207 FixingEntry fixingEntry = std::get<0>(inflationFixingEntry);
208 std::string indexName = std::get<0>(fixingEntry);
209 Date fixingDate = std::get<1>(fixingEntry);
210 Date payDate = std::get<2>(fixingEntry);
211 bool alwaysAddIfPaysOnSettlement = std::get<3>(fixingEntry);
212 bool indexInterpolated = std::get<1>(inflationFixingEntry);
213 Frequency indexFrequency = std::get<2>(inflationFixingEntry);
214 Period indexAvailabilityLag = std::get<3>(inflationFixingEntry);
215 CPI::InterpolationType couponInterpolation = std::get<1>(f);
216 Frequency couponFrequency = std::get<2>(f);
217 // add to result
218 SimpleCashFlow dummyCf(0.0, payDate);
219 if (!dummyCf.hasOccurred(d) || (alwaysAddIfPaysOnSettlement && dummyCf.date() == d)) {
220 std::set<Date> tmp;
221 addZeroInflationDates(tmp, fixingDate, d, indexInterpolated, indexFrequency, indexAvailabilityLag,
222 couponInterpolation, couponFrequency);
223 if (!tmp.empty())
224 result[indexName].insert(tmp.begin(), tmp.end());
225 }
226 }
227
228 // handle yoy inflation index based coupons
229 for (auto const& f : yoyInflationFixingDates_) {
230 // get the data
231 FixingEntry fixingEntry = std::get<0>(f);
232 std::string indexName = std::get<0>(fixingEntry);
233 Date fixingDate = std::get<1>(fixingEntry);
234 Date payDate = std::get<2>(fixingEntry);
235 bool alwaysAddIfPaysOnSettlement = std::get<3>(fixingEntry);
236 bool indexInterpolated = std::get<1>(f);
237 Frequency indexFrequency = std::get<2>(f);
238 Period indexAvailabilityLag = std::get<3>(f);
239 // add to result
240 SimpleCashFlow dummyCf(0.0, payDate);
241 if (!dummyCf.hasOccurred(d) || (alwaysAddIfPaysOnSettlement && dummyCf.date() == d)) {
242 auto fixingDates = needsForecast(fixingDate, d, indexInterpolated, indexFrequency, indexAvailabilityLag);
243 if (!fixingDates.empty())
244 result[indexName].insert(fixingDates.begin(), fixingDates.end());
245 // Add the previous year's date(s) also if any.
246 for (const auto& d : fixingDates) {
247 result[indexName].insert(d - 1 * Years);
248 }
249 }
250 }
251
252 return result;
253}

331void FixingDateGetter::visit(IborCoupon& c) {
332 if (auto bma = boost::dynamic_pointer_cast<QuantExt::BMAIndexWrapper>(c.index())) {
333 // Handle bma indices which we allow in IborCoupon as an approximation to BMA
334 // coupons. For these we allow fixing dates that are invalid as BMA fixing dates
335 // and adjust these dates to the last valid BMA fixing date in the BMAIndexWrapper.
336 // It is this adjusted date that we want to record here.
337 // Enforce fixing to be added even if coupon pays on settlement.
338 requiredFixings_.addFixingDate(bma->adjustedFixingDate(c.fixingDate()),
339 IndexNameTranslator::instance().oreName(c.index()->name()), c.date(), true);
340 } else {
341 auto fallback = boost::dynamic_pointer_cast<FallbackIborIndex>(c.index());
342 if (fallback != nullptr && c.fixingDate() >= fallback->switchDate()) {
343 requiredFixings_.addFixingDates(fallback->onCoupon(c.fixingDate())->fixingDates(),
344 IndexNameTranslator::instance().oreName(fallback->rfrIndex()->name()),
345 c.date());
346 } else {
347 visit(static_cast<FloatingRateCoupon&>(c));
348 }
349 }
350}

523void FixingDateGetter::visit(CommodityCashFlow& c) {
524 auto indices = c.indices();
525 for (const auto& kv : indices) {
526 // see above, the ql and ORE index names are identical
527 requiredFixings_.addFixingDate(kv.first, kv.second->name(), c.date());
528 // if the pricing date is > future expiry, add the future expiry itself as well
529 if (auto d = kv.second->expiryDate(); d != Date() && d < kv.first) {
530 requiredFixings_.addFixingDate(d, kv.second->name(), d);
531 }
532 if (auto baseFutureIndex = boost::dynamic_pointer_cast<CommodityBasisFutureIndex>(kv.second)) {
533 baseFutureIndex->baseCashflow(c.date())->accept(*this);

545void amendInflationFixingDates(map<string, set<Date>>& fixings) {
546 // Loop over indices and amend any that are of type InflationIndex
547 for (auto& kv : fixings) {
548 auto p = isInflationIndex(kv.first);
549 if (p.first) {
550 // We have an inflation index
551 set<Date> newDates;
552 Frequency f = p.second->frequency();
553 for (const Date& d : kv.second) {
554 auto period = inflationPeriod(d, f);
555 if (d == period.first) {
556 // If the fixing date is the start of the inflation period, move it to the end.
557 newDates.insert(period.second);
558 } else {
559 // If the fixing date is not the start of the inflation period, leave it as it is.
560 newDates.insert(d);
561 }
562 }
563 // Update the fixings map that was passed in with the new set of dates
564 kv.second = newDates;
565 }
566 }

569void addMarketFixingDates(const Date& asof, map<string, set<Date>>& fixings, const TodaysMarketParameters& mktParams,
570 const Period& iborLookback, const Period& oisLookback, const Period& bmaLookback,
571 const Period& inflationLookback, const string& configuration) {
572
573 if (mktParams.hasConfiguration(configuration)) {
574
575 LOG("Start adding market fixing dates for configuration '" << configuration << "'");
576
577 boost::shared_ptr<Conventions> conventions = InstrumentConventions::instance().conventions();
578
579 // If there are ibor indices in the market parameters, add the lookback fixings
580 // IF there are SIFMA / BMA indices, add lookback fixings for the Libor basis index
581 if (mktParams.hasMarketObject(MarketObject::IndexCurve)) {
582
583 QL_REQUIRE(iborLookback >= 0 * Days, "Ibor lookback period must be non-negative");
584
585 DLOG("Start adding market fixing dates for interest rate indices.");
586
587 set<Date> iborDates;
588 set<Date> oisDates;
589 set<Date> bmaDates;
590 WeekendsOnly calendar;
591
592 // For each of the IR indices in market parameters, insert the dates
593 for (const auto& kv : mktParams.mapping(MarketObject::IndexCurve, configuration)) {
594 if (isOvernightIndex(kv.first)) {
595 if (oisDates.empty()) {
596 TLOG("Generating fixing dates for overnight indices.");
597 oisDates = generateLookbackDates(asof, oisLookback, calendar);
598 }
599 TLOG("Adding extra fixing dates for overnight index " << kv.first);
600 fixings[kv.first].insert(oisDates.begin(), oisDates.end());
601 } else if (isBmaIndex(kv.first)) {
602 if (bmaDates.empty()) {
603 TLOG("Generating fixing dates for bma/sifma indices.");
604 bmaDates = generateLookbackDates(asof, bmaLookback, calendar);
605 }
606 fixings[kv.first].insert(bmaDates.begin(), bmaDates.end());
607 if (iborDates.empty()) {
608 TLOG("Generating fixing dates for ibor indices.");
609 iborDates = generateLookbackDates(asof, iborLookback, calendar);
610 }
611 std::set<string> liborNames;
612 for (auto const& c : conventions->get(Convention::Type::BMABasisSwap)) {
613 auto bma = boost::dynamic_pointer_cast<BMABasisSwapConvention>(c);
614 QL_REQUIRE(
615 bma, "internal error, could not cast to BMABasisSwapConvention in addMarketFixingDates()");
616 if (bma->bmaIndexName() == kv.first) {
617 liborNames.insert(bma->liborIndexName());
618 }
619 }
620 for (auto const& l : liborNames) {
621 TLOG("Adding extra fixing dates for libor index " << l << " from bma/sifma index " << kv.first);
622 fixings[l].insert(iborDates.begin(), iborDates.end());
623 }
624 } else {
625 if (iborDates.empty()) {
626 TLOG("Generating fixing dates for ibor indices.");
627 iborDates = generateLookbackDates(asof, iborLookback, calendar);
628 }
629 TLOG("Adding extra fixing dates for ibor index " << kv.first);
630 fixings[kv.first].insert(iborDates.begin(), iborDates.end());
631 }
632 }
633
634 DLOG("Finished adding market fixing dates for interest rate indices.");
635 }
636
637 // If there are inflation indices in the market parameters, add the lookback fixings
638 if (mktParams.hasMarketObject(MarketObject::ZeroInflationCurve) ||
639 mktParams.hasMarketObject(MarketObject::YoYInflationCurve)) {
640
641 QL_REQUIRE(inflationLookback >= 0 * Days, "Inflation lookback period must be non-negative");
642
643 // Dates that will be used for each of the inflation indices
644 Date today = Settings::instance().evaluationDate();
645 Date lookback = NullCalendar().advance(today, -inflationLookback);
646 lookback = Date(1, lookback.month(), lookback.year());
647 set<Date> dates;
648 do {
649 TLOG("Adding date " << io::iso_date(lookback) << " to fixings for inflation indices");
650 dates.insert(lookback);
651 lookback = NullCalendar().advance(lookback, 1 * Months);
652 } while (lookback <= today);
653
654 // For each of the inflation indices in market parameters, insert the dates
655 if (mktParams.hasMarketObject(MarketObject::ZeroInflationCurve)) {
656 for (const auto& kv : mktParams.mapping(MarketObject::ZeroInflationCurve, configuration)) {
657 TLOG("Adding extra fixing dates for (zero) inflation index " << kv.first);
658 fixings[kv.first].insert(dates.begin(), dates.end());
659 }
660 }
661
662 if (mktParams.hasMarketObject(MarketObject::YoYInflationCurve)) {
663 for (const auto& kv : mktParams.mapping(MarketObject::YoYInflationCurve, configuration)) {
664 TLOG("Adding extra fixing dates for (yoy) inflation index " << kv.first);
665 fixings[kv.first].insert(dates.begin(), dates.end());
666 }
667 }
668 }
669
670 // If there are commodity curves, add "fixings" for this month and two previous months. We add "fixings" for
671 // future contracts with expiry from two months hence to two months prior.
672 if (mktParams.hasMarketObject(MarketObject::CommodityCurve)) {
673
674 // "Fixing" dates for commodities.
675 Period commodityLookback = 2 * Months;
676 Date today = Settings::instance().evaluationDate();
677 Date lookback = today - commodityLookback;
678 lookback = Date(1, lookback.month(), lookback.year());
679 set<Date> dates;
680 do {
681 TLOG("Adding date " << io::iso_date(lookback) << " to fixings for commodities");
682 dates.insert(lookback++);
683 } while (lookback <= today);
684
685 // Expiry months and years for which we require future contract fixings. For our purposes here, using the
686 // 1st of the month does not matter. We will just use the date to get the appropriate commodity future
687 // index name below when adding the dates and the "-01" will be removed (for non-daily contracts).
688 Size numberMonths = 2;
689 vector<Date> contractExpiries;
690 Date startContract = today - numberMonths * Months;
691 Date endContract = today + numberMonths * Months;
692 do {
693 Month m = startContract.month();
694 Year y = startContract.year();
695 TLOG("Adding contract month and year (" << m << "," << y << ")");
696 contractExpiries.push_back(Date(1, m, y));
697 startContract += 1 * Months;
698 } while (startContract <= endContract);
699
700 // For each of the commodity names, create the future contract name with the relevant expiry and insert
701 // the dates. Skip commodity names that do not have future conventions.
702 for (const auto& kv : mktParams.mapping(MarketObject::CommodityCurve, configuration)) {
703
704 boost::shared_ptr<CommodityFutureConvention> cfc;
705 if (conventions->has(kv.first)) {
706 cfc = boost::dynamic_pointer_cast<CommodityFutureConvention>(conventions->get(kv.first));
707 }
708
709 auto commIdx = parseCommodityIndex(kv.first, false);
710 if (cfc) {
711 if (auto oppIdx = boost::dynamic_pointer_cast<OffPeakPowerIndex>(commIdx)) {
712 DLOG("Commodity " << kv.first << " is off-peak power so adding underlying daily contracts.");
713 const auto& opIndex = oppIdx->offPeakIndex();
714 const auto& pIndex = oppIdx->peakIndex();
715 for (const Date& expiry : dates) {
716 auto tmpIdx = oppIdx->clone(expiry);
717 auto opName = opIndex->clone(expiry)->name();
718 TLOG("Adding (date, id) = (" << io::iso_date(expiry) << "," << opName << ")");
719 fixings[opName].insert(expiry);
720 auto pName = pIndex->clone(expiry)->name();
721 TLOG("Adding (date, id) = (" << io::iso_date(expiry) << "," << pName << ")");
722 fixings[pName].insert(expiry);
723 }
724 } else if (cfc->contractFrequency() == Daily) {
725 DLOG("Commodity " << kv.first << " has daily frequency so adding daily contracts.");
726 for (const Date& expiry : dates) {
727 auto indexName = commIdx->clone(expiry)->name();
728 TLOG("Adding (date, id) = (" << io::iso_date(expiry) << "," << indexName << ")");
729 fixings[indexName].insert(expiry);
730 }
731 } else {
732 DLOG("Commodity " << kv.first << " is not daily so adding the monthly contracts.");
733 for (const Date& expiry : contractExpiries) {
734 auto indexName = commIdx->clone(expiry)->name();
735 TLOG("Adding extra fixing dates for commodity future " << indexName);
736 fixings[indexName].insert(dates.begin(), dates.end());
737 }
738 }
739 } else {
740 // Assumption here is that we have a spot index.
741 DLOG("Commodity " << kv.first << " does not have future conventions so adding daily fixings.");
742 auto indexName = commIdx->name();
743 TLOG("Adding extra fixing dates for commodity spot " << indexName);
744 fixings[indexName].insert(dates.begin(), dates.end());
745 }
746 }
747 }
748
749 LOG("Finished adding market fixing dates for configuration '" << configuration << "'");
750 }
751}
