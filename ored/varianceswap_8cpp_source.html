36void VarSwap::build(const boost::shared_ptr<ore::data::EngineFactory>& engineFactory) {
37 Currency ccy = ore::data::parseCurrency(currency_);
38 Position::Type longShort = ore::data::parsePositionType(longShort_);
39 start_ = ore::data::parseDate(startDate_);
40 Date endDate = ore::data::parseDate(endDate_);
41 cal_ = ore::data::parseCalendar(calendar_);
42 MomentType momentType = parseMomentType(momentType_);
43
44 if (cal_.empty()) {
45 cal_ = ore::data::parseCalendar(ccy.code());
46 }
47
48 QL_REQUIRE(strike_ > 0 && !close_enough(strike_, 0.0),
49 "VarSwap::build() strike must be positive (" << strike_ << ")");
50 QL_REQUIRE(notional_ > 0 || close_enough(notional_, 0.0),
51 "VarSwap::build() notional must be non-negative (" << notional_ << ")");
52
53 // Input strike is annualised Vol
54 // The quantlib strike and notional are in terms of variance, not volatility, so we convert here.
55 Real varianceStrike = strike_ * strike_;
56 Real varianceNotional = notional_ / (2 * 100 * strike_);
57
58 boost::shared_ptr<VarianceSwap> varSwap(new QuantExt::VarianceSwap(
59 longShort, varianceStrike, momentType == MomentType::Variance ? varianceNotional : notional_, start_, endDate,
60 cal_, addPastDividends_));
61
62 // Pricing Engine
63 boost::shared_ptr<ore::data::EngineBuilder> builder = engineFactory->builder(tradeType_);
64 QL_REQUIRE(builder, "No builder found for " << tradeType_);
65 boost::shared_ptr<VarSwapEngineBuilder> varSwapBuilder = boost::dynamic_pointer_cast<VarSwapEngineBuilder>(builder);
66
67 varSwap->setPricingEngine(varSwapBuilder->engine(name(), ccy, assetClassUnderlying_, momentType));
68
69 // set up other Trade details
70 instrument_ = boost::shared_ptr<ore::data::InstrumentWrapper>(new ore::data::VanillaInstrument(varSwap));
71
72 npvCurrency_ = currency_;
73 notionalCurrency_ = currency_;
74 maturity_ = endDate;
75
76 // add required fixings
77 for (Date d = cal_.advance(start_, -1 * Days); d <= endDate; d = cal_.advance(d, 1 * Days)) {
78 requiredFixings_.addFixingDate(d, indexName_, varSwap->maturityDate());
79 }
80}

82QuantLib::Real VarSwap::notional() const {
83 MomentType momentType = parseMomentType(momentType_);
84 if (momentType == MomentType::Variance) {
85 Real varianceNotional = notional_ / (2 * 100 * strike_);
86 return varianceNotional * 10000;
87 } else
88 return notional_ * 100;
89}

91void VarSwap::fromXML(XMLNode* node) {
92 Trade::fromXML(node);
93 XMLNode* vNode = XMLUtils::getChildNode(node, tradeType() + "Data");
94 if (!vNode) {
95 vNode = XMLUtils::getChildNode(node, "VarianceSwapData");
96 oldXml_ = true;
97 }
98 startDate_ = XMLUtils::getChildValue(vNode, "StartDate", true);
99 endDate_ = XMLUtils::getChildValue(vNode, "EndDate", true);
100 currency_ = XMLUtils::getChildValue(vNode, "Currency", true);
101
102 XMLNode* tmp = XMLUtils::getChildNode(vNode, "Underlying");
103 if (!tmp) {
104 tmp = XMLUtils::getChildNode(vNode, "Name");
105 QL_REQUIRE(tmp, "Must provide a valid Underlying or Name node");
106 }
107 UnderlyingBuilder underlyingBuilder;
108 underlyingBuilder.fromXML(tmp);
109 underlying_ = underlyingBuilder.underlying();
110
111 longShort_ = XMLUtils::getChildValue(vNode, "LongShort", true);
112 strike_ = XMLUtils::getChildValueAsDouble(vNode, "Strike", true);
113 notional_ = XMLUtils::getChildValueAsDouble(vNode, "Notional", true);
114 calendar_ = XMLUtils::getChildValue(vNode, "Calendar", true);
115 momentType_ = XMLUtils::getChildValue(vNode, "MomentType", false);
116 if (momentType_ == "")
117 momentType_ = "Variance";
118 string addPastDividendsStr = XMLUtils::getChildValue(vNode, "AddPastDividends", false);
119 if (addPastDividendsStr == "")
120 addPastDividends_ = false;
121 else
122 addPastDividends_ = parseBool(addPastDividendsStr);
123 initIndexName();
124}

126XMLNode* VarSwap::toXML(ore::data::XMLDocument& doc) {
127 XMLNode* node = Trade::toXML(doc);
128 XMLNode* vNode;
129 if (oldXml_) {
130 vNode = doc.allocNode("VarianceSwapData");
131 } else {
132 vNode = doc.allocNode(tradeType() + "Data");
133 }
134 XMLUtils::appendNode(node, vNode);
135 XMLUtils::addChild(doc, vNode, "StartDate", startDate_);
136 XMLUtils::addChild(doc, vNode, "EndDate", endDate_);
137 XMLUtils::addChild(doc, vNode, "Currency", currency_);
138 XMLUtils::appendNode(vNode, underlying_->toXML(doc));
139 XMLUtils::addChild(doc, vNode, "LongShort", longShort_);
140 XMLUtils::addChild(doc, vNode, "Strike", strike_);
141 XMLUtils::addChild(doc, vNode, "Notional", notional_);
142 XMLUtils::addChild(doc, vNode, "Calendar", calendar_);
143 XMLUtils::addChild(doc, vNode, "MomentType", momentType_);
144 XMLUtils::addChild(doc, vNode, "AddPastDividends", addPastDividends_);
145 return node;
146}

148void VarSwap::initIndexName() {
149 if (assetClassUnderlying_ == AssetClass::FX)
150 indexName_ = "FX-" + name();
151 else if (assetClassUnderlying_ == AssetClass::EQ)
152 indexName_ = "EQ-" + name();
153 else if (assetClassUnderlying_ == AssetClass::COM)
154 indexName_ = "COM-" + name();
155 else {
156 QL_FAIL("asset class " << assetClassUnderlying_ << " not supported.");
157 }
158}
