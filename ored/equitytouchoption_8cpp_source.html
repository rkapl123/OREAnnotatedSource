39EquityTouchOption::EquityTouchOption(Envelope& env, OptionData option, BarrierData barrier,
40 const EquityUnderlying& equityUnderlying, string payoffCurrency,
41 double payoffAmount, string startDate, string calendar, string eqIndex)
42 : ore::data::Trade("EquityTouchOption", env),
43 EquitySingleAssetDerivative("", env, equityUnderlying), option_(option), barrier_(barrier),
44 startDate_(startDate), calendar_(calendar), eqIndex_(eqIndex), payoffAmount_(payoffAmount),
45 payoffCurrency_(payoffCurrency) {
46 Barrier::Type barrierType = parseBarrierType(barrier_.type());
47 switch (barrierType) {
48 case Barrier::DownIn:
49 case Barrier::UpIn:
50 type_ = "One-Touch";
51 break;
52 case Barrier::DownOut:
53 case Barrier::UpOut:
54 type_ = "No-Touch";
55 break;
56 default:
57 QL_FAIL("unknown barrier type");
58 }
59}

61void EquityTouchOption::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
62
63 Date today = Settings::instance().evaluationDate();
64 const boost::shared_ptr<Market> market = engineFactory->market();
65
66 // Parse trade data
67 string assetName = equityName();
68 Currency ccy = parseCurrency(payoffCurrency_);
69 Real level = barrier_.levels()[0].value();
70 Date expiryDate = parseDate(option_.exerciseDates().front());
71 Barrier::Type barrierType = parseBarrierType(barrier_.type());
72 Option::Type type;
73 if (barrierType == Barrier::DownIn || barrierType == Barrier::DownOut)
74 type = Option::Type::Put;
75 else
76 type = Option::Type::Call;
77 bool payoffAtExpiry = option_.payoffAtExpiry();
78 Real rebate = barrier_.rebate();
79 Position::Type positionType = parsePositionType(option_.longShort());
80 Date start = ore::data::parseDate(startDate_);
81
82 QL_REQUIRE(tradeActions().empty(), "TradeActions not supported for EquityOption");
83 QL_REQUIRE(option_.exerciseDates().size() == 1, "Invalid number of excercise dates");
84 QL_REQUIRE(barrier_.levels().size() == 1, "Double barriers not supported for EquityTouchOptions");
85 QL_REQUIRE(barrier_.style().empty() || barrier_.style() == "American", "Only american barrier style suppported");
86 QL_REQUIRE(rebate == 0, "Rebates not supported for EquityTouchOptions");
87 QL_REQUIRE(payoffAtExpiry == true || barrierType == Barrier::Type::DownIn || barrierType == Barrier::Type::UpIn,
88 "Payoff at hit not supported for EquityNoTouchOptions");
89
90 // create payoff and exercise, as well as leg of underlying instrument
91 boost::shared_ptr<StrikedTypePayoff> payoff(new CashOrNothingPayoff(type, level, 1.0));
92 Leg leg;
93
94 leg.push_back(boost::shared_ptr<CashFlow>(new SimpleCashFlow(1.0, expiryDate)));
95
96 boost::shared_ptr<Exercise> exercise = boost::make_shared<AmericanExercise>(expiryDate, payoffAtExpiry);
97
98 boost::shared_ptr<Instrument> barrier = boost::make_shared<VanillaOption>(payoff, exercise);
99 boost::shared_ptr<Instrument> underlying = boost::make_shared<Swap>(Leg(), leg);
100
101 QL_REQUIRE(eqIndex_ != "", "No eqIndex provided");
102 QL_REQUIRE(calendar_ != "", "No calendar provided");
103
104 boost::shared_ptr<QuantExt::EquityIndex2> eqIndex = parseEquityIndex(eqIndex_);
105
106 // check if the barrier has been triggered already
107 bool triggered = false;
108 Calendar cal = eqIndex->fixingCalendar();
109 if (startDate_ != "" && start < today) {
110
111 Date d = start;
112
113 while (d < today && !triggered) {
114
115 Real fixing = eqIndex->pastFixing(d);
116
117 if (fixing == 0.0 || fixing == Null<Real>()) {
118 ALOG("Got invalid Equity fixing for index " << eqIndex_ << " on " << d
119 << "Skipping this date, assuming no trigger");
120 } else {
121 triggered = checkBarrier(fixing, barrierType, level);
122 }
123
124 d = cal.advance(d, 1, Days);
125 }
126 }
127
128 // set pricing engines
129 boost::shared_ptr<EngineBuilder> builder = engineFactory->builder(tradeType_);
130 QL_REQUIRE(builder, "No builder found for " << tradeType_);
131 boost::shared_ptr<EquityTouchOptionEngineBuilder> eqTouchOptBuilder =
132 boost::dynamic_pointer_cast<EquityTouchOptionEngineBuilder>(builder);
133 barrier->setPricingEngine(eqTouchOptBuilder->engine(assetName, ccy, type_));
134 if (type_ == "One-Touch") {
135 // if a one-touch option is triggered it becomes a simple forward cashflow
136 // which we price as a swap
137 builder = engineFactory->builder("Swap");
138 QL_REQUIRE(builder, "No builder found for Swap");
139 boost::shared_ptr<SwapEngineBuilderBase> swapBuilder =
140 boost::dynamic_pointer_cast<SwapEngineBuilderBase>(builder);
141 underlying->setPricingEngine(swapBuilder->engine(ccy));
142 }
143
144 bool isLong = (positionType == Position::Long) ? true : false;
145
146 std::vector<boost::shared_ptr<Instrument>> additionalInstruments;
147 std::vector<Real> additionalMultipliers;
148 Date lastPremiumDate =
149 addPremiums(additionalInstruments, additionalMultipliers, payoffAmount_, option_.premiumData(),
150 isLong ? -1.0 : 1.0, ccy, engineFactory, builder->configuration(MarketContext::pricing));
151
152 Handle<Quote> spot = market->equitySpot(assetName);
153 instrument_ = boost::make_shared<SingleBarrierOptionWrapper>(
154 barrier, isLong, expiryDate, false, underlying, barrierType, spot, level, rebate, ccy, start, eqIndex, cal, payoffAmount_,
155 payoffAmount_, additionalInstruments, additionalMultipliers);
156 npvCurrency_ = payoffCurrency_;
157 notional_ = payoffAmount_;
158 notionalCurrency_ = payoffCurrency_;
159 maturity_ = std::max(lastPremiumDate, expiryDate);
160
161 if (start != Date()) {
162 for (Date d = start; d <= expiryDate; d = cal.advance(d, 1 * Days))
163 requiredFixings_.addFixingDate(d, eqIndex_, expiryDate);
164 }
165
166 additionalData_["payoffAmount"] = payoffAmount_;
167 additionalData_["payoffCurrency"] = payoffCurrency_;
168
169 // ISDA taxonomy
170 additionalData_["isdaAssetClass"] = string("Equity");
171 additionalData_["isdaBaseProduct"] = string("Other");
172 additionalData_["isdaSubProduct"] = string("Price Return Basic Performance");
173 // skip the transaction level mapping for now
174 additionalData_["isdaTransaction"] = string("");
175}

177bool EquityTouchOption::checkBarrier(Real spot, Barrier::Type type, Real barrier) {
178 switch (type) {
179 case Barrier::DownIn:
180 case Barrier::DownOut:
181 return spot <= barrier;
182 case Barrier::UpIn:
183 case Barrier::UpOut:
184 return spot >= barrier;
185 default:
186 QL_FAIL("unknown barrier type " << type);
187 }
188}

190void EquityTouchOption::fromXML(XMLNode* node) {
191 Trade::fromXML(node);
192 XMLNode* eqNode = XMLUtils::getChildNode(node, "EquityTouchOptionData");
193 QL_REQUIRE(eqNode, "No EquityOptionData Node");
194 option_.fromXML(XMLUtils::getChildNode(eqNode, "OptionData"));
195 barrier_.fromXML(XMLUtils::getChildNode(eqNode, "BarrierData"));
196 Barrier::Type barrierType = parseBarrierType(barrier_.type());
197 switch (barrierType) {
198 case Barrier::DownIn:
199 case Barrier::UpIn:
200 type_ = "One-Touch";
201 break;
202 case Barrier::DownOut:
203 case Barrier::UpOut:
204 type_ = "No-Touch";
205 break;
206 default:
207 QL_FAIL("unknown barrier type");
208 }
209
210 XMLNode* tmp = XMLUtils::getChildNode(eqNode, "Underlying");
211 if (!tmp)
212 tmp = XMLUtils::getChildNode(eqNode, "Name");
213 equityUnderlying_.fromXML(tmp);
214 payoffAmount_ = XMLUtils::getChildValueAsDouble(eqNode, "PayoffAmount", true);
215 payoffCurrency_ = XMLUtils::getChildValue(eqNode, "PayoffCurrency", true);
216 startDate_ = XMLUtils::getChildValue(eqNode, "StartDate", false);
217 calendar_ = XMLUtils::getChildValue(eqNode, "Calendar", false);
218 eqIndex_ = XMLUtils::getChildValue(eqNode, "EQIndex", false);
219}

221XMLNode* EquityTouchOption::toXML(XMLDocument& doc) {
222 XMLNode* node = Trade::toXML(doc);
223 XMLNode* eqNode = doc.allocNode("EquityTouchOptionData");
224 XMLUtils::appendNode(node, eqNode);
225 XMLUtils::appendNode(eqNode, option_.toXML(doc));
226 XMLUtils::appendNode(eqNode, barrier_.toXML(doc));
227 XMLUtils::appendNode(eqNode, equityUnderlying_.toXML(doc));
228 XMLUtils::addChild(doc, eqNode, "PayoffCurrency", payoffCurrency_);
229 XMLUtils::addChild(doc, eqNode, "PayoffAmount", payoffAmount_);
230 if (startDate_ != "")
231 XMLUtils::addChild(doc, eqNode, "StartDate", startDate_);
232 if (eqIndex_ != "")
233 XMLUtils::addChild(doc, eqNode, "EQIndex", eqIndex_);
234 if (calendar_ != "")
235 XMLUtils::addChild(doc, eqNode, "Calendar", calendar_);
236
237 return node;
238}
