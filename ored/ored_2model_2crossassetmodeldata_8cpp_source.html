72  LOG("CrossAssetModelData: adding correlations.");
73  XMLNode* correlationNode = XMLUtils::locateNode(node, "InstantaneousCorrelations");

75  if (correlationNode) {

77  for (Size i = 0; i < nodes.size(); ++i) {

113 bool InstantaneousCorrelations::operator==(const InstantaneousCorrelations& rhs) {
114  // compare correlations by value (not the handle links)
115  map<CorrelationKey, Handle<Quote>>::const_iterator corr1, corr2;
116  for (corr1 = correlations_.begin(), corr2 = rhs.correlations_.begin();
117  corr1 != correlations_.end() && corr2 != rhs.correlations_.end(); ++corr1, ++corr2) {
118  if (corr1->first != corr2->first || !close_enough(corr1->second->value(), corr2->second->value()))
119  return false;
120  }
121  if (corr1 != correlations_.end() || corr2 != rhs.correlations_.end())
122  return false;
123 
124  return true;

192 void CrossAssetModelData::validate() {
193  QL_REQUIRE(irConfigs_.size() > 0, "no IR data provided");
194  QL_REQUIRE(fxConfigs_.size() == irConfigs_.size() - 1, "inconsistent number of FX data provided");
195  for (Size i = 0; i < fxConfigs_.size(); ++i)
196  QL_REQUIRE(fxConfigs_[i]->foreignCcy() == irConfigs_[i + 1]->ccy(),
197  "currency mismatch between IR and FX config vectors");
198 
199  if (measure_ == "BA") {
200  // ensure that the domestic LGM has shift = 0 and scaling = 1
201  for (Size i = 0; i < irConfigs_.size(); ++i)
202  if (irConfigs_[i]->ccy() == domesticCurrency_) {
203  QL_REQUIRE(close_enough(irConfigs_[i]->scaling(), 1.0),
204  "scaling for the domestic LGM must be 1 for BA measure simulations");
205  QL_REQUIRE(close_enough(irConfigs_[i]->shiftHorizon(), 0.0),
206  "shift horizon for the domestic LGM must be 0 for BA measure simulations");
207  }
208  }

447 void CrossAssetModelData::buildIrConfigs(std::map<std::string, boost::shared_ptr<IrLgmData>>& irDataMap) {
448  // Append IR configurations into the irConfigs vector in the order of the currencies
449  // in the currencies vector.
450  // If there is an IR configuration for any of the currencies missing, then we will
451  // look up the configuration with key "default" and use this instead. If this is
452  // not provided either we will throw an exception.
453  irConfigs_.resize(currencies_.size());
454  for (Size i = 0; i < currencies_.size(); i++) {
455  string ccy = currencies_[i];
456  std::string ccyKey;
457  for(auto const& d: irDataMap) {
458  if(d.second->ccy() == ccy) {
459  QL_REQUIRE(ccyKey.empty(), "CrossAssetModelData: duplicate ir config for ccy " << ccy);
460  ccyKey = d.first;
461  }
462  }
463  if (!ccyKey.empty())
464  irConfigs_[i] = irDataMap.at(ccyKey);
465  else { // copy from default
466  LOG("IR configuration missing for currency " << ccy << ", using default");
467  if (irDataMap.find("default") == irDataMap.end()) {
468  ALOG("Both default IR and " << ccy << " IR configuration missing");
469  QL_FAIL("Both default IR and " << ccy << " IR configuration missing");
470  }
471  boost::shared_ptr<IrLgmData> def = irDataMap["default"];
472  irConfigs_[i] = boost::make_shared<IrLgmData>(
473  ccy, // overwrite this and keep the others
474  def->calibrationType(), def->reversionType(), def->volatilityType(), def->calibrateH(),
475  def->hParamType(), def->hTimes(), def->hValues(), def->calibrateA(), def->aParamType(), def->aTimes(),
476  def->aValues(), def->shiftHorizon(), def->scaling(), def->optionExpiries(), def->optionTerms(),
477  def->optionStrikes());
478  }
479  LOG("CrossAssetModelData: IR config added for ccy " << ccy << " " << irConfigs_[i]->ccy());
480  }

539 void CrossAssetModelData::buildInfConfigs(const map<string, boost::shared_ptr<InflationModelData>>& mp) {
540 
541  // Append inflation model data to the infConfigs_ vector in the order of the inflation indices in the infindices_
542  // vector.
543 
544  // If for any of the inflation indices in the infindices_ vector, there is no inflation model data in mp then the
545  // default inflation model data is used. The default inflation model data should be in mp under the key name
546  // "default". If it is not provided either, an exception is thrown.
547 
548  for (const string& indexName : infindices_) {
549 
550  auto it = mp.find(indexName);
551  if (it != mp.end()) {
552  infConfigs_.push_back(it->second);
553  } else {
554 
555  LOG("Inflation index model data missing for index " << indexName << " so attempt to use default");
556 
557  auto itDefault = mp.find("default");
558  QL_REQUIRE(itDefault != mp.end(),

576 void CrossAssetModelData::buildCrConfigs(std::map<std::string, boost::shared_ptr<CrLgmData>>& crLgmDataMap,
577  std::map<std::string, boost::shared_ptr<CrCirData>>& crCirDataMap) {
578  // Append IR configurations into the irConfigs vector in the order of the currencies
579  // in the currencies vector.
580  // If there is an IR configuration for any of the currencies missing, then we will
581  // look up the configuration with key "default" and use this instead. If this is
582  // not provided either we will throw an exception.
583  crLgmConfigs_.clear();
584  crCirConfigs_.clear();
585 
586  for (Size i = 0; i < creditNames_.size(); i++) {
587  string name = creditNames_[i];
588  if (crLgmDataMap.find(name) != crLgmDataMap.end()) {
589  QL_REQUIRE(crCirDataMap.find(name) == crCirDataMap.end(), "");
590  crLgmConfigs_.push_back(crLgmDataMap[name]);
591  } else if (crCirDataMap.find(name) != crCirDataMap.end()) {
592  crCirConfigs_.push_back(crCirDataMap[name]);
593  } else { // copy from LGM default, CIR default is not used
594  LOG("CR configuration missing for name " << name << ", using default");
595  if (crLgmDataMap.find("default") == crLgmDataMap.end()) {
596  ALOG("Both default CR LGM and " << name << " CR configuration missing");
597  QL_FAIL("Both default CR and " << name << " CR configuration missing");
598  }
599  boost::shared_ptr<CrLgmData> def = crLgmDataMap["default"];
600  crLgmConfigs_.push_back(boost::make_shared<CrLgmData>(
601  name, // overwrite this and keep the others
602  def->calibrationType(), def->reversionType(), def->volatilityType(), def->calibrateH(),
603  def->hParamType(), def->hTimes(), def->hValues(), def->calibrateA(), def->aParamType(), def->aTimes(),
604  def->aValues(), def->shiftHorizon(), def->scaling(), def->optionExpiries(), def->optionTerms(),
605  def->optionStrikes()));
606  }
607  LOG("CrossAssetModelData: CR config added for name " << name << " " << name);
608  }
