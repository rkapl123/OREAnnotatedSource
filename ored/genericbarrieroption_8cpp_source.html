31 static const std::string mcscript =
32 " REQUIRE PayoffType == 0 OR PayoffType == 1;\n"
33 " REQUIRE TransatlanticBarrierType >= 0 AND TransatlanticBarrierType <= 4;\n"
34 " REQUIRE SIZE(BarrierTypes) == SIZE(BarrierLevels) / SIZE(Underlyings);\n"
35 " REQUIRE SIZE(BarrierTypes) == SIZE(BarrierRebates);\n"
36 " REQUIRE SIZE(BarrierTypes) == SIZE(BarrierRebateCurrencies);\n"
37 " REQUIRE SIZE(BarrierTypes) == SIZE(BarrierRebatePayTimes);\n"
38 " REQUIRE ExpiryDate >= BarrierMonitoringDates[SIZE(BarrierMonitoringDates)];\n"
39 "\n"
40 " NUMBER KnockedIn, KnockedOut, Active, rebate, TransatlanticActive;\n"
41 " NUMBER U, i, k, d, currentNotional, levelIndex;\n"
42 "\n"
43 " FOR d IN (1, SIZE(BarrierMonitoringDates), 1) DO\n"
44 "\n"
45 " FOR i IN (1, SIZE(BarrierTypes), 1) DO\n"
46 "\n"
47 " FOR k IN (1, SIZE(Underlyings), 1) DO\n"

49 "\n"

94 " { TransatlanticBarrierType == 2 AND Underlyings[k](ExpiryDate) <= TransatlanticBarrierLevel[k] } OR\n"
95 " { TransatlanticBarrierType == 3 AND Underlyings[k](ExpiryDate) < TransatlanticBarrierLevel[k] } OR\n"
96 " { TransatlanticBarrierType == 4 AND Underlyings[k](ExpiryDate) > TransatlanticBarrierLevel[k] } THEN\n"
97 " TransatlanticActive = 0;\n"
98 " END;\n"
99 " END;\n"
100 "\n"

102 "\n"
103 " IF PayoffType == 0 AND SIZE(Underlyings) == 1 THEN\n"
104 " value = Active * TransatlanticActive * PAY( LongShort * Quantity * max(0, PutCall * (Underlyings[1](ExpiryDate) - Strike)), ExpiryDate, SettlementDate, PayCurrency ) +\n"
105 " rebate;\n"

107 " value = Active * TransatlanticActive * PAY( LongShort * Amount, ExpiryDate, SettlementDate, PayCurrency ) +\n"
108 " rebate;\n"
109 " END;\n"
110 "\n"
111 " IF PayoffType == 0 THEN\n"
112 " currentNotional = Quantity * Strike;\n"
113 " ELSE\n"
114 " currentNotional = Amount;\n"
115 " END;";

185 " R_KOKI = NPV(R_KOKI, BarrierMonitoringDates[d]);\n"

187 "\n"
188 " FOR i IN (1, SIZE(BarrierTypes), 1) DO\n"
189 "\n"

194 " levelIndex = ((k - 1) * SIZE(BarrierTypes)) + i;\n"
195 " IF {BarrierTypes[i] == 1 AND U <= BarrierLevels[levelIndex]} OR\n"
196 " {BarrierTypes[i] == 2 AND U >= BarrierLevels[levelIndex]} THEN\n"
197 " IsKnockedIn = 1;"
198 " END;\n"
199 " IF {BarrierTypes[i] == 3 AND U < BarrierLevels[levelIndex]} OR\n"
200 " {BarrierTypes[i] == 4 AND U > BarrierLevels[levelIndex]} THEN\n"
201 " IsKnockedOut = 1;"
202 " END;\n"
203 " END;\n"
204 "\n"
205 " IF {IsKnockedIn == 1} THEN\n"
206 " V_KIKO = V_KO + V_KIKO + V_KOKI;\n"

222 " R_V = R_V + R_KIKO;\n"
223 " END;\n"
224 " END;\n"
225 "\n"
226 " IF { IsKnockedOut == 1 } THEN\n"
227 " V_KOKI = V_KI + V_KOKI + V_KIKO;\n"
228 " V_KIKO = V_KIKO * 0;\n"
229 " V_KO = V_NA + V_KO;\n"
230 " V_KI = V_KI * 0;\n"
231 " V_NA = V_NA * 0;\n"
232 " IF KikoType == 1 OR KikoType == 2 THEN\n"
233 " V_V = V_V * 0;\n"
234 " END;\n"
235 " R_KOKI = R_KI + R_KOKI + R_KIKO;\n"
236 " R_KIKO = R_KIKO * 0;\n"
237 " R_KO = R_NA + R_KO;\n"
238 " R_KI = R_KI * 0;\n"
239 " R_NA = R_NA * 0;\n"
240 " IF KikoType == 1 OR KikoType == 2 THEN\n"
241 " R_V = R_V * 0;\n"
242 " END;\n"
243 " IF BarrierRebatePayTimes[i] == 0 THEN\n"
244 " rebate = PAY( LongShort * BarrierRebates[i], BarrierMonitoringDates[d], BarrierMonitoringDates[d], BarrierRebateCurrencies[i] );\n"
245 " ELSE\n"
246 " rebate = PAY( LongShort * BarrierRebates[i], BarrierMonitoringDates[d], SettlementDate, BarrierRebateCurrencies[i] );\n"
247 " END;\n"
248 " END;\n"
249 "\n"
250 " END;\n"
251 "\n"
252 " END;\n"
253 "\n"
254 " rebate = NPV(rebate, TODAY);"
255 " R_V = NPV(R_V, TODAY);"
256 " V_V = NPV(V_V, TODAY);"
257 "\n"
258 "\n"
259 " rebate = rebate + ( PAY( LongShort * BarrierRebate, TODAY, SettlementDate, BarrierRebateCurrency ) - R_V );\n"
260 " value = V_V + rebate;\n"
261 "\n"
262 " IF PayoffType == 0 THEN\n"
263 " currentNotional = Quantity * Strike;\n"
264 " ELSE\n"
265 " currentNotional = Amount;\n"
266 " END;";

270void GenericBarrierOption::build(const boost::shared_ptr<EngineFactory>& factory) {
271
272 // set script parameters
273
274 clear();
275 initIndices();
276
277 if (underlyings_.size() > 1)
278 QL_REQUIRE(optionData_.payoffType() == "CashOrNothing",
279 "Only CashOrNothing payoff allowed for mulitple underlyings");
280
281 std::string payoffType;
282 if (optionData_.payoffType() == "Vanilla" || optionData_.payoffType() == "AssetOrNothing")
283 payoffType = "0";
284 else if (optionData_.payoffType() == "CashOrNothing")
285 payoffType = "1";
286 else {
287 QL_FAIL("PayoffTpye (" << optionData_.payoffType() << ") must be Vanilla or CashOrNothing");
288 }
289 numbers_.emplace_back("Number", "PayoffType", payoffType);
290
291 QL_REQUIRE(optionData_.payoffType() != "Vanilla" || (!quantity_.empty() && !strike_.empty() && amount_.empty()),
292 "Need Quantity, Strike, no Amount for PayoffType = Vanilla");
293 QL_REQUIRE(optionData_.payoffType() != "AssetOrNothing" ||
294 (!quantity_.empty() && strike_.empty() && amount_.empty()),
295 "Need Quantity, no Strike, no Amount for PayoffType = AssetOrNothing");
296 QL_REQUIRE(optionData_.payoffType() != "CashOrNothing" ||
297 (quantity_.empty() && strike_.empty() && !amount_.empty()),
298 "Need no Quantity, no Strike, Amount for PayoffType = CashOrNothing");
299
300 std::string transatlanticBarrierType = "0";
301 std::vector<std::string> transatlanticBarrierLevel(underlyings_.size(), "0");
302 std::string transatlanticBarrierRebate = "0.0";
303 std::string transatlanticBarrierRebateCurrency = payCurrency_;
304 if (!transatlanticBarrier_.type().empty()) {
305 if (transatlanticBarrier_.type() == "DownAndIn")
306 transatlanticBarrierType = "1";
307 else if (transatlanticBarrier_.type() == "UpAndIn")
308 transatlanticBarrierType = "2";
309 else if (transatlanticBarrier_.type() == "DownAndOut")
310 transatlanticBarrierType = "3";
311 else if (transatlanticBarrier_.type() == "UpAndOut")
312 transatlanticBarrierType = "4";
313 else {
314 QL_FAIL("Transatlantic BarrierType (" << transatlanticBarrier_.type()
315 << ") must be DownAndIn, UpAndIn, DownAndOut, UpAndOut");
316 }
317 QL_REQUIRE(transatlanticBarrier_.levels().size() == underlyings_.size(),
318 "Transatlantic Barrier must have exactly 1 level for each underlying, got " << transatlanticBarrier_.levels().size());
319 transatlanticBarrierLevel.clear();
320 for (const auto& l : transatlanticBarrier_.levels())
321 transatlanticBarrierLevel.push_back(boost::lexical_cast<std::string>(l.value()));
322 transatlanticBarrierRebate = boost::lexical_cast<std::string>(transatlanticBarrier_.rebate());
323 if (!transatlanticBarrier_.rebateCurrency().empty())
324 transatlanticBarrierRebateCurrency = transatlanticBarrier_.rebateCurrency();
325 }
326 numbers_.emplace_back("Number", "TransatlanticBarrierType", transatlanticBarrierType);
327 numbers_.emplace_back("Number", "TransatlanticBarrierLevel", transatlanticBarrierLevel);
328 numbers_.emplace_back("Number", "TransatlanticBarrierRebate", transatlanticBarrierRebate);
329 currencies_.emplace_back("Currency", "TransatlanticBarrierRebateCurrency", transatlanticBarrierRebateCurrency);
330
331 auto positionType = parsePositionType(optionData_.longShort());
332 numbers_.emplace_back("Number", "LongShort", positionType == Position::Long ? "1" : "-1");
333
334 numbers_.emplace_back("Number", "PutCall", parseOptionType(optionData_.callPut()) == Option::Call ? "1.0" : "-1.0");
335 numbers_.emplace_back("Number", "Quantity", quantity_.empty() ? "0.0" : quantity_);
336 if (!strike_.empty())
337 numbers_.emplace_back("Number", "Strike", strike_);
338 numbers_.emplace_back("Number", "Amount", amount_.empty() ? "0.0" : amount_);
339 currencies_.emplace_back("Currency", "PayCurrency", payCurrency_);
340
341 QL_REQUIRE(optionData_.exerciseDates().size() == 1,
342 "OptionData must contain exactly one ExerciseDate, got " << optionData_.exerciseDates().size());
343 events_.emplace_back("ExpiryDate", optionData_.exerciseDates().front());
344
345 if (!settlementDate_.empty()) {
346 QL_REQUIRE(
347 settlementLag_.empty() && settlementCalendar_.empty() && settlementConvention_.empty(),
348 "If SettlementDate is given, no SettlementLag, SettlementCalendar or SettlementConvention must be given.");
349 events_.emplace_back("SettlementDate", settlementDate_);
350 } else {
351 Date ref = parseDate(optionData_.exerciseDates().front());
352 Period p = settlementLag_.empty() ? 0 * Days : parsePeriod(settlementLag_);
353 Calendar cal =
354 settlementCalendar_.empty() ? getUnderlyingCalendar(factory) : parseCalendar(settlementCalendar_);
355 BusinessDayConvention conv =
356 settlementConvention_.empty() ? Following : parseBusinessDayConvention(settlementConvention_);
357 events_.emplace_back("SettlementDate", ore::data::to_string(cal.advance(ref, p, conv)));
358 }
359
360 if (barrierMonitoringDates_.hasData()) {
361 QL_REQUIRE(barrierMonitoringStartDate_.empty() && barrierMonitoringEndDate_.empty(),
362 "If ScheduleData is given, no StartDate or EndDate must be given");
363 events_.emplace_back("BarrierMonitoringDates", barrierMonitoringDates_);
364 } else {
365 /* build a daily schedule from the given start / end dates and deriving the calendar from
366 the underlying */
367 QL_REQUIRE(!barrierMonitoringStartDate_.empty() && !barrierMonitoringEndDate_.empty(),
368 "If no ScheduleData is given, StartDate and EndDate must be given");
369 events_.emplace_back("BarrierMonitoringDates",
370 ScheduleData(ScheduleRules(barrierMonitoringStartDate_, barrierMonitoringEndDate_, "1D",
371 getUnderlyingCalendar(factory).name(), "F", "F", "Forward")));
372 }
373
374 std::vector<std::string> barrierTypes, barrierLevels, barrierRebates, barrierRebateCurrencies,
375 barrierRebatePayTimes;
376 bool hasKi = false, hasKo = false;
377 for (auto const& b : barriers_) {
378 std::string barrierType;
379 if (b.type() == "DownAndIn") {
380 barrierType = "1";
381 hasKi = true;
382 } else if (b.type() == "UpAndIn") {
383 barrierType = "2";
384 hasKi = true;
385 } else if (b.type() == "DownAndOut") {
386 barrierType = "3";
387 hasKo = true;
388 } else if (b.type() == "UpAndOut") {
389 barrierType = "4";
390 hasKo = true;
391 } else {
392 QL_FAIL("BarrierType (" << b.type() << ") must be DownAndIn, UpAndIn, DownAndOut, UpAndOut");
393 }
394 barrierTypes.push_back(barrierType);
395 QL_REQUIRE(b.levels().size() == underlyings_.size(), "Barrier must have exactly number of levels as underlyings, got " << b.levels().size());
396 for (const auto& l : b.levels())
397 barrierLevels.push_back(boost::lexical_cast<std::string>(l.value()));
398 barrierRebates.push_back(boost::lexical_cast<std::string>(b.rebate()));
399 barrierRebateCurrencies.push_back(b.rebateCurrency().empty() ? payCurrency_ : b.rebateCurrency());
400 std::string rebatePayTime;
401 if (b.rebatePayTime() == "atHit")
402 rebatePayTime = "0";
403 else if (b.rebatePayTime() == "atExpiry" || b.rebatePayTime().empty())
404 rebatePayTime = "1";
405 else {
406 QL_FAIL("RebatePayTime (" << b.rebatePayTime() << ") must be atHit, atExpiry");
407 }
408 barrierRebatePayTimes.push_back(rebatePayTime);
409 }
410
411 // if there is at least one ki barrier, all rebates must be identical + atExpiry
412 // and are set via BarrierRebate, BarrierRebateCurrency
413
414 std::string barrierRebate = "0.0", barrierRebateCurrency = payCurrency_;
415 if (hasKi) {
416 for (Size i = 1; i < barrierTypes.size(); ++i) {
417 QL_REQUIRE(close_enough(parseReal(barrierRebates[i]), parseReal(barrierRebates[0])) &&
418 barrierRebateCurrencies[i] == barrierRebateCurrencies[0],
419 "If Knock-In barrier is present, all rebates must be identical, found "
420 << barrierRebates[0] << " " << barrierRebateCurrencies[0] << " and " << barrierRebates[i]
421 << " " << barrierRebateCurrencies[i]);
422 }
423 for (Size i = 0; i < barrierTypes.size(); ++i) {
424 QL_REQUIRE(barrierRebatePayTimes[i] == "1",
425 "If Knock-In barrier is present, all rebate pay times must be atExpiry");
426 }
427 barrierRebate = barrierRebates[0];
428 barrierRebateCurrency = barrierRebateCurrencies[0];
429 for (Size i = 0; i < barrierTypes.size(); ++i) {
430 barrierRebates[i] = "0.0";
431 barrierRebateCurrencies[i] = payCurrency_;
432 }
433 }
434
435 numbers_.emplace_back("Number", "BarrierTypes", barrierTypes);
436 numbers_.emplace_back("Number", "BarrierLevels", barrierLevels);
437 numbers_.emplace_back("Number", "BarrierRebates", barrierRebates);
438 currencies_.emplace_back("Currency", "BarrierRebateCurrencies", barrierRebateCurrencies);
439 numbers_.emplace_back("Number", "BarrierRebatePayTimes", barrierRebatePayTimes);
440 numbers_.emplace_back("Number", "BarrierRebate", barrierRebate);
441 currencies_.emplace_back("Currency", "BarrierRebateCurrency", barrierRebateCurrency);
442
443 std::string kikoType;
444 if (kikoType_ == "KoAlways" || kikoType_.empty())
445 kikoType = "1";
446 else if (kikoType_ == "KoBeforeKi")
447 kikoType = "2";
448 else if (kikoType_ == "KoAfterKi")
449 kikoType = "3";
450 else {
451 QL_FAIL("KikoType (" << kikoType_ << ") must be KoAlways, KoBeforeKi, KoAfterKi");
452 }
453 numbers_.emplace_back("Number", "KikoType", kikoType);
454
455 QL_REQUIRE((hasKi && hasKo) || kikoType == "1",
456 "KikoType (" << kikoType_ << ") must be KoAlways if there are only Ko or only Ki barriers");
457 QL_REQUIRE(!(hasKi && hasKo) || !kikoType_.empty(),
458 "KikoType must be given (KoAlways, KoBeforeKi, KoAfterKi) if both Ko and Ki barriers are present");
459
460 // set product tag
461
462 productTag_ = (underlyings_.size() < 2) ? "SingleAssetOptionBwd({AssetClass})"
463 : "MultiAssetOption({AssetClass})";
464
465 // set script
466
467 script_[""] = ScriptedTradeScriptData(
468 mcscript, "value",
469 {{"currentNotional", "currentNotional"},
470 {"notionalCurrency", "PayCurrency"},
471 {"Active", "Active"},
472 {"TransatlanticActive", "TransatlanticActive"}},
473 {}, {}, {ScriptedTradeScriptData::CalibrationData("Underlyings", {"Strike", "BarrierLevels"})});
474 script_["FD"] = ScriptedTradeScriptData(
475 fdscript, "value", {{"currentNotional", "currentNotional"}, {"notionalCurrency", "PayCurrency"}}, {}, {},
476 {ScriptedTradeScriptData::CalibrationData("Underlyings", {"Strike", "BarrierLevels"})});
477
478 // build trade
479

491QuantLib::Calendar GenericBarrierOption::getUnderlyingCalendar(const boost::shared_ptr<EngineFactory>& factory) const {
492 Calendar calendar = NullCalendar();
493 QL_REQUIRE(underlyings_.size() > 0, "No underlyings provided.");
494 IndexInfo ind(scriptedIndexName(underlyings_.front()));
495 if (ind.isFx()) {
496 /* just take the default calendars for the two currencies, this can be improved once we have
497 full fx indices available in the t0 market (TODO) */
498 calendar = parseCalendar(ind.fx()->sourceCurrency().code() + "," + ind.fx()->targetCurrency().code());
499 } else if (ind.isEq()) {
500 /* get the equity calendar from the market */
501 calendar = factory->market()
502 ->equityCurve(ind.eq()->name(), factory->configuration(MarketContext::pricing))
503 ->fixingCalendar();
504 } else if (ind.isComm()) {
505 /* again, just take the default calendar for the comm currency until we have the actual calendar
506 available via the market interface */
507 calendar =
508 parseCalendar(factory->market()
509 ->commodityPriceCurve(ind.commName(), factory->configuration(MarketContext::pricing))
510 ->currency()
511 .code());
512 }
513 return calendar;
514}

516void GenericBarrierOption::fromXML(XMLNode* node) {
517 Trade::fromXML(node);
518 XMLNode* dataNode = XMLUtils::getChildNode(node, tradeType() + "Data");
519 QL_REQUIRE(dataNode, tradeType() + "Data node not found");
520
521 if (XMLNode* underlyingsNode = XMLUtils::getChildNode(dataNode, "Underlyings")) {
522 QL_REQUIRE(underlyingsNode, "No Underlyings node");
523 auto underlyings = XMLUtils::getChildrenNodes(underlyingsNode, "Underlying");
524 for (auto const& n : underlyings) {
525 UnderlyingBuilder underlyingBuilder;
526 underlyingBuilder.fromXML(n);
527 underlyings_.push_back(underlyingBuilder.underlying());
528 }
529 } else {
530 XMLNode* tmp = XMLUtils::getChildNode(dataNode, "Underlying");
531 if (!tmp)
532 tmp = XMLUtils::getChildNode(dataNode, "Name");
533 UnderlyingBuilder underlyingBuilder;
534 underlyingBuilder.fromXML(tmp);
535 underlyings_.push_back(underlyingBuilder.underlying());
536 }
537
538 optionData_.fromXML(XMLUtils::getChildNode(dataNode, "OptionData"));
539
540 auto barriersNode = XMLUtils::getChildNode(dataNode, "Barriers");
541 QL_REQUIRE(barriersNode, "No Barriers node found");
542 if (auto sd = XMLUtils::getChildNode(barriersNode, "ScheduleData")) {
543 barrierMonitoringDates_.fromXML(sd);
544 }
545 barrierMonitoringStartDate_ = XMLUtils::getChildValue(barriersNode, "StartDate", false);
546 barrierMonitoringEndDate_ = XMLUtils::getChildValue(barriersNode, "EndDate", false);
547 kikoType_ = XMLUtils::getChildValue(barriersNode, "KikoType", false, "KoAlways");
548 auto barriers = XMLUtils::getChildrenNodes(barriersNode, "BarrierData");
549 for (auto const& n : barriers) {
550 barriers_.push_back(BarrierData());
551 barriers_.back().fromXML(n);
552 }
553
554 auto transatlanticBarrierNode = XMLUtils::getChildNode(dataNode, "TransatlanticBarrier");
555 if (transatlanticBarrierNode) {
556 auto b = XMLUtils::getChildNode(transatlanticBarrierNode, "BarrierData");
557 if (b)
558 transatlanticBarrier_.fromXML(b);
559 }
560
561 payCurrency_ = XMLUtils::getChildValue(dataNode, "PayCurrency", true);
562 settlementDate_ = XMLUtils::getChildValue(dataNode, "SettlementDate", false);
563 settlementLag_ = XMLUtils::getChildValue(dataNode, "SettlementLag", false);
564 settlementCalendar_ = XMLUtils::getChildValue(dataNode, "SettlementCalendar", false);
565 settlementConvention_ = XMLUtils::getChildValue(dataNode, "SettlementConvention", false);
566 quantity_ = XMLUtils::getChildValue(dataNode, "Quantity", false);
567 strike_ = XMLUtils::getChildValue(dataNode, "Strike", false);
568 amount_ = XMLUtils::getChildValue(dataNode, "Amount", false);
569
570 initIndices();
571}

573XMLNode* GenericBarrierOption::toXML(XMLDocument& doc) {

576 XMLUtils::appendNode(node, dataNode);

595 }
596 if (!kikoType_.empty())
597 XMLUtils::addChild(doc, barriers, "KikoType", kikoType_);

599
600 if (!transatlanticBarrier_.type().empty()) {
601 XMLNode* transatlanticBarrierNode = doc.allocNode("TransatlanticBarrier");
602 XMLUtils::appendNode(transatlanticBarrierNode, transatlanticBarrier_.toXML(doc));
603 XMLUtils::appendNode(dataNode, transatlanticBarrierNode);
604 }
605
606 XMLUtils::addChild(doc, dataNode, "PayCurrency", payCurrency_);
607 if (!settlementDate_.empty())
608 XMLUtils::addChild(doc, dataNode, "SettlementDate", settlementDate_);
609 if (!settlementLag_.empty())
610 XMLUtils::addChild(doc, dataNode, "SettlementLag", settlementLag_);
611 if (!settlementCalendar_.empty())
612 XMLUtils::addChild(doc, dataNode, "SettlementCalendar", settlementCalendar_);
613 if (!settlementConvention_.empty())
614 XMLUtils::addChild(doc, dataNode, "SettlementConvention", settlementConvention_);
615
616 if (!quantity_.empty())
617 XMLUtils::addChild(doc, dataNode, "Quantity", quantity_);
618 if (!strike_.empty())
619 XMLUtils::addChild(doc, dataNode, "Strike", strike_);
620 if (!amount_.empty())
621 XMLUtils::addChild(doc, dataNode, "Amount", amount_);
622
623 return node;
624}
