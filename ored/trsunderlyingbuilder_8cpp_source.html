49void BondTrsUnderlyingBuilder::build(
50 const std::string& parentId, const boost::shared_ptr<Trade>& underlying, const std::vector<Date>& valuationDates,
51 const boost::shared_ptr<EngineFactory>& engineFactory, boost::shared_ptr<QuantLib::Index>& underlyingIndex,
52 Real& underlyingMultiplier, std::map<std::string, double>& indexQuantities,
53 std::map<std::string, boost::shared_ptr<QuantExt::FxIndex>>& fxIndices, Real& initialPrice,
54 std::string& assetCurrency, std::string& creditRiskCurrency,
55 std::map<std::string, SimmCreditQualifierMapping>& creditQualifierMapping, Date& maturity,
56 const std::function<boost::shared_ptr<QuantExt::FxIndex>(
57 const boost::shared_ptr<Market> market, const std::string& configuration, const std::string& domestic,
58 const std::string& foreign, std::map<std::string, boost::shared_ptr<QuantExt::FxIndex>>& fxIndices)>&
59 getFxIndex,
60 const std::string& underlyingDerivativeId) const {
61 auto t = boost::dynamic_pointer_cast<ore::data::Bond>(underlying);
62 QL_REQUIRE(t, "could not cast to ore::data::Bond, this is unexpected");
63 auto qlBond = boost::dynamic_pointer_cast<QuantLib::Bond>(underlying->instrument()->qlInstrument());
64 QL_REQUIRE(qlBond, "expected QuantLib::Bond, could not cast");
65 underlyingIndex = boost::make_shared<QuantExt::BondIndex>(
66 t->bondData().securityId(), true, false, NullCalendar(), qlBond, Handle<YieldTermStructure>(),
67 Handle<DefaultProbabilityTermStructure>(), Handle<Quote>(), Handle<Quote>(), Handle<YieldTermStructure>(), true,
68 t->bondData().priceQuoteMethod(), t->bondData().priceQuoteBaseValue(), t->bondData().isInflationLinked());
69 underlyingMultiplier = t->bondData().bondNotional();
70 indexQuantities["BOND-" + t->bondData().securityId()] = underlyingMultiplier;
71 Real adj = t->bondData().priceQuoteMethod() == QuantExt::BondIndex::PriceQuoteMethod::CurrencyPerUnit
72 ? 1.0 / t->bondData().priceQuoteBaseValue()
73 : 1.0;
74 DLOG("BondTrsUnderlyingBuilder: price quote method adjustment for " << t->bondData().securityId() << " is " << adj);
75 if (initialPrice != Null<Real>())
76 initialPrice = initialPrice * qlBond->notional(valuationDates.front()) * adj;
77 assetCurrency = t->bondData().currency();
78 if (!t->bondData().creditCurveId().empty())
79 creditRiskCurrency = t->bondData().currency();
80 creditQualifierMapping[securitySpecificCreditCurveName(t->bondData().securityId(), t->bondData().creditCurveId())] =
81 SimmCreditQualifierMapping(t->bondData().securityId(), t->bondData().creditGroup());
82 creditQualifierMapping[t->bondData().creditCurveId()] =
83 SimmCreditQualifierMapping(t->bondData().securityId(), t->bondData().creditGroup());
84 // FIXME shouldn't we leave that empty and let TRS determine the maturity date based on valuation / funding dates?
85 maturity = qlBond->maturityDate();
86}

128template <class T>
129void AssetPositionTrsUnderlyingBuilder<T>::build(
130 const std::string& parentId, const boost::shared_ptr<Trade>& underlying, const std::vector<Date>& valuationDates,
131 const boost::shared_ptr<EngineFactory>& engineFactory, boost::shared_ptr<QuantLib::Index>& underlyingIndex,
132 Real& underlyingMultiplier, std::map<std::string, double>& indexQuantities,
133 std::map<std::string, boost::shared_ptr<QuantExt::FxIndex>>& fxIndices, Real& initialPrice,
134 std::string& assetCurrency, std::string& creditRiskCurrency,
135 std::map<std::string, SimmCreditQualifierMapping>& creditQualifierMapping, Date& maturity,
136 const std::function<boost::shared_ptr<QuantExt::FxIndex>(
137 const boost::shared_ptr<Market> market, const std::string& configuration, const std::string& domestic,
138 const std::string& foreign, std::map<std::string, boost::shared_ptr<QuantExt::FxIndex>>& fxIndices)>&
139 getFxIndex,
140 const std::string& underlyingDerivativeId) const {
141 auto t = boost::dynamic_pointer_cast<T>(underlying);
142 QL_REQUIRE(t, "could not cast to ore::data::EquityPosition, this is unexpected");
143 if (t->isSingleCurrency()) {
144 assetCurrency = t->npvCurrency();
145 DLOG("underlying equity position is single-currency, assetCurrency is " << assetCurrency);
146 } else {
147 // asset currency is set to funding currency data currency in trs as a default
148 // we use fxSpot() as opposed to fxRate() here to ensure consistency between NPV() and the fixing of an
149 // equivalent index representing the same basket
150 t->setNpvCurrencyConversion(
151 assetCurrency, engineFactory->market()->fxSpot(t->npvCurrency() + assetCurrency,
152 engineFactory->configuration(MarketContext::pricing)));
153 DLOG("underlying equity position is multi-currency, set assetCurrency to fundingCurrency = " << assetCurrency);
154 }
155 std::vector<boost::shared_ptr<QuantExt::FxIndex>> fxConversion(t->data().underlyings().size());
156 std::vector<boost::shared_ptr<QuantLib::Index>> indices;
157 for (auto const& i : t->indices()) {
158 indices.push_back(i);
159 DLOG("underlying equity index " << i->name() << " added.");
160 }
161 for (Size i = 0; i < t->data().underlyings().size(); ++i) {
162 fxConversion[i] = getFxIndex(engineFactory->market(), engineFactory->configuration(MarketContext::pricing),
163 assetCurrency, getIndexCurrencyFromPosition(t, i), fxIndices);
164 updateQuantities(indexQuantities, t->data().underlyings()[i].name(),
165 t->weights()[i] * t->data().quantity());
166 }
167 underlyingIndex = boost::make_shared<QuantExt::CompositeIndex>("Composite Index trade id " + parentId, indices,
168 t->weights(), fxConversion);
169 DLOG("underlying equity index built with " << indices.size() << " constituents.");
170 underlyingMultiplier = t->data().quantity();
171}

208void EquityOptionPositionTrsUnderlyingBuilder::build(
209 const std::string& parentId, const boost::shared_ptr<Trade>& underlying, const std::vector<Date>& valuationDates,
210 const boost::shared_ptr<EngineFactory>& engineFactory, boost::shared_ptr<QuantLib::Index>& underlyingIndex,
211 Real& underlyingMultiplier, std::map<std::string, double>& indexQuantities,
212 std::map<std::string, boost::shared_ptr<QuantExt::FxIndex>>& fxIndices, Real& initialPrice,
213 std::string& assetCurrency, std::string& creditRiskCurrency,
214 std::map<std::string, SimmCreditQualifierMapping>& creditQualifierMapping, Date& maturity,
215 const std::function<boost::shared_ptr<QuantExt::FxIndex>(
216 const boost::shared_ptr<Market> market, const std::string& configuration, const std::string& domestic,
217 const std::string& foreign, std::map<std::string, boost::shared_ptr<QuantExt::FxIndex>>& fxIndices)>&
218 getFxIndex,
219 const std::string& underlyingDerivativeId) const {
220 auto t = boost::dynamic_pointer_cast<ore::data::EquityOptionPosition>(underlying);
221 QL_REQUIRE(t, "could not cast to ore::data::EquityOptionPosition, this is unexpected");
222 if (t->isSingleCurrency()) {
223 assetCurrency = t->npvCurrency();
224 DLOG("underlying equity option position is single-currency, assetCurrency is " << assetCurrency);
225 } else {
226 // asset currency is set to funding currency data currency in trs as a default
227 // we use fxSpot() as opposed to fxRate() here to ensure consistency between NPV() and the fixing of an
228 // equivalent index representing the same basket
229 t->setNpvCurrencyConversion(
230 assetCurrency, engineFactory->market()->fxSpot(t->npvCurrency() + assetCurrency,
231 engineFactory->configuration(MarketContext::pricing)));
232 DLOG("underlying equity option position is multi-currency, set assetCurrency to fundingCurrency = "
233 << assetCurrency);
234 }
235 std::vector<boost::shared_ptr<QuantExt::FxIndex>> fxConversion(t->data().underlyings().size());
236 std::vector<boost::shared_ptr<QuantLib::Index>> indices;
237 for (auto const& i : t->historicalPriceIndices()) {
238 indices.push_back(i);
239 DLOG("underlying historical equity option price index " << i->name() << " added.");
240 }
241 QL_REQUIRE(indices.size() == t->data().underlyings().size(),
242 "underlying historical price indices size (" << indices.size() << ") must match underlyings size ("
243 << t->data().underlyings().size());
244 for (Size i = 0; i < t->data().underlyings().size(); ++i) {
245 fxConversion[i] = getFxIndex(engineFactory->market(), engineFactory->configuration(MarketContext::pricing),
246 assetCurrency, t->currencies()[i], fxIndices);
247 indexQuantities[indices[i]->name()] = t->weights()[i] * t->positions()[i];
248 }
249 std::vector<Real> w;
250 for (Size i = 0; i < t->weights().size(); ++i) {
251 w.push_back(t->weights()[i] * t->positions()[i]);
252 }
253 underlyingIndex =
254 boost::make_shared<QuantExt::CompositeIndex>("Composite Index trade id " + parentId, indices, w, fxConversion);
255 DLOG("underlying equity option historical price index built with " << indices.size() << " constituents.");
256 underlyingMultiplier = t->data().quantity();
257}

259void BondPositionTrsUnderlyingBuilder::build(
260 const std::string& parentId, const boost::shared_ptr<Trade>& underlying, const std::vector<Date>& valuationDates,
261 const boost::shared_ptr<EngineFactory>& engineFactory, boost::shared_ptr<QuantLib::Index>& underlyingIndex,
262 Real& underlyingMultiplier, std::map<std::string, double>& indexQuantities,
263 std::map<std::string, boost::shared_ptr<QuantExt::FxIndex>>& fxIndices, Real& initialPrice,
264 std::string& assetCurrency, std::string& creditRiskCurrency,
265 std::map<std::string, SimmCreditQualifierMapping>& creditQualifierMapping, Date& maturity,
266 const std::function<boost::shared_ptr<QuantExt::FxIndex>(
267 const boost::shared_ptr<Market> market, const std::string& configuration, const std::string& domestic,
268 const std::string& foreign, std::map<std::string, boost::shared_ptr<QuantExt::FxIndex>>& fxIndices)>&
269 getFxIndex,
270 const std::string& underlyingDerivativeId) const {
271 auto t = boost::dynamic_pointer_cast<ore::data::BondPosition>(underlying);
272 QL_REQUIRE(t, "could not cast to ore::data::BondPosition, this is unexpected");
273 if (t->isSingleCurrency()) {
274 assetCurrency = t->npvCurrency();
275 DLOG("underlying bond position is single-currency, assetCurrency is " << assetCurrency);
276 } else {
277 // asset currency is set to funding currency data currency in trs as a default
278 t->setNpvCurrencyConversion(
279 assetCurrency, engineFactory->market()->fxSpot(t->npvCurrency() + assetCurrency,
280 engineFactory->configuration(MarketContext::pricing)));
281 DLOG("underlying bond position is multi-currency, set assetCurrency to fundingCurrency = " << assetCurrency);
282 }
283
284 std::vector<boost::shared_ptr<QuantExt::FxIndex>> fxConversion(t->data().underlyings().size());
285 std::vector<boost::shared_ptr<QuantLib::Index>> indices;
286 bool hasCreditRisk = false;
287 for (Size i = 0; i < t->bonds().size(); ++i) {
288 // relative index, because weights are supposed to include any amortization factors
289 indices.push_back(boost::make_shared<QuantExt::BondIndex>(
290 t->data().underlyings()[i].name(), true, true, NullCalendar(), t->bonds()[i].bond,
291 Handle<YieldTermStructure>(), Handle<DefaultProbabilityTermStructure>(), Handle<Quote>(), Handle<Quote>(),
292 Handle<YieldTermStructure>(), true, t->bonds()[i].priceQuoteMethod, t->bonds()[i].priceQuoteBaseValue,
293 t->bonds()[i].isInflationLinked, t->data().underlyings()[i].bidAskAdjustment()));
294 DLOG("underlying bond index " << indices.back()->name() << " added.");
295 indexQuantities["BOND-" + t->data().underlyings()[i].name()] = t->weights()[i] * t->data().quantity();
296 creditQualifierMapping[ore::data::securitySpecificCreditCurveName(t->bonds()[i].securityId,
297 t->bonds()[i].creditCurveId)] =
298 SimmCreditQualifierMapping(t->bonds()[i].securityId, t->bonds()[i].creditGroup);
299 creditQualifierMapping[t->bonds()[i].creditCurveId] =
300 SimmCreditQualifierMapping(t->bonds()[i].securityId, t->bonds()[i].creditGroup);
301 hasCreditRisk = hasCreditRisk || t->bonds()[i].hasCreditRisk;
302 }
303 for (Size i = 0; i < t->data().underlyings().size(); ++i) {
304 fxConversion[i] = getFxIndex(engineFactory->market(), engineFactory->configuration(MarketContext::pricing),
305 assetCurrency, t->bonds()[i].currency, fxIndices);
306 }
307 std::vector<Real> w;
308 for (Size i = 0; i < t->weights().size(); ++i) {
309 w.push_back(t->weights()[i]);
310 }
311 underlyingIndex =
312 boost::make_shared<QuantExt::CompositeIndex>("Composite Index trade id " + parentId, indices, w, fxConversion);
313 DLOG("underlying bond position index built with " << indices.size() << " constituents.");
314 underlyingMultiplier = t->data().quantity();
315 if (hasCreditRisk)
316 creditRiskCurrency = assetCurrency;
317 // FIXME same question as for single bond underlying: shouldn't we leave that empty and let TRS determine the
318 // maturity date based on valuation / funding dates?
319 maturity = t->maturity();
320}
