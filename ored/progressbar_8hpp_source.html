36//! Abstract Base class for a Progress Indicator
37/*! \ingroup utilities

39class ProgressIndicator {

42 virtual ~ProgressIndicator() {}
43 virtual void updateProgress(const unsigned long progress, const unsigned long total) = 0;
44 virtual void reset() = 0;
45};
