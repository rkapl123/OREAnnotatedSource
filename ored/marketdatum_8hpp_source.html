64//! Base market data class
65/*!
66 This class holds a single market point, a SimpleQuote pointer and generic
67 additional information.
68
69 The market point is classified by an instrument type, a quote type and
70 a name string. The name's structure depends on the market point's type
71 with tokens separated by "/".
72
73 Specific market data classes are derived from this base class and hold
74 additional specific data that are represented by the market point's name.
75
76 \ingroup marketdata
77*/
78class MarketDatum {
79public:
80 MarketDatum() {}
81 //! Supported market instrument types

83 ZERO,

89 IMM_FRA,

92 BMA_SWAP,

94 CC_FIX_FLOAT_SWAP,
95 CDS,
96 CDS_INDEX,
97 FX_SPOT,
98 FX_FWD,
99 HAZARD_RATE,
100 RECOVERY_RATE,
101 SWAPTION,
102 CAPFLOOR,
103 FX_OPTION,
104 ZC_INFLATIONSWAP,
105 ZC_INFLATIONCAPFLOOR,
106 YY_INFLATIONSWAP,
107 YY_INFLATIONCAPFLOOR,
108 SEASONALITY,
109 EQUITY_SPOT,
110 EQUITY_FWD,
111 EQUITY_DIVIDEND,
112 EQUITY_OPTION,
113 BOND,
114 BOND_OPTION,
115 INDEX_CDS_OPTION,
116 COMMODITY_SPOT,
117 COMMODITY_FWD,
118 CORRELATION,

128 CREDIT_SPREAD,
129 CONV_CREDIT_SPREAD,

144 //! Constructor
145 MarketDatum(Real value, Date asofDate, const string& name, QuoteType quoteType, InstrumentType instrumentType)
146 : quote_(boost::make_shared<SimpleQuote>(value)), asofDate_(asofDate), name_(name),
147 instrumentType_(instrumentType), quoteType_(quoteType) {}
148
149 //! Default destructor
150 virtual ~MarketDatum() {}
151
152 //! Make a copy of the market datum
153 virtual boost::shared_ptr<MarketDatum> clone() {
154 return boost::make_shared<MarketDatum>(quote_->value(), asofDate_, name_, quoteType_, instrumentType_);
155 }
156
157 //! \name Inspectors
158 //@{
159 const string& name() const { return name_; }
160 const Handle<Quote>& quote() const { return quote_; }
161 Date asofDate() const { return asofDate_; }
162 InstrumentType instrumentType() const { return instrumentType_; }
163 QuoteType quoteType() const { return quoteType_; }
164 //@}
165protected:
166 Handle<Quote> quote_;
167 Date asofDate_;
168 string name_;
169 InstrumentType instrumentType_;
170 QuoteType quoteType_;
171
172private:
173 //! Serialization
174 friend class boost::serialization::access;
175 template <class Archive> void serialize(Archive& ar, const unsigned int version);
176};
