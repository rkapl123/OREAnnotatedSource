31void ForwardRateAgreement::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
32 const boost::shared_ptr<Market> market = engineFactory->market();
33
34 Date startDate = parseDate(startDate_);
35 Date endDate = parseDate(endDate_);
36 Position::Type positionType = parsePositionType(longShort_);
37
38 Handle<YieldTermStructure> discountTS = market->discountCurve(currency_);
39 Handle<QuantLib::IborIndex> index = market->iborIndex(index_);
40
41 boost::shared_ptr<QuantExt::ForwardRateAgreement> fra(
42 new QuantExt::ForwardRateAgreement(startDate, endDate, positionType, strike_, amount_, *index, discountTS));
43
44 Currency npvCcy = parseCurrency(currency_);
45 if (engineFactory->engineData()->hasProduct("ForwardRateAgreement")) {
46 // engine is optional for FRA instrument
47 boost::shared_ptr<EngineBuilder> builder = engineFactory->builder("ForwardRateAgreement");
48 boost::shared_ptr<FraEngineBuilderBase> fraBuilder =
49 boost::dynamic_pointer_cast<FraEngineBuilderBase>(builder);
50 QL_REQUIRE(fraBuilder, "No Builder found for ForwardRateAgreement " << id());
51 fra->setPricingEngine(fraBuilder->engine(npvCcy));
52 }
53
54 instrument_.reset(new VanillaInstrument(fra));
55 npvCurrency_ = currency_;
56 maturity_ = endDate;
57 instrument_->qlInstrument()->update();
58 notional_ = amount_;
59 notionalCurrency_ = currency_;
60 // the QL instrument reads the fixing in setupExpired() (bug?), so we don't add a payment date here to be safe
61 requiredFixings_.addFixingDate(fra->fixingDate(), index_);
62 // add required fixings for an Ibor fallback index
63 if (auto fallback = boost::dynamic_pointer_cast<QuantExt::FallbackIborIndex>(*index)) {
64 requiredFixings_.addFixingDates(fallback->onCoupon(fra->fixingDate())->fixingDates(),
65 engineFactory->iborFallbackConfig().fallbackData(index_).rfrIndex);
66 }
67
68 // ISDA taxonomy
69 additionalData_["isdaAssetClass"] = string("Interest Rate");
70 additionalData_["isdaBaseProduct"] = string("FRA");
71 additionalData_["isdaSubProduct"] = string("");
72 additionalData_["isdaTransaction"] = string("");
73}
