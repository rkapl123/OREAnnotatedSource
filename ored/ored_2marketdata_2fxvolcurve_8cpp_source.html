104 void FXVolCurve::buildSmileDeltaCurve(Date asof, FXVolatilityCurveSpec spec, const Loader& loader,
105  boost::shared_ptr<FXVolatilityCurveConfig> config, const FXLookup& fxSpots,
106  const map<string, boost::shared_ptr<YieldCurve>>& yieldCurves,
107  const Conventions& conventions) {
108  vector<Period> unsortedExp;
109 
110  vector<std::pair<Real, string>> putDeltas, callDeltas;
111  bool hasATM = false;
112 
113  for (auto const& delta : config->deltas()) {
114  DeltaString d(delta);
115  if (d.isAtm())
116  hasATM = true;
117  else if (d.isPut())
118  putDeltas.push_back(std::make_pair(d.delta(), delta));
119  else if (d.isCall())
120  callDeltas.push_back(std::make_pair(d.delta(), delta));
121  }
122 
123  Calendar cal = config->calendar();
124 
125  // sort puts 10P, 15P, 20P, ... and calls 45C, 40C, 35C, ... (notice put deltas have a negative sign)
126  auto comp = [](const std::pair<Real, string>& x, const std::pair<Real, string>& y) { return x.first > y.first; };
127  std::sort(putDeltas.begin(), putDeltas.end(), comp);
128  std::sort(callDeltas.begin(), callDeltas.end(), comp);
129 
130  vector<Date> dates;
131  Matrix blackVolMatrix;
132 
133  string base = "FX_OPTION/RATE_LNVOL/" + sourceCcy_ + "/" + targetCcy_ + "/";
134 
135  // build quote names
136  std::vector<std::string> deltaNames;
137  for (auto const& d : putDeltas) {
138  deltaNames.push_back(d.second);
139  }
140  if (hasATM) {
141  deltaNames.push_back("ATM");
142  }
143  for (auto const& d : callDeltas) {
144  deltaNames.push_back(d.second);
145  }
146 
147  if (expiriesWildcard_) {
148 
149  // we save relevant delta quotes to avoid looping twice
150  std::vector<boost::shared_ptr<MarketDatum>> data;
151  std::vector<std::string> expiriesStr;
152  // get list of possible expiries
153  for (auto& md : loader.loadQuotes(asof)) {
154  if (md->asofDate() == asof && md->instrumentType() == MarketDatum::InstrumentType::FX_OPTION) {
155  boost::shared_ptr<FXOptionQuote> q = boost::dynamic_pointer_cast<FXOptionQuote>(md);
156  Strike s = parseStrike(q->strike());
157  if (q->unitCcy() == spec.unitCcy() && q->ccy() == spec.ccy() &&
158  (s.type == Strike::Type::DeltaCall || s.type == Strike::Type::DeltaPut ||
159  s.type == Strike::Type::ATM)) {
160  vector<string> tokens;
161  boost::split(tokens, md->name(), boost::is_any_of("/"));
162  QL_REQUIRE(tokens.size() == 6, "6 tokens expected in " << md->name());
163  if ((*expiriesWildcard_).matches(tokens[4])) {
164  data.push_back(md);
165  auto it = std::find(expiries_.begin(), expiries_.end(), q->expiry());
166  if (it == expiries_.end()) {
167  expiries_.push_back(q->expiry());
168  expiriesStr.push_back(tokens[4]);
169  }
170  }
171  }
172  }
173  }
174  unsortedExp = expiries_;
175  std::sort(expiries_.begin(), expiries_.end());
176 
177  // we try to find all necessary quotes for each expiry
178  vector<Size> validExpiryIdx;
179  Matrix tmpMatrix(expiries_.size(), config->deltas().size());
180  for (Size i = 0; i < expiries_.size(); i++) {
181  Size idx = std::find(unsortedExp.begin(), unsortedExp.end(), expiries_[i]) - unsortedExp.begin();
182  string e = expiriesStr[idx];
183  for (Size j = 0; j < deltaNames.size(); ++j) {
184  string qs = base + e + "/" + deltaNames[j];
185  boost::shared_ptr<MarketDatum> md;
186  for (auto& m : data) {
187  if (m->name() == qs) {
188  md = m;
189  break;
190  }
191  }
192  boost::shared_ptr<FXOptionQuote> q = boost::dynamic_pointer_cast<FXOptionQuote>(md);
193  if (!q) {
194  DLOG("missing " << qs << ", expiry " << e << " will be excluded");
195  break;
196  }
197  tmpMatrix[i][j] = q->quote()->value();
198  // if we have found all the quotes then this is a valid expiry
199  if (j == deltaNames.size() - 1) {
200  dates.push_back(cal.advance(asof, expiries_[i]));
201  validExpiryIdx.push_back(i);
202  }
203  }
204  }
205 
206  QL_REQUIRE(validExpiryIdx.size() > 0, "no valid FxVol expiries found");
207  DLOG("found " << validExpiryIdx.size() << " valid expiries:");
208  for (auto& e : validExpiryIdx)
209  DLOG(expiries_[e]);
210  // we build a matrix with just the valid expiries
211  blackVolMatrix = Matrix(validExpiryIdx.size(), config->deltas().size());
212  for (Size i = 0; i < validExpiryIdx.size(); i++) {
213  for (Size j = 0; j < deltaNames.size(); ++j) {
214  blackVolMatrix[i][j] = tmpMatrix[validExpiryIdx[i]][j];
215  }
216  }
217 
218  } else {
219  expiries_ = parseVectorOfValues<Period>(expiriesNoDuplicates_, &parsePeriod);
220  unsortedExp = parseVectorOfValues<Period>(expiriesNoDuplicates_, &parsePeriod);
221  std::sort(expiries_.begin(), expiries_.end());
222 
223  blackVolMatrix = Matrix(expiries_.size(), config->deltas().size());
224  for (Size i = 0; i < expiries_.size(); i++) {
225  Size idx = std::find(unsortedExp.begin(), unsortedExp.end(), expiries_[i]) - unsortedExp.begin();
226  string e = expiriesNoDuplicates_[idx];
227  dates.push_back(cal.advance(asof, expiries_[i]));
228  for (Size j = 0; j < deltaNames.size(); ++j) {
229  string qs = base + e + "/" + deltaNames[j];
230  boost::shared_ptr<MarketDatum> md = loader.get(qs, asof);
231  boost::shared_ptr<FXOptionQuote> q = boost::dynamic_pointer_cast<FXOptionQuote>(md);
232  QL_REQUIRE(q, "quote not found, " << qs);
233  blackVolMatrix[i][j] = q->quote()->value();
234  }
235  }
236  }
237 
238  // daycounter used for interpolation in time.
239  // TODO: push into conventions or config
240  DayCounter dc = config->dayCounter();
241  std::vector<Real> putDeltasNum, callDeltasNum;
242  std::transform(putDeltas.begin(), putDeltas.end(), std::back_inserter(putDeltasNum),
243  [](const std::pair<Real, string>& x) { return x.first; });
244  std::transform(callDeltas.begin(), callDeltas.end(), std::back_inserter(callDeltasNum),
245  [](const std::pair<Real, string>& x) { return x.first; });
246  vol_ = boost::make_shared<QuantExt::BlackVolatilitySurfaceDelta>(
247  asof, dates, putDeltasNum, callDeltasNum, hasATM, blackVolMatrix, dc, cal, fxSpot_, domYts_, forYts_,
248  deltaType_, atmType_, boost::none, switchTenor_, longTermDeltaType_, longTermAtmType_);
249 
250  vol_->enableExtrapolation();
251 }

253 void FXVolCurve::buildSmileBfRrCurve(Date asof, FXVolatilityCurveSpec spec, const Loader& loader,
254  boost::shared_ptr<FXVolatilityCurveConfig> config, const FXLookup& fxSpots,
255  const map<string, boost::shared_ptr<YieldCurve>>& yieldCurves,
256  const Conventions& conventions) {
257 
258  // collect relevant market data and populate expiries (as per regex or configured list)
259 
260  std::set<Period> expiriesTmp;
261 
262  std::vector<boost::shared_ptr<FXOptionQuote>> data;
263  for (auto const& md : loader.loadQuotes(asof)) {
264  if (md->asofDate() == asof && md->instrumentType() == MarketDatum::InstrumentType::FX_OPTION) {
265  boost::shared_ptr<FXOptionQuote> q = boost::dynamic_pointer_cast<FXOptionQuote>(md);
266  QL_REQUIRE(q, "internal error: could not cast to FXOptionQuote");
267  Strike s = parseStrike(q->strike());
268  if (q->unitCcy() == spec.unitCcy() && q->ccy() == spec.ccy() &&
269  (s.type == Strike::Type::BF || s.type == Strike::Type::RR || s.type == Strike::Type::ATM)) {
270  vector<string> tokens;
271  boost::split(tokens, md->name(), boost::is_any_of("/"));
272  QL_REQUIRE(tokens.size() == 6, "6 tokens expected in " << md->name());
273  if (expiriesWildcard_ && (*expiriesWildcard_).matches(tokens[4]))
274  expiriesTmp.insert(q->expiry());
275  data.push_back(q);
276  }
277  }
278  }
279 
280  if (!expiriesWildcard_) {
281  auto tmp = parseVectorOfValues<Period>(expiriesNoDuplicates_, &parsePeriod);
282  expiriesTmp = std::set<Period>(tmp.begin(), tmp.end());
283  }
284 
285  // populate quotes
286 
287  std::vector<Size> smileDeltas = config->smileDelta();
288  std::sort(smileDeltas.begin(), smileDeltas.end());
289 
290  std::vector<std::vector<Real>> bfQuotesTmp(expiriesTmp.size(), std::vector<Real>(smileDeltas.size(), Null<Real>()));
291  std::vector<std::vector<Real>> rrQuotesTmp(expiriesTmp.size(), std::vector<Real>(smileDeltas.size(), Null<Real>()));
292  std::vector<Real> atmQuotesTmp(expiriesTmp.size(), Null<Real>());
293 
294  for (auto const& q : data) {
295  Size expiryIdx = std::distance(expiriesTmp.begin(), expiriesTmp.find(q->expiry()));
296  if (expiryIdx >= expiriesTmp.size())
297  continue;
298  Strike s = parseStrike(q->strike());
299  if (s.type == Strike::Type::ATM) {
300  atmQuotesTmp[expiryIdx] = q->quote()->value();
301  } else {
302  Size deltaIdx = std::distance(smileDeltas.begin(), std::find(smileDeltas.begin(), smileDeltas.end(),
303  static_cast<Size>(s.value + 0.5)));
304  if (deltaIdx >= smileDeltas.size())
305  continue;
306  if (s.type == Strike::Type::BF) {
307  bfQuotesTmp[expiryIdx][deltaIdx] = q->quote()->value();
308  } else if (s.type == Strike::Type::RR) {
309  rrQuotesTmp[expiryIdx][deltaIdx] = q->quote()->value();
310  }
311  }
312  }
313 
314  // identify the rows with complete data
315 
316  std::vector<bool> dataComplete(expiriesTmp.size(), true);
317 
318  for (Size i = 0; i < expiriesTmp.size(); ++i) {
319  for (Size j = 0; j < smileDeltas.size(); ++j) {
320  if (bfQuotesTmp[i][j] == Null<Real>() || rrQuotesTmp[i][j] == Null<Real>() ||
321  atmQuotesTmp[i] == Null<Real>())
322  dataComplete[i] = false;
323  }
324  }
325 
326  // if we have an explicitly configured expiry list, we require that the data is complete for all expiries
327 
328  if (!expiriesWildcard_) {
329  Size i = 0;
330  for (auto const& e : expiriesTmp) {
331  QL_REQUIRE(dataComplete[i++], "BFRR FX vol surface: incomplete data for expiry " << e);
332  }
333  }
334 
335  // build the final quotes for the expiries that have complete data
336 
337  Size i = 0;
338  for (auto const& e : expiriesTmp) {
339  if (dataComplete[i++]) {
340  expiries_.push_back(e);
341  TLOG("adding expiry " << e << " with complete data");
342  } else {
343  TLOG("removing expiry " << e << ", because data is not complete");
344  }
345  }
346 
347  std::vector<std::vector<Real>> bfQuotes(expiries_.size(), std::vector<Real>(smileDeltas.size()));
348  std::vector<std::vector<Real>> rrQuotes(expiries_.size(), std::vector<Real>(smileDeltas.size()));
349  std::vector<Real> atmQuotes(expiries_.size());
350 
351  Size row = 0;
352  for (Size i = 0; i < expiriesTmp.size(); ++i) {
353  if (!dataComplete[i])
354  continue;
355  atmQuotes[row] = atmQuotesTmp[i];
356  for (Size j = 0; j < smileDeltas.size(); ++j) {
357  bfQuotes[row][j] = bfQuotesTmp[i][j];
358  rrQuotes[row][j] = rrQuotesTmp[i][j];
359  }
360  ++row;
361  }
362 
363  // build BFRR surface
364 
365  DLOG("build BFRR fx vol surface with " << expiries_.size() << " expiries and " << smileDeltas.size()
366  << " delta(s)");
367 
368  QuantExt::BlackVolatilitySurfaceBFRR::SmileInterpolation interp;
369  if (config->smileInterpolation() == FXVolatilityCurveConfig::SmileInterpolation::Linear)
370  interp = QuantExt::BlackVolatilitySurfaceBFRR::SmileInterpolation::Linear;
371  else if (config->smileInterpolation() == FXVolatilityCurveConfig::SmileInterpolation::Cubic)
372  interp = QuantExt::BlackVolatilitySurfaceBFRR::SmileInterpolation::Cubic;
373  else {
374  QL_FAIL("BFRR FX vol surface: invalid interpolation, expected Linear, Cubic");
375  }
376 
377  std::vector<Date> dates;
378  std::transform(expiries_.begin(), expiries_.end(), std::back_inserter(dates),
379  [&asof, &config](const Period& p) { return config->calendar().advance(asof, p); });
380 
381  std::vector<Real> smileDeltasScaled;
382  std::transform(smileDeltas.begin(), smileDeltas.end(), std::back_inserter(smileDeltasScaled),
383  [](Size d) { return static_cast<Real>(d) / 100.0; });
384 
385  vol_ = boost::make_shared<QuantExt::BlackVolatilitySurfaceBFRR>(
386  asof, dates, smileDeltasScaled, bfQuotes, rrQuotes, atmQuotes, config->dayCounter(), config->calendar(),
387  fxSpot_, spotDays_, spotCalendar_, domYts_, forYts_, deltaType_, atmType_, switchTenor_, longTermDeltaType_,
388  longTermAtmType_, riskReversalInFavorOf_, butterflyIsBrokerStyle_, interp);
389 
390  vol_->enableExtrapolation();
391 }

393 void FXVolCurve::buildVannaVolgaOrATMCurve(Date asof, FXVolatilityCurveSpec spec, const Loader& loader,
394  boost::shared_ptr<FXVolatilityCurveConfig> config, const FXLookup& fxSpots,
395  const map<string, boost::shared_ptr<YieldCurve>>& yieldCurves,
396  const Conventions& conventions) {
397 
398  bool isATM = config->dimension() == FXVolatilityCurveConfig::Dimension::ATM;
399  Natural smileDelta = 0;
400  std::string deltaRr;
401  std::string deltaBf;
402  if (!isATM) {
403  QL_REQUIRE(config->smileDelta().size() == 1,
404  "Exactly one SmileDelta required for VannaVolga Curve (got " << config->smileDelta().size() << ")");
405  smileDelta = config->smileDelta().front();
406  deltaRr = to_string(smileDelta) + "RR";
407  deltaBf = to_string(smileDelta) + "BF";
408  }
409  // We loop over all market data, looking for quotes that match the configuration
410  // every time we find a matching expiry we remove it from the list
411  // we replicate this for all 3 types of quotes were applicable.
412  Size n = isATM ? 1 : 3; // [0] = ATM, [1] = RR, [2] = BF
413  vector<vector<boost::shared_ptr<FXOptionQuote>>> quotes(n);
414 
415  QL_REQUIRE(!expiriesWildcard_ || isATM, "wildcards only supported for ATM, Delta, BFRR FxVol Curves");
416 
417  vector<Period> cExpiries;
418  vector<vector<Period>> expiries;
419  // Create the regular expression
420  if (!expiriesWildcard_) {
421  cExpiries = parseVectorOfValues<Period>(expiriesNoDuplicates_, &parsePeriod);
422  expiries = vector<vector<Period>>(n, cExpiries);
423  }
424 
425  // Load the relevant quotes
426  for (auto& md : loader.loadQuotes(asof)) {
427  // skip irrelevant data
428  if (md->asofDate() == asof && md->instrumentType() == MarketDatum::InstrumentType::FX_OPTION) {
429 
430  boost::shared_ptr<FXOptionQuote> q = boost::dynamic_pointer_cast<FXOptionQuote>(md);
431 
432  if (q->unitCcy() == spec.unitCcy() && q->ccy() == spec.ccy()) {
433 
434  Size idx = 999999;
435  if (q->strike() == "ATM")
436  idx = 0;
437  else if (!isATM && q->strike() == deltaRr)
438  idx = 1;
439  else if (!isATM && q->strike() == deltaBf)
440  idx = 2;
441 
442  // silently skip unknown strike strings
443  if ((isATM && idx == 0) || (!isATM && idx <= 2)) {
444  if (expiriesWildcard_) {
445  vector<string> tokens;
446  boost::split(tokens, md->name(), boost::is_any_of("/"));
447  QL_REQUIRE(tokens.size() == 6, "6 tokens expected in " << md->name());
448  if ((*expiriesWildcard_).matches(tokens[4])) {
449  quotes[idx].push_back(q);
450  }
451  } else {
452  auto it = std::find(expiries[idx].begin(), expiries[idx].end(), q->expiry());
453  if (it != expiries[idx].end()) {
454  // we have a hit
455  quotes[idx].push_back(q);
456  // remove it from the list
457  expiries[idx].erase(it);
458  }
459 
460  // check if we are done
461  // for ATM we just check expiries[0], otherwise we check all 3
462  if (expiries[0].empty() && (isATM || (expiries[1].empty() && expiries[2].empty())))
463  break;
464  }
465  }
466  }
467  }
468  }
469 
470  // Check ATM first
471  // Check that we have all the expiries we need
472  LOG("FXVolCurve: read " << quotes[0].size() << " ATM vols");
473  if (!expiriesWildcard_) {
474  QL_REQUIRE(expiries[0].size() == 0,
475  "No ATM quote found for spec " << spec << " with expiry " << expiries[0].front());
476  }
477 
478  QL_REQUIRE(quotes[0].size() > 0, "No ATM quotes found for spec " << spec);
479  // No check the rest
480  if (!isATM) {
481  LOG("FXVolCurve: read " << quotes[1].size() << " RR and " << quotes[2].size() << " BF quotes");
482  QL_REQUIRE(expiries[1].size() == 0,
483  "No RR quote found for spec " << spec << " with expiry " << expiries[1].front());
484  QL_REQUIRE(expiries[2].size() == 0,
485  "No BF quote found for spec " << spec << " with expiry " << expiries[2].front());
486  }
487 
488  // sort all quotes
489  for (Size i = 0; i < n; i++) {
490  std::sort(quotes[i].begin(), quotes[i].end(),
491  [](const boost::shared_ptr<FXOptionQuote>& a, const boost::shared_ptr<FXOptionQuote>& b) -> bool {
492  return a->expiry() < b->expiry();
493  });
494  }
495 
496  // daycounter used for interpolation in time.
497  // TODO: push into conventions or config
498  DayCounter dc = config->dayCounter();
499  Calendar cal = config->calendar();
500 
501  // build vol curve
502  if (isATM && quotes[0].size() == 1) {
503  vol_ = boost::shared_ptr<BlackVolTermStructure>(
504  new BlackConstantVol(asof, config->calendar(), quotes[0].front()->quote()->value(), dc));
505  expiries_ = {quotes[0].front()->expiry()};
506  } else {
507 
508  Size numExpiries = quotes[0].size();
509  vector<Date> dates(numExpiries);
510  vector<vector<Volatility>> vols(n, vector<Volatility>(numExpiries)); // same as above: [0] = ATM, etc.
511 
512  for (Size i = 0; i < numExpiries; i++) {
513  dates[i] = cal.advance(asof, quotes[0][i]->expiry());
514  expiries_.push_back(quotes[0][i]->expiry());
515  DLOG("Spec Tenor Vol Variance");
516  for (Size idx = 0; idx < n; idx++) {
517  vols[idx][i] = quotes[idx][i]->quote()->value();
518  Real variance = vols[idx][i] * vols[idx][i] * (dates[i] - asof) / 365.0; // approximate variance
519  DLOG(spec << " " << quotes[0][i]->expiry() << " " << vols[idx][i] << " " << variance);
520  }
521  }
522 
523  if (isATM) {
524  // ATM
525  // Set forceMonotoneVariance to false - allowing decreasing variance
526  vol_ = boost::shared_ptr<BlackVolTermStructure>(new BlackVarianceCurve(asof, dates, vols[0], dc, false));
527  } else {
528  // Smile
529  bool vvFirstApprox = false; // default to VannaVolga second approximation
530  if (config->smileInterpolation() == FXVolatilityCurveConfig::SmileInterpolation::VannaVolga1) {
531  vvFirstApprox = true;
532  }
533 
534  vol_ = boost::make_shared<QuantExt::FxBlackVannaVolgaVolatilitySurface>(
535  asof, dates, vols[0], vols[1], vols[2], dc, cal, fxSpot_, domYts_, forYts_, false, vvFirstApprox,
536  atmType_, deltaType_, smileDelta / 100.0, switchTenor_, longTermAtmType_, longTermDeltaType_);
537  }
538  }
539  vol_->enableExtrapolation();
540 }

542 Handle<QuantExt::CorrelationTermStructure>
543 getCorrelationCurve(const std::string& index1, const std::string& index2,
544  const map<string, boost::shared_ptr<CorrelationCurve>>& correlationCurves) {
545  // straight pair
546  auto tmpCorr = correlationCurves.find("Correlation/" + index1 + "&" + index2);
547  if (tmpCorr != correlationCurves.end()) {
548  return Handle<QuantExt::CorrelationTermStructure>(tmpCorr->second->corrTermStructure());
549  }
550  // inverse pair
551  tmpCorr = correlationCurves.find("Correlation/" + index2 + "&" + index1);
552  if (tmpCorr != correlationCurves.end()) {
553  return Handle<QuantExt::CorrelationTermStructure>(tmpCorr->second->corrTermStructure());
554  }
555  // inverse fx index1
556  tmpCorr = correlationCurves.find("Correlation/" + inverseFxIndex(index1) + "&" + index2);
557  if (tmpCorr != correlationCurves.end()) {
558  Handle<QuantExt::CorrelationTermStructure> h =
559  Handle<QuantExt::CorrelationTermStructure>(tmpCorr->second->corrTermStructure());
560  return Handle<QuantExt::CorrelationTermStructure>(
561  boost::make_shared<QuantExt::NegativeCorrelationTermStructure>(h));
562  }
563  tmpCorr = correlationCurves.find("Correlation/" + index2 + "&" + inverseFxIndex(index1));
564  if (tmpCorr != correlationCurves.end()) {
565  Handle<QuantExt::CorrelationTermStructure> h =
566  Handle<QuantExt::CorrelationTermStructure>(tmpCorr->second->corrTermStructure());
567  return Handle<QuantExt::CorrelationTermStructure>(
568  boost::make_shared<QuantExt::NegativeCorrelationTermStructure>(h));
569  }
570  // inverse fx index2
571  tmpCorr = correlationCurves.find("Correlation/" + index1 + "&" + inverseFxIndex(index2));
572  if (tmpCorr != correlationCurves.end()) {
573  Handle<QuantExt::CorrelationTermStructure> h =
574  Handle<QuantExt::CorrelationTermStructure>(tmpCorr->second->corrTermStructure());
575  return Handle<QuantExt::CorrelationTermStructure>(
576  boost::make_shared<QuantExt::NegativeCorrelationTermStructure>(h));
577  }
578  tmpCorr = correlationCurves.find("Correlation/" + inverseFxIndex(index2) + "&" + index1);
579  if (tmpCorr != correlationCurves.end()) {
580  Handle<QuantExt::CorrelationTermStructure> h =
581  Handle<QuantExt::CorrelationTermStructure>(tmpCorr->second->corrTermStructure());
582  return Handle<QuantExt::CorrelationTermStructure>(
583  boost::make_shared<QuantExt::NegativeCorrelationTermStructure>(h));
584  }
585  // both fx indices inverted
586  tmpCorr = correlationCurves.find("Correlation/" + inverseFxIndex(index1) + "&" + inverseFxIndex(index2));
587  if (tmpCorr != correlationCurves.end()) {
588  return Handle<QuantExt::CorrelationTermStructure>(tmpCorr->second->corrTermStructure());
589  }
590  tmpCorr = correlationCurves.find("Correlation/" + inverseFxIndex(index2) + "&" + inverseFxIndex(index1));
591  if (tmpCorr != correlationCurves.end()) {
592  return Handle<QuantExt::CorrelationTermStructure>(tmpCorr->second->corrTermStructure());
593  }
594 
595  QL_FAIL("no correlation curve found for " << index1 << ":" << index2);
596 }

598 void FXVolCurve::buildATMTriangulated(Date asof, FXVolatilityCurveSpec spec, const Loader& loader,
599  boost::shared_ptr<FXVolatilityCurveConfig> config, const FXLookup& fxSpots,
600  const map<string, boost::shared_ptr<YieldCurve>>& yieldCurves,
601  const map<string, boost::shared_ptr<FXVolCurve>>& fxVols,
602  const map<string, boost::shared_ptr<CorrelationCurve>>& correlationCurves,
603  const Conventions& conventions) {
604 
605  DLOG("Triangulating FxVol curve " << config->curveID() << " from baseVols " << config->baseVolatility1() << ":"
606  << config->baseVolatility2());
607 
608  std::string baseCcy;
609  QL_REQUIRE(config->baseVolatility1().size() == 6, "invalid ccy pair length for baseVolatility1");
610  auto forBase1 = config->baseVolatility1().substr(0, 3);
611  auto domBase1 = config->baseVolatility1().substr(3);
612  std::string spec1 = "FXVolatility/" + forBase1 + "/" + domBase1 + "/" + config->baseVolatility1();
613 
614  bool base1Inverted = false;
615  if (forBase1 != sourceCcy_ && forBase1 != targetCcy_) {
616  // we invert the pair
617  base1Inverted = true;
618  std::string tmp = forBase1;
619  forBase1 = targetCcy_;
620  domBase1 = tmp;
621 
622  QL_REQUIRE(forBase1 == sourceCcy_ || forBase1 == targetCcy_,
623  "FxVol: mismatch in the baseVolatility1 " << config->baseVolatility1() << " and Target Pair "
624  << sourceCcy_ << targetCcy_);
625  }
626  baseCcy = domBase1;
627 
628  QL_REQUIRE(config->baseVolatility2().size() == 6, "invalid ccy pair length for baseVolatility2");
629  auto forBase2 = config->baseVolatility2().substr(0, 3);
630  auto domBase2 = config->baseVolatility2().substr(3);
631  std::string spec2 = "FXVolatility/" + forBase2 + "/" + domBase2 + "/" + config->baseVolatility2();
632  bool base2Inverted = false;
633 
634  QL_REQUIRE(forBase2 == baseCcy || domBase2 == baseCcy,
635  "baseVolatility2 must share a ccy code with the baseVolatility1");
636 
637  if (forBase2 != sourceCcy_ && forBase2 != targetCcy_) {
638  // we invert the pair
639  std::string tmp = forBase2;
640  forBase2 = targetCcy_;
641  domBase2 = tmp;
642  base2Inverted = true;
643  }
644 
645  auto tmp = fxVols.find(spec1);
646  QL_REQUIRE(tmp != fxVols.end(), "fx vol not found for " << config->baseVolatility1());
647  Handle<BlackVolTermStructure> forBaseVol;
648  if (base1Inverted) {
649  auto h = Handle<BlackVolTermStructure>(tmp->second->volTermStructure());
650  if (!h.empty())
651  forBaseVol = Handle<BlackVolTermStructure>(boost::make_shared<QuantExt::BlackInvertedVolTermStructure>(h));
652  } else {
653  forBaseVol = Handle<BlackVolTermStructure>(tmp->second->volTermStructure());
654  }
655  forBaseVol->enableExtrapolation();
656 
657  tmp = fxVols.find(spec2);
658  QL_REQUIRE(tmp != fxVols.end(), "fx vol not found for " << config->baseVolatility2());
659  Handle<BlackVolTermStructure> domBaseVol;
660  if (base2Inverted) {
661  auto h = Handle<BlackVolTermStructure>(tmp->second->volTermStructure());
662  if (!h.empty())
663  domBaseVol = Handle<BlackVolTermStructure>(boost::make_shared<QuantExt::BlackInvertedVolTermStructure>(h));
664 
665  } else {
666  domBaseVol = Handle<BlackVolTermStructure>(tmp->second->volTermStructure());
667  }
668  domBaseVol->enableExtrapolation();
669 
670  string forIndex = "FX-" + config->fxIndexTag() + "-" + sourceCcy_ + "-" + baseCcy;
671  string domIndex = "FX-" + config->fxIndexTag() + "-" + targetCcy_ + "-" + baseCcy;
672 
673  Handle<QuantExt::CorrelationTermStructure> rho = getCorrelationCurve(forIndex, domIndex, correlationCurves);
674 
675  vol_ = boost::make_shared<QuantExt::BlackTriangulationATMVolTermStructure>(forBaseVol, domBaseVol, rho);
676  vol_->enableExtrapolation();
677 }

679 void FXVolCurve::init(Date asof, FXVolatilityCurveSpec spec, const Loader& loader,
680  const CurveConfigurations& curveConfigs, const FXLookup& fxSpots,
681  const map<string, boost::shared_ptr<YieldCurve>>& yieldCurves,
682  const std::map<string, boost::shared_ptr<FXVolCurve>>& fxVols,
683  const map<string, boost::shared_ptr<CorrelationCurve>>& correlationCurves,
684  const Conventions& conventions) {
685  try {
686 
687  const boost::shared_ptr<FXVolatilityCurveConfig>& config = curveConfigs.fxVolCurveConfig(spec.curveConfigID());
688 
689  QL_REQUIRE(config->dimension() == FXVolatilityCurveConfig::Dimension::ATM ||
690  config->dimension() == FXVolatilityCurveConfig::Dimension::ATMTriangulated ||
691  config->dimension() == FXVolatilityCurveConfig::Dimension::SmileVannaVolga ||
692  config->dimension() == FXVolatilityCurveConfig::Dimension::SmileDelta ||
693  config->dimension() == FXVolatilityCurveConfig::Dimension::SmileBFRR,
694  "Unknown FX curve building dimension");
695 
696  expiriesWildcard_ = getUniqueWildcard(config->expiries());
697 
698  // remove expiries that would lead to duplicate expiry dates (keep the later expiry in this case)
699 
700  if (!expiriesWildcard_) {
701  std::vector<std::tuple<std::string, Period, Date>> tmp;
702  for (auto const& e : config->expiries()) {
703  auto p = parsePeriod(e);
704  tmp.push_back(std::make_tuple(e, p, config->calendar().advance(asof, p)));
705  }
706  std::sort(tmp.begin(), tmp.end(),
707  [](const std::tuple<std::string, Period, Date>& a,
708  const std::tuple<std::string, Period, Date>& b) -> bool {
709  if (std::get<2>(a) == std::get<2>(b)) {
710  try {
711  // equal dates => compare periods (might throw!)
712  return std::get<1>(a) < std::get<1>(b);
713  } catch (...) {
714  // period comparison not possible => compare the strings
715  return std::get<0>(a) < std::get<0>(b);
716  }
717  }
718  // different dates => compare them
719  return std::get<2>(a) < std::get<2>(b);
720  });
721  Date lastDate = Date::maxDate();
722  for (Size i = tmp.size(); i > 0; --i) {
723  if (std::get<2>(tmp[i - 1]) == lastDate)
724  continue;
725  expiriesNoDuplicates_.insert(expiriesNoDuplicates_.begin(), std::get<0>(tmp[i - 1]));
726  lastDate = std::get<2>(tmp[i - 1]);
727  }
728 
729  DLOG("expiries in configuration:")
730  for (auto const& e : config->expiries()) {
731  DLOG(e);
732  }
733 
734  DLOG("expiries after removing duplicate expiry dates and sorting:");
735  for (auto const& e : expiriesNoDuplicates_) {
736  DLOG(e);
737  }
738  } else {
739  DLOG("expiry wildcard is used: " << (*expiriesWildcard_).regex());
740  }
741 
742  QL_REQUIRE(config->dimension() == FXVolatilityCurveConfig::Dimension::ATMTriangulated || expiriesWildcard_ ||
743  !expiriesNoDuplicates_.empty(),
744  "no expiries after removing duplicate expiry dates");
745 
746  //
747 
748  std::vector<std::string> tokens;
749  boost::split(tokens, config->fxSpotID(), boost::is_any_of("/"));
750  QL_REQUIRE(tokens.size() == 3, "Expected 3 tokens in fx spot id '" << config->fxSpotID() << "'");
751  sourceCcy_ = tokens[1];
752  targetCcy_ = tokens[2];
753 
754  atmType_ = DeltaVolQuote::AtmType::AtmDeltaNeutral;
755  deltaType_ = DeltaVolQuote::DeltaType::Spot;
756  switchTenor_ = 2 * Years;
757  longTermAtmType_ = DeltaVolQuote::AtmType::AtmDeltaNeutral;
758  longTermDeltaType_ = DeltaVolQuote::DeltaType::Fwd;
759  riskReversalInFavorOf_ = Option::Call;
760  butterflyIsBrokerStyle_ = true;
761  spotDays_ = 2;
762  string calTmp = sourceCcy_ + "," + targetCcy_;
763  spotCalendar_ = parseCalendar(calTmp);
764 
765  if (config->conventionsID() != "") {
766  auto fxOptConv = boost::dynamic_pointer_cast<FxOptionConvention>(conventions.get(config->conventionsID()));
767  QL_REQUIRE(fxOptConv,
768  "unable to cast convention '" << config->conventionsID() << "' into FxOptionConvention");
769  boost::shared_ptr<FXConvention> fxConv;
770  if (!fxOptConv->fxConventionID().empty()) {
771  fxConv = boost::dynamic_pointer_cast<FXConvention>(conventions.get(fxOptConv->fxConventionID()));
772  QL_REQUIRE(fxConv, "unable to cast convention '" << fxOptConv->fxConventionID()
773  << "', from FxOptionConvention '"
774  << config->conventionsID() << "' into FxConvention");
775  }
776  atmType_ = fxOptConv->atmType();
777  deltaType_ = fxOptConv->deltaType();
778  longTermAtmType_ = fxOptConv->longTermAtmType();
779  longTermDeltaType_ = fxOptConv->longTermDeltaType();
780  switchTenor_ = fxOptConv->switchTenor();
781  riskReversalInFavorOf_ = fxOptConv->riskReversalInFavorOf();
782  butterflyIsBrokerStyle_ = fxOptConv->butterflyIsBrokerStyle();
783  if (fxConv) {
784  spotDays_ = fxConv->spotDays();
785  spotCalendar_ = fxConv->advanceCalendar();
786  }
787  } else {
788  WLOG("no fx option conventions given in fxvol curve condig for " << spec.curveConfigID()
789  << ", assuming defaults");
790  }
791 
792  fxSpot_ = fxSpots.fxPairLookup(config->fxSpotID());
793  if (!config->fxDomesticYieldCurveID().empty())
794  domYts_ = getHandle<YieldTermStructure>(config->fxDomesticYieldCurveID(), yieldCurves);
795  if (!config->fxForeignYieldCurveID().empty())
796  forYts_ = getHandle<YieldTermStructure>(config->fxForeignYieldCurveID(), yieldCurves);
797 
798  if (config->dimension() == FXVolatilityCurveConfig::Dimension::SmileDelta) {
799  buildSmileDeltaCurve(asof, spec, loader, config, fxSpots, yieldCurves, conventions);
800  } else if (config->dimension() == FXVolatilityCurveConfig::Dimension::SmileBFRR) {
801  buildSmileBfRrCurve(asof, spec, loader, config, fxSpots, yieldCurves, conventions);
802  } else if (config->dimension() == FXVolatilityCurveConfig::Dimension::ATMTriangulated) {
803  buildATMTriangulated(asof, spec, loader, config, fxSpots, yieldCurves, fxVols, correlationCurves,
804  conventions);
805  } else {
806  buildVannaVolgaOrATMCurve(asof, spec, loader, config, fxSpots, yieldCurves, conventions);
807  }
808 
809  // build calibration info
810 
811  DLOG("Building calibration info for fx vol surface");
812 
813  if (domYts_.empty() || forYts_.empty()) {
814  WLOG("no domestic / foreign yield curves given in fx vol curve config for "
815  << spec.curveConfigID() << ", skip building calibration info");
816  return;
817  }
818 
819  ReportConfig rc = effectiveReportConfig(curveConfigs.reportConfigFxVols(), config->reportConfig());
820 
821  bool reportOnDeltaGrid = *rc.reportOnDeltaGrid();
822  bool reportOnMoneynessGrid = *rc.reportOnMoneynessGrid();
823  std::vector<Real> moneyness = *rc.moneyness();
824  std::vector<std::string> deltas = *rc.deltas();
825  std::vector<Period> expiries = *rc.expiries();
826 
827  calibrationInfo_ = boost::make_shared<FxEqVolCalibrationInfo>();
828 
829  calibrationInfo_->dayCounter = config->dayCounter().empty() ? "na" : config->dayCounter().name();
830  calibrationInfo_->calendar = config->calendar().empty() ? "na" : config->calendar().name();
831  calibrationInfo_->atmType = ore::data::to_string(atmType_);
832  calibrationInfo_->deltaType = ore::data::to_string(deltaType_);
833  calibrationInfo_->longTermAtmType = ore::data::to_string(longTermAtmType_);
834  calibrationInfo_->longTermDeltaType = ore::data::to_string(longTermDeltaType_);
835  calibrationInfo_->switchTenor = ore::data::to_string(switchTenor_);
836  calibrationInfo_->riskReversalInFavorOf = riskReversalInFavorOf_ == Option::Call ? "Call" : "Put";
837  calibrationInfo_->butterflyStyle = butterflyIsBrokerStyle_ ? "Broker" : "Smile";
838 
839  std::vector<Real> times, forwards, domDisc, forDisc;
840  Date settl = spotCalendar_.advance(asof, spotDays_ * Days);
841  for (auto const& p : expiries) {
842  Date d = vol_->optionDateFromTenor(p);
843  Date settlFwd = spotCalendar_.advance(d, spotDays_ * Days);
844  calibrationInfo_->expiryDates.push_back(d);
845  times.push_back(vol_->dayCounter().empty() ? Actual365Fixed().yearFraction(asof, d)
846  : vol_->timeFromReference(d));
847  domDisc.push_back(domYts_->discount(settlFwd) / domYts_->discount(settl));
848  forDisc.push_back(forYts_->discount(settlFwd) / forYts_->discount(settl));
849  forwards.push_back(fxSpot_->value() / domDisc.back() * forDisc.back());
850  }
851 
852  calibrationInfo_->times = times;
853  calibrationInfo_->forwards = forwards;
854 
855  Date switchExpiry = vol_->calendar().empty() ? asof + switchTenor_ : vol_->optionDateFromTenor(switchTenor_);
856  Real switchTime = vol_->dayCounter().empty() ? Actual365Fixed().yearFraction(asof, switchExpiry)
857  : vol_->timeFromReference(switchExpiry);
858  if (switchTenor_ == 0 * Days)
859  switchTime = QL_MAX_REAL;
860 
861  std::vector<std::vector<Real>> callPricesDelta(times.size(), std::vector<Real>(deltas.size(), 0.0));
862  std::vector<std::vector<Real>> callPricesMoneyness(times.size(), std::vector<Real>(moneyness.size(), 0.0));
863 
864  calibrationInfo_->isArbitrageFree = true;
865 
866  if (reportOnDeltaGrid) {
867  calibrationInfo_->deltas = deltas;
868  calibrationInfo_->deltaGridStrikes =
869  std::vector<std::vector<Real>>(times.size(), std::vector<Real>(deltas.size(), 0.0));
870  calibrationInfo_->deltaGridProb =
871  std::vector<std::vector<Real>>(times.size(), std::vector<Real>(deltas.size(), 0.0));
872  calibrationInfo_->deltaGridImpliedVolatility =
873  std::vector<std::vector<Real>>(times.size(), std::vector<Real>(deltas.size(), 0.0));
874  calibrationInfo_->deltaGridCallSpreadArbitrage =
875  std::vector<std::vector<bool>>(times.size(), std::vector<bool>(deltas.size(), true));
876  calibrationInfo_->deltaGridButterflyArbitrage =
877  std::vector<std::vector<bool>>(times.size(), std::vector<bool>(deltas.size(), true));
878  DeltaVolQuote::DeltaType dt;
879  DeltaVolQuote::AtmType at;
880  TLOG("Delta surface arbitrage analysis result (no calendar spread arbitrage included):");
881  Real maxTime = QL_MAX_REAL;
882  if (!expiries_.empty())
883  maxTime = vol_->timeFromReference(vol_->optionDateFromTenor(expiries_.back()));
884  for (Size i = 0; i < times.size(); ++i) {
885  Real t = times[i];
886  if (t <= switchTime || close_enough(t, switchTime)) {
887  at = atmType_;
888  dt = deltaType_;
889  } else {
890  at = longTermAtmType_;
891  dt = longTermDeltaType_;
892  }
893  // for times after the last quoted expiry we use artificial conventions to avoid problems with strike
894  // from delta conversions: we keep the pa feature, but use fwd delta always and ATM DNS
895  if (t > maxTime) {
896  at = DeltaVolQuote::AtmDeltaNeutral;
897  dt = (deltaType_ == DeltaVolQuote::Spot || deltaType_ == DeltaVolQuote::Fwd) ? DeltaVolQuote::Fwd
898  : DeltaVolQuote::PaFwd;
899  }
900  bool validSlice = true;
901  for (Size j = 0; j < deltas.size(); ++j) {
902  DeltaString d(deltas[j]);
903  try {
904  Real strike;
905  if (d.isAtm()) {
906  strike = QuantExt::getAtmStrike(dt, at, fxSpot_->value(), domDisc[i], forDisc[i], vol_, t);
907  } else if (d.isCall()) {
908  strike = QuantExt::getStrikeFromDelta(Option::Call, d.delta(), dt, fxSpot_->value(),
909  domDisc[i], forDisc[i], vol_, t);
910  } else {
911  strike = QuantExt::getStrikeFromDelta(Option::Put, d.delta(), dt, fxSpot_->value(),
912  domDisc[i], forDisc[i], vol_, t);
913  }
914  Real stddev = std::sqrt(vol_->blackVariance(t, strike));
915  callPricesDelta[i][j] = blackFormula(Option::Call, strike, forwards[i], stddev);
916  calibrationInfo_->deltaGridStrikes[i][j] = strike;
917  calibrationInfo_->deltaGridImpliedVolatility[i][j] = stddev / std::sqrt(t);
918  } catch (const std::exception& e) {
919  validSlice = false;
920  TLOG("error for time " << t << " delta " << deltas[j] << ": " << e.what());
921  }
922  }
923  if (validSlice) {
924  try {
925  QuantExt::CarrMadanMarginalProbability cm(calibrationInfo_->deltaGridStrikes[i], forwards[i],
926  callPricesDelta[i]);
927  calibrationInfo_->deltaGridCallSpreadArbitrage[i] = cm.callSpreadArbitrage();
928  calibrationInfo_->deltaGridButterflyArbitrage[i] = cm.butterflyArbitrage();
929  if (!cm.arbitrageFree())
930  calibrationInfo_->isArbitrageFree = false;
931  calibrationInfo_->deltaGridProb[i] = cm.density();
932  TLOGGERSTREAM << arbitrageAsString(cm);
933  } catch (const std::exception& e) {
934  TLOG("error for time " << t << ": " << e.what());
935  calibrationInfo_->isArbitrageFree = false;
936  TLOGGERSTREAM << "..(invalid slice)..";
937  }
938  } else {
939  calibrationInfo_->isArbitrageFree = false;
940  TLOGGERSTREAM << "..(invalid slice)..";
941  }
942  }
943  TLOG("Delta surface arbitrage analysis completed.");
944  }
945 
946  if (reportOnMoneynessGrid) {
947  calibrationInfo_->moneyness = moneyness;
948  calibrationInfo_->moneynessGridStrikes =
949  std::vector<std::vector<Real>>(times.size(), std::vector<Real>(moneyness.size(), 0.0));
950  calibrationInfo_->moneynessGridProb =
951  std::vector<std::vector<Real>>(times.size(), std::vector<Real>(moneyness.size(), 0.0));
952  calibrationInfo_->moneynessGridImpliedVolatility =
953  std::vector<std::vector<Real>>(times.size(), std::vector<Real>(moneyness.size(), 0.0));
954  calibrationInfo_->moneynessGridCallSpreadArbitrage =
955  std::vector<std::vector<bool>>(times.size(), std::vector<bool>(moneyness.size(), true));
956  calibrationInfo_->moneynessGridButterflyArbitrage =
957  std::vector<std::vector<bool>>(times.size(), std::vector<bool>(moneyness.size(), true));
958  calibrationInfo_->moneynessGridCalendarArbitrage =
959  std::vector<std::vector<bool>>(times.size(), std::vector<bool>(moneyness.size(), true));
960  for (Size i = 0; i < times.size(); ++i) {
961  Real t = times[i];
962  for (Size j = 0; j < moneyness.size(); ++j) {
963  try {
964  Real strike = moneyness[j] * forwards[i];
965  calibrationInfo_->moneynessGridStrikes[i][j] = strike;
966  Real stddev = std::sqrt(vol_->blackVariance(t, strike));
967  callPricesMoneyness[i][j] = blackFormula(Option::Call, strike, forwards[i], stddev);
968  calibrationInfo_->moneynessGridImpliedVolatility[i][j] = stddev / std::sqrt(t);
969  } catch (const std::exception& e) {
970  TLOG("error for time " << t << " moneyness " << moneyness[j] << ": " << e.what());
971  }
972  }
973  }
974  if (!times.empty() && !moneyness.empty()) {
975  try {
976  QuantExt::CarrMadanSurface cm(times, moneyness, fxSpot_->value(), forwards, callPricesMoneyness);
977  for (Size i = 0; i < times.size(); ++i) {
978  calibrationInfo_->moneynessGridProb[i] = cm.timeSlices()[i].density();
979  }
