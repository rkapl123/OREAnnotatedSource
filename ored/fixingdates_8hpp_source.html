72 /*! Class holding the information on the fixings required to price a trade (or a portfolio of trades). */
73 class RequiredFixings {
74 public:
75  /*! Gives back the dates for which fixings will be required to price the trade assuming a given \p settlementDate.
76  If the \p settlementDate is not provided or is set equal to \c QuantLib::Date(), the settlement date in the
77  implementation is assumed to be the \c Settings::instance().evaluationDate().
78 
79  If a cashflow payment is deemed to have already occurred relative to the settlement date, then no fixing is
80  needed. The determination of whether a cashflow has or has not occurred will in general rely on a call to \c
81  CashFlow::hasOccurred which is important in cases where the cash flow payment date falls on the settlement date.
82 
83  Another important case is where a cash flow fixing date occurs on the settlement date. In this case, we should
84  always add the fixing date to the set of fixing dates regardless of
85  \c Settings::instance().enforcesTodaysHistoricFixings(). */
86  std::map<std::string, std::set<QuantLib::Date>>
87  fixingDatesIndices(const QuantLib::Date& settlementDate = QuantLib::Date()) const;
88 
89  /*! Adds a single fixing date \p fixingDate for an index given by its ORE index name \p indexName arising from a
90  coupon with payment date \p payDate. If \p alwaysAddIfPaysOnSettlement is true the fixing date will be added
91  if the coupon pays on the settlement date even if the cashflow returns hasOccured(settlementDate) as true. This
92  is conservative and necessary in some cases since some pricing engines in QL (e.g. CapFloor) do not respect
93  hasOccured() and ask for the fixing regardless. If the payDate is not given, it defaults to Date::maxDate()
94  meaning that the added fixing is relevant unconditional on a pay date */
95  void addFixingDate(const QuantLib::Date& fixingDate, const std::string& indexName,
96  const QuantLib::Date& payDate = Date::maxDate(), const bool alwaysAddIfPaysOnSettlement = false);
97 
98  /*! adds a vector of fixings dates \p fixingDates for an index given by is ORE index name \p indexName arising from
99  a coupon with payment date \p payDate */
100  void addFixingDates(const std::vector<QuantLib::Date>& fixingDates, const std::string& indexName,
101  const QuantLib::Date& payDate = Date::maxDate(),
102  const bool alwaysAddIfPaysOnSettlement = false);
103 
104  /*! add a single fixing date \p fixingDate for a coupon based on a zero inflation index given by its ORE index name
105  \p indexName with payment date \p payDate */
106  void addZeroInflationFixingDate(const QuantLib::Date& fixingDate, const std::string& indexName,
107  const bool indexInterpolated, const Frequency indexFrequency,
108  const Period& indexAvailabilityLag,
109  const CPI::InterpolationType coupopnInterpolation, const Frequency couponFrequency,
110  const QuantLib::Date& payDate = Date::maxDate(),
111  const bool alwaysAddIfPaysOnSettlement = false);
112 
113  /*! add a single fixing date \p fixingDate for a coupon based on a yoy inflation index given by its ORE index name
114  \p indexName with payment date \p payDate */
115  void addYoYInflationFixingDate(const QuantLib::Date& fixingDate, const std::string& indexName,
116  const bool indexInterpolated, const Frequency indexFrequency,
117  const Period& indexAvailabilityLag, const QuantLib::Date& payDate = Date::maxDate(),
118  const bool alwaysAddIfPaysOnSettlement = false);

120  /*! clear all data */
121  void clear();
122 
123  /*! add data from another RequiredFixings instance */
124  void addData(const RequiredFixings& requiredFixings);
125 
126  /*! Set all pay dates to Date::maxDate(), fixingDatesIndices() will then not filter the required fixings by the
127  given settlement date any more. Needed by total return swaps on bonds for example, where a cashflow in a bond with
128  past payment date can still be relevant for the payment of the current return period. */
129  void unsetPayDates();
130 
131 private:
132  // FixingEntry = indexName, fixingDate, payDate, alwaysAddIfPaysOnSettlement
133  using FixingEntry = std::tuple<std::string, QuantLib::Date, QuantLib::Date, bool>;
134  // InflationFixingEntry = FixingEntry, indexInterpolated, indexFrequency, indexAvailabilityLag
135  using InflationFixingEntry = std::tuple<FixingEntry, bool, Frequency, Period>;
136  // ZeroInflationFixingEntry = InflationFixingEntry, couponInterpolation, couponFrequency
137  using ZeroInflationFixingEntry = std::tuple<InflationFixingEntry, CPI::InterpolationType, Frequency>;
138 
139  // maps an ORE index name to a pair (fixing date, pay date, alwaysAddIfPaysOnSettlment)
140  std::set<FixingEntry> fixingDates_;
141  // same as above, but for zero inflation index based coupons
142  std::set<ZeroInflationFixingEntry> zeroInflationFixingDates_;
143  // same as above, but for yoy inflation index based coupons
144  std::set<InflationFixingEntry> yoyInflationFixingDates_;
145 
146  // grant access to stream output operator
147  friend std::ostream& operator<<(std::ostream&, const RequiredFixings&);
148 };

232 /*! Inflation fixings are generally available on a monthly, or coarser, frequency. When a portfolio is asked for its
233  fixings, and it contains inflation fixings, ORE will by convention put the fixing date as the 1st day of the
234  applicable inflation period. Some market data providers by convention supply the inflation fixings with the date
235  as the last date of the applicable inflation period. This function scans the \p fixings map, and moves any
236  inflation fixing dates from the 1st day of the inflation period to the last day of the inflation period. The key
237  in the \p fixings map is the index name and the value is the set of dates for which we require the fixings.
238 
239  If inflation indices have been set up via ZeroInflationIndex entries in the Conventions, the \p conventions
240  should be passed here. If not, the default \c nullptr parameter will be sufficient.
241 */
242 void amendInflationFixingDates(std::map<std::string, std::set<QuantLib::Date>>& fixings);
243 
244 /*! Add index and fixing date pairs to \p fixings that will be potentially needed to build a TodaysMarket.
245 
246  These additional index and fixing date pairs are found by scanning the \p mktParams and:
247  - for MarketObject::IndexCurve, take the ibor index name and add the dates for each weekday between settlement
248  date minus \p iborLookback period or \p oisLookback period and settlement date. The distinction between
249  Ibor and OIS is made here to cover the fixings necessary for OIS futures. The default value of 4 months
250  covers OIS futures with a contract period of up to 3 months. It would need to be configured differently
251  if OIS futures with a longer contract period are possible.
252  - for MarketObject::ZeroInflationCurve, take the inflation index and add the first of each month between
253  settlement date minus \p inflationLookback period and settlement date
254  - for MarketObject::YoYInflationCurve, take the inflation index and add the first of each month between
255  settlement date minus \p inflationLookback period and settlement date
256  - for MarketObject::CommodityCurve, add \e fixings for future contracts expiring 2 months either side of the
257  settlement date. The fixing dates are added for each weekday going back to the first day of the month that
258  precedes the settlement date by 2 months. The approach here will give rise to some spot commodities being
259  given a future contract name and dates added against them - this should not be a problem as there will be
260  no fixings found for them in any case.
261 
262  The original \p fixings map may be empty.
263 */
264 void addMarketFixingDates(std::map<std::string, std::set<QuantLib::Date>>& fixings,
265  const TodaysMarketParameters& mktParams,
266  const QuantLib::Period& iborLookback = 5 * QuantLib::Days,
267  const QuantLib::Period& oisLookback = 4 * QuantLib::Months,
268  const QuantLib::Period& bmaLookback = 2 * QuantLib::Weeks,
269  const QuantLib::Period& inflationLookback = 1 * QuantLib::Years,
270  const std::string& configuration = Market::defaultConfiguration);
