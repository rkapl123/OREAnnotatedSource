35 void AsianOption::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
36 
37  Currency payCcy = parseCurrency(currency_);
38 
39  QL_REQUIRE(tradeActions().empty(), "TradeActions not supported for AsianOption");
40 
41  Option::Type type = parseOptionType(option_.callPut());
42  QL_REQUIRE(option_.exerciseDates().size() == 1, "Expected exactly one exercise date");
43  Date expiryDate = parseDate(option_.exerciseDates().front());
44 
45  string tradeTypeBuilder = tradeType_;
46 
47  // Add Arithmetic/Geometric
48 
49  if (option_.payoffType2() == "Arithmetic" || option_.payoffType2().empty())
50  tradeTypeBuilder += "Arithmetic";
51  else if (option_.payoffType2() == "Geometric")
52  tradeTypeBuilder += "Geometric";
53  else {
54  QL_FAIL("payoff type 2 must be 'Arithmetic' or 'Geomtetric'");
55  }
56 
57  // Add Price/Strike
58 
59  if (option_.payoffType() == "Asian")
60  tradeTypeBuilder += "Price";
61  else if (option_.payoffType() == "AverageStrike")
62  tradeTypeBuilder += "Strike";
63  else {
64  QL_FAIL("payoff type must be 'Asian' or 'AverageStrike'");
65  }
66 
67  boost::shared_ptr<EngineBuilder> builder = engineFactory->builder(tradeTypeBuilder);
68  QL_REQUIRE(builder, "No builder found for " << tradeTypeBuilder);
69 
70  // check for delegating engine builder
71 
72  if (auto db = boost::dynamic_pointer_cast<DelegatingEngineBuilder>(builder)) {
73 
74  // let the delegating builder build the trade and link the results to this trade
75 
76  delegatingBuilderTrade_ = db->build(this, engineFactory);
77 
78  instrument_ = delegatingBuilderTrade_->instrument();
79  maturity_ = delegatingBuilderTrade_->maturity();
80  npvCurrency_ = delegatingBuilderTrade_->npvCurrency();
81  additionalData_ = delegatingBuilderTrade_->additionalData();
82  requiredFixings_ = delegatingBuilderTrade_->requiredFixings();
83 
84  // notional and notional currency are defined in overriden methods!
85 
86  return;
87  }
88 
89  // we do not have a delegating engine builder
90 
91  boost::shared_ptr<AsianOptionEngineBuilder> asianOptionBuilder =
92  boost::dynamic_pointer_cast<AsianOptionEngineBuilder>(builder);
93 
94  QL_REQUIRE(asianOptionBuilder, "engine builder is not an AsianOption engine builder" << tradeTypeBuilder);
95 
96  std::string processType = asianOptionBuilder->processType();
97  QL_REQUIRE(!processType.empty(), "ProcessType must be configured, this is unexpected");
98 
99  boost::shared_ptr<StrikedTypePayoff> payoff(new PlainVanillaPayoff(type, tradeStrike_.value()));
100 
101  auto index = parseIndex(indexName());
102 
103  if (auto fxIndex = boost::dynamic_pointer_cast<QuantExt::FxIndex>(index)) {
104  QL_REQUIRE(fxIndex->targetCurrency() == payCcy,
105  "FX domestic ccy " << fxIndex->targetCurrency() << " must match pay ccy " << payCcy);
106  assetName_ = fxIndex->sourceCurrency().code();
107  } else if (auto eqIndex = boost::dynamic_pointer_cast<QuantExt::EquityIndex>(index)) {
108  // FIXME for EQ and COMM indices check whether EQ, COMM ccy = payCcy (in the engine builders probably)
109  assetName_ = eqIndex->name();
110  } else if (auto commIndex = boost::dynamic_pointer_cast<QuantExt::CommodityIndex>(index)) {
111  assetName_ = commIndex->underlyingName();
112  }
113 
114  // FIXME the engine should handle the historical part of the averaging as well!
115  boost::shared_ptr<QuantLib::Instrument> asian;
116  auto exercise = boost::make_shared<QuantLib::EuropeanExercise>(expiryDate);
117  if (processType == "Discrete") {
118  QuantLib::Date today = engineFactory->market()->asofDate();
119  Real runningAccumulator = option_.payoffType2() == "Geometric" ? 1.0 : 0.0;
120  Size pastFixings = 0;
121  Schedule observationSchedule = makeSchedule(observationDates_);
122  std::vector<QuantLib::Date> observationDates = observationSchedule.dates();
123 
124  // Sort for the engine's sake. Not needed - instrument also sorts...?
125  std::sort(observationDates.begin(), observationDates.end());
126 
127  for (QuantLib::Date observationDate : observationDates) {
128  // TODO: Verify. Should today be read too? a enforcesTodaysHistoricFixings() be used?
129  if (observationDate < today ||
130  (observationDate == today && Settings::instance().enforcesTodaysHistoricFixings())) {
131  // FIXME all observation dates lead to a required fixing
132  requiredFixings_.addFixingDate(observationDate, indexName());
133  Real fixingValue = index->fixing(observationDate);
134  if (option_.payoffType2() == "Geometric") {
135  runningAccumulator *= fixingValue;
136  } else if (option_.payoffType2() == "Arithmetic") {
137  runningAccumulator += fixingValue;
138  }
139  ++pastFixings;
140  }
141  }
142  asian = boost::make_shared<QuantLib::DiscreteAveragingAsianOption>(
143  option_.payoffType2() == "Geometric" ? QuantLib::Average::Type::Geometric
144  : QuantLib::Average::Type::Arithmetic,
145  runningAccumulator, pastFixings, observationDates, payoff, exercise);
146  } else if (processType == "Continuous") {
147  // FIXME how is the accumulated average handled in this case?
148  asian = boost::make_shared<QuantLib::ContinuousAveragingAsianOption>(option_.payoffType2() == "Geometric"
149  ? QuantLib::Average::Type::Geometric
150  : QuantLib::Average::Type::Arithmetic,
151  payoff, exercise);
152  } else {
153  QL_FAIL("unexpected ProcessType, valid options are Discrete/Continuous");
154  }
155 
156  // Only try to set an engine on the option instrument if it is not expired. This avoids errors in
157  // engine builders that rely on the expiry date being in the future.
158  string configuration = asianOptionBuilder->configuration(MarketContext::pricing);
159  if (!asian->isExpired()) {
160  asian->setPricingEngine(asianOptionBuilder->engine(assetName_, payCcy, expiryDate));
161  } else {
162  DLOG("No engine attached for option on trade " << id() << " with expiry date " << io::iso_date(expiryDate)
163  << " because it is expired.");
164  }
165 
166  Position::Type positionType = parsePositionType(option_.longShort());
167  Real bsInd = (positionType == QuantLib::Position::Long ? 1.0 : -1.0);
168  Real mult = quantity_ * bsInd;
169 
170  std::vector<boost::shared_ptr<Instrument>> additionalInstruments;
171  std::vector<Real> additionalMultipliers;
172  maturity_ = expiryDate;
173  maturity_ =
174  std::max(maturity_, addPremiums(additionalInstruments, additionalMultipliers, 1.0, option_.premiumData(),
175  positionType == QuantLib::Position::Long ? -1.0 : 1.0, payCcy, engineFactory,
176  configuration));
177 
178  instrument_ = boost::make_shared<VanillaInstrument>(asian, mult, additionalInstruments, additionalMultipliers);
179 
180  npvCurrency_ = currency_;
181  notional_ = tradeStrike_.value() * quantity_;
182  notionalCurrency_ = currency_;
183 }

186  Trade::fromXML(node);
187  XMLNode* n = XMLUtils::getChildNode(node, tradeType() + "Data");
188  QL_REQUIRE(n, "No " + tradeType_ + "Data node found.");
189 
190  quantity_ = XMLUtils::getChildValueAsDouble(n, "Quantity", true);
191 
192  tradeStrike_.fromXML(n);
193 
194  currency_ = XMLUtils::getChildValue(n, "Currency", false);
195 
196  XMLNode* tmp = XMLUtils::getChildNode(n, "Underlying");
197  if (!tmp)
198  tmp = XMLUtils::getChildNode(n, "Name");
199  UnderlyingBuilder underlyingBuilder;
200  underlyingBuilder.fromXML(tmp);
201  underlying_ = underlyingBuilder.underlying();
202 
203  option_.fromXML(XMLUtils::getChildNode(n, "OptionData"));
204 
205  settlementDate_ = parseDate(XMLUtils::getChildValue(n, "Settlement", false));
206 
207  observationDates_.fromXML(XMLUtils::getChildNode(n, "ObservationDates"));
208 }

210 XMLNode* AsianOption::toXML(XMLDocument& doc) {
211  XMLNode* node = Trade::toXML(doc);
212  XMLNode* n = doc.allocNode(tradeType() + "Data");
213  XMLUtils::appendNode(node, n);
214  XMLUtils::addChild(doc, n, "Quantity", quantity_);
215  XMLUtils::appendNode(n, tradeStrike_.toXML(doc));
216  XMLUtils::addChild(doc, n, "Currency", currency_);
217  XMLUtils::appendNode(n, underlying_->toXML(doc));
218  XMLUtils::appendNode(n, option_.toXML(doc));
219  if (settlementDate_ != Null<Date>())
220  XMLUtils::addChild(doc, n, "Settlement", ore::data::to_string(settlementDate_));
221  auto tmp = observationDates_.toXML(doc);
222  XMLUtils::setNodeName(doc, tmp, "ObservationDates");
223  XMLUtils::appendNode(n, tmp);
224  return node;
225 }

240 void AsianOption::populateIndexName() const {
241  if (!indexName_.empty())
242  return;
243  if (underlying_->type() == "Equity") {
244  indexName_ = "EQ-" + underlying_->name();
245  } else if (underlying_->type() == "FX") {
246  indexName_ = "FX-" + underlying_->name();
247  } else if (underlying_->type() == "Commodity") {
248  boost::shared_ptr<CommodityUnderlying> comUnderlying =
249  boost::dynamic_pointer_cast<CommodityUnderlying>(underlying_);
250  std::string tmp = "COMM-" + comUnderlying->name();
251  if (comUnderlying->priceType().empty() || comUnderlying->priceType() == "Spot") {
252  indexName_ = tmp;
253  } else if (comUnderlying->priceType() == "FutureSettlement") {
254  auto conventions = InstrumentConventions::instance().conventions();
255  QL_REQUIRE(conventions->has(comUnderlying->name()),
256  "future settlement requires conventions for commodity '" << comUnderlying->name() << "'");
257  auto convention =
258  boost::dynamic_pointer_cast<CommodityFutureConvention>(conventions->get(comUnderlying->name()));
259  Size futureMonthsOffset =
260  comUnderlying->futureMonthOffset() == Null<Size>() ? 0 : comUnderlying->futureMonthOffset();
261  Size deliveryRollDays =
262  comUnderlying->deliveryRollDays() == Null<Size>() ? 0 : comUnderlying->deliveryRollDays();
263  Calendar cal = parseCalendar(comUnderlying->deliveryRollCalendar());
264  ConventionsBasedFutureExpiry expiryCalculator(*convention);
265  QL_REQUIRE(option_.exerciseDates().size() == 1, "expected exactly one exercise date");
266  Date expiryDate = parseDate(option_.exerciseDates().front());
267  Date adjustedObsDate =
268  deliveryRollDays != 0 ? cal.advance(expiryDate, deliveryRollDays * Days) : expiryDate;
269  auto tmp = parseCommodityIndex(comUnderlying->name(), false, Handle<QuantExt::PriceTermStructure>(),
270  convention->calendar(), true);
271  tmp = tmp->clone(expiryCalculator.nextExpiry(true, adjustedObsDate, futureMonthsOffset));
272  indexName_ = tmp->name();
273  } else {
274  QL_FAIL("underlying price type '" << comUnderlying->priceType() << "' for commodity underlying '"
275  << comUnderlying->name() << "' not handled.");
276  }
277  } else if (underlying_->type() == "Basic") {
278  indexName_ = underlying_->name();
279  } else {
280  QL_FAIL("invalid underlying type: " << underlying_->type());
281  }
282 }
