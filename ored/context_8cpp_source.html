29Size Context::varSize() const {
30 QL_REQUIRE(!scalars.empty() || !arrays.empty(), "Context::varSize(): context is empty()");
31 Size res = Null<Size>();
32 for (auto const& v : scalars) {
33 if (res != Null<Size>())
34 QL_REQUIRE(size(v.second) == res, "Context::varSize(): inconsistent var sizes");
35 else
36 res = size(v.second);
37 }
38 for (auto const& v : arrays) {
39 for (auto const& a : v.second) {
40 if (res != Null<Size>())
41 QL_REQUIRE(size(a) == res, "Context::varSize(): inconsistent var sizes");
42 else
43 res = size(a);
44 }
45 }
46 return res;
47}

49std::ostream& operator<<(std::ostream& out, const Context& context) {
50 constexpr Size wd1 = 30, wd2 = 10;
51 for (auto const& v : context.scalars) {
52 out << std::setw(wd1) << std::left << v.first;
53 out << "(" << std::setw(wd2) << valueTypeLabels.at(v.second.which()) << ")";
54 if (std::find(context.constants.begin(), context.constants.end(), v.first) != context.constants.end())
55 out << " const ";
56 else
57 out << " ";
58 out << std::left << v.second << '\n';
59 }
60 for (auto const& v : context.arrays) {
61 Size counter = 0;
62 for (auto const& d : v.second) {
63 out << std::setw(wd1) << std::left << (v.first + "[" + std::to_string(counter + 1) + "]");
64 out << "(" << std::setw(wd2) << valueTypeLabels.at(d.which()) << ")";
65 if (std::find(context.constants.begin(), context.constants.end(), v.first) != context.constants.end())
66 out << " const ";
67 else
68 out << " ";
69 out << std::left << d << '\n';
70 ++counter;
71 }
72 }
73 return out;
74}

77void resetSize(ValueType& v, const Size n) {
78 switch (v.which()) {
79 case ValueTypeWhich::Number:
80 boost::get<RandomVariable>(v).resetSize(n);
81 break;
82 case ValueTypeWhich::Filter:
83 boost::get<Filter>(v).resetSize(n);
84 break;
85 case ValueTypeWhich::Event:
86 boost::get<EventVec>(v).size = n;
87 break;
88 case ValueTypeWhich::Currency:
89 boost::get<CurrencyVec>(v).size = n;
90 break;
91 case ValueTypeWhich::Index:
92 boost::get<IndexVec>(v).size = n;
93 break;
94 case ValueTypeWhich::Daycounter:
95 boost::get<DaycounterVec>(v).size = n;
96 break;
97 default:
98 QL_FAIL("resetSize(): value type not handled. internal error, contact dev.");
99 }
100}
101} // namespace
