45QuantLib::Currency CurrencyParser::parseCurrency(const std::string& name) const {
46 boost::shared_lock<boost::shared_mutex> lock(mutex_);
47 {
48 auto it = currencies_.find(name);
49 if (it != currencies_.end()) {
50 return it->second;
51 }
52 }
53 {
54 auto it = preciousMetals_.find(name);
55 if (it != preciousMetals_.end()) {
56 return it->second;
57 }
58 }
59 {
60 auto it = cryptoCurrencies_.find(name);
61 if (it != cryptoCurrencies_.end()) {
62 return it->second;
63 }
64 }
65 QL_FAIL("Currency \"" << name << "\" not recognized");
66}

68QuantLib::Currency CurrencyParser::parseMinorCurrency(const std::string& name) const {
69 boost::shared_lock<boost::shared_mutex> lock(mutex_);
70 auto it = minorCurrencies_.find(name);
71 if (it != minorCurrencies_.end()) {
72 return it->second;
73 }
74 QL_FAIL("Currency \"" << name << "\" not recognized");
75}

77QuantLib::Currency CurrencyParser::parseCurrencyWithMinors(const std::string& name) const {
78 try {
79 return parseCurrency(name);
80 } catch (...) {
81 }
82 return parseMinorCurrency(name);
83}

85bool CurrencyParser::isValidCurrency(const std::string& name) const {
86 try {

113bool CurrencyParser::hasMinorCurrency(const std::string& name) const {
114 boost::shared_lock<boost::shared_mutex> lock(mutex_);
115 for (auto const& c : minorCurrencies_) {
116 if (c.second.code() == name)
117 return true;
118 }
119 return false;
120}

122std::string CurrencyParser::getMinorCurrency(const std::string& name) const {
123 boost::shared_lock<boost::shared_mutex> lock(mutex_);
124 for (auto const& c : minorCurrencies_) {
125 if (c.second.code() == name)
126 return c.first;
127 }
128 QL_FAIL("no minor currency found for '" << name << "'");
129}

131std::set<std::string> CurrencyParser::pseudoCurrencyCodes() const {
132 boost::shared_lock<boost::shared_mutex> lock(mutex_);
133 std::set<std::string> tmp;
134 for (auto const& c : preciousMetals_)
135 tmp.insert(c.first);
136 for (auto const& c : cryptoCurrencies_)
137 tmp.insert(c.first);

141QuantLib::Real CurrencyParser::convertMinorToMajorCurrency(const std::string& s, QuantLib::Real value) {
142 if (isMinorCurrency(s)) {
143 QuantLib::Currency ccy = parseMinorCurrency(s);
144 return value / ccy.fractionsPerUnit();
145 } else {
146 return value;
147 }
148}

150void CurrencyParser::addCurrency(const std::string& newName, const QuantLib::Currency& currency) {
151 boost::unique_lock<boost::shared_mutex> lock(mutex_);
152 if (currencies_.find(newName) != currencies_.end() || preciousMetals_.find(newName) != preciousMetals_.end() ||
153 cryptoCurrencies_.find(newName) != cryptoCurrencies_.end())
154 return;
155 currencies_[newName] = currency;
156 addMinorCurrencyCodes(currency);
157}
158
159void CurrencyParser::addMinorCurrencyCodes(const QuantLib::Currency& currency) {
160 for (auto const& c : currency.minorUnitCodes()) {
161 minorCurrencies_[c] = currency;
162 }
163}
