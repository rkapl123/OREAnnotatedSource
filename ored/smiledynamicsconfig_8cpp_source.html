29void SmileDynamicsConfig::validate() {
30 std::vector<string> v = { "StickyStrike", "StickyMoneyness" };
31 QL_REQUIRE(std::find(v.begin(), v.end(), swaption_) != v.end(), "swaption smile dynamics " << swaption_ << " invalid");
32 QL_REQUIRE(std::find(v.begin(), v.end(), capFloor_) != v.end(), "capfloor smile dynamics " << capFloor_ << " invalid");
33 QL_REQUIRE(std::find(v.begin(), v.end(), yield_) != v.end(), "yield vol smile dynamics " << yield_ << " invalid");
34 QL_REQUIRE(std::find(v.begin(), v.end(), zeroInflationCapFloor_) != v.end(), "ZCI vol smile dynamics " << zeroInflationCapFloor_ << " invalid");
35 QL_REQUIRE(std::find(v.begin(), v.end(), yoyInflationCapFloor_) != v.end(), "YOY smile dynamics " << yoyInflationCapFloor_ << " invalid");
36 QL_REQUIRE(std::find(v.begin(), v.end(), equity_) != v.end(), "equity vol smile dynamics " << equity_ << " invalid");
37 QL_REQUIRE(std::find(v.begin(), v.end(), commodity_) != v.end(), "commodity vol smile dynamics " << commodity_ << " invalid");
38 QL_REQUIRE(std::find(v.begin(), v.end(), fx_) != v.end(), "FX vol smile dynamics " << fx_ << " invalid");
39 QL_REQUIRE(std::find(v.begin(), v.end(), cds_) != v.end(), "CDS vol smile dynamics " << cds_ << " invalid");
40}
