42//! Linear Gauss Markov Model Parameters
43/*!
44 This class contains the description of a Linear Gauss Markov interest rate model
45 and instructions for how to calibrate it.
46
47 \ingroup models
48 */
49class LgmData : public IrModelData {
50public:
51 //! Supported mean reversion types
52 enum class ReversionType {
53 /*! Parametrize H(t) via Hull-White mean reversion speed,
54 LGM H(t) = int_0^t exp(-kappa(s) *s) ds with constant or piecewise kappa(s)
55 */
56 HullWhite,
57 //! Parametrize LGM H(t) as H(t) = int_0^t h(s) ds with constant or piecewise h(s)
58 Hagan
59 // FIXME: indent
60 };
61
62 //! Supported volatility types
63 enum class VolatilityType {
64 //! Parametrize volatility as HullWhite sigma(t)
65 HullWhite,
66 //! Parametrize volatility as Hagan alpha(t)
67 Hagan
68 };
69
70 //! Default constructor
71 LgmData()
72 : IrModelData("LGM", "", CalibrationType::None), revType_(ReversionType::Hagan),
73 volType_(VolatilityType::Hagan), calibrateH_(false), hType_(ParamType::Constant), calibrateA_(false),
74 aType_(ParamType::Constant), shiftHorizon_(0.0), scaling_(1.0) {}

79 ParamType aType, std::vector<Time> aTimes, std::vector<Real> aValues, Real shiftHorizon = 0.0,
80 Real scaling = 1.0, std::vector<std::string> optionExpiries = std::vector<std::string>(),
81 std::vector<std::string> optionTerms = std::vector<std::string>(),
82 std::vector<std::string> optionStrikes = std::vector<std::string>())
83 : IrModelData("LGM", qualifier, calibrationType), revType_(revType), volType_(volType),

85 aType_(aType), aTimes_(aTimes), aValues_(aValues), shiftHorizon_(shiftHorizon), scaling_(scaling),
86 optionExpiries_(optionExpiries), optionTerms_(optionTerms), optionStrikes_(optionStrikes) {}
