154 void ScheduleBuilder::makeSchedules(const Date& openEndDateReplacement) {
155  map<string, Schedule> builtSchedules;
156  map<string, ScheduleData> derivedSchedules;
157 
158  // First, we build all the rules-based and dates-based schedules
159  for (auto& s : schedules_) {
160  string schName = s.first;
161  ScheduleData& schData = s.second.first;
162  Schedule& sch = s.second.second;
163 
164  if (!schData.hasDerived()) {
165  sch = makeSchedule(schData, openEndDateReplacement);
166  builtSchedules[schName] = sch;
167  } else {
168  derivedSchedules[schName] = schData;
169  }
170  }
171 
172  // We then keep looping through the list of derived schedules and building these from the list of built schedules.
173  bool calculated;
174  while (derivedSchedules.size() > 0) {
175  calculated = false;
176  for (auto& ds : derivedSchedules) {
177  string dsName = ds.first;
178  ScheduleData& dsSchedData = ds.second;
179  vector<string> baseNames = dsSchedData.baseScheduleNames();
180  for (string& bn : baseNames) {
181  QL_REQUIRE(builtSchedules.find(bn) != builtSchedules.end(), "Could not find base schedule \" " << bn << "\" for derived schedule \" " << dsName << "\"");
182  }
183  Schedule schedule;
184  schedule = makeSchedule(dsSchedData, openEndDateReplacement, builtSchedules);
185  schedules_.find(dsName)->second.second = schedule;
186  builtSchedules[dsName] = schedule;
187  derivedSchedules.erase(dsName);
188  calculated = true;
189  break;
190  }
191 
192  // If we go through the whole list without having built a schedule, then assume that we cannot build them
193  // anymore.
194  if (!calculated) {
195  for (auto& ds : derivedSchedules)
196  ALOG("makeSchedules(): could not find base schedule \"" << ds.first << "\"");
197  QL_FAIL("makeSchedules(): failed to build at least one derived schedule");
198  break;
199  }
200  }
201 }

203 Schedule makeSchedule(const ScheduleDates& data) {
204  QL_REQUIRE(data.dates().size() > 0, "Must provide at least 1 date for Schedule");
205  Calendar calendar = parseCalendar(data.calendar());
206  BusinessDayConvention convention = ModifiedFollowing;
207  if (data.convention() != "")
208  convention = parseBusinessDayConvention(data.convention());
209  // Avoid compiler warning on gcc
210  // https://www.boost.org/doc/libs/1_74_0/libs/optional/doc/html/boost_optional/tutorial/
211  // gotchas/false_positive_with__wmaybe_uninitialized.html
212  auto tenor = boost::make_optional(false, Period());
213  if (data.tenor() != "")
214  tenor = parsePeriod(data.tenor());
215  bool endOfMonth = false;
216  if (data.endOfMonth() != "")
217  endOfMonth = parseBool(data.endOfMonth());
218 
219  // Ensure that Schedule ctor is passed a vector of unique ordered dates.
220  std::set<Date> uniqueDates;
221  for (const string& d : data.dates())
222  uniqueDates.insert(calendar.adjust(parseDate(d), convention));
223 
224  return Schedule(vector<Date>(uniqueDates.begin(), uniqueDates.end()), calendar, convention, boost::none, tenor,
225  boost::none, endOfMonth);
226 }

228 Schedule makeSchedule(const ScheduleDerived& data, const Schedule& baseSchedule) {
229 
230  string strCalendar = data.calendar();
231  Calendar calendar;
232  if (strCalendar.empty()) {
233  calendar = NullCalendar();
234  WLOG("No calendar provided in Schedule, attempting to use a null calendar.");
235  }
236  else
237  calendar = parseCalendar(strCalendar);
238 
239  BusinessDayConvention convention;
240  string strConvention = data.convention();
241  if (strConvention.empty())
242  convention = BusinessDayConvention::Unadjusted;
243  else
244  convention = parseBusinessDayConvention(strConvention);
245 
246  string strShift = data.shift();
247  Period shift;
248  if (strShift.empty())
249  shift = 0 * Days;
250  else
251  shift = parsePeriod(data.shift());
252 
253  const std::vector<QuantLib::Date>& baseDates = baseSchedule.dates();
254  std::vector<QuantLib::Date> derivedDates;
255  QuantLib::Date derivedDate;
256  for (const Date& d : baseDates) {
257  derivedDate = calendar.advance(d, shift, convention);
258  derivedDates.push_back(derivedDate);
259  }
260  return Schedule(vector<Date>(derivedDates.begin(), derivedDates.end()), calendar, convention, boost::none,
261  baseSchedule.tenor(), boost::none, baseSchedule.endOfMonth());

264 Schedule makeSchedule(const ScheduleRules& data, const Date& openEndDateReplacement) {
265  QL_REQUIRE(!data.endDate().empty() || openEndDateReplacement != Null<Date>(),
266  "makeSchedule(): Schedule does not have an end date, this is not supported in this context / for this "
267  "trade type. Please provide an end date.");
268  QL_REQUIRE(!data.endDate().empty() || data.lastDate().empty(),
269  "makeSchedule(): If no end date is given, a last date is not allowed either. Please remove the last "
270  "date from the schedule.");
271  Calendar calendar = parseCalendar(data.calendar());
272  if (calendar == NullCalendar())
273  WLOG("No calendar provided in Schedule, attempting to use a null calendar.");
274  Date startDate = parseDate(data.startDate());
275  Date endDate = data.endDate().empty() ? openEndDateReplacement : parseDate(data.endDate());
276  // Handle trivial case here
277  if (startDate == endDate)
278  return Schedule(vector<Date>(1, startDate), calendar);
279 
280  QL_REQUIRE(startDate < endDate, "StartDate " << startDate << " is ahead of EndDate " << endDate);
281 
282  Date firstDate = parseDate(data.firstDate());
283  Date lastDate = parseDate(data.lastDate());
284  Period tenor = parsePeriod(data.tenor());
285 
286  // defaults
287  BusinessDayConvention bdc = ModifiedFollowing;
288  BusinessDayConvention bdcEnd = ModifiedFollowing;
289  DateGeneration::Rule rule = DateGeneration::Forward;
290  bool endOfMonth = false;
291 
292  // now check the strings, if they are empty we take defaults
293  if (!data.convention().empty())
294  bdc = parseBusinessDayConvention(data.convention());
295  if (!data.termConvention().empty())
296  bdcEnd = parseBusinessDayConvention(data.termConvention());
297  else
298  bdcEnd = bdc; // except here
299  if (!data.rule().empty())
300  rule = parseDateGenerationRule(data.rule());
301  if (!data.endOfMonth().empty())
302  endOfMonth = parseBool(data.endOfMonth());
303 
304  if ((rule == DateGeneration::CDS || rule == DateGeneration::CDS2015) &&
305  (firstDate != Null<Date>() || lastDate != Null<Date>())) {
306  // Special handling of first date and last date in combination with CDS and CDS2015 rules:
307  // To be able to construct CDS schedules with front or back stub periods, we overwrite the
308  // first (last) date of the schedule built in QL with a given first (last) date
309  // The schedule builder in QL itself is not capable of doing this, it just throws an exception
310  // if a first (last) date is given in combination with a CDS / CDS2015 date generation rule.
311  std::vector<Date> dates = Schedule(startDate, endDate, tenor, calendar, bdc, bdcEnd, rule, endOfMonth).dates();
312  QL_REQUIRE(!dates.empty(),
313  "got empty CDS or CDS2015 schedule, startDate = " << startDate << ", endDate = " << endDate);
314  if (firstDate != Date())
315  dates.front() = firstDate;
316  if (lastDate != Date())
317  dates.back() = lastDate;
318  return Schedule(dates, calendar, bdc, bdcEnd, tenor, rule, endOfMonth);
319  } else {
320  return Schedule(startDate, endDate, tenor, calendar, bdc, bdcEnd, rule, endOfMonth, firstDate, lastDate);
321  }
322 }

342 Schedule makeSchedule(const ScheduleData& data, const Date& openEndDateReplacement, const map<string, QuantLib::Schedule>& baseSchedules) {
343  // only the last rule-based schedule is allowed to have an open end date, check this
344  for (Size i = 1; i < data.rules().size(); ++i) {
345  QL_REQUIRE(!data.rules()[i - 1].endDate().empty(),
346  "makeSchedule(): only last schedule is allowed to have an open end date");
347  }
348  // build all the date and rule based sub-schedules we have
349  vector<Schedule> schedules;
350  for (auto& d : data.dates())
351  schedules.push_back(makeSchedule(d));
352  for (auto& r : data.rules())
353  schedules.push_back(makeSchedule(r, openEndDateReplacement));
354  if (!baseSchedules.empty())
355  for (auto& dv : data.derived()) {
356  auto baseSchedule = baseSchedules.find(dv.baseSchedule());
357  QL_REQUIRE(baseSchedule != baseSchedules.end(), "makeSchedule(): could not find base schedule \"" << dv.baseSchedule() << "\"");
358  schedules.push_back(makeSchedule(dv, baseSchedule->second));
359  }
360  QL_REQUIRE(!schedules.empty(), "No dates or rules to build Schedule from");
361  if (schedules.size() == 1)
362  // if we have just one, use that (most common case)
363  return schedules.front();
364  else {
365  // if we have multiple, combine them
366 
367  // 1) sort by start date
368  std::sort(schedules.begin(), schedules.end(),
369  [](const Schedule& lhs, const Schedule& rhs) -> bool { return lhs.startDate() < rhs.startDate(); });
370 
371  // 2) check if meta data is present, and if yes if it is consistent across schedules;
372  // the only exception is the term date convention, this is taken from the last schedule always
373  BusinessDayConvention convention = Null<BusinessDayConvention>(),
374  termConvention = Unadjusted; // initialization prevents gcc warning
375  Calendar calendar;
376  Period tenor;
377  DateGeneration::Rule rule = DateGeneration::Zero; // initialization prevents gcc warning
378  bool endOfMonth = false; // initialization prevents gcc warning
379  bool hasCalendar = false, hasConvention = false, hasTermConvention = false, hasTenor = false, hasRule = false,
380  hasEndOfMonth = false, hasConsistentCalendar = true, hasConsistentConvention = true,
381  hasConsistentTenor = true, hasConsistentRule = true, hasConsistentEndOfMonth = true;
382  for (auto& d : data.dates()) {
383  updateData<Calendar>(d.calendar(), calendar, hasCalendar, hasConsistentCalendar, parseCalendarTemp);
384  updateData<BusinessDayConvention>(d.convention(), convention, hasConvention, hasConsistentConvention,
385  parseBusinessDayConvention);
386  updateData<Period>(d.tenor(), tenor, hasTenor, hasConsistentTenor, parsePeriod);
387  }
388  for (auto& d : data.rules()) {
389  updateData<Calendar>(d.calendar(), calendar, hasCalendar, hasConsistentCalendar, parseCalendarTemp);
390  updateData<BusinessDayConvention>(d.convention(), convention, hasConvention, hasConsistentConvention,
391  parseBusinessDayConvention);
392  updateData<Period>(d.tenor(), tenor, hasTenor, hasConsistentTenor, parsePeriod);
393  updateData<bool>(d.endOfMonth(), endOfMonth, hasEndOfMonth, hasConsistentEndOfMonth, parseBool);
394  updateData<DateGeneration::Rule>(d.rule(), rule, hasRule, hasConsistentRule, parseDateGenerationRule);
395  if (d.termConvention() != "") {
396  hasTermConvention = true;
397  termConvention = parseBusinessDayConvention(d.termConvention());
398  }
399  }
400 
401  // 3) combine dates and fill isRegular flag
402  const Schedule& s0 = schedules.front();
403  vector<Date> dates = s0.dates();
404  std::vector<bool> isRegular(s0.dates().size() - 1, false);
405  if (s0.hasIsRegular())
406  isRegular = s0.isRegular();
407  // will be removed, if next schedule's front date is matching the last date of current schedule
408  isRegular.push_back(false);
409  for (Size i = 1; i < schedules.size(); ++i) {
410  const Schedule& s = schedules[i];
411  QL_REQUIRE(dates.back() <= s.dates().front(), "Dates mismatch");
412  // if the end points match up, skip one to avoid duplicates, otherwise take both
413  Size offset = dates.back() == s.dates().front() ? 1 : 0;
414  isRegular.erase(isRegular.end() - offset,
415  isRegular.end()); // correct for superfluous flags from previous schedule
416  // add isRegular information, if available, otherwise assume irregular periods
417  if (s.hasIsRegular()) {
418  isRegular.insert(isRegular.end(), s.isRegular().begin(), s.isRegular().end());
419  } else {
420  for (Size ii = 0; ii < s.dates().size() - 1; ++ii)
421  isRegular.push_back(false);
422  }
423  if (i < schedules.size() - 1) {
424  // will be removed if next schedule's front date is matching last date of current schedule
425  isRegular.push_back(false);
426  }
427  // add the dates
428  dates.insert(dates.end(), s.dates().begin() + offset, s.dates().end());
429  }
430 
431  // 4) Build schedule
432  return Schedule(dates, hasCalendar && hasConsistentCalendar ? calendar : NullCalendar(),
433  hasConvention && hasConsistentConvention ? convention : Unadjusted,
434  hasTermConvention ? boost::optional<BusinessDayConvention>(termConvention) : boost::none,
435  hasTenor && hasConsistentTenor ? boost::optional<Period>(tenor) : boost::none,
436  hasRule && hasConsistentRule ? boost::optional<DateGeneration::Rule>(rule) : boost::none,
437  hasEndOfMonth && hasConsistentEndOfMonth ? boost::optional<bool>(endOfMonth) : boost::none,
438  isRegular);
439  }
