31std::vector<Date> everyThursdayDates(const Date& startDate, const Date& endDate, const Date& firstDate) {
32 std::vector<Date> result;
33 if (firstDate != Date())
34 result.push_back(firstDate);
35 Date d = startDate;
36 while (d <= endDate && (d.weekday() != QuantLib::Thursday || d < firstDate)) {
37 ++d;
38 }
39 if (d.weekday() == QuantLib::Thursday && (result.empty() || result.back() != d))
40 result.push_back(d);
41 while (d + 7 <= endDate) {
42 d += 7;
43 result.push_back(d);
44 }
45 return result;
46}

183void ScheduleBuilder::makeSchedules(const Date& openEndDateReplacement) {
184 map<string, Schedule> builtSchedules;
185 map<string, ScheduleData> derivedSchedules;
186
187 // First, we build all the rules-based and dates-based schedules
188 for (auto& s : schedules_) {
189 string schName = s.first;
190 ScheduleData& schData = s.second.first;
191 Schedule& sch = s.second.second;
192
193 if (!schData.hasDerived()) {
194 sch = makeSchedule(schData, openEndDateReplacement);
195 builtSchedules[schName] = sch;
196 } else {
197 derivedSchedules[schName] = schData;
198 }
199 }
200
201 // We then keep looping through the list of derived schedules and building these from the list of built schedules.
202 bool calculated;
203 while (derivedSchedules.size() > 0) {
204 calculated = false;
205 for (auto& ds : derivedSchedules) {
206 string dsName = ds.first;
207 ScheduleData& dsSchedData = ds.second;
208 vector<string> baseNames = dsSchedData.baseScheduleNames();
209 for (string& bn : baseNames) {
210 QL_REQUIRE(builtSchedules.find(bn) != builtSchedules.end(), "Could not find base schedule \" " << bn << "\" for derived schedule \" " << dsName << "\"");
211 }
212 Schedule schedule;
213 schedule = makeSchedule(dsSchedData, openEndDateReplacement, builtSchedules);
214 schedules_.find(dsName)->second.second = schedule;
215 builtSchedules[dsName] = schedule;
216 derivedSchedules.erase(dsName);
217 calculated = true;
218 break;
219 }
220
221 // If we go through the whole list without having built a schedule, then assume that we cannot build them
222 // anymore.
223 if (!calculated) {
224 for (auto& ds : derivedSchedules)
225 ALOG("makeSchedules(): could not find base schedule \"" << ds.first << "\"");
226 QL_FAIL("makeSchedules(): failed to build at least one derived schedule");
227 break;
228 }
229 }

232Schedule makeSchedule(const ScheduleDates& data) {
233 QL_REQUIRE(data.dates().size() > 0, "Must provide at least 1 date for Schedule");
234 Calendar calendar = parseCalendar(data.calendar());
235 BusinessDayConvention convention = ModifiedFollowing;
236 if (data.convention() != "")
237 convention = parseBusinessDayConvention(data.convention());
238 // Avoid compiler warning on gcc
239 // https://www.boost.org/doc/libs/1_74_0/libs/optional/doc/html/boost_optional/tutorial/
240 // gotchas/false_positive_with__wmaybe_uninitialized.html
241 auto tenor = boost::make_optional(false, Period());
242 if (data.tenor() != "")
243 tenor = parsePeriod(data.tenor());
244 bool endOfMonth = false;
245 if (data.endOfMonth() != "")
246 endOfMonth = parseBool(data.endOfMonth());
247
248 // Ensure that Schedule ctor is passed a vector of unique ordered dates.
249 std::set<Date> uniqueDates;
250 for (const string& d : data.dates())
251 uniqueDates.insert(calendar.adjust(parseDate(d), convention));
252
253 return Schedule(vector<Date>(uniqueDates.begin(), uniqueDates.end()), calendar, convention, boost::none, tenor,
254 boost::none, endOfMonth);
255}

257Schedule makeSchedule(const ScheduleDerived& data, const Schedule& baseSchedule) {
258
259 string strCalendar = data.calendar();
260 Calendar calendar;
261 if (strCalendar.empty()) {
262 calendar = NullCalendar();
263 WLOG("No calendar provided in Schedule, attempting to use a null calendar.");
264 }
265 else
266 calendar = parseCalendar(strCalendar);
267
268 BusinessDayConvention convention;
269 string strConvention = data.convention();
270 if (strConvention.empty())
271 convention = BusinessDayConvention::Unadjusted;
272 else
273 convention = parseBusinessDayConvention(strConvention);
274
275 string strShift = data.shift();
276 Period shift;
277 if (strShift.empty())
278 shift = 0 * Days;
279 else
280 shift = parsePeriod(data.shift());
281
282 const std::vector<QuantLib::Date>& baseDates = baseSchedule.dates();
283 std::vector<QuantLib::Date> derivedDates;
284 QuantLib::Date derivedDate;
285 for (const Date& d : baseDates) {
286 derivedDate = calendar.advance(d, shift, convention);
287 derivedDates.push_back(derivedDate);
288 }
289 return Schedule(vector<Date>(derivedDates.begin(), derivedDates.end()), calendar, convention, boost::none,
290 baseSchedule.tenor(), boost::none, baseSchedule.endOfMonth());
291}

293Schedule makeSchedule(const ScheduleRules& data, const Date& openEndDateReplacement) {
294 QL_REQUIRE(!data.endDate().empty() || openEndDateReplacement != Null<Date>(),
295 "makeSchedule(): Schedule does not have an end date, this is not supported in this context / for this "
296 "trade type. Please provide an end date.");
297 QL_REQUIRE(!data.endDate().empty() || data.lastDate().empty(),
298 "makeSchedule(): If no end date is given, a last date is not allowed either. Please remove the last "
299 "date from the schedule.");
300 Calendar calendar = parseCalendar(data.calendar());
301 if (calendar == NullCalendar())
302 WLOG("No calendar provided in Schedule, attempting to use a null calendar.");
303 Date startDate = parseDate(data.startDate());
304 Date endDate = data.endDate().empty() ? openEndDateReplacement : parseDate(data.endDate());
305 // Handle trivial case here
306 if (startDate == endDate)
307 return Schedule(vector<Date>(1, startDate), calendar);
308
309 QL_REQUIRE(startDate < endDate, "StartDate " << startDate << " is ahead of EndDate " << endDate);
310
311 Date firstDate = parseDate(data.firstDate());
312 Date lastDate = parseDate(data.lastDate());
313 Period tenor = parsePeriod(data.tenor());
314
315 // defaults
316 BusinessDayConvention bdc = ModifiedFollowing;
317 BusinessDayConvention bdcEnd = ModifiedFollowing;
318 DateGeneration::Rule rule = DateGeneration::Forward;
319 bool endOfMonth = false;
320
321 // now check the strings, if they are empty we take defaults
322 if (!data.convention().empty())
323 bdc = parseBusinessDayConvention(data.convention());
324 if (!data.termConvention().empty())
325 bdcEnd = parseBusinessDayConvention(data.termConvention());
326 else
327 bdcEnd = bdc; // except here
328
329 if (!data.endOfMonth().empty())
330 endOfMonth = parseBool(data.endOfMonth());
331
332 if (!data.rule().empty()) {
333
334 // handle special rules outside the QuantLib date generation rules
335
336 if (data.rule() == "EveryThursday") {
337 auto dates = everyThursdayDates(startDate, endDate, firstDate);
338 for (auto& d : dates)
339 d = calendar.adjust(d, bdc);
340 return Schedule(dates, calendar, bdc, bdcEnd, tenor, rule, endOfMonth);
341 }
342
343 // parse rule for further processing below
344
345 rule = parseDateGenerationRule(data.rule());
346 }
347
348 // handling of date generation rules that require special adjustments
349
350 if ((rule == DateGeneration::CDS || rule == DateGeneration::CDS2015) &&
351 (firstDate != Null<Date>() || lastDate != Null<Date>())) {
352 // Special handling of first date and last date in combination with CDS and CDS2015 rules:
353 // To be able to construct CDS schedules with front or back stub periods, we overwrite the
354 // first (last) date of the schedule built in QL with a given first (last) date
355 // The schedule builder in QL itself is not capable of doing this, it just throws an exception
356 // if a first (last) date is given in combination with a CDS / CDS2015 date generation rule.
357 std::vector<Date> dates = Schedule(startDate, endDate, tenor, calendar, bdc, bdcEnd, rule, endOfMonth).dates();
358 QL_REQUIRE(!dates.empty(),
359 "got empty CDS or CDS2015 schedule, startDate = " << startDate << ", endDate = " << endDate);
360 if (firstDate != Date())
361 dates.front() = firstDate;
362 if (lastDate != Date())
363 dates.back() = lastDate;
364 return Schedule(dates, calendar, bdc, bdcEnd, tenor, rule, endOfMonth);
365 }
366
367 // default handling (QuantLib scheduler)
368
369 return Schedule(startDate, endDate, tenor, calendar, bdc, bdcEnd, rule, endOfMonth, firstDate, lastDate);
370}

390Schedule makeSchedule(const ScheduleData& data, const Date& openEndDateReplacement, const map<string, QuantLib::Schedule>& baseSchedules) {
391 if(!data.hasData())
392 return Schedule();
393 // only the last rule-based schedule is allowed to have an open end date, check this
394 for (Size i = 1; i < data.rules().size(); ++i) {
395 QL_REQUIRE(!data.rules()[i - 1].endDate().empty(),
396 "makeSchedule(): only last schedule is allowed to have an open end date");
397 }
398 // build all the date and rule based sub-schedules we have
399 vector<Schedule> schedules;
400 for (auto& d : data.dates())
401 schedules.push_back(makeSchedule(d));
402 for (auto& r : data.rules())
403 schedules.push_back(makeSchedule(r, openEndDateReplacement));
404 if (!baseSchedules.empty())
405 for (auto& dv : data.derived()) {
406 auto baseSchedule = baseSchedules.find(dv.baseSchedule());
407 QL_REQUIRE(baseSchedule != baseSchedules.end(), "makeSchedule(): could not find base schedule \"" << dv.baseSchedule() << "\"");
408 schedules.push_back(makeSchedule(dv, baseSchedule->second));
409 }
410 QL_REQUIRE(!schedules.empty(), "No dates or rules to build Schedule from");
411 if (schedules.size() == 1)
412 // if we have just one, use that (most common case)
413 return schedules.front();
414 else {
415 // if we have multiple, combine them
416
417 // 1) sort by start date
418 std::sort(schedules.begin(), schedules.end(),
419 [](const Schedule& lhs, const Schedule& rhs) -> bool { return lhs.startDate() < rhs.startDate(); });
420
421 // 2) check if meta data is present, and if yes if it is consistent across schedules;
422 // the only exception is the term date convention, this is taken from the last schedule always
423 BusinessDayConvention convention = Null<BusinessDayConvention>(),
424 termConvention = Unadjusted; // initialization prevents gcc warning
425 Calendar calendar;
426 Period tenor;
427 DateGeneration::Rule rule = DateGeneration::Zero; // initialization prevents gcc warning
428 bool endOfMonth = false; // initialization prevents gcc warning
429 bool hasCalendar = false, hasConvention = false, hasTermConvention = false, hasTenor = false, hasRule = false,
430 hasEndOfMonth = false, hasConsistentCalendar = true, hasConsistentConvention = true,
431 hasConsistentTenor = true, hasConsistentRule = true, hasConsistentEndOfMonth = true;
432 for (auto& d : data.dates()) {
433 updateData<Calendar>(d.calendar(), calendar, hasCalendar, hasConsistentCalendar, parseCalendarTemp);
434 updateData<BusinessDayConvention>(d.convention(), convention, hasConvention, hasConsistentConvention,
435 parseBusinessDayConvention);
436 updateData<Period>(d.tenor(), tenor, hasTenor, hasConsistentTenor, parsePeriod);
437 }
438 for (auto& d : data.rules()) {
439 updateData<Calendar>(d.calendar(), calendar, hasCalendar, hasConsistentCalendar, parseCalendarTemp);
440 updateData<BusinessDayConvention>(d.convention(), convention, hasConvention, hasConsistentConvention,
441 parseBusinessDayConvention);
442 updateData<Period>(d.tenor(), tenor, hasTenor, hasConsistentTenor, parsePeriod);
443 updateData<bool>(d.endOfMonth(), endOfMonth, hasEndOfMonth, hasConsistentEndOfMonth, parseBool);
444 updateData<DateGeneration::Rule>(d.rule(), rule, hasRule, hasConsistentRule, parseDateGenerationRule);
445 if (d.termConvention() != "") {
446 hasTermConvention = true;
447 termConvention = parseBusinessDayConvention(d.termConvention());
448 }
449 }
450
451 // 3) combine dates and fill isRegular flag
452 const Schedule& s0 = schedules.front();
453 vector<Date> dates = s0.dates();
454 std::vector<bool> isRegular(s0.dates().size() - 1, false);
455 if (s0.hasIsRegular())
456 isRegular = s0.isRegular();
457 // will be removed, if next schedule's front date is matching the last date of current schedule
458 isRegular.push_back(false);
459 for (Size i = 1; i < schedules.size(); ++i) {
460 const Schedule& s = schedules[i];
461 QL_REQUIRE(dates.back() <= s.dates().front(), "Dates mismatch");
462 // if the end points match up, skip one to avoid duplicates, otherwise take both
463 Size offset = dates.back() == s.dates().front() ? 1 : 0;
464 isRegular.erase(isRegular.end() - offset,
465 isRegular.end()); // correct for superfluous flags from previous schedule
466 // add isRegular information, if available, otherwise assume irregular periods
467 if (s.hasIsRegular()) {
468 isRegular.insert(isRegular.end(), s.isRegular().begin(), s.isRegular().end());
469 } else {
470 for (Size ii = 0; ii < s.dates().size() - 1; ++ii)
471 isRegular.push_back(false);
472 }
473 if (i < schedules.size() - 1) {
474 // will be removed if next schedule's front date is matching last date of current schedule
475 isRegular.push_back(false);
476 }
477 // add the dates
478 dates.insert(dates.end(), s.dates().begin() + offset, s.dates().end());
479 }
480
481 // 4) Build schedule
482 return Schedule(dates, hasCalendar && hasConsistentCalendar ? calendar : NullCalendar(),
483 hasConvention && hasConsistentConvention ? convention : Unadjusted,
484 hasTermConvention ? boost::optional<BusinessDayConvention>(termConvention) : boost::none,
485 hasTenor && hasConsistentTenor ? boost::optional<Period>(tenor) : boost::none,
486 hasRule && hasConsistentRule ? boost::optional<DateGeneration::Rule>(rule) : boost::none,
487 hasEndOfMonth && hasConsistentEndOfMonth ? boost::optional<bool>(endOfMonth) : boost::none,
488 isRegular);
489 }
