60 class DiscountingForwardBondEngineBuilder : public fwdBondEngineBuilder {
61 public:
62  DiscountingForwardBondEngineBuilder()
63  : fwdBondEngineBuilder("DiscountedCashflows" /*the model; dont touch*/,
64  "DiscountingForwardBondEngine" /*the engine*/) {}
65 
66 protected:
67  virtual boost::shared_ptr<PricingEngine> engineImpl(const string& id, const Currency& ccy,
68  const string& creditCurveId, const string& securityId,
69  const string& referenceCurveId,
70  const string& incomeCurveId) override {
71 
72  string tsperiodStr = engineParameters_.at("TimestepPeriod");
73  Period tsperiod = parsePeriod(tsperiodStr);
74  Handle<YieldTermStructure> yts = market_->yieldCurve(referenceCurveId, configuration(MarketContext::pricing));
75 
76  Handle<YieldTermStructure> discountTS =
77  market_->discountCurve(ccy.code(), configuration(MarketContext::pricing));
78  // fall back on reference curve if no income curve is specified
79  Handle<YieldTermStructure> incomeTS = market_->yieldCurve(
80  incomeCurveId.empty() ? referenceCurveId : incomeCurveId, configuration(MarketContext::pricing));
81  Handle<DefaultProbabilityTermStructure> dpts;
82  // credit curve may not always be used. If credit curve ID is empty proceed without it
83  if (!creditCurveId.empty())
84  dpts = market_->defaultCurve(creditCurveId, configuration(MarketContext::pricing));
85  Handle<Quote> recovery;
86  try {
87  recovery = market_->recoveryRate(securityId, configuration(MarketContext::pricing));
88  } catch (...) {
89  // otherwise fall back on curve recovery
90  WLOG("security specific recovery rate not found for security ID "
91  << securityId << ", falling back on the recovery rate for credit curve Id " << creditCurveId);
92  if (!creditCurveId.empty())
93  recovery = market_->recoveryRate(creditCurveId, configuration(MarketContext::pricing));
94  }
95 
96  Handle<Quote> bondSpread;
97  try {
98  // spread is optional, pass empty handle to engine if not given (will be treated as 0 spread there)
99  bondSpread = market_->securitySpread(securityId, configuration(MarketContext::pricing));
100  } catch (...) {
101  }
102  return boost::make_shared<QuantExt::DiscountingForwardBondEngine>(discountTS, incomeTS, yts, bondSpread, dpts,
103  recovery, tsperiod);
104  }
105 };
