43FxTouchOption::FxTouchOption(Envelope& env, OptionData option, BarrierData barrier, string foreignCurrency,
44 string domesticCurrency, string payoffCurrency, double payoffAmount, string startDate,
45 string calendar, string fxIndex)
46 : ore::data::Trade("FxTouchOption", env),
47 FxSingleAssetDerivative("", env, foreignCurrency, domesticCurrency), option_(option),
48 barrier_(barrier), startDate_(startDate), calendar_(calendar), fxIndex_(fxIndex), payoffAmount_(payoffAmount),
49 payoffCurrency_(payoffCurrency) {
50 Barrier::Type barrierType = parseBarrierType(barrier_.type());
51 switch (barrierType) {
52 case Barrier::DownIn:
53 case Barrier::UpIn:
54 type_ = "One-Touch";
55 break;
56 case Barrier::DownOut:
57 case Barrier::UpOut:
58 type_ = "No-Touch";
59 break;
60 default:
61 QL_FAIL("unknown barrier type");
62 }
63}

65void FxTouchOption::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
66
67 const boost::shared_ptr<Market> market = engineFactory->market();
68
69 QL_REQUIRE(tradeActions().empty(), "TradeActions not supported for FxOption");
70 QL_REQUIRE(option_.exerciseDates().size() == 1, "Invalid number of excercise dates");
71 QL_REQUIRE(barrier_.levels().size() == 1, "Double barriers not supported for FxTouchOptions");
72 QL_REQUIRE(barrier_.style().empty() || barrier_.style() == "American", "Only american barrier style suppported");
73
74 // Parse trade data
75 Currency fgnCcy = parseCurrency(foreignCurrency_);
76 Currency domCcy = parseCurrency(domesticCurrency_);
77 Real level = barrier_.levels()[0].value();
78 Date expiryDate = parseDate(option_.exerciseDates().front());
79
80 Natural payLag = 0;
81 BusinessDayConvention payConvention = Unadjusted;
82 Calendar payCalendar = NullCalendar();
83 Date payDate = expiryDate;
84 const boost::optional<OptionPaymentData>& opd = option_.paymentData();
85 if (opd) {
86 if (opd->rulesBased()) {
87 payLag = opd->lag();
88 payConvention = opd->convention();
89 payCalendar = opd->calendar();
90 payDate = payCalendar.advance(expiryDate, opd->lag(), Days, opd->convention());
91 } else {
92 if (opd->dates().size() > 1)
93 WLOG(ore::data::StructuredTradeWarningMessage(
94 id(), tradeType(), "Trade build", "Found more than 1 payment date. The first one will be used."));
95 payDate = opd->dates().front();
96 }
97 }
98 QL_REQUIRE(payDate >= expiryDate, "Settlement date cannot be earlier than expiry date");
99
100 Barrier::Type barrierType = parseBarrierType(barrier_.type());
101 Option::Type type;
102 if (barrierType == Barrier::DownIn || barrierType == Barrier::DownOut)
103 type = Option::Type::Put;
104 else
105 type = Option::Type::Call;
106 option_.setCallPut(to_string(type));
107 bool payoffAtExpiry = option_.payoffAtExpiry();
108 Real rebate = barrier_.rebate();
109 Position::Type positionType = parsePositionType(option_.longShort());
110 Date start = ore::data::parseDate(startDate_);
111
112 QL_REQUIRE(rebate == 0, "Rebates not supported for FxTouchOptions");
113 QL_REQUIRE(payoffAtExpiry == true || barrierType == Barrier::Type::DownIn || barrierType == Barrier::Type::UpIn,
114 "Payoff at hit not supported for FxNoTouchOptions");
115 if ((barrierType == Barrier::Type::DownIn || barrierType == Barrier::Type::UpIn) && payoffAtExpiry == false)
116 QL_REQUIRE(
117 !opd || (opd->rulesBased() && opd->relativeTo() == OptionPaymentData::RelativeTo::Exercise),
118 "Option payment data must be rules-based and relative to Exercise for FxOneTouchOption with payoff at hit");
119
120 // Handle PayoffCurrency, we might have to flip the trade here
121 bool flipResults = false;
122 if (payoffCurrency_ == foreignCurrency_) {
123 // Invert the trade, switch dom and for and flip Put/Call
124 level = 1.0 / level;
125 std::swap(fgnCcy, domCcy);
126 type = type == Option::Call ? Option::Put : Option::Call;
127 switch (barrierType) {
128 case Barrier::DownIn:
129 barrierType = Barrier::UpIn;
130 break;
131 case Barrier::UpIn:
132 barrierType = Barrier::DownIn;
133 break;
134 case Barrier::DownOut:
135 barrierType = Barrier::UpOut;
136 break;
137 case Barrier::UpOut:
138 barrierType = Barrier::DownOut;
139 break;
140 }
141 flipResults = true;
142 } else if (payoffCurrency_ != domesticCurrency_) {
143 QL_FAIL("Invalid Payoff currency (" << payoffCurrency_ << ") for FxTouchOption " << foreignCurrency_
144 << domesticCurrency_);
145 }
146 DLOG("Setting up FxTouchOption with level " << level << " foreign/bought " << fgnCcy << " domestic/sold "
147 << domCcy);
148 // from this point on it's important not to use domesticCurrency_, foreignCurrency_, barrier_.level(), etc
149 // rather the local variables (fgnCcy, domCcy, level, etc) should be used as they may have been flipped.
150
151 boost::shared_ptr<QuantExt::FxIndex> fxIndex;
152 if (!fxIndex_.empty())
153 fxIndex = buildFxIndex(fxIndex_, domCcy.code(), fgnCcy.code(), engineFactory->market(),
154 engineFactory->configuration(MarketContext::pricing));
155 Calendar cal = ore::data::parseCalendar(calendar_);
156
157 auto buildBarrierOptionWrapperInstr = [this, type, level, engineFactory, domCcy, fgnCcy, flipResults, positionType,
158 market, barrierType, rebate, fxIndex, cal,
159 start](const Date& expiryDate, const Date& payDate) {
160 boost::shared_ptr<StrikedTypePayoff> payoff(new CashOrNothingPayoff(type, level, 1.0));
161 Leg leg;
162
163 leg.push_back(boost::shared_ptr<CashFlow>(new SimpleCashFlow(1.0, payDate)));
164 // Hard code payoff at expiry to true - we ignore in pricing; QPR-10669
165 bool payoffFlag = true;
166
167 boost::shared_ptr<Exercise> exercise = boost::make_shared<AmericanExercise>(expiryDate, payoffFlag);
168
169 boost::shared_ptr<Instrument> barrier = boost::make_shared<VanillaOption>(payoff, exercise);
170 boost::shared_ptr<Instrument> underlying = boost::make_shared<Swap>(Leg(), leg);
171
172 // set pricing engines
173 boost::shared_ptr<EngineBuilder> builder = engineFactory->builder(tradeType_);
174 QL_REQUIRE(builder, "No builder found for " << tradeType_);
175 boost::shared_ptr<FxTouchOptionEngineBuilder> fxTouchOptBuilder =
176 boost::dynamic_pointer_cast<FxTouchOptionEngineBuilder>(builder);
177 barrier->setPricingEngine(fxTouchOptBuilder->engine(fgnCcy, domCcy, type_, payDate, flipResults));
178 if (type_ == "One-Touch") {
179 // if a one-touch option is triggered it becomes a simple forward cashflow
180 // which we price as a swap
181 builder = engineFactory->builder("Swap");
182 QL_REQUIRE(builder, "No builder found for Swap");
183 boost::shared_ptr<SwapEngineBuilderBase> swapBuilder =
184 boost::dynamic_pointer_cast<SwapEngineBuilderBase>(builder);
185 underlying->setPricingEngine(swapBuilder->engine(domCcy));
186 }
187
188 bool isLong = (positionType == Position::Long) ? true : false;
189
190 std::vector<boost::shared_ptr<Instrument>> additionalInstruments;
191 std::vector<Real> additionalMultipliers;
192 Date lastPremiumDate = addPremiums(additionalInstruments, additionalMultipliers, payoffAmount_,
193 option_.premiumData(), isLong ? -1.0 : 1.0, parseCurrency(payoffCurrency_),
194 engineFactory, builder->configuration(MarketContext::pricing));
195
196 Handle<Quote> spot = market->fxRate(fgnCcy.code() + domCcy.code());
197
198 auto barrierOptionWrapper = boost::make_shared<SingleBarrierOptionWrapper>(
199 barrier, isLong, expiryDate, false, underlying, barrierType, spot, level, rebate, domCcy, start, fxIndex,
200 cal, payoffAmount_, payoffAmount_, additionalInstruments, additionalMultipliers);
201
202 maturity_ = std::max(lastPremiumDate, payDate);
203
204 return barrierOptionWrapper;
205 };
206
207 auto barrierOptionWrapper = buildBarrierOptionWrapperInstr(expiryDate, payDate);
208
209 // We make sure to add required fixings before checking for modifying the instrument's expiry date, to make sure the
210 // portfolio-analyzer gets all the fixings needed for the instrument at the first evaluation.
211 Calendar fixingCal = fxIndex ? fxIndex->fixingCalendar() : cal;
212 if (start != Date()) {
213 for (Date d = start; d <= expiryDate; d = fixingCal.advance(d, 1 * Days))
214 requiredFixings_.addFixingDate(d, fxIndex_, payDate);
215 }
216
217 // Check if the barrier has been triggered already. If payoff-at-hit, and barrier was touched in the past, then
218 // create instrument again, with expiry date and pay date corresponding to that past barrier exercise date
219 if (auto rt = engineFactory->engineData()->globalParameters().find("RunType");
220 rt != engineFactory->engineData()->globalParameters().end() && rt->second != "PortfolioAnalyser" &&
221 barrierOptionWrapper->exercise()) {
222 QL_REQUIRE(barrierOptionWrapper->exerciseDate() != Date(), "Option is exercised but exercise date was not defined");
223 expiryDate = barrierOptionWrapper->exerciseDate();
224 additionalData_["exerciseDate"] = expiryDate;
225
226 if (!payoffAtExpiry && type_ == "One-Touch") {
227 payDate = payCalendar.advance(expiryDate, payLag, Days, payConvention);
228 barrierOptionWrapper = buildBarrierOptionWrapperInstr(expiryDate, payDate);
229 additionalData_["settlementDate"] = payDate;
230 }
231 }
232
233 instrument_ = barrierOptionWrapper;
234 npvCurrency_ = payoffCurrency_;
235 notional_ = payoffAmount_;
236 notionalCurrency_ = payoffCurrency_;
237 // maturity_ is set in buildBarrierOptionWrapperInstr()
238
239 additionalData_["payoffAmount"] = payoffAmount_;
240 additionalData_["payoffCurrency"] = payoffCurrency_;
241
242 // ISDA taxonomy
243 additionalData_["isdaAssetClass"] = string("Foreign Exchange");
244 additionalData_["isdaBaseProduct"] = string("Simple Exotic");
245 additionalData_["isdaSubProduct"] = string("Barrier");
246 additionalData_["isdaTransaction"] = string("");
247}

249bool FxTouchOption::checkBarrier(Real spot, Barrier::Type type, Real barrier) {
250 switch (type) {
251 case Barrier::DownIn:
252 case Barrier::DownOut:
253 return spot <= barrier;
254 case Barrier::UpIn:
255 case Barrier::UpOut:
256 return spot >= barrier;
257 default:
258 QL_FAIL("unknown barrier type " << type);
259 }
260}

262void FxTouchOption::fromXML(XMLNode* node) {
263 Trade::fromXML(node);
264 XMLNode* fxNode = XMLUtils::getChildNode(node, "FxTouchOptionData");
265 QL_REQUIRE(fxNode, "No FxOptionData Node");
266 option_.fromXML(XMLUtils::getChildNode(fxNode, "OptionData"));
267 barrier_.fromXML(XMLUtils::getChildNode(fxNode, "BarrierData"));
268 Barrier::Type barrierType = parseBarrierType(barrier_.type());
269 switch (barrierType) {
270 case Barrier::DownIn:
271 case Barrier::UpIn:
272 type_ = "One-Touch";
273 break;
274 case Barrier::DownOut:
275 case Barrier::UpOut:
276 type_ = "No-Touch";
277 break;
278 default:
279 QL_FAIL("unknown barrier type");
280 }
281
282 foreignCurrency_ = XMLUtils::getChildValue(fxNode, "ForeignCurrency", true);
283 domesticCurrency_ = XMLUtils::getChildValue(fxNode, "DomesticCurrency", true);
284 payoffCurrency_ = XMLUtils::getChildValue(fxNode, "PayoffCurrency", true);
285 startDate_ = XMLUtils::getChildValue(fxNode, "StartDate", false);
286 calendar_ = XMLUtils::getChildValue(fxNode, "Calendar", false);
287 fxIndex_ = XMLUtils::getChildValue(fxNode, "FXIndex", false);
288 payoffAmount_ = XMLUtils::getChildValueAsDouble(fxNode, "PayoffAmount", true);
289}

291XMLNode* FxTouchOption::toXML(XMLDocument& doc) {
292 XMLNode* node = Trade::toXML(doc);
293 XMLNode* fxNode = doc.allocNode("FxTouchOptionData");
294 XMLUtils::appendNode(node, fxNode);
295 XMLUtils::appendNode(fxNode, option_.toXML(doc));
296 XMLUtils::appendNode(fxNode, barrier_.toXML(doc));
297 XMLUtils::addChild(doc, fxNode, "ForeignCurrency", foreignCurrency_);
298 XMLUtils::addChild(doc, fxNode, "DomesticCurrency", domesticCurrency_);
299 XMLUtils::addChild(doc, fxNode, "PayoffCurrency", payoffCurrency_);
300 XMLUtils::addChild(doc, fxNode, "PayoffAmount", payoffAmount_);
301 if (startDate_ != "")
302 XMLUtils::addChild(doc, fxNode, "StartDate", startDate_);
303 if (fxIndex_ != "")
304 XMLUtils::addChild(doc, fxNode, "FXIndex", fxIndex_);
305 if (calendar_ != "")
306 XMLUtils::addChild(doc, fxNode, "Calendar", calendar_);
307
308 return node;
309}
