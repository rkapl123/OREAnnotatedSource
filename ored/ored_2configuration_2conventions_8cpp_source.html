53 void checkContinuationMappings(const map<Natural, Natural>& mp, const string& name) {
54 
55  Natural previousValue = 0;
56  for (const auto& kv : mp) {
57  QL_REQUIRE(kv.first <= kv.second, "Not allowed a " << name << " continuation mapping where From (" <<
58  kv.first << ") is greater than To (" << kv.second << ").");
59  QL_REQUIRE(kv.second > previousValue, "The To " << name << " continuation mappings should be strictly " <<
60  "increasing but got " << kv.second << " <= " << previousValue);
61  previousValue = kv.second;
62  }
63 }

367 void IborIndexConvention::build() {
368  // just a check really that the id is consistent with the ibor index name rules
369  vector<string> tokens;
370  split(tokens, localId_, boost::is_any_of("-"));
371  QL_REQUIRE(tokens.size() == 2 || tokens.size() == 3,
372  "Two or three tokens required in IborIndexConvention " << localId_ << ": CCY-INDEX or CCY-INDEX-TERM");
373 
374  // set the Id - this converts the local id term from "7D" to "1W", "28D" to "1M" etc, so if can be picked
375  // up by searches
376  id_ = (tokens.size() == 3) ? (tokens[0] + "-" + tokens[1] + "-" + to_string(parsePeriod(tokens[2]))) : localId_;
377 }
