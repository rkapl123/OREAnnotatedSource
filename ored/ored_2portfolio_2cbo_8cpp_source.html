36void CBO::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
37 DLOG("CBO::build() called for trade " << id());
38
39 requiredFixings_.clear();
40
41 const boost::shared_ptr<Market> market = engineFactory->market();
42 boost::shared_ptr<EngineBuilder> builder = engineFactory->builder("CBO");
43
44 populateFromCboReferenceData(engineFactory->referenceData());
45 validateCbo();
46
47 Schedule schedule = makeSchedule(scheduleData_);
48
49 bondbasket_ = bondbasketdata_.build(engineFactory, parseCurrency(ccy_), reinvestmentEndDate_);
50 requiredFixings_.addData(bondbasketdata_.requiredFixings());
51
52 vector<QuantExt::Tranche> tranches;
53 size_t investCount = 0;
54 size_t idx = 0;
55 multiplier_ = 1.0;
56 for(size_t i = 0; i < trancheData_.size(); i++){
57
58 QuantExt::Tranche tranche;
59
60 tranche.name = trancheData_[i]->name();;
61 tranche.faceAmount = trancheData_[i]->faceAmount();
62 tranche.icRatio = trancheData_[i]->icRatio();
63 tranche.ocRatio = trancheData_[i]->ocRatio();
64
65 LegData legdata = LegData();
66 legdata.notionals() = {trancheData_[i]->faceAmount()};
67 legdata.schedule() = scheduleData_;
68 legdata.dayCounter() = daycounter_;
69 legdata.concreteLegData() = trancheData_[i]->concreteLegData();
70 legdata.paymentConvention() = paymentConvention_;
71
72 auto legBuilder = engineFactory->legBuilder(trancheData_[i]->concreteLegData()->legType());
73 RequiredFixings requiredFixingsLeg;
74 auto configuration = builder->configuration(MarketContext::pricing);
75 Leg leg = legBuilder->buildLeg(legdata, engineFactory, requiredFixingsLeg, configuration);
76 tranche.leg = leg;
77
78 requiredFixings_.addData(requiredFixingsLeg);
79 tranches.push_back(tranche);
80
81 //check if invested tranche
82 if(tranche.name == investedTrancheName_){
83 idx = i; //Set for leg output
84 multiplier_ = investedNotional_ / tranche.faceAmount;
85 if(multiplier_ > 1.0)
86 ALOG("Ratio bigger than 1 : investment=" << investedNotional_ << " vs. faceAmount=" << tranche.faceAmount);
87 investCount++;
88 }
89 }
90 //check if tranche name could be found...
91 QL_REQUIRE(investCount == 1, "Could not assign CBOInvestment TrancheName " << investedTrancheName_ << " to Names of CBOTranches.");
92
93 // CHECK DATES BONDS vs TRANCHES
94 Date longestBondDate = Settings::instance().evaluationDate();
95 for (const auto& bond : bondbasket_->bonds()) {
96 Date bondMat = bond.second->maturityDate();
97 if (bondMat > longestBondDate)
98 longestBondDate = bondMat;
99 }
100 QL_REQUIRE(schedule.endDate() > longestBondDate, " Tranche Maturity should be after Bond Maturity: Bond "
101 << longestBondDate << " vs. Tranche " << schedule.endDate());
102
103 // set QuantExt instrument
104 boost::shared_ptr<QuantExt::CBO> cbo =
105 boost::make_shared<QuantExt::CBO>(bondbasket_, schedule, parseReal(seniorFee_), parseDayCounter(feeDayCounter_),
106 tranches, parseReal(subordinatedFee_), parseReal(equityKicker_),
107 parseCurrency(ccy_), investedTrancheName_);
108 // set pricing engine...
109 boost::shared_ptr<CboMCEngineBuilder> cboBuilder = boost::dynamic_pointer_cast<CboMCEngineBuilder>(builder);
110 QL_REQUIRE(cboBuilder, "No Builder found for CBO: " << id());
111 cbo->setPricingEngine(cboBuilder->engine(bondbasket_->pool()));
112 instrument_.reset(new VanillaInstrument(cbo, multiplier_));
113
114 maturity_ = schedule.endDate();
115 npvCurrency_ = ccy_;
116 notional_ = investedNotional_;
117 legs_ = vector<QuantLib::Leg>(1, tranches[idx].leg);
118 legCurrencies_ = vector<string>(1, ccy_);
119 legPayers_ = vector<bool>(1, false);
120
121 // sensitivity...
122 for (const auto& bond : bondBasketData().bonds()) {
123 cbo->registerWith(securitySpecificCreditCurve(market, bond->bondData().securityId(),
124 bond->bondData().creditCurveId()));
125 cbo->registerWith(bond->instrument()->qlInstrument());
126
127 // to capture floater...
128 Leg bondleg = bond->legs().front();
129 for (auto const& c : bondleg)
130 cbo->registerWith(c);
131 }
132
133 std::map<string, boost::shared_ptr<QuantExt::FxIndex>> fxIndexMap = bondbasket_->fxIndexMap();
134 std::map<string, boost::shared_ptr<QuantExt::FxIndex>>::iterator it;
135 for (it = fxIndexMap.begin(); it != fxIndexMap.end(); ++it)
136 cbo->registerWith(it->second);
137
138 // ISDA taxonomy: not a derivative, but define the asset class at least
139 // so that we can determine a TRS asset class that has CBO underlyings
140 additionalData_["isdaAssetClass"] = string("Credit");
141 additionalData_["isdaBaseProduct"] = string("");
142 additionalData_["isdaSubProduct"] = string("");
143 additionalData_["isdaTransaction"] = string("");
144}

159void CBO::fromXML(XMLNode* node) {
160
161 Trade::fromXML(node);
162 XMLNode* cboData = XMLUtils::getChildNode(node, "CBOData");
163 QL_REQUIRE(cboData, "expected node CBOData");
164
165 //investment
166 XMLNode* cboInvestment = XMLUtils::getChildNode(cboData, "CBOInvestment");
167 QL_REQUIRE(cboInvestment, "expected node CBOInvestment");
168
169 investedTrancheName_ = XMLUtils::getChildValue(cboInvestment, "TrancheName", true);
170 investedNotional_ = XMLUtils::getChildValueAsDouble(cboInvestment, "Notional", true);
171 structureId_ = XMLUtils::getChildValue(cboInvestment, "StructureId", true);
172
173 //structure
174 XMLNode* cboStructure = XMLUtils::getChildNode(cboData, "CBOStructure");
175 //QL_REQUIRE(cboStructure, "expected node CBOStructure");
176 if(cboStructure){
177 daycounter_ = XMLUtils::getChildValue(cboStructure, "DayCounter", false);
178 paymentConvention_ = XMLUtils::getChildValue(cboStructure, "PaymentConvention", false);
179 ccy_ = XMLUtils::getChildValue(cboStructure, "Currency", false);
180 seniorFee_ = XMLUtils::getChildValue(cboStructure, "SeniorFee", false);
181 subordinatedFee_ = XMLUtils::getChildValue(cboStructure, "SubordinatedFee", false);
182 equityKicker_ = XMLUtils::getChildValue(cboStructure, "EquityKicker", false);
183 feeDayCounter_ = XMLUtils::getChildValue(cboStructure, "FeeDayCounter", false);
184 reinvestmentEndDate_ = XMLUtils::getChildValue(cboStructure, "ReinvestmentEndDate", false, "");
185
186 scheduleData_ = ScheduleData();
187 XMLNode* scheduleData = XMLUtils::getChildNode(cboStructure, "ScheduleData");
188 if (scheduleData)
189 scheduleData_.fromXML(scheduleData);
190
191 bondbasketdata_.clear();
192 XMLNode* bondbasketNode = XMLUtils::getChildNode(cboStructure, "BondBasketData");
193 if(bondbasketNode)
194 bondbasketdata_.fromXML(bondbasketNode);
195
196 trancheData_.clear();
197 XMLNode* tranchesNode = XMLUtils::getChildNode(cboStructure, "CBOTranches");
198 if(tranchesNode){
199 for (XMLNode* child = XMLUtils::getChildNode(tranchesNode, "Tranche"); child; child = XMLUtils::getNextSibling(child)) {
200 auto data = boost::make_shared<ore::data::TrancheData>();
201 data->fromXML(child);
202 trancheData_.push_back(data);
203 }
204 }
205 }

249void CBOTrsUnderlyingBuilder::build(
250 const std::string& parentId, const boost::shared_ptr<Trade>& underlying, const std::vector<Date>& valuationDates,
251 const std::vector<Date>& paymentDates, const std::string& fundingCurrency,
252 const boost::shared_ptr<EngineFactory>& engineFactory, boost::shared_ptr<QuantLib::Index>& underlyingIndex,
253 Real& underlyingMultiplier, std::map<std::string, double>& indexQuantities,
254 std::map<std::string, boost::shared_ptr<QuantExt::FxIndex>>& fxIndices, Real& initialPrice,
255 std::string& assetCurrency, std::string& creditRiskCurrency,
256 std::map<std::string, SimmCreditQualifierMapping>& creditQualifierMapping, Date& maturity,
257 const std::function<boost::shared_ptr<QuantExt::FxIndex>(
258 const boost::shared_ptr<Market> market, const std::string& configuration, const std::string& domestic,
259 const std::string& foreign, std::map<std::string, boost::shared_ptr<QuantExt::FxIndex>>& fxIndices)>&
260 getFxIndex,
261 const std::string& underlyingDerivativeId, RequiredFixings& fixings, std::vector<Leg>& returnLegs) const {
262 auto t = boost::dynamic_pointer_cast<ore::data::CBO>(underlying);
263 QL_REQUIRE(t, "could not cast to ore::data::CBO, this is unexpected");
264 string indexName = "GENERIC-" + t->investedTrancheName();
265 IndexNameTranslator::instance().add(indexName, indexName);
266 indexQuantities[indexName] = t->underlyingMultiplier();
267 underlyingIndex = boost::make_shared<QuantExt::GenericIndex>(indexName);
268 underlyingMultiplier = t->underlyingMultiplier();
269 assetCurrency = t->npvCurrency();
270 creditRiskCurrency = t->npvCurrency();
271
272 auto fxIndex = getFxIndex(engineFactory->market(), engineFactory->configuration(MarketContext::pricing),
273 assetCurrency, fundingCurrency, fxIndices);
274 returnLegs.push_back(QuantExt::TRSLeg(valuationDates, paymentDates, underlyingMultiplier, underlyingIndex, fxIndex)
275 .withInitialPrice(initialPrice));
276
277 //fill the SimmCreditQualifierMapping
278 auto bonds = t->bondBasketData().bonds();
279 for (Size i = 0; i < bonds.size(); ++i) {
280 creditQualifierMapping[ore::data::securitySpecificCreditCurveName(bonds[i]->bondData().securityId(),
281 bonds[i]->bondData().creditCurveId())] =
282 SimmCreditQualifierMapping(bonds[i]->bondData().securityId(), bonds[i]->bondData().creditGroup());
283 creditQualifierMapping[bonds[i]->bondData().creditCurveId()] =
284 SimmCreditQualifierMapping(bonds[i]->bondData().securityId(), bonds[i]->bondData().creditGroup());
285 }

364void CBO::populateFromCboReferenceData(const boost::shared_ptr<CboReferenceDatum>& cboReferenceDatum){
365
366 DLOG("populating data cbo from reference data");
367 QL_REQUIRE(cboReferenceDatum, "populateFromCboReferenceData(): empty cbo reference datum given");
368
369 if (seniorFee_.empty()) {
370 seniorFee_ = cboReferenceDatum->cbostructure().seniorFee;
371 TLOG("overwrite SeniorFee with '" << seniorFee_ << "'");
372 }
373
374 if (subordinatedFee_.empty()) {
375 subordinatedFee_ = cboReferenceDatum->cbostructure().subordinatedFee;
376 TLOG("overwrite SubordinatedFee with '" << subordinatedFee_ << "'");
377 }
378
379 if (equityKicker_.empty()) {
380 equityKicker_ = cboReferenceDatum->cbostructure().equityKicker;
381 TLOG("overwrite EquityKicker with '" << equityKicker_ << "'");
382 }
383
384 if (feeDayCounter_.empty()) {
385 feeDayCounter_ = cboReferenceDatum->cbostructure().feeDayCounter;
386 TLOG("overwrite FeeDayCounter with '" << feeDayCounter_ << "'");
387 }
388
389 if (ccy_.empty()) {
390 ccy_ = cboReferenceDatum->cbostructure().ccy;
391 TLOG("overwrite currency with '" << ccy_ << "'");
392 }
393
394 if (reinvestmentEndDate_.empty()) {
395 reinvestmentEndDate_ = cboReferenceDatum->cbostructure().reinvestmentEndDate;
396 TLOG("overwrite ReinvestmentEndDate with '" << reinvestmentEndDate_ << "'");
397 }
398
399 if (daycounter_.empty()) {
400 daycounter_ = cboReferenceDatum->cbostructure().daycounter;
401 TLOG("overwrite DayCounter with '" << daycounter_ << "'");
402 }
403
404 if (paymentConvention_.empty()) {
405 paymentConvention_ = cboReferenceDatum->cbostructure().paymentConvention;
406 TLOG("overwrite PaymentConvention with '" << paymentConvention_ << "'");
407 }
408
409 if (!scheduleData_.hasData()) {
410 scheduleData_ = cboReferenceDatum->cbostructure().scheduleData;
411 TLOG("overwrite ScheduleData");
412 }
413
414 if (bondbasketdata_.empty()) {
415 bondbasketdata_ = cboReferenceDatum->cbostructure().bondbasketdata;
416 TLOG("overwrite BondBasketData");
417 }
418
419 if (trancheData_.empty()) {
420 trancheData_ = cboReferenceDatum->cbostructure().trancheData;
421 TLOG("overwrite TrancheData");
422 }

426void CBO::validateCbo(){
427
428 //Check for mandatroy fields
429 //reinvestmentEndDate_ not required
430 //TrancheName, Notional, StructuredID already checked in fromXML
431
432 std::string missing = "";
433
434 if (seniorFee_.empty())
435 missing.append("SeniorFee ");
436
437 if (subordinatedFee_.empty())
438 missing.append("SubordinatedFee ");
439
440 if (equityKicker_.empty())
441 missing.append("EquityKicker ");
442
443 if (feeDayCounter_.empty())
444 missing.append("FeeDayCounter ");
445
446 if (ccy_.empty())
447 missing.append("Currency ");
448
449 if (daycounter_.empty())
450 missing.append("DayCounter ");
451
452 if (paymentConvention_.empty())
453 missing.append("PaymentConvention ");
454
455 if (!scheduleData_.hasData())
456 missing.append("ScheduleData ");
457
458 if (bondbasketdata_.empty())
459 missing.append("BondBasketData ");
460
461 if (trancheData_.empty())
462 missing.append("TrancheData ");
463
464 if(!missing.empty())
465 QL_FAIL("CBO " << structureId_ << " expects " + missing +"elements");
466
467}
