35void CBO::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
36 DLOG("CBO::build() called for trade " << id());
37
38 requiredFixings_.clear();
39
40 const boost::shared_ptr<Market> market = engineFactory->market();
41 boost::shared_ptr<EngineBuilder> builder = engineFactory->builder("CBO");
42
43 populateFromCboReferenceData(engineFactory->referenceData());
44 validateCbo();
45
46 Schedule schedule = makeSchedule(scheduleData_);
47
48 bondbasket_ = bondbasketdata_.build(engineFactory, parseCurrency(ccy_), reinvestmentEndDate_);
49 requiredFixings_.addData(bondbasketdata_.requiredFixings());
50
51 vector<QuantExt::Tranche> tranches;
52 size_t investCount = 0;
53 size_t idx = 0;
54 multiplier_ = 1.0;
55 for(size_t i = 0; i < trancheData_.size(); i++){
56
57 QuantExt::Tranche tranche;
58
59 tranche.name = trancheData_[i]->name();;
60 tranche.faceAmount = trancheData_[i]->faceAmount();
61 tranche.icRatio = trancheData_[i]->icRatio();
62 tranche.ocRatio = trancheData_[i]->ocRatio();
63
64 LegData legdata = LegData();
65 legdata.notionals() = {trancheData_[i]->faceAmount()};
66 legdata.schedule() = scheduleData_;
67 legdata.dayCounter() = daycounter_;
68 legdata.concreteLegData() = trancheData_[i]->concreteLegData();
69 legdata.paymentConvention() = paymentConvention_;
70
71 auto legBuilder = engineFactory->legBuilder(trancheData_[i]->concreteLegData()->legType());
72 RequiredFixings requiredFixingsLeg;
73 auto configuration = builder->configuration(MarketContext::pricing);
74 Leg leg = legBuilder->buildLeg(legdata, engineFactory, requiredFixingsLeg, configuration);
75 tranche.leg = leg;
76
77 requiredFixings_.addData(requiredFixingsLeg);
78 tranches.push_back(tranche);
79
80 //check if invested tranche
81 if(tranche.name == investedTrancheName_){
82 idx = i; //Set for leg output
83 multiplier_ = investedNotional_ / tranche.faceAmount;
84 if(multiplier_ > 1.0)
85 ALOG("Ratio bigger than 1 : investment=" << investedNotional_ << " vs. faceAmount=" << tranche.faceAmount);
86 investCount++;
87 }
88 }
89 //check if tranche name could be found...
90 QL_REQUIRE(investCount == 1, "Could not assign CBOInvestment TrancheName " << investedTrancheName_ << " to Names of CBOTranches.");
91
92 // CHECK DATES BONDS vs TRANCHES
93 Date longestBondDate = Settings::instance().evaluationDate();
94 for (const auto& bond : bondbasket_->bonds()) {
95 Date bondMat = bond.second->maturityDate();
96 if (bondMat > longestBondDate)
97 longestBondDate = bondMat;
98 }
99 QL_REQUIRE(schedule.endDate() > longestBondDate, " Tranche Maturity should be after Bond Maturity: Bond "
100 << longestBondDate << " vs. Tranche " << schedule.endDate());
101
102 // set QuantExt instrument
103 boost::shared_ptr<QuantExt::CBO> cbo =
104 boost::make_shared<QuantExt::CBO>(bondbasket_, schedule, parseReal(seniorFee_), parseDayCounter(feeDayCounter_),
105 tranches, parseReal(subordinatedFee_), parseReal(equityKicker_),
106 parseCurrency(ccy_), investedTrancheName_);
107 // set pricing engine...
108 boost::shared_ptr<CboMCEngineBuilder> cboBuilder = boost::dynamic_pointer_cast<CboMCEngineBuilder>(builder);
109 QL_REQUIRE(cboBuilder, "No Builder found for CBO: " << id());
110 cbo->setPricingEngine(cboBuilder->engine(bondbasket_->pool()));
111 instrument_.reset(new VanillaInstrument(cbo, multiplier_));
112
113 maturity_ = schedule.endDate();
114 npvCurrency_ = ccy_;
115 notional_ = investedNotional_;
116 legs_ = vector<QuantLib::Leg>(1, tranches[idx].leg);
117 legCurrencies_ = vector<string>(1, ccy_);
118 legPayers_ = vector<bool>(1, false);
119
120 // sensitivity...
121 for (const auto& bond : bondBasketData().bonds()) {
122 cbo->registerWith(securitySpecificCreditCurve(market, bond->bondData().securityId(),
123 bond->bondData().creditCurveId()));
124 cbo->registerWith(bond->instrument()->qlInstrument());
125
126 // to capture floater...
127 Leg bondleg = bond->legs().front();
128 for (auto const& c : bondleg)
129 cbo->registerWith(c);
130 }
131
132 std::map<string, boost::shared_ptr<QuantExt::FxIndex>> fxIndexMap = bondbasket_->fxIndexMap();
133 std::map<string, boost::shared_ptr<QuantExt::FxIndex>>::iterator it;
134 for (it = fxIndexMap.begin(); it != fxIndexMap.end(); ++it)
135 cbo->registerWith(it->second);
136
137 // ISDA taxonomy: not a derivative, but define the asset class at least
138 // so that we can determine a TRS asset class that has CBO underlyings
139 additionalData_["isdaAssetClass"] = string("Credit");
140 additionalData_["isdaBaseProduct"] = string("");
141 additionalData_["isdaSubProduct"] = string("");
142 additionalData_["isdaTransaction"] = string("");
143}

158void CBO::fromXML(XMLNode* node) {
159
160 Trade::fromXML(node);
161 XMLNode* cboData = XMLUtils::getChildNode(node, "CBOData");
162 QL_REQUIRE(cboData, "expected node CBOData");
163
164 //investment
165 XMLNode* cboInvestment = XMLUtils::getChildNode(cboData, "CBOInvestment");
166 QL_REQUIRE(cboInvestment, "expected node CBOInvestment");
167
168 investedTrancheName_ = XMLUtils::getChildValue(cboInvestment, "TrancheName", true);
169 investedNotional_ = XMLUtils::getChildValueAsDouble(cboInvestment, "Notional", true);
170 structureId_ = XMLUtils::getChildValue(cboInvestment, "StructureId", true);
171
172 //structure
173 XMLNode* cboStructure = XMLUtils::getChildNode(cboData, "CBOStructure");
174 //QL_REQUIRE(cboStructure, "expected node CBOStructure");
175 if(cboStructure){
176 daycounter_ = XMLUtils::getChildValue(cboStructure, "DayCounter", false);
177 paymentConvention_ = XMLUtils::getChildValue(cboStructure, "PaymentConvention", false);
178 ccy_ = XMLUtils::getChildValue(cboStructure, "Currency", false);
179 seniorFee_ = XMLUtils::getChildValue(cboStructure, "SeniorFee", false);
180 subordinatedFee_ = XMLUtils::getChildValue(cboStructure, "SubordinatedFee", false);
181 equityKicker_ = XMLUtils::getChildValue(cboStructure, "EquityKicker", false);
182 feeDayCounter_ = XMLUtils::getChildValue(cboStructure, "FeeDayCounter", false);
183 reinvestmentEndDate_ = XMLUtils::getChildValue(cboStructure, "ReinvestmentEndDate", false, "");
184
185 scheduleData_ = ScheduleData();
186 XMLNode* scheduleData = XMLUtils::getChildNode(cboStructure, "ScheduleData");
187 if (scheduleData)
188 scheduleData_.fromXML(scheduleData);
189
190 bondbasketdata_.clear();
191 XMLNode* bondbasketNode = XMLUtils::getChildNode(cboStructure, "BondBasketData");
192 if(bondbasketNode)
193 bondbasketdata_.fromXML(bondbasketNode);
194
195 trancheData_.clear();
196 XMLNode* tranchesNode = XMLUtils::getChildNode(cboStructure, "CBOTranches");
197 if(tranchesNode){
198 for (XMLNode* child = XMLUtils::getChildNode(tranchesNode, "Tranche"); child; child = XMLUtils::getNextSibling(child)) {
199 auto data = boost::make_shared<ore::data::TrancheData>();
200 data->fromXML(child);
201 trancheData_.push_back(data);
202 }
203 }
204 }

262 string indexName = "GENERIC-" + t->investedTrancheName();
263 indexQuantities[indexName] = t->underlyingMultiplier();
264 underlyingIndex = boost::make_shared<QuantExt::GenericIndex>(indexName);
265 underlyingMultiplier = t->underlyingMultiplier();
266 assetCurrency = t->npvCurrency();
267 creditRiskCurrency = t->npvCurrency();
268
269 //fill the SimmCreditQualifierMapping
270 auto bonds = t->bondBasketData().bonds();
271 for (Size i = 0; i < bonds.size(); ++i) {
272 creditQualifierMapping[ore::data::securitySpecificCreditCurveName(bonds[i]->bondData().securityId(),
273 bonds[i]->bondData().creditCurveId())] =
274 SimmCreditQualifierMapping(bonds[i]->bondData().securityId(), bonds[i]->bondData().creditGroup());
275 creditQualifierMapping[bonds[i]->bondData().creditCurveId()] =
276 SimmCreditQualifierMapping(bonds[i]->bondData().securityId(), bonds[i]->bondData().creditGroup());
277 }

356void CBO::populateFromCboReferenceData(const boost::shared_ptr<CboReferenceDatum>& cboReferenceDatum){
357
358 DLOG("populating data cbo from reference data");
359 QL_REQUIRE(cboReferenceDatum, "populateFromCboReferenceData(): empty cbo reference datum given");
360
361 if (seniorFee_.empty()) {
362 seniorFee_ = cboReferenceDatum->cbostructure().seniorFee;
363 TLOG("overwrite SeniorFee with '" << seniorFee_ << "'");
364 }
365
366 if (subordinatedFee_.empty()) {
367 subordinatedFee_ = cboReferenceDatum->cbostructure().subordinatedFee;
368 TLOG("overwrite SubordinatedFee with '" << subordinatedFee_ << "'");
369 }
370
371 if (equityKicker_.empty()) {
372 equityKicker_ = cboReferenceDatum->cbostructure().equityKicker;
373 TLOG("overwrite EquityKicker with '" << equityKicker_ << "'");
374 }
375
376 if (feeDayCounter_.empty()) {
377 feeDayCounter_ = cboReferenceDatum->cbostructure().feeDayCounter;
378 TLOG("overwrite FeeDayCounter with '" << feeDayCounter_ << "'");
379 }
380
381 if (ccy_.empty()) {
382 ccy_ = cboReferenceDatum->cbostructure().ccy;
383 TLOG("overwrite currency with '" << ccy_ << "'");
384 }
385
386 if (reinvestmentEndDate_.empty()) {
387 reinvestmentEndDate_ = cboReferenceDatum->cbostructure().reinvestmentEndDate;
388 TLOG("overwrite ReinvestmentEndDate with '" << reinvestmentEndDate_ << "'");
389 }
390
391 if (daycounter_.empty()) {
392 daycounter_ = cboReferenceDatum->cbostructure().daycounter;
393 TLOG("overwrite DayCounter with '" << daycounter_ << "'");
394 }
395
396 if (paymentConvention_.empty()) {
397 paymentConvention_ = cboReferenceDatum->cbostructure().paymentConvention;
398 TLOG("overwrite PaymentConvention with '" << paymentConvention_ << "'");
399 }
400
401 if (!scheduleData_.hasData()) {
402 scheduleData_ = cboReferenceDatum->cbostructure().scheduleData;
403 TLOG("overwrite ScheduleData");
404 }
405
406 if (bondbasketdata_.empty()) {
407 bondbasketdata_ = cboReferenceDatum->cbostructure().bondbasketdata;
408 TLOG("overwrite BondBasketData");
409 }
410
411 if (trancheData_.empty()) {
412 trancheData_ = cboReferenceDatum->cbostructure().trancheData;
413 TLOG("overwrite TrancheData");
414 }

418void CBO::validateCbo(){
419
420 //Check for mandatroy fields
421 //reinvestmentEndDate_ not required
422 //TrancheName, Notional, StructuredID already checked in fromXML
423
424 std::string missing = "";
425
426 if (seniorFee_.empty())
427 missing.append("SeniorFee ");
428
429 if (subordinatedFee_.empty())
430 missing.append("SubordinatedFee ");
431
432 if (equityKicker_.empty())
433 missing.append("EquityKicker ");
434
435 if (feeDayCounter_.empty())
436 missing.append("FeeDayCounter ");
437
438 if (ccy_.empty())
439 missing.append("Currency ");
440
441 if (daycounter_.empty())
442 missing.append("DayCounter ");
443
444 if (paymentConvention_.empty())
445 missing.append("PaymentConvention ");
446
447 if (!scheduleData_.hasData())
448 missing.append("ScheduleData ");
449
450 if (bondbasketdata_.empty())
451 missing.append("BondBasketData ");
452
453 if (trancheData_.empty())
454 missing.append("TrancheData ");
455
456 if(!missing.empty())
457 QL_FAIL("CBO " << structureId_ << " expects " + missing +"elements");
458
459}
