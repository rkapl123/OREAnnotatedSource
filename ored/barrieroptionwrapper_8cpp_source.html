38Real BarrierOptionWrapper::NPV() const {
39 Real addNPV = additionalInstrumentsNPV();
40
41 Date today = Settings::instance().evaluationDate();
42
43 // check the trigger on the first run, we only need to re check it if the
44 // instrument becomes un calculated. The trigger should only need to be rechecked
45 // for a change in valuation date or spot, and this ensures
46 if (!exercised_ || !instrument_->isCalculated()) {
47 exercise();
48 }
49
50 if (exercised_) {
51 Real npv;
52 if (barrierType_ == Barrier::DownOut || barrierType_ == Barrier::UpOut)
53 npv = (today == exerciseDate_) ? multiplier2() * rebate_ * undMultiplier_ : 0.0;
54 else
55 npv = multiplier2() * getTimedNPV(activeUnderlyingInstrument_) * undMultiplier_;
56
57 return npv + addNPV;
58 } else {
59 // if not exercised we just return the original option's NPV
60 Real npv = multiplier2() * getTimedNPV(instrument_) * multiplier_;
61 return npv + addNPV;
62 }
63}

65const std::map<std::string, boost::any>& BarrierOptionWrapper::additionalResults() const {
66 static std::map<std::string, boost::any> emptyMap;
67 NPV();
68 if (exercised_) {
69 if (!(barrierType_ == Barrier::DownOut || barrierType_ == Barrier::UpOut))
70 return activeUnderlyingInstrument_->additionalResults();
71 else
72 return emptyMap;
73 } else {
74 return instrument_->additionalResults();

78bool SingleBarrierOptionWrapper::checkBarrier(Real spot, Barrier::Type type, Real barrier) const {
79 switch (type) {
80 case Barrier::DownIn:
81 case Barrier::DownOut:
82 return spot <= barrier;
83 case Barrier::UpIn:
84 case Barrier::UpOut:
85 return spot >= barrier;
86 default:
87 QL_FAIL("unknown barrier type " << type);
88 }
89}

91bool SingleBarrierOptionWrapper::exercise() const {
92 bool trigger = false;
93 Date today = Settings::instance().evaluationDate();
94
95 // check historical fixings - only check if the instrument is not calculated
96 // really only needs to be checked if evaluation date changed
97 if (!instrument_->isCalculated()) {
98 if (startDate_ != Date() && startDate_ < today) {
99 QL_REQUIRE(index_, "no index provided");
100 QL_REQUIRE(calendar_ != Calendar(), "no calendar provided");
101
102 boost::shared_ptr<QuantExt::EqFxIndexBase> eqfxIndex =
103 boost::dynamic_pointer_cast<QuantExt::EqFxIndexBase>(index_);
104
105 if (eqfxIndex) {
106 Date d = calendar_.adjust(startDate_);
107 while (d < today && !trigger) {
108 Real fixing = eqfxIndex->pastFixing(d);
109 if (fixing == Null<Real>()) {
110 ALOG(StructuredMessage(
111 StructuredMessage::Category::Error, StructuredMessage::Group::Fixing,
112 "Missing fixing for index " + index_->name() + " on " + ore::data::to_string(d) +
113 ", Skipping this date, assuming no trigger",
114 std::map<string, string>({{"exceptionType", "Invalid or missing fixings"}})));
115 } else {
116 // This is so we can use pastIndex and not fail on a missing fixing to be
117 // consistent with previous implemention, however maybe we should use fixing
118 // and be strict on needed fixings being present
119 auto fxInd = boost::dynamic_pointer_cast<QuantExt::FxIndex>(eqfxIndex);
120 trigger = checkBarrier(fixing, barrierType_, barrier_);
121 if (trigger)
122 exerciseDate_ = d;
123 }
124 d = calendar_.advance(d, 1, Days);
125 }
126 }
127 }
128 }
129
130 // check todays spot, if triggered today set the exerciseDate, may have to pay a rebate
131 if (!trigger) {
132 trigger = checkBarrier(spot_->value(), barrierType_, barrier_);
133 if (trigger)
134 exerciseDate_ = today;
135 }
136
137 exercised_ = trigger;
138 return trigger;
139}
