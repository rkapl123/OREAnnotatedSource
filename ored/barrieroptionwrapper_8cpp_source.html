42 Real BarrierOptionWrapper::NPV() const {
43  Real addNPV = additionalInstrumentsNPV();
44 
45  Date today = Settings::instance().evaluationDate();
46 
47  // check the trigger on the first run, we only need to re check it if the
48  // instrument becomes un calculated. The trigger should only need to be rechecked
49  // for a change in valuation date or spot, and this ensures
50  if (!exercised_ || !instrument_->isCalculated()) {
51  exercise();
52  }
53 
54  if (exercised_) {
55  // if exercised, return underlying npv for physical settlement and also for
56  // cash settlement if we are still on the exercise date (since the cash
57  // settlement takes place strictly after the exercise date usually)
58  // FIXME: we assume that the settlement date lies strictly after the exercise
59  // date, but before or on the next simulation date. Check this explicitly
60  // by introducing the cash settlement date into the option wrapper (note
61  // that we will probably need an effective cash settlement date then to
62  // maintain the relative position to the effective exercise date).
63  Real npv;
64  if (barrierType_ == Barrier::DownOut || barrierType_ == Barrier::UpOut)
65  npv = (today == exerciseDate_) ? (isLong_ ? 1.0 : -1.0) * rebate_ * undMultiplier_ : 0.0;
66  else
67  npv = (isLong_ ? 1.0 : -1.0) * getTimedNPV(activeUnderlyingInstrument_) * undMultiplier_;
68 
69  return npv + addNPV;
70  } else {
71  // if not exercised we just return the original option's NPV
72  Real npv = (isLong_ ? 1.0 : -1.0) * getTimedNPV(instrument_) * multiplier_;
73  return npv + addNPV;
74  }
75 }

85 bool SingleBarrierOptionWrapper::checkBarrier(Real spot, Barrier::Type type, Real barrier) const {
86  switch (type) {
87  case Barrier::DownIn:
88  case Barrier::DownOut:
89  return spot <= barrier;
90  case Barrier::UpIn:
91  case Barrier::UpOut:
92  return spot >= barrier;
93  default:
94  QL_FAIL("unknown barrier type " << type);
95  }
96 }

98 bool SingleBarrierOptionWrapper::exercise() const {
99  bool trigger = false;
100  Date today = Settings::instance().evaluationDate();
101 
102  // check historical fixings - only check if the instrument is not calculated
103  // really only needs to be checked if evaluation date changed
104  if (!instrument_->isCalculated()) {
105  triggered_ = false;
106  if (startDate_ != Date() && startDate_ < today) {
107  QL_REQUIRE(index_, "no index provided");
108  QL_REQUIRE(calendar_ != Calendar(), "no calendar provided");
109 
110  boost::shared_ptr<QuantExt::EqFxIndexBase> eqfxIndex =
111  boost::dynamic_pointer_cast<QuantExt::EqFxIndexBase>(index_);
112 
113  if (eqfxIndex) {
114  Date d = calendar_.adjust(startDate_);
115  while (d < today && !triggered_) {
116  Real fixing = eqfxIndex->pastFixing(d);
117  if (fixing == 0.0 || fixing == Null<Real>()) {
118  ALOG("Got invalid fixing for index " << index_->name() << " on " << d
119  << "Skipping this date, assuming no trigger");
120  } else {
121  // This is so we can use pastIndex and not fail on a missing fixing to be
122  // consistent with previous implemention, however maybe we should use fixing
123  // and be strict on needed fixings being present
124  auto fxInd = boost::dynamic_pointer_cast<QuantExt::FxIndex>(eqfxIndex);
125  triggered_ = checkBarrier(fixing, barrierType_, barrier_);
126  }
127  d = calendar_.advance(d, 1, Days);
128  }
129  }
130  }
131  }
132 
133  trigger = triggered_;
134  // check todays spot, if triggered today set the exerciseDate, may have to pay a rebate
135  if (!trigger) {
136  trigger = checkBarrier(spot_->value(), barrierType_, barrier_);
137  exerciseDate_ = today;
138  }
139 
140  exercised_ = trigger;
141  return trigger;
142 }
