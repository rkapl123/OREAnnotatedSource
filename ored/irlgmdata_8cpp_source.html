27 void IrLgmData::fromXML(XMLNode* node) {
28  qualifier_ = XMLUtils::getAttribute(node, "key");
29  if(qualifier_.empty()) {
30  std::string ccy = XMLUtils::getAttribute(node, "ccy");
31  if(!ccy.empty()) {
32  qualifier_ = ccy;
33  WLOG("IrLgmData: attribute ccy is deprecated, use key instead.");
34  }
35  }
36  LOG("LGM with attribute (key) = " << qualifier_);
37 
38  // Calibration Swaptions
39 
40  XMLNode* optionsNode = XMLUtils::getChildNode(node, "CalibrationSwaptions");
41 
42  optionExpiries() = XMLUtils::getChildrenValuesAsStrings(optionsNode, "Expiries", false);
43  optionTerms() = XMLUtils::getChildrenValuesAsStrings(optionsNode, "Terms", false);
44  QL_REQUIRE(optionExpiries().size() == optionTerms().size(),
45  "vector size mismatch in swaption expiries/terms for ccy " << qualifier_);
46  optionStrikes() = XMLUtils::getChildrenValuesAsStrings(optionsNode, "Strikes", false);
47  if (optionStrikes().size() > 0) {
48  QL_REQUIRE(optionStrikes().size() == optionExpiries().size(),
49  "vector size mismatch in swaption expiries/strikes for ccy " << qualifier_);
50  } else // Default: ATM
51  optionStrikes().resize(optionExpiries().size(), "ATM");
52 
53  for (Size i = 0; i < optionExpiries().size(); i++) {
54  LOG("LGM calibration swaption " << optionExpiries()[i] << " x " << optionTerms()[i] << " "
55  << optionStrikes()[i]);
56  }
57 
58  LgmData::fromXML(node);
59 }

61 XMLNode* IrLgmData::toXML(XMLDocument& doc) {
62  XMLNode* node = LgmData::toXML(doc);
63  XMLUtils::addAttribute(doc, node, "key", qualifier_);
64 
65  // swaption calibration
66  XMLNode* calibrationSwaptionsNode = XMLUtils::addChild(doc, node, "CalibrationSwaptions");
67  XMLUtils::addGenericChildAsList(doc, calibrationSwaptionsNode, "Expiries", optionExpiries());
68  XMLUtils::addGenericChildAsList(doc, calibrationSwaptionsNode, "Terms", optionTerms());
69  XMLUtils::addGenericChildAsList(doc, calibrationSwaptionsNode, "Strikes", optionStrikes());
70 
71  return node;
72 }
