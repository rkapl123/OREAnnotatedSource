93 void CSVLoader::loadFile(const string& filename, DataType dataType) {
94  LOG("CSVLoader loading from " << filename);
95 
96  Date today = QuantLib::Settings::instance().evaluationDate();
97 
98  ifstream file;
99  file.open(filename.c_str());
100  QL_REQUIRE(file.is_open(), "error opening file " << filename);
101 
102  while (!file.eof()) {
103  string line;
104  getline(file, line);
105  boost::trim(line);
106  // skip blank and comment lines
107  if (line.size() > 0 && line[0] != '#') {
108 
109  vector<string> tokens;
110  boost::trim(line);
111  boost::split(tokens, line, boost::is_any_of(",;\t "), boost::token_compress_on);
112 
113  // TODO: should we try, catch and log any invalid lines?
114  QL_REQUIRE(tokens.size() == 3, "Invalid CSVLoader line, 3 tokens expected " << line);
115  Date date = parseDate(tokens[0]);
116  const string& key = tokens[1];
117  Real value = parseReal(tokens[2]);
118 
119  if (dataType == DataType::Market) {
120  // process market
121  // build market datum and add to map
122  try {
123  data_[date].push_back(parseMarketDatum(date, key, value));
124  TLOG("Added MarketDatum " << data_[date].back()->name());
125  } catch (std::exception& e) {
126  WLOG("Failed to parse MarketDatum " << key << ": " << e.what());
127  }
128  } else if (dataType == DataType::Fixing) {
129  // process fixings
130  if (date < today || (date == today && !implyTodaysFixings_)) {
131  if(!fixings_.insert(Fixing(date, key, value)).second) {
132  WLOG("Skipped Fixing " << key << "@" << QuantLib::io::iso_date(date)
133  << " - this is already present.");
134  }
135  }
136  } else if (dataType == DataType::Dividend) {
137  // process dividends
138  if (date <= today) {
139  if(!dividends_.insert(Fixing(date, key, value)).second) {
140  WLOG("Skipped Dividend " << key << "@" << QuantLib::io::iso_date(date)
141  << " - this is already present.");
142  }
143  }
144  } else {
145  QL_FAIL("unknown data type");
146  }
147  }
148  }
149  file.close();
150  LOG("CSVLoader completed processing " << filename);
151 }
