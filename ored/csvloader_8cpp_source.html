93void CSVLoader::loadFile(const string& filename, DataType dataType) {
94 LOG("CSVLoader loading from " << filename);
95
96 Date today = QuantLib::Settings::instance().evaluationDate();
97
98 ifstream file;
99 file.open(filename.c_str());
100 QL_REQUIRE(file.is_open(), "error opening file " << filename);
101
102 while (!file.eof()) {
103 string line;
104 getline(file, line);
105 boost::trim(line);
106 // skip blank and comment lines
107 if (line.size() > 0 && line[0] != '#') {
108
109 vector<string> tokens;
110 boost::trim(line);
111 boost::split(tokens, line, boost::is_any_of(",;\t "), boost::token_compress_on);
112
113 // TODO: should we try, catch and log any invalid lines?
114 QL_REQUIRE(tokens.size() == 3 || tokens.size() == 4, "Invalid CSVLoader line, 3 tokens expected " << line);
115 if (tokens.size() == 4)
116 QL_REQUIRE(dataType == DataType::Dividend, "CSVLoader, dataType must be of type Dividend");
117 Date date = parseDate(tokens[0]);
118 const string& key = tokens[1];
119 Real value = parseReal(tokens[2]);
120
121 if (dataType == DataType::Market) {
122 // process market
123 // build market datum and add to map
124 try {
125 boost::shared_ptr<MarketDatum> md;
126 try {
127 md = parseMarketDatum(date, key, value);
128 } catch (std::exception& e) {
129 WLOG("Failed to parse MarketDatum " << key << ": " << e.what());
130 }
131 if (md != nullptr) {
132 if (data_[date].insert(md).second) {
133 TLOG("Added MarketDatum " << key);
134 } else {
135 WLOG("Skipped MarketDatum " << key << " - this is already present.");
136 }
137 }
138 } catch (std::exception& e) {
139 WLOG("Failed to parse MarketDatum " << key << ": " << e.what());
140 }
141 } else if (dataType == DataType::Fixing) {
142 // process fixings
143 if (date < today || (date == today && !implyTodaysFixings_)) {
144 if(!fixings_.insert(Fixing(date, key, value)).second) {
145 WLOG("Skipped Fixing " << key << "@" << QuantLib::io::iso_date(date)
146 << " - this is already present.");
147 }
148 }
149 } else if (dataType == DataType::Dividend) {
150 Date payDate = date;
151 if (tokens.size() == 4)
152 payDate = parseDate(tokens[3]);
153 // process dividends
154 if (date <= today) {
155 if (!dividends_.insert(QuantExt::Dividend(date, key, value, payDate)).second) {
156 WLOG("Skipped Dividend " << key << "@" << QuantLib::io::iso_date(date)
157 << " - this is already present.");
158 }
159 }
160 } else {
161 QL_FAIL("unknown data type");
162 }
163 }
164 }
165 file.close();
166 LOG("CSVLoader completed processing " << filename);
167}

182boost::shared_ptr<MarketDatum> CSVLoader::get(const string& name, const QuantLib::Date& d) const {
183 auto it = data_.find(d);
184 QL_REQUIRE(it != data_.end(), "No datum for " << name << " on date " << d);
185 auto it2 = it->second.find(makeDummyMarketDatum(d, name));

190std::set<boost::shared_ptr<MarketDatum>> CSVLoader::get(const std::set<std::string>& names,
191 const QuantLib::Date& asof) const {
192 auto it = data_.find(asof);
193 if(it == data_.end())
194 return {};
195 std::set<boost::shared_ptr<MarketDatum>> result;
196 for (auto const& n : names) {
197 auto it2 = it->second.find(makeDummyMarketDatum(asof, n));
198 if (it2 != it->second.end())
199 result.insert(*it2);
200 }
201 return result;

204std::set<boost::shared_ptr<MarketDatum>> CSVLoader::get(const Wildcard& wildcard,
205 const QuantLib::Date& asof) const {
206 if (!wildcard.hasWildcard()) {
207 // no wildcard => use get by name function
208 try {
209 return {get(wildcard.pattern(), asof)};
210 } catch (...) {
211 }
212 return {};
213 }
214 auto it = data_.find(asof);
215 if (it == data_.end())
216 return {};
217 std::set<boost::shared_ptr<MarketDatum>> result;
218 std::set<boost::shared_ptr<MarketDatum>>::iterator it1, it2;
219 if (wildcard.wildcardPos() == 0) {
220 // wildcard at first position => we have to search all of the data
221 it1 = it->second.begin();
222 it2 = it->second.end();
223 } else {
224 // search the range matching the substring of the pattern until the wildcard
225 std::string prefix = wildcard.pattern().substr(0, wildcard.wildcardPos());
226 it1 = it->second.lower_bound(makeDummyMarketDatum(asof, prefix));
227 it2 = it->second.upper_bound(makeDummyMarketDatum(asof, prefix + "\xFF"));
228 }
229 for (auto it = it1; it != it2; ++it) {
230 if (wildcard.isPrefix() || wildcard.matches((*it)->name()))
231 result.insert(*it);
232 }
233 return result;
234}
