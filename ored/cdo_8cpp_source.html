62// Check that weight, prior Weight or recovery is in [0, 1]
63void validateWeightRec(Real value, const string& name, const string& varName) {
64 QL_REQUIRE(value <= 1.0,
65 "The " << varName << " value (" << value << ") for name " << name << " should not be greater than 1.0.");
66 QL_REQUIRE(value >= 0.0,
67 "The " << varName << " value (" << value << ") for name " << name << " should not be less than 0.0.");
68}

75void SyntheticCDO::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
76
77 DLOG("SyntheticCDO::build() called for trade " << id());
78
79 Date protectionStartDate = protectionStart_ == "" ? Date() : parseDate(protectionStart_);
80 Date upfrontDate = upfrontDate_ == "" ? Date() : parseDate(upfrontDate_);
81 Leg leg = makeFixedLeg(legData_);
82 Protection::Side side = legData_.isPayer() ? Protection::Buyer : Protection::Seller;
83 Schedule schedule = makeSchedule(legData_.schedule());
84 Real fixedRecovery = recoveryRate_;
85 auto fixedLegData = boost::dynamic_pointer_cast<FixedLegData>(legData_.concreteLegData());
86 QL_REQUIRE(fixedLegData, "Expected FixedLegData but got " << legData_.legType());
87 Real runningRate = fixedLegData->rates().front();
88 DayCounter dayCounter = parseDayCounter(legData_.dayCounter());
89 BusinessDayConvention bdc = parseBusinessDayConvention(legData_.paymentConvention());
90 Currency ccy = parseCurrency(legData_.currency());
91
92 // In general for CDS index trades, the standard day counter is Actual/360 and the final
93 // period coupon accrual includes the maturity date.
94 Actual360 standardDayCounter;
95 DayCounter lastPeriodDayCounter = dayCounter == standardDayCounter ? Actual360(true) : dayCounter;
96
97
98 // Set some trade variables
99 npvCurrency_ = legData_.currency();
100 maturity_ = leg.back()->date();
101 notionalCurrency_ = legData_.currency();
102 legs_ = {leg};
103 legPayers_ = {legData_.isPayer()};
104 legCurrencies_ = {legData_.currency()};
105
106 // Checks for upfront date and upfront fee
107 QL_REQUIRE(upfrontDate == Date() || upfrontFee_ != Null<Real>(),
108 "If upfront date is given (" << upfrontDate << "), upfront fee must be given.");
109 QL_REQUIRE(upfrontDate != Date() || upfrontFee_ == Null<Real>() || close_enough(upfrontFee_, 0.0),
110 "If no upfront date is given, no upfront fee should be given but got " << upfrontFee_ << ".");
111
112 // Get the original total notional using the contractual attachment point and detachment point and the contractual
113 // tranche notional. We will calculate the corresponding current amounts below from the basket or reference data.
114 QL_REQUIRE(attachmentPoint_ < detachmentPoint_, "Detachment point should be greater than attachment point.");
115 Real origTrancheNtl = legData_.notionals().front();
116 Real origTotalNtl = origTrancheNtl / (detachmentPoint_ - attachmentPoint_);
117 Real origEquityNtl = origTotalNtl * attachmentPoint_;
118 Real origSeniorNtl = origTotalNtl * (1.0 - detachmentPoint_);
119
120 DLOG("Original tranche notional: " << origTrancheNtl);
121 DLOG("Original equity notional: " << origEquityNtl);
122 DLOG("Original senior notional: " << origSeniorNtl);
123 DLOG("Original attachment point: " << attachmentPoint_);
124 DLOG("Original detachment point: " << detachmentPoint_);
125 DLOG("Original total notional: " << origTotalNtl);
126
127 // There may have been credit events up to the valuation date. Record the cumulative lost and recovered notional.
128 Real lostNotional = 0.0;
129 Real recoveredNotional = 0.0;
130 // clear the basketNotionals and credit names
131
132 vector<Real> basketNotionals;
133
134 vector<string> creditCurves;
135
136
137
138 if (basketData_.constituents().size() > 0) {
139
140 const auto& constituents = basketData_.constituents();
141 DLOG("Building constituents from basket data containing " << constituents.size() << " elements.");
142
143 Real totalRemainingNtl = 0.0;
144 Real totalPriorNtl = 0.0;
145
146 for (const auto& c : constituents) {
147 Real ntl = Null<Real>(), priorNotional = Null<Real>();
148 const auto& creditCurve = c.creditCurveId();
149 if (c.weightInsteadOfNotional()) {
150 ntl = c.weight() * origTotalNtl;
151 priorNotional = c.priorWeight();
152 if (priorNotional != Null<Real>()) {
153 priorNotional *= origTotalNtl;
154 }
155 } else {
156 ntl = c.notional();
157 priorNotional = c.priorNotional();
158 QL_REQUIRE(c.currency() == npvCurrency_, "The currency of basket constituent "
159 << creditCurve << " is " << c.currency()
160 << " and does not equal the trade leg currency "
161 << npvCurrency_);
162 }
163
164 if (!close(0.0, ntl) && ntl > 0.0) {
165 if (std::find(creditCurves.begin(), creditCurves.end(), creditCurve) == creditCurves.end()) {
166 DLOG("Adding underlying " << creditCurve << " with notional " << ntl);
167 creditCurves.push_back(creditCurve);
168 basketNotionals.push_back(ntl);
169 totalRemainingNtl += ntl;
170 } else {
171 ALOG(StructuredTradeErrorMessage(id(), "Synthetic CDO", "Error building trade",
172 ("Invalid Basket: found a duplicate credit curve " + creditCurve +
173 ", skip it. Check the basket data for possible errors.")
174 .c_str()));
175 }
176 } else {
177 DLOG("Underlying " << creditCurve << " notional is " << ntl << " so assuming a credit event occured.");
178 QL_REQUIRE(priorNotional != Null<Real>(),
179 "Expecting a valid prior notional for name " << creditCurve << ".");
180 auto recovery = c.recovery();
181 QL_REQUIRE(recovery != Null<Real>(), "Expecting a valid recovery for name " << creditCurve << ".");
182 validateWeightRec(recovery, creditCurve, "recovery");
183 lostNotional += (1.0 - recovery) * priorNotional;
184 recoveredNotional += recovery * priorNotional;
185 totalPriorNtl += priorNotional;
186 }
187 }
188
189 Real totalNtl = totalRemainingNtl + totalPriorNtl;
190 DLOG("All Underlyings added, total remaining notional = " << totalRemainingNtl);
191 DLOG("All Underlyings added, total prior notional = " << totalPriorNtl);
192 DLOG("All Underlyings added, total notional = " << totalNtl);
193
194 QL_REQUIRE(creditCurves.size() == basketNotionals.size(), "numbers of defaults curves ("
195 << creditCurves.size() << ") and notionals ("
196 << basketNotionals.size() << ") doesnt match");
197 Real notionalCorrectionFactor = origTotalNtl / totalNtl;
198 // Scaling to Notional if relative error is close less than 10^-4
199 if (!close(totalNtl, origTotalNtl) && (abs(notionalCorrectionFactor - 1.0) <= 1e-4)) {
200 ALOG("Trade " << id() << ", sum of notionals(" << totalNtl << ") is very close to total original notional ("
201 << origTotalNtl << "), will scale each notional by " << notionalCorrectionFactor
202 << ", check the basket data for possible errors.");
203 totalRemainingNtl = 0;
204 for (Size i = 0; i < basketNotionals.size(); i++) {
205 Real scaledNotional = basketNotionals[i] * notionalCorrectionFactor;
206 TLOG("Trade " << id() << ", Issuer" << creditCurves[i] << " unscaled Notional: " << basketNotionals[i]
207 << ", scaled Notional: " << scaledNotional);
208 basketNotionals[i] = scaledNotional;
209 totalRemainingNtl += scaledNotional;
210 }
211 lostNotional *= notionalCorrectionFactor;
212 recoveredNotional *= notionalCorrectionFactor;
213 totalNtl *= notionalCorrectionFactor;
214 }
215
216 if (!close(totalRemainingNtl, origTotalNtl) && totalRemainingNtl > origTotalNtl) {
217 ALOG(StructuredTradeErrorMessage(id(), "Synthetic CDO", "Error building trade",
218 ("Total remaining notional (" + std::to_string(totalRemainingNtl) +
219 ") is greater than total original notional (" +
220 std::to_string(origTotalNtl) +
221 "), check the basket data for possible errors.")
222 .c_str()));
223 }
224
225 if (!close(totalNtl, origTotalNtl)) {
226 ALOG(
227 StructuredTradeErrorMessage(id(), "Synthetic CDO", "Error building trade",
228 ("Expected the total notional (" + std::to_string(totalNtl) + " = " +
229 std::to_string(totalRemainingNtl) + " + " + std::to_string(totalPriorNtl) +
230 ") to equal the total original notional (" + std::to_string(origTotalNtl) +
231 "), check the basket data for possible errors.")
232 .c_str()));
233 }
234
235 DLOG("Finished building constituents using basket data.");
236
237 } else {
238
239 DLOG("Building constituents using CreditIndexReferenceDatum for ID " << qualifier_);
240
241 QL_REQUIRE(engineFactory->referenceData(),
242 "Trade " << id() << " has no basket data and there is no reference data manager.");
243 QL_REQUIRE(engineFactory->referenceData()->hasData(CreditIndexReferenceDatum::TYPE, qualifier_),
244 "Trade " << id() << " needs credit index reference data for ID " << qualifier_);
245 auto crd = boost::dynamic_pointer_cast<CreditIndexReferenceDatum>(
246 engineFactory->referenceData()->getData(CreditIndexReferenceDatum::TYPE, qualifier_));
247
248 Real totalRemainingWeight = 0.0;
249 Real totalPriorWeight = 0.0;
250 for (const auto& c : crd->constituents()) {
251
252 const auto& name = c.name();
253 auto weight = c.weight();
254 validateWeightRec(weight, name, "weight");
255
256 if (!close(0.0, weight)) {
257 DLOG("Adding underlying " << name << " with weight " << weight);
258 creditCurves.push_back(name);
259 basketNotionals.push_back(weight * origTotalNtl);
260 totalRemainingWeight += weight;
261 } else {
262 DLOG("Underlying " << name << " has weight " << weight << " so assuming a credit event occured.");
263 auto priorWeight = c.priorWeight();
264 QL_REQUIRE(priorWeight != Null<Real>(), "Expecting a valid prior weight for name " << name << ".");
265 validateWeightRec(priorWeight, name, "prior weight");
266 auto recovery = c.recovery();
267 QL_REQUIRE(recovery != Null<Real>(), "Expecting a valid recovery for name " << name << ".");
268 validateWeightRec(recovery, name, "recovery");
269 lostNotional += (1.0 - recovery) * priorWeight * origTotalNtl;
270 recoveredNotional += recovery * priorWeight * origTotalNtl;
271 totalPriorWeight += priorWeight;
272 }
273 }
274
275 Real totalWeight = totalRemainingWeight + totalPriorWeight;
276 DLOG("All Underlyings added, total remaining weight = " << totalRemainingWeight);
277 DLOG("All Underlyings added, total prior weight = " << totalPriorWeight);
278 DLOG("All Underlyings added, total weight = " << totalWeight);
279
280 if (!close(totalRemainingWeight, 1.0) && totalRemainingWeight > 1.0) {
281 ALOG("Total remaining weight is greater than 1, possible error in CreditIndexReferenceDatum");
282 }
283
284 if (!close(totalWeight, 1.0)) {
285 ALOG("Expected the total weight (" << totalWeight << " = " << totalRemainingWeight << " + "
286 << totalPriorWeight
287 << ") to equal 1, possible error in CreditIndexReferenceDatum");
288 }
289
290 DLOG("Finished building constituents using CreditIndexReferenceDatum for ID " << qualifier_);
291 }
292
293 // Lost notional eats into junior tranches and recovered amount reduces senior tranches.
294 // Throw exception if tranche has been completely written down. Should be removed from portfolio or will be
295 // ignored if continue on error is true. Maybe better to have instrument that gives an NPV of 0.
296 Real currTotalNtl = accumulate(basketNotionals.begin(), basketNotionals.end(), 0.0);
297 QL_REQUIRE(!close(currTotalNtl, 0.0), "Trade " << id() << " has a current total notional of 0.0.");
298 Real currEquityNtl = max(origEquityNtl - lostNotional, 0.0);
299 Real currSeniorNtl = max(origSeniorNtl - recoveredNotional, 0.0);
300 Real currTrancheNtl = origTrancheNtl - max(min(recoveredNotional - origSeniorNtl, origTrancheNtl), 0.0) -
301 max(min(lostNotional - origEquityNtl, origTrancheNtl), 0.0);
302 QL_REQUIRE(!close(currTrancheNtl, 0.0), "Trade " << id() << " has a current tranche notional of 0.0.");
303 Real adjAttachPoint = currEquityNtl / currTotalNtl;
304 Real adjDetachPoint = (currEquityNtl + currTrancheNtl) / currTotalNtl;
305 notional_ = currTrancheNtl;
306
307 DLOG("Current tranche notional: " << currTrancheNtl);
308 DLOG("Current equity notional: " << currEquityNtl);
309 DLOG("Current senior notional: " << currSeniorNtl);
310 DLOG("Current attachment point: " << adjAttachPoint);
311 DLOG("Current detachment point: " << adjDetachPoint);
312 DLOG("Current total notional: " << currTotalNtl);
313
314
315 const auto& market = engineFactory->market();
316 auto cdoEngineBuilder = boost::dynamic_pointer_cast<CdoEngineBuilder>(engineFactory->builder("SyntheticCDO"));
317 QL_REQUIRE(cdoEngineBuilder, "Trade " << id() << " needs a valid CdoEngineBuilder.");
318 const string& config = cdoEngineBuilder->configuration(MarketContext::pricing);
319
320
321 std::vector<Handle<DefaultProbabilityTermStructure>> dpts;
322 vector<Real> recoveryRates;
323
324 if (fixedRecovery != Null<Real>()) {
325 LOG("Set all recovery rates to " << fixedRecovery);
326 }
327 for (Size i = 0; i < creditCurves.size(); ++i) {
328 const string& cc = creditCurves[i];
329 Real mktRecoveryRate = market->recoveryRate(cc, config)->value();
330 recoveryRates.push_back(fixedRecovery != Null<Real>() ? fixedRecovery : mktRecoveryRate);
331 auto originalCurve = market->defaultCurve(cc, config)->curve();
332 dpts.push_back(originalCurve);
333 }
334
335 // TODO check if underlying is index cds
336
337 // Calibrate the underlying constituent curves so that the index cds pricing with underlying curves matches the
338 // prices of the index cds with flat index curve.
339
340 boost::shared_ptr<SimpleQuote> calibrationFactor = boost::make_shared<SimpleQuote>(1.0);
341
342 bool calibrateConstiuentCurves = cdoEngineBuilder->calibrateConstituentCurve() && isIndexTranche();
343
344 if (calibrateConstiuentCurves) {
345 // Adjustment factor is a simplified version of the O'Kane's Forward Default Probability Multiplier
346 // O'Kane 2008 - Modelling Single-name and Multi-name Credit Derivatives
347 // Chapter 10.6
348 LOG("Use calibrated constiuent curves");
349 boost::shared_ptr<IndexCreditDefaultSwap> indexCDS;
350 Real cdsFairSpreads = 0.0;
351 Real cdsNpvs = 0.0;
352 try {
353 Handle<YieldTermStructure> yts =
354 market->discountCurve(ccy.code(), cdoEngineBuilder->configuration(MarketContext::pricing));
355
356 Date cdsStartDate = indexStartDateHint() == Date() ? schedule.dates().front() : indexStartDateHint();
357
358 Schedule cdsSchedule(cdsStartDate, schedule.dates().back(),
359 schedule.tenor(), schedule.calendar(), Following,
360 schedule.terminationDateBusinessDayConvention(), schedule.rule(), false);
361 indexCDS = boost::make_shared<QuantExt::IndexCreditDefaultSwap>(
362 side, currTotalNtl, basketNotionals, 0.0, runningRate, cdsSchedule, bdc, dayCounter, settlesAccrual_,
363 protectionPaymentTime_, Date(), Date(), boost::shared_ptr<Claim>(),
364 lastPeriodDayCounter, rebatesAccrual_);
365 Handle<DefaultProbabilityTermStructure> indexCreditCurve =
366 indexCdsDefaultCurve(market, creditCurveIdWithTerm(), config)->curve();
367 Handle<Quote> indexCdsRecovery = market->recoveryRate(creditCurveIdWithTerm(), config);
368 auto indexPricingEngine =
369 boost::make_shared<QuantExt::MidPointIndexCdsEngine>(indexCreditCurve, indexCdsRecovery->value(), yts);
370 indexCDS->setPricingEngine(indexPricingEngine);
371 } catch (const std::exception& e) {
372 indexCDS = nullptr;
373 WLOG("CDO constiuent calibration failed to build index cds. Got "
374 << e.what());
375 }
376 if (indexCDS) {
377 cdsFairSpreads = indexCDS->fairSpreadClean();
378 cdsNpvs = indexCDS->NPV();
379 }
380
381 auto it = engineFactory->engineData()->globalParameters().find("RunType");
382 if (it != engineFactory->engineData()->globalParameters().end() && it->second != "PortfolioAnalyser") {
383
384 Handle<YieldTermStructure> yts =
385 market->discountCurve(ccy.code(), cdoEngineBuilder->configuration(MarketContext::pricing));
386
387 std::vector<Handle<DefaultProbabilityTermStructure>> wrapperCurves;
388 DLOG("Building wrapper curves for calibration");
389 for (size_t i = 0; i < creditCurves.size(); ++i) {
390 try {
391 const string& cc = creditCurves[i];
392 auto originalCurve = market->defaultCurve(cc, config)->curve();
393 wrapperCurves.push_back(
394 buildCalibratedConstiuentCurve(originalCurve, calibrationFactor));
395 } catch (const std::exception& e) {
396
397 WLOG("CDO constiuent calibration failed during building wrapper curve for "
398 << creditCurves[i] << ", skip this curve. Got "
399 << e.what());
400 }
401 }
402
403 if (wrapperCurves.size() == dpts.size() && indexCDS) {
404 LOG("Start bootstraping of the calibration factors");
405 dpts.swap(wrapperCurves);
406
407 auto cdsPricingEngineUnderlyingCurves =
408 boost::make_shared<QuantExt::MidPointIndexCdsEngine>(dpts, recoveryRates, yts);
409
410 indexCDS->setPricingEngine(cdsPricingEngineUnderlyingCurves);
411
412 try {
413 auto targetFunction = [&cdsNpvs, &calibrationFactor, &indexCDS](const double& factor) {
414 calibrationFactor->setValue(factor);
415 return cdsNpvs - indexCDS->NPV();
416 };
417
418 Brent solver;
419 double indexAdjustmentForUnderlyingCurves =
420 solver.solve(targetFunction, 1e-8, cdsFairSpreads / indexCDS->fairSpreadClean(), 0.001, 2);
421
422 DLOG("Calibration of indexterm " << io::iso_date(indexCDS->maturity())
423 << "successful, found solution "
424 << indexAdjustmentForUnderlyingCurves);
425 calibrationFactor->setValue(indexAdjustmentForUnderlyingCurves);
426 } catch (const std::exception& e) {
427 WLOG("Calibration failed, at pillar " << io::iso_date(indexCDS->maturity())
428 << ", set calibration factor to 1 (uncalibrated), got "
429 << e.what());
430 calibrationFactor->setValue(1.0);
431 }
432
433 LOG("Calibration results for creditCurve:" << creditCurveIdWithTerm());
434 LOG("Expiry \t CalibrationFactor \t NpvIntrinsic \t NpvIndexCurve \t NpvError \t "
435 "FairSpreadIntrinsic "
436 "\t FairSpreadIndexCurve \t FairSreadError");
437
438 LOG(io::iso_date(indexCDS->maturity())
439 << "\t" << calibrationFactor->value() << "\t" << indexCDS->NPV() << "\t" << cdsNpvs << "\t"
440 << indexCDS->NPV() - cdsNpvs << "\t" << indexCDS->fairSpreadClean() << "\t" << cdsFairSpreads
441 << "\t" << indexCDS->fairSpreadClean() - cdsFairSpreads);
442 }
443 }
444 }
445
446 // Create the instruments.
447 auto pool = boost::make_shared<Pool>();
448
449 CreditPortfolioSensitivityDecomposition sensitivityDecomposition = cdoEngineBuilder->sensitivityDecomposition();
450 useSensitivitySimplification_ = sensitivityDecomposition != CreditPortfolioSensitivityDecomposition::Underlying;
451 Handle<DefaultProbabilityTermStructure> clientCurve;
452 Handle<DefaultProbabilityTermStructure> baseCurve;
453 vector<Time> baseCurveTimes;
454 vector<Real> expLoss;
455
456 if (cdoEngineBuilder->optimizedSensitivityCalculation()) {
457 dpts = buildPerformanceOptimizedDefaultCurves(dpts);
458 }
459
460 for (Size i = 0; i < creditCurves.size(); ++i) {
461 const string& cc = creditCurves[i];
462 DefaultProbKey key = NorthAmericaCorpDefaultKey(ccy, SeniorSec, Period(), 1.0);
463 Real recoveryRate = recoveryRates[i];
464 auto defaultCurve = dpts[i];
465 expLoss.push_back((1 - recoveryRate) * defaultCurve->defaultProbability(maturity_, true) * basketNotionals[i]);
466 std::pair<DefaultProbKey, Handle<DefaultProbabilityTermStructure>> p(key, defaultCurve);
467 vector<pair<DefaultProbKey, Handle<DefaultProbabilityTermStructure>>> probabilities(1, p);
468 // Empty default set. Adjustments have been made above to account for existing credit events.
469 Issuer issuer(probabilities, DefaultEventSet());
470 pool->add(cc, issuer, key);
471 DLOG("Issuer " << cc << " added to the pool.");
472 }
473
474 // If we use the simplification, we need a list of all credit curves and their weight to the basket
475
476 if (sensitivityDecomposition == CreditPortfolioSensitivityDecomposition::LossWeighted) {
477 basketConstituents_.clear();
478 Real totalWeight = std::accumulate(expLoss.begin(), expLoss.end(), 0.0);
479 for (Size basketIdx = 0; basketIdx < creditCurves.size(); basketIdx++) {
480 string& creditCurve = creditCurves[basketIdx];
481 Real weight = expLoss[basketIdx];
482 if (basketConstituents_.count(creditCurve) == 0) {
483 basketConstituents_[creditCurve] = weight / totalWeight;
484 } else {
485 basketConstituents_[creditCurve] += weight / totalWeight;
486 }
487 }
488 } else if (sensitivityDecomposition == CreditPortfolioSensitivityDecomposition::NotionalWeighted) {
489 basketConstituents_.clear();
490 Real totalWeight = std::accumulate(basketNotionals.begin(), basketNotionals.end(), 0.0);
491 for (Size basketIdx = 0; basketIdx < creditCurves.size(); basketIdx++) {
492 string& creditCurve = creditCurves[basketIdx];
493 Real weight = basketNotionals[basketIdx];
494 if (basketConstituents_.count(creditCurve) == 0) {
495 basketConstituents_[creditCurve] = weight / totalWeight;
496 } else {
497 basketConstituents_[creditCurve] += weight / totalWeight;
498 }
499 }
500 } else if (sensitivityDecomposition == CreditPortfolioSensitivityDecomposition::DeltaWeighted) {
501 basketConstituents_.clear();
502 Real totalWeight = 0;
503 for (Size basketIdx = 0; basketIdx < creditCurves.size(); basketIdx++) {
504 string& creditCurve = creditCurves[basketIdx];
505 Real notional = basketNotionals[basketIdx];
506 auto defaultCurve = market->defaultCurve(creditCurve, config)->curve();
507 Real constituentSurvivalProb = defaultCurve->survivalProbability(maturity_);
508 Time t = defaultCurve->timeFromReference(maturity_);
509 Real CR01 = t * constituentSurvivalProb * notional;
510 if (basketConstituents_.find(creditCurve) == basketConstituents_.end())
511 basketConstituents_.emplace(creditCurve, CR01);
512 totalWeight += CR01;
513 }
514 // Normalize
515 for (auto& decompWeight : basketConstituents_) {
516 decompWeight.second /= totalWeight;
517 }
518 }
519
520 // We will use a homogeneous pool loss model below if all notionals and recoveries are the same.
521 bool homogeneous = all_of(basketNotionals.begin(), basketNotionals.end(),
522 [&basketNotionals](Real ntl) { return close_enough(ntl, basketNotionals[0]); });
523 homogeneous = homogeneous && all_of(recoveryRates.begin(), recoveryRates.end(),
524 [&recoveryRates](Real rr) { return close_enough(rr, recoveryRates[0]); });
525
526
527 // vanilla holds the representation of the CDO, i.e. 1 * [0, Detach] instrument - 1 * [0, Attach] instrument,
528 // without the upfront fee payment which is added below to create the full final instrument.
529 boost::shared_ptr<Instrument> vanilla;
530
531 // Tranche from 0 to detachment point.
532 // If detachment point is 1.0, build an index CDS, i.e. [0, 100%] tranche. Otherwise an actual tranche.
533 boost::shared_ptr<Instrument> cdoD;
534 if (!close_enough(adjDetachPoint, 1.0)) {
535 DLOG("Building detachment tranche [0," << adjDetachPoint << "].");
536 // Set up the basket loss model.
537 auto basket =
538 boost::make_shared<QuantExt::Basket>(schedule[0], creditCurves, basketNotionals, pool, 0.0, adjDetachPoint);
539 basket->setLossModel(
540 cdoEngineBuilder->lossModel(qualifier(), recoveryRates, adjDetachPoint,
541 maturity_, homogeneous));
542
543 auto cdoDetach =
544 boost::make_shared<QuantExt::SyntheticCDO>(basket, side, schedule, 0.0, runningRate, dayCounter, bdc,
545 settlesAccrual_, protectionPaymentTime_, protectionStartDate, parseDate(upfrontDate_), boost::none, Null<Real>(), lastPeriodDayCounter);
546
547 cdoDetach->setPricingEngine(
548 cdoEngineBuilder->engine(ccy, false, {}, calibrationFactor, fixedRecovery));
549 cdoD = cdoDetach;
550
551 DLOG("Detachment tranche [0," << adjDetachPoint << "] built.");
552
553 } else {
554 DLOG("Detachment point is 1.0 so building an index CDS for [0,1.0] 'tranche'.");
555
556 /*auto cdsBuilder = boost::dynamic_pointer_cast<IndexCreditDefaultSwapEngineBuilder>(
557 engineFactory->builder("IndexCreditDefaultSwap"));
558 QL_REQUIRE(cdsBuilder, "Trade " << id() << " needs a IndexCreditDefaultSwapEngineBuilder.");
559 */
560 auto cds = boost::make_shared<QuantExt::IndexCreditDefaultSwap>(
561 side, currTotalNtl, basketNotionals, 0.0, runningRate, schedule, bdc, dayCounter, settlesAccrual_,
562 protectionPaymentTime_, protectionStartDate, Date(), boost::shared_ptr<Claim>(),
563 lastPeriodDayCounter, rebatesAccrual_, protectionStartDate, 3);
564
565 cds->setPricingEngine(
566 cdoEngineBuilder->engine(ccy, true, creditCurves, calibrationFactor, fixedRecovery));
567 cdoD = cds;
568
569 DLOG("Index CDS for [0,1.0] 'tranche' built.");
570 }
571
572 // Tranche from 0 to attachment point.
573 // If attachment point is 0.0, the instrument is simply the 0 to detachment point CDO built above. If attachment
574 // point is greater than 0, we build the 0 to attachment point tranche and the value of the CDO is the
575 // difference between the detachment point CDO and attachment point CDO.
576 if (close_enough(adjAttachPoint, 0.0)) {
577 DLOG("Attachment point is 0 so the instrument is built.");
578 vanilla = cdoD;
579 } else {
580 DLOG("Building attachment tranche [0," << adjAttachPoint << "].");
581
582 // Set up the basket loss model.
583 auto basket =
584 boost::make_shared<QuantExt::Basket>(schedule[0], creditCurves, basketNotionals, pool, 0.0, adjAttachPoint);
585 basket->setLossModel(cdoEngineBuilder->lossModel(qualifier(), recoveryRates, adjAttachPoint,
586 maturity_, homogeneous));
587
588 auto cdoA =
589 boost::make_shared<QuantExt::SyntheticCDO>(basket, side, schedule, 0.0, runningRate, dayCounter, bdc, settlesAccrual_, protectionPaymentTime_,
590 protectionStartDate, parseDate(upfrontDate_), boost::none, fixedRecovery, lastPeriodDayCounter);
591
592 cdoA->setPricingEngine(
593 cdoEngineBuilder->engine(ccy, false, {}, calibrationFactor, fixedRecovery));
594
595 DLOG("Attachment tranche [0," << adjAttachPoint << "] built.");
596
597 DLOG("Building attachment and detachment composite instrument.");
598
599 auto composite = boost::make_shared<CompositeInstrument>();
600 composite->add(cdoD);
601 composite->subtract(cdoA);
602 vanilla = composite;
603
604 DLOG("Attachment and detachment composite instrument built.");
605 }
606
607 // Add the upfront fee payment here. Positive number indicates payment from buyer of protection to seller of
608 // protection and a negative value indicates a payment from seller of protection to buyer of protection. The
609 // upfront fee provided is interpreted as a fractional amount of the original tranche notional.
610 if (upfrontDate != Date()) {
611 vector<boost::shared_ptr<Instrument>> insts;
612 vector<Real> mults;
613 Real upfrontAmount = upfrontFee_ * origTrancheNtl;
614 string configuration = cdoEngineBuilder->configuration(MarketContext::pricing);
615 maturity_ =
616 std::max(maturity_, addPremiums(insts, mults, 1.0, PremiumData(upfrontAmount, ccy.code(), upfrontDate),
617 side == Protection::Buyer ? -1.0 : 1.0, ccy, engineFactory, configuration));
618
619 instrument_ = boost::make_shared<VanillaInstrument>(vanilla, 1.0, insts, mults);
620
621 } else {
622 // If no upfront payment, the instrument is simply the vanilla instrument.
623 instrument_ = boost::make_shared<VanillaInstrument>(vanilla);
624 }
625
626 additionalData_["originalNotional"] = origTrancheNtl;
627 additionalData_["currentNotional"] = currTrancheNtl;
628
629 // ISDA taxonomy
630 additionalData_["isdaAssetClass"] = string("Credit");
631 additionalData_["isdaBaseProduct"] = string("Index Tranche");
632 boost::shared_ptr<ReferenceDataManager> refData = engineFactory->referenceData();
633 if (refData && refData->hasData("CreditIndex", qualifier_)) {
634 auto refDatum = refData->getData("CreditIndex", qualifier_);
635 boost::shared_ptr<CreditIndexReferenceDatum> creditIndexRefDatum = boost::dynamic_pointer_cast<CreditIndexReferenceDatum>(refDatum);
636 additionalData_["isdaSubProduct"] = creditIndexRefDatum->indexFamily();
637 if (creditIndexRefDatum->indexFamily() == "") {
638 ALOG("IndexFamily is blank in credit index reference data for entity " << qualifier_);
639 }
640 } else {
641 ALOG("Credit index reference data missing for entity " << qualifier_ << ", isdaSubProduct left blank");
642 }
643 // skip the transaction level mapping for now
644 additionalData_["isdaTransaction"] = string("");
645
646 DLOG("CDO instrument built");
647}

649void SyntheticCDO::fromXML(XMLNode* node) {
650 Trade::fromXML(node);
651 XMLNode* cdoNode = XMLUtils::getChildNode(node, "CdoData");
652 QL_REQUIRE(cdoNode, "No CdoData Node");
653 qualifier_ = XMLUtils::getChildValue(cdoNode, "Qualifier", true);
654 protectionStart_ = XMLUtils::getChildValue(cdoNode, "ProtectionStart", true);
655 upfrontDate_ = XMLUtils::getChildValue(cdoNode, "UpfrontDate", false);
656
657 // zero if empty or missing
658 upfrontFee_ = Null<Real>();
659 string strUpfrontFee = XMLUtils::getChildValue(cdoNode, "UpfrontFee", false);
660 if (!strUpfrontFee.empty()) {
661 upfrontFee_ = parseReal(strUpfrontFee);
662 }
663 settlesAccrual_ = XMLUtils::getChildValueAsBool(cdoNode, "SettlesAccrual", false); // default = Y
664 rebatesAccrual_ = XMLUtils::getChildValueAsBool(cdoNode, "RebatesAccrual", false); // default = Y
665 protectionPaymentTime_ = QuantExt::CreditDefaultSwap::ProtectionPaymentTime::atDefault; // set default
666
667 // Recovery rate is Null<Real>() on a standard CDO i.e. if "FixedRecoveryRate" field is not populated.
668 recoveryRate_ = Null<Real>();
669 string strRecoveryRate = XMLUtils::getChildValue(node, "FixedRecoveryRate", false);
670 if (!strRecoveryRate.empty()) {
671 recoveryRate_ = parseReal(strRecoveryRate);
672 }
673
674 // for backwards compatibility only
675 if (auto c = XMLUtils::getChildNode(cdoNode, "PaysAtDefaultTime"))
676 if (!parseBool(XMLUtils::getNodeValue(c)))
677 protectionPaymentTime_ = QuantExt::CreditDefaultSwap::ProtectionPaymentTime::atPeriodEnd;
678 // new node overrides deprecated one, if both should be given
679 if (auto c = XMLUtils::getChildNode(cdoNode, "ProtectionPaymentTime")) {
680 if (XMLUtils::getNodeValue(c) == "atDefault")
681 protectionPaymentTime_ = QuantExt::CreditDefaultSwap::ProtectionPaymentTime::atDefault;
682 else if (XMLUtils::getNodeValue(c) == "atPeriodEnd")
683 protectionPaymentTime_ = QuantExt::CreditDefaultSwap::ProtectionPaymentTime::atPeriodEnd;
684 else if (XMLUtils::getNodeValue(c) == "atMaturity")
685 protectionPaymentTime_ = QuantExt::CreditDefaultSwap::ProtectionPaymentTime::atMaturity;
686 else {
687 QL_FAIL("protection payment time '" << XMLUtils::getNodeValue(c)
688 << "' not known, expected atDefault, atPeriodEnd, atMaturity");
689 }
690 }
691 attachmentPoint_ = XMLUtils::getChildValueAsDouble(cdoNode, "AttachmentPoint", true);
692 detachmentPoint_ = XMLUtils::getChildValueAsDouble(cdoNode, "DetachmentPoint", true);
693 XMLNode* legNode = XMLUtils::getChildNode(cdoNode, "LegData");
694 legData_.fromXML(legNode);
695 XMLNode* basketNode = XMLUtils::getChildNode(cdoNode, "BasketData");
696 if (basketNode)
697 basketData_.fromXML(basketNode);
698}

715 if (protectionPaymentTime_ == QuantExt::CreditDefaultSwap::ProtectionPaymentTime::atDefault)

717 else if (protectionPaymentTime_ == QuantExt::CreditDefaultSwap::ProtectionPaymentTime::atPeriodEnd)
718 XMLUtils::addChild(doc, cdoNode, "ProtectionPaymentTime", "atPeriodEnd");
719 else if (protectionPaymentTime_ == QuantExt::CreditDefaultSwap::ProtectionPaymentTime::atMaturity)
720 XMLUtils::addChild(doc, cdoNode, "ProtectionPaymentTime", "atMaturity");

726 XMLUtils::addChild(doc, cdoNode, "AttachmentPoint", attachmentPoint_);
727 XMLUtils::addChild(doc, cdoNode, "DetachmentPoint", detachmentPoint_);
728 XMLUtils::appendNode(cdoNode, legData_.toXML(doc));
729 XMLUtils::appendNode(cdoNode, basketData_.toXML(doc));
730 return node;

732vector<Time>
733SyntheticCDO::extractTimeGridDefaultCurve(const QuantLib::Handle<QuantLib::DefaultProbabilityTermStructure> &dpts) {
734 if (auto c = boost::dynamic_pointer_cast<SpreadedSurvivalProbabilityTermStructure>(*dpts)) {
735 return SyntheticCDO::extractTimeGridDefaultCurve(c->referenceCurve());
736 } else if (auto c = boost::dynamic_pointer_cast<InterpolatedSurvivalProbabilityCurve<LogLinear>>(*dpts)) {
737 return c->times();
738 } else if (auto c = boost::dynamic_pointer_cast<InterpolatedHazardRateCurve<BackwardFlat>>(*dpts)) {
739 return c->times();
740 } else if (auto c = boost::dynamic_pointer_cast<SurvivalProbabilityCurve<LogLinear>>(*dpts)) {
741 return c->times();
742 } else {
743 WLOG("Extraction of time points failed, unsupported default probability termstructure");
744 return vector<double>();
745 }

748QuantLib::Handle<QuantLib::DefaultProbabilityTermStructure>
749SyntheticCDO::buildCalibratedConstiuentCurve(const QuantLib::Handle<QuantLib::DefaultProbabilityTermStructure>& curve,
750 const boost::shared_ptr<SimpleQuote>& calibrationFactor) {
751 if (!calibrationFactor) {
752 return curve;
753 }
754 auto curveTimes = extractTimeGridDefaultCurve(curve);
755 vector<Handle<Quote>> spreads;
756 for (size_t timeIdx = 0; timeIdx < curveTimes.size(); ++timeIdx) {
757 auto sp = curve->survivalProbability(curveTimes[timeIdx]);
758 auto compQuote = boost::make_shared<CompositeQuote<std::function<double(double, double)>>>(
759 Handle<Quote>(calibrationFactor),
760 Handle<Quote>(boost::make_shared<SimpleQuote>(sp)),
761 [](const double q1, const double q2) -> double { return std::exp(-(1 - q1) * std::log(q2)); });
762 spreads.push_back(Handle<Quote>(compQuote));
763 }
764 Handle<DefaultProbabilityTermStructure> targetCurve = Handle<DefaultProbabilityTermStructure>(
765 boost::make_shared<SpreadedSurvivalProbabilityTermStructure>(curve, curveTimes, spreads));
766 if (curve->allowsExtrapolation()) {
767 targetCurve->enableExtrapolation();
768 }
769 return targetCurve;
770}

772std::string SyntheticCDO::creditCurveIdWithTerm() const {
773 auto p = ore::data::splitCurveIdWithTenor(qualifier());
774 if (p.second != 0 * Days || !isIndexTranche())
775 return qualifier();
776
777 QuantLib::Schedule s = makeSchedule(leg().schedule());
778 if (s.dates().empty())
779 return p.first;
780 QuantLib::Period t = QuantExt::implyIndexTerm(
781 indexStartDateHint_ == Date() ? s.dates().front() : indexStartDateHint_, s.dates().back());
782 if (t != 0 * Days)
783 return p.first + "_" + ore::data::to_string(t);
784 return p.first;
785}
