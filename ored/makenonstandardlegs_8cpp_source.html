28Leg makeNonStandardIborLeg(const boost::shared_ptr<IborIndex>& index, const std::vector<Date>& calcDates,
29 const std::vector<Date>& payDatesInput, const std::vector<Date>& fixingDatesInput,
30 const std::vector<Date>& resetDatesInput, const Size fixingDays,
31 const std::vector<Real>& notionals, const std::vector<Date>& notionalDatesInput,
32 const std::vector<Real>& spreads, const std::vector<Date>& spreadDatesInput,
33 const std::vector<Real>& gearings, const std::vector<Date>& gearingDatesInput,
34 const bool strictNotionalDates, const DayCounter& dayCounter, const Calendar& payCalendar,
35 const BusinessDayConvention payConv, const Period& payLag, const bool isInArrears) {
36
37 // checks
38
39 QL_REQUIRE(calcDates.size() >= 2,
40 "makeNonStandardIborLeg(): calc dates size (" << calcDates.size() << ") >= 2 required");
41 QL_REQUIRE(!notionals.empty(), "makeNonStandardIborLeg(): empty notinoals");
42 QL_REQUIRE(notionalDatesInput.empty() || notionalDatesInput.size() == notionals.size() - 1,
43 "makeNonStandardIborLeg(): notional dates (" << notionalDatesInput.size() << ") must match notional ("
44 << notionals.size() << ") minus 1");
45 QL_REQUIRE(spreadDatesInput.empty() || spreadDatesInput.size() == spreads.size() - 1,
46 "makeNonStandardIborLeg(): spread dates (" << spreadDatesInput.size() << ") must match spread ("
47 << spreads.size() << ") minus 1");
48 QL_REQUIRE(gearingDatesInput.empty() || gearingDatesInput.size() == gearings.size() - 1,
49 "makeNonStandardIborLeg(): gearing dates (" << gearingDatesInput.size() << ") must match gearing ("
50 << gearings.size() << ") minus 1");
51
52 // populate pay dates, reset dates, fixing dates, notional dates if not explicitly given
53
54 std::vector<Date> payDates(payDatesInput);
55 std::vector<Date> resetDates(resetDatesInput);
56 std::vector<Date> fixingDates(fixingDatesInput);
57 std::vector<Date> notionalDates(notionalDatesInput);
58 std::vector<Date> spreadDates(spreadDatesInput);
59 std::vector<Date> gearingDates(gearingDatesInput);
60
61 if (payDates.empty()) {
62 for (Size i = 1; i < calcDates.size(); ++i) {
63 payDates.push_back(payCalendar.advance(calcDates[i], payLag, payConv));
64 }
65 }
66
67 if (resetDates.empty() && fixingDates.empty()) {
68 for (Size i = 0; i < calcDates.size() - 1; ++i) {
69 resetDates.push_back(calcDates[i]);
70 fixingDates.push_back(index->fixingCalendar().advance(isInArrears ? calcDates[i + 1] : calcDates[i],
71 -static_cast<int>(fixingDays) * Days, Preceding));
72 }
73 } else if (resetDates.empty()) {
74 for (Size i = 0; i < fixingDates.size(); ++i) {
75 resetDates.push_back(index->fixingCalendar().advance(fixingDates[i], fixingDays * Days, Following));
76 }
77 } else if (fixingDates.empty()) {
78 for (Size i = 0; i < resetDates.size(); ++i) {
79 fixingDates.push_back(
80 index->fixingCalendar().advance(resetDates[i], -static_cast<int>(fixingDays) * Days, Preceding));
81 }
82 }
83
84 if (notionalDates.empty()) {
85 for (Size i = 1; i < notionals.size(); ++i)
86 notionalDates.push_back(calcDates[i]);
87 }
88
89 if (spreadDates.empty()) {
90 for (Size i = 1; i < spreads.size(); ++i)
91 spreadDates.push_back(calcDates[i]);
92 }
93
94 if (gearingDates.empty()) {
95 for (Size i = 1; i < gearings.size(); ++i)
96 gearingDates.push_back(calcDates[i]);
97 }
98
99 // more checks
100
101 QL_REQUIRE(payDates.size() == calcDates.size() - 1, "makeNonStandardIborLeg(): pay dates size ("
102 << payDates.size() << ") = calc dates size ("
103 << calcDates.size() << ") minus 1 required");
104 QL_REQUIRE(fixingDates.size() == resetDates.size(), "makeNonStandardIborLeg(): fixing dates ("
105 << fixingDates.size() << ") must match reset dates ("
106 << resetDates.size() << ")");
107
108 for (Size i = 0; i < fixingDates.size(); ++i) {
109 QL_REQUIRE(fixingDates[i] <= resetDates[i], "makeNonStandardIborLeg(): fixing date at "
110 << i << " (" << fixingDates[i]
111 << ") must be less or equal to reset date at " << i << " ("
112 << resetDates[i] << ")");
113 QL_REQUIRE(resetDates[i] <= calcDates.back(), "makeNonStandardIborLeg(): reset date at "
114 << i << " (" << resetDates[i]
115 << ") must be less or equal last calculation date ("
116 << calcDates.back());
117 }
118
119 for (Size i = 0; i < calcDates.size() - 1; ++i) {
120 QL_REQUIRE(calcDates[i] <= calcDates[i + 1], "makeNonStandardIborLeg(): calc date at "
121 << i << " (" << calcDates[i]
122 << ") must be less or equal calc date at " << (i + 1) << " ("
123 << calcDates[i + 1]);
124 }
125
126 for (Size i = 0; i < fixingDates.size() - 1; ++i) {
127 QL_REQUIRE(fixingDates[i] <= fixingDates[i + 1], "makeNonStandardIborLeg(): fixing date at "
128 << i << " (" << fixingDates[i]
129 << ") must be less or equal fixing date at " << (i + 1)
130 << " (" << fixingDates[i + 1] << ")");
131 }
132
133 for (Size i = 0; i < resetDates.size() - 1; ++i) {
134 QL_REQUIRE(resetDates[i] <= resetDates[i + 1], "makeNonStandardIborLeg(): reset date at "
135 << i << " (" << resetDates[i]
136 << ") must be less or equal reset date at " << (i + 1)
137 << " (" << resetDates[i + 1] << ")");
138 }
139
140 // build calculation periods including broken periods due to notional resets or fixing resets
141
142 std::set<Date> effCalcDates;
143
144 for (auto const& d : calcDates)
145 effCalcDates.insert(d);
146
147 for (auto const& d : resetDates)
148 effCalcDates.insert(d);
149
150 if (strictNotionalDates)
151 for (auto const& d : notionalDates) {
152 effCalcDates.insert(d);
153 }
154
155 // build coupons
156
157 Leg leg;
158
159 for (auto startDate = effCalcDates.begin(); startDate != std::next(effCalcDates.end(), -1); ++startDate) {
160
161 // determine calc end date from start date
162
163 Date endDate = *std::next(startDate, 1);
164
165 // determine pay date
166
167 auto nextCalcDate = std::lower_bound(calcDates.begin(), calcDates.end(), endDate);
168 QL_REQUIRE(nextCalcDate != calcDates.begin(),
169 "makeNonStandardIborLeg(): internal error, nextCalcDate == calcDates.begin(), contact dev.");
170 Date payDate = payDates[std::distance(calcDates.begin(), nextCalcDate) - 1];
171
172 // determine reset and thereby fixing date
173
174 auto nextReset = std::upper_bound(resetDates.begin(), resetDates.end(), *startDate);
175 QL_REQUIRE(nextReset != resetDates.begin(),
176 "makeNonStandardIborLeg(): calc start date "
177 << *startDate << " is before first reset date " << *resetDates.begin()
178 << ". Ensure that there is a reset date on or before the calc start date.");
179 Date fixingDate = fixingDates[std::distance(resetDates.begin(), nextReset) - 1];
180
181 // determine notional
182
183 auto notionalDate = std::upper_bound(notionalDates.begin(), notionalDates.end(), *startDate);
184 Real notional = notionals[std::distance(notionalDates.begin(), notionalDate)];
185
186 // determine spread
187
188 auto spreadDate = std::upper_bound(spreadDates.begin(), spreadDates.end(), *startDate);
189 Real spread = spreads[std::distance(spreadDates.begin(), spreadDate)];
190
191 // determine gearing
192
193 auto gearingDate = std::upper_bound(gearingDates.begin(), gearingDates.end(), *startDate);
194 Real gearing = gearings[std::distance(gearingDates.begin(), gearingDate)];
195
196 // build coupon
197
198 leg.push_back(boost::make_shared<IborCoupon>(payDate, notional, *startDate, endDate, fixingDate, index, gearing,
199 spread, Date(), Date(), dayCounter));
200 }
201
202 return leg;
203}
