148void FloatingLegData::fromXML(XMLNode* node) {
149 XMLUtils::checkNode(node, legNodeName());
150 index_ = internalIndexName(XMLUtils::getChildValue(node, "Index", true));
151 indices_.insert(index_);
152 // These are all optional

154 &parseReal);
155 isInArrears_ = boost::none;
156 lastRecentPeriod_ = boost::none;
157 isAveraged_ = hasSubPeriods_ = includeSpread_ = false;
158 if (XMLNode* arrNode = XMLUtils::getChildNode(node, "IsInArrears"))
159 isInArrears_ = parseBool(XMLUtils::getNodeValue(arrNode));
160 if (XMLNode* lrNode = XMLUtils::getChildNode(node, "LastRecentPeriod"))
161 lastRecentPeriod_ = parsePeriod(XMLUtils::getNodeValue(lrNode));
162 lastRecentPeriodCalendar_ = XMLUtils::getChildValue(node, "LastRecentPeriodCalendar", false);
163 if (XMLNode* avgNode = XMLUtils::getChildNode(node, "IsAveraged"))
164 isAveraged_ = parseBool(XMLUtils::getNodeValue(avgNode));
165 if (XMLNode* spNode = XMLUtils::getChildNode(node, "HasSubPeriods"))
166 hasSubPeriods_ = parseBool(XMLUtils::getNodeValue(spNode));
167 if (XMLNode* incSpNode = XMLUtils::getChildNode(node, "IncludeSpread"))
168 includeSpread_ = parseBool(XMLUtils::getNodeValue(incSpNode));
169 if (auto n = XMLUtils::getChildNode(node, "FixingDays"))
170 fixingDays_ = parseInteger(XMLUtils::getNodeValue(n));
171 else
172 fixingDays_ = Null<Size>();
173 if (auto n = XMLUtils::getChildNode(node, "Lookback"))
174 lookback_ = parsePeriod(XMLUtils::getNodeValue(n));
175 else
176 lookback_ = 0 * Days;
177 if (auto n = XMLUtils::getChildNode(node, "RateCutoff"))
178 rateCutoff_ = parseInteger(XMLUtils::getNodeValue(n));
179 else
180 rateCutoff_ = Null<Size>();
181 caps_ = XMLUtils::getChildrenValuesWithAttributes<Real>(node, "Caps", "Cap", "startDate", capDates_, &parseReal);
182 floors_ =
183 XMLUtils::getChildrenValuesWithAttributes<Real>(node, "Floors", "Floor", "startDate", floorDates_, &parseReal);
184 gearings_ = XMLUtils::getChildrenValuesWithAttributes<Real>(node, "Gearings", "Gearing", "startDate", gearingDates_,
185 &parseReal, false);
186 if (XMLUtils::getChildNode(node, "NakedOption"))
187 nakedOption_ = XMLUtils::getChildValueAsBool(node, "NakedOption", false);
188 else
189 nakedOption_ = false;
190
191 if (XMLUtils::getChildNode(node, "LocalCapFloor"))
192 localCapFloor_ = XMLUtils::getChildValueAsBool(node, "LocalCapFloor", false);
193 else
194 localCapFloor_ = false;
195
196 if (auto tmp = XMLUtils::getChildNode(node, "FixingSchedule")) {
197 fixingSchedule_.fromXML(tmp);
198 }
199 if (auto tmp = XMLUtils::getChildNode(node, "ResetSchedule")) {
200 resetSchedule_.fromXML(tmp);
201 }

216 if (fixingDays_ != Null<Size>())

218 if (lookback_ != 0 * Days)
219 XMLUtils::addChild(doc, node, "Lookback", ore::data::to_string(lookback_));
220 if (rateCutoff_ != Null<Size>())
221 XMLUtils::addChild(doc, node, "RateCutoff", static_cast<int>(rateCutoff_));
222 XMLUtils::addChildrenWithOptionalAttributes(doc, node, "Caps", "Cap", caps_, "startDate", capDates_);
223 XMLUtils::addChildrenWithOptionalAttributes(doc, node, "Floors", "Floor", floors_, "startDate", floorDates_);
224 XMLUtils::addChildrenWithOptionalAttributes(doc, node, "Gearings", "Gearing", gearings_, "startDate",
225 gearingDates_);
226 XMLUtils::addChildrenWithOptionalAttributes(doc, node, "Spreads", "Spread", spreads_, "startDate", spreadDates_);
227 XMLUtils::addChild(doc, node, "NakedOption", nakedOption_);
228 if (localCapFloor_)
229 XMLUtils::addChild(doc, node, "LocalCapFloor", localCapFloor_);
230 if (fixingSchedule_.hasData()) {
231 auto tmp = fixingSchedule_.toXML(doc);
232 XMLUtils::setNodeName(doc, tmp, "FixingSchedule");
233 XMLUtils::appendNode(node, tmp);
234 }
235 if (resetSchedule_.hasData()) {
236 auto tmp = resetSchedule_.toXML(doc);
237 XMLUtils::setNodeName(doc, tmp, "ResetSchedule");
238 XMLUtils::appendNode(node, tmp);
239 }
240 return node;
241}

243void CPILegData::fromXML(XMLNode* node) {
244 XMLUtils::checkNode(node, legNodeName());
245 index_ = XMLUtils::getChildValue(node, "Index", true);
246 startDate_ = XMLUtils::getChildValue(node, "StartDate", false);
247 indices_.insert(index_);
248 baseCPI_ = XMLUtils::getChildValueAsDouble(node, "BaseCPI", false, QuantLib::Null<QuantLib::Real>());
249 observationLag_ = XMLUtils::getChildValue(node, "ObservationLag", false, "");

251 if (auto c = XMLUtils::getChildNode(node, "Interpolated")) {
252 QL_REQUIRE(XMLUtils::getChildNode(node, "Interpolation") == nullptr,
253 "can not have both Interpolated and Interpolation node in CPILegData");
254 interpolation_ = parseBool(XMLUtils::getNodeValue(c)) ? "Linear" : "Flat";
255 } else {
256 interpolation_ = XMLUtils::getChildValue(node, "Interpolation", false, "");
257 }
258 XMLNode* subNomNode = XMLUtils::getChildNode(node, "SubtractInflationNotional");
259 if (subNomNode)
260 subtractInflationNominal_ = XMLUtils::getChildValueAsBool(node, "SubtractInflationNotional", true);
261 else
262 subtractInflationNominal_ = false;
263
264 XMLNode* subNomCpnsNode = XMLUtils::getChildNode(node, "SubtractInflationNotionalAllCoupons");
265 if (subNomCpnsNode)
266 subtractInflationNominalCoupons_ =
267 XMLUtils::getChildValueAsBool(node, "SubtractInflationNotionalAllCoupons", true);
268 else
269 subtractInflationNominalCoupons_ = false;
270
271 rates_ = XMLUtils::getChildrenValuesWithAttributes<Real>(node, "Rates", "Rate", "startDate", rateDates_, &parseReal,
272 true);
273 caps_ = XMLUtils::getChildrenValuesWithAttributes<Real>(node, "Caps", "Cap", "startDate", capDates_, &parseReal);
274 floors_ =
275 XMLUtils::getChildrenValuesWithAttributes<Real>(node, "Floors", "Floor", "startDate", floorDates_, &parseReal);
276
277 finalFlowCap_ = Null<Real>();
278 if (auto n = XMLUtils::getChildNode(node, "FinalFlowCap")) {
279 string v = XMLUtils::getNodeValue(n);
280 if (!v.empty())
281 finalFlowCap_ = parseReal(XMLUtils::getNodeValue(n));
282 }
283
284 finalFlowFloor_ = Null<Real>();
285 if (auto n = XMLUtils::getChildNode(node, "FinalFlowFloor")) {
286 string v = XMLUtils::getNodeValue(n);
287 if (!v.empty())
288 finalFlowFloor_ = parseReal(XMLUtils::getNodeValue(n));
289 }
290
291 if (XMLUtils::getChildNode(node, "NakedOption"))
292 nakedOption_ = XMLUtils::getChildValueAsBool(node, "NakedOption", false);
293 else
294 nakedOption_ = false;
295}

928Leg makeFixedLeg(const LegData& data, const QuantLib::Date& openEndDateReplacement) {
929
930 boost::shared_ptr<FixedLegData> fixedLegData = boost::dynamic_pointer_cast<FixedLegData>(data.concreteLegData());
931 QL_REQUIRE(fixedLegData, "Wrong LegType, expected Fixed, got " << data.legType());
932
933 // build schedules
934
935 Schedule schedule;
936 Schedule paymentSchedule;
937 ScheduleBuilder scheduleBuilder;
938 scheduleBuilder.add(schedule, data.schedule());
939 scheduleBuilder.add(paymentSchedule, data.paymentSchedule());
940 scheduleBuilder.makeSchedules(openEndDateReplacement);
941
942 // Get explicit payment dates, if given
943
944 vector<Date> paymentDates;
945
946 if (!paymentSchedule.empty()) {
947 paymentDates = paymentSchedule.dates();
948 } else if (!data.paymentDates().empty()) {
949 BusinessDayConvention paymentDatesConvention =
950 data.paymentConvention().empty() ? Unadjusted : parseBusinessDayConvention(data.paymentConvention());
951 Calendar paymentDatesCalendar =
952 data.paymentCalendar().empty() ? NullCalendar() : parseCalendar(data.paymentCalendar());
953 paymentDates = parseVectorOfValues<Date>(data.paymentDates(), &parseDate);
954 for (Size i = 0; i < paymentDates.size(); i++)
955 paymentDates[i] = paymentDatesCalendar.adjust(paymentDates[i], paymentDatesConvention);
956 }
957
958 // set payment calendar
959
960 Calendar paymentCalendar;
961 if (!data.paymentCalendar().empty())
962 paymentCalendar = parseCalendar(data.paymentCalendar());
963 else if (!paymentSchedule.calendar().empty())
964 paymentCalendar = paymentSchedule.calendar();
965 else if (!schedule.calendar().empty())
966 paymentCalendar = schedule.calendar();
967
968 // set day counter and bdc
969
970 DayCounter dc = parseDayCounter(data.dayCounter());
971 BusinessDayConvention bdc = parseBusinessDayConvention(data.paymentConvention());
972
973 // build standard schedules (for non-strict notional dates)
974
975 vector<double> rates = buildScheduledVector(fixedLegData->rates(), fixedLegData->rateDates(), schedule);
976 vector<double> notionals = buildScheduledVector(data.notionals(), data.notionalDates(), schedule);
977
978 // parse payment lag
979
980 PaymentLag paymentLag = parsePaymentLag(data.paymentLag());
981
982 // apply amortization
983
984 applyAmortization(notionals, data, schedule, true, rates);
985
986 // build leg
987
988 if (!data.strictNotionalDates()) {
989
990 // no strict notional dates
991
992 Leg leg = FixedRateLeg(schedule)
993 .withNotionals(notionals)
994 .withCouponRates(rates, dc)
995 .withPaymentAdjustment(bdc)
996 .withPaymentLag(boost::apply_visitor(PaymentLagInteger(), paymentLag))
997 .withPaymentCalendar(paymentCalendar)
998 .withLastPeriodDayCounter(data.lastPeriodDayCounter().empty()
999 ? DayCounter()
1000 : parseDayCounter(data.lastPeriodDayCounter()))
1001 .withPaymentDates(paymentDates);
1002 return leg;
1003
1004 } else {
1005
1006 // strict notional dates
1007
1008 std::vector<Date> notionalDatesAsDates;
1009 std::vector<Date> rateDatesAsDates;
1010
1011 for (auto const& d : data.notionalDates()) {
1012 if (!d.empty())
1013 notionalDatesAsDates.push_back(parseDate(d));
1014 }
1015
1016 for (auto const& d : fixedLegData->rateDates()) {
1017 if (!d.empty())
1018 rateDatesAsDates.push_back(parseDate(d));
1019 }
1020
1021 return makeNonStandardFixedLeg(schedule.dates(), paymentDates, data.notionals(), notionalDatesAsDates,
1022 fixedLegData->rates(), rateDatesAsDates, data.strictNotionalDates(), dc,
1023 paymentCalendar, bdc, boost::apply_visitor(PaymentLagPeriod(), paymentLag));
1024 }

1028 boost::shared_ptr<ZeroCouponFixedLegData> zcFixedLegData =

1030 QL_REQUIRE(zcFixedLegData, "Wrong LegType, expected Zero Coupon Fixed, got " << data.legType());
1031
1032 Schedule schedule = makeSchedule(data.schedule(), openEndDateReplacement);
1033
1034 Calendar paymentCalendar;
1035 if (data.paymentCalendar().empty())
1036 paymentCalendar = schedule.calendar();
1037 else
1038 paymentCalendar = parseCalendar(data.paymentCalendar());
1039
1040 BusinessDayConvention payConvention = parseBusinessDayConvention(data.paymentConvention());
1041
1042 DayCounter dc = parseDayCounter(data.dayCounter());
1043
1044 Size numNotionals = data.notionals().size();
1045 Size numRates = zcFixedLegData->rates().size();
1046 Size numDates = schedule.size();
1047

1049 QL_REQUIRE(numNotionals >= 1,
1050 "Incorrect number of notional values entered, expected at least1, got " << numNotionals);
1051 QL_REQUIRE(numRates >= 1, "Incorrect number of rate values entered, expected at least 1, got " << numRates);

1054
1055 vector<double> rates = buildScheduledVector(zcFixedLegData->rates(), zcFixedLegData->rateDates(), schedule);

1057

1063 vector<Date> cpnDates;
1064 cpnDates.push_back(dates.front());
1065
1066 for (Size i = 0; i < numDates - 1; i++) {
1067
1068 double currentNotional = i < notionals.size() ? notionals[i] : notionals.back();
1069 double currentRate = i < rates.size() ? rates[i] : rates.back();
1070 cpnDates.push_back(dates[i + 1]);
1071 Date paymentDate = paymentCalendar.adjust(dates[i + 1], payConvention);
1072 leg.push_back(boost::make_shared<ZeroFixedCoupon>(paymentDate, currentNotional, currentRate, dc, cpnDates, comp,
1073 zcFixedLegData->subtractNotional()));
1074 }

1120 paymentCalendar = paymentSchedule.calendar();
1121 else if (!schedule.calendar().empty())
1122 paymentCalendar = schedule.calendar();

1147

1162 ? Date::minDate()
1163 : parseDate(data.amortizationData().front().startDate());

1165 bool underflow = data.amortizationData().front().underflow();
1166 vector<boost::shared_ptr<Coupon>> coupons;
1167 for (Size i = 0; i < schedule.size() - 1; i++) {
1168 Date paymentDate = paymentCalendar.adjust(schedule[i + 1], bdc);
1169 if (schedule[i] < startDate || i == 0) {
1170 boost::shared_ptr<FloatingRateCoupon> coupon;

1180 : QuantExt::SubPeriodsCoupon1::Compounding,

1188 "FloatingAnnuityCoupon needs at least one predecessor, e.g. a plain IborCoupon");

1190 << " " << coupons.back()->amount());

1200 for (Size i = 0; i < coupons.size(); i++)

1202 LOG("Floating annuity notional schedule done");

1204 }
1205 }
1206
1207 // handle sub periods leg
1208
1209 if (floatData->hasSubPeriods()) {
1210 QL_REQUIRE(floatData->caps().empty() && floatData->floors().empty(),
1211 "SubPeriodsLegs does not support caps or floors");
1212 QL_REQUIRE(!isInArrears, "SubPeriodLegs do not support in arrears fixings");

1214 .withNotionals(notionals)
1215 .withPaymentDayCounter(dc)
1216 .withPaymentAdjustment(bdc)
1217 .withGearings(gearings)
1218 .withSpreads(spreads)
1219 .withType(floatData->isAveraged() ? QuantExt::SubPeriodsCoupon1::Averaging
1220 : QuantExt::SubPeriodsCoupon1::Compounding)
1221 .includeSpread(floatData->includeSpread());

1224 }
1225
1226 // parse payment lag
1227
1228 PaymentLag paymentLag = parsePaymentLag(data.paymentLag());
1229
1230 // handle ibor leg
1231
1232 Leg tmpLeg;
1233 bool isNonStandard;
1234
1235 if (!data.strictNotionalDates() && fixingSchedule.empty() && resetSchedule.empty()) {
1236
1237 // no strict notional dates, no fixing or reset schedule
1238
1239 IborLeg iborLeg = IborLeg(schedule, index)
1240 .withNotionals(notionals)
1241 .withSpreads(spreads)
1242 .withPaymentCalendar(paymentCalendar)
1243 .withPaymentDayCounter(dc)
1244 .withPaymentAdjustment(bdc)
1245 .withFixingDays(fixingDays)
1246 .inArrears(isInArrears)
1247 .withGearings(gearings)

1251 iborLeg.withCaps(buildScheduledVector(floatData->caps(), floatData->capDates(), schedule));

1253 iborLeg.withFloors(buildScheduledVector(floatData->floors(), floatData->floorDates(), schedule));
1254
1255 tmpLeg = iborLeg;
1256 isNonStandard = false;
1257
1258 } else {
1259
1260 // strict notional dates, fixing or reset schedule present
1261
1262 QL_REQUIRE(!hasCapsFloors, "Ibor leg with strict notional or reset dates, explicit fixing or reset schedule "
1263 "does not support cap / floors");

1266 std::vector<Date> spreadDatesAsDates;
1267 std::vector<Date> gearingDatesAsDates;

1271 notionalDatesAsDates.push_back(parseDate(d));

1273
1274 for (auto const& d : floatData->spreadDates()) {
1275 if (!d.empty())
1276 spreadDatesAsDates.push_back(parseDate(d));

1278
1279 for (auto const& d : floatData->gearingDates()) {
1280 if (!d.empty())
1281 gearingDatesAsDates.push_back(parseDate(d));
1282 }
1283
1284 tmpLeg = makeNonStandardIborLeg(index, schedule.dates(), paymentDates, fixingSchedule.dates(),
1285 resetSchedule.dates(), fixingDays, data.notionals(), notionalDatesAsDates,
1286 floatData->spreads(), spreadDatesAsDates, floatData->gearings(),
1287 gearingDatesAsDates, data.strictNotionalDates(), dc, paymentCalendar, bdc,
1288 boost::apply_visitor(PaymentLagPeriod(), paymentLag), isInArrears);
1289
1290 isNonStandard = true;
1291 }
1292
1293 if (attachPricer && (hasCapsFloors || isInArrears || isNonStandard)) {

1295 QL_REQUIRE(builder, "No builder found for CapFlooredIborLeg");

1299 }
1300
1301 // build naked option leg if required
1302
1303 if (floatData->nakedOption()) {
1304 tmpLeg = StrippedCappedFlooredCouponLeg(tmpLeg);
1305 }
1306
1307 // return the leg
1308
1309 return tmpLeg;

1505Leg makeNotionalLeg(const Leg& refLeg, const bool initNomFlow, const bool finalNomFlow, const bool amortNomFlow,
1506 const Natural paymentLag, const BusinessDayConvention paymentConvention,
1507 const Calendar paymentCalendar, const bool excludeIndexing) {
1508
1509 // Assumption - Cashflows on Input Leg are all coupons
1510 // This is the Leg to be populated
1511 Leg leg;
1512
1513 // Initial Flow Amount
1514 if (initNomFlow) {
1515 auto coupon = boost::dynamic_pointer_cast<QuantLib::Coupon>(refLeg[0]);
1516 QL_REQUIRE(coupon, "makeNotionalLeg does not support non-coupon legs");
1517 double initFlowAmt = (excludeIndexing ? unpackIndexedCoupon(coupon) : coupon)->nominal();
1518 Date initDate = coupon->accrualStartDate();
1519 initDate = paymentCalendar.advance(initDate, paymentLag, Days, paymentConvention);
1520 if (initFlowAmt != 0)
1521 leg.push_back(boost::shared_ptr<CashFlow>(new SimpleCashFlow(-initFlowAmt, initDate)));
1522 }
1523
1524 // Amortization Flows
1525 if (amortNomFlow) {
1526 for (Size i = 1; i < refLeg.size(); i++) {
1527 auto coupon = boost::dynamic_pointer_cast<QuantLib::Coupon>(refLeg[i]);
1528 QL_REQUIRE(coupon, "makeNotionalLeg does not support non-coupon legs");
1529 auto coupon2 = boost::dynamic_pointer_cast<QuantLib::Coupon>(refLeg[i - 1]);
1530 QL_REQUIRE(coupon, "makeNotionalLeg does not support non-coupon legs");
1531 Date flowDate = coupon->accrualStartDate();
1532 flowDate = paymentCalendar.advance(flowDate, paymentLag, Days, paymentConvention);
1533 Real initNom = (excludeIndexing ? unpackIndexedCoupon(coupon2) : coupon2)->nominal();
1534 Real newNom = (excludeIndexing ? unpackIndexedCoupon(coupon) : coupon)->nominal();
1535 Real flow = initNom - newNom;
1536 if (flow != 0)
1537 leg.push_back(boost::shared_ptr<CashFlow>(new SimpleCashFlow(flow, flowDate)));
1538 }
1539 }
1540
1541 // Final Nominal Return at Maturity
1542 if (finalNomFlow) {
1543 auto coupon = boost::dynamic_pointer_cast<QuantLib::Coupon>(refLeg.back());
1544 QL_REQUIRE(coupon, "makeNotionalLeg does not support non-coupon legs");
1545 double finalNomFlow = (excludeIndexing ? unpackIndexedCoupon(coupon) : coupon)->nominal();
1546 Date finalDate = coupon->accrualEndDate();
1547 finalDate = paymentCalendar.advance(finalDate, paymentLag, Days, paymentConvention);
1548 if (finalNomFlow != 0)
1549 leg.push_back(boost::shared_ptr<CashFlow>(new SimpleCashFlow(finalNomFlow, finalDate)));
1550 }
1551
1552 return leg;
1553}

2481 lastAmortDate = schedule[i];
2482 } else if (i > 0 && lastAmortDate > Date::minDate())
2483 nominals[i] = nominals[i - 1];
2484 Real dcf = dc.yearFraction(schedule[i], schedule[i + 1]);
2485 Real rate = i < rates.size() ? rates[i] : rates.back();
2486 amort = annuity - rate * nominals[i] * dcf;
2487 if (amort > nominals[i] && underflow == false) {
2488 amort = std::max(nominals[i], 0.0);
2489 }
2490 }
2491 DLOG("Fixed Annuity notional schedule done");
2492 return nominals;
2493}

2495vector<double> buildAmortizationScheduleLinearToMaturity(const vector<double>& notionals, const Schedule& schedule,
2496 const AmortizationData& data) {
2497 DLOG("Build linear-to-maturity notional schedule");
2498 vector<double> nominals = normaliseToSchedule(notionals, schedule, (Real)Null<Real>());
2499 Date startDate = data.startDate().empty() ? Date::minDate() : parseDate(data.startDate());
2500 Date endDate = data.endDate().empty() ? Date::maxDate() : parseDate(data.endDate());
2501 Period amortPeriod = data.frequency().empty() ? 0 * Days : parsePeriod(data.frequency());
2502 Date lastAmortDate = Date::minDate();
2503 Real periodAmortization = Null<Real>();
2504 Real accumulatedAmortization = 0.0;
2505 for (Size i = 0; i < schedule.size() - 1; ++i) {
2506 if (schedule[i] >= startDate && periodAmortization == Null<Real>()) {
2507 periodAmortization = nominals[i] / static_cast<Real>(schedule.size() - i);
2508 }
2509 if (i > 0 && schedule[i] >= startDate && schedule[i] < endDate) {
2510 accumulatedAmortization += periodAmortization;
2511 }
2512 if (i > 0 && (lastAmortDate == Date::minDate() || schedule[i] > lastAmortDate + amortPeriod - 4 * Days) &&
2513 schedule[i] >= startDate && schedule[i] < endDate) {
2514 nominals[i] = nominals[i - 1] - accumulatedAmortization;
2515 accumulatedAmortization = 0.0;
2516 lastAmortDate = schedule[i];
2517 } else if (i > 0 && lastAmortDate > Date::minDate()) {
2518 nominals[i] = nominals[i - 1];
2519 }
2520 }
2521 DLOG("Linear-to-maturity notional schedule done");
2522 return nominals;
2523}

2525void applyAmortization(std::vector<Real>& notionals, const LegData& data, const Schedule& schedule,
2526 const bool annuityAllowed, const std::vector<Real>& rates) {
2527 Date lastEndDate = Date::minDate();
2528 for (Size i = 0; i < data.amortizationData().size(); ++i) {
2529 const AmortizationData& amort = data.amortizationData()[i];
2530 if (!amort.initialized())
2531 continue;
2532 QL_REQUIRE(i == 0 || !amort.startDate().empty(),
2533 "All AmortizationData blocks except the first require a StartDate");
2534 Date startDate = amort.startDate().empty() ? Date::minDate() : parseDate(amort.startDate());
2535 QL_REQUIRE(startDate >= lastEndDate, "Amortization start date ("
2536 << startDate << ") is earlier than last end date (" << lastEndDate
2537 << ")");
2538 lastEndDate = amort.endDate().empty() ? Date::minDate() : parseDate(amort.endDate());
2539 AmortizationType amortizationType = parseAmortizationType(amort.type());
2540 if (amortizationType == AmortizationType::FixedAmount)
2541 notionals = buildAmortizationScheduleFixedAmount(notionals, schedule, amort);
2542 else if (amortizationType == AmortizationType::RelativeToInitialNotional)
2543 notionals = buildAmortizationScheduleRelativeToInitialNotional(notionals, schedule, amort);
2544 else if (amortizationType == AmortizationType::RelativeToPreviousNotional)
2545 notionals = buildAmortizationScheduleRelativeToPreviousNotional(notionals, schedule, amort);
2546 else if (amortizationType == AmortizationType::Annuity) {
2547 QL_REQUIRE(annuityAllowed, "Amortization type Annuity not allowed for leg type " << data.legType());
2548 if (!rates.empty())
2549 notionals = buildAmortizationScheduleFixedAnnuity(notionals, rates, schedule, amort,
2550 parseDayCounter(data.dayCounter()));
2551 } else if (amortizationType == AmortizationType::LinearToMaturity) {
2552 notionals = buildAmortizationScheduleLinearToMaturity(notionals, schedule, amort);
2553 } else {
2554 QL_FAIL("AmortizationType " << amort.type() << " not supported");
2555 }
2556 // check that for a floating leg we only have one amortization block, if the type is annuity
2557 // we recognise a floating leg by an empty (fixed) rates vector
2558 if (rates.empty() && amortizationType == AmortizationType::Annuity) {
2559 QL_REQUIRE(data.amortizationData().size() == 1,
2560 "Floating Leg supports only one amortisation block of type Annuity");
2561 }
2562 }
2563}

2565void applyIndexing(Leg& leg, const LegData& data, const boost::shared_ptr<EngineFactory>& engineFactory,
2566 RequiredFixings& requiredFixings, const QuantLib::Date& openEndDateReplacement,
2567 const bool useXbsCurves) {
2568 for (auto const& indexing : data.indexing()) {
2569 if (indexing.hasData()) {
2570 DLOG("apply indexing (index='" << indexing.index() << "') to leg of type " << data.legType());
2571 QL_REQUIRE(engineFactory, "applyIndexing: engineFactory required");
2572
2573 // we allow indexing by equity, commodity and FX indices (technically any QuantLib::Index
2574 // will work, so the list of index types can be extended here if required)
2575 boost::shared_ptr<Index> index;
2576 string config = engineFactory->configuration(MarketContext::pricing);
2577 if (boost::starts_with(indexing.index(), "EQ-")) {
2578 string eqName = indexing.index().substr(3);
2579 index = *engineFactory->market()->equityCurve(eqName, config);
2580 } else if (boost::starts_with(indexing.index(), "FX-")) {
2581 auto tmp = parseFxIndex(indexing.index());
2582 Currency ccy1 = tmp->targetCurrency();
2583 Currency ccy2 = tmp->sourceCurrency();
2584 QL_REQUIRE(ccy1.code() == data.currency() || ccy2.code() == data.currency(),
2585 "applyIndexing: fx index '" << indexing.index() << "' ccys do not match leg ccy ("
2586 << data.currency() << ")");
2587 std::string domestic = data.currency();
2588 std::string foreign = ccy1.code() == domestic ? ccy2.code() : ccy1.code();
2589 index = buildFxIndex(indexing.index(), domestic, foreign, engineFactory->market(),
2590 engineFactory->configuration(MarketContext::pricing), useXbsCurves);
2591
2592 } else if (boost::starts_with(indexing.index(), "COMM-")) {
2593 auto tmp = parseCommodityIndex(indexing.index());
2594 index = parseCommodityIndex(indexing.index(), true,
2595 engineFactory->market()->commodityPriceCurve(tmp->underlyingName(), config),
2596 tmp->fixingCalendar());
2597 } else if (boost::starts_with(indexing.index(), "BOND-")) {
2598 // if we build a bond index, we add the required fixings for the bond underlying
2599 boost::shared_ptr<BondIndex> bi = parseBondIndex(indexing.index());
2600 QL_REQUIRE(!(boost::dynamic_pointer_cast<BondFuturesIndex>(bi)),
2601 "BondFuture Legs are not yet supported");
2602 BondData bondData(bi->securityName(), 1.0);
2603 index = buildBondIndex(bondData, indexing.indexIsDirty(), indexing.indexIsRelative(),
2604 parseCalendar(indexing.fixingCalendar()),
2605 indexing.indexIsConditionalOnSurvival(), engineFactory, requiredFixings);
2606 } else {
2607 QL_FAIL("invalid index '" << indexing.index()
2608 << "' in indexing data, expected EQ-, FX-, COMM-, BOND- index");
2609 }
2610
2611 QL_REQUIRE(index, "applyIndexing(): index is null, this is unexpected");
2612
2613 // apply the indexing
2614 IndexedCouponLeg indLeg(leg, indexing.quantity(), index);
2615 indLeg.withInitialFixing(indexing.initialFixing());
2616 // we set the initial notional fixing only if we have an initial exchange, otherwise this is applied to the
2617 // first notional payment appearing in the leg
2618 if (data.notionalInitialExchange())
2619 indLeg.withInitialNotionalFixing(indexing.initialNotionalFixing());
2620 indLeg.withFixingDays(indexing.fixingDays());
2621 indLeg.inArrearsFixing(indexing.inArrearsFixing());
2622 if (indexing.valuationSchedule().hasData())
2623 indLeg.withValuationSchedule(makeSchedule(indexing.valuationSchedule(), openEndDateReplacement));
2624 if (!indexing.fixingCalendar().empty())
2625 indLeg.withFixingCalendar(parseCalendar(indexing.fixingCalendar()));
2626 if (!indexing.fixingConvention().empty())
2627 indLeg.withFixingConvention(parseBusinessDayConvention(indexing.fixingConvention()));
2628 leg = indLeg;
2629 }
2630 }

2702Leg joinLegs(const std::vector<Leg>& legs) {
2703 Leg masterLeg;
2704 Size lastLeg = Null<Size>();
2705 for (Size i = 0; i < legs.size(); ++i) {
2706 // skip empty legs
2707 if (legs[i].empty())
2708 continue;
2709 // check if the periods of adjacent legs are consistent
2710 if (lastLeg != Null<Size>()) {
2711 auto lcpn = boost::dynamic_pointer_cast<Coupon>(legs[lastLeg].back());
2712 auto fcpn = boost::dynamic_pointer_cast<Coupon>(legs[i].front());
2713 QL_REQUIRE(lcpn, "joinLegs: expected coupon as last cashflow in leg #" << lastLeg);
2714 QL_REQUIRE(fcpn, "joinLegs: expected coupon as first cashflow in leg #" << i);
2715 QL_REQUIRE(lcpn->accrualEndDate() == fcpn->accrualStartDate(),
2716 "joinLegs: accrual end date of last coupon in leg #"
2717 << lastLeg << " (" << lcpn->accrualEndDate()
2718 << ") is not equal to accrual start date of first coupon in leg #" << i << " ("
2719 << fcpn->accrualStartDate() << ")");
2720 lastLeg = i;
2721 }
2722 // copy legs together
2723 masterLeg.insert(masterLeg.end(), legs[i].begin(), legs[i].end());
2724 }
2725 return masterLeg;

2728Leg buildNotionalLeg(const LegData& data, const Leg& leg, RequiredFixings& requiredFixings,
2729 const boost::shared_ptr<Market>& market, const std::string& configuration) {
2730
2731 if (!data.isNotResetXCCY()) {
2732 // If we have an FX resetting leg, add the notional amount at the start and end of each coupon period.
2733 DLOG("Building Resetting XCCY Notional leg");
2734 Real foreignNotional = data.foreignAmount();
2735
2736 QL_REQUIRE(!data.fxIndex().empty(), "buildNotionalLeg(): need fx index for fx resetting leg");
2737 auto fxIndex =
2738 buildFxIndex(data.fxIndex(), data.currency(), data.foreignCurrency(), market, configuration, true);
2739
2740 Leg resettingLeg;
2741 for (Size j = 0; j < leg.size(); j++) {
2742
2743 boost::shared_ptr<Coupon> c = boost::dynamic_pointer_cast<Coupon>(leg[j]);
2744 QL_REQUIRE(c, "Expected each cashflow in FX resetting leg to be of type Coupon");
2745
2746 // Build a pair of notional flows, one at the start and one at the end of the accrual period.
2747 // They both have the same FX fixing date => same amount in this leg's currency.
2748 boost::shared_ptr<CashFlow> outCf;
2749 boost::shared_ptr<CashFlow> inCf;
2750 Date fixingDate;
2751 if (j == 0) {
2752 // Two possibilities for first coupon:
2753 // 1. we have not been given a domestic notional so it is an FX linked coupon
2754 // 2. we have been given an explicit domestic notional so it is a simple cashflow
2755 if (data.notionals().size() == 0) {
2756 fixingDate = fxIndex->fixingDate(c->accrualStartDate());
2757 if (data.notionalInitialExchange()) {
2758 outCf = boost::make_shared<FXLinkedCashFlow>(c->accrualStartDate(), fixingDate,
2759 -foreignNotional, fxIndex);
2760 }
2761 // if there is only one period we generate the cash flow at the period end
2762 // only if there is a final notional exchange
2763 if (leg.size() > 1 || data.notionalFinalExchange()) {
2764 inCf = boost::make_shared<FXLinkedCashFlow>(c->accrualEndDate(), fixingDate, foreignNotional,
2765 fxIndex);
2766 }
2767 } else {
2768 if (data.notionalInitialExchange()) {
2769 outCf = boost::make_shared<SimpleCashFlow>(-c->nominal(), c->accrualStartDate());
2770 }
2771 if (leg.size() > 1 || data.notionalFinalExchange()) {
2772 inCf = boost::make_shared<SimpleCashFlow>(c->nominal(), c->accrualEndDate());
2773 }
2774 }
2775 } else {
2776 fixingDate = fxIndex->fixingDate(c->accrualStartDate());
2777 outCf =
2778 boost::make_shared<FXLinkedCashFlow>(c->accrualStartDate(), fixingDate, -foreignNotional, fxIndex);
2779 // we don't want a final one, unless there is notional exchange
2780 if (j < leg.size() - 1 || data.notionalFinalExchange()) {
2781 inCf =
2782 boost::make_shared<FXLinkedCashFlow>(c->accrualEndDate(), fixingDate, foreignNotional, fxIndex);
2783 }
2784 }
2785
2786 // Add the cashflows to the notional leg if they have been populated
2787 if (outCf) {
2788 resettingLeg.push_back(outCf);
2789 if (fixingDate != Date())
2790 requiredFixings.addFixingDate(fixingDate, data.fxIndex(), outCf->date());
2791 }
2792 if (inCf) {
2793 resettingLeg.push_back(inCf);
2794 if (fixingDate != Date())
2795 requiredFixings.addFixingDate(fixingDate, data.fxIndex(), inCf->date());
2796 }
2797 }
2798
2799 if (data.notionalAmortizingExchange()) {
2800 QL_FAIL("Cannot have an amortizing notional with FX reset");
2801 }
2802
2803 return resettingLeg;
2804
2805 } else if ((data.notionalInitialExchange() || data.notionalFinalExchange() || data.notionalAmortizingExchange()) &&
2806 (data.legType() != "CPI")) {
2807
2808 // check for notional exchanges on non FX reseting trades
2809
2810 PaymentLag payLag = parsePaymentLag(data.paymentLag());
2811 Natural payLagInteger = boost::apply_visitor(PaymentLagInteger(), payLag);
2812
2813 return makeNotionalLeg(leg, data.notionalInitialExchange(), data.notionalFinalExchange(),
2814 data.notionalAmortizingExchange(), payLagInteger,
2815 parseBusinessDayConvention(data.paymentConvention()),
2816 parseCalendar(data.paymentCalendar()), true);
2817 } else {
2818 return Leg();
2819 }
2820}
