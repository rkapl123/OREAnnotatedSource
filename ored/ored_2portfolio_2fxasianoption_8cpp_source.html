24  void FxAsianOption::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
25  // Checks
26  QL_REQUIRE(quantity_ > 0, "Fx Asian option requires a positive quatity");
27  QL_REQUIRE(strike_ > 0, "Fx Asian option requires a positive strike");
28 
29  const boost::shared_ptr<Market>& market = engineFactory->market();
30 
31  // Check that we can populate index_. Reused logic from AutomaticExercise FX vanilla options
32  QL_REQUIRE(!fxIndex_.empty(), "FX Asian option trade "
33  << id()
34  << " has automatic exercise so the FXIndex node needs to be populated.");
35 
36  // The strike is the number of units of sold currency (currency_) per unit of bought currency (assetName_).
37  // So, the convention here is that the sold currency is domestic and the bought currency is foreign.
38  // Note: intentionally use null calendar and 0 day fixing lag here because we will ask the FX index for its
39  // value on the expiry date without adjustment.
40  index_ = buildFxIndex(fxIndex_, currency_, assetName_, market,
41  engineFactory->configuration(MarketContext::pricing), "NullCalendar", 0);
42 
43  // Populate the external index name so that fixings work.
44  indexName_ = fxIndex_;
45 
46  // Build the trade using the shared functionality in the base class.
47  AsianOptionTrade::build(engineFactory);
48  }

50  void FxAsianOption::fromXML(XMLNode* node) {
51  AsianOptionTrade::fromXML(node);
52 
53  XMLNode* fxNode = XMLUtils::getChildNode(node, "FxAsianOptionData");
54  QL_REQUIRE(fxNode, "No FxAsianOptionData Node");
55 
56  option_.fromXML(XMLUtils::getChildNode(fxNode, "OptionData"));
57  QL_REQUIRE(option_.payoffType() == "Asian", "Expected PayoffType Asian for FxAsianOption.");
58  XMLNode* asianNode = XMLUtils::getChildNode(fxNode, "AsianData");
59  if (asianNode)
60  asianData_.fromXML(asianNode);
61 
62  XMLNode* scheduleDataNode = XMLUtils::getChildNode(fxNode, "ObservationDates");
63  observationDates_.fromXML(scheduleDataNode);
64 
65  XMLNode* tmpBought = XMLUtils::getChildNode(fxNode, "BoughtAmount");
66  XMLNode* tmpSold = XMLUtils::getChildNode(fxNode, "SoldAmount");
67  bool vanillaFlavour = tmpBought && tmpSold;
68  if (vanillaFlavour) {
69  LOG("Vanilla Flavour");
70  assetName_ = XMLUtils::getChildValue(fxNode, "BoughtCurrency", true);
71  currency_ = XMLUtils::getChildValue(fxNode, "SoldCurrency", true);
72  double boughtAmount = XMLUtils::getChildValueAsDouble(fxNode, "BoughtAmount", true);
73  double soldAmount = XMLUtils::getChildValueAsDouble(fxNode, "SoldAmount", true);
74  strike_ = soldAmount / boughtAmount;
75  quantity_ = boughtAmount;
76  fxIndex_ = XMLUtils::getChildValue(fxNode, "FXIndex", true);
77  } else {
78  LOG("Scripted Flavour");
79  currency_ = XMLUtils::getChildValue(fxNode, "Currency", true);
80  quantity_ = XMLUtils::getChildValueAsDouble(fxNode, "Quantity", true);
81  strike_ = XMLUtils::getChildValueAsDouble(fxNode, "Strike", false);
82  // Get fx index and asset name from the underlying node
83  XMLNode* tmp = XMLUtils::getChildNode(fxNode, "Underlying");
84  FXUnderlying underlying;
85  underlying.fromXML(tmp);
86  fxIndex_ = underlying.name();
87  boost::shared_ptr<QuantExt::FxIndex> fxi = parseFxIndex(fxIndex_);
88  assetName_ = fxi->sourceCurrency().code() == currency_ ?
89  fxi->targetCurrency().code() : fxi->sourceCurrency().code();
90  }
91  }
