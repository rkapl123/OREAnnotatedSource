37 std::string to_string(const QuantLib::Date& date) {
38  if (date == QuantLib::Date())
39  return "1900-01-01";
40 
41  char buf[11];
42  int y = date.year();
43  int m = static_cast<int>(date.month());
44  int d = date.dayOfMonth();
45 
46  int n = snprintf(buf, sizeof(buf), "%04d-%02d-%02d", y, m, d);
47  QL_REQUIRE(n == 10, "Failed to convert date " << date << " to_string() n:" << n);
48  return std::string(buf);
49 }
