38 std::string to_string(const QuantLib::Date& date) {
39  if (date == QuantLib::Date())
40  return "1900-01-01";
41 
42  char buf[11];
43  int y = date.year();
44  int m = static_cast<int>(date.month());
45  int d = date.dayOfMonth();
46 
47  int n = snprintf(buf, sizeof(buf), "%04d-%02d-%02d", y, m, d);
48  QL_REQUIRE(n == 10, "Failed to convert date " << date << " to_string() n:" << n);
49  return std::string(buf);
50 }

54 std::string to_string(const QuantLib::Period& period) {
55  Integer n = period.length();
56  Integer m = 0;
57  std::ostringstream o;
58  switch (period.units()) {
59  case Days:
60  if (n>=7) {
61  m = n/7;
62  o << m << "W";
63  n = n%7;
64  }
65  if (n != 0 || m == 0) {
66  o << n << "D";
67  return o.str();
68  }
69  else {
70  return o.str();
71  }
72  case Weeks: {
73  o << n << "W";
74  return o.str();
75  }
76  case Months:
77  if (n>=12) {
78  m = n/12;
79  o << n/12 << "Y";
80  n = n%12;
81  }
82  if (n != 0 || m == 0) {
83  o << n << "M";
84  return o.str();
85  }
86  else {
87  return o.str();
88  }
89  case Years: {
90  o << n << "Y";
91  return o.str();
92  }
93  default:
94  ALOG("unknown time unit (" << Integer(period.units()) << ")");
95  o << period;
96  return o.str();
97  }
98 }
