30template <typename T> std::string toCommaSeparatedList(const std::vector<T>& d) {
31 std::ostringstream o;
32 for (Size i = 0; i < d.size(); ++i)
33 o << d[i] << (i < d.size() - 1 ? "," : "");
34 return o.str();
35}

349void ConvertibleBondData::populateFromBondReferenceData(
350 const boost::shared_ptr<ore::data::ReferenceDataManager>& referenceData) {
351
352 QL_REQUIRE(!bondData_.securityId().empty(),
353 "ConvertibleBondData::populateFromBondReferenceData(): no security id given");
354 if (!referenceData || !referenceData->hasData(ConvertibleBondReferenceDatum::TYPE, bondData_.securityId())) {
355 DLOG("could not get ConvertibleBondReferenceDatum for name " << bondData_.securityId()
356 << " leave data in trade unchanged");
357 } else {
358 auto bondRefData = boost::dynamic_pointer_cast<ConvertibleBondReferenceDatum>(
359 referenceData->getData(ConvertibleBondReferenceDatum::TYPE, bondData_.securityId()));
360 QL_REQUIRE(bondRefData, "could not cast to ConvertibleBondReferenceDatum, this is unexpected");
361 DLOG("Got ConvertibleBondReferenceDatum for name " << bondData_.securityId()
362 << " overwrite empty elements in trade");
363 bondData_.populateFromBondReferenceData(
364 boost::make_shared<BondReferenceDatum>(bondData_.securityId(), bondRefData->bondData()));
365 if (!callData_.initialised()) {
366 DLOG("overwrite CallData from reference data")
367 callData_ = bondRefData->callData();
368 }
369 if (!putData_.initialised()) {
370 DLOG("overwrite PutData from reference data")
371 putData_ = bondRefData->putData();
372 }
373 if (!conversionData_.initialised()) {
374 DLOG("overwrite ConversionData from reference data")
375 conversionData_ = bondRefData->conversionData();
376 }
377 if (!dividendProtectionData_.initialised()) {
378 DLOG("overwrite DividendProtectionData from reference data")
379 dividendProtectionData_ = bondRefData->dividendProtectionData();
380 }
381 if (detachable_.empty()) {
382 DLOG("overwrite detachable from reference data");
383 detachable_ = bondRefData->detachable();
384 }
385 }
