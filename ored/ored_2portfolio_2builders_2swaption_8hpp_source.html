37 //! European Swaption Engine Builder
38 /*! European Swaptions are priced with Black or Bachelier pricing engines,
39  * depending on the volatility type provided by Market (if it is normal, this
40  * builder returns a Bachelier engine, if it is lognormal (or lognormal shiffed)
41  * it will be a Black engine.
42  *
43  * Engines are cached based on currency
44 
45  \ingroup builders
46  */
47 class EuropeanSwaptionEngineBuilder : public CachingPricingEngineBuilder<string, const Currency&> {
48 public:
49  EuropeanSwaptionEngineBuilder()
50  : CachingEngineBuilder("BlackBachelier", "BlackBachelierSwaptionEngine", {"EuropeanSwaption"}) {}
51 
52 protected:
53  virtual string keyImpl(const Currency& ccy) override { return ccy.code(); }
54 
55  virtual boost::shared_ptr<PricingEngine> engineImpl(const Currency& ccy) override;
56 };
