35 QuantLib::Date getInflationSwapStart(const QuantLib::Date& asof, const InflationSwapConvention& conv) {
36 
37  using IPR = InflationSwapConvention::PublicationRoll;
38 
39  // If no roll schedule, just return (as of, convention's obs lag).
40  if (conv.publicationRoll() == IPR::None) {
41  return asof;
42  }
43 
44  // Get schedule and check not empty
45  const Schedule& ps = conv.publicationSchedule();
46  QL_REQUIRE(!ps.empty(), "InflationCurve: roll on publication is true for "
47  << conv.id() << " but the publication schedule is empty.");
48 
49  // Check the schedule dates cover the as of date.
50  const vector<Date>& ds = ps.dates();
51  QL_REQUIRE(ds.front() < asof, "InflationCurve: first date in the publication schedule ("
52  << io::iso_date(ds.front()) << ") should be before the as of date ("
53  << io::iso_date(asof) << ").");
54  QL_REQUIRE(asof < ds.back(), "InflationCurve: last date in the publication schedule ("
55  << io::iso_date(ds.back()) << ") should be after the as of date ("
56  << io::iso_date(asof) << ").");
57 
58  // Find d such that d_- < asof <= d. If necessary, move to the next publication schedule date. We
59  // know that there is another date because asof < ds.back() is checked above.
60  auto it = lower_bound(ds.begin(), ds.end(), asof);
61  Date d = *it;
62  if (asof == d && conv.publicationRoll() == IPR::OnPublicationDate) {
63  d = *next(it);
64  }
65 
66  // Move d back availability lag and the 15th of that month is the helper's start date.
67  // Note: the 15th of the month is specific to AU CPI. We may need to generalise later.
68  d -= conv.index()->availabilityLag();
69 
70  return Date(15, d.month(), d.year());
71 }
