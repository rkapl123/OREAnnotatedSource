28 CSA::Type parseCsaType(const string& s) {
29  static map<string, CSA::Type> t = {
30  {"Bilateral", CSA::Bilateral}, {"CallOnly", CSA::CallOnly}, {"PostOnly", CSA::PostOnly}};
31 
32  auto it = t.find(s);
33  if (it != t.end()) {
34  return it->second;
35  } else {
36  QL_FAIL("Cannot convert \"" << s << "\" to CSA::Type");
37  }
38 }

53 void CSA::validate(string nettingSetId) {
54  QL_REQUIRE(csaCurrency_.size() == 3, "NettingSetDefinition build error;"
55  << " csa currency should be a three-letter ISO code");
56 
57  QL_REQUIRE(thresholdPay_ >= 0, "NettingSetDefinition build error; negative thresholdPay");
58  QL_REQUIRE(thresholdRcv_ >= 0, "NettingSetDefinition build error; negative thresholdRcv");
59  QL_REQUIRE(mtaPay_ >= 0, "NettingSetDefinition build error; negative mtaPay");
60  QL_REQUIRE(mtaRcv_ >= 0, "NettingSetDefinition build error; negative mtaRcv");
61  QL_REQUIRE(iaType_ == "FIXED", "NettingSetDefinition build error;"
62  << " unsupported independent amount type; " << iaType_);
63 
64  QL_REQUIRE(marginCallFreq_ > Period(0, Days) && marginPostFreq_ > Period(0, Days),
65  "NettingSetDefinition build error;"
66  << " non-positive margining frequency");
67  QL_REQUIRE(mpr_ >= Period(0, Days), "NettingSetDefinition build error;"
68  << " negative margin period of risk");
69  if (mpr_ < marginCallFreq_ || mpr_ < marginPostFreq_) {
70  LOG("NettingSet " << nettingSetId << " has CSA margining frequency (" << marginCallFreq_ << ", "
71  << marginPostFreq_ << ") longer than assumed margin period of risk " << mpr_);
72  }
73 
74  for (Size i = 0; i < eligCollatCcys_.size(); i++) {
75  QL_REQUIRE(eligCollatCcys_[i].size() == 3, "NettingSetDefinition build error;"
76  << " three-letter ISO code expected");
77  }
78 
79  // unilateral CSA - set threshold near infinity to disable margining
80  switch (type_) {
81  case CallOnly: {
82  thresholdPay_ = std::numeric_limits<double>::max();
83  break;
84  }
85  case PostOnly: {
86  thresholdRcv_ = std::numeric_limits<double>::max();
87  }
88  default:
89  break;
90  }
91 }

123 void NettingSetDefinition::fromXML(XMLNode* node) {
124  XMLUtils::checkNode(node, "NettingSet");
125 
126  // Read in the mandatory nodes.
127  nettingSetId_ = XMLUtils::getChildValue(node, "NettingSetId", true);
128  ctp_ = XMLUtils::getChildValue(node, "Counterparty", true);
129  activeCsaFlag_ = XMLUtils::getChildValueAsBool(node, "ActiveCSAFlag", true);
130 
131  // Load "CSA" information, if necessary
132  if (activeCsaFlag_) {
133  XMLNode* csaChild = XMLUtils::getChildNode(node, "CSADetails");
134  XMLUtils::checkNode(csaChild, "CSADetails");
135 
136  string csaTypeStr = XMLUtils::getChildValue(csaChild, "Bilateral", true);
137  string csaCurrency = XMLUtils::getChildValue(csaChild, "CSACurrency", true);
138  string index = XMLUtils::getChildValue(csaChild, "Index", true);
139  Real thresholdPay = XMLUtils::getChildValueAsDouble(csaChild, "ThresholdPay", true);
140  Real thresholdRcv = XMLUtils::getChildValueAsDouble(csaChild, "ThresholdReceive", true);
141  Real mtaPay = XMLUtils::getChildValueAsDouble(csaChild, "MinimumTransferAmountPay", true);
142  Real mtaRcv = XMLUtils::getChildValueAsDouble(csaChild, "MinimumTransferAmountReceive", true);
143  string mprStr = XMLUtils::getChildValue(csaChild, "MarginPeriodOfRisk", true);
144  Real collatSpreadRcv = XMLUtils::getChildValueAsDouble(csaChild, "CollateralCompoundingSpreadReceive", true);
145  Real collatSpreadPay = XMLUtils::getChildValueAsDouble(csaChild, "CollateralCompoundingSpreadPay", true);
146 
147  XMLNode* freqChild = XMLUtils::getChildNode(csaChild, "MarginingFrequency");
148  XMLUtils::checkNode(freqChild, "MarginingFrequency");
149  string marginCallFreqStr = XMLUtils::getChildValue(freqChild, "CallFrequency", true);
150  string marginPostFreqStr = XMLUtils::getChildValue(freqChild, "PostFrequency", true);
151 
152  XMLNode* iaChild = XMLUtils::getChildNode(csaChild, "IndependentAmount");
153  XMLUtils::checkNode(iaChild, "IndependentAmount");
154  Real iaHeld = XMLUtils::getChildValueAsDouble(iaChild, "IndependentAmountHeld", true);
155  string iaType = XMLUtils::getChildValue(iaChild, "IndependentAmountType", true);
156 
157  XMLNode* collatChild = XMLUtils::getChildNode(csaChild, "EligibleCollaterals");
158  XMLUtils::checkNode(collatChild, "EligibleCollaterals");
159  vector<string> eligCollatCcys = XMLUtils::getChildrenValues(collatChild, "Currencies", "Currency", true);
160 
161  bool applyInitialMargin = false;
162  XMLNode* applyInitialMarginNode = XMLUtils::getChildNode(csaChild, "ApplyInitialMargin");
163  if (applyInitialMarginNode)
164  applyInitialMargin = XMLUtils::getChildValueAsBool(csaChild, "ApplyInitialMargin", true);
165 
166  string initialMarginType = XMLUtils::getChildValue(csaChild, "InitialMarginType", false);
167  if (initialMarginType == "")
168  initialMarginType = "Bilateral";
169 
170  csa_ = boost::make_shared<CSA>(parseCsaType(csaTypeStr), csaCurrency, index, thresholdPay, thresholdRcv, mtaPay,
171  mtaRcv, iaHeld, iaType, parsePeriod(marginCallFreqStr),
172  parsePeriod(marginPostFreqStr), parsePeriod(mprStr), collatSpreadPay,
173  collatSpreadRcv, eligCollatCcys, applyInitialMargin,
174  parseCsaType(initialMarginType));
175  }
176 
177  validate();
178 }
