28CSA::Type parseCsaType(const string& s) {
29 static map<string, CSA::Type> t = {
30 {"Bilateral", CSA::Bilateral}, {"CallOnly", CSA::CallOnly}, {"PostOnly", CSA::PostOnly}};
31
32 auto it = t.find(s);
33 if (it != t.end()) {
34 return it->second;
35 } else {
36 QL_FAIL("Cannot convert \"" << s << "\" to CSA::Type");
37 }
38}

40std::ostream& operator<<(std::ostream& out, CSA::Type t) {
41 switch (t) {
42 case CSA::Bilateral:
43 return out << "Bilateral";
44 case CSA::CallOnly:
45 return out << "CallOnly";
46 case CSA::PostOnly:
47 return out << "PostOnly";
48 default:
49 QL_FAIL("csa type not covered");
50 }
51}

53void CSA::invertCSA() {
54 if (type_ != Bilateral) {
55 type_ = (type_ == CallOnly ? PostOnly : CallOnly);
56 }
57 if (initialMarginType_ != Bilateral) {
58 initialMarginType_ = (initialMarginType_ == CallOnly ? PostOnly : CallOnly);
59 }
60 std::swap<Real>(collatSpreadPay_, collatSpreadRcv_);
61 std::swap<Real>(thresholdPay_, thresholdRcv_);
62 std::swap<Real>(mtaPay_, mtaRcv_);
63 iaHeld_ *= -1;
64 std::swap<Period>(marginCallFreq_, marginPostFreq_);
65}

67void CSA::validate() {
68 QL_REQUIRE(csaCurrency_.size() == 3, "NettingSetDefinition build error;"
69 << " CSA currency should be a three-letter ISO code");
70
71 QL_REQUIRE(thresholdPay_ >= 0, "NettingSetDefinition build error; negative thresholdPay");
72 QL_REQUIRE(thresholdRcv_ >= 0, "NettingSetDefinition build error; negative thresholdRcv");
73 QL_REQUIRE(mtaPay_ >= 0, "NettingSetDefinition build error; negative mtaPay");
74 QL_REQUIRE(mtaRcv_ >= 0, "NettingSetDefinition build error; negative mtaRcv");
75 QL_REQUIRE(iaType_ == "FIXED", "NettingSetDefinition build error;"
76 << " unsupported independent amount type; " << iaType_);
77
78 QL_REQUIRE(marginCallFreq_ > Period(0, Days) && marginPostFreq_ > Period(0, Days),
79 "NettingSetDefinition build error;"
80 << " non-positive margining frequency");
81 QL_REQUIRE(mpr_ >= Period(0, Days), "NettingSetDefinition build error;"
82 << " negative margin period of risk");
83 if (mpr_ < marginCallFreq_ || mpr_ < marginPostFreq_) {
84 LOG("NettingSetDefinition has CSA margining frequency ("
85 << marginCallFreq_ << ", " << marginPostFreq_ << ") longer than assumed margin period of risk " << mpr_);
86 }
87
88 for (Size i = 0; i < eligCollatCcys_.size(); i++) {
89 QL_REQUIRE(eligCollatCcys_[i].size() == 3,
90 "NettingSetDefinition build error;"
91 << "EligibleCollaterals currency should be a three-letter ISO code");
92 }
93
94 // unilateral CSA - set threshold near infinity to disable margining
95 switch (type_) {
96 case CallOnly: {
97 thresholdPay_ = std::numeric_limits<double>::max();
98 break;
99 }
100 case PostOnly: {
101 thresholdRcv_ = std::numeric_limits<double>::max();
102 }
103 default:
104 break;
105 }
106}

138void NettingSetDefinition::fromXML(XMLNode* node) {
139 XMLUtils::checkNode(node, "NettingSet");
140
141 // Read in the mandatory nodes.
142 XMLNode* nettingSetDetailsNode = XMLUtils::getChildNode(node, "NettingSetDetails");
143 if (nettingSetDetailsNode) {
144 nettingSetDetails_.fromXML(nettingSetDetailsNode);
145 } else {
146 nettingSetId_ = XMLUtils::getChildValue(node, "NettingSetId", false);
147 nettingSetDetails_ = NettingSetDetails(nettingSetId_);
148 }
149
150 activeCsaFlag_ = XMLUtils::getChildValueAsBool(node, "ActiveCSAFlag", false, true);
151
152 // Load "CSA" information, if necessary
153 if (activeCsaFlag_) {
154 XMLNode* csaChild = XMLUtils::getChildNode(node, "CSADetails");
155 XMLUtils::checkNode(csaChild, "CSADetails");
156
157 string csaTypeStr = XMLUtils::getChildValue(csaChild, "Bilateral", false);
158 if (csaTypeStr.empty())
159 csaTypeStr = "Bilateral";
160 string csaCurrency = XMLUtils::getChildValue(csaChild, "CSACurrency", false);
161 string index = XMLUtils::getChildValue(csaChild, "Index", false);
162 Real thresholdPay = XMLUtils::getChildValueAsDouble(csaChild, "ThresholdPay", false, 0.0);
163 Real thresholdRcv = XMLUtils::getChildValueAsDouble(csaChild, "ThresholdReceive", false, 0.0);
164 Real mtaPay = XMLUtils::getChildValueAsDouble(csaChild, "MinimumTransferAmountPay", false, 0.0);
165 Real mtaRcv = XMLUtils::getChildValueAsDouble(csaChild, "MinimumTransferAmountReceive", false, 0.0);
166 string mprStr = XMLUtils::getChildValue(csaChild, "MarginPeriodOfRisk", false);
167 if (mprStr.empty())
168 mprStr = "2W";
169 Real collatSpreadRcv =
170 XMLUtils::getChildValueAsDouble(csaChild, "CollateralCompoundingSpreadReceive", false, 0.0);
171 Real collatSpreadPay = XMLUtils::getChildValueAsDouble(csaChild, "CollateralCompoundingSpreadPay", false, 0.0);
172
173 string marginCallFreqStr, marginPostFreqStr;
174 if (XMLNode* freqChild = XMLUtils::getChildNode(csaChild, "MarginingFrequency")) {
175 marginCallFreqStr = XMLUtils::getChildValue(freqChild, "CallFrequency", false);
176 marginPostFreqStr = XMLUtils::getChildValue(freqChild, "PostFrequency", false);
177 }
178 if (marginCallFreqStr.empty())
179 marginCallFreqStr = "1D";
180 if (marginPostFreqStr.empty())
181 marginPostFreqStr = "1D";
182
183 Real iaHeld = 0.0;
184 string iaType;
185 if (XMLNode* iaChild = XMLUtils::getChildNode(csaChild, "IndependentAmount")) {
186 iaHeld = XMLUtils::getChildValueAsDouble(iaChild, "IndependentAmountHeld", false, 0.0);
187 iaType = XMLUtils::getChildValue(iaChild, "IndependentAmountType", false);
188 }
189 if (iaType.empty())
190 iaType = "FIXED";
191
192 vector<string> eligCollatCcys;
193 if (XMLNode* collatChild = XMLUtils::getChildNode(csaChild, "EligibleCollaterals")) {
194 eligCollatCcys = XMLUtils::getChildrenValues(collatChild, "Currencies", "Currency", false);
195 }
196
197 bool applyInitialMargin = XMLUtils::getChildValueAsBool(csaChild, "ApplyInitialMargin", false, false);
198
199 string initialMarginType = XMLUtils::getChildValue(csaChild, "InitialMarginType", false);
200 if (initialMarginType.empty())
201 initialMarginType = "Bilateral";
202
203 bool calculateIMAmount = XMLUtils::getChildValueAsBool(csaChild, "CalculateIMAmount", false, false);
204 bool calculateVMAmount = XMLUtils::getChildValueAsBool(csaChild, "CalculateVMAmount", false, false);
205
206 csa_ = boost::make_shared<CSA>(parseCsaType(csaTypeStr), csaCurrency, index, thresholdPay, thresholdRcv, mtaPay,
207 mtaRcv, iaHeld, iaType, parsePeriod(marginCallFreqStr),
208 parsePeriod(marginPostFreqStr), parsePeriod(mprStr), collatSpreadPay,
209 collatSpreadRcv, eligCollatCcys, applyInitialMargin,
210 parseCsaType(initialMarginType), calculateIMAmount, calculateVMAmount);
211 }
212
213 validate();
214}

216XMLNode* NettingSetDefinition::toXML(XMLDocument& doc) {
217 // Allocate a node.
218 XMLNode* node = doc.allocNode("NettingSet");
219
220 // Add the mandatory members.
221 if (nettingSetDetails_.emptyOptionalFields()) {
222 XMLUtils::addChild(doc, node, "NettingSetId", nettingSetId_);
223 } else {
224 XMLUtils::appendNode(node, nettingSetDetails_.toXML(doc));
225 }
226 XMLUtils::addChild(doc, node, "ActiveCSAFlag", activeCsaFlag_);
227
228 XMLNode* csaSubNode = doc.allocNode("CSADetails");
229 XMLUtils::appendNode(node, csaSubNode);
230
231 if (activeCsaFlag_) {
232 QL_REQUIRE(csa_, "CSA details not defined");
233
234 XMLUtils::addChild(doc, csaSubNode, "Bilateral", to_string(csa_->type()));
235 XMLUtils::addChild(doc, csaSubNode, "CSACurrency", csa_->csaCurrency());
236 XMLUtils::addChild(doc, csaSubNode, "ThresholdPay", csa_->thresholdPay());
237 XMLUtils::addChild(doc, csaSubNode, "ThresholdReceive", csa_->thresholdRcv());
238 XMLUtils::addChild(doc, csaSubNode, "MinimumTransferAmountPay", csa_->mtaPay());
239 XMLUtils::addChild(doc, csaSubNode, "MinimumTransferAmountReceive", csa_->mtaRcv());
240 XMLUtils::addChild(doc, csaSubNode, "MarginPeriodOfRisk", to_string(csa_->marginPeriodOfRisk()));
241 XMLUtils::addChild(doc, csaSubNode, "CollateralCompoundingSpreadPay", csa_->collatSpreadPay());
242 XMLUtils::addChild(doc, csaSubNode, "CollateralCompoundingSpreadReceive", csa_->collatSpreadRcv());
243
244 XMLNode* freqSubNode = doc.allocNode("MarginingFrequency");
245 XMLUtils::appendNode(csaSubNode, freqSubNode);
246 XMLUtils::addChild(doc, freqSubNode, "CallFrequency", to_string(csa_->marginCallFrequency()));
247 XMLUtils::addChild(doc, freqSubNode, "PostFrequency", to_string(csa_->marginPostFrequency()));
248
249 XMLNode* iaSubNode = doc.allocNode("IndependentAmount");
250 XMLUtils::appendNode(csaSubNode, iaSubNode);
251 XMLUtils::addChild(doc, iaSubNode, "IndependentAmountHeld", csa_->independentAmountHeld());
252 XMLUtils::addChild(doc, iaSubNode, "IndependentAmountType", csa_->independentAmountType());
253
254 XMLNode* collatSubNode = doc.allocNode("EligibleCollaterals");
255 XMLUtils::appendNode(csaSubNode, collatSubNode);
256 XMLUtils::addChildren(doc, collatSubNode, "Currencies", "Currency", csa_->eligCollatCcys());
257
258 XMLUtils::addChild(doc, csaSubNode, "ApplyInitialMargin", csa_->applyInitialMargin());
259 XMLUtils::addChild(doc, csaSubNode, "InitialMarginType", to_string(csa_->initialMarginType()));
260 XMLUtils::addChild(doc, csaSubNode, "CalculateIMAmount", csa_->calculateIMAmount());
261 XMLUtils::addChild(doc, csaSubNode, "CalculateVMAmount", csa_->calculateVMAmount());
262 }
263
264 return node;
265}

267void NettingSetDefinition::validate() {

269 LOG(nettingSetLog << ": Validating netting set definition");
270 QL_REQUIRE(nettingSetId_.size() > 0 || !nettingSetDetails_.empty(),
271 "NettingSetDefinition build error; no netting set ID or netting set details");
272
273 if (activeCsaFlag_) {
274 QL_REQUIRE(csa_, "CSA not defined yet");
275 string nettingSetLog = nettingSetDetails_.empty() ? nettingSetId_ : ore::data::to_string(nettingSetDetails_);
276 LOG(nettingSetLog << ": Validating netting set definition's CSA details");
277 csa_->validate();
278 }
