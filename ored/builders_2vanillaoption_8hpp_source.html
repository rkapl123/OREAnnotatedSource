45template <class T, typename... Args> class CachingOptionEngineBuilder : public CachingPricingEngineBuilder<T, Args...> {
46public:
47 CachingOptionEngineBuilder(const string& model, const string& engine, const set<string>& tradeTypes,
48 const AssetClass& assetClass)
49 : CachingPricingEngineBuilder<T, Args...>(model, engine, tradeTypes), assetClass_(assetClass) {}
50
51protected:
52 boost::shared_ptr<QuantLib::GeneralizedBlackScholesProcess> getBlackScholesProcess(const string& assetName,
53 const Currency& ccy,
54 const AssetClass& assetClassUnderlying,
55 const std::vector<Time>& timePoints = {}) {
56
57 using VVTS = QuantExt::BlackMonotoneVarVolTermStructure;
58 string config = this->configuration(ore::data::MarketContext::pricing);
59
60 if (assetClassUnderlying == AssetClass::EQ) {
61 Handle<BlackVolTermStructure> vol = this->market_->equityVol(assetName, config);
62 if (!timePoints.empty()) {
63 vol = Handle<BlackVolTermStructure>(boost::make_shared<VVTS>(vol, timePoints));
64 vol->enableExtrapolation();
65 }
66 return boost::make_shared<QuantLib::GeneralizedBlackScholesProcess>(
67 this->market_->equitySpot(assetName, config), this->market_->equityDividendCurve(assetName, config),
68 this->market_->equityForecastCurve(assetName, config), vol);
69
70 } else if (assetClassUnderlying == AssetClass::FX) {
71 const string& ccyPairCode = assetName + ccy.code();
72 Handle<BlackVolTermStructure> vol = this->market_->fxVol(ccyPairCode, config);
73 if (!timePoints.empty()) {
74 vol = Handle<BlackVolTermStructure>(boost::make_shared<VVTS>(vol, timePoints));
75 vol->enableExtrapolation();
76 }
77 return boost::make_shared<QuantLib::GeneralizedBlackScholesProcess>(
78 this->market_->fxSpot(ccyPairCode, config), this->market_->discountCurve(assetName, config),
79 this->market_->discountCurve(ccy.code(), config), vol);
80
81 } else if (assetClassUnderlying == AssetClass::COM) {
82
83 Handle<BlackVolTermStructure> vol = this->market_->commodityVolatility(assetName, config);
84 if (!timePoints.empty()) {
85 vol = Handle<BlackVolTermStructure>(boost::make_shared<VVTS>(vol, timePoints));
86 vol->enableExtrapolation();
87 }
88
89 // Create the commodity convenience yield curve for the process
90 Handle<QuantExt::PriceTermStructure> priceCurve = this->market_->commodityPriceCurve(assetName, config);
91 Handle<Quote> commoditySpot(boost::make_shared<QuantExt::DerivedPriceQuote>(priceCurve));
92 Handle<YieldTermStructure> discount = this->market_->discountCurve(ccy.code(), config);
93 Handle<YieldTermStructure> yield(
94 boost::make_shared<QuantExt::PriceTermStructureAdapter>(*priceCurve, *discount));
95 yield->enableExtrapolation();
96
97 return boost::make_shared<QuantLib::GeneralizedBlackScholesProcess>(commoditySpot, yield, discount, vol);
98
99 } else {
100 QL_FAIL("Asset class of " << (int)assetClassUnderlying << " not recognized.");
101 }
102 }
103 AssetClass assetClass_;
104};

107/*! Pricing engines are cached by asset/currency
108
109 \ingroup builders
110 */
111class VanillaOptionEngineBuilder
112 : public CachingOptionEngineBuilder<string, const string&, const Currency&, const AssetClass&, const Date&> {
113public:
114 VanillaOptionEngineBuilder(const string& model, const string& engine, const set<string>& tradeTypes,
115 const AssetClass& assetClass, const Date& expiryDate)
116 : CachingOptionEngineBuilder(model, engine, tradeTypes, assetClass), expiryDate_(expiryDate) {}
117
118 boost::shared_ptr<PricingEngine> engine(const string& assetName, const Currency& ccy, const Date& expiryDate) {
119 return CachingPricingEngineBuilder<string, const string&, const Currency&, const AssetClass&,
120 const Date&>::engine(assetName, ccy, assetClass_, expiryDate);
121 }
122
123 boost::shared_ptr<PricingEngine> engine(const Currency& ccy1, const Currency& ccy2, const Date& expiryDate) {
124 return CachingPricingEngineBuilder<string, const string&, const Currency&, const AssetClass&,
125 const Date&>::engine(ccy1.code(), ccy2, assetClass_, expiryDate);
126 }
127
128protected:
129 virtual string keyImpl(const string& assetName, const Currency& ccy, const AssetClass& assetClassUnderlying,
130 const Date& expiryDate) override {
131 return assetName + "/" + ccy.code() + "/" + to_string(expiryDate);
132 }
133
134 Date expiryDate_;

213//! Abstract Engine Builder for American Vanilla Options using Finite Difference Method
214/*! Pricing engines are cached by asset/currency
215
216 \ingroup builders
217 */
218class AmericanOptionFDEngineBuilder : public AmericanOptionEngineBuilder {
219public:
220 AmericanOptionFDEngineBuilder(const string& model, const set<string>& tradeTypes, const AssetClass& assetClass,
221 const Date& expiryDate)
222 : AmericanOptionEngineBuilder(model, "FdBlackScholesVanillaEngine", tradeTypes, assetClass, expiryDate) {}
223
224protected:
225 virtual boost::shared_ptr<PricingEngine> engineImpl(const string& assetName, const Currency& ccy,
226 const AssetClass& assetClass, const Date& expiryDate) override {
227 // We follow the way FdBlackScholesBarrierEngine determines maturity for time grid generation
228 Handle<YieldTermStructure> riskFreeRate =
229 market_->discountCurve(ccy.code(), configuration(ore::data::MarketContext::pricing));
230 Time expiry = riskFreeRate->dayCounter().yearFraction(riskFreeRate->referenceDate(),
231 std::max(riskFreeRate->referenceDate(), expiryDate));
232
233 FdmSchemeDesc scheme = parseFdmSchemeDesc(engineParameter("Scheme"));
234 Size tGrid = (Size)(parseInteger(engineParameter("TimeGridPerYear")) * expiry);
235 Size xGrid = parseInteger(engineParameter("XGrid"));
236 Size dampingSteps = parseInteger(engineParameter("DampingSteps"));
237 bool monotoneVar = parseBool(engineParameter("EnforceMonotoneVariance", {}, false, "true"));
238 Size tGridMin = parseInteger(engineParameter("TimeGridMinimumSize", {}, false, "1"));
239 tGrid = std::max(tGridMin, tGrid);
240
241 boost::shared_ptr<QuantLib::GeneralizedBlackScholesProcess> gbsp;
242
243 if (monotoneVar) {
244 // Replicate the construction of time grid in FiniteDifferenceModel::rollbackImpl
245 // This time grid is required to build a BlackMonotoneVarVolTermStructure which
246 // ensures monotonic variance along the time grid
247 const Size totalSteps = tGrid + dampingSteps;
248 std::vector<Time> timePoints(totalSteps + 1);
249 Array timePointsArray(totalSteps, expiry, -expiry / totalSteps);
250 timePoints[0] = 0.0;
251 for (Size i = 0; i < totalSteps; i++)
252 timePoints[timePoints.size() - i - 1] = timePointsArray[i];
253 timePoints.insert(std::upper_bound(timePoints.begin(), timePoints.end(), 0.99 / 365), 0.99 / 365);
254 gbsp = getBlackScholesProcess(assetName, ccy, assetClass, timePoints);
255 } else {
256 gbsp = getBlackScholesProcess(assetName, ccy, assetClass);
257 }
258 return boost::make_shared<FdBlackScholesVanillaEngine>(gbsp, tGrid, xGrid, dampingSteps, scheme);
259 }
260};
