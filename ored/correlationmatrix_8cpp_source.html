58CorrelationFactor parseCorrelationFactor(const string& name) {
59
60 Size pos = name.find(':');
61 QL_REQUIRE(pos != string::npos, "Expected the factor to be of the form 'type:name'");
62
63 CrossAssetModel::AssetType factorType = parseCamAssetType(name.substr(0, pos));
64 string factorName = name.substr(pos + 1);
65
66 return { factorType, factorName, 0 };
67}

91void CorrelationMatrixBuilder::addCorrelation(const CorrelationFactor& f_1, const CorrelationFactor& f_2,
92 const Handle<Quote>& correlation) {
93
94 // Check the factors
95 checkFactor(f_1);
96 checkFactor(f_2);
97
98 // Store the correlation.
99 CorrelationKey ck = createKey(f_1, f_2);
100 QL_REQUIRE(corrs_.find(ck) == corrs_.end(), "Correlation for key [" <<
101 ck.first << "," << ck.second << "] already set");
102 QL_REQUIRE(correlation->value() >= -1.0 && correlation->value() <= 1.0, "Correlation value, " <<
103 correlation->value() << ", for key [" << ck.first << "," << ck.second << "] should be in [-1.0,1.0]");
104 corrs_[ck] = correlation;
105 DLOG("Added correlation: (" << f_1 << "," << f_2 << ") = " << correlation->value() << ".");
106}

131Matrix CorrelationMatrixBuilder::correlationMatrix(const ProcessInfo& processInfo) {
132
133 // Get the dimension of the matrix to create and create a list of factors.
134 Size dim = 0;
135 vector<CorrelationFactor> factors;
136 for (const auto& kv : processInfo) {
137 for (const auto& p : kv.second) {
138
139 // Don't allow multiple factors for FX for now. Need to check later the FX inversion in the lookup below
140 // if we want to extend the builder to multiple factors for each FX process.
141 if (kv.first == CrossAssetModel::AssetType::FX) {
142 QL_REQUIRE(p.second == 1, "CorrelationMatrixBuilder does not support multiple factors for FX. " <<
143 p.first << " is set up with " << p.second << " factors.");
144 }
145
146 dim += p.second;
147 for (Size i = 0; i < p.second; ++i) {
148 factors.push_back({ kv.first, p.first, i });
149 }
150 }
151 }
152
153 // Start with the identity matrix
154 Matrix corr(dim, dim, 0.0);
155 for (Size i = 0; i < dim; i++)
156 corr[i][i] = 1.0;
157
158 // Populate all of the off-diagonal elements
159 for (Size i = 0; i < dim; ++i) {
160 for (Size j = 0; j < i; ++j) {
161 corr[i][j] = corr[j][i] = getCorrelation(factors[i], factors[j])->value();
162 }
163 }
164
165 return corr;
166}

168CorrelationMatrixBuilder::ProcessInfo CorrelationMatrixBuilder::createProcessInfo(
169 const vector<string>& ccys,
170 const std::vector<std::string>& inflationIndices,
171 const std::vector<std::string>& creditNames,
172 const std::vector<std::string>& equityNames) {
173
174 // Check the currencies.
175 QL_REQUIRE(!ccys.empty(), "At least one currency required to build correlation matrix");
176 for (const string& ccy : ccys) {
177 QL_REQUIRE(ccy.size() == 3, "Invalid currency code " << ccy);
178 }
179
180 // Hold the resulting process information.
181 // Supporting a legacy method, assumed that there is 1 factor per process.
182 ProcessInfo result;
183
184 // Add process information for each currency.
185 for (const string& ccy : ccys) {
186 result[CrossAssetModel::AssetType::IR].emplace_back(ccy, 1);
187 }
188
189 // Add process information for each FX pair.
190 for (Size i = 1; i < ccys.size(); ++i) {
191 string ccyPair = ccys[i] + ccys[0];
192 result[CrossAssetModel::AssetType::FX].emplace_back(ccyPair, 1);
193 }
194
195 // Add process information for inflation indices.
196 for (const string& inflationIndex : inflationIndices) {
197 result[CrossAssetModel::AssetType::INF].emplace_back(inflationIndex, 1);
198 }
199
200 // Add process information for credit names.
201 for (const string& creditName : creditNames) {
202 result[CrossAssetModel::AssetType::CR].emplace_back(creditName, 1);
203 }
204
205 // Add process information for equity names.
206 for (const string& equityName : equityNames) {
207 result[CrossAssetModel::AssetType::EQ].emplace_back(equityName, 1);
208 }
209
210 return result;
211}

213void CorrelationMatrixBuilder::checkFactor(const CorrelationFactor& f) const {
214 switch (f.type) {
215 case CrossAssetModel::AssetType::IR:
216 QL_REQUIRE(f.name.size() == 3, "Expected IR factor name to be 3 character currency code but got: " << f.name);
217 break;
218 case CrossAssetModel::AssetType::FX:
219 QL_REQUIRE(f.name.size() == 6, "Expected FX factor name to be 6 character currency pair but got: " << f.name);
220 break;
221 case CrossAssetModel::AssetType::INF:
222 case CrossAssetModel::AssetType::CR:
223 case CrossAssetModel::AssetType::EQ:
224 case CrossAssetModel::AssetType::COM:
225 case CrossAssetModel::AssetType::CrState:
226 QL_REQUIRE(!f.name.empty(), "Expected non-empty factor name for factor type " << f.type);
227 break;
228 default:
229 QL_FAIL("Did not recognise factor type " << static_cast<int>(f.type) << ".");
230 }
231}

233CorrelationKey CorrelationMatrixBuilder::createKey(const CorrelationFactor& f_1,
234 const CorrelationFactor& f_2) const {
235
236 QL_REQUIRE(f_1 != f_2, "Correlation factors must be unique: " << f_1 << ".");
237
238 if (f_1 < f_2)
239 return make_pair(f_1, f_2);
240 else
241 return make_pair(f_2, f_1);
242}

250//! Get the correlation between the factor \p f_1 and \p f_2.
251Handle<Quote> CorrelationMatrixBuilder::getCorrelation(const CorrelationFactor& f_1,
252 const CorrelationFactor& f_2) const {
253
254 // Create key with the correct order for lookup
255 CorrelationKey ck = createKey(f_1, f_2);
256
257 // If we have the correlation via direct lookup, return it.
258 if (corrs_.find(ck) != corrs_.end())
259 return corrs_.at(ck);
260
261 // If one or both of the factors are FX, we may still be able to generate a correlation by using the inverse of
262 // a provided FX quote. Note that we have FX restricted to 1 factor in this class above.
263
264 typedef DerivedQuote<negate<Real>> InvQuote;
265
266 // If factor 1 is FX
267 if (f_1.type == CrossAssetModel::AssetType::FX) {
268 CorrelationFactor if_1{ CrossAssetModel::AssetType::FX, invertFx(f_1.name), f_1.index };
269 ck = createKey(if_1, f_2);
270 auto it = corrs_.find(ck);
271 if (it != corrs_.end())
272 return Handle<Quote>(boost::make_shared<InvQuote>(it->second, negate<Real>()));
273 }
274
275 // If factor 2 is FX
276 if (f_2.type == CrossAssetModel::AssetType::FX) {
277 CorrelationFactor if_2{ CrossAssetModel::AssetType::FX, invertFx(f_2.name), f_2.index };
278 ck = createKey(f_1, if_2);
279 auto it = corrs_.find(ck);
280 if (it != corrs_.end())
281 return Handle<Quote>(boost::make_shared<InvQuote>(it->second, negate<Real>()));
282 }
283
284 // If factor 1 and factor 2 are both FX
285 if (f_1.type == CrossAssetModel::AssetType::FX && f_2.type == CrossAssetModel::AssetType::FX) {
286 CorrelationFactor if_1{ CrossAssetModel::AssetType::FX, invertFx(f_1.name), f_1.index };
287 CorrelationFactor if_2{ CrossAssetModel::AssetType::FX, invertFx(f_2.name), f_2.index };
288 ck = createKey(if_1, if_2);
289 auto it = corrs_.find(ck);
290 if (it != corrs_.end())
291 return it->second;
292 }
293
294 // If we still haven't found anything, return a correlation of 0.
295 return Handle<Quote>(boost::make_shared<SimpleQuote>(0.0));
296}
