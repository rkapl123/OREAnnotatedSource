34void FxSwap::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
35 Currency nearBoughtCcy = data::parseCurrency(nearBoughtCurrency_);
36 Currency nearSoldCcy = data::parseCurrency(nearSoldCurrency_);
37 Date nearDate = data::parseDate(nearDate_);
38 Date farDate = data::parseDate(farDate_);
39
40 try {
41 DLOG("FxSwap::build() called for trade " << id());
42 //boost::shared_ptr<QuantLib::Instrument> instNear;
43 // builds two fxforwards and sums the npvs
44 // so that both npvs are in the same currency, the value of the first forward is taken to be the negative of the
45 // counterparty's npv
46 // npv_total= -npv1+npv2
47 instNear_.reset(
48 new QuantExt::FxForward(nearSoldAmount_, nearSoldCcy, nearBoughtAmount_, nearBoughtCcy, nearDate, false));
49 boost::shared_ptr<EngineBuilder> builder = engineFactory->builder("FxForward");
50 QL_REQUIRE(builder, "No builder found for " << tradeType_);
51 boost::shared_ptr<FxForwardEngineBuilderBase> fxBuilder =
52 boost::dynamic_pointer_cast<FxForwardEngineBuilderBase>(builder);
53 instNear_->setPricingEngine(fxBuilder->engine(nearSoldCcy, nearBoughtCcy));
54 //boost::shared_ptr<QuantLib::Instrument> instFar;
55 instFar_.reset(
56 new QuantExt::FxForward(farBoughtAmount_, nearSoldCcy, farSoldAmount_, nearBoughtCcy, farDate, false));
57 instFar_->setPricingEngine(fxBuilder->engine(nearSoldCcy, nearBoughtCcy));
58
59 DLOG("FxSwap::build(): Near NPV = " << instNear_->NPV());
60 DLOG("FxSwap::build(): Far NPV = " << instFar_->NPV());
61 // TODO: cannot use a CompositeInstrument
62 boost::shared_ptr<CompositeInstrument> composite(new CompositeInstrument());
63 composite->add(instNear_, -1.0);
64 composite->add(instFar_, 1.0);
65 instrument_.reset(new VanillaInstrument(composite));
66 npvCurrency_ = nearBoughtCurrency_;
67 notional_ = Null<Real>();//nearBoughtAmount_;
68 notionalCurrency_ = "";//nearBoughtCurrency_;
69 maturity_ = farDate;
70
71 } catch (std::exception&) {
72 instrument_.reset();
73 throw;
74 }
75 // Set up Legs
76 legs_.clear();
77 legs_.resize(4);
78 legCurrencies_.resize(4);
79 legPayers_.resize(4);
80 legs_[0].push_back(boost::shared_ptr<CashFlow>(new SimpleCashFlow(nearBoughtAmount_, nearDate)));
81 legs_[1].push_back(boost::shared_ptr<CashFlow>(new SimpleCashFlow(nearSoldAmount_, nearDate)));
82 legs_[2].push_back(boost::shared_ptr<CashFlow>(new SimpleCashFlow(farBoughtAmount_, farDate)));
83 legs_[3].push_back(boost::shared_ptr<CashFlow>(new SimpleCashFlow(farSoldAmount_, farDate)));
84
85 legCurrencies_[0] = nearBoughtCurrency_;
86 legCurrencies_[1] = nearSoldCurrency_;
87 legCurrencies_[2] = nearSoldCurrency_;
88 legCurrencies_[3] = nearBoughtCurrency_;
89
90 legPayers_[0] = false;
91 legPayers_[1] = true;
92 legPayers_[2] = false;
93 legPayers_[3] = true;
94
95 additionalData_["farSoldCurrency"] = nearBoughtCurrency_;
96 additionalData_["farBoughtCurrency"] = nearSoldCurrency_;
97 additionalData_["farSoldAmount"] = farSoldAmount_;
98 additionalData_["farBoughtAmount"] = farBoughtAmount_;
99 additionalData_["nearSoldCurrency"] = nearSoldCurrency_;
100 additionalData_["nearBoughtCurrency"] = nearBoughtCurrency_;
101 additionalData_["nearSoldAmount"] = nearSoldAmount_;
102 additionalData_["nearBoughtAmount"] = nearBoughtAmount_;
103
104 // ISDA taxonomy
105 additionalData_["isdaAssetClass"] = string("Foreign Exchange");
106 additionalData_["isdaBaseProduct"] = string(settlement_ == "Cash" ? "NDF" : "Forward");
107 additionalData_["isdaSubProduct"] = string("");
108 additionalData_["isdaTransaction"] = string("");
109
110 DLOG("FxSwap leg 0: " << nearDate_ << " " << legs_[0][0]->amount());
111 DLOG("FxSwap leg 1: " << nearDate_ << " " << legs_[1][0]->amount());
112 DLOG("FxSwap leg 2: " << farDate_ << " " << legs_[2][0]->amount());
113 DLOG("FxSwap leg 3: " << farDate_ << " " << legs_[3][0]->amount());
114}

117QuantLib::Real FxSwap::notional() const {
118 // try to get the notional from the additional results of the instrument
119 try {
120 return instFar_->result<Real>("currentNotional");
121 } catch (const std::exception& e) {
122 if (strcmp(e.what(), "currentNotional not provided"))
123 ALOG("error when retrieving notional: " << e.what());
124 }
125 // if not provided, return null
126 return Null<Real>();
127}

141void FxSwap::fromXML(XMLNode* node) {
142 Trade::fromXML(node);
143 XMLNode* fxNode = XMLUtils::getChildNode(node, "FxSwapData");
144 nearDate_ = XMLUtils::getChildValue(fxNode, "NearDate", true);
145 farDate_ = XMLUtils::getChildValue(fxNode, "FarDate", true);
146 nearBoughtCurrency_ = XMLUtils::getChildValue(fxNode, "NearBoughtCurrency", true);
147 nearSoldCurrency_ = XMLUtils::getChildValue(fxNode, "NearSoldCurrency", true);
148 nearBoughtAmount_ = XMLUtils::getChildValueAsDouble(fxNode, "NearBoughtAmount", true);
149 nearSoldAmount_ = XMLUtils::getChildValueAsDouble(fxNode, "NearSoldAmount", true);
150 farBoughtAmount_ = XMLUtils::getChildValueAsDouble(fxNode, "FarBoughtAmount", true);
151 farSoldAmount_ = XMLUtils::getChildValueAsDouble(fxNode, "FarSoldAmount", true);
152 settlement_ = XMLUtils::getChildValue(fxNode, "Settlement", false);
153 if (settlement_ == "")
154 settlement_ = "Physical";
155}

157XMLNode* FxSwap::toXML(XMLDocument& doc) {
158 XMLNode* node = Trade::toXML(doc);
159 XMLNode* fxNode = doc.allocNode("FxSwapData");
160 XMLUtils::appendNode(node, fxNode);
161 XMLUtils::addChild(doc, fxNode, "NearDate", nearDate_);
162 XMLUtils::addChild(doc, fxNode, "FarDate", farDate_);
163 XMLUtils::addChild(doc, fxNode, "NearBoughtCurrency", nearBoughtCurrency_);
164 XMLUtils::addChild(doc, fxNode, "NearBoughtAmount", nearBoughtAmount_);
165 XMLUtils::addChild(doc, fxNode, "NearSoldCurrency", nearSoldCurrency_);
166 XMLUtils::addChild(doc, fxNode, "NearSoldAmount", nearSoldAmount_);
167 XMLUtils::addChild(doc, fxNode, "FarBoughtAmount", farBoughtAmount_);
168 XMLUtils::addChild(doc, fxNode, "FarSoldAmount", farSoldAmount_);
169 XMLUtils::addChild(doc, fxNode, "Settlement", settlement_);
170
171 return node;
172}
