34void FxDigitalOption::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
35
36 // Only European Vanilla supported for now
37 QL_REQUIRE(option_.style() == "European", "Option Style unknown: " << option_.style());
38 QL_REQUIRE(option_.exerciseDates().size() == 1, "Invalid number of excercise dates");
39 QL_REQUIRE(option_.payoffAtExpiry() == true, "PayoffAtExpiry must be True for FxDigitalOption");
40 QL_REQUIRE(tradeActions().empty(), "TradeActions not supported for FxDigitalOption");
41 QL_REQUIRE(strike_ > 0.0 && strike_ != Null<Real>(), "Invalid strike " << strike_);
42
43 Currency domCcy = parseCurrency(domesticCurrency_);
44 Currency forCcy = parseCurrency(foreignCurrency_);
45
46 // Payoff Type
47 Option::Type type = parseOptionType(option_.callPut());
48
49 // Handle PayoffCurrency, we might have to flip the trade here
50 Real strike = strike_;
51 bool flipResults = false;
52 if (payoffCurrency_ == "") {
53 DLOG("PayoffCurrency defaulting to " << domesticCurrency_ << " for FxDigitalOption " << id());
54 } else if (payoffCurrency_ == foreignCurrency_) {
55 // Invert the trade, switch dom and for and flip Put/Call
56 strike = 1.0 / strike;
57 std::swap(domCcy, forCcy);
58 type = type == Option::Call ? Option::Put : Option::Call;
59 flipResults = true;
60 } else if (payoffCurrency_ != domesticCurrency_) {
61 QL_FAIL("Invalid Payoff currency (" << payoffCurrency_ << ") for FxDigitalOption " << forCcy << domCcy);
62 }
63 DLOG("Setting up FxDigitalOption with strike " << strike << " foreign " << forCcy << " domestic " << domCcy);
64
65 // Set up the CashOrNothing
66 boost::shared_ptr<StrikedTypePayoff> payoff(new CashOrNothingPayoff(type, strike, payoffAmount_));
67
68 // Exercise
69 Date expiryDate = parseDate(option_.exerciseDates().front());
70 boost::shared_ptr<Exercise> exercise = boost::make_shared<EuropeanExercise>(expiryDate);
71
72 // QL does not have an FXDigitalOption, so we add a vanilla one here and wrap
73 // it in a composite.
74 boost::shared_ptr<Instrument> vanilla = boost::make_shared<VanillaOption>(payoff, exercise);
75
76 // set pricing engines
77 boost::shared_ptr<EngineBuilder> builder = engineFactory->builder(tradeType_);
78 QL_REQUIRE(builder, "No builder found for " << tradeType_);
79 boost::shared_ptr<FxDigitalOptionEngineBuilder> fxOptBuilder =
80 boost::dynamic_pointer_cast<FxDigitalOptionEngineBuilder>(builder);
81 vanilla->setPricingEngine(fxOptBuilder->engine(forCcy, domCcy, flipResults));
82
83 Position::Type positionType = parsePositionType(option_.longShort());
84 Real bsInd = (positionType == QuantLib::Position::Long ? 1.0 : -1.0);
85 Real mult = bsInd;
86
87 std::vector<boost::shared_ptr<Instrument>> additionalInstruments;
88 std::vector<Real> additionalMultipliers;
89
90 Date lastPremiumDate =
91 addPremiums(additionalInstruments, additionalMultipliers, mult, option_.premiumData(), -bsInd, domCcy,
92 engineFactory, fxOptBuilder->configuration(MarketContext::pricing));
93
94 instrument_ = boost::shared_ptr<InstrumentWrapper>(
95 new VanillaInstrument(vanilla, mult, additionalInstruments, additionalMultipliers));
96
97 npvCurrency_ = domCcy.code(); // don't use domesticCurrency_ as it might be flipped
98 notional_ = payoffAmount_;
99 notionalCurrency_ = payoffCurrency_ != "" ? payoffCurrency_ : domesticCurrency_; // see logic above
100 maturity_ = std::max(lastPremiumDate, expiryDate);
101
102 additionalData_["payoffAmount"] = payoffAmount_;
103 additionalData_["payoffCurrency"] = payoffCurrency_;
104
105 // ISDA taxonomy
106 additionalData_["isdaAssetClass"] = string("Foreign Exchange");
107 additionalData_["isdaBaseProduct"] = string("Simple Exotic");
108 additionalData_["isdaSubProduct"] = string("Digital");
109 additionalData_["isdaTransaction"] = string("");
110}

112void FxDigitalOption::fromXML(XMLNode* node) {
113 Trade::fromXML(node);
114 XMLNode* fxNode = XMLUtils::getChildNode(node, "FxDigitalOptionData");
115 QL_REQUIRE(fxNode, "No FxDigitalOptionData Node");
116 option_.fromXML(XMLUtils::getChildNode(fxNode, "OptionData"));
117 strike_ = XMLUtils::getChildValueAsDouble(fxNode, "Strike", true);
118 payoffCurrency_ = XMLUtils::getChildValue(fxNode, "PayoffCurrency", false);
119 payoffAmount_ = XMLUtils::getChildValueAsDouble(fxNode, "PayoffAmount", true);
120 foreignCurrency_ = XMLUtils::getChildValue(fxNode, "ForeignCurrency", true);
121 domesticCurrency_ = XMLUtils::getChildValue(fxNode, "DomesticCurrency", true);
122}

124XMLNode* FxDigitalOption::toXML(XMLDocument& doc) {
125 XMLNode* node = Trade::toXML(doc);
126 XMLNode* fxNode = doc.allocNode("FxDigitalOptionData");
127 XMLUtils::appendNode(node, fxNode);
128
129 XMLUtils::appendNode(fxNode, option_.toXML(doc));
130 XMLUtils::addChild(doc, fxNode, "Strike", strike_);
131 XMLUtils::addChild(doc, fxNode, "PayoffCurrency", payoffCurrency_);
132 XMLUtils::addChild(doc, fxNode, "PayoffAmount", payoffAmount_);
133 XMLUtils::addChild(doc, fxNode, "ForeignCurrency", foreignCurrency_);
134 XMLUtils::addChild(doc, fxNode, "DomesticCurrency", domesticCurrency_);
135
136 return node;
137}
