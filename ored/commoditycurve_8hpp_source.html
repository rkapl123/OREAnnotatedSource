140template <template <class> class CurveType, typename... Args> void CommodityCurve::populateCurve(Args... args) {
141
142 if (interpolationMethod_ == "Linear") {
143 commodityPriceCurve_ = boost::make_shared<CurveType<QuantLib::Linear>>(args...);
144 } else if (interpolationMethod_ == "LogLinear") {
145 commodityPriceCurve_ = boost::make_shared<CurveType<QuantLib::LogLinear>>(args...);
146 } else if (interpolationMethod_ == "Cubic") {
147 commodityPriceCurve_ = boost::make_shared<CurveType<QuantLib::Cubic>>(args...);
148 } else if (interpolationMethod_ == "Hermite") {
149 commodityPriceCurve_ = boost::make_shared<CurveType<QuantLib::Cubic>>(
150 args..., QuantLib::Cubic(QuantLib::CubicInterpolation::Parabolic));
151 } else if (interpolationMethod_ == "LinearFlat") {
152 commodityPriceCurve_ = boost::make_shared<CurveType<QuantExt::LinearFlat>>(args...);
153 } else if (interpolationMethod_ == "LogLinearFlat") {
154 commodityPriceCurve_ = boost::make_shared<CurveType<QuantExt::LogLinearFlat>>(args...);
155 } else if (interpolationMethod_ == "CubicFlat") {
156 commodityPriceCurve_ = boost::make_shared<CurveType<QuantExt::CubicFlat>>(args...);
157 } else if (interpolationMethod_ == "HermiteFlat") {
158 commodityPriceCurve_ = boost::make_shared<CurveType<QuantExt::HermiteFlat>>(args...);
159 } else if (interpolationMethod_ == "BackwardFlat") {
160 commodityPriceCurve_ = boost::make_shared<CurveType<QuantLib::BackwardFlat>>(args...);
161 } else {
162 QL_FAIL("The interpolation method, " << interpolationMethod_ << ", is not supported.");
163 }
164}
