136 template <template <class> class CurveType, typename... Args> void CommodityCurve::populateCurve(Args... args) {
137 
138  if (interpolationMethod_ == "Linear") {
139  commodityPriceCurve_ = boost::make_shared<CurveType<QuantLib::Linear>>(args...);
140  } else if (interpolationMethod_ == "LogLinear") {
141  commodityPriceCurve_ = boost::make_shared<CurveType<QuantLib::LogLinear>>(args...);
142  } else if (interpolationMethod_ == "Cubic") {
143  commodityPriceCurve_ = boost::make_shared<CurveType<QuantLib::Cubic>>(args...);
144  } else if (interpolationMethod_ == "Hermite") {
145  commodityPriceCurve_ = boost::make_shared<CurveType<QuantLib::Cubic>>(
146  args..., QuantLib::Cubic(QuantLib::CubicInterpolation::Parabolic));
147  } else if (interpolationMethod_ == "LinearFlat") {
148  commodityPriceCurve_ = boost::make_shared<CurveType<QuantExt::LinearFlat>>(args...);
149  } else if (interpolationMethod_ == "LogLinearFlat") {
150  commodityPriceCurve_ = boost::make_shared<CurveType<QuantExt::LogLinearFlat>>(args...);
151  } else if (interpolationMethod_ == "CubicFlat") {
152  commodityPriceCurve_ = boost::make_shared<CurveType<QuantExt::CubicFlat>>(args...);
153  } else if (interpolationMethod_ == "HermiteFlat") {
154  commodityPriceCurve_ = boost::make_shared<CurveType<QuantExt::HermiteFlat>>(args...);
155  } else if (interpolationMethod_ == "BackwardFlat") {
156  commodityPriceCurve_ = boost::make_shared<CurveType<QuantLib::BackwardFlat>>(args...);
157  } else {
158  QL_FAIL("The interpolation method, " << interpolationMethod_ << ", is not supported.");
159  }
160 }
