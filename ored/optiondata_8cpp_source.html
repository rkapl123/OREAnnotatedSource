32void OptionData::fromXML(XMLNode* node) {
33 XMLUtils::checkNode(node, "OptionData");
34 longShort_ = XMLUtils::getChildValue(node, "LongShort", true);
35 callPut_ = XMLUtils::getChildValue(node, "OptionType", false);
36 payoffType_ = XMLUtils::getChildValue(node, "PayoffType", false);
37 payoffType2_ = XMLUtils::getChildValue(node, "PayoffType2", false);
38 style_ = XMLUtils::getChildValue(node, "Style", false);
39 noticePeriod_ = XMLUtils::getChildValue(node, "NoticePeriod", false);
40 noticeCalendar_ = XMLUtils::getChildValue(node, "NoticeCalendar", false);
41 noticeConvention_ = XMLUtils::getChildValue(node, "NoticeConvention", false);
42 settlement_ = XMLUtils::getChildValue(node, "Settlement", false);
43 settlementMethod_ = XMLUtils::getChildValue(node, "SettlementMethod", false);
44 payoffAtExpiry_ = XMLUtils::getChildValueAsBool(node, "PayOffAtExpiry", false);
45 premiumData_.fromXML(node);
46 exerciseFeeTypes_.clear();
47 exerciseFeeDates_.clear();
48 vector<std::reference_wrapper<vector<string>>> attrs;
49 attrs.push_back(exerciseFeeTypes_);
50 attrs.push_back(exerciseFeeDates_);
51 exerciseFees_ = XMLUtils::getChildrenValuesWithAttributes<Real>(node, "ExerciseFees", "ExerciseFee",
52 {"type", "startDate"}, attrs, &parseReal);
53 exerciseFeeSettlementPeriod_ = XMLUtils::getChildValue(node, "ExerciseFeeSettlementPeriod", false);
54 exerciseFeeSettlementCalendar_ = XMLUtils::getChildValue(node, "ExerciseFeeSettlementCalendar", false);
55 exerciseFeeSettlementConvention_ = XMLUtils::getChildValue(node, "ExerciseFeeSettlementConvention", false);
56 exercisePrices_ = XMLUtils::getChildrenValuesAsDoubles(node, "ExercisePrices", "ExercisePrice", false);
57 exerciseDates_ = XMLUtils::getChildrenValues(node, "ExerciseDates", "ExerciseDate", false);
58
59 automaticExercise_ = boost::none;
60 if (XMLNode* n = XMLUtils::getChildNode(node, "AutomaticExercise"))
61 automaticExercise_ = parseBool(XMLUtils::getNodeValue(n));
62
63 exerciseData_ = boost::none;
64 if (XMLNode* n = XMLUtils::getChildNode(node, "ExerciseData")) {
65 exerciseData_ = OptionExerciseData();
66 exerciseData_->fromXML(n);
67 }
68
69 paymentData_ = boost::none;
70 if (XMLNode* n = XMLUtils::getChildNode(node, "PaymentData")) {
71 paymentData_ = OptionPaymentData();
72 paymentData_->fromXML(n);
73 }
74}

79 if (callPut_ != "")

81 if (payoffType_ != "")
82 XMLUtils::addChild(doc, node, "PayoffType", payoffType_);

86 XMLUtils::addChild(doc, node, "Style", style_);

88 if (noticeCalendar_ != "")
89 XMLUtils::addChild(doc, node, "NoticeCalendar", noticeCalendar_);
90 if (noticeConvention_ != "")
91 XMLUtils::addChild(doc, node, "NoticeConvention", noticeConvention_);

93 XMLUtils::addChild(doc, node, "Settlement", settlement_);
94 if (settlementMethod_ != "")
95 XMLUtils::addChild(doc, node, "SettlementMethod", settlementMethod_);
96 XMLUtils::addChild(doc, node, "PayOffAtExpiry", payoffAtExpiry_);
97 XMLUtils::appendNode(node, premiumData_.toXML(doc));
98 XMLUtils::addChildrenWithOptionalAttributes(doc, node, "ExerciseFees", "ExerciseFee", exerciseFees_,
99 {"type", "startDate"}, {exerciseFeeTypes_, exerciseFeeDates_});
100 if (exerciseFeeSettlementPeriod_ != "")
101 XMLUtils::addChild(doc, node, "ExerciseFeeSettlementPeriod", exerciseFeeSettlementPeriod_);
102 if (exerciseFeeSettlementCalendar_ != "")
103 XMLUtils::addChild(doc, node, "ExerciseFeeSettlementCalendar", exerciseFeeSettlementCalendar_);
104 if (exerciseFeeSettlementConvention_ != "")
105 XMLUtils::addChild(doc, node, "ExerciseFeeSettlementConvention", exerciseFeeSettlementConvention_);
106 XMLUtils::addChildren(doc, node, "ExercisePrices", "ExercisePrice", exercisePrices_);
107 XMLUtils::addChildren(doc, node, "ExerciseDates", "ExerciseDate", exerciseDates_);
108
109 if (automaticExercise_)
110 XMLUtils::addChild(doc, node, "AutomaticExercise", *automaticExercise_);
111
112 if (exerciseData_) {
113 XMLUtils::appendNode(node, exerciseData_->toXML(doc));
114 }
115
116 if (paymentData_) {
117 XMLUtils::appendNode(node, paymentData_->toXML(doc));
118 }
119
120 return node;
121}

123ExerciseBuilder::ExerciseBuilder(const OptionData& optionData, const std::vector<Leg> legs) {
124
125 // only keep a) future exercise dates and b) exercise dates that exercise into a whole
126 // accrual period of the underlying; TODO handle exercises into broken periods?
127
128 // determine last accrual start date present in the underlying legs
129
130 Date lastAccrualStartDate = Date::minDate();
131 for (auto const& l : legs) {
132 for (auto const& c : l) {
133 if (auto cpn = boost::dynamic_pointer_cast<Coupon>(c))
134 lastAccrualStartDate = std::max(lastAccrualStartDate, cpn->accrualStartDate());
135 }
136 }
137
138 // get notice period, calendar, bdc
139
140 Period noticePeriod = optionData.noticePeriod().empty() ? 0 * Days : parsePeriod(optionData.noticePeriod());
141 Calendar noticeCal =
142 optionData.noticeCalendar().empty() ? NullCalendar() : parseCalendar(optionData.noticeCalendar());
143 BusinessDayConvention noticeBdc =
144 optionData.noticeConvention().empty() ? Unadjusted : parseBusinessDayConvention(optionData.noticeConvention());
145
146 // build vector of sorted exercise dates
147
148 std::vector<QuantLib::Date> sortedExerciseDates;
149 for (auto const& d : optionData.exerciseDates())
150 sortedExerciseDates.push_back(parseDate(d));
151 std::sort(sortedExerciseDates.begin(), sortedExerciseDates.end());
152
153 // build vector of alive exercise dates and corresponding native dates
154
155 std::vector<bool> isExerciseDateAlive(sortedExerciseDates.size(), false);
156
157 for (Size i = 0; i < sortedExerciseDates.size(); i++) {
158 Date noticeDate = noticeCal.advance(sortedExerciseDates[i], -noticePeriod, noticeBdc);
159 if (noticeDate > Settings::instance().evaluationDate() && noticeDate <= lastAccrualStartDate) {
160 isExerciseDateAlive[i] = true;
161 noticeDates_.push_back(noticeDate);
162 exerciseDates_.push_back(sortedExerciseDates[i]);
163 DLOG("Got notice date " << QuantLib::io::iso_date(noticeDate) << " using notice period " << noticePeriod
164 << ", convention " << noticeBdc << ", calendar " << noticeCal.name()
165 << " from exercise date " << exerciseDates_.back());
166 }
167 if (noticeDate > lastAccrualStartDate)
168 WLOG("Remove notice date " << ore::data::to_string(noticeDate) << " (exercise date "
169 << sortedExerciseDates[i] << ") after last accrual start date "
170 << ore ::data::to_string(lastAccrualStartDate));
171 }
172
173 // if we do not have any notice dates left, we are done (a nullptr will be returned by exercise())
174
175 if (noticeDates_.empty())
176 return;
177
178 // build exercise instance
179
180 if (optionData.style() == "European")
181 exercise_ = boost::make_shared<EuropeanExercise>(sortedExerciseDates.back());
182 else
183 exercise_ = boost::make_shared<BermudanExercise>(noticeDates_);
184
185
186 // build fee and rebated exercise instance, if any fees are present
187
188 if (!optionData.exerciseFees().empty()) {
189
190 // build an exercise date "schedule" by adding the maximum possible date at the end
191
192 std::vector<Date> exDatesPlusInf(sortedExerciseDates);
193 exDatesPlusInf.push_back(Date::maxDate());
194 vector<double> allRebates = buildScheduledVectorNormalised(optionData.exerciseFees(),
195 optionData.exerciseFeeDates(), exDatesPlusInf, 0.0);
196
197 // filter on alive rebates, so that we can a vector of rebates corresponding to the exerciseDates vector
198
199 vector<double> rebates;
200 for (Size i = 0; i < sortedExerciseDates.size(); ++i) {
201 if (isExerciseDateAlive[i])
202 rebates.push_back(allRebates[i]);
203 }
204
205 // flip the sign of the fee to get a rebate
206
207 for (auto& r : rebates)
208 r = -r;
209 vector<string> feeType = buildScheduledVectorNormalised<string>(
210 optionData.exerciseFeeTypes(), optionData.exerciseFeeDates(), exDatesPlusInf, "");
211
212 // convert relative to absolute fees if required
213
214 for (Size i = 0; i < rebates.size(); ++i) {
215
216 // default to Absolute
217
218 if (feeType[i].empty())
219 feeType[i] = "Absolute";
220
221 if (feeType[i] == "Percentage") {
222
223 // get next coupon after exercise to determine relevant notional
224
225 std::set<std::pair<Date, Real>> notionals;
226 for (auto const& l : legs) {
227 for (auto const& c : l) {
228 if (auto cpn = boost::dynamic_pointer_cast<Coupon>(c)) {
229 if (cpn->accrualStartDate() >= exerciseDates_[i])
230 notionals.insert(std::make_pair(cpn->accrualStartDate(), cpn->nominal()));
231 }
232 }
233 }
234
235 if (notionals.empty())
236 rebates[i] = 0.0; // notional is zero
237 else {
238 Real feeNotional = notionals.begin()->second;
239 DLOG("Convert percentage rebate "
240 << rebates[i] << " to absolute rebate " << rebates[i] * feeNotional << " using nominal "
241 << feeNotional << " for exercise date " << QuantLib::io::iso_date(exerciseDates_[i]));
242 rebates[i] *= feeNotional; // multiply percentage fee by relevant notional
243 }
244
245 } else {
246 QL_REQUIRE(feeType[i] == "Absolute", "fee type must be Absolute or Relative");
247 }
248 }
249
250 Period feeSettlPeriod = optionData.exerciseFeeSettlementPeriod().empty()
251 ? 0 * Days
252 : parsePeriod(optionData.exerciseFeeSettlementPeriod());
253
254 Calendar feeSettlCal = optionData.exerciseFeeSettlementCalendar().empty()
255 ? NullCalendar()
256 : parseCalendar(optionData.exerciseFeeSettlementCalendar());
257
258 BusinessDayConvention feeSettlBdc =
259 optionData.exerciseFeeSettlementConvention().empty()
260 ? Unadjusted
261 : parseBusinessDayConvention(optionData.exerciseFeeSettlementConvention());
262
263 exercise_ = boost::make_shared<QuantExt::RebatedExercise>(*exercise_, exerciseDates_, rebates, feeSettlPeriod,
264 feeSettlCal, feeSettlBdc);
265
266 // log rebates
267
268 auto dbgEx = boost::static_pointer_cast<QuantExt::RebatedExercise>(exercise_);
269
270 for (Size i = 0; i < exerciseDates_.size(); ++i) {
271 DLOG("Got rebate " << dbgEx->rebate(i) << " with payment date "
272 << QuantLib::io::iso_date(dbgEx->rebatePaymentDate(i)) << " (exercise date="
273 << QuantLib::io::iso_date(exerciseDates_[i]) << ") using rebate settl period "
274 << feeSettlPeriod << ", calendar " << feeSettlCal << ", convention " << feeSettlBdc);
275 }
276 }
277}
