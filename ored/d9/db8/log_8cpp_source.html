100 void Log::header(unsigned m, const char* filename, int lineNo) {
101  // 1. Reset stringstream
102  ls_.str(string());
103  ls_.clear();
104 
105  // Write the header to the stream
106  // TYPE [Time Stamp] (file:line)
107  switch (m) {
108  case ORE_ALERT:
109  ls_ << "ALERT ";
110  break;
111  case ORE_CRITICAL:
112  ls_ << "CRITICAL ";
113  break;
114  case ORE_ERROR:
115  ls_ << "ERROR ";
116  break;
117  case ORE_WARNING:
118  ls_ << "WARNING ";
119  break;
120  case ORE_NOTICE:
121  ls_ << "NOTICE ";
122  break;
123  case ORE_DEBUG:
124  ls_ << "DEBUG ";
125  break;
126  case ORE_DATA:
127  ls_ << "DATA ";
128  break;
129  case ORE_MEMORY:
130  ls_ << "MEMORY ";
131  break;
132  }
133 
134  // Timestamp
135  // Use boost::posix_time microsecond clock to get better precision (when available).
136  // format is "2014-Apr-04 11:10:16.179347"
137  ls_ << '[' << to_simple_string(microsec_clock::local_time()) << ']';
138 
139  // Filename & line no
140  // format is " (file:line)"
141  int lineNoLen = (int)log10((double)lineNo) + 1; // Length of the int as a string
142  int len = 2 + strlen(filename) + 1 + lineNoLen + 1; // " (" + file + ':' + line + ')'
143 
144  int maxLen = 30; // gives about 23 chars for the filename
145  if (len <= maxLen) {
146  ls_ << " (" << filename << ':' << lineNo << ')';
147  // pad out spaces
148  ls_ << string(maxLen - len, ' ');
149  } else {
150  // need to trim the filename to fit into maxLen chars
151  // need to remove (len - maxLen) chars + 3 for the "..."
152  ls_ << " (..." << string(filename).substr(3 + len - maxLen) << ':' << lineNo << ')';
153  }
154 
155  ls_ << " : ";
156 
157  // log pid if given
158  if (pid_ > 0)
159  ls_ << " [" << pid_ << "] ";
160 }
