101 Schedule makeSchedule(const ScheduleDates& data) {
102  QL_REQUIRE(data.dates().size() > 0, "Must provide at least 1 date for Schedule");
103  Calendar calendar = parseCalendar(data.calendar());
104  BusinessDayConvention convention = Unadjusted;
105  if (data.convention() != "")
106  convention = parseBusinessDayConvention(data.convention());
107  // Avoid compiler warning on gcc
108  // https://www.boost.org/doc/libs/1_74_0/libs/optional/doc/html/boost_optional/tutorial/
109  // gotchas/false_positive_with__wmaybe_uninitialized.html
110  auto tenor = boost::make_optional(false, Period());
111  if (data.tenor() != "")
112  tenor = parsePeriod(data.tenor());
113  bool endOfMonth = false;
114  if (data.endOfMonth() != "")
115  endOfMonth = parseBool(data.endOfMonth());
116 
117  // Ensure that Schedule ctor is passed a vector of unique ordered dates.
118  std::set<Date> uniqueDates;
119  for (const string& d : data.dates())
120  uniqueDates.insert(calendar.adjust(parseDate(d), convention));
121 
122  return Schedule(vector<Date>(uniqueDates.begin(), uniqueDates.end()), calendar, convention, boost::none, tenor,
123  boost::none, endOfMonth);
124 }

126 Schedule makeSchedule(const ScheduleRules& data, const Date& openEndDateReplacement) {
127  QL_REQUIRE(!data.endDate().empty() || openEndDateReplacement != Null<Date>(),
128  "makeSchedule(): Schedule does not have an end date, this is not supported in this context / for this "
129  "trade type. Please provide an end date.");
130  QL_REQUIRE(!data.endDate().empty() || data.lastDate().empty(),
131  "makeSchedule(): If no end date is given, a last date is not allowed either. Please remove the last "
132  "date from the schedule.");
133  Calendar calendar = parseCalendar(data.calendar());
134  Date startDate = parseDate(data.startDate());
135  Date endDate = data.endDate().empty() ? openEndDateReplacement : parseDate(data.endDate());
136  // Handle trivial case here
137  if (startDate == endDate)
138  return Schedule(vector<Date>(1, startDate), calendar);
139 
140  QL_REQUIRE(startDate < endDate, "StartDate " << startDate << " is ahead of EndDate " << endDate);
141 
142  Date firstDate = parseDate(data.firstDate());
143  Date lastDate = parseDate(data.lastDate());
144  Period tenor = parsePeriod(data.tenor());
145 
146  // defaults
147  BusinessDayConvention bdc = ModifiedFollowing;
148  BusinessDayConvention bdcEnd = ModifiedFollowing;
149  DateGeneration::Rule rule = DateGeneration::Forward;
150  bool endOfMonth = false;
151 
152  // now check the strings, if they are empty we take defaults
153  if (!data.convention().empty())
154  bdc = parseBusinessDayConvention(data.convention());
155  if (!data.termConvention().empty())
156  bdcEnd = parseBusinessDayConvention(data.termConvention());
157  else
158  bdcEnd = bdc; // except here
159  if (!data.rule().empty())
160  rule = parseDateGenerationRule(data.rule());
161  if (!data.endOfMonth().empty())
162  endOfMonth = parseBool(data.endOfMonth());
163 
164  if ((rule == DateGeneration::CDS || rule == DateGeneration::CDS2015) &&
165  (firstDate != Null<Date>() || lastDate != Null<Date>())) {
166  // Special handling of first date and last date in combination with CDS and CDS2015 rules:
167  // To be able to construct CDS schedules with front or back stub periods, we overwrite the
168  // first (last) date of the schedule built in QL with a given first (last) date
169  // The schedule builder in QL itself is not capable of doing this, it just throws an exception
170  // if a first (last) date is given in combination with a CDS / CDS2015 date generation rule.
171  std::vector<Date> dates = Schedule(startDate, endDate, tenor, calendar, bdc, bdcEnd, rule, endOfMonth).dates();
172  QL_REQUIRE(!dates.empty(),
173  "got empty CDS or CDS2015 schedule, startDate = " << startDate << ", endDate = " << endDate);
174  if (firstDate != Date())
175  dates.front() = firstDate;
176  if (lastDate != Date())
177  dates.back() = lastDate;
178  return Schedule(dates, calendar, bdc, bdcEnd, tenor, rule, endOfMonth);
179  } else {
180  return Schedule(startDate, endDate, tenor, calendar, bdc, bdcEnd, rule, endOfMonth, firstDate, lastDate);
181  }
182 }

202 Schedule makeSchedule(const ScheduleData& data, const Date& openEndDateReplacement) {
203  // only the last rule-based schedule is allowed to have an open end date, check this
204  for (Size i = 1; i < data.rules().size(); ++i) {
205  QL_REQUIRE(!data.rules()[i - 1].endDate().empty(),
206  "makeSchedule: only last schedule is allowed to have an open end date");
207  }
208  // build all the date and rule based sub-schedules we have
209  vector<Schedule> schedules;
210  for (auto& d : data.dates())
211  schedules.push_back(makeSchedule(d));
212  for (auto& r : data.rules())
213  schedules.push_back(makeSchedule(r, openEndDateReplacement));
214  QL_REQUIRE(!schedules.empty(), "No dates or rules to build Schedule from");
215  if (schedules.size() == 1)
216  // if we have just one, use that (most common case)
217  return schedules.front();
218  else {
219  // if we have multiple, combine them
220 
221  // 1) sort by start date
222  std::sort(schedules.begin(), schedules.end(),
223  [](const Schedule& lhs, const Schedule& rhs) -> bool { return lhs.startDate() < rhs.startDate(); });
224 
225  // 2) check if meta data is present, and if yes if it is consistent across schedules;
226  // the only exception is the term date convention, this is taken from the last schedule always
227  BusinessDayConvention convention = Null<BusinessDayConvention>(),
228  termConvention = Unadjusted; // initialization prevents gcc warning
229  Calendar calendar;
230  Period tenor;
231  DateGeneration::Rule rule = DateGeneration::Zero; // initialization prevents gcc warning
232  bool endOfMonth = false; // initialization prevents gcc warning
233  bool hasCalendar = false, hasConvention = false, hasTermConvention = false, hasTenor = false, hasRule = false,
234  hasEndOfMonth = false, hasConsistentCalendar = true, hasConsistentConvention = true,
235  hasConsistentTenor = true, hasConsistentRule = true, hasConsistentEndOfMonth = true;
236  for (auto& d : data.dates()) {
237  updateData<Calendar>(d.calendar(), calendar, hasCalendar, hasConsistentCalendar, parseCalendarTemp);
238  updateData<BusinessDayConvention>(d.convention(), convention, hasConvention, hasConsistentConvention,
239  parseBusinessDayConvention);
240  updateData<Period>(d.tenor(), tenor, hasTenor, hasConsistentTenor, parsePeriod);
241  }
242  for (auto& d : data.rules()) {
243  updateData<Calendar>(d.calendar(), calendar, hasCalendar, hasConsistentCalendar, parseCalendarTemp);
244  updateData<BusinessDayConvention>(d.convention(), convention, hasConvention, hasConsistentConvention,
245  parseBusinessDayConvention);
246  updateData<Period>(d.tenor(), tenor, hasTenor, hasConsistentTenor, parsePeriod);
247  updateData<bool>(d.endOfMonth(), endOfMonth, hasEndOfMonth, hasConsistentEndOfMonth, parseBool);
248  updateData<DateGeneration::Rule>(d.rule(), rule, hasRule, hasConsistentRule, parseDateGenerationRule);
249  if (d.termConvention() != "") {
250  hasTermConvention = true;
251  termConvention = parseBusinessDayConvention(d.termConvention());
252  }
253  }
254 
255  // 3) combine dates and fill isRegular flag
256  const Schedule& s0 = schedules.front();
257  vector<Date> dates = s0.dates();
258  std::vector<bool> isRegular(s0.dates().size() - 1, false);
259  if (s0.hasIsRegular())
260  isRegular = s0.isRegular();
261  // will be removed, if next schedule's front date is matching the last date of current schedule
262  isRegular.push_back(false);
263  for (Size i = 1; i < schedules.size(); ++i) {
264  const Schedule& s = schedules[i];
265  QL_REQUIRE(dates.back() <= s.dates().front(), "Dates mismatch");
266  // if the end points match up, skip one to avoid duplicates, otherwise take both
267  Size offset = dates.back() == s.dates().front() ? 1 : 0;
268  isRegular.erase(isRegular.end() - offset,
269  isRegular.end()); // correct for superfluous flags from previous schedule
270  // add isRegular information, if available, otherwise assume irregular periods
271  if (s.hasIsRegular()) {
272  isRegular.insert(isRegular.end(), s.isRegular().begin(), s.isRegular().end());
273  } else {
274  for (Size ii = 0; ii < s.dates().size() - 1; ++ii)
275  isRegular.push_back(false);
276  }
277  if (i < schedules.size() - 1) {
278  // will be removed if next schedule's front date is matching last date of current schedule
279  isRegular.push_back(false);
280  }
281  // add the dates
282  dates.insert(dates.end(), s.dates().begin() + offset, s.dates().end());
283  }
284 
285  // 4) Build schedule
286  return Schedule(dates, hasCalendar && hasConsistentCalendar ? calendar : NullCalendar(),
287  hasConvention && hasConsistentConvention ? convention : Unadjusted,
288  hasTermConvention ? boost::optional<BusinessDayConvention>(termConvention) : boost::none,
289  hasTenor && hasConsistentTenor ? boost::optional<Period>(tenor) : boost::none,
290  hasRule && hasConsistentRule ? boost::optional<DateGeneration::Rule>(rule) : boost::none,
291  hasEndOfMonth && hasConsistentEndOfMonth ? boost::optional<bool>(endOfMonth) : boost::none,
292  isRegular);
293  }
