48 void ForwardBond::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
49  DLOG("ForwardBond::build() called for trade " << id());
50 
51  const boost::shared_ptr<Market> market = engineFactory->market();
52 
53  boost::shared_ptr<EngineBuilder> builder_fwd = engineFactory->builder("ForwardBond");
54  boost::shared_ptr<EngineBuilder> builder_bd = engineFactory->builder("Bond");
55 
56  bondData_.populateFromBondReferenceData(engineFactory->referenceData());
57 
58  QL_REQUIRE(!bondData_.referenceCurveId().empty(), "reference curve id required");
59  QL_REQUIRE(!bondData_.settlementDays().empty(), "settlement days required");
60 
61  Date issueDate = parseDate(bondData_.issueDate());
62  Calendar calendar = parseCalendar(bondData_.calendar());
63  Natural settlementDays = boost::lexical_cast<Natural>(bondData_.settlementDays());
64 
65  Date fwdMaturityDate = parseDate(fwdMaturityDate_);
66  Date fwdSettlementDate = fwdSettlementDate_.empty() ? fwdMaturityDate : parseDate(fwdSettlementDate_);
67  bool isPhysicallySettled;
68  if (settlement_ == "Physical" || settlement_.empty())
69  isPhysicallySettled = true;
70  else if (settlement_ == "Cash")
71  isPhysicallySettled = false;
72  else {
73  QL_FAIL("ForwardBond: invalid settlement '" << settlement_ << "', expected Cash or Physical");
74  }
75  Real amount = amount_.empty() ? Null<Real>() : parseReal(amount_);
76  Real lockRate = lockRate_.empty() ? Null<Real>() : parseReal(lockRate_);
77  DayCounter lockRateDayCounter = lockRateDayCounter_.empty() ? Actual360() : parseDayCounter(lockRateDayCounter_);
78  bool settlementDirty = settlementDirty_.empty() ? true : parseBool(settlementDirty_);
79  Real compensationPayment = parseReal(compensationPayment_);
80  Date compensationPaymentDate = parseDate(compensationPaymentDate_);
81  bool longInForward = parseBool(longInForward_);
82 
83  QL_REQUIRE((amount == Null<Real>() && lockRate != Null<Real>()) ||
84  (amount != Null<Real>() && lockRate == Null<Real>()),
85  "ForwardBond: exactly one of Amount of LockRate must be given");
86  QL_REQUIRE(compensationPaymentDate <= fwdMaturityDate, "Premium cannot be paid after forward contract maturity");
87 
88  if (lockRate != Null<Real>())
89  isPhysicallySettled = false;
90 
91  QL_REQUIRE(!bondData_.coupons().empty(), "ForwardBond: No LegData given. If you want to represent a zero bond, "
92  "set it up as a coupon bond with zero fixed rate");
93 
94  bool firstLegIsPayer = bondData_.coupons()[0].isPayer();
95  QL_REQUIRE(firstLegIsPayer == false, "ForwardBond: The underlying bond must be entered with a receiver leg. Use "
96  "LongInBond to specify pay direction of forward payoff");
97  QL_REQUIRE(compensationPayment > 0.0 || close_enough(compensationPayment, 0.0),
98  "ForwardBond: Negative compensation payments ("
99  << compensationPayment
100  << ") are not allowed. Notice that we will ensure that a positive compensation amount will be paid "
101  "by the party being long in the forward contract.");
102 
103  boost::shared_ptr<Payoff> payoff;
104  if (amount != Null<Real>()) {
105  payoff = longInForward ? boost::make_shared<QuantExt::ForwardBondTypePayoff>(Position::Long, amount)
106  : boost::make_shared<QuantExt::ForwardBondTypePayoff>(Position::Short, amount);
107  }
108  compensationPayment = longInForward ? compensationPayment : -compensationPayment;
109 
110  std::vector<Leg> separateLegs;
111  for (Size i = 0; i < bondData_.coupons().size(); ++i) {
112  Leg leg;
113  auto configuration = builder_bd->configuration(MarketContext::pricing);
114  auto legBuilder = engineFactory->legBuilder(bondData_.coupons()[i].legType());
115  leg = legBuilder->buildLeg(bondData_.coupons()[i], engineFactory, requiredFixings_, configuration);
116  separateLegs.push_back(leg);
117  }
118  Leg leg = joinLegs(separateLegs);
119  auto bond = boost::make_shared<QuantLib::Bond>(settlementDays, calendar, issueDate, leg);
120 
121  npvCurrency_ = currency_ = bondData_.coupons().front().currency();
122  maturity_ = bond->cashflows().back()->date();
123  notional_ = currentNotional(bond->cashflows()) * bondData_.bondNotional();
124  notionalCurrency_ = currency_;
125 
126  // cashflows will be generated as additional results in the pricing engine
127  legs_ = {};
128  legCurrencies_ = {npvCurrency_};
129  legPayers_ = {firstLegIsPayer};
130  Currency currency = parseCurrency(currency_);
131 
132  // first ctor is for vanilla fwd bonds, second for tlocks with a lock rate specifying the payoff
133  boost::shared_ptr<QuantLib::Instrument> fwdBond =
134  payoff ? boost::make_shared<QuantExt::ForwardBond>(bond, payoff, fwdMaturityDate, fwdSettlementDate,
135  isPhysicallySettled, settlementDirty, compensationPayment,
136  compensationPaymentDate, bondData_.bondNotional())
137  : boost::make_shared<QuantExt::ForwardBond>(bond, lockRate, lockRateDayCounter, longInForward,
138  fwdMaturityDate, fwdSettlementDate, isPhysicallySettled,
139  settlementDirty, compensationPayment,
140  compensationPaymentDate, bondData_.bondNotional());
141 
142  boost::shared_ptr<fwdBondEngineBuilder> fwdBondBuilder =
143  boost::dynamic_pointer_cast<fwdBondEngineBuilder>(builder_fwd);
144  QL_REQUIRE(fwdBondBuilder, "ForwardBond::build(): could not cast builder: " << id());
145 
146  fwdBond->setPricingEngine(fwdBondBuilder->engine(id(), currency, bondData_.creditCurveId(), bondData_.securityId(),
147  bondData_.referenceCurveId(), bondData_.incomeCurveId()));
148  instrument_.reset(new VanillaInstrument(fwdBond, 1.0));
149 
150  additionalData_["currentNotional"] = currentNotional(bond->cashflows()) * bondData_.bondNotional();
151  additionalData_["originalNotional"] = originalNotional(bond->cashflows()) * bondData_.bondNotional();
152  additionalData_["currency"] = currency_;
153 }

155 void ForwardBond::fromXML(XMLNode* node) {
156  Trade::fromXML(node);
157  XMLNode* fwdBondNode = XMLUtils::getChildNode(node, "ForwardBondData");
158  QL_REQUIRE(fwdBondNode, "No ForwardBondData Node");
159  bondData_.fromXML(XMLUtils::getChildNode(fwdBondNode, "BondData"));
160 
161  XMLNode* fwdSettlementNode = XMLUtils::getChildNode(fwdBondNode, "SettlementData");
162  QL_REQUIRE(fwdSettlementNode, "No fwdSettlementNode Node");
163 
164  fwdMaturityDate_ = XMLUtils::getChildValue(fwdSettlementNode, "ForwardMaturityDate", true);
165  fwdSettlementDate_ = XMLUtils::getChildValue(fwdSettlementNode, "ForwardSettlementDate", false);
166  settlement_ = XMLUtils::getChildValue(fwdSettlementNode, "Settlement", false);
167  amount_ = XMLUtils::getChildValue(fwdSettlementNode, "Amount", false);
168  lockRate_ = XMLUtils::getChildValue(fwdSettlementNode, "LockRate", false);
169  lockRateDayCounter_ = XMLUtils::getChildValue(fwdSettlementNode, "LockRateDayCounter", false);
170  settlementDirty_ = XMLUtils::getChildValue(fwdSettlementNode, "SettlementDirty", false);
171 
172  XMLNode* fwdPremiumNode = XMLUtils::getChildNode(fwdBondNode, "PremiumData");
173  if (fwdPremiumNode) {
174  compensationPayment_ = XMLUtils::getChildValue(fwdPremiumNode, "Amount", true);
175  compensationPaymentDate_ = XMLUtils::getChildValue(fwdPremiumNode, "Date", true);
176  } else {
177  compensationPayment_ = "0.0";
178  compensationPaymentDate_ = fwdMaturityDate_;
179  }
180 
181  longInForward_ = XMLUtils::getChildValue(fwdBondNode, "LongInForward", true);
182 }

184 XMLNode* ForwardBond::toXML(XMLDocument& doc) {
185  XMLNode* node = Trade::toXML(doc);
186  XMLNode* fwdBondNode = doc.allocNode("ForwardBondData");
187  XMLUtils::appendNode(node, fwdBondNode);
188  XMLUtils::appendNode(fwdBondNode, bondData_.toXML(doc));
189 
190  XMLNode* fwdSettlementNode = doc.allocNode("SettlementData");
191  XMLUtils::appendNode(fwdBondNode, fwdSettlementNode);
192  XMLUtils::addChild(doc, fwdSettlementNode, "ForwardMaturityDate", fwdMaturityDate_);
193  if (!fwdSettlementDate_.empty())
194  XMLUtils::addChild(doc, fwdSettlementNode, "ForwardSettlementDate", fwdSettlementDate_);
195  if (!settlement_.empty())
196  XMLUtils::addChild(doc, fwdSettlementNode, "Settlement", settlement_);
197  if (!amount_.empty())
198  XMLUtils::addChild(doc, fwdSettlementNode, "Amount", amount_);
199  if (!lockRate_.empty())
200  XMLUtils::addChild(doc, fwdSettlementNode, "LockRate", lockRate_);
201  if (!lockRateDayCounter_.empty())
202  XMLUtils::addChild(doc, fwdSettlementNode, "LockRateDayCounter", lockRateDayCounter_);
203  if (!settlementDirty_.empty())
204  XMLUtils::addChild(doc, fwdSettlementNode, "SettlementDirty", settlementDirty_);
205 
206  XMLNode* fwdPremiumNode = doc.allocNode("PremiumData");
207  XMLUtils::appendNode(fwdBondNode, fwdPremiumNode);
208  XMLUtils::addChild(doc, fwdPremiumNode, "Amount", compensationPayment_);
209  XMLUtils::addChild(doc, fwdPremiumNode, "Date", compensationPaymentDate_);
210 
211  XMLUtils::addChild(doc, fwdBondNode, "LongInForward", longInForward_);
212 
213  return node;
214 }
