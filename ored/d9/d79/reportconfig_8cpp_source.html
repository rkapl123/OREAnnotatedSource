28 void ReportConfig::fromXML(XMLNode* node) {

30 
31  if (auto tmp = XMLUtils::getChildNode(node, "ReportOnDeltaGrid")) {
32  reportOnDeltaGrid_ = parseBool(XMLUtils::getNodeValue(tmp));

34  reportOnDeltaGrid_ = boost::none;

36 
37  if (auto tmp = XMLUtils::getChildNode(node, "ReportOnMoneynessGrid")) {
38  reportOnMoneynessGrid_ = parseBool(XMLUtils::getNodeValue(tmp));

42

44  deltas_ = parseListOfValues(XMLUtils::getNodeValue(tmp));

46  deltas_ = boost::none;

62 XMLNode* ReportConfig::toXML(XMLDocument& doc) {
63  XMLNode* node = doc.allocNode("Report");
64  if (reportOnDeltaGrid_)
65  XMLUtils::addChild(doc, node, "ReportOnDeltaGrid", *reportOnDeltaGrid_);
66  if (reportOnMoneynessGrid_)
67  XMLUtils::addChild(doc, node, "ReportOnMoneynessGrid", *reportOnMoneynessGrid_);
68  if (deltas_)
69  XMLUtils::addGenericChildAsList(doc, node, "Deltas", *deltas_);
70  if (moneyness_)
71  XMLUtils::addGenericChildAsList(doc, node, "Moneyness", *moneyness_);
72  if (expiries_)
73  XMLUtils::addGenericChildAsList(doc, node, "Expiries", *expiries_);
74  return node;
75 }
