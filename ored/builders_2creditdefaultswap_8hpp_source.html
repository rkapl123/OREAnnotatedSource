37/*! This class provides a key with which we will cache the CDS engine builders
38
39 In general, the CDS engine builders will be cached by the credit curve Id of the reference entity and the currency
40 of the trade that needs to be priced. If we are caching by credit curve Id and currency only, the recovery rate
41 member should be \c Null<Real>().
42
43 In some cases, for fixed recovery CDS trades for example, we need to cache the CDS engine builder not only by
44 credit curve Id and currency but also with an exogenous recovery rate that we wish to use instead of the market
45 supplied recovery rate.
46*/
47class CDSEngineKey {
48public:
49 /*! Constructor that takes a credit curve Id, \p creditCurveId, a currency, \p ccy, and optionally a recovery
50 rate, \p recoveryRate.
51 */
52 CDSEngineKey(const std::string& creditCurveId, const QuantLib::Currency& ccy,
53 QuantLib::Real recoveryRate = QuantLib::Null<QuantLib::Real>())
54 : creditCurveId_(creditCurveId), ccy_(ccy), recoveryRate_(recoveryRate) {}
55
56 //! Return the credit curve Id
57 const std::string& creditCurveId() const { return creditCurveId_; }
58
59 //! Return the currency
60 const QuantLib::Currency& currency() const { return ccy_; }
61
62 //! Return the recovery rate if it is set, otherwise \c Null<Real>()
63 QuantLib::Real recoveryRate() const { return recoveryRate_; }
64
65private:
66 std::string creditCurveId_;
67 QuantLib::Currency ccy_;
68 QuantLib::Real recoveryRate_;
69};

71inline bool operator==(const CDSEngineKey& lhs, const CDSEngineKey& rhs) {
72
73 // Check the credit curve IDs first
74 if (lhs.creditCurveId() != rhs.creditCurveId())
75 return false;
76
77 // Check the currencies
78 if (lhs.currency() != rhs.currency())
79 return false;
80
81 // Now check recovery rates.
82 if (lhs.recoveryRate() == QuantLib::Null<QuantLib::Real>() &&
83 rhs.recoveryRate() == QuantLib::Null<QuantLib::Real>()) {
84 // Most usual case is both are Null<Real> and using the recovery rate from the market.
85 return true;
86 } else if ((lhs.recoveryRate() != QuantLib::Null<QuantLib::Real>() &&
87 rhs.recoveryRate() == QuantLib::Null<QuantLib::Real>()) ||
88 (lhs.recoveryRate() == QuantLib::Null<QuantLib::Real>() &&
89 rhs.recoveryRate() != QuantLib::Null<QuantLib::Real>())) {
90 return false;
91 } else {
92 return QuantLib::close(lhs.recoveryRate(), rhs.recoveryRate());
93 }
94}

96inline bool operator<(const CDSEngineKey& lhs, const CDSEngineKey& rhs) {
97
98 // Check equality first
99 if (lhs == rhs)
100 return false;
101
102 // Now check credit curve Ids.
103 if (lhs.creditCurveId() != rhs.creditCurveId())
104 return lhs.creditCurveId() < rhs.creditCurveId();
105
106 // Now check currencies.
107 if (lhs.currency() != rhs.currency())
108 return lhs.currency().name() < rhs.currency().name();
109
110 // Now use the recovery rates
111 return lhs.recoveryRate() < rhs.recoveryRate();
112}

114inline bool operator!=(const CDSEngineKey& lhs, const CDSEngineKey& rhs) { return !(lhs == rhs); }
115inline bool operator>(const CDSEngineKey& lhs, const CDSEngineKey& rhs) { return rhs < lhs; }
116inline bool operator<=(const CDSEngineKey& lhs, const CDSEngineKey& rhs) { return !(lhs > rhs); }
117inline bool operator>=(const CDSEngineKey& lhs, const CDSEngineKey& rhs) { return !(lhs < rhs); }

136//! Midpoint engine builder class for credit default swaps
137/*! This class creates a MidPointCdsEngine
138 \ingroup builders
139*/
140class MidPointCdsEngineBuilder : public CreditDefaultSwapEngineBuilder {
141public:
142 MidPointCdsEngineBuilder() : CreditDefaultSwapEngineBuilder("DiscountedCashflows", "MidPointCdsEngine") {}
143
144protected:
145 boost::shared_ptr<PricingEngine>
146 engineImpl(QuantLib::Currency ccy, std::string creditCurveId,
147 QuantLib::Real recoveryRate = QuantLib::Null<QuantLib::Real>()) override {
148
149 auto cfg = configuration(MarketContext::pricing);
150 auto yts = market_->discountCurve(ccy.code(), cfg);
151 auto dpts = market_->defaultCurve(creditCurveId, cfg);
152
153 if (recoveryRate == QuantLib::Null<QuantLib::Real>()) {
154 // If recovery rate is null, get it from the market for the given reference entity
155 recoveryRate = market_->recoveryRate(creditCurveId, cfg)->value();
156 }
157
158 return boost::make_shared<MidPointCdsEngine>(dpts->curve(), recoveryRate, yts);
159 }
160};
