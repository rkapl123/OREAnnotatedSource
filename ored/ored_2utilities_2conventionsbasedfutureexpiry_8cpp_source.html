32 Date ConventionsBasedFutureExpiry::nextExpiry(bool includeExpiry, const Date& referenceDate, Natural offset,
33  bool forOption) {
34 
35  // Set the date relative to which we are calculating the next expiry
36  Date today = referenceDate == Date() ? Settings::instance().evaluationDate() : referenceDate;
37 
38  // Get the next expiry date relative to referenceDate
39  Date expiryDate = nextExpiry(today, forOption);
40 
41  // If expiry date equals today and we have asked not to include expiry, return next contract's expiry
42  if (expiryDate == today && !includeExpiry && offset == 0) {
43  expiryDate = nextExpiry(expiryDate + 1 * Days, forOption);
44  }
45 
46  // If offset is greater than 0, keep getting next expiry out
47  while (offset > 0) {
48  expiryDate = nextExpiry(expiryDate + 1 * Days, forOption);
49  offset--;
50  }
51 
52  return expiryDate;
53 }

55 Date ConventionsBasedFutureExpiry::priorExpiry(bool includeExpiry, const Date& referenceDate, bool forOption) {
56 
57  // Set the date relative to which we are calculating the preceding expiry
58  Date today = referenceDate == Date() ? Settings::instance().evaluationDate() : referenceDate;
59 
60  // Get the next expiry relative to the reference date (including the reference date)
61  Date expiry = nextExpiry(true, today, 0, forOption);
62 
63  // If that expiry is equal to reference date and we have set includeExpiry to true, we are done.
64  if (includeExpiry && expiry == today)
65  return expiry;
66 
67  // Get the preceding expiry.
68  Period p(convention_.contractFrequency());
69  Date baseDate = convention_.expiryCalendar().advance(expiry, -p);
70  expiry = nextExpiry(true, baseDate, 0, forOption);
71 
72  // May still not have the preceding expiry but must be close
73  Size counter = maxIterations_;
74  while (expiry >= today && counter > 0) {
75  baseDate--;
76  counter--;
77  expiry = nextExpiry(true, baseDate, 0, forOption);
78  }
79  QL_REQUIRE(expiry < today, "Expected that expiry " << io::iso_date(expiry) << " would be less than reference date "
80  << io::iso_date(today) << ".");
81 
82  return expiry;
83 }

93 Date ConventionsBasedFutureExpiry::expiry(Month contractMonth, Year contractYear, QuantLib::Natural monthOffset,
94  bool forOption) const {
95 
96  Date expiry;
97 
98  // Apply month offset if non-zero
99  if (monthOffset > 0) {
100  Date newDate = Date(15, contractMonth, contractYear) + monthOffset * Months;
101  contractMonth = newDate.month();
102  contractYear = newDate.year();
103  }
104 
105  // Move n months before (+ve) or after (-ve) for the expiry if necessary
106  if (convention_.expiryMonthLag() != 0) {
107  Date newDate = Date(15, contractMonth, contractYear) - convention_.expiryMonthLag() * Months;
108  contractMonth = newDate.month();
109  contractYear = newDate.year();
110  }
111 
112  // Calculate the relevant date in the expiry month and year
113  if (convention_.anchorType() == CommodityFutureConvention::AnchorType::DayOfMonth) {
114  Date last = Date::endOfMonth(Date(1, contractMonth, contractYear));
115  if (convention_.dayOfMonth() > static_cast<Natural>(last.dayOfMonth())) {
116  expiry = last;
117  } else {
118  expiry = Date(convention_.dayOfMonth(), contractMonth, contractYear);
119  }
120  } else if (convention_.anchorType() == CommodityFutureConvention::AnchorType::NthWeekday) {
121  expiry = Date::nthWeekday(convention_.nth(), convention_.weekday(), contractMonth, contractYear);
122  } else if (convention_.anchorType() == CommodityFutureConvention::AnchorType::CalendarDaysBefore) {
123  expiry = Date(1, contractMonth, contractYear) - convention_.calendarDaysBefore() * Days;
124  } else {
125  QL_FAIL("Did not recognise the commodity future convention's anchor type");
126  }
127 
128  // If expiry date is not a good business day, adjust it before applying the offset.
129  if (convention_.adjustBeforeOffset()) {
130  expiry = convention_.expiryCalendar().adjust(expiry, convention_.businessDayConvention());
131  }
132 
133  // Apply offset adjustments if necessary. A negative integer indicates that we move forward that number of days.
134  expiry = convention_.expiryCalendar().advance(expiry, -convention_.offsetDays(), Days);
135 
136  // If we want the option contract expiry, do the extra work here.
137  if (forOption) {
138  if (convention_.optionExpiryDay() != Null<Natural>()) {
139 
140  // Option expiry may be specified as n months before future expiry. Adjust here if necessary.
141  auto optionMonth = expiry.month();
142  auto optionYear = expiry.year();
143 
144  if (convention_.optionExpiryMonthLag() != 0) {
145  Date newDate = Date(15, optionMonth, optionYear) - convention_.optionExpiryMonthLag() * Months;
146  optionMonth = newDate.month();
147  optionYear = newDate.year();
148  }
149 
150  auto d = convention_.optionExpiryDay();
151  expiry = Date(d, optionMonth, optionYear);
152  expiry = convention_.expiryCalendar().adjust(expiry, convention_.optionBusinessDayConvention());
153 
154  } else {
155 
156  QL_REQUIRE(convention_.optionExpiryMonthLag() == 0 ||
157  convention_.expiryMonthLag() == convention_.optionExpiryMonthLag(), "The expiry month lag " <<
158  "and the option expiry month lag should be the same if using option expiry offset days.");
159 
160  expiry = convention_.expiryCalendar().advance(expiry,
161  -static_cast<Integer>(convention_.optionExpiryOffset()), Days);
162  }
163 
164  expiry = avoidProhibited(expiry, true);
165 
166  } else {
167  // If expiry date is one of the prohibited dates, move to preceding or following business day
168  expiry = avoidProhibited(expiry, false);
169  }
170 
171  return expiry;
172 }

174 Date ConventionsBasedFutureExpiry::nextExpiry(const Date& referenceDate, bool forOption) const {
175 
176  // If contract frequency is daily, next expiry is simply the next valid date on expiry calendar.
177  if (convention_.contractFrequency() == Daily) {
178  Date expiry = convention_.expiryCalendar().adjust(referenceDate, Following);
179  return avoidProhibited(expiry, false);
180  }
181 
182  // Get a contract expiry before today and increment until expiryDate is >= today
183  Date guideDate(15, convention_.oneContractMonth(), referenceDate.year() - 1);
184  Date expiryDate = expiry(convention_.oneContractMonth(), referenceDate.year() - 1, 0, forOption);
185  QL_REQUIRE(expiryDate < referenceDate, "Expected the expiry date in the previous year to be before reference");
186  while (expiryDate < referenceDate) {
187  guideDate += Period(convention_.contractFrequency());
188  expiryDate = expiry(guideDate.month(), guideDate.year(), 0, forOption);
189  }
190 
191  return expiryDate;
192 }

202 Date ConventionsBasedFutureExpiry::avoidProhibited(const Date& expiry, bool forOption) const {
203 
204  // If expiry date is one of the prohibited dates, move to preceding or following business day.
205  using PE = CommodityFutureConvention::ProhibitedExpiry;
206  Date result = expiry;
207  const auto& pes = convention_.prohibitedExpiries();
208  auto it = pes.find(PE(result));
209 
210  while (it != pes.end()) {
211 
212  // If not relevant, exit.
213  if ((!forOption && !it->forFuture()) || (forOption && !it->forOption()))
214  break;
215 
216  auto bdc = !forOption ? it->futureBdc() : it->optionBdc();
217 
218  if (bdc == Preceding || bdc == ModifiedPreceding) {
219  result = convention_.calendar().advance(result, -1, Days, bdc);
220  } else if (bdc == Following || bdc == ModifiedFollowing) {
221  result = convention_.calendar().advance(result, 1, Days, bdc);
222  } else {
223  QL_FAIL("Convention " << bdc << " associated with prohibited expiry " <<
224  io::iso_date(result) << " is not supported.");
225  }
226 
227  it = pes.find(PE(result));
228  }
229 
230  return result;
231 }
