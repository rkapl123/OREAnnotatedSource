44//! Serializable object holding generic trade data, reporting dimensions
45/*!
46 \ingroup tradedata
47*/
48class Envelope : public XMLSerializable {
49public:
50 //! Default constructor
51 Envelope() {}
52
53 //! Constructor with netting set id and portfolio ids, without additional fields
54 Envelope(const string& counterparty, const string& nettingSetId,
55 const set<string>& portfolioIds = set<string>())
56 : counterparty_(counterparty), nettingSetDetails_(NettingSetDetails(nettingSetId)),
57 portfolioIds_(portfolioIds) {}
58
59 //! Constructor with netting set details and portfolio ids, without additional fields
60 Envelope(const string& counterparty, const NettingSetDetails& nettingSetDetails = NettingSetDetails(),
61 const set<string>& portfolioIds = set<string>())
62 : counterparty_(counterparty), nettingSetDetails_(nettingSetDetails), portfolioIds_(portfolioIds) {}
63
64 //! Constructor without netting set / portfolio ids, with additional fields
65 Envelope(const string& counterparty, const map<string, string>& additionalFields)
66 : counterparty_(counterparty), nettingSetDetails_(NettingSetDetails()) {
67 for (const auto& addField : additionalFields)
68 additionalFields_[addField.first] = addField.second;
69 }
70
71 //! Constructor with netting set, with additional fields
72 Envelope(const string& counterparty, const string& nettingSetId, const map<string, string>& additionalFields,
73 const set<string>& portfolioIds = set<string>())
74 : counterparty_(counterparty), nettingSetDetails_(NettingSetDetails(nettingSetId)), portfolioIds_(portfolioIds) {
75 for (const auto& addField : additionalFields)
76 additionalFields_[addField.first] = addField.second;
77 }
78
79 //! Constructor with netting set details, with additional fields
80 Envelope(const string& counterparty, const NettingSetDetails& nettingSetDetails,
81 const map<string, string>& additionalFields, const set<string>& portfolioIds = set<string>())
82 : counterparty_(counterparty), nettingSetDetails_(nettingSetDetails), portfolioIds_(portfolioIds) {
83 for (const auto& addField : additionalFields)
84 additionalFields_[addField.first] = addField.second;
85 }
86
87 //! \name Serialisation
88 //@{
89 virtual void fromXML(XMLNode* node) override;
90 virtual XMLNode* toXML(XMLDocument& doc) override;
91 //@}
92
93 //! \name Inspectors
94 //@{
95 const string& counterparty() const { return counterparty_; }
96 const string& nettingSetId() const { return nettingSetDetails_.nettingSetId(); }
97 const NettingSetDetails nettingSetDetails() { return nettingSetDetails_; }
98 const set<string>& portfolioIds() const { return portfolioIds_; }
99 const map<string, string> additionalFields() const {
100 map<string, string> stringAddFields;
101 for (const auto& f : additionalFields_)
102 if (f.second.type() == typeid(string))
103 stringAddFields[f.first] = boost::any_cast<string>(f.second);
104 return stringAddFields;
105 }
106 const map<string, boost::any>& fullAdditionalFields() const { return additionalFields_; }
107 //@}
108
109 //! \name Utility
110 //@{
111 //! Check if the envelope has been populated
112 bool empty() const { return counterparty_ == ""; }
113 //! Check if the netting set details have been populated
114 bool hasNettingSetDetails() const { return !nettingSetDetails_.empty(); }
115 //@}
116
117private:
118 string counterparty_;
119 NettingSetDetails nettingSetDetails_;
120 set<string> portfolioIds_;
121 map<string, boost::any> additionalFields_;
122};
