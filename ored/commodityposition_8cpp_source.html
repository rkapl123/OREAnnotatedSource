47void CommodityPosition::build(const boost::shared_ptr<ore::data::EngineFactory>& engineFactory) {
48 DLOG("CommodityPosition::build() called for " << id());
49 QL_REQUIRE(!data_.underlyings().empty(), "CommodityPosition::build(): no underlyings given");
50 indices_.clear();
51 weights_.clear();
52 fxConversion_.clear();
53
54 std::vector<std::string> currencies;
55 for (auto const& u : data_.underlyings()) {
56 auto pts = engineFactory->market()->commodityPriceCurve(u.name(), engineFactory->configuration(MarketContext::pricing));
57 QL_REQUIRE(!pts.empty() && *pts != nullptr, "CommodityPosition, curve missing for '"<<u.name()<<"'");
58 QL_REQUIRE(!pts->currency().empty(),
59 "CommodityPosition, Currency not set in curve config for commodity curve'" << u.name() << "'. Skip this trade.");
60 auto index = parseCommodityIndex(u.name(), false, pts, NullCalendar(), u.priceType() == "FutureSettlement");
61 boost::shared_ptr<Conventions> conventions = InstrumentConventions::instance().conventions();
62 pair<bool, boost::shared_ptr<Convention>> p = conventions->get(u.name(), Convention::Type::CommodityFuture);
63 if (u.priceType() == "FutureSettlement" && p.first) {
64 auto convention = boost::dynamic_pointer_cast<CommodityFutureConvention>(p.second);
65 ConventionsBasedFutureExpiry feCalc(*convention);
66 Date expiry = Settings::instance().evaluationDate();
67 if (u.deliveryRollDays() != Null<Size>()) {
68 auto cal = u.deliveryRollCalendar().empty() ? convention->calendar() : parseCalendar(u.deliveryRollCalendar());
69 expiry = cal.advance(expiry, u.deliveryRollDays() * Days, convention->businessDayConvention());
70 }
71
72 Size nOffset = u.futureMonthOffset() == Null<Size>() ? 0 : u.futureMonthOffset();
73
74 expiry = feCalc.nextExpiry(true, expiry, nOffset);
75
76 index = index->clone(expiry, pts);
77 }
78 indices_.push_back(index);
79 weights_.push_back(u.weight());
80 currencies.push_back(pts->currency().code());
81 }
82
83 // get fx quotes
84
85 isSingleCurrency_ = true;
86 npvCurrency_ = currencies.front();
87 for (auto const& c : currencies) {
88 // we use fxSpot() as opposed to fxRate() here to ensure consistency between NPV() and the fixing of an
89 // equivalent index representing the same basket
90 fxConversion_.push_back(
91 engineFactory->market()->fxSpot(c + npvCurrency_, engineFactory->configuration(MarketContext::pricing)));
92 if (npvCurrency_ != c)
93 isSingleCurrency_ = false;
94 }
95
96 // set instrument
97 auto qlInstr =
98 boost::make_shared<CommodityPositionInstrumentWrapper>(data_.quantity(), indices_, weights_, fxConversion_);
99 qlInstr->setPricingEngine(boost::make_shared<CommodityPositionInstrumentWrapperEngine>());
100 instrument_ = boost::make_shared<VanillaInstrument>(qlInstr);
101
102 // no sensible way to set these members
103 maturity_ = Date::maxDate();
104 notional_ = Null<Real>();
105 notionalCurrency_ = "";
106
107 // ISDA taxonomy: not a derivative, but define the asset class at least
108 // so that we can determine a TRS asset class that has an EQ position underlying
109 additionalData_["isdaAssetClass"] = string("Commodity");
110 additionalData_["isdaBaseProduct"] = string("");
111 additionalData_["isdaSubProduct"] = string("");
112 additionalData_["isdaTransaction"] = string("");
113
114 // leave legs empty
115}

143CommodityPositionInstrumentWrapper::CommodityPositionInstrumentWrapper(
144 const Real quantity, const std::vector<boost::shared_ptr<QuantExt::CommodityIndex>>& commodities,
145 const std::vector<Real>& weights, const std::vector<Handle<Quote>>& fxConversion)
146 : quantity_(quantity), commodities_(commodities), weights_(weights), fxConversion_(fxConversion) {
147 QL_REQUIRE(commodities_.size() == weights_.size(), "CommodityPositionInstrumentWrapper: commodities size ("
148 << commodities_.size() << ") must match weights size ("
149 << weights_.size() << ")");
150 QL_REQUIRE(fxConversion_.empty() || fxConversion_.size() == commodities_.size(),
151 "CommodityPositionInstrumentWrapper: fxConversion size ("
152 << fxConversion_.size() << ") must match commodities size (" << commodities_.size() << ")");
153 for (auto const& i : commodities_)
154 registerWith(i);
155 for (auto const& f : fxConversion_)
156 registerWith(f);
157 registerWith(npvCcyConversion_);
158}

182void CommodityPositionInstrumentWrapperEngine::calculate() const {
183 Real result = 0.0;
184 Date today = Settings::instance().evaluationDate();
185 for (Size i = 0; i < arguments_.commodities_.size(); ++i) {
186 // TODO: if refering spot price need lookup the spot date instead of today
187 // for future settlement the fixing date is not relvant, always lookup future expiry date
188 Real tmp = arguments_.quantity_ * arguments_.commodities_[i]->fixing(today, true);
189 if (!arguments_.fxConversion_[i].empty()) {
190 tmp *= arguments_.fxConversion_[i]->value();
191 }
192 result += tmp * arguments_.weights_[i];
193 }
194 if (!arguments_.npvCcyConversion_.empty()) {
195 result *= arguments_.npvCcyConversion_->value();
196 }
197 results_.value = result;
198}
