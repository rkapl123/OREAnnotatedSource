44void CliquetOption::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
45
46 Currency ccy = parseCurrency(currency_);
47
48 QL_REQUIRE(tradeActions().empty(), "TradeActions not supported for VanillaOption");
49
50 // Payoff
51 Option::Type type = parseOptionType(callPut_);
52 boost::shared_ptr<PercentageStrikePayoff> payoff = boost::make_shared<PercentageStrikePayoff>(type, moneyness_);
53
54 Schedule schedule;
55 schedule = makeSchedule(scheduleData_);
56 Date expiryDate = schedule.dates().back();
57
58 // Vanilla European/American.
59 // If price adjustment is necessary we build a simple EU Option
60 boost::shared_ptr<QuantLib::Instrument> instrument;
61 boost::shared_ptr<EuropeanExercise> exerciseDate = boost::make_shared<EuropeanExercise>(expiryDate);
62
63 Date paymentDate = schedule.calendar().advance(expiryDate, settlementDays_, Days);
64
65 for (auto d : schedule.dates())
66 valuationDates_.insert(schedule.calendar().adjust(d, schedule.businessDayConvention()));
67
68 Position::Type position = parsePositionType(longShort_);
69
70 // Create QuantExt Cliquet Option
71 boost::shared_ptr<Instrument> cliquet = boost::make_shared<QuantExt::CliquetOption>(
72 payoff, exerciseDate, valuationDates_, paymentDate, cliquetNotional_, position, localCap_, localFloor_,
73 globalCap_, globalFloor_, premium_, parseDate(premiumPayDate_), premiumCcy_);
74
75 boost::shared_ptr<EngineBuilder> builder = engineFactory->builder(tradeType_);
76 QL_REQUIRE(builder, "No builder found for " << tradeType_);
77 boost::shared_ptr<CliquetOptionEngineBuilder> cliquetOptionBuilder =
78 boost::dynamic_pointer_cast<CliquetOptionEngineBuilder>(builder);
79
80 cliquet->setPricingEngine(cliquetOptionBuilder->engine(name(), ccy));
81
82 instrument_ = boost::shared_ptr<InstrumentWrapper>(new VanillaInstrument(cliquet));
83
84 npvCurrency_ = currency_;
85 maturity_ = expiryDate;
86 notional_ = cliquetNotional_;
87 notionalCurrency_ = currency_;
88
89 // add required fixings (all valuation dates)
90 for (auto const& d : valuationDates_) {
91 requiredFixings_.addFixingDate(d, "EQ-" + name(), paymentDate);
92 }
93
94 additionalData_["notional"] = cliquetNotional_;
95 additionalData_["currency"] = currency_;
96}

98void CliquetOption::fromXML(XMLNode* node) {
99 Trade::fromXML(node);
100
101 XMLNode* clNode = XMLUtils::getChildNode(node, tradeType() + "Data");
102 QL_REQUIRE(clNode, "No EquityCliquetOptionData Node");
103
104 XMLNode* tmp = XMLUtils::getChildNode(clNode, "Underlying");
105 if (!tmp)
106 tmp = XMLUtils::getChildNode(node, "Name");
107 UnderlyingBuilder underlyingBuilder;
108 underlyingBuilder.fromXML(tmp);
109 underlying_ = underlyingBuilder.underlying();
110
111 currency_ = XMLUtils::getChildValue(clNode, "Currency", true);
112 cliquetNotional_ = XMLUtils::getChildValueAsDouble(clNode, "Notional", true);
113 longShort_ = XMLUtils::getChildValue(clNode, "LongShort", true);
114 callPut_ = XMLUtils::getChildValue(clNode, "OptionType", true);
115 XMLNode* scheduleNode = XMLUtils::getChildNode(clNode, "ScheduleData");
116 scheduleData_.fromXML(scheduleNode);
117 moneyness_ = XMLUtils::getChildValueAsDouble(clNode, "Moneyness", false);
118 localCap_ = getRealOrNull(clNode, "LocalCap");
119 localFloor_ = getRealOrNull(clNode, "LocalFloor");
120 globalCap_ = getRealOrNull(clNode, "GlobalCap");
121 globalFloor_ = getRealOrNull(clNode, "GlobalFloor");
122 settlementDays_ = XMLUtils::getChildValueAsInt(clNode, "SettlementDays", false);
123 premium_ = XMLUtils::getChildValueAsDouble(clNode, "Premium", false);
124 premiumCcy_ = XMLUtils::getChildValue(clNode, "PremiumCurrency", false);
125 premiumPayDate_ = XMLUtils::getChildValue(clNode, "PremiumPaymentDate", false);
126}

128XMLNode* CliquetOption::toXML(XMLDocument& doc) {
129 XMLNode* node = Trade::toXML(doc);
130 XMLNode* clNode = doc.allocNode(tradeType() + "Data");
131 XMLUtils::appendNode(node, clNode);
132 XMLUtils::appendNode(clNode, underlying_->toXML(doc));
133 XMLUtils::addChild(doc, clNode, "Currency", currency_);
134 XMLUtils::addChild(doc, clNode, "Notional", cliquetNotional_);
135 XMLUtils::addChild(doc, clNode, "LongShort", longShort_);
136 XMLUtils::addChild(doc, clNode, "OptionType", callPut_);
137 XMLUtils::appendNode(clNode, scheduleData_.toXML(doc));
138 if (moneyness_ != Null<Real>())
139 XMLUtils::addChild(doc, clNode, "Moneyness", moneyness_);
140 if (localCap_ != Null<Real>())
141 XMLUtils::addChild(doc, clNode, "LocalCap", localCap_);
142 if (localFloor_ != Null<Real>())
143 XMLUtils::addChild(doc, clNode, "LocalFloor", localFloor_);
144 if (globalCap_ != Null<Real>())
145 XMLUtils::addChild(doc, clNode, "GlobalCap", globalCap_);
146 if (globalFloor_ != Null<Real>())
147 XMLUtils::addChild(doc, clNode, "GlobalFloor", globalFloor_);
148 if (settlementDays_ != Null<Size>())
149 XMLUtils::addChild(doc, clNode, "SettlementDays", static_cast<int>(settlementDays_));
150 if (premium_ != Null<Real>())
151 XMLUtils::addChild(doc, clNode, "Premium", premium_);
152 if (premiumCcy_ != "")
153 XMLUtils::addChild(doc, clNode, "PremiumCurrency", premiumCcy_);
154 if (premiumPayDate_ != "")
155 XMLUtils::addChild(doc, clNode, "PremiumPaymentDate", premiumPayDate_);
156 return node;
157}
