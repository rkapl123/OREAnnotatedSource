44void CliquetOption::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
45
46 Currency ccy = parseCurrency(currency_);
47
48 QL_REQUIRE(tradeActions().empty(), "TradeActions not supported for VanillaOption");
49
50 // Payoff
51 Option::Type type = parseOptionType(callPut_);
52 boost::shared_ptr<PercentageStrikePayoff> payoff = boost::make_shared<PercentageStrikePayoff>(type, moneyness_);
53
54 Schedule schedule;
55 schedule = makeSchedule(scheduleData_);
56 Date expiryDate = schedule.dates().back();
57
58 // Vanilla European/American.
59 // If price adjustment is necessary we build a simple EU Option
60 boost::shared_ptr<QuantLib::Instrument> instrument;
61 boost::shared_ptr<EuropeanExercise> exerciseDate = boost::make_shared<EuropeanExercise>(expiryDate);
62
63 Date paymentDate = schedule.calendar().advance(expiryDate, settlementDays_, Days);
64
65 for (auto d : schedule.dates())
66 valuationDates_.insert(schedule.calendar().adjust(d, schedule.businessDayConvention()));
67
68 Position::Type position = parsePositionType(longShort_);
69
70 // Create QuantExt Cliquet Option
71 boost::shared_ptr<Instrument> cliquet = boost::make_shared<QuantExt::CliquetOption>(
72 payoff, exerciseDate, valuationDates_, paymentDate, cliquetNotional_, position, localCap_, localFloor_,
73 globalCap_, globalFloor_, premium_, parseDate(premiumPayDate_), premiumCcy_);
74
75 boost::shared_ptr<EngineBuilder> builder = engineFactory->builder(tradeType_);
76 QL_REQUIRE(builder, "No builder found for " << tradeType_);
77 boost::shared_ptr<CliquetOptionEngineBuilder> cliquetOptionBuilder =
78 boost::dynamic_pointer_cast<CliquetOptionEngineBuilder>(builder);
79
80 cliquet->setPricingEngine(cliquetOptionBuilder->engine(name(), ccy));
81
82 instrument_ = boost::shared_ptr<InstrumentWrapper>(new VanillaInstrument(cliquet));
83
84 npvCurrency_ = currency_;
85 maturity_ = expiryDate;
86 notional_ = cliquetNotional_;
87 notionalCurrency_ = currency_;
88
89 // add required fixings (all valuation dates)
90 for (auto const& d : valuationDates_) {
91 requiredFixings_.addFixingDate(d, "EQ-" + name(), paymentDate);
92 }
93
94 additionalData_["notional"] = cliquetNotional_;
95 additionalData_["currency"] = currency_;
96
97 // ISDA taxonomy
98 if (underlying_->type() == "Equity") {
99 additionalData_["isdaAssetClass"] = string("Equity");
100 additionalData_["isdaBaseProduct"] = string("Other");
101 additionalData_["isdaSubProduct"] = string("Price Return Basic Performance");
102 }
103 else if (underlying_->type() == "Commodity") {
104 // assuming that Commoditiy is treated like Equity
105 additionalData_["isdaAssetClass"] = string("Commodity");
106 additionalData_["isdaBaseProduct"] = string("Other");
107 additionalData_["isdaSubProduct"] = string("Price Return Basic Performance");
108 }
109 else if (underlying_->type() == "FX") {
110 additionalData_["isdaAssetClass"] = string("Foreign Exchange");
111 additionalData_["isdaBaseProduct"] = string("Complex Exotic");
112 additionalData_["isdaSubProduct"] = string("Generic");
113 }
114 else {
115 WLOG("ISDA taxonomy not set for trade " << id());
116 }
117
118 additionalData_["isdaTransaction"] = string("");
119}

121void CliquetOption::fromXML(XMLNode* node) {
122 Trade::fromXML(node);
123
124 XMLNode* clNode = XMLUtils::getChildNode(node, tradeType() + "Data");
125 QL_REQUIRE(clNode, "No EquityCliquetOptionData Node");
126
127 XMLNode* tmp = XMLUtils::getChildNode(clNode, "Underlying");
128 if (!tmp)
129 tmp = XMLUtils::getChildNode(node, "Name");
130 UnderlyingBuilder underlyingBuilder;
131 underlyingBuilder.fromXML(tmp);
132 underlying_ = underlyingBuilder.underlying();
133
134 currency_ = XMLUtils::getChildValue(clNode, "Currency", true);
135 cliquetNotional_ = XMLUtils::getChildValueAsDouble(clNode, "Notional", true);
136 longShort_ = XMLUtils::getChildValue(clNode, "LongShort", true);
137 callPut_ = XMLUtils::getChildValue(clNode, "OptionType", true);
138 XMLNode* scheduleNode = XMLUtils::getChildNode(clNode, "ScheduleData");
139 scheduleData_.fromXML(scheduleNode);
140 moneyness_ = XMLUtils::getChildValueAsDouble(clNode, "Moneyness", false);
141 localCap_ = getRealOrNull(clNode, "LocalCap");
142 localFloor_ = getRealOrNull(clNode, "LocalFloor");
143 globalCap_ = getRealOrNull(clNode, "GlobalCap");
144 globalFloor_ = getRealOrNull(clNode, "GlobalFloor");
145 settlementDays_ = XMLUtils::getChildValueAsInt(clNode, "SettlementDays", false);
146 premium_ = XMLUtils::getChildValueAsDouble(clNode, "Premium", false);
147 premiumCcy_ = XMLUtils::getChildValue(clNode, "PremiumCurrency", false);
148 premiumPayDate_ = XMLUtils::getChildValue(clNode, "PremiumPaymentDate", false);
149}

151XMLNode* CliquetOption::toXML(XMLDocument& doc) {
152 XMLNode* node = Trade::toXML(doc);
153 XMLNode* clNode = doc.allocNode(tradeType() + "Data");
154 XMLUtils::appendNode(node, clNode);
155 XMLUtils::appendNode(clNode, underlying_->toXML(doc));
156 XMLUtils::addChild(doc, clNode, "Currency", currency_);
157 XMLUtils::addChild(doc, clNode, "Notional", cliquetNotional_);
158 XMLUtils::addChild(doc, clNode, "LongShort", longShort_);
159 XMLUtils::addChild(doc, clNode, "OptionType", callPut_);
160 XMLUtils::appendNode(clNode, scheduleData_.toXML(doc));
161 if (moneyness_ != Null<Real>())
162 XMLUtils::addChild(doc, clNode, "Moneyness", moneyness_);
163 if (localCap_ != Null<Real>())
164 XMLUtils::addChild(doc, clNode, "LocalCap", localCap_);
165 if (localFloor_ != Null<Real>())
166 XMLUtils::addChild(doc, clNode, "LocalFloor", localFloor_);
167 if (globalCap_ != Null<Real>())
168 XMLUtils::addChild(doc, clNode, "GlobalCap", globalCap_);
169 if (globalFloor_ != Null<Real>())
170 XMLUtils::addChild(doc, clNode, "GlobalFloor", globalFloor_);
171 if (settlementDays_ != Null<Size>())
172 XMLUtils::addChild(doc, clNode, "SettlementDays", static_cast<int>(settlementDays_));
173 if (premium_ != Null<Real>())
174 XMLUtils::addChild(doc, clNode, "Premium", premium_);
175 if (premiumCcy_ != "")
176 XMLUtils::addChild(doc, clNode, "PremiumCurrency", premiumCcy_);
177 if (premiumPayDate_ != "")
178 XMLUtils::addChild(doc, clNode, "PremiumPaymentDate", premiumPayDate_);
179 return node;
180}
