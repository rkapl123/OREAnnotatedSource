38class CSA {
39public:
40 /*!
41 Enumeration 'Type' specifies what type of
42 collateral agreement is in place.
43 - 'Bilateral' => both sides can request collateral margins
44 - 'PostOnly' => only the counterparty is allowed to issue a call for additional collateral
45 - 'CallOnly' => only WE are allowed to issue a margin call for additional collateral
46 */
47 enum Type { Bilateral, CallOnly, PostOnly };
48
49 CSA(const Type& type,
50 const string& csaCurrency, // three letter ISO code
51 const string& index, const Real& thresholdPay, const Real& thresholdRcv, const Real& mtaPay, const Real& mtaRcv,
52 const Real& iaHeld, const string& iaType, const Period& marginCallFreq, const Period& marginPostFreq,
53 const Period& mpr, const Real& collatSpreadPay, const Real& collatSpreadRcv,
54 const vector<string>& eligCollatCcys, // vector of three letter ISO codes
55 bool applyInitialMargin, Type initialMarginType, const bool calculateIMAmount, const bool calculateVMAmount)
56 : type_(type), csaCurrency_(csaCurrency), index_(index), thresholdPay_(thresholdPay),
57 thresholdRcv_(thresholdRcv), mtaPay_(mtaPay), mtaRcv_(mtaRcv), iaHeld_(iaHeld), iaType_(iaType),
58 marginCallFreq_(marginCallFreq), marginPostFreq_(marginPostFreq), mpr_(mpr),
59 collatSpreadPay_(collatSpreadPay), collatSpreadRcv_(collatSpreadRcv), eligCollatCcys_(eligCollatCcys),
60 applyInitialMargin_(applyInitialMargin), initialMarginType_(initialMarginType),
61 calculateIMAmount_(calculateIMAmount), calculateVMAmount_(calculateVMAmount) {}
62
63 //! Inspectors
64 //@{
65 /*! Nature of CSA margining agreement (e.g. Bilateral, PostOnly, CallOnly) */
66 const Type& type() const { return type_; }
67 /*! Master Currency of CSA */
68 const string& csaCurrency() const { return csaCurrency_; }
69 /*! Index that determines the compounding rate */
70 const string& index() const { return index_; }
71 /*! Threshold amount for margin calls issued by counterparty */
72 Real thresholdPay() const { return thresholdPay_; }
73 /*! Threshold amount when issuing calls to counterparty */
74 Real thresholdRcv() const { return thresholdRcv_; }
75 /*! Minimum Transfer Amount when posting collateral to counterparty */
76 Real mtaPay() const { return mtaPay_; }
77 /*! Minimum Transfer Amount when receiving collateral from counterparty */
78 Real mtaRcv() const { return mtaRcv_; }
79 /*! Sum of Independent Amounts covered by the CSA (positive => we hold the amount) */
80 Real independentAmountHeld() const { return iaHeld_; }
81 /*! 'Type' of Independent Amount as specified in the CSA */
82 const string& independentAmountType() const { return iaType_; }
83 /*! Margining Frequency when requesting collateral from counterparty (e.g. "1D", "1W") */
84 const Period& marginCallFrequency() const { return marginCallFreq_; }
85 /*! Margining Frequency when counterparty is requesting collateral (e.g. "1D", "1W") */
86 const Period& marginPostFrequency() const { return marginPostFreq_; }
87 /*! Margin Period of Risk (e.g. "1M") */
88 const Period& marginPeriodOfRisk() const { return mpr_; }
89 /*! Spread for interest accrual on held collateral */
90 Real collatSpreadRcv() const { return collatSpreadRcv_; }
91 /*! Spread for interest accrual on posted collateral */
92 Real collatSpreadPay() const { return collatSpreadPay_; }
93 /*! Eligible Collateral Currencies */
94 vector<string> eligCollatCcys() const { return eligCollatCcys_; }
95 /*! Apply (dynamic) initial margin in addition to variation margin */
96 bool applyInitialMargin() { return applyInitialMargin_; }
97 /*! Direction of (dynamic) initial margin */
98 Type initialMarginType() { return initialMarginType_; }
99 //@}
100 /*! Calculate SIMM as IM (currently used only for SA-CCR) */
101 bool calculateIMAmount() { return calculateIMAmount_; }
102 /*! Calculate VM from NPV (currently used only for SA-CCR) */
103 bool calculateVMAmount() { return calculateVMAmount_; }
104
105 /*! invert all relevant aspects of the CSA */
106 void invertCSA();
107
108 void validate();
