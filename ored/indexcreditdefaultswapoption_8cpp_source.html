55void IndexCreditDefaultSwapOption::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
56
57 DLOG("IndexCreditDefaultSwapOption::build() called for trade " << id());
58
59 // Dates
60 const boost::shared_ptr<Market>& market = engineFactory->market();
61 Date asof = market->asofDate();
62 if (asof == Null<Date>() || asof == Date()) {
63 asof = Settings::instance().evaluationDate();
64 }
65
66 if (tradeDate_ == Date()) {
67 tradeDate_ = asof;
68 } else {
69 QL_REQUIRE(tradeDate_ <= asof, "Trade date (" << io::iso_date(tradeDate_) << ") should be on or "
70 << "before the valuation date (" << io::iso_date(asof) << ")");
71 }
72
73 if (fepStartDate_ == Date()) {
74 fepStartDate_ = tradeDate_;
75 } else {
76 QL_REQUIRE(fepStartDate_ <= tradeDate_, "Front end protection start date ("
77 << io::iso_date(fepStartDate_)
78 << ") should be on or before the trade date ("
79 << io::iso_date(tradeDate_) << ")");
80 }
81
82 // Option trade notional. This is the full notional of the the index that is being traded not reduced by any
83 // defaults. The notional on the trade date will be a fraction of this if there are defaults by trade date.
84 auto legData = swap_.leg();
85 const auto& ntls = legData.notionals();
86 QL_REQUIRE(ntls.size() == 1, "IndexCreditDefaultSwapOption requires a single notional.");
87 notionals_ = Notionals();
88 notionals_.full = ntls.front();
89 notionalCurrency_ = legData.currency();
90
91 // Populate the constituents and determine the various notional amounts.
92 constituents_.clear();
93 if (swap_.basket().constituents().size() > 1) {
94 fromBasket(asof, constituents_);
95 } else {
96 fromReferenceData(asof, constituents_, engineFactory->referenceData());
97 }
98
99 // Transfer to vectors for ctors below
100 vector<string> constituentIds;
101 constituentIds.reserve(constituents_.size());
102 vector<Real> constituentNtls;
103 constituentNtls.reserve(constituents_.size());
104 for (const auto& kv : constituents_) {
105 constituentIds.push_back(kv.first);
106 constituentNtls.push_back(kv.second);
107 }
108
109 // Need fixed leg data with one rate. This should be the standard running coupon on the index CDS e.g.
110 // 100bp for CDX IG and 500bp for CDX HY.
111 QL_REQUIRE(legData.legType() == "Fixed", "Index CDS option " << id() << " requires fixed leg.");
112 auto fixedLegData = boost::dynamic_pointer_cast<FixedLegData>(legData.concreteLegData());
113 QL_REQUIRE(fixedLegData->rates().size() == 1, "Index CDS option " << id() << " requires single fixed rate.");
114 auto runningCoupon = fixedLegData->rates().front();
115 Real upfrontFee = swap_.upfrontFee();
116
117 // Payer (Receiver) swaption if the leg is paying (receiving).
118 auto side = legData.isPayer() ? Protection::Side::Buyer : Protection::Side::Seller;
119
120 // Day counter. In general for CDS and CDS index trades, the standard day counter is Actual/360 and the final
121 // period coupon accrual includes the maturity date.
122 DayCounter dc = parseDayCounter(legData.dayCounter());
123 Actual360 standardDayCounter;
124 DayCounter lastPeriodDayCounter = dc == standardDayCounter ? Actual360(true) : dc;
125
126 // Checks on the option data
127 QL_REQUIRE(option_.style() == "European", "IndexCreditDefaultSwapOption option style must"
128 << " be European but got " << option_.style() << ".");
129 QL_REQUIRE(option_.exerciseFees().empty(), "IndexCreditDefaultSwapOption cannot handle exercise fees.");
130
131 // Exercise must be European
132 const auto& exerciseDates = option_.exerciseDates();
133 QL_REQUIRE(exerciseDates.size() == 1, "IndexCreditDefaultSwapOption expects one exercise date"
134 << " but got " << exerciseDates.size() << " exercise dates.");
135 Date exerciseDate = parseDate(exerciseDates.front());
136 boost::shared_ptr<Exercise> exercise = boost::make_shared<EuropeanExercise>(exerciseDate);
137
138 QL_REQUIRE(parseDate(legData.schedule().rules().front().endDate()) > exerciseDate,
139 "IndexCreditDefaultSwapOption: ExerciseDate must be before EndDate");
140
141 // We apply an automatic correction to a common mistake in the input data, where the full index underlying
142 // is provided and not only the part of the underlying into which we exercise.
143 if (legData.schedule().rules().size() == 1 && legData.schedule().dates().empty()) {
144 // The start date should be >= exercise date, this will produce correct coupons for both
145 // - post big bang rules CDS, CDS2015 (full first coupon) and
146 // - pre big bang rules (short first coupon)
147 if (parseDate(legData.schedule().rules().front().startDate()) < exerciseDate) {
148 legData.schedule().modifyRules().front().modifyStartDate() = ore::data::to_string(exerciseDate);
149 }
150 }
151
152 // Schedule
153 Schedule schedule = makeSchedule(legData.schedule());
154 BusinessDayConvention payConvention = parseBusinessDayConvention(legData.paymentConvention());
155
156 // Usually, we expect a Strike and StrikeType. However, for backwards compatibility we also allow for
157 // empty values and populate Strike, StrikeType from the underlying upfront and coupon.
158 QL_REQUIRE(strikeType_ == "" || strikeType_ == "Spread" || strikeType_ == "Price",
159 "invalid StrikeType (" << strikeType_ << "), expected 'Spread' or 'Price' or empty value");
160 if (strike_ == Null<Real>() && strikeType_ == "" && upfrontFee == Null<Real>()) {
161 effectiveStrike_ = runningCoupon;
162 effectiveStrikeType_ = "Spread";
163 } else if (strike_ == Null<Real>() && strikeType_ == "Spread" && upfrontFee == Null<Real>()) {
164 effectiveStrike_ = runningCoupon;
165 effectiveStrikeType_ = "Spread";
166 } else if (strike_ == Null<Real>() && strikeType_ == "Price" && upfrontFee == Null<Real>()) {
167 effectiveStrike_ = 1.0;
168 effectiveStrikeType_ = "Price";
169 } else if (strike_ != Null<Real>() && strikeType_ == "" && upfrontFee == Null<Real>()) {
170 effectiveStrike_ = strike_;
171 effectiveStrikeType_ = "Spread";
172 } else if (strike_ != Null<Real>() && strikeType_ == "Spread" && upfrontFee == Null<Real>()) {
173 effectiveStrike_ = strike_;
174 effectiveStrikeType_ = "Spread";
175 } else if (strike_ != Null<Real>() && strikeType_ == "Price" && upfrontFee == Null<Real>()) {
176 effectiveStrike_ = strike_;
177 effectiveStrikeType_ = "Price";
178 } else if (strike_ == Null<Real>() && strikeType_ == "" && upfrontFee != Null<Real>()) {
179 effectiveStrike_ = 1.0 - upfrontFee;
180 effectiveStrikeType_ = "Price";
181 } else if (strike_ == Null<Real>() && strikeType_ == "Spread" && upfrontFee != Null<Real>()) {
182 if (close_enough(upfrontFee, 0.0)) {
183 effectiveStrike_ = runningCoupon;
184 effectiveStrikeType_ = "Spread";
185 } else {
186 QL_FAIL("StrikeType 'Spread' and non-zero upfront fee can not be combined.");
187 }
188 } else if (strike_ == Null<Real>() && strikeType_ == "Price" && upfrontFee != Null<Real>()) {
189 effectiveStrike_ = 1.0 - upfrontFee;
190 effectiveStrikeType_ = "Price";
191 } else if (strike_ != Null<Real>() && strikeType_ == "" && upfrontFee != Null<Real>()) {
192 if (close_enough(upfrontFee, 0.0)) {
193 effectiveStrike_ = strike_;
194 effectiveStrikeType_ = "Spread";
195 } else {
196 QL_FAIL("Strike and non-zero upfront can not be combined.");
197 }
198 } else if (strike_ != Null<Real>() && strikeType_ == "Spread" && upfrontFee != Null<Real>()) {
199 if (close_enough(upfrontFee, 0.0)) {
200 effectiveStrike_ = strike_;
201 effectiveStrikeType_ = "Spread";
202 } else {
203 QL_FAIL("Strike and non-zero upfront can not be combined.");
204 }
205 } else if (strike_ != Null<Real>() && strikeType_ == "Price" && upfrontFee != Null<Real>()) {
206 if (close_enough(upfrontFee, 0.0)) {
207 effectiveStrike_ = strike_;
208 effectiveStrikeType_ = "Price";
209 } else {
210 QL_FAIL("Strike and non-zero upfront can not be combined.");
211 }
212 } else {
213 QL_FAIL("internal error, impossible branch in strike / strike type deduction.");
214 }
215 DLOG("Will use strike = " << effectiveStrike_ << ", strikeType = " << effectiveStrikeType_);
216
217 // Populate trade date and protection start date of underlying swap
218 QL_REQUIRE(!schedule.dates().empty(),
219 "IndexCreditDefaultSwapOption: underlying swap schedule does not contain any dates");
220 Date underlyingTradeDate =
221 swap_.tradeDate() == Date() ? std::max(exerciseDate, schedule.dates().front()) : swap_.tradeDate();
222 Date underlyingProtectionStart;
223 if (swap_.protectionStart() != Date()) {
224 underlyingProtectionStart = swap_.protectionStart();
225 } else if (legData.schedule().rules().size() == 1 && legData.schedule().dates().empty()) {
226 auto rule = parseDateGenerationRule(legData.schedule().rules().front().rule());
227 if (rule == DateGeneration::CDS || rule == DateGeneration::CDS2015) {
228 underlyingProtectionStart = std::max(exerciseDate, schedule.dates().front());
229 } else {
230 underlyingProtectionStart = schedule.dates().front();
231 }
232 } else {
233 underlyingProtectionStart = std::max(exerciseDate, schedule.dates().front());
234 }
235
236 // get engine builders for option and underlying swap
237 auto iCdsOptionEngineBuilder = boost::dynamic_pointer_cast<IndexCreditDefaultSwapOptionEngineBuilder>(
238 engineFactory->builder("IndexCreditDefaultSwapOption"));
239 QL_REQUIRE(iCdsOptionEngineBuilder,
240 "IndexCreditDefaultSwapOption: internal error, expected IndexCreditDefaultSwapOptionEngineBuilder");
241 auto iCdsEngineBuilder = boost::dynamic_pointer_cast<IndexCreditDefaultSwapEngineBuilder>(
242 engineFactory->builder("IndexCreditDefaultSwap"));
243 QL_REQUIRE(iCdsEngineBuilder,
244 "IndexCreditDefaultSwap: internal error, expected IndexCreditDefaultSwapEngineBuilder");
245
246 // The underlying index CDS as it looks on the valuation date i.e. outstanding notional is the valuation
247 // date notional and the basket of notionals contains only those reference entities not defaulted (i.e.
248 // those with auction date in the future to be more precise).
249 auto cds = boost::make_shared<QuantExt::IndexCreditDefaultSwap>(
250 side, notionals_.valuationDate, constituentNtls, runningCoupon, schedule, payConvention, dc,
251 swap_.settlesAccrual(), swap_.protectionPaymentTime(), underlyingProtectionStart, boost::shared_ptr<Claim>(),
252 lastPeriodDayCounter, true, underlyingTradeDate, swap_.cashSettlementDays());
253
254 // Set engine on the underlying CDS.
255 npvCurrency_ = legData.currency();
256 auto ccy = parseCurrency(npvCurrency_);
257 std::string overrideCurve = iCdsOptionEngineBuilder->engineParameter("Curve", {}, false, "Underlying");
258
259 auto creditCurveId = this->creditCurveId();
260 // warn if that is not possible, except for trades on custom baskets
261 if (swap_.basket().constituents().empty() && splitCurveIdWithTenor(creditCurveId).second == 0 * Days) {
262 ALOG(StructuredTradeWarningMessage(id(), tradeType(), "Could not imply Index CDS term.",
263 "Index CDS term could not be derived from start, end date, are these "
264 "dates correct (credit curve id is '" +
265 swap_.creditCurveId() + "')"));
266 }
267
268 // for cash settlement build the underlying swap with the inccy discount curve
269 Settlement::Type settleType = parseSettlementType(option_.settlement());
270 cds->setPricingEngine(iCdsEngineBuilder->engine(ccy, creditCurveId, constituentIds, overrideCurve,
271 swap_.recoveryRate(), settleType == Settlement::Cash));
272
273 // Strike may be in terms of spread or price
274 auto strikeType = parseCdsOptionStrikeType(effectiveStrikeType_);
275
276 // Determine the index term;
277 effectiveIndexTerm_ = 5 * Years;
278 if (!indexTerm_.empty()) {
279 // if the option has an explicit index term set, we use that
280 effectiveIndexTerm_ = parsePeriod(indexTerm_);
281 } else {
282 // otherwise we derive the index term from the start date (or an externally set hint for that)
283 effectiveIndexTerm_ = QuantExt::implyIndexTerm(swap_.indexStartDateHint() == Date() ? schedule.dates().front()
284 : swap_.indexStartDateHint(),
285 schedule.dates().back());
286 }
287
288 // Build the option
289 auto option = boost::make_shared<QuantExt::IndexCdsOption>(cds, exercise, effectiveStrike_, strikeType, settleType,
290 notionals_.tradeDate, notionals_.realisedFep, knockOut_,
291 effectiveIndexTerm_);
292 // the vol curve id is the credit curve id stripped by a term, if the credit curve id should contain one
293 auto p = splitCurveIdWithTenor(swap_.creditCurveId());
294 volCurveId_ = p.first;
295 option->setPricingEngine(iCdsOptionEngineBuilder->engine(ccy, creditCurveId, volCurveId_, constituentIds));
296
297 // Keep this comment about the maturity date being the underlying maturity instead of the option expiry.
298 // [RL] Align option product maturities with ISDA AANA/GRID guidance as of November 2020.
299 maturity_ = cds->coupons().back()->date();
300
301 // Set Trade members _before_ possibly adding the premium payment below.
302 legs_ = {cds->coupons()};
303 legCurrencies_ = {npvCurrency_};
304 legPayers_ = {legData.isPayer()};
305
306 // Long or short the option
307 Position::Type positionType = parsePositionType(option_.longShort());
308 Real indicatorLongShort = positionType == Position::Long ? 1.0 : -1.0;
309
310 // Include premium if enough information is provided
311 vector<boost::shared_ptr<Instrument>> additionalInstruments;
312 vector<Real> additionalMultipliers;
313 string configuration = iCdsOptionEngineBuilder->configuration(MarketContext::pricing);
314 maturity_ =
315 std::max(maturity_, addPremiums(additionalInstruments, additionalMultipliers, 1.0, option_.premiumData(),
316 -indicatorLongShort, ccy, engineFactory, configuration));
317
318 // Instrument wrapper depends on the settlement type.
319 // The instrument build should be indpednent of the evaluation date. However, the general behavior
320 // in ORE (e.g. IR swaptions) for normal pricing runs is that the option is considered expired on
321 // the expiry date with no assumptions on an (automatic) exercise. Therefore we build a vanilla
322 // instrument if the exercise date is <= the eval date at build time.
323 if (settleType == Settlement::Cash || exerciseDate <= Settings::instance().evaluationDate()) {
324 instrument_ = boost::make_shared<VanillaInstrument>(option, indicatorLongShort, additionalInstruments,
325 additionalMultipliers);
326 } else {
327 bool isLong = positionType == Position::Long;
328 bool isPhysical = settleType == Settlement::Physical;
329 instrument_ = boost::make_shared<EuropeanOptionWrapper>(option, isLong, exerciseDate, isPhysical, cds, 1.0, 1.0,
330 additionalInstruments, additionalMultipliers);
331 }
332
333 sensitivityDecomposition_ = iCdsOptionEngineBuilder->sensitivityDecomposition();
334
335 // ISDA taxonomy
336 additionalData_["isdaAssetClass"] = string("Credit");
337 additionalData_["isdaBaseProduct"] = string("Swaptions");
338 string entity = swap_.creditCurveId();
339 boost::shared_ptr<ReferenceDataManager> refData = engineFactory->referenceData();
340 if (refData && refData->hasData("CreditIndex", entity)) {
341 auto refDatum = refData->getData("CreditIndex", entity);
342 boost::shared_ptr<CreditIndexReferenceDatum> creditIndexRefDatum = boost::dynamic_pointer_cast<CreditIndexReferenceDatum>(refDatum);
343 additionalData_["isdaSubProduct"] = creditIndexRefDatum->indexFamily();
344 if (creditIndexRefDatum->indexFamily() == "") {
345 ALOG("IndexFamily is blank in credit index reference data for entity " << entity);
346 }
347 } else {
348 ALOG("Credit index reference data missing for entity " << entity << ", isdaSubProduct left blank");
349 }
350 // skip the transaction level mapping for now
351 additionalData_["isdaTransaction"] = string("");
352}

364void IndexCreditDefaultSwapOption::fromXML(XMLNode* node) {
365
366 Trade::fromXML(node);
367
368 XMLNode* iCdsOptionData = XMLUtils::getChildNode(node, "IndexCreditDefaultSwapOptionData");
369 QL_REQUIRE(iCdsOptionData, "Expected IndexCreditDefaultSwapOptionData node on trade " << id() << ".");
370 strike_ = XMLUtils::getChildValueAsDouble(iCdsOptionData, "Strike", false, Null<Real>());
371 knockOut_ = XMLUtils::getChildValueAsBool(iCdsOptionData, "KnockOut", false, false);
372 indexTerm_ = XMLUtils::getChildValue(iCdsOptionData, "IndexTerm", false);
373 strikeType_ = XMLUtils::getChildValue(iCdsOptionData, "StrikeType", false);
374 tradeDate_ = Date();
375 if (auto n = XMLUtils::getChildNode(iCdsOptionData, "TradeDate")) {
376 tradeDate_ = parseDate(XMLUtils::getNodeValue(n));
377 }
378 fepStartDate_ = Date();
379 if (auto n = XMLUtils::getChildNode(iCdsOptionData, "FrontEndProtectionStartDate")) {
380 fepStartDate_ = parseDate(XMLUtils::getNodeValue(n));
381 }
382
383 XMLNode* iCdsData = XMLUtils::getChildNode(iCdsOptionData, "IndexCreditDefaultSwapData");
384 QL_REQUIRE(iCdsData, "Expected IndexCreditDefaultSwapData node on trade " << id() << ".");
385 swap_.fromXML(iCdsData);
386
387 XMLNode* optionData = XMLUtils::getChildNode(iCdsOptionData, "OptionData");
388 QL_REQUIRE(optionData, "Expected OptionData node on trade " << id() << ".");
389 option_.fromXML(optionData);
390}

414 // Add IndexCreditDefaultSwapOptionData node to Trade node

417 return node;
418}

436void IndexCreditDefaultSwapOption::fromBasket(const Date& asof, map<string, Real>& outConstituents) {
437
438 const auto& constituents = swap_.basket().constituents();
439 DLOG("Building constituents from basket data containing " << constituents.size() << " elements.");
440
441 Real totalNtl = 0.0;
442 Real fullNtl = notionals_.full;
443 for (const auto& c : constituents) {
444 Real ntl = Null<Real>();
445 Real priorNotional = Null<Real>();
446
447 const auto& creditCurve = c.creditCurveId();
448
449 if (c.weightInsteadOfNotional()) {
450 ntl = c.weight() * fullNtl;
451 } else {
452 ntl = c.notional();
453 }
454
455 if (ntl == 0.0 || close(0.0, ntl)) {
456 if (c.weightInsteadOfNotional()) {
457 priorNotional = c.priorWeight();
458 if (priorNotional != Null<Real>()) {
459 priorNotional *= fullNtl;
460 }
461 } else {
462 priorNotional = c.priorNotional();
463 }
464 // Entity is not in the index. Its auction date is in the past.
465 QL_REQUIRE(priorNotional != Null<Real>(), "Constituent "
466 << creditCurve << " in index CDS option trade " << id()
467 << " has defaulted so expecting a prior notional.");
468 QL_REQUIRE(c.recovery() != Null<Real>(), "Constituent " << creditCurve << " in index CDS option trade "
469 << id()
470 << " has defaulted so expecting a recovery.");
471 QL_REQUIRE(c.auctionDate() != Date(), "Constituent " << creditCurve << " in index CDS option trade " << id()
472 << " has defaulted so expecting an auction date.");
473 QL_REQUIRE(c.auctionDate() <= asof, "Constituent " << creditCurve << " in index CDS option trade " << id()
474 << " has defaulted so expecting the auction date ("
475 << io::iso_date(c.auctionDate())
476 << ") to be before or on the valuation date ("
477 << io::iso_date(asof) << ".");
478
479 totalNtl += priorNotional * notional_;
480
481 if (tradeDate_ < c.auctionDate()) {
482 TLOG("Trade date (" << io::iso_date(tradeDate_) << ") is before auction date ("
483 << io::iso_date(c.auctionDate()) << ") of " << creditCurve
484 << " so updating trade date "
485 << "notional by amount " << priorNotional);
486 notionals_.tradeDate += priorNotional;
487 }
488
489 if (fepStartDate_ < c.auctionDate()) {
490 Real recovery = swap_.recoveryRate() != Null<Real>() ? swap_.recoveryRate() : c.recovery();
491 Real fepAmount = (1 - recovery) * priorNotional;
492 TLOG("FEP start date (" << io::iso_date(fepStartDate_) << ") is before auction date ("
493 << io::iso_date(c.auctionDate()) << ") of " << creditCurve
494 << " so updating realised "
495 << "FEP by amount " << fepAmount);
496 notionals_.realisedFep += fepAmount;
497 }
498
499 } else if (ntl > 0.0) {
500
501 // Entity is still in the index.
502 // Note that it may have defaulted but its auction date is still in the future.
503 auto it = outConstituents.find(creditCurve);
504 if (it == outConstituents.end()) {
505 outConstituents[creditCurve] = ntl;
506 TLOG("Adding underlying " << creditCurve << " with notional " << ntl);
507 totalNtl += ntl;
508 notionals_.tradeDate += ntl;
509 notionals_.valuationDate += ntl;
510 } else {
511 ALOG(StructuredTradeErrorMessage(id(), "IndexCDSOption", "Error building trade",
512 ("Invalid Basket: found a duplicate credit curve " + creditCurve +
513 ".Skip it. Check the basket data for possible errors.")
514 .c_str()));
515 }
516 } else {
517 QL_FAIL("Constituent " << creditCurve << " in index CDS option trade " << id()
518 << " has a negative notional " << ntl << ".");
519 }
520 }
521
522 Real correctionFactor = fullNtl / totalNtl;
523 // Scaling to Notional if relative error is close less than 10^-4
524 if (!close(fullNtl, totalNtl) && (std::abs(correctionFactor - 1.0) <= 1e-4)) {
525 DLOG("Trade " << id() << ", sum of notionals (" << totalNtl << ") is very close to " << fullNtl
526 << ",will scale it by " << correctionFactor << ". Check the basket data for possible errors.");
527
528 for (auto& name_notional : outConstituents) {
529 TLOG("Trade " << id() << ", Issuer" << name_notional.first << " unscaled Notional: " << name_notional.second
530 << ", scaled Notional: " << name_notional.second * correctionFactor);
531 name_notional.second *= correctionFactor;
532 }
533
534 totalNtl *= correctionFactor;
535 notionals_.tradeDate *= correctionFactor;
536 notionals_.valuationDate *= correctionFactor;
537 notionals_.realisedFep *= correctionFactor;
538 }
539
540 DLOG("All underlyings added, total notional = " << totalNtl);
541 if (!close(fullNtl, totalNtl) && totalNtl > fullNtl) {
542 ALOG(StructuredTradeErrorMessage(id(), "IndexCDSOption", "Error building trade",
543 ("Sum of basket notionals (" + std::to_string(totalNtl) +
544 ") is greater than trade notional (" + std::to_string(fullNtl) +
545 "). Check the basket data for possible errors.")
546 .c_str()));
547 }
548
549 DLOG("Finished building constituents using basket data.");
550}

552void IndexCreditDefaultSwapOption::fromReferenceData(const Date& asof, map<string, Real>& outConstituents,
553 const boost::shared_ptr<ReferenceDataManager>& refData) {
554
555 const string& iCdsId = swap_.creditCurveId();
556 DLOG("Start building constituents using credit reference data for " << iCdsId << ".");
557
558 QL_REQUIRE(refData, "Building index CDS option " << id() << " ReferenceDataManager is null.");
559 QL_REQUIRE(refData->hasData(CreditIndexReferenceDatum::TYPE, iCdsId),
560 "No CreditIndex reference data for " << iCdsId);
561 auto referenceData = boost::dynamic_pointer_cast<CreditIndexReferenceDatum>(
562 refData->getData(CreditIndexReferenceDatum::TYPE, iCdsId));
563 DLOG("Got CreditIndexReferenceDatum for id " << iCdsId);
564
565 Real fullNtl = notionals_.full;
566 Real totalWeight = 0.0;
567 for (const auto& c : referenceData->constituents()) {
568
569 const auto& name = c.name();
570 auto weight = c.weight();
571
572 if (weight == 0.0 || close(0.0, weight)) {
573
574 // Entity is not in the index. Its auction date is in the past.
575 QL_REQUIRE(c.priorWeight() != Null<Real>(), "Constituent "
576 << name << " in index CDS option trade " << id()
577 << " has defaulted so expecting a prior weight.");
578 QL_REQUIRE(c.recovery() != Null<Real>(), "Constituent " << name << " in index CDS option trade " << id()
579 << " has defaulted so expecting a recovery.");
580 QL_REQUIRE(c.auctionDate() != Date(), "Constituent " << name << " in index CDS option trade " << id()
581 << " has defaulted so expecting an auction date.");
582 QL_REQUIRE(c.auctionDate() <= asof, "Constituent " << name << " in index CDS option trade " << id()
583 << " has defaulted so expecting the auction date ("
584 << io::iso_date(c.auctionDate())
585 << ") to be before or on the valuation date ("
586 << io::iso_date(asof) << ").");
587
588 totalWeight += c.priorWeight();
589
590 if (tradeDate_ < c.auctionDate()) {
591 Real entityNtl = c.priorWeight() * fullNtl;
592 TLOG("Trade date (" << io::iso_date(tradeDate_) << ") is before auction date ("
593 << io::iso_date(c.auctionDate()) << ") of " << name << " so updating trade date "
594 << "notional by amount " << entityNtl);
595 notionals_.tradeDate += entityNtl;
596 }
597
598 if (fepStartDate_ < c.auctionDate()) {
599 Real recovery = swap_.recoveryRate() != Null<Real>() ? swap_.recoveryRate() : c.recovery();
600 Real fepAmount = (1 - recovery) * c.priorWeight() * fullNtl;
601 TLOG("FEP start date (" << io::iso_date(fepStartDate_) << ") is before auction date ("
602 << io::iso_date(c.auctionDate()) << ") of " << name << " so updating realised "
603 << "FEP by amount " << fepAmount);
604 notionals_.realisedFep += fepAmount;
605 }
606
607 } else if (weight > 0.0) {
608
609 // Entity is still in the index.
610 // Note that it may have defaulted but its auction date is still in the future.
611 Real entityNtl = weight * fullNtl;
612 auto it = outConstituents.find(name);
613 if (it == outConstituents.end()) {
614 outConstituents[name] = entityNtl;
615 TLOG("Adding underlying " << name << " with weight " << weight << " (notional = " << entityNtl << ")");
616 } else {
617 it->second += entityNtl;
618 TLOG("Updating underlying " << name << " with weight " << weight << " (notional = " << entityNtl
619 << ")");
620 }
621
622 totalWeight += weight;
623 notionals_.tradeDate += entityNtl;
624 notionals_.valuationDate += entityNtl;
625
626 } else {
627 QL_FAIL("Constituent " << name << " in index CDS option trade " << id() << " has a negative weight "
628 << weight << ".");
629 }
630 }
631
632 DLOG("All underlyings added, total weight = " << totalWeight);
633 if (!close(1.0, totalWeight) && totalWeight > 1.0) {
634 ALOG("Total weight is greater than 1, possible error in CreditIndexReferenceDatum for "
635 << iCdsId << " while building constituents for trade " << id() << ".");
636 }
637
638 DLOG("Finished building constituents using credit reference data.");
639}
