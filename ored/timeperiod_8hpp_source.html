60inline TimePeriod::TimePeriod(const std::vector<Date>& dates) {
61 QL_REQUIRE(dates.size() % 2 == 0, "TimePeriod: dates size must be an even number, got " << dates.size());
62 for (Size i = 0; i < dates.size(); ++i) {
63 if (i % 2 == 0)
64 startDates_.push_back(dates[i]);
65 else
66 endDates_.push_back(dates[i]);
67 }
68}
69
70inline bool TimePeriod::contains(const Date& d) const {
71 for (Size i = 0; i < startDates_.size(); ++i) {
72 if (d >= startDates_[i] && d <= endDates_[i])
73 return true;
74 }
75 return false;
76}
77
78inline std::ostream& operator<<(std::ostream& out, const TimePeriod& t) {
79 for (Size i = 0; i < t.numberOfContiguousParts(); ++i) {
80 out << QuantLib::io::iso_date(t.startDates()[i]) << " to " << QuantLib::io::iso_date(t.endDates()[i]);
81 if (i < t.numberOfContiguousParts() - 1)
82 out << " + ";
83 }
84 return out;
85}
