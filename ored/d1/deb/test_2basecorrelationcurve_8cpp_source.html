79 BOOST_DATA_TEST_CASE(testBaseCorrelationStructureBuilding, bdata::make(setups), setup) {
80 
81  BOOST_TEST_MESSAGE("Testing base correlation structure building using setup in " << setup);
82 
83  Date asof(19, Oct, 2020);
84  Settings::instance().evaluationDate() = asof;
85 
86  auto todaysMarket = createTodaysMarket(asof, setup);
87 
88  // Get the built base correlation structure.
89  auto bc = todaysMarket->baseCorrelation("BASE_CORR_TEST");
90 
91  // These are the values used in the test configurations.
92  Calendar calendar = parseCalendar("US settlement");
93  BusinessDayConvention bdc = Following;
94 
95  // Tolerance for comparison.
96  Real tol = 1e-12;
97 
98  // Read in the expected results.
99  string filename = setup + "/expected.csv";
100  CSVFileReader reader(TEST_INPUT_FILE(filename), true, ",");
101  BOOST_REQUIRE_EQUAL(reader.numberOfColumns(), 3);
102 
103  BOOST_TEST_MESSAGE("term,detachment,expected_bc,calculated_bc,difference");
104  while (reader.next()) {
105 
106  // Get the term, detachment point and expected base correlation
107  Period term = parsePeriod(reader.get(0));
108  Real dp = parseReal(reader.get(1));
109  Real expBc = parseReal(reader.get(2));
110 
111  // Check
112  Date d = calendar.advance(asof, term, bdc);
113  Real calcBc = bc->correlation(d, dp);
114  Real difference = expBc - calcBc;
115  BOOST_TEST_MESSAGE(term << "," << fixed << setprecision(12) << dp << "," <<
116  expBc << "," << calcBc << "," << difference);
117  BOOST_CHECK_SMALL(difference, tol);
118  }
119 }
