31 // Local class for printing each report type via fprintf
32 class ReportTypePrinter : public boost::static_visitor<> {
33 public:
34  ReportTypePrinter(FILE* fp, int prec, char quoteChar = '\0', const string& nullString = "#N/A")
35  : fp_(fp), rounding_(prec, QuantLib::Rounding::Closest), quoteChar_(quoteChar), null_(nullString) {}
36 
37  void operator()(const Size i) const {
38  if (i == QuantLib::Null<Size>()) {
39  fprintNull();
40  } else {
41  fprintf(fp_, "%zu", i);
42  }
43  }
44  void operator()(const Real d) const {
45  if (d == QuantLib::Null<Real>() || !std::isfinite(d)) {
46  fprintNull();
47  } else {
48  Real r = rounding_(d);
49  fprintf(fp_, "%.*f", rounding_.precision(), QuantLib::close_enough(r, 0.0) ? 0.0 : r);
50  }
51  }
52  void operator()(const string& s) const { fprintString(s); }
53  void operator()(const Date& d) const {
54  if (d == QuantLib::Null<Date>()) {
55  fprintNull();
56  } else {
57  string s = to_string(d);
58  fprintString(s);
59  }
60  }
61  void operator()(const Period& p) const {
62  string s = to_string(p);
63  fprintString(s);
64  }
65 
66 private:
67  void fprintNull() const { fprintf(fp_, "%s", null_.c_str()); }
68 
69  // Shared implementation to include the quote character.
70  void fprintString(const string& s) const {
71  if (quoteChar_ != '\0')
72  fputc(quoteChar_, fp_);
73  fprintf(fp_, "%s", s.c_str());
74  if (quoteChar_ != '\0')
75  fputc(quoteChar_, fp_);
76  }
77 
78  FILE* fp_;
79  QuantLib::Rounding rounding_;
80  char quoteChar_;
81  string null_;
82 };

96 Report& CSVFileReport::addColumn(const string& name, const ReportType& rt, Size precision) {
97  columnTypes_.push_back(rt);
98  printers_.push_back(ReportTypePrinter(fp_, precision, quoteChar_, nullString_));
99  if (i_ == 0 && commentCharacter_)
100  fprintf(fp_, "#");
101  if (i_ > 0)
102  fprintf(fp_, "%c", sep_);
103  string cpName = name;
104  if (lowerHeader_ && !cpName.empty())
105  cpName[0] = std::tolower(static_cast<unsigned char>(cpName[0]));
106  fprintf(fp_, "%s", cpName.c_str());
107  i_++;
108  return *this;
109 }

118 Report& CSVFileReport::add(const ReportType& rt) {
119  QL_REQUIRE(i_ < columnTypes_.size(), "No column to add [" << rt << "] to.");
120  QL_REQUIRE(rt.which() == columnTypes_[i_].which(), "Cannot add value " << rt << " of type " << rt.which()

122  << columnTypes_[i_].which());
123 
124  if (i_ != 0)
125  fprintf(fp_, "%c", sep_);
126  boost::apply_visitor(printers_[i_], rt);
127  i_++;
128  return *this;
