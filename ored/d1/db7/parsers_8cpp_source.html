76 Date parseDate(const string& s) {
77  // TODO: review
78 
79  if (s == "")
80  return Date();
81 
82  // guess formats from token number and sizes
83  // check permissible lengths
84  QL_REQUIRE((s.size() >= 3 && s.size() <= 6) || s.size() == 8 || s.size() == 10,
85  "invalid date format of \"" << s << "\", date string length 8 or 10 or between 3 and 6 required");
86 
87  vector<string> tokens;
88  boost::split(tokens, s, boost::is_any_of("-/.:"));
89 
90  if (tokens.size() == 1) {
91  if (s.size() == 8) {
92  // yyyymmdd
93  int y = parseInteger(s.substr(0, 4));
94  int m = parseInteger(s.substr(4, 2));
95  int d = parseInteger(s.substr(6, 2));
96  return Date(d, Month(m), y);
97  } else if (s.size() >= 3 && s.size() <= 6) {
98  // Excel format
99  // Boundaries will be checked by Date constructor
100  // Boundaries are minDate = 367 i.e. Jan 1st, 1901
101  // and maxDate = 109574 i.e. Dec 31st, 2199
102  BigInteger serial = parseInteger(s);
103  return Date(serial);
104  }
105  } else if (tokens.size() == 3) {
106  if (tokens[0].size() == 4) {
107  // yyyy-mm-dd
108  // yyyy/mm/dd
109  // yyyy.mm.dd
110  int y = parseInteger(tokens[0]);
111  int m = parseInteger(tokens[1]);
112  int d = parseInteger(tokens[2]);
113  return Date(d, Month(m), y);
114  } else if (tokens[0].size() == 2) {
115  // dd-mm-yy
116  // dd/mm/yy
117  // dd.mm.yy
118  // dd-mm-yyyy
119  // dd/mm/yyyy
120  // dd.mm.yyyy
121  int d = parseInteger(tokens[0]);
122  int m = parseInteger(tokens[1]);
123  int y = parseInteger(tokens[2]);
124  if (y < 100) {
125  if (y > 80)
126  y += 1900;
127  else
128  y += 2000;
129  }
130  return Date(d, Month(m), y);
131  }
132  }
133 
134  QL_FAIL("Cannot convert \"" << s << "\" to Date.");
135 }
