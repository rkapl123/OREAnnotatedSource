60 std::string getParameter(const std::map<std::string, std::string>& m, const std::string& p, const std::string& q,
61  const bool mandatory, const std::string& defaultValue) {
62  // first look for p_q if a qualifier is given
63  if (!q.empty()) {
64  auto r = m.find(p + "_" + q);
65  if (r != m.end())
66  return r->second;
67  }
68  // no qualifier given, or fall back on p because p_q was not found
69  auto r = m.find(p);
70  if (r != m.end()) {
71  return r->second;
72  }
73  // if parameter is mandatory throw, otherwise return the default value
74  if (mandatory) {
75  QL_FAIL("parameter " << p << " not found (qualifier was \"" << q << "\")");
76  }
77  return defaultValue;
78 }

112 boost::shared_ptr<EngineBuilder> EngineFactory::builder(const string& tradeType) {
113  // Check that we have a model/engine for tradetype
114  QL_REQUIRE(engineData_->hasProduct(tradeType),
115  "No Pricing Engine configuration was provided for trade type " << tradeType);
116 
117  // Find a builder for the model/engine/tradeType
118  const string& model = engineData_->model(tradeType);
119  const string& engine = engineData_->engine(tradeType);
120  typedef pair<tuple<string, string, set<string>>, boost::shared_ptr<EngineBuilder>> map_type;
121  auto it =
122  std::find_if(builders_.begin(), builders_.end(), [&model, &engine, &tradeType](const map_type& v) -> bool {
123  const set<string>& types = std::get<2>(v.first);
124  return std::get<0>(v.first) == model && std::get<1>(v.first) == engine &&
125  std::find(types.begin(), types.end(), tradeType) != types.end();
126  });
127  QL_REQUIRE(it != builders_.end(), "No EngineBuilder for " << model << "/" << engine << "/" << tradeType);
128 
129  boost::shared_ptr<EngineBuilder> builder = it->second;
130 
131  builder->init(market_, configurations_, engineData_->modelParameters(tradeType),
132  engineData_->engineParameters(tradeType), engineData_->globalParameters());
133 
134  return builder;
135 }

240 void EngineFactory::addExtraBuilders(const std::vector<boost::shared_ptr<EngineBuilder>> extraEngineBuilders,
241  const std::vector<boost::shared_ptr<LegBuilder>> extraLegBuilders) {
242 
243  if (extraEngineBuilders.size() > 0) {
244  DLOG("adding " << extraEngineBuilders.size() << " extra engine builders");
245  for (auto eb : extraEngineBuilders)
246  registerBuilder(eb);
247  }
248  if (extraLegBuilders.size() > 0) {
249  DLOG("adding " << extraLegBuilders.size() << " extra leg builders");
250  for (auto elb : extraLegBuilders)
251  registerLegBuilder(elb);
252  }
