31  void CommodityAsianOption::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
32  // Checks
33  QL_REQUIRE(quantity_ > 0, "Commodity Asian option requires a positive quatity");
34  QL_REQUIRE(strike_ >= 0, "Commodity Asian option requires a strike >= 0");
35 
36  // Get the price curve for the commodity.
37  const boost::shared_ptr<Market>& market = engineFactory->market();
38  Handle<QuantExt::PriceTermStructure> priceCurve =
39  market->commodityPriceCurve(assetName_, engineFactory->configuration(MarketContext::pricing));
40 
41  // Populate the index_ in case the option is automatic exercise.
42  // Intentionally use null calendar because we will ask for index value on the expiry date without adjustment.
43  if (!isFuturePrice_ || *isFuturePrice_) {
44  // Assume future price if isFuturePrice_ is not explicitly set or if it is and true.
45 
46  // If we are given an explicit future contract expiry date, use it, otherwise use option's expiry.
47  Date expiryDate;
48  if (futureExpiryDate_ != Date()) {
49  expiryDate = futureExpiryDate_;
50  } else {
51  // Get the expiry date of the option. This is the expiry date of the commodity future index.
52  const vector<string>& expiryDates = option_.exerciseDates();
53  QL_REQUIRE(expiryDates.size() == 1, "Expected exactly one expiry date for CommodityAsianOption but got "
54  << expiryDates.size() << ".");
55  expiryDate = parseDate(expiryDates[0]);
56  }
57 
58  index_ = boost::make_shared<QuantExt::CommodityFuturesIndex>(assetName_, expiryDate, NullCalendar(), priceCurve);
59  } else {
60  // If the underlying is a commodity spot, create a spot index.
61  index_ = boost::make_shared<QuantExt::CommoditySpotIndex>(assetName_, NullCalendar(), priceCurve);
62  }
63 
64  AsianOptionTrade::build(engineFactory);
65  }

71  void CommodityAsianOption::fromXML(XMLNode* node) {
72  Trade::fromXML(node);
73 
74  XMLNode* commodityNode = XMLUtils::getChildNode(node, "CommodityAsianOptionData");
75  QL_REQUIRE(commodityNode, "A commodity Asian option needs a 'CommodityAsianOptionData' node");
76 
77  option_.fromXML(XMLUtils::getChildNode(commodityNode, "OptionData"));
78  QL_REQUIRE(option_.payoffType() == "Asian", "Expected PayoffType Asian for CommodityAsianOption.");
79  XMLNode* asianNode = XMLUtils::getChildNode(commodityNode, "AsianData");
80  if (asianNode)
81  asianData_.fromXML(asianNode);
82 
83  XMLNode* scheduleDataNode = XMLUtils::getChildNode(commodityNode, "ObservationDates");
84  observationDates_.fromXML(scheduleDataNode);
85 
86  XMLNode* tmp = XMLUtils::getChildNode(commodityNode, "Underlying");
87  if (!tmp)
88  tmp = XMLUtils::getChildNode(commodityNode, "Name");
89  commodityUnderlying_.fromXML(tmp);
90  assetName_ = commodityUnderlying_.name();
91 
92  currency_ = XMLUtils::getChildValue(commodityNode, "Currency", true);
93  // Require explicit Strike
94  strike_ = XMLUtils::getChildValueAsDouble(commodityNode, "Strike", true);
95  quantity_ = XMLUtils::getChildValueAsDouble(commodityNode, "Quantity", true);
96 
97  isFuturePrice_ = boost::none;
98  if (XMLNode* n = XMLUtils::getChildNode(commodityNode, "IsFuturePrice"))
99  isFuturePrice_ = parseBool(XMLUtils::getNodeValue(n));
100 
101  futureExpiryDate_ = Date();
102  if (XMLNode* n = XMLUtils::getChildNode(commodityNode, "FutureExpiryDate"))
103  futureExpiryDate_ = parseDate(XMLUtils::getNodeValue(n));
104  }

116  XMLUtils::appendNode(comNode, tmp);

119  XMLUtils::addChild(doc, comNode, "Currency", currency_);
120  XMLUtils::addChild(doc, comNode, "Strike", strike_);
121  XMLUtils::addChild(doc, comNode, "Quantity", quantity_);
122 
123  if (isFuturePrice_)
124  XMLUtils::addChild(doc, comNode, "IsFuturePrice", *isFuturePrice_);
125 
126  if (futureExpiryDate_ != Date())
127  XMLUtils::addChild(doc, comNode, "FutureExpiryDate", to_string(futureExpiryDate_));
128 
129  return node;
130  }
