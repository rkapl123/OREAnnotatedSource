43 InflationCapFloorPriceSurface::InflationCapFloorPriceSurface(
44  Date asof, InflationCapFloorPriceSurfaceSpec spec, const Loader& loader, const CurveConfigurations& curveConfigs,
45  map<string, boost::shared_ptr<YieldCurve>>& yieldCurves,
46  map<string, boost::shared_ptr<InflationCurve>>& inflationCurves) {
47 
48  try {
49 
50  const boost::shared_ptr<InflationCapFloorPriceSurfaceConfig>& config =
51  curveConfigs.inflationCapFloorPriceSurfaceConfig(spec.curveConfigID());
52 
53  QL_REQUIRE((config->type() == InflationCapFloorPriceSurfaceConfig::Type::ZC) ||
54  (config->type() == InflationCapFloorPriceSurfaceConfig::Type::YY),
55  "Inflation cap floor price surfaces must be of type 'ZC' or 'YY'");
56 
57  Handle<YieldTermStructure> yts;
58  auto it = yieldCurves.find(config->yieldTermStructure());
59  if (it != yieldCurves.end()) {
60  yts = it->second->handle();
61  } else {
62  QL_FAIL("The nominal term structure, " << config->yieldTermStructure()
63  << ", required in the building "
64  "of the curve, "
65  << spec.name() << ", was not found.");
66  }
67 
68  std::vector<Period> terms = parseVectorOfValues<Period>(config->maturities(), &parsePeriod);
69  std::vector<Real> capStrikes = parseVectorOfValues<Real>(config->capStrikes(), &parseReal);
70  std::vector<Real> floorStrikes = parseVectorOfValues<Real>(config->floorStrikes(), &parseReal);
71 
72  Matrix cPrice(capStrikes.size(), capStrikes.size() == 0 ? 0 : terms.size(), Null<Real>()),
73  fPrice(floorStrikes.size(), floorStrikes.size() == 0 ? 0 : terms.size(), Null<Real>());
74 
75  // We loop over all market data, looking for quotes that match the configuration
76  for (auto& md : loader.loadQuotes(asof)) {
77 
78  if (md->asofDate() == asof && (md->instrumentType() == MarketDatum::InstrumentType::ZC_INFLATIONCAPFLOOR ||
79  md->instrumentType() == MarketDatum::InstrumentType::YY_INFLATIONCAPFLOOR)) {
80 
81  boost::shared_ptr<InflationCapFloorQuote> q;
82 
83  if (config->type() == InflationCapFloorPriceSurfaceConfig::Type::ZC) {
84  q = boost::dynamic_pointer_cast<ZcInflationCapFloorQuote>(md);
85  } else {
86  q = boost::dynamic_pointer_cast<YyInflationCapFloorQuote>(md);
87  }
88 
89  if (q != NULL && q->index() == spec.index() && md->quoteType() == MarketDatum::QuoteType::PRICE) {
90  auto it1 = std::find(terms.begin(), terms.end(), q->term());
91  Real strike = parseReal(q->strike());
92  Size strikeIdx = Null<Size>();
93  if (q->isCap()) {
94  for (Size i = 0; i < capStrikes.size(); ++i) {
95  if (close_enough(capStrikes[i], strike))
96  strikeIdx = i;
97  }
98  } else {
99  for (Size i = 0; i < floorStrikes.size(); ++i) {
100  if (close_enough(floorStrikes[i], strike))
101  strikeIdx = i;
102  }
103  }
104  if (it1 != terms.end() && strikeIdx != Null<Size>()) {
105  if (q->isCap()) {
106  cPrice[strikeIdx][it1 - terms.begin()] = q->quote()->value();
107  } else {
108  fPrice[strikeIdx][it1 - terms.begin()] = q->quote()->value();
109  }
110  }
111  }
112  }
113  }
114 
115  for (Size j = 0; j < terms.size(); ++j) {
116  for (Size i = 0; i < capStrikes.size(); ++i)
117  QL_REQUIRE(cPrice[i][j] != Null<Real>(), "quote for cap floor price surface, type cap, strike "
118  << capStrikes[i] << ", term " << terms[j]
119  << ", not found.");
120  for (Size i = 0; i < floorStrikes.size(); ++i)
121  QL_REQUIRE(fPrice[i][j] != Null<Real>(), "quote for cap floor price surface, type floor, strike "
122  << floorStrikes[i] << ", term " << terms[j]
123  << ", not found.");
124  }
125 
126  // The strike grids have some minimum requirements which we fulfill here at
127  // least technically; note that the extrapolated prices will not be sensible,
128  // instead only the given strikes for the given option type may be sensible
129  // in the end
130 
131  static const Real largeStrike = 1.0;
132  static const Real largeStrikeFactor = 0.99;
133 
134  Size addFloor = 0, addCap = 0;
135  if (floorStrikes.size() == 0) {
136  floorStrikes.push_back(-largeStrike);
137  floorStrikes.push_back(-(largeStrike * largeStrikeFactor));
138  addFloor = 2;
139  }
140  if (floorStrikes.size() == 1) {
141  floorStrikes.insert(floorStrikes.begin(), -largeStrike);
142  addFloor = 1;
143  }
144  if (capStrikes.size() == 0) {
145  capStrikes.push_back(largeStrike * largeStrikeFactor);
146  capStrikes.push_back(largeStrike);
147  addCap = 2;
148  }
149  if (capStrikes.size() == 1) {
150  capStrikes.push_back(largeStrike);
151  addCap = 1;
152  }
153  if (addFloor > 0) {
154  Matrix tmp(fPrice.rows() + addFloor, terms.size(), 1E-10);
155  for (Size i = addFloor; i < fPrice.rows() + addFloor; ++i) {
156  for (Size j = 0; j < fPrice.columns(); ++j) {
157  tmp[i][j] = fPrice[i - addFloor][j];
158  }
159  }
160  fPrice = tmp;
161  }
162  if (addCap > 0) {
163  Matrix tmp(cPrice.rows() + addCap, terms.size(), 1E-10);
164  for (Size i = 0; i < cPrice.rows(); ++i) {
165  for (Size j = 0; j < cPrice.columns(); ++j) {
166  tmp[i][j] = cPrice[i][j];
167  }
168  }
169  cPrice = tmp;
170  }
171 
172  ostringstream capStrikesString, floorStrikesString;
173  for (Size i = 0; i < floorStrikes.size(); ++i)
174  floorStrikesString << floorStrikes[i] << ",";
175  for (Size i = 0; i < capStrikes.size(); ++i)
176  capStrikesString << capStrikes[i] << ",";
177  DLOG("Building inflation cap floor price surface:");
178  DLOG("Cap Strikes are: " << capStrikesString.str());
179  DLOG("Floor Strikes are: " << floorStrikesString.str());
180  DLOGGERSTREAM << "Cap Price Matrix:\n" << cPrice << "Floor Price Matrix:\n" << fPrice;
181 
182  if (config->type() == InflationCapFloorPriceSurfaceConfig::Type::ZC) {
183  // ZC Curve
184 
185  boost::shared_ptr<ZeroInflationIndex> index;
186  auto it2 = inflationCurves.find(config->indexCurve());
187  if (it2 != inflationCurves.end()) {
188  boost::shared_ptr<ZeroInflationTermStructure> ts =
189  boost::dynamic_pointer_cast<ZeroInflationTermStructure>(it2->second->inflationTermStructure());
190  QL_REQUIRE(ts, "inflation term structure " << config->indexCurve()
191  << " was expected to be zero, but is not");
192  index = parseZeroInflationIndex(config->index(), it2->second->interpolatedIndex(),
193  Handle<ZeroInflationTermStructure>(ts));
194  } else {
195  QL_FAIL("The zero inflation curve, " << config->indexCurve()
196  << ", required in building the inflation cap floor price surface "
197  << spec.name() << ", was not found");
198  }
199  // Build the term structure
200  surface_ = boost::shared_ptr<InterpolatedCPICapFloorTermPriceSurface<QuantLib::Bilinear>>(
201  new InterpolatedCPICapFloorTermPriceSurface<QuantLib::Bilinear>(
202  1.0, config->startRate(), config->observationLag(), config->calendar(),
203  config->businessDayConvention(), config->dayCounter(), Handle<ZeroInflationIndex>(index), yts,
204  capStrikes, floorStrikes, terms, cPrice, fPrice));
205 
206  boost::shared_ptr<InterpolatedCPICapFloorTermPriceSurface<QuantLib::Bilinear>> cpiPriceSurfacePtr =
207  boost::make_shared<InterpolatedCPICapFloorTermPriceSurface<QuantLib::Bilinear>>(
208  1.0, config->startRate(), config->observationLag(), config->calendar(),
209  config->businessDayConvention(), config->dayCounter(), Handle<ZeroInflationIndex>(index), yts,
210  capStrikes, floorStrikes, terms, cPrice, fPrice);
211 
212  // cast
213  surface_ = cpiPriceSurfacePtr;
214 
215  Handle<YieldTermStructure> nominalTS = index->zeroInflationTermStructure()->nominalTermStructure();
216  boost::shared_ptr<QuantExt::CPIBlackCapFloorEngine> engine =
217  boost::make_shared<QuantExt::CPIBlackCapFloorEngine>(
218  nominalTS, QuantLib::Handle<QuantLib::CPIVolatilitySurface>()); // vol surface can be empty, will be
219  // set in the striping process
220 
221  try {
222 
223  QuantLib::Handle<CPICapFloorTermPriceSurface> cpiPriceSurfaceHandle(cpiPriceSurfacePtr);
224  boost::shared_ptr<QuantExt::StrippedCPIVolatilitySurface<QuantLib::Bilinear>> cpiCapFloorVolSurface;
225  cpiCapFloorVolSurface = boost::make_shared<QuantExt::StrippedCPIVolatilitySurface<QuantLib::Bilinear>>(
226  QuantExt::PriceQuotePreference::CapFloor, cpiPriceSurfaceHandle, index, engine);
227 
228  // cast
229  cpiCapFloorVolSurface_ = cpiCapFloorVolSurface;
230 
231  cpiCapFloorVolSurface_->enableExtrapolation();
232 
233  for (Size i = 0; i < cpiCapFloorVolSurface->strikes().size(); ++i) {
234  for (Size j = 0; j < cpiCapFloorVolSurface->maturities().size(); ++j) {
235  DLOG("Implied CPI CapFloor BlackVol,strike,"
236  << cpiCapFloorVolSurface->strikes()[i] << ",maturity,"
237  << cpiCapFloorVolSurface->maturities()[j] << ",index," << index->name() << ",Vol,"
238  << cpiCapFloorVolSurface->volData()[i][j]);
239  }
240  }
241 
242  DLOG("CPIVolSurfaces built for spec " << spec.name());
243  } catch (std::exception& e) {
244  ALOG("Building CPIVolSurfaces failed for spec " << spec.name() << ": " << e.what());
245  }
246  }
247  if (config->type() == InflationCapFloorPriceSurfaceConfig::Type::YY) {
248 
249  boost::shared_ptr<YoYInflationIndex> index;
250  auto it2 = inflationCurves.find(config->indexCurve());
251  if (it2 != inflationCurves.end()) {
252  boost::shared_ptr<InflationTermStructure> ts = it2->second->inflationTermStructure();
253  // Check if the Index curve is a YoY curve - if not it must be a zero curve
254  boost::shared_ptr<YoYInflationTermStructure> yyTs =
255  boost::dynamic_pointer_cast<YoYInflationTermStructure>(ts);
256 
257  if (yyTs) {
258  useMarketYoyCurve_ = true;
259  index = boost::make_shared<QuantExt::YoYInflationIndexWrapper>(
260  parseZeroInflationIndex(config->index(), true), true, Handle<YoYInflationTermStructure>(yyTs));
261  } else {
262  useMarketYoyCurve_ = false;
263  boost::shared_ptr<ZeroInflationTermStructure> zeroTs =
264  boost::dynamic_pointer_cast<ZeroInflationTermStructure>(ts);
265  QL_REQUIRE(zeroTs,
266  "Inflation term structure " << config->indexCurve() << "must be of type YoY or Zero");
267  index = boost::make_shared<QuantExt::YoYInflationIndexWrapper>(
268  parseZeroInflationIndex(config->index(), true, Handle<ZeroInflationTermStructure>(zeroTs)),
269  true, Handle<YoYInflationTermStructure>());
270  }
271  } else {
272  QL_FAIL("The inflation curve, " << config->indexCurve()
273  << ", required in building the inflation cap floor price surface "
274  << spec.name() << ", was not found");
275  }
276 
277  // Build the term structure
278  boost::shared_ptr<QuantExt::InterpolatedYoYCapFloorTermPriceSurface<QuantLib::Bilinear, QuantLib::Linear>>
279  yoySurface = boost::make_shared<
280  QuantExt::InterpolatedYoYCapFloorTermPriceSurface<QuantLib::Bilinear, QuantLib::Linear>>(
281  0, config->observationLag(), index, config->startRate(), yts, config->dayCounter(),
282  config->calendar(), config->businessDayConvention(), capStrikes, floorStrikes, terms, cPrice,
283  fPrice);
284 
285  std::vector<Period> optionletTerms = {yoySurface->maturities().front()};
286  while (optionletTerms.back() != terms.back()) {
287  optionletTerms.push_back(optionletTerms.back() + Period(1, Years));
288  }
289  yoySurface->setMaturities(optionletTerms);
290  surface_ = yoySurface;
291 
292  boost::shared_ptr<InterpolatedYoYOptionletStripper<QuantLib::Linear>> yoyStripper =
293  boost::make_shared<InterpolatedYoYOptionletStripper<QuantLib::Linear>>();
294 
295  // Create an empty volatlity surface to pass to the engine
296  boost::shared_ptr<QuantLib::YoYOptionletVolatilitySurface> ovs =
297  boost::dynamic_pointer_cast<QuantLib::YoYOptionletVolatilitySurface>(
298  boost::make_shared<QuantLib::ConstantYoYOptionletVolatility>(
299  0.0, yoySurface->settlementDays(), yoySurface->calendar(), yoySurface->businessDayConvention(),
300  yoySurface->dayCounter(), yoySurface->observationLag(), yoySurface->frequency(),
301  yoySurface->indexIsInterpolated()));
302  Handle<QuantLib::YoYOptionletVolatilitySurface> hovs(ovs);
303 
304  // create a yoy Index from the surfaces termstructure
305  yoyTs_ = yoySurface->YoYTS();
306  boost::shared_ptr<YoYInflationIndex> yoyIndex = index->clone(Handle<YoYInflationTermStructure>(yoyTs_));
307 
308  boost::shared_ptr<YoYInflationBachelierCapFloorEngine> cfEngine =
309  boost::make_shared<YoYInflationBachelierCapFloorEngine>(yoyIndex, hovs);
310 
311  boost::shared_ptr<QuantExt::StrikeInterpolatedYoYOptionletVolatilitySurface<Linear>> interpVolSurface =
312  boost::make_shared<QuantExt::StrikeInterpolatedYoYOptionletVolatilitySurface<Linear>>(
313  yoySurface->settlementDays(), yoySurface->calendar(), yoySurface->businessDayConvention(),
314  yoySurface->dayCounter(), yoySurface->observationLag(), yoySurface, cfEngine, yoyStripper, 0);
315 
316  boost::shared_ptr<QuantExt::YoYOptionletVolatilitySurface> newSurface =
317  boost::make_shared<QuantExt::YoYOptionletVolatilitySurface>(interpVolSurface, VolatilityType::Normal);
318  yoyVolSurface_ = newSurface;
319  }
320 
321  } catch (std::exception& e) {
322  QL_FAIL("inflation cap floor price surface building failed: " << e.what());
323  } catch (...) {
324  QL_FAIL("inflation cap floor price surface building failed: unknown error");
325  }
