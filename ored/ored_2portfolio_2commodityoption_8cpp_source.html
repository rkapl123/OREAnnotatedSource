51void CommodityOption::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
52
53 // Checks
54 QL_REQUIRE(quantity_ > 0, "Commodity option requires a positive quatity");
55 QL_REQUIRE((strike_.value() > 0) || close_enough(strike_.value(),0.0), "Commodity option requires a non-negative strike");
56 if (close_enough(strike_.value(), 0.0)) {
57 strike_.setValue(0.0);
58 }
59
60 // Populate the index_ in case the option is automatic exercise.
61 // Intentionally use null calendar because we will ask for index value on the expiry date without adjustment.
62 const boost::shared_ptr<Market>& market = engineFactory->market();
63 index_ = *market->commodityIndex(assetName_, engineFactory->configuration(MarketContext::pricing));
64 if (!isFuturePrice_ || *isFuturePrice_) {
65
66 // Assume future price if isFuturePrice_ is not explicitly set or if it is and true.
67
68 auto index = *market->commodityIndex(assetName_, engineFactory->configuration(MarketContext::pricing));
69
70 // If we are given an explicit future contract expiry date, use it, otherwise use option's expiry.
71 Date expiryDate;
72 if (futureExpiryDate_ != Date()) {
73 expiryDate = futureExpiryDate_;
74 } else {
75 // Get the expiry date of the option. This is the expiry date of the commodity future index.
76 const vector<string>& expiryDates = option_.exerciseDates();
77 QL_REQUIRE(expiryDates.size() == 1,
78 "Expected exactly one expiry date for CommodityOption but got " << expiryDates.size() << ".");
79 expiryDate = parseDate(expiryDates[0]);
80 }
81
82 // Clone the index with the relevant expiry date.
83 index_ = index->clone(expiryDate);
84
85 // Set the VanillaOptionTrade forwardDate_ if the index is a CommodityFuturesIndex - we possibly still have a
86 // CommoditySpotIndex at this point so check. Also, will only work for European exercise.
87 auto et = parseExerciseType(option_.style());
88 if (et == Exercise::European && boost::dynamic_pointer_cast<CommodityFuturesIndex>(index_)) {
89 forwardDate_ = expiryDate;
90 }
91
92 }
93
94 VanillaOptionTrade::build(engineFactory);
95
96 // LOG the volatility if the trade expiry date is in the future.
97 if (expiryDate_ > Settings::instance().evaluationDate()) {
98 DLOG("Implied vol for " << tradeType_ << " on " << assetName_ << " with expiry " << expiryDate_
99 << " and strike " << strike_.value() << " is "
100 << market->commodityVolatility(assetName_)->blackVol(expiryDate_, strike_.value()));
101 }
102
103 additionalData_["quantity"] = quantity_;
104 additionalData_["strike"] = strike_.value();
105 additionalData_["strikeCurrency"] = currency_;
106
107 // ISDA taxonomy, assuming Commodity follows the Equity template
108 additionalData_["isdaAssetClass"] = std::string("Commodity");
109 additionalData_["isdaBaseProduct"] = std::string("Option");
110 additionalData_["isdaSubProduct"] = std::string("Price Return Basic Performance");
111 // skip the transaction level mapping for now
112 additionalData_["isdaTransaction"] = std::string("");
113}

120void CommodityOption::fromXML(XMLNode* node) {
121
122 Trade::fromXML(node);
123
124 XMLNode* commodityNode = XMLUtils::getChildNode(node, "CommodityOptionData");
125 QL_REQUIRE(commodityNode, "A commodity option needs a 'CommodityOptionData' node");
126
127 option_.fromXML(XMLUtils::getChildNode(commodityNode, "OptionData"));
128
129 assetName_ = XMLUtils::getChildValue(commodityNode, "Name", true);
130 currency_ = XMLUtils::getChildValue(commodityNode, "Currency", true);
131 strike_.fromXML(commodityNode);
132 quantity_ = XMLUtils::getChildValueAsDouble(commodityNode, "Quantity", true);
133
134 isFuturePrice_ = boost::none;
135 if (XMLNode* n = XMLUtils::getChildNode(commodityNode, "IsFuturePrice"))
136 isFuturePrice_ = parseBool(XMLUtils::getNodeValue(n));
137
138 futureExpiryDate_ = Date();
139 if (XMLNode* n = XMLUtils::getChildNode(commodityNode, "FutureExpiryDate"))
140 futureExpiryDate_ = parseDate(XMLUtils::getNodeValue(n));
141}

143XMLNode* CommodityOption::toXML(XMLDocument& doc) {
144
145 XMLNode* node = Trade::toXML(doc);
146
147 XMLNode* eqNode = doc.allocNode("CommodityOptionData");
148 XMLUtils::appendNode(node, eqNode);
149
150 XMLUtils::appendNode(eqNode, option_.toXML(doc));
151
152 XMLUtils::addChild(doc, eqNode, "Name", assetName_);
153 XMLUtils::addChild(doc, eqNode, "Currency", currency_);
154 XMLUtils::appendNode(eqNode, strike_.toXML(doc));
155 XMLUtils::addChild(doc, eqNode, "Quantity", quantity_);
156
157 if (isFuturePrice_)
158 XMLUtils::addChild(doc, eqNode, "IsFuturePrice", *isFuturePrice_);
159
160 if (futureExpiryDate_ != Date())
161 XMLUtils::addChild(doc, eqNode, "FutureExpiryDate", to_string(futureExpiryDate_));
162
163 return node;
164}
