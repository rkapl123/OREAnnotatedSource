43 const vector<string>& InflationCapFloorPriceSurfaceConfig::quotes() {
44  if (quotes_.size() == 0) {
45  string type;
46  if (type_ == Type::ZC)
47  type = "ZC";
48  else if (type_ == Type::YY)
49  type = "YY";
50 
51  string base = type + "_INFLATIONCAPFLOOR/PRICE/" + index_ + "/";
52  for (auto m : maturities_) {
53  for (auto f : floorStrikes_) {
54  quotes_.push_back(base + m + "/F/" + f);
55  }
56  for (auto c : capStrikes_) {
57  quotes_.push_back(base + m + "/C/" + c);
58  }
59  }
60  }
61  return quotes_;
62 }
