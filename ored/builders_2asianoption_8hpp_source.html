42 //! Abstract Engine Builder for Asian Options
43 /*! Pricing engines are cached by asset/currency/expiry, where
44 expiry is null (Date()) if irrelevant.
45 \ingroup builders
46 */
47 class AsianOptionEngineBuilder
48 : public CachingOptionEngineBuilder<string, const string&, const Currency&, const AssetClass&, const Date&> {
49 public:
50 AsianOptionEngineBuilder(const string& model, const string& engine, const set<string>& tradeTypes,
51 const AssetClass& assetClass, const Date& expiryDate)
52 : CachingOptionEngineBuilder(model, engine, tradeTypes, assetClass), expiryDate_(expiryDate) {}
53
54 boost::shared_ptr<PricingEngine> engine(const string& assetName, const Currency& ccy, const Date& expiryDate) {
55 return CachingPricingEngineBuilder<string, const string&, const Currency&, const AssetClass&,
56 const Date&>::engine(assetName, ccy, assetClass_, expiryDate);
57 }
58
59 boost::shared_ptr<PricingEngine> engine(const Currency& ccy1, const Currency& ccy2, const Date& expiryDate) {
60 return CachingPricingEngineBuilder<string, const string&, const Currency&, const AssetClass&,
61 const Date&>::engine(ccy1.code(), ccy2, assetClass_, expiryDate);
62 }

67 protected:

69 const Date& expiryDate) override {

71 }

76 //! Discrete Monte Carlo Engine Builder for European Asian Arithmetic Average Price Options
77 /*! Pricing engines are cached by asset/currency/expiry, where
78 expiry is null (Date()) if irrelevant.
79 \ingroup builders
80 */
81 class EuropeanAsianOptionMCDAAPEngineBuilder : public AsianOptionEngineBuilder {
82 public:
83 EuropeanAsianOptionMCDAAPEngineBuilder(const string& model, const set<string>& tradeTypes,
84 const AssetClass& assetClass, const Date& expiryDate)
85 : AsianOptionEngineBuilder(model, "MCDiscreteArithmeticAPEngine", tradeTypes, assetClass, expiryDate) {}
86
87 std::string processType() override { return "Discrete"; }
88
89 protected:
90 virtual boost::shared_ptr<PricingEngine> engineImpl(const string& assetName, const Currency& ccy,
91 const AssetClass& assetClassUnderlying,
92 const Date& expiryDate) override {
93 bool brownianBridge = ore::data::parseBool(engineParameter("BrownianBridge", {}, false, "true"));
94 bool antitheticVariate = ore::data::parseBool(engineParameter("AntitheticVariate", {}, false, "true"));
95 bool controlVariate = ore::data::parseBool(engineParameter("ControlVariate", {}, false, "true"));
96 Size requiredSamples = ore::data::parseInteger(engineParameter("RequiredSamples", {}, false, "0"));
97 Real requiredTolerance = ore::data::parseReal(engineParameter("RequiredTolerance", {}, false, "0"));
98 Size maxSamples = ore::data::parseInteger(engineParameter("MaxSamples", {}, false, "0"));
99 BigNatural seed = ore::data::parseInteger(engineParameter("Seed", {}, false, "123456"));
100
101 // Check if values defaulted to 0, if so replace by Null<T>().
102 if (requiredSamples == 0)
103 requiredSamples = Null<Size>();
104 if (requiredTolerance == 0)
105 requiredTolerance = Null<Real>();
106 if (maxSamples == 0)
107 maxSamples = Null<Size>();
108 QL_REQUIRE(requiredSamples != QuantLib::Null<Size>() || requiredTolerance != QuantLib::Null<Real>(),
109 "RequiredSamples or RequiredTolerance must be set for engine MCDiscreteArithmeticAPEngine.");
110
111 boost::shared_ptr<GeneralizedBlackScholesProcess> gbsp =
112 getBlackScholesProcess(assetName, ccy, assetClassUnderlying);
113 return boost::make_shared<MCDiscreteArithmeticAPEngine<LowDiscrepancy>>(gbsp, brownianBridge, antitheticVariate,
114 controlVariate, requiredSamples,
115 requiredTolerance, maxSamples, seed);
116 }
117 };
