64 const vector<string>& SwaptionVolatilityCurveConfig::quotes() {
65 
66  if (quotes_.size() == 0) {
67  std::vector<string> tokens;
68  split(tokens, swapIndexBase_, boost::is_any_of("-"));
69 
70  Currency ccy = parseCurrency(tokens[0]);
71 
72  std::stringstream ssBase;
73  ssBase << "SWAPTION/" << volatilityType_ << "/" << ccy.code() << "/";
74  string base = ssBase.str();
75 
76  if (dimension_ == Dimension::ATM) {
77  for (auto o : optionTenors_) {
78  for (auto s : swapTenors_) {
79  std::stringstream ss;
80  ss << base << to_string(o) << "/" << to_string(s) << "/ATM";
81  quotes_.push_back(ss.str());
82  }
83  }
84 
85  } else {
86  for (auto o : smileOptionTenors_) {
87  for (auto s : smileSwapTenors_) {
88  for (auto sp : smileSpreads_) {
89  std::stringstream ss;
90  ss << base << to_string(o) << "/" << to_string(s) << "/Smile/" << to_string(sp);
91  quotes_.push_back(ss.str());
92  }
93  }
94  }
95  }
96  }
97  return quotes_;
98 }

100 void SwaptionVolatilityCurveConfig::fromXML(XMLNode* node) {
101  XMLUtils::checkNode(node, "SwaptionVolatility");
102 
103  curveID_ = XMLUtils::getChildValue(node, "CurveId", true);
104  curveDescription_ = XMLUtils::getChildValue(node, "CurveDescription", true);
105 
106  string dim = XMLUtils::getChildValue(node, "Dimension", true);
107  if (dim == "ATM") {
108  dimension_ = Dimension::ATM;
109  } else if (dim == "Smile") {
110  dimension_ = Dimension::Smile;
111  } else {
112  QL_FAIL("Dimension " << dim << " not recognized");
113  }
114 
115  string volType = XMLUtils::getChildValue(node, "VolatilityType", true);
116  if (volType == "Normal") {
117  volatilityType_ = VolatilityType::Normal;
118  } else if (volType == "Lognormal") {
119  volatilityType_ = VolatilityType::Lognormal;
120  } else if (volType == "ShiftedLognormal") {
121  volatilityType_ = VolatilityType::ShiftedLognormal;
122  } else {
123  QL_FAIL("Volatility type " << volType << " not recognized");
124  }
125 
126  string extr = XMLUtils::getChildValue(node, "Extrapolation", true);
127  extrapolate_ = true;
128  flatExtrapolation_ = true;
129  if (extr == "Linear") {
130  flatExtrapolation_ = false;
131  } else if (extr == "Flat") {
132  flatExtrapolation_ = true;
133  } else if (extr == "None") {
134  extrapolate_ = false;
135  } else {
136  QL_FAIL("Extrapolation " << extr << " not recognized");
137  }
138 
139  optionTenors_ = XMLUtils::getChildrenValuesAsPeriods(node, "OptionTenors", true);
140  swapTenors_ = XMLUtils::getChildrenValuesAsPeriods(node, "SwapTenors", true);
141 
142  string cal = XMLUtils::getChildValue(node, "Calendar", true);
143  calendar_ = parseCalendar(cal);
144 
145  string dc = XMLUtils::getChildValue(node, "DayCounter", true);
146  dayCounter_ = parseDayCounter(dc);
147 
148  string bdc = XMLUtils::getChildValue(node, "BusinessDayConvention", true);
149  businessDayConvention_ = parseBusinessDayConvention(bdc);
150 
151  shortSwapIndexBase_ = XMLUtils::getChildValue(node, "ShortSwapIndexBase", true);
152  swapIndexBase_ = XMLUtils::getChildValue(node, "SwapIndexBase", true);
153 
154  // optional smile stuff
155  smileOptionTenors_ = XMLUtils::getChildrenValuesAsPeriods(node, "SmileOptionTenors");
156  smileSwapTenors_ = XMLUtils::getChildrenValuesAsPeriods(node, "SmileSwapTenors");
157  smileSpreads_ = XMLUtils::getChildrenValuesAsDoublesCompact(node, "SmileSpreads");
158 }

160 XMLNode* SwaptionVolatilityCurveConfig::toXML(XMLDocument& doc) {
161  XMLNode* node = doc.allocNode("SwaptionVolatility");
162 
163  XMLUtils::addChild(doc, node, "CurveId", curveID_);
164  XMLUtils::addChild(doc, node, "CurveDescription", curveDescription_);
165 
166  if (dimension_ == Dimension::ATM) {
167  XMLUtils::addChild(doc, node, "Dimension", "ATM");
168  } else if (dimension_ == Dimension::Smile) {
169  XMLUtils::addChild(doc, node, "Dimension", "Smile");
170  } else {
171  QL_FAIL("Unkown Dimension in SwaptionVolatilityCurveConfig::toXML()");
172  }
173 
174  if (volatilityType_ == VolatilityType::Normal) {
175  XMLUtils::addChild(doc, node, "VolatilityType", "Normal");
176  } else if (volatilityType_ == VolatilityType::Lognormal) {
177  XMLUtils::addChild(doc, node, "VolatilityType", "Lognormal");
178  } else if (volatilityType_ == VolatilityType::ShiftedLognormal) {
179  XMLUtils::addChild(doc, node, "VolatilityType", "ShiftedLognormal");
180  } else {
181  QL_FAIL("Unknown VolatilityType in SwaptionVolatilityCurveConfig::toXML()");
182  }
183 
184  string extr_str = flatExtrapolation_ ? "Flat" : "Linear";
185  if (!extrapolate_)
186  extr_str = "None";
187  XMLUtils::addChild(doc, node, "Extrapolation", extr_str);
188 
189  XMLUtils::addGenericChildAsList(doc, node, "OptionTenors", optionTenors_);
190  XMLUtils::addGenericChildAsList(doc, node, "SwapTenors", swapTenors_);
191  XMLUtils::addChild(doc, node, "Calendar", to_string(calendar_));
192  XMLUtils::addChild(doc, node, "DayCounter", to_string(dayCounter_));
193  XMLUtils::addChild(doc, node, "BusinessDayConvention", to_string(businessDayConvention_));
194  XMLUtils::addChild(doc, node, "ShortSwapIndexBase", shortSwapIndexBase_);
195  XMLUtils::addChild(doc, node, "SwapIndexBase", swapIndexBase_);
196 
197  if (dimension_ == Dimension::Smile) {
198  XMLUtils::addGenericChildAsList(doc, node, "SmileOptionTenors", smileOptionTenors_);
199  XMLUtils::addGenericChildAsList(doc, node, "SmileSwapTenors", smileSwapTenors_);
200  XMLUtils::addGenericChildAsList(doc, node, "SmileSpreads", smileSpreads_);
201  }
202 
203  return node;
204 }
