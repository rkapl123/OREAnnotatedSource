76void CreditDefaultSwapOption::build(const boost::shared_ptr<EngineFactory>& engineFactory) {
77
78 DLOG("CreditDefaultSwapOption::build() called for trade " << id());
79
80 // Notionals
81 const auto& legData = swap_.leg();
82 const auto& ntls = legData.notionals();
83 QL_REQUIRE(ntls.size() == 1, "CreditDefaultSwapOption requires a single notional.");
84 notional_ = ntls.front();
85 notionalCurrency_ = legData.currency();
86
87 // Type of instrument we build depends on whether the reference entity has already defaulted.
88 if (asi_) {
89 buildDefaulted(engineFactory);
90 } else {
91 buildNoDefault(engineFactory);
92 }
93
94 // ISDA taxonomy
95 additionalData_["isdaAssetClass"] = string("Credit");
96 additionalData_["isdaBaseProduct"] = string("Swaptions");
97 // set isdaSubProduct to entityType in credit reference data
98 additionalData_["isdaSubProduct"] = string("");
99 string entity = swap_.referenceInformation() ?
100 swap_.referenceInformation()->referenceEntityId() :
101 swap_.creditCurveId();
102 boost::shared_ptr<ReferenceDataManager> refData = engineFactory->referenceData();
103 if (refData && refData->hasData("Credit", entity)) {
104 auto refDatum = refData->getData("Credit", entity);
105 boost::shared_ptr<CreditReferenceDatum> creditRefDatum = boost::dynamic_pointer_cast<CreditReferenceDatum>(refDatum);
106 additionalData_["isdaSubProduct"] = creditRefDatum->creditData().entityType;
107 if (creditRefDatum->creditData().entityType == "") {
108 ALOG("EntityType is blank in credit reference data for entity " << entity);
109 }
110 } else {
111 ALOG("Credit reference data missing for entity " << entity << ", isdaSubProduct left blank");
112 }
113 // Skip the transaction level mapping for now
114 additionalData_["isdaTransaction"] = string("");
115}

147

168 QL_REQUIRE(optionData, "Expected OptionData node on trade " << id() << ".");
169 option_.fromXML(optionData);

199void CreditDefaultSwapOption::buildNoDefault(const boost::shared_ptr<EngineFactory>& engineFactory) {
200
201 DLOG("CreditDefaultSwapOption: building CDS option trade " << id() << " given no default.");
202
203 // Need fixed leg data with one rate. This should be the standard running coupon on the CDS e.g.
204 // generally 100bp for IG CDS and 500bp for HY CDS. For single name CDS options, one can use this field to give
205 // the strike spread. It may matter for the resulting valuation depending on the engine that is used - see
206 // "A CDS Option Miscellany, Richard J. Martin, 2019, Section 2.4".
207 const auto& legData = swap_.leg();
208 QL_REQUIRE(legData.legType() == "Fixed", "CDS option " << id() << " requires fixed leg.");
209 auto fixedLegData = boost::dynamic_pointer_cast<FixedLegData>(legData.concreteLegData());
210 QL_REQUIRE(fixedLegData->rates().size() == 1, "Index CDS option " << id() << " requires single fixed rate.");
211 auto runningCoupon = fixedLegData->rates().front();
212
213 // Payer (Receiver) swaption if the leg is paying (receiving).
214 auto side = legData.isPayer() ? Protection::Side::Buyer : Protection::Side::Seller;
215
216 // Day counter. In general for CDS, the standard day counter is Actual/360 and the final
217 // period coupon accrual includes the maturity date.
218 Actual360 standardDayCounter;
219 const std::string& strDc = legData.dayCounter();
220 DayCounter dc = strDc.empty() ? standardDayCounter : parseDayCounter(strDc);
221 DayCounter lastPeriodDayCounter = dc == standardDayCounter ? Actual360(true) : dc;
222
223 // Schedule
224 Schedule schedule = makeSchedule(legData.schedule());
225 BusinessDayConvention payConvention = legData.paymentConvention().empty() ? Following :
226 parseBusinessDayConvention(legData.paymentConvention());
227
228 // Don't support upfront fee on the underlying CDS for the moment.
229 QL_REQUIRE(swap_.upfrontFee() == Null<Real>() || close(swap_.upfrontFee(), 0.0),
230 "Upfront fee on the CDS underlying a CDS option is not supported.");
231
232 // The underlying CDS trade
233 auto cds = boost::make_shared<QuantLib::CreditDefaultSwap>(side, notional_, runningCoupon, schedule,
234 payConvention, dc, swap_.settlesAccrual(), swap_.protectionPaymentTime(), swap_.protectionStart(),
235 boost::shared_ptr<Claim>(), lastPeriodDayCounter, true, swap_.tradeDate(), swap_.cashSettlementDays());
236
237 // Set engine on the underlying CDS.
238 auto cdsBuilder = boost::dynamic_pointer_cast<CreditDefaultSwapEngineBuilder>(
239 engineFactory->builder("CreditDefaultSwap"));
240 QL_REQUIRE(cdsBuilder, "CreditDefaultSwapOption expected CDS engine " <<
241 " builder for underlying while building trade " << id() << ".");
242 npvCurrency_ = legData.currency();
243 auto ccy = parseCurrency(npvCurrency_);
244 cds->setPricingEngine(cdsBuilder->engine(ccy, swap_.creditCurveId(), swap_.recoveryRate()));
245
246 // Check option data
247 QL_REQUIRE(option_.style() == "European", "CreditDefaultSwapOption option style must" <<
248 " be European but got " << option_.style() << ".");
249 QL_REQUIRE(!option_.payoffAtExpiry(), "CreditDefaultSwapOption payoff must be at exercise.");
250 QL_REQUIRE(option_.exerciseFees().empty(), "CreditDefaultSwapOption cannot handle exercise fees.");
251
252 // Exercise must be European
253 const auto& exerciseDates = option_.exerciseDates();
254 QL_REQUIRE(exerciseDates.size() == 1, "CreditDefaultSwapOption expects one exercise date" <<
255 " but got " << exerciseDates.size() << " exercise dates.");
256 Date exerciseDate = parseDate(exerciseDates.front());
257 boost::shared_ptr<Exercise> exercise = boost::make_shared<EuropeanExercise>(exerciseDate);
258
259 // Limit strike type to Spread for now.
260 auto strikeType = parseCdsOptionStrikeType(strikeType_);
261 QL_REQUIRE(strikeType == QuantExt::CdsOption::Spread, "CreditDefaultSwapOption strike type must be Spread.");
262
263 // If the strike_ is Null, the strike is taken as the running coupon.
264 Real strike = strike_ == Null<Real>() ? runningCoupon : strike_;
265
266 // Build the option instrument
267 auto cdsOption = boost::make_shared<QuantExt::CdsOption>(cds, exercise, knockOut_, strike, strikeType);
268
269 // Set the option engine
270 auto cdsOptionEngineBuilder = boost::dynamic_pointer_cast<CreditDefaultSwapOptionEngineBuilder>(
271 engineFactory->builder("CreditDefaultSwapOption"));
272 QL_REQUIRE(cdsOptionEngineBuilder, "CreditDefaultSwapOption expected CDS option engine " <<
273 " builder for underlying while building trade " << id() << ".");
274 cdsOption->setPricingEngine(cdsOptionEngineBuilder->engine(ccy, swap_.creditCurveId(), term_));
275
276 // Copying here what is done for the index CDS option. The comment there is:
277 // Align option product maturities with ISDA AANA/GRID guidance as of November 2020.
278 maturity_ = cds->coupons().back()->date();
279
280 // Set Trade members.
281 legs_ = { cds->coupons() };
282 legCurrencies_ = { npvCurrency_ };
283 legPayers_ = { legData.isPayer() };
284
285 // Include premium if enough information is provided
286 vector<boost::shared_ptr<Instrument>> additionalInstruments;
287 vector<Real> additionalMultipliers;
288 string marketConfig = cdsOptionEngineBuilder->configuration(MarketContext::pricing);
289 maturity_ =
290 std::max(maturity_, addPremium(engineFactory, ccy, marketConfig, additionalInstruments, additionalMultipliers));
291
292 // Instrument wrapper depends on the settlement type.
293 Position::Type positionType = parsePositionType(option_.longShort());
294 Settlement::Type settleType = parseSettlementType(option_.settlement());
295 // The instrument build should be indpednent of the evaluation date. However, the general behavior
296 // in ORE (e.g. IR swaptions) for normal pricing runs is that the option is considered expired on
297 // the expiry date with no assumptions on an (automatic) exercise. Therefore we build a vanilla
298 // instrument if the exercise date is <= the eval date at build time.
299 if (settleType == Settlement::Cash || exerciseDate <= Settings::instance().evaluationDate()) {
300 Real indicatorLongShort = positionType == Position::Long ? 1.0 : -1.0;
301 instrument_ = boost::make_shared<VanillaInstrument>(cdsOption, indicatorLongShort,
302 additionalInstruments, additionalMultipliers);
303 } else {
304 bool isLong = positionType == Position::Long;
305 bool isPhysical = settleType == Settlement::Physical;
306 instrument_ = boost::make_shared<EuropeanOptionWrapper>(cdsOption, isLong, exerciseDate,
307 isPhysical, cds, 1.0, 1.0, additionalInstruments, additionalMultipliers);
308 }
309}

311void CreditDefaultSwapOption::buildDefaulted(const boost::shared_ptr<EngineFactory>& engineFactory) {
312
313 DLOG("CreditDefaultSwapOption: building CDS option trade " << id() << " given default occurred.");
314
315 // We add a simple payment for CDS options where the reference entity has already defaulted.
316 // If it is a knock-out CDS option, we add a dummy payment of 0.0 with date today instead of throwing.
317 const auto& legData = swap_.leg();
318 Date paymentDate = engineFactory->market()->asofDate();
319 Real amount = 0.0;
320 if (!knockOut_) {
321 paymentDate = asi_->auctionSettlementDate();
322 amount = notional_ * (1 - asi_->auctionFinalPrice());
323 // If it is a receiver option, i.e. selling protection, the FEP is paid out.
324 if (!legData.isPayer())
325 amount *= -1.0;
326 }
327
328 // Use the add premiums method to add the payment.
329 string marketConfig = Market::defaultConfiguration;
330 auto ccy = parseCurrency(notionalCurrency_);
331 vector<boost::shared_ptr<Instrument>> additionalInstruments;
332 vector<Real> additionalMultipliers;
333 Date premiumPayDate =
334 addPremiums(additionalInstruments, additionalMultipliers, 1.0,
335 PremiumData(amount, notionalCurrency_, paymentDate), 1.0, ccy, engineFactory, marketConfig);
336 DLOG("FEP payment (date = " << paymentDate << ", amount = " << amount << ") added for CDS option " << id() << ".");
337
338 // Use the instrument added as the main instrument and clear the vectors
339 auto qlInst = additionalInstruments.back();
340 QL_REQUIRE(qlInst, "Expected a FEP payment to have been added for CDS option " << id() << ".");
341 maturity_ = std::max(paymentDate, premiumPayDate);
342 additionalInstruments.clear();
343 additionalMultipliers.clear();
344
345 // Include premium if enough information is provided
346 addPremium(engineFactory, ccy, marketConfig, additionalInstruments, additionalMultipliers);
347
348 // Instrument wrapper.
349 Position::Type positionType = parsePositionType(option_.longShort());
350 Real indicatorLongShort = positionType == Position::Long ? 1.0 : -1.0;
351 instrument_ = boost::make_shared<VanillaInstrument>(qlInst, indicatorLongShort,
352 additionalInstruments, additionalMultipliers);
353}

355Date CreditDefaultSwapOption::addPremium(const boost::shared_ptr<EngineFactory>& ef,
356 const Currency& tradeCurrency,
357 const string& marketConfig,
358 vector<boost::shared_ptr<Instrument>>& additionalInstruments,
359 vector<Real>& additionalMultipliers) {
360 // The premium amount is always provided as a non-negative amount. Assign the correct sign here i.e.
361 // pay the premium if long the option and receive the premium if short the option.
362 Position::Type positionType = parsePositionType(option_.longShort());
363 Real indicatorLongShort = positionType == Position::Long ? 1.0 : -1.0;
364 return addPremiums(additionalInstruments, additionalMultipliers, 1.0, option_.premiumData(), indicatorLongShort,
365 tradeCurrency, ef, marketConfig);
366}
